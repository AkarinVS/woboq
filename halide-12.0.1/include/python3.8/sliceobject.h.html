15/*
16
17A slice object containing start, stop, and step data members (the
18names are from range). After much talk with Guido, it was decided to
19let these be any arbitrary python type. Py_None stands for omitted values.
20*/
21#ifndef Py_LIMITED_API
22typedef struct {
23 PyObject_HEAD
24 PyObject *start, *stop, *step; /* not NULL */
25} PySliceObject;
26#endif
