1// This file defines a generator for a first order IIR low pass filter
2// for a 2D image.
3
4#include "Halide.h"
5
6using namespace Halide;
7using namespace Halide::BoundaryConditions;
8
9Var x, y, c;
10
11// Defines a func to blur the columns of an input with a first order low
12// pass IIR filter, followed by a transpose.
13Func blur_cols_transpose(Func input, Expr height, Expr alpha, bool skip_schedule, Target target) {
14 Func blur("blur");
15
16 const int vec = target.natural_vector_size<float>();
17
18 // Pure definition: do nothing.
19 blur(x, y, c) = undef<float>();
20 // Update 0: set the top row of the result to the input.
21 blur(x, 0, c) = input(x, 0, c);
22 // Update 1: run the IIR filter down the columns.
23 RDom ry(1, height - 1);
24 blur(x, ry, c) =
25 (1 - alpha) * blur(x, ry - 1, c) + alpha * input(x, ry, c);
26 // Update 2: run the IIR blur up the columns.
27 Expr flip_ry = height - ry - 1;
28 blur(x, flip_ry, c) =
29 (1 - alpha) * blur(x, flip_ry + 1, c) + alpha * blur(x, flip_ry, c);
30
31 // Transpose the blur.
32 Func transpose("transpose");
33 transpose(x, y, c) = blur(y, x, c);
34
35 // Schedule
36 if (!skip_schedule) {
37 if (!target.has_gpu_feature()) {
38 // CPU schedule.
39 // 8.2ms on an Intel i9-9960X using 16 threads
40 // Split the transpose into tiles of rows. Parallelize over channels
41 // and strips (Halide supports nested parallelism).
42 Var xo, yo, t;
43 transpose.compute_root()
44 .tile(x, y, xo, yo, x, y, vec, vec * 4)
45 .vectorize(x)
46 .parallel(yo)
47 .parallel(c);
48
49 // Run the filter on each row of tiles (which corresponds to a strip of
50 // columns in the input).
51 blur.compute_at(transpose, yo);
52
53 // Vectorize computations within the strips.
54 blur.update(1)
55 .reorder(x, ry)
56 .vectorize(x);
57 blur.update(2)
58 .reorder(x, ry)
59 .vectorize(x);
60 } else if (target.has_feature(Target::CUDA)) {
61 // CUDA-specific GPU schedule (using gpu_lanes)
62
63 // Really for an IIR on the GPU you should use a more
64 // specialized DSL like RecFilter, but we can schedule it
65 // in Halide adequately, we just can't extract any
66 // parallelism from the scan dimension. Most GPUs will be
67 // heavily under-utilized with this schedule and thus
68 // unable to hide the memory latencies to L2.
69
70 const int warp_size = 32;
71
72 // 2.06ms on a 2060 RTX
73 Var xi, yi;
74 transpose.compute_root()
75 .tile(x, y, xi, yi, warp_size, warp_size)
76 .gpu_blocks(y, c)
77 .gpu_lanes(xi);
78
79 blur.compute_at(transpose, y)
80 .store_in(MemoryType::Heap) // Too large to fit into shared memory
81 .gpu_lanes(x);
82 blur.update(0)
83 .gpu_lanes(x);
84
85 // We can't hide load latencies by swapping in other warps
86 // because we don't have enough available parallelism for
87 // that, but if we unroll the scan loop a little then the
88 // ptx compiler can reorder the loads earlier than the
89 // fmas, and cover latency that way. Saves 1.7ms!
90 blur.update(1)
91 .unroll(ry, 8)
92 .gpu_lanes(x);
93 blur.update(2)
94 .unroll(ry, 8)
95 .gpu_lanes(x);
96
97 // Stage the transpose input through shared so that we do
98 // strided loads out of shared instead of global. By
99 // default the stride would be the width of the
100 // allocation, which is the warp size. This can cause bank
101 // conflicts. We can improve matters by padding out the
102 // storage horizontally to make the stride coprime with
103 // the warp size, so that each load has a distinct
104 // remainder modulo the warp size. warp_size + 1 will
105 // do. This saves 0.05 ms
106 blur.in()
107 .align_storage(x, warp_size + 1)
108 .compute_at(transpose, x)
109 .gpu_lanes(x);
110 } else {
111 // Generic GPU schedule (for gpus without gpu_lanes() support)
112 Var xi, yi;
113 blur.compute_root();
114 blur.update(0)
115 .split(x, x, xi, 32)
116 .gpu_blocks(x, c)
117 .gpu_threads(xi);
118 blur.update(1)
119 .split(x, x, xi, 32)
120 .gpu_blocks(x, c)
121 .gpu_threads(xi);
122 blur.update(2)
123 .split(x, x, xi, 32)
124 .gpu_blocks(x, c)
125 .gpu_threads(xi);
126 }
127 }
128
129 return transpose;
130}

132class IirBlur : public Generator<IirBlur> {
133public:
134 // This is the input image: a 3D (color) image with 32 bit float
135 // pixels.
136 Input<Buffer<float>> input{"input", 3};
137 // The filter coefficient, alpha is the weight of the input to the
138 // filter.
139 Input<float> alpha{"alpha"};
140
141 Output<Buffer<float>> output{"output", 3};
142
143 void generate() {
144 Expr width = input.width();
145 Expr height = input.height();
146
147 // First, blur the columns of the input.
148 Func blury_T = blur_cols_transpose(input, height, alpha, auto_schedule, get_target());
149
150 // Blur the columns again (the rows of the original).
151 Func blur = blur_cols_transpose(blury_T, width, alpha, auto_schedule, get_target());
152
153 // Scheduling is done inside blur_cols_transpose.
154 output = blur;
155
156 // Estimates
157 {
158 input.dim(0).set_estimate(0, 1536);
159 input.dim(1).set_estimate(0, 2560);
160 input.dim(2).set_estimate(0, 3);
161 alpha.set_estimate(0.1f);
162 output.dim(0).set_estimate(0, 1536);
163 output.dim(1).set_estimate(0, 2560);
164 output.dim(2).set_estimate(0, 3);
165 }
166 }
167};
