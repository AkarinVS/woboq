30template<int rows, int cols>
31struct Matrix {
32 Expr exprs[rows][cols];
33
34 Expr operator()(int i, int j) const {
35 return exprs[i][j];

37

55// Matrix-matrix multiply.
56template<int R, int S, int T>
57Matrix<R, T> mat_mul(const Matrix<R, S> &A, const Matrix<S, T> &B) {
58 Matrix<R, T> result;
59 for (int r = 0; r < R; r++) {
60 for (int t = 0; t < T; t++) {
61 result(r, t) = 0.0f;
62 for (int s = 0; s < S; s++) {
63 result(r, t) += A(r, s) * B(s, t);
64 }
65 }
66 }
67 return result;
68}

70// Solve Ax = b at each x, y, z. Compute the result at the given Func
71// and Var. Not currently used, but preserved for reference.
72template<int M, int N>
73Matrix<M, N> solve(Matrix<M, M> A, Matrix<M, N> b, Func compute, Var at, bool skip_schedule, Target target) {
74 // Put the input matrices in a Func to do the Gaussian elimination.
75 Var vi, vj;
76 Func f;
77 f(x, y, z, vi, vj) = undef<float>();
78 for (int i = 0; i < M; i++) {
79 for (int j = 0; j < M; j++) {
80 f(x, y, z, i, j) = A(i, j);
81 }
82 for (int j = 0; j < N; j++) {
83 f(x, y, z, i, j + M) = b(i, j);
84 }
85 }
86
87 // eliminate lower left
88 for (int k = 0; k < M - 1; k++) {
89 for (int i = k + 1; i < M; i++) {
90 f(x, y, z, -1, 0) = f(x, y, z, i, k) / f(x, y, z, k, k);
91 for (int j = k + 1; j < M + N; j++) {
92 f(x, y, z, i, j) -= f(x, y, z, k, j) * f(x, y, z, -1, 0);
93 }
94 f(x, y, z, i, k) = 0.0f;
95 }
96 }
97
98 // eliminate upper right
99 for (int k = M - 1; k > 0; k--) {
100 for (int i = 0; i < k; i++) {
101 f(x, y, z, -1, 0) = f(x, y, z, i, k) / f(x, y, z, k, k);
102 for (int j = k + 1; j < M + N; j++) {
103 f(x, y, z, i, j) -= f(x, y, z, k, j) * f(x, y, z, -1, 0);
104 }
105 f(x, y, z, i, k) = 0.0f;
106 }
107 }
108
109 // Divide by diagonal and put it in the output matrix.
110 for (int i = 0; i < M; i++) {
111 f(x, y, z, i, i) = 1.0f / f(x, y, z, i, i);
112 for (int j = 0; j < N; j++) {
113 b(i, j) = f(x, y, z, i, j + M) * f(x, y, z, i, i);
114 }
115 }
116
117 if (!skip_schedule) {
118 if (!target.has_gpu_feature()) {
119 for (int i = 0; i < f.num_update_definitions(); i++) {
120 f.update(i).vectorize(x);
121 }
122 }
123
124 f.compute_at(compute, at);
125 }
126
127 return b;
128};

130// Solve Ax = b at each x, y, z exploiting the fact that A is
131// symmetric. Compute the result at the given Func and Var.
132template<int M, int N>
133Matrix<M, N> solve_symmetric(Matrix<M, M> A_, Matrix<M, N> b_,
134 Func compute, Var at, bool skip_schedule, Target target) {
135
136 // Put the input matrices in a Func to do sqrt-free Cholesky.
137 // See https://users.wpi.edu/~walker/MA514/HANDOUTS/cholesky.pdf
138 // for an explanation of sqrt-free Cholesky.
139
140 Var vi, vj;
141 Func f;
142 f(x, y, z, vi, vj) = undef<float>();
143
144 // Add more usefully-named accessors.
145 auto A = [&](int i, int j) {
146 return f(x, y, z, i, j);
147 };
148 auto b = [&](int i, int j) {
149 return f(x, y, z, i, M + j);
150 };
151
152 for (int i = 0; i < M; i++) {
153 for (int j = 0; j < M; j++) {
154 A(i, j) = A_(i, j);
155 }
156 for (int j = 0; j < N; j++) {
157 b(i, j) = b_(i, j);
158 }
159 }
160
161 // L D L' factorization, packed into a single matrix. We'll store
162 // L in the lower triangle, 1/D on the diagonal, and ??? TODO in
163 // the upper triangle.
164 for (int j = 0; j < M; j++) {
165 // Normalize the jth column starting at the jth row, storing
166 // the normalization factor on the diagonal. Because A(i, j)
167 // is symmetric, the unnormalized version stays in the jth
168 // row.
169 A(j, j) = fast_inverse(A(j, j));
170 for (int i = j + 1; i < M; i++) {
171 A(i, j) *= A(j, j);
172 }
173
174 // Subtract the outer product of the jth column with its
175 // unnormalized version from the rest of the matrix down and
176 // to the right of it.
177 for (int i = j + 1; i < M; i++) {
178 for (int k = j + 1; k < M; k++) {
179 if (k < i) {
180 // We already did this one. Exploit symmetry
181 A(i, k) = A(k, i);
182 } else {
183 A(i, k) -= A(k, j) * A(j, i);
184 }
185 }
186 }
187 }
188
189 // We're done with the upper (unnormalized) triangle
190 // now.
191
192 // Back substitute to solve:
193 // LDL' x = b
194 // We're going to peel the matrices off the left-hand-side from
195 // left to right, updating b as we go.
196 Matrix<M, N> result;
197 for (int k = 0; k < N; k++) {
198 // First remove the leftmost L, by solving Lz = b.
199 for (int j = 0; j < M; j++) {
200 for (int i = 0; i < j; i++) {
201 b(j, k) -= A(j, i) * b(i, k);
202 }
203 }
204 // L has a unit diagonal, so there's no scaling step
205
206 // The problem is now DL' x = b
207
208 // Multiply both sizes by D inverse, which we have stored on the
209 // diagonal.
210 for (int j = 0; j < M; j++) {
211 b(j, k) *= A(j, j);
212 }
213
214 // The problem is now L' x = b
215
216 // Multiply both sides by L transpose inverse.
217 for (int j = M - 1; j >= 0; j--) {
218 for (int i = j + 1; i < M; i++) {
219 b(j, k) -= A(i, j) * b(i, k);
220 }
221 }
222
223 for (int j = 0; j < M; j++) {
224 result(j, k) = b(j, k);
225 }
226 }
227
228 if (!skip_schedule) {
229 if (!target.has_gpu_feature()) {
230 for (int i = 0; i < f.num_update_definitions(); i++) {
231 f.update(i).vectorize(x);
232 }
233 }
234
235 f.compute_at(compute, at);
236 }
237
238 return result;
239}

241template<int N, int M>
242Matrix<M, N> transpose(const Matrix<N, M> &in) {
243 Matrix<M, N> out;
244 for (int i = 0; i < N; i++) {
245 for (int j = 0; j < M; j++) {
246 out(j, i) = in(i, j);
247 }
248 }
249 return out;
250}

252Expr pack_channels(Var c, std::vector<Expr> exprs) {
253 Expr e = exprs.back();
254 for (int i = (int)exprs.size() - 2; i >= 0; i--) {
255 e = select(c == i, exprs[i], e);
256 }
257 return e;
258}

260class BGU : public Generator<BGU> {
261public:
262 // Size of each luma bin in the grid. Typically 1/8.
263 Input<float> r_sigma{"r_sigma"};
264
265 // Size of each spatial bin in the grid. Typically 16.
266 Input<int> s_sigma{"s_sigma"};
267
268 Input<Buffer<float>> splat_loc{"splat_loc", 3};
269 Input<Buffer<float>> values{"values", 3};
270 Input<Buffer<float>> slice_loc{"slice_loc", 3};
271
272 Output<Buffer<float>> output{"output", 3};
273
274 void generate() {
275 // Algorithm
276
277 // Add a boundary condition to the inputs.
278 Func clamped_values = BoundaryConditions::repeat_edge(values);
279 Func clamped_splat_loc = BoundaryConditions::repeat_edge(splat_loc);
280
281 // Figure out how much we're upsampling by. Not relevant if we're
282 // just fitting curves.
283 Expr upsample_factor_x =
284 i32(ceil(f32(slice_loc.width()) / splat_loc.width()));
285 Expr upsample_factor_y =
286 i32(ceil(f32(slice_loc.height()) / splat_loc.height()));
287 Expr upsample_factor = max(upsample_factor_x, upsample_factor_y);
288
289 Func gray_splat_loc;
290 gray_splat_loc(x, y) = (0.25f * clamped_splat_loc(x, y, 0) +
291 0.5f * clamped_splat_loc(x, y, 1) +
292 0.25f * clamped_splat_loc(x, y, 2));
293
294 Func gray_slice_loc;
295 gray_slice_loc(x, y) = (0.25f * slice_loc(x, y, 0) +
296 0.5f * slice_loc(x, y, 1) +
297 0.25f * slice_loc(x, y, 2));
298
299 // Construct the bilateral grid
300 Func histogram("histogram");
301 RDom r(0, s_sigma, 0, s_sigma);
302 {
303 histogram(x, y, z, c) = 0.0f;
304
305 Expr sx = x * s_sigma + r.x - s_sigma / 2, sy = y * s_sigma + r.y - s_sigma / 2;
306 Expr pos = gray_splat_loc(sx, sy);
307 pos = clamp(pos, 0.0f, 1.0f);
308 Expr zi = cast<int>(round(pos * (1.0f / r_sigma)));
309
310 // Sum all the terms we need to fit a line relating
311 // low-res input to low-res output within this bilateral grid
312 // cell
313 Expr vr = clamped_values(sx, sy, 0), vg = clamped_values(sx, sy, 1), vb = clamped_values(sx, sy, 2);
314 Expr sr = clamped_splat_loc(sx, sy, 0), sg = clamped_splat_loc(sx, sy, 1), sb = clamped_splat_loc(sx, sy, 2);
315
316 histogram(x, y, zi, c) +=
317 pack_channels(c,
318 {sr * sr, sr * sg, sr * sb, sr,
319 sg * sg, sg * sb, sg,
320 sb * sb, sb,
321 1.0f,
322 vr * sr, vr * sg, vr * sb, vr,
323 vg * sr, vg * sg, vg * sb, vg,
324 vb * sr, vb * sg, vb * sb, vb});
325 }
326
327 // Convolution pyramids (Farbman et al.) suggests convolving by
328 // something 1/d^3-like to get an interpolating membrane, so we do
329 // that. We could also just use a convolution pyramid here, but
330 // these grids are really small, so it's OK for the filter to drop
331 // sharply and truncate early.
332 Expr t0 = 1.0f / 64, t1 = 1.0f / 27, t2 = 1.0f / 8, t3 = 1.0f;
333
334 // Blur the grid using a seven-tap filter
335 Func blurx("blurx"), blury("blury"), blurz("blurz");
336
337 blurz(x, y, z, c) = (histogram(x, y, z - 3, c) * t0 +
338 histogram(x, y, z - 2, c) * t1 +
339 histogram(x, y, z - 1, c) * t2 +
340 histogram(x, y, z, c) * t3 +
341 histogram(x, y, z + 1, c) * t2 +
342 histogram(x, y, z + 2, c) * t1 +
343 histogram(x, y, z + 3, c) * t0);
344 blury(x, y, z, c) = (blurz(x, y - 3, z, c) * t0 +
345 blurz(x, y - 2, z, c) * t1 +
346 blurz(x, y - 1, z, c) * t2 +
347 blurz(x, y, z, c) * t3 +
348 blurz(x, y + 1, z, c) * t2 +
349 blurz(x, y + 2, z, c) * t1 +
350 blurz(x, y + 3, z, c) * t0);
351 blurx(x, y, z, c) = (blury(x - 3, y, z, c) * t0 +
352 blury(x - 2, y, z, c) * t1 +
353 blury(x - 1, y, z, c) * t2 +
354 blury(x, y, z, c) * t3 +
355 blury(x + 1, y, z, c) * t2 +
356 blury(x + 2, y, z, c) * t1 +
357 blury(x + 3, y, z, c) * t0);
358
359 // Do the solve, to convert the accumulated values to the affine
360 // matrices.
361 Func line("line");
362 {
363 // Pull out the 4x4 symmetric matrix from the values we've
364 // accumulated.
365 Matrix<4, 4> A;
366 A(0, 0) = blurx(x, y, z, 0);
367 A(0, 1) = blurx(x, y, z, 1);
368 A(0, 2) = blurx(x, y, z, 2);
369 A(0, 3) = blurx(x, y, z, 3);
370 A(1, 0) = A(0, 1);
371 A(1, 1) = blurx(x, y, z, 4);
372 A(1, 2) = blurx(x, y, z, 5);
373 A(1, 3) = blurx(x, y, z, 6);
374 A(2, 0) = A(0, 2);
375 A(2, 1) = A(1, 2);
376 A(2, 2) = blurx(x, y, z, 7);
377 A(2, 3) = blurx(x, y, z, 8);
378 A(3, 0) = A(0, 3);
379 A(3, 1) = A(1, 3);
380 A(3, 2) = A(2, 3);
381 A(3, 3) = blurx(x, y, z, 9);
382
383 // Pull out the rhs
384 Matrix<4, 3> b;
385 b(0, 0) = blurx(x, y, z, 10);
386 b(1, 0) = blurx(x, y, z, 11);
387 b(2, 0) = blurx(x, y, z, 12);
388 b(3, 0) = blurx(x, y, z, 13);
389 b(0, 1) = blurx(x, y, z, 14);
390 b(1, 1) = blurx(x, y, z, 15);
391 b(2, 1) = blurx(x, y, z, 16);
392 b(3, 1) = blurx(x, y, z, 17);
393 b(0, 2) = blurx(x, y, z, 18);
394 b(1, 2) = blurx(x, y, z, 19);
395 b(2, 2) = blurx(x, y, z, 20);
396 b(3, 2) = blurx(x, y, z, 21);
397
398 // Regularize by pushing the solution towards the average gain
399 // in this cell = (average output luma + eps) / (average input luma + eps).
400 const float lambda = 1e-6f;
401 const float epsilon = 1e-6f;
402
403 // The bottom right entry of A is a count of the number of
404 // constraints affecting this cell.
405 Expr N = A(3, 3);
406
407 // The last row of each matrix is the sum of input and output
408 // RGB values for the pixels affecting this cell. Instead of
409 // dividing them by N+1 to get averages, we'll multiply
410 // epsilon by N+1. This saves two divisions.
411 Expr output_luma = b(3, 0) + 2 * b(3, 1) + b(3, 2) + epsilon * (N + 1);
412 Expr input_luma = A(3, 0) + 2 * A(3, 1) + A(3, 2) + epsilon * (N + 1);
413 Expr gain = output_luma / input_luma;
414
415 // Add lambda and lambda*gain to the diagonal of the
416 // matrices. The matrices are sums/moments rather than
417 // means/covariances, so just like above we need to multiply
418 // lambda by N+1 so that it's equivalent to adding a constant
419 // to the diagonal of a covariance matrix. Otherwise it does
420 // nothing in cells with lots of linearly-dependent
421 // constraints.
422 Expr weighted_lambda = lambda * (N + 1);
423 A(0, 0) += weighted_lambda;
424 A(1, 1) += weighted_lambda;
425 A(2, 2) += weighted_lambda;
426 A(3, 3) += weighted_lambda;
427
428 b(0, 0) += weighted_lambda * gain;
429 b(1, 1) += weighted_lambda * gain;
430 b(2, 2) += weighted_lambda * gain;
431
432 // Now solve Ax = b
433 Matrix<3, 4> result = transpose(solve_symmetric(A, b, line, x, auto_schedule, get_target()));
434
435 // Pack the resulting matrix into the output Func.
436 line(x, y, z, c) = pack_channels(c, {result(0, 0),
437 result(0, 1),
438 result(0, 2),
439 result(0, 3),
440 result(1, 0),
441 result(1, 1),
442 result(1, 2),
443 result(1, 3),
444 result(2, 0),
445 result(2, 1),
446 result(2, 2),
447 result(2, 3)});
448 }
449
450 // If using the shader we stop there, and the Func "line" is the
451 // output. We also compile a more convenient but slower version
452 // that does the trilerp and evaluates the model inside the same
453 // Halide pipeline.
454
455 // We'll take trilinear samples to compute the output. We factor
456 // this into several stages to make better use of SIMD.
457 Func interpolated("interpolated");
458 Func slice_loc_z("slice_loc_z");
459 Func interpolated_matrix_x("interpolated_matrix_x");
460 Func interpolated_matrix_y("interpolated_matrix_y");
461 Func interpolated_matrix_z("interpolated_matrix_z");
462 {
463 // Spatial bin size in the high-res image.
464 Expr big_sigma = s_sigma * upsample_factor;
465
466 // Upsample the matrices in x and y.
467 Expr yf = cast<float>(y) / big_sigma;
468 Expr yi = cast<int>(floor(yf));
469 yf -= yi;
470 interpolated_matrix_y(x, y, z, c) =
471 lerp(line(x, yi, z, c),
472 line(x, yi + 1, z, c),
473 yf);
474
475 Expr xf = cast<float>(x) / big_sigma;
476 Expr xi = cast<int>(floor(xf));
477 xf -= xi;
478 interpolated_matrix_x(x, y, z, c) =
479 lerp(interpolated_matrix_y(xi, y, z, c),
480 interpolated_matrix_y(xi + 1, y, z, c),
481 xf);
482
483 // Sample it along the z direction using intensity.
484 Expr num_intensity_bins = cast<int>(1.0f / r_sigma);
485 Expr val = gray_slice_loc(x, y);
486 val = clamp(val, 0.0f, 1.0f);
487 Expr zv = val * num_intensity_bins;
488 Expr zi = cast<int>(zv);
489 Expr zf = zv - zi;
490 slice_loc_z(x, y) = {zi, zf};
491
492 interpolated_matrix_z(x, y, c) =
493 lerp(interpolated_matrix_x(x, y, slice_loc_z(x, y)[0], c),
494 interpolated_matrix_x(x, y, slice_loc_z(x, y)[0] + 1, c),
495 slice_loc_z(x, y)[1]);
496
497 // Multiply by 3x4 by 4x1.
498 interpolated(x, y, c) =
499 (interpolated_matrix_z(x, y, 4 * c + 0) * slice_loc(x, y, 0) +
500 interpolated_matrix_z(x, y, 4 * c + 1) * slice_loc(x, y, 1) +
501 interpolated_matrix_z(x, y, 4 * c + 2) * slice_loc(x, y, 2) +
502 interpolated_matrix_z(x, y, 4 * c + 3));
503 }
504
505 // Normalize
506 Func slice("slice");
507 slice(x, y, c) = clamp(interpolated(x, y, c), 0.0f, 1.0f);
508
509 output = slice;
510
511 // Schedule
512 if (!auto_schedule) {
513 if (!get_target().has_gpu_feature()) {
514 // 7.09 ms on an Intel i9-9960X using 16 threads
515 //
516 // The original manual version of this schedule was
517 // more than 10x slower than the autoscheduled one
518 // from Adams et al. 2018. It's probable that because
519 // the slicing half of the algorithm is intended to be
520 // run on GPU using a shader, the CPU schedule was
521 // never seriously optimized.
522 //
523 // This new CPU schedule takes inspiration from that
524 // solution. In particular, certain stages were poorly or
525 // not parallelized, and others had unhelpful storage
526 // layout reorderings.
527
528 // AVX512 vector widths seem to hurt performance, here so
529 // we stick to 8 instead of using natural_vector_size.
530 const int vec = 8;
531
532 // Fitting the curves.
533
534 // Compute the per tile histograms and splatting locations within
535 // rows of the blur in z.
536 histogram
537 .compute_at(blurz, y);
538 histogram.update()
539 .reorder(c, r.x, r.y, x, y)
540 .unroll(c);
541
542 gray_splat_loc
543 .compute_at(blurz, y)
544 .vectorize(x, vec);
545
546 // Compute the blur in z at root
547 blurz
548 .compute_root()
549 .reorder(c, z, x, y)
550 .parallel(y)
551 .vectorize(x, vec);
552
553 // The blurs of the Gram matrices across x and y will be computed
554 // within the outer loops of the matrix solve.
555 blury
556 .compute_at(line, z)
557 .store_in(MemoryType::Stack)
558 .vectorize(x, vec);
559
560 blurx
561 .compute_at(line, x)
562 .vectorize(x);
563
564 line
565 .compute_root()
566 .parallel(y)
567 .vectorize(x, vec)
568 .reorder(c, x, y)
569 .unroll(c);
570
571 // Applying the curves.
572 interpolated_matrix_y
573 .compute_root()
574 .parallel(c)
575 .parallel(z)
576 .vectorize(x, vec);
577
578 interpolated_matrix_z
579 .store_at(slice, x)
580 .compute_at(slice, c)
581 .reorder(c, x, y)
582 .unroll(c)
583 .vectorize(x);
584
585 slice
586 .compute_root()
587 .parallel(y)
588 .vectorize(x, vec)
589 .reorder(c, x, y)
590 .bound(c, 0, 3)
591 .unroll(c);
592
593 } else {
594 // 0.92ms on a 2060 RTX
595
596 // This app is somewhat sensitive to atomic adds
597 // getting lowered correctly, so if runtime is
598 // mysteriously slow, check the ptx to see if there
599 // are atomic adds vs cas loops. If it's the latter,
600 // please file a bug.
601
602 Var xi, yi, zi, xo, yo, t;
603 histogram
604 .compute_root()
605 .reorder(c, x, y, z)
606 .unroll(c)
607 .tile(x, y, xo, yo, xi, yi, 16, 8)
608 .gpu_threads(xi, yi)
609 .gpu_blocks(xo, yo, z);
610
611 if (get_target().has_feature(Target::CUDA)) {
612 // The CUDA backend supports atomics. Useful for
613 // histograms.
614 histogram
615 .update()
616 .atomic()
617 .split(x, xo, xi, 16)
618 .reorder(c, r.x, xi, r.y, xo, y)
619 .unroll(c)
620 .gpu_blocks(r.y, xo, y)
621 .gpu_threads(xi);
622 } else {
623 histogram
624 .update()
625 .split(x, xo, xi, 16)
