3namespace {
4
5using namespace Halide;
6
7class ConvolutionLayer : public Halide::Generator<ConvolutionLayer> {
8public:
9 Input<Buffer<float>> input{"input", 4};
10 Input<Buffer<float>> filter{"filter", 4};
11 Input<Buffer<float>> bias{"bias", 1};
12
13 Output<Buffer<float>> relu{"relu", 4};
14
15 void generate() {
16 const int N = 5, CI = 128, CO = 128, W = 100, H = 80;
17
18 /* THE ALGORITHM */
19
20 Var x("x"), y("y"), c("c"), n("n");
21
22 Func conv("conv");
23 RDom r(0, CI, 0, 3, 0, 3);
24
25 conv(c, x, y, n) = bias(c);
26 conv(c, x, y, n) += filter(c, r.y, r.z, r.x) * input(r.x, x + r.y, y + r.z, n);
27
28 relu(c, x, y, n) = max(0, conv(c, x, y, n));
29
30 /* THE SCHEDULE */
31
32 // MKL JITs code for the specific size and strides, so we'll
33 // do the same and ask Halide to compile for this specific
34 // size:
35
36 relu.dim(0).set_bounds(0, CO).set_stride(1);
37 relu.dim(1).set_bounds(0, W).set_stride(CO);
38 relu.dim(2).set_bounds(0, H).set_stride(CO * W);
39 relu.dim(3).set_bounds(0, N).set_stride(CO * H * W);
40
41 input.dim(0).set_bounds(0, CI).set_stride(1);
42 input.dim(1).set_bounds(0, W + 2).set_stride(CI);
43 input.dim(2).set_bounds(0, H + 2).set_stride(CI * (W + 2));
44 input.dim(3).set_bounds(0, N).set_stride(CI * (W + 2) * (H + 2));
45
46 filter.dim(0).set_bounds(0, CO).set_stride(1);
47 filter.dim(1).set_bounds(0, 3).set_stride(CO);
48 filter.dim(2).set_bounds(0, 3).set_stride(CO * 3);
49 filter.dim(3).set_bounds(0, CI).set_stride(CO * 3 * 3);
50
51 bias.dim(0).set_bounds(0, CO).set_stride(1);
52
53 if (auto_schedule) {
54 input.dim(0).set_estimate(0, CI);
55 input.dim(1).set_estimate(0, W + 2);
56 input.dim(2).set_estimate(0, H + 2);
57 input.dim(3).set_estimate(0, N);
58
59 filter.dim(0).set_estimate(0, CO);
60 filter.dim(1).set_estimate(0, 3);
61 filter.dim(2).set_estimate(0, 3);
62 filter.dim(3).set_estimate(0, CI);
63
64 bias.dim(0).set_estimate(0, CO);
65
66 relu.dim(0).set_estimate(0, W);
67 relu.dim(1).set_estimate(0, H);
68 relu.dim(2).set_estimate(0, CO);
69 relu.dim(3).set_estimate(0, N);
70
71 } else if (get_target().has_feature(Target::CUDA)) {
72 // GPU schedule, tuned for a GTX 980. Seems to be good on
73 // an RTX 2060 too (About 90% peak flops on both cards).
74
75 // 1.87 ms on an RTX 2060. According to NVIDIA Nsight
76 // Compute we're at 91.5% utilization of the FMA units
77
78 // 2.41 ms on a GTX 980. According to nvprof this is about
79 // 88% of peak flops.
80
81 // We use cuda-specific scheduling directives (gpu_lanes),
82 // so this is not a general GPGPU schedule.
83
84 Var ni, no, xi, xo, yi, yo, ci, co, t;
85 RVar rxo, rxi, rxii;
86 relu.compute_root()
87 .split(x, xo, xi, 5)
88 .split(y, yo, yi, 5)
89 .split(c, co, ci, 32)
90 .reorder(xi, yi, ci, xo, yo, co, n)
91 .gpu_lanes(ci)
92 .unroll(xi)
93 .unroll(yi)
94 .fuse(co, n, t)
95 .gpu_blocks(xo, yo, t);
96
97 conv.compute_at(relu, xo)
98 .store_in(MemoryType::Register)
99 .gpu_lanes(c)
100 .unroll(x)
101 .unroll(y)
102 .update()
103 .split(r.x, rxo, rxi, 16)
104 .split(rxi, rxi, rxii, 2)
105 .reorder(c, rxii, x, y, r.y, r.z, rxi, rxo)
106 .gpu_lanes(c)
107 .unroll(x)
108 .unroll(y)
109 .unroll(r.y)
110 .unroll(r.z)
111 .unroll(rxii);
112
113 input.in()
114 .compute_at(conv, rxo)
115 .vectorize(_0, 2)
116 .split(_1, xo, xi, 4)
117 .fuse(_0, xi, t)
118 .gpu_lanes(t)
119 .unroll(xo)
120 .unroll(_2);
121
122 } else {
123
124 // 4.06ms on an Intel i9-9960X using 16 threads at 3.0 GHz,
125 // which is 94.5% of peak flops assuming the math below is correct:
126
127 // 16 cores times 2 FMAs per cycle times 3G cycles per
128 // second times 16 vector lanes is a peak throughput of
129 // 1.536 TFlops.
130
131 // This conv does N * CI * CO * W * H * 3 * 3 = 5 * 128 *
132 // 128 * 100 * 80 * 3 * 3 FMAs in 4.06ms is 1.453 TFlops.
133
134 // The ratio of actual to theoretical flops hit is 0.9458
135
136 int tile_w = 1;
137 int tile_h = 1;
138 const int vec = natural_vector_size<float>();
139
140 if (get_target().has_feature(Target::AVX512_Skylake) ||
141 (get_target().arch == Target::ARM &&
142 get_target().bits == 64)) {
143 // On Skylake we have one load per fma and 32
144 // registers available, so there's considerable
145 // flexibility in the schedule. We'll use 20 accumulator
146 // registers in a 4x5 tile. This is also a reasonable
147 // choice for ARMv8, which also has 32 registers.
148 tile_w = 4;
149 tile_h = 5;
150 } else if (get_target().arch == Target::X86) {
151 // With 16-register ISAs like x86 with AVX2, we can
152 // only do one load per two fmas, which constrains the
153 // schedule to have to be a squarish 12-register tile
154 // of the output.
155 tile_w = 3;
156 tile_h = 4;
157 } else {
158 // The above should also be reasonable schedule for
159 // ARMv7 and other 16-register machines, but I see
160 // some spills on arm-32, so we use a 2x4 block of 8
161 // accumulators instead. This could probably be better
162 // tuned, because in principle 12 accumulators should
163 // be possible. I believe the issue is that there's no
164 // fused multiply-add instruction, and so we're
165 // fighting llvm's instruction scheduler, which wants
166 // to move the muls well ahead of the adds to cover
167 // instruction latencies.
168 tile_w = 2;
169 tile_h = 4;
170 }
171
172 Var co, ci, xo, xi, yo, yi, t;
173 relu.split(c, co, ci, vec * tile_w)
174 .split(x, xo, xi, tile_h)
175 .reorder(ci, xi, xo, y, n, co)
176 .vectorize(ci, vec)
177 .unroll(ci)
178 .unroll(xi)
179 .parallel(y)
180 .parallel(n)
181 .parallel(co);
182 conv.compute_at(relu, xo)
183 .vectorize(c, vec)
184 .unroll(c)
185 .unroll(x)
186 .unroll(y)
187 .update()
188 .reorder(c, x, y, r.x, r.y, r.z, n)
189 .vectorize(c, vec)
190 .unroll(c)
191 .unroll(x)
192 .unroll(y)
193 .unroll(r.x, 2);
194 filter.in()
195 .compute_at(conv, r.x)
196 .vectorize(_0, vec)
197 .unroll(_0)
198 .unroll(_3);
199 input.in()
200 .compute_at(conv, x)
201 .unroll(_0);
202 }
203 }
204};
