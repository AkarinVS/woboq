5class StencilChain : public Halide::Generator<StencilChain> {
6public:
7 GeneratorParam<int> stencils{"stencils", 32, 1, 100};
8
9 Input<Buffer<uint16_t>> input{"input", 2};
10 Output<Buffer<uint16_t>> output{"output", 2};
11
12 void generate() {
13
14 std::vector<Func> stages;
15
16 Var x("x"), y("y");
17
18 Func f = Halide::BoundaryConditions::repeat_edge(input);
19
20 stages.push_back(f);
21
22 for (int s = 0; s < (int)stencils; s++) {
23 Func f("stage_" + std::to_string(s));
24 Expr e = cast<uint16_t>(0);
25 for (int i = -2; i <= 2; i++) {
26 for (int j = -2; j <= 2; j++) {
27 e += ((i + 3) * (j + 3)) * stages.back()(x + i, y + j);
28 }
29 }
30 f(x, y) = e;
31 stages.push_back(f);
32 }
33
34 output(x, y) = stages.back()(x, y);
35
36 /* ESTIMATES */
37 // (This can be useful in conjunction with RunGen and benchmarks as well
38 // as auto-schedule, so we do it in all cases.)
39 {
40 const int width = 1536;
41 const int height = 2560;
42 // Provide estimates on the input image
43 input.set_estimates({{0, width}, {0, height}});
44 // Provide estimates on the pipeline output
45 output.set_estimates({{0, width}, {0, height}});
46 }
47
48 if (auto_schedule) {
49 // nothing
50 } else if (get_target().has_gpu_feature()) {
51 // GPU schedule
52
53 // 2.9 ms on a 2060 RTX
54
55 // It seems that just compute-rooting all the stencils is
56 // fastest on this GPU, plus some unrolling and aggressive
57 // staging to share loads between adjacent pixels.
58 Var xi, yi, xii, yii;
59 stages.pop_back(); // Inline the second-last stage into the output
60 stages.push_back(output);
61 for (size_t i = 1; i < stages.size(); i++) {
62 Func &s = stages[i];
63 Func prev = stages[i - 1];
64 x = s.args()[0];
65 y = s.args()[1];
66 s.compute_root()
67 .gpu_tile(x, y, xi, yi, 30 * 2, 12)
68 .tile(xi, yi, xii, yii, 2, 2)
69 .unroll(xii)
70 .unroll(yii);
71
72 // Pre-load the entire region required of the previous
73 // stage into shared memory by adding a wrapper Func
74 // and scheduling it at blocks. This way instead of
75 // every pixel doing 25 loads from global memory, many of
76 // which overlap, we load each unique value from
77 // global into shared once, and then we use faster
78 // loads from shared in the actual stencil.
79 prev.in()
80 .compute_at(s, x)
81 .tile(prev.args()[0], prev.args()[1], xi, yi, 2, 2)
82 .vectorize(xi)
83 .unroll(yi)
84 .gpu_threads(prev.args()[0], prev.args()[1]);
85
86 // A similar benefit applies for the
87 // vectorized/unrolled 2x2 tiles. Instead of having
88 // each unrolled iteration do its own mix of scalar
89 // and vector loads from shared memory in a 5x5
90 // window, many of which get deduped across the block,
91 // we load a 6x6 window of shared into registers using
92 // only aligned vector loads, and then the actual
93 // stencil pulls from those registers. We're adding
94 // another wrapper Func around the wrapper Func we
95 // created above, so we say .in().in()
96 prev.in()
97 .in()
98 .compute_at(s, xi)
99 .vectorize(prev.args()[0], 2)
100 .unroll(prev.args()[0])
101 .unroll(prev.args()[1]);
102 }
103
104 } else {
105 // CPU schedule
106 // 4.23ms on an Intel i9-9960X using 16 threads at 3.5
107 // GHz.
108
109 // Runtime is pretty noisy, so benchmarked over 1000
110 // trials instead of the default of 10 in the
111 // Makefile. This uses AVX-512 instructions, but not
112 // floating-point ones. My CPU seems to hover at 3.5GHz on
113 // this workload.
114
115 const int vec = natural_vector_size<uint16_t>();
116
117 // How many stencils in between each compute-root
118 const int group_size = 11;
119 Var yi, yo, xo, xi, t;
120
121 const int last_stage_idx = (int)stages.size() - 1;
122 for (int j = last_stage_idx; j > 0; j -= group_size) {
123 Func out = (j == last_stage_idx) ? output : stages[j];
124
125 const int stages_to_output = last_stage_idx - j;
126 const int expansion = 4 * stages_to_output;
127 const int w = 1536 + expansion;
128 const int h = 2560 + expansion;
129
130 out.compute_root()
131 // Break into 16 tiles for our 16 threads
132 .tile(x, y, xo, yo, xi, yi, w / 4, h / 4)
133 .fuse(xo, yo, t)
134 .parallel(t)
135 .vectorize(xi, vec);
136
137 for (int i = std::max(0, j - group_size + 1); i < j; i++) {
138 Func s = stages[i];
139 s.store_at(out, t)
140 .compute_at(out, yi)
141 .vectorize(s.args()[0], vec);
142 }
143 }
144 }
145 }
146};
