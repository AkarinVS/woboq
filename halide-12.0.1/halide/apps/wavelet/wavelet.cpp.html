26template<typename T>
27T clamp(T x, T min, T max) {
28 if (x < min) return min;
29 if (x > max) return max;
30 return x;
31}

39template<typename T>
40void save_transformed(Buffer<T> t, const std::string &filename) {
41 Buffer<T> rearranged(t.width() * 2, t.height(), 1);
42 for (int y = 0; y < t.height(); y++) {
43 for (int x = 0; x < t.width(); x++) {
44 rearranged(x, y, 0) = clamp(t(x, y, 0), 0.0f, 1.0f);
45 rearranged(x + t.width(), y, 0) = clamp(t(x, y, 1) * 4.f + 0.5f, 0.0f, 1.0f);
46 }
47 }
48 convert_and_save_image(rearranged, filename);
49 printf("Saved %s\n", filename.c_str());
50}
