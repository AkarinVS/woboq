7std::ostream &operator<<(std::ostream &stream, const halide_type_t &type) {
8 if (type.code == halide_type_uint && type.bits == 1) {
9 stream << "bool";
10 } else {
11 static const char *const names[5] = {"int", "uint", "float", "handle", "bfloat"};
12 assert(type.code >= 0 && type.code < size(names));
13 stream << names[type.code] << (int)type.bits;
14 }
15 if (type.lanes > 1) {
16 stream << "x" << (int)type.lanes;
17 }
18 return stream;
19}
