20inline constexpr int halide_type_code(halide_type_code_t code, int bits) {
21 return (((int)code) << 8) | bits;
22}

24// dynamic_type_dispatch is a utility for functors that want to be able
25// to dynamically dispatch a halide_type_t to type-specialized code.
26// To use it, a functor must be a *templated* class, e.g.
27//
28// template<typename T> class MyFunctor { int operator()(arg1, arg2...); };
29//
30// dynamic_type_dispatch() is called with a halide_type_t as the first argument,
31// followed by the arguments to the Functor's operator():
32//
33// auto result = dynamic_type_dispatch<MyFunctor>(some_halide_type, arg1, arg2);
34//
35// Note that this means that the functor must be able to instantiate its
36// operator() for all the Halide scalar types; it also means that all those
37// variants *will* be instantiated (increasing code size), so this approach
38// should only be used when strictly necessary.
39template<template<typename> class Functor, typename... Args>
40auto dynamic_type_dispatch(const halide_type_t &type, Args &&...args)
41 -> decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...)) {
42
43#define HANDLE_CASE(CODE, BITS, TYPE) \
44 case halide_type_code(CODE, BITS): \
45 return Functor<TYPE>()(std::forward<Args>(args)...);
46 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
47 // HANDLE_CASE(halide_type_float, 16, float) // TODO
48 HANDLE_CASE(halide_type_float, 32, float)
49 HANDLE_CASE(halide_type_float, 64, double)
50 HANDLE_CASE(halide_type_int, 8, int8_t)
51 HANDLE_CASE(halide_type_int, 16, int16_t)
52 HANDLE_CASE(halide_type_int, 32, int32_t)
53 HANDLE_CASE(halide_type_int, 64, int64_t)
54 HANDLE_CASE(halide_type_uint, 1, bool)
55 HANDLE_CASE(halide_type_uint, 8, uint8_t)
56 HANDLE_CASE(halide_type_uint, 16, uint16_t)
57 HANDLE_CASE(halide_type_uint, 32, uint32_t)
58 HANDLE_CASE(halide_type_uint, 64, uint64_t)
59 // Omitted because we don't expect to see this here and adding would
60 // require handling pointer types in our functors
61 // HANDLE_CASE(halide_type_handle, 64, void *)
62 default:
63 HCHECK(0) << "Unsupported type";
64 using ReturnType = decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...));
65 return ReturnType();
66 }
67#undef HANDLE_CASE
68}

70inline void check_shapes_match(const HalideBuffer<const void> &a,
71 const HalideBuffer<const void> &b) {
72 HCHECK(a.dimensions() == b.dimensions());
73 for (int d = 0; d < a.dimensions(); d++) {
74 HCHECK(a.dim(d).min() == b.dim(d).min());
75 HCHECK(a.dim(d).extent() == b.dim(d).extent());
76 }
77}

79struct CompareBuffersOptions {
80 // Threshold at which values are an 'exact' match.
81 // For integral types this should always be 0.0.
82 // For FP types it should be a small epsilon.
83 double exact_thresh = 0.0;
84 // Threshold at which values are 'close enough' to be considered ok
85 // some part of the time.
86 // For integral types this should always be 1.0.
87 // For FP types it should be an epsilon.
88 double close_thresh = 1.0;
89 // What percentage (0..1) of elements in the result can be off by
90 // more than exact_thresh (but <= close_thresh) and still have the
91 // result be considered correct.
92 double max_close_percent = 0.001; // 0.1% by default. TODO: tweak as needed
93 // If true, log info about failures to stderr.
94 // If false, log nothing, stay silent.
95 bool verbose = true;
96 // Max number of diffs to log
97 uint64_t max_diffs_to_log = 32; // somewhat arbitrary
98
99 void require_exact() {
100 exact_thresh = 0.0;
101 close_thresh = 0.0;
102 max_close_percent = 0.0;
103 }
104};

106struct CompareBuffersResult {
107 uint64_t num_close = 0;
108 uint64_t num_wrong = 0;
109 bool ok = true;
110};

112// Functor for use with dynamic_type_dispatch() to compare two buffers.
113// Requires that the buffers have the same type and same shape (ignoring strides);
114// type/shape mismatch will check-fail immediately.
115template<typename T>
116struct CompareBuffers {
117 CompareBuffersResult operator()(const HalideBuffer<const void> &expected_buf_dynamic,
118 const HalideBuffer<const void> &actual_buf_dynamic,
119 const CompareBuffersOptions &opts) {
120 HalideBuffer<const T> expected_buf = expected_buf_dynamic;
121 HalideBuffer<const T> actual_buf = actual_buf_dynamic;
122 check_shapes_match(expected_buf, actual_buf);
123
124 assert(opts.exact_thresh >= 0.0);
125 assert(opts.close_thresh >= opts.exact_thresh);
126 assert(opts.max_close_percent >= 0.0 && opts.max_close_percent <= 1.0);
127 const T exact_thresh = (T)opts.exact_thresh;
128 const T close_thresh = (T)opts.close_thresh;
129
130 const uint64_t max_close = std::ceil(expected_buf.number_of_elements() * opts.max_close_percent);
131
132 const auto do_compare = [&](bool verbose) -> CompareBuffersResult {
133 CompareBuffersResult r;
134 expected_buf.for_each_element([&](const int *pos) {
135 T expected = expected_buf(pos);
136 T actual = actual_buf(pos);
137 T diff = (expected > actual) ? (expected - actual) : (actual - expected);
138 if (diff > close_thresh) {
139 bool do_log = verbose;
140 const char *msg;
141 if (diff > exact_thresh) {
142 r.num_wrong++;
143 do_log &= (r.num_wrong < opts.max_diffs_to_log);
144 msg = "WRONG";
145 } else {
146 r.num_close++;
147 do_log &= (r.num_close < opts.max_diffs_to_log);
148 msg = "Inexact";
149 }
150 if (do_log) {
151 std::cerr << "*** " << msg << " at (";
152 for (int i = 0; i < expected_buf.dimensions(); ++i) {
153 if (i > 0) std::cerr << ", ";
154 std::cerr << pos[i];
155 }
156 std::cerr << "): expected " << 0 + expected << " actual " << 0 + actual << " diff " << 0 + diff << "\n";
157 }
158 }
159 });
160 return r;
161 };
162
163 CompareBuffersResult r = do_compare(false);
164 if (r.num_wrong > 0 || r.num_close > max_close) {
165 r.ok = false;
166 }
167 if (opts.verbose) {
168 if (!r.ok) {
169 // Run again just to log the diffs
170 std::cerr << "*** TOO MANY WRONG/INEXACT ELEMENTS (wrong " << r.num_wrong
171 << ", close " << r.num_close << " vs " << max_close << "):\n";
172 (void)do_compare(true);
173 }
174 if (r.num_wrong > opts.max_diffs_to_log) {
175 std::cerr << "(" << (r.num_wrong - opts.max_diffs_to_log) << " wrong values omitted)\n";
176 }
177 if (r.num_close > opts.max_diffs_to_log) {
178 std::cerr << "(" << (r.num_close - opts.max_diffs_to_log) << " inexact values omitted)\n";
179 }
180 }
181 return r;
182 }
183};

185// Functor for use with dynamic_type_dispatch() to fill a buffer
186// with pseudorandom data.
187template<typename T>
188struct FillWithRandom {
189 void operator()(HalideBuffer<void> &b_dynamic, int seed) {
190 HalideBuffer<T> b = b_dynamic;
191 std::mt19937 rng(seed);
192 fill_with_random_impl(b, rng);
193 }
194
195private:
196 inline static void fill_with_random_impl(HalideBuffer<T> &b, std::mt19937 &rng) {
197 std::uniform_int_distribution<T> dis(std::numeric_limits<T>::min(),
198 std::numeric_limits<T>::max());
199 b.for_each_value([&rng, &dis](T &value) {
200 value = dis(rng);
201 });
202 }
203};
204
205// Specializations must be at namespace scope, not class scope
206template<>
207inline /*static*/ void FillWithRandom<float>::fill_with_random_impl(HalideBuffer<float> &b, std::mt19937 &rng) {
208 // Floating point. We arbitrarily choose to use the range [0.0, 1.0].
209 std::uniform_real_distribution<float> dis(0.0, 1.0);
210 b.for_each_value([&rng, &dis](float &value) {
211 value = dis(rng);
212 });
213}
214
215template<>
216inline /*static*/ void FillWithRandom<double>::fill_with_random_impl(HalideBuffer<double> &b, std::mt19937 &rng) {
217 // Floating point. We arbitrarily choose to use the range [0.0, 1.0].
218 std::uniform_real_distribution<double> dis(0.0, 1.0);
219 b.for_each_value([&rng, &dis](double &value) {
220 value = dis(rng);
221 });
222}
223
224template<>
225inline /*static*/ void FillWithRandom<bool>::fill_with_random_impl(HalideBuffer<bool> &b, std::mt19937 &rng) {
226 std::uniform_int_distribution<int> dis(0, 1);
227 b.for_each_value([&rng, &dis](bool &value) {
228 value = static_cast<bool>(dis(rng));
229 });
230}

232// Functor for use with dynamic_type_dispatch() to dump a buffer's contents
233// to std::cerr in a very simple way. Intended only for temporary debugging.
234template<typename T>
235struct DumpBuffer {
236 void operator()(const HalideBuffer<const void> &buf_dynamic) {
237 HalideBuffer<const T> buf = buf_dynamic;
238 buf.for_each_element([&](const int *pos) {
239 T val = buf(pos);
240 std::cerr << "Value at (";
241 for (int i = 0; i < buf.dimensions(); ++i) {
242 if (i > 0) std::cerr << ", ";
243 std::cerr << pos[i];
244 }
245 std::cerr << "): " << 0 + val << "\n";
246 });
247 }
248};
