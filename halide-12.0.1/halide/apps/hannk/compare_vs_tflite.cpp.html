10using FlagFn = std::function<int(const std::string &value)>;
11using FlagFnMap = std::map<std::string, FlagFn>;

13int process_args(int argc, char **argv, const FlagFnMap &m, FlagFn nonflags) {
14 for (int i = 1; i < argc; i++) {
15 std::string flag = argv[i];
16 if (flag[0] != '-') {
17 nonflags(flag);
18 continue;
19 }
20 flag = flag.substr(1);
21 if (flag[0] == '-') {
22 flag = flag.substr(1);
23 }
24
25 std::string value;
26 auto eq = flag.find('=');
27 if (eq != std::string::npos) {
28 value = flag.substr(eq + 1);
29 flag = flag.substr(0, eq);
30 } else if (i + 1 < argc) {
31 value = argv[++i];
32 } else {
33 std::cerr << "Missing value for flag '" << flag << "'\n";
34 return -1;
35 }
36 auto it = m.find(flag);
37 if (it == m.end()) {
38 std::cerr << "Unknown flag '" << flag << "'\n";
39 return -1;
40 }
41 int r = it->second(value);
42 if (r != 0) {
43 return r;
44 }
45 }
46 return 0;
47}

102 }},
