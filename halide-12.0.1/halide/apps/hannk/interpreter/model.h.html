18struct QuantizationInfo {
19 std::vector<float> scale;
20 std::vector<int32_t> zero;
21 int32_t dimension = -1;
22
23 bool operator==(const QuantizationInfo &r) const {
24 return dimension == r.dimension && scale == r.scale && zero == r.zero;
25 }
26
27 float uniform_scale() const {
28 assert(scale.size() == 1);
29 return scale[0];
30 }
31
32 int32_t uniform_zero() const {
33 assert(zero.size() == 1);
34 return zero[0];
35 }
36};
37
38inline std::ostream &operator<<(std::ostream &s, const QuantizationInfo &q) {
39 return s << "{" << q.scale << ", " << q.zero << ", " << q.dimension << "}";
40}

238// A mapping from an output x to required input coordinates [min, max].
239// [min, max] = (x / inv_stride) * stride + bounds
240struct DimMap {
241 Interval pre_bounds;
242 int stride;
243 int inv_stride;
244 Interval bounds;
245
246 DimMap()
247 : pre_bounds(0, 0), stride(0), inv_stride(1), bounds(0, -1) {
248 }
249 DimMap(int stride, int inv_stride, const Interval &bounds)
250 : pre_bounds(0, 0), stride(stride), inv_stride(inv_stride), bounds(bounds) {
251 }
252
253 Interval evaluate(Interval x) const {
254 Interval result = x;
255 result += pre_bounds;
256 result /= inv_stride;
257 result *= stride;
258 result += bounds;
259 return result;
260 }
261 Interval evaluate(int at) const {
262 return evaluate(Interval(at));
263 }
264
265 bool is_elementwise() const {
266 return stride == 1 && inv_stride == 1 && bounds.extent() == 1;
267 }
268
269 bool is_upsample() const {
270 return stride == 1 && inv_stride > 1;
271 }
272
273 bool is_downsample() const {
274 return stride > 1 && inv_stride == 1;
275 }
276
277 bool is_constant() const {
278 return stride == 0;
279 }
280
281 // A dependency where the input `bounds` do not depend on the output.
282 DimMap &constant(const Interval &bounds) {
283 stride = 0;
284 inv_stride = 1;
285 this->bounds = bounds;
286 return *this;
287 }
288
289 DimMap &constant(int extent) {
290 return constant(Interval(0, extent - 1));
291 }
292
293 DimMap &downsample(int factor, const Interval &filter) {
294 stride = factor;
295 inv_stride = 1;
296 bounds = filter;
297 return *this;
298 }
299
300 DimMap &upsample(int factor, const Interval &filter) {
301 stride = 1;
302 inv_stride = factor;
303 bounds = filter;
304 return *this;
305 }
306
307 DimMap &downsample(int factor) {
308 return downsample(factor, Interval(0, 0));
309 }
310
311 DimMap &upsample(int factor) {
312 return upsample(factor, Interval(0, 0));
313 }
314
315 DimMap &elementwise(int offset = 0) {
316 return upsample(1);
317 }
318
319 DimMap &stencil(const Interval &filter) {
320 return upsample(1, filter);
321 }
322
323 DimMap &align(int alignment) {
324 pre_bounds.max += alignment - 1;
325 stride *= alignment;
326 inv_stride *= alignment;
327 bounds.min = align_down(bounds.min, alignment);
328 bounds.max = align_up(bounds.max + 1, alignment) - 1;
329 return *this;
330 }

333class BoundsMap {
334 int dims_in_;
335 int dims_out_;
336 DimMap data_[max_rank * (max_rank + 1)];
337
338public:
339 BoundsMap(int dims_in, int dims_out)
340 : dims_in_(dims_in), dims_out_(dims_out) {
341 assert(dims_in <= max_rank);
342 assert(dims_out <= max_rank);
343 }
344
345 DimMap &at(int dim_in, int dim_out) {
346 return data_[dim_in * (max_rank + 1) + dim_out];
347 }
348 const DimMap &at(int dim_in, int dim_out) const {
349 return data_[dim_in * (max_rank + 1) + dim_out];
350 }
351
352 DimMap &at(int dim_in) {
353 DimMap &result = at(dim_in, dims_out_);
354 assert(result.stride == 0);
355 return result;
356 }
357 const DimMap &at(int dim_in) const {
358 const DimMap &result = at(dim_in, dims_out_);
359 assert(result.stride == 0);
360 return result;
361 }
362
363 Interval evaluate(int dim_in, const Box &output) const {
364 Interval result = at(dim_in).bounds;
365 for (int i = 0; i < (int)output.size(); i++) {
366 result = Union(result, at(dim_in, i).evaluate(output[i]));
367 }
368 return result;
369 }
370
371 Box evaluate(const Box &output) const {
372 Box input(dims_in_);
373 for (int i = 0; i < dims_in_; i++) {
374 input[i] = evaluate(i, output);
375 }
376 return input;
377 }
378
379 // Check if this bounds map is solely an elementwise mapping from dim_in to dim_out.
380 bool is_elementwise(int dim_in, int dim_out) const {
381 bool result = true;
382 for (int i = 0; i < dims_out_ + 1; i++) {
383 const DimMap &map = at(dim_in, i);
384 result = result && (i == dim_out ? map.is_elementwise() : map.is_constant());
385 }
386 return result;
387 }
388
389 bool is_constant(int dim_in, int dim_out) const {
390 bool result = true;
391 for (int i = 0; i < dims_out_ + 1; i++) {
392 const DimMap &map = at(dim_in, i);
393 result = result && (i == dim_out ? map.is_elementwise() : map.is_constant());
394 }
395 return result;
396 }
397
398 // Add bounds for an elementwise mapping of x of dim_in to y of dim_out,
399 // where x maps to y + offset.
400 BoundsMap &elementwise(int dim_in, int dim_out, int offset = 0) {
401 at(dim_in, dim_out).elementwise(offset);
402 return *this;
403 }
404
405 BoundsMap &stencil(int dim_in, int dim_out, const Interval &filter) {
406 at(dim_in, dim_out).stencil(filter);
407 return *this;
408 }
409
410 BoundsMap &upsample(int dim_in, int dim_out, int factor) {
411 at(dim_in, dim_out).upsample(factor);
412 return *this;
413 }
414
415 BoundsMap &upsample(int dim_in, int dim_out, int factor, const Interval &filter) {
416 at(dim_in, dim_out).upsample(factor, filter);
417 return *this;
418 }
419
420 BoundsMap &downsample(int dim_in, int dim_out, int factor) {
421 at(dim_in, dim_out).downsample(factor);
422 return *this;
423 }
424
425 BoundsMap &downsample(int dim_in, int dim_out, int factor, const Interval &filter) {
426 at(dim_in, dim_out).downsample(factor, filter);
427 return *this;
428 }
429
430 BoundsMap &constant(int dim_in, int extent) {
431 at(dim_in).constant(extent);
432 return *this;
433 }
434
435 BoundsMap &constant(int dim_in, const Interval &bounds) {
436 at(dim_in).constant(bounds);
437 return *this;
438 }
439
440 BoundsMap &align_input(int dim_in, int alignment) {
441 for (int i = 0; i < dims_out_ + 1; i++) {
442 at(dim_in, i).align(alignment);
443 }
444 return *this;
445 }
446
447 // Producing an element of the output requires the corresponding element of
448 // the input.
449 static BoundsMap elementwise(int rank) {
450 BoundsMap result(rank, rank);
451 for (int i = 0; i < rank; i++) {
452 result.elementwise(i, i);
453 }
454 return result;
455 }
456
457 // Producing any point of any output dimension requires all of the input.
458 static BoundsMap all(const Box &bounds_in, int dims_out) {
459 BoundsMap result(bounds_in.size(), dims_out);
460 for (int j = 0; j < (int)bounds_in.size(); j++) {
461 result.constant(j, bounds_in[j]);
462 }
463 return result;
464 }
