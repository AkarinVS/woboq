5bool is_subset_of(const Box &a, const Box &b) {
6 assert(a.size() == b.size());
7 for (int i = 0; i < (int)a.size(); i++) {

10 }

15bool is_union_exact(const Box &a, const Box &b) {
16 if (is_subset_of(a, b) || is_subset_of(b, a)) {
17 return true;
18 }
19 assert(a.size() == b.size());
20 int different_dims = 0;
21 int dim = -1;
22 for (int i = 0; i < (int)a.size(); i++) {
23 if (a[i] != b[i]) {
24 different_dims++;
25 dim = i;
26 }
27 }
28 if (different_dims == 0) {
29 // The shapes are the same, the union is trivial.
30 return true;
31 } else if (different_dims == 1) {
32 // One dim is different. We might be able to produce an exact union.
33 return is_union_exact(a[dim], b[dim]);
34 } else {
35 // More than one dim is different, the union is not a rectangle.
36 return false;
37 }
38}

40Box Union(const Box &a, const Box &b) {
41 assert(a.size() == b.size());
42 Box result;
43 result.resize(a.size());
44 for (int i = 0; i < (int)a.size(); i++) {
45 result[i] = Union(a[i], b[i]);
46 }
