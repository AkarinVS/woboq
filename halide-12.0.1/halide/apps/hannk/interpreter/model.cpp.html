10const TensorPtr &apply(TensorMap &map, const TensorPtr &t) {
11 auto i = map.find(t);
12 if (i != map.end()) {
13 return i->second;
14 }
15
16 if (t->is_constant()) {
17 // Share constant tensors across users.
18 return t;
19 } else {
20 // Remember this cloned tensor for later applications of the mapping.
21 return map[t] = std::make_shared<Tensor>(*t);
22 }
23}

27HalideBuffer<void> make_buffer(halide_type_t type, const Box &bounds) {
28 // TODO: Avoid this dynamic allocation. Halide's API requires std::vector here.
29 SmallVector<halide_dimension_t, max_rank> dims(bounds.size());
30 int stride = 1;
31 for (int i = 0; i < (int)bounds.size(); i++) {
32 dims[i].min = bounds[i].min;
33 dims[i].extent = bounds[i].extent();
34 dims[i].stride = stride;
35 stride *= dims[i].extent;
36 }
37 return HalideBuffer<void>(type, nullptr, (int)dims.size(), dims.data());
38}

49void TensorStorage::add_use(halide_type_t type, const Box &bounds) {
50 if (buffer_.dimensions() == 0) {
51 buffer_ = make_buffer(type, bounds);
52 } else {
53 assert(buffer_.type() == type);
54 assert(buffer_.dimensions() == (int)bounds.size());
55 assert(!buffer_.data());
56
57 // Check that the storage is big enough for this buffer.
58 for (int i = 0; i < rank(); i++) {
59 assert(bounds[i].min >= buffer_.dim(i).min());
60 assert(bounds[i].max <= buffer_.dim(i).max());
61 }
62 }

130// Copy a Halide buffer without the internal reference counting.
131// This reduces overhead of buffer copies, and is unnecessary because
132// we do our own reference counting.
133template<typename T>
134HalideBuffer<T> drop_reference(const HalideBuffer<T> &buf) {
135 return HalideBuffer<T>(buf.type(), buf.data(), buf.dimensions(), buf.raw_buffer()->dim);
136}

140void Tensor::allocate() {
141 if (buffer_.data()) {
142 return;
143 }
144
145 if (is_dynamic()) {
146 return;
147 }
148
149 storage()->allocate();
150 HalideBuffer<void> buffer = drop_reference(storage()->buffer());
151 for (int i = 0; i < buffer.dimensions(); i++) {
152 Interval dim_i(buffer_.dim(i).min(), buffer_.dim(i).max());
153 if (i < (int)storage_offset_.size()) {
154 dim_i += storage_offset_[i];
155 }
156 assert(buffer.dim(i).min() <= dim_i.min);
157 assert(buffer.dim(i).max() >= dim_i.max);
158 buffer.crop(i, dim_i.min, dim_i.extent());
159 buffer.translate(i, -dim_i.min);
160 assert(buffer.dim(i).min() == buffer_.dim(i).min());
161 assert(buffer.dim(i).max() == buffer_.dim(i).max());
162 }
163 buffer_ = buffer;
164}

166void Tensor::resize(const Box &new_shape) {
167 assert(is_dynamic());
168
169 SmallVector<halide_dimension_t, max_rank> new_dims;
170
171 const halide_dimension_t *old_dims = buffer_.raw_buffer()->dim;
172
173 bool all_same = (buffer_.dimensions() == (int)new_shape.size());
174 // Resizing a dynamic tensor shouldn't (AFAICT) ever change the
175 // number of dimensions -- just the extents -- but let's guard
176 // against that just in case, because it's easy to do.
177 assert(all_same);
178
179 int stride = 1;
180 for (const auto &d : new_shape) {
181 const int d_min = d.min;
182 const int d_extent = d.extent();
183 if (all_same && (d_min != old_dims->min || d_extent != old_dims->extent)) {
184 all_same = false;
185 }
186 new_dims.emplace_back(d_min, d_extent, stride);
187 stride *= d_extent;
188 }
189 if (all_same) {
190 return;
191 }
192
193 HalideBuffer<void> new_buffer(buffer_.type(), nullptr, (int)new_dims.size(), new_dims.data());
194 new_buffer.allocate();
195 if (buffer_.data()) {
196 new_buffer.copy_from(buffer_);
197 }
198 buffer_ = std::move(new_buffer);
199 storage_ = nullptr;
200}

209 assert(!is_dynamic());

221void Tensor::replace_all_consumers_with(const TensorPtr &other) {
222 // We need to make a copy of the list of consumers so it doesn't get invalidated
223 // by set_input below.
224 auto consumers = consumers_;
225 for (Op *i : consumers) {
226 for (int j = 0; j < i->input_count(); j++) {
227 if (i->input(j).get() == this) {
228 i->set_input(j, other);
229 }
230 }
231 }

323void OpGroup::add(OpPtr to_add, const Op *before) {
324 for (auto i = ops_.begin(); i != ops_.end(); ++i) {
325 if (i->get() == before) {
326 ops_.insert(i, std::move(to_add));
327 return;
328 } else {
329 for (int j = 0; j < to_add->output_count(); ++j) {
330 for (int k = 0; k < (*i)->input_count(); ++k) {
331 if ((*i)->input(k) == to_add->output(j)) {
332 // i consumes an output of to_add.
333 ops_.insert(i, std::move(to_add));
334 return;
335 }
336 }
337 }
338 }
339 }
340 ops_.push_back(std::move(to_add));
341}

343void OpGroup::remove(const Op *op) {
344 for (auto i = ops_.begin(); i != ops_.end(); ++i) {
345 if (i->get() == op) {
346 ops_.erase(i);
347 return;
348 }
349 }

352OpPtr OpGroup::clone(TensorMap &tensor_map) const {
353 std::vector<TensorPtr> inputs;
354 for (int i = 0; i < input_count(); i++) {
355 inputs.push_back(apply(tensor_map, input(i)));
356 }
357 std::vector<TensorPtr> outputs;
358 for (int i = 0; i < output_count(); i++) {
359 outputs.push_back(apply(tensor_map, output(i)));
360 }
361
362 std::vector<OpPtr> ops;
363 for (int i = 0; i < op_count(); i++) {
364 ops.push_back(op(i)->clone(tensor_map));
365 }
366
367 return make_op<OpGroup>(std::move(inputs), std::move(outputs), std::move(ops));
368}
