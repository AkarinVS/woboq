20class AllocateAll : public OpVisitor {
21 void visit(OpGroup *g) {
22 for (int i = 0; i < g->op_count(); i++) {
23 Op *op = g->op(i);
24 for (int j = 0; j < op->input_count(); j++) {
25 op->input(j)->allocate();
26 }
27 for (int j = 0; j < op->output_count(); j++) {
28 op->output(j)->allocate();
29 }
30 op->accept(this);
31 }
32 }
33};

53TensorPtr Interpreter::get_tensor(const std::string &name) {
54 for (int i = 0; i < model_->op_count(); i++) {
55 Op *op = model_->op(i);
56 for (int j = 0; j < op->input_count(); j++) {
57 if (op->input(j)->name() == name) {
58 return op->input(j);
59 }
60 }
61 for (int j = 0; j < op->output_count(); j++) {
62 if (op->output(j)->name() == name) {
63 return op->output(j);
64 }
65 }
66 }
67 return nullptr;
68}
