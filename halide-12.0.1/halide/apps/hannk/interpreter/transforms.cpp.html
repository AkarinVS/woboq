7template<typename T>
8T *cast_op(Op *x) {
9 class Caster : public OpVisitor {
10 public:
11 T *result = nullptr;
12
13 void visit(T *op) {
14 result = op;
15 }
16 };
17
18 Caster caster;
19 x->accept(&caster);
20 if (caster.result == x) {
21 return caster.result;
22 } else {
23 return nullptr;
24 }
25}

29void remove_dead_ops(OpGroup *root) {
30 // Find ops with outputs that are unused.
31 // Go in reverse order so removing a dead op
32 // enables earlier ops to be seen as dead.
33 for (int i = root->op_count() - 1; i >= 0; --i) {
34 Op *op = root->op(i);
35 if (OpGroup *group = cast_op<OpGroup>(op)) {
36 remove_dead_ops(group);
37 }
38 bool dead = true;
39 for (int j = 0; dead && j < op->input_count(); j++) {
40 if (op->input(j)->is_output()) {
41 dead = false;
42 break;
43 }
44 }
45 for (int j = 0; dead && j < op->output_count(); j++) {
46 // An op isn't dead if its output is an output
47 // of the graph.
48 if (op->output(j)->is_output()) {
49 dead = false;
50 break;
51 }
52
53 if (!op->output(j)->consumers().empty()) {
54 dead = false;
55 break;
56 }
57 }
58
59 if (dead) {
60 root->remove(op);
61 }
62 }
63}

67// We can alias two tensors if the input is not used after the output is written,
68// and we meet a number of other requirements.
69bool maybe_alias_tensors(TensorPtr input, TensorPtr output, SmallVector<int, max_rank> offset = {}) {
70 // If the input is used anywhere else, we should not alias it.
71 // TODO: This is conservative, we could alias it if it is the *last* use.
72 if (input->consumers().size() != 1) {
73 return false;
74 }
75
76 // If either tensor is dynamic, can't alias them.
77 if (input->is_dynamic() || output->is_dynamic()) {
78 return false;
79 }
80
81 if (input->rank() != output->rank()) {
82 // TODO: We should be able to alias reshapes.
83 return false;
84 }
85
86 if (input->type().bytes() != output->type().bytes()) {
87 // We can't alias tensors with types of different size.
88 return false;
89 }
90
91 // We can't alias an input that is an input or output.
92 // TODO: We could, if we don't change the shape.
93 if (input->is_input() || input->is_output()) {
94 return false;
95 }
96
97 // We can't grow the bounds of the tensor we alias with.
98 // TODO: We could, if we allowed non-zero mins. We also
99 // could allow the max to grow, just not the min.
100 Box input_bounds_with_offset = input->bounds();
101 Box output_bounds_with_negative_offset = output->bounds();
102 for (int i = 0; i < (int)offset.size(); ++i) {
103 input_bounds_with_offset[i] += offset[i];
104 output_bounds_with_negative_offset[i] -= offset[i];
105 }
106 bool input_subset_of_output = is_subset_of(input_bounds_with_offset, output->bounds());
107 bool output_subset_of_input = is_subset_of(output_bounds_with_negative_offset, input->bounds());
108 if (input_subset_of_output && !input->is_alias()) {
109 input->set_alias_of(output, offset);
110 return true;
111 } else if (output_subset_of_input && !output->is_alias()) {
112 for (int &i : offset) {
113 i = -i;
114 }
115 output->set_alias_of(input, offset);
116 return true;
117 }
118
119 return true;
120}

122// Try to alias outputs to inputs when it is safe.
123class InPlace : public OpVisitor {
124 using OpVisitor::visit;
125
126 void maybe_alias_elementwise(ElementwiseOp *op) {
127 for (int j = 0; j < op->output_count(); j++) {
128 for (int i = 0; i < op->input_count(); i++) {
129 if (maybe_alias_tensors(op->input(i), op->output(j))) {
130 // We can only alias one of the input to each output.
131 break;
132 }
133 }
134 }
135 }
136
137 void visit(BinaryOp *op) {
138 maybe_alias_elementwise(op);
139 }
140
141 void visit(UnaryOp *op) {
142 maybe_alias_elementwise(op);
143 }
144
145 void visit(ElementwiseProgramOp *op) {
146 maybe_alias_elementwise(op);
147 }
148
149 void visit(ConcatenationOp *op) {
150 bool is_no_op = true;
151 SmallVector<int, max_rank> offset(op->axis() + 1);
152 for (int i = 0; i < op->input_count(); i++) {
153 is_no_op = is_no_op && maybe_alias_tensors(op->input(i), op->output(), offset);
154 is_no_op = is_no_op && op->input(i)->quantization() == op->output()->quantization();
155 offset[op->axis()] += op->input(i)->extent(op->axis());
156 }
157 if (is_no_op) {
158 // TODO: Try actually deleting the op?
159 op->set_no_op();
160 }
161 }
162
163 void visit(SplitOp *op) {
164 bool is_no_op = true;
165 SmallVector<int, max_rank> offset(op->axis() + 1);
166 for (int i = 0; i < op->output_count(); i++) {
167 is_no_op = is_no_op && maybe_alias_tensors(op->input(), op->output(i), offset);
168 is_no_op = is_no_op && op->output(i)->quantization() == op->input()->quantization();
169 offset[op->axis()] -= op->output(i)->extent(op->axis());
170 }
171 if (is_no_op) {
172 // TODO: Try actually deleting the op?
173 op->set_no_op();
174 }
175 }
176
177 void visit(PadOp *op) {
178 if (!op->input(1) || !op->input(1)->is_constant()) {
179 return;
180 }
181 assert(op->input(1)->is_allocated());
182
183 auto padding = op->input(1)->buffer<const int32_t>();
184
185 SmallVector<int, max_rank> offset(padding.extent(1));
186 for (int d = 0; d < padding.extent(1); d++) {
187 offset[d] = padding(0, d);
188 }
189
190 maybe_alias_tensors(op->input(), op->output(), offset);
191 }
192
193 void visit(ReshapeOp *op) {
194 maybe_alias_tensors(op->input(), op->output());
195 }
196
197 void visit(OpGroup *op) {
198 for (int i = 0; i < op->op_count(); i++) {
199 op->op(i)->accept(this);
200 }
201 }
202};

213// Find ops that need padding and add an explicit pad op.
214class PadForOps : public OpVisitor {
215 void pad_for_op(Op *op, int input_idx, int output_idx) {
216 TensorPtr input = op->input(input_idx);
217 TensorPtr output = op->output(output_idx);
218 BoundsMap deps = op->map_bounds(input_idx, output_idx);
219 Box required = deps.evaluate(output->bounds());
220
221 if (is_subset_of(required, input->bounds())) {
222 return;
223 }
224
225 // Make a PadOp and a new tensor for the padded result.
226 TensorPtr padded =
227 std::make_shared<Tensor>(input->name() + "_padded", input->type(), required, input->quantization());
228 op->set_input(padded);
229
230 HalideBuffer<int32_t> padding_data(2, input->rank());
231 // Center the crop, except for the channel dimension.
232 // TODO: Is this always correct?
233 const int r = input->rank();
234 padding_data(0, r - 1) = 0;
235 padding_data(1, r - 1) = 0;
236 for (int i = 1; i < input->rank(); i++) {
237 padding_data(0, r - i - 1) = (required[i].extent() - input->extent(i)) / 2;
238 padding_data(1, r - i - 1) = (required[i].extent() - input->extent(i) + 1) / 2;
239 }
240 TensorPtr padding = std::make_shared<Tensor>(input->name() + "_padding", padding_data);
241
242 // Add the new tensor, op, and update the input.
243 OpPtr pad = make_op<PadOp>(input, padding, padded);
244 new_ops.emplace_back(std::move(pad));
245 }
246
247 void visit(Conv2DOp *op) {
248 pad_for_op(op, 0, 0);
249
250 // We also need to tile the filter.
251 TensorPtr filter = op->filter();
252 if (op->filter()->rank() == 4) {
253 BoundsMap bounds = op->map_bounds(1, 0);
254 Box tiled_shape = bounds.evaluate(op->output()->bounds());
255
256 halide_type_t type = op->filter_type();
257 QuantizationInfo quantization = filter->quantization();
258 if (type.bits > filter->type().bits) {
259 // We're widening the filter. Subtract the offset.
260 std::fill(quantization.zero.begin(), quantization.zero.end(), 0);
261 }
262 TensorPtr tiled =
263 std::make_shared<Tensor>(filter->name() + "_tiled", type, tiled_shape, quantization);
264 // Maybe more than one op uses this same filter...?
265 filter->replace_all_consumers_with(tiled);
266
267 OpPtr tile = make_op<TileConvFilterOp>(filter, tiled);
268 new_ops.emplace_back(std::move(tile));
269 }
270 }
271
272 void visit(DepthwiseConv2DOp *op) {
273 pad_for_op(op, 0, 0);
274 }
275
276 void visit(OpGroup *op) {
277 for (int i = 0; i < op->op_count(); i++) {
278 op->op(i)->accept(this);
279 }
280 }
281
282public:
283 std::vector<OpPtr> new_ops;
284};

286class FusePadOps : public OpVisitor {
287 void visit(PadOp *op) {
288 if (op->input()->producers().size() != 1 || op->input()->consumers().size() != 1) {
289 return;
290 }
291 PadOp *prev_pad = cast_op<PadOp>(op->input()->producers().front());
292 if (!prev_pad) {
293 return;
294 }
295
296 op->set_input(prev_pad->input());
297
298 auto prev_padding = prev_pad->input(1)->buffer<const int32_t>();
299 auto padding = op->input(1)->buffer<int32_t>();
300
301 for (int d = 0; d < std::min(prev_padding.dimensions(), padding.dimensions()); d++) {
302 padding(0, d) += prev_padding(0, d);
303 padding(1, d) += prev_padding(1, d);
304 }
305 }
306
307 void visit(OpGroup *op) {
308 for (int i = 0; i < op->op_count(); i++) {
309 op->op(i)->accept(this);
310 }
311 }
312};

316void pad_for_ops(OpGroup *op) {
317 PadForOps padder;
318 op->accept(&padder);
319 for (auto &i : padder.new_ops) {
320 op->add(std::move(i));
321 }
322
323 // Some networks use padding already for other reasons, so
324 // we might have introduced two paddings in a row, which is
325 // a waste.
326 FusePadOps fuser;
327 op->accept(&fuser);
328}

332bool can_execute(const Op *op) {
333 for (int i = 0; i < op->input_count(); i++) {
334 if (!op->input(i)->is_constant()) {
335 return false;
336 }
337 assert(op->input(i)->is_allocated());
338 }
339 return true;
340}

344void fold_constants(OpGroup *root) {
345 std::vector<const Op *> to_remove;
346 for (int i = 0; i < root->op_count(); i++) {
347 Op *op = root->op(i);
348 if (OpGroup *group = cast_op<OpGroup>(op)) {
349 fold_constants(group);
350 }
351 if (can_execute(op)) {
352 // Allocate all the outputs.
353 for (int j = 0; j < op->output_count(); j++) {
354 op->output(j)->allocate();
355 }
356
357 // Run the whole op.
358 op->execute();
359
360 // Mark the outputs constant.
361 for (int j = 0; j < op->output_count(); j++) {
362 op->output(j)->set_constant();
363 }
364
365 to_remove.push_back(op);
366 }
367 }
368
369 for (const Op *i : to_remove) {
370 root->remove(i);
371 }
