9class DepthwiseConv : public Generator<DepthwiseConv> {
10public:
11 // If positive, a constant inverse depth multiplier.
12 GeneratorParam<int> inv_depth_multiplier_{"inv_depth_multiplier", -1};
13
14 // Unsigned 8-bit input tensor, indexed by ci, x, y, b.
15 Input<Buffer<uint8_t>> input_{"input", 4};
16 Input<uint8_t> input_zero_{"input_zero"};
17
18 // A 3D array of 8-bit filter coefficients indexed by co, x, y.
19 Input<Buffer<uint8_t>> filter_{"filter", 3};
20 Input<uint8_t> filter_zero_{"filter_zero"};
21
22 // A 1D array of 32-bit biases indexed by co.
23 Input<Buffer<int32_t>> bias_{"bias", 1};
24
25 // The depth multiplier specifies the ratio between co and ci.
26 Input<int> depth_multiplier_{"depth_multiplier"};
27
28 // The stride specifies how the input [x, y] are sub-subsampled. For every
29 // spatial location [x, y] in the output buffer, the input buffer is sampled
30 // spatially at [x * stride, y * stride]. The caller should ensure that
31 // [x * stride, y * stride] is a valid spatial location in the input buffer.
32 // Generally, this means setting the output buffer's [width, height] to be
33 // the input buffer's [width, height] / stride.
34 Input<int> stride_x_{"stride_x"};
35 Input<int> stride_y_{"stride_y"};
36 Input<int> dilation_x_{"dilation_x"};
37 Input<int> dilation_y_{"dilation_y"};
38
39 Input<int32_t> output_multiplier_{"output_multiplier"};
40 Input<uint32_t> output_shift_{"output_shift"};
41 Input<uint8_t> output_zero_{"output_zero"};
42 Input<uint8_t> output_min_{"output_min"};
43 Input<uint8_t> output_max_{"output_max"};
44
45 Output<Buffer<uint8_t>> output_{"output", 4};
46
47 void generate() {
48 // The algorithm.
49
50 // Some free variables, where x and y represent the spatial dimensions.
51 Var x("x"), y("y"), c("c"), b("b");
52
53 // Apply the c multiplier.
54 Func resampled_input("resampled_input");
55 Expr c_resampled = inv_depth_multiplier_ >= 0 ? c * inv_depth_multiplier_ : c / depth_multiplier_;
56 resampled_input(c, x, y, b) = input_(c_resampled, x, y, b);
57
58 Func filter_zeroed("filter_zeroed");
59 filter_zeroed(c, x, y) = i16(filter_(c, x, y)) - i16(filter_zero_);
60
61 // Do the convolution in 32-bit.
62 filter_.dim(1).set_min(0);
63 filter_.dim(2).set_min(0);
64 Expr filter_width = filter_.dim(1).extent();
65 Expr filter_height = filter_.dim(2).extent();
66 RDom r(0, filter_width, 0, filter_height);
67 Expr filter_zeroed_rdxy = filter_zeroed(c, r.x, r.y);
68
69 // We want to compute the reduction:
70 // convolved(c, x, y, b) = bias_(c)
71 // convolved(c, x, y, b) +=
72 // i32(filter_zeroed_rdxy) *
73 // (i32(input_rdxy) - i32(input_zero_))
74 //
75 // However, this requires subtracting the input zero at every output.
76 // We can factor the reduction like so:
77 //
78 // convolved(c, x, y, b) = bias_(c)
79 // convolved(c, x, y, b) +=
80 // i32(filter_zeroed_rdxy) * i32(input_rdxyc) -
81 // i32(filter_zeroed_rdxy) * i32(input_zero_)
82 //
83 // The latter reduction can be computed once per output channel.
84 Func sum_filter("sum_filter");
85 sum_filter(c) += i32(filter_zeroed_rdxy);
86
87 Func offset_c("offset_c");
88 offset_c(c) = bias_(c) - sum_filter(c) * i32(input_zero_);
89
90 Expr input_rdxy =
91 resampled_input(c, x * stride_x_ + r.x * dilation_x_, y * stride_y_ + r.y * dilation_y_, b);
92 Func convolved("convolved");
93 convolved(c, x, y, b) = offset_c(c);
94 convolved(c, x, y, b) += i32(filter_zeroed_rdxy) * i32(input_rdxy);
95
96 // Saturate and narrow the output.
97 Expr output = multiply_2x_high(convolved(c, x, y, b), output_multiplier_);
98 output = i16_sat(rounding_shift_right(output, output_shift_));
99 output = u8_sat(saturating_add(output, output_zero_));
100 output_(c, x, y, b) = clamp(output, output_min_, output_max_);
101
102 // Schedule.
103 interpret_as_tensor(input_);
104 interpret_as_tensor(filter_);
105 interpret_as_tensor(bias_);
106 interpret_as_tensor(output_);
107 require_same_min_extent(3, input_, output_);
108 require_same_min_extent(0, bias_, output_);
109 require_same_min_extent(0, filter_, output_);
110
111 int vector_size = natural_vector_size<uint8_t>();
112 if (get_register_count(target) < 32) {
113 vector_size = natural_vector_size<int16_t>();
114 }
115
116 if (inv_depth_multiplier_ == 0) {
117 // When we're broadcasting input channels, require that the input has only
118 // one channel.
119 input_.dim(0).set_extent(1);
120 } else if (inv_depth_multiplier_ == 1) {
121 // Require the input to be aligned.
122 const int input_alignment = vector_size;
123 input_.set_host_alignment(input_alignment);
124 for (int d = 1; d < input_.dimensions(); d++) {
125 input_.dim(d).set_stride(align(input_.dim(d).stride(), input_alignment));
126 }
127 }
128
129 // Tile the output, so we can try to re-use loads spatially when performing
130 // convolution. This also helps because we can schedule the input and not
131 // waste work for strides less than the tile size.
132 // We split co and reorder it outermost, so we can maximize locality of the
133 // filter. We even put it outside of the batch loop, so we can compute the
134 // boundary condition on the filter at co and reuse it across batches.
135 const int kTileW = 2;
136 const int kTileH = 2;
137 // When the output is small, the overhead from shift inwards can be large.
138 // Only tile when the input is at least this many tiles to avoid this.
139 const int kMinTiles = 4;
140 Var xo("xo"), yo("yo"), co("co");
141 Expr output_channels = output_.dim(0).extent();
142 Expr output_width = output_.dim(1).extent();
143 Expr output_height = output_.dim(2).extent();
144 Expr use_tiles =
145 (output_width >= kTileW * kMinTiles || output_width % kTileW == 0) &&
146 (output_height >= kTileH * kMinTiles || output_height % kTileH == 0);
147 output_.compute_root()
148 .specialize(output_channels >= vector_size && use_tiles)
149 .tile(x, y, xo, yo, x, y, kTileW, kTileH, TailStrategy::ShiftInwards)
150 .split(c, co, c, vector_size, TailStrategy::ShiftInwards)
151 .reorder(x, y, c, xo, yo, b, co)
152 .unroll(x)
153 .unroll(y)
154 .vectorize(c);
155
156 // Enable 1x1 outputs to work.
157 output_
158 .tile(x, y, xo, yo, x, y, 1, 1, TailStrategy::RoundUp)
159 .unroll(x)
160 .unroll(y);
161
162 // Vectorize c, using predication only for small numbers of channels.
163 output_
164 .specialize(output_channels >= vector_size)
165 .split(c, co, c, vector_size, TailStrategy::ShiftInwards)
166 .reorder(x, y, c, xo, yo, b, co)
167 .vectorize(c);
168 output_
169 .split(c, co, c, vector_size, TailStrategy::Predicate)
170 .reorder(x, y, c, xo, yo, b, co)
171 .vectorize(c);
172
173 convolved.compute_at(output_, xo)
174 .store_in(MemoryType::Register)
175 .bound_extent(c, vector_size)
176 .unroll(x)
177 .unroll(y)
178 .vectorize(c);
179 convolved.update()
180 .reorder(x, y, r.x, r.y)
181 .unroll(x)
182 .unroll(y)
183 .vectorize(c);
184 convolved.update()
185 .specialize(filter_width == 3 && filter_height == 3)
186 .unroll(r.x)
187 .unroll(r.y);
188
189 if (inv_depth_multiplier_ < 0) {
190 // The reason inv_depth_multiplier_ is a GeneratorParam and not a
191 // specialization is that we can't specialize the (lack of) compute_at here.
192 resampled_input
193 .compute_at(output_, b)
194 .store_in(MemoryType::Stack)
195 .vectorize(c, vector_size, TailStrategy::GuardWithIf);
196
197 resampled_input.specialize(depth_multiplier_ == 1);
198 }
199
200 filter_zeroed.compute_at(output_, co)
201 .store_in(MemoryType::Stack)
202 .align_storage(c, natural_vector_size<int16_t>())
203 .vectorize(c, natural_vector_size<int16_t>(), TailStrategy::GuardWithIf)
204 .unroll(c, 2, TailStrategy::GuardWithIf);
205
206 offset_c.compute_at(output_, co)
207 .store_in(MemoryType::Stack)
208 .align_storage(c, natural_vector_size<int32_t>())
209 .vectorize(c, vector_size, TailStrategy::GuardWithIf);
210 }
211};
