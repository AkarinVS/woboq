9// There are two codepaths in this generator. On targets with widening
10// 8-bit multiplies, we implement the reduction by expanding the subtraction
11// of the offsets into 4 reductions involving 8-bit multiplies. On targets
12// without widening 8-bit multiplication, it's faster to just subtract the
13// offsets and use 16-bit multiplications.
14bool use_8bit_multiply(const Target &target) {
15 return target.arch != Target::X86 || target.has_feature(Target::AVX512_SapphireRapids);
16}
17
18class FullyConnected : public Generator<FullyConnected> {
19public:
20 Input<Buffer<uint8_t>> input_{"input", 2};
21 Input<uint8_t> input_zero_{"input_zero"};
22
23 Input<Buffer<uint8_t>> filter_{"filter", 2};
24 Input<uint8_t> filter_zero_{"filter_zero"};
25
26 Input<Buffer<int32_t>> bias_{"bias", 1};
27
28 Input<uint8_t> output_zero_{"output_zero"};
29 Input<int32_t> output_multiplier_{"output_multiplier"};
30 // TODO: We only need this to be a signed shift for exactly one known network.
31 // Figure out if there is something else we should be doing instead.
32 Input<int32_t> output_shift_{"output_shift"};
33 Input<uint8_t> output_min_{"output_min"};
34 Input<uint8_t> output_max_{"output_max"};
35
36 Output<Buffer<>> output_{"output", 2};
37
38 void generate() {
39 Var c("c"), b("b");
40
41 // We require the reduction dimension to be aligned to a uint8 vector.
42 const int filter_alignment = natural_vector_size<uint8_t>();
43 Expr filter_extent = align(filter_.dim(0).extent(), filter_alignment);
44 filter_.dim(0).set_min(0);
45 RDom rc(0, filter_extent);
46
47 Func sum_input("sum_input");
48 Func sum_filter("sum_filter");
49 Func multiplied("multiplied");
50 if (use_8bit_multiply(target)) {
51 // We want to compute the reduction:
52 // multiplied(c, b) = bias_(c)
53 // multiplied(c, b) +=
54 // (i32(input) - i32(input_zero_)) *
55 // (i32(filter) - i32(filter_zero_))
56 //
57 // However, this precludes using efficient dot product instructions. To
58 // fix this, expand the expression:
59 //
60 // multiplied(c, b) = bias_(c)
61 // multiplied(c, b) +=
62 // i32(filter(rc, c)) * i32(input(rc, b)) -
63 // i32(filter(rc, c)) * i32(input_zero_) -
64 // i32(filter_zero_) * i32(input(rc, b)) +
65 // i32(filter_zero_) * i32(input_zero_)
66 //
67 // We can then separate this into several reductions. The last reduction
68 // is a constant, and the middle two reductions can be computed once for
69 // each c or b, instead of each (c, b).
70 sum_input(b) += u32(input_(rc, b));
71 sum_filter(c) += u32(filter_(rc, c));
72
73 multiplied(c, b) =
74 bias_(c) + filter_extent * filter_zero_ * input_zero_ -
75 i32(sum_input(b)) * filter_zero_;
76
77 multiplied(c, b) += i32(u16(filter_(rc, c)) * u16(input_(rc, b)));
78
79 // TODO: This subtract happens after the total vector reductions from the
80 // above reduction. It would be a lot better if we could do this subtract
81 // first somehow.
82 multiplied(c, b) = multiplied(c, b) - i32(sum_filter(c)) * i32(input_zero_);
83 } else {
84 multiplied(c, b) = bias_(c);
85 multiplied(c, b) +=
86 i32(i16(filter_(rc, c)) - i16(filter_zero_)) *
87 i32(i16(input_(rc, b)) - i16(input_zero_));
88 }
89
90 // Saturate and narrow the output.
91 Expr output = multiply_2x_high(multiplied(c, b), output_multiplier_);
92 output = i16_sat(rounding_shift_right(output, output_shift_));
93 if (output_.type() == halide_type_of<uint8_t>()) {
94 output = u8_sat(saturating_add(output, output_zero_));
95 output = clamp(output, output_min_, output_max_);
96 }
97 output_(c, b) = output;
98
99 // Schedule.
100 // Reorder batches inside the outer loop over channels to improve locality
101 // of accesses to the filter. This also allows us to compute the sum of the
102 // filter for only a subset of channels at a time.
103 Var co("co"), bo("bo");
104 Expr output_channels = output_.dim(0).extent();
105 Expr output_batches = output_.dim(1).extent();
106 // Use half of the registers as accumulators.
107 const int accum_registers = get_register_count(target) / 2;
108 // When we have enough batches, compute a few of them at a time, so we can
109 // re-use the filter a few times.
110 const int tile_batches = 4;
111 output_.compute_root()
112 .specialize(output_channels >= accum_registers / tile_batches && output_batches >= tile_batches)
113 .split(c, co, c, accum_registers / tile_batches, TailStrategy::ShiftInwards)
114 .split(b, bo, b, tile_batches, TailStrategy::ShiftInwards)
115 .reorder(c, b, bo, co)
116 .vectorize(c)
117 .unroll(b);
118
119 // Handle the batch 1 case. In this case, we need an accumulator for both
120 // the filter and the input.
121 output_
122 .specialize(output_channels >= accum_registers)
123 .split(c, co, c, accum_registers / 2, TailStrategy::ShiftInwards)
124 .split(b, bo, b, 1)
125 .reorder(c, b, bo, co)
126 .vectorize(c)
127 .unroll(b);
128
129 // Make dummy outer loops if there aren't enough channels or batches.
130 output_
131 .split(c, co, c, 1)
132 .split(b, bo, b, 1)
133 .reorder(c, b, bo, co);
134
135 multiplied.compute_at(output_, bo)
136 .vectorize(c)
137 .unroll(b);
138 // Enable sum_filter to be skipped if it isn't needed.
139 multiplied.specialize(filter_zero_ == 0);
140
141 // The schedule here splits the reduction into 3 parts:
142 // 1. The inner vector reduction factor (rci)
143 // 2. The outer vector reduction factor (rc)
144 // 3. The reduction of whole vectors (rco).
145 // Step 2 is saved for the end, which is a total reduction of an int32 vector.
146 // The other two steps map nicely to vector reductions like udot or pmaddwd.
147 const int accum_vector_size = natural_vector_size<int32_t>();
148 const int vector_reduction_factor = get_vector_reduction_factor(target, UInt(8));
149 RVar rci, rco;
150 multiplied.update()
151 .split(rc, rc, rci, vector_reduction_factor)
152 .split(rc, rco, rc, accum_vector_size);
153 Func multiplied_intm = multiplied.update().rfactor(rc, co);
154
155 multiplied_intm.compute_at(output_, bo)
156 .reorder_storage(co, c)
157 .vectorize(co)
158 .unroll(c)
159 .unroll(b)
160 .update()
161 .reorder(rci, co, c, b, rco)
162 .unroll(c)
163 .unroll(b)
164 .vectorize(co)
165 .atomic()
166 .vectorize(rci)
167 .specialize(input_zero_ == 0);
168
169 // We could transpose here by adding a wrapper to multiplied_intm and reordering
170 // the storage, which would enable the reduction below to be a pure vectorize
171 // instead of a vector reduction, but this didn't seem to be better on either x86
172 // or ARM.
173
174 multiplied.update()
175 .reorder(c, rc, b)
176 .unroll(c)
177 .atomic()
178 .vectorize(rc);
179
180 if (use_8bit_multiply(target)) {
181 // We schedule this to use the same loops as multiplied_intm above, so we can
182 // compute_with it.
183 sum_filter.compute_at(output_, bo)
184 .vectorize(c);
185 sum_filter.update()
186 .split(rc, rc, rci, vector_reduction_factor)
187 .split(rc, rco, rc, accum_vector_size);
188 Func sum_filter_intm = sum_filter.update().rfactor(rc, co);
189
190 sum_filter_intm.compute_at(output_, bo)
191 .reorder_storage(co, c)
192 .vectorize(co)
193 .unroll(c)
194 .update()
195 .reorder(rci, co, c, rco)
196 .unroll(c)
197 .vectorize(co)
198 .atomic()
199 .vectorize(rci);
200 sum_filter_intm.update().compute_with(multiplied_intm.update(), rco);
201
202 sum_filter.update()
203 .reorder(c, rc)
204 .unroll(c)
205 .atomic()
206 .vectorize(rc);
207
208 multiplied.update(1)
209 .vectorize(c)
210 .unroll(b);
211
212 // This reduction could be optimized better, but it rarely matters much.
213 const int reduce_vector_size = natural_vector_size<uint8_t>();
214 sum_input.compute_root()
215 .update()
216 .atomic()
217 .reorder(rc, b)
218 .vectorize(rc, reduce_vector_size);
219 }
220
221 if (get_target().arch == Target::ARM) {
222 // For some absurd reason, making a copy of this buffer makes this pipeline
223 // up to 10x faster on ARM (e.g. the fully connected layer of resnet with 8
224 // batches). It also appears to have negligible impact in other cases.
225 // TODO: Figure out why this is, and figure out how to avoid this dumb copy.
226 input_.in().compute_root();
227 }
228 }
229};
