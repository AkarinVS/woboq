10class L2Normalization : public Generator<L2Normalization> {
11public:
12 Input<Buffer<uint8_t>> input_{"input", 2};
13 Input<uint8_t> input_zero_{"input_zero"};
14
15 Output<Buffer<uint8_t>> output_{"output", 2};
16
17 void generate() {
18 Var x("x"), y("y");
19
20 // We don't need the input scale, because the result of L2
21 // normalization doesn't depend on it.
22 Func input_zeroed("input_zeroed");
23 input_zeroed(x, y) = i16(input_(x, y)) - i16(input_zero_);
24
25 Func sum_input_sq("sum_input_sq");
26 RDom rx(input_.dim(0).min(), input_.dim(0).extent());
27 sum_input_sq(y) = i32(0);
28 sum_input_sq(y) += pow(i32(input_zeroed(rx, y)), 2);
29
30 Func inv_sqrt("inv_sqrt");
31 const int q = 15;
32 inv_sqrt(y) = approx_reciprocal_sqrt(q, sum_input_sq(y));
33
34 // The output has a scale of 2^7 = 128 and offset of 128.
35 Expr output = i32(input_zeroed(x, y)) * i32(inv_sqrt(y));
36 output = i16_sat(rounding_shift_right(output, q - 7));
37 output_(x, y) = u8_sat(saturating_add(output, i16(128)));
38
39 // Schedule.
40 const int vector_size = natural_vector_size<uint8_t>();
41
42 output_.compute_root()
43 .vectorize(x, vector_size, TailStrategy::Predicate);
44
45 inv_sqrt.compute_at(output_, y);
46
47 sum_input_sq.compute_at(output_, y)
48 .update()
49 .atomic()
50 .vectorize(rx, vector_size);
51 }
52};

54class Softmax : public Generator<Softmax> {
55public:
56 Input<Buffer<uint8_t>> input_{"input", 2};
57 // The beta multiplier and shift should have an extra factor of log2(e).
58 Input<int16_t> beta_multiplier_{"beta_multiplier"};
59 Input<uint16_t> beta_shift_{"beta_shift"};
60
61 Input<uint8_t> output_zero_{"output_zero"};
62 Input<int16_t> output_multiplier_{"output_multiplier"};
63 Input<uint16_t> output_shift_{"output_shift"};
64 Output<Buffer<uint8_t>> output_{"output", 2};
65
66 void generate() {
67 // The algorithm.
68 Var x("x"), y("y");
69
70 // On x86, this fixed point approximation is actually much slower
71 // than just using floats, but producing identical results on all
72 // targets is nice, and this op doesn't appear to be a significant
73 // factor in overall performance.
74
75 // Compute 2^input_(x, y) / sum(2^input_(rx, y)) by rewriting it
76 // to 2^(input_(x, y) - max_x(y)) / sum(2^(input_(rx, y) - max_x(y)).
77 // This makes it easier to compute in fixed point, because we know
78 // that 2^x is less than 1.
79 Func max_x("max_x");
80 RDom rx(input_.dim(0).min(), input_.dim(0).extent());
81 max_x(y) = u8(0);
82 max_x(y) = max(max_x(y), input_(rx, y));
83
84 Expr diff = (i16(input_(x, y)) - i16(max_x(y))) << softmax_input_shift;
85 Expr diff_beta = multiply_2x_high(diff, beta_multiplier_);
86
87 // Since we know that diff_beta is less than 0, we can use the full
88 // range of an integer for the fractional part.
89 constexpr int q = 15;
90 Func exp2_diff("exp2_diff");
91 exp2_diff(x, y) = approx_exp2(q, diff_beta, beta_shift_, Int(16));
92
93 // This could overflow if there are more than 2^16 values of x.
94 Func sum_exp_row("sum_exp_row");
95 sum_exp_row(y) += i32(exp2_diff(rx, y));
96
97 // Below, we compute exp2_diff * inv_sum_exp_row / 2^15, so we need to
98 // multiply by 2^(q + 15) to get a result of the correct quantization.
99 // This doesn't overflow because we know the sum is greater than or equal
100 // to 2^0*2^q, because we subtracted the max from the input.
101 Func inv_sum_exp_row("inv_sum_exp_row");
102 inv_sum_exp_row(y) = approx_reciprocal(q + 15, sum_exp_row(y), Int(16));
103
104 static_assert(q == 15, "");
105 Expr output = multiply_2x_high(exp2_diff(x, y), inv_sum_exp_row(y));
106 output = multiply_2x_high(output, output_multiplier_);
107 output = rounding_shift_right(output, output_shift_);
108 output_(x, y) = u8_sat(saturating_add(output, output_zero_));
109
110 // Schedule.
111 const int vector_size = natural_vector_size<uint8_t>();
112
113 output_.vectorize(x, vector_size, TailStrategy::Predicate);
114
115 max_x.compute_at(output_, y)
116 .update()
117 .atomic()
118 .vectorize(rx, vector_size, TailStrategy::GuardWithIf);
119
120 sum_exp_row.compute_at(output_, y)
121 .update()
122 .atomic()
123 .vectorize(rx, vector_size, TailStrategy::GuardWithIf);
124
125 inv_sum_exp_row.compute_at(output_, y);
126 }
127};
