126halide_type_t ConvertTfLiteType(TfLiteType t) {

130 case kTfLiteFloat16:
131 return halide_type_t(halide_type_float, 16);
132 case kTfLiteFloat32:
133 return halide_type_t(halide_type_float, 32);

187std::vector<int> ConvertTfLiteShape(const TfLiteTensor &tensor) {
188 assert(tensor.dims);
189 const int shape_size = tensor.dims->size;
190 std::vector<int> shape;
191 shape.reserve(shape_size);
192 for (int i = 0; i < shape_size; i++) {
193 shape.push_back(tensor.dims->data[shape_size - 1 - i]);
194 }
195 return shape;
196}

198TensorPtr ConvertTfLiteTensor(const TfLiteTensor &tensor) {
199 auto shape = ConvertTfLiteShape(tensor);
200
201 halide_type_t type = ConvertTfLiteType(tensor.type);
202
203 QuantizationInfo quantization;
204 if (tensor.quantization.type == kTfLiteAffineQuantization) {
205 const TfLiteAffineQuantization *q = (const TfLiteAffineQuantization *)tensor.quantization.params;
206 for (int i = 0; i < q->scale->size; ++i) {
207 const float scale = q->scale->data[i];
208 quantization.scale.emplace_back(scale);
209 }
210 for (int i = 0; i < q->zero_point->size; i++) {
211 const int zero = q->zero_point->data[i];
212 quantization.zero.emplace_back(zero);
213 }
214 quantization.dimension = tensor.dims->size - q->quantized_dimension;
215 }
216
217 // tensor.name can be null, apparently. I don't think we have any requirement
218 // for unique or non-empty names in our code, so let's just map that to
219 // an empty string.
220 const char *name = tensor.name ? tensor.name : "";
221
222 if (IsConstantTensor(tensor)) {
223 const void *read_only_data = (const uint8_t *)tensor.data.data;
224 assert(read_only_data != nullptr);
225 // Construct a HalideBuffer that points to read_only_data (but does not copy or own it).
226 // Since TFLite will ensure that the TfLiteTensor remains valid while we're using it,
227 // this should be completely safe.
228 HalideBuffer<void> buffer(type, const_cast<void *>(read_only_data), shape);
229 assert(tensor.bytes == buffer.size_in_bytes());
230
231 return std::make_shared<Tensor>(name, std::move(buffer), std::move(quantization));
232 }
233
234 // Create an "unallocated" Buffer, which points to null.
235 HalideBuffer<void> buffer(type, nullptr, shape);
236 return std::make_shared<Tensor>(name, std::move(buffer), std::move(quantization));
237}

247 // Init() will be called exactly once per instance.
248 TfLiteStatus Init(TfLiteContext *context,
249 const TfLiteDelegateParams *params) {
250 if (options_.verbosity >= 1) {
251 HLOG(INFO) << "Delegate " << (void *)this << " Init\n";
252 }
253
254 if (interpreter_ != nullptr) {
255 TF_LITE_KERNEL_LOG(context, "Init must not be called twice.");
256 return kTfLiteDelegateError;
257 }
258
259 std::vector<int> node_indices(params->nodes_to_replace->size);
260 for (int i = 0; i < params->nodes_to_replace->size; i++) {
261 const int node_index = params->nodes_to_replace->data[i];
262 node_indices[i] = node_index;
263 }
264 if (options_.verbosity >= 1) {
265 HLOG(INFO) << "Delegate " << (void *)this << " Init nodes: " << node_indices << "\n";
266 }
267
268 // Pre-emptively map *all* the TFLiteTensors into our Tensor type.
269 for (size_t tensor_id = 0; tensor_id < context->tensors_size; tensor_id++) {
270 const TfLiteTensor &tensor = context->tensors[tensor_id];
271 if (tensor.dims == nullptr) {
272 // Can't convert a TfLiteTensor with no dimension info
273 continue;
274 }
275 auto t = ConvertTfLiteTensor(tensor);
276 assert(!tensors_.count(tensor_id));
277 tensors_[tensor_id] = t;
278 }
279
280 // Be careful with params->input_tensors and params->output_tensors here;
281 // in particular, params->input_tensors will contain all of the 'constant'
282 // input tensors (which are generally inputs only to a specific node).
283
284 // Mark the input and output tensors correctly, as code in our interpreter
285 // relies upon it.
286 std::vector<TensorPtr> inputs;
287 for (int i = 0; i < params->input_tensors->size; i++) {
288 const int tensor_id = params->input_tensors->data[i];
289 if (tensor_id == kTfLiteOptionalTensor) {
290 continue;
291 }
292 auto t = GetTensorById(context, tensor_id);
293 t->set_input(true);
294 inputs.push_back(t);
295 if (options_.verbosity >= 2) {
296 HLOG(INFO) << "Delegate " << (void *)this << (t->is_constant() ? " Const" : "") << " Input tensor: " << tensor_id << "\n";
297 }
298 }
299
300 // Add the output tensors.
301 std::vector<TensorPtr> outputs;
302 for (int i = 0; i < params->output_tensors->size; i++) {
303 const int tensor_id = params->output_tensors->data[i];
304 if (tensor_id == kTfLiteOptionalTensor) {
305 continue;
306 }
307 if (options_.verbosity >= 2) {
308 HLOG(INFO) << "Delegate " << (void *)this << " Output tensor: " << tensor_id << "\n";
309 }
310 auto t = GetTensorById(context, tensor_id);
311 t->set_output(true);
312 outputs.push_back(t);
313 }
314
315 // Add all ops.
316 TfLiteNode *node;
317 TfLiteRegistration *reg;
318 std::vector<OpPtr> ops;
319 for (int node_index : node_indices) {
320 TF_LITE_ENSURE_STATUS(context->GetNodeAndRegistration(context, node_index, &node, &reg));
321 const int op_type = reg->builtin_code;
322 OpPtr op;
323
324 // clang-format off
325 switch (op_type) {
326 #define KNOWN_OP(OP) case kTfLiteBuiltin##OP: op = Build##OP(context, node); break;
327 ALL_KNOWN_OPS
328 #undef KNOWN_OP
329
330 default:
331 TF_LITE_KERNEL_LOG(context, "Op not supported: %d", op_type);
332 return kTfLiteDelegateError;
333 }
334 // clang-format on
335
336 if (op == nullptr) {
337 TF_LITE_KERNEL_LOG(context, "Op factory returned null: %s", op_type);
338 return kTfLiteDelegateError;
339 }
340 ops.push_back(std::move(op));
341 }
342 model_ = make_op<OpGroup>(std::move(inputs), std::move(outputs), std::move(ops));
343
344 return kTfLiteOk;
345 }
346
347 // Prepare() will be called at least once, prior to any calls to Eval().
348 // It will be called again if tensor shape(s) change. It is preferable
349 // to do all memory allocation in Prepare(), rather than Eval(), if possible.
350 TfLiteStatus Prepare(TfLiteContext *context, TfLiteNode *node) {
351 if (options_.verbosity >= 1) {
352 HLOG(INFO) << "Delegate " << (void *)this << " Prepare\n";
353 }
354
355 assert(model_ != nullptr);
356
357 if (interpreter_ != nullptr) {
358 TF_LITE_KERNEL_LOG(context, "Calling Prepare() multiple times");
359 return kTfLiteDelegateError;
360 }
361
362 interpreter_ = std::unique_ptr<Interpreter>(new Interpreter(std::move(model_)));
363
364 for (int i = 0; i < node->outputs->size; i++) {
365 const int tensor_id = node->outputs->data[i];
366 if (tensor_id == kTfLiteOptionalTensor) {
367 continue;
368 }
369 auto t = GetTensorById(context, tensor_id);
370 if (t && t->is_dynamic()) {
371 if (options_.verbosity >= 2) {
372 HLOG(INFO) << "SetTensorToDynamic " << tensor_id;
373 }
374 SetTensorToDynamic(context, tensor_id);
375 }
376 }
377
378 return kTfLiteOk;
379 }
380
381 // Eval() will be called at least once. It can expect that Prepare() will
382 // have been called for the current set of tensor shape(s).
383 TfLiteStatus Eval(TfLiteContext *context, TfLiteNode *node) {
384 if (options_.verbosity >= 3) {
385 HLOG(INFO) << "Delegate " << (void *)this << " Eval\n";
386 }
387
388 if (interpreter_ == nullptr) {
389 TF_LITE_KERNEL_LOG(context, "interpreter_ is not built in Eval");
390 return kTfLiteDelegateError;
391 }
392
393 // Copy the non-constant Tensor inputs. TODO: avoid this by sharing pointers.
394 for (int i = 0; i < node->inputs->size; i++) {
395 const int tensor_id = node->inputs->data[i];
396 if (tensor_id == kTfLiteOptionalTensor) {
397 continue;
398 }
399 assert(tensor_id >= 0 && tensor_id < (int)context->tensors_size);
400 const TfLiteTensor &tensor = context->tensors[tensor_id];
401 auto t = GetTensorById(context, tensor_id);
402 assert(t->is_constant() == IsConstantTensor(tensor));
403 if (t->is_constant()) {
404 continue;
405 }
406 assert(t->is_input() && !t->is_constant() && t->is_allocated());
407 auto buf = t->buffer();
408 assert(buf.size_in_bytes() == tensor.bytes);
409 memcpy(buf.data(), tensor.data.data, tensor.bytes);
410 }
411
412 // TODO: execute needs to return an error code.
413 interpreter_->execute();
414
415 // Copy the Tensor outputs. TODO: avoid this by sharing pointers.
416 for (int i = 0; i < node->outputs->size; i++) {
417 const int tensor_id = node->outputs->data[i];
418 if (tensor_id == kTfLiteOptionalTensor) {
419 continue;
420 }
421 assert(tensor_id >= 0 && tensor_id < (int)context->tensors_size);
422 TfLiteTensor &tensor = context->tensors[tensor_id];
423 assert(!IsConstantTensor(tensor));
424 auto t = GetTensorById(context, tensor_id);
425 assert(t->is_output() && !t->is_constant() && t->is_allocated());
426 if (t->is_dynamic()) {
427 HCHECK(IsDynamicTensor(tensor));
428 const Box b = t->bounds();
429 TfLiteIntArray *new_size = TfLiteIntArrayCreate((int)b.size());
430 for (size_t i = 0; i < b.size(); i++) {
431 new_size->data[b.size() - i - 1] = b[i].extent();
432 }
433 // (Note that ResizeTensor takes ownership of new_size, even if an error is returned.)
434 auto status = context->ResizeTensor(context, &tensor, new_size);
435 if (status != kTfLiteOk) {
436 TF_LITE_KERNEL_LOG(context, "ResizeTensor() failed:", status);
437 return status;
438 }
439 }
440 auto buf = t->buffer();
441 assert(tensor.data.data != nullptr);
442 assert(buf.data() != nullptr);
443 assert(buf.size_in_bytes() == tensor.bytes);
444
445 memcpy(tensor.data.data, buf.data(), tensor.bytes);
446 }
447
448 // Eval() could be called again with the same graph -- don't destroy the interpreter_ yet.
449
450 return kTfLiteOk;
451 }

592 OpPtr BuildConcatenation(TfLiteContext *context, TfLiteNode *node) {
593 std::vector<TensorPtr> inputs(node->inputs->size);
594 for (int i = 0; i < node->inputs->size; i++) {
595 inputs[i] = GetTensorById(context, node->inputs->data[i]);
596 }
597 auto output = GetTensorById(context, node->outputs->data[0]);
598 const TfLiteConcatenationParams *params = (const TfLiteConcatenationParams *)(node->builtin_data);
599 auto activation = ConvertTfLiteActivation(params->activation);
600 HCHECK(activation == ActivationFunction::None);
601 int axis = params->axis;
602 // Handle negative values, which are legal
603 if (axis < 0) {
604 axis = (int)output->rank() + axis;
605 }
606 // Now 'flip' the axis so that it refers to the right dimension in
607 // the Tensor (since we reverse the dimension order)
608 axis = (int)output->rank() - axis - 1;
609 return make_op<ConcatenationOp>(inputs, output, axis);
610 }

620 params->stride_width,
621 params->stride_height,

1604/*static*/ TfLiteStatus HannkDelegate::DelegatePrepare(TfLiteContext *context, TfLiteDelegate *delegate) {
1605 HannkDelegate *self = (HannkDelegate *)delegate;
1606 const int verbosity = self->options_.verbosity;
1607
1608 TfLiteStatus status;
1609
1610 TfLiteIntArray *plan = nullptr;
1611 if ((status = context->GetExecutionPlan(context, &plan)) != kTfLiteOk) {
1612 HLOG(ERROR) << "GetExecutionPlan failed";
1613 return status;
1614 }
1615
1616 // Build up a list of the nodes we want to handle.
1617 std::vector<int> supported_nodes;
1618 for (int i = 0; i < plan->size; i++) {
1619 const int node_index = plan->data[i];
1620 TfLiteNode *node;
1621 TfLiteRegistration *registration;
1622 if ((status = context->GetNodeAndRegistration(context, node_index, &node, &registration)) != kTfLiteOk) {
1623 HLOG(ERROR) << "GetNodeAndRegistration failed";
1624 return status;
1625 }
1626
1627 NodeSupport support(context, node, registration, verbosity >= 1);
1628 if (support.IsNodeSupported()) {
1629 if (verbosity >= 2) {
1630 HLOG(INFO) << "Handling node, index=" << node_index
1631 << " code=" << registration->builtin_code
1632 << " (" << GetOpName(registration->builtin_code) << ")";
1633 }
1634 supported_nodes.push_back(node_index);
1635 } else {
1636 if (verbosity >= 1) {
1637 HLOG(INFO) << "Skipping unsupported node, index=" << node_index
1638 << " code=" << registration->builtin_code
1639 << " (" << GetOpName(registration->builtin_code) << ")"
1640 << " version=" << registration->version
1641 << " custom_name=(" << (registration->custom_name ? registration->custom_name : "null") << "); "
1642 << "Reason(s): " << support.Failures();
1643 }
1644 }
1645 }
1646
1647 if ((status = context->ReplaceNodeSubsetsWithDelegateKernels(context,
1648 HannkDelegateKernel::GetRegistration(),
1649 BuildTfLiteIntArray(supported_nodes).get(),
1650 delegate)) != kTfLiteOk) {
1651 HLOG(ERROR) << "ReplaceNodeSubsetsWithDelegateKernels failed";
1652 return status;
1653 }
1654
1655 return kTfLiteOk;
1656}
