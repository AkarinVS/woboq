8class HannkDelegateProvider : public DelegateProvider {
9public:
10 HannkDelegateProvider() {
11 default_params_.AddParam("use_hannk", ToolParam::Create<bool>(false));
12 default_params_.AddParam("hannk_verbosity", ToolParam::Create<int>(0));
13 }
14
15 std::vector<Flag> CreateFlags(ToolParams *params) const final {
16 std::vector<Flag> flags = {
17 CreateFlag<bool>("use_hannk", params, "use HANNK"),
18 CreateFlag<int>("hannk_verbosity", params, "Verbosity of HANNK debug logging"),
19 };
20 return flags;
21 }
22
23 void LogParams(const ToolParams &params, bool verbose) const final {
24 LOG_TOOL_PARAM(params, bool, "use_hannk", "Use HANNK", verbose);
25 LOG_TOOL_PARAM(params, int, "hannk_verbosity", "HANNK verbosity", verbose);
26 }
27
28 TfLiteDelegatePtr CreateTfLiteDelegate(const ToolParams &params) const final {
29 if (params.Get<bool>("use_hannk")) {
30 HannkDelegateOptions options = {};
31 options.verbosity = params.Get<int32_t>("hannk_verbosity");
32 if (options.verbosity >= 1) {
33 HLOG(INFO) << "Registrar HannkDelegate: verbosity set to "
34 << options.verbosity << ".";
35 }
36 return TfLiteDelegatePtr(HannkDelegateCreate(&options), &HannkDelegateDelete);
37 } else {
38 return TfLiteDelegatePtr(nullptr, [](TfLiteDelegate *) {});
39 }
40 }
41
42 std::string GetName() const final {
43 return "HANNK";
44 }
45};
