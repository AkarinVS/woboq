38 // Timing code

40 // Manually-tuned version
41 double min_t_manual = benchmark(timing_iterations, 10, [&]() {
42 lens_blur(left_im, right_im, slices, focus_depth, blur_radius_scale,

44 output.device_sync();

46 printf("Manually-tuned time: %gms\n", min_t_manual * 1e3);
