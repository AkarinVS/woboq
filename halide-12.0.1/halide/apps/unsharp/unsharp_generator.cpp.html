1#include "Halide.h"
2
3namespace {
4
5class Unsharp : public Halide::Generator<Unsharp> {
6public:
7 GeneratorParam<float> sigma{"sigma", 1.5f};
8
9 Input<Buffer<float>> input{"input", 3};
10 Output<Buffer<float>> output{"output", 3};
11
12 void generate() {
13 Var x("x"), y("y"), c("c");
14
15 const float kPi = 3.14159265358979310000f;
16
17 Func kernel("kernel");
18 kernel(x) = exp(-x * x / (2 * sigma * sigma)) / (sqrtf(2 * kPi) * sigma);
19
20 Func input_bounded = Halide::BoundaryConditions::repeat_edge(input);
21
22 Func gray("gray");
23 gray(x, y) = (0.299f * input_bounded(x, y, 0) +
24 0.587f * input_bounded(x, y, 1) +
25 0.114f * input_bounded(x, y, 2));
26
27 Func blur_y("blur_y");
28 blur_y(x, y) = (kernel(0) * gray(x, y) +
29 kernel(1) * (gray(x, y - 1) +
30 gray(x, y + 1)) +
31 kernel(2) * (gray(x, y - 2) +
32 gray(x, y + 2)) +
33 kernel(3) * (gray(x, y - 3) +
34 gray(x, y + 3)));
35
36 Func blur_x("blur_x");
37 blur_x(x, y) = (kernel(0) * blur_y(x, y) +
38 kernel(1) * (blur_y(x - 1, y) +
39 blur_y(x + 1, y)) +
40 kernel(2) * (blur_y(x - 2, y) +
41 blur_y(x + 2, y)) +
42 kernel(3) * (blur_y(x - 3, y) +
43 blur_y(x + 3, y)));
44
45 Func sharpen("sharpen");
46 sharpen(x, y) = 2 * gray(x, y) - blur_x(x, y);
47
48 Func ratio("ratio");
49 ratio(x, y) = sharpen(x, y) / gray(x, y);
50
51 output(x, y, c) = ratio(x, y) * input(x, y, c);
52
53 // Estimates (for autoscheduler; ignored otherwise)
54 {
55 input.dim(0).set_estimate(0, 1536);
56 input.dim(1).set_estimate(0, 2560);
57 input.dim(2).set_estimate(0, 3);
58 output.dim(0).set_estimate(0, 1536);
59 output.dim(1).set_estimate(0, 2560);
60 output.dim(2).set_estimate(0, 3);
61 }
62
63 // Schedule
64 if (!auto_schedule) {
65 // Some Intel Mac Minis have GPUs that require tile sizes smaller than 32x32
66 // for this pipeline because they have too few registers. Drop to 16x16 to
67 // avoid unexpected crashes in CI.
68 const int tile_size = get_target().has_feature(Target::Metal) ? 16 : 32;
69
70 if (get_target().has_gpu_feature()) {
71 // The timing of this schedule is oddly noisy. Runs
72 // from 0.4ms to 0.5ms on a 2060 RTX. Oddly, the
73 // better runtimes occur when running under nvprof.
74 Var xi, yi;
75 output.compute_root()
76 .reorder(c, x, y)
77 .gpu_tile(x, y, xi, yi, tile_size, tile_size)
78 .bound(c, 0, 3)
79 .unroll(c);
80 ratio.compute_at(output, xi);
81 gray.compute_at(output, x)
82 .tile(x, y, xi, yi, 2, 2)
83 .unroll(xi)
84 .unroll(yi)
85 .gpu_threads(x, y);
86 blur_y.compute_at(output, x)
87 .unroll(x, 2)
88 .gpu_threads(x, y);
89 } else {
90 // 1.93ms on an Intel i9-9960X using 16 threads
91 Var yo, yi;
92 const int vec = natural_vector_size<float>();
93
94 output.split(y, yo, yi, 32)
95 .vectorize(x, vec)
96 .parallel(yo)
97 .reorder(x, c, yi, yo);
98 gray.compute_at(output, yi)
99 .store_at(output, yo)
100 .vectorize(x, vec);
101 blur_y.compute_at(output, yi)
102 .store_at(output, yo)
103 .vectorize(x, vec);
104 ratio.compute_at(output, yi)
105 .store_at(output, yo)
106 .vectorize(x, vec);
107 }
108 }
109 }
110};
