3namespace {
4
5using namespace Halide::ConciseCasts;
6using namespace Halide::BoundaryConditions;
7
8class Max : public Halide::Generator<Max> {
9public:
10 GeneratorParam<int> radius_{"radius", 26};
11 Input<Buffer<float>> input_{"input", 3};
12 Output<Buffer<float>> output_{"output", 3};
13
14 void generate() {
15 Var x("x"), y("y"), c("c"), t("t");
16
17 Func input = repeat_edge(input_,
18 {{input_.dim(0).min(), input_.dim(0).extent()},
19 {input_.dim(1).min(), input_.dim(1).extent()}});
20
21 const int radius = radius_;
22 const int slices = (int)(ceilf(logf(radius) / logf(2))) + 1;
23
24 // A sequence of vertically-max-filtered versions of the input,
25 // each filtered twice as tall as the previous slice. All filters
26 // are downward-looking.
27 Func vert_log("vert_log");
28 vert_log(x, y, c, t) = input(x, y, c);
29 RDom r(-radius, input_.height() + radius, 1, slices - 1);
30 vert_log(x, r.x, c, r.y) = max(vert_log(x, r.x, c, r.y - 1),
31 vert_log(x, r.x + clamp((1 << (r.y - 1)), 0, radius * 2), c, r.y - 1));
32
33 // We're going to take a max filter of arbitrary diameter
34 // by maxing two samples from its floor log 2 (e.g. maxing two
35 // 8-high overlapping samples). This next Func tells us which
36 // slice to draw from for a given radius:
37 Func slice_for_radius("slice_for_radius");
38 slice_for_radius(t) = cast<int>(floor(log(2 * t + 1) / logf(2)));
39
40 // Produce every possible vertically-max-filtered version of the image:
41 Func vert("vert");
42 // t is the blur radius
43 Expr slice = clamp(slice_for_radius(t), 0, slices);
44 Expr first_sample = vert_log(x, y - t, c, slice);
45 Expr second_sample = vert_log(x, y + t + 1 - clamp(1 << slice, 0, 2 * radius), c, slice);
46 vert(x, y, c, t) = max(first_sample, second_sample);
47
48 Func filter_height("filter_height");
49 RDom dy(0, radius + 1);
50 filter_height(x) = sum(select(x * x + dy * dy < (radius + 0.25f) * (radius + 0.25f), 1, 0));
51
52 // Now take an appropriate horizontal max of them at each output pixel
53 RDom dx(-radius, 2 * radius + 1);
54 output_(x, y, c) = maximum(vert(x + dx, y, c, clamp(filter_height(dx), 0, radius + 1)));
55
56 // Estimates (for autoscheduler; ignored otherwise)
57 {
58 input_.dim(0).set_estimate(0, 1536);
59 input_.dim(1).set_estimate(0, 2560);
60 input_.dim(2).set_estimate(0, 3);
61 output_.dim(0).set_estimate(0, 1536);
62 output_.dim(1).set_estimate(0, 2560);
63 output_.dim(2).set_estimate(0, 3);
64 }
65
66 // Schedule
67 if (!auto_schedule) {
68 if (get_target().has_gpu_feature()) {
69 // 11.8ms on a 2060 RTX
70
71 slice_for_radius.compute_root();
72 filter_height.compute_root();
73 Var xi, xo, yi;
74
75 output_.gpu_tile(x, y, xi, yi, 32, 8)
76 .gpu_blocks(x, y, c);
77
78 // There's a boundary condition on the input, so let's
79 // keep things simple with a RoundUp tail strategy.
80 vert_log.compute_root()
81 .reorder(c, t, x, y)
82 .gpu_tile(x, y, xi, yi, 16, 16, TailStrategy::RoundUp)
83 .update()
84 .split(x, xo, xi, 32, TailStrategy::RoundUp)
85 .reorder(r.x, r.y, xi, xo, c)
86 .gpu_blocks(xo, c)
87 .gpu_threads(xi);
88
89 } else {
90 // 47ms on an Intel i9-9960X using 16 threads
91
92 Var tx;
93 // These don't matter, just LUTs
94 slice_for_radius.compute_root();
95 filter_height.compute_root();
96
97 // vert_log.update() doesn't have enough parallelism, but I
98 // can't figure out how to give it more... Split whole image
99 // into slices.
100
101 output_.compute_root()
102 .split(x, tx, x, 256)
103 .reorder(x, y, c, tx)
104 .fuse(c, tx, t)
105 .parallel(t)
106 .vectorize(x, 8);
107 vert_log.compute_at(output_, t);
108 vert_log.vectorize(x, 8);
109 vert_log.update()
110 .reorder(x, r.x, r.y, c)
111 .vectorize(x, 8);
112 vert.compute_at(output_, y)
113 .vectorize(x, 8);
114 }
115 }
116 }
117};

119} // namespace
