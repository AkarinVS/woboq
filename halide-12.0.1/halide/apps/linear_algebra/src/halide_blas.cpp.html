15template<typename T>
16Buffer<T> init_scalar_buffer(T *x) {
17 return Buffer<T>::make_scalar(x);
18}
19
20template<typename T>
21Buffer<T> init_vector_buffer(const int N, T *x, const int incx) {
22 halide_dimension_t shape = {0, N, incx};
23 return Buffer<T>(x, 1, &shape);
24}
25
26template<typename T>
27Buffer<T> init_matrix_buffer(const int M, const int N, T *A, const int lda) {
28 halide_dimension_t shape[] = {{0, M, 1}, {0, N, lda}};
29 return Buffer<T>(A, 2, shape);
30}
