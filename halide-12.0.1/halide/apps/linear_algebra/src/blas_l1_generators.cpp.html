8// Generator class for BLAS axpy operations.
9template<class T>
10class AXPYGenerator : public Generator<AXPYGenerator<T>> {
11public:
12 typedef Generator<AXPYGenerator<T>> Base;
13 using Base::get_target;
14 using Base::natural_vector_size;
15 using Base::target;
16 template<typename T2>
17 using Input = typename Base::template Input<T2>;
18 template<typename T2>
19 using Output = typename Base::template Output<T2>;
20
21 GeneratorParam<bool> vectorize_ = {"vectorize", true};
22 GeneratorParam<int> block_size_ = {"block_size", 1024};
23 GeneratorParam<bool> scale_x_ = {"scale_x", true};
24 GeneratorParam<bool> add_to_y_ = {"add_to_y", true};
25
26 // Standard ordering of parameters in AXPY functions.
27 Input<T> a_ = {"a", 1};
28 Input<Buffer<T>> x_ = {"x", 1};
29 Input<Buffer<T>> y_ = {"y", 1};
30
31 Output<Buffer<T>> result_ = {"result", 1};
32
33 template<class Arg>
34 Expr calc(Arg i) {
35 if (static_cast<bool>(scale_x_) && static_cast<bool>(add_to_y_)) {
36 return a_ * x_(i) + y_(i);
37 } else if (static_cast<bool>(scale_x_)) {
38 return a_ * x_(i);
39 } else {
40 return x_(i);
41 }
42 }
43
44 void generate() {
45 assert(get_target().has_feature(Target::NoBoundsQuery));
46
47 const int vec_size = vectorize_ ? natural_vector_size(type_of<T>()) : 1;
48 Expr size = x_.width();
49 Expr size_vecs = (size / vec_size) * vec_size;
50 Expr size_tail = size - size_vecs;
51
52 Var i("i");
53 RDom vecs(0, size_vecs, "vec");
54 RDom tail(size_vecs, size_tail, "tail");
55 result_(i) = undef(type_of<T>());
56 result_(vecs) = calc(vecs);
57 result_(tail) = calc(tail);
58
59 if (vectorize_) {
60 Var ii("ii");
61 result_.update().vectorize(vecs, vec_size);
62 }
63
64 result_.bound(i, 0, x_.width());
65 result_.dim(0).set_bounds(0, x_.width());
66
67 x_.dim(0).set_min(0);
68 y_.dim(0).set_bounds(0, x_.width());
69 }
70};

72// Generator class for BLAS dot operations.
73template<class T>
74class DotGenerator : public Generator<DotGenerator<T>> {
75public:
76 typedef Generator<DotGenerator<T>> Base;
77 using Base::get_target;
78 using Base::natural_vector_size;
79 using Base::target;
80 template<typename T2>
81 using Input = typename Base::template Input<T2>;
82 template<typename T2>
83 using Output = typename Base::template Output<T2>;
84
85 GeneratorParam<bool> vectorize_ = {"vectorize", true};
86 GeneratorParam<bool> parallel_ = {"parallel", true};
87 GeneratorParam<int> block_size_ = {"block_size", 1024};
88
89 Input<Buffer<T>> x_ = {"x", 1};
90 Input<Buffer<T>> y_ = {"y", 1};
91
92 Output<Buffer<T>> result_ = {"result", 0};
93
94 void generate() {
95 assert(get_target().has_feature(Target::NoBoundsQuery));
96
97 const int vec_size = vectorize_ ? natural_vector_size(type_of<T>()) : 1;
98 Expr size = x_.width();
99 Expr size_vecs = size / vec_size;
100 Expr size_tail = size - size_vecs * vec_size;
101
102 Var i("i");
103 if (vectorize_) {
104 Func dot;
105
106 RDom k(0, size_vecs);
107 dot(i) += x_(k * vec_size + i) * y_(k * vec_size + i);
108
109 RDom lanes(0, vec_size);
110 RDom tail(size_vecs * vec_size, size_tail);
111 result_() = sum(dot(lanes));
112 result_() += sum(x_(tail) * y_(tail));
113
114 dot.compute_root().vectorize(i);
115 dot.update(0).vectorize(i);
116 } else {
117 RDom k(0, size);
118 result_() = sum(x_(k) * y_(k));
119 }
120
121 x_.dim(0).set_bounds(0, size);
122 y_.dim(0).set_bounds(0, size);
123 }
124};

140 GeneratorParam<bool> parallel_ = {"parallel", true};

144
145 Output<Buffer<T>> result_ = {"result", 0};
146
147 void generate() {

149
150 const int vec_size = vectorize_ ? natural_vector_size(type_of<T>()) : 1;
151 Expr size = x_.width();
152 Expr size_vecs = size / vec_size;
153 Expr size_tail = size - size_vecs * vec_size;
154
155 Var i("i");
156 if (vectorize_) {
157 Func norm;
158
159 RDom k(0, size_vecs);
160 norm(i) += abs(x_(k * vec_size + i));
161
162 RDom lanes(0, vec_size);
163 RDom tail(size_vecs * vec_size, size_tail);
164 result_() = sum(norm(lanes));
165 result_() += sum(abs(x_(tail)));
166
167 norm.compute_root().vectorize(i);
168 norm.update(0).vectorize(i);
169 } else {
170 RDom k(0, x_.width());
171 result_() = sum(abs(x_(k)));
172 }
173
174 x_.dim(0).set_min(0);
175 }
