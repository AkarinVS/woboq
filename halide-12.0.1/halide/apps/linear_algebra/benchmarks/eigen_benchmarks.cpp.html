1// USAGE: eigen_benchmarks <subroutine> <size>
2//
3// Benchmarks BLAS subroutines using Eigen's implementation. Will
4// construct random size x size matrices and/or size x 1 vectors
5// to test the subroutine with.
6//
7// Accepted values for subroutine are:
8// L1: scal, copy, axpy, dot, nrm2
9// L2: gemv_notrans, gemv_trans
10// L3: gemm_notrans, gemm_trans_A, gemm_trans_B, gemm_trans_AB
11//
12
13#include "clock.h"
14#include "macros.h"
15#include <Eigen/Eigen>
16#include <iomanip>
17#include <iostream>
18#include <string>
19
20template<class T>
21std::string type_name();
22
23template<>
24std::string type_name<float>() {
25 return "s";
26}
27
28template<>
29std::string type_name<double>() {
30 return "d";
31}
32
33struct BenchmarksBase {
34 virtual void bench_copy(int N) = 0;
35 virtual void bench_scal(int N) = 0;
36 virtual void bench_axpy(int N) = 0;
37 virtual void bench_dot(int N) = 0;
38 virtual void bench_asum(int N) = 0;
39 virtual void bench_gemv_notrans(int N) = 0;
40 virtual void bench_gemv_trans(int N) = 0;
41 virtual void bench_ger(int N) = 0;
42 virtual void bench_gemm_notrans(int N) = 0;
43 virtual void bench_gemm_transA(int N) = 0;
44 virtual void bench_gemm_transB(int N) = 0;
45 virtual void bench_gemm_transAB(int N) = 0;
46};
47
48template<class T>
49struct Benchmarks : BenchmarksBase {
50 typedef T Scalar;
51 typedef Eigen::Matrix<T, Eigen::Dynamic, 1> Vector;
52 typedef Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Matrix;
53
54 Scalar random_scalar() {
55 Vector x(1);
56 x.setRandom();
57 return x[0];
58 }
59
60 Vector random_vector(int N) {
61 Vector x(N);
62 x.setRandom();
63 return x;
64 }
65
66 Matrix random_matrix(int N) {
67 Matrix A(N, N);
68 A.setRandom();
69 return A;
70 }
71
72 Benchmarks(std::string n)
73 : name(n) {
74 }
75
76 void run(std::string benchmark, int size) {
77 if (benchmark == "copy") {
78 bench_copy(size);
79 } else if (benchmark == "scal") {
80 bench_scal(size);
81 } else if (benchmark == "axpy") {
82 bench_axpy(size);
83 } else if (benchmark == "dot") {
84 bench_dot(size);
85 } else if (benchmark == "asum") {
86 bench_asum(size);
87 } else if (benchmark == "gemv_notrans") {
88 bench_gemv_notrans(size);
89 } else if (benchmark == "gemv_trans") {
90 bench_gemv_trans(size);
91 } else if (benchmark == "ger") {
92 bench_ger(size);
93 } else if (benchmark == "gemm_notrans") {
94 bench_gemm_notrans(size);
95 } else if (benchmark == "gemm_transA") {
96 bench_gemm_transA(size);
97 } else if (benchmark == "gemm_transB") {
98 bench_gemm_transB(size);
99 } else if (benchmark == "gemm_transAB") {
100 bench_gemm_transAB(size);
101 }
102 }
103
104 Scalar result;
105
106 L1Benchmark(copy, type_name<T>(), y = x);
107 L1Benchmark(scal, type_name<T>(), x = alpha * x);
108 L1Benchmark(axpy, type_name<T>(), y = alpha * x + y);
109 L1Benchmark(dot, type_name<T>(), result = x.dot(y));
110 L1Benchmark(asum, type_name<T>(), result = x.array().abs().sum());
111
112 L2Benchmark(gemv_notrans, type_name<T>(), y = alpha * A * x + beta * y);
113 L2Benchmark(gemv_trans, type_name<T>(), y = alpha * A.transpose() * x + beta * y);
114 L2Benchmark(ger, type_name<T>(), A = alpha * x * y.transpose() + A);
115
116 L3Benchmark(gemm_notrans, type_name<T>(), C = alpha * A * B + beta * C);
117 L3Benchmark(gemm_transA, type_name<T>(), C = alpha * A.transpose() * B + beta * C);
118 L3Benchmark(gemm_transB, type_name<T>(), C = alpha * A * B.transpose() + beta * C);
119 L3Benchmark(gemm_transAB, type_name<T>(), C = alpha * A.transpose() * B.transpose() + beta * C);
120
121private:
122 std::string name;
123};
124
125int main(int argc, char *argv[]) {
126 if (argc != 3) {
127 std::cout << "USAGE: eigen_benchmarks <subroutine> <size>\n";
128 return 0;
129 }
130
131 std::string subroutine = argv[1];
132 char type = subroutine[0];
133 int size = std::stoi(argv[2]);
134
135 subroutine = subroutine.substr(1);
136 if (type == 's') {
137 Benchmarks<float>("Eigen").run(subroutine, size);
138 } else if (type == 'd') {
139 Benchmarks<double>("Eigen").run(subroutine, size);
140 }
141
142 return 0;
143}
