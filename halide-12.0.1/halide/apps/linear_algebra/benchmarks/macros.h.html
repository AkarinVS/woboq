10inline void set_math_flags() {
11#ifdef ENABLE_FTZ_DAZ
12
13#if defined(__i386__) || defined(__x86_64__)
14 // Flush denormals to zero (the FTZ flag).
15 _MM_SET_FLUSH_ZERO_MODE(_MM_FLUSH_ZERO_ON);
16 // Interpret denormal inputs as zero (the DAZ flag).
17 _MM_SET_DENORMALS_ZERO_MODE(_MM_DENORMALS_ZERO_ON);
18#endif // defined(__i386__) || defined(__x86_64__)
19
20#if defined(__arm__) || defined(__aarch64__)
21 intptr_t fpsr = 0;
22
23 // Get the FP status register
24#if defined(__aarch64__)
25 asm volatile("mrs %0, fpcr"
26 : "=r"(fpsr));
27#else
28 asm volatile("vmrs %0, fpscr"
29 : "=r"(fpsr));
30#endif
31
32 // Setting this is like setting FTZ+DAZ on x86
33 constexpr intptr_t flush_to_zero = (1 << 24 /* FZ */);
34 fpsr |= flush_to_zero;
35
36 // Set the FP status register
37#if defined(__aarch64__)
38 asm volatile("msr fpcr, %0"
39 :
40 : "ri"(fpsr));
41#else
42 asm volatile("vmsr fpscr, %0"
43 :
44 : "ri"(fpsr));
45#endif
46
47#endif // defined(__arm__) || defined(__aarch64__)
48
49#endif // ENABLE_FTZ_DAZ
50}

52#define time_it(code) \
53 set_math_flags(); \
54 double elapsed = 0; \
55 for (int iters = 1;; iters *= 2) { \
56 /* Best of 5 */ \
57 elapsed = 1e6 * Halide::Tools::benchmark(5, iters, [&]() { code; }); \
58 /* spend at least 5x20ms benchmarking */ \
59 if (elapsed * iters > 20000) { \
60 break; \
61 } \
62 }
