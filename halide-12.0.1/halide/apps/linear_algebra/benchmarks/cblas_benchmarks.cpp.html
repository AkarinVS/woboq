1// USAGE: halide_benchmarks <subroutine> <size>
2//
3// Benchmarks BLAS subroutines using Halide's implementation. Will
4// construct random size x size matrices and/or size x 1 vectors
5// to test the subroutine with.
6//
7// Accepted values for subroutine are:
8// L1: scal, copy, axpy, dot, nrm2
9// L2: gemv_notrans, gemv_trans
10// L3: gemm_notrans, gemm_transA, gemm_transB, gemm_transAB
11//
12

25template<class T>
26struct BenchmarksBase {
27 typedef T Scalar;
28 typedef std::vector<T> Vector;
29 typedef std::vector<T> Matrix;
30
31 std::random_device rand_dev;
32 std::default_random_engine rand_eng{rand_dev()};
33
34 std::string name;
35
36 Scalar random_scalar() {
37 std::uniform_real_distribution<T> uniform_dist(0.0, 1.0);
38 return uniform_dist(rand_eng);
39 }
40
41 Vector random_vector(int N) {
42 Vector buff(N);
43 for (int i = 0; i < N; ++i) {
44 buff[i] = random_scalar();
45 }
46 return buff;
47 }
48
49 Matrix random_matrix(int N) {
50 Matrix buff(N * N);
51 for (int i = 0; i < N * N; ++i) {
52 buff[i] = random_scalar();
53 }
54 return buff;
55 }
56
57 BenchmarksBase(std::string n)
58 : name(n) {
59 }
60
61 void run(std::string benchmark, int size) {
62 if (benchmark == "copy") {
63 bench_copy(size);
64 } else if (benchmark == "scal") {
65 bench_scal(size);
66 } else if (benchmark == "axpy") {
67 bench_axpy(size);
68 } else if (benchmark == "dot") {
69 bench_dot(size);
70 } else if (benchmark == "asum") {
71 bench_asum(size);
72 } else if (benchmark == "gemv_notrans") {
73 this->bench_gemv_notrans(size);
74 } else if (benchmark == "gemv_trans") {
75 this->bench_gemv_trans(size);
76 } else if (benchmark == "ger") {
77 this->bench_ger(size);
78 } else if (benchmark == "gemm_notrans") {
79 this->bench_gemm_notrans(size);
80 } else if (benchmark == "gemm_transA") {
81 this->bench_gemm_transA(size);
82 } else if (benchmark == "gemm_transB") {
83 this->bench_gemm_transB(size);
84 } else if (benchmark == "gemm_transAB") {
85 this->bench_gemm_transAB(size);
86 }
87 }
88
89 virtual void bench_copy(int N) = 0;
90 virtual void bench_scal(int N) = 0;
91 virtual void bench_axpy(int N) = 0;
92 virtual void bench_dot(int N) = 0;
93 virtual void bench_asum(int N) = 0;
94 virtual void bench_gemv_notrans(int N) = 0;
95 virtual void bench_gemv_trans(int N) = 0;
96 virtual void bench_ger(int N) = 0;
97 virtual void bench_gemm_notrans(int N) = 0;
98 virtual void bench_gemm_transA(int N) = 0;
99 virtual void bench_gemm_transB(int N) = 0;
100 virtual void bench_gemm_transAB(int N) = 0;
101};

159int main(int argc, char *argv[]) {
160 if (argc != 3) {
161 std::cout << "USAGE: cblas_benchmarks <subroutine> <size>\n";
162 return 0;
163 }
164
165 std::string subroutine = argv[1];
166 char type = subroutine[0];
167 int size = std::stoi(argv[2]);
168
169 subroutine = subroutine.substr(1);
170 if (type == 's') {
171 BenchmarksFloat(BLAS_NAME).run(subroutine, size);
172 } else if (type == 'd') {
173 BenchmarksDouble(BLAS_NAME).run(subroutine, size);
174 }
175
176 return 0;
177}
