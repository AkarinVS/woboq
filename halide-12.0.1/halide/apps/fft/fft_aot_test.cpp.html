14namespace {
15const float kPi = 3.14159265358979310000f;
16
17const int32_t kSize = 16;
18} // namespace

20using Halide::Runtime::Buffer;
21
22Buffer<float, 3> real_buffer(int32_t y_size = kSize) {
23 return Buffer<float, 3>::make_interleaved(kSize, y_size, 1);
24}
25
26Buffer<float, 3> complex_buffer(int32_t y_size = kSize) {
27 return Buffer<float, 3>::make_interleaved(kSize, y_size, 2);
28}
29
30float &re(Buffer<float, 3> &b, int x, int y) {
31 return b(x, y, 0);
32}
33
34float &im(Buffer<float, 3> &b, int x, int y) {
35 return b(x, y, 1);
36}
37
38float re(const Buffer<float, 3> &b, int x, int y) {
39 return b(x, y, 0);
40}
41
42float im(const Buffer<float, 3> &b, int x, int y) {
43 return b(x, y, 1);
44}

46int main(int argc, char **argv) {
47 std::cout << std::fixed << std::setprecision(2);
48
49 // Forward real to complex test.
50 {
51 std::cout << "Forward real to complex test.\n";
52
53 float signal_1d[kSize];
54 for (size_t i = 0; i < kSize; i++) {
55 signal_1d[i] = 0;
56 for (size_t k = 1; k < 5; k++) {
57 signal_1d[i] += cos(2 * kPi * (k * (i / (float)kSize) + (k / 16.0f)));
58 }
59 }
60
61 auto in = real_buffer();
62 for (int j = 0; j < kSize; j++) {
63 for (int i = 0; i < kSize; i++) {
64 in(i, j) = signal_1d[i] + signal_1d[j];
65 }
66 }
67
68 auto out = complex_buffer(kSize / 2 + 1);
69
70 int halide_result;
71 halide_result = fft_forward_r2c(in, out);
72 if (halide_result != 0) {
73 std::cerr << "fft_forward_r2c failed returning " << halide_result << "\n";
74 exit(1);
75 }
76
77 for (size_t i = 1; i < 5; i++) {
78 // Check horizontal bins
79 float real = re(out, i, 0);
80 float imaginary = im(out, i, 0);
81 float magnitude = sqrt(real * real + imaginary * imaginary);
82 if (fabs(magnitude - .5f) > .001) {
83 std::cerr << "fft_forward_r2c bad magnitude for horizontal bin " << i << ":" << magnitude << "\n";
84 exit(1);
85 }
86 float phase_angle = atan2(imaginary, real);
87 if (fabs(phase_angle - (i / 16.0f) * 2 * kPi) > .001) {
88 std::cerr << "fft_forward_r2c bad phase angle for horizontal bin " << i << ": " << phase_angle << "\n";
89 exit(1);
90 }
91 // Check vertical bins
92 real = re(out, 0, i);
93 imaginary = im(out, 0, i);
94 magnitude = sqrt(real * real + imaginary * imaginary);
95 if (fabs(magnitude - .5f) > .001) {
96 std::cerr << "fft_forward_r2c bad magnitude for vertical bin " << i << ":" << magnitude << "\n";
97 exit(1);
98 }
99 phase_angle = atan2(imaginary, real);
100 if (fabs(phase_angle - (i / 16.0f) * 2 * kPi) > .001) {
101 std::cerr << "fft_forward_r2c bad phase angle for vertical bin " << i << ": " << phase_angle << "\n";
102 exit(1);
103 }
104 }
105
106 // Check all other components are close to zero.
107 for (size_t j = 0; j < kSize / 2 + 1; j++) {
108 for (size_t i = 0; i < kSize; i++) {
109 // The first four non-DC bins in x and y have non-zero
110 // values. The horizontal ones are mirrored into the
111 // negative frequency components as well.
112 if (!((j == 0 && ((i > 0 && i < 5) || (i > kSize - 5))) ||
113 (i == 0 && j > 0 && j < 5))) {
114 float real = re(out, i, j);
115 float imaginary = im(out, i, j);
116 if (fabs(real) > .001) {
117 std::cerr << "fft_forward_r2c real component at (" << i << ", " << j << ") is non-zero: " << real << "\n";
118 exit(1);
119 }
120 if (fabs(imaginary) > .001) {
121 std::cerr << "fft_forward_r2c imaginary component at (" << i << ", " << j << ") is non-zero: " << imaginary << "\n";
122 exit(1);
123 }
124 }
125 }
126 }
127 }
128
129 // Inverse complex to real test.
130 {
131 std::cout << "Inverse complex to real test.\n";
132
133 auto in = complex_buffer();
134 in.fill(0);
135
136 // There are four components that get summed to form the magnitude, which we want to be 1.
137 // The components are each of the positive and negative frequencies and each of the
138 // real and complex components. The +/- frequencies sum algebraically and the complex
139 // components contribute to the magnitude as the sides of triangle like any 2D vector.
140 float term_magnitude = 1.0f / (2.0f * sqrt(2.0f));
141 re(in, 1, 0) = term_magnitude;
142 im(in, 1, 0) = term_magnitude;
143 // Negative frequencies count backward from end, no DC term
144 re(in, kSize - 1, 0) = term_magnitude;
145 im(in, kSize - 1, 0) = -term_magnitude; // complex conjugate
146
147 auto out = real_buffer();
148
149 int halide_result;
150 halide_result = fft_inverse_c2r(in, out);
151 if (halide_result != 0) {
152 std::cerr << "fft_inverse_c2r failed returning " << halide_result << "\n";
153 exit(1);
154 }
155
156 for (size_t j = 0; j < kSize; j++) {
157 for (size_t i = 0; i < kSize; i++) {
158 float sample = out(i, j);
159 float expected = cos(2 * kPi * (i / 16.0f + .125f));
160 if (fabs(sample - expected) > .001) {
161 std::cerr << "fft_inverse_c2r mismatch at (" << i << ", " << j << ") " << sample << " vs. " << expected << "\n";
162 exit(1);
163 }
164 }
165 }
166 }
167
168 // Forward complex to complex test.
169 {
170 std::cout << "Forward complex to complex test.\n";
171
172 auto in = complex_buffer();
173
174 float signal_1d_real[kSize];
175 float signal_1d_complex[kSize];
176 for (size_t i = 0; i < kSize; i++) {
177 signal_1d_real[i] = 0;
178 signal_1d_complex[i] = 0;
179 for (size_t k = 1; k < 5; k++) {
180 signal_1d_real[i] += cos(2 * kPi * (k * (i / (float)kSize) + (k / 16.0f)));
181 signal_1d_complex[i] += sin(2 * kPi * (k * (i / (float)kSize) + (k / 16.0f)));
182 }
183 }
184
185 for (int j = 0; j < kSize; j++) {
186 for (int i = 0; i < kSize; i++) {
187 re(in, i, j) = signal_1d_real[i] + signal_1d_real[j];
188 im(in, i, j) = signal_1d_complex[i] + signal_1d_complex[j];
189 }
190 }
191
192 auto out = complex_buffer();
193
194 int halide_result;
195 halide_result = fft_forward_c2c(in, out);
196 if (halide_result != 0) {
197 std::cerr << "fft_forward_c2c failed returning " << halide_result << "\n";
198 exit(1);
199 }
200
201 for (size_t i = 1; i < 5; i++) {
202 // Check horizontal bins
203 float real = re(out, i, 0);
204 float imaginary = im(out, i, 0);
205 float magnitude = sqrt(real * real + imaginary * imaginary);
206 if (fabs(magnitude - 1.0f) > .001) {
207 std::cerr << "fft_forward_c2c bad magnitude for horizontal bin " << i << ":" << magnitude << "\n";
208 exit(1);
209 }
210 float phase_angle = atan2(imaginary, real);
211 if (fabs(phase_angle - (i / 16.0f) * 2 * kPi) > .001) {
212 std::cerr << "fft_forward_c2c bad phase angle for horizontal bin " << i << ": " << phase_angle << "\n";
213 exit(1);
214 }
215 // Check vertical bins
216 real = re(out, 0, i);
217 imaginary = im(out, 0, i);
218 magnitude = sqrt(real * real + imaginary * imaginary);
219 if (fabs(magnitude - 1.0f) > .001) {
220 std::cerr << "fft_forward_c2c bad magnitude for vertical bin " << i << ":" << magnitude << "\n";
221 exit(1);
222 }
223 phase_angle = atan2(imaginary, real);
224 if (fabs(phase_angle - (i / 16.0f) * 2 * kPi) > .001) {
225 std::cerr << "fft_forward_c2c bad phase angle for vertical bin " << i << ": " << phase_angle << "\n";
226 exit(1);
227 }
228 }
229
230 // Check all other components are close to zero.
231 for (size_t j = 0; j < kSize; j++) {
232 for (size_t i = 0; i < kSize; i++) {
233 // The first four non-DC bins in x and y have non-zero
234 // values. The input is chose so the mirrored negative
235 // frequency components are all zero due to
236 // interference of the real and complex parts.
237 if (!((j == 0 && (i > 0 && i < 5)) ||
238 (i == 0 && j > 0 && j < 5))) {
239 float real = re(out, i, j);
240 float imaginary = im(out, i, j);
241 if (fabs(real) > .001) {
242 std::cerr << "fft_forward_c2c real component at (" << i << ", " << j << ") is non-zero: " << real << "\n";
243 exit(1);
244 }
245 if (fabs(imaginary) > .001) {
246 std::cerr << "fft_forward_c2c imaginary component at (" << i << ", " << j << ") is non-zero: " << imaginary << "\n";
247 exit(1);
248 }
249 }
250 }
251 }
252 }
253
254 // Inverse complex to complex test.
255 {
256 std::cout << "Inverse complex to complex test.\n";
257
258 auto in = complex_buffer();
259 in.fill(0);
260
261 re(in, 1, 0) = .5f;
262 im(in, 1, 0) = .5f;
263 re(in, kSize - 1, 0) = .5f;
264 im(in, kSize - 1, 0) = .5f; // Not conjugate. Result will not be real
265
266 auto out = complex_buffer();
267
268 int halide_result;
269 halide_result = fft_inverse_c2c(in, out);
270 if (halide_result != 0) {
271 std::cerr << "fft_inverse_c2c failed returning " << halide_result << "\n";
272 exit(1);
273 }
274
275 for (size_t j = 0; j < kSize; j++) {
276 for (size_t i = 0; i < kSize; i++) {
277 float real_sample = re(out, i, j);
278 float imaginary_sample = im(out, i, j);
279 float real_expected = 1 / sqrt(2) * (cos(2 * kPi * (i / 16.0f + .125)) + cos(2 * kPi * (i * (kSize - 1) / 16.0f + .125)));
280 float imaginary_expected = 1 / sqrt(2) * (sin(2 * kPi * (i / 16.0f + .125)) + sin(2 * kPi * (i * (kSize - 1) / 16.0f + .125)));
281
282 if (fabs(real_sample - real_expected) > .001) {
283 std::cerr << "fft_inverse_c2c real mismatch at (" << i << ", " << j << ") " << real_sample << " vs. " << real_expected << "\n";
284 exit(1);
285 }
286
287 if (fabs(imaginary_sample - imaginary_expected) > .001) {
288 std::cerr << "fft_inverse_c2c imaginary mismatch at (" << i << ", " << j << ") " << imaginary_sample << " vs. " << imaginary_expected << "\n";
289 exit(1);
290 }
291 }
292 }
293 }
294
295 std::cout << "Success!\n";
296 exit(0);
