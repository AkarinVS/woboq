8// Complex number expression in Halide. This maps complex number Tuples
9// to a type we can use for function overloading (especially operator
10// overloading).
11struct ComplexExpr {
12 Halide::Expr x, y;
13
14 explicit ComplexExpr(Halide::Tuple z)
15 : x(z[0]), y(z[1]) {
16 }
17
18 // A default constructed complex number is zero.
19 ComplexExpr()
20 : x(0.0f), y(0.0f) {
21 }
22 ComplexExpr(float x, float y)
23 : x(x), y(y) {
24 }
25 // This constructor will implicitly convert a real number (either
26 // Halide::Expr or constant float) to a complex number with an
27 // imaginary part of zero.
28 ComplexExpr(Halide::Expr x, Halide::Expr y = 0.0f)
29 : x(x), y(y) {
30 }
31
32 Halide::Expr re() {
33 return x;
34 }
35 Halide::Expr im() {
36 return y;
37 }
38
39 operator Halide::Tuple() const {
40 return Halide::Tuple(x, y);
41 }
42
43 ComplexExpr &operator+=(ComplexExpr r) {
44 x += r.x;
45 y += r.y;
46 return *this;
47 }
48};

50// Define a typed Func for complex numbers.
51typedef FuncT<ComplexExpr> ComplexFunc;

53// Function style real/imaginary part of a complex number (and real
54// numbers too).
55inline Halide::Expr re(ComplexExpr z) {
56 return z.re();
57}
58inline Halide::Expr im(ComplexExpr z) {
59 return z.im();
60}
61inline Halide::Expr re(Halide::Expr x) {
62 return x;
63}
64inline Halide::Expr im(Halide::Expr x) {
65 return 0.0f;
66}
67
68inline ComplexExpr conj(ComplexExpr z) {
69 return ComplexExpr(re(z), -im(z));
70}
71
72// Unary negation.
73inline ComplexExpr operator-(ComplexExpr z) {
74 return ComplexExpr(-re(z), -im(z));
75}
76// Complex arithmetic.
77inline ComplexExpr operator+(ComplexExpr a, ComplexExpr b) {
78 return ComplexExpr(re(a) + re(b), im(a) + im(b));
79}
80inline ComplexExpr operator+(ComplexExpr a, Halide::Expr b) {
81 return ComplexExpr(re(a) + b, im(a));
82}
83inline ComplexExpr operator+(Halide::Expr a, ComplexExpr b) {
84 return ComplexExpr(a + re(b), im(b));
85}
86inline ComplexExpr operator-(ComplexExpr a, ComplexExpr b) {
87 return ComplexExpr(re(a) - re(b), im(a) - im(b));
88}
89inline ComplexExpr operator-(ComplexExpr a, Halide::Expr b) {
90 return ComplexExpr(re(a) - b, im(a));
91}
92inline ComplexExpr operator-(Halide::Expr a, ComplexExpr b) {
93 return ComplexExpr(a - re(b), -im(b));
94}
95inline ComplexExpr operator*(ComplexExpr a, ComplexExpr b) {
96 return ComplexExpr(re(a) * re(b) - im(a) * im(b),
97 re(a) * im(b) + im(a) * re(b));
98}
99inline ComplexExpr operator*(ComplexExpr a, Halide::Expr b) {
100 return ComplexExpr(re(a) * b, im(a) * b);
101}
102inline ComplexExpr operator*(Halide::Expr a, ComplexExpr b) {
103 return ComplexExpr(a * re(b), a * im(b));
104}
105inline ComplexExpr operator/(ComplexExpr a, Halide::Expr b) {
106 return ComplexExpr(re(a) / b, im(a) / b);
107}

109// Compute exp(j*x)
