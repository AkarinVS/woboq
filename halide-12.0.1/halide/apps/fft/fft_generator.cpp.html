9enum class FFTNumberType { Real,
10 Complex };
11std::map<std::string, FFTNumberType> fft_number_type_enum_map() {
12 return {{"real", FFTNumberType::Real},
13 {"complex", FFTNumberType::Complex}};
14}

16// Direction of FFT. Samples can be read as "time" or "spatial" depending
17// on the meaning of the input domain.
18enum class FFTDirection { SamplesToFrequency,
19 FrequencyToSamples };
20std::map<std::string, FFTDirection> fft_direction_enum_map() {
21 return {{"samples_to_frequency", FFTDirection::SamplesToFrequency},
22 {"frequency_to_samples", FFTDirection::FrequencyToSamples}};
23}
24
25class FFTGenerator : public Halide::Generator<FFTGenerator> {
26public:
27 // Gain to apply to the FFT. This is folded into gains already
28 // being applied to the FFT. A gain of 1.0f indicates an
29 // unnormalized FFT. 1 / sqrt(N) gives a unitary transform such that
30 // forward and inverse operations have the same gain without changing
31 // signal magnitude.
32 // A common convention is 1/N for the forward direction and 1 for the
33 // inverse.
34 // "N" above is the size of the input, which is the product of
35 // the dimensions.
36 GeneratorParam<float> gain{"gain", 1.0f};
37
38 // The following option specifies that a particular vector width should be
39 // used when the vector width can change the results of the FFT.
40 // Some parts of the FFT algorithm use the vector width to change the way
41 // floating point operations are ordered and grouped, which causes the results
42 // to vary with respect to the target architecture. Setting this option forces
43 // such stages to use the specified vector width (independent of the actual
44 // architecture's vector width), which eliminates the architecture specific
45 // behavior.
46 GeneratorParam<int32_t> vector_width{"vector_width", 0};
47
48 // The following option indicates that the FFT should parallelize within a
49 // single FFT. This only makes sense to use on large FFTs, and generally only
50 // if there is no outer loop around FFTs that can be parallelized.
51 GeneratorParam<bool> parallel{"parallel", false};
52
53 // Indicates forward or inverse Fourier transform --
54 // "samples_to_frequency" maps to a forward FFT. (Other packages sometimes call this a sign of -1)
55 // "frequency_to_samples" maps to a backward FFT. (Other packages sometimes call this a sign of +1)
56 GeneratorParam<FFTDirection> direction{"direction", FFTDirection::SamplesToFrequency,
57 fft_direction_enum_map()};
58
59 // Whether the input is "real" or "complex".
60 GeneratorParam<FFTNumberType> input_number_type{"input_number_type",
61 FFTNumberType::Real, fft_number_type_enum_map()};
62 // Whether the output is "real" or "complex".
63 GeneratorParam<FFTNumberType> output_number_type{"output_number_type",
64 FFTNumberType::Real, fft_number_type_enum_map()};
65
66 // Size of first dimension, required to be greater than zero.
67 GeneratorParam<int32_t> size0{"size0", 1};
68 // Size of second dimension, may be zero for 1D FFT.
69 GeneratorParam<int32_t> size1{"size1", 0};
70 // TODO(zalman): Add support for 3D and maybe 4D FFTs
71
72 // The input buffer. Must be separate from the output.
73 // Only Float(32) is supported.
74 //
75 // For a real input FFT, this should have the following shape:
76 // Dim0: extent = size0, stride = 1
77 // Dim1: extent = size1 / 2 - 1, stride = size0
78 // Dim2: extent = 1, stride = 1
79 //
80 // For a complex input FFT, this should have the following shape:
81 // Dim0: extent = size0, stride = 2
82 // Dim1: extent = size1, stride = size0 * 2
83 // Dim2: extent = 2, stride = 1 (real followed by imaginary components)
84 Input<Buffer<float>> input{"input", 3};
85 Output<Buffer<float>> output{"output", 3};
86
87 void generate() {
88 _halide_user_assert(size0 > 0) << "FFT must be at least 1D\n";
89
90 Fft2dDesc desc;
91
92 desc.gain = gain;
93 desc.vector_width = vector_width;
94 desc.parallel = parallel;
95
96 // The logic below calls the specialized r2c or c2r version if
97 // applicable to take advantage of better scheduling. It is
98 // assumed that projecting a real Func to a ComplexFunc and
99 // immediately back has zero cost.
100
101 const int sign = (direction == FFTDirection::SamplesToFrequency) ? -1 : 1;
102
103 if (input_number_type == FFTNumberType::Real) {
104 if (direction == FFTDirection::SamplesToFrequency) {
105 // TODO: Not sure why this is necessary as ImageParam
106 // -> Func conversion should happen, It may not work
107 // with implicit dimension (use of _) logic in FFT.
108 Func in;
109 in(x, y) = input(x, y, 0);
110
111 complex_result = fft2d_r2c(in, size0, size1, target, desc);
112 } else {
113 ComplexFunc in;
114 in(x, y) = ComplexExpr(input(x, y, 0), 0);
115
116 complex_result = fft2d_c2c(in, size0, size1, sign, target, desc);
117 }
118 } else {
119 ComplexFunc in;
120 in(x, y) = ComplexExpr(input(x, y, 0), input(x, y, 1));
121 if (output_number_type == FFTNumberType::Real &&
122 direction == FFTDirection::FrequencyToSamples) {
123 real_result = fft2d_c2r(in, size0, size1, target, desc);
124 } else {
125 complex_result = fft2d_c2c(in, size0, size1, sign, target, desc);
126 }
127 }
128
129 if (output_number_type == FFTNumberType::Real) {
130 if (real_result.defined()) {
131 output(x, y, c) = real_result(x, y);
132 } else {
133 output(x, y, c) = re(complex_result(x, y));
134 }
135 } else {
136 output(x, y, c) = mux(c, {re(complex_result(x, y)), im(complex_result(x, y))});
137 }
138 }
139
140 void schedule() {
141 const int input_comps = (input_number_type == FFTNumberType::Real) ? 1 : 2;
142 const int output_comps = (output_number_type == FFTNumberType::Real) ? 1 : 2;
143
144 input.dim(0).set_stride(input_comps);
145 input.dim(2).set_min(0).set_extent(input_comps).set_stride(1);
146
147 output.dim(0).set_stride(output_comps);
148 output.dim(2).set_min(0).set_extent(output_comps).set_stride(1);
149
150 if (output_comps != 1) {
151 output.reorder(c, x, y).unroll(c);
152 }
153
154 if (real_result.defined()) {
155 real_result.compute_at(output, Var::outermost());
156 } else {
157 assert(complex_result.defined());
158 complex_result.compute_at(output, Var::outermost());
159 }
160 }
161
162private:
163 Var x{"x"}, y{"y"}, c{"c"};
164 Func real_result;
165 ComplexFunc complex_result;
166};
