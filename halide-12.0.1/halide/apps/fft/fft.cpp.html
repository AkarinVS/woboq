43int gcd(int x, int y) {
44 while (y != 0) {
45 int r = x % y;
46 x = y;
47 y = r;
48 }
49 return x;
50}
51
52int lcm(int x, int y) {
53 return std::min(x, y) * (std::max(x, y) / gcd(x, y));
54}

56// Compute the product of the integers in R.
57int product(const vector<int> &R) {
58 int p = 1;
59 for (size_t i = 0; i < R.size(); i++) {
60 p *= R[i];
61 }
62 return p;
63}

65// These tersely named functions concatenate vectors of Var/Expr for use
66// in generating argument lists to Halide functions. They are named to avoid
67// bloating the code, since these are used extremely frequently, and often many
68// times within one line.
69vector<Var> A(vector<Var> l, const vector<Var> &r) {
70 for (const Var &i : r) {
71 l.push_back(i);
72 }
73 return l;
74}

76template<typename T>
77vector<Expr> A(vector<Expr> l, const vector<T> &r) {
78 for (const Var &i : r) {
79 l.push_back(i);
80 }
81 return l;

84// Get call references to the first N elements of dimension dim of x. If temps
85// is set, grab references to elements [-N, -1] instead.
86typedef FuncRefT<ComplexExpr> ComplexFuncRef;
87vector<ComplexFuncRef> get_func_refs(ComplexFunc x, int N, bool temps = false) {
88 vector<Var> args(x.args());
89 args.erase(args.begin());
90
91 vector<ComplexFuncRef> refs;
92 for (int i = 0; i < N; i++) {
93 if (temps) {
94 refs.push_back(x(A({Expr(-i - 1)}, args)));
95 } else {
96 refs.push_back(x(A({Expr(i)}, args)));
97 }
98 }
99 return refs;
100}

107// Specializations for some small DFTs of the first dimension of a
108// Func f.
109ComplexFunc dft2(ComplexFunc f, const string &prefix) {
110 Type type = f.output_types()[0];
111
112 ComplexFunc F(prefix + "X2");
113 F(f.args()) = undef_z(type);
114
115 vector<ComplexFuncRef> x = get_func_refs(f, 2);
116 vector<ComplexFuncRef> X = get_func_refs(F, 2);
117
118 X[0] = x[0] + x[1];
119 X[1] = x[0] - x[1];
120
121 return F;
122}

124ComplexFunc dft4(ComplexFunc f, int sign, const string &prefix) {
125 Type type = f.output_types()[0];
126
127 ComplexFunc F(prefix + "X4");
128 F(f.args()) = undef_z(type);
129
130 vector<ComplexFuncRef> x = get_func_refs(f, 4);
131 vector<ComplexFuncRef> X = get_func_refs(F, 4);
132 vector<ComplexFuncRef> T = get_func_refs(F, 2, true);
133 // We can re-use these two temps. T[0], T[2] and T[1], T[3] do not have
134 // overlapping lifetime.
135 T.push_back(T[1]);
136 T.push_back(T[0]);
137
138 T[0] = (x[0] + x[2]);
139 T[2] = (x[1] + x[3]);
140 X[0] = (T[0] + T[2]);
141 X[2] = (T[0] - T[2]);
142
143 T[1] = (x[0] - x[2]);
144 T[3] = (x[1] - x[3]) * j * sign;
145 X[1] = (T[1] + T[3]);
146 X[3] = (T[1] - T[3]);
147
148 return F;
149}

151ComplexFunc dft6(ComplexFunc f, int sign, const string &prefix) {
152 const float re_W1_3 = -0.5f;
153 const float im_W1_3 = sign * 0.866025404f;
154
155 ComplexExpr W1_3(re_W1_3, im_W1_3);
156 ComplexExpr W2_3(re_W1_3, -im_W1_3);
157 ComplexExpr W4_3 = W1_3;
158
159 Type type = f.output_types()[0];
160
161 ComplexFunc F(prefix + "X8");
162 F(f.args()) = undef_z(type);
163
164 vector<ComplexFuncRef> x = get_func_refs(f, 6);
165 vector<ComplexFuncRef> X = get_func_refs(F, 6);
166 vector<ComplexFuncRef> T = get_func_refs(F, 6, true);
167
168 // Prime factor FFT, N=2*3, no twiddle factors!
169 T[0] = (x[0] + x[3]);
170 T[3] = (x[0] - x[3]);
171 T[1] = (x[1] + x[4]);
172 T[4] = (x[1] - x[4]);
173 T[2] = (x[2] + x[5]);
174 T[5] = (x[2] - x[5]);
175
176 X[0] = T[0] + T[2] + T[1];
177 X[4] = T[0] + T[2] * W1_3 + T[1] * W2_3;
178 X[2] = T[0] + T[2] * W2_3 + T[1] * W4_3;
179
180 X[3] = T[3] + T[5] - T[4];
181 X[1] = T[3] + T[5] * W1_3 - T[4] * W2_3;
182 X[5] = T[3] + T[5] * W2_3 - T[4] * W4_3;
183
184 return F;
185}

187ComplexFunc dft8(ComplexFunc f, int sign, const string &prefix) {
188 const float sqrt2_2 = 0.70710678f;
189
190 Type type = f.output_types()[0];
191
192 ComplexFunc F(prefix + "X8");
193 F(f.args()) = undef_z(type);
194
195 vector<ComplexFuncRef> x = get_func_refs(f, 8);
196 vector<ComplexFuncRef> X = get_func_refs(F, 8);
197 vector<ComplexFuncRef> T = get_func_refs(F, 8, true);
198
199 X[0] = (x[0] + x[4]);
200 X[2] = (x[2] + x[6]);
201 T[0] = (X[0] + X[2]);
202 T[2] = (X[0] - X[2]);
203
204 X[1] = (x[0] - x[4]);
205 X[3] = (x[2] - x[6]) * j * sign;
206 T[1] = (X[1] + X[3]);
207 T[3] = (X[1] - X[3]);
208
209 X[4] = (x[1] + x[5]);
210 X[6] = (x[3] + x[7]);
211 T[4] = (X[4] + X[6]);
212 T[6] = (X[4] - X[6]) * j * sign;
213
214 X[5] = (x[1] - x[5]);
215 X[7] = (x[3] - x[7]) * j * sign;
216 T[5] = mul(X[5] + X[7], sqrt2_2, sign * sqrt2_2);
217 T[7] = mul(X[5] - X[7], -sqrt2_2, sign * sqrt2_2);
218
219 X[0] = (T[0] + T[4]);
220 X[1] = (T[1] + T[5]);
221 X[2] = (T[2] + T[6]);
222 X[3] = (T[3] + T[7]);
223 X[4] = (T[0] - T[4]);
224 X[5] = (T[1] - T[5]);
225 X[6] = (T[2] - T[6]);
226 X[7] = (T[3] - T[7]);
227
228 return F;
229}

231// Compute the complex DFT of size N on dimension 0 of x.
232ComplexFunc dftN(ComplexFunc x, int N, int sign, const string &prefix) {
233 vector<Var> args(x.args());
234 args.erase(args.begin());
235
236 Var n("n");
237 ComplexFunc X(prefix + "XN");
238 if (N < 10) {
239 // If N is small, unroll the loop.
240 ComplexExpr dft = x(A({Expr(0)}, args));
241 for (int k = 1; k < N; k++) {
242 dft += expj((sign * 2 * kPi * k * n) / N) * x(A({Expr(k)}, args));
243 }
244 X(A({n}, args)) = dft;
245 } else {
246 // If N is larger, we really shouldn't be using this algorithm for the DFT anyways.
247 RDom k(0, N);
248 X(A({n}, args)) = sum(expj((sign * 2 * kPi * k * n) / N) * x(A({k}, args)));
249 }
250 X.unroll(n);
251 return X;
252}

254ComplexFunc dft1d_c2c(ComplexFunc x, int N, int sign,
255 const string &prefix) {
256 switch (N) {
257 case 2:
258 return dft2(x, prefix);
259 case 4:
260 return dft4(x, sign, prefix);
261 case 6:
262 return dft6(x, sign, prefix);
263 case 8:
264 return dft8(x, sign, prefix);
265 default:
266 return dftN(x, N, sign, prefix);
267 }
268}

270// Map to remember previously computed twiddle factors.
271typedef std::map<int, ComplexFunc> TwiddleFactorSet;
272
273// Return a function defining the twiddle factors.
274ComplexFunc twiddle_factors(int N, Expr gain, int sign,
275 const string &prefix,
276 TwiddleFactorSet *cache) {
277 // If the gain is one, we can use the cache. Otherwise, always define a new
278 // function. Generally, any given FFT will only have one set of twiddle
279 // factors where gain != 1.
280 ComplexFunc W(prefix + "W");
281 if (is_const_one(gain)) {
282 W = (*cache)[N];
283 }
284 if (!W.defined()) {
285 Var n("n");
286 W(n) = expj((sign * 2 * kPi * n) / N) * gain;
287 W.compute_root();
288 }
289
290 return W;
291}

293// Compute the N point DFT of dimension 1 (columns) of x using
294// radix R.
295ComplexFunc fft_dim1(ComplexFunc x,
296 const vector<int> &NR,
297 int sign,
298 int extent_0,
299 Expr gain,
300 bool parallel,
301 const string &prefix,
302 const Target &target,
303 TwiddleFactorSet *twiddle_cache) {
304 int N = product(NR);
305
306 vector<Var> args = x.args();
307 Var n0(args[0]), n1(args[1]);
308 args.erase(args.begin());
309 args.erase(args.begin());
310
311 vector<std::pair<Func, RDom>> stages;
312
313 RVar r_, s_;
314 int S = 1;
315 int vector_width = 1;
316 for (size_t i = 0; i < NR.size(); i++) {
317 int R = NR[i];
318 assert(R != 1);
319
320 std::stringstream stage_id;
321 stage_id << prefix;
322 if (S == N / R) {
323 stage_id << "fft1";
324 } else {
325 stage_id << "x";
326 }
327 stage_id << "_S" << S << "_R" << R << "_" << n1.name();
328
329 ComplexFunc exchange(stage_id.str());
330 Var r("r"), s("s");
331
332 // Load the points from each subtransform and apply the
333 // twiddle factors. Twiddle factors for S = 1 are all expj(0) = 1.
334 ComplexFunc v("v_" + stage_id.str());
335 ComplexExpr x_rs = x(A({n0, s + r * (N / R)}, args));
336 if (S > 1) {
337 x_rs = cast<float>(x_rs);
338 ComplexFunc W = twiddle_factors(R * S, gain, sign, prefix, twiddle_cache);
339 v(A({r, s, n0}, args)) = select(r > 0, likely(x_rs * W(r * (s % S))), x_rs * gain);
340
341 // Set the gain to 1 so it is only applied once.
342 gain = 1.0f;
343 } else {
344 v(A({r, s, n0}, args)) = x_rs;
345 }
346
347 // The vector width is the least common multiple of the previous vector
348 // width and the natural vector size for this stage.
349 vector_width = lcm(vector_width, target.natural_vector_size(v.output_types()[0]));
350
351 // Compute the R point DFT of the subtransform.
352 ComplexFunc V = dft1d_c2c(v, R, sign, prefix);
353
354 // Write the subtransform and use it as input to the next
355 // pass. Since the pure stage is undef, we explicitly generate the
356 // arg list (because we can't use placeholders in an undef
357 // definition).
358 exchange(A({n0, n1}, args)) = undef_z(V.output_types()[0]);
359
360 RDom rs(0, R, 0, N / R);
361 r_ = rs.x;
362 s_ = rs.y;
363 ComplexExpr V_rs = V(A({r_, s_, n0}, args));
364 if (S == N / R) {
365 // In case we haven't yet applied the requested gain (i.e. there were no
366 // twiddle factor steps), do so now. If gain is one, this will be a no-op.
367 V_rs = V_rs * gain;
368 gain = 1.0f;
369 }
370 exchange(A({n0, ((s_ / S) * R * S) + (s_ % S) + (r_ * S)}, args)) = V_rs;
371 exchange.bound(n1, 0, N);
372
373 if (S > 1) {
374 v.compute_at(exchange, s_).unroll(r);
375 v.reorder_storage(n0, r, s);
376 } else {
377 // On the first stage, the twiddle factors are 1, so we can inline this (no-op).
378 }
379
380 V.compute_at(exchange, s_);
381 V.reorder_storage(V.args()[2], V.args()[0], V.args()[1]);
382
383 // The last stage needs explicit vectorization, because it doesn't get computed
384 // at the vectorized context exchange (below).
385 if (S == N / R) {
386 if (S > 1) {
387 v.vectorize(n0);
388 }
389 V.vectorize(V.args()[2]);
390 for (int i = 0; i < V.num_update_definitions(); i++) {
391 V.update(i).vectorize(V.args()[2]);
392 }
393 }
394
395 exchange.update().unroll(r_);
396 // Remember this stage for scheduling later.
397 stages.push_back({exchange, rs});
398
399 x = exchange;
400 S *= R;
401 }
402
403 // Ensure that the vector width divides the vectorization dimension extent.
404 vector_width = gcd(vector_width, extent_0);
405
406 // Split the tile into groups of DFTs, and vectorize within the
407 // group.
408 x.update()
409 .split(n0, group, n0, vector_width)
410 .reorder(n0, r_, s_, group)
411 .vectorize(n0);
412 if (parallel) {
413 x.update().parallel(group);
414 }
415 for (size_t i = 0; i + 1 < stages.size(); i++) {
416 Func stage = stages[i].first;
417 stage.compute_at(x, group).update().vectorize(n0);
418 }
419
420 return x;
421}

423// transpose the first two dimensions of x.
424template<typename FuncType>
425FuncType transpose(FuncType f) {
426 vector<Halide::Var> argsT(f.args());
427 std::swap(argsT[0], argsT[1]);
428 FuncType fT;
429 fT(argsT) = f(f.args());
430 return fT;
431}

433template<typename FuncType>
434std::pair<FuncType, FuncType> tiled_transpose(FuncType f, int max_tile_size,
435 const Target &target,
436 const string &prefix,
437 bool always_tile = false) {
438 // ARM can do loads of up to stride 4. We can use these loads to write a more
439 // efficient transpose. The strategy is to break the transpose into 4x4 tiles,
440 // transpose the tiles themselves (dense vector load/stores), then transpose
441 // the data within each tile (stride 4 loads).
442 if (target.arch != Target::ARM && !always_tile) {
443 return {transpose(f), FuncType()};
444 }
445
446 const int tile_size =
447 std::min(max_tile_size, target.natural_vector_size(f.output_types()[0]));
448
449 vector<Var> args = f.args();
450 Var x(args[0]), y(args[1]);
451 args.erase(args.begin());
452 args.erase(args.begin());
453
454 Var xo(x.name() + "o");
455 Var yo(y.name() + "o");
456
457 // Break the transposed DFT into 4x4 tiles.
458 FuncType f_tiled(prefix + "tiled");
459 f_tiled(A({x, y, xo, yo}, args)) = f(A({xo * tile_size + x, yo * tile_size + y}, args));
460
461 // transpose the values within each tile.
462 FuncType f_tiledT(prefix + "tiledT");
463 f_tiledT(A({y, x, xo, yo}, args)) = f_tiled(A({x, y, xo, yo}, args));
464
465 FuncType fT_tiled(prefix + "T_tiled");
466 fT_tiled(A({y, x, yo, xo}, args)) = f_tiledT(A({y, x, xo, yo}, args));
467
468 // Produce the untiled result.
469 FuncType fT(prefix + "T");
470 fT(A({y, x}, args)) = fT_tiled(A({y % tile_size, x % tile_size, y / tile_size, x / tile_size}, args));
471
472 f_tiledT
473 .vectorize(x, tile_size)
474 .unroll(y, tile_size);
475
476 return {fT, f_tiledT};
477}

481ComplexFunc fft2d_c2c(ComplexFunc x,
482 vector<int> R0,
483 vector<int> R1,
484 int sign,
485 const Target &target,
486 const Fft2dDesc &desc) {
487 string prefix = desc.name.empty() ? "c2c_" : desc.name + "_";
488
489 int N0 = product(R0);
490 int N1 = product(R1);
491
492 // Get the innermost variable outside the FFT.
493 Var outer = Var::outermost();
494 if (x.dimensions() > 2) {
495 outer = x.args()[2];
496 }
497 Var n0 = x.args()[0];
498 Var n1 = x.args()[1];
499
500 // Cache of twiddle factors for this FFT.
501 TwiddleFactorSet twiddle_cache;
502
503 // transpose the input to the FFT.
504 ComplexFunc xT, x_tiled;
505 std::tie(xT, x_tiled) = tiled_transpose(x, N1, target, prefix);
506
507 // Compute the DFT of dimension 1 (originally dimension 0).
508 ComplexFunc dft1T = fft_dim1(xT,
509 R0,
510 sign,
511 N1, // extent of dim 0.
512 1.0f,
513 desc.parallel,
514 prefix,
515 target,
516 &twiddle_cache);
517
518 // transpose back.
519 ComplexFunc dft1, dft1_tiled;
520 std::tie(dft1, dft1_tiled) = tiled_transpose(dft1T, N0, target, prefix);
521
522 // Compute the DFT of dimension 1.
523 ComplexFunc dft = fft_dim1(dft1,
524 R1,
525 sign,
526 N0, // extent of dim 0
527 desc.gain,
528 desc.parallel,
529 prefix,
530 target,
531 &twiddle_cache);
532
533 // Schedule the tiled transposes at each group.
534 if (dft1_tiled.defined()) {
535 dft1_tiled.compute_at(dft, group);
536 } else {
537 xT.compute_at(dft, outer).vectorize(n0).unroll(n1);
538 }
539 if (x_tiled.defined()) {
540 x_tiled.compute_at(dft1T, group);
541 }
542
543 // Schedule the input, if requested.
544 if (desc.schedule_input) {
545 x.compute_at(dft1T, group);
546 }
547
548 dft1T.compute_at(dft, outer);
549
550 dft.bound(dft.args()[0], 0, N0);
551 dft.bound(dft.args()[1], 0, N1);
552
553 return dft;
554}

556// The next two functions implement real to complex or complex to real FFTs. To
557// understand the real to complex FFT, we need some background on the properties
558// of FFTs of real data. If X = DFT[x] for a real sequence x of length N, then
559// the following relationship holds:
560//
561// X_n = (X_(N-n))* (1)
562//
563// This means that for N even, N/2 - 1 of the elements of X are redundant with
564// another element of X. This property allows us to store only roughly half of
565// a DFT of a real sequence, because the remaining half is fully determined by
566// the first.
567//
568// Also note that for any DFT (not just real):
569//
570// Z*_n = sum[ (z_n*) e^(-2*pi*i*n/N) ]
571// = sum[ z_n (e^(-2*pi*i*n/N))* ]*
572// = sum[ z_n e^(-2*pi*i*(N - n)/N) ]*
573// Z*_n = (Z_(N-n))* (2)
574//
575// Using these relationships, we can more efficiently compute two real FFTs by
576// using one complex FFT. Let x and y be two real sequences of length N, and
577// let z = x + j*y. We can compute the FFT of x and y using one complex FFT of
578// z; let X + j*Y = Z = DFT[z] = DFT[x + j*y], then by the linearity of the DFT
579// and equations (1) and (2):
580//
581// x_n = (z_n + (z_n)*)/2
582// -> X_n = (Z_n + Z*_n)/2
583// = (Z_n + (Z_(N-n))*)/2 (3)
584//
585// and
586//
587// y_n = (z_n - (z_n)*)/(2*j)
588// -> Y_n = (Z_n - Z*_n)/(2*j)
589// = (Z_n + (Z_(N-n))*)/(2*j) (4)
590//
591// This gives 2 real DFTs for the cost of computing 1 complex FFT.
592//
593// As a side note, a consequence of (1) is that Z_0 and Z_(N/2) must be real.
594// Note that Z_N = Z_0 by periodicity of the DFT:
595//
596// Z_0 = (Z_N)* = (Z_0)*
597// Z_(N/2) = (Z_(N/2))*
598//
599// The only way z = z* can be true is if im(z) = 0, i.e. z is real.
600//
601// We want an efficient 2D FFT. Applying the above tools to a 2D DFT leads to
602// some interesting results. First, note that the FFT of a 2D sequence x_(m, n)
603// with extents MxN (rows x cols) is a 1D FFT of the rows, followed by a 1D FFT
604// of the columns. Suppose x is real. One way to use the above tools is to
605// combine pairs of columns into a set of half as many complex columns, and
606// compute the FFT of these complex columns. This gives a result laid out like
607// so:
608//
609// N/2
610// +---------+
611// | a | m = 0
612// +---------+
613// | |
614// | b |
615// | |
616// +---------+
617// | c | m = M/2
618// +---------+
619// | |
620// | d |
621// | |
622// +---------+
623//
624// When we unzip the columns using (3) and (4) from above, we get data with the
625// following layout.
626//
627// N
628// +-------------------+
629// | a' | m = 0
630// +-------------------+
631// | |
632// | b' |
633// | |
634// +-------------------+
635// | c' | m = M/2
636// +-------------------+
637// | |
638// | b'* |
639// | |
640// +-------------------+
641//
642// Because b'* is redundant with b, we don't need to compute or store it,
643// leaving M/2 + 1 rows.
644//
645// Now, we want to compute the DFT of the rows of this data. Because there are
646// M/2 + 1 of them, and we are going to compute the FFTs using SIMD
647// instructions, the extra 1 row can be quite expensive. If M is 16, then we
648// have 9 rows. If the SIMD width is 4, we will compute 3 SIMD vectors worth of
649// FFTs instead of 2, 33% more work than necessary. This is even worse if the
650// SIMD width is 8 (AVX).
651//
652// We can fix this by recognizing that a' and c' are both real, and combining
653// them together into one row in the same manner we did for the columns. This
654// gives a block of data that looks like:
655//
656// N
657// +-------------------+
658// | a' + j c' |
659// +-------------------+
660// | | M/2
661// | b' |
662// | |
663// +-------------------+
664//
665// This way, we have M/2 rows to compute the FFT of, which is likely to be
666// efficient to compute without wasted SIMD work. The DFTs of the rows a and c
667// can be recovered using (3) and (4) again.
668
669ComplexFunc fft2d_r2c(Func r,
670 const vector<int> &R0,
671 const vector<int> &R1,
672 const Target &target,
673 const Fft2dDesc &desc) {
674 string prefix = desc.name.empty() ? "r2c_" : desc.name + "_";
675
676 vector<Var> args(r.args());
677 Var n0(args[0]), n1(args[1]);
678 args.erase(args.begin());
679 args.erase(args.begin());
680
681 // Get the innermost variable outside the FFT.
682 Var outer = Var::outermost();
683 if (!args.empty()) {
684 outer = args.front();
685 }
686
687 int N0 = product(R0);
688 int N1 = product(R1);
689
690 const int natural_vector_size = target.natural_vector_size(r.output_types()[0]);
691
692 // If this FFT is small, the logic related to zipping and unzipping
693 // the FFT may be expensive compared to just brute forcing with a complex
694 // FFT.
695 bool skip_zip = N0 < natural_vector_size * 2;
696 // We also are bad at handling zipping when the zip size is a small non-integer
697 // factor of the vector size.
698 skip_zip = skip_zip || (N0 < natural_vector_size * 4 && (N0 % (natural_vector_size * 2) != 0));
699 if (skip_zip) {
700 ComplexFunc r_complex("r_complex");
701 r_complex(A({n0, n1}, args)) = ComplexExpr(r(A({n0, n1}, args)), 0.0f);
702 ComplexFunc dft = fft2d_c2c(r_complex, R0, R1, -1, target, desc);
703
704 // fft2d_c2c produces a N0 x N1 buffer, but the caller of this probably only expects
705 // an N0 x N1 / 2 + 1 buffer.
706 ComplexFunc result(prefix + "r2c");
707 result(A({n0, n1}, args)) = dft(A({n0, n1}, args));
708 result.bound(n0, 0, N0);
709 result.bound(n1, 0, (N1 + 1) / 2 + 1);
710 result.vectorize(n0, std::min(N0, target.natural_vector_size(result.output_types()[0])));
711 dft.compute_at(result, outer);
712 return result;
713 }
714
715 // Cache of twiddle factors for this FFT.
716 TwiddleFactorSet twiddle_cache;
717
718 // The gain requested of the FFT.
719 Expr gain = desc.gain;
720
721 // Combine pairs of real columns x, y into complex columns z = x + j y. This
722 // allows us to compute two real DFTs using one complex FFT. See the large
723 // comment above this function for more background.
724 //
725 // An implementation detail is that we zip the columns in groups from the
726 // input data to enable the loads to be dense vectors. x is taken from the
727 // even indexed groups columns, y is taken from the odd indexed groups of
728 // columns.
729 //
730 // Changing the group size can (insignificantly) numerically change the result
731 // due to regrouping floating point operations. To avoid this, if the FFT
732 // description specified a vector width, use it as the group size.
733 ComplexFunc zipped(prefix + "zipped");
734 int zip_width = desc.vector_width;
735 if (zip_width <= 0) {
736 zip_width = target.natural_vector_size(r.output_types()[0]);
737 }
738 // Ensure the zip width divides the zipped extent.
739 zip_width = gcd(zip_width, N0 / 2);
740 Expr zip_n0 = (n0 / zip_width) * zip_width * 2 + (n0 % zip_width);
741 zipped(A({n0, n1}, args)) =
742 ComplexExpr(r(A({zip_n0, n1}, args)),
743 r(A({zip_n0 + zip_width, n1}, args)));
744
745 // DFT down the columns first.
746 ComplexFunc dft1 = fft_dim1(zipped,
747 R1,
748 -1, // sign
749 N0 / 2, // extent of dim 0
750 1.0f,
751 false, // We parallelize unzipped below instead.
752 prefix,
753 target,
754 &twiddle_cache);
755
756 // Unzip the two groups of real DFTs we zipped together above. For more
757 // information about the unzipping operation, see the large comment above this
758 // function.
759 ComplexFunc unzipped(prefix + "unzipped");
760 {
761 Expr unzip_n0 = (n0 / (zip_width * 2)) * zip_width + (n0 % zip_width);
762 ComplexExpr Z = dft1(A({unzip_n0, n1}, args));
763 ComplexExpr conjsymZ = conj(dft1(A({unzip_n0, (N1 - n1) % N1}, args)));
764
765 ComplexExpr X = Z + conjsymZ;
766 ComplexExpr Y = -j * (Z - conjsymZ);
767 // Rather than divide the above expressions by 2 here, adjust the gain
768 // instead.
769 gain /= 2;
770
771 unzipped(A({n0, n1}, args)) =
772 select(n0 % (zip_width * 2) < zip_width, X, Y);
773 }
774
775 // Zip the DC and Nyquist DFT bin rows, which should be real.
776 ComplexFunc zipped_0(prefix + "zipped_0");
777 zipped_0(A({n0, n1}, args)) =
778 select(n1 > 0, likely(unzipped(A({n0, n1}, args))),
779 ComplexExpr(re(unzipped(A({n0, 0}, args))),
780 re(unzipped(A({n0, N1 / 2}, args)))));
781
782 // The vectorization of the columns must not exceed this value.
783 int zipped_extent0 = std::min((N1 + 1) / 2, zip_width);
784
785 // transpose so we can FFT dimension 0 (by making it dimension 1).
786 ComplexFunc unzippedT, unzippedT_tiled;
787 std::tie(unzippedT, unzippedT_tiled) = tiled_transpose(zipped_0, zipped_extent0, target, prefix);
788
789 // DFT down the columns again (the rows of the original).
790 ComplexFunc dftT = fft_dim1(unzippedT,
791 R0,
792 -1, // sign
793 zipped_extent0,
794 gain,
795 desc.parallel,
796 prefix,
797 target,
798 &twiddle_cache);
799
800 // transpose the result back to the original orientation, unless the caller
801 // requested a transposed DFT.
802 ComplexFunc dft = transpose(dftT);
803
804 // We are going to add a row to the result (with update steps) by unzipping
805 // the DC and Nyquist bin rows. To avoid unnecessarily computing some junk for
806 // this row before we overwrite it, pad the pure definition with undef.
807 dft = ComplexFunc(constant_exterior((Func)dft, Tuple(undef_z()), Expr(), Expr(), Expr(0), Expr(N1 / 2)));
808
809 // Unzip the DFTs of the DC and Nyquist bin DFTs. Unzip the Nyquist DFT first,
810 // because the DC bin DFT is updated in-place. For more information about
811 // this, see the large comment above this function.
812 RDom n0z1(1, N0 / 2);
813 RDom n0z2(N0 / 2, N0 / 2);
814 // Update 0: Unzip the DC bin of the DFT of the Nyquist bin row.
815 dft(A({0, N1 / 2}, args)) = im(dft(A({0, 0}, args)));
816 // Update 1: Unzip the rest of the DFT of the Nyquist bin row.
817 dft(A({n0z1, N1 / 2}, args)) =
818 0.5f * -j * (dft(A({n0z1, 0}, args)) - conj(dft(A({N0 - n0z1, 0}, args))));
819 // Update 2: Compute the rest of the Nyquist bin row via conjugate symmetry.
820 // Note that this redundantly computes n0 = N0/2, but that's faster and easier
821 // than trying to deal with N0/2 - 1 bins.
822 dft(A({n0z2, N1 / 2}, args)) = conj(dft(A({N0 - n0z2, N1 / 2}, args)));
823
824 // Update 3: Unzip the DC bin of the DFT of the DC bin row.
825 dft(A({0, 0}, args)) = re(dft(A({0, 0}, args)));
826 // Update 4: Unzip the rest of the DFT of the DC bin row.
827 dft(A({n0z1, 0}, args)) =
828 0.5f * (dft(A({n0z1, 0}, args)) + conj(dft(A({N0 - n0z1, 0}, args))));
829 // Update 5: Compute the rest of the DC bin row via conjugate symmetry.
830 // Note that this redundantly computes n0 = N0/2, but that's faster and easier
831 // than trying to deal with N0/2 - 1 bins.
832 dft(A({n0z2, 0}, args)) = conj(dft(A({N0 - n0z2, 0}, args)));
833
834 // Schedule.
835 dftT.compute_at(dft, outer);
836
837 // Schedule the tiled transposes.
838 if (unzippedT_tiled.defined()) {
839 unzippedT_tiled.compute_at(dftT, group);
840 }
841
842 // Schedule the input, if requested.
843 if (desc.schedule_input) {
844 r.compute_at(dft1, group);
845 }
846
847 // Vectorize the zip groups, and unroll by a factor of 2 to simplify the
848 // even/odd selection.
849 Var n0o("n0o"), n0i("n0i");
850 unzipped.compute_at(dft, outer)
851 .split(n0, n0o, n0i, zip_width * 2)
852 .reorder(n0i, n1, n0o)
853 .vectorize(n0i, zip_width)
854 .unroll(n0i);
855 dft1.compute_at(unzipped, n0o);
856 if (desc.parallel) {
857 // Note that this also parallelizes dft1, which is computed inside this loop
858 // of unzipped.
859 unzipped.parallel(n0o);
860 }
861
862 // Schedule the final DFT transpose and unzipping updates.
863 int vector_size = gcd(target.natural_vector_size<float>(), N0);
864 dft.vectorize(n0, vector_size)
865 .unroll(n0, gcd(N0 / vector_size, 4));
866
867 // The Nyquist bin at n0z = N0/2 looks like a race condition because it
868 // simplifies to an expression similar to the DC bin. However, we include it
869 // in the reduction because it makes the reduction have length N/2, which is
870 // convenient for vectorization, and just ignore the resulting appearance of
871 // a race condition.
872 dft.update(1).allow_race_conditions().vectorize(n0z1, vector_size);
873 dft.update(2).allow_race_conditions().vectorize(n0z2, vector_size);
874 dft.update(4).allow_race_conditions().vectorize(n0z1, vector_size);
875 dft.update(5).allow_race_conditions().vectorize(n0z2, vector_size);
876
877 // Our result is undefined outside these bounds.
878 dft.bound(n0, 0, N0);
879 dft.bound(n1, 0, (N1 + 1) / 2 + 1);
880
881 return dft;
882}

884Func fft2d_c2r(ComplexFunc c,
885 vector<int> R0,
886 vector<int> R1,
887 const Target &target,
888 const Fft2dDesc &desc) {
889 string prefix = desc.name.empty() ? "c2r_" : desc.name + "_";
890
891 vector<Var> args = c.args();
892 Var n0(args[0]), n1(args[1]);
893 args.erase(args.begin());
894 args.erase(args.begin());
895
896 // Get the innermost variable outside the FFT.
897 Var outer = Var::outermost();
898 if (!args.empty()) {
899 outer = args.front();
900 }
901
902 int N0 = product(R0);
903 int N1 = product(R1);
904
905 // Add a boundary condition to prevent scheduling from causing the
906 // algorithms below to reach out of the bounds we promise to define in
907 // forward FFTs.
908 c = ComplexFunc(repeat_edge((Func)c, {{Expr(0), Expr(N0)}, {Expr(0), Expr((N1 + 1) / 2 + 1)}}));
909
910 // If this FFT is small, the logic related to zipping and unzipping
911 // the FFT may be expensive compared to just brute forcing with a complex
912 // FFT.
913 const int natural_vector_size = target.natural_vector_size(c.output_types()[0]);
914
915 bool skip_zip = N0 < natural_vector_size * 2;
916
917 ComplexFunc dft;
918 Func unzipped(prefix + "unzipped");
919 if (skip_zip) {
920 // Because fft2d_c2c expects the full complex domain, we need to reconstruct
921 // it via conjugate symmetry.
922 ComplexFunc c_extended(prefix + "c_extended");
923 c_extended(A({n0, n1}, args)) =
924 select(n1 <= (N1 + 1) / 2, c(A({n0, n1}, args)), conj(c(A({(N0 - n0) % N0, (N1 - n1) % N1}, args))));
925 dft = fft2d_c2c(c_extended, R0, R1, 1, target, desc);
926 unzipped(A({n0, n1}, args)) = re(dft(A({n0, n1}, args)));
927
928 dft.compute_at(unzipped, outer);
929
930 // We want to unroll by at least two zip_widths to simplify the zip group
931 // logic.
932 int vector_size = std::min(N0, natural_vector_size);
933 unzipped.vectorize(n0, vector_size);
934 } else {
935 // Cache of twiddle factors for this FFT.
936 TwiddleFactorSet twiddle_cache;
937
938 int zipped_extent0 = (N1 + 1) / 2;
939
940 // The DC and Nyquist bins must be real, so we zip those two DFTs together
941 // into one complex DFT. Note that this select gets eliminated due to the
942 // scheduling done by tiled_transpose below.
943 ComplexFunc c_zipped(prefix + "c_zipped");
944 {
945 // Stuff the Nyquist bin DFT into the imaginary part of the DC bin DFT.
946 ComplexExpr X = c(A({n0, 0}, args));
947 ComplexExpr Y = c(A({n0, N1 / 2}, args));
948 c_zipped(A({n0, n1}, args)) = select(n1 > 0, likely(c(A({n0, n1}, args))), X + j * Y);
949 }
950
951 // transpose the input.
952 ComplexFunc cT, cT_tiled;
953 std::tie(cT, cT_tiled) =
954 tiled_transpose(c_zipped, zipped_extent0, target, prefix);
955
956 // Take the inverse DFT of the columns (rows in the final result).
957 ComplexFunc dft0T = fft_dim1(cT,
958 R0,
959 1, // sign
960 zipped_extent0,
961 1.0f,
962 desc.parallel,
963 prefix,
964 target,
965 &twiddle_cache);
966
967 // The vector width of the zipping performed below.
968 int zip_width = desc.vector_width;
969 if (zip_width <= 0) {
970 zip_width = gcd(target.natural_vector_size(dft0T.output_types()[0]), N1 / 2);
971 }
972
973 // transpose so we can take the DFT of the columns again.
974 ComplexFunc dft0, dft0_tiled;
975 std::tie(dft0, dft0_tiled) = tiled_transpose(dft0T, zip_width, target, prefix, true);
976
977 // Unzip the DC and Nyquist DFTs.
978 ComplexFunc dft0_unzipped("dft0_unzipped");
979 {
980 dft0_unzipped(A({n0, n1}, args)) =
981 select(n1 <= 0, re(dft0(A({n0, 0}, args))),
982 n1 >= N1 / 2, im(dft0(A({n0, 0}, args))),
983 likely(dft0(A({n0, min(n1, (N1 / 2) - 1)}, args))));
984 }
985
986 // Zip two real DFTs X and Y into one complex DFT Z = X + j Y. For more
987 // information, see the large comment above fft2d_r2c.
988 //
989 // As an implementation detail, this zip operation is done in groups of
990 // columns to enable dense vector loads. X is taken from the even indexed
991 // groups of columns, Y is taken from the odd indexed groups of columns.
992 //
993 // Ensure the zip width divides the zipped extent.
994 zip_width = gcd(zip_width, N0 / 2);
995
996 ComplexFunc zipped(prefix + "zipped");
997 {
998 // Construct the whole DFT domain of X and Y via conjugate symmetry.
999 Expr n0_X = (n0 / zip_width) * zip_width * 2 + (n0 % zip_width);
1000 Expr n1_sym = (N1 - n1) % N1;
1001 ComplexExpr X = select(n1 < N1 / 2,
1002 dft0_unzipped(A({n0_X, n1}, args)),
1003 conj(dft0_unzipped(A({n0_X, n1_sym}, args))));
1004
1005 Expr n0_Y = n0_X + zip_width;
1006 ComplexExpr Y = select(n1 < N1 / 2,
1007 dft0_unzipped(A({n0_Y, n1}, args)),
1008 conj(dft0_unzipped(A({n0_Y, n1_sym}, args))));
1009 zipped(A({n0, n1}, args)) = X + j * Y;
1010 }
1011
1012 // Take the inverse DFT of the columns again.
1013 dft = fft_dim1(zipped,
1014 R1,
1015 1, // sign
1016 std::min(zip_width, N0 / 2), // extent of dim 0
1017 desc.gain,
1018 desc.parallel,
1019 prefix,
1020 target,
1021 &twiddle_cache);
1022
1023 ComplexFunc dft_padded = ComplexFunc(repeat_edge((Func)dft, {{Expr(), Expr()}, {Expr(0), Expr(N1)}}));
1024
1025 // Extract the real inverse DFTs.
1026 Expr unzip_n0 = (n0 / (zip_width * 2)) * zip_width + (n0 % zip_width);
1027 unzipped(A({n0, n1}, args)) =
1028 select(n0 % (zip_width * 2) < zip_width,
1029 re(dft_padded(A({unzip_n0, n1}, args))),
1030 im(dft_padded(A({unzip_n0, n1}, args))));
1031
1032 // Schedule.
1033
1034 // Schedule the transpose step.
1035 if (cT_tiled.defined()) {
1036 cT_tiled.compute_at(dft0T, group);
1037 }
1038 dft0_tiled.compute_at(dft, outer);
1039
1040 // Schedule the input, if requested.
1041 if (desc.schedule_input) {
1042 // We should want to compute this at dft0T, group. However, due to the zip
1043 // operation, the bounds are bigger than we'd like (we need the last row for
1044 // the first group).
1045 c.compute_at(dft, outer);
1046 }
1047
1048 dft0T.compute_at(dft, outer);
1049
1050 // We want to unroll by at least two zip_widths to simplify the zip group
1051 // logic.
1052 unzipped
1053 .vectorize(n0, zip_width)
1054 .unroll(n0, gcd(N0 / zip_width, 4));
1055 }
1056 dft.compute_at(unzipped, outer);
1057
1058 unzipped.bound(n0, 0, N0);
1059 unzipped.bound(n1, 0, N1);
1060
1061 return unzipped;
1062}

1066// Compute a factorization of N suitable for use in the FFT.
1067vector<int> radix_factor(int N) {
1068 // Some special cases to optimize.
1069 switch (N) {
1070 case 16:
1071 return {4, 4};
1072 case 32:
1073 return {8, 4};
1074 case 64:
1075 return {8, 8};
1076 case 128:
1077 return {8, 4, 4};
1078 case 256:
1079 return {8, 8, 4};
1080 }
1081
1082 // Factor N into factors found in the 'radices' set.
1083 static const int radices[] = {8, 6, 4, 2};
1084 vector<int> R;
1085 for (int r : radices) {
1086 while (N % r == 0) {
1087 R.push_back(r);
1088 N /= r;
1089 }
1090 }
1091
1092 // If there are still factors left over, just include them as a radix.
1093 if (N != 1 || R.empty()) {
1094 R.push_back(N);
1095 }
1096
1097 return R;
1098}
