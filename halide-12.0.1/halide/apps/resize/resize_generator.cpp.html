5enum InterpolationType {
6 Box,
7 Linear,
8 Cubic,
9 Lanczos
10};
11
12Expr kernel_box(Expr x) {
13 Expr xx = abs(x);
14 return select(xx <= 0.5f, 1.0f, 0.0f);
15}
16
17Expr kernel_linear(Expr x) {
18 Expr xx = abs(x);
19 return select(xx < 1.0f, 1.0f - xx, 0.0f);
20}
21
22Expr kernel_cubic(Expr x) {
23 Expr xx = abs(x);
24 Expr xx2 = xx * xx;
25 Expr xx3 = xx2 * xx;
26 float a = -0.5f;
27
28 return select(xx < 1.0f, (a + 2.0f) * xx3 - (a + 3.0f) * xx2 + 1,
29 select(xx < 2.0f, a * xx3 - 5 * a * xx2 + 8 * a * xx - 4.0f * a,
30 0.0f));
31}
32
33Expr sinc(Expr x) {
34 x *= 3.14159265359f;
35 return sin(x) / x;
36}
37
38Expr kernel_lanczos(Expr x) {
39 Expr value = sinc(x) * sinc(x / 3);
40 value = select(x == 0.0f, 1.0f, value); // Take care of singularity at zero
41 value = select(x > 3 || x < -3, 0.0f, value); // Clamp to zero out of bounds
42 return value;
43}

45struct KernelInfo {
46 const char *name;
47 int taps;
48 Expr (*kernel)(Expr);
49};
50
51static KernelInfo kernel_info[] = {
52 {"box", 1, kernel_box},
53 {"linear", 2, kernel_linear},
54 {"cubic", 4, kernel_cubic},
55 {"lanczos", 6, kernel_lanczos}};

57class Resize : public Halide::Generator<Resize> {
58public:
59 GeneratorParam<InterpolationType> interpolation_type{"interpolation_type", Cubic, {{"box", Box}, {"linear", Linear}, {"cubic", Cubic}, {"lanczos", Lanczos}}};
60
61 // If we statically know whether we're upsampling or downsampling,
62 // we can generate different pipelines (we want to reorder the
63 // resample in x and in y).
64 GeneratorParam<bool> upsample{"upsample", false};
65
66 Input<Buffer<>> input{"input", 3};
67 Input<float> scale_factor{"scale_factor"};
68 Output<Buffer<>> output{"output", 3};
69
70 // Common Vars
71 Var x, y, c, k;
72
73 // Intermediate Funcs
74 Func as_float, resized_x, resized_y,
75 unnormalized_kernel_x, unnormalized_kernel_y,
76 kernel_x, kernel_y,
77 kernel_sum_x, kernel_sum_y;
78
79 void generate() {
80
81 // Handle different types by just casting to float
82 as_float(x, y, c) = cast<float>(input(x, y, c));
83
84 // For downscaling, widen the interpolation kernel to perform lowpass
85 // filtering.
86
87 // Invert the scale factor in a single place and do it
88 // strictly, to avoid getting different ratios showing up in
89 // different places.
90 Expr inverse_scale_factor = strict_float(1.0f / scale_factor);
91
92 Expr kernel_scaling = upsample ? Expr(1.0f) : scale_factor;
93 Expr inverse_kernel_scaling = upsample ? Expr(1.0f) : inverse_scale_factor;
94
95 Expr kernel_radius = 0.5f * kernel_info[interpolation_type].taps * inverse_kernel_scaling;
96
97 Expr kernel_taps = cast<int>(ceil(kernel_info[interpolation_type].taps * inverse_kernel_scaling));
98
99 // source[xy] are the (non-integer) coordinates inside the source image
100 Expr sourcex = (x + 0.5f) * inverse_scale_factor - 0.5f;
101 Expr sourcey = (y + 0.5f) * inverse_scale_factor - 0.5f;
102
103 // Initialize interpolation kernels. Since we allow an
104 // arbitrary scaling factor, the filter coefficients are
105 // different for each x and y coordinate. Use strict-float to
106 // ensure fast-math doesn't mess up our bounds inference.
107 Expr beginx = cast<int>(strict_float(ceil(sourcex - kernel_radius)));
108 Expr beginy = cast<int>(strict_float(ceil(sourcey - kernel_radius)));
109 beginx = clamp(beginx, input.dim(0).min(), input.dim(0).max() + 1 - kernel_taps);
110 beginy = clamp(beginy, input.dim(1).min(), input.dim(1).max() + 1 - kernel_taps);
111
112 RDom r(0, kernel_taps);
113 const KernelInfo &info = kernel_info[interpolation_type];
114
115 unnormalized_kernel_x(x, k) = info.kernel((k + beginx - sourcex) * kernel_scaling);
116 unnormalized_kernel_y(y, k) = info.kernel((k + beginy - sourcey) * kernel_scaling);
117
118 kernel_sum_x(x) = sum(unnormalized_kernel_x(x, r), "kernel_sum_x");
119 kernel_sum_y(y) = sum(unnormalized_kernel_y(y, r), "kernel_sum_y");
120
121 kernel_x(x, k) = unnormalized_kernel_x(x, k) / kernel_sum_x(x);
122 kernel_y(y, k) = unnormalized_kernel_y(y, k) / kernel_sum_y(y);
123
124 // Perform separable resizing. The resize in x vectorizes
125 // poorly compared to the resize in y, so do it first if we're
126 // upsampling, and do it second if we're downsampling.
127 Func resized;
128 if (upsample) {
129 resized_x(x, y, c) = sum(kernel_x(x, r) * as_float(r + beginx, y, c), "resized_x");
130 resized_y(x, y, c) = sum(kernel_y(y, r) * resized_x(x, r + beginy, c), "resized_y");
131 resized = resized_y;
132 } else {
133 resized_y(x, y, c) = sum(kernel_y(y, r) * as_float(x, r + beginy, c), "resized_y");
134 resized_x(x, y, c) = sum(kernel_x(x, r) * resized_y(r + beginx, y, c), "resized_x");
135 resized = resized_x;
136 }
137
138 if (input.type().is_float()) {
139 output(x, y, c) = clamp(resized(x, y, c), 0.0f, 1.0f);
140 } else {
141 output(x, y, c) = saturating_cast(input.type(), resized(x, y, c));
142 }
143 }
144
145 void schedule() {
146 const int vec = natural_vector_size<float>();
147
148 Var xi, yi;
149 unnormalized_kernel_x
150 .compute_at(kernel_x, x)
151 .vectorize(x);
152 kernel_sum_x
153 .compute_at(kernel_x, x)
154 .vectorize(x);
155 kernel_x
156 .compute_root()
157 .reorder(k, x)
158 .vectorize(x, vec);
159
160 unnormalized_kernel_y
161 .compute_at(kernel_y, y)
162 .vectorize(y, vec);
163 kernel_sum_y
164 .compute_at(kernel_y, y)
165 .vectorize(y);
166 kernel_y
167 .compute_at(output, y)
168 .reorder(k, y)
169 .vectorize(y, vec);
170
171 if (upsample) {
172 output
173 .tile(x, y, xi, yi, 16, 64)
174 .parallel(y)
175 .vectorize(xi);
176 resized_x
177 .compute_at(output, x)
178 .store_in(MemoryType::Stack)
179 .vectorize(x);
180 resized_y
181 .compute_at(output, xi)
182 .unroll(c);
183 } else {
184 output
185 .tile(x, y, xi, yi, 32, 8)
186 .parallel(y)
187 .vectorize(xi);
188 resized_y
189 .compute_at(output, y)
190 .vectorize(x, vec);
191 resized_x
192 .compute_at(output, xi)
193 .unroll(c);
194 }
195
196 // Allow the input and output to have arbitrary memory layout,
197 // and add some specializations for a few common cases. If
198 // your case is not covered (e.g. planar input, packed rgb
199 // output), you could add a new specialization here.
200 output.dim(0).set_stride(Expr());
201 input.dim(0).set_stride(Expr());
202
203 Expr planar = (output.dim(0).stride() == 1 &&
204 input.dim(0).stride() == 1);
205 Expr packed_rgb = (output.dim(0).stride() == 3 &&
206 output.dim(2).stride() == 1 &&
207 output.dim(2).min() == 0 &&
208 output.dim(2).extent() == 3 &&
209 input.dim(0).stride() == 3 &&
210 input.dim(2).stride() == 1 &&
211 input.dim(2).min() == 0 &&
212 input.dim(2).extent() == 3);
213 Expr packed_rgba = (output.dim(0).stride() == 4 &&
214 output.dim(2).stride() == 1 &&
215 output.dim(2).min() == 0 &&
216 output.dim(2).extent() == 4 &&
217 input.dim(0).stride() == 4 &&
218 input.dim(2).stride() == 1 &&
219 input.dim(2).min() == 0 &&
220 input.dim(2).extent() == 4);
221
222 output.specialize(planar);
223
224 output.specialize(packed_rgb)
225 .reorder(c, xi, yi, x, y)
226 .unroll(c);
227
228 output.specialize(packed_rgba)
229 .reorder(c, xi, yi, x, y)
230 .unroll(c);
231 }
232};
