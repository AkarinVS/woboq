1// Small stub to produce a module-specific entry point
2// for a Python module. It assumes/requires that it is
3// linked with PyStubImpl.o to be useful.
4//
5// Note that this quite deliberately doesn't include any PyBind11 headers;
6// it is designed to be compiled without PyBind11 being available at compile time,
7// to simplify build requirements in downstream environments.
8
9#include <Python.h>
10#include <functional>
11#include <memory>
12
13#include "Halide.h"
14
15#ifndef HALIDE_PYSTUB_GENERATOR_NAME
16#error "HALIDE_PYSTUB_GENERATOR_NAME must be defined"
17#endif
18
19#ifndef HALIDE_PYSTUB_MODULE_NAME
20#define HALIDE_PYSTUB_MODULE_NAME HALIDE_PYSTUB_GENERATOR_NAME
21#endif
22
23extern "C" PyObject *_halide_pystub_impl(const char *module_name, const Halide::Internal::GeneratorFactory &factory);
24
25#define HALIDE_STRINGIFY(x) #x
26#define HALIDE_TOSTRING(x) HALIDE_STRINGIFY(x)
27
28#define _HALIDE_CONCAT(first, second) first##second
29#define HALIDE_CONCAT(first, second) _HALIDE_CONCAT(first, second)
30
31// Don't use HALIDE_EXPORT: Halide.h will already have defined it,
32// but it might be defined the wrong way (import rather than export).
33#if defined(WIN32) || defined(_WIN32)
34#define HALIDE_PLUGIN_EXPORT __declspec(dllexport)
35#else
36#define HALIDE_PLUGIN_EXPORT __attribute__((visibility("default")))
37#endif
38
39static_assert(PY_MAJOR_VERSION >= 3, "Python bindings for Halide require Python 3+");
40
41#define _HALIDE_PLUGIN_IMPL(name) extern "C" HALIDE_PLUGIN_EXPORT PyObject *PyInit_##name() /* NOLINT(bugprone-macro-parentheses) */
42#define HALIDE_PLUGIN_IMPL(name) _HALIDE_PLUGIN_IMPL(name)
43
44// clang-format off
45
46namespace halide_register_generator {
47namespace HALIDE_CONCAT(HALIDE_PYSTUB_GENERATOR_NAME, _ns) {
48 extern std::unique_ptr<Halide::Internal::GeneratorBase> factory(const Halide::GeneratorContext &context);
49} // namespace HALIDE_CONCAT(HALIDE_PYSTUB_GENERATOR_NAME,_ns)
50} // namespace halide_register_generator
51
52HALIDE_PLUGIN_IMPL(HALIDE_PYSTUB_MODULE_NAME) {
53 const auto factory = halide_register_generator::HALIDE_CONCAT(HALIDE_PYSTUB_GENERATOR_NAME, _ns)::factory;
54 return _halide_pystub_impl(HALIDE_TOSTRING(HALIDE_PYSTUB_MODULE_NAME), factory);
55}
