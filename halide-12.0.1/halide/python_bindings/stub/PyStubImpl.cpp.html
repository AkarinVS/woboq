68// Anything that defines __getitem__ looks sequencelike to pybind,
69// so also check for __len_ to avoid things like Buffer and Func here.
70bool is_real_sequence(const py::object &o) {
71 return py::isinstance<py::sequence>(o) && py::hasattr(o, "__len__");
72}
73
74StubInput to_stub_input(const py::object &o) {
75 // Don't use isinstance: we want to get things that
76 // can be implicitly converted as well (eg ImageParam -> Func)
77 try {
78 return StubInput(StubInputBuffer(o.cast<Buffer<>>()));
79 } catch (...) {
80 // Not convertible to Buffer. Fall thru and try next.
81 }
82
83 try {
84 return StubInput(o.cast<Func>());
85 } catch (...) {
86 // Not convertible to Func. Fall thru and try next.
87 }
88
89 return StubInput(o.cast<Expr>());
90}

92std::vector<StubInput> to_stub_inputs(const py::object &value) {
93 if (is_real_sequence(value)) {
94 std::vector<StubInput> v;
95 for (const auto &o : py::reinterpret_borrow<py::sequence>(value)) {
96 v.push_back(to_stub_input(o));
97 }
98 return v;
99 } else {
100 return {to_stub_input(value)};
101 }
102}

104py::object generate_impl(const GeneratorFactory &factory, const GeneratorContext &context, const py::args &args, const py::kwargs &kwargs) {
105 Stub stub(context, [factory](const GeneratorContext &context) -> std::unique_ptr<Halide::Internal::GeneratorBase> {
106 return factory(context);
107 });
108 auto names = stub.get_names();
109 _halide_user_assert(!names.outputs.empty())
110 << "Generators that use build() (instead of generate()+Output<>) are not supported in the Python bindings.";
111
112 // Inputs can be specified by either positional or named args,
113 // but may not be mixed. (i.e., if any inputs are specified as a named
114 // arg, they all must be specified that way; otherwise they must all be
115 // positional, in the order declared in the Generator.)
116 //
117 // GeneratorParams can only be specified by name, and are always optional.
118
119 std::map<std::string, std::vector<StubInput>> kw_inputs;
120 for (const auto &name : names.inputs) {
121 _halide_user_assert(kw_inputs.count(name) == 0); // internal error
122 kw_inputs[name] = std::vector<StubInput>{};
123 }
124 size_t kw_inputs_specified = 0;
125
126 GeneratorParamsMap generator_params;
127
128 // Process the kwargs first.
129 for (auto kw : kwargs) {
130 // If the kwarg is the name of a known input, stick it in the input
131 // vector. If not, stick it in the GeneratorParamsMap (if it's invalid,
132 // an error will be reported further downstream).
133 std::string name = kw.first.cast<std::string>();
134 py::handle value = kw.second;
135 auto it = kw_inputs.find(name);
136 if (it != kw_inputs.end()) {
137 _halide_user_assert(it->second.empty())
138 << "Generator Input named '" << it->first << "' was specified more than once.";
139 it->second = to_stub_inputs(py::cast<py::object>(value));
140 kw_inputs_specified++;
141 } else {
142 if (py::isinstance<LoopLevel>(value)) {
143 generator_params[name] = value.cast<LoopLevel>();
144 } else {
145 generator_params[name] = py::str(value).cast<std::string>();
146 }
147 }
148 }
149
150 std::vector<std::vector<StubInput>> inputs;
151 inputs.reserve(names.inputs.size());
152
153 if (args.empty()) {
154 // No arguments specified positionally, so they must all be via keywords.
155 _halide_user_assert(kw_inputs_specified == names.inputs.size())
156 << "Expected exactly " << names.inputs.size() << " keyword args for inputs, but saw " << kw_inputs_specified << ".";
157 for (const auto &name : names.inputs) {
158 inputs.push_back(std::move(kw_inputs[name]));
159 }
160 } else {
161 // Some positional arguments, so all inputs must be positional (and none via keyword).
162 _halide_user_assert(kw_inputs_specified == 0)
163 << "Cannot use both positional and keyword arguments for inputs.";
164 _halide_user_assert(args.size() == names.inputs.size())
165 << "Expected exactly " << names.inputs.size() << " positional args for inputs, but saw " << args.size() << ".";
166 for (auto arg : args) {
167 inputs.push_back(to_stub_inputs(py::cast<py::object>(arg)));
168 }
169 }
170
171 // Verify everything is there
172 _halide_user_assert(inputs.size() == names.inputs.size());
173 for (size_t i = 0; i < inputs.size(); ++i) {
174 _halide_user_assert(!inputs[i].empty())
175 << "Generator Input named '" << names.inputs[i] << "' was not specified.";
176 }
177
178 const std::vector<std::vector<Func>> outputs = stub.generate(generator_params, inputs);
179
180 py::tuple py_outputs(outputs.size());
181 for (size_t i = 0; i < outputs.size(); i++) {
182 py::object o;
183 if (outputs[i].size() == 1) {
184 // convert list-of-1 into single element
185 o = py::cast(outputs[i][0]);
186 } else {
187 o = py::cast(outputs[i]);
188 }
189 if (outputs.size() == 1) {
190 // bail early, return the single object rather than a dict
191 return o;
192 }
193 py_outputs[i] = o;
194 }
195 // An explicit "std::move" is needed here because there's
196 // an implicit tuple->object conversion that inhibits it otherwise.
197 return std::move(py_outputs);
198}

200void pystub_init(pybind11::module &m, const GeneratorFactory &factory) {
201 m.def(
202 "generate", [factory](const Halide::Target &target, const py::args &args, const py::kwargs &kwargs) -> py::object {
203 return generate_impl(factory, Halide::GeneratorContext(target), args, kwargs);
204 },
205 py::arg("target"));
206}

212extern "C" PyObject *_halide_pystub_impl(const char *module_name, const Halide::Internal::GeneratorFactory &factory) {
213 int major, minor;
214 if (sscanf(Py_GetVersion(), "%i.%i", &major, &minor) != 2) {
215 PyErr_SetString(PyExc_ImportError, "Can't parse Python version.");
216 return nullptr;
217 } else if (major != PY_MAJOR_VERSION || minor != PY_MINOR_VERSION) {
218 PyErr_Format(PyExc_ImportError,
219 "Python version mismatch: module was compiled for "
220 "version %i.%i, while the interpreter is running "
221 "version %i.%i.",
222 PY_MAJOR_VERSION, PY_MINOR_VERSION,
223 major, minor);
224 return nullptr;
225 }
226
227 // TODO: do something meaningful with the PyModuleDef & add a doc string
228 auto m = pybind11::module_::create_extension_module(module_name, nullptr, new PyModuleDef());
229 try {
230 Halide::PythonBindings::install_error_handlers(m);
231 Halide::PythonBindings::pystub_init(m, factory);
232 return m.ptr();
233 } catch (pybind11::error_already_set &e) {
234 PyErr_SetString(PyExc_ImportError, e.what());
235 return nullptr;
236 } catch (const std::exception &e) {
237 PyErr_SetString(PyExc_ImportError, e.what());
238 return nullptr;
239 }
