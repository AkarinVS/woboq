23std::vector<T> args_to_vector(const py::args &args, size_t start_offset = 0, size_t end_offset = 0) {
24 if (args.size() < start_offset + end_offset) {
25 throw py::value_error("Not enough arguments");
26 }
27 std::vector<T> v;
28 v.reserve(args.size() - (start_offset + end_offset));
29 for (size_t i = start_offset; i < args.size() - end_offset; ++i) {
30 v.push_back(args[i].cast<T>());
31 }
32 return v;
33}
