8void define_rvar(py::module &m) {
9 auto rvar_class =
10 py::class_<RVar>(m, "RVar")
11 .def(py::init<>())
12 .def(py::init<std::string>(), py::arg("name"))
13 .def(py::init([](const RDom &r) -> RVar { return r; }))
14 .def("min", &RVar::min)
15 .def("extent", &RVar::extent)
16 .def("name", &RVar::name)
17 .def("__repr__", [](const RVar &v) -> std::string {
18 std::ostringstream o;
19 o << "<halide.RVar " << v << ">";
20 return o.str();
21 });
22
23 py::implicitly_convertible<RDom, RVar>();
24
25 add_binary_operators(rvar_class);
26}

28void define_rdom(py::module &m) {
29 define_rvar(m);
30
31 auto rdom_class =
32 py::class_<RDom>(m, "RDom")
33 .def(py::init<>())
34 .def(py::init<Buffer<>>(), py::arg("buffer"))
35 .def(py::init<OutputImageParam>(), py::arg("image_param"))
36 .def(py::init<const Region &, std::string>(), py::arg("region"), py::arg("name") = "")
37 .def("domain", &RDom::domain)
38 .def("defined", &RDom::defined)
39 .def("same_as", &RDom::same_as)
40 .def("dimensions", &RDom::dimensions)
41 .def("__len__", &RDom::dimensions)
42 .def("where", &RDom::where, py::arg("predicate"))
43 .def("__getitem__", [](RDom &r, const int i) -> RVar {
44 if (i < 0 || i >= r.dimensions()) {
45 throw pybind11::key_error();
46 }
47 return r[i];
48 })
49 .def_readonly("x", &RDom::x)
50 .def_readonly("y", &RDom::y)
51 .def_readonly("z", &RDom::z)
52 .def_readonly("w", &RDom::w)
53 .def("__repr__", [](const RDom &r) -> std::string {
54 std::ostringstream o;
55 o << "<halide.RDom " << r << ">";
56 return o.str();
57 });
58
59 add_binary_operators(rdom_class);
60}
