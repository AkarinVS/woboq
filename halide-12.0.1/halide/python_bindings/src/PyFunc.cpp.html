51// Special case: there is no implicit conversion of double in C++ Halide
52// but there is implicit conversion of double in Python.
53template<>
54void define_set_func_ref<double>(py::class_<Func> &func_class) {
55 func_class
56 .def("__setitem__",
57 [](Func &func, const FuncRef &lhs, const double &rhs) -> Stage {
58 // Implicitly convert rhs to single precision. Issue warnings if
59 // we detect loss of precision.
60 float f = rhs;
61 if (Internal::reinterpret_bits<uint64_t>(rhs) !=
62 Internal::reinterpret_bits<uint64_t>((double)f)) {
63 double diff = rhs - f;
64 std::ostringstream os;
65 os << "Loss of precision detected when casting " << rhs << " to a single precision float. The difference is " << diff << ".";
66 std::string msg = os.str();
67 PyErr_WarnEx(nullptr, msg.c_str(), 1);
68 }
69 return func(lhs) = Expr(f);
70 });
71}

73py::object realization_to_object(const Realization &r) {
74 // Only one Buffer -> just return it
75 if (r.size() == 1) {
76 return py::cast(r[0]);
77 }
78
79 // Multiple -> return as Python tuple
80 return to_python_tuple(r);
81}

361 // Special cases of f[g[...]] = ...
362 // We need to capture this case here since otherwise
363 // pybind11 would try to convert g[...], which is a FuncRef
364 // to a list of Var.
365 // However, to do this it has to check g[...][0] is a Var or not.
366 // If g is not defined as a Tuple this results in a runtime error
367 // inside Halide.
368 // We also can't rely on implicit conversion in pybind11 since it
369 // happens after all function overloads have been visited.
