43 .def("set_max_value", &Param<>::set_max_value)
44 .def("min_value", &Param<>::min_value)
45 .def("max_value", &Param<>::max_value)
46
47 .def("__repr__", [](const Param<> &param) -> std::string {
48 std::ostringstream o;
49 o << "<halide.Param '" << param.name() << "'"
