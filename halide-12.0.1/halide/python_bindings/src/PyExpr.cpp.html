11void define_expr(py::module &m) {
12 auto to_bool = [](const Expr &e) -> bool {
13 std::ostringstream o;
14 o << e;
15 throw py::value_error("The halide.Expr (" +
16 o.str() +
17 ") cannot be converted to a bool. "
18 "If this error occurs using the 'and'/'or' keywords, "
19 "consider using the '&'/'|' operators instead.");
20 return false;
21 };
22
23 auto expr_class =
24 py::class_<Expr>(m, "Expr")
25 .def(py::init<>())
26 // PyBind11 searches in declared order,
27 // int should be tried before float conversion
28 .def(py::init<int>())
29 // Python float is implemented by double
30 // But Halide prohibits implicitly construct by double.
31 .def(py::init([](double v) {
32 return double_to_expr_check(v);
33 }))
34 .def(py::init<std::string>())
35
36 // for implicitly_convertible
37 .def(py::init([](const FuncRef &f) -> Expr { return f; }))
38 .def(py::init([](const FuncTupleElementRef &f) -> Expr { return f; }))
39 .def(py::init([](const Param<> &p) -> Expr { return p; }))
40 .def(py::init([](const RDom &r) -> Expr { return r; }))
41 .def(py::init([](const RVar &r) -> Expr { return r; }))
42 .def(py::init([](const Var &v) -> Expr { return v; }))
43
44 .def("__bool__", to_bool)
45 .def("__nonzero__", to_bool)
46
47 .def("type", &Expr::type)
48 .def("__repr__", [](const Expr &e) -> std::string {
49 std::ostringstream o;
50 o << "<halide.Expr of type " << halide_type_to_string(e.type()) << ": " << e << ">";
51 return o.str();
52 });
53
54 add_binary_operators(expr_class);
55
56 // implicitly_convertible declaration order matters,
57 // int should be tried before float conversion
58 py::implicitly_convertible<int, Expr>();
59 py::implicitly_convertible<float, Expr>();
60 py::implicitly_convertible<double, Expr>();
61
62 // There must be an Expr() ctor available for each of these
63 py::implicitly_convertible<FuncRef, Expr>();
64 py::implicitly_convertible<FuncTupleElementRef, Expr>();
65 py::implicitly_convertible<Param<>, Expr>();
66 py::implicitly_convertible<RDom, Expr>();
67 py::implicitly_convertible<RVar, Expr>();
68 py::implicitly_convertible<Var, Expr>();
69
70 auto range_class =
71 py::class_<Range>(m, "Range")
72 .def(py::init<>())
73 .def(py::init([](const Expr &min, const Expr &extent) -> Range {
74 return Range(min, extent);
75 }))
76 // Allow implicit conversion from py::tuple -> Range, iff py::tuple.size() == 2
77 .def(py::init([](const py::tuple &t) -> Range {
78 if (t.size() != 2) {
79 throw py::value_error("Halide::Range requires exactly two values");
80 }
81 Expr min = t[0].cast<Expr>();
82 Expr extent = t[1].cast<Expr>();
83 return Range(min, extent);
84 }))
85 .def_readwrite("min", &Range::min)
86 .def_readwrite("extent", &Range::extent);
87
88 // Allow implicit tuple->Range conversion, so that things like
89 // [(0, W), (0, H)] work for BoundaryConditions and Estimates
90 py::implicitly_convertible<py::tuple, Range>();
91}
