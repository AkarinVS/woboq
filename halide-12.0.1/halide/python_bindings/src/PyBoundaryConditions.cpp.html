14void define_boundary_conditions(py::module &m) {
15 using namespace BoundaryConditions;

17 py::module bc = m.def_submodule("BoundaryConditions");
18
19 // This code could be made less redundant with some templating,
20 // but because of the templated nature of BoundaryConditions itself,
21 // we'd need at least a couple of template levels; not clear it would

24 // ----- constant_exterior
25
26 bc.def(
27 "constant_exterior", [](const ImageParam &im, const Expr &exterior) -> Func {
28 return constant_exterior(im, exterior);
29 },
30 py::arg("f"), py::arg("exterior"));
31 bc.def(
32 "constant_exterior", [](const Buffer<> &b, const Expr &exterior) -> Func {
33 return constant_exterior(b, exterior);
34 },
35 py::arg("f"), py::arg("exterior"));
36 bc.def(
37 "constant_exterior", [](const py::object &target, const Expr &exterior, const Region &bounds) -> Func {
38 try {
39 return constant_exterior(target.cast<Func>(), exterior, bounds);
40 } catch (...) {
41 // fall thru
42 }
43 try {
44 return constant_exterior(to_func(target.cast<Buffer<>>()), exterior, bounds);
45 } catch (...) {
46 // fall thru
47 }
48 throw py::value_error("Invalid arguments to constant_exterior");
49 return Func();
50 },
51 py::arg("f"), py::arg("exterior"), py::arg("bounds"));
