13std::vector<Expr> args_to_vector_for_print(const py::args &args, size_t start_offset = 0) {
14 if (args.size() < start_offset) {
15 throw py::value_error("Not enough arguments");
16 }
17 std::vector<Expr> v;
18 v.reserve(args.size() - (start_offset));
19 for (size_t i = start_offset; i < args.size(); ++i) {
20 // No way to see if a cast will work: just have to try
21 // and fail. Normally we don't want string to be convertible
22 // to Expr, but in this unusual case we do.
23 try {
24 v.emplace_back(args[i].cast<std::string>());
25 } catch (...) {
26 v.push_back(args[i].cast<Expr>());
27 }
28 }
29 return v;
30}

35 m.def("max", [](const py::args &args) -> Expr {
36 if (args.size() < 2) {
37 throw py::value_error("max() must have at least 2 arguments");
38 }
39 int pos = (int)args.size() - 1;
40 Expr value = args[pos--].cast<Expr>();
41 while (pos >= 0) {
42 value = max(args[pos--].cast<Expr>(), value);
43 }

63 m.def("select", [](const py::args &args) -> Expr {
64 if (args.size() < 3) {
65 throw py::value_error("select() must have at least 3 arguments");
66 }
67 if ((args.size() % 2) == 0) {
68 throw py::value_error("select() must have an odd number of arguments");
69 }
70 int pos = (int)args.size() - 1;
71 Expr false_value = args[pos--].cast<Expr>();
72 while (pos > 0) {
73 Expr true_value = args[pos--].cast<Expr>();
74 Expr condition = args[pos--].cast<Expr>();
75 false_value = select(condition, true_value, false_value);
76 }
77 return false_value;
78 });

80 m.def("tuple_select", [](const py::args &args) -> py::tuple {
81 _halide_user_assert(args.size() >= 3)
82 << "tuple_select() must have at least 3 arguments";
83 _halide_user_assert((args.size() % 2) != 0)
84 << "tuple_select() must have an odd number of arguments";
85
86 int pos = (int)args.size() - 1;
87 Tuple false_value = args[pos--].cast<Tuple>();
88 bool has_tuple_cond = false, has_expr_cond = false;
89 while (pos > 0) {
90 Tuple true_value = args[pos--].cast<Tuple>();
91 ;
92 // Note that 'condition' can be either Expr or Tuple, but must be consistent across all
93 py::object py_cond = args[pos--];
94 Expr expr_cond;
95 Tuple tuple_cond(expr_cond);
96 try {
97 tuple_cond = py_cond.cast<Tuple>();
98 has_tuple_cond = true;
99 } catch (...) {
100 expr_cond = py_cond.cast<Expr>();
101 has_expr_cond = true;
102 }
103
104 if (expr_cond.defined()) {
105 false_value = tuple_select(expr_cond, true_value, false_value);
106 } else {
107 false_value = tuple_select(tuple_cond, true_value, false_value);
108 }
109 }
110 _halide_user_assert(!(has_tuple_cond && has_expr_cond))
111 << "tuple_select() may not mix Expr and Tuple for the condition elements.";
112 return to_python_tuple(false_value);
113 });
