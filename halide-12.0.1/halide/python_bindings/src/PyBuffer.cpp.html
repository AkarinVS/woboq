34// Given a Buffer<>, return its shape in the form of a vector<halide_dimension_t>.
35// (Oddly, Buffer<> has no API to do this directly.)
36std::vector<halide_dimension_t> get_buffer_shape(const Buffer<> &b) {
37 if (!b.defined()) {
38 // Return an empty vector if the buffer is not defined.
39 return {};
40 }
41 std::vector<halide_dimension_t> s;
42 for (int i = 0; i < b.dimensions(); ++i) {
43 s.push_back(b.raw_buffer()->dim[i]);
44 }
45 return s;
46}

233// Use an alias class so that if we are created via a py::buffer, we can
234// keep the py::buffer_info class alive for the life of the Buffer<>,
235// ensuring the data isn't collected out from under us.
236class PyBuffer : public Buffer<> {
237 py::buffer_info info;
238
239 static std::vector<halide_dimension_t> make_dim_vec(const py::buffer_info &info) {
240 const Type t = format_descriptor_to_type(info.format);
241 std::vector<halide_dimension_t> dims;
242 dims.reserve(info.ndim);
243 for (int i = 0; i < info.ndim; i++) {
244 if (INT_MAX < info.shape[i] || INT_MAX < (info.strides[i] / t.bytes())) {
245 throw py::value_error("Out of range arguments to make_dim_vec.");
246 }
247 dims.emplace_back(0, (int32_t)info.shape[i], (int32_t)(info.strides[i] / t.bytes()));
248 }
249 return dims;
250 }
251
252 PyBuffer(py::buffer_info &&info, const std::string &name)
253 : Buffer<>(
254 format_descriptor_to_type(info.format),
255 info.ptr,
256 (int)info.ndim,
257 make_dim_vec(info).data(),
258 name),
259 info(std::move(info)) {
260 }
261
262public:
263 PyBuffer()
264 : Buffer<>(), info() {
265 }
266
267 explicit PyBuffer(const Buffer<> &b)
268 : Buffer<>(b), info() {
269 }
270
271 PyBuffer(const py::buffer &buffer, const std::string &name)
272 : PyBuffer(buffer.request(/*writable*/ true), name) {
273 // Default to setting host-dirty on any PyBuffer we create from an existing py::buffer;
274 // this allows (e.g.) code like
275 //
276 // input = hl.Buffer(imageio.imread(image_path))
277 //
278 // to work as you'd expected when the buffer is used for input. (Without
279 // host_dirty being set, copying to GPU can be skipped and produce surprising
280 // results.)
281 //
282 // Note that this is a bit different from the C++ Buffer<> ctors that
283 // take pointer-to-existing data (which do *not* set host dirty); a crucial
284 // difference there is that those also do not take ownership, but here,
285 // we always take (shared) ownership of the underlying buffer.
286 this->set_host_dirty();
287 }
288
289 ~PyBuffer() override = default;
290};

308 // most notably, we can convert to an ndarray by calling numpy.array()
309 .def_buffer([](Buffer<> &b) -> py::buffer_info {
310 if (b.data() == nullptr) {
311 throw py::value_error("Cannot convert a Buffer<> with null host ptr to a Python buffer.");
312 }
313
314 const int d = b.dimensions();
315 const int bytes = b.type().bytes();
316 std::vector<ssize_t> shape, strides;
317 for (int i = 0; i < d; i++) {
318 shape.push_back((ssize_t)b.raw_buffer()->dim[i].extent);
319 strides.push_back((ssize_t)(b.raw_buffer()->dim[i].stride * bytes));
320 }
321
322 return py::buffer_info(
323 b.data(), // Pointer to buffer
324 bytes, // Size of one scalar
325 type_to_format_descriptor(b.type()), // Python struct-style format descriptor
326 d, // Number of dimensions
327 shape, // Buffer dimensions
328 strides // Strides (in bytes) for each index
329 );
330 })
