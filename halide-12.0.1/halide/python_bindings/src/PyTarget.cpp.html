16void define_target(py::module &m) {
17 // Disambiguate some ambigious methods
18 int (Target::*natural_vector_size_method)(const Type &t) const = &Target::natural_vector_size;
19 bool (Target::*supports_type1_method)(const Type &t) const = &Target::supports_type;
20 bool (Target::*supports_type2_method)(const Type &t, DeviceAPI device) const = &Target::supports_type;
21
22 auto target_class =
23 py::class_<Target>(m, "Target")
24 .def(py::init<>())
25 .def(py::init<const std::string &>())
26 .def(py::init<Target::OS, Target::Arch, int>())
27 .def(py::init<Target::OS, Target::Arch, int, std::vector<Target::Feature>>())
28
29 .def("__eq__", [](const Target &value, Target *value2) { return value2 && value == *value2; })
30 .def("__ne__", [](const Target &value, Target *value2) { return !value2 || value != *value2; })
31
32 .def_readwrite("os", &Target::os)
33 .def_readwrite("arch", &Target::arch)
34 .def_readwrite("bits", &Target::bits)
35
36 .def("__repr__", &target_repr)
37 .def("__str__", &Target::to_string)
38 .def("to_string", &Target::to_string)
39
40 .def("has_feature", (bool (Target::*)(Target::Feature) const) & Target::has_feature)
41 .def("features_any_of", &Target::features_any_of, py::arg("features"))
42 .def("features_all_of", &Target::features_all_of, py::arg("features"))
43
44 .def("set_feature", &Target::set_feature, py::arg("f"), py::arg("value") = true)
45 .def("set_features", &Target::set_features, py::arg("features"), py::arg("value") = true)
46 .def("with_feature", &Target::with_feature, py::arg("feature"))
47 .def("without_feature", &Target::without_feature, py::arg("feature"))
48 .def("has_gpu_feature", &Target::has_gpu_feature)
49 .def("supports_type", supports_type1_method, py::arg("type"))
50 .def("supports_type", supports_type2_method, py::arg("type"), py::arg("device"))
51 .def("supports_device_api", &Target::supports_device_api, py::arg("device"))
52 .def("natural_vector_size", natural_vector_size_method, py::arg("type"))
53 .def("has_large_buffers", &Target::has_large_buffers)
54 .def("maximum_buffer_size", &Target::maximum_buffer_size)
55 .def("supported", &Target::supported)
56 .def_static("validate_target_string", &Target::validate_target_string, py::arg("name"));
57 ;
58
59 m.def("get_host_target", &get_host_target);
60 m.def("get_target_from_environment", &get_target_from_environment);
61 m.def("get_jit_target_from_environment", &get_jit_target_from_environment);
62 m.def("target_feature_for_device_api", &target_feature_for_device_api);
63
64 // TODO: this really belong in PyDeviceInterface.cpp (once it exists);
65 // it's here as an expedient to make our tutorials work more cleanly.
66 m.def("host_supports_target_device", &host_supports_target_device);
67}
