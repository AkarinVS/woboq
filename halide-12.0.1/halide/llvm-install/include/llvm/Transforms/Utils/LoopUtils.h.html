67/// Ensures LCSSA form for every instruction from the Worklist in the scope of
68/// innermost containing loop.
69///
70/// For the given instruction which have uses outside of the loop, an LCSSA PHI
71/// node is inserted and the uses outside the loop are rewritten to use this
72/// node.
73///
74/// LoopInfo and DominatorTree are required and, since the routine makes no
75/// changes to CFG, preserved.
76///
77/// Returns true if any modifications are made.
78///
79/// This function may introduce unused PHI nodes. If \p PHIsToRemove is not
80/// nullptr, those are added to it (before removing, the caller has to check if
81/// they still do not have any uses). Otherwise the PHIs are directly removed.
82bool formLCSSAForInstructions(
83 SmallVectorImpl<Instruction *> &Worklist, const DominatorTree &DT,
84 const LoopInfo &LI, ScalarEvolution *SE, IRBuilderBase &Builder,
85 SmallVectorImpl<PHINode *> *PHIsToRemove = nullptr);

87/// Put loop into LCSSA form.
88///
89/// Looks at all instructions in the loop which have uses outside of the
90/// current loop. For each, an LCSSA PHI node is inserted and the uses outside
91/// the loop are rewritten to use this node. Sub-loops must be in LCSSA form
92/// already.
93///
94/// LoopInfo and DominatorTree are required and preserved.
95///
96/// If ScalarEvolution is passed in, it will be preserved.
97///
98/// Returns true if any modifications are made to the loop.
99bool formLCSSA(Loop &L, const DominatorTree &DT, const LoopInfo *LI,
100 ScalarEvolution *SE);

268/// The mode sets how eager a transformation should be applied.
269enum TransformationMode {
270 /// The pass can use heuristics to determine whether a transformation should
271 /// be applied.
272 TM_Unspecified,
273
274 /// The transformation should be applied without considering a cost model.
275 TM_Enable,
276
277 /// The transformation should not be applied.
278 TM_Disable,
279
280 /// Force is a flag and should not be used alone.
281 TM_Force = 0x04,
282
283 /// The transformation was directed by the user, e.g. by a #pragma in
284 /// the source code. If the transformation could not be applied, a
285 /// warning should be emitted.
286 TM_ForcedByUser = TM_Enable | TM_Force,
287
288 /// The transformation must not be applied. For instance, `#pragma clang loop
289 /// unroll(disable)` explicitly forbids any unrolling to take place. Unlike
290 /// general loop metadata, it must not be dropped. Most passes should not
291 /// behave differently under TM_Disable and TM_SuppressedByUser.
292 TM_SuppressedByUser = TM_Disable | TM_Force
293};

444/// Utility that implements appending of loops onto a worklist given a range.
445/// We want to process loops in postorder, but the worklist is a LIFO data
446/// structure, so we append to it in *reverse* postorder.
447/// For trees, a preorder traversal is a viable reverse postorder, so we
448/// actually append using a preorder walk algorithm.
449template <typename RangeT>
450void appendLoopsToWorklist(RangeT &&, SmallPriorityWorklist<Loop *, 4> &);
451/// Utility that implements appending of loops onto a worklist given a range.
452/// It has the same behavior as appendLoopsToWorklist, but assumes the range of
453/// loops has already been reversed, so it processes loops in the given order.
454template <typename RangeT>
455void appendReversedLoopsToWorklist(RangeT &&,
456 SmallPriorityWorklist<Loop *, 4> &);

458/// Utility that implements appending of loops onto a worklist given LoopInfo.
459/// Calls the templated utility taking a Range of loops, handing it the Loops
460/// in LoopInfo, iterated in reverse. This is because the loops are stored in
461/// RPO w.r.t. the control flow graph in LoopInfo. For the purpose of unrolling,
462/// loop deletion, and LICM, we largely want to work forward across the CFG so
463/// that we visit defs before uses and can propagate simplifications from one
464/// loop nest into the next. Calls appendReversedLoopsToWorklist with the
465/// already reversed loops in LI.
466/// FIXME: Consider changing the order in LoopInfo.
467void appendLoopsToWorklist(LoopInfo &, SmallPriorityWorklist<Loop *, 4> &);
