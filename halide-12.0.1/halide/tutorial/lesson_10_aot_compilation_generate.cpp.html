1// Halide tutorial lesson 10: AOT compilation part 1
2
3// This lesson demonstrates how to use Halide as an more traditional
4// ahead-of-time (AOT) compiler.
5
6// This lesson is split across two files. The first (this one), builds
7// a Halide pipeline and compiles it to a static library and
8// header. The second (lesson_10_aot_compilation_run.cpp), uses that
9// static library to actually run the pipeline. This means that
10// compiling this code is a multi-step process.
11
12// On linux, you can compile and run it like so:
13// g++ lesson_10*generate.cpp -g -std=c++11 -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_10_generate
14// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_10_generate
15// g++ lesson_10*run.cpp lesson_10_halide.a -std=c++11 -I <path/to/Halide.h> -lpthread -ldl -o lesson_10_run
16// ./lesson_10_run
17
18// On os x:
19// g++ lesson_10*generate.cpp -g -std=c++11 -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_10_generate
20// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_10_generate
21// g++ lesson_10*run.cpp lesson_10_halide.a -o lesson_10_run -I <path/to/Halide.h>
22// ./lesson_10_run
23
24// The benefits of this approach are that the final program:
25// - Doesn't do any jit compilation at runtime, so it's fast.
26// - Doesn't depend on libHalide at all, so it's a small, easy-to-deploy binary.
27
28// If you have the entire Halide source tree, you can also build it by
29// running:
30// make tutorial_lesson_10_aot_compilation_run
31// in a shell with the current directory at the top of the halide
32// source tree.
33
34#include "Halide.h"
35#include <stdio.h>
36using namespace Halide;
37
38int main(int argc, char **argv) {
39
40 // We'll define a simple one-stage pipeline:
41 Func brighter;
42 Var x, y;
43
44 // The pipeline will depend on one scalar parameter.
45 Param<uint8_t> offset;
46
47 // And take one grayscale 8-bit input buffer. The first
48 // constructor argument gives the type of a pixel, and the second
49 // specifies the number of dimensions (not the number of
50 // channels!). For a grayscale image this is two; for a color
51 // image it's three. Currently, four dimensions is the maximum for
52 // inputs and outputs.
53 ImageParam input(type_of<uint8_t>(), 2);
54
55 // If we were jit-compiling, these would just be an int and a
56 // Buffer, but because we want to compile the pipeline once and
57 // have it work for any value of the parameter, we need to make a
58 // Param object, which can be used like an Expr, and an ImageParam
59 // object, which can be used like a Buffer.
60
61 // Define the Func.
62 brighter(x, y) = input(x, y) + offset;
63
64 // Schedule it.
65 brighter.vectorize(x, 16).parallel(y);
66
67 // This time, instead of calling brighter.realize(...), which
68 // would compile and run the pipeline immediately, we'll call a
69 // method that compiles the pipeline to a static library and header.
70 //
71 // For AOT-compiled code, we need to explicitly declare the
72 // arguments to the routine. This routine takes two. Arguments are
73 // usually Params or ImageParams.
74 brighter.compile_to_static_library("lesson_10_halide", {input, offset}, "brighter");
75
76 printf("Halide pipeline compiled, but not yet run.\n");
77
78 // To continue this lesson, look in the file lesson_10_aot_compilation_run.cpp
79
80 return 0;
81}
