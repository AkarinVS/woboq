1// Halide tutorial lesson 21: Auto-Scheduler
2
3// Before reading this file, see lesson_21_auto_scheduler_generate.cpp
4
5// This is the code that actually uses the Halide pipeline we've
6// compiled. It does not depend on libHalide, so we won't be including
7// Halide.h.
8//
9// Instead, it depends on the header files that lesson_21_auto_scheduler_generator produced.
10#include "auto_schedule_false.h"
11#include "auto_schedule_true.h"
12
13// We'll use the Halide::Runtime::Buffer class for passing data into and out of
14// the pipeline.
15#include "HalideBuffer.h"
16#include "halide_benchmark.h"
17
18#include <assert.h>
19#include <stdio.h>
20#include <stdlib.h>
21#include <string.h>
22
23int main(int argc, char **argv) {
24 // Let's declare and initialize the input images
25 Halide::Runtime::Buffer<float> input(1024, 1024, 3);
26
27 for (int c = 0; c < input.channels(); ++c) {
28 for (int y = 0; y < input.height(); ++y) {
29 for (int x = 0; x < input.width(); ++x) {
30 input(x, y, c) = rand();
31 }
32 }
33 }
34
35 Halide::Runtime::Buffer<float> output1(1024, 1024);
36 Halide::Runtime::Buffer<float> output2(1024, 1024);
37 // Run each version of the codes (with no auto-schedule and with
38 // auto-schedule) multiple times for benchmarking.
39 double auto_schedule_off = Halide::Tools::benchmark(2, 5, [&]() {
40 auto_schedule_false(input, 2.0f, output1, output2);
41 });
42 printf("Manual schedule: %gms\n", auto_schedule_off * 1e3);
43
44 double auto_schedule_on = Halide::Tools::benchmark(2, 5, [&]() {
45 auto_schedule_true(input, 2.0f, output1, output2);
46 });
47 printf("Auto schedule: %gms\n", auto_schedule_on * 1e3);
48
49 // auto_schedule_on should be faster since in the auto_schedule_off version,
50 // the schedule is very simple.
51 if (!(auto_schedule_on < auto_schedule_off)) {
52 fprintf(stderr, "Warning: expected auto_schedule_on < auto_schedule_off , "
53 "saw auto_schedule_on=%f auto_schedule_off=%f\n", auto_schedule_on, auto_schedule_off); \
54 }
55
56 return 0;
57}
