1// Halide tutorial lesson 10: AOT compilation part 2
2
3// Before reading this file, see lesson_10_aot_compilation_generate.cpp
4
5// This is the code that actually uses the Halide pipeline we've
6// compiled. It does not depend on libHalide, so we won't be including
7// Halide.h.
8//
9// Instead, it depends on the header file that lesson_10_generate
10// produced when we ran it:
11#include "lesson_10_halide.h"
12
13// We want to continue to use our Halide::Buffer with AOT-compiled
14// code, so we explicitly include it. It's a header-only class, and
15// doesn't require libHalide.
16#include "HalideBuffer.h"
17
18#include <stdio.h>
19
20int main(int argc, char **argv) {
21 // Have a look in the header file above (it won't exist until you've run
22 // lesson_10_generate). At the bottom is the signature of the function we generated:
23
24 // int brighter(halide_buffer_t *_input_buffer, uint8_t _offset, halide_buffer_t *_brighter_buffer);
25
26 // The ImageParam inputs have become pointers to "halide_buffer_t"
27 // structs. This is struct that Halide uses to represent arrays of
28 // data. Unless you're calling the Halide pipeline from pure C
29 // code, you don't want to use it
30 // directly. Halide::Runtime::Buffer is a simple wrapper around
31 // halide_buffer_t that will implicitly convert to a
32 // halide_buffer_t *. We will pass Halide::Runtime::Buffer objects
33 // in those slots.
34
35 // The Halide::Buffer class we have been using in JIT code is in
36 // fact just a shared pointer to the simpler
37 // Halide::Runtime::Buffer class. They share the same API.
38
39 // Finally, the return value of "brighter" is an error code. It's
40 // zero on success.
41
42 // Let's make a buffer for our input and output.
43 Halide::Runtime::Buffer<uint8_t> input(640, 480), output(640, 480);
44
45 // Halide::Runtime::Buffer also has constructors that wrap
46 // existing data instead of allocating new memory. Use these if
47 // you have your own Image type that you want to use.
48
49 int offset = 5;
50 int error = brighter(input, offset, output);
51
52 if (error) {
53 printf("Halide returned an error: %d\n", error);
54 return -1;
55 }
56
57 // Now let's check the filter performed as advertised. It was
58 // supposed to add the offset to every input pixel.
59 for (int y = 0; y < 480; y++) {
60 for (int x = 0; x < 640; x++) {
61 uint8_t input_val = input(x, y);
62 uint8_t output_val = output(x, y);
63 uint8_t correct_val = input_val + offset;
64 if (output_val != correct_val) {
65 printf("output(%d, %d) was %d instead of %d\n",
66 x, y, output_val, correct_val);
67 return -1;
68 }
69 }
70 }
71
72 // Everything worked!
73 printf("Success!\n");
74 return 0;
75}
