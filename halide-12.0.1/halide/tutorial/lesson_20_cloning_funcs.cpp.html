1// Halide tutorial lesson 20: Cloning Funcs
2
3// This lesson demonstrates how to use Func::clone_in to create a clone of
4// a Func.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_20*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_20 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_20
9
10// On os x:
11// g++ lesson_20*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_20 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_20
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_20_cloning_funcs
17// in a shell at the top of the halide source tree.
18
19// The only Halide header file you need is Halide.h. It includes all of Halide.
20#include "Halide.h"
21
22// We'll also include stdio for printf.
23#include <stdio.h>
24
25using namespace Halide;
26
27int main(int argc, char **argv) {
28 // First we'll declare some Vars to use below.
29 Var x("x"), y("y"), xo("xo"), yo("yo"), xi("xi"), yi("yi");
30
31 // This lesson will be about cloning a Func using the Func::clone_in
32 // directive.
33 {
34 // Consider a simple two-stage pipeline:
35 Func f("f_single"), g("g_single"), h("h_single");
36 f(x, y) = x + y;
37 g(x, y) = 2 * f(x, y) + 3;
38 h(x, y) = f(x, y) + g(x, y) + 10;
39
40 f.compute_root();
41 g.compute_root();
42 h.compute_root();
43
44 // This produces the following loop nests:
45 // for y:
46 // for x:
47 // f(x, y) = x + y
48 // for y:
49 // for x:
50 // g(x, y) = 2 * f(x, y) + 3
51 // for y:
52 // for x:
53 // h(x, y) = f(x, y) + g(x, y) + 10
54
55 // Using Func::clone_in, we can replace calls to 'f' inside 'g' with
56 // a clone of 'f' using the schedule alone:
57 Func f_clone_in_g = f.clone_in(g);
58 f_clone_in_g.compute_root();
59
60 // Equivalently, we could also chain the schedules like so:
61 // f.clone_in(g).compute_root();
62
63 // This produces the following loop nests:
64 // for y:
65 // for x:
66 // f(x, y) = x + y
67 // for y:
68 // for x:
69 // f_clone_in_g(x, y) = x + y
70 // for y:
71 // for x:
72 // g(x, y) = 2 * f_clone_in_g(x, y) + 3
73 // for y:
74 // for x:
75 // h(x, y) = f(x, y) + g(x, y) + 10
76
77 h.realize({5, 5});
78
79 // The schedule directive f.clone_in(g) replaces all calls to 'f'
80 // inside 'g' with a clone of 'f' and then returns that clone.
81 // Essentially, it rewrites the original pipeline above into the
82 // following:
83 {
84 Func f_clone_in_g("f_clone_in_g"), f("f"), g("g"), h("h");
85 f(x, y) = x + y;
86 f_clone_in_g(x, y) = x + y;
87 g(x, y) = 2 * f_clone_in_g(x, y) + 3;
88 h(x, y) = f(x, y) + g(x, y) + 10;
89
90 f.compute_root();
91 f_clone_in_g.compute_root();
92 g.compute_root();
93 h.compute_root();
94 }
95 }
96
97 {
98 // In the schedule above, only the calls to 'f' made by 'g' are
99 // replaced. Other calls made to 'f' would still call 'f' directly
100 // (i.e. 'h' still calls 'f' and not the clone). If we wish to
101 // replace all calls to 'f' made by both 'g' and 'h' with a single
102 // clone, we simply say f.clone_in({g, h}).
103
104 // Consider a three stage pipeline, with two consumers of f:
105 Func f("f_group"), g("g_group"), h("h_group"), out("out_group");
106 f(x, y) = x + y;
107 g(x, y) = 2 * f(x, y);
108 h(x, y) = f(x, y) + 10;
109 out(x, y) = f(x, y) + g(x, y) + h(x, y);
110
111 f.compute_root();
112 g.compute_root();
113 h.compute_root();
114 out.compute_root();
115
116 // We will replace all calls to 'f' inside both 'g' and 'h'
117 // with calls to a single clone:
118 f.clone_in({g, h}).compute_root();
119
120 // The equivalent loop nests are:
121 // for y:
122 // for x:
123 // f(x, y) = x + y
124 // for y:
125 // for x:
126 // f_clone(x, y) = x + y
127 // for y:
128 // for x:
129 // g(x, y) = 2 * f_clone(x, y)
130 // for y:
131 // for x:
132 // h(x, y) = f_clone(x, y) + 10
133 // for y:
134 // for x:
135 // out(x, y) = f(x, y) + g(x, y) + h(x, y)
136
137 out.realize({5, 5});
138 }
139
140 {
141 // One use case of Func::clone_in() is when two consumers of a producer
142 // consume regions of the producer that are very disjoint. Consider
143 // the following case for example:
144 Func f("f"), g("g"), h("h");
145 f(x) = x;
146 g(x) = 2 * f(0);
147 h(x) = f(99) + 10;
148
149 // Let's schedule 'f' to be computed at root.
150 f.compute_root();
151 // Since both 'g' and 'h' consume 'f', the region required of 'f'
152 // in the x-dimension is [0, 99]. The equivalent loop nests are:
153 // for x = 0 to 99
154 // f(x) = x
155 // for x:
156 // g(x) = 2 * f(0)
157 // for x:
158 // h(x) = f(99) + 10
159
160 // If 'f' is very expensive to compute, we might be better off with
161 // having distinct copies of 'f' for each consumer, 'g' and 'h', to
162 // avoid unnecessary computations. To create separate copies of 'f'
163 // for each consumer, we can do the following:
164 f.clone_in(g).compute_root();
165
166 // The equivalent loop nests are:
167 // f(0) = x
168 // f_clone(99) = x
169 // for x:
170 // g(x) = 2 * f_clone(0)
171 // for x:
172 // h(x) = f(99) + 10
173 }
174
175 printf("Success!\n");
176
177 return 0;
178}
