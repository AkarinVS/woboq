1// Halide tutorial lesson 4: Debugging with tracing, print, and print_when
2
3// This lesson demonstrates how to follow what Halide is doing at runtime.
4
5// On linux, you can compile and run it like so:
6// g++ lesson_04*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_04 -std=c++11
7// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_04
8
9// On os x:
10// g++ lesson_04*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_04 -std=c++11
11// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_04
12
13// If you have the entire Halide source tree, you can also build it by
14// running:
15// make tutorial_lesson_04_debugging_2
16// in a shell with the current directory at the top of the halide
17// source tree.
18
19#include "Halide.h"
20#include <stdio.h>
21using namespace Halide;
22
23int main(int argc, char **argv) {
24
25 Var x("x"), y("y");
26
27 // Printing out the value of Funcs as they are computed.
28 {
29 // We'll define our gradient function as before.
30 Func gradient("gradient");
31 gradient(x, y) = x + y;
32
33 // And tell Halide that we'd like to be notified of all
34 // evaluations.
35 gradient.trace_stores();
36
37 // Realize the function over an 8x8 region.
38 printf("Evaluating gradient\n");
39 Buffer<int> output = gradient.realize({8, 8});
40
41 // This will print out all the times gradient(x, y) gets
42 // evaluated.
43
44 // Now that we can snoop on what Halide is doing, let's try our
45 // first scheduling primitive. We'll make a new version of
46 // gradient that processes each scanline in parallel.
47 Func parallel_gradient("parallel_gradient");
48 parallel_gradient(x, y) = x + y;
49
50 // We'll also trace this function.
51 parallel_gradient.trace_stores();
52
53 // Things are the same so far. We've defined the algorithm, but
54 // haven't said anything about how to schedule it. In general,
55 // exploring different scheduling decisions doesn't change the code
56 // that describes the algorithm.
57
58 // Now we tell Halide to use a parallel for loop over the y
59 // coordinate. On Linux we run this using a thread pool and a task
60 // queue. On OS X we call into grand central dispatch, which does
61 // the same thing for us.
62 parallel_gradient.parallel(y);
63
64 // This time the printfs should come out of order, because each
65 // scanline is potentially being processed in a different
66 // thread. The number of threads should adapt to your system, but
67 // on linux you can control it manually using the environment
68 // variable HL_NUM_THREADS.
69 printf("\nEvaluating parallel_gradient\n");
70 parallel_gradient.realize({8, 8});
71 }
72
73 // Printing individual Exprs.
74 {
75 // trace_stores() can only print the value of a
76 // Func. Sometimes you want to inspect the value of
77 // sub-expressions rather than the entire Func. The built-in
78 // function 'print' can be wrapped around any Expr to print
79 // the value of that Expr every time it is evaluated.
80
81 // For example, say we have some Func that is the sum of two terms:
82 Func f;
83 f(x, y) = sin(x) + cos(y);
84
85 // If we want to inspect just one of the terms, we can wrap
86 // 'print' around it like so:
87 Func g;
88 g(x, y) = sin(x) + print(cos(y));
89
90 printf("\nEvaluating sin(x) + cos(y), and just printing cos(y)\n");
91 g.realize({4, 4});
92 }
93
94 // Printing additional context.
95 {
96 // print can take multiple arguments. It prints all of them
97 // and evaluates to the first one. The arguments can be Exprs
98 // or constant strings. This can be used to print additional
99 // context alongside the value:
100 Func f;
101 f(x, y) = sin(x) + print(cos(y), "<- this is cos(", y, ") when x =", x);
102
103 printf("\nEvaluating sin(x) + cos(y), and printing cos(y) with more context\n");
104 f.realize({4, 4});
105
106 // It can be useful to split expressions like the one above
107 // across multiple lines to make it easier to turn on and off
108 // printing certain values while debugging.
109 Expr e = cos(y);
110 // Uncomment the following line to print the value of cos(y)
111 // e = print(e, "<- this is cos(", y, ") when x =", x);
112 Func g;
113 g(x, y) = sin(x) + e;
114 g.realize({4, 4});
115 }
116
117 // Conditional printing
118 {
119 // Both print and trace_stores can produce a lot of output. If
120 // you're looking for a rare event, or just want to see what
121 // happens at a single pixel, this amount of output can be
122 // difficult to dig through. Instead, the function print_when
123 // can be used to conditionally print an Expr. The first
124 // argument to print_when is a boolean Expr. If the Expr
125 // evaluates to true, it returns the second argument and
126 // prints all of the arguments. If the Expr evaluates to false
127 // it just returns the second argument and does not print.
128
129 Func f;
130 Expr e = cos(y);
131 e = print_when(x == 37 && y == 42, e, "<- this is cos(y) at x, y == (37, 42)");
132 f(x, y) = sin(x) + e;
133 printf("\nEvaluating sin(x) + cos(y), and printing cos(y) at a single pixel\n");
134 f.realize({640, 480});
135
136 // print_when can also be used to check for values you're not expecting:
137 Func g;
138 e = cos(y);
139 e = print_when(e < 0, e, "cos(y) < 0 at y ==", y);
140 g(x, y) = sin(x) + e;
141 printf("\nEvaluating sin(x) + cos(y), and printing whenever cos(y) < 0\n");
142 g.realize({4, 4});
143 }
144
145 // Printing expressions at compile-time.
146 {
147 // The code above builds up a Halide Expr across several lines
148 // of code. If you're programmatically constructing a complex
149 // expression, and you want to check the Expr you've created
150 // is what you think it is, you can also print out the
151 // expression itself using C++ streams:
152 Var fizz("fizz"), buzz("buzz");
153 Expr e = 1;
154 for (int i = 2; i < 100; i++) {
155 if (i % 3 == 0 && i % 5 == 0) {
156 e += fizz * buzz;
157 } else if (i % 3 == 0) {
158 e += fizz;
159 } else if (i % 5 == 0) {
160 e += buzz;
161 } else {
162 e += i;
163 }
164 }
165 std::cout << "Printing a complex Expr: " << e << "\n";
166 }
167
168 printf("Success!\n");
169 return 0;
170}
