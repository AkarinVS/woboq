1// Halide tutorial lesson 19: Wrapper Funcs
2
3// This lesson demonstrates how to use Func::in and ImageParam::in to
4// schedule a Func differently in different places, and to stage loads
5// from a Func or an ImageParam.
6
7// On linux, you can compile and run it like so:
8// g++ lesson_19*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_19 -std=c++11
9// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_19
10
11// On os x:
12// g++ lesson_19*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_19 -std=c++11
13// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_19
14
15// If you have the entire Halide source tree, you can also build it by
16// running:
17// make tutorial_lesson_19_wrapper_funcs
18// in a shell at the top of the halide source tree.
19
20// The only Halide header file you need is Halide.h. It includes all of Halide.
21#include "Halide.h"
22
23// We'll also include stdio for printf.
24#include <stdio.h>
25
26using namespace Halide;
27
28int main(int argc, char **argv) {
29 // First we'll declare some Vars to use below.
30 Var x("x"), y("y"), xo("xo"), yo("yo"), xi("xi"), yi("yi");
31
32 // This lesson will be about "wrapping" a Func or an ImageParam using the
33 // Func::in and ImageParam::in directives
34 {
35 // Consider a simple two-stage pipeline:
36 Func f("f_local"), g("g_local");
37 f(x, y) = x + y;
38 g(x, y) = 2 * f(x, y) + 3;
39
40 f.compute_root();
41
42 // This produces the following loop nests:
43 // for y:
44 // for x:
45 // f(x, y) = x + y
46 // for y:
47 // for x:
48 // g(x, y) = 2 * f(x, y) + 3
49
50 // Using Func::in, we can interpose a new Func in between f
51 // and g using the schedule alone:
52 Func f_in_g = f.in(g);
53 f_in_g.compute_root();
54
55 // Equivalently, we could also chain the schedules like so:
56 // f.in(g).compute_root();
57
58 // This produces the following three loop nests:
59 // for y:
60 // for x:
61 // f(x, y) = x + y
62 // for y:
63 // for x:
64 // f_in_g(x, y) = f(x, y)
65 // for y:
66 // for x:
67 // g(x, y) = 2 * f_in_g(x, y) + 3
68
69 g.realize({5, 5});
70
71 // See figures/lesson_19_wrapper_local.mp4 for a visualization.
72
73 // The schedule directive f.in(g) replaces all calls to 'f'
74 // inside 'g' with a wrapper Func and then returns that
75 // wrapper. Essentially, it rewrites the original pipeline
76 // above into the following:
77 {
78 Func f_in_g("f_in_g"), f("f"), g("g");
79 f(x, y) = x + y;
80 f_in_g(x, y) = f(x, y);
81 g(x, y) = 2 * f_in_g(x, y) + 3;
82
83 f.compute_root();
84 f_in_g.compute_root();
85 g.compute_root();
86 }
87
88 // In isolation, such a transformation seems pointless, but it
89 // can be used for a variety of scheduling tricks.
90 }
91
92 {
93 // In the schedule above, only the calls to 'f' made by 'g'
94 // are replaced. Other calls made to f would still call 'f'
95 // directly. If we wish to globally replace all calls to 'f'
96 // with a single wrapper, we simply say f.in().
97
98 // Consider a three stage pipeline, with two consumers of f:
99 Func f("f_global"), g("g_global"), h("h_global");
100 f(x, y) = x + y;
101 g(x, y) = 2 * f(x, y);
102 h(x, y) = 3 + g(x, y) - f(x, y);
103 f.compute_root();
104 g.compute_root();
105 h.compute_root();
106
107 // We will replace all calls to 'f' inside both 'g' and 'h'
108 // with calls to a single wrapper:
109 f.in().compute_root();
110
111 // The equivalent loop nests are:
112 // for y:
113 // for x:
114 // f(x, y) = x + y
115 // for y:
116 // for x:
117 // f_in(x, y) = f(x, y)
118 // for y:
119 // for x:
120 // g(x, y) = 2 * f_in(x, y)
121 // for y:
122 // for x:
123 // h(x, y) = 3 + g(x, y) - f_in(x, y)
124
125 h.realize({5, 5});
126
127 // See figures/lesson_19_wrapper_global.mp4 and for a
128 // visualization of what this did.
129 }
130
131 {
132 // We could also give g and h their own unique wrappers of
133 // f. This time we'll schedule them each inside the loop nests
134 // of the consumer, which is not something we could do with a
135 // single global wrapper.
136
137 Func f("f_unique"), g("g_unique"), h("h_unique");
138 f(x, y) = x + y;
139 g(x, y) = 2 * f(x, y);
140 h(x, y) = 3 + g(x, y) - f(x, y);
141
142 f.compute_root();
143 g.compute_root();
144 h.compute_root();
145
146 f.in(g).compute_at(g, y);
147 f.in(h).compute_at(h, y);
148
149 // This creates the loop nests:
150 // for y:
151 // for x:
152 // f(x, y) = x + y
153 // for y:
154 // for x:
155 // f_in_g(x, y) = f(x, y)
156 // for x:
157 // g(x, y) = 2 * f_in_g(x, y)
158 // for y:
159 // for x:
160 // f_in_h(x, y) = f(x, y)
161 // for x:
162 // h(x, y) = 3 + g(x, y) - f_in_h(x, y)
163
164 h.realize({5, 5});
165 // See figures/lesson_19_wrapper_unique.mp4 for a visualization.
166 }
167
168 {
169 // So far this may seem like a lot of pointless copying of
170 // memory. Func::in can be combined with other scheduling
171 // directives for a variety of purposes. The first we will
172 // examine is creating distinct realizations of a Func for
173 // several consumers and scheduling each differently.
174
175 // We'll start with nearly the same pipeline.
176 Func f("f_sched"), g("g_sched"), h("h_sched");
177 f(x, y) = x + y;
178 g(x, y) = 2 * f(x, y);
179 // h will use a far-away region of f
180 h(x, y) = 3 + g(x, y) - f(x + 93, y - 87);
181
182 // This time we'll inline f.
183 // f.compute_root();
184 g.compute_root();
185 h.compute_root();
186
187 f.in(g).compute_at(g, y);
188 f.in(h).compute_at(h, y);
189
190 // g and h now call f via distinct wrappers. The wrappers are
191 // scheduled, but f is not, which means that f is inlined into
192 // its two wrappers. They will each independently compute the
193 // region of f required by their consumer. If we had scheduled
194 // f compute_root, we'd be computing the bounding box of the
195 // region required by g and the region required by h, which
196 // would mostly be unused data.
197
198 // We can also schedule each of these wrappers
199 // differently. For scheduling purposes, wrappers inherit the
200 // pure vars of the Func they wrap, so we use the same x and y
201 // that we used when defining f:
202 f.in(g).vectorize(x, 4);
203 f.in(h).split(x, xo, xi, 2).reorder(xo, xi);
204
205 // Note that calling f.in(g) a second time returns the wrapper
206 // already created by the first call, it doesn't make a new one.
207
208 h.realize({8, 8});
209 // See figures/lesson_19_wrapper_vary_schedule.mp4 for a
210 // visualization.
211
212 // Note that because f is inlined into its two wrappers, it is
213 // the wrappers that do the work of computing f, rather than
214 // just loading from an existing computed realization.
215 }
216
217 {
218 // Func::in is useful to stage loads from a Func via some
219 // smaller intermediate buffer, perhaps on the stack or in
220 // shared GPU memory.
221
222 // Consider a pipeline that transposes some compute_root'd Func:
223
224 Func f("f_transpose"), g("g_transpose");
225 f(x, y) = sin(((x + y) * sqrt(y)) / 10);
226 f.compute_root();
227
228 g(x, y) = f(y, x);
229
230 // The execution strategy we want is to load an 4x4 tile of f
231 // into registers, transpose it in-register, and then write it
232 // out as an 4x4 tile of g. We will use Func::in to express this:
233
234 Func f_tile = f.in(g);
235
236 // We now have a three stage pipeline:
237 // f -> f_tile -> g
238
239 // f_tile will load vectors of f, and store them transposed
240 // into registers. g will then write this data back to main
241 // memory.
242 g.tile(x, y, xo, yo, xi, yi, 4, 4)
243 .vectorize(xi)
244 .unroll(yi);
245
246 // We will compute f_transpose at tiles of g, and use
247 // Func::reorder_storage to state that f_transpose should be
248 // stored column-major, so that the loads to it done by g can
249 // be dense vector loads.
250 f_tile.compute_at(g, xo)
251 .reorder_storage(y, x)
252 .vectorize(x)
253 .unroll(y);
254
255 // We take care to make sure f_transpose is only ever accessed
256 // at constant indicies. The full unrolling/vectorization of
257 // all loops that exist inside its compute_at level has this
258 // effect. Allocations that are only ever accessed at constant
259 // indices can be promoted into registers.
260
261 g.realize({16, 16});
262 // See figures/lesson_19_transpose.mp4 for a visualization
263 }
264
265 {
266 // ImageParam::in behaves the same way as Func::in, and you
267 // can use it to stage loads in similar ways. Instead of
268 // transposing again, we'll use ImageParam::in to stage tiles
269 // of an input image into GPU shared memory, effectively using
270 // shared/local memory as an explicitly-managed cache.
271
272 ImageParam img(Int(32), 2);
273
274 // We will compute a small blur of the input.
275 Func blur("blur");
276 blur(x, y) = (img(x - 1, y - 1) + img(x, y - 1) + img(x + 1, y - 1) +
277 img(x - 1, y) + img(x, y) + img(x + 1, y) +
278 img(x - 1, y + 1) + img(x, y + 1) + img(x + 1, y + 1));
279
280 blur.compute_root().gpu_tile(x, y, xo, yo, xi, yi, 8, 8);
281
282 // The wrapper Func created by ImageParam::in has pure vars
283 // named _0, _1, etc. Schedule it per tile of "blur", and map
284 // _0 and _1 to gpu threads.
285 img.in(blur).compute_at(blur, xo).gpu_threads(_0, _1);
286
287 // Without Func::in, computing an 8x8 tile of blur would do
288 // 8*8*9 loads to global memory. With Func::in, the wrapper
289 // does 10*10 loads to global memory up front, and then blur
290 // does 8*8*9 loads to shared/local memory.
291
292 // Select an appropriate GPU API, as we did in lesson 12
293 Target target = get_host_target();
294 if (target.os == Target::OSX) {
295 target.set_feature(Target::Metal);
296 } else {
297 target.set_feature(Target::OpenCL);
298 }
299
300 // This check isn't strictly necessary, but it allows a more graceful
301 // exit if running on a system that doesn't have the expected drivers
302 // and/or hardware present.
303 if (!host_supports_target_device(target)) {
304 printf("Requested GPU is not supported; skipping this test. (Do you have the proper hardware and/or driver installed?)\n");
305 return 0;
306 }
307
308 // Create an interesting input image to use.
309 Buffer<int> input(258, 258);
310 input.set_min(-1, -1);
311 for (int y = input.top(); y <= input.bottom(); y++) {
312 for (int x = input.left(); x <= input.right(); x++) {
313 input(x, y) = x * 17 + y % 4;
314 }
315 }
316
317 img.set(input);
318 blur.compile_jit(target);
319 Buffer<int> out = blur.realize({256, 256});
320
321 // Check the output is what we expected
322 for (int y = out.top(); y <= out.bottom(); y++) {
323 for (int x = out.left(); x <= out.right(); x++) {
324 int val = out(x, y);
325 int expected = (input(x - 1, y - 1) + input(x, y - 1) + input(x + 1, y - 1) +
326 input(x - 1, y) + input(x, y) + input(x + 1, y) +
327 input(x - 1, y + 1) + input(x, y + 1) + input(x + 1, y + 1));
328 if (val != expected) {
329 printf("out(%d, %d) = %d instead of %d\n",
330 x, y, val, expected);
331 return -1;
332 }
333 }
334 }
335 }
336
337 {
338 // Func::in can also be used to group multiple stages of a
339 // Func into the same loop nest. Consider the following
340 // pipeline, which computes a value per pixel, then sweeps
341 // from left to right and back across each scanline.
342 Func f("f_group"), g("g_group"), h("h_group");
343
344 // Initialize f
345 f(x, y) = sin(x - y);
346 RDom r(1, 7);
347
348 // Sweep from left to right
349 f(r, y) = (f(r, y) + f(r - 1, y)) / 2;
350
351 // Sweep from right to left
352 f(7 - r, y) = (f(7 - r, y) + f(8 - r, y)) / 2;
353
354 // Then we do something with a complicated access pattern: A
355 // 45 degree rotation with wrap-around
356 g(x, y) = f((x + y) % 8, (x - y) % 8);
357
358 // f should be scheduled compute_root, because its consumer
359 // accesses it in a complicated way. But that means all stages
360 // of f are computed in separate loop nests:
361
362 // for y:
363 // for x:
364 // f(x, y) = sin(x - y)
365 // for y:
366 // for r:
367 // f(r, y) = (f(r, y) + f(r - 1, y)) / 2
368 // for y:
369 // for r:
370 // f(7 - r, y) = (f(7 - r, y) + f(8 - r, y)) / 2
371 // for y:
372 // for x:
373 // g(x, y) = f((x + y) % 8, (x - y) % 8);
374
375 // We can get better locality if we schedule the work done by
376 // f to share a common loop over y. We can do this by
377 // computing f at scanlines of a wrapper like so:
378
379 f.in(g).compute_root();
380 f.compute_at(f.in(g), y);
381
382 // f has the default schedule for a Func with update stages,
383 // which is to be computed at the innermost loop of its
384 // consumer, which is now the wrapper f.in(g). This therefore
385 // generates the following loop nest, which has better
386 // locality:
387
388 // for y:
389 // for x:
390 // f(x, y) = sin(x - y)
391 // for r:
392 // f(r, y) = (f(r, y) + f(r - 1, y)) / 2
393 // for r:
394 // f(7 - r, y) = (f(7 - r, y) + f(8 - r, y)) / 2
395 // for x:
396 // f_in_g(x, y) = f(x, y)
397 // for y:
398 // for x:
399 // g(x, y) = f_in_g((x + y) % 8, (x - y) % 8);
400
401 // We'll additionally vectorize the initialization of, and
402 // then transfer of pixel values from f into its wrapper:
403 f.vectorize(x, 4);
404 f.in(g).vectorize(x, 4);
405
406 g.realize({8, 8});
407 // See figures/lesson_19_group_updates.mp4 for a visualization.
408 }
409
410 printf("Success!\n");
411
412 return 0;
