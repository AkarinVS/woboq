34int main(int argc, char **argv) {
35 // Declare some Vars to use below.
36 Var x("x"), y("y");
37
38 // Load a grayscale image to use as an input.
39 Buffer<uint8_t> input = load_image("images/gray.png");

41 // You can define a Func in multiple passes. Let's see a toy
42 // example first.
43 {
44 // The first definition must be one like we have seen already
45 // - a mapping from Vars to an Expr:
46 Func f;
47 f(x, y) = x + y;
48 // We call this first definition the "pure" definition.
49
50 // But the later definitions can include computed expressions on
51 // both sides. The simplest example is modifying a single point:
52 f(3, 7) = 42;
53
54 // We call these extra definitions "update" definitions, or
55 // "reduction" definitions. A reduction definition is an
56 // update definition that recursively refers back to the
57 // function's current value at the same site:
58 f(x, y) = f(x, y) + 17;
59
60 // If we confine our update to a single row, we can
61 // recursively refer to values in the same column:
62 f(x, 3) = f(x, 0) * f(x, 10);
63
64 // Similarly, if we confine our update to a single column, we
65 // can recursively refer to other values in the same row.
66 f(0, y) = f(0, y) / f(3, y);
67
68 // The general rule is: Each Var used in an update definition
69 // must appear unadorned in the same position as in the pure
70 // definition in all references to the function on the left-
71 // and right-hand sides. So the following definitions are
72 // legal updates:
73 f(x, 17) = x + 8;
74 f(0, y) = y * 8;
75 f(x, x + 1) = x + 8;
76 f(y / 2, y) = f(0, y) * 17;
77
78 // But these ones would cause an error:
79
80 // f(x, 0) = f(x + 1, 0);
81 // First argument to f on the right-hand-side must be 'x', not 'x + 1'.
82
83 // f(y, y + 1) = y + 8;
84 // Second argument to f on the left-hand-side must be 'y', not 'y + 1'.
85
86 // f(y, x) = y - x;
87 // Arguments to f on the left-hand-side are in the wrong places.
88
89 // f(3, 4) = x + y;
90 // Free variables appear on the right-hand-side but not the left-hand-side.
91
92 // We'll realize this one just to make sure it compiles. The
93 // second-to-last definition forces us to realize over a
94 // domain that is taller than it is wide.
95 f.realize({100, 101});
96
97 // For each realization of f, each step runs in its entirety
98 // before the next one begins. Let's trace the loads and
99 // stores for a simpler example:
100 Func g("g");
101 g(x, y) = x + y; // Pure definition
102 g(2, 1) = 42; // First update definition
103 g(x, 0) = g(x, 1); // Second update definition
104
105 g.trace_loads();
106 g.trace_stores();
107
108 g.realize({4, 4});
109
110 // See figures/lesson_09_update.gif for a visualization.
111
112 // Reading the log, we see that each pass is applied in
113 // turn. The equivalent C is:
114 int result[4][4];
115 // Pure definition
116 for (int y = 0; y < 4; y++) {
117 for (int x = 0; x < 4; x++) {
118 result[y][x] = x + y;
119 }
120 }
121 // First update definition
122 result[1][2] = 42;
123 // Second update definition
124 for (int x = 0; x < 4; x++) {
125 result[0][x] = result[1][x];
126 }
127 }
128
129 // Putting update passes inside loops.
130 {
131 // Starting with this pure definition:
132 Func f;
133 f(x, y) = (x + y) / 100.0f;
134
135 // Say we want an update that squares the first fifty rows. We
136 // could do this by adding 50 update definitions:
137
138 // f(x, 0) = f(x, 0) * f(x, 0);
139 // f(x, 1) = f(x, 1) * f(x, 1);
140 // f(x, 2) = f(x, 2) * f(x, 2);
141 // ...
142 // f(x, 49) = f(x, 49) * f(x, 49);
143
144 // Or equivalently using a compile-time loop in our C++:
145 // for (int i = 0; i < 50; i++) {
146 // f(x, i) = f(x, i) * f(x, i);
147 // }
148
149 // But it's more manageable and more flexible to put the loop
150 // in the generated code. We do this by defining a "reduction
151 // domain" and using it inside an update definition:
152 RDom r(0, 50);
153 f(x, r) = f(x, r) * f(x, r);
154 Buffer<float> halide_result = f.realize({100, 100});
155
156 // See figures/lesson_09_update_rdom.mp4 for a visualization.
157
158 // The equivalent C is:
159 float c_result[100][100];
160 for (int y = 0; y < 100; y++) {
161 for (int x = 0; x < 100; x++) {
162 c_result[y][x] = (x + y) / 100.0f;
163 }
164 }
165 for (int x = 0; x < 100; x++) {
166 for (int r = 0; r < 50; r++) {
167 // The loop over the reduction domain occurs inside of
168 // the loop over any pure variables used in the update
169 // step:
170 c_result[r][x] = c_result[r][x] * c_result[r][x];
171 }
172 }
173
174 // Check the results match:
175 for (int y = 0; y < 100; y++) {
176 for (int x = 0; x < 100; x++) {
177 if (fabs(halide_result(x, y) - c_result[y][x]) > 0.01f) {
178 printf("halide_result(%d, %d) = %f instead of %f\n",
179 x, y, halide_result(x, y), c_result[y][x]);
180 return -1;
181 }
182 }
183 }
184 }
185
186 // Now we'll examine a real-world use for an update definition:
187 // computing a histogram.
188 {
189
190 // Some operations on images can't be cleanly expressed as a pure
191 // function from the output coordinates to the value stored
192 // there. The classic example is computing a histogram. The
193 // natural way to do it is to iterate over the input image,
194 // updating histogram buckets. Here's how you do that in Halide:
195 Func histogram("histogram");
196
197 // Histogram buckets start as zero.
198 histogram(x) = 0;
199
200 // Define a multi-dimensional reduction domain over the input image:
201 RDom r(0, input.width(), 0, input.height());
202
203 // For every point in the reduction domain, increment the
204 // histogram bucket corresponding to the intensity of the
205 // input image at that point.
206 histogram(input(r.x, r.y)) += 1;
207
208 Buffer<int> halide_result = histogram.realize({256});
209
210 // The equivalent C is:
211 int c_result[256];
212 for (int x = 0; x < 256; x++) {
213 c_result[x] = 0;
214 }
215 for (int r_y = 0; r_y < input.height(); r_y++) {
216 for (int r_x = 0; r_x < input.width(); r_x++) {
217 c_result[input(r_x, r_y)] += 1;
218 }
219 }
220
221 // Check the answers agree:
222 for (int x = 0; x < 256; x++) {
223 if (c_result[x] != halide_result(x)) {
224 printf("halide_result(%d) = %d instead of %d\n",
225 x, halide_result(x), c_result[x]);
226 return -1;
227 }
228 }
229 }

231 // Scheduling update steps
232 {
233 // The pure variables in an update step and can be
234 // parallelized, vectorized, split, etc as usual.
235
236 // Vectorizing, splitting, or parallelize the variables that
237 // are part of the reduction domain is trickier. We'll cover
238 // that in a later lesson.
239
240 // Consider the definition:
241 Func f;
242 f(x, y) = x * y;
243 // Set row zero to each row 8
244 f(x, 0) = f(x, 8);
245 // Set column zero equal to column 8 plus 2
246 f(0, y) = f(8, y) + 2;
247
248 // The pure variables in each stage can be scheduled
249 // independently. To control the pure definition, we schedule
250 // as we have done in the past. The following code vectorizes
251 // and parallelizes the pure definition only.
252 f.vectorize(x, 4).parallel(y);
253
254 // We use Func::update(int) to get a handle to an update step
255 // for the purposes of scheduling. The following line
256 // vectorizes the first update step across x. We can't do
257 // anything with y for this update step, because it doesn't
258 // use y.
259 f.update(0).vectorize(x, 4);
260
261 // Now we parallelize the second update step in chunks of size
262 // 4.
263 Var yo, yi;
264 f.update(1).split(y, yo, yi, 4).parallel(yo);
265
266 Buffer<int> halide_result = f.realize({16, 16});
267
268 // See figures/lesson_09_update_schedule.mp4 for a visualization.
269
270 // Here's the equivalent (serial) C:
271 int c_result[16][16];
272
273 // Pure step. Vectorized in x and parallelized in y.
274 for (int y = 0; y < 16; y++) { // Should be a parallel for loop
275 for (int x_vec = 0; x_vec < 4; x_vec++) {
276 int x[] = {x_vec * 4, x_vec * 4 + 1, x_vec * 4 + 2, x_vec * 4 + 3};
277 c_result[y][x[0]] = x[0] * y;
278 c_result[y][x[1]] = x[1] * y;
279 c_result[y][x[2]] = x[2] * y;
280 c_result[y][x[3]] = x[3] * y;
281 }
282 }
283
284 // First update. Vectorized in x.
285 for (int x_vec = 0; x_vec < 4; x_vec++) {
286 int x[] = {x_vec * 4, x_vec * 4 + 1, x_vec * 4 + 2, x_vec * 4 + 3};
287 c_result[0][x[0]] = c_result[8][x[0]];
288 c_result[0][x[1]] = c_result[8][x[1]];
289 c_result[0][x[2]] = c_result[8][x[2]];
290 c_result[0][x[3]] = c_result[8][x[3]];
291 }
292
293 // Second update. Parallelized in chunks of size 4 in y.
294 for (int yo = 0; yo < 4; yo++) { // Should be a parallel for loop
295 for (int yi = 0; yi < 4; yi++) {
296 int y = yo * 4 + yi;
297 c_result[y][0] = c_result[y][8] + 2;
298 }
299 }
300
301 // Check the C and Halide results match:
302 for (int y = 0; y < 16; y++) {
303 for (int x = 0; x < 16; x++) {
304 if (halide_result(x, y) != c_result[y][x]) {
305 printf("halide_result(%d, %d) = %d instead of %d\n",
306 x, y, halide_result(x, y), c_result[y][x]);
307 return -1;
308 }
309 }
310 }
311 }

313 // That covers how to schedule the variables within a Func that
314 // uses update steps, but what about producer-consumer
315 // relationships that involve compute_at and store_at? Let's
316 // examine a reduction as a producer, in a producer-consumer pair.
317 {
318 // Because an update does multiple passes over a stored array,
319 // it's not meaningful to inline them. So the default schedule
320 // for them does the closest thing possible. It computes them
321 // in the innermost loop of their consumer. Consider this
322 // trivial example:
323 Func producer, consumer;
324 producer(x) = x * 2;
325 producer(x) += 10;
326 consumer(x) = 2 * producer(x);
327 Buffer<int> halide_result = consumer.realize({10});
328
329 // See figures/lesson_09_inline_reduction.gif for a visualization.
330
331 // The equivalent C is:
332 int c_result[10];
333 for (int x = 0; x < 10; x++) {
334 int producer_storage[1];
335 // Pure step for producer
336 producer_storage[0] = x * 2;
337 // Update step for producer
338 producer_storage[0] = producer_storage[0] + 10;
339 // Pure step for consumer
340 c_result[x] = 2 * producer_storage[0];
341 }
342
343 // Check the results match
344 for (int x = 0; x < 10; x++) {
345 if (halide_result(x) != c_result[x]) {
346 printf("halide_result(%d) = %d instead of %d\n",
347 x, halide_result(x), c_result[x]);
348 return -1;
349 }
350 }
351
352 // For all other compute_at/store_at options, the reduction
353 // gets placed where you would expect, somewhere in the loop
354 // nest of the consumer.
355 }
356
357 // Now let's consider a reduction as a consumer in a
358 // producer-consumer pair. This is a little more involved.
359 // Case 1: The consumer references the producer in the pure step only.
360 {
361 Func producer, consumer;
362
363 // The producer is pure.
364 producer(x) = x * 17;
365 consumer(x) = 2 * producer(x);
366 consumer(x) += 50;
367
368 // The valid schedules for the producer in this case are
369 // the default schedule - inlined, and also:
370 //
371 // 1) producer.compute_at(x), which places the computation of
372 // the producer inside the loop over x in the pure step of the
373 // consumer.
374 //
375 // 2) producer.compute_root(), which computes all of the
376 // producer ahead of time.
377 //
378 // 3) producer.store_root().compute_at(x), which allocates
379 // space for the consumer outside the loop over x, but fills
380 // it in as needed inside the loop.
381 //
382 // Let's use option 1.
383
384 producer.compute_at(consumer, x);
385
386 Buffer<int> halide_result = consumer.realize({10});
387
388 // See figures/lesson_09_compute_at_pure.gif for a visualization.
389
390 // The equivalent C is:
391 int c_result[10];
392 // Pure step for the consumer
393 for (int x = 0; x < 10; x++) {
394 // Pure step for producer
395 int producer_storage[1];
396 producer_storage[0] = x * 17;
397 c_result[x] = 2 * producer_storage[0];
398 }
399 // Update step for the consumer
400 for (int x = 0; x < 10; x++) {
401 c_result[x] += 50;
402 }
403
404 // All of the pure step is evaluated before any of the
405 // update step, so there are two separate loops over x.
406
407 // Check the results match
408 for (int x = 0; x < 10; x++) {
409 if (halide_result(x) != c_result[x]) {
410 printf("halide_result(%d) = %d instead of %d\n",
411 x, halide_result(x), c_result[x]);
412 return -1;
413 }
414 }
415 }
416
417 {
418 // Case 2: The consumer references the producer in the update step only
419 Func producer, consumer;
420 producer(x) = x * 17;
421 consumer(x) = 100 - x * 10;
422 consumer(x) += producer(x);
423
424 // Again we compute the producer per x coordinate of the
425 // consumer. This places producer code inside the update
426 // step of the consumer, because that's the only step that
427 // uses the producer.
428 producer.compute_at(consumer, x);
429
430 // Note however, that we didn't say:
431 //
432 // producer.compute_at(consumer.update(0), x).
433 //
434 // Scheduling is done with respect to Vars of a Func, and
435 // the Vars of a Func are shared across the pure and
436 // update steps.
437
438 Buffer<int> halide_result = consumer.realize({10});
439
440 // See figures/lesson_09_compute_at_update.gif for a visualization.
441
442 // The equivalent C is:
443 int c_result[10];
444 // Pure step for the consumer
445 for (int x = 0; x < 10; x++) {
446 c_result[x] = 100 - x * 10;
447 }
448 // Update step for the consumer
449 for (int x = 0; x < 10; x++) {
450 // Pure step for producer
451 int producer_storage[1];
452 producer_storage[0] = x * 17;
453 c_result[x] += producer_storage[0];
454 }
455
456 // Check the results match
457 for (int x = 0; x < 10; x++) {
458 if (halide_result(x) != c_result[x]) {
459 printf("halide_result(%d) = %d instead of %d\n",
460 x, halide_result(x), c_result[x]);
461 return -1;
462 }
463 }
464 }
465
466 {
467 // Case 3: The consumer references the producer in
468 // multiple steps that share common variables
469 Func producer, consumer;
470 producer(x) = x * 17;
471 consumer(x) = 170 - producer(x);
472 consumer(x) += producer(x) / 2;
473
474 // Again we compute the producer per x coordinate of the
475 // consumer. This places producer code inside both the
476 // pure and the update step of the consumer. So there end
477 // up being two separate realizations of the producer, and
478 // redundant work occurs.
479 producer.compute_at(consumer, x);
480
481 Buffer<int> halide_result = consumer.realize({10});
482
483 // See figures/lesson_09_compute_at_pure_and_update.gif for a visualization.
484
485 // The equivalent C is:
486 int c_result[10];
487 // Pure step for the consumer
488 for (int x = 0; x < 10; x++) {
489 // Pure step for producer
490 int producer_storage[1];
491 producer_storage[0] = x * 17;
492 c_result[x] = 170 - producer_storage[0];
493 }
494 // Update step for the consumer
495 for (int x = 0; x < 10; x++) {
496 // Another copy of the pure step for producer
497 int producer_storage[1];
498 producer_storage[0] = x * 17;
499 c_result[x] += producer_storage[0] / 2;
500 }
501
502 // Check the results match
503 for (int x = 0; x < 10; x++) {
504 if (halide_result(x) != c_result[x]) {
505 printf("halide_result(%d) = %d instead of %d\n",
506 x, halide_result(x), c_result[x]);
507 return -1;
508 }
509 }
510 }
511
512 {
513 // Case 4: The consumer references the producer in
514 // multiple steps that do not share common variables
515 Func producer, consumer;
516 producer(x, y) = (x * y) / 10 + 8;
517 consumer(x, y) = x + y;
518 consumer(x, 0) += producer(x, x);
519 consumer(0, y) += producer(y, 9 - y);
520
521 // In this case neither producer.compute_at(consumer, x)
522 // nor producer.compute_at(consumer, y) will work, because
523 // either one fails to cover one of the uses of the
524 // producer. So we'd have to inline producer, or use
525 // producer.compute_root().
526
527 // Let's say we really really want producer to be
528 // compute_at the inner loops of both consumer update
529 // steps. Halide doesn't allow multiple different
530 // schedules for a single Func, but we can work around it
531 // by making two wrappers around producer, and scheduling
532 // those instead:
533
534 // Attempt 2:
535 Func producer_1, producer_2, consumer_2;
536 producer_1(x, y) = producer(x, y);
537 producer_2(x, y) = producer(x, y);
538
539 consumer_2(x, y) = x + y;
540 consumer_2(x, 0) += producer_1(x, x);
541 consumer_2(0, y) += producer_2(y, 9 - y);
542
543 // The wrapper functions give us two separate handles on
544 // the producer, so we can schedule them differently.
545 producer_1.compute_at(consumer_2, x);
546 producer_2.compute_at(consumer_2, y);
547
548 Buffer<int> halide_result = consumer_2.realize({10, 10});
549
550 // See figures/lesson_09_compute_at_multiple_updates.mp4 for a visualization.
551
552 // The equivalent C is:
553 int c_result[10][10];
554 // Pure step for the consumer
555 for (int y = 0; y < 10; y++) {
556 for (int x = 0; x < 10; x++) {
557 c_result[y][x] = x + y;
558 }
559 }
560 // First update step for consumer
561 for (int x = 0; x < 10; x++) {
562 int producer_1_storage[1];
563 producer_1_storage[0] = (x * x) / 10 + 8;
564 c_result[0][x] += producer_1_storage[0];
565 }
566 // Second update step for consumer
567 for (int y = 0; y < 10; y++) {
568 int producer_2_storage[1];
569 producer_2_storage[0] = (y * (9 - y)) / 10 + 8;
570 c_result[y][0] += producer_2_storage[0];
571 }
572
573 // Check the results match
574 for (int y = 0; y < 10; y++) {
575 for (int x = 0; x < 10; x++) {
576 if (halide_result(x, y) != c_result[y][x]) {
577 printf("halide_result(%d, %d) = %d instead of %d\n",
578 x, y, halide_result(x, y), c_result[y][x]);
579 return -1;
580 }
581 }
582 }
583 }
584
585 {
586 // Case 5: Scheduling a producer under a reduction domain
587 // variable of the consumer.
588
589 // We are not just restricted to scheduling producers at
590 // the loops over the pure variables of the consumer. If a
591 // producer is only used within a loop over a reduction
592 // domain (RDom) variable, we can also schedule the
593 // producer there.
594
595 Func producer, consumer;
596
597 RDom r(0, 5);
598 producer(x) = x % 8;
599 consumer(x) = x + 10;
600 consumer(x) += r + producer(x + r);
601
602 producer.compute_at(consumer, r);
603
604 Buffer<int> halide_result = consumer.realize({10});
605
606 // See figures/lesson_09_compute_at_rvar.gif for a visualization.
607
608 // The equivalent C is:
609 int c_result[10];
610 // Pure step for the consumer.
611 for (int x = 0; x < 10; x++) {
612 c_result[x] = x + 10;
613 }
614 // Update step for the consumer.
615 for (int x = 0; x < 10; x++) {
616 // The loop over the reduction domain is always the inner loop.
617 for (int r = 0; r < 5; r++) {
618 // We've schedule the storage and computation of
619 // the producer here. We just need a single value.
620 int producer_storage[1];
621 // Pure step of the producer.
622 producer_storage[0] = (x + r) % 8;
623
624 // Now use it in the update step of the consumer.
625 c_result[x] += r + producer_storage[0];
626 }
627 }
628
629 // Check the results match
630 for (int x = 0; x < 10; x++) {
631 if (halide_result(x) != c_result[x]) {
632 printf("halide_result(%d) = %d instead of %d\n",
633 x, halide_result(x), c_result[x]);
634 return -1;
635 }
636 }

639 // A real-world example of a reduction inside a producer-consumer chain.
640 {
641 // The default schedule for a reduction is a good one for
642 // convolution-like operations. For example, the following
643 // computes a 5x5 box-blur of our grayscale test image with a
644 // clamp-to-edge boundary condition:
645
646 // First add the boundary condition.
647 Func clamped = BoundaryConditions::repeat_edge(input);
648
649 // Define a 5x5 box that starts at (-2, -2)
650 RDom r(-2, 5, -2, 5);
651
652 // Compute the 5x5 sum around each pixel.
653 Func local_sum;
654 local_sum(x, y) = 0; // Compute the sum as a 32-bit integer
655 local_sum(x, y) += clamped(x + r.x, y + r.y);
656
657 // Divide the sum by 25 to make it an average
658 Func blurry;
659 blurry(x, y) = cast<uint8_t>(local_sum(x, y) / 25);
660
661 Buffer<uint8_t> halide_result = blurry.realize({input.width(), input.height()});
662
663 // The default schedule will inline 'clamped' into the update
664 // step of 'local_sum', because clamped only has a pure
665 // definition, and so its default schedule is fully-inlined.
666 // We will then compute local_sum per x coordinate of blurry,
667 // because the default schedule for reductions is
668 // compute-innermost. Here's the equivalent C:
669
670 Buffer<uint8_t> c_result(input.width(), input.height());
671 for (int y = 0; y < input.height(); y++) {
672 for (int x = 0; x < input.width(); x++) {
673 int local_sum[1];
674 // Pure step of local_sum
675 local_sum[0] = 0;
676 // Update step of local_sum
677 for (int r_y = -2; r_y <= 2; r_y++) {
678 for (int r_x = -2; r_x <= 2; r_x++) {
679 // The clamping has been inlined into the update step.
680 int clamped_x = std::min(std::max(x + r_x, 0), input.width() - 1);
681 int clamped_y = std::min(std::max(y + r_y, 0), input.height() - 1);
682 local_sum[0] += input(clamped_x, clamped_y);
683 }
684 }
685 // Pure step of blurry
686 c_result(x, y) = (uint8_t)(local_sum[0] / 25);
687 }
688 }
689
690 // Check the results match
691 for (int y = 0; y < input.height(); y++) {
692 for (int x = 0; x < input.width(); x++) {
693 if (halide_result(x, y) != c_result(x, y)) {
694 printf("halide_result(%d, %d) = %d instead of %d\n",
695 x, y, halide_result(x, y), c_result(x, y));
696 return -1;
697 }
698 }
699 }
700 }
701
702 // Reduction helpers.
703 {
704 // There are several reduction helper functions provided in
705 // Halide.h, which compute small reductions and schedule them
706 // innermost into their consumer. The most useful one is
707 // "sum".
708 Func f1;
709 RDom r(0, 100);
710 f1(x) = sum(r + x) * 7;
711
712 // Sum creates a small anonymous Func to do the reduction. It's equivalent to:
713 Func f2;
714 Func anon;
715 anon(x) = 0;
716 anon(x) += r + x;
717 f2(x) = anon(x) * 7;
718
719 // So even though f1 references a reduction domain, it is a
720 // pure function. The reduction domain has been swallowed to
721 // define the inner anonymous reduction.
722
723 Buffer<int> halide_result_1 = f1.realize({10});
724 Buffer<int> halide_result_2 = f2.realize({10});
725
726 // The equivalent C is:
727 int c_result[10];
728 for (int x = 0; x < 10; x++) {
729 int anon[1];
730 anon[0] = 0;
731 for (int r = 0; r < 100; r++) {
732 anon[0] += r + x;
733 }
734 c_result[x] = anon[0] * 7;
735 }
736
737 // Check they all match.
738 for (int x = 0; x < 10; x++) {
739 if (halide_result_1(x) != c_result[x]) {
740 printf("halide_result_1(%d) = %d instead of %d\n",
741 x, halide_result_1(x), c_result[x]);
742 return -1;
743 }
744 if (halide_result_2(x) != c_result[x]) {
745 printf("halide_result_2(%d) = %d instead of %d\n",
746 x, halide_result_2(x), c_result[x]);
747 return -1;
748 }
749 }
750 }
751
752 // A complex example that uses reduction helpers.
753 {
754 // Other reduction helpers include "product", "minimum",
755 // "maximum", "argmin", and "argmax". Using argmin and argmax
756 // requires understanding tuples, which come in a later
757 // lesson. Let's use minimum and maximum to compute the local
758 // spread of our grayscale image.
759
760 // First, add a boundary condition to the input.
761 Func clamped;
762 Expr x_clamped = clamp(x, 0, input.width() - 1);
763 Expr y_clamped = clamp(y, 0, input.height() - 1);
764 clamped(x, y) = input(x_clamped, y_clamped);
765
766 RDom box(-2, 5, -2, 5);
767 // Compute the local maximum minus the local minimum:
768 Func spread;
769 spread(x, y) = (maximum(clamped(x + box.x, y + box.y)) -
770 minimum(clamped(x + box.x, y + box.y)));
771
772 // Compute the result in strips of 32 scanlines
773 Var yo, yi;
774 spread.split(y, yo, yi, 32).parallel(yo);
775
776 // Vectorize across x within the strips. This implicitly
777 // vectorizes stuff that is computed within the loop over x in
778 // spread, which includes our minimum and maximum helpers, so
779 // they get vectorized too.
780 spread.vectorize(x, 16);
781
782 // We'll apply the boundary condition by padding each scanline
783 // as we need it in a circular buffer (see lesson 08).
784 clamped.store_at(spread, yo).compute_at(spread, yi);
785
786 Buffer<uint8_t> halide_result = spread.realize({input.width(), input.height()});
787
788// The C equivalent is almost too horrible to contemplate (and
789// took me a long time to debug). This time I want to time
790// both the Halide version and the C version, so I'll use sse
791// intrinsics for the vectorization, and openmp to do the
792// parallel for loop (you'll need to compile with -fopenmp or
793// similar to get correct timing).
794#ifdef __SSE2__
795 // Don't include the time required to allocate the output buffer.
796 Buffer<uint8_t> c_result(input.width(), input.height());
797
798#ifdef _OPENMP
799 double t1 = current_time();
800#endif
801
802 // Run this one hundred times so we can average the timing results.
803 for (int iters = 0; iters < 100; iters++) {
804
805#pragma omp parallel for
806 for (int yo = 0; yo < (input.height() + 31) / 32; yo++) {
807 int y_base = std::min(yo * 32, input.height() - 32);
808
809 // Compute clamped in a circular buffer of size 8
810 // (smallest power of two greater than 5). Each thread
811 // needs its own allocation, so it must occur here.
812
813 int clamped_width = input.width() + 4;
814 uint8_t *clamped_storage = (uint8_t *)malloc(clamped_width * 8);
815
816 for (int yi = 0; yi < 32; yi++) {
817 int y = y_base + yi;
818
819 uint8_t *output_row = &c_result(0, y);
820
821 // Compute clamped for this scanline, skipping rows
822 // already computed within this slice.
823 int min_y_clamped = (yi == 0) ? (y - 2) : (y + 2);
824 int max_y_clamped = (y + 2);
825 for (int cy = min_y_clamped; cy <= max_y_clamped; cy++) {
826 // Figure out which row of the circular buffer
827 // we're filling in using bitmasking:
828 uint8_t *clamped_row =
829 clamped_storage + (cy & 7) * clamped_width;
830
831 // Figure out which row of the input we're reading
832 // from by clamping the y coordinate:
833 int clamped_y = std::min(std::max(cy, 0), input.height() - 1);
834 uint8_t *input_row = &input(0, clamped_y);
835
836 // Fill it in with the padding.
837 for (int x = -2; x < input.width() + 2; x++) {
838 int clamped_x = std::min(std::max(x, 0), input.width() - 1);
839 *clamped_row++ = input_row[clamped_x];
840 }
841 }
842
843 // Now iterate over vectors of x for the pure step of the output.
844 for (int x_vec = 0; x_vec < (input.width() + 15) / 16; x_vec++) {
845 int x_base = std::min(x_vec * 16, input.width() - 16);
846
847 // Allocate storage for the minimum and maximum
848 // helpers. One vector is enough.
849 __m128i minimum_storage, maximum_storage;
850
851 // The pure step for the maximum is a vector of zeros
852 maximum_storage = _mm_setzero_si128();
853
854 // The update step for maximum
855 for (int max_y = y - 2; max_y <= y + 2; max_y++) {
856 uint8_t *clamped_row =
857 clamped_storage + (max_y & 7) * clamped_width;
858 for (int max_x = x_base - 2; max_x <= x_base + 2; max_x++) {
859 __m128i v = _mm_loadu_si128(
860 (__m128i const *)(clamped_row + max_x + 2));
861 maximum_storage = _mm_max_epu8(maximum_storage, v);
862 }
863 }
864
865 // The pure step for the minimum is a vector of
866 // ones. Create it by comparing something to
867 // itself.
868 minimum_storage = _mm_cmpeq_epi32(_mm_setzero_si128(),
869 _mm_setzero_si128());
870
871 // The update step for minimum.
872 for (int min_y = y - 2; min_y <= y + 2; min_y++) {
873 uint8_t *clamped_row =
874 clamped_storage + (min_y & 7) * clamped_width;
875 for (int min_x = x_base - 2; min_x <= x_base + 2; min_x++) {
876 __m128i v = _mm_loadu_si128(
877 (__m128i const *)(clamped_row + min_x + 2));
878 minimum_storage = _mm_min_epu8(minimum_storage, v);
879 }
880 }
881
882 // Now compute the spread.
883 __m128i spread = _mm_sub_epi8(maximum_storage, minimum_storage);
884
885 // Store it.
886 _mm_storeu_si128((__m128i *)(output_row + x_base), spread);
887 }
888 }
889
890 free(clamped_storage);
891 }
892 }
893
894// Skip the timing comparison if we don't have openmp
895// enabled. Otherwise it's unfair to C.
896#ifdef _OPENMP
897 double t2 = current_time();
898
899 // Now run the Halide version again without the
900 // jit-compilation overhead. Also run it one hundred times.
901 for (int iters = 0; iters < 100; iters++) {
902 spread.realize(halide_result);
903 }
904
905 double t3 = current_time();
906
907 // Report the timings. On my machine they both take about 3ms
908 // for the 4-megapixel input (fast!), which makes sense,
909 // because they're using the same vectorization and
910 // parallelization strategy. However I find the Halide easier
911 // to read, write, debug, modify, and port.
912 printf("Halide spread took %f ms. C equivalent took %f ms\n",
913 (t3 - t2) / 100, (t2 - t1) / 100);
914
915#endif // _OPENMP
916
917 // Check the results match:
918 for (int y = 0; y < input.height(); y++) {
919 for (int x = 0; x < input.width(); x++) {
920 if (halide_result(x, y) != c_result(x, y)) {
921 printf("halide_result(%d, %d) = %d instead of %d\n",
922 x, y, halide_result(x, y), c_result(x, y));
923 return -1;
924 }
925 }
926 }
927
928#endif // __SSE2__
929 }
930
931 printf("Success!\n");
932 return 0;
