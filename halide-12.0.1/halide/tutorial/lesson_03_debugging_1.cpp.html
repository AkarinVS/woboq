1// Halide tutorial lesson 3: Inspecting the generated code
2
3// This lesson demonstrates how to inspect what the Halide compiler is producing.
4
5// On linux, you can compile and run it like so:
6// g++ lesson_03*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_03 -std=c++11
7// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_03
8
9// On os x:
10// g++ lesson_03*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_03 -std=c++11
11// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_03
12
13// If you have the entire Halide source tree, you can also build it by
14// running:
15// make tutorial_lesson_03_debugging_1
16// in a shell with the current directory at the top of the halide
17// source tree.
18
19#include "Halide.h"
20#include <stdio.h>
21
22// This time we'll just import the entire Halide namespace
23using namespace Halide;
24
25int main(int argc, char **argv) {
26
27 // We'll start by defining the simple single-stage imaging
28 // pipeline from lesson 1.
29
30 // This lesson will be about debugging, but unfortunately in C++,
31 // objects don't know their own names, which makes it hard for us
32 // to understand the generated code. To get around this, you can
33 // pass a string to the Func and Var constructors to give them a
34 // name for debugging purposes.
35 Func gradient("gradient");
36 Var x("x"), y("y");
37 gradient(x, y) = x + y;
38
39 // Realize the function to produce an output image. We'll keep it
40 // very small for this lesson.
41 Buffer<int> output = gradient.realize({8, 8});
42
43 // That line compiled and ran the pipeline. Try running this
44 // lesson with the environment variable HL_DEBUG_CODEGEN set to
45 // 1. It will print out the various stages of compilation, and a
46 // pseudocode representation of the final pipeline.
47
48 // If you set HL_DEBUG_CODEGEN to a higher number, you can see
49 // more and more details of how Halide compiles your pipeline.
50 // Setting HL_DEBUG_CODEGEN=2 shows the Halide code at each stage
51 // of compilation, and also the llvm bitcode we generate at the
52 // end.
53
54 // Halide will also output an HTML version of this output, which
55 // supports syntax highlighting and code-folding, so it can be
56 // nicer to read for large pipelines. Open gradient.html with your
57 // browser after running this tutorial.
58 gradient.compile_to_lowered_stmt("gradient.html", {}, HTML);
59
60 // You can usually figure out what code Halide is generating using
61 // this pseudocode. In the next lesson we'll see how to snoop on
62 // Halide at runtime.
63
64 printf("Success!\n");
65 return 0;
66}
