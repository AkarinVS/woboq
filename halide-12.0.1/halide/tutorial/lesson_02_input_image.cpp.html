1// Halide tutorial lesson 2: Processing images
2
3// This lesson demonstrates how to pass in input images and manipulate
4// them.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_02*.cpp -g -I <path/to/Halide.h> -I <path/to/tools/halide_image_io.h> -L <path/to/libHalide.so> -lHalide `libpng-config --cflags --ldflags` -ljpeg -lpthread -ldl -o lesson_02 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_02
9
10// On os x:
11// g++ lesson_02*.cpp -g -I <path/to/Halide.h> -I <path/to/tools/halide_image_io.h> -L <path/to/libHalide.so> -lHalide `libpng-config --cflags --ldflags` -ljpeg -o lesson_02 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_02
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_02_input_image
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20// The only Halide header file you need is Halide.h. It includes all of Halide.
21#include "Halide.h"
22
23// Include some support code for loading pngs.
24#include "halide_image_io.h"
25using namespace Halide::Tools;
26
27int main(int argc, char **argv) {
28
29 // This program defines a single-stage imaging pipeline that
30 // brightens an image.
31
32 // First we'll load the input image we wish to brighten.
33 Halide::Buffer<uint8_t> input = load_image("images/rgb.png");
34
35 // See figures/lesson_02_input.jpg for a smaller version.
36
37 // Next we define our Func object that represents our one pipeline
38 // stage.
39 Halide::Func brighter;
40
41 // Our Func will have three arguments, representing the position
42 // in the image and the color channel. Halide treats color
43 // channels as an extra dimension of the image.
44 Halide::Var x, y, c;
45
46 // Normally we'd probably write the whole function definition on
47 // one line. Here we'll break it apart so we can explain what
48 // we're doing at every step.
49
50 // For each pixel of the input image.
51 Halide::Expr value = input(x, y, c);
52
53 // Cast it to a floating point value.
54 value = Halide::cast<float>(value);
55
56 // Multiply it by 1.5 to brighten it. Halide represents real
57 // numbers as floats, not doubles, so we stick an 'f' on the end
58 // of our constant.
59 value = value * 1.5f;
60
61 // Clamp it to be less than 255, so we don't get overflow when we
62 // cast it back to an 8-bit unsigned int.
63 value = Halide::min(value, 255.0f);
64
65 // Cast it back to an 8-bit unsigned integer.
66 value = Halide::cast<uint8_t>(value);
67
68 // Define the function.
69 brighter(x, y, c) = value;
70
71 // The equivalent one-liner to all of the above is:
72 //
73 // brighter(x, y, c) = Halide::cast<uint8_t>(min(input(x, y, c) * 1.5f, 255));
74 //
75 // In the shorter version:
76 // - I skipped the cast to float, because multiplying by 1.5f does
77 // that automatically.
78 // - I also used an integer constant as the second argument in the
79 // call to min, because it gets cast to float to be compatible
80 // with the first argument.
81 // - I left the Halide:: off the call to min. It's unnecessary due
82 // to Koenig lookup.
83
84 // Remember, all we've done so far is build a representation of a
85 // Halide program in memory. We haven't actually processed any
86 // pixels yet. We haven't even compiled that Halide program yet.
87
88 // So now we'll realize the Func. The size of the output image
89 // should match the size of the input image. If we just wanted to
90 // brighten a portion of the input image we could request a
91 // smaller size. If we request a larger size Halide will throw an
92 // error at runtime telling us we're trying to read out of bounds
93 // on the input image.
94 Halide::Buffer<uint8_t> output =
95 brighter.realize({input.width(), input.height(), input.channels()});
96
97 // Save the output for inspection. It should look like a bright parrot.
98 save_image(output, "brighter.png");
99
100 // See figures/lesson_02_output.jpg for a small version of the output.
101
102 printf("Success!\n");
103 return 0;
104}
