1// Halide tutorial lesson 13: Tuples
2
3// This lesson describes how to write Funcs that evaluate to multiple
4// values.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_13*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_13 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_13
9
10// On os x:
11// g++ lesson_13*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_13 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_13
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_13_tuples
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20#include "Halide.h"
21#include <algorithm>
22#include <stdio.h>
23using namespace Halide;
24
25int main(int argc, char **argv) {
26
27 // So far Funcs (such as the one below) have evaluated to a single
28 // scalar value for each point in their domain.
29 Func single_valued;
30 Var x, y;
31 single_valued(x, y) = x + y;
32
33 // One way to write a Func that returns a collection of values is
34 // to add an additional dimension that indexes that
35 // collection. This is how we typically deal with color. For
36 // example, the Func below represents a collection of three values
37 // for every x, y coordinate indexed by c.
38 Func color_image;
39 Var c;
40 color_image(x, y, c) = select(c == 0, 245, // Red value
41 c == 1, 42, // Green value
42 132); // Blue value
43
44 // Since this pattern appears quite often, Halide provides a
45 // syntatic sugar to write the code above as the following,
46 // using the "mux" function.
47 // color_image(x, y, c) = mux(c, {245, 42, 132});
48
49 // This method is often convenient because it makes it easy to
50 // operate on this Func in a way that treats each item in the
51 // collection equally:
52 Func brighter;
53 brighter(x, y, c) = color_image(x, y, c) + 10;
54
55 // However this method is also inconvenient for three reasons.
56 //
57 // 1) Funcs are defined over an infinite domain, so users of this
58 // Func can for example access color_image(x, y, -17), which is
59 // not a meaningful value and is probably indicative of a bug.
60 //
61 // 2) It requires a select, which can impact performance if not
62 // bounded and unrolled:
63 // brighter.bound(c, 0, 3).unroll(c);
64 //
65 // 3) With this method, all values in the collection must have the
66 // same type. While the above two issues are merely inconvenient,
67 // this one is a hard limitation that makes it impossible to
68 // express certain things in this way.
69
70 // It is also possible to represent a collection of values as a
71 // collection of Funcs:
72 Func func_array[3];
73 func_array[0](x, y) = x + y;
74 func_array[1](x, y) = sin(x);
75 func_array[2](x, y) = cos(y);
76
77 // This method avoids the three problems above, but introduces a
78 // new annoyance. Because these are separate Funcs, it is
79 // difficult to schedule them so that they are all computed
80 // together inside a single loop over x, y.
81
82 // A third alternative is to define a Func as evaluating to a
83 // Tuple instead of an Expr. A Tuple is a fixed-size collection of
84 // Exprs. Each Expr in a Tuple may have a different type. The
85 // following function evaluates to an integer value (x+y), and a
86 // floating point value (sin(x*y)).
87 Func multi_valued;
88 multi_valued(x, y) = Tuple(x + y, sin(x * y));
89
90 // Realizing a tuple-valued Func returns a collection of
91 // Buffers. We call this a Realization. It's equivalent to a
92 // std::vector of Buffer objects:
93 {
94 Realization r = multi_valued.realize({80, 60});
95 assert(r.size() == 2);
96 Buffer<int> im0 = r[0];
97 Buffer<float> im1 = r[1];
98 assert(im0(30, 40) == 30 + 40);
99 assert(im1(30, 40) == sinf(30 * 40));
100 }
101
102 // All Tuple elements are evaluated together over the same domain
103 // in the same loop nest, but stored in distinct allocations. The
104 // equivalent C++ code to the above is:
105 {
106 int multi_valued_0[80 * 60];
107 float multi_valued_1[80 * 60];
108 for (int y = 0; y < 80; y++) {
109 for (int x = 0; x < 60; x++) {
110 multi_valued_0[x + 60 * y] = x + y;
111 multi_valued_1[x + 60 * y] = sinf(x * y);
112 }
113 }
114 }
115
116 // When compiling ahead-of-time, a Tuple-valued Func evaluates
117 // into multiple distinct output halide_buffer_t structs. These appear in
118 // order at the end of the function signature:
119 // int multi_valued(...input buffers and params...,
120 // halide_buffer_t *output_1, halide_buffer_t *output_2);
121
122 // You can construct a Tuple by passing multiple Exprs to the
123 // Tuple constructor as we did above. Perhaps more elegantly, you
124 // can also take advantage of C++11 initializer lists and just
125 // enclose your Exprs in braces:
126 Func multi_valued_2;
127 multi_valued_2(x, y) = {x + y, sin(x * y)};
128
129 // Calls to a multi-valued Func cannot be treated as Exprs. The
130 // following is a syntax error:
131 // Func consumer;
132 // consumer(x, y) = multi_valued_2(x, y) + 10;
133
134 // Instead you must index a Tuple with square brackets to retrieve
135 // the individual Exprs:
136 Expr integer_part = multi_valued_2(x, y)[0];
137 Expr floating_part = multi_valued_2(x, y)[1];
138 Func consumer;
139 consumer(x, y) = {integer_part + 10, floating_part + 10.0f};
140
141 // Tuple reductions.
142 {
143 // Tuples are particularly useful in reductions, as they allow
144 // the reduction to maintain complex state as it walks along
145 // its domain. The simplest example is an argmax.
146
147 // First we create a Buffer to take the argmax over.
148 Func input_func;
149 input_func(x) = sin(x);
150 Buffer<float> input = input_func.realize({100});
151
152 // Then we define a 2-valued Tuple which tracks the index of
153 // the maximum value and the value itself.
154 Func arg_max;
155
156 // Pure definition.
157 arg_max() = {0, input(0)};
158
159 // Update definition.
160 RDom r(1, 99);
161 Expr old_index = arg_max()[0];
162 Expr old_max = arg_max()[1];
163 Expr new_index = select(old_max < input(r), r, old_index);
164 Expr new_max = max(input(r), old_max);
165 arg_max() = {new_index, new_max};
166
167 // The equivalent C++ is:
168 int arg_max_0 = 0;
169 float arg_max_1 = input(0);
170 for (int r = 1; r < 100; r++) {
171 int old_index = arg_max_0;
172 float old_max = arg_max_1;
173 int new_index = old_max < input(r) ? r : old_index;
174 float new_max = std::max(input(r), old_max);
175 // In a tuple update definition, all loads and computation
176 // are done before any stores, so that all Tuple elements
177 // are updated atomically with respect to recursive calls
178 // to the same Func.
179 arg_max_0 = new_index;
180 arg_max_1 = new_max;
181 }
182
183 // Let's verify that the Halide and C++ found the same maximum
184 // value and index.
185 {
186 Realization r = arg_max.realize();
187 Buffer<int> r0 = r[0];
188 Buffer<float> r1 = r[1];
189 assert(arg_max_0 == r0(0));
190 assert(arg_max_1 == r1(0));
191 }
192
193 // Halide provides argmax and argmin as built-in reductions
194 // similar to sum, product, maximum, and minimum. They return
195 // a Tuple consisting of the point in the reduction domain
196 // corresponding to that value, and the value itself. In the
197 // case of ties they return the first value found. We'll use
198 // one of these in the following section.
199 }
200
201 // Tuples for user-defined types.
202 {
203 // Tuples can also be a convenient way to represent compound
204 // objects such as complex numbers. Defining an object that
205 // can be converted to and from a Tuple is one way to extend
206 // Halide's type system with user-defined types.
207 struct Complex {
208 Expr real, imag;
209
210 // Construct from a Tuple
211 Complex(Tuple t)
212 : real(t[0]), imag(t[1]) {
213 }
214
215 // Construct from a pair of Exprs
216 Complex(Expr r, Expr i)
217 : real(r), imag(i) {
218 }
219
220 // Construct from a call to a Func by treating it as a Tuple
221 Complex(FuncRef t)
222 : Complex(Tuple(t)) {
223 }
224
225 // Convert to a Tuple
226 operator Tuple() const {
227 return {real, imag};
228 }
229
230 // Complex addition
231 Complex operator+(const Complex &other) const {
232 return {real + other.real, imag + other.imag};
233 }
234
235 // Complex multiplication
236 Complex operator*(const Complex &other) const {
237 return {real * other.real - imag * other.imag,
238 real * other.imag + imag * other.real};
239 }
240
241 // Complex magnitude, squared for efficiency
242 Expr magnitude_squared() const {
243 return real * real + imag * imag;
244 }
245
246 // Other complex operators would go here. The above are
247 // sufficient for this example.
248 };
249
250 // Let's use the Complex struct to compute a Mandelbrot set.
251 Func mandelbrot;
252
253 // The initial complex value corresponding to an x, y coordinate
254 // in our Func.
255 Complex initial(x / 15.0f - 2.5f, y / 6.0f - 2.0f);
256
257 // Pure definition.
258 Var t;
259 mandelbrot(x, y, t) = Complex(0.0f, 0.0f);
260
261 // We'll use an update definition to take 12 steps.
262 RDom r(1, 12);
263 Complex current = mandelbrot(x, y, r - 1);
264
265 // The following line uses the complex multiplication and
266 // addition we defined above.
267 mandelbrot(x, y, r) = current * current + initial;
268
269 // We'll use another tuple reduction to compute the iteration
270 // number where the value first escapes a circle of radius 4.
271 // This can be expressed as an argmin of a boolean - we want
272 // the index of the first time the given boolean expression is
273 // false (we consider false to be less than true). The argmax
274 // would return the index of the first time the expression is
275 // true.
276
277 Expr escape_condition = Complex(mandelbrot(x, y, r)).magnitude_squared() < 16.0f;
278 Tuple first_escape = argmin(escape_condition);
279
280 // We only want the index, not the value, but argmin returns
281 // both, so we'll index the argmin Tuple expression using
282 // square brackets to get the Expr representing the index.
283 Func escape;
284 escape(x, y) = first_escape[0];
285
286 // Realize the pipeline and print the result as ascii art.
287 Buffer<int> result = escape.realize({61, 25});
288 const char *code = " .:-~*={}&%#@";
289 for (int y = 0; y < result.height(); y++) {
290 for (int x = 0; x < result.width(); x++) {
291 printf("%c", code[result(x, y)]);
292 }
293 printf("\n");
294 }
295 }
296
297 printf("Success!\n");
298
299 return 0;
300}
