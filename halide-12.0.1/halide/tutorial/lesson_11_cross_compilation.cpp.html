1// Halide tutorial lesson 11: Cross-compilation
2
3// This lesson demonstrates how to use Halide as a cross-compiler to
4// generate code for any platform from any platform.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_11*.cpp -g -std=c++11 -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_11
9
10// On os x:
11// g++ lesson_11*.cpp -g -std=c++11 -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_11
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_11_cross_compilation
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20#include "Halide.h"
21#include <stdio.h>
22using namespace Halide;
23
24int main(int argc, char **argv) {
25
26 // We'll define the simple one-stage pipeline that we used in lesson 10.
27 Func brighter;
28 Var x, y;
29
30 // Declare the arguments.
31 Param<uint8_t> offset;
32 ImageParam input(type_of<uint8_t>(), 2);
33 std::vector<Argument> args(2);
34 args[0] = input;
35 args[1] = offset;
36
37 // Define the Func.
38 brighter(x, y) = input(x, y) + offset;
39
40 // Schedule it.
41 brighter.vectorize(x, 16).parallel(y);
42
43 // The following line is what we did in lesson 10. It compiles an
44 // object file suitable for the system that you're running this
45 // program on. For example, if you compile and run this file on
46 // 64-bit linux on an x86 cpu with sse4.1, then the generated code
47 // will be suitable for 64-bit linux on x86 with sse4.1.
48 brighter.compile_to_file("lesson_11_host", args, "brighter");
49
50 // We can also compile object files suitable for other cpus and
51 // operating systems. You do this with an optional third argument
52 // to compile_to_file which specifies the target to compile for.
53
54 // Let's use this to compile a 32-bit arm android version of this code:
55 Target target;
56 target.os = Target::Android; // The operating system
57 target.arch = Target::ARM; // The CPU architecture
58 target.bits = 32; // The bit-width of the architecture
59 std::vector<Target::Feature> arm_features; // A list of features to set
60 target.set_features(arm_features);
61 // We then pass the target as the last argument to compile_to_file.
62 brighter.compile_to_file("lesson_11_arm_32_android", args, "brighter", target);
63
64 // And now a Windows object file for 64-bit x86 with AVX and SSE 4.1:
65 target.os = Target::Windows;
66 target.arch = Target::X86;
67 target.bits = 64;
68 std::vector<Target::Feature> x86_features;
69 x86_features.push_back(Target::AVX);
70 x86_features.push_back(Target::SSE41);
71 target.set_features(x86_features);
72 brighter.compile_to_file("lesson_11_x86_64_windows", args, "brighter", target);
73
74 // And finally an iOS mach-o object file for one of Apple's 32-bit
75 // ARM processors - the A6. It's used in the iPhone 5. The A6 uses
76 // a slightly modified ARM architecture called ARMv7s. We specify
77 // this using the target features field. Support for Apple's
78 // 64-bit ARM processors is very new in llvm, and still somewhat
79 // flaky.
80 target.os = Target::IOS;
81 target.arch = Target::ARM;
82 target.bits = 32;
83 std::vector<Target::Feature> armv7s_features;
84 armv7s_features.push_back(Target::ARMv7s);
85 target.set_features(armv7s_features);
86 brighter.compile_to_file("lesson_11_arm_32_ios", args, "brighter", target);
87
88 // Now let's check these files are what they claim, by examining
89 // their first few bytes.
90
91 // 32-arm android object files start with the magic bytes:
92 uint8_t arm_32_android_magic[] = {0x7f, 'E', 'L', 'F', // ELF format
93 1, // 32-bit
94 1, // 2's complement little-endian
95 1}; // Current version of elf
96
97 FILE *f = fopen("lesson_11_arm_32_android.o", "rb");
98 uint8_t header[32];
99 if (!f || fread(header, 32, 1, f) != 1) {
100 printf("Object file not generated\n");
101 return -1;
102 }
103 fclose(f);
104
105 if (memcmp(header, arm_32_android_magic, sizeof(arm_32_android_magic))) {
106 printf("Unexpected header bytes in 32-bit arm object file.\n");
107 return -1;
108 }
109
110 // 64-bit windows object files start with the magic 16-bit value 0x8664
111 // (presumably referring to x86-64)
112 uint8_t win_64_magic[] = {0x64, 0x86};
113
114 f = fopen("lesson_11_x86_64_windows.obj", "rb");
115 if (!f || fread(header, 32, 1, f) != 1) {
116 printf("Object file not generated\n");
117 return -1;
118 }
119 fclose(f);
120
121 if (memcmp(header, win_64_magic, sizeof(win_64_magic))) {
122 printf("Unexpected header bytes in 64-bit windows object file.\n");
123 return -1;
124 }
125
126 // 32-bit arm iOS mach-o files start with the following magic bytes:
127 uint32_t arm_32_ios_magic[] = {0xfeedface, // Mach-o magic bytes
128 12, // CPU type is ARM
129 11, // CPU subtype is ARMv7s
130 1}; // It's a relocatable object file.
131 f = fopen("lesson_11_arm_32_ios.o", "rb");
132 if (!f || fread(header, 32, 1, f) != 1) {
133 printf("Object file not generated\n");
134 return -1;
135 }
136 fclose(f);
137
138 if (memcmp(header, arm_32_ios_magic, sizeof(arm_32_ios_magic))) {
139 printf("Unexpected header bytes in 32-bit arm ios object file.\n");
140 return -1;
141 }
142
143 // It looks like the object files we produced are plausible for
144 // those targets. We'll count that as a success for the purposes
145 // of this tutorial. For a real application you'd then need to
146 // figure out how to integrate Halide into your cross-compilation
147 // toolchain. There are several small examples of this in the
148 // Halide repository under the apps folder. See HelloAndroid and
149 // HelloiOS here:
150 // https://github.com/halide/Halide/tree/master/apps/
151 printf("Success!\n");
152 return 0;
153}
