1// Halide tutorial lesson 6: Realizing Funcs over arbitrary domains
2
3// This lesson demonstrates how to evaluate a Func over a domain that
4// does not start at (0, 0).
5
6// On linux, you can compile and run it like so:
7// g++ lesson_06*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_06 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_06
9
10// On os x:
11// g++ lesson_06*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_06 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_06
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_06_realizing_over_shifted_domains
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20#include "Halide.h"
21#include <stdio.h>
22
23using namespace Halide;
24
25int main(int argc, char **argv) {
26
27 // The last lesson was quite involved, and scheduling complex
28 // multi-stage pipelines is ahead of us. As an interlude, let's
29 // consider something easy: evaluating funcs over rectangular
30 // domains that do not start at the origin.
31
32 // We define our familiar gradient function.
33 Func gradient("gradient");
34 Var x("x"), y("y");
35 gradient(x, y) = x + y;
36
37 // And turn on tracing so we can see how it is being evaluated.
38 gradient.trace_stores();
39
40 // Previously we've realized gradient like so:
41 //
42 // gradient.realize({8, 8});
43 //
44 // This does three things internally:
45 // 1) Generates code that can evaluate gradient over an arbitrary
46 // rectangle.
47 // 2) Allocates a new 8 x 8 image.
48 // 3) Runs the generated code to evaluate gradient for all x, y
49 // from (0, 0) to (7, 7) and puts the result into the image.
50 // 4) Returns the new image as the result of the realize call.
51
52 // What if we're managing memory carefully and don't want Halide
53 // to allocate a new image for us? We can call realize another
54 // way. We can pass it an image we would like it to fill in. The
55 // following evaluates our Func into an existing image:
56 printf("Evaluating gradient from (0, 0) to (7, 7)\n");
57 Buffer<int> result(8, 8);
58 gradient.realize(result);
59
60 // Let's check it did what we expect:
61 for (int y = 0; y < 8; y++) {
62 for (int x = 0; x < 8; x++) {
63 if (result(x, y) != x + y) {
64 printf("Something went wrong!\n");
65 return -1;
66 }
67 }
68 }
69
70 // Now let's evaluate gradient over a 5 x 7 rectangle that starts
71 // somewhere else -- at position (100, 50). So x and y will run
72 // from (100, 50) to (104, 56) inclusive.
73
74 // We start by creating an image that represents that rectangle:
75 Buffer<int> shifted(5, 7); // In the constructor we tell it the size.
76 shifted.set_min(100, 50); // Then we tell it the top-left corner.
77
78 printf("Evaluating gradient from (100, 50) to (104, 56)\n");
79
80 // Note that this won't need to compile any new code, because when
81 // we realized it the first time, we generated code capable of
82 // evaluating gradient over an arbitrary rectangle.
83 gradient.realize(shifted);
84
85 // From C++, we also access the image object using coordinates
86 // that start at (100, 50).
87 for (int y = 50; y < 57; y++) {
88 for (int x = 100; x < 105; x++) {
89 if (shifted(x, y) != x + y) {
90 printf("Something went wrong!\n");
91 return -1;
92 }
93 }
94 }
95 // The image 'shifted' stores the value of our Func over a domain
96 // that starts at (100, 50), so asking for shifted(0, 0) would in
97 // fact read out-of-bounds and probably crash.
98
99 // What if we want to evaluate our Func over some region that
100 // isn't rectangular? Too bad. Halide only does rectangles :)
101
102 printf("Success!\n");
103 return 0;
104}
