22int main(int argc, char **argv) {
23 // First we'll declare some Vars to use below.
24 Var x("x"), y("y");

26 // Let's examine various scheduling options for a simple two stage
27 // pipeline. We'll start with the default schedule:
28 {
29 Func producer("producer_default"), consumer("consumer_default");
30
31 // The first stage will be some simple pointwise math similar
32 // to our familiar gradient function. The value at position x,
33 // y is the sin of product of x and y.
34 producer(x, y) = sin(x * y);
35
36 // Now we'll add a second stage which averages together multiple
37 // points in the first stage.
38 consumer(x, y) = (producer(x, y) +
39 producer(x, y + 1) +
40 producer(x + 1, y) +
41 producer(x + 1, y + 1)) / 4;
42
43 // We'll turn on tracing for both functions.
44 consumer.trace_stores();
45 producer.trace_stores();
46
47 // And evaluate it over a 4x4 box.
48 printf("\nEvaluating producer-consumer pipeline with default schedule\n");
49 consumer.realize({4, 4});
50
51 // There were no messages about computing values of the
52 // producer. This is because the default schedule fully
53 // inlines 'producer' into 'consumer'. It is as if we had
54 // written the following code instead:
55
56 // consumer(x, y) = (sin(x * y) +
57 // sin(x * (y + 1)) +
58 // sin((x + 1) * y) +
59 // sin((x + 1) * (y + 1))/4);
60
61 // All calls to 'producer' have been replaced with the body of
62 // 'producer', with the arguments substituted in for the
63 // variables.
64
65 // The equivalent C code is:
66 float result[4][4];
67 for (int y = 0; y < 4; y++) {
68 for (int x = 0; x < 4; x++) {
69 result[y][x] = (sin(x * y) +
70 sin(x * (y + 1)) +
71 sin((x + 1) * y) +
72 sin((x + 1) * (y + 1))) / 4;
73 }
74 }
75 printf("\n");
76
77 // If we look at the loop nest, the producer doesn't appear
78 // at all. It has been inlined into the consumer.
79 printf("Pseudo-code for the schedule:\n");
80 consumer.print_loop_nest();
81 printf("\n");
82 }
83
84 // Next we'll examine the next simplest option - computing all
85 // values required in the producer before computing any of the
86 // consumer. We call this schedule "root".
87 {
88 // Start with the same function definitions:
89 Func producer("producer_root"), consumer("consumer_root");
90 producer(x, y) = sin(x * y);
91 consumer(x, y) = (producer(x, y) +
92 producer(x, y + 1) +
93 producer(x + 1, y) +
94 producer(x + 1, y + 1)) / 4;
95
96 // Tell Halide to evaluate all of producer before any of consumer.
97 producer.compute_root();
98
99 // Turn on tracing.
100 consumer.trace_stores();
101 producer.trace_stores();
102
103 // Compile and run.
104 printf("\nEvaluating producer.compute_root()\n");
105 consumer.realize({4, 4});
106
107 // Reading the output we can see that:
108 // A) There were stores to producer.
109 // B) They all happened before any stores to consumer.
110
111 // See figures/lesson_08_compute_root.gif for a visualization.
112 // The producer is on the left and the consumer is on the
113 // right. Stores are marked in orange and loads are marked in
114 // blue.
115
116 // Equivalent C:
117
118 float result[4][4];
119
120 // Allocate some temporary storage for the producer.
121 float producer_storage[5][5];
122
123 // Compute the producer.
124 for (int y = 0; y < 5; y++) {
125 for (int x = 0; x < 5; x++) {
126 producer_storage[y][x] = sin(x * y);
127 }
128 }
129
130 // Compute the consumer. Skip the prints this time.
131 for (int y = 0; y < 4; y++) {
132 for (int x = 0; x < 4; x++) {
133 result[y][x] = (producer_storage[y][x] +
134 producer_storage[y + 1][x] +
135 producer_storage[y][x + 1] +
136 producer_storage[y + 1][x + 1]) / 4;
137 }
138 }

140 // Note that consumer was evaluated over a 4x4 box, so Halide
141 // automatically inferred that producer was needed over a 5x5
142 // box. This is the same 'bounds inference' logic we saw in
143 // the previous lesson, where it was used to detect and avoid
144 // out-of-bounds reads from an input image.
145
146 // If we print the loop nest, we'll see something very
147 // similar to the C above.
148 printf("Pseudo-code for the schedule:\n");
149 consumer.print_loop_nest();
150 printf("\n");
151 }

153 // Let's compare the two approaches above from a performance
154 // perspective.
155
156 // Full inlining (the default schedule):
157 // - Temporary memory allocated: 0
158 // - Loads: 0
159 // - Stores: 16
160 // - Calls to sin: 64
161
162 // producer.compute_root():
163 // - Temporary memory allocated: 25 floats
164 // - Loads: 64
165 // - Stores: 41
166 // - Calls to sin: 25
167
168 // There's a trade-off here. Full inlining used minimal temporary
169 // memory and memory bandwidth, but did a whole bunch of redundant
170 // expensive math (calling sin). It evaluated most points in
171 // 'producer' four times. The second schedule,
172 // producer.compute_root(), did the mimimum number of calls to
173 // sin, but used more temporary memory and more memory bandwidth.
174
175 // In any given situation the correct choice can be difficult to
176 // make. If you're memory-bandwidth limited, or don't have much
177 // memory (e.g. because you're running on an old cell-phone), then
178 // it can make sense to do redundant math. On the other hand, sin
179 // is expensive, so if you're compute-limited then fewer calls to
180 // sin will make your program faster. Adding vectorization or
181 // multi-core parallelism tilts the scales in favor of doing
182 // redundant work, because firing up multiple cpu cores increases
183 // the amount of math you can do per second, but doesn't increase
184 // your system memory bandwidth or capacity.
185
186 // We can make choices in between full inlining and
187 // compute_root. Next we'll alternate between computing the
188 // producer and consumer on a per-scanline basis:
189 {
190 // Start with the same function definitions:
191 Func producer("producer_y"), consumer("consumer_y");
192 producer(x, y) = sin(x * y);
193 consumer(x, y) = (producer(x, y) +
194 producer(x, y + 1) +
195 producer(x + 1, y) +
196 producer(x + 1, y + 1)) / 4;
197
198 // Tell Halide to evaluate producer as needed per y coordinate
199 // of the consumer:
200 producer.compute_at(consumer, y);
201
202 // This places the code that computes the producer just
203 // *inside* the consumer's for loop over y, as in the
204 // equivalent C below.
205
206 // Turn on tracing.
207 producer.trace_stores();
208 consumer.trace_stores();
209
210 // Compile and run.
211 printf("\nEvaluating producer.compute_at(consumer, y)\n");
212 consumer.realize({4, 4});
213
214 // See figures/lesson_08_compute_y.gif for a visualization.
215
216 // Reading the log or looking at the figure you should see
217 // that producer and consumer alternate on a per-scanline
218 // basis. Let's look at the equivalent C:
219
220 float result[4][4];
221
222 // There's an outer loop over scanlines of consumer:
223 for (int y = 0; y < 4; y++) {
224
225 // Allocate space and compute enough of the producer to
226 // satisfy this single scanline of the consumer. This
227 // means a 5x2 box of the producer.
228 float producer_storage[2][5];
229 for (int py = y; py < y + 2; py++) {
230 for (int px = 0; px < 5; px++) {
231 producer_storage[py - y][px] = sin(px * py);
232 }
233 }
234
235 // Compute a scanline of the consumer.
236 for (int x = 0; x < 4; x++) {
237 result[y][x] = (producer_storage[0][x] +
238 producer_storage[1][x] +
239 producer_storage[0][x + 1] +
240 producer_storage[1][x + 1]) / 4;
241 }
242 }
243
244 // Again, if we print the loop nest, we'll see something very
245 // similar to the C above.
246 printf("Pseudo-code for the schedule:\n");
247 consumer.print_loop_nest();
248 printf("\n");

250 // The performance characteristics of this strategy are in
251 // between inlining and compute root. We still allocate some
252 // temporary memory, but less than compute_root, and with
253 // better locality (we load from it soon after writing to it,
254 // so for larger images, values should still be in cache). We
255 // still do some redundant work, but less than full inlining:
256
257 // producer.compute_at(consumer, y):
258 // - Temporary memory allocated: 10 floats
259 // - Loads: 64
260 // - Stores: 56
261 // - Calls to sin: 40
262 }
263
264 // We could also say producer.compute_at(consumer, x), but this
265 // would be very similar to full inlining (the default
266 // schedule). Instead let's distinguish between the loop level at
267 // which we allocate storage for producer, and the loop level at
268 // which we actually compute it. This unlocks a few optimizations.
269 {
270 Func producer("producer_root_y"), consumer("consumer_root_y");
271 producer(x, y) = sin(x * y);
272 consumer(x, y) = (producer(x, y) +
273 producer(x, y + 1) +
274 producer(x + 1, y) +
275 producer(x + 1, y + 1)) / 4;
276
277 // Tell Halide to make a buffer to store all of producer at
278 // the outermost level:
279 producer.store_root();
280 // ... but compute it as needed per y coordinate of the
281 // consumer.
282 producer.compute_at(consumer, y);
283
284 producer.trace_stores();
285 consumer.trace_stores();
286
287 printf("\nEvaluating producer.store_root().compute_at(consumer, y)\n");
288 consumer.realize({4, 4});
289
290 // See figures/lesson_08_store_root_compute_y.gif for a
291 // visualization.
292
293 // Reading the log or looking at the figure you should see
294 // that producer and consumer again alternate on a
295 // per-scanline basis. It computes a 5x2 box of the producer
296 // to satisfy the first scanline of the consumer, but after
297 // that it only computes a 5x1 box of the output for each new
298 // scanline of the consumer!
299 //
300 // Halide has detected that for all scanlines except for the
301 // first, it can reuse the values already sitting in the
302 // buffer we've allocated for producer. Let's look at the
303 // equivalent C:
304
305 float result[4][4];
306
307 // producer.store_root() implies that storage goes here:
308 float producer_storage[5][5];
309
310 // There's an outer loop over scanlines of consumer:
311 for (int y = 0; y < 4; y++) {
312
313 // Compute enough of the producer to satisfy this scanline
314 // of the consumer.
315 for (int py = y; py < y + 2; py++) {
316
317 // Skip over rows of producer that we've already
318 // computed in a previous iteration.
319 if (y > 0 && py == y) continue;
320
321 for (int px = 0; px < 5; px++) {
322 producer_storage[py][px] = sin(px * py);
323 }
324 }
325
326 // Compute a scanline of the consumer.
327 for (int x = 0; x < 4; x++) {
328 result[y][x] = (producer_storage[y][x] +
329 producer_storage[y + 1][x] +
330 producer_storage[y][x + 1] +
331 producer_storage[y + 1][x + 1]) / 4;
332 }
333 }
334
335 printf("Pseudo-code for the schedule:\n");
336 consumer.print_loop_nest();
337 printf("\n");
338
339 // The performance characteristics of this strategy are pretty
340 // good! The numbers are similar to compute_root, except locality
341 // is better. We're doing the minimum number of sin calls,
342 // and we load values soon after they are stored, so we're
343 // probably making good use of the cache:
344
345 // producer.store_root().compute_at(consumer, y):
346 // - Temporary memory allocated: 10 floats
347 // - Loads: 64
348 // - Stores: 41
349 // - Calls to sin: 25
350
351 // Note that my claimed amount of memory allocated doesn't
352 // match the reference C code. Halide is performing one more
353 // optimization under the hood. It folds the storage for the
354 // producer down into a circular buffer of two
355 // scanlines. Equivalent C would actually look like this:
356
357 {
358 // Actually store 2 scanlines instead of 5
359 float producer_storage[2][5];
360 for (int y = 0; y < 4; y++) {
361 for (int py = y; py < y + 2; py++) {
362 if (y > 0 && py == y) continue;
363 for (int px = 0; px < 5; px++) {
364 // Stores to producer_storage have their y coordinate bit-masked.
365 producer_storage[py & 1][px] = sin(px * py);
366 }
367 }
368
369 // Compute a scanline of the consumer.
370 for (int x = 0; x < 4; x++) {
371 // Loads from producer_storage have their y coordinate bit-masked.
372 result[y][x] = (producer_storage[y & 1][x] +
373 producer_storage[(y + 1) & 1][x] +
374 producer_storage[y & 1][x + 1] +
375 producer_storage[(y + 1) & 1][x + 1]) / 4;
376 }
377 }
378 }
379 }
380
381 // We can do even better, by leaving the storage in the outermost
382 // loop, but moving the computation into the innermost loop:
383 {
384 Func producer("producer_root_x"), consumer("consumer_root_x");
385 producer(x, y) = sin(x * y);
386 consumer(x, y) = (producer(x, y) +
387 producer(x, y + 1) +
388 producer(x + 1, y) +
389 producer(x + 1, y + 1)) / 4;
390
391 // Store outermost, compute innermost.
392 producer.store_root().compute_at(consumer, x);
393
394 producer.trace_stores();
395 consumer.trace_stores();
396
397 printf("\nEvaluating producer.store_root().compute_at(consumer, x)\n");
398 consumer.realize({4, 4});
399
400 // See figures/lesson_08_store_root_compute_x.gif for a
401 // visualization.
402
403 // You should see that producer and consumer now alternate on
404 // a per-pixel basis. Here's the equivalent C:
405
406 float result[4][4];
407
408 // producer.store_root() implies that storage goes here, but
409 // we can fold it down into a circular buffer of two
410 // scanlines:
411 float producer_storage[2][5];
412
413 // For every pixel of the consumer:
414 for (int y = 0; y < 4; y++) {
415 for (int x = 0; x < 4; x++) {
416
417 // Compute enough of the producer to satisfy this
418 // pixel of the consumer, but skip values that we've
419 // already computed:
420 if (y == 0 && x == 0) {
421 producer_storage[y & 1][x] = sin(x * y);
422 }
423 if (y == 0) {
424 producer_storage[y & 1][x + 1] = sin((x + 1) * y);
425 }
426 if (x == 0) {
427 producer_storage[(y + 1) & 1][x] = sin(x * (y + 1));
428 }
429 producer_storage[(y + 1) & 1][x + 1] = sin((x + 1) * (y + 1));
430
431 result[y][x] = (producer_storage[y & 1][x] +
432 producer_storage[(y + 1) & 1][x] +
433 producer_storage[y & 1][x + 1] +
434 producer_storage[(y + 1) & 1][x + 1]) / 4;
435 }
436 }
437
438 printf("Pseudo-code for the schedule:\n");
439 consumer.print_loop_nest();
440 printf("\n");
441
442 // The performance characteristics of this strategy are the
443 // best so far. One of the four values of the producer we need
444 // is probably still sitting in a register, so I won't count
445 // it as a load:
446 // producer.store_root().compute_at(consumer, x):
447 // - Temporary memory allocated: 10 floats
448 // - Loads: 48
449 // - Stores: 41
450 // - Calls to sin: 25
451 }
452
453 // So what's the catch? Why not always do
454 // producer.store_root().compute_at(consumer, x) for this type of
455 // code?
456 //
457 // The answer is parallelism. In both of the previous two
458 // strategies we've assumed that values computed in previous
459 // iterations are lying around for us to reuse. This assumes that
460 // previous values of x or y happened earlier in time and have
461 // finished. This is not true if you parallelize or vectorize
462 // either loop. Darn. If you parallelize, Halide won't inject the
463 // optimizations that skip work already done if there's a parallel
464 // loop in between the store_at level and the compute_at level,
465 // and won't fold the storage down into a circular buffer either,
466 // which makes our store_root pointless.
467
468 // We're running out of options. We can make new ones by
469 // splitting. We can store_at or compute_at at the natural
470 // variables of the consumer (x and y), or we can split x or y
471 // into new inner and outer sub-variables and then schedule with
472 // respect to those. We'll use this to express fusion in tiles:
473 {
474 Func producer("producer_tile"), consumer("consumer_tile");
475 producer(x, y) = sin(x * y);
476 consumer(x, y) = (producer(x, y) +
477 producer(x, y + 1) +
478 producer(x + 1, y) +
479 producer(x + 1, y + 1)) / 4;
480
481 // We'll compute 8x8 of the consumer, in 4x4 tiles.
482 Var x_outer, y_outer, x_inner, y_inner;
483 consumer.tile(x, y, x_outer, y_outer, x_inner, y_inner, 4, 4);
484
485 // Compute the producer per tile of the consumer
486 producer.compute_at(consumer, x_outer);
487
488 // Notice that I wrote my schedule starting from the end of
489 // the pipeline (the consumer). This is because the schedule
490 // for the producer refers to x_outer, which we introduced
491 // when we tiled the consumer. You can write it in the other
492 // order, but it tends to be harder to read.
493
494 // Turn on tracing.
495 producer.trace_stores();
496 consumer.trace_stores();
497
498 printf("\nEvaluating:\n"
499 "consumer.tile(x, y, x_outer, y_outer, x_inner, y_inner, 4, 4);\n"
500 "producer.compute_at(consumer, x_outer);\n");
501 consumer.realize({8, 8});
502
503 // See figures/lesson_08_tile.gif for a visualization.
504
505 // The producer and consumer now alternate on a per-tile
506 // basis. Here's the equivalent C:
507
508 float result[8][8];
509
510 // For every tile of the consumer:
511 for (int y_outer = 0; y_outer < 2; y_outer++) {
512 for (int x_outer = 0; x_outer < 2; x_outer++) {
513 // Compute the x and y coords of the start of this tile.
514 int x_base = x_outer * 4;
515 int y_base = y_outer * 4;
516
517 // Compute enough of producer to satisfy this tile. A
518 // 4x4 tile of the consumer requires a 5x5 tile of the
519 // producer.
520 float producer_storage[5][5];
521 for (int py = y_base; py < y_base + 5; py++) {
522 for (int px = x_base; px < x_base + 5; px++) {
523 producer_storage[py - y_base][px - x_base] = sin(px * py);
524 }
525 }
526
527 // Compute this tile of the consumer
528 for (int y_inner = 0; y_inner < 4; y_inner++) {
529 for (int x_inner = 0; x_inner < 4; x_inner++) {
530 int x = x_base + x_inner;
531 int y = y_base + y_inner;
532 result[y][x] =
533 (producer_storage[y - y_base][x - x_base] +
534 producer_storage[y - y_base + 1][x - x_base] +
535 producer_storage[y - y_base][x - x_base + 1] +
536 producer_storage[y - y_base + 1][x - x_base + 1]) / 4;
537 }
538 }
539 }
540 }
541
542 printf("Pseudo-code for the schedule:\n");
543 consumer.print_loop_nest();
544 printf("\n");
545
546 // Tiling can make sense for problems like this one with
547 // stencils that reach outwards in x and y. Each tile can be
548 // computed independently in parallel, and the redundant work
549 // done by each tile isn't so bad once the tiles get large
550 // enough.
551 }
552
553 // Let's try a mixed strategy that combines what we have done with
554 // splitting, parallelizing, and vectorizing. This is one that
555 // often works well in practice for large images. If you
556 // understand this schedule, then you understand 95% of scheduling
557 // in Halide.
558 {
559 Func producer("producer_mixed"), consumer("consumer_mixed");
560 producer(x, y) = sin(x * y);
561 consumer(x, y) = (producer(x, y) +
562 producer(x, y + 1) +
563 producer(x + 1, y) +
564 producer(x + 1, y + 1)) / 4;
565
566 // Split the y coordinate of the consumer into strips of 16 scanlines:
567 Var yo, yi;
568 consumer.split(y, yo, yi, 16);
569 // Compute the strips using a thread pool and a task queue.
570 consumer.parallel(yo);
571 // Vectorize across x by a factor of four.
572 consumer.vectorize(x, 4);
573
574 // Now store the producer per-strip. This will be 17 scanlines
575 // of the producer (16+1), but hopefully it will fold down
576 // into a circular buffer of two scanlines:
577 producer.store_at(consumer, yo);
578 // Within each strip, compute the producer per scanline of the
579 // consumer, skipping work done on previous scanlines.
580 producer.compute_at(consumer, yi);
581 // Also vectorize the producer (because sin is vectorizable on x86 using SSE).
582 producer.vectorize(x, 4);
583
584 // Let's leave tracing off this time, because we're going to
585 // evaluate over a larger image.
586 // consumer.trace_stores();
587 // producer.trace_stores();
588
589 Buffer<float> halide_result = consumer.realize({160, 160});
590
591 // See figures/lesson_08_mixed.mp4 for a visualization.
592
593 // Here's the equivalent (serial) C:
594
595 float c_result[160][160];
596
597 // For every strip of 16 scanlines (this loop is parallel in
598 // the Halide version)
599 for (int yo = 0; yo < 160 / 16 + 1; yo++) {
600
601 // 16 doesn't divide 160, so push the last slice upwards
602 // to fit within [0, 159] (see lesson 05).
603 int y_base = yo * 16;
604 if (y_base > 160 - 16) y_base = 160 - 16;
605
606 // Allocate a two-scanline circular buffer for the producer
607 float producer_storage[2][161];
608
609 // For every scanline in the strip of 16:
610 for (int yi = 0; yi < 16; yi++) {
611 int y = y_base + yi;
612
613 for (int py = y; py < y + 2; py++) {
614 // Skip scanlines already computed *within this task*
615 if (yi > 0 && py == y) continue;
616
617 // Compute this scanline of the producer in 4-wide vectors
618 for (int x_vec = 0; x_vec < 160 / 4 + 1; x_vec++) {
619 int x_base = x_vec * 4;
620 // 4 doesn't divide 161, so push the last vector left
621 // (see lesson 05).
622 if (x_base > 161 - 4) x_base = 161 - 4;
623 // If you're on x86, Halide generates SSE code for this part:
624 int x[] = {x_base, x_base + 1, x_base + 2, x_base + 3};
625 float vec[4] = {sinf(x[0] * py), sinf(x[1] * py),
626 sinf(x[2] * py), sinf(x[3] * py)};
627 producer_storage[py & 1][x[0]] = vec[0];
628 producer_storage[py & 1][x[1]] = vec[1];
629 producer_storage[py & 1][x[2]] = vec[2];
630 producer_storage[py & 1][x[3]] = vec[3];
631 }
632 }
633
634 // Now compute consumer for this scanline:
635 for (int x_vec = 0; x_vec < 160 / 4; x_vec++) {
636 int x_base = x_vec * 4;
637 // Again, Halide's equivalent here uses SSE.
638 int x[] = {x_base, x_base + 1, x_base + 2, x_base + 3};
639 float vec[] = {
640 (producer_storage[y & 1][x[0]] +
641 producer_storage[(y + 1) & 1][x[0]] +
642 producer_storage[y & 1][x[0] + 1] +
643 producer_storage[(y + 1) & 1][x[0] + 1]) /
644 4,
645 (producer_storage[y & 1][x[1]] +
646 producer_storage[(y + 1) & 1][x[1]] +
647 producer_storage[y & 1][x[1] + 1] +
648 producer_storage[(y + 1) & 1][x[1] + 1]) /
649 4,
650 (producer_storage[y & 1][x[2]] +
651 producer_storage[(y + 1) & 1][x[2]] +
652 producer_storage[y & 1][x[2] + 1] +
653 producer_storage[(y + 1) & 1][x[2] + 1]) /
654 4,
655 (producer_storage[y & 1][x[3]] +
656 producer_storage[(y + 1) & 1][x[3]] +
657 producer_storage[y & 1][x[3] + 1] +
658 producer_storage[(y + 1) & 1][x[3] + 1]) /
