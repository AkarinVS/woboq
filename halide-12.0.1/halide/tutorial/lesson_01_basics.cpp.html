1// Halide tutorial lesson 1: Getting started with Funcs, Vars, and Exprs
2
3// This lesson demonstrates basic usage of Halide as a JIT compiler for imaging.
4
5// On linux, you can compile and run it like so:
6// g++ lesson_01*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_01 -std=c++11
7// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_01
8
9// On os x:
10// g++ lesson_01*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_01 -std=c++11
11// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_01
12
13// If you have the entire Halide source tree, you can also build it by
14// running:
15// make tutorial_lesson_01_basics
16// in a shell with the current directory at the top of the halide
17// source tree.
18
19// The only Halide header file you need is Halide.h. It includes all of Halide.
20#include "Halide.h"
21
22// We'll also include stdio for printf.
23#include <stdio.h>
24
25int main(int argc, char **argv) {
26
27 // This program defines a single-stage imaging pipeline that
28 // outputs a grayscale diagonal gradient.
29
30 // A 'Func' object represents a pipeline stage. It's a pure
31 // function that defines what value each pixel should have. You
32 // can think of it as a computed image.
33 Halide::Func gradient;
34
35 // Var objects are names to use as variables in the definition of
36 // a Func. They have no meaning by themselves.
37 Halide::Var x, y;
38
39 // We typically use Vars named 'x' and 'y' to correspond to the x
40 // and y axes of an image, and we write them in that order. If
41 // you're used to thinking of images as having rows and columns,
42 // then x is the column index, and y is the row index.
43
44 // Funcs are defined at any integer coordinate of its variables as
45 // an Expr in terms of those variables and other functions.
46 // Here, we'll define an Expr which has the value x + y. Vars have
47 // appropriate operator overloading so that expressions like
48 // 'x + y' become 'Expr' objects.
49 Halide::Expr e = x + y;
50
51 // Now we'll add a definition for the Func object. At pixel x, y,
52 // the image will have the value of the Expr e. On the left hand
53 // side we have the Func we're defining and some Vars. On the right
54 // hand side we have some Expr object that uses those same Vars.
55 gradient(x, y) = e;
56
57 // This is the same as writing:
58 //
59 // gradient(x, y) = x + y;
60 //
61 // which is the more common form, but we are showing the
62 // intermediate Expr here for completeness.
63
64 // That line of code defined the Func, but it didn't actually
65 // compute the output image yet. At this stage it's just Funcs,
66 // Exprs, and Vars in memory, representing the structure of our
67 // imaging pipeline. We're meta-programming. This C++ program is
68 // constructing a Halide program in memory. Actually computing
69 // pixel data comes next.
70
71 // Now we 'realize' the Func, which JIT compiles some code that
72 // implements the pipeline we've defined, and then runs it. We
73 // also need to tell Halide the domain over which to evaluate the
74 // Func, which determines the range of x and y above, and the
75 // resolution of the output image. Halide.h also provides a basic
76 // templatized image type we can use. We'll make an 800 x 600
77 // image.
78 Halide::Buffer<int32_t> output = gradient.realize({800, 600});
79
80 // Halide does type inference for you. Var objects represent
81 // 32-bit integers, so the Expr object 'x + y' also represents a
82 // 32-bit integer, and so 'gradient' defines a 32-bit image, and
83 // so we got a 32-bit signed integer image out when we call
84 // 'realize'. Halide types and type-casting rules are equivalent
85 // to C.
86
87 // Let's check everything worked, and we got the output we were
88 // expecting:
89 for (int j = 0; j < output.height(); j++) {
90 for (int i = 0; i < output.width(); i++) {
91 // We can access a pixel of an Buffer object using similar
92 // syntax to defining and using functions.
93 if (output(i, j) != i + j) {
94 printf("Something went wrong!\n"
95 "Pixel %d, %d was supposed to be %d, but instead it's %d\n",
96 i, j, i + j, output(i, j));
97 return -1;
98 }
99 }
100 }
101
102 // Everything worked! We defined a Func, then called 'realize' on
103 // it to generate and run machine code that produced an Buffer.
104 printf("Success!\n");
105
106 return 0;
107}
