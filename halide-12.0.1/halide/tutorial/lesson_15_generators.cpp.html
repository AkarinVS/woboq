20#include "Halide.h"
21#include <stdio.h>
22
23using namespace Halide;
24
25// Generators are a more structured way to do ahead-of-time
26// compilation of Halide pipelines. Instead of writing an int main()
27// with an ad-hoc command-line interface like we did in lesson 10, we
28// define a class that inherits from Halide::Generator.
29class MyFirstGenerator : public Halide::Generator<MyFirstGenerator> {
30public:
31 // We declare the Inputs to the Halide pipeline as public
32 // member variables. They'll appear in the signature of our generated
33 // function in the same order as we declare them.
34 Input<uint8_t> offset{"offset"};
35 Input<Buffer<uint8_t>> input{"input", 2};
36
37 // We also declare the Outputs as public member variables.
38 Output<Buffer<uint8_t>> brighter{"brighter", 2};
39
40 // Typically you declare your Vars at this scope as well, so that
41 // they can be used in any helper methods you add later.
42 Var x, y;
43
44 // We then define a method that constructs and return the Halide
45 // pipeline:
46 void generate() {
47 // In lesson 10, here is where we called
48 // Func::compile_to_file. In a Generator, we just need to
49 // define the Output(s) representing the output of the pipeline.
50 brighter(x, y) = input(x, y) + offset;
51
52 // Schedule it.
53 brighter.vectorize(x, 16).parallel(y);
54 }
55};
56
57// We compile this file along with tools/GenGen.cpp. That file defines
58// an "int main(...)" that provides the command-line interface to use
59// your generator class. We need to tell that code about our
60// generator. We do this like so:
61HALIDE_REGISTER_GENERATOR(MyFirstGenerator, my_first_generator)
62
63// If you like, you can put multiple Generators in the one file. This
64// could be a good idea if they share some common code. Let's define
65// another more complex generator:
66class MySecondGenerator : public Halide::Generator<MySecondGenerator> {
67public:
68 // This generator will take some compile-time parameters
69 // too. These let you compile multiple variants of a Halide
70 // pipeline. We'll define one that tells us whether or not to
71 // parallelize in our schedule:
72 GeneratorParam<bool> parallel{"parallel", /* default value */ true};
73
74 // ... and another representing a constant scale factor to use:
75 GeneratorParam<float> scale{"scale",
76 1.0f /* default value */,
77 0.0f /* minimum value */,
78 100.0f /* maximum value */};
79
80 // You can define GeneratorParams of all the basic scalar
81 // types. For numeric types you can optionally provide a minimum
82 // and maximum value, as we did for scale above.
83
84 // You can also define GeneratorParams for enums. To make this
85 // work you must provide a mapping from strings to your enum
86 // values.
87 enum class Rotation { None,
88 Clockwise,
89 CounterClockwise };
90 GeneratorParam<Rotation> rotation{"rotation",
91 /* default value */
92 Rotation::None,
93 /* map from names to values */
94 {{"none", Rotation::None},
95 {"cw", Rotation::Clockwise},
96 {"ccw", Rotation::CounterClockwise}}};
97
98 // We'll use the same Inputs as before:
99 Input<uint8_t> offset{"offset"};
100 Input<Buffer<uint8_t>> input{"input", 2};
101
102 // And a similar Output. Note that we don't specify a type for the Buffer:
103 // at compile-time, we must specify an explicit type via the "output.type"
104 // GeneratorParam (which is implicitly defined for this Output).
105 Output<Buffer<>> output{"output", 2};
106
107 // And we'll declare our Vars here as before.
108 Var x, y;
109
110 void generate() {
111 // Define the Func. We'll use the compile-time scale factor as
112 // well as the runtime offset param.
113 Func brighter;
114 brighter(x, y) = scale * (input(x, y) + offset);
115
116 // We'll possibly do some sort of rotation, depending on the
117 // enum. To get the value of a GeneratorParam, cast it to the
118 // corresponding type. This cast happens implicitly most of
119 // the time (e.g. with scale above).
120
121 Func rotated;
122 switch ((Rotation)rotation) {
123 case Rotation::None:
124 rotated(x, y) = brighter(x, y);
125 break;
126 case Rotation::Clockwise:
127 rotated(x, y) = brighter(y, 100 - x);
128 break;
129 case Rotation::CounterClockwise:
130 rotated(x, y) = brighter(100 - y, x);
131 break;
132 }
133
134 // We'll then cast to the desired output type.
135 output(x, y) = cast(output.type(), rotated(x, y));
136
137 // The structure of the pipeline depended on the generator
138 // params. So will the schedule.
139
140 // Let's start by vectorizing the output. We don't know the
141 // type though, so it's hard to pick a good factor. Generators
142 // provide a helper called "natural_vector_size" which will
143 // pick a reasonable factor for you given the type and the
144 // target you're compiling to.
145 output.vectorize(x, natural_vector_size(output.type()));
146
147 // Now we'll possibly parallelize it:
148 if (parallel) {
149 output.parallel(y);
150 }
151
152 // If there was a rotation, we'll schedule that to occur per
153 // scanline of the output and vectorize it according to its
154 // type.
155 if (rotation != Rotation::None) {
156 rotated
157 .compute_at(output, y)
158 .vectorize(x, natural_vector_size(rotated.output_types()[0]));
159 }
160 }
161};
162
163// Register our second generator:
164HALIDE_REGISTER_GENERATOR(MySecondGenerator, my_second_generator)
