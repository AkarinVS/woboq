1// Halide tutorial lesson 5: Vectorize, parallelize, unroll and tile your code
2
3// This lesson demonstrates how to manipulate the order in which you
4// evaluate pixels in a Func, including vectorization,
5// parallelization, unrolling, and tiling.
6
7// On linux, you can compile and run it like so:
8// g++ lesson_05*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_05 -std=c++11
9// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_05
10
11// On os x:
12// g++ lesson_05*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_05 -std=c++11
13// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_05
14
15// If you have the entire Halide source tree, you can also build it by
16// running:
17// make tutorial_lesson_05_scheduling_1
18// in a shell with the current directory at the top of the halide
19// source tree.
20
21#include "Halide.h"
22#include <algorithm>
23#include <stdio.h>
24using namespace Halide;
25
26int main(int argc, char **argv) {
27
28 // We're going to define and schedule our gradient function in
29 // several different ways, and see what order pixels are computed
30 // in.
31
32 Var x("x"), y("y");
33
34 // First we observe the default ordering.
35 {
36 Func gradient("gradient");
37 gradient(x, y) = x + y;
38 gradient.trace_stores();
39
40 // By default we walk along the rows and then down the
41 // columns. This means x varies quickly, and y varies
42 // slowly. x is the column and y is the row, so this is a
43 // row-major traversal.
44 printf("Evaluating gradient row-major\n");
45 Buffer<int> output = gradient.realize({4, 4});
46
47 // See figures/lesson_05_row_major.gif for a visualization of
48 // what this did.
49
50 // The equivalent C is:
51 printf("Equivalent C:\n");
52 for (int y = 0; y < 4; y++) {
53 for (int x = 0; x < 4; x++) {
54 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
55 }
56 }
57 printf("\n\n");
58
59 // Tracing is one useful way to understand what a schedule is
60 // doing. You can also ask Halide to print out pseudocode
61 // showing what loops Halide is generating:
62 printf("Pseudo-code for the schedule:\n");
63 gradient.print_loop_nest();
64 printf("\n");
65
66 // Because we're using the default ordering, it should print:
67 // compute gradient:
68 // for y:
69 // for x:
70 // gradient(...) = ...
71 }
72
73 // Reorder variables.
74 {
75 Func gradient("gradient_col_major");
76 gradient(x, y) = x + y;
77 gradient.trace_stores();
78
79 // If we reorder x and y, we can walk down the columns
80 // instead. The reorder call takes the arguments of the func,
81 // and sets a new nesting order for the for loops that are
82 // generated. The arguments are specified from the innermost
83 // loop out, so the following call puts y in the inner loop:
84 gradient.reorder(y, x);
85
86 // This means y (the row) will vary quickly, and x (the
87 // column) will vary slowly, so this is a column-major
88 // traversal.
89
90 printf("Evaluating gradient column-major\n");
91 Buffer<int> output = gradient.realize({4, 4});
92
93 // See figures/lesson_05_col_major.gif for a visualization of
94 // what this did.
95
96 printf("Equivalent C:\n");
97 for (int x = 0; x < 4; x++) {
98 for (int y = 0; y < 4; y++) {
99 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
100 }
101 }
102 printf("\n");
103
104 // If we print pseudo-code for this schedule, we'll see that
105 // the loop over y is now inside the loop over x.
106 printf("Pseudo-code for the schedule:\n");
107 gradient.print_loop_nest();
108 printf("\n");
109 }
110
111 // Split a variable into two.
112 {
113 Func gradient("gradient_split");
114 gradient(x, y) = x + y;
115 gradient.trace_stores();
116
117 // The most powerful primitive scheduling operation you can do
118 // to a var is to split it into inner and outer sub-variables:
119 Var x_outer, x_inner;
120 gradient.split(x, x_outer, x_inner, 2);
121
122 // This breaks the loop over x into two nested loops: an outer
123 // one over x_outer, and an inner one over x_inner. The last
124 // argument to split was the "split factor". The inner loop
125 // runs from zero to the split factor. The outer loop runs
126 // from zero to the extent required of x (4 in this case)
127 // divided by the split factor. Within the loops, the old
128 // variable is defined to be outer * factor + inner. If the
129 // old loop started at a value other than zero, then that is
130 // also added within the loops.
131
132 printf("Evaluating gradient with x split into x_outer and x_inner \n");
133 Buffer<int> output = gradient.realize({4, 4});
134
135 printf("Equivalent C:\n");
136 for (int y = 0; y < 4; y++) {
137 for (int x_outer = 0; x_outer < 2; x_outer++) {
138 for (int x_inner = 0; x_inner < 2; x_inner++) {
139 int x = x_outer * 2 + x_inner;
140 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
141 }
142 }
143 }
144 printf("\n");
145
146 printf("Pseudo-code for the schedule:\n");
147 gradient.print_loop_nest();
148 printf("\n");
149
150 // Note that the order of evaluation of pixels didn't actually
151 // change! Splitting by itself does nothing, but it does open
152 // up all of the scheduling possibilities that we will explore
153 // below.
154 }
155
156 // Fuse two variables into one.
157 {
158 Func gradient("gradient_fused");
159 gradient(x, y) = x + y;
160
161 // The opposite of splitting is 'fusing'. Fusing two variables
162 // merges the two loops into a single for loop over the
163 // product of the extents. Fusing is less important than
164 // splitting, but it also sees use (as we'll see later in this
165 // lesson). Like splitting, fusing by itself doesn't change
166 // the order of evaluation.
167 Var fused;
168 gradient.fuse(x, y, fused);
169
170 printf("Evaluating gradient with x and y fused\n");
171 Buffer<int> output = gradient.realize({4, 4});
172
173 printf("Equivalent C:\n");
174 for (int fused = 0; fused < 4 * 4; fused++) {
175 int y = fused / 4;
176 int x = fused % 4;
177 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
178 }
179 printf("\n");
180
181 printf("Pseudo-code for the schedule:\n");
182 gradient.print_loop_nest();
183 printf("\n");
184 }
185
186 // Evaluating in tiles.
187 {
188 Func gradient("gradient_tiled");
189 gradient(x, y) = x + y;
190 gradient.trace_stores();
191
192 // Now that we can both split and reorder, we can do tiled
193 // evaluation. Let's split both x and y by a factor of four,
194 // and then reorder the vars to express a tiled traversal.
195 //
196 // A tiled traversal splits the domain into small rectangular
197 // tiles, and outermost iterates over the tiles, and within
198 // that iterates over the points within each tile. It can be
199 // good for performance if neighboring pixels use overlapping
200 // input data, for example in a blur. We can express a tiled
201 // traversal like so:
202 Var x_outer, x_inner, y_outer, y_inner;
203 gradient.split(x, x_outer, x_inner, 4);
204 gradient.split(y, y_outer, y_inner, 4);
205 gradient.reorder(x_inner, y_inner, x_outer, y_outer);
206
207 // This pattern is common enough that there's a shorthand for it:
208 // gradient.tile(x, y, x_outer, y_outer, x_inner, y_inner, 4, 4);
209
210 printf("Evaluating gradient in 4x4 tiles\n");
211 Buffer<int> output = gradient.realize({8, 8});
212
213 // See figures/lesson_05_tiled.gif for a visualization of this
214 // schedule.
215
216 printf("Equivalent C:\n");
217 for (int y_outer = 0; y_outer < 2; y_outer++) {
218 for (int x_outer = 0; x_outer < 2; x_outer++) {
219 for (int y_inner = 0; y_inner < 4; y_inner++) {
220 for (int x_inner = 0; x_inner < 4; x_inner++) {
221 int x = x_outer * 4 + x_inner;
222 int y = y_outer * 4 + y_inner;
223 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
224 }
225 }
226 }
227 }
228 printf("\n");
229
230 printf("Pseudo-code for the schedule:\n");
231 gradient.print_loop_nest();
232 printf("\n");
233 }
234
235 // Evaluating in vectors.
236 {
237 Func gradient("gradient_in_vectors");
238 gradient(x, y) = x + y;
239 gradient.trace_stores();
240
241 // The nice thing about splitting is that it guarantees the
242 // inner variable runs from zero to the split factor. Most of
243 // the time the split-factor will be a compile-time constant,
244 // so we can replace the loop over the inner variable with a
245 // single vectorized computation. This time we'll split by a
246 // factor of four, because on X86 we can use SSE to compute in
247 // 4-wide vectors.
248 Var x_outer, x_inner;
249 gradient.split(x, x_outer, x_inner, 4);
250 gradient.vectorize(x_inner);
251
252 // Splitting and then vectorizing the inner variable is common
253 // enough that there's a short-hand for it. We could have also
254 // said:
255 //
256 // gradient.vectorize(x, 4);
257 //
258 // which is equivalent to:
259 //
260 // gradient.split(x, x, x_inner, 4);
261 // gradient.vectorize(x_inner);
262 //
263 // Note that in this case we reused the name 'x' as the new
264 // outer variable. Later scheduling calls that refer to x
265 // will refer to this new outer variable named x.
266
267 // This time we'll evaluate over an 8x4 box, so that we have
268 // more than one vector of work per scanline.
269 printf("Evaluating gradient with x_inner vectorized \n");
270 Buffer<int> output = gradient.realize({8, 4});
271
272 // See figures/lesson_05_vectors.gif for a visualization.
273
274 printf("Equivalent C:\n");
275 for (int y = 0; y < 4; y++) {
276 for (int x_outer = 0; x_outer < 2; x_outer++) {
277 // The loop over x_inner has gone away, and has been
278 // replaced by a vectorized version of the
279 // expression. On x86 processors, Halide generates SSE
280 // for all of this.
281 int x_vec[] = {x_outer * 4 + 0,
282 x_outer * 4 + 1,
283 x_outer * 4 + 2,
284 x_outer * 4 + 3};
285 int val[] = {x_vec[0] + y,
286 x_vec[1] + y,
287 x_vec[2] + y,
288 x_vec[3] + y};
289 printf("Evaluating at <%d, %d, %d, %d>, <%d, %d, %d, %d>:"
290 " <%d, %d, %d, %d>\n",
291 x_vec[0], x_vec[1], x_vec[2], x_vec[3],
292 y, y, y, y,
293 val[0], val[1], val[2], val[3]);
294 }
295 }
296 printf("\n");
297
298 printf("Pseudo-code for the schedule:\n");
299 gradient.print_loop_nest();
300 printf("\n");
301 }
302
303 // Unrolling a loop.
304 {
305 Func gradient("gradient_unroll");
306 gradient(x, y) = x + y;
307 gradient.trace_stores();
308
309 // If multiple pixels share overlapping data, it can make
310 // sense to unroll a computation so that shared values are
311 // only computed or loaded once. We do this similarly to how
312 // we expressed vectorizing. We split a dimension and then
313 // fully unroll the loop of the inner variable. Unrolling
314 // doesn't change the order in which things are evaluated.
315 Var x_outer, x_inner;
316 gradient.split(x, x_outer, x_inner, 2);
317 gradient.unroll(x_inner);
318
319 // The shorthand for this is:
320 // gradient.unroll(x, 2);
321
322 printf("Evaluating gradient unrolled by a factor of two\n");
323 Buffer<int> result = gradient.realize({4, 4});
324
325 printf("Equivalent C:\n");
326 for (int y = 0; y < 4; y++) {
327 for (int x_outer = 0; x_outer < 2; x_outer++) {
328 // Instead of a for loop over x_inner, we get two
329 // copies of the innermost statement.
330 {
331 int x_inner = 0;
332 int x = x_outer * 2 + x_inner;
333 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
334 }
335 {
336 int x_inner = 1;
337 int x = x_outer * 2 + x_inner;
338 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
339 }
340 }
341 }
342 printf("\n");
343
344 printf("Pseudo-code for the schedule:\n");
345 gradient.print_loop_nest();
346 printf("\n");
347 }
348
349 // Splitting by factors that don't divide the extent.
350 {
351 Func gradient("gradient_split_7x2");
352 gradient(x, y) = x + y;
353 gradient.trace_stores();
354
355 // Splitting guarantees that the inner loop runs from zero to
356 // the split factor, which is important for the uses we saw
357 // above. So what happens when the total extent we wish to
358 // evaluate x over isn't a multiple of the split factor? We'll
359 // split by a factor three, and we'll evaluate gradient over a
360 // 7x2 box instead of the 4x4 box we've been using.
361 Var x_outer, x_inner;
362 gradient.split(x, x_outer, x_inner, 3);
363
364 printf("Evaluating gradient over a 7x2 box with x split by three \n");
365 Buffer<int> output = gradient.realize({7, 2});
366
367 // See figures/lesson_05_split_7_by_3.gif for a visualization
368 // of what happened. Note that some points get evaluated more
369 // than once!
370
371 printf("Equivalent C:\n");
372 for (int y = 0; y < 2; y++) {
373 for (int x_outer = 0; x_outer < 3; x_outer++) { // Now runs from 0 to 2
374 for (int x_inner = 0; x_inner < 3; x_inner++) {
375 int x = x_outer * 3;
376 // Before we add x_inner, make sure we don't
377 // evaluate points outside of the 7x2 box. We'll
378 // clamp x to be at most 4 (7 minus the split
379 // factor).
380 if (x > 4) x = 4;
381 x += x_inner;
382 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
383 }
384 }
385 }
386 printf("\n");
387
388 printf("Pseudo-code for the schedule:\n");
389 gradient.print_loop_nest();
390 printf("\n");
391
392 // If you read the output, you'll see that some coordinates
393 // were evaluated more than once. That's generally OK, because
394 // pure Halide functions have no side-effects, so it's safe to
395 // evaluate the same point multiple times. If you're calling
396 // out to C functions like we are, it's your responsibility to
397 // make sure you can handle the same point being evaluated
398 // multiple times.
399
400 // The general rule is: If we require x from x_min to x_min + x_extent, and
401 // we split by a factor 'factor', then:
402 //
403 // x_outer runs from 0 to (x_extent + factor - 1)/factor
404 // x_inner runs from 0 to factor
405 // x = min(x_outer * factor, x_extent - factor) + x_inner + x_min
406 //
407 // In our example, x_min was 0, x_extent was 7, and factor was 3.
408
409 // However, if you write a Halide function with an update
410 // definition (see lesson 9), then it is not safe to evaluate
411 // the same point multiple times, so we won't apply this
412 // trick. Instead the range of values computed will be rounded
413 // up to the next multiple of the split factor.
414 }
415
416 // Fusing, tiling, and parallelizing.
417 {
418 // We saw in the previous lesson that we can parallelize
419 // across a variable. Here we combine it with fusing and
420 // tiling to express a useful pattern - processing tiles in
421 // parallel.
422
423 // This is where fusing shines. Fusing helps when you want to
424 // parallelize across multiple dimensions without introducing
425 // nested parallelism. Nested parallelism (parallel for loops
426 // within parallel for loops) is supported by Halide, but
427 // often gives poor performance compared to fusing the
428 // parallel variables into a single parallel for loop.
429
430 Func gradient("gradient_fused_tiles");
431 gradient(x, y) = x + y;
432 gradient.trace_stores();
433
434 // First we'll tile, then we'll fuse the tile indices and
435 // parallelize across the combination.
436 Var x_outer, y_outer, x_inner, y_inner, tile_index;
437 gradient.tile(x, y, x_outer, y_outer, x_inner, y_inner, 4, 4);
438 gradient.fuse(x_outer, y_outer, tile_index);
439 gradient.parallel(tile_index);
440
441 // The scheduling calls all return a reference to the Func, so
442 // you can also chain them together into a single statement to
443 // make things slightly clearer:
444 //
445 // gradient
446 // .tile(x, y, x_outer, y_outer, x_inner, y_inner, 4, 4)
447 // .fuse(x_outer, y_outer, tile_index)
448 // .parallel(tile_index);
449
450 printf("Evaluating gradient tiles in parallel\n");
451 Buffer<int> output = gradient.realize({8, 8});
452
453 // The tiles should occur in arbitrary order, but within each
454 // tile the pixels will be traversed in row-major order. See
455 // figures/lesson_05_parallel_tiles.gif for a visualization.
456
457 printf("Equivalent (serial) C:\n");
458 // This outermost loop should be a parallel for loop, but that's hard in C.
459 for (int tile_index = 0; tile_index < 4; tile_index++) {
460 int y_outer = tile_index / 2;
461 int x_outer = tile_index % 2;
462 for (int y_inner = 0; y_inner < 4; y_inner++) {
463 for (int x_inner = 0; x_inner < 4; x_inner++) {
464 int y = y_outer * 4 + y_inner;
465 int x = x_outer * 4 + x_inner;
466 printf("Evaluating at x = %d, y = %d: %d\n", x, y, x + y);
467 }
468 }
469 }
470 printf("\n");
471
472 printf("Pseudo-code for the schedule:\n");
473 gradient.print_loop_nest();
474 printf("\n");
475 }
476
477 // Putting it all together.
478 {
479 // Are you ready? We're going to use all of the features above now.
480 Func gradient_fast("gradient_fast");
481 gradient_fast(x, y) = x + y;
482
483 // We'll process 64x64 tiles in parallel.
484 Var x_outer, y_outer, x_inner, y_inner, tile_index;
485 gradient_fast
486 .tile(x, y, x_outer, y_outer, x_inner, y_inner, 64, 64)
487 .fuse(x_outer, y_outer, tile_index)
488 .parallel(tile_index);
489
490 // We'll compute two scanlines at once while we walk across
491 // each tile. We'll also vectorize in x. The easiest way to
492 // express this is to recursively tile again within each tile
493 // into 4x2 subtiles, then vectorize the subtiles across x and
494 // unroll them across y:
495 Var x_inner_outer, y_inner_outer, x_vectors, y_pairs;
496 gradient_fast
497 .tile(x_inner, y_inner, x_inner_outer, y_inner_outer, x_vectors, y_pairs, 4, 2)
498 .vectorize(x_vectors)
499 .unroll(y_pairs);
500
501 // Note that we didn't do any explicit splitting or
502 // reordering. Those are the most important primitive
503 // operations, but mostly they are buried underneath tiling,
504 // vectorizing, or unrolling calls.
505
506 // Now let's evaluate this over a range which is not a
507 // multiple of the tile size.
508
509 // If you like you can turn on tracing, but it's going to
510 // produce a lot of printfs. Instead we'll compute the answer
511 // both in C and Halide and see if the answers match.
512 Buffer<int> result = gradient_fast.realize({350, 250});
513
514 // See figures/lesson_05_fast.mp4 for a visualization.
515
516 printf("Checking Halide result against equivalent C...\n");
517 for (int tile_index = 0; tile_index < 6 * 4; tile_index++) {
518 int y_outer = tile_index / 4;
519 int x_outer = tile_index % 4;
520 for (int y_inner_outer = 0; y_inner_outer < 64 / 2; y_inner_outer++) {
521 for (int x_inner_outer = 0; x_inner_outer < 64 / 4; x_inner_outer++) {
522 // We're vectorized across x
523 int x = std::min(x_outer * 64, 350 - 64) + x_inner_outer * 4;
524 int x_vec[4] = {x + 0,
525 x + 1,
526 x + 2,
527 x + 3};
528
529 // And we unrolled across y
530 int y_base = std::min(y_outer * 64, 250 - 64) + y_inner_outer * 2;
531 {
532 // y_pairs = 0
533 int y = y_base + 0;
534 int y_vec[4] = {y, y, y, y};
535 int val[4] = {x_vec[0] + y_vec[0],
536 x_vec[1] + y_vec[1],
537 x_vec[2] + y_vec[2],
538 x_vec[3] + y_vec[3]};
539
540 // Check the result.
541 for (int i = 0; i < 4; i++) {
542 if (result(x_vec[i], y_vec[i]) != val[i]) {
543 printf("There was an error at %d %d!\n",
544 x_vec[i], y_vec[i]);
545 return -1;
546 }
547 }
548 }
549 {
550 // y_pairs = 1
551 int y = y_base + 1;
552 int y_vec[4] = {y, y, y, y};
553 int val[4] = {x_vec[0] + y_vec[0],
554 x_vec[1] + y_vec[1],
555 x_vec[2] + y_vec[2],
556 x_vec[3] + y_vec[3]};
557
558 // Check the result.
559 for (int i = 0; i < 4; i++) {
560 if (result(x_vec[i], y_vec[i]) != val[i]) {
561 printf("There was an error at %d %d!\n",
562 x_vec[i], y_vec[i]);
563 return -1;
564 }
565 }
566 }
567 }
568 }
569 }
570 printf("\n");
571
572 printf("Pseudo-code for the schedule:\n");
573 gradient_fast.print_loop_nest();
574 printf("\n");
575
576 // Note that in the Halide version, the algorithm is specified
577 // once at the top, separately from the optimizations, and there
578 // aren't that many lines of code total. Compare this to the C
579 // version. There's more code (and it isn't even parallelized or
580 // vectorized properly). More annoyingly, the statement of the
581 // algorithm (the result is x plus y) is buried in multiple places
582 // within the mess. This C code is hard to write, hard to read,
583 // hard to debug, and hard to optimize further. This is why Halide
584 // exists.
585 }
586
587 printf("Success!\n");
588 return 0;
589}
