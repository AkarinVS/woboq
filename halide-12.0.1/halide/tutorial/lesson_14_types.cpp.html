1// Halide tutorial lesson 14: The Halide type system
2
3// This lesson more precisely describes Halide's type system.
4
5// On linux, you can compile and run it like so:
6// g++ lesson_14*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_14 -std=c++11
7// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_14
8
9// On os x:
10// g++ lesson_14*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_14 -std=c++11
11// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_14

13// If you have the entire Halide source tree, you can also build it by
14// running:
15// make tutorial_lesson_14_types
16// in a shell with the current directory at the top of the halide
17// source tree.
18
19#include "Halide.h"
20#include <stdio.h>
21using namespace Halide;
22
23// This function is used to demonstrate generic code at the end of
24// this lesson.
25Expr average(Expr a, Expr b);
26
27int main(int argc, char **argv) {
28
29 // All Exprs have a scalar type, and all Funcs evaluate to one or
30 // more scalar types. The scalar types in Halide are unsigned
31 // integers of various bit widths, signed integers of the same set
32 // of bit widths, floating point numbers in single and double
33 // precision, and opaque handles (equivalent to void *). The
34 // following array contains all the legal types.
35
36 Type valid_halide_types[] = {
37 UInt(8), UInt(16), UInt(32), UInt(64),
38 Int(8), Int(16), Int(32), Int(64),
39 Float(32), Float(64), Handle()};
40
41 // Constructing and inspecting types.
42 {
43 // You can programmatically examine the properties of a Halide
44 // type. This is useful when you write a C++ function that has
45 // Expr arguments and you wish to check their types:
46 assert(UInt(8).bits() == 8);
47 assert(Int(8).is_int());
48
49 // You can also programmatically construct Types as a function of other Types.
50 Type t = UInt(8);
51 t = t.with_bits(t.bits() * 2);
52 assert(t == UInt(16));
53
54 // Or construct a Type from a C++ scalar type
55 assert(type_of<float>() == Float(32));
56
57 // The Type struct is also capable of representing vector types,
58 // but this is reserved for Halide's internal use. You should
59 // vectorize code by using Func::vectorize, not by attempting to
60 // construct vector expressions directly. You may encounter vector
61 // types if you programmatically manipulate lowered Halide code,
62 // but this is an advanced topic (see Func::add_custom_lowering_pass).
63
64 // You can query any Halide Expr for its type. An Expr
65 // representing a Var has type Int(32):
66 Var x;
67 assert(Expr(x).type() == Int(32));
68
69 // Most transcendental functions in Halide cast their inputs to a
70 // Float(32) and return a Float(32):
71 assert(sin(x).type() == Float(32));
72
73 // You can cast an Expr from one Type to another using the cast operator:
74 assert(cast(UInt(8), x).type() == UInt(8));
75
76 // This also comes in a template form that takes a C++ type.
77 assert(cast<uint8_t>(x).type() == UInt(8));
78
79 // You can also query any defined Func for the types it produces.
80 Func f1;
81 f1(x) = cast<uint8_t>(x);
82 assert(f1.output_types()[0] == UInt(8));
83
84 Func f2;
85 f2(x) = {x, sin(x)};
86 assert(f2.output_types()[0] == Int(32) &&
87 f2.output_types()[1] == Float(32));
88 }
89
90 // Type promotion rules.
91 {
92 // When you combine Exprs of different types (e.g. using '+',
93 // '*', etc), Halide uses a system of type promotion
94 // rules. These differ to C's rules. To demonstrate these
95 // we'll make some Exprs of each type.
96 Var x;
97 Expr u8 = cast<uint8_t>(x);
98 Expr u16 = cast<uint16_t>(x);
99 Expr u32 = cast<uint32_t>(x);
100 Expr u64 = cast<uint64_t>(x);
101 Expr s8 = cast<int8_t>(x);
102 Expr s16 = cast<int16_t>(x);
103 Expr s32 = cast<int32_t>(x);
104 Expr s64 = cast<int64_t>(x);
105 Expr f32 = cast<float>(x);
106 Expr f64 = cast<double>(x);
107
108 // The rules are as follows, and are applied in the order they are
109 // written below.
110
111 // 1) It is an error to cast or use arithmetic operators on Exprs of type Handle().
112
113 // 2) If the types are the same, then no type conversions occur.
114 for (Type t : valid_halide_types) {
115 // Skip the handle type.
116 if (t.is_handle()) continue;
117 Expr e = cast(t, x);
118 assert((e + e).type() == e.type());
119 }
120
121 // 3) If one type is a float but the other is not, then the
122 // non-float argument is promoted to a float (possibly causing a
123 // loss of precision for large integers).
124 assert((u8 + f32).type() == Float(32));
125 assert((f32 + s64).type() == Float(32));
126 assert((u16 + f64).type() == Float(64));
127 assert((f64 + s32).type() == Float(64));
128
129 // 4) If both types are float, then the narrower argument is
130 // promoted to the wider bit-width.
131 assert((f64 + f32).type() == Float(64));
132
133 // The rules above handle all the floating-point cases. The
134 // following three rules handle the integer cases.
135
136 // 5) If one of the arguments is an C++ int, and the other is
137 // a Halide::Expr, then the int is coerced to the type of the
138 // expression.
139 assert((u32 + 3).type() == UInt(32));
140 assert((3 + s16).type() == Int(16));
141
142 // If this rule would cause the integer to overflow, then Halide
143 // will trigger an error, e.g. uncommenting the following line
144 // will cause this program to terminate with an error.
145 // Expr bad = u8 + 257;
146
147 // 6) If both types are unsigned integers, or both types are
148 // signed integers, then the narrower argument is promoted to
149 // wider type.
150 assert((u32 + u8).type() == UInt(32));
151 assert((s16 + s64).type() == Int(64));
152
153 // 7) If one type is signed and the other is unsigned, both
154 // arguments are promoted to a signed integer with the greater of
155 // the two bit widths.
156 assert((u8 + s32).type() == Int(32));
157 assert((u32 + s8).type() == Int(32));
158
159 // Note that this may silently overflow the unsigned type in the
160 // case where the bit widths are the same.
161 assert((u32 + s32).type() == Int(32));
162
163 // When an unsigned Expr is converted to a wider signed type in
164 // this way, it is first widened to a wider unsigned type
165 // (zero-extended), and then reinterpreted as a signed
166 // integer. I.e. casting the UInt(8) value 255 to an Int(32)
167 // produces 255, not -1.
168 int32_t result32 = evaluate<int>(cast<int32_t>(cast<uint8_t>(255)));
169 assert(result32 == 255);
170
171 // When a signed type is explicitly converted to a wider unsigned
172 // type with the cast operator (the type promotion rules will
173 // never do this automatically), it is first converted to the
174 // wider signed type (sign-extended), and then reinterpreted as
175 // an unsigned integer. I.e. casting the Int(8) value -1 to a
176 // UInt(16) produces 65535, not 255.
177 uint16_t result16 = evaluate<uint16_t>(cast<uint16_t>(cast<int8_t>(-1)));
178 assert(result16 == 65535);
179 }
180
181 // The type Handle().
182 {
183 // Handle is used to represent opaque pointers. Applying
184 // type_of to any pointer type will return Handle()
185 assert(type_of<void *>() == Handle());
186 assert(type_of<const char *const **>() == Handle());
187
188 // Handles are always stored as 64-bit, regardless of the compilation
189 // target.
190 assert(Handle().bits() == 64);
191
192 // The main use of an Expr of type Handle is to pass
193 // it through Halide to other external code.
194 }
195
196 // Generic code.
197 {
198 // The main explicit use of Type in Halide is to write Halide
199 // code parameterized by a Type. In C++ you'd do this with
200 // templates. In Halide there's no need - you can inspect and
201 // modify the types dynamically at C++ runtime instead. The
202 // function defined below averages two expressions of any
203 // equal numeric type.
204 Var x;
205 assert(average(cast<float>(x), 3.0f).type() == Float(32));
206 assert(average(x, 3).type() == Int(32));
207 assert(average(cast<uint8_t>(x), cast<uint8_t>(3)).type() == UInt(8));
208 }
209
210 printf("Success!\n");
211
212 return 0;
213}
214
215Expr average(Expr a, Expr b) {
216 // Types must match.
217 assert(a.type() == b.type());
218
219 // For floating point types:
220 if (a.type().is_float()) {
221 // The '2' will be promoted to the floating point type due to
222 // rule 3 above.
223 return (a + b) / 2;
224 }
225
226 // For integer types, we must compute the intermediate value in a
227 // wider type to avoid overflow.
228 Type narrow = a.type();
229 Type wider = narrow.with_bits(narrow.bits() * 2);
230 a = cast(wider, a);
231 b = cast(wider, b);
232 return cast(narrow, (a + b) / 2);
233}
