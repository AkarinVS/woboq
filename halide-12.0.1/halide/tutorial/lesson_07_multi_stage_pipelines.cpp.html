1// Halide tutorial lesson 7: Multi-stage pipelines
2
3// On linux, you can compile and run it like so:
4// g++ lesson_07*.cpp -g -std=c++11 -I <path/to/Halide.h> -I <path/to/tools/halide_image_io.h> -L <path/to/libHalide.so> -lHalide `libpng-config --cflags --ldflags` -ljpeg -lpthread -ldl -o lesson_07
5// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_07
6
7// On os x:
8// g++ lesson_07*.cpp -g -std=c++11 -I <path/to/Halide.h> -I <path/to/tools/halide_image_io.h> -L <path/to/libHalide.so> -lHalide `libpng-config --cflags --ldflags` -ljpeg -o lesson_07
9// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_07
10
11// If you have the entire Halide source tree, you can also build it by
12// running:
13// make tutorial_lesson_07_multi_stage_pipelines
14// in a shell with the current directory at the top of the halide
15// source tree.
16
17#include "Halide.h"
18#include <stdio.h>
19
20using namespace Halide;
21
22// Support code for loading pngs.
23#include "halide_image_io.h"
24
25using namespace Halide::Tools;
26
27int main(int argc, char **argv) {
28 // First we'll declare some Vars to use below.
29 Var x("x"), y("y"), c("c");
30
31 // Now we'll express a multi-stage pipeline that blurs an image
32 // first horizontally, and then vertically.
33 {
34 // Take a color 8-bit input
35 Buffer<uint8_t> input = load_image("images/rgb.png");
36
37 // Upgrade it to 16-bit, so we can do math without it overflowing.
38 Func input_16("input_16");
39 input_16(x, y, c) = cast<uint16_t>(input(x, y, c));
40
41 // Blur it horizontally:
42 Func blur_x("blur_x");
43 blur_x(x, y, c) = (input_16(x - 1, y, c) +
44 2 * input_16(x, y, c) +
45 input_16(x + 1, y, c)) / 4;
46
47 // Blur it vertically:
48 Func blur_y("blur_y");
49 blur_y(x, y, c) = (blur_x(x, y - 1, c) +
50 2 * blur_x(x, y, c) +
51 blur_x(x, y + 1, c)) / 4;
52
53 // Convert back to 8-bit.
54 Func output("output");
55 output(x, y, c) = cast<uint8_t>(blur_y(x, y, c));
56
57 // Each Func in this pipeline calls a previous one using
58 // familiar function call syntax (we've overloaded operator()
59 // on Func objects). A Func may call any other Func that has
60 // been given a definition. This restriction prevents
61 // pipelines with loops in them. Halide pipelines are always
62 // feed-forward graphs of Funcs.
63
64 // Now let's realize it...
65
66 // Buffer<uint8_t> result = output.realize({input.width(), input.height(), 3});
67
68 // Except that the line above is not going to work. Uncomment
69 // it to see what happens.
70
71 // Realizing this pipeline over the same domain as the input
72 // image requires reading pixels out of bounds in the input,
73 // because the blur_x stage reaches outwards horizontally, and
74 // the blur_y stage reaches outwards vertically. Halide
75 // detects this by injecting a piece of code at the top of the
76 // pipeline that computes the region over which the input will
77 // be read. When it starts to run the pipeline it first runs
78 // this code, determines that the input will be read out of
79 // bounds, and refuses to continue. No actual bounds checks
80 // occur in the inner loop; that would be slow.
81 //
82 // So what do we do? There are a few options. If we realize
83 // over a domain shifted inwards by one pixel, we won't be
84 // asking the Halide routine to read out of bounds. We saw how
85 // to do this in the previous lesson:
86 Buffer<uint8_t> result(input.width() - 2, input.height() - 2, 3);
87 result.set_min(1, 1);
88 output.realize(result);
89
90 // Save the result. It should look like a slightly blurry
91 // parrot, and it should be two pixels narrower and two pixels
92 // shorter than the input image.
93 save_image(result, "blurry_parrot_1.png");
94
95 // This is usually the fastest way to deal with boundaries:
96 // don't write code that reads out of bounds :) The more
97 // general solution is our next example.
98 }
99
100 // The same pipeline, with a boundary condition on the input.
101 {
102 // Take a color 8-bit input
103 Buffer<uint8_t> input = load_image("images/rgb.png");
104
105 // This time, we'll wrap the input in a Func that prevents
106 // reading out of bounds:
107 Func clamped("clamped");
108
109 // Define an expression that clamps x to lie within the
110 // range [0, input.width()-1].
111 Expr clamped_x = clamp(x, 0, input.width() - 1);
112 // clamp(x, a, b) is equivalent to max(min(x, b), a).
113
114 // Similarly clamp y.
115 Expr clamped_y = clamp(y, 0, input.height() - 1);
116 // Load from input at the clamped coordinates. This means that
117 // no matter how we evaluated the Func 'clamped', we'll never
118 // read out of bounds on the input. This is a clamp-to-edge
119 // style boundary condition, and is the simplest boundary
120 // condition to express in Halide.
121 clamped(x, y, c) = input(clamped_x, clamped_y, c);
122
123 // Defining 'clamped' in that way can be done more concisely
124 // using a helper function from the BoundaryConditions
125 // namespace like so:
126 //
127 // clamped = BoundaryConditions::repeat_edge(input);
128 //
129 // These are important to use for other boundary conditions,
130 // because they are expressed in the way that Halide can best
131 // understand and optimize. When used correctly they are as
132 // cheap as having no boundary condition at all.
133
134 // Upgrade it to 16-bit, so we can do math without it
135 // overflowing. This time we'll refer to our new Func
136 // 'clamped', instead of referring to the input image
137 // directly.
138 Func input_16("input_16");
139 input_16(x, y, c) = cast<uint16_t>(clamped(x, y, c));
140
141 // The rest of the pipeline will be the same...
142
143 // Blur it horizontally:
144 Func blur_x("blur_x");
145 blur_x(x, y, c) = (input_16(x - 1, y, c) +
146 2 * input_16(x, y, c) +
147 input_16(x + 1, y, c)) / 4;
148
149 // Blur it vertically:
150 Func blur_y("blur_y");
151 blur_y(x, y, c) = (blur_x(x, y - 1, c) +
152 2 * blur_x(x, y, c) +
153 blur_x(x, y + 1, c)) / 4;
154
155 // Convert back to 8-bit.
156 Func output("output");
157 output(x, y, c) = cast<uint8_t>(blur_y(x, y, c));
158
159 // This time it's safe to evaluate the output over the same
160 // domain as the input, because we have a boundary condition.
161 Buffer<uint8_t> result = output.realize({input.width(), input.height(), 3});
162
163 // Save the result. It should look like a slightly blurry
164 // parrot, but this time it will be the same size as the
165 // input.
166 save_image(result, "blurry_parrot_2.png");
167 }
168
169 printf("Success!\n");
170 return 0;
171}
