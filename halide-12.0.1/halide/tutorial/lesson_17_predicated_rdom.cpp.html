1// Halide tutorial lesson 17: Reductions over non-rectangular domains
2
3// This lesson demonstrates how to define updates that iterate over
4// subsets of a reduction domain using predicates.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_17*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_17 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_17
9
10// On os x:
11// g++ lesson_17*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_17 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_17
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_17_predicated_rdom
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20#include "Halide.h"
21#include <stdio.h>
22
23using namespace Halide;
24
25int main(int argc, char **argv) {
26
27 // In lesson 9, we learned how to use RDom to define a "reduction
28 // domain" to use in a Halide update definition. The domain
29 // defined by an RDom, however, is always rectangular, and the
30 // update occurs at every point in that rectangular domain. In
31 // some cases, we might want to iterate over some non-rectangular
32 // domain, e.g. a circle. We can achieve this behavior by using
33 // the RDom::where directive.
34
35 {
36 // Starting with this pure definition:
37 Func circle("circle");
38 Var x("x"), y("y");
39 circle(x, y) = x + y;
40
41 // Say we want an update that squares the values inside a
42 // circular region centered at (3, 3) with radius of 3. To do
43 // this, we first define the minimal bounding box over the
44 // circular region using an RDom.
45 RDom r(0, 7, 0, 7);
46
47 // The bounding box does not have to be minimal. In fact, the
48 // box can be of any size, as long it covers the region we'd
49 // like to update. However, the tighter the bounding box, the
50 // tighter the generated loop bounds will be. Halide will
51 // tighten the loop bounds automatically when possible, but in
52 // general, it is better to define a minimal bounding box.
53
54 // Then, we use RDom::where to define the predicate over that
55 // bounding box, such that the update is performed only if the
56 // given predicate evaluates to true, i.e. within the circular
57 // region.
58 r.where((r.x - 3) * (r.x - 3) + (r.y - 3) * (r.y - 3) <= 10);
59
60 // After defining the predicate, we then define the update.
61 circle(r.x, r.y) *= 2;
62
63 Buffer<int> halide_result = circle.realize({7, 7});
64
65 // See figures/lesson_17_rdom_circular.mp4 for a visualization of
66 // what this did.
67
68 // The equivalent C is:
69 int c_result[7][7];
70 for (int y = 0; y < 7; y++) {
71 for (int x = 0; x < 7; x++) {
72 c_result[y][x] = x + y;
73 }
74 }
75 for (int r_y = 0; r_y < 7; r_y++) {
76 for (int r_x = 0; r_x < 7; r_x++) {
77 // Update is only performed if the predicate evaluates to true.
78 if ((r_x - 3) * (r_x - 3) + (r_y - 3) * (r_y - 3) <= 10) {
79 c_result[r_y][r_x] *= 2;
80 }
81 }
82 }
83
84 // Check the results match:
85 for (int y = 0; y < 7; y++) {
86 for (int x = 0; x < 7; x++) {
87 if (halide_result(x, y) != c_result[y][x]) {
88 printf("halide_result(%d, %d) = %d instead of %d\n",
89 x, y, halide_result(x, y), c_result[y][x]);
90 return -1;
91 }
92 }
93 }
94 }
95
96 {
97 // We can also define multiple predicates over an RDom. Let's
98 // say now we want the update to happen within some triangular
99 // region. To do this we define three predicates, where each
100 // corresponds to one side of the triangle.
101 Func triangle("triangle");
102 Var x("x"), y("y");
103 triangle(x, y) = x + y;
104 // First, let's define the minimal bounding box over the triangular
105 // region.
106 RDom r(0, 8, 0, 10);
107 // Next, let's add the three predicates to the RDom using
108 // multiple calls to RDom::where
109 r.where(r.x + r.y > 5);
110 r.where(3 * r.y - 2 * r.x < 15);
111 r.where(4 * r.x - r.y < 20);
112
113 // We can also pack the multiple predicates into one like so:
114 // r.where((r.x + r.y > 5) && (3*r.y - 2*r.x < 15) && (4*r.x - r.y < 20));
115
116 // Then define the update.
117 triangle(r.x, r.y) *= 2;
118
119 Buffer<int> halide_result = triangle.realize({10, 10});
120
121 // See figures/lesson_17_rdom_triangular.mp4 for a
122 // visualization of what this did.
123
124 // The equivalent C is:
125 int c_result[10][10];
126 for (int y = 0; y < 10; y++) {
127 for (int x = 0; x < 10; x++) {
128 c_result[y][x] = x + y;
129 }
130 }
131 for (int r_y = 0; r_y < 10; r_y++) {
132 for (int r_x = 0; r_x < 8; r_x++) {
133 // Update is only performed if the predicate evaluates to true.
134 if ((r_x + r_y > 5) && (3 * r_y - 2 * r_x < 15) && (4 * r_x - r_y < 20)) {
135 c_result[r_y][r_x] *= 2;
136 }
137 }
138 }
139
140 // Check the results match:
141 for (int y = 0; y < 10; y++) {
142 for (int x = 0; x < 10; x++) {
143 if (halide_result(x, y) != c_result[y][x]) {
144 printf("halide_result(%d, %d) = %d instead of %d\n",
145 x, y, halide_result(x, y), c_result[y][x]);
146 return -1;
147 }
148 }
149 }
150 }
151
152 {
153 // The predicate is not limited to the RDom's variables only
154 // (r.x, r.y, ...). It can also refer to free variables in
155 // the update definition, and even make calls to other Funcs,
156 // or make recursive calls to the same Func. For example:
157 Func f("f"), g("g");
158 Var x("x"), y("y");
159 f(x, y) = 2 * x + y;
160 g(x, y) = x + y;
161
162 // This RDom's predicates depend on the initial value of 'f'.
163 RDom r1(0, 5, 0, 5);
164 r1.where(f(r1.x, r1.y) >= 4);
165 r1.where(f(r1.x, r1.y) <= 7);
166 f(r1.x, r1.y) /= 10;
167
168 f.compute_root();
169
170 // While this one involves calls to another Func.
171 RDom r2(1, 3, 1, 3);
172 r2.where(f(r2.x, r2.y) < 1);
173 g(r2.x, r2.y) += 17;
174
175 Buffer<int> halide_result_g = g.realize({5, 5});
176
177 // See figures/lesson_17_rdom_calls_in_predicate.mp4 for a
178 // visualization of what this did.
179
180 // The equivalent C for 'f' is:
181 int c_result_f[5][5];
182 for (int y = 0; y < 5; y++) {
183 for (int x = 0; x < 5; x++) {
184 c_result_f[y][x] = 2 * x + y;
185 }
186 }
187 for (int r1_y = 0; r1_y < 5; r1_y++) {
188 for (int r1_x = 0; r1_x < 5; r1_x++) {
189 // Update is only performed if the predicate evaluates to true.
190 if ((c_result_f[r1_y][r1_x] >= 4) && (c_result_f[r1_y][r1_x] <= 7)) {
191 c_result_f[r1_y][r1_x] /= 10;
192 }
193 }
194 }
195
196 // And, the equivalent C for 'g' is:
197 int c_result_g[5][5];
198 for (int y = 0; y < 5; y++) {
199 for (int x = 0; x < 5; x++) {
200 c_result_g[y][x] = x + y;
201 }
202 }
203 for (int r2_y = 1; r2_y < 4; r2_y++) {
204 for (int r1_x = 1; r1_x < 4; r1_x++) {
205 // Update is only performed if the predicate evaluates to true.
206 if (c_result_f[r2_y][r1_x] < 1) {
207 c_result_g[r2_y][r1_x] += 17;
208 }
209 }
210 }
211
212 // Check the results match:
213 for (int y = 0; y < 5; y++) {
214 for (int x = 0; x < 5; x++) {
215 if (halide_result_g(x, y) != c_result_g[y][x]) {
216 printf("halide_result_g(%d, %d) = %d instead of %d\n",
217 x, y, halide_result_g(x, y), c_result_g[y][x]);
218 return -1;
219 }
220 }
221 }
222 }
223
224 printf("Success!\n");
225
226 return 0;
227}
