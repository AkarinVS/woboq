267/** A parallel task to be passed to halide_do_parallel_tasks. This
268 * task may recursively call halide_do_parallel_tasks, and there may
269 * be complex dependencies between seemingly unrelated tasks expressed
270 * using semaphores. If you are using a custom task system, care must
271 * be taken to avoid potential deadlock. This can be done by carefully
272 * respecting the static metadata at the end of the task struct.*/
273struct halide_parallel_task_t {
274 // The function to call. It takes a user context, a min and
275 // extent, a closure, and a task system pass through argument.
276 halide_loop_task_t fn;
277
278 // The closure to pass it
279 uint8_t *closure;
280
281 // The name of the function to be called. For debugging purposes only.
282 const char *name;
283
284 // An array of semaphores that must be acquired before the
285 // function is called. Must be reacquired for every call made.
286 struct halide_semaphore_acquire_t *semaphores;
287 int num_semaphores;
288
289 // The entire range the function should be called over. This range
290 // may be sliced up and the function called multiple times.
291 int min, extent;
292
293 // A parallel task provides several pieces of metadata to prevent
294 // unbounded resource usage or deadlock.
295
296 // The first is the minimum number of execution contexts (call
297 // stacks or threads) necessary for the function to run to
298 // completion. This may be greater than one when there is nested
299 // parallelism with internal producer-consumer relationships
300 // (calling the function recursively spawns and blocks on parallel
301 // sub-tasks that communicate with each other via semaphores). If
302 // a parallel runtime calls the function when fewer than this many
303 // threads are idle, it may need to create more threads to
304 // complete the task, or else risk deadlock due to committing all
305 // threads to tasks that cannot complete without more.
306 //
307 // FIXME: Note that extern stages are assumed to only require a
308 // single thread to complete. If the extern stage is itself a
309 // Halide pipeline, this may be an underestimate.
310 int min_threads;
311
312 // The calls to the function should be in serial order from min to min+extent-1, with only
313 // one executing at a time. If false, any order is fine, and
314 // concurrency is fine.
315 bool serial;
316};

497/** A runtime tag for a type in the halide type system. Can be ints,
498 * unsigned ints, or floats of various bit-widths (the 'bits'
499 * field). Can also be vectors of the same (by setting the 'lanes'
500 * field to something larger than one). This struct should be
501 * exactly 32-bits in size. */
502struct halide_type_t {
503 /** The basic type code: signed integer, unsigned integer, or floating point. */
504#if (__cplusplus >= 201103L || _MSVC_LANG >= 201103L)
505 HALIDE_ATTRIBUTE_ALIGN(1)
506 halide_type_code_t code; // halide_type_code_t
507#else
508 HALIDE_ATTRIBUTE_ALIGN(1)
509 uint8_t code; // halide_type_code_t
510#endif

512 /** The number of bits of precision of a single scalar value of this type. */
513 HALIDE_ATTRIBUTE_ALIGN(1)
514 uint8_t bits;
515
516 /** How many elements in a vector. This is 1 for scalar types. */
517 HALIDE_ATTRIBUTE_ALIGN(2)
518 uint16_t lanes;
519
520#if (__cplusplus >= 201103L || _MSVC_LANG >= 201103L)
521 /** Construct a runtime representation of a Halide type from:
522 * code: The fundamental type from an enum.
523 * bits: The bit size of one element.
524 * lanes: The number of vector elements in the type. */
525 HALIDE_ALWAYS_INLINE halide_type_t(halide_type_code_t code, uint8_t bits, uint16_t lanes = 1)
526 : code(code), bits(bits), lanes(lanes) {
527 }

529 /** Default constructor is required e.g. to declare halide_trace_event
530 * instances. */
531 HALIDE_ALWAYS_INLINE halide_type_t()
532 : code((halide_type_code_t)0), bits(0), lanes(0) {
533 }
534
535 HALIDE_ALWAYS_INLINE halide_type_t with_lanes(uint16_t new_lanes) const {
536 return halide_type_t((halide_type_code_t)code, bits, new_lanes);
537 }
538
539 /** Compare two types for equality. */
540 HALIDE_ALWAYS_INLINE bool operator==(const halide_type_t &other) const {
541 return as_u32() == other.as_u32();
542 }
543
544 HALIDE_ALWAYS_INLINE bool operator!=(const halide_type_t &other) const {
545 return !(*this == other);
546 }
547
548 HALIDE_ALWAYS_INLINE bool operator<(const halide_type_t &other) const {
549 return as_u32() < other.as_u32();
550 }
551
552 /** Size in bytes for a single element, even if width is not 1, of this type. */
553 HALIDE_ALWAYS_INLINE int bytes() const {
554 return (bits + 7) / 8;
555 }
556
557 HALIDE_ALWAYS_INLINE uint32_t as_u32() const {
558 uint32_t u;
559 memcpy(&u, this, sizeof(u));
560 return u;
561 }

1944// Allocating and freeing device memory is often very slow. The
1945// methods below give Halide's runtime permission to hold onto device
1946// memory to service future requests instead of returning it to the
1947// underlying device API. The API does not manage an allocation pool,
1948// all it does is provide access to a shared counter that acts as a
1949// limit on the unused memory not yet returned to the underlying
1950// device API. It makes callbacks to participants when memory needs to
1951// be released because the limit is about to be exceeded (either
1952// because the limit has been reduced, or because the memory owned by
1953// some participant becomes unused).
1954
1955/** Tell Halide whether or not it is permitted to hold onto device
1956 * allocations to service future requests instead of returning them
1957 * eagerly to the underlying device API. Many device allocators are
1958 * quite slow, so it can be beneficial to set this to true. The
1959 * default value for now is false.
1960 *
1961 * Note that if enabled, the eviction policy is very simplistic. The
1962 * 32 most-recently used allocations are preserved, regardless of
1963 * their size. Additionally, if a call to cuMalloc results in an
1964 * out-of-memory error, the entire cache is flushed and the allocation
1965 * is retried. See https://github.com/halide/Halide/issues/4093
1966 *
1967 * If set to false, releases all unused device allocations back to the
1968 * underlying device APIs. For finer-grained control, see specific
1969 * methods in each device api runtime. */
