61int _convert_py_buffer_to_halide(
62 PyObject* pyobj, int dimensions, int flags,
63 halide_dimension_t* dim, // array of size `dimensions`
64 halide_buffer_t* out, Py_buffer &buf, const char* name) {
65 int ret = PyObject_GetBuffer(
66 pyobj, &buf, PyBUF_FORMAT | PyBUF_STRIDED_RO | PyBUF_ANY_CONTIGUOUS | flags);
67 if (ret < 0) {
68 return ret;
69 }
70 if (dimensions && buf.ndim != dimensions) {
71 PyErr_Format(PyExc_ValueError, "Invalid argument %s: Expected %d dimensions, got %d",
72 name, dimensions, buf.ndim);
73 PyBuffer_Release(&buf);
74 return -1;
75 }
76 /* We'll get a buffer that's either:
77 * C_CONTIGUOUS (last dimension varies the fastest, i.e., has stride=1) or
78 * F_CONTIGUOUS (first dimension varies the fastest, i.e., has stride=1).
79 * The latter is preferred, since it's already in the format that Halide
80 * needs. It can can be achieved in numpy by passing order='F' during array
81 * creation. However, if we do get a C_CONTIGUOUS buffer, flip the dimensions
82 * (transpose) so we can process it without having to reallocate.
83 */
84 int i, j, j_step;
85 if (PyBuffer_IsContiguous(&buf, 'F')) {
86 j = 0;
87 j_step = 1;
88 } else if (PyBuffer_IsContiguous(&buf, 'C')) {
89 j = buf.ndim - 1;
90 j_step = -1;
91 } else {
92 /* Python checks all dimensions and strides, so this typically indicates
93 * a bug in the array's buffer protocol. */
94 PyErr_Format(PyExc_ValueError, "Invalid buffer: neither C nor Fortran contiguous");
95 PyBuffer_Release(&buf);
96 return -1;
97 }
98 for (i = 0; i < buf.ndim; ++i, j += j_step) {
99 dim[i].min = 0;
100 dim[i].stride = (int)(buf.strides[j] / buf.itemsize); // strides is in bytes
101 dim[i].extent = (int)buf.shape[j];
102 dim[i].flags = 0;
103 if (buf.suboffsets && buf.suboffsets[i] >= 0) {
104 // Halide doesn't support arrays of pointers. But we should never see this
105 // anyway, since we specified PyBUF_STRIDED.
106 PyErr_Format(PyExc_ValueError, "Invalid buffer: suboffsets not supported");
107 PyBuffer_Release(&buf);
108 return -1;
109 }
110 }
111 if (dim[buf.ndim - 1].extent * dim[buf.ndim - 1].stride * buf.itemsize != buf.len) {
112 PyErr_Format(PyExc_ValueError, "Invalid buffer: length %ld, but computed length %ld",
113 buf.len, buf.shape[0] * buf.strides[0]);
114 PyBuffer_Release(&buf);
115 return -1;
116 }
117 *out = halide_buffer_t();
118 if (!buf.format) {
119 out->type.code = halide_type_uint;
120 out->type.bits = 8;
121 } else {
122 /* Convert struct type code. See
123 * https://docs.python.org/2/library/struct.html#module-struct */
124 char* p = buf.format;
125 while (strchr("@<>!=", *p)) {
126 p++; // ignore little/bit endian (and alignment)
127 }
128 if (*p == 'f' || *p == 'd') {
129 // 'f' and 'd' are float and double, respectively.
130 out->type.code = halide_type_float;
131 } else if (*p >= 'a' && *p <= 'z') {
132 // lowercase is signed int.
133 out->type.code = halide_type_int;
134 } else {
135 // uppercase is unsigned int.
136 out->type.code = halide_type_uint;
137 }
138 const char* type_codes = "bB?hHiIlLqQfd"; // integers and floats
139 if (strchr(type_codes, *p)) {
140 out->type.bits = (uint8_t)buf.itemsize * 8;
141 } else {
142 // We don't handle 's' and 'p' (char[]) and 'P' (void*)
143 PyErr_Format(PyExc_ValueError, "Invalid data type for %s: %s", name, buf.format);
144 PyBuffer_Release(&buf);
145 return -1;
146 }
147 }
148 out->type.lanes = 1;
149 out->dimensions = buf.ndim;
150 out->dim = dim;
151 out->host = (uint8_t*)buf.buf;
152 return 0;
153}
