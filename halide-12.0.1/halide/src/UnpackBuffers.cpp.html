18struct BufferInfo {
19 Expr handle;
20 int dimensions;
21};

23class FindBufferSymbols : public IRVisitor {
24 using IRVisitor::visit;
25
26 void visit_param(const string &ref_name, const Parameter &param) {
27 if (param.defined() && param.is_buffer()) {
28 const string &name = param.name();
29 buffers[name] =
30 BufferInfo{Variable::make(type_of<halide_buffer_t *>(), name + ".buffer", param),
31 param.dimensions()};
32 }
33 }
34
35 void visit_buffer(const string &ref_name, const Buffer<> &buffer) {
36 if (buffer.defined()) {
37 const string &name = buffer.name();
38 buffers[name] =
39 BufferInfo{Variable::make(type_of<halide_buffer_t *>(), name + ".buffer", buffer),
40 buffer.dimensions()};
41 }
42 }
43
44 void visit(const Variable *op) override {
45 visit_param(op->name, op->param);
46 visit_buffer(op->name, op->image);
47 symbols.insert(op->name);
48 }
49
50 void visit(const Load *op) override {
51 visit_param(op->name, op->param);
52 visit_buffer(op->name, op->image);
53 symbols.insert(op->name);
54 IRVisitor::visit(op);
55 }
56
57 void visit(const Store *op) override {
58 visit_param(op->name, op->param);
59 symbols.insert(op->name);
60 IRVisitor::visit(op);
61 }
62
63public:
64 set<string> symbols;
65 map<string, BufferInfo> buffers;
66};

68} // namespace

70Stmt unpack_buffers(Stmt s) {
71 FindBufferSymbols finder;
72 s.accept(&finder);
73
74 vector<pair<string, Expr>> lets;
75
76 for (auto &p : finder.buffers) {
77 const string &name = p.first;
78 const BufferInfo &info = p.second;
79 vector<Expr> args = {info.handle};
80
81 string host_var = name;
82 Expr host_val = Call::make(type_of<void *>(), Call::buffer_get_host, args, Call::Extern);
83 lets.emplace_back(host_var, host_val);
84
85 string dev_var = name + ".device";
86 Expr dev_val = Call::make(type_of<uint64_t>(), Call::buffer_get_device, args, Call::Extern);
87 lets.emplace_back(dev_var, dev_val);
88
89 string dev_interface_var = name + ".device_interface";
90 Expr dev_interface_val = Call::make(type_of<const halide_device_interface_t *>(),
91 Call::buffer_get_device_interface, args, Call::Extern);
92 lets.emplace_back(dev_interface_var, dev_interface_val);
93
94 string type_code_var = name + ".type";
95 Expr type_code_val = Call::make(UInt(32), Call::buffer_get_type, args, Call::Extern);
96 lets.emplace_back(type_code_var, type_code_val);
97
98 string host_dirty_var = name + ".host_dirty";
99 Expr host_dirty_val = Call::make(Bool(), Call::buffer_get_host_dirty, args, Call::Extern);
100 lets.emplace_back(host_dirty_var, host_dirty_val);
101
102 string dev_dirty_var = name + ".device_dirty";
103 Expr dev_dirty_val = Call::make(Bool(), Call::buffer_get_device_dirty, args, Call::Extern);
104 lets.emplace_back(dev_dirty_var, dev_dirty_val);
105
106 string dimensions_var = name + ".dimensions";
107 Expr dimensions_val = Call::make(Int(32), Call::buffer_get_dimensions, args, Call::Extern);
108 lets.emplace_back(dimensions_var, dimensions_val);
109
110 for (int i = 0; i < info.dimensions; i++) {
111 vector<Expr> args = {info.handle, i};
112
113 string min_var = name + ".min." + std::to_string(i);
114 Expr min_val = Call::make(Int(32), Call::buffer_get_min, args, Call::Extern);
115 lets.emplace_back(min_var, min_val);
116
117 string extent_var = name + ".extent." + std::to_string(i);
118 Expr extent_val = Call::make(Int(32), Call::buffer_get_extent, args, Call::Extern);
119 lets.emplace_back(extent_var, extent_val);
120
121 string stride_var = name + ".stride." + std::to_string(i);
122 Expr stride_val = Call::make(Int(32), Call::buffer_get_stride, args, Call::Extern);
123 lets.emplace_back(stride_var, stride_val);
124 }
125 }
126
127 while (!lets.empty()) {
128 pair<string, Expr> l = lets.back();
129 lets.pop_back();
130 if (finder.symbols.count(l.first)) {
131 s = LetStmt::make(l.first, l.second, s);
132 }
133 }
134
135 // Create buffer is not null assertions
136 for (auto &p : finder.buffers) {
137 Expr buf = p.second.handle;
138 Expr cond = reinterpret<uint64_t>(buf) != 0;
139 Expr error = Call::make(Int(32), "halide_error_buffer_argument_is_null",
140 {p.first}, Call::Extern);
141 Stmt check = AssertStmt::make(cond, error);
142 s = Block::make(check, s);
143 }
144
145 return s;
146}
