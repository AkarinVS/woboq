127void IRVisitor::visit(const Call *op) {
128 for (size_t i = 0; i < op->args.size(); i++) {
129 op->args[i].accept(this);
130 }
131
132 // Consider extern call args
133 if (op->func.defined()) {
134 Function f(op->func);
135 if (op->call_type == Call::Halide && f.has_extern_definition()) {
136 for (size_t i = 0; i < f.extern_arguments().size(); i++) {
137 ExternFuncArgument arg = f.extern_arguments()[i];
138 if (arg.is_expr()) {
139 arg.expr.accept(this);
140 }
141 }
142 }
143 }
144}
