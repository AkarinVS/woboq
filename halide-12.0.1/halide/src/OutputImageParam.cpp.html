94OutputImageParam &OutputImageParam::set_estimates(const Region &estimates) {
95 const int d = dimensions();
96 user_assert((int)estimates.size() == d)
97 << "ImageParam " << name() << " has " << d << " dimensions, "
98 << "but the estimates passed to set_estimates contains " << estimates.size() << " pairs.\n";
99 for (int i = 0; i < d; i++) {
100 dim(i).set_estimate(estimates[i].min, estimates[i].extent);
101 }
102 return *this;
103}
