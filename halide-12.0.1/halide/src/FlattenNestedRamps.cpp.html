15class FlattenRamps : public IRMutator {
16 using IRMutator::visit;
17
18 Expr visit(const Ramp *op) override {
19 if (op->base.type().is_vector()) {
20 Expr base = mutate(op->base);
21 Expr stride = mutate(op->stride);
22 std::vector<Expr> ramp_elems;
23 for (int ix = 0; ix < op->lanes; ix++) {
24 ramp_elems.push_back(base + ix * stride);
25 }
26
27 return Shuffle::make_concat(ramp_elems);
28 }
29
30 return IRMutator::visit(op);
31 }
32
33 Expr visit(const Broadcast *op) override {
34 if (op->value.type().is_vector()) {
35 Expr value = mutate(op->value);
36 return Shuffle::make_broadcast(value, op->lanes);
37 }
38
39 return IRMutator::visit(op);
40 }
41
42 Expr visit(const Load *op) override {
43 // Convert a load of a bounded span of indices into a shuffle
44 // of a dense or strided load if possible.
45 const int lanes = op->type.lanes();
46 // This is about converting *to* a dense ramp, so we don't
47 // want to do this if it's already a dense ramp.
48 const Ramp *ramp = op->index.as<Ramp>();
49 if (lanes > 1 &&
50 is_const_one(op->predicate) &&
51 (ramp == nullptr || ramp->lanes < lanes)) {
52
53 Interval bounds_of_lanes = bounds_of_expr_in_scope(op->index, Scope<Interval>::empty_scope());
54 Expr min_lane;
55 if (!bounds_of_lanes.has_lower_bound()) {
56 return IRMutator::visit(op);
57 } else {
58 min_lane = bounds_of_lanes.min;
59 }
60
61 // Extract each index as a scalar
62 vector<Expr> indices(lanes);
63 for (int i = 0; i < lanes; i++) {
64 indices[i] = extract_lane(op->index, i);
65 }
66
67 // Check if the other indices are just the min index plus a constant
68 vector<int> const_indices;
69 const_indices.reserve(lanes);
70 int max_constant_offset = 0;
71 for (Expr &idx : indices) {
72 idx = simplify(common_subexpression_elimination(idx - min_lane));
73 const int64_t *i = as_const_int(idx);
74 if (i) {
75 const_indices.push_back((int)(*i));
76 max_constant_offset = std::max((int)(*i), max_constant_offset);
77 } else {
78 break;
79 }
80 }
81
82 // If they are, we'll have a full vector of const_indices
83 if ((int)const_indices.size() == lanes) {
84
85 // Compute the stride for the underlying strided load
86 int stride = 0;
87 for (int c : const_indices) {
88 stride = (int)gcd(stride, c);
89 }
90 for (int &c : const_indices) {
91 c /= stride;
92 }
93
94 // Compute the number of elements loaded
95 int extent = (int)((max_constant_offset / stride) + 1);
96
97 // If we're gathering from a very large range, it
98 // might be better to just do the gather rather than
99 // doing a big dense load and then shuffling. We
100 // currently do the big-load-and-shuffle if we're
101 // going to use at least a quarter of the values
102 // loaded.
103 //
104 // TODO: It would be good to be able to control this
105 // in the schedule somehow.
106 const int max_unused_lane_factor = 4;
107 if (extent < max_unused_lane_factor * lanes) {
108 Expr dense_index = Ramp::make(min_lane, make_const(min_lane.type(), stride), extent);
109 Expr dense_load =
110 Load::make(op->type.with_lanes(extent), op->name, dense_index,
111 op->image, op->param,
112 const_true(extent), ModulusRemainder{});
113 return Shuffle::make({dense_load}, const_indices);
114 }
115 }
116 }
117 return IRMutator::visit(op);
118 }
119};
