62class SplitTuples : public IRMutator {
63 using IRMutator::visit;
64
65 map<string, set<int>> func_value_indices;
66
67 Stmt visit(const Realize *op) override {
68 ScopedBinding<int> bind(realizations, op->name, 0);
69 if (op->types.size() > 1) {
70 // Make a nested set of realize nodes for each tuple element
71 Stmt body = mutate(op->body);
72 for (int i = (int)op->types.size() - 1; i >= 0; i--) {
73 body = Realize::make(op->name + "." + std::to_string(i),
74 {op->types[i]}, op->memory_type,
75 op->bounds, op->condition, body);
76 }
77 return body;
78 } else {
79 return IRMutator::visit(op);
80 }
81 }
82
83 Stmt visit(const For *op) override {
84 FindCallValueIndices find;
85 op->body.accept(&find);
86
87 ScopedValue<map<string, set<int>>> func_value_indices_v(func_value_indices, find.func_value_indices);
88 return IRMutator::visit(op);
89 }
90
91 Stmt visit(const Prefetch *op) override {
92 if (!op->prefetch.param.defined() && (op->types.size() > 1)) {
93 Stmt body = mutate(op->body);
94 // Split the prefetch from a multi-dimensional halide tuple to
95 // prefetches of each tuple element. Keep only prefetches of
96 // elements that are actually used in the loop body.
97 const auto &indices = func_value_indices.find(op->name);
98 internal_assert(indices != func_value_indices.end());
99
100 for (const auto &idx : indices->second) {
101 internal_assert(idx < (int)op->types.size());
102 body = Prefetch::make(op->name + "." + std::to_string(idx), {op->types[(idx)]}, op->bounds, op->prefetch, op->condition, body);
103 }
104 return body;
105 } else {
106 return IRMutator::visit(op);
107 }
108 }
109
110 Expr visit(const Call *op) override {
111 if (op->call_type == Call::Halide) {
112 auto it = env.find(op->name);
113 internal_assert(it != env.end());
114 Function f = it->second;
115 string name = op->name;
116 bool changed = false;
117 if (f.outputs() > 1) {
118 name += "." + std::to_string(op->value_index);
119 changed = true;
120 }
121 vector<Expr> args;
122 for (const Expr &e : op->args) {
123 args.push_back(mutate(e));
124 changed = changed || !args.back().same_as(e);
125 }
126 // It's safe to hook up the pointer to the function
127 // unconditionally. This expr never gets held by a
128 // Function, so there can't be a cycle. We do this even
129 // for scalar provides.
130 if (changed) {
131 return Call::make(op->type, name, args, op->call_type, f.get_contents());
132 } else {
133 return op;
134 }
135 } else {
136 return IRMutator::visit(op);
137 }
138 }

140 Stmt visit_provide(const Provide *op, const Atomic *atomic) {
141
142 if (op->values.size() == 1) {
143 if (atomic) {
144 return IRMutator::visit(atomic);
145 } else {
146 return IRMutator::visit(op);
147 }
148 }
149
150 // Mutate the args
151 vector<Expr> args;
152 for (const Expr &e : op->args) {
153 args.push_back(mutate(e));
154 }
155
156 // Get the Function
157 auto it = env.find(op->name);
158 internal_assert(it != env.end());
159 Function f = it->second;
160
161 // For the new value of each tuple component, what existing
162 // tuple components does it already depend on?
163 vector<set<int>> dependencies(op->values.size());
164 for (int i = 0; i < (int)op->values.size(); i++) {
165 class Checker : public IRVisitor {
166 using IRVisitor::visit;
167 vector<pair<string, Expr>> lets;
168
169 void visit(const Let *op) override {
170 op->value.accept(this);
171 lets.emplace_back(op->name, op->value);
172 op->body.accept(this);
173 lets.pop_back();
174 }
175
176 void visit(const Call *op) override {
177 if (op->call_type == Call::Halide &&
178 op->name == func_name &&
179 could_alias(op->args, store_args)) {
180 deps.insert(op->value_index);
181 }
182 }
183
184 bool could_alias(const vector<Expr> &a, const vector<Expr> &b) {
185 internal_assert(a.size() == b.size());
186 // Construct a boolean Expr that says the addresses are equal
187 Expr aliases = const_true();
188 for (size_t i = 0; i < a.size(); i++) {
189 aliases = aliases && (a[i] == b[i]);
190 }
191 // Might need some of the containing lets
192 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
193 if (expr_uses_var(aliases, it->first)) {
194 aliases = Let::make(it->first, it->second, aliases);
195 }
196 }
197 return !can_prove(!aliases);
198 }
199
200 public:
201 set<int> &deps;
202 const string &func_name;
203 const vector<Expr> &store_args;
204 Checker(set<int> &deps,
205 const string &func_name,
206 const vector<Expr> &store_args)
207 : deps(deps), func_name(func_name), store_args(store_args) {
208 }
209 } checker(dependencies[i], op->name, args);
210 op->values[i].accept(&checker);
211 }
212
213 // Build clusters of tuple components where two components
214 // belong to the same cluster if any of their loads or stores
215 // may alias
216 vector<vector<int>> clusters;
217 // Reserve space so that we can use pointers to clusters.
218 clusters.reserve(op->values.size());
219 for (int i = 0; i < (int)op->values.size(); i++) {
220 // What clusters does it already belong to?
221 vector<int> *owning_cluster = nullptr;
222 for (auto &c : clusters) {
223 bool belongs_to_this_cluster = false;
224 for (int j : c) {
225 if (dependencies[j].count(i) ||
226 dependencies[i].count(j)) {
227 belongs_to_this_cluster = true;
228 break;
229 }
230 }
231 if (belongs_to_this_cluster) {
232 if (owning_cluster) {
233 // It's already in a cluster! We need to merge the clusters.
234 owning_cluster->insert(owning_cluster->end(), c.begin(), c.end());
235 c.clear();
236 } else {
237 owning_cluster = &c;
238 c.push_back(i);
239 }
240 }
241 }
242 if (!owning_cluster) {
243 // Make a new cluster
244 clusters.emplace_back();
245 clusters.back().push_back(i);
246 }
247 }
248
249 // If each cluster has only a single store in it, we can use
250 // CAS loops or atomic adds and avoid ever needing to wrap
251 // things in a mutex. We express this using separate atomic
252 // nodes per store. If there's no mutex involved at all, then
253 // there's no benefit in packing things together into a single
254 // critical section.
255 bool separate_atomic_nodes_per_store =
256 ((atomic && atomic->mutex_name.empty()) ||
257 (clusters.size() == op->values.size()));
258
259 // For each cluster, build a list of scalar provide
260 // statements, and a list of lets to wrap them.
261 vector<Stmt> result;
262 for (auto &c : clusters) {
263 if (c.empty()) {
264 continue;
265 }
266 std::sort(c.begin(), c.end());
267 vector<Stmt> provides;
268 vector<pair<string, Expr>> lets;
269
270 Stmt s;
271
272 if (c.size() == 1) {
273 // Just make a provide node
274 int i = *c.begin();
275 string name = op->name + "." + std::to_string(i);
276 s = Provide::make(name, {mutate(op->values[i])}, args);
277 } else {
278 // Make a list of let statements that compute the
279 // values (doing any loads), and then a block of
280 // provide statements that do the stores.
281 for (auto i : c) {
282 string name = op->name + "." + std::to_string(i);
283 string var_name = name + ".value";
284 Expr val = mutate(op->values[i]);
285 if (!is_undef(val)) {
286 lets.emplace_back(var_name, val);
287 val = Variable::make(val.type(), var_name);
288 }
289 provides.push_back(Provide::make(name, {val}, args));
290 }
291
292 s = Block::make(provides);
293
294 while (!lets.empty()) {
295 auto p = lets.back();
296 lets.pop_back();
297 s = LetStmt::make(p.first, p.second, s);
298 }
299 }
300
301 if (atomic && separate_atomic_nodes_per_store) {
302 s = Atomic::make(atomic->producer_name, atomic->mutex_name, s);
303 }
304
305 internal_assert(s.defined());
306 result.push_back(s);
307 }
308
309 {
310 Stmt s = Block::make(result);
311 if (atomic && !separate_atomic_nodes_per_store) {
312 s = Atomic::make(atomic->producer_name, atomic->mutex_name, s);
313 }
314 return s;
315 }
316 }

319 return visit_provide(op, nullptr);
320 }

322 Stmt visit(const Atomic *op) override {
323 // At this point in lowering, the only child of an atomic node
324 // should be a single provide node. We haven't many any
325 // statement mutations yet that would put things in between
326 // the provide and the atomic.
327 if (const Provide *p = op->body.as<Provide>()) {
328 return visit_provide(p, op);
329 } else {
330 return IRMutator::visit(op);
331 }

349 class GetScatterGatherSize : public IRVisitor {
350 bool permitted = true;
351 using IRVisitor::visit;
352 void visit(const Call *op) override {
353 if (op->is_intrinsic(Call::scatter_gather)) {
354 user_assert(permitted)
355 << "Can't nest an expression tuple inside another in definition of "
356 << producer_name;
357 if (result == 0) {
358 result = (int)op->args.size();
359 } else {
360 user_assert((int)op->args.size() == result)
361 << "Expression tuples of mismatched sizes used in definition of "
362 << producer_name << ": " << result << " vs " << op->args.size();
363 }
364 // No nesting tuples
365 permitted = false;
366 IRVisitor::visit(op);
367 permitted = true;
368 } else {
369 IRVisitor::visit(op);
370 }
371 }
372
373 public:
374 string producer_name;
375 int result = 0;
376 } get_scatter_gather_size_visitor;
377
378 int get_scatter_gather_size(const IRNode *op) {
379 if (producer) {
380 get_scatter_gather_size_visitor.producer_name = producer->name;
381 }
382 get_scatter_gather_size_visitor.result = 0;
383 op->accept(&get_scatter_gather_size_visitor);
384
385 // Maybe the user did something like jam a gather op into a
386 // constraint or tile size, so this is a user_assert.
387 user_assert(producer || get_scatter_gather_size_visitor.result == 0)
388 << "scatter/gather expression used outside of a Func definition";
389
390 return get_scatter_gather_size_visitor.result;
391 }

415 Stmt visit_gather_let_stmt(const LetStmt *op, int size) {
416 // Split this variable into the gather components
417 vector<pair<string, Expr>> lets;
418 vector<Expr> vars;
419 for (extractor.idx = 0; extractor.idx < size; extractor.idx++) {
420 string name = unique_name(op->name + "." + std::to_string(extractor.idx));
421 lets.emplace_back(name, extractor.mutate(op->value));
422 vars.push_back(Variable::make(op->value.type(), name));
423 }
424
425 Stmt body = op->body;
426 Expr gather_replacement = Call::make(op->value.type(),
427 Call::scatter_gather,
428 vars,
429 Call::PureIntrinsic);
430 body = substitute(op->name, gather_replacement, body);
431 body = mutate(body);
432
433 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
434 body = LetStmt::make(it->first, it->second, body);
435 }
436
437 return body;
438 }
439
440 Stmt visit(const LetStmt *op) override {
441 vector<pair<string, Expr>> lets;
442 int size = 0;
443 Stmt body;
444 do {
445 body = op->body;
446 size = get_scatter_gather_size(op->value.get());
447 if (size != 0) {
448 break;
449 }
450 lets.emplace_back(op->name, op->value);
451 op = body.as<LetStmt>();
452 } while (op);
453
454 if (size) {
455 internal_assert(op);
456 body = visit_gather_let_stmt(op, size);
457 } else {
458 internal_assert(op == nullptr);
459 body = mutate(body);
460 }
461
462 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
463 body = LetStmt::make(it->first, it->second, body);
464 }
465
466 return body;
467 }
468
469 Stmt visit(const Provide *op) override {
470 int size = get_scatter_gather_size(op);
471
472 if (size == 0) {
473 return IRMutator::visit(op);
474 }
475
476 // The LHS should contain at least one scatter op, or our scatters
477 // all go to the same place. Is it worth asserting this? It
478 // could be a bug, or it could be some sort of degenerate base case.
479
480 // Fork the args and the RHS into their various versions
481 vector<Stmt> provides;
482 vector<string> names;
483 vector<Expr> exprs;
484 for (extractor.idx = 0; extractor.idx < size; extractor.idx++) {
485 vector<Expr> args = op->args;
486 for (Expr &a : args) {
487 string name = unique_name('t');
488 exprs.push_back(extractor.mutate(a));
489 names.push_back(name);
490 a = Variable::make(a.type(), name);
491 }
492 vector<Expr> values = op->values;
493 for (Expr &v : values) {
494 v = extractor.mutate(v);
495 string name = unique_name('t');
496 exprs.push_back(extractor.mutate(v));
497 names.push_back(name);
498 v = Variable::make(v.type(), name);
499 }
500 provides.push_back(Provide::make(op->name, values, args));
501 }
502
503 Stmt s = Block::make(provides);
504 // We just duplicated all the non-scatter/gather stuff too,
505 // so do joint CSE on the exprs
506 Expr bundle = Call::make(Int(32), Call::bundle, exprs, Call::PureIntrinsic);
507 bundle = common_subexpression_elimination(bundle);
508
509 vector<pair<string, Expr>> lets;
510 while (const Let *let = bundle.as<Let>()) {
511 lets.emplace_back(let->name, let->value);
512 bundle = let->body;
513 }
514 const Call *c = bundle.as<Call>();
515 internal_assert(c && c->is_intrinsic(Call::bundle));
516 for (size_t i = 0; i < exprs.size(); i++) {
517 if (is_pure(c->args[i])) {
518 // names[i] is only used once, so if the value is pure
519 // it should be substituted in
520 s = substitute(names[i], c->args[i], s);
521 } else {
522 lets.emplace_back(names[i], c->args[i]);
523 }
524 }
525
526 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
527 s = LetStmt::make(it->first, it->second, s);
528 }
529
530 return s;
531 }
