29 ParameterContents(Type t, bool b, int d, const std::string &n)
30 : type(t), dimensions(d), name(n), buffer(Buffer<>()), data(0),
31 host_alignment(t.bytes()), buffer_constraints(dimensions), is_buffer(b) {
32 // stride_constraint[0] defaults to 1. This is important for
33 // dense vectorization. You can unset it by setting it to a
34 // null expression. (param.set_stride(0, Expr());)
35 if (dimensions > 0) {
36 buffer_constraints[0].stride = 1;
37 }
38 }
39};

107Expr Parameter::scalar_expr() const {
108 check_is_scalar();
109 const Type t = type();
110 if (t.is_float()) {
111 switch (t.bits()) {
112 case 16:
113 if (t.is_bfloat()) {
114 return Expr(scalar<bfloat16_t>());
115 } else {
116 return Expr(scalar<float16_t>());
117 }
118 case 32:
119 return Expr(scalar<float>());
120 case 64:
121 return Expr(scalar<double>());
122 }
123 } else if (t.is_int()) {
124 switch (t.bits()) {
125 case 8:
126 return Expr(scalar<int8_t>());
127 case 16:
128 return Expr(scalar<int16_t>());
129 case 32:
130 return Expr(scalar<int32_t>());
131 case 64:
132 return Expr(scalar<int64_t>());
133 }
134 } else if (t.is_uint()) {
135 switch (t.bits()) {
136 case 1:
137 return make_bool(scalar<bool>());
138 case 8:
139 return Expr(scalar<uint8_t>());
140 case 16:
141 return Expr(scalar<uint16_t>());
142 case 32:
143 return Expr(scalar<uint32_t>());
144 case 64:
145 return Expr(scalar<uint64_t>());
146 }
147 } else if (t.is_handle()) {
148 // handles are always uint64 internally.
149 switch (t.bits()) {
150 case 64:
151 return Expr(scalar<uint64_t>());
152 }
153 }
154 internal_error << "Unsupported type " << t << " in scalar_expr\n";
155 return Expr();
156}

340ArgumentEstimates Parameter::get_argument_estimates() const {
341 ArgumentEstimates argument_estimates;
342 if (!is_buffer()) {
343 argument_estimates.scalar_def = default_value();
344 argument_estimates.scalar_min = min_value();
345 argument_estimates.scalar_max = max_value();
346 argument_estimates.scalar_estimate = estimate();
347 } else {
348 argument_estimates.buffer_estimates.resize(dimensions());
349 for (int i = 0; i < dimensions(); i++) {
350 argument_estimates.buffer_estimates[i].min = min_constraint_estimate(i);
351 argument_estimates.buffer_estimates[i].extent = extent_constraint_estimate(i);
352 }
353 }
354 return argument_estimates;
355}

357void check_call_arg_types(const std::string &name, std::vector<Expr> *args, int dims) {
358 user_assert(args->size() == (size_t)dims)
359 << args->size() << "-argument call to \""
360 << name << "\", which has " << dims << " dimensions.\n";
361
362 for (size_t i = 0; i < args->size(); i++) {
363 user_assert((*args)[i].defined())
364 << "Argument " << i << " to call to \"" << name << "\" is an undefined Expr\n";
365 Type t = (*args)[i].type();
366 if (t.is_float() || (t.is_uint() && t.bits() >= 32) || (t.is_int() && t.bits() > 32)) {
367 user_error << "Implicit cast from " << t << " to int in argument " << (i + 1)
368 << " in call to \"" << name << "\" is not allowed. Use an explicit cast.\n";
369 }
370 // We're allowed to implicitly cast from other varieties of int
371 if (t != Int(32)) {
372 (*args)[i] = Cast::make(Int(32), (*args)[i]);
373 }
374 }
