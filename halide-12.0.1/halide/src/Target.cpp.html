44#ifdef _LP64
45void cpuid(int info[4], int infoType, int extra) {
46 __asm__ __volatile__(
47 "cpuid \n\t"
48 : "=a"(info[0]), "=b"(info[1]), "=c"(info[2]), "=d"(info[3])
49 : "0"(infoType), "2"(extra));
50}
51#else
52static void cpuid(int info[4], int infoType, int extra) {
53 // We save %ebx in case it's the PIC register
54 __asm__ __volatile__(
55 "mov{l}\t{%%}ebx, %1 \n\t"
56 "cpuid \n\t"
57 "xchg{l}\t{%%}ebx, %1 \n\t"
58 : "=a"(info[0]), "=r"(info[1]), "=c"(info[2]), "=d"(info[3])
59 : "0"(infoType), "2"(extra));
60}
61#endif
62#endif

65Target calculate_host_target() {
66 Target::OS os = Target::OSUnknown;
67#ifdef __linux__
68 os = Target::Linux;
69#endif
70#ifdef _WIN32
71 os = Target::Windows;
72#endif
73#ifdef __APPLE__
74 os = Target::OSX;
75#endif
76
77 bool use_64_bits = (sizeof(size_t) == 8);
78 int bits = use_64_bits ? 64 : 32;
79 std::vector<Target::Feature> initial_features;
80
81#if __riscv__
82 Target::Arch arch = Target::RISCV;
83#else
84#if __mips__ || __mips || __MIPS__
85 Target::Arch arch = Target::MIPS;
86#else
87#if defined(__arm__) || defined(__aarch64__)
88 Target::Arch arch = Target::ARM;
89#else
90#if defined(__powerpc__) && (defined(__FreeBSD__) || defined(__linux__))
91 Target::Arch arch = Target::POWERPC;
92
93#if defined(__linux__)
94 unsigned long hwcap = getauxval(AT_HWCAP);
95 unsigned long hwcap2 = getauxval(AT_HWCAP2);
96#elif defined(__FreeBSD__)
97 unsigned long hwcap, hwcap2;
98 elf_aux_info(AT_HWCAP, &hwcap, sizeof(hwcap));
99 elf_aux_info(AT_HWCAP2, &hwcap2, sizeof(hwcap2));
100#endif
101 bool have_altivec = (hwcap & PPC_FEATURE_HAS_ALTIVEC) != 0;
102 bool have_vsx = (hwcap & PPC_FEATURE_HAS_VSX) != 0;
103 bool arch_2_07 = (hwcap2 & PPC_FEATURE2_ARCH_2_07) != 0;
104
105 user_assert(have_altivec)
106 << "The POWERPC backend assumes at least AltiVec support. This machine does not appear to have AltiVec.\n";
107
108 if (have_vsx) initial_features.push_back(Target::VSX);
109 if (arch_2_07) initial_features.push_back(Target::POWER_ARCH_2_07);
110#else
111 Target::Arch arch = Target::X86;
112
113 int info[4];
114 cpuid(info, 1, 0);
115 bool have_sse41 = (info[2] & (1 << 19)) != 0;
116 bool have_sse2 = (info[3] & (1 << 26)) != 0;
117 bool have_avx = (info[2] & (1 << 28)) != 0;
118 bool have_f16c = (info[2] & (1 << 29)) != 0;
119 bool have_rdrand = (info[2] & (1 << 30)) != 0;
120 bool have_fma = (info[2] & (1 << 12)) != 0;
121
122 user_assert(have_sse2)
123 << "The x86 backend assumes at least sse2 support. This machine does not appear to have sse2.\n"
124 << "cpuid returned: "
125 << std::hex << info[0]
126 << ", " << info[1]
127 << ", " << info[2]
128 << ", " << info[3]
129 << std::dec << "\n";
130
131 if (have_sse41) {
132 initial_features.push_back(Target::SSE41);
133 }
134 if (have_avx) {
135 initial_features.push_back(Target::AVX);
136 }
137 if (have_f16c) {
138 initial_features.push_back(Target::F16C);
139 }
140 if (have_fma) {
141 initial_features.push_back(Target::FMA);
142 }
143
144 if (use_64_bits && have_avx && have_f16c && have_rdrand) {
145 // So far, so good. AVX2/512?
146 // Call cpuid with eax=7, ecx=0
147 int info2[4];
148 cpuid(info2, 7, 0);
149 const uint32_t avx2 = 1U << 5;
150 const uint32_t avx512f = 1U << 16;
151 const uint32_t avx512dq = 1U << 17;
152 const uint32_t avx512pf = 1U << 26;
153 const uint32_t avx512er = 1U << 27;
154 const uint32_t avx512cd = 1U << 28;
155 const uint32_t avx512bw = 1U << 30;
156 const uint32_t avx512vl = 1U << 31;
157 const uint32_t avx512ifma = 1U << 21;
158 const uint32_t avx512 = avx512f | avx512cd;
159 const uint32_t avx512_knl = avx512 | avx512pf | avx512er;
160 const uint32_t avx512_skylake = avx512 | avx512vl | avx512bw | avx512dq;
161 const uint32_t avx512_cannonlake = avx512_skylake | avx512ifma; // Assume ifma => vbmi
162 if ((info2[1] & avx2) == avx2) {
163 initial_features.push_back(Target::AVX2);
164 }
165 if ((info2[1] & avx512) == avx512) {
166 initial_features.push_back(Target::AVX512);
167 if ((info2[1] & avx512_knl) == avx512_knl) {
168 initial_features.push_back(Target::AVX512_KNL);
169 }
170 if ((info2[1] & avx512_skylake) == avx512_skylake) {
171 initial_features.push_back(Target::AVX512_Skylake);
172 }
173 if ((info2[1] & avx512_cannonlake) == avx512_cannonlake) {
174 initial_features.push_back(Target::AVX512_Cannonlake);
175
176#if LLVM_VERSION >= 120
177 // Sapphire Rapids support was added in LLVM 12, so earlier versions cannot support this CPU's features.
178 const uint32_t avx512vnni = 1U << 11; // vnni result in ecx
179 const uint32_t avx512bf16 = 1U << 5; // bf16 result in eax, with cpuid(eax=7, ecx=1)
180 int info3[4];
181 cpuid(info3, 7, 1);
182 if ((info2[2] & avx512vnni) == avx512vnni &&
183 (info3[0] & avx512bf16) == avx512bf16) {
184 initial_features.push_back(Target::AVX512_SapphireRapids);
185 }
186#endif
187 }
188 }
189 }
190#endif
191#endif
192#endif
193#endif
194
195 return {os, arch, bits, initial_features};
196}

237Target::Feature calculate_host_cuda_capability(Target t) {
238 const auto *interface = get_device_interface_for_device_api(DeviceAPI::CUDA, t);
239 internal_assert(interface->compute_capability);
240 int major, minor;
241 int err = interface->compute_capability(nullptr, &major, &minor);
242 internal_assert(err == 0) << "Failed to query cuda compute capability\n";
243 int ver = major * 10 + minor;
244 if (ver < 30) {
245 return Target::FeatureEnd;
246 } else if (ver < 32) {
247 return Target::CUDACapability30;
248 } else if (ver < 35) {
249 return Target::CUDACapability32;
250 } else if (ver < 50) {
251 return Target::CUDACapability35;
252 } else if (ver < 61) {
253 return Target::CUDACapability50;
254 } else if (ver < 70) {
255 return Target::CUDACapability61;
256 } else if (ver < 75) {
257 return Target::CUDACapability70;
258 } else if (ver < 80) {
259 return Target::CUDACapability75;
260 } else {
261 return Target::CUDACapability80;
262 }
263}

265Target::Feature get_host_cuda_capability(Target t) {
266 static Target::Feature cap = calculate_host_cuda_capability(t);
267 return cap;
268}

438bool merge_string(Target &t, const std::string &target) {
439 string rest = target;
440 vector<string> tokens;
441 size_t first_dash;
442 while ((first_dash = rest.find('-')) != string::npos) {
443 //Internal::debug(0) << first_dash << ", " << rest << "\n";
444 tokens.push_back(rest.substr(0, first_dash));
445 rest = rest.substr(first_dash + 1);
446 }
447 tokens.push_back(rest);
448
449 bool os_specified = false, arch_specified = false, bits_specified = false, features_specified = false;
450 bool is_host = false;
451
452 for (size_t i = 0; i < tokens.size(); i++) {
453 const string &tok = tokens[i];
454 Target::Feature feature;
455
456 if (tok == "host") {
457 if (i > 0) {
458 // "host" is now only allowed as the first token.
459 return false;
460 }
461 is_host = true;
462 t = get_host_target();
463 } else if (tok == "32" || tok == "64" || tok == "0") {
464 if (bits_specified) {
465 return false;
466 }
467 bits_specified = true;
468 t.bits = std::stoi(tok);
469 } else if (lookup_arch(tok, t.arch)) {
470 if (arch_specified) {
471 return false;
472 }
473 arch_specified = true;
474 } else if (lookup_os(tok, t.os)) {
475 if (os_specified) {
476 return false;
477 }
478 os_specified = true;
479 } else if (lookup_feature(tok, feature)) {
480 t.set_feature(feature);
481 features_specified = true;
482 } else if (tok == "trace_all") {
483 t.set_features({Target::TraceLoads, Target::TraceStores, Target::TraceRealizations});
484 features_specified = true;
485 } else {
486 return false;
487 }
488 }
489
490 if (is_host &&
491 t.has_feature(Target::CUDA) &&
492 !t.has_feature(Target::CUDACapability30) &&
493 !t.has_feature(Target::CUDACapability32) &&
494 !t.has_feature(Target::CUDACapability35) &&
495 !t.has_feature(Target::CUDACapability50) &&
496 !t.has_feature(Target::CUDACapability61) &&
497 !t.has_feature(Target::CUDACapability70) &&
498 !t.has_feature(Target::CUDACapability75) &&
499 !t.has_feature(Target::CUDACapability80)) {
500 // Detect host cuda capability
501 t.set_feature(get_host_cuda_capability(t));
502 }
503
504 if (arch_specified && !bits_specified) {
505 return false;
506 }
507
508 if (bits_specified && t.bits == 0) {
509 // bits == 0 is allowed iff arch and os are "unknown" and no features are set,
510 // to allow for roundtripping the string for default Target() ctor.
511 if (!(arch_specified && t.arch == Target::ArchUnknown) ||
512 !(os_specified && t.os == Target::OSUnknown) ||
513 features_specified) {
514 return false;
515 }
516 }
517
518 return true;
519}

521void bad_target_string(const std::string &target) {
522 const char *separator = "";
523 std::string architectures;
524 for (const auto &arch_entry : arch_name_map) {
525 architectures += separator + arch_entry.first;
526 separator = ", ";
527 }
528 separator = "";
529 std::string oses;
530 for (const auto &os_entry : os_name_map) {
531 oses += separator + os_entry.first;
532 separator = ", ";
533 }
534 separator = "";
535 // Format the features to go one feature over 70 characters per line,
536 // assume the first line starts with "Features are ".
537 int line_char_start = -(int)sizeof("Features are");
538 std::string features;
539 for (const auto &feature_entry : feature_name_map) {
540 features += separator + feature_entry.first;
541 if (features.length() - line_char_start > 70) {
542 separator = "\n";
543 line_char_start = features.length();
544 } else {
545 separator = ", ";
546 }
547 }
548 user_error << "Did not understand Halide target " << target << "\n"
549 << "Expected format is arch-bits-os-feature1-feature2-...\n"
550 << "Where arch is: " << architectures << ".\n"
551 << "bits is either 32 or 64.\n"
552 << "os is: " << oses << ".\n"
553 << "\n"
554 << "If arch, bits, or os are omitted, they default to the host.\n"
555 << "\n"
556 << "Features are: " << features << ".\n"
557 << "\n"
558 << "The target can also begin with \"host\", which sets the "
559 << "host's architecture, os, and feature set, with the "
560 << "exception of the GPU runtimes, which default to off.\n"
561 << "\n"
562 << "On this platform, the host target is: " << get_host_target().to_string() << "\n";
563}

576 }

607std::string Target::to_string() const {
608 string result;
609 for (const auto &arch_entry : arch_name_map) {
610 if (arch_entry.second == arch) {
611 result += arch_entry.first;
612 break;
613 }
614 }
615 result += "-" + std::to_string(bits);
616 for (const auto &os_entry : os_name_map) {
617 if (os_entry.second == os) {
618 result += "-" + os_entry.first;
619 break;
620 }
621 }
622 for (const auto &feature_entry : feature_name_map) {
623 if (has_feature(feature_entry.second)) {
624 result += "-" + feature_entry.first;
625 }
626 }
627 // Use has_feature() multiple times (rather than features_any_of())
628 // to avoid constructing a temporary vector for this rather-common call.
629 if (has_feature(Target::TraceLoads) && has_feature(Target::TraceStores) && has_feature(Target::TraceRealizations)) {
630 result = Internal::replace_all(result, "trace_loads-trace_realizations-trace_stores", "trace_all");
631 }
632 return result;
633}

635/** Was libHalide compiled with support for this target? */
636bool Target::supported() const {
637 bool bad = false;
638#if !defined(WITH_ARM)
639 bad |= arch == Target::ARM && bits == 32;
640#endif
641#if !defined(WITH_AARCH64)
642 bad |= arch == Target::ARM && bits == 64;
643#endif
644#if !defined(WITH_X86)
645 bad |= arch == Target::X86;
646#endif
647#if !defined(WITH_MIPS)
648 bad |= arch == Target::MIPS;
649#endif
650#if !defined(WITH_POWERPC)
651 bad |= arch == Target::POWERPC;
652#endif
653#if !defined(WITH_HEXAGON)
654 bad |= arch == Target::Hexagon;
655#endif
656#if !defined(WITH_WEBASSEMBLY)
657 bad |= arch == Target::WebAssembly;
658#endif
659#if !defined(WITH_RISCV)

791bool Target::supports_type(const Type &t, DeviceAPI device) const {
792 if (device == DeviceAPI::Default_GPU) {
793 device = get_default_device_api_for_target(*this);
794 }
795
796 if (device == DeviceAPI::Hexagon) {
797 // HVX supports doubles and long long in the scalar unit only.
798 if (t.is_float() || t.bits() == 64) {
799 return t.lanes() == 1;
800 }
801 } else if (device == DeviceAPI::Metal) {
802 // Metal spec says no double or long long.
803 if (t.bits() == 64) {
804 return false;
805 }
806 } else if (device == DeviceAPI::OpenCL) {
807 if (t.is_float() && t.bits() == 64) {
808 return has_feature(Target::CLDoubles);
809 }
810 } else if (device == DeviceAPI::D3D12Compute) {
811 // Shader Model 5.x can optionally support double-precision; 64-bit int
812 // types are not supported.
813 return t.bits() < 64;
814 } else if (device == DeviceAPI::OpenGLCompute) {
815 return t.bits() < 64;
816 }
817
818 return true;

882int Target::natural_vector_size(const Halide::Type &t) const {
883 user_assert(!has_unknowns())
884 << "natural_vector_size cannot be used on a Target with Unknown values.\n";
885
886 const bool is_integer = t.is_int() || t.is_uint();
887 const int data_size = t.bytes();
888
889 if (arch == Target::Hexagon) {
890 if (is_integer) {
891 if (has_feature(Halide::Target::HVX)) {
892 return 128 / data_size;
893 } else {
894 user_error << "Target uses hexagon arch without target feature hvx set.\n";
895 return 0;
896 }
897 } else {
898 // HVX does not have vector float instructions.
899 return 1;
900 }
901 } else if (arch == Target::X86) {
902 if (is_integer && (has_feature(Halide::Target::AVX512_Skylake) ||
903 has_feature(Halide::Target::AVX512_Cannonlake))) {
904 // AVX512BW exists on Skylake and Cannonlake
905 return 64 / data_size;
906 } else if (t.is_float() && (has_feature(Halide::Target::AVX512) ||
907 has_feature(Halide::Target::AVX512_KNL) ||
908 has_feature(Halide::Target::AVX512_Skylake) ||
909 has_feature(Halide::Target::AVX512_Cannonlake))) {
910 // AVX512F is on all AVX512 architectures
911 return 64 / data_size;
912 } else if (has_feature(Halide::Target::AVX2)) {
913 // AVX2 uses 256-bit vectors for everything.
914 return 32 / data_size;
915 } else if (!is_integer && has_feature(Halide::Target::AVX)) {
916 // AVX 1 has 256-bit vectors for float, but not for
917 // integer instructions.
918 return 32 / data_size;
919 } else {
920 // SSE was all 128-bit. We ignore MMX.
921 return 16 / data_size;
922 }
923 } else if (arch == Target::WebAssembly) {
924 if (has_feature(Halide::Target::WasmSimd128)) {
925 // 128-bit vectors for other types.
926 return 16 / data_size;
927 } else {
928 // No vectors, sorry.
929 return 1;
930 }
931 } else {
932 // Assume 128-bit vectors on other targets.
933 return 16 / data_size;
934 }
935}

937bool Target::get_runtime_compatible_target(const Target &other, Target &result) {
938 // Create mask to select features that:
939 // (a) must be included if either target has the feature (union)
940 // (b) must be included if both targets have the feature (intersection)
941 // (c) must match across both targets; it is an error if one target has the feature and the other doesn't
942
943 // clang-format off
944 const std::array<Feature, 18> union_features = {{
945 // These are true union features.
946 CUDA,
947 D3D12Compute,
948 Metal,
949 NoNEON,
950 OpenCL,
951 OpenGLCompute,
952
953 // These features are actually intersection-y, but because targets only record the _highest_,
954 // we have to put their union in the result and then take a lower bound.
955 CUDACapability30,
956 CUDACapability32,
957 CUDACapability35,
958 CUDACapability50,
959 CUDACapability61,
960 CUDACapability70,
961 CUDACapability75,
962 CUDACapability80,
963 HVX_v62,
964 HVX_v65,
965 HVX_v66,
966 }};
967 // clang-format on
968
969 // clang-format off
970 const std::array<Feature, 14> intersection_features = {{
971 ARMv7s,
972 ARMv81a,
973 AVX,
974 AVX2,
975 AVX512,
976 AVX512_Cannonlake,
977 AVX512_KNL,
978 AVX512_SapphireRapids,
979 AVX512_Skylake,
980 F16C,
981 FMA,
982 FMA4,
983 SSE41,
984 VSX,
985 }};
986 // clang-format on
987
988 // clang-format off
989 const std::array<Feature, 12> matching_features = {{
990 ASAN,
991 Debug,
992 HexagonDma,
993 HVX,
994 HVX_shared_object,
995 MSAN,
996 SoftFloatABI,
997 TSAN,
998 WasmThreads,
999 }};
1000 // clang-format on
1001
1002 // bitsets need to be the same width.
1003 decltype(result.features) union_mask;
1004 decltype(result.features) intersection_mask;
1005 decltype(result.features) matching_mask;
1006
1007 for (const auto &feature : union_features) {
1008 union_mask.set(feature);
1009 }
1010
1011 for (const auto &feature : intersection_features) {
1012 intersection_mask.set(feature);
1013 }
1014
1015 for (const auto &feature : matching_features) {
1016 matching_mask.set(feature);
1017 }
1018
1019 if (arch != other.arch || bits != other.bits || os != other.os) {
1020 Internal::debug(1) << "runtime targets must agree on platform (arch-bits-os)\n"
1021 << " this: " << *this << "\n"
1022 << " other: " << other << "\n";
1023 return false;
1024 }
1025
1026 if ((features & matching_mask) != (other.features & matching_mask)) {
1027 Internal::debug(1) << "runtime targets must agree on SoftFloatABI, Debug, TSAN, ASAN, MSAN, HVX, HexagonDma, and HVX_shared_object\n"
1028 << " this: " << *this << "\n"
1029 << " other: " << other << "\n";
1030 return false;
1031 }
1032
1033 // Union of features is computed through bitwise-or, and masked away by the features we care about
1034 // Intersection of features is computed through bitwise-and and masked away, too.
1035 // We merge the bits via bitwise or.
1036 Target output = Target{os, arch, bits};
1037 output.features = ((features | other.features) & union_mask) | ((features | other.features) & matching_mask) | ((features & other.features) & intersection_mask);
1038
1039#if LLVM_VERSION < 120
1040 // We require LLVM 12+ to compile SapphireRapids features.
1041 output.features.reset(AVX512_SapphireRapids);
1042#endif
1043
1044 // Pick tight lower bound for CUDA capability. Use fall-through to clear redundant features
1045 int cuda_a = get_cuda_capability_lower_bound();
1046 int cuda_b = other.get_cuda_capability_lower_bound();
1047
1048 // get_cuda_capability_lower_bound returns -1 when unused. Casting to unsigned makes this
1049 // large, so min selects the true lower bound when one target doesn't specify a capability,
1050 // and the other doesn't use CUDA at all.
1051 int cuda_capability = std::min((unsigned)cuda_a, (unsigned)cuda_b);
1052 if (cuda_capability < 30) {
1053 output.features.reset(CUDACapability30);
1054 }
1055 if (cuda_capability < 32) {
1056 output.features.reset(CUDACapability32);
1057 }
1058 if (cuda_capability < 35) {
1059 output.features.reset(CUDACapability35);
1060 }
1061 if (cuda_capability < 50) {
1062 output.features.reset(CUDACapability50);
1063 }
1064 if (cuda_capability < 61) {
1065 output.features.reset(CUDACapability61);
1066 }
1067 if (cuda_capability < 70) {
1068 output.features.reset(CUDACapability70);
1069 }
1070 if (cuda_capability < 75) {
1071 output.features.reset(CUDACapability75);
1072 }
1073 if (cuda_capability < 80) {
1074 output.features.reset(CUDACapability80);
1075 }
1076
1077 // Pick tight lower bound for HVX version. Use fall-through to clear redundant features
1078 int hvx_a = get_hvx_lower_bound(*this);
1079 int hvx_b = get_hvx_lower_bound(other);
1080
1081 // Same trick as above for CUDA
1082 int hvx_version = std::min((unsigned)hvx_a, (unsigned)hvx_b);
1083 if (hvx_version < 62) {
1084 output.features.reset(HVX_v62);
1085 }
1086 if (hvx_version < 65) {
1087 output.features.reset(HVX_v65);
1088 }
1089 if (hvx_version < 66) {
1090 output.features.reset(HVX_v66);
1091 }
1092
1093 result = output;
1094 return true;
1095}

1099void target_test() {
1100 Target t;
1101 for (const auto &feature : feature_name_map) {
1102 t.set_feature(feature.second);
1103 }
1104 for (int i = 0; i < (int)(Target::FeatureEnd); i++) {
1105 internal_assert(t.has_feature((Target::Feature)i)) << "Feature " << i << " not in feature_names_map.\n";
1106 }
1107
1108 // 3 targets: {A,B,C}. Want gcd(A,B)=C
1109 std::vector<std::array<std::string, 3>> gcd_tests = {
1110 {{"x86-64-linux-sse41-fma", "x86-64-linux-sse41-fma", "x86-64-linux-sse41-fma"}},
1111 {{"x86-64-linux-sse41-fma-no_asserts-no_runtime", "x86-64-linux-sse41-fma", "x86-64-linux-sse41-fma"}},
1112 {{"x86-64-linux-avx2-sse41", "x86-64-linux-sse41-fma", "x86-64-linux-sse41"}},
1113 {{"x86-64-linux-avx2-sse41", "x86-32-linux-sse41-fma", ""}},
1114 {{"x86-64-linux-cuda", "x86-64-linux", "x86-64-linux-cuda"}},
1115 {{"x86-64-linux-cuda-cuda_capability_50", "x86-64-linux-cuda", "x86-64-linux-cuda"}},
1116 {{"x86-64-linux-cuda-cuda_capability_50", "x86-64-linux-cuda-cuda_capability_30", "x86-64-linux-cuda-cuda_capability_30"}},
1117 {{"hexagon-32-qurt-hvx_v65", "hexagon-32-qurt-hvx_v62", "hexagon-32-qurt-hvx_v62"}},
1118 {{"hexagon-32-qurt-hvx_v62", "hexagon-32-qurt", "hexagon-32-qurt"}},
1119 {{"hexagon-32-qurt-hvx_v62-hvx", "hexagon-32-qurt", ""}},
1120 {{"hexagon-32-qurt-hvx_v62-hvx", "hexagon-32-qurt-hvx", "hexagon-32-qurt-hvx"}},
1121 };
1122
1123 for (const auto &test : gcd_tests) {
1124 Target result{};
1125 Target a{test[0]};
1126 Target b{test[1]};
1127 if (a.get_runtime_compatible_target(b, result)) {
1128 internal_assert(!test[2].empty() && result == Target{test[2]})
1129 << "Targets " << a.to_string() << " and " << b.to_string() << " were computed to have gcd "
1130 << result.to_string() << " but expected '" << test[2] << "'\n";
1131 } else {
1132 internal_assert(test[2].empty())
1133 << "Targets " << a.to_string() << " and " << b.to_string() << " were computed to have no gcd "
1134 << "but " << test[2] << " was expected.";
1135 }
1136 }
1137
1138 std::cout << "Target test passed" << std::endl;
1139}
