33 InjectProfiling(const string &pipeline_name)
34 : pipeline_name(pipeline_name) {
35 stack.push_back(get_func_id("overhead"));
36 // ID 0 is treated specially in the runtime as overhead
37 internal_assert(stack.back() == 0);
38
39 malloc_id = get_func_id("halide_malloc");
40 free_id = get_func_id("halide_free");
41 profiler_pipeline_state = Variable::make(Handle(), "profiler_pipeline_state");
42 profiler_state = Variable::make(Handle(), "profiler_state");
43 profiler_token = Variable::make(Int(32), "profiler_token");

66 // Strip down the tuple name, e.g. f.0 into f
67 string normalize_name(const string &name) {
68 vector<string> v = split_string(name, ".");
69 internal_assert(!v.empty());
70 return v[0];
71 }

73 int get_func_id(const string &name) {
74 string norm_name = normalize_name(name);
75 int idx = -1;
76 map<string, int>::iterator iter = indices.find(norm_name);
77 if (iter == indices.end()) {
78 idx = (int)indices.size();
79 indices[norm_name] = idx;
80 } else {
81 idx = iter->second;
82 }
83 return idx;
84 }

86 Stmt set_current_func(const Expr &id) {
87 // This call gets inlined and becomes a single store instruction.
88 return Evaluate::make(Call::make(Int(32), "halide_profiler_set_current_func",
89 {profiler_state, profiler_token, id}, Call::Extern));

92 Expr compute_allocation_size(const vector<Expr> &extents,
93 const Expr &condition,
94 const Type &type,
95 const std::string &name,
96 bool &on_stack) {
97 on_stack = true;
98
99 Expr cond = simplify(condition);
100 if (is_const_zero(cond)) { // Condition always false
101 return make_zero(UInt(64));
102 }
103
104 int64_t constant_size = Allocate::constant_allocation_size(extents, name);
105 if (constant_size > 0) {
106 int64_t stack_bytes = constant_size * type.bytes();
107 if (can_allocation_fit_on_stack(stack_bytes)) { // Allocation on stack
108 return make_const(UInt(64), stack_bytes);
109 }
110 }
111
112 // Check that the allocation is not scalar (if it were scalar
113 // it would have constant size).
114 internal_assert(!extents.empty());
115
116 on_stack = false;
117 Expr size = cast<uint64_t>(extents[0]);
118 for (size_t i = 1; i < extents.size(); i++) {
119 size *= extents[i];
120 }
121 size = simplify(Select::make(condition, size * type.bytes(), make_zero(UInt(64))));
122 return size;
123 }
124
125 Stmt visit(const Allocate *op) override {
126 int idx = get_func_id(op->name);
127
128 vector<Expr> new_extents;
129 bool all_extents_unmodified = true;
130 for (size_t i = 0; i < op->extents.size(); i++) {
131 new_extents.push_back(mutate(op->extents[i]));
132 all_extents_unmodified &= new_extents[i].same_as(op->extents[i]);
133 }
134 Expr condition = mutate(op->condition);
135
136 bool on_stack;
137 Expr size = compute_allocation_size(new_extents, condition, op->type, op->name, on_stack);
138 internal_assert(size.type() == UInt(64));
139 func_alloc_sizes.push(op->name, {on_stack, size});
140
141 // compute_allocation_size() might return a zero size, if the allocation is
142 // always conditionally false. remove_dead_allocations() is called after
143 // inject_profiling() so this is a possible scenario.
144 if (!is_const_zero(size) && on_stack) {
145 const uint64_t *int_size = as_const_uint(size);
146 internal_assert(int_size != nullptr); // Stack size is always a const int
147 func_stack_current[idx] += *int_size;
148 func_stack_peak[idx] = std::max(func_stack_peak[idx], func_stack_current[idx]);
149 debug(3) << " Allocation on stack: " << op->name
150 << "(" << size << ") in pipeline " << pipeline_name
151 << "; current: " << func_stack_current[idx]
152 << "; peak: " << func_stack_peak[idx] << "\n";
153 }
154
155 Stmt body = mutate(op->body);
156 Expr new_expr;
157 Stmt stmt;
158 if (op->new_expr.defined()) {
159 new_expr = mutate(op->new_expr);
160 }
161 if (all_extents_unmodified &&
162 body.same_as(op->body) &&
163 condition.same_as(op->condition) &&
164 new_expr.same_as(op->new_expr)) {
165 stmt = op;
166 } else {
167 stmt = Allocate::make(op->name, op->type, op->memory_type,
168 new_extents, condition, body, new_expr, op->free_function);
169 }
170
171 if (!is_const_zero(size) && !on_stack && profiling_memory) {
172 debug(3) << " Allocation on heap: " << op->name
173 << "(" << size << ") in pipeline "
174 << pipeline_name << "\n";
175
176 vector<Stmt> tasks{
177 set_current_func(malloc_id),
178 Evaluate::make(Call::make(Int(32), "halide_profiler_memory_allocate",
179 {profiler_pipeline_state, idx, size}, Call::Extern)),
180 stmt,
181 set_current_func(stack.back())};
182
183 stmt = Block::make(tasks);
184 }
185
186 return stmt;
187 }
188
189 Stmt visit(const Free *op) override {
190 int idx = get_func_id(op->name);
191
192 AllocSize alloc = func_alloc_sizes.get(op->name);
193 internal_assert(alloc.size.type() == UInt(64));
194 func_alloc_sizes.pop(op->name);
195
196 Stmt stmt = IRMutator::visit(op);
197
198 if (!is_const_zero(alloc.size)) {
199 if (!alloc.on_stack) {
200 if (profiling_memory) {
201 debug(3) << " Free on heap: " << op->name << "(" << alloc.size << ") in pipeline " << pipeline_name << "\n";
202
203 vector<Stmt> tasks{
204 set_current_func(free_id),
205 Evaluate::make(Call::make(Int(32), "halide_profiler_memory_free",
206 {profiler_pipeline_state, idx, alloc.size}, Call::Extern)),
207 stmt,
208 set_current_func(stack.back())};
209
210 stmt = Block::make(tasks);
211 }
212 } else {
213 const uint64_t *int_size = as_const_uint(alloc.size);
214 internal_assert(int_size != nullptr);
215
216 func_stack_current[idx] -= *int_size;
217 debug(3) << " Free on stack: " << op->name << "(" << alloc.size << ") in pipeline " << pipeline_name
218 << "; current: " << func_stack_current[idx] << "; peak: " << func_stack_peak[idx] << "\n";
219 }
220 }
221 return stmt;
222 }

224 Stmt visit(const ProducerConsumer *op) override {
225 int idx;
226 Stmt body;
227 if (op->is_producer) {
228 idx = get_func_id(op->name);
229 stack.push_back(idx);
230 body = mutate(op->body);
231 stack.pop_back();
232 } else {
233 body = mutate(op->body);
234 // At the beginning of the consume step, set the current task
235 // back to the outer one.
236 idx = stack.back();
237 }
238
239 body = Block::make(set_current_func(idx), body);
240
241 return ProducerConsumer::make(op->name, op->is_producer, body);
242 }

254 Stmt visit_parallel_task(const Stmt &s) {
255 if (const Fork *f = s.as<Fork>()) {
256 return Fork::make(visit_parallel_task(f->first), visit_parallel_task(f->rest));
257 } else if (const Acquire *a = s.as<Acquire>()) {
258 return Acquire::make(a->semaphore, a->count, visit_parallel_task(a->body));
259 } else {
260 return Block::make({incr_active_threads(), mutate(s), decr_active_threads()});
261 }
262 }
263
264 Stmt visit(const Acquire *op) override {
265 Stmt s = visit_parallel_task(op);
266 return Block::make({decr_active_threads(), s, incr_active_threads()});
267 }
268
269 Stmt visit(const Fork *op) override {
270 Stmt s = visit_parallel_task(op);
271 return Block::make({decr_active_threads(), s, incr_active_threads()});
272 }
273
274 Stmt visit(const For *op) override {
275 Stmt body = op->body;
276
277 // The for loop indicates a device transition or a
278 // parallel job launch. Decrement the number of active
279 // threads outside the loop, and increment it inside the
280 // body.
281 bool update_active_threads = (op->device_api == DeviceAPI::Hexagon ||
282 op->is_unordered_parallel());
283
284 if (update_active_threads) {
285 body = Block::make({incr_active_threads(), body, decr_active_threads()});
286 }
287
288 // We profile by storing a token to global memory, so don't enter GPU loops
289 if (op->device_api == DeviceAPI::Hexagon) {
290 // TODO: This is for all offload targets that support
291 // limited internal profiling, which is currently just
292 // hexagon. We don't support per-func stats remotely,
293 // which means we can't do memory accounting.
294 bool old_profiling_memory = profiling_memory;
295 profiling_memory = false;
296 body = mutate(body);
297 profiling_memory = old_profiling_memory;
298
299 // Get the profiler state pointer from scratch inside the
300 // kernel. There will be a separate copy of the state on
301 // the DSP that the host side will periodically query.
302 Expr get_state = Call::make(Handle(), "halide_profiler_get_state", {}, Call::Extern);
303 body = substitute("profiler_state", Variable::make(Handle(), "hvx_profiler_state"), body);
304 body = LetStmt::make("hvx_profiler_state", get_state, body);
305 } else if (op->device_api == DeviceAPI::None ||
306 op->device_api == DeviceAPI::Host) {
307 body = mutate(body);
308 } else {
309 body = op->body;
310 }
311
312 Stmt stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
313
314 if (update_active_threads) {
315 stmt = Block::make({decr_active_threads(), stmt, incr_active_threads()});
316 }
317 return stmt;
318 }

323Stmt inject_profiling(Stmt s, const string &pipeline_name) {
324 InjectProfiling profiling(pipeline_name);
325 s = profiling.mutate(s);
326
327 int num_funcs = (int)(profiling.indices.size());
328
329 Expr func_names_buf = Variable::make(Handle(), "profiling_func_names");
330
331 Expr start_profiler = Call::make(Int(32), "halide_profiler_pipeline_start",
332 {pipeline_name, num_funcs, func_names_buf}, Call::Extern);
333
334 Expr get_state = Call::make(Handle(), "halide_profiler_get_state", {}, Call::Extern);
335
336 Expr get_pipeline_state = Call::make(Handle(), "halide_profiler_get_pipeline_state", {pipeline_name}, Call::Extern);
337
338 Expr profiler_token = Variable::make(Int(32), "profiler_token");
339
340 Expr stop_profiler = Call::make(Handle(), Call::register_destructor,
341 {Expr("halide_profiler_pipeline_end"), get_state}, Call::Intrinsic);
342
343 bool no_stack_alloc = profiling.func_stack_peak.empty();
344 if (!no_stack_alloc) {
345 Expr func_stack_peak_buf = Variable::make(Handle(), "profiling_func_stack_peak_buf");
346
347 Expr profiler_pipeline_state = Variable::make(Handle(), "profiler_pipeline_state");
348 Stmt update_stack = Evaluate::make(Call::make(Int(32), "halide_profiler_stack_peak_update",
349 {profiler_pipeline_state, func_stack_peak_buf}, Call::Extern));
350 s = Block::make(update_stack, s);
351 }
352
353 Expr profiler_state = Variable::make(Handle(), "profiler_state");
354 Stmt incr_active_threads =
355 Evaluate::make(Call::make(Int(32), "halide_profiler_incr_active_threads",
356 {profiler_state}, Call::Extern));
357 Stmt decr_active_threads =
358 Evaluate::make(Call::make(Int(32), "halide_profiler_decr_active_threads",
359 {profiler_state}, Call::Extern));
360 s = Block::make({incr_active_threads, s, decr_active_threads});
361
362 s = LetStmt::make("profiler_pipeline_state", get_pipeline_state, s);
363 s = LetStmt::make("profiler_state", get_state, s);
364 // If there was a problem starting the profiler, it will call an
365 // appropriate halide error function and then return the
366 // (negative) error code as the token.
367 s = Block::make(AssertStmt::make(profiler_token >= 0, profiler_token), s);
368 s = LetStmt::make("profiler_token", start_profiler, s);
369
370 if (!no_stack_alloc) {
371 for (int i = num_funcs - 1; i >= 0; --i) {
372 s = Block::make(Store::make("profiling_func_stack_peak_buf",
373 make_const(UInt(64), profiling.func_stack_peak[i]),
374 i, Parameter(), const_true(), ModulusRemainder()),
375 s);
376 }
377 s = Block::make(s, Free::make("profiling_func_stack_peak_buf"));
378 s = Allocate::make("profiling_func_stack_peak_buf", UInt(64),
379 MemoryType::Auto, {num_funcs}, const_true(), s);
380 }
381
382 for (const auto &p : profiling.indices) {
383 s = Block::make(Store::make("profiling_func_names", p.first, p.second, Parameter(), const_true(), ModulusRemainder()), s);
384 }
385
386 s = Block::make(s, Free::make("profiling_func_names"));
387 s = Allocate::make("profiling_func_names", Handle(),
388 MemoryType::Auto, {num_funcs}, const_true(), s);
389 s = Block::make(Evaluate::make(stop_profiler), s);
390
391 return s;
392}
