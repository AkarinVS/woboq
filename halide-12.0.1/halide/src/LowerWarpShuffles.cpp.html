14// In CUDA, allocations stored in registers and shared across lanes
15// look like private per-lane allocations, even though communication
16// across lanes is possible. So while we model then as allocations
17// outside the loop over lanes, we need to codegen them as allocations
18// inside the loop over lanes. So the lanes collectively share
19// responsibility for storing the allocation. We will stripe the
20// storage across the lanes (think RAID 0). This is basically the
21// opposite of RewriteAccessToVectorAlloc in Vectorize.cpp.
22//
23// If there were no constraints, we could just arbitrarily slice
24// things up, e.g. on a per-element basis (stride one), but we have
25// the added wrinkle that while threads can load from anywhere, they
26// can only store into their own stripe, so we need to analyze the
27// existing stores in order to figure out a striping that corresponds
28// to the stores taking place. In fact, a common pattern is having
29// lanes responsible for an adjacent pair of values, which gives us a
30// stride of two.
31//
32// This lowering pass determines a good stride for each allocation,
33// then moves the allocation inside the loop over lanes. Loads and
34// stores have their indices rewritten to reflect the striping, and
35// loads from outside a lane's own stripe become warp shuffle
36// intrinsics. Finally, warp shuffles must be hoisted outside of
37// conditionals, because they return undefined values if either the
38// source or destination lanes are inactive.

49// Try to reduce all terms in an affine expression modulo a given
50// modulus, making as many simplifications as possible. Used for
51// eliminating terms from nested affine expressions. This is much more
52// aggressive about eliminating terms than using % and then
53// calling the simplifier.
54Expr reduce_expr_helper(Expr e, const Expr &modulus) {
55 if (is_const_one(modulus)) {
56 return make_zero(e.type());
57 } else if (is_const(e)) {
58 return simplify(e % modulus);
59 } else if (const Add *add = e.as<Add>()) {
60 return (reduce_expr_helper(add->a, modulus) + reduce_expr_helper(add->b, modulus));
61 } else if (const Sub *sub = e.as<Sub>()) {
62 return (reduce_expr_helper(sub->a, modulus) - reduce_expr_helper(sub->b, modulus));
63 } else if (const Mul *mul = e.as<Mul>()) {
64 if (is_const(mul->b) && can_prove(modulus % mul->b == 0)) {
65 return reduce_expr_helper(mul->a, simplify(modulus / mul->b)) * mul->b;
66 } else {
67 return reduce_expr_helper(mul->a, modulus) * reduce_expr_helper(mul->b, modulus);
68 }
69 } else if (const Ramp *ramp = e.as<Ramp>()) {
70 return Ramp::make(reduce_expr_helper(ramp->base, modulus), reduce_expr_helper(ramp->stride, modulus), ramp->lanes);
71 } else if (const Broadcast *b = e.as<Broadcast>()) {
72 return Broadcast::make(reduce_expr_helper(b->value, modulus), b->lanes);
73 } else {
74 return e;
75 }
76}
77
78Expr reduce_expr(Expr e, const Expr &modulus, const Scope<Interval> &bounds) {
79 e = reduce_expr_helper(simplify(e, true, bounds), modulus);
80 if (is_const_one(simplify(e >= 0 && e < modulus, true, bounds))) {
81 return e;
82 } else {
83 return e % modulus;
84 }

126// Determine a good striping stride for an allocation, by inspecting
127// loads and stores.
128class DetermineAllocStride : public IRVisitor {
129
130 using IRVisitor::visit;
131
132 const string &alloc, &lane_var;
133 Expr warp_size;
134 bool single_thread = false;
135 vector<Expr> loads, stores, single_stores;
136
137 // The derivatives of all the variables in scope w.r.t the
138 // lane_var. If something isn't in this scope, the derivative can
139 // be assumed to be zero.
140 Scope<Expr> dependent_vars;
141
142 Scope<Interval> bounds;
143
144 // Get the derivative of an integer expression w.r.t the warp
145 // lane. Returns an undefined Expr if the result is non-trivial.
146 Expr warp_stride(const Expr &e) {
147 if (is_const(e)) {
148 return 0;
149 } else if (const Variable *var = e.as<Variable>()) {
150 if (var->name == lane_var) {
151 return 1;
152 } else if (dependent_vars.contains(var->name)) {
153 return dependent_vars.get(var->name);
154 } else {
155 return 0;
156 }
157 } else if (const Add *add = e.as<Add>()) {
158 Expr sa = warp_stride(add->a), sb = warp_stride(add->b);
159 if (sa.defined() && sb.defined()) {
160 return sa + sb;
161 }
162 } else if (const Sub *sub = e.as<Sub>()) {
163 Expr sa = warp_stride(sub->a), sb = warp_stride(sub->b);
164 if (sa.defined() && sb.defined()) {
165 return sa - sb;
166 }
167 } else if (const Mul *mul = e.as<Mul>()) {
168 Expr sa = warp_stride(mul->a), sb = warp_stride(mul->b);
169 if (sa.defined() && sb.defined() && is_const_zero(sb)) {
170 return sa * mul->b;
171 }
172 } else if (const Broadcast *b = e.as<Broadcast>()) {
173 return warp_stride(b->value);
174 } else if (const Ramp *r = e.as<Ramp>()) {
175 Expr sb = warp_stride(r->base);
176 Expr ss = warp_stride(r->stride);
177 if (sb.defined() && ss.defined() && is_const_zero(ss)) {
178 return sb;
179 }
180 } else if (const Let *let = e.as<Let>()) {
181 ScopedBinding<Expr> bind(dependent_vars, let->name, warp_stride(let->value));
182 return warp_stride(let->body);
183 } else if (!expr_uses_vars(e, dependent_vars)) {
184 return 0;
185 }
186
187 return Expr();
188 }
189
190 void visit(const Let *op) override {
191 ScopedBinding<Expr> bind(dependent_vars, op->name, warp_stride(op->value));
192 IRVisitor::visit(op);
193 }

218 void visit(const IfThenElse *op) override {
219 // When things drop down to a single thread, we have different
220 // constraints, so notice that. Check if the condition implies
221 // the lane var is at most one.
222 if (can_prove(!op->condition || Variable::make(Int(32), lane_var) <= 1)) {
223 bool old_single_thread = single_thread;
224 single_thread = true;
225 op->then_case.accept(this);
226 single_thread = old_single_thread;
227 if (op->else_case.defined()) {
228 op->else_case.accept(this);
229 }
230 } else {
231 IRVisitor::visit(op);
232 }

236 ScopedBinding<Interval>
237 bind_bounds_if(is_const(op->min) && is_const(op->extent),
238 bounds, op->name, Interval(op->min, simplify(op->min + op->extent - 1)));
239 ScopedBinding<Expr>

241 expr_uses_vars(op->extent, dependent_vars)),

246 void fail(const vector<Expr> &bad) {
247 std::ostringstream message;
248 message
249 << "Access pattern for " << alloc << " does not meet the requirements for its store_at location. "
250 << "All access to an allocation scheduled inside a loop over GPU "
251 << "threads and outside a loop over GPU lanes must obey the following constraint:\n"
252 << "The index must be affine in " << lane_var << " with a consistent linear "
253 << "term across all stores, and a constant term which, when divided by the stride "
254 << "(rounding down), becomes a multiple of the warp size (" << warp_size << ").\n";
255 if (!stores.empty()) {
256 message << alloc << " is stored to at the following indices by multiple lanes:\n";
257 for (const Expr &e : stores) {
258 message << " " << e << "\n";
259 }
260 }
261 if (!single_stores.empty()) {
262 message << "And the following indicies by lane zero:\n";
263 for (const Expr &e : single_stores) {
264 message << " " << e << "\n";
265 }
266 }
267 if (!loads.empty()) {
268 message << "And loaded from at the following indices:\n";
269 for (const Expr &e : loads) {
270 message << " " << e << "\n";
271 }
272 }
273 message << "The problematic indices are:\n";
274 for (const Expr &e : bad) {
275 message << " " << e << "\n";
276 }
277 user_error << message.str();
278 }

280public:
281 DetermineAllocStride(const string &alloc, const string &lane_var, const Expr &warp_size)
282 : alloc(alloc), lane_var(lane_var), warp_size(warp_size) {
283 dependent_vars.push(lane_var, 1);
284 }
285
286 // A version of can_prove which exploits the constant bounds we've been tracking
287 bool can_prove(const Expr &e) {
288 return is_const_one(simplify(e, true, bounds));
289 }
290
291 Expr get_stride() {
292 bool ok = true;
293 Expr stride;
294 Expr var = Variable::make(Int(32), lane_var);
295 vector<Expr> bad;
296 for (const Expr &e : stores) {
297 Expr s = warp_stride(e);
298 if (s.defined()) {
299 // Constant-fold
300 s = simplify(s);
301 }
302 if (!stride.defined()) {
303 stride = s;
304 }
305
306 // Check the striping pattern of this store corresponds to
307 // any already discovered on previous stores.
308 bool this_ok = (s.defined() &&
309 (can_prove(stride == s) &&
310 can_prove(reduce_expr(e / stride - var, warp_size, bounds) == 0)));
311
312 internal_assert(stride.defined());
313
314 if (!this_ok) {
315 bad.push_back(e);
316 }
317 ok = ok && this_ok;
318 }
319
320 for (const Expr &e : loads) {
321 // We can handle any access pattern for loads, but it's
322 // better if the stride matches up because then it's just
323 // a register access, not a warp shuffle.
324 Expr s = warp_stride(e);
325 if (!stride.defined()) {
326 stride = s;
327 }
328 }
329
330 if (stride.defined()) {
331 for (const Expr &e : single_stores) {
332 // If only thread zero was active for the store, that makes the proof simpler.
333 Expr simpler = substitute(lane_var, 0, e);
334 bool this_ok = can_prove(reduce_expr(simpler / stride, warp_size, bounds) == 0);
335 if (!this_ok) {
336 bad.push_back(e);
337 }
338 ok = ok && this_ok;
339 }
340 }
341
342 if (!ok) {
343 fail(bad);
344 }
345
346 if (!stride.defined()) {
347 // This allocation must only accessed via single-threaded stores.
348 stride = 1;
349 }
350
351 return stride;
352 }

355// Move allocations outside the loop over lanes into the loop over
356// lanes (using the striping described above), and rewrites
357// stores/loads to them as cuda register shuffle intrinsics.
358class LowerWarpShuffles : public IRMutator {
359 using IRMutator::visit;
360
361 Expr warp_size, this_lane;
362 string this_lane_name;
363 bool may_use_warp_shuffle;
364 vector<Stmt> allocations;
365 struct AllocInfo {
366 int size;
367 Expr stride;
368 };
369 Scope<AllocInfo> allocation_info;
370 Scope<Interval> bounds;
371
372 Stmt visit(const For *op) override {
373 ScopedBinding<Interval>
374 bind_if(is_const(op->min) && is_const(op->extent),
375 bounds, op->name, Interval(op->min, simplify(op->min + op->extent - 1)));
376 if (!this_lane.defined() && op->for_type == ForType::GPULane) {
377
378 bool should_mask = false;
379 ScopedValue<Expr> old_warp_size(warp_size);
380 if (op->for_type == ForType::GPULane) {
381 const int64_t *loop_size = as_const_int(op->extent);
382 user_assert(loop_size && *loop_size <= 32)
383 << "CUDA gpu lanes loop must have constant extent of at most 32: " << op->extent << "\n";
384
385 // Select a warp size - the smallest power of two that contains the loop size
386 int64_t ws = 1;
387 while (ws < *loop_size) {
388 ws *= 2;
389 }
390 should_mask = (ws != *loop_size);
391 warp_size = make_const(Int(32), ws);
392 } else {
393 warp_size = op->extent;
394 }
395 this_lane_name = op->name;
396 this_lane = Variable::make(Int(32), op->name);
397 may_use_warp_shuffle = (op->for_type == ForType::GPULane);
398
399 Stmt body = op->body;
400
401 // Figure out the shrunken size of the hoisted allocations
402 // and populate the scope.
403 for (const Stmt &s : allocations) {
404 const Allocate *alloc = s.as<Allocate>();
405 internal_assert(alloc && alloc->extents.size() == 1);
406 // The allocation has been moved into the lane loop,
407 // with storage striped across the warp lanes, so the
408 // size required per-lane is the old size divided by
409 // the number of lanes (rounded up).
410 Expr new_size = (alloc->extents[0] + op->extent - 1) / op->extent;
411 new_size = simplify(new_size, true, bounds);
412 new_size = find_constant_bound(new_size, Direction::Upper, bounds);
413 const int64_t *sz = as_const_int(new_size);
414 user_assert(sz) << "Warp-level allocation with non-constant size: "
415 << alloc->extents[0] << ". Use Func::bound_extent.";
416 DetermineAllocStride stride(alloc->name, op->name, warp_size);
417 body.accept(&stride);
418 allocation_info.push(alloc->name, {(int)(*sz), stride.get_stride()});
419 }
420
421 body = mutate(op->body);
422
423 if (should_mask) {
424 // Mask off the excess lanes in the warp
425 body = IfThenElse::make(this_lane < op->extent, body, Stmt());
426 }
427
428 // Wrap the hoisted warp-level allocations, at their new
429 // reduced size.
430 for (const Stmt &s : allocations) {
431 const Allocate *alloc = s.as<Allocate>();
432 internal_assert(alloc && alloc->extents.size() == 1);
433 int new_size = allocation_info.get(alloc->name).size;
434 allocation_info.pop(alloc->name);
435 body = Allocate::make(alloc->name, alloc->type, alloc->memory_type,
436 {new_size}, alloc->condition,
437 body, alloc->new_expr, alloc->free_function);
438 }
439 allocations.clear();
440
441 this_lane = Expr();
442 this_lane_name.clear();
443 may_use_warp_shuffle = false;
444
445 // Mutate the body once more to apply the same transformation to any inner loops
446 body = mutate(body);
447
448 // Rewrap any hoisted allocations that weren't placed outside some inner loop
449 for (const Stmt &s : allocations) {
450 const Allocate *alloc = s.as<Allocate>();
451 body = Allocate::make(alloc->name, alloc->type, alloc->memory_type,
452 alloc->extents, alloc->condition,
453 body, alloc->new_expr, alloc->free_function);
454 }
455 allocations.clear();
456
457 return For::make(op->name, op->min, warp_size,
458 op->for_type, op->device_api, body);
459 } else {
460 return IRMutator::visit(op);
461 }
462 }
463
464 Stmt visit(const IfThenElse *op) override {
465 // Consider lane-masking if-then-elses when determining the
466 // active bounds of the lane index.
467 //
468 // FuseGPULoopNests injects conditionals of the form lane <
469 // limit_val when portions parts of the kernel to certain
470 // threads, so we need to match that pattern. Things that come
471 // from GuardWithIf can also inject <=.
472 const LT *lt = op->condition.as<LT>();
473 const LE *le = op->condition.as<LE>();
474 if ((lt && equal(lt->a, this_lane) && is_const(lt->b)) ||
475 (le && equal(le->a, this_lane) && is_const(le->b))) {
476 Expr condition = mutate(op->condition);
477 internal_assert(bounds.contains(this_lane_name));
478 Interval interval = bounds.get(this_lane_name);
479 interval.max = lt ? simplify(lt->b - 1) : le->b;
480 ScopedBinding<Interval> bind(bounds, this_lane_name, interval);
481 Stmt then_case = mutate(op->then_case);
482 Stmt else_case = mutate(op->else_case);
483 return IfThenElse::make(condition, then_case, else_case);
484 } else {
485 return IRMutator::visit(op);
486 }
487 }
488
489 Stmt visit(const Store *op) override {
490 if (allocation_info.contains(op->name)) {
491 Expr idx = mutate(op->index);
492 Expr value = mutate(op->value);
493 Expr stride = allocation_info.get(op->name).stride;
494 internal_assert(stride.defined() && warp_size.defined());
495
496 // Reduce the index to an index in my own stripe. We have
497 // already validated the legality of this in
498 // DetermineAllocStride. We split the flat index into into
499 // a three-dimensional index using warp_size and
500 // stride. The innermost dimension is at most the stride,
501 // and is in the index within one contiguous chunk stored
502 // by a lane. The middle dimension corresponds to
503 // lanes. It's the one we're striping across, so it should
504 // be eliminated. The outermost dimension is whatever bits
505 // are left over. If everything is a power of two, you can
506 // think of this as erasing some of the bits in the middle
507 // of the index and shifting the high bits down to cover
508 // them. Reassembling the result into a flat address gives
509 // the expression below.
510 Expr in_warp_idx = simplify((idx / (warp_size * stride)) * stride + reduce_expr(idx, stride, bounds), true, bounds);
511 return Store::make(op->name, value, in_warp_idx, op->param, op->predicate, ModulusRemainder());
512 } else {
513 return IRMutator::visit(op);
514 }
515 }

517 Expr make_warp_load(Type type, const string &name, const Expr &idx, Expr lane) {
518 // idx: The index of the value within the local allocation
519 // lane: Which thread's value we want. If it's our own, we can just use a load.
520
521 // Do the equivalent load, and then ask for another lane's
522 // value of the result. For this to work idx
523 // must not depend on the thread ID.
524
525 // We handle other cases by converting it to a select tree
526 // that muxes between all possible values.
527
528 if (expr_uses_var(idx, this_lane_name)) {
529 Expr equiv = make_warp_load(type, name, make_zero(idx.type()), lane);
530 int elems = allocation_info.get(name).size;
531 for (int i = 1; i < elems; i++) {
532 // Load the right lanes from stripe number i
533 equiv = select(idx >= i, make_warp_load(type, name, make_const(idx.type(), i), lane), equiv);
534 }
535 return simplify(equiv, true, bounds);
536 }
537
538 // Load the value to be shuffled
539 Expr base_val = Load::make(type, name, idx, Buffer<>(),
540 Parameter(), const_true(idx.type().lanes()), ModulusRemainder());
541
542 Expr scalar_lane = lane;
543 if (const Broadcast *b = scalar_lane.as<Broadcast>()) {
544 scalar_lane = b->value;
545 }
546 if (equal(scalar_lane, this_lane)) {
547 // This is a regular load. No shuffling required.
548 return base_val;
549 }
550
551 // Make 32-bit with a combination of reinterprets and zero extension
552 Type shuffle_type = type;
553 if (type.bits() < 32) {
554 shuffle_type = UInt(32, type.lanes());
555 base_val = cast(shuffle_type, reinterpret(type.with_code(Type::UInt), base_val));
556 } else if (type.bits() == 64) {
557 // TODO: separate shuffles of the low and high halves and then recombine.
558 user_error << "Warp shuffles of 64-bit types not yet implemented\n";
559 } else {
560 user_assert(type.bits() == 32) << "Warp shuffles not supported for this type: " << type << "\n";
561 }
562
563 internal_assert(may_use_warp_shuffle) << name << ", " << idx << ", " << lane << "\n";
564
565 string intrin_suffix;
566 if (shuffle_type.is_float()) {
567 intrin_suffix = ".f32";
568 } else {
569 intrin_suffix = ".i32";
570 }
571
572 Expr wild = Variable::make(Int(32), "*");
573 vector<Expr> result;
574 int bits = 0;
575
576 // Move this_lane as far left as possible in the expression to
577 // reduce the number of cases to check below.
578 lane = solve_expression(lane, this_lane_name).result;
579
580 Expr shuffled;
581
582 if (expr_match(this_lane + wild, lane, result)) {
583 // We know that 0 <= lane + wild < warp_size by how we
584 // constructed it, so we can just do a shuffle down.
585 Expr down = Call::make(shuffle_type, "llvm.nvvm.shfl.down" + intrin_suffix,
586 {base_val, result[0], 31}, Call::PureExtern);
587 shuffled = down;
588 } else if (expr_match((this_lane + wild) % wild, lane, result) &&
589 is_const_power_of_two_integer(result[1], &bits) &&
590 bits <= 5) {
591 result[0] = simplify(result[0] % result[1], true, bounds);
592 // Rotate. Mux a shuffle up and a shuffle down. Uses fewer
593 // intermediate registers than using a general gather for
594 // this.
595 Expr mask = (1 << bits) - 1;
596 Expr down = Call::make(shuffle_type, "llvm.nvvm.shfl.down" + intrin_suffix,
597 {base_val, result[0], mask}, Call::PureExtern);
598 Expr up = Call::make(shuffle_type, "llvm.nvvm.shfl.up" + intrin_suffix,
599 {base_val, (1 << bits) - result[0], 0}, Call::PureExtern);
600 Expr cond = (this_lane >= (1 << bits) - result[0]);
601 Expr equiv = select(cond, up, down);
602 shuffled = simplify(equiv, true, bounds);
603 } else {
604 // The format of the mask is a pain. The high bits tell
605 // you how large the a warp is for this instruction
606 // (i.e. is it a shuffle within groups of 8, or a shuffle
607 // within groups of 16?). The low bits serve as a clamp on
608 // the max value pulled from. We don't use that, but it
609 // could hypothetically be used for boundary conditions.
610 Expr mask = simplify(((31 & ~(warp_size - 1)) << 8) | 31);
611 // The idx variant can do a general gather. Use it for all other cases.
612 shuffled = Call::make(shuffle_type, "llvm.nvvm.shfl.idx" + intrin_suffix,
613 {base_val, lane, mask}, Call::PureExtern);
614 }
615 // TODO: There are other forms, like butterfly and clamp, that
616 // don't need to use the general gather
617
618 if (shuffled.type() != type) {
619 user_assert(shuffled.type().bits() > type.bits());
620 // Narrow it back down
621 shuffled = reinterpret(type, cast(type.with_code(Type::UInt), shuffled));
622 }
623 return shuffled;
624 }

626 Expr visit(const Load *op) override {
627 if (allocation_info.contains(op->name)) {
628 Expr idx = mutate(op->index);
629 Expr stride = allocation_info.get(op->name).stride;
630
631 // Break the index into lane and stripe components
632 Expr lane = simplify(reduce_expr(idx / stride, warp_size, bounds), true, bounds);
633 idx = simplify((idx / (warp_size * stride)) * stride + reduce_expr(idx, stride, bounds), true, bounds);
634 // We don't want the idx to depend on the lane var, so try to eliminate it
635 idx = simplify(solve_expression(idx, this_lane_name).result, true, bounds);
636 return make_warp_load(op->type, op->name, idx, lane);
637 } else {
638 return IRMutator::visit(op);
639 }
640 }
641
642 Stmt visit(const Allocate *op) override {
643 if (this_lane.defined() || op->memory_type == MemoryType::GPUShared) {
644 // Not a warp-level allocation
645 return IRMutator::visit(op);
646 } else {
647 // Pick up this allocation and deposit it inside the loop over lanes at reduced size.
648 allocations.emplace_back(op);
649 return mutate(op->body);
650 }
651 }
652
653public:
654 LowerWarpShuffles() = default;
655};

657class HoistWarpShufflesFromSingleIfStmt : public IRMutator {
658 using IRMutator::visit;
659
660 Scope<int> stored_to;
661 vector<pair<string, Expr>> lifted_lets;
662
663 Expr visit(const Call *op) override {
664 // If it was written outside this if clause but read inside of
665 // it, we need to hoist it.
666 if (starts_with(op->name, "llvm.nvvm.shfl.") &&
667 !expr_uses_vars(op, stored_to)) {
668 string name = unique_name('t');
669 lifted_lets.emplace_back(name, op);
670 return Variable::make(op->type, name);
671 } else {
672 return IRMutator::visit(op);
673 }
674 }
675
676 template<typename ExprOrStmt, typename LetOrLetStmt>
677 ExprOrStmt visit_let(const LetOrLetStmt *op) {
678 Expr value = mutate(op->value);
679 ExprOrStmt body = mutate(op->body);
680
681 // If any of the lifted expressions use this, we also need to
682 // lift this.
683 bool should_lift = false;
684 for (const auto &p : lifted_lets) {
685 should_lift |= expr_uses_var(p.second, op->name);
686 }
687
688 if (should_lift) {
689 lifted_lets.push_back({op->name, value});
690 return body;
691 } else {
692 return LetOrLetStmt::make(op->name, value, body);
693 }
694 }
695
696 Expr visit(const Let *op) override {
697 return visit_let<Expr>(op);
698 }
699
700 Stmt visit(const LetStmt *op) override {
701 return visit_let<Stmt>(op);
702 }
703
704 Stmt visit(const For *op) override {
705 Stmt body = mutate(op->body);
706 bool fail = false;
707 for (const auto &p : lifted_lets) {
708 fail |= expr_uses_var(p.second, op->name);
709 }
710 if (fail) {
711 // We can't hoist. We need to bail out here.
712 body = rewrap(body);
713 success = false;
714 } else {
715 debug(3) << "Successfully hoisted shuffle out of for loop\n";
716 }
717 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
718 }
719
720 Stmt visit(const Store *op) override {
721 stored_to.push(op->name, 0);
722 return IRMutator::visit(op);
723 }
724
725public:
726 bool success = true;
727
728 Stmt rewrap(Stmt s) {
729 while (!lifted_lets.empty()) {
730 const pair<string, Expr> &p = lifted_lets.back();
731 s = LetStmt::make(p.first, p.second, s);
732 lifted_lets.pop_back();
733 }
734 return s;
735 }
736};

739class MoveIfStatementInwards : public IRMutator {
740 using IRMutator::visit;

742 Stmt visit(const Store *op) override {

744 return IfThenElse::make(condition, op, Stmt());

755// The destination *and source* for warp shuffles must be active
756// threads, or the value is undefined, so we want to lift them out of
757// if statements.
758class HoistWarpShuffles : public IRMutator {
759 using IRMutator::visit;
760
761 Stmt visit(const IfThenElse *op) override {
762 // Move all Exprs that contain a shuffle out of the body of
763 // the if.
764 Stmt then_case = mutate(op->then_case);
765 Stmt else_case = mutate(op->else_case);
766
767 HoistWarpShufflesFromSingleIfStmt hoister;
768 then_case = hoister.mutate(then_case);
769 else_case = hoister.mutate(else_case);
770 Stmt s = IfThenElse::make(op->condition, then_case, else_case);
771 if (hoister.success) {
772 return hoister.rewrap(s);
773 } else {
774 // Need to move the ifstmt further inwards instead.
775 internal_assert(!else_case.defined()) << "Cannot hoist warp shuffle out of " << s << "\n";
776 string pred_name = unique_name('p');
777 s = MoveIfStatementInwards(Variable::make(op->condition.type(), pred_name)).mutate(then_case);
778 return LetStmt::make(pred_name, op->condition, s);
779 }
780 }
781};
