23/** Remove identity functions, even if they have side-effects. */
24class StripIdentities : public IRMutator {
25 using IRMutator::visit;
26
27 Expr visit(const Call *op) override {
28 if (Call::as_tag(op) || op->is_intrinsic(Call::return_second)) {
29 return mutate(op->args.back());
30 } else {
31 return IRMutator::visit(op);
32 }
33 }

36/** Check if an Expr loads from the given buffer. */
37class LoadsFromBuffer : public IRVisitor {
38 using IRVisitor::visit;
39
40 void visit(const Load *op) override {
41 if (op->name == buffer) {
42 result = true;
43 } else {
44 IRVisitor::visit(op);
45 }
46 }
47
48 string buffer;
49
50public:
51 bool result = false;
52 LoadsFromBuffer(const string &b)
53 : buffer(b) {
54 }
55};

57bool loads_from_buffer(const Expr &e, const string &buf) {
58 LoadsFromBuffer l(buf);
59 e.accept(&l);
60 return l.result;

63/** Construct a sufficient condition for the visited stmt to be a no-op. */
64class IsNoOp : public IRVisitor {
65 using IRVisitor::visit;
66
67 Expr make_and(Expr a, Expr b) {
68 if (is_const_zero(a) || is_const_one(b)) {
69 return a;
70 }
71 if (is_const_zero(b) || is_const_one(a)) {
72 return b;
73 }
74 return a && b;
75 }
76
77 Expr make_or(Expr a, Expr b) {
78 if (is_const_zero(a) || is_const_one(b)) {
79 return b;
80 }
81 if (is_const_zero(b) || is_const_one(a)) {
82 return a;
83 }
84 return a || b;
85 }
86
87 void visit(const Store *op) override {
88 if (op->value.type().is_handle() || is_const_zero(op->predicate)) {
89 condition = const_false();
90 } else {
91 if (is_const_zero(condition)) {
92 return;
93 }
94 // If the value being stored is the same as the value loaded,
95 // this is a no-op
96 debug(3) << "Considering store: " << Stmt(op) << "\n";
97
98 // Early-out: There's no way for that to be true if the
99 // RHS does not load from the buffer being stored to.
100 if (!loads_from_buffer(op->value, op->name)) {
101 condition = const_false();
102 return;
103 }
104
105 Expr equivalent_load = Load::make(op->value.type(), op->name, op->index,
106 Buffer<>(), Parameter(), op->predicate, op->alignment);
107 Expr is_no_op = equivalent_load == op->value;
108 is_no_op = StripIdentities().mutate(is_no_op);
109 // We need to call CSE since sometimes we have "let" stmt on the RHS
110 // that makes the expr harder to solve, i.e. the solver will just give up
111 // and return a conservative false on call to and_condition_over_domain().
112 is_no_op = simplify(common_subexpression_elimination(is_no_op));
113 debug(3) << "Anding condition over domain... " << is_no_op << "\n";
114 is_no_op = and_condition_over_domain(is_no_op, Scope<Interval>::empty_scope());
115 condition = make_and(condition, is_no_op);
116 debug(3) << "Condition is now " << condition << "\n";
117 }
118 }
119
120 void visit(const For *op) override {
121 if (is_const_zero(condition)) {
122 return;
123 }
124 Expr old_condition = condition;
125 condition = const_true();
126 op->body.accept(this);
127 Scope<Interval> varying;
128 varying.push(op->name, Interval(op->min, op->min + op->extent - 1));
129 condition = simplify(common_subexpression_elimination(condition));
130 debug(3) << "About to relax over " << op->name << " : " << condition << "\n";
131 condition = and_condition_over_domain(condition, varying);
132 debug(3) << "Relaxed: " << condition << "\n";
133 condition = make_and(old_condition, make_or(condition, simplify(op->extent <= 0)));
134 }
135
136 void visit(const IfThenElse *op) override {
137 if (is_const_zero(condition)) {
138 return;
139 }
140 Expr total_condition = condition;
141 condition = const_true();
142 op->then_case.accept(this);
143 // This is a no-op if we're previously a no-op, and the
144 // condition is false or the if body is a no-op.
145 total_condition = make_and(total_condition, make_or(!op->condition, condition));
146 condition = const_true();
147 if (op->else_case.defined()) {
148 op->else_case.accept(this);
149 total_condition = make_and(total_condition, make_or(op->condition, condition));
150 }
151 condition = total_condition;
152 }
153
154 void visit(const Call *op) override {
155 // If the loop calls an impure function, we can't remove the
156 // call to it. Most notably: image_store.
157 if (!op->is_pure()) {
158 condition = const_false();
159 return;
160 }
161 IRVisitor::visit(op);
162 }
163
164 void visit(const Acquire *op) override {
165 condition = const_false();
166 }
167
168 template<typename LetOrLetStmt>
169 void visit_let(const LetOrLetStmt *op) {
170 IRVisitor::visit(op);
171 if (expr_uses_var(condition, op->name)) {
172 condition = Let::make(op->name, op->value, condition);
173 }
174 }
175
176 void visit(const LetStmt *op) override {
177 visit_let(op);
178 }
179
180 void visit(const Let *op) override {
181 visit_let(op);
182 }
183
184public:
185 Expr condition = const_true();
186};

188class SimplifyUsingBounds : public IRMutator {
189 struct ContainingLoop {
190 string var;
191 Interval i;
192 };
193 vector<ContainingLoop> containing_loops;
194
195 using IRMutator::visit;
196
197 // Can we prove a condition over the non-rectangular domain of the for loops we're in?
198 bool provably_true_over_domain(Expr test) {
199 debug(3) << "Attempting to prove: " << test << "\n";
200 for (size_t i = containing_loops.size(); i > 0; i--) {
201 // Because the domain is potentially non-rectangular, we
202 // need to take each variable one-by-one, simplifying in
203 // between to allow for cancellations of the bounds of
204 // inner loops with outer loop variables.
205 auto loop = containing_loops[i - 1];
206 if (is_const(test)) {
207 break;
208 } else if (!expr_uses_var(test, loop.var)) {
209 continue;
210 } else if (loop.i.is_bounded() &&
211 can_prove(loop.i.min == loop.i.max) &&
212 expr_uses_var(test, loop.var)) {
213 // If min == max then either the domain only has one correct value, which we
214 // can substitute directly.
215 // Need to call CSE here since simplify() is sometimes unable to simplify expr with
216 // non-trivial 'let' value, e.g. (let x = min(10, y-1) in (x < y))
217 test = common_subexpression_elimination(Let::make(loop.var, loop.i.min, test));
218 } else if (loop.i.is_bounded() &&
219 can_prove(loop.i.min >= loop.i.max) &&
220 expr_uses_var(test, loop.var)) {
221 // If min >= max then either the domain only has one correct value,
222 // or the domain is empty, which implies both min/max are true under
223 // the domain.
224 // Need to call CSE here since simplify() is sometimes unable to simplify expr with
225 // non-trivial 'let' value, e.g. (let x = 10 in x < y) || (let x = min(10, y-1) in (x < y))
226 test = common_subexpression_elimination(Let::make(loop.var, loop.i.min, test) ||
227 Let::make(loop.var, loop.i.max, test));
228 } else {
229 Scope<Interval> s;
230 // Rearrange the expression if possible so that the
231 // loop var only occurs once.
232 SolverResult solved = solve_expression(test, loop.var);
233 if (solved.fully_solved) {
234 test = solved.result;
235 }
236 s.push(loop.var, loop.i);
237 test = and_condition_over_domain(test, s);
238 }
239 test = simplify(test);
240 debug(3) << " -> " << test << "\n";
241 }
242 return is_const_one(test);
243 }
244
245 Expr visit(const Min *op) override {
246 if (!op->type.is_int() || op->type.bits() < 32) {
247 return IRMutator::visit(op);
248 } else {
249 Expr a = mutate(op->a);
250 Expr b = mutate(op->b);
251 Expr test = a <= b;
252 if (provably_true_over_domain(a <= b)) {
253 return a;
254 } else if (provably_true_over_domain(b <= a)) {
255 return b;
256 } else {
257 return Min::make(a, b);
258 }
259 }
260 }
261
262 Expr visit(const Max *op) override {
263 if (!op->type.is_int() || op->type.bits() < 32) {
264 return IRMutator::visit(op);
265 } else {
266 Expr a = mutate(op->a);
267 Expr b = mutate(op->b);
268 if (provably_true_over_domain(a >= b)) {
269 return a;
270 } else if (provably_true_over_domain(b >= a)) {
271 return b;
272 } else {
273 return Max::make(a, b);
274 }
275 }
276 }

278 template<typename Cmp>
279 Expr visit_cmp(const Cmp *op) {
280 Expr expr = IRMutator::visit(op);
281 if (provably_true_over_domain(expr)) {
282 expr = make_one(op->type);
283 } else if (provably_true_over_domain(!expr)) {
284 expr = make_zero(op->type);
285 }
286 return expr;
287 }
288
289 Expr visit(const LE *op) override {
290 return visit_cmp(op);
291 }
292
293 Expr visit(const LT *op) override {
294 return visit_cmp(op);
295 }
296
297 Expr visit(const GE *op) override {
298 return visit_cmp(op);
299 }
300
301 Expr visit(const GT *op) override {
302 return visit_cmp(op);
303 }
304
305 Expr visit(const EQ *op) override {
306 return visit_cmp(op);
307 }
308
309 Expr visit(const NE *op) override {
310 return visit_cmp(op);
311 }

313 template<typename StmtOrExpr, typename LetStmtOrLet>
314 StmtOrExpr visit_let(const LetStmtOrLet *op) {
315 Expr value = mutate(op->value);
316 StmtOrExpr body;
317 if (value.type() == Int(32) && is_pure(value)) {
318 containing_loops.push_back({op->name, {value, value}});
319 body = mutate(op->body);
320 containing_loops.pop_back();
321 } else {
322 body = mutate(op->body);
323 }
324 return LetStmtOrLet::make(op->name, value, body);
325 }
326
327 Expr visit(const Let *op) override {
328 return visit_let<Expr, Let>(op);
329 }
330
331 Stmt visit(const LetStmt *op) override {
332 return visit_let<Stmt, LetStmt>(op);
333 }
334
335 Stmt visit(const For *op) override {
336 // Simplify the loop bounds.
337 Expr min = mutate(op->min);
338 Expr extent = mutate(op->extent);
339 containing_loops.push_back({op->name, {min, min + extent - 1}});
340 Stmt body = mutate(op->body);
341 containing_loops.pop_back();
342 return For::make(op->name, min, extent, op->for_type, op->device_api, body);
343 }
344
345public:
346 SimplifyUsingBounds(const string &v, const Interval &i) {
347 containing_loops.push_back({v, i});
348 }
349
350 SimplifyUsingBounds() = default;

353class TrimNoOps : public IRMutator {
354 using IRMutator::visit;
355
356 Stmt visit(const For *op) override {
357 // Bounds of GPU loops can't depend on outer gpu loop vars
358 if (CodeGen_GPU_Dev::is_gpu_var(op->name)) {
359 debug(3) << "TrimNoOps found gpu loop var: " << op->name << "\n";
360 return IRMutator::visit(op);
361 }
362
363 Stmt body = mutate(op->body);
364
365 debug(3) << "\n\n ***** Trim no ops in loop over " << op->name << "\n";
366
367 IsNoOp is_no_op;
368 body.accept(&is_no_op);
369 debug(3) << "Condition is " << is_no_op.condition << "\n";
370 is_no_op.condition = simplify(simplify(common_subexpression_elimination(is_no_op.condition)));
371
372 debug(3) << "Simplified condition is " << is_no_op.condition << "\n";
373
374 if (is_const_one(is_no_op.condition)) {
375 // This loop is definitely useless
376 debug(3) << "Removed empty loop.\n"
377 << "Old: " << Stmt(op) << "\n";
378 return Evaluate::make(0);
379 } else if (is_const_zero(is_no_op.condition)) {
380 // This loop is definitely needed
381 if (body.same_as(op->body)) {
382 return op;
383 } else {
384 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
385 }
386 }
387
388 // The condition is something interesting. Try to see if we
389 // can trim the loop bounds over which the loop does
390 // something.
391 Interval i = solve_for_outer_interval(!is_no_op.condition, op->name);
392
393 debug(3) << "Interval is: " << i.min << ", " << i.max << "\n";
394
395 if (i.is_everything()) {
396 // Nope.
397 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
398 }
399
400 if (i.is_empty()) {
401 // Empty loop
402 debug(3) << "Removed empty loop.\n"
403 << "Old: " << Stmt(op) << "\n";
404 return Evaluate::make(0);
405 }
406
407 // Simplify the body to take advantage of the fact that the
408 // loop range is now truncated
409 body = simplify(SimplifyUsingBounds(op->name, i).mutate(body));
410
411 string new_min_name = unique_name(op->name + ".new_min");
412 string new_max_name = unique_name(op->name + ".new_max");
413 string old_max_name = unique_name(op->name + ".old_max");
414 Expr new_min_var = Variable::make(Int(32), new_min_name);
415 Expr new_max_var = Variable::make(Int(32), new_max_name);
416 Expr old_max_var = Variable::make(Int(32), old_max_name);
417
418 // Convert max to max-plus-one
419 if (i.has_upper_bound()) {
420 i.max = i.max + 1;
421 }
422
423 // Truncate the loop bounds to the region over which it's not
424 // a no-op.
425 Expr old_max = op->min + op->extent;
426 Expr new_min, new_max;
427 if (i.has_lower_bound()) {
428 new_min = clamp(i.min, op->min, old_max_var);
429 } else {
430 new_min = op->min;
431 }
432 if (i.has_upper_bound()) {
433 new_max = clamp(i.max, new_min_var, old_max_var);
434 } else {
435 new_max = old_max;
436 }
437
438 Expr new_extent = new_max_var - new_min_var;
439
440 Stmt stmt = For::make(op->name, new_min_var, new_extent, op->for_type, op->device_api, body);
441 stmt = LetStmt::make(new_max_name, new_max, stmt);
442 stmt = LetStmt::make(new_min_name, new_min, stmt);
443 stmt = LetStmt::make(old_max_name, old_max, stmt);
444 stmt = simplify(stmt);
445
446 debug(3) << "Rewrote loop.\n"
447 << "Old: " << Stmt(op) << "\n"
448 << "New: " << stmt << "\n";
449
450 return stmt;
451 }
