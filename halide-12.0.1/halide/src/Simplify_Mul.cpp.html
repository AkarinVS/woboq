6Expr Simplify::visit(const Mul *op, ExprInfo *bounds) {
7 ExprInfo a_bounds, b_bounds;
8 Expr a = mutate(op->a, &a_bounds);
9 Expr b = mutate(op->b, &b_bounds);
10
11 if (bounds && no_overflow_int(op->type)) {
12 bool a_positive = a_bounds.min_defined && a_bounds.min > 0;
13 bool b_positive = b_bounds.min_defined && b_bounds.min > 0;
14 bool a_bounded = a_bounds.min_defined && a_bounds.max_defined;
15 bool b_bounded = b_bounds.min_defined && b_bounds.max_defined;
16
17 if (a_bounded && b_bounded) {
18 bounds->min_defined = bounds->max_defined = true;
19 int64_t v1 = saturating_mul(a_bounds.min, b_bounds.min);
20 int64_t v2 = saturating_mul(a_bounds.min, b_bounds.max);
21 int64_t v3 = saturating_mul(a_bounds.max, b_bounds.min);
22 int64_t v4 = saturating_mul(a_bounds.max, b_bounds.max);
23 bounds->min = std::min(std::min(v1, v2), std::min(v3, v4));
24 bounds->max = std::max(std::max(v1, v2), std::max(v3, v4));
25 } else if ((a_bounds.max_defined && b_bounded && b_positive) ||
26 (b_bounds.max_defined && a_bounded && a_positive)) {
27 bounds->max_defined = true;
28 bounds->max = saturating_mul(a_bounds.max, b_bounds.max);
29 } else if ((a_bounds.min_defined && b_bounded && b_positive) ||
30 (b_bounds.min_defined && a_bounded && a_positive)) {
31 bounds->min_defined = true;
32 bounds->min = saturating_mul(a_bounds.min, b_bounds.min);
33 }
34
35 if (bounds->max_defined && bounds->max == INT64_MAX) {
36 // Assume it saturated to avoid overflow. This gives up a
37 // single representable value at the top end of the range
38 // to represent infinity.
39 bounds->max_defined = false;
40 bounds->max = 0;
41 }
42 if (bounds->min_defined && bounds->min == INT64_MIN) {
43 bounds->min_defined = false;
44 bounds->min = 0;
45 }
46
47 bounds->alignment = a_bounds.alignment * b_bounds.alignment;
48 bounds->trim_bounds_using_alignment();
49 }
50
51 if (may_simplify(op->type)) {
52
53 // Order commutative operations by node type
54 if (should_commute(a, b)) {
55 std::swap(a, b);
56 std::swap(a_bounds, b_bounds);
57 }
58
59 auto rewrite = IRMatcher::rewriter(IRMatcher::mul(a, b), op->type);
60 if (rewrite(c0 * c1, fold(c0 * c1)) ||
61 rewrite(IRMatcher::Overflow() * x, a) ||
62 rewrite(x * IRMatcher::Overflow(), b) ||
63 rewrite(0 * x, 0) ||
64 rewrite(1 * x, x) ||
65 rewrite(x * 0, 0) ||
66 rewrite(x * 1, x)) {
67 return rewrite.result;
68 }
69
70 if (rewrite((x + c0) * c1, x * c1 + fold(c0 * c1), !overflows(c0 * c1)) ||
71 rewrite((c0 - x) * c1, x * fold(-c1) + fold(c0 * c1), !overflows(c0 * c1)) ||
72 rewrite((0 - x) * y, 0 - x * y) ||
73 rewrite(x * (0 - y), 0 - x * y) ||
74 rewrite((x - y) * c0, (y - x) * fold(-c0), c0 < 0 && -c0 > 0) ||
75 rewrite((x * c0) * c1, x * fold(c0 * c1), !overflows(c0 * c1)) ||
76 rewrite((x * c0) * y, (x * y) * c0, !is_const(y)) ||
77 rewrite(x * (y * c0), (x * y) * c0) ||
78 rewrite(max(x, y) * min(x, y), x * y) ||
79 rewrite(max(x, y) * min(y, x), y * x) ||
80 rewrite(broadcast(x, c0) * broadcast(y, c0), broadcast(x * y, c0)) ||
81 rewrite(broadcast(x, c0) * broadcast(y, c1), broadcast(x * broadcast(y, fold(c1 / c0)), c0), c1 % c0 == 0) ||
82 rewrite(broadcast(y, c1) * broadcast(x, c0), broadcast(broadcast(y, fold(c1 / c0)) * x, c0), c1 % c0 == 0) ||
83 rewrite(ramp(x, y, c0) * broadcast(z, c0), ramp(x * z, y * z, c0)) ||
84 rewrite(ramp(broadcast(x, c0), broadcast(y, c0), c1) * broadcast(z, c2),
85 ramp(broadcast(x * z, c0), broadcast(y * z, c0), c1), c2 == c0 * c1) ||
86 false) {
87 return mutate(rewrite.result, bounds);
88 }
89 }
90
91 const Shuffle *shuffle_a = a.as<Shuffle>();
92 const Shuffle *shuffle_b = b.as<Shuffle>();
93 if (shuffle_a && shuffle_b &&
94 shuffle_a->is_slice() &&
95 shuffle_b->is_slice()) {
96 if (a.same_as(op->a) && b.same_as(op->b)) {
97 return hoist_slice_vector<Mul>(op);
98 } else {
99 return hoist_slice_vector<Mul>(Mul::make(a, b));
100 }
101 }
102
103 if (a.same_as(op->a) && b.same_as(op->b)) {
104 return op;
105 } else {
106 return Mul::make(a, b);
107 }
108}
