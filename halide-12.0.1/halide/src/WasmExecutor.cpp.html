92// BDMalloc aka BrainDeadMalloc. This is an *extremely* simple-minded implementation
93// of malloc/free on top of a WasmMemoryObject, and is intended to be just barely adequate
94// to allow Halide's JIT-based tests to pass. It is neither memory-efficient nor performant,
95// nor has it been particularly well-vetted for potential buffer overruns and such.
96class BDMalloc {
97 struct Region {
98 uint32_t size : 31;
99 uint32_t used : 1;
100 };
101
102 uint32_t total_size = 0;
103 std::map<uint32_t, Region> regions;
104
105public:
106 BDMalloc() = default;
107
108 uint32_t get_total_size() const {
109 return total_size;
110 }
111
112 bool inited() const {
113 return total_size > 0;
114 }
115
116 void init(uint32_t total_size, uint32_t heap_start = 1) {
117 this->total_size = total_size;
118 regions.clear();
119
120 internal_assert(heap_start < total_size);
121 // Area before heap_start is permanently off-limits
122 regions[0] = {heap_start, true};
123 // Everything else is free
124 regions[heap_start] = {total_size - heap_start, false};
125 }
126
127 void reset() {
128 this->total_size = 0;
129 regions.clear();
130 }

132 uint32_t alloc_region(uint32_t requested_size) {
133 internal_assert(requested_size > 0);
134
135 bddebug(1) << "begin alloc_region " << requested_size << "\n";
136 validate();
137
138 // TODO: this would be faster with a basic free list,
139 // but for most Halide test code, there aren't enough allocations
140 // for this to be worthwhile, or at least that's my observation from
141 // a first run-thru. Consider adding it if any of the JIT-based tests
142 // seem unreasonably slow; as it is, a linear search for the first free
143 // block of adequate size is (apparently) performant enough.
144
145 // alignment and min-block-size are the same for our purposes here.
146 constexpr uint32_t kAlignment = 32;
147 const uint32_t size = std::max(align_up((uint32_t)requested_size, kAlignment), kAlignment);
148
149 constexpr uint32_t kMaxAllocSize = 0x7fffffff;
150 internal_assert(size <= kMaxAllocSize);
151 bddebug(2) << "size -> " << size << "\n";
152
153 for (auto it = regions.begin(); it != regions.end(); it++) {
154 const uint32_t start = it->first;
155 Region &r = it->second;
156 if (!r.used && r.size >= size) {
157 bddebug(2) << "alloc @ " << start << "," << (uint32_t)r.size << "\n";
158 if (r.size > size + kAlignment) {
159 // Split the block
160 const uint32_t r2_start = start + size;
161 const uint32_t r2_size = r.size - size;
162 regions[r2_start] = {r2_size, false};
163 r.size = size;
164 bddebug(2) << "split: r-> " << start << "," << (uint32_t)r.size << "," << (start + r.size) << "\n";
165 bddebug(2) << "split: r2-> " << r2_start << "," << r2_size << "," << (r2_start + r2_size) << "\n";
166 }
167 // Just return the block
168 r.used = true;
169 bddebug(1) << "end alloc_region " << requested_size << "\n";
170 validate();
171 return start;
172 }
173 }
174 bddebug(1) << "fail alloc_region " << requested_size << "\n";
175 validate();
176 return 0;
177 }
178
179 void free_region(uint32_t start) {
180 bddebug(1) << "begin free_region " << start << "\n";
181 validate();
182
183 // Can't free region at zero
184 if (!start) {
185 return;
186 }
187
188 internal_assert(start > 0);
189 auto it = regions.find(start);
190 internal_assert(it != regions.end());
191 internal_assert(it->second.used);
192 it->second.used = false;
193 // If prev region is free, combine with it
194 if (it != regions.begin()) {
195 auto prev = std::prev(it);
196 if (!prev->second.used) {
197 bddebug(2) << "combine prev: " << prev->first << " w/ " << it->first << "\n";
198 prev->second.size += it->second.size;
199 regions.erase(it);
200 it = prev;
201 }
202 }
203 // If next region is free, combine with it
204 auto next = std::next(it);
205 if (next != regions.end() && !next->second.used) {
206 bddebug(2) << "combine next: " << next->first << " w/ " << it->first << " "
207 << "\n";
208 it->second.size += next->second.size;
209 regions.erase(next);
210 }
211
212 bddebug(1) << "end free_region " << start << "\n";
213 validate();
214 }
215
216 void grow_total_size(uint32_t new_total_size) {
217 bddebug(1) << "begin grow_total_size " << new_total_size << "\n";
218 validate();
219
220 internal_assert(new_total_size > total_size);
221 auto it = regions.rbegin();
222 const uint32_t start = it->first;
223 Region &r = it->second;
224 uint32_t r_end = start + r.size;
225 internal_assert(r_end == total_size);
226 uint32_t delta = new_total_size - r_end;
227 if (r.used) {
228 // Add a free region after the last one
229 regions[r_end] = {delta, false};
230 } else {
231 // Just extend the last (free) region
232 r.size += delta;
233 }
234
235 // bookkeeping
236 total_size = new_total_size;
237
238 bddebug(1) << "end grow_total_size " << new_total_size << "\n";
239 validate();
240 }
241
242 void validate() const {
243 internal_assert(total_size > 0);
244#if (BDMALLOC_DEBUG_LEVEL >= 1) || (WASM_DEBUG_LEVEL >= 1)
245 uint32_t prev_end = 0;
246 bool prev_used = false;
247 for (auto it : regions) {
248 const uint32_t start = it.first;
249 const Region &r = it.second;
250 bddebug(2) << "R: " << start << ".." << (start + r.size - 1) << "," << r.used << "\n";
251 wassert(start == prev_end) << "start " << start << " prev_end " << prev_end << "\n";
252 // it's OK to have two used regions in a row, but not two free ones
253 wassert(!(!prev_used && !r.used));
254 prev_end = start + r.size;
255 prev_used = r.used;
256 }
257 wassert(prev_end == total_size) << "prev_end " << prev_end << " total_size " << total_size << "\n";
258 bddebug(2) << "\n";
259#endif
260 }

276std::vector<char> compile_to_wasm(const Module &module, const std::string &fn_name) {
277 static std::mutex link_lock;
278 std::lock_guard<std::mutex> lock(link_lock);
279
280 llvm::LLVMContext context;
281 std::unique_ptr<llvm::Module> fn_module;
282
283 // Default wasm stack size is ~64k, but schedules with lots of
284 // alloca usage (heavily inlined, or tracing enabled) can blow thru
285 // this, which crashes in amusing ways, so ask for extra stack space
286 // for the alloca usage.
287 size_t stack_size = 65536;
288 {
289 std::unique_ptr<CodeGen_Posix> cg(new_CodeGen_WebAssembly(module.target()));
290 cg->set_context(context);
291 fn_module = cg->compile(module);
292 stack_size += cg->get_requested_alloca_total();
293 }
294
295 stack_size = align_up(stack_size);
296 wdebug(1) << "Requesting stack size of " << stack_size << "\n";
297
298 std::unique_ptr<llvm::Module> llvm_module =
299 link_with_wasm_jit_runtime(&context, module.target(), std::move(fn_module));
300
301 llvm::SmallVector<char, 4096> object;
302 llvm::raw_svector_ostream object_stream(object);
303 compile_llvm_module_to_object(*llvm_module, object_stream);
304
305 // TODO: surely there's a better way that doesn't require spooling things
306 // out to temp files
307 TemporaryFile obj_file("", ".o");
308 write_entire_file(obj_file.pathname(), object.data(), object.size());
309#if WASM_DEBUG_LEVEL
310 obj_file.detach();
311 wdebug(1) << "Dumping obj_file to " << obj_file.pathname() << "\n";
312#endif
313
314 TemporaryFile wasm_output("", ".wasm");
315
316 std::string lld_arg_strs[] = {
317 "HalideJITLinker",
318 // For debugging purposes:
319 // "--verbose",
320 // "-error-limit=0",
321 // "--print-gc-sections",
322 "--export=__heap_base",
323 "--allow-undefined",
324 "-zstack-size=" + std::to_string(stack_size),
325 obj_file.pathname(),
326 "--entry=" + fn_name,
327 "-o",
328 wasm_output.pathname()};
329
330 constexpr int c = sizeof(lld_arg_strs) / sizeof(lld_arg_strs[0]);
331 const char *lld_args[c];
332 for (int i = 0; i < c; ++i) {
333 lld_args[i] = lld_arg_strs[i].c_str();
334 }
335
336 // lld will temporarily hijack the signal handlers to ensure that temp files get cleaned up,
337 // but rather than preserving custom handlers in place, it restores the default handlers.
338 // This conflicts with some of our testing infrastructure, which relies on a SIGABRT handler
339 // set at global-ctor time to stay set. Therefore we'll save and restore this ourselves.
340 // Note that we must restore it before using internal_error (and also on the non-error path).
341 auto old_abort_handler = std::signal(SIGABRT, SIG_DFL);
342
343 if (!lld::wasm::link(lld_args, /*CanExitEarly*/ false, llvm::outs(), llvm::errs())) {
344 std::signal(SIGABRT, old_abort_handler);
345 internal_error << "lld::wasm::link failed\n";
346 }
347
348 std::signal(SIGABRT, old_abort_handler);
349
350#if WASM_DEBUG_LEVEL
351 wasm_output.detach();
352 wdebug(1) << "Dumping linked wasm to " << wasm_output.pathname() << "\n";
353#endif
354
355 return read_entire_file(wasm_output.pathname());
356}

362// dynamic_type_dispatch is a utility for functors that want to be able
363// to dynamically dispatch a halide_type_t to type-specialized code.
364// To use it, a functor must be a *templated* class, e.g.
365//
366// template<typename T> class MyFunctor { int operator()(arg1, arg2...); };
367//
368// dynamic_type_dispatch() is called with a halide_type_t as the first argument,
369// followed by the arguments to the Functor's operator():
370//
371// auto result = dynamic_type_dispatch<MyFunctor>(some_halide_type, arg1, arg2);
372//
373// Note that this means that the functor must be able to instantiate its
374// operator() for all the Halide scalar types; it also means that all those
375// variants *will* be instantiated (increasing code size), so this approach
376// should only be used when strictly necessary.
377
378// clang-format off
379template<template<typename> class Functor, typename... Args>
380auto dynamic_type_dispatch(const halide_type_t &type, Args &&... args) -> decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...)) {
381
382#define HANDLE_CASE(CODE, BITS, TYPE) \
383 case halide_type_code(CODE, BITS): \
384 return Functor<TYPE>()(std::forward<Args>(args)...);
385
386 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
387 HANDLE_CASE(halide_type_bfloat, 16, bfloat16_t)
388 HANDLE_CASE(halide_type_float, 16, float16_t)
389 HANDLE_CASE(halide_type_float, 32, float)
390 HANDLE_CASE(halide_type_float, 64, double)
391 HANDLE_CASE(halide_type_int, 8, int8_t)
392 HANDLE_CASE(halide_type_int, 16, int16_t)
393 HANDLE_CASE(halide_type_int, 32, int32_t)
394 HANDLE_CASE(halide_type_int, 64, int64_t)
395 HANDLE_CASE(halide_type_uint, 1, bool)
396 HANDLE_CASE(halide_type_uint, 8, uint8_t)
397 HANDLE_CASE(halide_type_uint, 16, uint16_t)
398 HANDLE_CASE(halide_type_uint, 32, uint32_t)
399 HANDLE_CASE(halide_type_uint, 64, uint64_t)
400 HANDLE_CASE(halide_type_handle, 64, void *)
401 default:
402 internal_error;
403 using ReturnType = decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...));
404 return ReturnType();
405 }
406
407#undef HANDLE_CASE
408}

442 wasm32_ptr_t p = wabt_context.bdmalloc.alloc_region(size);

554// Given a halide_buffer_t on the host, allocate a wasm_halide_buffer_t in wasm
555// memory space and copy all relevant data. The resulting buf is laid out in
556// contiguous memory, and can be free with a single free().
557wasm32_ptr_t hostbuf_to_wasmbuf(WabtContext &wabt_context, const halide_buffer_t *src) {
558 static_assert(sizeof(halide_type_t) == 4, "halide_type_t");
559 static_assert(sizeof(halide_dimension_t) == 16, "halide_dimension_t");
560 static_assert(sizeof(wasm_halide_buffer_t) == 40, "wasm_halide_buffer_t");
561
562 wdebug(2) << "\nhostbuf_to_wasmbuf:\n";
563 if (!src) {
564 return 0;
565 }
566
567 dump_hostbuf(wabt_context, src, "src");
568
569 wassert(src->device == 0);
570 wassert(src->device_interface == nullptr);
571
572 // Assume our malloc() has everything 32-byte aligned,
573 // and insert enough padding for host to also be 32-byte aligned.
574 const size_t dims_size_in_bytes = sizeof(halide_dimension_t) * src->dimensions;
575 const size_t dims_offset = sizeof(wasm_halide_buffer_t);
576 const size_t mem_needed_base = sizeof(wasm_halide_buffer_t) + dims_size_in_bytes;
577 const size_t host_offset = align_up(mem_needed_base);
578 const size_t host_size_in_bytes = src->size_in_bytes();
579 const size_t mem_needed = host_offset + host_size_in_bytes;
580
581 const wasm32_ptr_t dst_ptr = wabt_malloc(wabt_context, mem_needed);
582 wassert(dst_ptr);
583
584 uint8_t *base = get_wasm_memory_base(wabt_context);
585

612

630 dst_tmp.dimensions = src->dimensions;

1161 wabt::interp::Trap::Ptr *trap) {
1162 WabtContext &wabt_context = get_wabt_context(thread);
1163
1164 wassert(arg_types.size() >= 1);
1165 const size_t arg_types_len = arg_types.size() - 1;
1166 const ExternArgType &ret_type = arg_types[0];
1167
1168 // There's wasted space here, but that's ok.
1169 std::vector<Halide::Runtime::Buffer<>> buffers(arg_types_len);
1170 std::vector<uint64_t> scalars(arg_types_len, 0);
1171 std::vector<void *> trampoline_args(arg_types_len, nullptr);
1172
1173 for (size_t i = 0; i < arg_types_len; ++i) {
1174 const auto &a = arg_types[i + 1];
1175 if (a.is_ucon) {
1176 // We have to special-case ucon because Halide considers it an int64 everywhere
1177 // (even for wasm, where pointers are int32), and trying to extract it as an
1178 // int64 from a Value that is int32 will assert-fail. In JIT mode the value
1179 // doesn't even matter (except for guarding that it is our predicted constant).
1180 wassert(args[i].Get<int32_t>() == 0 || args[i].Get<int32_t>() == kMagicJitUserContextValue);
1181 store_value(Int(32), args[i], &scalars[i]);
1182 trampoline_args[i] = &scalars[i];
1183 } else if (a.is_buffer) {
1184 const wasm32_ptr_t buf_ptr = args[i].Get<int32_t>();
1185 wasmbuf_to_hostbuf(wabt_context, buf_ptr, buffers[i]);
1186 trampoline_args[i] = buffers[i].raw_buffer();
1187 } else {
1188 store_value(a.type, args[i], &scalars[i]);
1189 trampoline_args[i] = &scalars[i];
1190 }
1191 }
1192
1193 // The return value (if any) is always scalar.
1194 uint64_t ret_val = 0;
1195 const bool has_retval = !ret_type.is_void;
1196 internal_assert(!ret_type.is_buffer);
1197 if (has_retval) {
1198 trampoline_args.push_back(&ret_val);
1199 }
1200 (*trampoline_fn)(trampoline_args.data());
1201
1202 if (has_retval) {
1203 results[0] = dynamic_type_dispatch<LoadValue>(ret_type.type, (void *)&ret_val);
1204 }
1205
1206 // Progagate buffer data backwards. Note that for arbitrary extern functions,
1207 // we have no idea which buffers might be "input only", so we copy all data for all of them.
1208 for (size_t i = 0; i < arg_types_len; ++i) {
1209 const auto &a = arg_types[i + 1];
1210 if (a.is_buffer) {
1211 const wasm32_ptr_t buf_ptr = args[i].Get<int32_t>();
1212 copy_hostbuf_to_existing_wasmbuf(wabt_context, buffers[i], buf_ptr);
1213 }
1214 }
1215
1216 return wabt::Result::Ok;
1217}

1226wabt::interp::HostFunc::Ptr make_extern_callback(wabt::interp::Store &store,
1227 const std::map<std::string, Halide::JITExtern> &jit_externs,
1228 const JITModule &trampolines,
1229 const wabt::interp::ImportDesc &import) {
1230 const std::string &fn_name = import.type.name;
1231 if (should_skip_extern_symbol(fn_name)) {
1232 wdebug(1) << "Skipping extern symbol: " << fn_name << "\n";
1233 return wabt::interp::HostFunc::Ptr();
1234 }
1235
1236 const auto it = jit_externs.find(fn_name);
1237 if (it == jit_externs.end()) {
1238 wdebug(1) << "Extern symbol not found in JIT Externs: " << fn_name << "\n";
1239 return wabt::interp::HostFunc::Ptr();
1240 }
1241 const ExternSignature &sig = it->second.extern_c_function().signature();
1242
1243 const auto &tramp_it = trampolines.exports().find(fn_name + kTrampolineSuffix);
1244 if (tramp_it == trampolines.exports().end()) {
1245 wdebug(1) << "Extern symbol not found in trampolines: " << fn_name << "\n";
1246 return wabt::interp::HostFunc::Ptr();
1247 }
1248 TrampolineFn trampoline_fn = (TrampolineFn)tramp_it->second.address;
1249
1250 const size_t arg_count = sig.arg_types().size();
1251
1252 std::vector<ExternArgType> arg_types;
1253
1254 if (sig.is_void_return()) {
1255 const bool is_void = true;
1256 const bool is_buffer = false;
1257 const bool is_ucon = false;
1258 // Specifying a type here with bits == 0 should trigger a proper 'void' return type
1259 arg_types.push_back(ExternArgType{{halide_type_int, 0, 0}, is_void, is_buffer, is_ucon});
1260 } else {
1261 const Type &t = sig.ret_type();
1262 const bool is_void = false;
1263 const bool is_buffer = (t == type_of<halide_buffer_t *>());
1264 const bool is_ucon = false;
1265 user_assert(t.lanes() == 1) << "Halide Extern functions cannot return vector values.";
1266 user_assert(!is_buffer) << "Halide Extern functions cannot return halide_buffer_t.";
1267 arg_types.push_back(ExternArgType{t, is_void, is_buffer, is_ucon});
1268 }
1269 for (size_t i = 0; i < arg_count; ++i) {
1270 const Type &t = sig.arg_types()[i];
1271 const bool is_void = false;
1272 const bool is_buffer = (t == type_of<halide_buffer_t *>());
1273 // Since arbitrary pointer args aren't legal for extern calls,
1274 // assume that anything that is a void* is a user context.
1275 const bool is_ucon = (t == type_of<void *>());
1276 user_assert(t.lanes() == 1) << "Halide Extern functions cannot accept vector values as arguments.";
1277 arg_types.push_back(ExternArgType{t, is_void, is_buffer, is_ucon});
1278 }
1279
1280 const auto callback_wrapper =
1281 [arg_types, trampoline_fn](wabt::interp::Thread &thread,
1282 const wabt::interp::Values &args,
1283 wabt::interp::Values &results,
1284 wabt::interp::Trap::Ptr *trap) -> wabt::Result {
1285 return extern_callback_wrapper(arg_types, trampoline_fn, thread, args, results, trap);
1286 };
1287
1288 auto func_type = *wabt::cast<wabt::interp::FuncType>(import.type.type.get());
1289 auto host_func = wabt::interp::HostFunc::New(store, func_type, callback_wrapper);
1290
1291 return host_func;
1292}

1345 const std::string &fn_name,
1346 const std::map<std::string, Halide::JITExtern> &jit_externs,

1354#if WITH_WABT
1355 // TODO: we should probably move this to LLVM 12 or 13, since LLVM11 won't support SIMD usefully enough for Halide
1356 user_assert(LLVM_VERSION >= 110) << "Using the WebAssembly JIT is only supported under LLVM 11+.";
1357
1358 user_assert(!target.has_feature(Target::WasmThreads)) << "wasm_threads requires Emscripten (or a similar compiler); it will never be supported under JIT.";
1359
1360 wdebug(1) << "Compiling wasm function " << fn_name << "\n";
1361
1362 // Compile halide into wasm bytecode.
1363 std::vector<char> final_wasm = compile_to_wasm(halide_module, fn_name);
1364
1365 store = wabt::interp::Store(calc_features(halide_module.target()));
1366
1367 // Create a wabt Module for it.
1368 wabt::MemoryStream log_stream;
1369 constexpr bool kReadDebugNames = true;
1370 constexpr bool kStopOnFirstError = true;
1371 constexpr bool kFailOnCustomSectionError = true;
1372 wabt::ReadBinaryOptions options(store.features(),
1373 &log_stream,
1374 kReadDebugNames,
1375 kStopOnFirstError,
1376 kFailOnCustomSectionError);
1377 wabt::Errors errors;
1378 wabt::interp::ModuleDesc module_desc;
1379 wabt::Result r = wabt::interp::ReadBinaryInterp(final_wasm.data(),
1380 final_wasm.size(),
1381 options,
1382 &errors,
1383 &module_desc);
1384 internal_assert(Succeeded(r))
1385 << "ReadBinaryInterp failed:\n"
1386 << wabt::FormatErrorsToString(errors, wabt::Location::Type::Binary) << "\n"
1387 << " log: " << to_string(log_stream) << "\n";
1388
1389 if (WASM_DEBUG_LEVEL >= 2) {
1390 wabt::MemoryStream dis_stream;
1391 module_desc.istream.Disassemble(&dis_stream);
1392 wdebug(WASM_DEBUG_LEVEL) << "Disassembly:\n"
1393 << to_string(dis_stream) << "\n";
1394 }
1395
1396 module = wabt::interp::Module::New(store, module_desc);
1397
1398 // Bind all imports to our callbacks.
1399 wabt::interp::RefVec imports;
1400 const HostCallbackMap &host_callback_map = get_host_callback_map();
1401 for (const auto &import : module->desc().imports) {
1402 wdebug(1) << "import=" << import.type.module << "." << import.type.name << "\n";
1403 if (import.type.type->kind == wabt::interp::ExternKind::Func && import.type.module == "env") {
1404 auto it = host_callback_map.find(import.type.name);
1405 if (it != host_callback_map.end()) {
1406 auto func_type = *wabt::cast<wabt::interp::FuncType>(import.type.type.get());
1407 auto host_func = wabt::interp::HostFunc::New(store, func_type, it->second);
1408 imports.push_back(host_func.ref());
1409 continue;
1410 }
1411
1412 // If it's not one of the standard host callbacks, assume it must be
1413 // a define_extern, and look for it in the jit_externs.
1414 auto host_func = make_extern_callback(store, jit_externs, trampolines, import);
1415 imports.push_back(host_func.ref());
1416 continue;
1417 }
1418 // By default, just push a null reference. This won't resolve, and
1419 // instantiation will fail.
1420 imports.push_back(wabt::interp::Ref::Null);
1421 }

1425 internal_assert(instance) << "Error initializing module: " << trap->message() << "\n";

1429 for (const auto &e : module_desc.exports) {
1430 if (e.type.name == "__heap_base") {
1431 internal_assert(e.type.type->kind == wabt::ExternalKind::Global);
1432 heap_base = store.UnsafeGet<wabt::interp::Global>(instance->globals()[e.index])->Get().Get<int32_t>();
1433 wdebug(1) << "__heap_base is " << heap_base << "\n";
1434 continue;
1435 }
1436 if (e.type.name == "memory") {
1437 internal_assert(e.type.type->kind == wabt::ExternalKind::Memory);
1438 internal_assert(!memory.get()) << "Expected exactly one memory object but saw " << (void *)memory.get();
1439 memory = store.UnsafeGet<wabt::interp::Memory>(instance->memories()[e.index]);
1440 wdebug(1) << "heap_size is " << memory->ByteSize() << "\n";
1441 continue;
1442 }
1443 }
1444 internal_assert(heap_base >= 0) << "__heap_base not found";
1445 internal_assert(memory->ByteSize() > 0) << "memory size is unlikely";
1446
1447 bdmalloc.init(memory->ByteSize(), heap_base);
1448
1449#endif // WITH_WABT

1452int WasmModuleContents::run(const void **args) {
1453#if WITH_WABT
1454 const auto &module_desc = module->desc();
1455
1456 wabt::interp::FuncType *func_type = nullptr;
1457 wabt::interp::RefPtr<wabt::interp::Func> func;
1458 std::string func_name;
1459
1460 for (const auto &e : module_desc.exports) {
1461 if (e.type.type->kind == wabt::ExternalKind::Func) {
1462 wdebug(1) << "Selecting export '" << e.type.name << "'\n";
1463 internal_assert(!func_type && !func) << "Multiple exported funcs found";
1464 func_type = wabt::cast<wabt::interp::FuncType>(e.type.type.get());
1465 func = store.UnsafeGet<wabt::interp::Func>(instance->funcs()[e.index]);
1466 func_name = e.type.name;
1467 continue;
1468 }
1469 }
1470
1471 JITUserContext *jit_user_context = nullptr;
1472 for (size_t i = 0; i < arguments.size(); i++) {
1473 const Argument &arg = arguments[i];
1474 const void *arg_ptr = args[i];
1475 if (arg.name == "__user_context") {
1476 jit_user_context = *(JITUserContext **)const_cast<void *>(arg_ptr);
1477 }
1478 }
1479
1480 WabtContext wabt_context(jit_user_context, *memory, bdmalloc);
1481
1482 wabt::interp::Values wabt_args;
1483 wabt::interp::Values wabt_results;
1484 wabt::interp::Trap::Ptr trap;
1485
1486 std::vector<wasm32_ptr_t> wbufs(arguments.size(), 0);
1487
1488 for (size_t i = 0; i < arguments.size(); i++) {
1489 const Argument &arg = arguments[i];
1490 const void *arg_ptr = args[i];
1491 if (arg.is_buffer()) {
1492 halide_buffer_t *buf = (halide_buffer_t *)const_cast<void *>(arg_ptr);
1493 // It's OK for this to be null (let Halide asserts handle it)
1494 wasm32_ptr_t wbuf = hostbuf_to_wasmbuf(wabt_context, buf);
1495 wbufs[i] = wbuf;
1496 wabt_args.push_back(load_value(wbuf));
1497 } else {
1498 if (arg.name == "__user_context") {
1499 wabt_args.push_back(wabt::interp::Value::Make(kMagicJitUserContextValue));
1500 } else {
1501 wabt_args.push_back(load_value(arg.type, arg_ptr));
1502 }
1503 }
1504 }
1505
1506 wabt::interp::Thread::Options options;
1507 wabt::interp::Thread::Ptr thread = wabt::interp::Thread::New(store, options);
1508 thread->set_host_info(&wabt_context);
1509
1510 auto r = func->Call(*thread, wabt_args, wabt_results, &trap);
1511 if (WASM_DEBUG_LEVEL >= 2) {
1512 wabt::MemoryStream call_stream;
1513 WriteCall(&call_stream, func_name, *func_type, wabt_args, wabt_results, trap);
1514 wdebug(WASM_DEBUG_LEVEL) << to_string(call_stream) << "\n";
1515 }
1516 internal_assert(Succeeded(r)) << "Func::Call failed: " << trap->message() << "\n";
1517 internal_assert(wabt_results.size() == 1);
1518 int32_t result = wabt_results[0].Get<int32_t>();
1519
1520 wdebug(1) << "Result is " << result << "\n";
1521
1522 if (result == 0) {
1523 // Update any output buffers
1524 for (size_t i = 0; i < arguments.size(); i++) {
1525 const Argument &arg = arguments[i];
1526 const void *arg_ptr = args[i];
1527 if (arg.is_buffer()) {
1528 halide_buffer_t *buf = (halide_buffer_t *)const_cast<void *>(arg_ptr);
1529 copy_wasmbuf_to_existing_hostbuf(wabt_context, wbufs[i], buf);
1530 }
1531 }
1532 }
1533
1534 for (wasm32_ptr_t p : wbufs) {
1535 wabt_free(wabt_context, p);
1536 }
1537
1538 // Don't do this: things allocated by Halide runtime might need to persist
1539 // between multiple invocations of the same function.
1540 // bdmalloc.reset();
1541
1542 return result;
1543
1544#endif
1545
1546 internal_error << "WasmExecutor is not configured correctly";
1547 return -1;
1548}
