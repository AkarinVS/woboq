10Expr Simplify::visit(const IntImm *op, ExprInfo *bounds) {
11 if (bounds && no_overflow_int(op->type)) {
12 bounds->min_defined = bounds->max_defined = true;
13 bounds->min = bounds->max = op->value;
14 bounds->alignment.remainder = op->value;
15 bounds->alignment.modulus = 0;
16 }
17 return op;
18}

38Expr Simplify::visit(const Broadcast *op, ExprInfo *bounds) {
39 Expr value = mutate(op->value, bounds);
40
41 const int lanes = op->lanes;
42
43 auto rewrite = IRMatcher::rewriter(IRMatcher::broadcast(value, lanes), op->type);
44 if (rewrite(broadcast(broadcast(x, c0), lanes), broadcast(x, c0 * lanes)) ||
45 false) {
46 return mutate(rewrite.result, bounds);
47 }
48
49 if (value.same_as(op->value)) {
50 return op;
51 } else {
52 return Broadcast::make(value, op->lanes);
53 }

56Expr Simplify::visit(const VectorReduce *op, ExprInfo *bounds) {
57 Expr value = mutate(op->value, bounds);
58
59 const int lanes = op->type.lanes();
60 const int arg_lanes = op->value.type().lanes();
61 const int factor = arg_lanes / lanes;
62 if (factor == 1) {
63 return value;
64 }
65
66 if (bounds && op->type.is_int()) {
67 switch (op->op) {
68 case VectorReduce::Add:
69 // Alignment of result is the alignment of the arg. Bounds
70 // of the result can grow according to the reduction
71 // factor.
72 if (bounds->min_defined) {
73 bounds->min *= factor;
74 }
75 if (bounds->max_defined) {
76 bounds->max *= factor;
77 }
78 break;
79 case VectorReduce::SaturatingAdd:
80 if (bounds->min_defined) {
81 bounds->min = saturating_mul(bounds->min, factor);
82 }
83 if (bounds->max_defined) {
84 bounds->max = saturating_mul(bounds->max, factor);
85 }
86 break;
87 case VectorReduce::Mul:
88 // Don't try to infer anything about bounds. Leave the
89 // alignment unchanged even though we could theoretically
90 // upgrade it.
91 bounds->min_defined = bounds->max_defined = false;
92 break;
93 case VectorReduce::Min:
94 case VectorReduce::Max:
95 // Bounds and alignment of the result are just the bounds and alignment of the arg.
96 break;
97 case VectorReduce::And:
98 case VectorReduce::Or:
99 // For integer types this is a bitwise operator. Don't try
100 // to infer anything for now.
101 bounds->min_defined = bounds->max_defined = false;
102 bounds->alignment = ModulusRemainder{};
103 break;
104 }
105 }
106
107 // We can pull multiplications by a broadcast out of horizontal
108 // additions and do the horizontal addition earlier. This means we
109 // do the multiplication on a vector with fewer lanes. This
110 // approach applies whenever we have a distributive law. We'll
111 // exploit the following distributive laws here:
112 // - Multiplication distributes over addition
113 // - min/max distributes over min/max
114 // - and/or distributes over and/or
115
116 // Further, we can collapse min/max/and/or of a broadcast down to
117 // a narrower broadcast.
118
119 // TODO: There are other rules we could apply here if they ever
120 // come up in practice:
121 // - a horizontal min/max/add of a ramp is a different ramp
122 // - horizontal add of a broadcast is a broadcast + multiply
123 // - horizontal reduce of an shuffle_vectors may be simplifiable to the
124 // underlying op on different shuffle_vectors calls
125
126 switch (op->op) {
127 case VectorReduce::Add: {
128 auto rewrite = IRMatcher::rewriter(IRMatcher::h_add(value, lanes), op->type);
129 if (rewrite(h_add(x * broadcast(y, arg_lanes), lanes), h_add(x, lanes) * broadcast(y, lanes)) ||
130 rewrite(h_add(broadcast(x, arg_lanes) * y, lanes), h_add(y, lanes) * broadcast(x, lanes))) {
131 return mutate(rewrite.result, bounds);
132 }
133 break;
134 }
135 case VectorReduce::Min: {
136 auto rewrite = IRMatcher::rewriter(IRMatcher::h_min(value, lanes), op->type);
137 if (rewrite(h_min(min(x, broadcast(y, arg_lanes)), lanes), min(h_min(x, lanes), broadcast(y, lanes))) ||
138 rewrite(h_min(min(broadcast(x, arg_lanes), y), lanes), min(h_min(y, lanes), broadcast(x, lanes))) ||
139 rewrite(h_min(max(x, broadcast(y, arg_lanes)), lanes), max(h_min(x, lanes), broadcast(y, lanes))) ||
140 rewrite(h_min(max(broadcast(x, arg_lanes), y), lanes), max(h_min(y, lanes), broadcast(x, lanes))) ||
141 rewrite(h_min(broadcast(x, arg_lanes), lanes), broadcast(x, lanes)) ||
142 rewrite(h_min(broadcast(x, c0), lanes), h_min(x, lanes), factor % c0 == 0) ||
143 rewrite(h_min(ramp(x, y, arg_lanes), lanes), x + min(y * (arg_lanes - 1), 0)) ||
144 false) {
145 return mutate(rewrite.result, bounds);
146 }
147 break;
148 }
149 case VectorReduce::Max: {
150 auto rewrite = IRMatcher::rewriter(IRMatcher::h_max(value, lanes), op->type);
151 if (rewrite(h_max(min(x, broadcast(y, arg_lanes)), lanes), min(h_max(x, lanes), broadcast(y, lanes))) ||
152 rewrite(h_max(min(broadcast(x, arg_lanes), y), lanes), min(h_max(y, lanes), broadcast(x, lanes))) ||
153 rewrite(h_max(max(x, broadcast(y, arg_lanes)), lanes), max(h_max(x, lanes), broadcast(y, lanes))) ||
154 rewrite(h_max(max(broadcast(x, arg_lanes), y), lanes), max(h_max(y, lanes), broadcast(x, lanes))) ||
155 rewrite(h_max(broadcast(x, arg_lanes), lanes), broadcast(x, lanes)) ||
156 rewrite(h_max(broadcast(x, c0), lanes), h_max(x, lanes), factor % c0 == 0) ||
157 rewrite(h_max(ramp(x, y, arg_lanes), lanes), x + max(y * (arg_lanes - 1), 0)) ||
158 false) {
159 return mutate(rewrite.result, bounds);
160 }
161 break;
162 }
163 case VectorReduce::And: {
164 auto rewrite = IRMatcher::rewriter(IRMatcher::h_and(value, lanes), op->type);
165 if (rewrite(h_and(x || broadcast(y, arg_lanes), lanes), h_and(x, lanes) || broadcast(y, lanes)) ||
166 rewrite(h_and(broadcast(x, arg_lanes) || y, lanes), h_and(y, lanes) || broadcast(x, lanes)) ||
167 rewrite(h_and(x && broadcast(y, arg_lanes), lanes), h_and(x, lanes) && broadcast(y, lanes)) ||
168 rewrite(h_and(broadcast(x, arg_lanes) && y, lanes), h_and(y, lanes) && broadcast(x, lanes)) ||
169 rewrite(h_and(broadcast(x, arg_lanes), lanes), broadcast(x, lanes)) ||
170 rewrite(h_and(broadcast(x, c0), lanes), h_and(x, lanes), factor % c0 == 0) ||
171 rewrite(h_and(ramp(x, y, arg_lanes) < broadcast(z, arg_lanes), lanes),
172 x + max(y * (arg_lanes - 1), 0) < z) ||
173 rewrite(h_and(ramp(x, y, arg_lanes) <= broadcast(z, arg_lanes), lanes),
174 x + max(y * (arg_lanes - 1), 0) <= z) ||
175 rewrite(h_and(broadcast(x, arg_lanes) < ramp(y, z, arg_lanes), lanes),
176 x < y + min(z * (arg_lanes - 1), 0)) ||
177 rewrite(h_and(broadcast(x, arg_lanes) < ramp(y, z, arg_lanes), lanes),
178 x <= y + min(z * (arg_lanes - 1), 0)) ||
179 false) {
180 return mutate(rewrite.result, bounds);
181 }
182 break;
183 }
184 case VectorReduce::Or: {
185 auto rewrite = IRMatcher::rewriter(IRMatcher::h_or(value, lanes), op->type);
186 if (rewrite(h_or(x || broadcast(y, arg_lanes), lanes), h_or(x, lanes) || broadcast(y, lanes)) ||
187 rewrite(h_or(broadcast(x, arg_lanes) || y, lanes), h_or(y, lanes) || broadcast(x, lanes)) ||
188 rewrite(h_or(x && broadcast(y, arg_lanes), lanes), h_or(x, lanes) && broadcast(y, lanes)) ||
189 rewrite(h_or(broadcast(x, arg_lanes) && y, lanes), h_or(y, lanes) && broadcast(x, lanes)) ||
190 rewrite(h_or(broadcast(x, arg_lanes), lanes), broadcast(x, lanes)) ||
191 rewrite(h_or(broadcast(x, c0), lanes), h_or(x, lanes), factor % c0 == 0) ||
192 // type of arg_lanes is somewhat indeterminate
193 rewrite(h_or(ramp(x, y, arg_lanes) < broadcast(z, arg_lanes), lanes),
194 x + min(y * (arg_lanes - 1), 0) < z) ||
195 rewrite(h_or(ramp(x, y, arg_lanes) <= broadcast(z, arg_lanes), lanes),
196 x + min(y * (arg_lanes - 1), 0) <= z) ||
197 rewrite(h_or(broadcast(x, arg_lanes) < ramp(y, z, arg_lanes), lanes),
198 x < y + max(z * (arg_lanes - 1), 0)) ||
199 rewrite(h_or(broadcast(x, arg_lanes) < ramp(y, z, arg_lanes), lanes),
200 x <= y + max(z * (arg_lanes - 1), 0)) ||
201 false) {
202 return mutate(rewrite.result, bounds);
203 }
204 break;
205 }
206 default:
207 break;
208 }
209
210 if (value.same_as(op->value)) {
211 return op;
212 } else {
213 return VectorReduce::make(op->op, value, op->type.lanes());
214 }

217Expr Simplify::visit(const Variable *op, ExprInfo *bounds) {
218 if (bounds_and_alignment_info.contains(op->name)) {
219 const ExprInfo &b = bounds_and_alignment_info.get(op->name);
220 if (bounds) {
221 *bounds = b;
222 }
223 if (b.min_defined && b.max_defined && b.min == b.max) {
224 return make_const(op->type, b.min);
225 }
226 }
227
228 if (var_info.contains(op->name)) {
229 auto &info = var_info.ref(op->name);
230
231 // if replacement is defined, we should substitute it in (unless
232 // it's a var that has been hidden by a nested scope).
233 if (info.replacement.defined()) {
234 internal_assert(info.replacement.type() == op->type)
235 << "Cannot replace variable " << op->name
236 << " of type " << op->type
237 << " with expression of type " << info.replacement.type() << "\n";
238 info.new_uses++;
239 // We want to remutate the replacement, because we may be
240 // injecting it into a context where it is known to be a
241 // constant (e.g. due to an if).
242 return mutate(info.replacement, bounds);
243 } else {
244 // This expression was not something deemed
245 // substitutable - no replacement is defined.
246 info.old_uses++;
247 return op;
248 }
249 } else {
250 // We never encountered a let that defines this var. Must
251 // be a uniform. Don't touch it.
252 return op;
253 }

256Expr Simplify::visit(const Ramp *op, ExprInfo *bounds) {
257 ExprInfo base_bounds, stride_bounds;
258 Expr base = mutate(op->base, &base_bounds);
259 Expr stride = mutate(op->stride, &stride_bounds);
260 const int lanes = op->lanes;
261
262 if (bounds && no_overflow_int(op->type)) {
263 bounds->min_defined = base_bounds.min_defined && stride_bounds.min_defined;
264 bounds->max_defined = base_bounds.max_defined && stride_bounds.max_defined;
265 bounds->min = std::min(base_bounds.min, base_bounds.min + (lanes - 1) * stride_bounds.min);
266 bounds->max = std::max(base_bounds.max, base_bounds.max + (lanes - 1) * stride_bounds.max);
267 // A ramp lane is b + l * s. Expanding b into mb * x + rb and s into ms * y + rs, we get:
268 // mb * x + rb + l * (ms * y + rs)
269 // = mb * x + ms * l * y + rs * l + rb
270 // = gcd(rs, ms, mb) * z + rb
271 int64_t m = stride_bounds.alignment.modulus;
272 m = gcd(m, stride_bounds.alignment.remainder);
273 m = gcd(m, base_bounds.alignment.modulus);
274 int64_t r = base_bounds.alignment.remainder;
275 if (m != 0) {
276 r = mod_imp(base_bounds.alignment.remainder, m);
277 }
278 bounds->alignment = {m, r};
279 }
280
281 // A somewhat torturous way to check if the stride is zero,
282 // but it helps to have as many rules as possible written as
283 // formal rewrites, so that they can be formally verified,
284 // etc.
285 auto rewrite = IRMatcher::rewriter(IRMatcher::ramp(base, stride, lanes), op->type);
286 if (rewrite(ramp(x, 0, lanes), broadcast(x, lanes)) ||
287 rewrite(ramp(ramp(x, c0, c2), broadcast(c1, c4), c3),
288 ramp(x, c0, c2 * c3),
289 c1 == c0 * fold(c2)) ||
290 false) {
291
292 return rewrite.result;
293 }
294
295 if (base.same_as(op->base) &&
296 stride.same_as(op->stride)) {
297 return op;
298 } else {
299 return Ramp::make(base, stride, op->lanes);
300 }
301}

303Expr Simplify::visit(const Load *op, ExprInfo *bounds) {
304 found_buffer_reference(op->name);
305
306 Expr predicate = mutate(op->predicate, nullptr);
307
308 ExprInfo index_info;
309 Expr index = mutate(op->index, &index_info);
310
311 // If the load is fully out of bounds, replace it with undef.
312 // This should only occur inside branches that make the load unreachable,
313 // but perhaps the branch was hard to prove constant true or false. This
314 // provides an alternative mechanism to simplify these unreachable loads.
315 string alloc_extent_name = op->name + ".total_extent_bytes";
316 if (bounds_and_alignment_info.contains(alloc_extent_name)) {
317 if (index_info.max_defined && index_info.max < 0) {
318 in_unreachable = true;
319 return unreachable(op->type);
320 }
321 const ExprInfo &alloc_info = bounds_and_alignment_info.get(alloc_extent_name);
322 if (alloc_info.max_defined && index_info.min_defined) {
323 int index_min_bytes = index_info.min * op->type.bytes();
324 if (index_min_bytes > alloc_info.max) {
325 in_unreachable = true;
326 return unreachable(op->type);
327 }
328 }
329 }
330
331 ExprInfo base_info;
332 if (const Ramp *r = index.as<Ramp>()) {
333 mutate(r->base, &base_info);
334 }
335
336 base_info.alignment = ModulusRemainder::intersect(base_info.alignment, index_info.alignment);
337
338 ModulusRemainder align = ModulusRemainder::intersect(op->alignment, base_info.alignment);
339
340 const Broadcast *b_index = index.as<Broadcast>();
341 const Shuffle *s_index = index.as<Shuffle>();
342 if (is_const_zero(predicate)) {
343 // Predicate is always false
344 return undef(op->type);
345 } else if (b_index && is_const_one(predicate)) {
346 // Load of a broadcast should be broadcast of the load
347 Expr new_index = b_index->value;
348 int new_lanes = new_index.type().lanes();
349 Expr load = Load::make(op->type.with_lanes(new_lanes), op->name, b_index->value,
350 op->image, op->param, const_true(new_lanes), align);
351 return Broadcast::make(load, b_index->lanes);
352 } else if (s_index &&
353 is_const_one(predicate) &&
354 (s_index->is_concat() ||
355 s_index->is_interleave())) {
356 // Loads of concats/interleaves should be concats/interleaves of loads
357 std::vector<Expr> loaded_vecs;
358 for (const Expr &new_index : s_index->vectors) {
359 int new_lanes = new_index.type().lanes();
360 Expr load = Load::make(op->type.with_lanes(new_lanes), op->name, new_index,
361 op->image, op->param, const_true(new_lanes), ModulusRemainder{});
362 loaded_vecs.emplace_back(std::move(load));
363 }
364 return Shuffle::make(loaded_vecs, s_index->indices);
365 } else if (predicate.same_as(op->predicate) && index.same_as(op->index) && align == op->alignment) {
366 return op;
367 } else {
368 return Load::make(op->type, op->name, index, op->image, op->param, predicate, align);
369 }
