10class LowerUnsafePromises : public IRMutator {
11 using IRMutator::visit;

13 Expr visit(const Call *op) override {
14 if (op->is_intrinsic(Call::unsafe_promise_clamped)) {
15 if (check) {
16 Expr is_clamped = op->args[0] >= op->args[1] && op->args[0] <= op->args[2];
17 std::ostringstream promise_expr_text;
18 promise_expr_text << is_clamped;
19 Expr cond_as_string = StringImm::make(promise_expr_text.str());
20 Expr promise_broken_error =
21 Call::make(Int(32),
22 "halide_error_requirement_failed",
23 {cond_as_string, StringImm::make("from unsafe_promise_clamped")},
24 Call::Extern);
25 return Call::make(op->args[0].type(),
26 Call::require,
27 {mutate(is_clamped), mutate(op->args[0]), promise_broken_error},
28 Call::PureIntrinsic);
29 } else {
30 return mutate(op->args[0]);
31 }
32 } else {
33 return IRMutator::visit(op);
34 }
35 }
36
37 bool check;
38
39public:
40 LowerUnsafePromises(bool check)
41 : check(check) {
42 }
43};
