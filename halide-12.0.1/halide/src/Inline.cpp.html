19// Sanity check that this is a reasonable function to inline
20void validate_schedule_inlined_function(Function f) {
21 const FuncSchedule &func_s = f.schedule();
22 const StageSchedule &stage_s = f.definition().schedule();
23
24 if (!func_s.store_level().is_inlined()) {
25 user_error << "Function " << f.name() << " is scheduled to be computed inline, "
26 << "but is not scheduled to be stored inline. A storage schedule "
27 << "is meaningless for functions computed inline.\n";
28 }
29
30 // Inlining is allowed only if there is no specialization.
31 user_assert(f.definition().specializations().empty())
32 << "Function " << f.name() << " is scheduled inline, so it"
33 << " must not have any specializations. Specialize on the"
34 << " scheduled function instead.\n";
35
36 if (func_s.memoized()) {
37 user_error << "Cannot memoize function "
38 << f.name() << " because the function is scheduled inline.\n";
39 }
40
41 for (size_t i = 0; i < stage_s.dims().size(); i++) {
42 Dim d = stage_s.dims()[i];
43 if (d.is_unordered_parallel()) {
44 user_error << "Cannot parallelize dimension "
45 << d.var << " of function "
46 << f.name() << " because the function is scheduled inline.\n";
47 } else if (d.for_type == ForType::Unrolled) {
48 user_error << "Cannot unroll dimension "
49 << d.var << " of function "
50 << f.name() << " because the function is scheduled inline.\n";
51 } else if (d.for_type == ForType::Vectorized) {
52 user_error << "Cannot vectorize dimension "
53 << d.var << " of function "
54 << f.name() << " because the function is scheduled inline.\n";
55 }
56 }
57
58 for (size_t i = 0; i < stage_s.splits().size(); i++) {
59 if (stage_s.splits()[i].is_rename()) {
60 user_warning << "It is meaningless to rename variable "
61 << stage_s.splits()[i].old_var << " of function "
62 << f.name() << " to " << stage_s.splits()[i].outer
63 << " because " << f.name() << " is scheduled inline.\n";
64 } else if (stage_s.splits()[i].is_fuse()) {
65 user_warning << "It is meaningless to fuse variables "
66 << stage_s.splits()[i].inner << " and " << stage_s.splits()[i].outer
67 << " because " << f.name() << " is scheduled inline.\n";
68 } else {
69 user_warning << "It is meaningless to split variable "
70 << stage_s.splits()[i].old_var << " of function "
71 << f.name() << " into "
72 << stage_s.splits()[i].outer << " * "
73 << stage_s.splits()[i].factor << " + "
74 << stage_s.splits()[i].inner << " because "
75 << f.name() << " is scheduled inline.\n";
76 }
77 }
78
79 for (size_t i = 0; i < func_s.bounds().size(); i++) {
80 if (func_s.bounds()[i].min.defined()) {
81 user_warning << "It is meaningless to bound dimension "
82 << func_s.bounds()[i].var << " of function "
83 << f.name() << " to be within ["
84 << func_s.bounds()[i].min << ", "
85 << func_s.bounds()[i].extent << "] because the function is scheduled inline.\n";
86 } else if (func_s.bounds()[i].modulus.defined()) {
87 user_warning << "It is meaningless to align the bounds of dimension "
88 << func_s.bounds()[i].var << " of function "
89 << f.name() << " to have modulus/remainder ["
90 << func_s.bounds()[i].modulus << ", "
91 << func_s.bounds()[i].remainder << "] because the function is scheduled inline.\n";
92 }
93 }
94}

96class Inliner : public IRMutator {
97 using IRMutator::visit;
98
99 Function func;
100
101 Expr visit(const Call *op) override {
102 if (op->name == func.name()) {
103
104 // Mutate the args
105 vector<Expr> args(op->args.size());
106 for (size_t i = 0; i < args.size(); i++) {
107 args[i] = mutate(op->args[i]);
108 }
109 // Grab the body
110 Expr body = qualify(func.name() + ".", func.values()[op->value_index]);
111
112 const vector<string> func_args = func.args();
113
114 // Bind the args using Let nodes
115 internal_assert(args.size() == func_args.size());
116
117 for (size_t i = 0; i < args.size(); i++) {
118 if (is_const(args[i]) || args[i].as<Variable>()) {
119 body = substitute(func.name() + "." + func_args[i], args[i], body);
120 } else {
121 body = Let::make(func.name() + "." + func_args[i], args[i], body);
122 }
123 }
124
125 found++;
126
127 return body;
128
129 } else {
130 return IRMutator::visit(op);
131 }
132 }
133
134 Expr visit(const Variable *op) override {
135 if (op->name == func.name() + ".buffer") {
136 const Call *call = func.is_wrapper();
137 internal_assert(call);
138 // Do a whole-image inline. Substitute the .buffer symbol
139 // for the wrapped object's .buffer symbol.
140 string buf_name;
141 if (call->call_type == Call::Halide) {
142 buf_name = call->name;
143 if (Function(call->func).outputs() > 1) {
144 buf_name += "." + std::to_string(call->value_index);
145 }
146 buf_name += ".buffer";
147 return Variable::make(type_of<halide_buffer_t *>(), buf_name);
148 } else if (call->param.defined()) {
149 return Variable::make(type_of<halide_buffer_t *>(), call->name + ".buffer", call->param);
150 } else {
151 internal_assert(call->image.defined());
152 return Variable::make(type_of<halide_buffer_t *>(), call->name + ".buffer", call->image);
153 }
154 } else {
155 return op;
156 }
157 }
158
159 Stmt visit(const Provide *op) override {
160 ScopedValue<int> old_found(found, 0);
161 Stmt stmt = IRMutator::visit(op);
162
163 // TODO: making this > 1 should be desirable,
164 // but explodes compiletimes in some situations.
165 if (found > 0) {
166 stmt = common_subexpression_elimination(stmt);
167 }
168
169 return stmt;
170 }
171
172public:
173 int found = 0;
174
175 Inliner(const Function &f)
176 : func(f) {
177 internal_assert(f.can_be_inlined()) << "Illegal to inline " << f.name() << "\n";
178 validate_schedule_inlined_function(f);
179 }
180};

182Stmt inline_function(Stmt s, const Function &f) {
183 Inliner i(f);
184 s = i.mutate(s);
185 return s;
186}
187
188Expr inline_function(Expr e, const Function &f) {
189 Inliner i(f);
190 e = i.mutate(e);
191 // TODO: making this > 1 should be desirable,
192 // but explodes compiletimes in some situations.
193 if (i.found > 0) {
194 e = common_subexpression_elimination(e);
195 }
196 return e;
197}

199// Inline all calls to 'f' inside 'caller'
200void inline_function(Function caller, const Function &f) {
201 Inliner i(f);
202 caller.mutate(&i);
203 if (caller.has_extern_definition()) {
204 for (ExternFuncArgument &arg : caller.extern_arguments()) {
205 if (arg.is_func() && arg.func.same_as(f.get_contents())) {
206 const Call *call = f.is_wrapper();
207 internal_assert(call);
208 arg.func = call->func;
209 }
210 }
211 }
