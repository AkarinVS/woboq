31class CountGPUBlocksThreads : public IRVisitor {
32 string prefix; // Producer name + stage
33
34 using IRVisitor::visit;
35
36 void visit(const For *op) override {
37 if (starts_with(op->name, prefix)) {
38 if (op->for_type == ForType::GPUBlock) {
39 nblocks++;
40 } else if (op->for_type == ForType::GPUThread) {
41 nthreads++;
42 } else if (op->for_type == ForType::GPULane) {
43 nlanes++;
44 }
45 }
46 IRVisitor::visit(op);
47 }
48
49 void visit(const IfThenElse *op) override {
50 op->condition.accept(this);
51
52 int old_nblocks = nblocks;
53 int old_nthreads = nthreads;
54 int old_nlanes = nlanes;
55 op->then_case.accept(this);
56
57 if (op->else_case.defined()) {
58 int then_nblocks = nblocks;
59 int then_nthreads = nthreads;
60 int then_nlanes = nlanes;
61 nblocks = old_nblocks;
62 nthreads = old_nthreads;
63 nlanes = old_nlanes;
64 op->else_case.accept(this);
65 nblocks = std::max(then_nblocks, nblocks);
66 nthreads = std::max(then_nthreads, nthreads);
67 nlanes = std::max(then_nlanes, nlanes);
68 }
69 }
70
71public:
72 CountGPUBlocksThreads(const string &p)
73 : prefix(p) {
74 }
75 int nblocks = 0;
76 int nthreads = 0;
77 int nlanes = 0;
78};

80class CanonicalizeGPUVars : public IRMutator {
81 map<string, string> gpu_vars;
82
83 using IRMutator::visit;
84
85 string gpu_name(vector<string> v, const string &new_var) {
86 v.push_back(new_var);
87
88 std::ostringstream stream;
89 for (size_t i = 0; i < v.size(); ++i) {
90 stream << v[i];
91 if (i != v.size() - 1) {
92 stream << ".";
93 }
94 }
95 return stream.str();
96 }
97
98 string find_replacement(const string &suffix, const string &name) {
99 vector<string> v = split_string(name, suffix);
100 internal_assert(v.size() == 2);
101 const auto &iter = gpu_vars.find(v[0]);
102 if (iter != gpu_vars.end()) {
103 return iter->second + suffix;
104 }
105 return name;
106 }
107
108 string canonicalize_let(const string &name) {
109 if (ends_with(name, ".loop_max")) {
110 return find_replacement(".loop_max", name);
111 } else if (ends_with(name, ".loop_min")) {
112 return find_replacement(".loop_min", name);
113 } else if (ends_with(name, ".loop_extent")) {
114 return find_replacement(".loop_extent", name);
115 } else {
116 return name;
117 }
118 }
119
120 Stmt visit(const For *op) override {
121 string name = op->name;
122 Expr min = mutate(op->min);
123 Expr extent = mutate(op->extent);
124 Stmt body = mutate(op->body);
125
126 if ((op->for_type == ForType::GPUBlock) ||
127 (op->for_type == ForType::GPUThread) ||
128 (op->for_type == ForType::GPULane)) {
129
130 vector<string> v = split_string(op->name, ".");
131 internal_assert(v.size() > 2);
132
133 CountGPUBlocksThreads counter(v[0] + "." + v[1]);
134 op->body.accept(&counter);
135 internal_assert(counter.nblocks <= 4)
136 << op->name << " can only have maximum of 4 block dimensions\n";
137 internal_assert(counter.nthreads <= 4)
138 << op->name << " can only have maximum of 4 thread dimensions\n";
139
140 if (op->for_type == ForType::GPUBlock) {
141 name = gpu_name(v, get_block_name(counter.nblocks));
142 debug(5) << "Replacing " << op->name << " with GPU block name " << name << "\n";
143 } else if (op->for_type == ForType::GPUThread) {
144 name = gpu_name(v, get_thread_name(counter.nlanes + counter.nthreads));
145 debug(5) << "Replacing " << op->name << " with GPU thread name " << name << "\n";
146 } else if (op->for_type == ForType::GPULane) {
147 user_assert(counter.nlanes == 0) << "Cannot nest multiple loops over gpu lanes: " << name << "\n";
148 name = gpu_name(v, get_thread_name(0));
149 }
150
151 if (name != op->name) {
152 // Canonicalize the GPU for loop name
153 gpu_vars.emplace(op->name, name);
154 Expr new_var = Variable::make(Int(32), name);
155 min = substitute(op->name, new_var, min);
156 extent = substitute(op->name, new_var, extent);
157 body = substitute(op->name, new_var, body);
158 }
159 }
160
161 if ((name == op->name) &&
162 min.same_as(op->min) &&
163 extent.same_as(op->extent) &&
164 body.same_as(op->body)) {
165 return op;
166 } else {
167 return For::make(name, min, extent, op->for_type, op->device_api, body);
168 }
169 }
170
171 Stmt visit(const LetStmt *op) override {
172 vector<std::pair<string, Expr>> lets;
173 Stmt result;
174
175 do {
176 lets.emplace_back(op->name, mutate(op->value));
177 result = op->body;
178 } while ((op = op->body.as<LetStmt>()));
179
180 result = mutate(result);
181
182 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
183 string name = canonicalize_let(it->first);
184 if (name != it->first) {
185 Expr new_var = Variable::make(Int(32), name);
186 result = substitute(it->first, new_var, result);
187 }
188 result = LetStmt::make(name, it->second, result);
189 }
190
191 return result;
192 }
193
194 Stmt visit(const IfThenElse *op) override {
195 Expr condition = mutate(op->condition);
196
197 map<string, string> old_gpu_vars;
198 old_gpu_vars.swap(gpu_vars);
199 Stmt then_case = mutate(op->then_case);
200
201 gpu_vars = old_gpu_vars;
202 Stmt else_case = mutate(op->else_case);
203
204 if (condition.same_as(op->condition) &&
205 then_case.same_as(op->then_case) &&
206 else_case.same_as(op->else_case)) {
207 return op;
208 } else {
209 return IfThenElse::make(condition, then_case, else_case);
210 }
211 }
212};
