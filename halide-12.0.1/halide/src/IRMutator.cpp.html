41template<typename T>
42Expr mutate_binary_operator(IRMutator *mutator, const T *op) {
43 Expr a = mutator->mutate(op->a);
44 Expr b = mutator->mutate(op->b);
45 if (a.same_as(op->a) &&
46 b.same_as(op->b)) {
47 return op;
48 }
49 return T::make(std::move(a), std::move(b));
50}

148Expr IRMutator::visit(const Call *op) {
149 vector<Expr> new_args(op->args.size());
150 bool changed = false;
151
152 // Mutate the args
153 for (size_t i = 0; i < op->args.size(); i++) {
154 const Expr &old_arg = op->args[i];
155 Expr new_arg = mutate(old_arg);
156 if (!new_arg.same_as(old_arg)) {
157 changed = true;
158 }
159 new_args[i] = std::move(new_arg);
160 }
161
162 if (!changed) {
163 return op;
164 }
165 return Call::make(op->type, op->name, new_args, op->call_type,
166 op->func, op->value_index, op->image, op->param);
167}

230Stmt IRMutator::visit(const Provide *op) {
231 vector<Expr> new_args(op->args.size());
232 vector<Expr> new_values(op->values.size());
233 bool changed = false;
234
235 // Mutate the args
236 for (size_t i = 0; i < op->args.size(); i++) {
237 const Expr &old_arg = op->args[i];
238 Expr new_arg = mutate(old_arg);
239 if (!new_arg.same_as(old_arg)) {
240 changed = true;
241 }
242 new_args[i] = new_arg;
243 }
244
245 for (size_t i = 0; i < op->values.size(); i++) {
246 const Expr &old_value = op->values[i];
247 Expr new_value = mutate(old_value);
248 if (!new_value.same_as(old_value)) {
249 changed = true;
250 }
251 new_values[i] = new_value;
252 }
253
254 if (!changed) {
255 return op;
256 }
257 return Provide::make(op->name, new_values, new_args);
258}

260Stmt IRMutator::visit(const Allocate *op) {
261 std::vector<Expr> new_extents;
262 bool all_extents_unmodified = true;
263 for (size_t i = 0; i < op->extents.size(); i++) {
264 new_extents.push_back(mutate(op->extents[i]));
265 all_extents_unmodified &= new_extents[i].same_as(op->extents[i]);
266 }
267 Stmt body = mutate(op->body);
268 Expr condition = mutate(op->condition);
269 Expr new_expr;
270 if (op->new_expr.defined()) {
271 new_expr = mutate(op->new_expr);
272 }
273 if (all_extents_unmodified &&
274 body.same_as(op->body) &&
275 condition.same_as(op->condition) &&
276 new_expr.same_as(op->new_expr)) {
277 return op;
278 }
279 return Allocate::make(op->name, op->type, op->memory_type,
280 new_extents, std::move(condition),
281 std::move(body), std::move(new_expr), op->free_function);
282}

288Stmt IRMutator::visit(const Realize *op) {
289 Region new_bounds;
290 bool bounds_changed;
291
292 // Mutate the bounds
293 std::tie(new_bounds, bounds_changed) = mutate_region(this, op->bounds);
294
295 Stmt body = mutate(op->body);
296 Expr condition = mutate(op->condition);
297 if (!bounds_changed &&
298 body.same_as(op->body) &&
299 condition.same_as(op->condition)) {
300 return op;
301 }
302 return Realize::make(op->name, op->types, op->memory_type, new_bounds,
303 std::move(condition), std::move(body));
304}

417Stmt IRGraphMutator::mutate(const Stmt &s) {
418 auto p = stmt_replacements.emplace(s, Stmt());
419 if (p.second) {
420 // N.B: Inserting into a map (as the recursive mutate call
421 // does), does not invalidate existing iterators.
422 p.first->second = IRMutator::mutate(s);
423 }
424 return p.first->second;
425}

427Expr IRGraphMutator::mutate(const Expr &e) {
428 auto p = expr_replacements.emplace(e, Expr());
429 if (p.second) {
430 p.first->second = IRMutator::mutate(e);
431 }
432 return p.first->second;
