18class UniquifyVariableNames : public IRMutator {
19
20 using IRMutator::visit;
21
22 // The mapping from old names to new names
23 Scope<string> renaming;
24
25 // Get a new previously unused name for a let binding or for loop,
26 // and push it onto the renaming. Will return the original name if
27 // possible, but pushes unconditionally to simplify cleanup.
28 string make_new_name(const string &base) {
29 if (!renaming.contains(base)) {
30 renaming.push(base, base);
31 return base;
32 }
33 for (size_t i = std::max((size_t)1, renaming.count(base));; i++) {
34 string candidate = base + "_" + std::to_string(i);
35 if (!renaming.contains(candidate)) {
36 // Reserve this name for this base name
37 renaming.push(base, candidate);
38 // And reserve the generated name forever more (will not be popped)
39 renaming.push(candidate, candidate);
40 return candidate;
41 }
42 }
43 }
44
45 template<typename LetOrLetStmt>
46 auto visit_let(const LetOrLetStmt *op) -> decltype(op->body) {
47 struct Frame {
48 const LetOrLetStmt *op;
49 Expr value;
50 string new_name;
51 };
52
53 vector<Frame> frames;
54 decltype(op->body) result;
55 while (op) {
56 frames.emplace_back();
57 auto &f = frames.back();
58 f.op = op;
59 f.value = mutate(op->value);
60 f.new_name = make_new_name(op->name);
61 result = op->body;
62 op = result.template as<LetOrLetStmt>();
63 }
64
65 result = mutate(result);
66
67 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
68 renaming.pop(it->op->name);
69 if (it->new_name == it->op->name &&
70 result.same_as(it->op->body) &&
71 it->op->value.same_as(it->value)) {
72 result = it->op;
73 } else {
74 result = LetOrLetStmt::make(it->new_name, it->value, result);
75 }
76 }
77
78 return result;
79 }
80
81 Stmt visit(const LetStmt *op) override {
82 return visit_let(op);
83 }
84
85 Expr visit(const Let *op) override {
86 return visit_let(op);
87 }
88
89 Stmt visit(const For *op) override {
90 Expr min = mutate(op->min);
91 Expr extent = mutate(op->extent);
92 string new_name = make_new_name(op->name);
93 Stmt body = mutate(op->body);
94 renaming.pop(op->name);
95
96 if (new_name == op->name &&
97 body.same_as(op->body) &&
98 min.same_as(op->min) &&
99 extent.same_as(op->extent)) {
100 return op;
101 } else {
102 return For::make(new_name, min, extent, op->for_type, op->device_api, body);
103 }
104 }
105
106 Expr visit(const Variable *op) override {
107 if (renaming.contains(op->name)) {
108 string new_name = renaming.get(op->name);
109 if (new_name != op->name) {
110 return Variable::make(op->type, new_name);
111 }
112 }
113 return op;
114 }
115
116public:
117 UniquifyVariableNames(const Scope<string> *free_vars) {
118 renaming.set_containing_scope(free_vars);
119 }
120};

122class FindFreeVars : public IRVisitor {
123
124 using IRVisitor::visit;
125
126 Scope<> scope;
127
128 void visit(const Variable *op) override {
129 if (!scope.contains(op->name)) {
130 free_vars.push(op->name, op->name);
131 }
132 }
133
134 void visit(const Let *op) override {
135 op->value.accept(this);
136 {
137 ScopedBinding<> bind(scope, op->name);
138 op->body.accept(this);
139 }
140 }
141
142 void visit(const LetStmt *op) override {
143 op->value.accept(this);
144 {
145 ScopedBinding<> bind(scope, op->name);
146 op->body.accept(this);
147 }
148 }
149
150 void visit(const For *op) override {
151 op->min.accept(this);
152 op->extent.accept(this);
153 {
154 ScopedBinding<> bind(scope, op->name);
155 op->body.accept(this);
156 }
157 }
158
159public:
160 Scope<string> free_vars;
161};

164Stmt uniquify_variable_names(const Stmt &s) {
165 FindFreeVars finder;
166 s.accept(&finder);
167 UniquifyVariableNames u(&finder.free_vars);
168 return u.mutate(s);
169}

171void check(vector<pair<Var, Expr>> in,
172 vector<pair<Var, Expr>> out) {
173 Stmt in_stmt = Evaluate::make(0), out_stmt = Evaluate::make(0);
174 for (auto it = in.rbegin(); it != in.rend(); it++) {
175 in_stmt = LetStmt::make(it->first.name(), it->second, in_stmt);
176 }
177 for (auto it = out.rbegin(); it != out.rend(); it++) {
178 out_stmt = LetStmt::make(it->first.name(), it->second, out_stmt);
179 }
180
181 Stmt s = uniquify_variable_names(in_stmt);
182
183 internal_assert(equal(s, out_stmt))
184 << "Failure in uniquify_variable_names\n"
185 << "Input:\n"
186 << in_stmt << "\n"
187 << "Produced:\n"
188 << s << "\n"
189 << "Correct output:\n"
190 << out_stmt << "\n";
191}

193void uniquify_variable_names_test() {
194 Var x("x"), x_1("x_1"), x_2("x_2"), x_3{"x_3"};
195 Var y("y"), y_1("y_1"), y_2("y_2"), y_3{"y_3"};
196
197 // Stmts with all names already unique should be unchanged
198 check({{x, 3},
199 {y, x}},
200 {{x, 3},
201 {y, x}});
202
203 // Shadowed definitions of Vars should be given unique names
204 check({{x, 3},
205 {y, x},
206 {x, x + y},
207 {y, x + y},
208 {x, x + y},
209 {y, x + y}},
210 {{x, 3},
211 {y, x},
212 {x_1, x + y},
213 {y_1, x_1 + y},
214 {x_2, x_1 + y_1},
215 {y_2, x_2 + y_1}});
216
217 // Check a case with a free var after then end of the scope of a let of the same name
218 check({{x, Let::make(y.name(), 3, y)}, // y is bound
219 {x, y}}, // This is not the same y. It's free and can't be renamed.
220 {{x, Let::make(y_1.name(), 3, y_1)}, // We rename the bound one
221 {x_1, y}});
222
223 // An existing in-scope use of one of the names that would be
224 // autogenerated should be skipped over
225 check({{x_1, 8},
226 {x, 3},
227 {y, x},
228 {x, x + y},
229 {y, x + y},
230 {x, x + y},
231 {y, x + y}},
232 {{x_1, 8},
233 {x, 3},
234 {y, x},
235 {x_2, x + y},
236 {y_1, x_2 + y},
237 {x_3, x_2 + y_1},
238 {y_2, x_3 + y_1}});
239
240 // Check parallel bindings. The scope doesn't overlap so they can keep their name
241 check({{x, Let::make(y.name(), 3, y)},
242 {x, Let::make(y.name(), 4, y)}},
243 {{x, Let::make(y.name(), 3, y)},
244 {x_1, Let::make(y.name(), 4, y)}});
245
246 std::cout << "uniquify_variable_names test passed" << std::endl;
247}
