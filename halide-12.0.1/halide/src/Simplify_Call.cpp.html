21int popcount64(uint64_t x) {
22#ifdef _MSC_VER
23#if defined(_WIN64)
24 return __popcnt64(x);
25#else
26 return __popcnt((uint32_t)(x >> 32)) + __popcnt((uint32_t)(x & 0xffffffff));
27#endif
28#else
29 static_assert(sizeof(unsigned long long) >= sizeof(uint64_t), "");
30 return __builtin_popcountll(x);
31#endif
32}

34int clz64(uint64_t x) {
35 internal_assert(x != 0);
36#ifdef _MSC_VER
37 unsigned long r = 0;
38#if defined(_WIN64)
39 return _BitScanReverse64(&r, x) ? (63 - r) : 64;
40#else
41 if (_BitScanReverse(&r, (uint32_t)(x >> 32))) {
42 return (63 - (r + 32));
43 } else if (_BitScanReverse(&r, (uint32_t)(x & 0xffffffff))) {
44 return 63 - r;
45 } else {
46 return 64;
47 }
48#endif
49#else
50 static_assert(sizeof(unsigned long long) >= sizeof(uint64_t), "");
51 constexpr int offset = (sizeof(unsigned long long) - sizeof(uint64_t)) * 8;
52 return __builtin_clzll(x) + offset;
53#endif
54}

56int ctz64(uint64_t x) {
57 internal_assert(x != 0);
58#ifdef _MSC_VER
59 unsigned long r = 0;
60#if defined(_WIN64)
61 return _BitScanForward64(&r, x) ? r : 64;
62#else
63 if (_BitScanForward(&r, (uint32_t)(x & 0xffffffff))) {
64 return r;
65 } else if (_BitScanForward(&r, (uint32_t)(x >> 32))) {
66 return r + 32;
67 } else {
68 return 64;
69 }
70#endif
71#else
72 static_assert(sizeof(unsigned long long) >= sizeof(uint64_t), "");
73 return __builtin_ctzll(x);
74#endif
75}

77// Rewrite name(broadcast(args)) to broadcast(name(args)).
78// Assumes that scalars are implicitly broadcast.
79Expr lift_elementwise_broadcasts(Type type, const std::string &name, std::vector<Expr> args, Call::CallType call_type) {
80 if (type.lanes() == 1) {
81 return Expr();
82 }
83 int lanes = 0;
84 for (Expr &i : args) {
85 if (const Broadcast *b = i.as<Broadcast>()) {
86 i = b->value;
87 if (lanes == 0) {
88 lanes = i.type().lanes();
89 } else if (lanes != i.type().lanes()) {
90 // This is a broadcast of another vector, and does not match another vector argument.
91 return Expr();
92 }
93 } else if (!i.type().is_scalar()) {
94 // This is not a scalar or broadcast scalar, we can't lift broadcasts.
95 return Expr();
96 }
97 }
98 if (lanes != type.lanes()) {
99 return Broadcast::make(Call::make(type.with_lanes(lanes), name, args, call_type), type.lanes() / lanes);
100 } else {
101 return Expr();
102 }
103}

107Expr Simplify::visit(const Call *op, ExprInfo *bounds) {
108 // Calls implicitly depend on host, dev, mins, and strides of the buffer referenced
109 if (op->call_type == Call::Image || op->call_type == Call::Halide) {
110 found_buffer_reference(op->name, op->args.size());
111 }
112
113 if (op->is_intrinsic(Call::unreachable)) {
114 in_unreachable = true;
115 return op;
116 } else if (op->is_intrinsic(Call::strict_float)) {
117 if (Call::as_intrinsic(op->args[0], {Call::strict_float})) {
118 // Always simplify strict_float(strict_float(x)) -> strict_float(x).
119 Expr arg = mutate(op->args[0], nullptr);
120 return arg.same_as(op->args[0]) ? op->args[0] : arg;
121 } else {
122 ScopedValue<bool> save_no_float_simplify(no_float_simplify, true);
123 Expr arg = mutate(op->args[0], nullptr);
124 if (arg.same_as(op->args[0])) {
125 return op;
126 } else {
127 return strict_float(arg);
128 }
129 }
130 } else if (op->is_intrinsic(Call::popcount) ||
131 op->is_intrinsic(Call::count_leading_zeros) ||
132 op->is_intrinsic(Call::count_trailing_zeros)) {
133 Expr a = mutate(op->args[0], nullptr);
134
135 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a}, op->call_type);
136 if (unbroadcast.defined()) {
137 return mutate(unbroadcast, bounds);
138 }
139
140 uint64_t ua = 0;
141 if (const_int(a, (int64_t *)(&ua)) || const_uint(a, &ua)) {
142 const int bits = op->type.bits();
143 const uint64_t mask = std::numeric_limits<uint64_t>::max() >> (64 - bits);
144 ua &= mask;
145 static_assert(sizeof(unsigned long long) >= sizeof(uint64_t), "");
146 int r = 0;
147 if (op->is_intrinsic(Call::popcount)) {
148 // popcount *is* well-defined for ua = 0
149 r = popcount64(ua);
150 } else if (op->is_intrinsic(Call::count_leading_zeros)) {
151 // clz64() is undefined for 0, but Halide's count_leading_zeros defines clz(0) = bits
152 r = ua == 0 ? bits : (clz64(ua) - (64 - bits));
153 } else /* if (op->is_intrinsic(Call::count_trailing_zeros)) */ {
154 // ctz64() is undefined for 0, but Halide's count_trailing_zeros defines clz(0) = bits
155 r = ua == 0 ? bits : (ctz64(ua));
156 }
157 return make_const(op->type, r);
158 }
159
160 if (a.same_as(op->args[0])) {
161 return op;
162 } else {
163 return Call::make(op->type, op->name, {std::move(a)}, Internal::Call::PureIntrinsic);
164 }
165
166 } else if (op->is_intrinsic(Call::shift_left) ||
167 op->is_intrinsic(Call::shift_right)) {
168 Expr a = mutate(op->args[0], nullptr);
169 // TODO: When simplifying b, it would be nice to specify the min/max useful bounds, so
170 // stronger simplifications could occur. For example, x >> min(-i8, 0) should be simplified
171 // to x >> -max(i8, 0) and then x << max(i8, 0). This isn't safe because -i8 can overflow.
172 ExprInfo b_info;
173 Expr b = mutate(op->args[1], &b_info);
174
175 if (is_const_zero(b)) {
176 return a;
177 }
178
179 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a, b}, op->call_type);
180 if (unbroadcast.defined()) {
181 return mutate(unbroadcast, bounds);
182 }
183
184 const Type t = op->type;
185
186 // We might swap from a right to left shift or the reverse.
187 std::string result_op = op->name;
188
189 // If we know the sign of this shift, change it to an unsigned shift.
190 if (b_info.min_defined && b_info.min >= 0) {
191 b = mutate(cast(b.type().with_code(halide_type_uint), b), nullptr);
192 } else if (b_info.max_defined && b_info.max <= 0) {
193 result_op = Call::get_intrinsic_name(op->is_intrinsic(Call::shift_right) ? Call::shift_left : Call::shift_right);
194 b = mutate(cast(b.type().with_code(halide_type_uint), -b), nullptr);
195 }
196
197 // If the shift is by a constant, it should now be unsigned.
198 uint64_t ub = 0;
199 if (const_uint(b, &ub)) {
200 // LLVM shl and shr instructions produce poison for
201 // shifts >= typesize, so we will follow suit in our simplifier.
202 if (ub >= (uint64_t)(t.bits())) {
203 return make_signed_integer_overflow(t);
204 }
205 if (a.type().is_uint() || ub < ((uint64_t)t.bits() - 1)) {
206 b = make_const(t, ((int64_t)1LL) << ub);
207 if (result_op == Call::get_intrinsic_name(Call::shift_left)) {
208 return mutate(Mul::make(a, b), bounds);
209 } else {
210 return mutate(Div::make(a, b), bounds);
211 }
212 } else {
213 // For signed types, (1 << (t.bits() - 1)) will overflow into the sign bit while
214 // (-32768 >> (t.bits() - 1)) propagates the sign bit, making decomposition
215 // into mul or div problematic, so just special-case them here.
216 if (result_op == Call::get_intrinsic_name(Call::shift_left)) {
217 return mutate(select((a & 1) != 0, make_const(t, ((int64_t)1LL) << ub), make_zero(t)), bounds);
218 } else {
219 return mutate(select(a < 0, make_const(t, -1), make_zero(t)), bounds);
220 }
221 }
222 }
223
224 // Rewrite shifts with negated RHSes as shifts of the other direction.
225 if (const Sub *sub = b.as<Sub>()) {
226 if (is_const_zero(sub->a)) {
227 result_op = Call::get_intrinsic_name(op->is_intrinsic(Call::shift_right) ? Call::shift_left : Call::shift_right);
228 b = sub->b;
229 return mutate(Call::make(op->type, result_op, {a, b}, Call::PureIntrinsic), bounds);
230 }
231 }
232
233 if (a.same_as(op->args[0]) && b.same_as(op->args[1])) {
234 internal_assert(result_op == op->name);
235 return op;
236 } else {
237 return Call::make(op->type, result_op, {a, b}, Call::PureIntrinsic);
238 }
239 } else if (op->is_intrinsic(Call::bitwise_and)) {
240 Expr a = mutate(op->args[0], nullptr);
241 Expr b = mutate(op->args[1], nullptr);
242
243 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a, b}, op->call_type);
244 if (unbroadcast.defined()) {
245 return mutate(unbroadcast, bounds);
246 }
247
248 int64_t ia, ib = 0;
249 uint64_t ua, ub = 0;
250 int bits;
251
252 if (const_int(a, &ia) &&
253 const_int(b, &ib)) {
254 return make_const(op->type, ia & ib);
255 } else if (const_uint(a, &ua) &&
256 const_uint(b, &ub)) {
257 return make_const(op->type, ua & ub);
258 } else if (const_int(b, &ib) &&
259 !b.type().is_max(ib) &&
260 is_const_power_of_two_integer(make_const(a.type(), ib + 1), &bits)) {
261 return Mod::make(a, make_const(a.type(), ib + 1));
262 } else if (const_uint(b, &ub) &&
263 b.type().is_max(ub)) {
264 return a;
265 } else if (const_int(b, &ib) &&
266 ib == -1) {
267 return a;
268 } else if (const_uint(b, &ub) &&
269 is_const_power_of_two_integer(make_const(a.type(), ub + 1), &bits)) {
270 return Mod::make(a, make_const(a.type(), ub + 1));
271 } else if (a.same_as(op->args[0]) && b.same_as(op->args[1])) {
272 return op;
273 } else {
274 return a & b;
275 }
276 } else if (op->is_intrinsic(Call::bitwise_or)) {
277 Expr a = mutate(op->args[0], nullptr);
278 Expr b = mutate(op->args[1], nullptr);
279
280 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a, b}, op->call_type);
281 if (unbroadcast.defined()) {
282 return mutate(unbroadcast, bounds);
283 }
284
285 int64_t ia, ib;
286 uint64_t ua, ub;
287 if (const_int(a, &ia) &&
288 const_int(b, &ib)) {
289 return make_const(op->type, ia | ib);
290 } else if (const_uint(a, &ua) &&
291 const_uint(b, &ub)) {
292 return make_const(op->type, ua | ub);
293 } else if (a.same_as(op->args[0]) && b.same_as(op->args[1])) {
294 return op;
295 } else {
296 return a | b;
297 }
298 } else if (op->is_intrinsic(Call::bitwise_not)) {
299 Expr a = mutate(op->args[0], nullptr);
300
301 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a}, op->call_type);
302 if (unbroadcast.defined()) {
303 return mutate(unbroadcast, bounds);
304 }
305
306 int64_t ia;
307 uint64_t ua;
308 if (const_int(a, &ia)) {
309 return make_const(op->type, ~ia);
310 } else if (const_uint(a, &ua)) {
311 return make_const(op->type, ~ua);
312 } else if (a.same_as(op->args[0])) {
313 return op;
314 } else {
315 return ~a;
316 }
317 } else if (op->is_intrinsic(Call::bitwise_xor)) {
318 Expr a = mutate(op->args[0], nullptr);
319 Expr b = mutate(op->args[1], nullptr);
320
321 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a, b}, op->call_type);
322 if (unbroadcast.defined()) {
323 return mutate(unbroadcast, bounds);
324 }
325
326 int64_t ia, ib;
327 uint64_t ua, ub;
328 if (const_int(a, &ia) &&
329 const_int(b, &ib)) {
330 return make_const(op->type, ia ^ ib);
331 } else if (const_uint(a, &ua) &&
332 const_uint(b, &ub)) {
333 return make_const(op->type, ua ^ ub);
334 } else if (a.same_as(op->args[0]) && b.same_as(op->args[1])) {
335 return op;
336 } else {
337 return a ^ b;
338 }
339 } else if (op->is_intrinsic(Call::reinterpret)) {
340 Expr a = mutate(op->args[0], nullptr);
341
342 int64_t ia;
343 uint64_t ua;
344 bool vector = op->type.is_vector() || a.type().is_vector();
345 if (op->type == a.type()) {
346 return a;
347 } else if (const_int(a, &ia) && op->type.is_uint() && !vector) {
348 // int -> uint
349 return make_const(op->type, (uint64_t)ia);
350 } else if (const_uint(a, &ua) && op->type.is_int() && !vector) {
351 // uint -> int
352 return make_const(op->type, (int64_t)ua);
353 } else if (a.same_as(op->args[0])) {
354 return op;
355 } else {
356 return reinterpret(op->type, a);
357 }
358 } else if (op->is_intrinsic(Call::abs)) {
359 // Constant evaluate abs(x).
360 ExprInfo a_bounds;
361 Expr a = mutate(op->args[0], &a_bounds);
362
363 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a}, op->call_type);
364 if (unbroadcast.defined()) {
365 return mutate(unbroadcast, bounds);
366 }
367
368 Type ta = a.type();
369 int64_t ia = 0;
370 double fa = 0;
371 if (ta.is_int() && const_int(a, &ia)) {
372 if (ia < 0 && !(Int(64).is_min(ia))) {
373 ia = -ia;
374 }
375 return make_const(op->type, ia);
376 } else if (ta.is_uint()) {
377 // abs(uint) is a no-op.
378 return a;
379 } else if (const_float(a, &fa)) {
380 if (fa < 0) {
381 fa = -fa;
382 }
383 return make_const(a.type(), fa);
384 } else if (a.type().is_int() && a_bounds.min_defined && a_bounds.min >= 0) {
385 return cast(op->type, a);
386 } else if (a.type().is_int() && a_bounds.max_defined && a_bounds.max <= 0) {
387 return cast(op->type, -a);
388 } else if (a.same_as(op->args[0])) {
389 return op;
390 } else {
391 return abs(a);
392 }
393 } else if (op->is_intrinsic(Call::absd)) {
394 // Constant evaluate absd(a, b).
395 ExprInfo a_bounds, b_bounds;
396 Expr a = mutate(op->args[0], &a_bounds);
397 Expr b = mutate(op->args[1], &b_bounds);
398
399 Expr unbroadcast = lift_elementwise_broadcasts(op->type, op->name, {a, b}, op->call_type);
400 if (unbroadcast.defined()) {
401 return mutate(unbroadcast, bounds);
402 }
403
404 Type ta = a.type();
405 // absd() should enforce identical types for a and b when the node is created
406 internal_assert(ta == b.type());
407
408 int64_t ia = 0, ib = 0;
409 uint64_t ua = 0, ub = 0;
410 double fa = 0, fb = 0;
411 if (ta.is_int() && const_int(a, &ia) && const_int(b, &ib)) {
412 // Note that absd(int, int) always produces a uint result
413 internal_assert(op->type.is_uint());
414 const uint64_t d = ia > ib ? (uint64_t)(ia - ib) : (uint64_t)(ib - ia);
415 return make_const(op->type, d);
416 } else if (ta.is_uint() && const_uint(a, &ua) && const_uint(b, &ub)) {
417 const uint64_t d = ua > ub ? ua - ub : ub - ua;
418 return make_const(op->type, d);
419 } else if (const_float(a, &fa) && const_float(b, &fb)) {
420 const double d = fa > fb ? fa - fb : fb - fa;
421 return make_const(op->type, d);
422 } else if (a.same_as(op->args[0]) && b.same_as(op->args[1])) {
423 return op;
424 } else {
425 return absd(a, b);
426 }
427 } else if (op->is_intrinsic(Call::stringify)) {
428 // Eagerly concat constant arguments to a stringify.
429 bool changed = false;
430 vector<Expr> new_args;
431 const StringImm *last = nullptr;
432 for (size_t i = 0; i < op->args.size(); i++) {
433 Expr arg = mutate(op->args[i], nullptr);
434 if (!arg.same_as(op->args[i])) {
435 changed = true;
436 }
437 const StringImm *string_imm = arg.as<StringImm>();
438 const IntImm *int_imm = arg.as<IntImm>();
439 const FloatImm *float_imm = arg.as<FloatImm>();
440 // We use snprintf here rather than stringstreams,
441 // because the runtime's float printing is guaranteed
442 // to match snprintf.
443 char buf[64]; // Large enough to hold the biggest float literal.
444 if (last && string_imm) {
445 new_args.back() = last->value + string_imm->value;
446 changed = true;
447 } else if (int_imm) {
448 snprintf(buf, sizeof(buf), "%lld", (long long)int_imm->value);
449 if (last) {
450 new_args.back() = last->value + buf;
451 } else {
452 new_args.emplace_back(string(buf));
453 }
454 changed = true;
455 } else if (last && float_imm) {
456 snprintf(buf, sizeof(buf), "%f", float_imm->value);
457 if (last) {
458 new_args.back() = last->value + buf;
459 } else {
460 new_args.emplace_back(string(buf));
461 }
462 changed = true;
463 } else {
464 new_args.push_back(arg);
465 }
466 last = new_args.back().as<StringImm>();
467 }
468
469 if (new_args.size() == 1 && new_args[0].as<StringImm>()) {
470 // stringify of a string constant is just the string constant
471 return new_args[0];
472 } else if (changed) {
473 return Call::make(op->type, op->name, new_args, op->call_type);
474 } else {
475 return op;
476 }
477 } else if (op->is_intrinsic(Call::prefetch)) {
478 // Collapse the prefetched region into lower dimension whenever is possible.
479 // TODO(psuriana): Deal with negative strides and overlaps.
480
481 internal_assert(op->args.size() % 2 == 0); // Format: {base, offset, extent0, min0, ...}
482
483 vector<Expr> args(op->args);
484 bool changed = false;
485 for (size_t i = 0; i < op->args.size(); ++i) {
486 args[i] = mutate(op->args[i], nullptr);
487 if (!args[i].same_as(op->args[i])) {
488 changed = true;
489 }
490 }
491
492 // The {extent, stride} args in the prefetch call are sorted
493 // based on the storage dimension in ascending order (i.e. innermost
494 // first and outermost last), so, it is enough to check for the upper
495 // triangular pairs to see if any contiguous addresses exist.
496 for (size_t i = 2; i < args.size(); i += 2) {
497 Expr extent_0 = args[i];
498 Expr stride_0 = args[i + 1];
499 for (size_t j = i + 2; j < args.size(); j += 2) {
500 Expr extent_1 = args[j];
501 Expr stride_1 = args[j + 1];
502
503 if (is_const_one(mutate(extent_0 * stride_0 == stride_1, nullptr))) {
504 Expr new_extent = mutate(extent_0 * extent_1, nullptr);
505 args.erase(args.begin() + j, args.begin() + j + 2);
506 args[i] = new_extent;
507 args[i + 1] = stride_0;
508 i -= 2;
509 break;
510 }
511 }
512 }
513 internal_assert(args.size() <= op->args.size());
514
515 if (changed || (args.size() != op->args.size())) {
516 return Call::make(op->type, Call::prefetch, args, Call::Intrinsic);
517 } else {
518 return op;
519 }
520 } else if (op->is_intrinsic(Call::require)) {
521 Expr cond = mutate(op->args[0], nullptr);
522 // likely(const-bool) is deliberately not reduced
523 // by the simplify(), but for our purposes here, we want
524 // to ignore the likely() wrapper. (Note that this is
525 // equivalent to calling can_prove() without needing to
526 // create a new Simplifier instance.)
527 if (const Call *c = cond.as<Call>()) {
528 if (c->is_intrinsic(Call::likely)) {
529 cond = c->args[0];
530 }
531 }
532
533 if (is_const_zero(cond)) {
534 // (We could simplify this to avoid evaluating the provably-false
535 // expression, but since this is a degenerate condition, don't bother.)
536 user_warning << "This pipeline is guaranteed to fail a require() expression at runtime: \n"
537 << Expr(op) << "\n";
538 }
539
540 Expr result;
541 {
542 // Can assume the condition is true when evaluating the value.
543 auto t = scoped_truth(cond);
544 result = mutate(op->args[1], bounds);
545 }
546
547 if (is_const_one(cond)) {
548 return result;
549 }
550
551 Expr message = mutate(op->args[2], nullptr);
552
553 if (cond.same_as(op->args[0]) &&
554 result.same_as(op->args[1]) &&
555 message.same_as(op->args[2])) {
556 return op;
557 } else {
558 return Internal::Call::make(op->type,
559 Internal::Call::require,
560 {std::move(cond), std::move(result), std::move(message)},
561 Internal::Call::PureIntrinsic);
562 }
563 } else if (op->is_intrinsic(Call::promise_clamped) ||
564 op->is_intrinsic(Call::unsafe_promise_clamped)) {
565 // If the simplifier can infer that the clamp is unnecessary,
566 // we should be good to discard the promise.
567 internal_assert(op->args.size() == 3);
568 ExprInfo arg_info, lower_info, upper_info;
569 Expr arg = mutate(op->args[0], &arg_info);
570 Expr lower = mutate(op->args[1], &lower_info);
571 Expr upper = mutate(op->args[2], &upper_info);
572
573 const Broadcast *b_arg = arg.as<Broadcast>();
574 const Broadcast *b_lower = lower.as<Broadcast>();
575 const Broadcast *b_upper = upper.as<Broadcast>();
576
577 if (arg_info.min_defined &&
578 arg_info.max_defined &&
579 lower_info.max_defined &&
580 upper_info.min_defined &&
581 arg_info.min >= lower_info.max &&
582 arg_info.max <= upper_info.min) {
583 return arg;
584 } else if (b_arg && b_lower && b_upper) {
585 // Move broadcasts outwards
586 return Broadcast::make(Call::make(b_arg->value.type(), op->name,
587 {b_arg->value, b_lower->value, b_upper->value},
588 Call::Intrinsic),
589 b_arg->lanes);
590 } else if (arg.same_as(op->args[0]) &&
591 lower.same_as(op->args[1]) &&
592 upper.same_as(op->args[2])) {
593 return op;
594
595 } else {
596 return Call::make(op->type, op->name,
597 {arg, lower, upper},
598 Call::Intrinsic);
599 }
600 } else if (Call::as_tag(op)) {
601 // The bounds of the result are the bounds of the arg
602 internal_assert(op->args.size() == 1);
603 Expr arg = mutate(op->args[0], bounds);
604 if (arg.same_as(op->args[0])) {
605 return op;
606 } else {
607 return Call::make(op->type, op->name, {arg}, op->call_type);
608 }
609 } else if (op->is_intrinsic(Call::if_then_else)) {
610 // Note that this call promises to evaluate exactly one of the conditions,
611 // so this optimization should be safe.
612
613 internal_assert(op->args.size() == 3);
614 Expr cond_value = mutate(op->args[0], nullptr);
615
616 // Ignore tags for our purposes here
617 Expr cond = unwrap_tags(cond_value);
618 if (in_unreachable) {
619 return op;
620 }
621
622 if (is_const_one(cond)) {
623 return mutate(op->args[1], bounds);
624 } else if (is_const_zero(cond)) {
625 return mutate(op->args[2], bounds);
626 } else {
627 Expr true_value = mutate(op->args[1], nullptr);
628 bool true_unreachable = in_unreachable;
629 in_unreachable = false;
630 Expr false_value = mutate(op->args[2], nullptr);
631 bool false_unreachable = in_unreachable;
632
633 if (true_unreachable && false_unreachable) {
634 return false_value;
635 }
636 in_unreachable = false;
637 if (true_unreachable) {
638 return false_value;
639 } else if (false_unreachable) {
640 return true_value;
641 }
642
643 if (cond_value.same_as(op->args[0]) &&
644 true_value.same_as(op->args[1]) &&
645 false_value.same_as(op->args[2])) {
646 return op;
647 } else {
648 return Internal::Call::make(op->type,
649 Call::if_then_else,
650 {std::move(cond_value), std::move(true_value), std::move(false_value)},
651 op->call_type);
652 }
653 }
654 } else if (op->is_intrinsic(Call::mux)) {
655 internal_assert(op->args.size() >= 2);
656 int num_values = (int)op->args.size() - 1;
657 if (num_values == 1) {
658 // Mux of a single value
659 return mutate(op->args[1], bounds);
660 }
661 ExprInfo index_info;
662 Expr index = mutate(op->args[0], &index_info);
663
664 // Check if the mux has statically resolved
665 if (index_info.min_defined &&
666 index_info.max_defined &&
667 index_info.min == index_info.max) {
668 if (index_info.min >= 0 && index_info.min < num_values) {
669 // In-range, return the (simplified) corresponding value.
670 return mutate(op->args[index_info.min + 1], bounds);
671 } else {
672 // It's out-of-range, so return the last value.
673 return mutate(op->args.back(), bounds);
674 }
675 }
676
677 // The logic above could be extended to also truncate the
678 // range of values in the case where the mux index has a
679 // constant bound. This seems unlikely to ever come up though.
680
681 bool unchanged = index.same_as(op->args[0]);
682 vector<Expr> mutated_args(op->args.size());
683 mutated_args[0] = index;
684 for (size_t i = 1; i < op->args.size(); ++i) {
685 mutated_args[i] = mutate(op->args[i], nullptr);
686 unchanged &= mutated_args[i].same_as(op->args[i]);
687 }
688
689 if (unchanged) {
690 return op;
691 } else {
692 return Call::make(op->type, Call::mux, mutated_args, op->call_type);
693 }
694 } else if (op->call_type == Call::PureExtern) {
695 // TODO: This could probably be simplified into a single map-lookup
696 // with a bit more cleverness; not sure if the reduced lookup time
697 // would pay for itself (in comparison with the possible lost code clarity).
698
699 // Handle all the PureExtern cases of float -> bool
700 {
701 using FnType = bool (*)(double);
702 // Some GCC versions are unable to resolve std::isnan (etc) directly, so
703 // wrap them in lambdas.
704 const FnType is_finite = [](double a) -> bool { return std::isfinite(a); };
705 const FnType is_inf = [](double a) -> bool { return std::isinf(a); };
706 const FnType is_nan = [](double a) -> bool { return std::isnan(a); };
707 static const std::unordered_map<std::string, FnType>
708 pure_externs_f1b = {
709 {"is_finite_f16", is_finite},
710 {"is_finite_f32", is_finite},
711 {"is_finite_f64", is_finite},
712 {"is_inf_f16", is_inf},
713 {"is_inf_f32", is_inf},
714 {"is_inf_f64", is_inf},
715 {"is_nan_f16", is_nan},
716 {"is_nan_f32", is_nan},
717 {"is_nan_f64", is_nan},
718 };
719 auto it = pure_externs_f1b.find(op->name);
720 if (it != pure_externs_f1b.end()) {
721 Expr arg = mutate(op->args[0], nullptr);
722 double f = 0.0;
723 if (const_float(arg, &f)) {
724 auto fn = it->second;
725 return make_bool(fn(f));
726 } else if (arg.same_as(op->args[0])) {
727 return op;
728 } else {
729 return Call::make(op->type, op->name, {arg}, op->call_type);
730 }
731 }
732 // else fall thru
733 }
734
735 // Handle all the PureExtern cases of float -> float
736 // TODO: should we handle the f16 and f64 cases here? (We never did before.)
737 // TODO: should we handle fast_inverse and/or fast_inverse_sqrt here?
738 {
739 using FnType = double (*)(double);
740 static const std::unordered_map<std::string, FnType>
741 pure_externs_f1 = {
742 {"acos_f32", std::acos},
743 {"acosh_f32", std::acosh},
744 {"asin_f32", std::asin},
745 {"asinh_f32", std::asinh},
746 {"atan_f32", std::atan},
747 {"atanh_f32", std::atanh},
748 {"cos_f32", std::cos},
749 {"cosh_f32", std::cosh},
750 {"exp_f32", std::exp},
751 {"log_f32", std::log},
752 {"sin_f32", std::sin},
753 {"sinh_f32", std::sinh},
754 {"sqrt_f32", std::sqrt},
755 {"tan_f32", std::tan},
756 {"tanh_f32", std::tanh},
757 };
758 auto it = pure_externs_f1.find(op->name);
759 if (it != pure_externs_f1.end()) {
760 Expr arg = mutate(op->args[0], nullptr);
761 if (const double *f = as_const_float(arg)) {
762 auto fn = it->second;
763 return make_const(arg.type(), fn(*f));
764 } else if (arg.same_as(op->args[0])) {
765 return op;
766 } else {
767 return Call::make(op->type, op->name, {arg}, op->call_type);
768 }
769 }
770 // else fall thru
771 }
772
773 // Handle all the PureExtern cases of float -> integerized-float
774 {
775 using FnType = double (*)(double);
776 static const std::unordered_map<std::string, FnType>
777 pure_externs_truncation = {
778 {"ceil_f32", std::ceil},
779 {"floor_f32", std::floor},
780 {"round_f32", std::nearbyint},
781 {"trunc_f32", [](double a) -> double { return (a < 0 ? std::ceil(a) : std::floor(a)); }},
782 };
783 auto it = pure_externs_truncation.find(op->name);
784 if (it != pure_externs_truncation.end()) {
785 internal_assert(op->args.size() == 1);
786 Expr arg = mutate(op->args[0], nullptr);
787
788 const Call *call = arg.as<Call>();
789 if (const double *f = as_const_float(arg)) {
790 auto fn = it->second;
791 return make_const(arg.type(), fn(*f));
792 } else if (call && call->call_type == Call::PureExtern &&
793 (it = pure_externs_truncation.find(call->name)) != pure_externs_truncation.end()) {
794 // For any combination of these integer-valued functions, we can
795 // discard the outer function. For example, floor(ceil(x)) == ceil(x).
796 return call;
797 } else if (!arg.same_as(op->args[0])) {
798 return Call::make(op->type, op->name, {arg}, op->call_type);
799 } else {
800 return op;
801 }
802 }
803 // else fall thru
804 }
805
806 // Handle all the PureExtern cases of (float, float) -> integerized-float
807 {
808 using FnType = double (*)(double, double);
809 static const std::unordered_map<std::string, FnType>
810 pure_externs_f2 = {
811 {"atan2_f32", std::atan2},
812 {"pow_f32", std::pow},
813 };
814 auto it = pure_externs_f2.find(op->name);
815 if (it != pure_externs_f2.end()) {
816 Expr arg0 = mutate(op->args[0], nullptr);
817 Expr arg1 = mutate(op->args[1], nullptr);
818
819 const double *f0 = as_const_float(arg0);
820 const double *f1 = as_const_float(arg1);
821 if (f0 && f1) {
822 auto fn = it->second;
823 return make_const(arg0.type(), fn(*f0, *f1));
824 } else if (!arg0.same_as(op->args[0]) || !arg1.same_as(op->args[1])) {
825 return Call::make(op->type, op->name, {arg0, arg1}, op->call_type);
826 } else {
827 return op;
828 }
829 }
830 // else fall thru
831 }
832
833 // There are other PureExterns we don't bother with (e.g. fast_inverse_f32)...
834 // just fall thru and take the general case.
835 debug(2) << "Simplifier: unhandled PureExtern: " << op->name;
836 }
837
838 // No else: we want to fall thru from the PureExtern clause.
839 {
840 vector<Expr> new_args(op->args.size());
841 bool changed = false;
842
843 // Mutate the args
844 for (size_t i = 0; i < op->args.size(); i++) {
845 const Expr &old_arg = op->args[i];
846 Expr new_arg = mutate(old_arg, nullptr);
847 if (!new_arg.same_as(old_arg)) {
848 changed = true;
849 }
850 new_args[i] = std::move(new_arg);
851 }
852
853 if (!changed) {
854 return op;
855 } else {
856 return Call::make(op->type, op->name, new_args, op->call_type,
857 op->func, op->value_index, op->image, op->param);
858 }
859 }
860}
