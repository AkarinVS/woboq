10uint64_t max_uint(int bits) {
11 uint64_t max_val = 0xffffffffffffffffULL;
12 return max_val >> (64 - bits);
13}

15int64_t max_int(int bits) {
16 int64_t max_val = 0x7fffffffffffffffLL;
17 return max_val >> (64 - bits);
18}

20int64_t min_int(int bits) {

26/** Return an expression which is the maximum value of this type */
27Halide::Expr Type::max() const {
28 if (is_vector()) {
29 return Internal::Broadcast::make(element_of().max(), lanes());
30 } else if (is_int()) {
31 return Internal::IntImm::make(*this, max_int(bits()));
32 } else if (is_uint()) {
33 return Internal::UIntImm::make(*this, max_uint(bits()));
34 } else {
35 internal_assert(is_float());
36 if (bits() == 16) {
37 return Internal::FloatImm::make(*this, 65504.0);
38 } else if (bits() == 32) {
39 return Internal::FloatImm::make(*this, std::numeric_limits<float>::infinity());
40 } else if (bits() == 64) {
41 return Internal::FloatImm::make(*this, std::numeric_limits<double>::infinity());
42 } else {
43 internal_error
44 << "Unknown float type: " << (*this) << "\n";
45 return 0;
46 }
47 }
48}

50/** Return an expression which is the minimum value of this type */
51Halide::Expr Type::min() const {

102bool Type::can_represent(Type other) const {
103 if (*this == other) {
104 return true;
105 }
106 if (lanes() != other.lanes()) {
107 return false;
108 }
109 if (is_int()) {
110 return ((other.is_int() && other.bits() <= bits()) ||
111 (other.is_uint() && other.bits() < bits()));
112 } else if (is_uint()) {
113 return other.is_uint() && other.bits() <= bits();
114 } else if (is_bfloat()) {
115 return (other.is_bfloat() && other.bits() <= bits());
116 } else if (is_float()) {
117 if (other.is_bfloat()) {
118 return bits() > other.bits();
119 } else {
120 return ((other.is_float() && other.bits() <= bits()) ||
121 (bits() == 64 && other.bits() <= 32) ||
122 (bits() == 32 && other.bits() <= 16));
123 }
124 } else {
125 return false;
126 }

129bool Type::can_represent(int64_t x) const {
130 if (is_int()) {
131 return x >= min_int(bits()) && x <= max_int(bits());
132 } else if (is_uint()) {
133 return x >= 0 && (uint64_t)x <= max_uint(bits());
134 } else if (is_bfloat()) {
135 switch (bits()) {
136 case 16:
137 // Round-trip from int64_t to bfloat16_t and back to see
138 // if the value was preserved. This round-tripping must be
139 // done via float in both directions, which gives us the
140 // following ridiculous chain of casts:
141 return (int64_t)(float)(bfloat16_t)(float)x == x;
142 default:
143 return false;
144 }
145 } else if (is_float()) {
146 switch (bits()) {
147 case 16:
148 return (int64_t)(float)(float16_t)(float)x == x;
149 case 32:
150 return (int64_t)(float)x == x;
151 case 64:
152 return (int64_t)(double)x == x;
153 default:
154 return false;
155 }
156 } else {
157 return false;
158 }
159}

189bool Type::can_represent(double x) const {
190 if (is_int()) {
191 int64_t i = Internal::safe_numeric_cast<int64_t>(x);
192 return (x >= min_int(bits())) && (x <= max_int(bits())) && (x == (double)i);
193 } else if (is_uint()) {
194 uint64_t u = Internal::safe_numeric_cast<uint64_t>(x);
195 return (x >= 0) && (x <= max_uint(bits())) && (x == (double)u);
196 } else if (is_bfloat()) {
197 switch (bits()) {
198 case 16:
199 return (double)(bfloat16_t)x == x;
200 default:
201 return false;
202 }
203 } else if (is_float()) {
204 switch (bits()) {
205 case 16:
206 return (double)(float16_t)x == x;
207 case 32:
208 return (double)(float)x == x;
209 case 64:
210 return true;
211 default:
212 return false;
213 }
214 } else {
215 return false;
216 }

219bool Type::same_handle_type(const Type &other) const {
220 const halide_handle_cplusplus_type *first = handle_type;
221 const halide_handle_cplusplus_type *second = other.handle_type;
222
223 if (first == second) {
224 return true;
225 }
226
227 if (first == nullptr) {
228 first = halide_handle_traits<void *>::type_info();
229 }
230 if (second == nullptr) {
231 second = halide_handle_traits<void *>::type_info();
232 }
233
234 return first->inner_name == second->inner_name &&
235 first->namespaces == second->namespaces &&
236 first->enclosing_types == second->enclosing_types &&
237 first->cpp_type_modifiers == second->cpp_type_modifiers &&
238 first->reference_type == second->reference_type;
239}

241std::string type_to_c_type(Type type, bool include_space, bool c_plus_plus) {
242 bool needs_space = true;
243 ostringstream oss;
244
245 if (type.is_bfloat()) {
246 oss << "bfloat" << type.bits() << "_t";
247 } else if (type.is_float()) {
248 if (type.bits() == 32) {
249 oss << "float";
250 } else if (type.bits() == 64) {
251 oss << "double";
252 } else {
253 oss << "float" << type.bits() << "_t";
254 }
255 if (type.is_vector()) {
256 oss << type.lanes();
257 }
258 } else if (type.is_handle()) {
259 needs_space = false;
260
261 // If there is no type info or is generating C (not C++) and
262 // the type is a class or in an inner scope, just use void *.
263 if (type.handle_type == nullptr ||
264 (!c_plus_plus &&
265 (!type.handle_type->namespaces.empty() ||
266 !type.handle_type->enclosing_types.empty() ||
267 type.handle_type->inner_name.cpp_type_type == halide_cplusplus_type_name::Class))) {
268 oss << "void *";
269 } else {
270 if (type.handle_type->inner_name.cpp_type_type ==
271 halide_cplusplus_type_name::Struct) {
272 oss << "struct ";
273 }
274
275 if (!type.handle_type->namespaces.empty() ||
276 !type.handle_type->enclosing_types.empty()) {
277 oss << "::";
278 for (size_t i = 0; i < type.handle_type->namespaces.size(); i++) {
279 oss << type.handle_type->namespaces[i] << "::";
280 }
281 for (size_t i = 0; i < type.handle_type->enclosing_types.size(); i++) {
282 oss << type.handle_type->enclosing_types[i].name << "::";
283 }
284 }
285 oss << type.handle_type->inner_name.name;
286 if (type.handle_type->reference_type == halide_handle_cplusplus_type::LValueReference) {
287 oss << " &";
288 } else if (type.handle_type->reference_type == halide_handle_cplusplus_type::RValueReference) {
289 oss << " &&";
290 }
291 for (auto modifier : type.handle_type->cpp_type_modifiers) {
292 if (modifier & halide_handle_cplusplus_type::Const) {
293 oss << " const";
294 }
295 if (modifier & halide_handle_cplusplus_type::Volatile) {
296 oss << " volatile";
297 }
298 if (modifier & halide_handle_cplusplus_type::Restrict) {
299 oss << " restrict";
300 }
301 if (modifier & halide_handle_cplusplus_type::Pointer) {
302 oss << " *";
303 }
304 }
305 }
306 } else {
307 // This ends up using different type names than OpenCL does
308 // for the integer vector types. E.g. uint16x8_t rather than
309 // OpenCL's short8. Should be fine as CodeGen_C introduces
310 // typedefs for them and codegen always goes through this
311 // routine or its override in CodeGen_OpenCL to make the
312 // names. This may be the better bet as the typedefs are less
313 // likely to collide with built-in types (e.g. the OpenCL
314 // ones for a C compiler that decides to compile OpenCL).
315 // This code also supports arbitrary vector sizes where the
316 // OpenCL ones must be one of 2, 3, 4, 8, 16, which is too
317 // restrictive for already existing architectures.
318 switch (type.bits()) {
319 case 1:
320 // bool vectors are always emitted as uint8 in the C++ backend
321 if (type.is_vector()) {
322 oss << "uint8x" << type.lanes() << "_t";
323 } else {
324 oss << "bool";
325 }
326 break;
327 default:
328 if (type.is_uint()) {
329 oss << "u";
330 }
331 oss << "int" << type.bits();
332 if (type.is_vector()) {
333 oss << "x" << type.lanes();
334 }
335 oss << "_t";
336 }
337 }
338 if (include_space && needs_space) {
339 oss << " ";
340 }
341 return oss.str();
342}
