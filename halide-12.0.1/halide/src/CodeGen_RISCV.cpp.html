12/** A code generator that emits mips code from a given Halide stmt. */
13class CodeGen_RISCV : public CodeGen_Posix {
14public:
15 /** Create a mips code generator. Processor features can be

37string CodeGen_RISCV::mattrs() const {
38 // Note: the default march is "rv[32|64]imafdc",
39 // which includes standard extensions:
40 // +m Integer Multiplication and Division,
41 // +a Atomic Instructions,
42 // +f Single-Precision Floating-Point,
43 // +d Double-Precision Floating-Point,
44 // +c Compressed Instructions,
45 string arch_flags = "+m,+a,+f,+d,+c";
46
47 if (target.has_feature(Target::RVV)) {
48 arch_flags += ",+experimental-v";
49 }
50 return arch_flags;
51}
52
53string CodeGen_RISCV::mabi() const {
54 string abi;
55 if (target.bits == 32) {
56 abi = "ilp32";
57 } else {
58 abi = "lp64";
59 }
60 if (!target.has_feature(Target::SoftFloatABI)) {
61 abi += "d";
62 }
63 return abi;
64}
65
66bool CodeGen_RISCV::use_soft_float_abi() const {
67 return target.has_feature(Target::SoftFloatABI);
68}
69
70int CodeGen_RISCV::native_vector_bits() const {
71 return 128;
72}
