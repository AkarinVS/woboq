59 string simplify_name(const string &s, bool is_func) {
60 // Trim the function name and stage number from the for loop,
61 // as well as any uniqueness $n suffixes on variables.
62 std::ostringstream trimmed_name;
63
64 bool keep = is_func;
65 int dot_count = 0;
66 for (size_t i = 0; i < s.size(); i++) {
67 if (s[i] == '.') {
68 dot_count++;
69 if (dot_count >= 2) {
70 if (dot_count == 2) {
71 i++;
72 }
73 keep = true;
74 }
75 }
76 if (s[i] == '$') {
77 keep = false;
78 }
79 if (keep) {
80 trimmed_name << s[i];
81 }
82 }
83
84 return trimmed_name.str();
85 }

87 void visit(const For *op) override {
88 string simplified_loop_var_name = simplify_var_name(op->name);
89
90 out << get_indent() << op->for_type << " " << simplified_loop_var_name;
91
92 // If the min or extent are constants, print them. At this
93 // stage they're all variables.
94 Expr min_val = op->min, extent_val = op->extent;
95 const Variable *min_var = min_val.as<Variable>();
96 const Variable *extent_var = extent_val.as<Variable>();
97 if (min_var && constants.contains(min_var->name)) {
98 min_val = constants.get(min_var->name);
99 }
100
101 if (extent_var && constants.contains(extent_var->name)) {
102 extent_val = constants.get(extent_var->name);
103 }
104
105 if (extent_val.defined() && is_const(extent_val) &&
106 min_val.defined() && is_const(min_val)) {
107 Expr max_val = simplify(min_val + extent_val - 1);
108 out << " in [" << min_val << ", " << max_val << "]";
109 }
110
111 out << op->device_api;
112
113 out << ":\n";
114 indent += 2;
115 op->body.accept(this);
116 indent -= 2;
117 }
118
119 void visit(const Realize *op) override {
120 // If the storage and compute levels for this function are
121 // distinct, print the store level too.
122 auto it = env.find(op->name);
123 if (it != env.end() &&
124 !(it->second.schedule().store_level() ==
125 it->second.schedule().compute_level())) {
126 out << get_indent();
127 out << "store " << simplify_func_name(op->name) << ":\n";
128 indent += 2;
129 op->body.accept(this);
130 indent -= 2;
131 } else {
132 op->body.accept(this);
133 }
134 }
135
136 void visit(const ProducerConsumer *op) override {
137 out << get_indent();
138 if (op->is_producer) {
139 out << "produce " << simplify_func_name(op->name) << ":\n";
140 } else {
141 out << "consume " << simplify_func_name(op->name) << ":\n";
142 }
143 indent += 2;
144 op->body.accept(this);
145 indent -= 2;
146 }
147
148 void visit(const Provide *op) override {
149 out << get_indent() << simplify_func_name(op->name) << "(...) = ...\n";
150 }
151
152 void visit(const LetStmt *op) override {
153 if (is_const(op->value)) {
154 constants.push(op->name, op->value);
155 op->body.accept(this);
156 constants.pop(op->name);
157 } else {
158 op->body.accept(this);
159 }
160 }

165string print_loop_nest(const vector<Function> &output_funcs) {
166 // Do the first part of lowering:
167
168 // Compute an environment
169 map<string, Function> env;
170 for (const Function &f : output_funcs) {
171 populate_environment(f, env);
172 }
173
174 // Create a deep-copy of the entire graph of Funcs.
175 vector<Function> outputs;
176 std::tie(outputs, env) = deep_copy(output_funcs, env);
177
178 // Output functions should all be computed and stored at root.
179 for (const Function &f : outputs) {
180 Func(f).compute_root().store_root();
181 }
182
183 // Finalize all the LoopLevels
184 for (auto &iter : env) {
185 iter.second.lock_loop_levels();
186 }
187
188 // Substitute in wrapper Funcs
189 env = wrap_func_calls(env);
190
191 // Compute a realization order and determine group of functions which loops
192 // are to be fused together
193 vector<string> order;
194 vector<vector<string>> fused_groups;
195 std::tie(order, fused_groups) = realization_order(outputs, env);
196
197 // Try to simplify the RHS/LHS of a function definition by propagating its
198 // specializations' conditions
199 simplify_specializations(env);
200
201 // For the purposes of printing the loop nest, we don't want to
202 // worry about which features are and aren't enabled.
203 Target target = get_host_target();
204 for (DeviceAPI api : all_device_apis) {
205 target.set_feature(target_feature_for_device_api(DeviceAPI(api)));
206 }
207
208 bool any_memoized = false;
209 // Schedule the functions.
210 Stmt s = schedule_functions(outputs, fused_groups, env, target, any_memoized);
211
212 // Compute the maximum and minimum possible value of each
213 // function. Used in later bounds inference passes.
214 FuncValueBounds func_bounds = compute_function_value_bounds(order, env);
215
216 // This pass injects nested definitions of variable names, so we
217 // can't simplify statements from here until we fix them up. (We
218 // can still simplify Exprs).
219 s = bounds_inference(s, outputs, order, fused_groups, env, func_bounds, target);
220 s = remove_extern_loops(s);
221 s = sliding_window(s, env);
222 s = simplify_correlated_differences(s);
223 s = allocation_bounds_inference(s, env, func_bounds);
224 s = remove_undef(s);
225 s = uniquify_variable_names(s);
226 s = simplify(s, false);
227
228 // Now convert that to pseudocode
229 std::ostringstream sstr;
230 PrintLoopNest pln(sstr, env);
231 s.accept(&pln);
232 return sstr.str();
233}
