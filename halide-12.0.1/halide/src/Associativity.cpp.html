28template<typename T>
29vector<T> get_subvector(const vector<T> &v, const set<int> &indices) {
30 vector<T> sub;
31 for (const auto &index : indices) {
32 internal_assert(index < (int)v.size());
33 sub.push_back(v[index]);
34 }
35 return sub;

38// Replace self-references to 'func' with arguments 'args' at
39// 'value_index' in the Expr/Stmt with some Var
40class ConvertSelfRef : public IRGraphMutator {
41 using IRGraphMutator::visit;
42
43 const string &func;
44 const vector<Expr> &args;
45 // If that function has multiple values, which value does this
46 // call node refer to?
47 const int value_index;
48 const vector<string> &op_x_names;
49
50 Expr visit(const Call *op) override {
51 if (!is_solvable) {
52 return op;
53 }
54 Expr expr = IRGraphMutator::visit(op);
55 op = expr.as<Call>();
56 internal_assert(op);
57
58 if ((op->call_type == Call::Halide) && (func == op->name)) {
59 internal_assert(args.size() == op->args.size())
60 << "Self-reference should have the same number of args as the original\n";
61 for (size_t i = 0; i < op->args.size(); i++) {
62 if (!graph_equal(op->args[i], args[i])) {
63 debug(5) << "Self-reference of " << op->name
64 << " with different args from the LHS. Operation is not associative\n";
65 is_solvable = false;
66 return expr;
67 }
68 }
69 // Substitute the call
70 internal_assert(op->value_index < (int)op_x_names.size());
71 debug(5) << " Substituting Call " << op->name << " at value index "
72 << op->value_index << " with " << op_x_names[op->value_index] << "\n";
73 expr = Variable::make(op->type, op_x_names[op->value_index]);
74
75 if (op->value_index == value_index) {
76 x_part = op;
77 } else {
78 x_dependencies.insert(op->value_index);
79 }
80 }
81 return expr;
82 }
83
84public:
85 ConvertSelfRef(const string &f, const vector<Expr> &args, int idx,
86 const vector<string> &x_names)
87 : func(f), args(args), value_index(idx), op_x_names(x_names) {
88 }
89
90 bool is_solvable = true;
91 set<int> x_dependencies; // Contains dependencies on self-reference at different tuple indices
92 Expr x_part; // Undefined if there is no self-reference at value_index
93};

95bool associative_op_pattern_match(const Expr &e,
96 const Expr &op,
97 const vector<string> &x_names,
98 const vector<string> &y_names,
99 const Scope<> &x_scope,
100 map<string, Expr> &match) {
101
102 internal_assert(e.type() == op.type())
103 << "Expr has type " << e.type() << ", while pattern has type " << op.type() << "\n";
104 map<string, Expr> result;
105 if (expr_match(op, e, result)) {
106 debug(5) << "Found associative ops for " << e << " -> " << op
107 << ", y_part: " << result["y0"] << "\n";
108
109 for (size_t i = 0; i < x_names.size(); ++i) {
110 const auto &iter = result.find("x" + std::to_string(i));
111 if (iter != result.end()) {
112 const Variable *xvar = iter->second.as<Variable>();
113 if ((xvar == nullptr) || (xvar->name != x_names[i])) {
114 debug(5) << "...Skipping match since the x_part is different than expected. "
115 << "Expect: " << x_names[i] << "; get: " << iter->second << "\n";
116 return false;
117 }
118 }
119 }
120 for (size_t i = 0; i < y_names.size(); ++i) {
121 const auto &iter = result.find("y" + std::to_string(i));
122 if (iter != result.end()) {
123 // Make sure that y_part should not depend on x vars
124 if (expr_uses_vars(iter->second, x_scope)) {
125 debug(5) << "...Skipping match since the y_part depends on x vars\n";
126 return false;
127 }
128 }
129 }
130 for (size_t i = 0; i < x_names.size(); ++i) {
131 const auto &iter = result.find("k" + std::to_string(i));
132 if (iter != result.end()) {
133 // Make sure that k_part is constant
134 if (!is_const(iter->second)) {
135 debug(5) << "...Skipping match since the k_part is not constant\n";
136 return false;
137 }
138 }
139 }
140
141 // Make sure that the new matches are in agreement with any previous matches
142 for (const auto &iter : result) {
143 const auto &match_iter = match.find(iter.first);
144 if (match_iter == match.end()) {
145 debug(5) << "Adding result: " << iter.first << " -> " << iter.second << "\n";
146 match.emplace(iter.first, iter.second);
147 } else {
148 if (!equal(iter.first, match_iter->first) || !equal(iter.second, match_iter->second)) {
149 return false;
150 }
151 }
152 }
153 return true;
154 }
155 return false;
156}

158// Return true if we are able to find a match in the table (i.e. the op can be
159// proven associative) and update 'assoc_op'.
160bool find_match(const vector<AssociativePattern> &table, const vector<string> &op_x_names,
161 const vector<string> &op_y_names, const vector<Expr> &x_parts,
162 const vector<Expr> &exprs, AssociativeOp &assoc_op) {
163 internal_assert(op_x_names.size() == op_y_names.size());
164 internal_assert(op_x_names.size() == x_parts.size());
165 internal_assert(op_x_names.size() == exprs.size());
166 internal_assert(op_x_names.size() == assoc_op.size());
167
168 Scope<> x_scope;
169 for (const auto &x : op_x_names) {
170 x_scope.push(x);
171 }
172
173 for (const AssociativePattern &pattern : table) {
174 internal_assert(pattern.size() == op_x_names.size());
175 map<string, Expr> pattern_match;
176 bool matched = true;
177 // If any of element in 'pattern' does not match, try the next thing in
178 // the table.
179 for (size_t i = 0; i < pattern.size(); ++i) {
180 if (!associative_op_pattern_match(exprs[i], pattern.ops[i], op_x_names,
181 op_y_names, x_scope, pattern_match)) {
182 matched = false;
183 break;
184 }
185 }
186 if (!matched) {
187 continue;
188 }
189
190 vector<pair<Expr, Expr>> replacement; // find -> replacement
191 for (size_t index = 0; index < op_y_names.size(); ++index) {
192 const auto &y_iter = pattern_match.find("y" + std::to_string(index));
193 if (y_iter == pattern_match.end()) {
194 // Didn't find y{index} during pattern matching. Try next pattern.
195 matched = false;
196 break;
197 }
198 Expr y_part = y_iter->second;
199 debug(5) << "Pattern at index " << index << ":\n " << op_x_names[index]
200 << " -> " << x_parts[index] << "\n " << op_y_names[index]
201 << " -> " << y_part << "\n";
202
203 assoc_op.xs[index] = {op_x_names[index], x_parts[index]};
204 assoc_op.ys[index] = {op_y_names[index], y_part};
205 replacement.emplace_back(y_part, Variable::make(y_part.type(), op_y_names[index]));
206 }
207 if (!matched) {
208 continue;
209 }
210 for (size_t index = 0; index < exprs.size(); ++index) {
211 Expr e = exprs[index];
212 // Order of substitution matters, e.g. in the argmin case, _y_0 -> g(rx)[0]
213 // and _y_1 -> rx. If we substitute the 2nd element rx first, substitution
214 // of g(rx)[0] will fail.
215 for (const auto &iter : replacement) {
216 e = substitute(iter.first, iter.second, e);
217 }
218 assoc_op.pattern.ops[index] = e;
219 assoc_op.pattern.identities[index] = pattern.identities[index];
220 }
221 assoc_op.pattern.is_commutative = pattern.is_commutative;
222 return true;
223 }
224 return false;
225}

227// Return a pair of booleans indicating if an operator is associative.
228// 'assoc_op' contains the the equivalent associative binary/unary operator
229// for that operator. If the operator is non-associative, 'assoc_op' is not valid.
230bool extract_associative_op(const vector<Expr> &exprs, const vector<string> &op_x_names,
231 const vector<string> &op_y_names, const vector<Expr> &x_parts,
232 AssociativeOp &assoc_op) {
233 if (exprs.size() == 1) {
234 Type t = exprs[0].type();
235 if (!x_parts[0].defined()) {
236 // Update with no self-recurrence is associative and the identity
237 // can be anything since it's going to be replaced anyway, but it's
238 // not commutative
239 assoc_op.pattern.ops[0] = Variable::make(t, op_y_names[0]);
240 assoc_op.pattern.identities[0] = make_const(t, 0);
241 assoc_op.pattern.is_commutative = false;
242 assoc_op.xs[0] = {"", Expr()};
243 assoc_op.ys[0] = {op_y_names[0], exprs[0]};
244 return true;
245 } else if (equal(exprs[0], Variable::make(t, op_x_names[0]))) {
246 // Self assignment, f(x) = f(x), is both associative
247 // and commutative. The identity can be anything since it's
248 // going to be replaced by itself.
249 debug(5) << "Self assignment: " << x_parts[0] << " = " << x_parts[0] << "\n";
250 assoc_op.pattern.ops[0] = Variable::make(t, op_x_names[0]);
251 assoc_op.pattern.identities[0] = make_const(t, 0);
252 assoc_op.pattern.is_commutative = true;
253 assoc_op.xs[0] = {op_x_names[0], x_parts[0]};
254 assoc_op.ys[0] = {"", Expr()};
255 return true;
256 }
257 }
258 return find_match(get_ops_table(exprs), op_x_names, op_y_names,
259 x_parts, exprs, assoc_op);
260}

262void add_transitive_dependencies(vector<set<int>> &dependencies) {
263 // TODO(psuriana): there might be a better way to find all the transitive dependencies
264 bool change = true;
265 while (change) {
266 change = false;
267 for (size_t i = 0; i < dependencies.size(); ++i) {
268 for (size_t j = 0; j < dependencies.size(); ++j) {
269 if (i == j) {
270 continue;
271 }
272 if (dependencies[i].count(j)) {
273 for (const auto &idx : dependencies[j]) {
274 if (dependencies[i].count(idx) == 0) {
275 dependencies[i].insert(idx);
276 change = true;
277 }
278 }
279 }
280 }
281 }
282 }

285// Given dependencies of each tuple element, compute the set of subgraphs:
286// all vertices that are reachable from a given vertex. If a subgraph is fully
287// contained in another subgraph, remove it from the final output.
288vector<set<int>> compute_subgraphs(vector<set<int>> dependencies) {
289 vector<set<int>> subgraphs(dependencies.size());
290 for (size_t i = 0; i < dependencies.size(); ++i) {
291 // Check if the current subgraph is a subset of another
292 const auto &current = dependencies[i];
293 if (current.empty()) {
294 continue;
295 }
296 bool should_remove = false;
297 for (size_t j = 0; j < dependencies.size(); ++j) {
298 const auto &other = dependencies[j];
299 if ((i == j) || (current.size() > other.size()) || (j < i && subgraphs[i].empty())) {
300 continue;
301 }
302 vector<int> diff;
303 // Compute the vertices in the current set that are not contained in the other
304 std::set_difference(current.begin(), current.end(), other.begin(), other.end(),
305 std::inserter(diff, diff.begin()));
306 if (diff.empty()) {
307 // 'current' is fully contained in 'other'
308 should_remove = true;
309 break;
310 }
311 }
312 if (!should_remove) {
313 subgraphs[i] = current;
314 }
315 }
316 return subgraphs;
317}

321AssociativeOp prove_associativity(const string &f, vector<Expr> args, vector<Expr> exprs) {
322 AssociativeOp assoc_op(exprs.size());
323
324 for (Expr &arg : args) {
325 // Undo the existing CSE pass done at function definition time
326 // to ensure things like += are in the expected form. Make no
327 // further transformations so that the LHS and RHS don't
328 // diverge.
329 arg = substitute_in_all_lets(arg);
330 }
331
332 vector<string> op_x_names(exprs.size()), op_y_names(exprs.size());
333 for (size_t idx = 0; idx < exprs.size(); ++idx) {
334 op_x_names[idx] = unique_name("_x_" + std::to_string(idx));
335 op_y_names[idx] = unique_name("_y_" + std::to_string(idx));
336 }
337
338 vector<set<int>> dependencies(exprs.size());
339 vector<Expr> x_parts(exprs.size());
340 bool all_independent = true;
341
342 // For a Tuple of exprs to be associative, each element of the Tuple
343 // has to be associative.
344 for (int idx = exprs.size() - 1; idx >= 0; --idx) {
345 // Undo the existing CSE pass done at function definition time.
346 exprs[idx] = substitute_in_all_lets(exprs[idx]);
347
348 // Replace any self-reference to Func 'f' with a Var
349 ConvertSelfRef csr(f, args, idx, op_x_names);
350 exprs[idx] = csr.mutate(exprs[idx]);
351 if (!csr.is_solvable) {
352 return AssociativeOp();
353 }
354 if (!csr.x_dependencies.empty()) {
355 all_independent = false;
356 }
357 x_parts[idx] = csr.x_part;
358 dependencies[idx] = csr.x_dependencies;
359 // Add dependency on itself (regardless whether it actually depends on
360 // its previous values) for the purpose of computing the subgraph
361 dependencies[idx].insert(idx);
362
363 exprs[idx] = common_subexpression_elimination(exprs[idx]);
364 exprs[idx] = simplify(exprs[idx]);
365 exprs[idx] = solve_expression(exprs[idx], op_x_names[idx]).result; // Move 'x' to the left as possible
366 exprs[idx] = substitute_in_all_lets(exprs[idx]);
367 }
368 internal_assert((exprs.size() != 1) || all_independent) << "1D tuple should be all independent\n";
369
370 vector<set<int>> subgraphs;
371 if (!all_independent) {
372 debug(5) << "There are cross-dependencies. Need to prove associativity in bulk.\n";
373 // Find all transitive dependencies and add them to the graph
374 add_transitive_dependencies(dependencies);
375 // Decompose the tuple into subgraphs and solve for each separately
376 subgraphs = compute_subgraphs(dependencies);
377 } else {
378 debug(5) << "All tuple elements are independent. Try proving associativity of "
379 << "each element separately.\n";
380 // If all elements are independent, the subgraph is equal to the dependencies graph
381 subgraphs = dependencies;
382 }
383 internal_assert(subgraphs.size() == exprs.size());
384
385 for (size_t i = 0; i < subgraphs.size(); ++i) {
386 if (subgraphs[i].empty()) {
387 debug(5) << "Empty subgraph " << i << "\n";
388 continue;
389 }
390 if (subgraphs[i].size() > 2) {
391 // TODO(psuriana): Currently only support max of 2 tuple elements
392 debug(5) << "Subgraph " << i << " size is " << subgraphs[i].size() << " which is bigger than 2\n";
393 return AssociativeOp();
394 }
395
396 vector<Expr> sub_exprs = get_subvector(exprs, subgraphs[i]);
397 vector<string> sub_op_x_names = get_subvector(op_x_names, subgraphs[i]);
398 vector<string> sub_op_y_names = get_subvector(op_y_names, subgraphs[i]);
399 vector<Expr> sub_x_parts = get_subvector(x_parts, subgraphs[i]);
400 AssociativeOp sub_assoc_op(sub_exprs.size());
401
402 // TODO(psuriana): In general, if we fail to find a match for the
403 // set of initial subgraphs, we need to consider other possible
404 // grouping of those initial subgraphs. Since only the 'x' is
405 // apparent from the Halide update definition, the compute_subgraphs
406 // method over-partitions the graph (e.g. 2x2 matrix multiplication
407 // written as a four-dimensional reduction).
408
409 if (!extract_associative_op(sub_exprs, sub_op_x_names, sub_op_y_names,
410 sub_x_parts, sub_assoc_op)) {
411 debug(5) << "Cannot find matching associative ops\n";
412 return AssociativeOp();
413 }
414
415 debug(5) << "...Proving associativity of subgraph " << i << "\n";
416 const set<int> &indices = subgraphs[i];
417 for (auto iter = indices.begin(); iter != indices.end(); ++iter) {
418 int index = *iter;
419 int j = std::distance(indices.begin(), iter);
420
421 // If the ops/x/y have been extracted previously, we have to make sure
422 // they are consistent with the new extracted values.
423 if (assoc_op.pattern.ops[index].defined()) {
424 if (!equal(assoc_op.pattern.ops[index], sub_assoc_op.pattern.ops[j]) ||
425 !equal(assoc_op.pattern.identities[index], sub_assoc_op.pattern.identities[j])) {
426 debug(5) << "Conflicting associative ops/identities from different subgraphs\n";
427 return AssociativeOp();
428 }
429 }
430 if (assoc_op.xs[index].expr.defined()) {
431 if (assoc_op.xs[index] != sub_assoc_op.xs[j]) {
432 debug(5) << "Conflicting associative x-replacements from different subgraphs\n";
433 return AssociativeOp();
434 }
435 }
436 if (assoc_op.ys[index].expr.defined()) {
437 if (assoc_op.ys[index] != sub_assoc_op.ys[j]) {
438 debug(5) << "Conflicting associative y-replacements from different subgraphs\n";
439 return AssociativeOp();
440 }
441 }
442
443 assoc_op.pattern.ops[index] = sub_assoc_op.pattern.ops[j];
444 assoc_op.pattern.identities[index] = sub_assoc_op.pattern.identities[j];
445 assoc_op.pattern.is_commutative = sub_assoc_op.pattern.is_commutative;
446 assoc_op.xs[index] = sub_assoc_op.xs[j];
447 assoc_op.ys[index] = sub_assoc_op.ys[j];
448 }
449 }
450
451 assoc_op.is_associative = true;
452 debug(5) << "Found associative ops:\n"
453 << assoc_op << "\n";
454 return assoc_op;
455}

459std::string print_args(const string &f, const vector<Expr> &args, const vector<Expr> &exprs) {
460 std::ostringstream stream;
461 stream << f << "(";
462 for (size_t i = 0; i < args.size(); ++i) {
463 stream << args[i];
464 if (i != args.size() - 1) {
465 stream << ", ";
466 }
467 }
468 stream << ") = ";
469
470 if (exprs.size() == 1) {
471 stream << exprs[0];
472 } else if (exprs.size() > 1) {
473 stream << "Tuple(";
474 for (size_t i = 0; i < exprs.size(); ++i) {
475 stream << exprs[i];
476 if (i != exprs.size() - 1) {
477 stream << ", ";
478 }
479 }
480 stream << ")";
481 }
482 return stream.str();
483}

485void check_associativity(const string &f, const vector<Expr> &args, const vector<Expr> &exprs,
486 const AssociativeOp &assoc_op) {
487 auto result = prove_associativity(f, args, exprs);
488 internal_assert(result.associative() == assoc_op.associative())
489 << "Checking associativity: " << print_args(f, args, exprs) << "\n"
490 << " Expect is associative: " << assoc_op.associative() << "\n"
491 << " instead of " << result.associative() << "\n";
492 if (assoc_op.associative()) {
493 map<string, Expr> replacement;
494 for (size_t i = 0; i < assoc_op.size(); ++i) {
495 internal_assert(equal(result.pattern.identities[i], assoc_op.pattern.identities[i]))
496 << "Checking associativity: " << print_args(f, args, exprs) << "\n"
497 << " Index: " << i << "\n"
498 << " Expect identity: " << assoc_op.pattern.identities[i] << "\n"
499 << " instead of " << result.pattern.identities[i] << "\n";
500 internal_assert(equal(result.xs[i].expr, assoc_op.xs[i].expr))
501 << "Checking associativity: " << print_args(f, args, exprs) << "\n"
502 << " Index: " << i << "\n"
503 << " Expect x: " << assoc_op.xs[i].expr << "\n"
504 << " instead of " << result.xs[i].expr << "\n";
505 internal_assert(equal(result.ys[i].expr, assoc_op.ys[i].expr))
506 << "Checking associativity: " << print_args(f, args, exprs) << "\n"
507 << " Index: " << i << "\n"
508 << " Expect y: " << assoc_op.ys[i].expr << "\n"
509 << " instead of " << result.ys[i].expr << "\n";
510
511 if (result.xs[i].expr.defined()) {
512 replacement.emplace(assoc_op.xs[i].var, Variable::make(result.xs[i].expr.type(), result.xs[i].var));
513 }
514 if (result.ys[i].expr.defined()) {
515 replacement.emplace(assoc_op.ys[i].var, Variable::make(result.ys[i].expr.type(), result.ys[i].var));
516 }
517 }
518 for (size_t i = 0; i < assoc_op.size(); ++i) {
519 Expr expected_op = substitute(replacement, assoc_op.pattern.ops[i]);
520
521 internal_assert(equal(result.pattern.ops[i], expected_op))
522 << "Checking associativity: " << print_args(f, args, exprs) << "\n"
523 << " Index: " << i << "\n"
524 << " Expect bin op: " << expected_op << "\n"
525 << " instead of " << result.pattern.ops[i] << "\n";
526
527 debug(5) << "\nExpected op: " << expected_op << "\n";
528 debug(5) << "Operator: " << result.pattern.ops[i] << "\n";
529 debug(5) << " identity: " << result.pattern.identities[i] << "\n";
530 debug(5) << " x: " << result.xs[i].var << " -> " << result.xs[i].expr << "\n";
531 debug(5) << " y: " << result.ys[i].var << " -> " << result.ys[i].expr << "\n";
532 }
533 }

538void associativity_test() {
539 typedef AssociativeOp::Replacement Replacement;
540
541 {
542 // Tests for saturating addition
543 Type t = UInt(8);
544 Expr x = Variable::make(t, "x");
545 Expr y = Variable::make(t, "y");
546 Expr x_idx = Variable::make(Int(32), "x_idx");
547 Expr f_call_0 = Call::make(t, "f", {x_idx}, Call::CallType::Halide, FunctionPtr(), 0);
548
549 // f(x) = uint8(uint16(x + y), 255)
550 check_associativity("f", {x_idx}, {Cast::make(UInt(8), min(Cast::make(UInt(16), y + f_call_0), make_const(t, 255)))},
551 AssociativeOp(
552 AssociativePattern(Cast::make(UInt(8), min(Cast::make(UInt(16), x + y), make_const(t, 255))), make_const(t, 0), true),
553 {Replacement("x", f_call_0)},
554 {Replacement("y", y)},
555 true));
556
557 // f(x) = uint8(uint16(x + y), uint16(255))
558 check_associativity("f", {x_idx}, {Cast::make(UInt(8), min(Cast::make(UInt(16), y + f_call_0), Cast::make(UInt(16), make_const(t, 255))))},
559 AssociativeOp(
560 AssociativePattern(Cast::make(UInt(8), min(Cast::make(UInt(16), x + y), make_const(t, 255))), make_const(t, 0), true),
561 {Replacement("x", f_call_0)},
562 {Replacement("y", y)},
563 true));
564
565 // f(x) = select(x > 255 - y, 255, y)
566 check_associativity("f", {x_idx}, {select(f_call_0 > make_const(t, 255) - y, make_const(t, 255), y)},
567 AssociativeOp(
568 AssociativePattern(select(x > make_const(t, 255) - y, make_const(t, 255), y), make_const(t, 0), true),
569 {Replacement("x", f_call_0)},
570 {Replacement("y", y)},
571 true));
572
573 // f(x) = select(x >= -y, 255, y)
574 check_associativity("f", {x_idx}, {select(f_call_0 >= -y, make_const(t, 255), y)},
575 AssociativeOp(
576 AssociativePattern(select(x < -y, y, make_const(t, 255)), make_const(t, 0), true),
