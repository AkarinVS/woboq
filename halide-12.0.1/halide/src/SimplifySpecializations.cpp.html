22void substitute_value_in_var(const string &var, const Expr &value, vector<Definition> &definitions) {
23 for (Definition &def : definitions) {
24 for (auto &def_arg : def.args()) {
25 def_arg = simplify(substitute(var, value, def_arg));
26 }
27 for (auto &def_val : def.values()) {
28 def_val = simplify(substitute(var, value, def_val));

33class SimplifyUsingFact : public IRMutator {
34public:
35 using IRMutator::mutate;
36
37 Expr mutate(const Expr &e) override {
38 if (e.type().is_bool()) {
39 if (equal(fact, e) ||
40 can_prove(!fact || e)) {
41 // fact implies e
42 return const_true();
43 }
44 if (equal(fact, !e) ||
45 equal(!fact, e) ||
46 can_prove(!fact || !e)) {
47 // fact implies !e
48 return const_false();
49 }
50 }
51 return IRMutator::mutate(e);
52 }
53
54 Expr fact;
55 SimplifyUsingFact(Expr f)
56 : fact(std::move(f)) {
57 }
58};

60void simplify_using_fact(const Expr &fact, vector<Definition> &definitions) {
61 for (Definition &def : definitions) {
62 for (auto &def_arg : def.args()) {
63 def_arg = simplify(SimplifyUsingFact(fact).mutate(def_arg));
64 }
65 for (auto &def_val : def.values()) {
66 def_val = simplify(SimplifyUsingFact(fact).mutate(def_val));
67 }
68 }

71vector<Definition> propagate_specialization_in_definition(Definition &def, const string &name) {
72 vector<Definition> result;
73
74 result.push_back(def);
75
76 vector<Specialization> &specializations = def.specializations();
77
78 // Prune specializations based on constants:
79 // -- Any Specializations that have const-false as a condition
80 // can never trigger; go ahead and prune them now to save time & energy
81 // during later phases.
82 // -- Once we encounter a Specialization that is const-true, no subsequent
83 // Specializations can ever trigger (since we evaluate them in order),
84 // so erase them.
85 bool seen_const_true = false;
86 for (auto it = specializations.begin(); it != specializations.end(); /*no-increment*/) {
87 Expr old_c = it->condition;
88 Expr c = simplify(it->condition);
89 // Go ahead and save the simplified condition now
90 it->condition = c;
91 if (is_const_zero(c) || seen_const_true) {
92 debug(1) << "Erasing unreachable specialization ("
93 << old_c << ") -> (" << c << ") for function \"" << name << "\"\n";
94 it = specializations.erase(it);
95 } else {
96 it++;
97 }
98 seen_const_true |= is_const_one(c);
99 }
100
101 // If the final Specialization is const-true, then the default schedule
102 // for the definition will never be run: replace the definition's main
103 // schedule with the one from the final Specialization and prune it from
104 // the list. This may leave the list of Specializations empty.
105 if (!specializations.empty() && is_const_one(specializations.back().condition) && specializations.back().failure_message.empty()) {
106 debug(1) << "Replacing default Schedule with const-true specialization for function \"" << name << "\"\n";
107 const Definition s_def = specializations.back().definition;
108 specializations.pop_back();
109
110 // The values/args needs to be copied over since they might have
111 // been simplified based on the predicate of the specialization.
112 def.values() = s_def.values();
113 def.args() = s_def.args();
114
115 // Copy over the schedule.
116 def.schedule() = s_def.schedule().get_copy();
117
118 // Append our sub-specializations to the Definition's list
119 specializations.insert(specializations.end(), s_def.specializations().begin(), s_def.specializations().end());
120 }
121
122 for (size_t i = specializations.size(); i > 0; i--) {
123 Expr c = specializations[i - 1].condition;
124 Definition &s_def = specializations[i - 1].definition;
125 const EQ *eq = c.as<EQ>();
126 const Variable *var = eq ? eq->a.as<Variable>() : c.as<Variable>();
127
128 internal_assert(s_def.defined());
129
130 vector<Definition> s_result = propagate_specialization_in_definition(s_def, name);
131
132 if (var && eq) {
133 // Then case
134 substitute_value_in_var(var->name, eq->b, s_result);
135
136 // Else case
137 if (eq->b.type().is_bool()) {
138 substitute_value_in_var(var->name, !eq->b, result);
139 }
140 } else if (var) {
141 // Then case
142 substitute_value_in_var(var->name, const_true(), s_result);
143
144 // Else case
145 substitute_value_in_var(var->name, const_false(), result);
146 } else {
147 simplify_using_fact(c, s_result);
148 simplify_using_fact(!c, result);
149 }
150
151 result.insert(result.end(), s_result.begin(), s_result.end());
152 }
153
154 return result;
155}

159void simplify_specializations(map<string, Function> &env) {
160 for (auto &iter : env) {
161 Function &func = iter.second;
162 if (func.definition().defined()) {
163 propagate_specialization_in_definition(func.definition(), func.name());
164 }
165 }
