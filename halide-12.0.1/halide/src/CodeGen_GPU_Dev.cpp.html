29// Check to see if an expression is uniform within a block.
30// This is done by checking to see if the expression depends on any GPU
31// thread indices.
32class IsBlockUniform : public IRVisitor {
33 using IRVisitor::visit;
34
35 void visit(const Variable *op) override {
36 if (CodeGen_GPU_Dev::is_gpu_thread_var(op->name)) {
37 result = false;
38 }
39 }
40
41public:
42 bool result = true;
43
44 IsBlockUniform() = default;
45};
46} // namespace
47
48bool CodeGen_GPU_Dev::is_block_uniform(const Expr &expr) {
49 IsBlockUniform v;
50 expr.accept(&v);
51 return v.result;
52}

54namespace {
55// Check to see if a buffer is a candidate for constant memory storage.
56// A buffer is a candidate for constant memory if it is never written to,
57// and loads are uniform within the workgroup.
58class IsBufferConstant : public IRVisitor {
59 using IRVisitor::visit;
60
61 void visit(const Store *op) override {
62 if (op->name == buffer) {
63 result = false;
64 }
65 if (result) {
66 IRVisitor::visit(op);
67 }
68 }
69
70 void visit(const Load *op) override {
71 if (op->name == buffer &&
72 !CodeGen_GPU_Dev::is_block_uniform(op->index)) {
73 result = false;
74 }
75 if (result) {
76 IRVisitor::visit(op);
77 }
78 }
79
80public:
81 bool result;
82 const std::string &buffer;
83
84 IsBufferConstant(const std::string &b)
85 : result(true), buffer(b) {
86 }
87};
88} // namespace
89
90bool CodeGen_GPU_Dev::is_buffer_constant(const Stmt &kernel,
91 const std::string &buffer) {
92 IsBufferConstant v(buffer);
93 kernel.accept(&v);
94 return v.result;
95}
