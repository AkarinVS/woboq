137template<typename T>
138struct Rel {
139 typedef typename T::addr_t addr_t;
140 typedef typename T::addr_off_t addr_off_t;
141
142 addr_t r_offset;
143 addr_t r_info;
144
145 uint32_t r_type() const {
146 if (sizeof(addr_t) == 8) {
147 return r_info & 0xffffffff;
148 } else {
149 return r_info & 0xff;
150 }
151 }
152
153 uint32_t r_sym() const {
154 if (sizeof(addr_t) == 8) {
155 return (uint64_t)r_info >> 32;
156 } else {
157 return r_info >> 8;
158 }
159 }
160
161 static addr_t make_info(uint32_t type, uint32_t sym) {
162 if (sizeof(addr_t) == 8) {
163 return (uint64_t)type | ((uint64_t)sym << 32);
164 } else {
165 return (type & 0xff) | (sym << 8);
166 }
167 }
168
169 void set_r_type(uint32_t type) {
170 r_info = make_info(type, r_sym());
171 }
172
173 void set_r_sym(uint32_t sym) {
174 r_info = make_info(r_type(), sym);
175 }
176
177 Rel(addr_t offset, addr_t info)
178 : r_offset(offset), r_info(info) {
179 }
180
181 Rel(addr_t offset, uint32_t type, uint32_t sym)
182 : r_offset(offset), r_info(make_info(type, sym)) {
183 }
184};

245class StringTable {
246 // TODO: We could be smarter and find substrings in the existing
247 // table, not just whole strings. It would probably be fine to just
248 // put every substring of each new string into the cache.
249 std::map<std::string, uint32_t> cache;
250
251public:
252 std::vector<char> table;
253
254 StringTable() {
255 // For our cache to work, we need something in the table to
256 // start with so index 0 isn't valid (it will be the empty
257 // string).
258 table.push_back(0);
259 }
260
261 uint32_t get(const std::string &str) {
262 uint32_t &index = cache[str];
263 if (index == 0) {
264 index = table.size();
265 table.insert(table.end(), str.begin(), str.end());
266 table.push_back(0);
267 }
268 return index;
269 }
270};

282template<typename It>
283void append(std::vector<char> &buf, It begin, It end) {
284 buf.reserve(buf.size() + std::distance(begin, end) * sizeof(*begin));
285 for (It i = begin; i != end; i++) {
286 append_object(buf, *i);
287 }
288}

313unsigned long elf_hash(const char *name) {
314 unsigned long h = 0;
315 unsigned long g;
316 for (char c = *name; c; c = *name++) {
317 h = (h << 4) + c;
318 g = h & 0xf0000000;
319 if (g != 0) {
320 h ^= g >> 24;
321 }
322 h &= ~g;
323 }
324 return h;
325}

327template<typename T>
328std::unique_ptr<Object> parse_object_internal(const char *data, size_t size) {
329 Ehdr<T> header = *(const Ehdr<T> *)data;
330 internal_assert(memcmp(header.e_ident, elf_magic, sizeof(elf_magic)) == 0);
331 internal_assert(header.e_type == Object::ET_REL || header.e_type == Object::ET_DYN);
332
333 std::unique_ptr<Object> obj(new Object());
334 obj->set_type((Object::Type)header.e_type)
335 .set_machine(header.e_machine)
336 .set_version(header.e_version)
337 .set_entry(header.e_entry)
338 .set_flags(header.e_flags);
339
340 auto get_section_header = [&](int idx) -> const Shdr<T> * {
341 const char *at = data + header.e_shoff + idx * header.e_shentsize;
342 internal_assert(data <= at && at + sizeof(Shdr<T>) <= data + size)
343 << "Section header out of bounds.\n";
344 return (const Shdr<T> *)at;
345 };
346
347 // Find the string table.
348 const char *strings = nullptr;
349 for (int i = 0; i < header.e_shnum; i++) {
350 const Shdr<T> *sh = get_section_header(i);
351 if (sh->sh_type == Section::SHT_STRTAB) {
352 internal_assert(!strings) << "Found more than one string table.\n";
353 strings = data + sh->sh_offset;
354 internal_assert(data <= strings && strings + sh->sh_size <= data + size);
355 }
356 }
357 internal_assert(strings)
358 << "String table not found.\n";
359
360 // Load the rest of the sections.
361 std::map<int, Section *> section_map;
362 for (uint16_t i = 0; i < header.e_shnum; i++) {
363 const Shdr<T> *sh = get_section_header(i);
364 if (sh->sh_type != Section::SHT_SYMTAB && sh->sh_type != Section::SHT_STRTAB &&
365 sh->sh_type != Section::SHT_REL && sh->sh_type != Section::SHT_RELA) {
366 const char *name = assert_string_valid(&strings[sh->sh_name], data, size);
367 auto section = obj->add_section(name, (Section::Type)sh->sh_type);
368 section->set_flags(sh->sh_flags)
369 .set_size(sh->sh_size)
370 .set_alignment(sh->sh_addralign);
371 if (sh->sh_type == Section::SHT_NOBITS) {
372 // This section doesn't have any data to load.
373 } else if (sh->sh_type == Section::SHT_NULL) {
374 } else {
375 const char *sh_data = data + sh->sh_offset;
376 internal_assert(data <= sh_data && sh_data + sh->sh_size <= data + size);
377 section->set_contents(sh_data, sh_data + sh->sh_size);
378 }
379 section_map[i] = &*section;
380 }
381 }
382
383 // Find and load the symbols.
384 std::map<int, Symbol *> symbol_map;
385 for (uint16_t i = 0; i < header.e_shnum; i++) {
386 const Shdr<T> *sh = get_section_header(i);
387 if (sh->sh_type == Section::SHT_SYMTAB) {
388 internal_assert(sh->sh_entsize == sizeof(Sym<T>));
389 // Skip symbol 0, which is a null symbol.
390 for (uint64_t j = 1; j < sh->sh_size / sizeof(Sym<T>); ++j) {
391 const char *sym_ptr = data + sh->sh_offset + j * sizeof(Sym<T>);
392 internal_assert(data <= sym_ptr && sym_ptr + sizeof(Sym<T>) <= data + size);
393 const Sym<T> &sym = *(const Sym<T> *)sym_ptr;
394 const char *name = assert_string_valid(&strings[sym.st_name], data, size);
395 auto symbol = obj->add_symbol(name);
396 symbol->set_type((Symbol::Type)sym.get_type())
397 .set_binding((Symbol::Binding)sym.get_binding())
398 .set_visibility((Symbol::Visibility)sym.st_other);
399 if (sym.st_shndx != 0) {
400 symbol->define(section_map[sym.st_shndx], sym.st_value, sym.st_size);
401 }
402 symbol_map[j] = &*symbol;
403 }
404 }
405 }
406
407 // Load relocations.
408 for (uint16_t i = 0; i < header.e_shnum; i++) {
409 const Shdr<T> *sh = get_section_header(i);
410 internal_assert(sh->sh_type != Section::SHT_REL) << "Section::SHT_REL not supported\n";
411 if (sh->sh_type == Section::SHT_RELA) {
412 const char *name = assert_string_valid(&strings[sh->sh_name], data, size);
413 internal_assert(strncmp(name, ".rela.", 6) == 0);
414 internal_assert(sh->sh_entsize == sizeof(Rela<T>));
415 auto to_relocate = obj->find_section(name + 5);
416 internal_assert(to_relocate != obj->sections_end());
417 // TODO: This assert should work, but it seems like this
418 // isn't a reliable test. We rely on the names intead.
419 //internal_assert(&*to_relocate == section_map[sh->sh_link]);
420 for (uint64_t i = 0; i < sh->sh_size / sh->sh_entsize; i++) {
421 const char *rela_ptr = data + sh->sh_offset + i * sh->sh_entsize;
422 internal_assert(data <= rela_ptr && rela_ptr + sizeof(Rela<T>) <= data + size);
423 const Rela<T> &rela = *(const Rela<T> *)rela_ptr;
424 Relocation reloc(rela.r_type(), rela.r_offset, rela.r_addend, symbol_map[rela.r_sym()]);
425 to_relocate->add_relocation(reloc);
426 }
427 }
428 }
429
430 return obj;
431}

475Object::section_iterator Object::merge_sections(const std::vector<section_iterator> &to_merge) {
476 internal_assert(!to_merge.empty());
477 section_iterator merged = *to_merge.begin();
478
479 std::vector<char> contents = merged->get_contents();
480
481 for (auto i = to_merge.begin() + 1; i != to_merge.end(); ++i) {
482 section_iterator s = *i;
483 internal_assert(s->get_type() == merged->get_type());
484
485 // Make the new text section have an alignment that
486 // satisfies all sections. This should be gcd, not max,
487 // but we assume that all of the alignments are powers of
488 // 2.
489 uint64_t alignment = std::max(merged->get_alignment(), s->get_alignment());
490 merged->set_alignment(alignment);
491
492 append_padding(contents, alignment);
493 // The offset of the section in the new merged section.
494 uint64_t offset = contents.size();
495 append(contents, s->contents_begin(), s->contents_end());
496
497 for (auto j = s->relocations_begin(); j != s->relocations_end(); j++) {
498 Elf::Relocation reloc = *j;
499 reloc.set_offset(reloc.get_offset() + offset);
500 merged->add_relocation(reloc);
501 }
502
503 // Find all of the symbols that were defined in this section, and update them.
504 for (auto j = symbols_begin(); j != symbols_end(); j++) {
505 if (j->get_section() == &*s) {
506 j->define(&*merged, j->get_offset() + offset, j->get_size());
507 }
508 }
509 }
510
511 merged->set_contents(contents.begin(), contents.end());
512
513 // Remove all of the sections we merged.
514 for (auto i = to_merge.begin() + 1; i != to_merge.end(); ++i) {
515 erase_section(*i);
516 }
517
518 return merged;
519}

521Object::section_iterator Object::merge_text_sections() {
522 std::vector<section_iterator> text_sections;
523 for (auto i = sections_begin(); i != sections_end(); i++) {
524 if (i->get_type() == Section::SHT_PROGBITS && starts_with(i->get_name(), ".text")) {
525 text_sections.push_back(i);
526 }
527 }
528 section_iterator text = merge_sections(text_sections);
529 text->set_name(".text");
530 return text;

533template<typename T>
534std::vector<char> write_shared_object_internal(Object &obj, Linker *linker, const std::vector<std::string> &dependencies,
535 const std::string &soname) {
536 typedef typename T::addr_t addr_t;
537
538 // The buffer we will be writing to.
539 std::vector<char> output;
540
541 // Declare the things we need to put in the shared object.
542 Ehdr<T> ehdr;
543 std::array<Phdr<T>, 3> phdrs;
544 memset(&ehdr, 0, sizeof(ehdr));
545 memset(&phdrs[0], 0, sizeof(phdrs));
546 auto &text_phdr = phdrs[0];
547 auto &data_phdr = phdrs[1];
548 auto &dyn_phdr = phdrs[2];
549
550 // The text program header starts at the beginning of the object.
551 text_phdr.p_type = PT_LOAD;
552 text_phdr.p_flags = PF_X | PF_R;
553 text_phdr.p_offset = 0;
554 text_phdr.p_align = 4096;
555
556 // We need to build a string table as we go.
557 StringTable strings;
558
559 // And build a list of section headers.
560 std::vector<Shdr<T>> shdrs;
561 // Add the null section now.
562 Shdr<T> sh_null;
563 memset(&sh_null, 0, sizeof(sh_null));
564 shdrs.push_back(sh_null);
565
566 // We also need a mapping of section objects to section headers.
567 std::map<const Section *, uint16_t> section_idxs;
568
569 // Define a helper function to write a section to the shared
570 // object, making a section header for it.
571 auto write_section = [&](const Section &s, uint64_t entsize) {
572 uint64_t alignment = s.get_alignment();
573
574 append_padding(output, alignment);
575 uint64_t offset = output.size();
576 debug(2) << "Writing section " << s.get_name() << " at offset " << offset << "\n";
577 const std::vector<char> &contents = s.get_contents();
578 append(output, contents.begin(), contents.end());
579 if (contents.size() < s.get_size()) {
580 append_zeros(output, s.get_size() - contents.size());
581 }
582 append_padding(output, alignment);
583
584 Shdr<T> shdr;
585 shdr.sh_name = strings.get(s.get_name());
586 safe_assign(shdr.sh_type, s.get_type());
587 safe_assign(shdr.sh_flags, s.get_flags());
588 safe_assign(shdr.sh_offset, offset);
589 safe_assign(shdr.sh_addr, offset);
590 safe_assign(shdr.sh_size, s.get_size());
591 safe_assign(shdr.sh_addralign, alignment);
592
593 shdr.sh_link = 0;
594 shdr.sh_info = 0;
595 safe_assign(shdr.sh_entsize, entsize);
596
597 uint16_t shndx = safe_cast<uint16_t>(shdrs.size());
598 section_idxs[&s] = shndx;
599 shdrs.push_back(shdr);
600 return shndx;
601 };
602
603 // And a helper to get the offset we've given a section.
604 auto get_section_offset = [&](const Section &s) -> uint64_t {
605 uint16_t idx = section_idxs[&s];
606 return shdrs[idx].sh_offset;
607 };
608
609 // We need to define the GOT symbol.
610 uint64_t max_got_size = obj.symbols_size() * 2 * sizeof(addr_t);
611 Section got(".got", Section::SHT_PROGBITS);
612 got.set_alignment(4);
613 got.set_size(max_got_size);
614 got.set_flags(Section::SHF_ALLOC);
615 Symbol got_sym("_GLOBAL_OFFSET_TABLE_");
616 got_sym.define(&got, 0, max_got_size);
617 got_sym.set_type(Symbol::STT_OBJECT);
618 got_sym.set_visibility(Symbol::STV_HIDDEN);
619 Symbol dynamic_sym("_DYNAMIC");
620 dynamic_sym.define(&got, 0, 4);
621 dynamic_sym.set_type(Symbol::STT_OBJECT);
622 got.append_contents((addr_t)0);
623 // On some platforms, GOT slots 1 and 2 are also reserved.
624 got.append_contents((addr_t)0);
625 got.append_contents((addr_t)0);
626
627 // Since we can't change the object, start a map of all of the
628 // symbols that we can mutate. If a symbol from the object is a
629 // key in this map, we use the mapped value instead.
630 std::map<const Symbol *, const Symbol *> symbols;
631 symbols[&dynamic_sym] = &dynamic_sym;
632
633 Object::section_iterator iter_dtors = obj.find_section(".dtors");
634 Symbol dtor_list_sym("__DTOR_LIST__");
635 if (iter_dtors != obj.sections_end()) {
636 Section *dtors = &(*iter_dtors);
637 dtor_list_sym.define(dtors, 0, 0);
638 dtor_list_sym.set_type(Symbol::STT_NOTYPE);
639 dtor_list_sym.set_visibility(Symbol::STV_DEFAULT);
640 dtor_list_sym.set_binding(Symbol::STB_GLOBAL);
641 }
642
643 Object::section_iterator iter_ctors = obj.find_section(".ctors");
644 Symbol ctor_end_sym("__CTOR_END__");
645 if (iter_ctors != obj.sections_end()) {
646 Section *ctors = &(*iter_ctors);
647 internal_assert(ctors->get_size() == ctors->contents_size())
648 << "There should no padding at the end of the .ctors section\n";
649 ctor_end_sym.define(ctors, ctors->get_size(), 0);
650 ctor_end_sym.set_type(Symbol::STT_NOTYPE);
651 ctor_end_sym.set_visibility(Symbol::STV_DEFAULT);
652 ctor_end_sym.set_binding(Symbol::STB_GLOBAL);
653 }
654
655 for (const Symbol &i : obj.symbols()) {
656 if (i.get_name() == "_GLOBAL_OFFSET_TABLE_") {
657 symbols[&i] = &got_sym;
658 } else if (i.get_name() == "__DTOR_LIST__") {
659 // It is our job to create this symbol. So, a defined __DTOR_LIST__
660 // symbol shouldn't be present already.
661 internal_assert(!i.is_defined()) << "__DTOR_LIST__ already defined\n";
662 symbols[&i] = &dtor_list_sym;
663 } else if (i.get_name() == "__CTOR_END__") {
664 internal_assert(!i.is_defined()) << "__CTOR_END__ already defined\n";
665 symbols[&i] = &ctor_end_sym;
666 } else {
667 symbols[&i] = &i;
668 }
669 }
670 // Get a symbol from a relocation, accounting for the symbol map
671 // above.
672 auto get_symbol = [&](const Relocation &r) {
673 const Symbol *sym = r.get_symbol();
674 if (!sym) {
675 return sym;
676 }
677
678 auto i = symbols.find(sym);
679 if (i != symbols.end()) {
680 return i->second;
681 }
682 return sym;
683 };
684
685 // Check if a relocation needs a PLT entry, which adds some
686 // additional conditions on top of what the linker implementation
687 // wants.
688 auto needs_plt_entry = [&](const Relocation &r) {
689 const Symbol *s = get_symbol(r);
690 if (!s || s->is_defined()) {
691 return false;
692 }
693
694 if (s->get_type() != Symbol::STT_NOTYPE) {
695 return false;
696 }
697
698 return linker->needs_plt_entry(r);
699 };
700
701 // We need to build the PLT, so it can be positioned along with
702 // the rest of the text sections.
703 Section plt(".plt", Section::SHT_PROGBITS);
704 plt.set_alignment(16);
705 plt.set_flags(Section::SHF_ALLOC | Section::SHF_EXECINSTR);
706 std::list<Symbol> plt_symbols;
707 std::map<const Symbol *, const Symbol *> plt_defs;
708 // Hack: We're defining the global offset table, so it shouldn't be treated as an external symbol.
709 plt_defs[&got_sym] = &got_sym;
710 for (const Section &s : obj.sections()) {
711 for (const Relocation &r : s.relocations()) {
712 if (!needs_plt_entry(r)) {
713 continue;
714 }
715
716 const Symbol *sym = get_symbol(r);
717 const Symbol *&plt_def = plt_defs[sym];
718 if (plt_def) {
719 // We already made a PLT entry for this symbol.
720 continue;
721 }
722
723 debug(2) << "Defining PLT entry for " << sym->get_name() << "\n";
724 plt_symbols.push_back(linker->add_plt_entry(*sym, plt, got, got_sym));
725
726 plt_def = &plt_symbols.back();
727 symbols[plt_def] = plt_def;
728 }
729 }
730
731 // Start placing the sections into the shared object.
732
733 // Leave room for the header, and program headers at the beginning of the file.
734 append_zeros(output, sizeof(ehdr));
735 append_zeros(output, sizeof(phdrs[0]) * 3);
736
737 // We need to perform the relocations. To do that, we need to position the sections
738 // where they will go in the final shared object.
739 write_section(plt, 0);
740 for (const Section &s : obj.sections()) {
741 if (s.is_alloc() && !s.is_writable()) {
742 write_section(s, 0);
743 }
744 }
745 append_padding(output, 4096);
746 text_phdr.p_filesz = output.size() - text_phdr.p_offset;
747
748 data_phdr.p_type = PT_LOAD;
749 data_phdr.p_flags = PF_W | PF_R;
750 safe_assign(data_phdr.p_offset, output.size());
751 data_phdr.p_align = 4096;
752 for (const Section &s : obj.sections()) {
753 if (s.is_alloc() && s.is_writable()) {
754 write_section(s, 0);
755 }
756 }
757
758 // The got will be written again later, after we add entries to it.
759 write_section(got, 0);
760
761 /// Now that we've written the sections that define symbols, we
762 // can generate the symbol table.
763 Section symtab(".symtab", Section::SHT_SYMTAB);
764 symtab.set_alignment(4);
765 symtab.set_flag(Section::SHF_ALLOC);
766 std::vector<Sym<T>> syms;
767 Sym<T> undef_sym;
768 memset(&undef_sym, 0, sizeof(undef_sym));
769 syms.push_back(undef_sym);
770
771 // Ensure that we output the symbols deterministically, since a map of pointers
772 // will vary in ordering from run to tun.
773 std::vector<std::pair<const Symbol *, const Symbol *>> sorted_symbols;
774 for (const auto &i : symbols) {
775 sorted_symbols.emplace_back(i);
776 }
777 std::sort(sorted_symbols.begin(), sorted_symbols.end(),
778 [&](const std::pair<const Symbol *, const Symbol *> &lhs, const std::pair<const Symbol *, const Symbol *> &rhs) {
779 return lhs.first->get_name() < rhs.first->get_name();
780 });
781
782 std::map<const Symbol *, uint16_t> symbol_idxs;
783 uint64_t local_count = 0;
784 for (bool is_local : {true, false}) {
785 for (const auto &i : sorted_symbols) {
786 const Symbol *s = i.second;
787 if ((s->get_binding() == Symbol::STB_LOCAL) != is_local) {
788 continue;
789 }
790
791 uint64_t value = s->get_offset();
792 // In shared objects, the symbol value is a virtual address,
793 // not a section offset.
794 if (s->is_defined()) {
795 value += get_section_offset(*s->get_section());
796 }
797 Sym<T> sym;
798 safe_assign(sym.st_name, strings.get(s->get_name()));
799 safe_assign(sym.st_value, value);
800 safe_assign(sym.st_size, s->get_size());
801 sym.set_type(s->get_type());
802 sym.set_binding(s->get_binding());
803 safe_assign(sym.st_other, s->get_visibility());
804 sym.st_shndx = section_idxs[s->get_section()];
805
806 safe_assign(symbol_idxs[s], syms.size());
807 syms.push_back(sym);
808 }
809 if (is_local) {
810 local_count = syms.size();
811 }
812 }
813 symtab.set_contents(syms);
814 uint16_t symtab_idx = write_section(symtab, sizeof(syms[0]));
815 safe_assign(shdrs[symtab_idx].sh_info, local_count);
816
817 // Also write the symbol table as SHT_DYNSYM.
818 Section dynsym = symtab;
819 dynsym.set_name(".dynsym");
820 dynsym.set_type(Section::SHT_DYNSYM);
821 uint16_t dynsym_idx = write_section(dynsym, sizeof(syms[0]));
822 shdrs[dynsym_idx].sh_info = local_count;
823
824 // We really do need to make a hash table. Make a trivial one with one bucket.
825 Section hash(".hash", Section::SHT_HASH);
826 hash.set_alignment(4);
827 hash.set_flag(Section::SHF_ALLOC);
828 size_t sym_count = syms.size();
829 // TODO: Fix non-trivial hash tables so they work with dlsym.
830 size_t bucket_count = 1;
831 std::vector<uint32_t> hash_table(bucket_count + sym_count + 2);
832 safe_assign(hash_table[0], bucket_count);
833 safe_assign(hash_table[1], sym_count);
834 uint32_t *buckets = &hash_table[2];
835 uint32_t *chains = buckets + bucket_count;
836 for (size_t i = 0; i < sym_count; i++) {
837 const char *name = &strings.table[syms[i].st_name];
838 uint32_t hash = elf_hash(name) % bucket_count;
839 chains[i] = buckets[hash];
840 safe_assign(buckets[hash], i);
841 }
842 hash.set_contents(hash_table);
843 uint16_t hash_idx = write_section(hash, sizeof(hash_table[0]));
844
845 auto do_relocations = [&](const Section &s) {
846 debug(2) << "Processing relocations for section " << s.get_name() << "\n";
847 for (const Relocation &r : s.relocations()) {
848 const Symbol *sym = get_symbol(r);
849 if (needs_plt_entry(r)) {
850 // This relocation is a function call, we need to use the PLT entry for this symbol.
851 auto plt_def = plt_defs.find(sym);
852 internal_assert(plt_def != plt_defs.end());
853 debug(2) << "Using PLT entry " << plt_def->second->get_name() << " for symbol " << sym->get_name() << "\n";
854 sym = plt_def->second;
855 }
856
857 uint64_t fixup_offset = get_section_offset(s) + r.get_offset();
858 char *fixup_addr = output.data() + fixup_offset;
859 uint64_t sym_offset = 0;
860 if (sym && sym->is_defined()) {
861 sym_offset = get_section_offset(*sym->get_section()) + sym->get_offset();
862 debug(2) << "Symbol " << sym->get_name() << " is defined at " << sym_offset << "\n";
863 }
864 Relocation new_reloc = linker->relocate(fixup_offset, fixup_addr, r.get_type(), sym, sym_offset, r.get_addend(), got);
865 if (new_reloc.get_type() != 0) {
866 // The linker wants a dynamic relocation here. This
867 // section must be writable at runtime.
868 internal_assert(s.is_writable());
869 debug(2) << "Linker returned new relocation type " << new_reloc.get_type() << "\n";
870 new_reloc.set_offset(new_reloc.get_offset() - get_section_offset(got));
871 got.add_relocation(new_reloc);
872 }
873 }
874 };
875
876 // Now that we've generated the symbol table, we can do relocations.
877 do_relocations(plt);
878 for (const Section &s : obj.sections()) {
879 do_relocations(s);
880 }
881
882 // Now we can write the GOT.
883 internal_assert(got.contents_size() <= max_got_size);
884 memcpy(output.data() + get_section_offset(got), got.contents_data(), got.contents_size());
885
886 auto write_relocation_section = [&](const Section &s) {
887 uint64_t alignment = 8;
888 append_padding(output, alignment);
889 uint64_t offset = output.size();
890 for (const Relocation &r : s.relocations()) {
891 uint64_t i_offset = get_section_offset(s) + r.get_offset();
892 Rela<T> rela(i_offset, r.get_type(), symbol_idxs[get_symbol(r)], r.get_addend());
893 append_object(output, rela);
894 }
895 uint64_t size = output.size() - offset;
896 append_padding(output, alignment);
897
898 Shdr<T> shdr;
899 safe_assign(shdr.sh_name, strings.get(".rela" + s.get_name()));
900 shdr.sh_type = Section::SHT_RELA;
901 shdr.sh_flags = Section::SHF_ALLOC;
902 safe_assign(shdr.sh_offset, offset);
903 safe_assign(shdr.sh_addr, offset);
904 safe_assign(shdr.sh_size, size);
905 safe_assign(shdr.sh_addralign, alignment);
906
907 safe_assign(shdr.sh_link, symtab_idx);
908 safe_assign(shdr.sh_info, section_idxs[&s]);
909 shdr.sh_entsize = sizeof(Rela<T>);
910
911 uint16_t shndx = safe_cast<uint16_t>(shdrs.size());
912 shdrs.push_back(shdr);
913 return shndx;
914 };
915
916 addr_t rela_got_idx = write_relocation_section(got);
917
918 // Add some strings we know we'll need in the string table after we write it.
919 strings.get(soname);
920 for (const auto &i : dependencies) {
921 strings.get(i);
922 }
923
924 Section dynamic(".dynamic", Section::SHT_DYNAMIC);
925 strings.get(dynamic.get_name());
926 dynamic.set_alignment(4);
927 dynamic.set_flag(Section::SHF_ALLOC);
928 Section strtab(".strtab", Section::SHT_STRTAB);
929 strings.get(strtab.get_name());
930 strtab.set_flag(Section::SHF_ALLOC);
931 strtab.set_contents(strings.table);
932 uint16_t strtab_idx = write_section(strtab, 0);
933
934 std::vector<Dyn<T>> dyn;
935 auto make_dyn = [](int32_t tag, addr_t val) {
936 Dyn<T> d;
937 d.d_tag = tag;
938 d.d_val = val;
939 return d;
940 };
941
942 for (const auto &i : dependencies) {
943 dyn.push_back(make_dyn(DT_NEEDED, strings.get(i)));
944 }
945 if (!soname.empty()) {
946 dyn.push_back(make_dyn(DT_SONAME, strings.get(soname)));
947 }
948 dyn.push_back(make_dyn(DT_SYMBOLIC, 0));
949
950 // This is really required...
951 dyn.push_back(make_dyn(DT_HASH, get_section_offset(hash)));
952
953 // Address of the symbol table.
954 dyn.push_back(make_dyn(DT_SYMTAB, shdrs[dynsym_idx].sh_offset));
955 dyn.push_back(make_dyn(DT_SYMENT, shdrs[dynsym_idx].sh_entsize));
956
957 // Address of the string table.
958 dyn.push_back(make_dyn(DT_STRTAB, get_section_offset(strtab)));
959 dyn.push_back(make_dyn(DT_STRSZ, strtab.get_size()));
960
961 // Offset to the GOT.
962 dyn.push_back(make_dyn(DT_PLTGOT, get_section_offset(got)));
963
964 // Relocations associated with the PLT.
965 addr_t pltrelsz = sizeof(Rela<T>) * plt_symbols.size();
966 dyn.push_back(make_dyn(DT_JMPREL, shdrs[rela_got_idx].sh_offset));
967 dyn.push_back(make_dyn(DT_PLTREL, DT_RELA));
968 dyn.push_back(make_dyn(DT_PLTRELSZ, pltrelsz));
969
970 // Other relocations.
971 dyn.push_back(make_dyn(DT_RELA, shdrs[rela_got_idx].sh_offset + pltrelsz));
972 dyn.push_back(make_dyn(DT_RELASZ, shdrs[rela_got_idx].sh_size - pltrelsz));
973 dyn.push_back(make_dyn(DT_RELAENT, sizeof(Rela<T>)));
974
975 // DT_FINI
976 Object::section_iterator iter_fini = obj.find_section(".fini.halide");
977 if (iter_fini != obj.sections_end()) {
978 Section &fini = *iter_fini;
979 dyn.push_back(make_dyn(DT_FINI, get_section_offset(fini)));
980 }
981
982 // DT_INIT
983 Object::section_iterator iter_init = obj.find_section(".init.halide");
984 if (iter_init != obj.sections_end()) {
985 Section &init = *iter_init;
986 dyn.push_back(make_dyn(DT_INIT, get_section_offset(init)));
987 }
988
989 dynamic.set_contents(dyn);
990
991 // Add any target specific stuff.
992 linker->append_dynamic(dynamic);
993
994 // Null terminator.
995 dynamic.append_contents((uint32_t)DT_NULL);
996 dynamic.append_contents((addr_t)0);
997
998 uint16_t dyn_idx = write_section(dynamic, sizeof(dyn[0]));
999 dyn_phdr.p_type = PT_DYNAMIC;
1000 dyn_phdr.p_offset = shdrs[dyn_idx].sh_offset;
1001 dyn_phdr.p_flags = PF_R;
1002 dyn_phdr.p_filesz = shdrs[dyn_idx].sh_size;
1003 dyn_phdr.p_memsz = dyn_phdr.p_filesz;
1004 dyn_phdr.p_align = 4;
1005
1006 append_padding(output, 4096);
1007 safe_assign(data_phdr.p_filesz, output.size() - data_phdr.p_offset);
1008
1009 // Setup the section headers.
1010 shdrs[symtab_idx].sh_link = strtab_idx;
1011 shdrs[dynsym_idx].sh_link = strtab_idx;
1012 shdrs[dyn_idx].sh_link = strtab_idx;
1013 shdrs[hash_idx].sh_link = dynsym_idx;
1014
1015 // Write the section header table.
1016 ehdr.e_shoff = output.size();
1017 ehdr.e_shnum = shdrs.size();
1018 ehdr.e_shentsize = sizeof(shdrs[0]);
1019 for (auto &i : shdrs) {
1020 append_object(output, i);
1021 }
1022
1023 // Now go back and write the headers.
1024 memcpy(ehdr.e_ident, elf_magic, 4);
1025 ehdr.e_ident[4] = 1;
1026 ehdr.e_ident[5] = 1;
1027 ehdr.e_ident[6] = 1;
1028 ehdr.e_type = Object::ET_DYN;
1029 ehdr.e_machine = linker->get_machine();
1030 ehdr.e_ehsize = sizeof(ehdr);
1031 ehdr.e_version = linker->get_version();
1032 ehdr.e_entry = obj.get_entry();
1033 ehdr.e_flags = linker->get_flags();
1034 ehdr.e_phoff = sizeof(ehdr);
1035 ehdr.e_phentsize = sizeof(phdrs[0]);
1036 ehdr.e_phnum = phdrs.size();
1037 ehdr.e_shstrndx = strtab_idx;
1038
1039 memcpy(output.data(), &ehdr, sizeof(ehdr));
1040 for (auto &i : phdrs) {
1041 i.p_vaddr = i.p_offset;
1042 i.p_paddr = i.p_offset;
1043 i.p_memsz = i.p_filesz;
1044 }
1045 memcpy(output.data() + ehdr.e_phoff, phdrs.data(), sizeof(phdrs));
1046
1047 return output;
1048}
