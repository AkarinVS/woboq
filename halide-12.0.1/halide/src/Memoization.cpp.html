19class FindParameterDependencies : public IRGraphVisitor {
20public:
21 FindParameterDependencies() = default;
22 ~FindParameterDependencies() override = default;
23
24 void visit_function(const Function &function) {
25 function.accept(this);
26
27 if (function.has_extern_definition()) {
28 const std::vector<ExternFuncArgument> &extern_args =
29 function.extern_arguments();
30 for (size_t i = 0; i < extern_args.size(); i++) {
31 if (extern_args[i].is_buffer()) {
32 // Function with an extern definition
33 record(Halide::Internal::Parameter(extern_args[i].buffer.type(), true,
34 extern_args[i].buffer.dimensions(),
35 extern_args[i].buffer.name()));
36 } else if (extern_args[i].is_image_param()) {
37 record(extern_args[i].image_param);
38 }
39 }
40 }
41 }
42
43 using IRGraphVisitor::visit;
44
45 void visit(const Call *call) override {
46 if (call->param.defined()) {
47 record(call->param);
48 }
49
50 if (call->is_intrinsic(Call::memoize_expr)) {
51 internal_assert(!call->args.empty());
52 if (call->args.size() == 1) {
53 record(call->args[0]);
54 } else {
55 // Do not look at anything inside a memoize_expr bracket.
56 for (size_t i = 1; i < call->args.size(); i++) {
57 record(call->args[i]);
58 }
59 }
60 } else if (call->func.defined()) {
61 Function fn(call->func);
62 visit_function(fn);
63 IRGraphVisitor::visit(call);
64 } else {
65 IRGraphVisitor::visit(call);
66 }
67 }
68
69 void visit(const Load *load) override {
70 if (load->param.defined()) {
71 record(load->param);
72 }
73 IRGraphVisitor::visit(load);
74 }
75
76 void visit(const Variable *var) override {
77 if (var->param.defined()) {
78 if (var->param.is_buffer() &&
79 !var->type.is_handle()) {
80 record(memoize_tag(var));
81 } else {
82 record(var->param);
83 }
84 }
85
86 IRGraphVisitor::visit(var);
87 }
88
89 void record(const Parameter &parameter) {
90 struct DependencyInfo info;
91
92 info.type = parameter.type();
93
94 if (parameter.is_buffer()) {
95 internal_error
96 << "Buffer parameter " << parameter.name()
97 << " encountered in computed_cached computation.\n"
98 << "Computations which depend on buffer parameters "
99 << "cannot be scheduled compute_cached.\n"
100 << "Use memoize_tag to provide cache key information for buffer.\n";
101 } else if (info.type.is_handle()) {
102 internal_error
103 << "Handle parameter " << parameter.name()
104 << " encountered in computed_cached computation.\n"
105 << "Computations which depend on handle parameters "
106 << "cannot be scheduled compute_cached.\n"
107 << "Use memoize_tag to provide cache key information for handle.\n";
108 } else {
109 info.size_expr = info.type.bytes();
110 info.value_expr = Internal::Variable::make(info.type, parameter.name(), parameter);
111 }
112
113 dependency_info[DependencyKey(info.type.bytes(), parameter.name())] = info;
114 }
115
116 void record(const Expr &expr) {
117 struct DependencyInfo info;
118 info.type = expr.type();
119 info.size_expr = info.type.bytes();
120 info.value_expr = expr;
121 dependency_info[DependencyKey(info.type.bytes(), unique_name("memoize_tag"))] = info;
122 }

124 // Used to make sure larger parameters come before smaller ones
125 // for alignment reasons.
126 struct DependencyKey {
127 uint32_t size;
128 std::string name;
129
130 bool operator<(const DependencyKey &rhs) const {
131 if (size < rhs.size) {
132 return true;
133 } else if (size == rhs.size) {
134 return name < rhs.name;
135 }
136 return false;
137 }
138
139 DependencyKey(uint32_t size_arg, const std::string &name_arg)
140 : size(size_arg), name(name_arg) {
141 }
142 };
143
144 struct DependencyInfo {
145 Type type;
146 Expr size_expr;
147 Expr value_expr;
148 };
149
150 std::map<DependencyKey, DependencyInfo> dependency_info;
151};

153typedef std::pair<FindParameterDependencies::DependencyKey, FindParameterDependencies::DependencyInfo> DependencyKeyInfoPair;
154typedef std::pair<const FindParameterDependencies::DependencyKey, FindParameterDependencies::DependencyInfo> ConstDependencyKeyInfoPair;

156class KeyInfo {
157 FindParameterDependencies dependencies;
158 Expr key_size_expr;
159 const std::string &top_level_name;
160 const std::string &function_name;
161 int memoize_instance;
162
163 size_t parameters_alignment() {
164 int32_t max_alignment = 0;
165 // Find maximum natural alignment needed.
166 for (const ConstDependencyKeyInfoPair &i : dependencies.dependency_info) {
167 int alignment = i.second.type.bytes();
168 if (alignment > max_alignment) {
169 max_alignment = alignment;
170 }
171 }
172 // Make sure max_alignment is a power of two and has maximum value of 32
173 int i = 0;
174 while (i < 4 && max_alignment > (1 << i)) {
175 i = i + 1;
176 }
177 return size_t(1) << i;
178 }
179
180 // TODO: Using the full names in the key results in a (hopefully incredibly
181 // slight) performance difference based on how one names filters and
182 // functions. It is arguably a little easier to debug if something
183 // goes wrong as one doesn't need to destructure the cache key by hand
184 // in the debugger. Also, if a pointer is used, a counter must also be
185 // put in the cache key to avoid aliasing on reuse of the address in
186 // JIT situations where code is regenerated into the same region of
187 // memory.
188 //
189 // There is a plan to change the hash function used in the cache and
190 // after that happens, we'll measure performance again and maybe decide
191 // to choose one path or the other (see Git history for the implementation.
192 // It was deleted as part of the address_of intrinsic cleanup).
193
194public:
195 KeyInfo(const Function &function, const std::string &name, int memoize_instance)
196 : top_level_name(name),
197 function_name(function.origin_name()),
198 memoize_instance(memoize_instance) {
199 dependencies.visit_function(function);
200 size_t size_so_far = 0;
201 size_so_far += Handle().bytes() + 4;
202
203 size_t needed_alignment = parameters_alignment();
204 if (needed_alignment > 1) {
205 size_so_far = (size_so_far + needed_alignment - 1) & ~(needed_alignment - 1);
206 }
207 key_size_expr = (int32_t)size_so_far;
208
209 for (const ConstDependencyKeyInfoPair &i : dependencies.dependency_info) {
210 key_size_expr += i.second.size_expr;
211 }
212 }
213
214 // Return the number of bytes needed to store the cache key
215 // for the target function. Make sure it takes 4 bytes in cache key.
216 Expr key_size() {
217 return cast<int32_t>(key_size_expr);
218 }
219
220 // Code to fill in the Allocation named key_name with the byte of
221 // the key. The Allocation is guaranteed to be 1d, of type uint8_t
222 // and of the size returned from key_size
223 Stmt generate_key(const std::string &key_name) {
224 std::vector<Stmt> writes;
225 Expr index = Expr(0);
226
227 // Store a pointer to a string identifying the filter and
228 // function. Assume this will be unique due to CSE. This can
229 // break with loading and unloading of code, though the name
230 // mechanism can also break in those conditions.
231 writes.push_back(Store::make(key_name,
232 StringImm::make(std::to_string(top_level_name.size()) + ":" + top_level_name +
233 std::to_string(function_name.size()) + ":" + function_name),
234 (index / Handle().bytes()), Parameter(), const_true(), ModulusRemainder()));
235 size_t alignment = Handle().bytes();
236 index += Handle().bytes();
237
238 // Halide compilation is not threadsafe anyway...
239 writes.push_back(Store::make(key_name,
240 memoize_instance,
241 (index / Int(32).bytes()),
242 Parameter(), const_true(), ModulusRemainder()));
243 alignment += 4;
244 index += 4;
245
246 size_t needed_alignment = parameters_alignment();
247 if (needed_alignment > 1) {
248 while (alignment % needed_alignment) {
249 writes.push_back(Store::make(key_name, Cast::make(UInt(8), 0),
250 index, Parameter(), const_true(), ModulusRemainder()));
251 index = index + 1;
252 alignment++;
253 }
254 }
255
256 for (const ConstDependencyKeyInfoPair &i : dependencies.dependency_info) {
257 writes.push_back(Store::make(key_name,
258 i.second.value_expr,
259 (index / i.second.size_expr),
260 Parameter(), const_true(), ModulusRemainder()));
261 index += i.second.size_expr;
262 }
263 Stmt blocks = Block::make(writes);
264
265 return blocks;
266 }
267
268 // Returns a bool expression, which either evaluates to true,
269 // in which case the Allocation named by storage will be computed,
270 // or false, in which case it will be assumed the buffer was populated
271 // by the code in this call.
272 Expr generate_lookup(const std::string &key_allocation_name, const std::string &computed_bounds_name,
273 int32_t tuple_count, const std::string &storage_base_name) {
274 std::vector<Expr> args;
275 args.push_back(Variable::make(type_of<uint8_t *>(), key_allocation_name));
276 args.push_back(key_size());
277 args.push_back(Variable::make(type_of<halide_buffer_t *>(), computed_bounds_name));
278 args.emplace_back(tuple_count);
279 std::vector<Expr> buffers;
280 if (tuple_count == 1) {
281 buffers.push_back(Variable::make(type_of<halide_buffer_t *>(), storage_base_name + ".buffer"));
282 } else {
283 for (int32_t i = 0; i < tuple_count; i++) {
284 buffers.push_back(Variable::make(type_of<halide_buffer_t *>(), storage_base_name + "." + std::to_string(i) + ".buffer"));
285 }
286 }
287 args.push_back(Call::make(type_of<halide_buffer_t **>(), Call::make_struct, buffers, Call::Intrinsic));
288
289 return Call::make(Int(32), "halide_memoization_cache_lookup", args, Call::Extern);
290 }
291
292 // Returns a statement which will store the result of a computation under this key
293 Stmt store_computation(const std::string &key_allocation_name, const std::string &computed_bounds_name,
294 const std::string &eviction_key_name, int32_t tuple_count, const std::string &storage_base_name) {
295 std::vector<Expr> args;
296 args.push_back(Variable::make(type_of<uint8_t *>(), key_allocation_name));
297 args.push_back(key_size());
298 args.push_back(Variable::make(type_of<halide_buffer_t *>(), computed_bounds_name));
299 args.emplace_back(tuple_count);
300 std::vector<Expr> buffers;
301 if (tuple_count == 1) {
302 buffers.push_back(Variable::make(type_of<halide_buffer_t *>(), storage_base_name + ".buffer"));
303 } else {
304 for (int32_t i = 0; i < tuple_count; i++) {
305 buffers.push_back(Variable::make(type_of<halide_buffer_t *>(), storage_base_name + "." + std::to_string(i) + ".buffer"));
306 }
307 }
308 args.push_back(Call::make(type_of<halide_buffer_t **>(), Call::make_struct, buffers, Call::Intrinsic));
309 if (!eviction_key_name.empty()) {
310 args.push_back(make_const(Bool(), true));
311 args.push_back(Variable::make(UInt(64), eviction_key_name));
312 } else {
313 args.push_back(make_const(Bool(), false));
314 args.push_back(make_const(UInt(64), 0));
315 }
316 // This is actually a void call. How to indicate that? Look at Extern_ stuff.
317 return Evaluate::make(Call::make(Int(32), "halide_memoization_cache_store", args, Call::Extern));
318 }

321// Inject caching structure around memoized realizations.
322class InjectMemoization : public IRMutator {
323public:
324 const std::map<std::string, Function> &env;
325 int memoize_instance;
326 const std::string &top_level_name;
327 const std::vector<Function> &outputs;
328
329 InjectMemoization(const std::map<std::string, Function> &e,
330 int memoize_instance,
331 const std::string &name,
332 const std::vector<Function> &outputs)
333 : env(e), memoize_instance(memoize_instance), top_level_name(name), outputs(outputs) {
334 }
335
336private:
337 using IRMutator::visit;
338
339 Stmt visit(const Realize *op) override {
340 std::map<std::string, Function>::const_iterator iter = env.find(op->name);
341 if (iter != env.end() &&
342 iter->second.schedule().memoized()) {
343
344 const Function f(iter->second);
345
346 for (const Function &o : outputs) {
347 if (f.same_as(o)) {
348 user_error << "Function " << f.name() << " cannot be memoized because "
349 << "it an output of pipeline " << top_level_name << ".\n";
350 }
351 }
352
353 // There are currently problems with the cache key
354 // construction getting moved above the scope of use if
355 // the the compute and store levels are different. It also
356 // has implications for the cache compute/allocated bounds
357 // logic. And it isn't clear it is useful for
358 // anything. Hence this is currently an error.
359 if (!f.schedule().compute_level().match(f.schedule().store_level())) {
360 user_error << "Function " << f.name() << " cannot be memoized because "
361 << "it has compute and storage scheduled at different loop levels.\n";
362 }
363
364 Stmt mutated_body = mutate(op->body);
365
366 KeyInfo key_info(f, top_level_name, memoize_instance);
367
368 std::string cache_key_name = op->name + ".cache_key";
369 std::string cache_result_name = op->name + ".cache_result";
370 std::string cache_miss_name = op->name + ".cache_miss";
371 std::string computed_bounds_name = op->name + ".computed_bounds.buffer";
372 std::string eviction_key_name = op->name + ".cache_eviction_key";
373
374 Stmt eviction_key_marker;
375 const Expr &eviction_key = iter->second.schedule().memoize_eviction_key();
376 bool has_eviction_key = eviction_key.defined();
377 if (has_eviction_key) {
378 internal_assert(eviction_key.type() == UInt(64)) << "Logic error: bad type for memoization eviction key in expr: " << eviction_key << " .\n";
379 eviction_key_marker = LetStmt::make(eviction_key_name, eviction_key, mutated_body);
380 }
381
382 Stmt cache_miss_marker = LetStmt::make(cache_miss_name,
383 Cast::make(Bool(), Variable::make(Int(32), cache_result_name)),
384 has_eviction_key ? eviction_key_marker : mutated_body);
385 Stmt cache_lookup_check = Block::make(AssertStmt::make(NE::make(Variable::make(Int(32), cache_result_name), -1),
386 Call::make(Int(32), "halide_error_out_of_memory", {}, Call::Extern)),
387 cache_miss_marker);
388
389 Stmt cache_lookup = LetStmt::make(cache_result_name,
390 key_info.generate_lookup(cache_key_name, computed_bounds_name, f.outputs(), op->name),
391 cache_lookup_check);
392
393 BufferBuilder builder;
394 builder.dimensions = f.dimensions();
395 std::string max_stage_num = std::to_string(f.updates().size());
396 for (const std::string &arg : f.args()) {
397 std::string prefix = op->name + ".s" + max_stage_num + "." + arg;
398 Expr min = Variable::make(Int(32), prefix + ".min");
399 Expr max = Variable::make(Int(32), prefix + ".max");
400 builder.mins.push_back(min);
401 builder.extents.push_back(max + 1 - min);
402 }
403 Expr computed_bounds = builder.build();
404
405 Stmt computed_bounds_let = LetStmt::make(computed_bounds_name, computed_bounds, cache_lookup);
406
407 Stmt generate_key = Block::make(key_info.generate_key(cache_key_name), computed_bounds_let);
408 Stmt cache_key_alloc =
409 Allocate::make(cache_key_name, UInt(8), MemoryType::Stack, {key_info.key_size()},
410 const_true(), generate_key);
411
412 return Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, cache_key_alloc);
413 } else {
414 return IRMutator::visit(op);
415 }
416 }
417
418 Stmt visit(const ProducerConsumer *op) override {
419 std::map<std::string, Function>::const_iterator iter = env.find(op->name);
420 if (iter != env.end() &&
421 iter->second.schedule().memoized()) {
422
423 // The error checking should have been done inside Realization node
424 // of this producer, so no need to do it here.
425
426 Stmt body = mutate(op->body);
427
428 std::string cache_miss_name = op->name + ".cache_miss";
429 Expr cache_miss = Variable::make(Bool(), cache_miss_name);
430
431 if (op->is_producer) {
432 Stmt mutated_body = IfThenElse::make(cache_miss, body);
433 return ProducerConsumer::make(op->name, op->is_producer, mutated_body);
434 } else {
435 const Function f(iter->second);
436 KeyInfo key_info(f, top_level_name, memoize_instance);
437
438 std::string cache_key_name = op->name + ".cache_key";
439 std::string computed_bounds_name = op->name + ".computed_bounds.buffer";
440 std::string eviction_key_name;
441
442 if (f.schedule().memoize_eviction_key().defined()) {
443 eviction_key_name = op->name + ".cache_eviction_key";
444 }
445 Stmt cache_store_back =
446 IfThenElse::make(cache_miss,
447 key_info.store_computation(cache_key_name, computed_bounds_name,
448 eviction_key_name, f.outputs(), op->name));
449
450 Stmt mutated_body = Block::make(cache_store_back, body);
451 return ProducerConsumer::make(op->name, op->is_producer, mutated_body);
452 }
453 } else {
454 return IRMutator::visit(op);
455 }
456 }
457};

461Stmt inject_memoization(const Stmt &s, const std::map<std::string, Function> &env,
462 const std::string &name,
463 const std::vector<Function> &outputs) {
464 // Cache keys use the addresses of names of Funcs. For JIT, a
465 // counter for the pipeline is needed as the address may be reused
466 // across pipelines. This isn't a problem when using full names as
467 // the function names already are uniquefied by a counter.
468 static std::atomic<int> memoize_instance{0};
469
470 InjectMemoization injector(env, memoize_instance++, name, outputs);
471
472 return injector.mutate(s);
473}

477class RewriteMemoizedAllocations : public IRMutator {
478public:
479 RewriteMemoizedAllocations(const std::map<std::string, Function> &e)
480 : env(e) {
481 }
482
483private:
484 const std::map<std::string, Function> &env;
485 std::map<std::string, std::vector<const Allocate *>> pending_memoized_allocations;
486 std::string innermost_realization_name;
487
488 std::string get_realization_name(const std::string &allocation_name) {
489 std::string realization_name = allocation_name;
490 size_t off = realization_name.rfind('.');
491 if (off != std::string::npos) {
492 size_t i = off + 1;
493 while (i < realization_name.size() && isdigit(realization_name[i])) {
494 i++;
495 }
496 if (i == realization_name.size()) {
497 realization_name = realization_name.substr(0, off);
498 }
499 }
500 return realization_name;
501 }
502
503 using IRMutator::visit;
504
505 Stmt visit(const Allocate *allocation) override {
506 std::string realization_name = get_realization_name(allocation->name);
507 std::map<std::string, Function>::const_iterator iter = env.find(realization_name);
508
509 if (iter != env.end() && iter->second.schedule().memoized()) {
510 ScopedValue<std::string> old_innermost_realization_name(innermost_realization_name, realization_name);
511 pending_memoized_allocations[innermost_realization_name].push_back(allocation);
512 return mutate(allocation->body);
513 } else {
514 return IRMutator::visit(allocation);
515 }
516 }
517
518 Expr visit(const Call *call) override {
519 if (!innermost_realization_name.empty() &&
520 call->name == Call::buffer_init) {
521 internal_assert(call->args.size() >= 3)
522 << "RewriteMemoizedAllocations: _halide_buffer_init call with fewer than two args.\n";
523
524 // Grab the host pointer argument
525 const Variable *var = call->args[2].as<Variable>();
526 if (var && get_realization_name(var->name) == innermost_realization_name) {
527 // Rewrite _halide_buffer_init to use a nullptr handle for address.
528 std::vector<Expr> args = call->args;
529 args[2] = make_zero(Handle());
530 return Call::make(type_of<struct halide_buffer_t *>(), Call::buffer_init,
531 args, Call::Extern);
532 }
533 }
534
535 // If any part of the match failed, do default mutator action.
536 return IRMutator::visit(call);
537 }
538
539 Stmt visit(const LetStmt *let) override {
540 if (let->name == innermost_realization_name + ".cache_miss") {
541 Expr value = mutate(let->value);
542 Stmt body = mutate(let->body);
543
544 std::vector<const Allocate *> &allocations = pending_memoized_allocations[innermost_realization_name];
545
546 for (size_t i = allocations.size(); i > 0; i--) {
547 const Allocate *allocation = allocations[i - 1];
548
549 // Make the allocation node
550 body = Allocate::make(allocation->name, allocation->type, allocation->memory_type, allocation->extents, allocation->condition, body,
551 Call::make(Handle(), Call::buffer_get_host,
552 {Variable::make(type_of<struct halide_buffer_t *>(), allocation->name + ".buffer")}, Call::Extern),
553 "halide_memoization_cache_release");
554 }
555
556 pending_memoized_allocations.erase(innermost_realization_name);
557
558 return LetStmt::make(let->name, value, body);
559 } else {
560 return IRMutator::visit(let);
561 }
562 }
563};
