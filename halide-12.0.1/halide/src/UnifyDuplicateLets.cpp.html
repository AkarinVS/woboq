14class UnifyDuplicateLets : public IRMutator {
15 using IRMutator::visit;
16
17 map<Expr, string, IRDeepCompare> scope;
18 map<string, string> rewrites;
19 string producing;
20
21public:
22 using IRMutator::mutate;
23
24 Expr mutate(const Expr &e) override {
25 if (e.defined()) {
26 map<Expr, string, IRDeepCompare>::iterator iter = scope.find(e);
27 if (iter != scope.end()) {
28 return Variable::make(e.type(), iter->second);
29 } else {
30 return IRMutator::mutate(e);
31 }
32 } else {
33 return Expr();
34 }
35 }
36
37protected:
38 Expr visit(const Variable *op) override {
39 map<string, string>::iterator iter = rewrites.find(op->name);
40 if (iter != rewrites.end()) {
41 return Variable::make(op->type, iter->second);
42 } else {
43 return op;
44 }
45 }
46
47 // Can't unify lets where the RHS might be not be pure
48 bool is_impure;
49 Expr visit(const Call *op) override {
50 is_impure |= !op->is_pure();
51 return IRMutator::visit(op);
52 }
53
54 Expr visit(const Load *op) override {
55 is_impure = true;
56 return IRMutator::visit(op);
57 }
58
59 Stmt visit(const ProducerConsumer *op) override {
60 if (op->is_producer) {
61 string old_producing = producing;
62 producing = op->name;
63 Stmt stmt = IRMutator::visit(op);
64 producing = old_producing;
65 return stmt;
66 } else {
67 return IRMutator::visit(op);
68 }
69 }
70
71 template<typename LetStmtOrLet>
72 auto visit_let(const LetStmtOrLet *op) -> decltype(op->body) {
73 is_impure = false;
74 Expr value = mutate(op->value);
75 auto body = op->body;
76
77 bool should_pop = false;
78 bool should_erase = false;
79
80 if (!is_impure) {
81 map<Expr, string, IRDeepCompare>::iterator iter = scope.find(value);
82 if (iter == scope.end()) {
83 scope[value] = op->name;
84 should_pop = true;
85 } else {
86 value = Variable::make(value.type(), iter->second);
87 rewrites[op->name] = iter->second;
88 should_erase = true;
89 }
90 }
91
92 body = mutate(op->body);
93
94 if (should_pop) {
95 scope.erase(value);
96 }
97 if (should_erase) {
98 rewrites.erase(op->name);
99 }
100
101 if (value.same_as(op->value) && body.same_as(op->body)) {
102 return op;
103 } else {
104 return LetStmtOrLet::make(op->name, value, body);
105 }
106 }
107
108 Expr visit(const Let *op) override {
109 return visit_let(op);
110 }
111
112 Stmt visit(const LetStmt *op) override {
113 return visit_let(op);
114 }
115};

119Stmt unify_duplicate_lets(const Stmt &s) {
120 return UnifyDuplicateLets().mutate(s);
121}
