7Func repeat_edge(const Func &source,
8 const Region &bounds) {
9 std::vector<Var> args(source.args());
10 user_assert(args.size() >= bounds.size())
11 << "repeat_edge called with more bounds (" << bounds.size()
12 << ") than dimensions (" << args.size()
13 << ") Func " << source.name() << " has.\n";
14
15 std::vector<Expr> actuals;
16 for (size_t i = 0; i < bounds.size(); i++) {
17 Var arg_var = args[i];
18 Expr min = bounds[i].min;
19 Expr extent = bounds[i].extent;
20
21 if (min.defined() && extent.defined()) {
22 actuals.push_back(clamp(likely(arg_var), min, min + extent - 1));
23 } else if (!min.defined() && !extent.defined()) {
24 actuals.push_back(arg_var);
25 } else {
26 user_error << "Partially undefined bounds for dimension " << arg_var
27 << " of Func " << source.name() << "\n";
28 }
29 }
30
31 // If there were fewer bounds than dimensions, regard the ones at the end as unbounded.
32 actuals.insert(actuals.end(), args.begin() + actuals.size(), args.end());
33
34 Func bounded("repeat_edge");
35 bounded(args) = source(actuals);
36
37 return bounded;
38}

40Func constant_exterior(const Func &source, const Tuple &value,
41 const Region &bounds) {
42 std::vector<Var> source_args = source.args();
43 std::vector<Var> args(source_args);
44 user_assert(args.size() >= bounds.size())
45 << "constant_exterior called with more bounds (" << bounds.size()
46 << ") than dimensions (" << source_args.size()
47 << ") Func " << source.name() << " has.\n";
48
49 Expr out_of_bounds = cast<bool>(false);
50 for (size_t i = 0; i < bounds.size(); i++) {
51 Var arg_var = source_args[i];
52 Expr min = bounds[i].min;
53 Expr extent = bounds[i].extent;
54
55 if (min.defined() && extent.defined()) {
56 out_of_bounds = (out_of_bounds ||
57 arg_var < min ||
58 arg_var >= min + extent);
59 } else if (min.defined() || extent.defined()) {
60 user_error << "Partially undefined bounds for dimension " << arg_var
61 << " of Func " << source.name() << "\n";
62 }
63 }
64
65 Func bounded("constant_exterior");
66 if (value.as_vector().size() > 1) {
67 std::vector<Expr> def;
68 for (size_t i = 0; i < value.as_vector().size(); i++) {
69 def.push_back(select(out_of_bounds, value[i], repeat_edge(source, bounds)(args)[i]));
70 }
71 bounded(args) = Tuple(def);
72 } else {
73 bounded(args) = select(out_of_bounds, value[0], repeat_edge(source, bounds)(args));
74 }
75
76 return bounded;
77}

84Func repeat_image(const Func &source,
85 const Region &bounds) {
86 std::vector<Var> args(source.args());
87 user_assert(args.size() >= bounds.size())
88 << "repeat_image called with more bounds (" << bounds.size()
89 << ") than dimensions (" << args.size()
90 << ") Func " << source.name() << " has.\n";
91
92 std::vector<Expr> actuals;
93 for (size_t i = 0; i < bounds.size(); i++) {
94 Var arg_var = args[i];
95 Expr min = bounds[i].min;
96 Expr extent = bounds[i].extent;
97
98 if (min.defined() && extent.defined()) {
99 Expr coord = arg_var - min; // Enforce zero origin.
100 coord = coord % extent; // Range is 0 to w-1
101 coord = coord + min; // Restore correct min
102
103 coord = select(arg_var < min || arg_var >= min + extent, coord,
104 clamp(likely(arg_var), min, min + extent - 1));
105
106 actuals.push_back(coord);
107 } else if (!min.defined() && !extent.defined()) {
108 actuals.push_back(arg_var);
109 } else {
110 user_error << "Partially undefined bounds for dimension " << arg_var
111 << " of Func " << source.name() << "\n";
112 }
113 }
114
115 // If there were fewer bounds than dimensions, regard the ones at the end as unbounded.
116 actuals.insert(actuals.end(), args.begin() + actuals.size(), args.end());

124Func mirror_image(const Func &source,
125 const Region &bounds) {
126 std::vector<Var> args(source.args());
127 user_assert(args.size() >= bounds.size())
128 << "mirror_image called with more bounds (" << bounds.size()
129 << ") than dimensions (" << args.size() << ") Func "
130 << source.name() << " has.\n";
131
132 std::vector<Expr> actuals;
133 for (size_t i = 0; i < bounds.size(); i++) {
134 Var arg_var = args[i];
135
136 Expr min = bounds[i].min;
137 Expr extent = bounds[i].extent;
138
139 if (min.defined() && extent.defined()) {
140 Expr coord = arg_var - min; // Enforce zero origin.
141 coord = coord % (2 * extent); // Range is 0 to 2w-1
142 coord = select(coord >= extent, 2 * extent - 1 - coord, coord); // Range is -w+1, w
143 coord = coord + min; // Restore correct min
144 coord = clamp(coord, min, min + extent - 1);
145 coord = select(arg_var < min || arg_var >= min + extent, coord,
146 clamp(likely(arg_var), min, min + extent - 1));
147 actuals.push_back(coord);
148 } else if (!min.defined() && !extent.defined()) {
149 actuals.push_back(arg_var);
150 } else {
151 user_error << "Partially undefined bounds for dimension " << arg_var
152 << " of Func " << source.name() << "\n";
153 }
154 }
155
156 // If there were fewer bounds than dimensions, regard the ones at the end as unbounded.
157 actuals.insert(actuals.end(), args.begin() + actuals.size(), args.end());
158
159 Func bounded("mirror_image");
160 bounded(args) = source(actuals);

162 return bounded;

165Func mirror_interior(const Func &source,
166 const Region &bounds) {
167 std::vector<Var> args(source.args());
168 user_assert(args.size() >= bounds.size())
169 << "mirror_interior called with more bounds (" << bounds.size()
170 << ") than dimensions (" << args.size()
171 << ") Func " << source.name() << " has.\n";
172
173 std::vector<Expr> actuals;
174 for (size_t i = 0; i < bounds.size(); i++) {
175 Var arg_var = args[i];
176
177 Expr min = bounds[i].min;
178 Expr extent = bounds[i].extent;
179
180 if (min.defined() && extent.defined()) {
181 Expr limit = extent - 1;
182 Expr coord = arg_var - min; // Enforce zero origin.
183 coord = coord % (2 * limit); // Range is 0 to 2w-1
184 coord = coord - limit; // Range is -w, w
185 coord = abs(coord); // Range is 0, w
186 coord = limit - coord; // Range is 0, w
187 coord = coord + min; // Restore correct min
188
189 // The boundary condition probably doesn't apply
190 coord = select(arg_var < min || arg_var >= min + extent, coord,
191 clamp(likely(arg_var), min, min + extent - 1));
192
193 actuals.push_back(coord);
194 } else if (!min.defined() && !extent.defined()) {
195 actuals.push_back(arg_var);
196 } else {
197 user_error << "Partially undefined bounds for dimension " << arg_var
198 << " of Func " << source.name() << "\n";
199 }
200 }
201
202 // If there were fewer bounds than dimensions, regard the ones at the end as unbounded.
203 actuals.insert(actuals.end(), args.begin() + actuals.size(), args.end());
