27// This is the One True Source of the known output types for halide,
28// and the appropriate file extension for each output type. If you are
29// explicitly managing file extensions somewhere else, you are probably
30// doing it wrong; please prefer to use this table as the source of truth.
31//
32// Note that we deliberately default to ".py.cpp" (rather than .py.c) here for python_extension;
33// in theory, the Python extension file we generate can be compiled just
34// fine as a plain-C file... but if we are building with cpp-name-mangling
35// enabled in the target, we will include generated .h files that can't be compiled.
36// We really don't want to vary the file extensions based on target flags,
37// and in practice, it's extremely unlikely that anyone needs to rely on this
38// being pure C output (vs possibly C++).
39std::map<Output, const OutputInfo> get_output_info(const Target &target) {
40 constexpr bool IsMulti = true;
41 constexpr bool IsSingle = false;
42 const bool is_windows_coff = target.os == Target::Windows;
43 std::map<Output, const OutputInfo> ext = {
44 {Output::assembly, {"assembly", ".s", IsMulti}},
45 {Output::bitcode, {"bitcode", ".bc", IsMulti}},
46 {Output::c_header, {"c_header", ".h", IsSingle}},
47 {Output::c_source, {"c_source", ".halide_generated.cpp", IsSingle}},
48 {Output::compiler_log, {"compiler_log", ".halide_compiler_log", IsSingle}},
49 {Output::cpp_stub, {"cpp_stub", ".stub.h", IsSingle}},
50 {Output::featurization, {"featurization", ".featurization", IsMulti}},
51 {Output::llvm_assembly, {"llvm_assembly", ".ll", IsMulti}},
52 {Output::object, {"object", is_windows_coff ? ".obj" : ".o", IsMulti}},
53 {Output::python_extension, {"python_extension", ".py.cpp", IsSingle}},
54 {Output::pytorch_wrapper, {"pytorch_wrapper", ".pytorch.h", IsSingle}},
55 {Output::registration, {"registration", ".registration.cpp", IsSingle}},
56 {Output::schedule, {"schedule", ".schedule.h", IsSingle}},
57 {Output::static_library, {"static_library", is_windows_coff ? ".lib" : ".a", IsSingle}},
58 {Output::stmt, {"stmt", ".stmt", IsMulti}},
59 {Output::stmt_html, {"stmt_html", ".stmt.html", IsMulti}},
60 };
61 return ext;
62}

79 std::string add_temp_file(const std::string &base_path_name,
80 const std::string &suffix,
81 const Target &target,
82 bool in_front = false) {
83 size_t slash_idx = base_path_name.rfind('/');
84 size_t backslash_idx = base_path_name.rfind('\\');
85 if (slash_idx == std::string::npos) {
86 slash_idx = 0;
87 } else {
88 slash_idx++;
89 }
90 if (backslash_idx == std::string::npos) {
91 backslash_idx = 0;
92 } else {
93 backslash_idx++;
94 }
95 std::string base_name = base_path_name.substr(std::max(slash_idx, backslash_idx));
96 std::string name = dir_path + "/" + base_name + suffix;
97 debug(1) << "add_temp_object_file: " << name << "\n";
98 if (in_front) {
99 dir_files.insert(dir_files.begin(), name);
100 } else {
101 dir_files.push_back(name);
102 }
103 return name;
104 }
105
106 std::string add_temp_object_file(const std::string &base_path_name,
107 const std::string &suffix,
108 const Target &target,
109 bool in_front = false) {
110 const char *ext = (target.os == Target::Windows) ? ".obj" : ".o";
111 return add_temp_file(base_path_name, suffix + ext, target, in_front);
112 }

114 const std::vector<std::string> &files() {

129// Given a pathname of the form /path/to/name.ext, append suffix before ext to produce /path/to/namesuffix.ext
130std::string add_suffix(const std::string &path, const std::string &suffix) {
131 size_t last_path = std::min(path.rfind('/'), path.rfind('\\'));
132 if (last_path == std::string::npos) {
133 last_path = 0;
134 }
135 size_t dot = path.find('.', last_path);
136 if (dot == std::string::npos) {
137 return path + suffix;
138 } else {
139 return path.substr(0, dot) + suffix + path.substr(dot);
140 }
141}

155void emit_registration(const Module &m, std::ostream &stream) {
156 /*
157 This relies on the filter library being linked in a way that doesn't
158 dead-strip "unused" initialization code; this may mean that you need to
159 explicitly link with with --whole-archive (or the equivalent) to ensure
160 that the registration code isn't omitted. Sadly, there's no portable way
161 to do this, so you may need to take care in your make/build/etc files:
162
163 Linux: -Wl,--whole-archive "/path/to/library" -Wl,-no-whole-archive
164 Darwin/OSX: -Wl,-force_load,/path/to/library
165 VS2015 R2+: /WHOLEARCHIVE:/path/to/library.lib
166 Bazel: alwayslink=1
167
168 Note also that registration files deliberately have no #includes, and
169 are specifically designed to be legal to concatenate into a single
170 source file; it should be equivalent to compile-and-link multiple
171 registration files separately, or to concatenate multiple registration
172 files into a single one which is then compiled.
173 */
174
175 const std::string registration_template = R"INLINE_CODE(
176// MACHINE GENERATED -- DO NOT EDIT
177
178extern "C" {
179struct halide_filter_metadata_t;
180void halide_register_argv_and_metadata(
181 int (*filter_argv_call)(void **),
182 const struct halide_filter_metadata_t *filter_metadata,
183 const char * const *extra_key_value_pairs
184);
185}
186
187$NAMESPACEOPEN$
188extern int $SHORTNAME$_argv(void **args);
189extern const struct halide_filter_metadata_t *$SHORTNAME$_metadata();
190$NAMESPACECLOSE$
191
192#ifdef HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC
193extern "C" const char * const *HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC();
194#endif // HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC
195
196namespace $NREGS$ {
197namespace {
198struct Registerer {
199 Registerer() {
200#ifdef HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC
201 halide_register_argv_and_metadata(::$FULLNAME$_argv, ::$FULLNAME$_metadata(), HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC());
202#else
203 halide_register_argv_and_metadata(::$FULLNAME$_argv, ::$FULLNAME$_metadata(), nullptr);
204#endif // HALIDE_REGISTER_EXTRA_KEY_VALUE_PAIRS_FUNC
205 }
206};
207static Registerer registerer;
208} // namespace
209} // $NREGS$
210
211)INLINE_CODE";
212
213 for (const auto &f : m.functions()) {
214 if (f.linkage == LinkageType::ExternalPlusMetadata) {
215 std::vector<std::string> namespaces;
216 std::string simple_name = extract_namespaces(f.name, namespaces);
217 std::string nsopen, nsclose;
218 for (const auto &ns : namespaces) {
219 nsopen += "namespace " + ns + " { ";
220 nsclose += "}";
221 }
222 if (!m.target().has_feature(Target::CPlusPlusMangling)) {
223 internal_assert(namespaces.empty());
224 nsopen = "extern \"C\" {";
225 nsclose = "}";
226 }
227 std::string nsreg = "halide_nsreg_" + replace_all(f.name, "::", "_");
228 std::string s = replace_all(registration_template, "$NAMESPACEOPEN$", nsopen);
229 s = replace_all(s, "$SHORTNAME$", simple_name);
230 s = replace_all(s, "$NAMESPACECLOSE$", nsclose);
231 s = replace_all(s, "$FULLNAME$", f.name);
232 s = replace_all(s, "$NREGS$", nsreg);
233 stream << s;
234 }
235 }
236}

238std::string indent_string(const std::string &src, const std::string &indent) {
239 std::ostringstream o;
240 bool prev_was_newline = true;
241 for (size_t i = 0; i < src.size(); i++) {
242 const char c = src[i];
243 const bool is_newline = (c == '\n');
244 if (prev_was_newline && !is_newline) {
245 o << indent;
246 }
247 o << c;
248 prev_was_newline = is_newline;
249 }
250 return o.str();
251}

253void emit_schedule_file(const std::string &name,
254 const std::vector<Target> &targets,
255 const std::string &scheduler_name,
256 const std::string &machine_params_string,
257 const std::string &body,
258 std::ostream &stream) {
259 std::string s = R"INLINE_CODE(#ifndef $CLEANNAME$_SCHEDULE_H
260#define $CLEANNAME$_SCHEDULE_H
261
262// MACHINE GENERATED -- DO NOT EDIT
263// This schedule was automatically generated by $SCHEDULER$
264// for target=$TARGET$ // NOLINT
265// with machine_params=$MACHINEPARAMS$
266
267#include "Halide.h"
268
269$NAMESPACEOPEN$
270inline void apply_schedule_$SHORTNAME$(
271 ::Halide::Pipeline pipeline,
272 ::Halide::Target target
273) {
274 using ::Halide::Func;
275 using ::Halide::MemoryType;
276 using ::Halide::RVar;
277 using ::Halide::TailStrategy;
278 using ::Halide::Var;
279$BODY$
280}
281$NAMESPACECLOSE$
282#endif // $CLEANNAME$_SCHEDULE_H
283)INLINE_CODE";
284
285 // For logging in the comment, strip out features that are almost
286 // certainly irrelevant to scheduling issues, to make for easier reading
287 const Target::Feature irrelevant_features[] = {
288 Target::CPlusPlusMangling,
289 Target::NoRuntime,
290 Target::UserContext,
291 };
292
293 std::vector<std::string> namespaces;
294 std::string simple_name = extract_namespaces(name, namespaces);
295 std::string nsopen, nsclose;
296 for (const auto &ns : namespaces) {
297 nsopen += "namespace " + ns + " {\n";
298 nsclose += "} // namespace " + ns + "\n";
299 }
300 std::string clean_name = replace_all(name, "::", "_");
301 std::string target_string;
302 for (Target t : targets) {
303 if (!target_string.empty()) {
304 target_string += ",";
305 }
306 for (auto f : irrelevant_features) {
307 t = t.without_feature(f);
308 }
309 target_string += t.to_string();
310 }
311 std::string body_text = indent_string(body, " ");
312 s = replace_all(s, "$SCHEDULER$", scheduler_name);
313 s = replace_all(s, "$NAMESPACEOPEN$", nsopen);
314 s = replace_all(s, "$SHORTNAME$", simple_name);
315 s = replace_all(s, "$CLEANNAME$", clean_name);
316 s = replace_all(s, "$NAMESPACECLOSE$", nsclose);
317 s = replace_all(s, "$TARGET$", target_string);
318 s = replace_all(s, "$BODY$", body_text);
319 s = replace_all(s, "$MACHINEPARAMS$", machine_params_string);
320 stream << s;
321}

448Module link_modules(const std::string &name, const std::vector<Module> &modules) {
449 Module output(name, modules.front().target());
450
451 for (size_t i = 0; i < modules.size(); i++) {
452 const Module &input = modules[i];
453
454 if (output.target() != input.target()) {
455 user_error << "Mismatched targets in modules to link ("
456 << output.name() << ", " << output.target().to_string()
457 << "), ("
458 << input.name() << ", " << input.target().to_string() << ")\n";
459 }
460
461 // TODO(dsharlet): Check for naming collisions, maybe rename
462 // internal linkage declarations in the case of collision.
463 for (const auto &b : input.buffers()) {
464 output.append(b);
465 }
466 for (const auto &f : input.functions()) {
467 output.append(f);
468 }
469 }
470
471 return output;
472}

474Buffer<uint8_t> Module::compile_to_buffer() const {
475 // TODO: This Hexagon specific code should be removed as soon as possible.
476 // This may involve adding more general support for post-processing and
477 // a way of specifying to use it.
478 if (target().arch == Target::Hexagon) {
479 return compile_module_to_hexagon_shared_object(*this);
480 }
481
482 llvm::LLVMContext context;
483 std::unique_ptr<llvm::Module> llvm_module(compile_module_to_llvm_module(*this, context));
484
485 llvm::SmallVector<char, 4096> object;
486 llvm::raw_svector_ostream object_stream(object);
487 compile_llvm_module_to_object(*llvm_module, object_stream);
488
489 if (debug::debug_level() >= 2) {
490 debug(2) << "Submodule assembly for " << name() << ": "
491 << "\n";
492 llvm::SmallString<4096> assembly;
493 llvm::raw_svector_ostream assembly_stream(assembly);
494 compile_llvm_module_to_assembly(*llvm_module, assembly_stream);
495 debug(2) << assembly.c_str() << "\n";
496 }
497
498 Buffer<uint8_t> result(object.size(), name());
499 memcpy(result.data(), reinterpret_cast<uint8_t *>(&object[0]), object.size());
500 return result;
501}

503Module Module::resolve_submodules() const {
504 if (submodules().empty()) {
505 return *this;
506 }
507
508 Module lowered_module(name(), target());
509
510 for (const auto &f : functions()) {
511 lowered_module.append(f);
512 }
513 for (const auto &buf : buffers()) {
514 lowered_module.append(buf);
515 }
516 for (const auto &ec : external_code()) {
517 lowered_module.append(ec);
518 }
519 for (const auto &m : submodules()) {
520 Module copy(m.resolve_submodules());
521
522 // Propagate external code blocks.
523 for (const auto &ec : external_code()) {
524 // TODO(zalman): Is this the right thing to do?
525 bool already_in_list = false;
526 for (const auto &ec_sub : copy.external_code()) {
527 if (ec_sub.name() == ec.name()) {
528 already_in_list = true;
529 break;
530 }
531 }
532 if (!already_in_list) {
533 copy.append(ec);
534 }
535 }
536
537 auto buf = copy.compile_to_buffer();
538 lowered_module.append(buf);
539 }
540 // Copy the autoscheduler results back into the lowered module after resolving the submodules.
541 if (auto *r = contents->auto_scheduler_results.get()) {
542 lowered_module.set_auto_scheduler_results(*r);
543 }
544 return lowered_module;
545}

557void Module::compile(const std::map<Output, std::string> &output_files) const {
558 validate_outputs(output_files);
559
560 // output stmt and html prior to resolving submodules. We need to
561 // clear the output after writing it, otherwise the output will
562 // be overwritten by recursive calls after submodules are resolved.
563 if (contains(output_files, Output::stmt)) {
564 debug(1) << "Module.compile(): stmt " << output_files.at(Output::stmt) << "\n";
565 std::ofstream file(output_files.at(Output::stmt));
566 file << *this;
567 }
568 if (contains(output_files, Output::stmt_html)) {
569 debug(1) << "Module.compile(): stmt_html " << output_files.at(Output::stmt_html) << "\n";
570 Internal::print_to_html(output_files.at(Output::stmt_html), *this);
571 }
572
573 // If there are submodules, recursively lower submodules to
574 // buffers on a copy of the module being compiled, then compile
575 // the copied module.
576 if (!submodules().empty()) {
577 std::map<Output, std::string> output_files_copy = output_files;
578 output_files_copy.erase(Output::stmt);
579 output_files_copy.erase(Output::stmt_html);
580 resolve_submodules().compile(output_files_copy);
581 return;
582 }
583
584 auto *logger = get_compiler_logger();
585 if (contains(output_files, Output::object) || contains(output_files, Output::assembly) ||
586 contains(output_files, Output::bitcode) || contains(output_files, Output::llvm_assembly) ||
587 contains(output_files, Output::static_library)) {
588 llvm::LLVMContext context;
589 std::unique_ptr<llvm::Module> llvm_module(compile_module_to_llvm_module(*this, context));
590
591 if (contains(output_files, Output::object)) {
592 const auto &f = output_files.at(Output::object);
593 debug(1) << "Module.compile(): object " << f << "\n";
594 auto out = make_raw_fd_ostream(f);
595 compile_llvm_module_to_object(*llvm_module, *out);
596 if (logger) {
597 out->flush();
598 logger->record_object_code_size(file_stat(f).file_size);
599 }
600 }
601 if (contains(output_files, Output::static_library)) {
602 // To simplify the code, we always create a temporary object output
603 // here, even if output_files.at(Output::object) was also set: in practice,
604 // no real-world code ever sets both object and static_library
605 // at the same time, so there is no meaningful performance advantage
606 // to be had.
607 TemporaryObjectFileDir temp_dir;
608 {
609 std::string object = temp_dir.add_temp_object_file(output_files.at(Output::static_library), "", target());
610 debug(1) << "Module.compile(): temporary object " << object << "\n";
611 auto out = make_raw_fd_ostream(object);
612 compile_llvm_module_to_object(*llvm_module, *out);
613 out->flush(); // create_static_library() is happier if we do this
614 if (logger && !contains(output_files, Output::object)) {
615 // Don't double-record object-code size if we already recorded it for object
616 logger->record_object_code_size(file_stat(object).file_size);
617 }
618 }
619 debug(1) << "Module.compile(): static_library " << output_files.at(Output::static_library) << "\n";
620 Target base_target(target().os, target().arch, target().bits);
621 create_static_library(temp_dir.files(), base_target, output_files.at(Output::static_library));
622 }
623 if (contains(output_files, Output::assembly)) {
624 debug(1) << "Module.compile(): assembly " << output_files.at(Output::assembly) << "\n";
625 auto out = make_raw_fd_ostream(output_files.at(Output::assembly));
626 compile_llvm_module_to_assembly(*llvm_module, *out);
627 }
628 if (contains(output_files, Output::bitcode)) {
629 debug(1) << "Module.compile(): bitcode " << output_files.at(Output::bitcode) << "\n";
630 auto out = make_raw_fd_ostream(output_files.at(Output::bitcode));
631 compile_llvm_module_to_llvm_bitcode(*llvm_module, *out);
632 }
633 if (contains(output_files, Output::llvm_assembly)) {
634 debug(1) << "Module.compile(): llvm_assembly " << output_files.at(Output::llvm_assembly) << "\n";
635 auto out = make_raw_fd_ostream(output_files.at(Output::llvm_assembly));
636 compile_llvm_module_to_llvm_assembly(*llvm_module, *out);
637 }
638 }
639 if (contains(output_files, Output::c_header)) {
640 debug(1) << "Module.compile(): c_header " << output_files.at(Output::c_header) << "\n";
641 std::ofstream file(output_files.at(Output::c_header));
642 Internal::CodeGen_C cg(file,
643 target(),
644 target().has_feature(Target::CPlusPlusMangling) ? Internal::CodeGen_C::CPlusPlusHeader : Internal::CodeGen_C::CHeader,
645 output_files.at(Output::c_header));
646 cg.compile(*this);
647 }
648 if (contains(output_files, Output::c_source)) {
649 debug(1) << "Module.compile(): c_source " << output_files.at(Output::c_source) << "\n";
650 std::ofstream file(output_files.at(Output::c_source));
651 Internal::CodeGen_C cg(file,
652 target(),
653 target().has_feature(Target::CPlusPlusMangling) ? Internal::CodeGen_C::CPlusPlusImplementation : Internal::CodeGen_C::CImplementation);
654 cg.compile(*this);
655 }
656 if (contains(output_files, Output::python_extension)) {
657 debug(1) << "Module.compile(): python_extension " << output_files.at(Output::python_extension) << "\n";
658 std::ofstream file(output_files.at(Output::python_extension));
659 Internal::PythonExtensionGen python_extension_gen(file);
660 python_extension_gen.compile(*this);
661 }
662 if (contains(output_files, Output::schedule)) {
663 debug(1) << "Module.compile(): schedule " << output_files.at(Output::schedule) << "\n";
664 std::ofstream file(output_files.at(Output::schedule));
665 auto *r = contents->auto_scheduler_results.get();
666 std::string scheduler = r ? r->scheduler_name : "(None)";
667 std::string machine_params = r ? r->machine_params_string : "(None)";
668 std::string body = r && !r->schedule_source.empty() ? r->schedule_source : "// No autoscheduler has been run for this Generator.\n";
669 emit_schedule_file(name(), {target()}, scheduler, machine_params, body, file);
670 }
671 if (contains(output_files, Output::featurization)) {
672 debug(1) << "Module.compile(): featurization " << output_files.at(Output::featurization) << "\n";
673 // If the featurization data is empty, just write an empty file
674 std::ofstream binfile(output_files.at(Output::featurization), std::ios::binary | std::ios_base::trunc);
675 auto *r = contents->auto_scheduler_results.get();
676 if (r) {
677 binfile.write((const char *)r->featurization.data(), r->featurization.size());
678 }
679 binfile.close();
680 }
681 if (contains(output_files, Output::registration)) {
682 debug(1) << "Module.compile(): registration " << output_files.at(Output::registration) << "\n";
683 std::ofstream file(output_files.at(Output::registration));
684 emit_registration(*this, file);
685 file.close();
686 internal_assert(!file.fail());
687 }
688 if (contains(output_files, Output::pytorch_wrapper)) {
689 debug(1) << "Module.compile(): pytorch_wrapper " << output_files.at(Output::pytorch_wrapper) << "\n";
690
691 std::ofstream file(output_files.at(Output::pytorch_wrapper));
692 Internal::CodeGen_PyTorch cg(file);
693 cg.compile(*this);
694 file.close();
695 internal_assert(!file.fail());
696 }
697 if (contains(output_files, Output::compiler_log)) {
698 debug(1) << "Module.compile(): compiler_log " << output_files.at(Output::compiler_log) << "\n";
699 std::ofstream file(output_files.at(Output::compiler_log));
700 internal_assert(get_compiler_logger() != nullptr);
701 get_compiler_logger()->emit_to_stream(file);
702 file.close();
703 internal_assert(!file.fail());
704 }
705 // If HL_DEBUG_COMPILER_LOGGER is set, dump the log (if any) to stderr now, whether or it is required
706 if (get_env_variable("HL_DEBUG_COMPILER_LOGGER") == "1" && get_compiler_logger() != nullptr) {
707 get_compiler_logger()->emit_to_stream(std::cerr);
708 }
709}

711std::map<Output, std::string> compile_standalone_runtime(const std::map<Output, std::string> &output_files, const Target &t) {
712 validate_outputs(output_files);
713
714 Module empty("standalone_runtime", t.without_feature(Target::NoRuntime).without_feature(Target::JIT));
715 // For runtime, it only makes sense to output object files or static_library, so ignore
716 // everything else.
717 std::map<Output, std::string> actual_outputs;
718 for (auto key : {Output::object, Output::static_library}) {
719 auto it = output_files.find(key);
720 if (it != output_files.end()) {
721 actual_outputs[key] = it->second;
722 }
723 }
724 empty.compile(actual_outputs);
725 return actual_outputs;
726}

752void compile_multitarget(const std::string &fn_name,
753 const std::map<Output, std::string> &output_files,
754 const std::vector<Target> &targets,
755 const std::vector<std::string> &suffixes,
756 const ModuleFactory &module_factory,
757 const CompilerLoggerFactory &compiler_logger_factory) {
758 validate_outputs(output_files);
759
760 user_assert(!fn_name.empty()) << "Function name must be specified.\n";
761 user_assert(!targets.empty()) << "Must specify at least one target.\n";
762 user_assert(suffixes.empty() || suffixes.size() == targets.size())
763 << "The suffixes list must be empty or the same length as the targets list.\n";
764
765 // The final target in the list is considered "baseline", and is used
766 // for (e.g.) the runtime and shared code. It is often just arch-bits-os
767 // with no other features (though this is *not* a requirement).
768 const Target &base_target = targets.back();
769
770 // JIT makes no sense.
771 user_assert(!base_target.has_feature(Target::JIT)) << "JIT not allowed for compile_multitarget.\n";
772
773 const auto suffix_for_entry = [&](int i) -> std::string {
774 return "-" + (suffixes.empty() ? targets[i].to_string() : suffixes[i]);
775 };
776
777 const auto add_suffixes = [&](const std::map<Output, std::string> &in, const std::string &suffix) -> std::map<Output, std::string> {
778 // is_multi doesn't vary by Target, so we can pass an empty target here safely
779 auto output_info = get_output_info(Target());
780 std::map<Output, std::string> out = in;
781 for (auto &it : out) {
782 if (output_info[it.first].is_multi) {
783 out[it.first] = add_suffix(it.second, suffix);
784 }
785 }
786 return out;
787 };
788
789 // If only one target, don't bother with the runtime feature detection wrapping.
790 const bool needs_wrapper = (targets.size() > 1);
791 if (targets.size() == 1) {
792 debug(1) << "compile_multitarget: single target is " << base_target.to_string() << "\n";
793 ScopedCompilerLogger activate(compiler_logger_factory, fn_name, base_target);
794
795 // If we want to have single-output object files use the target suffix, we'd
796 // want to do this instead:
797 //
798 // auto sub_out = add_suffixes(output_files, suffix_for_entry(0));
799 // module_factory(fn_name, base_target).compile(sub_out);
800 //
801 // This would make the filename outputs more symmetrical (ie the same for n=1 as for n>1)
802 // but at the expense of breaking existing users. So for now, we're going to continue
803 // with the legacy treatment below:
804 module_factory(fn_name, base_target).compile(output_files);
805 return;
806 }
807
808 user_assert(((int)contains(output_files, Output::object) + (int)contains(output_files, Output::static_library)) == 1)
809 << "compile_multitarget() expects exactly one of 'object' and 'static_library' to be specified when multiple targets are specified.\n";
810
811 // For safety, the runtime must be built only with features common to all
812 // of the targets; given an unusual ordering like
813 //
814 // x86-64-linux,x86-64-sse41
815 //
816 // we should still always be *correct*: this ordering would never select sse41
817 // (since x86-64-linux would be selected first due to ordering), but could
818 // crash on non-sse41 machines (if we generated a runtime with sse41 instructions
819 // included). So we'll keep track of the common features as we walk thru the targets.
820
821 // Using something like std::bitset would be arguably cleaner here, but we need an
822 // array-of-uint64 for calls to halide_can_use_target_features() anyway,
823 // so we'll just build and maintain in that form to avoid extra conversion.
824 constexpr int kFeaturesWordCount = (Target::FeatureEnd + 63) / (sizeof(uint64_t) * 8);
825 uint64_t runtime_features[kFeaturesWordCount] = {(uint64_t)-1LL};
826
827 TemporaryObjectFileDir temp_obj_dir, temp_compiler_log_dir;
828 std::vector<Expr> wrapper_args;
829 std::vector<LoweredArgument> base_target_args;
830 std::vector<AutoSchedulerResults> auto_scheduler_results;
831
832 for (size_t i = 0; i < targets.size(); ++i) {
833 const Target &target = targets[i];
834
835 // arch-bits-os must be identical across all targets.
836 if (target.os != base_target.os ||
837 target.arch != base_target.arch ||
838 target.bits != base_target.bits) {
839 user_error << "All Targets must have matching arch-bits-os for compile_multitarget.\n";
840 }
841 // Some features must match across all targets.
842 static const std::array<Target::Feature, 9> must_match_features = {{
843 Target::ASAN,
844 Target::CPlusPlusMangling,
845 Target::Debug,
846 Target::JIT,
847 Target::Matlab,
848 Target::MSAN,
849 Target::NoRuntime,
850 Target::TSAN,
851 Target::UserContext,
852 }};
853 for (auto f : must_match_features) {
854 if (target.has_feature(f) != base_target.has_feature(f)) {
855 user_error << "All Targets must have feature '" << Target::feature_to_name(f) << "'' set identically for compile_multitarget.\n";
856 break;
857 }
858 }
859
860 // Each sub-target has a function name that is the 'real' name plus a suffix
861 std::string suffix = suffix_for_entry(i);
862 std::string sub_fn_name = needs_wrapper ? (fn_name + suffix) : fn_name;
863
864 // We always produce the runtime separately, so add NoRuntime explicitly.
865 // Matlab should be added to the wrapper pipeline below, instead of each sub-pipeline.
866 Target sub_fn_target = target.with_feature(Target::NoRuntime);
867 if (needs_wrapper) {
868 sub_fn_target = sub_fn_target.without_feature(Target::Matlab);
869 }
870
871 {
872 ScopedCompilerLogger activate(compiler_logger_factory, sub_fn_name, sub_fn_target);
873 Module sub_module = module_factory(sub_fn_name, sub_fn_target);
874 // Re-assign every time -- should be the same across all targets anyway,
875 // but base_target is always the last one we encounter.
876 base_target_args = sub_module.get_function_by_name(sub_fn_name).args;
877
878 auto sub_out = add_suffixes(output_files, suffix);
879 if (contains(output_files, Output::static_library)) {
880 sub_out[Output::object] = temp_obj_dir.add_temp_object_file(output_files.at(Output::static_library), suffix, target);
881 sub_out.erase(Output::static_library);
882 }
883 sub_out.erase(Output::registration);
884 sub_out.erase(Output::schedule);
885 sub_out.erase(Output::c_header);
886 if (contains(sub_out, Output::compiler_log)) {
887 sub_out[Output::compiler_log] = temp_compiler_log_dir.add_temp_file(output_files.at(Output::compiler_log), suffix, target);
888 }
889 debug(1) << "compile_multitarget: compile_sub_target " << sub_out[Output::object] << "\n";
890 sub_module.compile(sub_out);
891 const auto *r = sub_module.get_auto_scheduler_results();
892 auto_scheduler_results.push_back(r ? *r : AutoSchedulerResults());
893 }
894
895 uint64_t cur_target_features[kFeaturesWordCount] = {0};
896 for (int i = 0; i < Target::FeatureEnd; ++i) {
897 if (target.has_feature((Target::Feature)i)) {
898 cur_target_features[i >> 6] |= ((uint64_t)1) << (i & 63);
899 }
900 }
901
902 Expr can_use;
903 if (target != base_target) {
904 std::vector<Expr> features_struct_args;
905 for (int i = 0; i < kFeaturesWordCount; ++i) {
906 features_struct_args.emplace_back(UIntImm::make(UInt(64), cur_target_features[i]));
907 }
908 can_use = Call::make(Int(32), "halide_can_use_target_features",
909 {kFeaturesWordCount, Call::make(type_of<uint64_t *>(), Call::make_struct, features_struct_args, Call::Intrinsic)},
910 Call::Extern);
911 } else {
912 can_use = IntImm::make(Int(32), 1);
913 }
914
915 for (int i = 0; i < kFeaturesWordCount; ++i) {
916 runtime_features[i] &= cur_target_features[i];
917 }
918
919 wrapper_args.push_back(can_use != 0);
920 wrapper_args.emplace_back(sub_fn_name);
921 }
922
923 // If we haven't specified "no runtime", build a runtime with the base target
924 // and add that to the result.
925 if (!base_target.has_feature(Target::NoRuntime)) {
926 // Start with a bare Target, set only the features we know are common to all.
927 Target runtime_target(base_target.os, base_target.arch, base_target.bits);
928 for (int i = 0; i < Target::FeatureEnd; ++i) {
929 // We never want NoRuntime set here.
930 if (i == Target::NoRuntime) {
931 continue;
932 }
933 const int word = i >> 6;
934 const int bit = i & 63;
935 if (runtime_features[word] & (((uint64_t)1) << bit)) {
936 runtime_target.set_feature((Target::Feature)i);
937 }
938 }
939 std::string runtime_path = contains(output_files, Output::static_library) ?
940 temp_obj_dir.add_temp_object_file(output_files.at(Output::static_library), "_runtime", runtime_target) :
941 add_suffix(output_files.at(Output::object), "_runtime");
942
943 std::map<Output, std::string> runtime_out =
944 {{Output::object, runtime_path}};
945 debug(1) << "compile_multitarget: compile_standalone_runtime " << runtime_out.at(Output::object) << "\n";
946 compile_standalone_runtime(runtime_out, runtime_target);
947 }
948
949 if (needs_wrapper) {
950 Expr indirect_result = Call::make(Int(32), Call::call_cached_indirect_function, wrapper_args, Call::Intrinsic);
951 std::string private_result_name = unique_name(fn_name + "_result");
952 Expr private_result_var = Variable::make(Int(32), private_result_name);
953 Stmt wrapper_body = AssertStmt::make(private_result_var == 0, private_result_var);
954 wrapper_body = LetStmt::make(private_result_name, indirect_result, wrapper_body);
955
956 // Always build with NoRuntime: that's handled as a separate module.
957 //
958 // Always build with NoBoundsQuery: underlying code will implement that (or not).
959 //
960 // Always build *without* NoAsserts (ie, with Asserts enabled): that's the
961 // only way to propagate a nonzero result code to our caller.
962 Target wrapper_target = base_target
963 .with_feature(Target::NoRuntime)
964 .with_feature(Target::NoBoundsQuery)
965 .without_feature(Target::NoAsserts);
966
967 // If the base target specified the Matlab target, we want the Matlab target
968 // on the wrapper instead.
969 if (base_target.has_feature(Target::Matlab)) {
970 wrapper_target = wrapper_target.with_feature(Target::Matlab);
971 }
972
973 Module wrapper_module(fn_name, wrapper_target);
974 wrapper_module.append(LoweredFunc(fn_name, base_target_args, wrapper_body, LinkageType::ExternalPlusMetadata));
975
976 std::string wrapper_path = contains(output_files, Output::static_library) ?
977 temp_obj_dir.add_temp_object_file(output_files.at(Output::static_library), "_wrapper", base_target, /* in_front*/ true) :
978 add_suffix(output_files.at(Output::object), "_wrapper");
979
980 std::map<Output, std::string> wrapper_out = {{Output::object, wrapper_path}};
981 debug(1) << "compile_multitarget: wrapper " << wrapper_out.at(Output::object) << "\n";
982 wrapper_module.compile(wrapper_out);
983 }
984
985 if (contains(output_files, Output::c_header)) {
986 Module header_module(fn_name, base_target);
987 header_module.append(LoweredFunc(fn_name, base_target_args, {}, LinkageType::ExternalPlusMetadata));
988 std::map<Output, std::string> header_out = {{Output::c_header, output_files.at(Output::c_header)}};
989 debug(1) << "compile_multitarget: c_header " << header_out.at(Output::c_header) << "\n";
990 header_module.compile(header_out);
991 }
992
993 if (contains(output_files, Output::registration)) {
994 debug(1) << "compile_multitarget: registration " << output_files.at(Output::registration) << "\n";
995 Module registration_module(fn_name, base_target);
996 registration_module.append(LoweredFunc(fn_name, base_target_args, {}, LinkageType::ExternalPlusMetadata));
997 std::map<Output, std::string> registration_out = {{Output::registration, output_files.at(Output::registration)}};
998 debug(1) << "compile_multitarget: registration " << registration_out.at(Output::registration) << "\n";
999 registration_module.compile(registration_out);
1000 }
1001
1002 if (contains(output_files, Output::schedule)) {
1003 debug(1) << "compile_multitarget: schedule " << output_files.at(Output::schedule) << "\n";
1004 std::string scheduler = auto_scheduler_results.front().scheduler_name;
1005 if (scheduler.empty()) {
1006 scheduler = "(None)";
1007 }
1008 std::string machine_params = auto_scheduler_results.front().machine_params_string;
1009 if (machine_params.empty()) {
1010 machine_params = "(None)";
1011 }
1012
1013 // Find the features that are unique to each stage (vs the baseline case).
1014 const auto &baseline_target = auto_scheduler_results.back().target;
1015 const auto &baseline_features = baseline_target.get_features_bitset();
1016
1017 // Autoscheduling should be all-or-none across the subtargets;
1018 // if code tries to somehow only autoschedule some subtargets,
1019 // this code may break, and that's ok.
1020 std::ostringstream body;
1021 if (baseline_target.os == Target::OSUnknown && baseline_target.arch == Target::ArchUnknown) {
1022 body << "// No autoscheduler has been run for this Generator.";
1023 } else {
1024 for (size_t i = 0; i < auto_scheduler_results.size(); i++) {
1025 const auto &a = auto_scheduler_results[i];
1026 body << "\n\n";
1027 if (i == auto_scheduler_results.size() - 1) {
1028 body << "// default schedule\n";
1029 body << "{\n";
1030 } else {
1031 auto cur_features = a.target.get_features_bitset() & ~baseline_features;
1032 user_assert(cur_features.count() > 0) << "Multitarget subtargets must be distinct";
1033 std::ostringstream condition;
1034 for (int i = 0; i < Target::FeatureEnd; ++i) {
1035 if (!cur_features[i]) {
1036 continue;
1037 }
1038 if (!condition.str().empty()) {
1039 condition << " &&\n ";
1040 }
1041 condition << "target.has_feature(halide_target_feature_"
1042 << Target::feature_to_name((Target::Feature)i) << ")";
1043 }
1044 body << "if (" << condition.str() << ") {\n";
1045 }
1046 body << indent_string(a.schedule_source, " ");
1047 body << " return;\n";
1048 body << "}";
1049 }
1050 }
1051
1052 std::ofstream file(output_files.at(Output::schedule));
1053 emit_schedule_file(fn_name, targets, scheduler, machine_params, body.str(), file);
1054 }
1055
1056 if (contains(output_files, Output::static_library)) {
1057 debug(1) << "compile_multitarget: static_library " << output_files.at(Output::static_library) << "\n";
1058 create_static_library(temp_obj_dir.files(), base_target, output_files.at(Output::static_library));
1059 }
1060
1061 if (contains(output_files, Output::compiler_log)) {
1062 debug(1) << "compile_multitarget: compiler_log " << output_files.at(Output::compiler_log) << "\n";
1063
1064 std::ofstream compiler_log_file(output_files.at(Output::compiler_log));
1065 compiler_log_file << "[\n";
1066 const auto &f = temp_compiler_log_dir.files();
1067 for (size_t i = 0; i < f.size(); i++) {
1068 auto d = read_entire_file(f[i]);
1069 compiler_log_file.write(d.data(), d.size());
1070 if (i < f.size() - 1) {
1071 compiler_log_file << ",\n";
1072 }
1073 }
1074 compiler_log_file << "]\n";
1075 compiler_log_file.close();
1076 internal_assert(!compiler_log_file.fail());
1077 }
