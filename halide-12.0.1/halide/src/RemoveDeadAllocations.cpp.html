11class RemoveDeadAllocations : public IRMutator {
12 using IRMutator::visit;
13
14 Scope<int> allocs;
15
16 Expr visit(const Call *op) override {
17 if (op->is_extern()) {
18 for (size_t i = 0; i < op->args.size(); i++) {
19 const Variable *var = op->args[i].as<Variable>();
20 if (var && ends_with(var->name, ".buffer")) {
21 std::string func = var->name.substr(0, var->name.find_first_of('.'));
22 if (allocs.contains(func)) {
23 allocs.pop(func);
24 }
25 }
26 }
27 }
28
29 return IRMutator::visit(op);
30 }
31
32 Expr visit(const Load *op) override {
33 if (allocs.contains(op->name)) {
34 allocs.pop(op->name);
35 }
36
37 return IRMutator::visit(op);
38 }
39
40 Stmt visit(const Store *op) override {
41 if (allocs.contains(op->name)) {
42 allocs.pop(op->name);
43 }
44
45 return IRMutator::visit(op);
46 }
47
48 Expr visit(const Variable *op) override {
49 if (allocs.contains(op->name)) {
50 allocs.pop(op->name);
51 }
52 return op;
53 }
54
55 Stmt visit(const Allocate *op) override {
56 allocs.push(op->name, 1);
57 Stmt body = mutate(op->body);
58
59 if (allocs.contains(op->name) && op->free_function.empty()) {
60 allocs.pop(op->name);
61 return body;
62 } else if (body.same_as(op->body)) {
63 return op;
64 } else {
65 return Allocate::make(op->name, op->type, op->memory_type, op->extents,
66 op->condition, body, op->new_expr, op->free_function);
67 }
68 }
69
70 Stmt visit(const Free *op) override {
71 if (allocs.contains(op->name)) {
72 // We have reached a Free Stmt without ever using this buffer, do nothing.
73 return Evaluate::make(0);
74 } else {
75 return op;
76 }
77 }
78
79 Stmt visit(const Atomic *op) override {
80 if (allocs.contains(op->mutex_name)) {
81 allocs.pop(op->mutex_name);
82 }
83
84 return IRMutator::visit(op);
85 }
86};

90Stmt remove_dead_allocations(const Stmt &s) {
91 return RemoveDeadAllocations().mutate(s);
92}
