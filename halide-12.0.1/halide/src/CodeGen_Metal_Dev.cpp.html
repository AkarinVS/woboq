107string CodeGen_Metal_Dev::CodeGen_Metal_C::print_type_maybe_storage(Type type, bool storage, AppendSpaceIfNeeded space) {
108 ostringstream oss;
109
110 // Storage uses packed vector types.
111 if (storage && type.lanes() != 1) {
112 oss << "packed_";
113 }
114 if (type.is_float()) {
115 if (type.bits() == 16) {
116 oss << "half";
117 } else if (type.bits() == 32) {
118 oss << "float";
119 } else if (type.bits() == 64) {
120 oss << "double";
121 } else {
122 user_error << "Can't represent a float with this many bits in Metal C: " << type << "\n";
123 }
124
125 } else {
126 if (type.is_uint() && type.bits() > 1) {
127 oss << "u";
128 }
129 switch (type.bits()) {
130 case 1:
131 oss << "bool";
132 break;
133 case 8:
134 oss << "char";
135 break;
136 case 16:
137 oss << "short";
138 break;
139 case 32:
140 oss << "int";
141 break;
142 case 64:
143 user_error << "Metal does not support 64-bit integers.\n";
144 break;
145 default:
146 user_error << "Can't represent an integer with this many bits in Metal C: " << type << "\n";
147 }
148 }
149 if (type.lanes() != 1) {
150 switch (type.lanes()) {
151 case 2:
152 case 3:
153 case 4:
154 oss << type.lanes();
155 break;
156 default:
157 user_error << "Unsupported vector width in Metal C: " << type << "\n";
158 }
159 }
160 if (space == AppendSpace) {
161 oss << " ";
162 }
163 return oss.str();
164}

185string simt_intrinsic(const string &name) {
186 if (ends_with(name, ".__thread_id_x")) {
187 return "tid_in_tgroup.x";
188 } else if (ends_with(name, ".__thread_id_y")) {
189 return "tid_in_tgroup.y";
190 } else if (ends_with(name, ".__thread_id_z")) {
191 return "tid_in_tgroup.z";
192 } else if (ends_with(name, ".__thread_id_w")) {
193 user_error << "Metal does not support more than three dimensions in a kernel (threads).\n";
194 } else if (ends_with(name, ".__block_id_x")) {
195 return "tgroup_index.x";
196 } else if (ends_with(name, ".__block_id_y")) {
197 return "tgroup_index.y";
198 } else if (ends_with(name, ".__block_id_z")) {
199 return "tgroup_index.z";
200 } else if (ends_with(name, ".__block_id_w")) {
201 user_error << "Metal does not support more than three dimensions in a kernel (groups).\n";
202 }
203 internal_error << "simt_intrinsic called on bad variable name: " << name << "\n";
204 return "";
205}

228 int bits;
229 if (is_const_power_of_two_integer(op->b, &bits)) {
230 ostringstream oss;

298void CodeGen_Metal_Dev::CodeGen_Metal_C::visit(const Call *op) {
299 if (op->is_intrinsic(Call::gpu_thread_barrier)) {
300 internal_assert(op->args.size() == 1) << "gpu_thread_barrier() intrinsic must specify memory fence type.\n";
301
302 const auto *fence_type_ptr = as_const_int(op->args[0]);
303 internal_assert(fence_type_ptr) << "gpu_thread_barrier() parameter is not a constant integer.\n";
304 auto fence_type = *fence_type_ptr;
305
306 // This is quite annoying: even though the MSL docs claim these flags can be combined,
307 // Metal compilers prior to Metal 1.2 give compiler errors. So, we do not combine them,
308 // and rather use a preprocessor definition to do the right thing.
309
310 stream << get_indent() << "threadgroup_barrier(";
311 if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device &&
312 fence_type & CodeGen_GPU_Dev::MemoryFenceType::Shared) {
313 stream << "_halide_mem_fence_device_and_threadgroup";
314 } else if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device) {
315 stream << "mem_flags::mem_device";
316 } else if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Shared) {
317 stream << "mem_flags::mem_threadgroup";
318 } else {
319 stream << "mem_flags::mem_none";
320 }
321 stream << ");\n";
322 print_assignment(op->type, "0");
323 } else {
324 CodeGen_C::visit(op);
325 }

330// If e is a ramp expression with stride 1, return the base, otherwise undefined.
331Expr is_ramp_one(const Expr &e) {
332 const Ramp *r = e.as<Ramp>();
333 if (r == nullptr) {
334 return Expr();
335 }
336
337 if (is_const_one(r->stride)) {
338 return r->base;
339 }
340
341 return Expr();
342}

353void CodeGen_Metal_Dev::CodeGen_Metal_C::visit(const Load *op) {
354 user_assert(is_const_one(op->predicate)) << "Predicated load is not supported inside Metal kernel.\n";
355 user_assert(op->type.lanes() <= 4) << "Vectorization by widths greater than 4 is not supported by Metal -- type is " << op->type << ".\n";
356
357 // If we're loading a contiguous ramp, load from a vector type pointer.
358 Expr ramp_base = is_ramp_one(op->index);
359 if (ramp_base.defined()) {
360 internal_assert(op->type.is_vector());
361 string id_ramp_base = print_expr(ramp_base);
362
363 ostringstream rhs;
364 rhs << "*(" << get_memory_space(op->name) << " " << print_storage_type(op->type) << " *)(("
365 << get_memory_space(op->name) << " " << print_type(op->type.element_of()) << " *)" << print_name(op->name)
366 << " + " << id_ramp_base << ")";
367 print_assignment(op->type, rhs.str());
368
369 return;
370 }
371
372 string id_index = print_expr(op->index);
373
374 // Get the rhs just for the cache.
375 bool type_cast_needed = !(allocations.contains(op->name) &&
376 allocations.get(op->name).type == op->type);
377 ostringstream rhs;
378 if (type_cast_needed) {
379 rhs << "((" << get_memory_space(op->name) << " "
380 << print_storage_type(op->type) << " *)"
381 << print_name(op->name)
382 << ")";
383 } else {
384 rhs << print_name(op->name);
385 }
386 rhs << "[" << id_index << "]";
387
388 std::map<string, string>::iterator cached = cache.find(rhs.str());
389 if (cached != cache.end()) {
390 id = cached->second;
391 return;
392 }
393
394 if (op->index.type().is_vector()) {
395 // If index is a vector, gather vector elements.
396 internal_assert(op->type.is_vector());
397
398 // This has to be underscore as print_name prepends an underscore to
399 // names without one and that results in a name mismatch if a Load
400 // appears as the value of a Let
401 id = unique_name('_');
402 cache[rhs.str()] = id;
403
404 stream << get_indent() << print_type(op->type)
405 << " " << id << ";\n";
406
407 for (int i = 0; i < op->type.lanes(); ++i) {
408 stream << get_indent();
409 stream
410 << id << "[" << i << "]"
411 << " = ((" << get_memory_space(op->name) << " "
412 << print_type(op->type.element_of()) << "*)"
413 << print_name(op->name) << ")"
414 << "[" << id_index << "[" << i << "]];\n";
415 }
416 } else {
417 print_assignment(op->type, rhs.str());
418 }
419}

421void CodeGen_Metal_Dev::CodeGen_Metal_C::visit(const Store *op) {
422 user_assert(is_const_one(op->predicate)) << "Predicated store is not supported inside Metal kernel.\n";
423 user_assert(op->value.type().lanes() <= 4) << "Vectorization by widths greater than 4 is not supported by Metal -- type is " << op->value.type() << ".\n";
424
425 string id_value = print_expr(op->value);
426 Type t = op->value.type();
427
428 // If we're writing a contiguous ramp, store through a pointer of vector type.
429 Expr ramp_base = is_ramp_one(op->index);
430 if (ramp_base.defined()) {
431 internal_assert(op->value.type().is_vector());
432 string id_ramp_base = print_expr(ramp_base);
433
434 stream << get_indent() << "*(" << get_memory_space(op->name) << " " << print_storage_type(t) << " *)(("
435 << get_memory_space(op->name) << " " << print_type(t.element_of()) << " *)" << print_name(op->name)
436 << " + " << id_ramp_base << ") = " << id_value << ";\n";
437 } else if (op->index.type().is_vector()) {
438 // If index is a vector, scatter vector elements.
439 internal_assert(t.is_vector());
440
441 string id_index = print_expr(op->index);
442
443 for (int i = 0; i < t.lanes(); ++i) {
444 stream << get_indent() << "((" << get_memory_space(op->name) << " "
445 << print_storage_type(t.element_of()) << " *)"
446 << print_name(op->name)
447 << ")["
448 << id_index << "[" << i << "]] = "
449 << id_value << "[" << i << "];\n";
450 }
451 } else {
452 bool type_cast_needed = !(allocations.contains(op->name) &&
453 allocations.get(op->name).type == t);
454
455 string id_index = print_expr(op->index);
456 string id_value = print_expr(op->value);
457 stream << get_indent();
458
459 if (type_cast_needed) {
460 stream << "(("
461 << get_memory_space(op->name) << " "
462 << print_storage_type(t)
463 << " *)"
464 << print_name(op->name)
465 << ")";
466 } else {
467 stream << print_name(op->name);
468 }
469 stream << "[" << id_index << "] = "
470 << id_value << ";\n";
471 }
472
473 cache.clear();
474}

489void CodeGen_Metal_Dev::CodeGen_Metal_C::visit(const Allocate *op) {
490
491 if (op->memory_type == MemoryType::GPUShared) {
492 // Already handled
493 op->body.accept(this);
494 } else {
495 open_scope();
496
497 debug(2) << "Allocate " << op->name << " on device\n";
498
499 debug(3) << "Pushing allocation called " << op->name << " onto the symbol table\n";
500
501 // Allocation is not a shared memory allocation, just make a local declaration.
502 // It must have a constant size.
503 int32_t size = op->constant_allocation_size();
504 user_assert(size > 0)
505 << "Allocation " << op->name << " has a dynamic size. "
506 << "Only fixed-size allocations are supported on the gpu. "
507 << "Try storing into shared memory instead.";
508
509 stream << get_indent() << print_storage_type(op->type) << " "
510 << print_name(op->name) << "[" << size << "];\n";
511 stream << get_indent() << "#define " << get_memory_space(op->name) << " thread\n";
512
513 Allocation alloc;
514 alloc.type = op->type;
515 allocations.push(op->name, alloc);
516
517 op->body.accept(this);
518
519 // Should have been freed internally
520 internal_assert(!allocations.contains(op->name));
521
522 close_scope("alloc " + print_name(op->name));
523 }

573void CodeGen_Metal_Dev::CodeGen_Metal_C::add_kernel(const Stmt &s,
574 const string &name,
575 const vector<DeviceArgument> &args) {
576
577 debug(2) << "Adding Metal kernel " << name << "\n";
578
579 // Figure out which arguments should be passed in constant.
580 // Such arguments should be:
581 // - not written to,
582 // - loads are block-uniform,
583 // - constant size,
584 // - and all allocations together should be less than the max constant
585 // buffer size given by CL_DEVICE_MAX_CONSTANT_BUFFER_SIZE.
586 // The last condition is handled via the preprocessor in the kernel
587 // declaration.
588 vector<BufferSize> constants;
589 for (size_t i = 0; i < args.size(); i++) {
590 if (args[i].is_buffer &&
591 CodeGen_GPU_Dev::is_buffer_constant(s, args[i].name) &&
592 args[i].size > 0) {
593 constants.emplace_back(args[i].name, args[i].size);
594 }
595 }
596
597 // Sort the constant candidates from smallest to largest. This will put
598 // as many of the constant allocations in constant as possible.
599 // Ideally, we would prioritize constant buffers by how frequently they
600 // are accessed.
601 sort(constants.begin(), constants.end());
602
603 // Compute the cumulative sum of the constants.
604 for (size_t i = 1; i < constants.size(); i++) {
605 constants[i].size += constants[i - 1].size;
606 }
607
608 // Create preprocessor replacements for the address spaces of all our buffers.
609 stream << "// Address spaces for " << name << "\n";
610 for (size_t i = 0; i < args.size(); i++) {
611 if (args[i].is_buffer) {
612 vector<BufferSize>::iterator constant = constants.begin();
613 while (constant != constants.end() &&
614 constant->name != args[i].name) {
615 constant++;
616 }
617
618 if (constant != constants.end()) {
619 stream << "#if " << constant->size << " < MAX_CONSTANT_BUFFER_SIZE && "
620 << constant - constants.begin() << " < MAX_CONSTANT_ARGS\n";
621 stream << "#define " << get_memory_space(args[i].name) << " constant\n";
622 stream << "#else\n";
623 stream << "#define " << get_memory_space(args[i].name) << " device\n";
624 stream << "#endif\n";
625 } else {
626 stream << "#define " << get_memory_space(args[i].name) << " device\n";
627 }
628 }
629 }
630
631 // Emit a struct to hold the scalar args of the kernel
632 bool any_scalar_args = false;
633 for (size_t i = 0; i < args.size(); i++) {
634 if (!args[i].is_buffer) {
635 if (!any_scalar_args) {
636 stream << "struct " + name + "_args {\n";
637 any_scalar_args = true;
638 }
639 stream << print_type(args[i].type)
640 << " "
641 << print_name(args[i].name)
642 << ";\n";
643 }
644 }
645 if (any_scalar_args) {
646 stream << "};\n";
647 }
648
649 // Emit the function prototype
650 stream << "kernel void " << name << "(\n";
651 stream << "uint3 tgroup_index [[ threadgroup_position_in_grid ]],\n"
652 << "uint3 tid_in_tgroup [[ thread_position_in_threadgroup ]]";
653 size_t buffer_index = 0;
654 if (any_scalar_args) {
655 stream << ",\nconst device " << name << "_args *_scalar_args [[ buffer(0) ]]";
656 buffer_index++;
657 }
658
659 for (size_t i = 0; i < args.size(); i++) {
660 if (args[i].is_buffer) {
661 stream << ",\n";
662 stream << " " << get_memory_space(args[i].name) << " ";
663 if (!args[i].write) {
664 stream << "const ";
665 }
666 stream << print_storage_type(args[i].type) << " *"
667 << print_name(args[i].name) << " [[ buffer(" << buffer_index++ << ") ]]";
668 Allocation alloc;
669 alloc.type = args[i].type;
670 allocations.push(args[i].name, alloc);
671 }
672 }
673
674 class FindShared : public IRVisitor {
675 using IRVisitor::visit;
676 void visit(const Allocate *op) override {
677 if (op->memory_type == MemoryType::GPUShared) {
678 internal_assert(alloc == nullptr)
679 << "Found multiple shared allocations in metal kernel\n";
680 alloc = op;
681 }
682 }
683
684 public:
685 const Allocate *alloc = nullptr;
686 } find_shared;
687 s.accept(&find_shared);
688
689 if (find_shared.alloc) {
690 shared_name = find_shared.alloc->name;
691 } else {
692 shared_name = "__shared";
693 }
694 // Note that int4 below is an int32x4, not an int4_t. The type
695 // is chosen to be large to maximize alignment.
696 stream << ",\n"
697 << " threadgroup int4* "
698 << print_name(shared_name) << " [[ threadgroup(0) ]]"
699 << ")\n";
700
701 open_scope();
702
703 // Unpack args struct into local variables to match naming of generated code.
704 for (size_t i = 0; i < args.size(); i++) {
705 if (!args[i].is_buffer) {
706 stream << print_type(args[i].type)
707 << " "
708 << print_name(args[i].name)
709 << " = _scalar_args->" << print_name(args[i].name)
710 << ";\n";
711 }
712 }
713
714 print(s);
715 close_scope("kernel " + name);
716
717 for (size_t i = 0; i < args.size(); i++) {
718 // Remove buffer arguments from allocation scope
719 if (args[i].is_buffer) {
720 allocations.pop(args[i].name);
721 }
722 }
723
724 // Undef all the buffer address spaces, in case they're different in another kernel.
725 for (size_t i = 0; i < args.size(); i++) {
726 if (args[i].is_buffer) {
727 stream << "#undef " << get_memory_space(args[i].name) << "\n";
728 }
729 }
730}

732void CodeGen_Metal_Dev::init_module() {
733 debug(2) << "Metal device codegen init_module\n";
734
735 // wipe the internal kernel source
736 src_stream.str("");
737 src_stream.clear();
738
739 // Write out the Halide math functions.
740 src_stream << "#pragma clang diagnostic ignored \"-Wunused-function\"\n"
741 << "#include <metal_stdlib>\n"
742 << "using namespace metal;\n" // Seems like the right way to go.
743 << "namespace {\n"
744 << "constexpr float float_from_bits(unsigned int x) {return as_type<float>(x);}\n"
745 << "constexpr float nan_f32() { return as_type<float>(0x7fc00000); }\n" // Quiet NaN with minimum fractional value.
746 << "constexpr float neg_inf_f32() { return float_from_bits(0xff800000); }\n"
747 << "constexpr float inf_f32() { return float_from_bits(0x7f800000); }\n"
748 << "float fast_inverse_f32(float x) { return 1.0f / x; }\n"
749 << "#define is_nan_f32 isnan\n"
750 << "#define is_inf_f32 isinf\n"
751 << "#define is_finite_f32 isfinite\n"
752 << "#define sqrt_f32 sqrt\n"
753 << "#define sin_f32 sin\n"
754 << "#define cos_f32 cos\n"
755 << "#define exp_f32 exp\n"
756 << "#define log_f32 log\n"
757 << "#define abs_f32 fabs\n"
758 << "#define floor_f32 floor\n"
759 << "#define ceil_f32 ceil\n"
760 << "#define round_f32 round\n"
761 << "#define trunc_f32 trunc\n"
762 << "#define pow_f32 pow\n"
763 << "#define asin_f32 asin\n"
764 << "#define acos_f32 acos\n"
765 << "#define tan_f32 tan\n"
766 << "#define atan_f32 atan\n"
767 << "#define atan2_f32 atan2\n"
768 << "#define sinh_f32 sinh\n"
769 << "#define asinh_f32 asinh\n"
770 << "#define cosh_f32 cosh\n"
771 << "#define acosh_f32 acosh\n"
772 << "#define tanh_f32 tanh\n"
773 << "#define atanh_f32 atanh\n"
774 << "#define fast_inverse_sqrt_f32 rsqrt\n"
775 // This is quite annoying: even though the MSL docs claim
776 // all versions of Metal support the same memory fence
777 // names, the truth is that 1.0 does not.
778 << "#if __METAL_VERSION__ >= 120\n"
779 << "#define _halide_mem_fence_device_and_threadgroup (mem_flags::mem_device | mem_flags::mem_threadgroup)\n"
780 << "#else\n"
781 << "#define _halide_mem_fence_device_and_threadgroup mem_flags::mem_device_and_threadgroup\n"
782 << "#endif\n"
783 << "}\n"; // close namespace
784
785 src_stream << "#define halide_unused(x) (void)(x)\n";
786
787 src_stream << "\n";
788
789 cur_kernel_name = "";
790}
