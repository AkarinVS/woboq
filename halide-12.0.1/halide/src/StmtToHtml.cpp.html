40 // All spans and divs will have an id of the form "x-y", where x
41 // is shared among all spans/divs in the same context, and y is unique.
42 std::vector<int> context_stack;
43 string open_tag(const string &tag, const string &cls, int id = -1) {
44 std::stringstream s;
45 s << "<" << tag << " class='" << cls << "' id='";
46 if (id == -1) {
47 s << context_stack.back() << "-";
48 s << unique_id();
49 } else {
50 s << id;
51 }
52 s << "'>";
53 context_stack.push_back(unique_id());
54 return s.str();
55 }
56 string tag(const string &tag, const string &cls, const string &body, int id = -1) {
57 std::stringstream s;
58 s << open_tag(tag, cls, id);
59 s << body;
60 s << close_tag(tag);
61 return s.str();
62 }
63 string close_tag(const string &tag) {
64 context_stack.pop_back();
65 return "</" + tag + ">";
66 }

108 Scope<int> scope;
109 string var(const string &x) {
110 int id;
111 if (scope.contains(x)) {
112 id = scope.get(x);
113 } else {
114 id = unique_id();
115 scope.push(x, id);
116 }
117
118 std::stringstream s;
119 s << "<b class='Variable Matched' id='" << id << "-" << unique_id() << "'>";
120 s << x;
121 s << "</b>";
122 return s.str();
123 }

173 void visit(const StringImm *op) override {
174 stream << open_span("StringImm");
175 stream << "\"";
176 for (size_t i = 0; i < op->value.size(); i++) {
177 unsigned char c = op->value[i];
178 if (c >= ' ' && c <= '~' && c != '\\' && c != '"') {
179 stream << c;
180 } else {
181 stream << "\\";
182 switch (c) {
183 case '"':
184 stream << "\"";
185 break;
186 case '\\':
187 stream << "\\";
188 break;
189 case '\t':
190 stream << "t";
191 break;
192 case '\r':
193 stream << "r";
194 break;
195 case '\n':
196 stream << "n";
197 break;
198 default:
199 string hex_digits = "0123456789ABCDEF";
200 stream << "x" << hex_digits[c >> 4] << hex_digits[c & 0xf];
201 }
202 }
203 }
204 stream << "\""
205 << close_span();
206 }

392 stream << open_expand_button(id);

394 if (op->for_type == ForType::Serial) {
395 stream << keyword("for");
396 } else if (op->for_type == ForType::Parallel) {
397 stream << keyword("parallel");
398 } else if (op->for_type == ForType::Vectorized) {
399 stream << keyword("vectorized");
400 } else if (op->for_type == ForType::Unrolled) {
401 stream << keyword("unrolled");

624 void visit(const IfThenElse *op) override {
625 stream << open_div("IfThenElse");
626 int id = unique_id();
627 stream << open_expand_button(id);
628 stream << open_span("Matched");
629 stream << keyword("if") << " (";
630 stream << close_span();
631 while (true) {
632 print(op->condition);
633 stream << matched(")");
634 stream << close_expand_button() << " ";
635 stream << matched("{"); // close if (or else if) span
636
637 stream << open_div("ThenBody Indent", id);
638 print(op->then_case);
639 stream << close_div(); // close thenbody div
640
641 if (!op->else_case.defined()) {
642 stream << matched("}");
643 break;
644 }
645
646 id = unique_id();
647
648 if (const IfThenElse *nested_if = op->else_case.as<IfThenElse>()) {
649 stream << matched("}") << " ";
650 stream << open_expand_button(id);
651 stream << open_span("Matched");
652 stream << keyword("else if") << " (";
653 stream << close_span();
654 op = nested_if;
655 } else {
656 stream << open_span("Matched") << "} ";
657 stream << open_expand_button(id);
658 stream << keyword("else");
659 stream << close_expand_button() << "{";
660 stream << close_span();
661 stream << open_div("ElseBody Indent", id);
662 print(op->else_case);
663 stream << close_div() << matched("}");
664 break;
665 }
666 }
667 stream << close_div(); // Closing ifthenelse div.
668 }

676 void visit(const Shuffle *op) override {
677 stream << open_span("Shuffle");
678 if (op->is_concat()) {
679 print_list(symbol("concat_vectors("), op->vectors, ")");
680 } else if (op->is_interleave()) {
681 print_list(symbol("interleave_vectors("), op->vectors, ")");
682 } else if (op->is_extract_element()) {
683 std::vector<Expr> args = op->vectors;
684 args.emplace_back(op->slice_begin());
685 print_list(symbol("extract_element("), args, ")");
686 } else if (op->is_slice()) {
687 std::vector<Expr> args = op->vectors;
688 args.emplace_back(op->slice_begin());
689 args.emplace_back(op->slice_stride());
690 args.emplace_back(static_cast<int>(op->indices.size()));
691 print_list(symbol("slice_vectors("), args, ")");
692 } else {
693 std::vector<Expr> args = op->vectors;
694 for (int i : op->indices) {
695 args.emplace_back(i);
696 }
697 print_list(symbol("shuffle("), args, ")");
698 }
699 stream << close_span();
700 }
