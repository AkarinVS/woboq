31inline int64_t saturating_mul(int64_t a, int64_t b) {
32 if (mul_would_overflow(64, a, b)) {
33 if ((a > 0) == (b > 0)) {
34 return INT64_MAX;
35 } else {
36 return INT64_MIN;
37 }
38 } else {
39 return a * b;
40 }
41}

49 struct ExprInfo {
50 // We track constant integer bounds when they exist
51 // TODO: Use ConstantInterval?
52 int64_t min = 0, max = 0;
53 bool min_defined = false, max_defined = false;
54 // And the alignment of integer variables
55 ModulusRemainder alignment;
56
57 void trim_bounds_using_alignment() {
58 if (alignment.modulus == 0) {
59 min_defined = max_defined = true;
60 min = max = alignment.remainder;
61 } else if (alignment.modulus > 1) {
62 if (min_defined) {
63 int64_t new_min = min - mod_imp(min, alignment.modulus) + alignment.remainder;
64 if (new_min < min) {
65 new_min += alignment.modulus;
66 }
67 min = new_min;
68 }
69 if (max_defined) {
70 int64_t new_max = max - mod_imp(max, alignment.modulus) + alignment.remainder;
71 if (new_max > max) {
72 new_max -= alignment.modulus;
73 }
74 max = new_max;
75 }
76 }
77
78 if (min_defined && max_defined && min == max) {
79 alignment.modulus = 0;
80 alignment.remainder = min;
81 }
82 }
83
84 // Mix in existing knowledge about this Expr
85 void intersect(const ExprInfo &other) {
86 if (min_defined && other.min_defined) {
87 min = std::max(min, other.min);
88 } else if (other.min_defined) {
89 min_defined = true;
90 min = other.min;
91 }
92
93 if (max_defined && other.max_defined) {
94 max = std::min(max, other.max);
95 } else if (other.max_defined) {
96 max_defined = true;
97 max = other.max;
98 }
99
100 alignment = ModulusRemainder::intersect(alignment, other.alignment);
101
102 trim_bounds_using_alignment();
103 }
104 };

111 Expr mutate(const Expr &e, ExprInfo *b) {
112 const std::string spaces(debug_indent, ' ');
113 debug(1) << spaces << "Simplifying Expr: " << e << "\n";
114 debug_indent++;
115 Expr new_e = Super::dispatch(e, b);
116 debug_indent--;
117 if (!new_e.same_as(e)) {
118 debug(1)
119 << spaces << "Before: " << e << "\n"
120 << spaces << "After: " << new_e << "\n";
121 }
122 internal_assert(e.type() == new_e.type());
123 return new_e;
124 }

162 // Returns true iff t is an integral type where overflow is undefined
163 HALIDE_ALWAYS_INLINE
164 bool no_overflow_int(Type t) {
165 return t.is_int() && t.bits() >= 32;
166 }
167
168 HALIDE_ALWAYS_INLINE
169 bool no_overflow_scalar_int(Type t) {
170 return t.is_scalar() && no_overflow_int(t);
171 }
172
173 // Returns true iff t does not have a well defined overflow behavior.
174 HALIDE_ALWAYS_INLINE
175 bool no_overflow(Type t) {
176 return t.is_float() || no_overflow_int(t);

223 // Put the args to a commutative op in a canonical order
224 HALIDE_ALWAYS_INLINE
225 bool should_commute(const Expr &a, const Expr &b) {
226 if (a.node_type() < b.node_type()) {
227 return true;
228 }
229 if (a.node_type() > b.node_type()) {
230 return false;
231 }
232
233 if (a.node_type() == IRNodeType::Variable) {
234 const Variable *va = a.as<Variable>();
235 const Variable *vb = b.as<Variable>();
236 return va->name.compare(vb->name) > 0;
237 }
238
239 return false;
240 }
