142 /** Given a string of the form used in HL_TARGET
143 * (e.g. "x86-64-avx"), construct the Target it specifies. Note
144 * that this always starts with the result of get_host_target(),
145 * replacing only the parts found in the target string, so if you
146 * omit (say) an OS specification, the host OS will be used
147 * instead. An empty string is exactly equivalent to
148 * get_host_target().
149 *
150 * Invalid target strings will fail with a user_error.
151 */
152 // @{
153 explicit Target(const std::string &s);
154 explicit Target(const char *s);
155 // @}

190 /** Is a fully feature GPU compute runtime enabled? I.e. is
191 * Func::gpu_tile and similar going to work? Currently includes
192 * CUDA, OpenCL, Metal and D3D12Compute. We do not include OpenGL,
193 * because it is not capable of gpgpu, and is not scheduled via
194 * Func::gpu_tile.
195 * TODO: Should OpenGLCompute be included here? */
196 bool has_gpu_feature() const;
197
198 /** Does this target allow using a certain type. Generally all
199 * types except 64-bit float and int/uint should be supported by
200 * all backends.
201 *
202 * It is likely better to call the version below which takes a DeviceAPI.
203 */
204 bool supports_type(const Type &t) const;
205
206 /** Does this target allow using a certain type on a certain device.
207 * This is the prefered version of this routine.
208 */
209 bool supports_type(const Type &t, DeviceAPI device) const;
210
211 /** Returns whether a particular device API can be used with this
212 * Target. */
213 bool supports_device_api(DeviceAPI api) const;
214
215 /** If this Target (including all Features) requires a specific DeviceAPI,
216 * return it. If it doesn't, return DeviceAPI::None. If the Target has
217 * features with multiple (different) DeviceAPI requirements, the result
218 * will be an arbitrary DeviceAPI. */
219 DeviceAPI get_required_device_api() const;
220
221 bool operator==(const Target &other) const {
222 return os == other.os &&
223 arch == other.arch &&
224 bits == other.bits &&
225 features == other.features;
226 }
227
228 bool operator!=(const Target &other) const {
229 return !(*this == other);
230 }

232 /**
233 * Create a "greatest common denominator" runtime target that is compatible with
234 * both this target and \p other. Used by generators to conveniently select a suitable
235 * runtime when linking together multiple functions.
236 *
237 * @param other The other target from which we compute the gcd target.
238 * @param[out] result The gcd target if we return true, otherwise unmodified. Can be the same as *this.
239 * @return Whether it was possible to find a compatible target (true) or not.
240 */
241 bool get_runtime_compatible_target(const Target &other, Target &result);
242
243 /** Convert the Target into a string form that can be reconstituted
244 * by merge_string(), which will always be of the form
245 *
246 * arch-bits-os-feature1-feature2...featureN.
247 *
248 * Note that is guaranteed that Target(t1.to_string()) == t1,
249 * but not that Target(s).to_string() == s (since there can be
250 * multiple strings that parse to the same Target)...
251 * *unless* t1 contains 'unknown' fields (in which case you'll get a string
252 * that can't be parsed, which is intentional).
253 */
254 std::string to_string() const;

256 /** Given a data type, return an estimate of the "natural" vector size
257 * for that data type when compiling for this Target. */
258 int natural_vector_size(const Halide::Type &t) const;
259
260 /** Given a data type, return an estimate of the "natural" vector size
261 * for that data type when compiling for this Target. */
262 template<typename data_t>
263 int natural_vector_size() const {
264 return natural_vector_size(type_of<data_t>());
265 }
266
267 /** Return true iff 64 bits and has_feature(LargeBuffers). */
268 bool has_large_buffers() const {
269 return bits == 64 && has_feature(LargeBuffers);
270 }
271
272 /** Return the maximum buffer size in bytes supported on this
273 * Target. This is 2^31 - 1 except on 64-bit targets when the LargeBuffers
274 * feature is enabled, which expands the maximum to 2^63 - 1. */
275 int64_t maximum_buffer_size() const {
276 if (has_large_buffers()) {
277 return (((uint64_t)1) << 63) - 1;
278 } else {
279 return (((uint64_t)1) << 31) - 1;
280 }
281 }

283 /** Get the minimum cuda capability found as an integer. Returns

288 /** Was libHalide compiled with support for this target? */

291 /** Return a bitset of the Featuress set in this Target (set = 1).
292 * Note that while this happens to be the current internal representation,
293 * that might not always be the case. */
294 const std::bitset<FeatureEnd> &get_features_bitset() const {
