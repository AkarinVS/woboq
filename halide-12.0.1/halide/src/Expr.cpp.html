7const IntImm *IntImm::make(Type t, int64_t value) {
8 internal_assert(t.is_int() && t.is_scalar())
9 << "IntImm must be a scalar Int\n";
10 internal_assert(t.bits() >= 1 && t.bits() <= 64)
11 << "IntImm must have between 1 and 64 bits\n";
12
13 // Normalize the value by dropping the high bits.
14 // Since left-shift of negative value is UB in C++, cast to uint64 first;
15 // it's unlikely any compilers we care about will misbehave, but UBSan will complain.
16 value = (int64_t)(((uint64_t)value) << (64 - t.bits()));
17
18 // Then sign-extending to get them back
19 value >>= (64 - t.bits());
20
21 IntImm *node = new IntImm;
22 node->type = t;
23 node->value = value;
24 return node;
25}

27const UIntImm *UIntImm::make(Type t, uint64_t value) {
28 internal_assert(t.is_uint() && t.is_scalar())
29 << "UIntImm must be a scalar UInt\n";
30 internal_assert(t.bits() >= 1 && t.bits() <= 64)
31 << "UIntImm must have between 1 and 64 bits\n";
32
33 // Normalize the value by dropping the high bits
34 value <<= (64 - t.bits());
35 value >>= (64 - t.bits());
36
37 UIntImm *node = new UIntImm;
38 node->type = t;
39 node->value = value;
40 return node;
41}

43const FloatImm *FloatImm::make(Type t, double value) {
44 internal_assert(t.is_float() && t.is_scalar())
45 << "FloatImm must be a scalar Float\n";
46 FloatImm *node = new FloatImm;
47 node->type = t;
48 switch (t.bits()) {
49 case 16:
50 if (t.is_bfloat()) {
51 node->value = (double)((bfloat16_t)value);
52 } else {
53 node->value = (double)((float16_t)value);
54 }
55 break;
56 case 32:
57 node->value = (float)value;
58 break;
59 case 64:
60 node->value = value;
61 break;
62 default:
63 internal_error << "FloatImm must be 16, 32, or 64-bit\n";
64 }
65
66 return node;
67}

76/** Check if for_type executes for loop iterations in parallel and unordered. */
77bool is_unordered_parallel(ForType for_type) {
78 return (for_type == ForType::Parallel ||
79 for_type == ForType::GPUBlock ||
80 for_type == ForType::GPUThread);
81}
82
83/** Returns true if for_type executes for loop iterations in parallel. */
84bool is_parallel(ForType for_type) {
85 return (is_unordered_parallel(for_type) ||
86 for_type == ForType::Vectorized ||
87 for_type == ForType::GPULane);
88}
