20class FindFreeVars : public IRMutator {
21public:
22 vector<Var> free_vars;
23 vector<Expr> call_args;
24 RDom rdom;
25
26 FindFreeVars(const RDom &r, const string &n)
27 : rdom(r), explicit_rdom(r.defined()), name(n) {
28 }
29
30private:
31 bool explicit_rdom;
32 const string &name;
33
34 Scope<> internal;
35
36 using IRMutator::visit;
37
38 Expr visit(const Let *op) override {
39 Expr value = mutate(op->value);
40 internal.push(op->name);
41 Expr body = mutate(op->body);
42 internal.pop(op->name);
43 if (value.same_as(op->value) &&
44 body.same_as(op->body)) {
45 return op;
46 } else {
47 return Let::make(op->name, value, body);
48 }
49 }
50
51 Expr visit(const Variable *v) override {
52 string var_name = v->name;
53 Expr expr = v;
54
55 if (internal.contains(var_name)) {
56 // Don't capture internally defined vars
57 return expr;
58 }
59
60 if (v->reduction_domain.defined()) {
61 if (explicit_rdom) {
62 if (v->reduction_domain.same_as(rdom.domain())) {
63 // This variable belongs to the explicit reduction domain, so
64 // skip it.
65 return expr;
66 } else {
67 // This should be converted to a pure variable and
68 // added to the free vars list.
69 var_name = unique_name('v');
70 expr = Variable::make(v->type, var_name);
71 }
72 } else {
73 if (!rdom.defined()) {
74 // We're looking for a reduction domain, and this variable
75 // has one. Capture it.
76 rdom = RDom(v->reduction_domain);
77 return expr;
78 } else if (!rdom.domain().same_as(v->reduction_domain)) {
79 // We were looking for a reduction domain, and already
80 // found one. This one is different!
81 user_error << "Inline reduction \"" << name
82 << "\" refers to reduction variables from multiple reduction domains: "
83 << v->name << ", " << rdom.x.name() << "\n";
84 } else {
85 // Recapturing an already-known reduction domain
86 return expr;
87 }
88 }
89 }
90
91 if (v->param.defined()) {
92 // Skip parameters
93 return expr;
94 }
95
96 for (size_t i = 0; i < free_vars.size(); i++) {
97 if (var_name == free_vars[i].name()) {
98 return expr;
99 }
100 }
101
102 free_vars.emplace_back(var_name);
103 call_args.emplace_back(v);
104 return expr;
105 }
106};

123Expr sum(const RDom &r, Expr e, const Func &f) {
124 user_assert(!f.defined())
125 << "Func " << f.name()
126 << " passed to sum already has a definition";
127
128 Internal::FindFreeVars v(r, f.name());
129 e = v.mutate(common_subexpression_elimination(e));
130
131 user_assert(v.rdom.defined()) << "Expression passed to sum must reference a reduction domain";
132
133 f(v.free_vars) += e;
134 return f(v.call_args);
135}

256Tuple argmax(const RDom &r, Expr e, const Func &f) {
257 user_assert(!f.defined())
258 << "Func " << f.name()
259 << " passed to argmax already has a definition";
260
261 Internal::FindFreeVars v(r, f.name());
262 e = v.mutate(common_subexpression_elimination(e));
263
264 user_assert(v.rdom.defined()) << "Expression passed to argmax must reference a reduction domain";
265
266 Tuple initial_tup(vector<Expr>(v.rdom.dimensions() + 1));
267 Tuple update_tup(vector<Expr>(v.rdom.dimensions() + 1));
268 for (int i = 0; i < v.rdom.dimensions(); i++) {
269 initial_tup[i] = 0;
270 update_tup[i] = v.rdom[i];
271 }
272 int value_index = (int)initial_tup.size() - 1;
273 initial_tup[value_index] = e.type().min();
274 update_tup[value_index] = e;
275
276 f(v.free_vars) = initial_tup;
277 Expr better = e > f(v.free_vars)[value_index];
278 Tuple update = tuple_select(better, update_tup, f(v.free_vars));
279 f(v.free_vars) = update;
280 return f(v.call_args);
281}
