23// T is an ImageParam, Buffer<>, Input<Buffer<>>
24template<typename T>
25Internal::ReductionDomain make_dom_from_dimensions(const T &t, const std::string &name) {
26 std::vector<Internal::ReductionVariable> vars;
27 for (int i = 0; i < t.dimensions(); i++) {
28 vars.push_back({name + dom_var_names[i],
29 t.dim(i).min(),
30 t.dim(i).extent()});
31 }
32
33 return Internal::ReductionDomain(vars);
34}

76// This just initializes the predefined x, y, z, w members of RDom.
77void RDom::init_vars(const string &name) {
78 const std::vector<ReductionVariable> &dom_vars = dom.domain();
79 std::array<RVar *, 4> vars = {{&x, &y, &z, &w}};
80
81 for (size_t i = 0; i < vars.size(); i++) {
82 if (i < dom_vars.size()) {
83 *(vars[i]) = RVar(dom, i);
84 } else {
85 *(vars[i]) = RVar(name + dom_var_names[i]);
86 }
87 }
88}

98class CheckRDomBounds : public IRGraphVisitor {
99
100 using IRGraphVisitor::visit;
101
102 void visit(const Call *op) override {
103 IRGraphVisitor::visit(op);
104 if (op->call_type == Call::Halide) {
105 offending_func = op->name;
106 }
107 }
108
109 void visit(const Variable *op) override {
110 if (!op->param.defined() &&
111 !op->image.defined() &&
112 !internal_vars.contains(op->name)) {
113 offending_free_var = op->name;
114 }
115 }
116
117 void visit(const Let *op) override {
118 ScopedBinding<int> bind(internal_vars, op->name, 0);
119 IRGraphVisitor::visit(op);
120 }
121 Scope<int> internal_vars;
122
123public:
124 string offending_func;
125 string offending_free_var;
126};

129void RDom::initialize_from_region(const Region &region, string name) {
130 if (name.empty()) {
131 name = make_entity_name(this, "Halide:.*:RDom", 'r');
132 }
133
134 std::vector<ReductionVariable> vars;
135 for (size_t i = 0; i < region.size(); i++) {
136 CheckRDomBounds checker;
137 user_assert(region[i].min.defined() && region[i].extent.defined())
138 << "The RDom " << name << " may not be constructed with undefined Exprs.\n";
139 region[i].min.accept(&checker);
140 region[i].extent.accept(&checker);
141 user_assert(checker.offending_func.empty())
142 << "The bounds of the RDom " << name
143 << " in dimension " << i
144 << " are:\n"
145 << " " << region[i].min << " ... " << region[i].extent << "\n"
146 << "These depend on a call to the Func " << checker.offending_func << ".\n"
147 << "The bounds of an RDom may not depend on a call to a Func.\n";
148 user_assert(checker.offending_free_var.empty())
149 << "The bounds of the RDom " << name
150 << " in dimension " << i
151 << " are:\n"
152 << " " << region[i].min << " ... " << region[i].extent << "\n"
153 << "These depend on the variable " << checker.offending_free_var << ".\n"
154 << "The bounds of an RDom may not depend on a free variable.\n";
155
156 std::string rvar_uniquifier;
157 switch (i) {
158 case 0:
159 rvar_uniquifier = "x";
160 break;
161 case 1:
162 rvar_uniquifier = "y";
163 break;
164 case 2:
165 rvar_uniquifier = "z";
166 break;
167 case 3:
168 rvar_uniquifier = "w";
169 break;
170 default:
171 rvar_uniquifier = std::to_string(i);
172 break;
173 }
174 ReductionVariable rv;
175 rv.var = name + "$" + rvar_uniquifier;
176 rv.min = cast<int32_t>(region[i].min);
177 rv.extent = cast<int32_t>(region[i].extent);
178 vars.push_back(rv);
179 }
180 dom = ReductionDomain(vars);
181 init_vars(name);
182}

200RVar RDom::operator[](int i) const {
201 switch (i) {
202 case 0:
203 return x;
204 case 1:
205 return y;
206 case 2:
207 return z;
208 case 3:
209 return w;
210 default:
211 if (i < dimensions()) {
212 return RVar(dom, i);
213 } else {
214 user_error << "Reduction domain index out of bounds: " << i << "\n";
215 return x; // Keep the compiler happy
216 }
217 }

252/** Emit an RDom in a human-readable form. */
253std::ostream &operator<<(std::ostream &stream, const RDom &dom) {
254 stream << "RDom(\n";
255 for (int i = 0; i < dom.dimensions(); i++) {
256 stream << " " << dom[i] << "\n";
257 }
258 stream << ")";
259 Expr pred = simplify(dom.domain().predicate());
260 if (!equal(const_true(), pred)) {
261 stream << " where (\n " << pred << ")";
262 }
263 stream << "\n";
264 return stream;
265}
