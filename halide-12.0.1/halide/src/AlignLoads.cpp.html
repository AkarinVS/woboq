18// This mutator attempts to rewrite unaligned or strided loads to
19// sequences of aligned loads by loading aligned vectors that cover
20// the original unaligned load, and then slicing or shuffling the
21// intended vector out of the aligned vector.
22class AlignLoads : public IRMutator {
23public:
24 AlignLoads(int alignment)
25 : alignment_analyzer(alignment), required_alignment(alignment) {
26 }
27
28private:
29 HexagonAlignmentAnalyzer alignment_analyzer;
30
31 // Loads and stores should ideally be aligned to the vector width in bytes.
32 int required_alignment;
33
34 using IRMutator::visit;
35
36 // Rewrite a load to have a new index, updating the type if necessary.
37 Expr make_load(const Load *load, const Expr &index, ModulusRemainder alignment) {
38 internal_assert(is_const_one(load->predicate)) << "Load should not be predicated.\n";
39 return mutate(Load::make(load->type.with_lanes(index.type().lanes()), load->name,
40 index, load->image, load->param,
41 const_true(index.type().lanes()),
42 alignment));
43 }
44
45 Expr visit(const Load *op) override {
46 if (!is_const_one(op->predicate)) {
47 // TODO(psuriana): Do nothing to predicated loads for now.
48 return IRMutator::visit(op);
49 }
50
51 if (!op->type.is_vector()) {
52 // Nothing to do for scalar loads.
53 return IRMutator::visit(op);
54 }
55
56 if (op->image.defined()) {
57 // We can't reason about the alignment of external images.
58 return IRMutator::visit(op);
59 }
60
61 if (required_alignment % op->type.bytes() != 0) {
62 return IRMutator::visit(op);
63 }
64
65 if (op->type.bytes() * op->type.lanes() <= 8) {
66 // These can probably be treated as scalars instead.
67 return IRMutator::visit(op);
68 }
69
70 Expr index = mutate(op->index);
71 const Ramp *ramp = index.as<Ramp>();
72 const int64_t *const_stride = ramp ? as_const_int(ramp->stride) : nullptr;
73 if (!ramp || !const_stride) {
74 // We can't handle indirect loads, or loads with
75 // non-constant strides.
76 return IRMutator::visit(op);
77 }
78 if (!(*const_stride == 1 || *const_stride == 2 || *const_stride == 3)) {
79 // Handle ramps with stride 1, 2 or 3 only.
80 return IRMutator::visit(op);
81 }
82
83 int64_t aligned_offset = 0;
84 bool is_aligned =
85 alignment_analyzer.is_aligned(op, &aligned_offset);
86 // We know the alignment_analyzer has been able to reason about alignment
87 // if the following is true.
88 bool known_alignment = is_aligned || (!is_aligned && aligned_offset != 0);
89 int lanes = ramp->lanes;
90 int native_lanes = required_alignment / op->type.bytes();
91 int stride = static_cast<int>(*const_stride);
92 if (stride != 1) {
93 internal_assert(stride >= 0);
94
95 // If we know the offset of this strided load is smaller
96 // than the stride, we can just make the load aligned now
97 // without requiring more vectors from the dense
98 // load. This makes loads like f(2*x + 1) into an aligned
99 // load of double length, with a single shuffle.
100 int shift = known_alignment && aligned_offset < stride ? aligned_offset : 0;
101
102 // Load a dense vector covering all of the addresses in the load.
103 Expr dense_base = simplify(ramp->base - shift);
104 ModulusRemainder alignment = op->alignment - shift;
105 Expr dense_index = Ramp::make(dense_base, 1, lanes * stride);
106 Expr dense = make_load(op, dense_index, alignment);
107
108 // Shuffle the dense load.
109 return Shuffle::make_slice(dense, shift, stride, lanes);
110 }
111
112 // We now have a dense vector load to deal with.
113 internal_assert(stride == 1);
114 if (lanes < native_lanes) {
115 // This load is smaller than a native vector. Load a
116 // native vector.
117 Expr ramp_base = ramp->base;
118 ModulusRemainder alignment = op->alignment;
119 int slice_offset = 0;
120
121 // If load is smaller than a native vector and can fully fit inside of it and offset is known,
122 // we can simply offset the native load and slice.
123 if (!is_aligned && aligned_offset != 0 && Int(32).can_represent(aligned_offset) && (aligned_offset + lanes <= native_lanes)) {
124 ramp_base = simplify(ramp_base - (int)aligned_offset);
125 alignment = alignment - aligned_offset;
126 slice_offset = aligned_offset;
127 }
128
129 Expr native_load = make_load(op, Ramp::make(ramp_base, 1, native_lanes), alignment);
130
131 // Slice the native load.
132 return Shuffle::make_slice(native_load, slice_offset, 1, lanes);
133 }
134
135 if (lanes > native_lanes) {
136 // This load is larger than a native vector. Load native
137 // vectors, and concatenate the results.
138 vector<Expr> slices;
139 for (int i = 0; i < lanes; i += native_lanes) {
140 int slice_lanes = std::min(native_lanes, lanes - i);
141 Expr slice_base = simplify(ramp->base + i);
142 ModulusRemainder alignment = op->alignment + i;
143 slices.push_back(make_load(op, Ramp::make(slice_base, 1, slice_lanes), alignment));
144 }
145 return Shuffle::make_concat(slices);
146 }
147
148 if (!is_aligned && aligned_offset != 0 && Int(32).can_represent(aligned_offset)) {
149 // We know the offset of this load from an aligned
150 // address. Rewrite this is an aligned load of two
151 // native vectors, followed by a shuffle.
152 Expr aligned_base = simplify(ramp->base - (int)aligned_offset);
153 ModulusRemainder alignment = op->alignment - (int)aligned_offset;
154 Expr aligned_load = make_load(op, Ramp::make(aligned_base, 1, lanes * 2), alignment);
155
156 return Shuffle::make_slice(aligned_load, (int)aligned_offset, 1, lanes);
157 }
158
159 return IRMutator::visit(op);
160 }
161};
162
163} // namespace
164
165Stmt align_loads(const Stmt &s, int alignment) {
166 return AlignLoads(alignment).mutate(s);
167}
