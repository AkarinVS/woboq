28// Sniff the contents of a kernel to extracts the bounds of all the
29// thread indices (so we know how many threads to launch), and the
30// amount of shared memory to allocate.
31class ExtractBounds : public IRVisitor {
32public:
33 Expr num_threads[4];
34 Expr num_blocks[4];
35 Expr shared_mem_size;
36
37 ExtractBounds()
38 : shared_mem_size(0) {
39 for (int i = 0; i < 4; i++) {
40 num_threads[i] = num_blocks[i] = 1;
41 }
42 }
43
44private:
45 bool found_shared = false;
46
47 using IRVisitor::visit;
48
49 void visit(const For *op) override {
50 if (CodeGen_GPU_Dev::is_gpu_var(op->name)) {
51 internal_assert(is_const_zero(op->min));
52 }
53
54 if (ends_with(op->name, ".__thread_id_x")) {
55 num_threads[0] = op->extent;
56 } else if (ends_with(op->name, ".__thread_id_y")) {
57 num_threads[1] = op->extent;
58 } else if (ends_with(op->name, ".__thread_id_z")) {
59 num_threads[2] = op->extent;
60 } else if (ends_with(op->name, ".__thread_id_w")) {
61 num_threads[3] = op->extent;
62 } else if (ends_with(op->name, ".__block_id_x")) {
63 num_blocks[0] = op->extent;
64 } else if (ends_with(op->name, ".__block_id_y")) {
65 num_blocks[1] = op->extent;
66 } else if (ends_with(op->name, ".__block_id_z")) {
67 num_blocks[2] = op->extent;
68 } else if (ends_with(op->name, ".__block_id_w")) {
69 num_blocks[3] = op->extent;
70 }
71
72 op->body.accept(this);
73 }
74
75 void visit(const LetStmt *op) override {
76 if (expr_uses_var(shared_mem_size, op->name)) {
77 shared_mem_size = Let::make(op->name, op->value, shared_mem_size);
78 }
79 op->body.accept(this);
80 }
81
82 void visit(const Allocate *allocate) override {
83 user_assert(!allocate->new_expr.defined()) << "Allocate node inside GPU kernel has custom new expression.\n"
84 << "(Memoization is not supported inside GPU kernels at present.)\n";
85
86 if (allocate->memory_type == MemoryType::GPUShared) {
87 internal_assert(allocate->extents.size() == 1);
88 shared_mem_size += allocate->extents[0] * allocate->type.bytes();
89 found_shared = true;
90 }
91 allocate->body.accept(this);
92 }
93};

95class InjectGpuOffload : public IRMutator {
96 /** Child code generator for device kernels. */
97 map<DeviceAPI, unique_ptr<CodeGen_GPU_Dev>> cgdev;
98
99 map<string, bool> state_needed;
100
101 const Target &target;
102
103 Expr get_state_var(const string &name) {
104 // Expr v = Variable::make(type_of<void *>(), name);
105 state_needed[name] = true;
106 return Load::make(type_of<void *>(), name, 0,
107 Buffer<>(), Parameter(), const_true(), ModulusRemainder());
108 }
109
110 Expr make_state_var(const string &name) {
111 auto storage = Buffer<void *>::make_scalar(name + "_buf");
112 storage() = nullptr;
113 Expr buf = Variable::make(type_of<halide_buffer_t *>(), storage.name() + ".buffer", storage);
114 return Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
115 }
116
117 // Create a Buffer containing the given vector, and return an
118 // expression for a pointer to the first element.
119 Expr make_buffer_ptr(const vector<char> &data, const string &name) {
120 Buffer<uint8_t> code((int)data.size(), name);
121 memcpy(code.data(), data.data(), (int)data.size());
122 Expr buf = Variable::make(type_of<halide_buffer_t *>(), name + ".buffer", code);
123 return Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
124 }
125
126 using IRMutator::visit;
127
128 Stmt visit(const For *loop) override {
129 if (!CodeGen_GPU_Dev::is_gpu_var(loop->name)) {
130 return IRMutator::visit(loop);
131 }
132
133 // We're in the loop over outermost block dimension
134 debug(2) << "Kernel launch: " << loop->name << "\n";
135
136 internal_assert(loop->device_api != DeviceAPI::Default_GPU)
137 << "A concrete device API should have been selected before codegen.";
138
139 ExtractBounds bounds;
140 loop->accept(&bounds);
141 debug(2) << "Kernel bounds: ("
142 << bounds.num_threads[0] << ", "
143 << bounds.num_threads[1] << ", "
144 << bounds.num_threads[2] << ", "
145 << bounds.num_threads[3] << ") threads, ("
146 << bounds.num_blocks[0] << ", "
147 << bounds.num_blocks[1] << ", "
148 << bounds.num_blocks[2] << ", "
149 << bounds.num_blocks[3] << ") blocks\n";
150
151 // compute a closure over the state passed into the kernel
152 HostClosure c(loop->body, loop->name);
153
154 // Determine the arguments that must be passed into the halide function
155 vector<DeviceArgument> closure_args = c.arguments();
156
157 // Sort the args by the size of the underlying type. This is
158 // helpful for avoiding struct-packing ambiguities in metal,
159 // which passes the scalar args as a struct.
160 sort(closure_args.begin(), closure_args.end(),
161 [](const DeviceArgument &a, const DeviceArgument &b) {
162 if (a.is_buffer == b.is_buffer) {
163 return a.type.bits() > b.type.bits();
164 } else {
165 // Ensure that buffer arguments come first:
166 // for many OpenGL/Compute systems, the
167 // legal indices for buffer args are much
168 // more restrictive than for scalar args,
169 // and scalar args can be 'grown' by
170 // LICM. Putting buffers first makes it much
171 // more likely we won't fail on some
172 // hardware.
173 return a.is_buffer > b.is_buffer;
174 }
175 });
176
177 // compile the kernel
178 string kernel_name = c_print_name(unique_name("kernel_" + loop->name));
179
180 CodeGen_GPU_Dev *gpu_codegen = cgdev[loop->device_api].get();
181 user_assert(gpu_codegen != nullptr)
182 << "Loop is scheduled on device " << loop->device_api
183 << " which does not appear in target " << target.to_string() << "\n";
184 gpu_codegen->add_kernel(loop, kernel_name, closure_args);
185
186 // get the actual name of the generated kernel for this loop
187 kernel_name = gpu_codegen->get_current_kernel_name();
188 debug(2) << "Compiled launch to kernel \"" << kernel_name << "\"\n";
189
190 bool runtime_run_takes_types = gpu_codegen->kernel_run_takes_types();
191 Type target_size_t_type = target.bits == 32 ? Int(32) : Int(64);
192
193 vector<Expr> args, arg_types_or_sizes, arg_is_buffer;
194 for (const DeviceArgument &i : closure_args) {
195 Expr val;
196 if (i.is_buffer) {
197 val = Variable::make(Handle(), i.name + ".buffer");
198 } else {
199 val = Variable::make(i.type, i.name);
200 val = Call::make(type_of<void *>(), Call::make_struct, {val}, Call::Intrinsic);
201 }
202 args.emplace_back(val);
203
204 if (runtime_run_takes_types) {
205 arg_types_or_sizes.emplace_back(((halide_type_t)i.type).as_u32());
206 } else {
207 arg_types_or_sizes.emplace_back(cast(target_size_t_type, i.is_buffer ? 8 : i.type.bytes()));
208 }
209
210 arg_is_buffer.emplace_back(cast<uint8_t>(i.is_buffer));
211 }
212
213 // nullptr-terminate the lists
214 args.emplace_back(reinterpret(Handle(), cast<uint64_t>(0)));
215 if (runtime_run_takes_types) {
216 internal_assert(sizeof(halide_type_t) == sizeof(uint32_t));
217 arg_types_or_sizes.emplace_back(cast<uint32_t>(0));
218 } else {
219 arg_types_or_sizes.emplace_back(cast(target_size_t_type, 0));
220 }
221 arg_is_buffer.emplace_back(cast<uint8_t>(0));
222
223 // TODO: only three dimensions can be passed to
224 // cuLaunchKernel. How should we handle blkid[3]?
225 internal_assert(is_const_one(bounds.num_threads[3]) && is_const_one(bounds.num_blocks[3]))
226 << bounds.num_threads[3] << ", " << bounds.num_blocks[3] << "\n";
227 debug(3) << "bounds.num_blocks[0] = " << bounds.num_blocks[0] << "\n";
228 debug(3) << "bounds.num_blocks[1] = " << bounds.num_blocks[1] << "\n";
229 debug(3) << "bounds.num_blocks[2] = " << bounds.num_blocks[2] << "\n";
230 debug(3) << "bounds.num_threads[0] = " << bounds.num_threads[0] << "\n";
231 debug(3) << "bounds.num_threads[1] = " << bounds.num_threads[1] << "\n";
232 debug(3) << "bounds.num_threads[2] = " << bounds.num_threads[2] << "\n";
233
234 string api_unique_name = gpu_codegen->api_unique_name();
235 vector<Expr> run_args = {
236 get_state_var(api_unique_name),
237 kernel_name,
238 Expr(bounds.num_blocks[0]),
239 Expr(bounds.num_blocks[1]),
240 Expr(bounds.num_blocks[2]),
241 Expr(bounds.num_threads[0]),
242 Expr(bounds.num_threads[1]),
243 Expr(bounds.num_threads[2]),
244 Expr(bounds.shared_mem_size),
245 Call::make(Handle(), Call::make_struct, arg_types_or_sizes, Call::Intrinsic),
246 Call::make(Handle(), Call::make_struct, args, Call::Intrinsic),
247 Call::make(Handle(), Call::make_struct, arg_is_buffer, Call::Intrinsic),
248 };
249 return call_extern_and_assert("halide_" + api_unique_name + "_run", run_args);
250 }
251
252public:
253 InjectGpuOffload(const Target &target)
254 : target(target) {
255 if (target.has_feature(Target::OpenGLCompute)) {
256 cgdev[DeviceAPI::OpenGLCompute] = new_CodeGen_OpenGLCompute_Dev(target);
257 }
258 if (target.has_feature(Target::CUDA)) {
259 cgdev[DeviceAPI::CUDA] = new_CodeGen_PTX_Dev(target);
260 }
261 if (target.has_feature(Target::OpenCL)) {
262 cgdev[DeviceAPI::OpenCL] = new_CodeGen_OpenCL_Dev(target);
263 }
264 if (target.has_feature(Target::Metal)) {
265 cgdev[DeviceAPI::Metal] = new_CodeGen_Metal_Dev(target);
266 }
267 if (target.has_feature(Target::D3D12Compute)) {
268 cgdev[DeviceAPI::D3D12Compute] = new_CodeGen_D3D12Compute_Dev(target);
269 }
270
271 internal_assert(!cgdev.empty()) << "Requested unknown GPU target: " << target.to_string() << "\n";
272 }
273
274 Stmt inject(const Stmt &s) {
275 // Create a new module for all of the kernels we find in this function.
276 for (auto &i : cgdev) {
277 i.second->init_module();
278 }
279
280 Stmt result = mutate(s);
281
282 for (auto &i : cgdev) {
283 string api_unique_name = i.second->api_unique_name();
284
285 // If the module state for this API/function did not get created, there were
286 // no kernels using this API.
287 if (!state_needed[api_unique_name]) {
288 continue;
289 }
290 Expr state_ptr = make_state_var(api_unique_name);
291 Expr state_ptr_var = Variable::make(type_of<void *>(), api_unique_name);
292
293 debug(2) << "Generating init_kernels for " << api_unique_name << "\n";
294 vector<char> kernel_src = i.second->compile_to_src();
295 Expr kernel_src_buf = make_buffer_ptr(kernel_src, api_unique_name + "_gpu_source_kernels");
296
297 string init_kernels_name = "halide_" + api_unique_name + "_initialize_kernels";
298 vector<Expr> init_args = {state_ptr_var, kernel_src_buf, Expr((int)kernel_src.size())};
299 Stmt init_kernels = call_extern_and_assert(init_kernels_name, init_args);
300
301 string destructor_name = "halide_" + api_unique_name + "_finalize_kernels";
302 vector<Expr> finalize_args = {Expr(destructor_name), get_state_var(api_unique_name)};
303 Stmt register_destructor = Evaluate::make(
304 Call::make(Handle(), Call::register_destructor, finalize_args, Call::Intrinsic));
305
306 result = LetStmt::make(api_unique_name, state_ptr, Block::make({init_kernels, register_destructor, result}));
307 }
308 return result;
309 }
