12bool should_rebase(ForType type) {
13 switch (type) {
14 case ForType::Extern:
15 case ForType::GPUBlock:
16 case ForType::GPUThread:
17 case ForType::GPULane:
18 return false;
19 default:
20 return true;
21 }
22}
23
24class RebaseLoopsToZero : public IRMutator {
25 using IRMutator::visit;
26
27 Stmt visit(const For *op) override {
28 if (!should_rebase(op->for_type)) {
29 return IRMutator::visit(op);
30 }
31 Stmt body = mutate(op->body);
32 string name = op->name;
33 if (!is_const_zero(op->min)) {
34 // Renaming the loop (intentionally) invalidates any .loop_min/.loop_max lets.
35 name = op->name + ".rebased";
36 Expr loop_var = Variable::make(Int(32), name);
37 body = LetStmt::make(op->name, loop_var + op->min, body);
38 }
39 if (body.same_as(op->body)) {
40 return op;
41 } else {
42 return For::make(name, 0, op->extent, op->for_type, op->device_api, body);
43 }
44 }
45};
