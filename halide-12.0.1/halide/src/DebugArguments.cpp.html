12void debug_arguments(LoweredFunc *func, const Target &t) {
13 internal_assert(func);
14 vector<Stmt> stmts;
15 stmts.push_back(Evaluate::make(print("Entering Pipeline " + func->name)));
16 stmts.push_back(Evaluate::make(print("Target: " + t.to_string())));
17 for (const LoweredArgument &arg : func->args) {
18 std::ostringstream name;
19 Expr scalar_var = Variable::make(arg.type, arg.name);
20 Expr buffer_var = Variable::make(type_of<halide_buffer_t *>(), arg.name + ".buffer");
21 Expr value;
22 switch (arg.kind) {
23 case Argument::InputScalar:
24 name << " Input " << arg.type << " " << arg.name << ":";
25 value = scalar_var;
26 break;
27 case Argument::InputBuffer:
28 name << " Input Buffer " << arg.name << ":";
29 value = buffer_var;
30 break;
31 case Argument::OutputBuffer:
32 name << " Output Buffer " << arg.name << ":";
33 value = buffer_var;
34 break;
35 }
36 stmts.push_back(Evaluate::make(print(name.str(), value)));
37 }
38 stmts.push_back(func->body);
39 stmts.push_back(Evaluate::make(print("Exiting Pipeline " + func->name)));
40 func->body = Block::make(stmts);
41}
