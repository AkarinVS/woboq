35void Closure::found_buffer_ref(const string &name, Type type,
36 bool read, bool written, const Halide::Buffer<> &image) {
37 if (!ignore.contains(name)) {
38 debug(3) << "Adding buffer " << name << " to closure\n";
39 Buffer &ref = buffers[name];
40 ref.type = type.element_of(); // TODO: Validate type is the same as existing refs?
41 ref.read = ref.read || read;
42 ref.write = ref.write || written;
43
44 // If reading an image/buffer, compute the size.
45 if (image.defined()) {
46 ref.size = image.size_in_bytes();
47 ref.dimensions = image.dimensions();
48 }
49 } else {
50 debug(3) << "Not adding " << name << " to closure\n";
51 }
52}

67void Closure::visit(const Allocate *op) {
68 if (op->new_expr.defined()) {
69 op->new_expr.accept(this);
70 }
71 ScopedBinding<> p(ignore, op->name);
72 for (size_t i = 0; i < op->extents.size(); i++) {
73 op->extents[i].accept(this);
74 }
75 op->condition.accept(this);
76 op->body.accept(this);
77}

79void Closure::visit(const Variable *op) {
80 if (ignore.contains(op->name)) {
81 debug(3) << "Not adding " << op->name << " to closure\n";
82 } else {
83 debug(3) << "Adding " << op->name << " to closure\n";
84 vars[op->name] = op->type;
85 }

88void Closure::visit(const Atomic *op) {
89 if (!op->mutex_name.empty()) {
90 found_buffer_ref(op->mutex_name, type_of<void *>(), true, true, Halide::Buffer<>());
91 }
92 op->body.accept(this);
