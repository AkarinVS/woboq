11Expr bfloat16_to_float32(Expr e) {
12 if (e.type().is_bfloat()) {
13 e = reinterpret(e.type().with_code(Type::UInt), e);
14 }
15 e = cast(UInt(32, e.type().lanes()), e);
16 e = e << 16;
17 e = reinterpret(Float(32, e.type().lanes()), e);
18 e = strict_float(e);
19 return e;
20}

22Expr float32_to_bfloat16(Expr e) {
23 internal_assert(e.type().bits() == 32);
24 e = strict_float(e);
25 e = reinterpret(UInt(32, e.type().lanes()), e);
26 // We want to round ties to even, so before truncating either
27 // add 0x8000 (0.5) to odd numbers or 0x7fff (0.499999) to
28 // even numbers.
29 e += 0x7fff + ((e >> 16) & 1);
30 e = (e >> 16);
31 e = cast(UInt(16, e.type().lanes()), e);
32 e = reinterpret(BFloat(16, e.type().lanes()), e);
33 return e;
34}

36Expr float16_to_float32(Expr value) {
37 value = strict_float(value);
38 Type f32_t = Float(32, value.type().lanes());
39 Type u32_t = UInt(32, value.type().lanes());
40 Type u16_t = UInt(16, value.type().lanes());
41
42 Expr f16_bits = value;
43 if (!(value.type() == u16_t)) {
44 f16_bits = reinterpret(u16_t, f16_bits);
45 }
46
47 Expr magnitude = f16_bits & make_const(u16_t, 0x7fff);
48 Expr sign = f16_bits & make_const(u16_t, 0x8000);
49
50 // Denorms are linearly spaced, so we should just use an
51 // int->float cast and then scale down by reducing the
52 // exponent.
53 Expr denorm = reinterpret(u32_t, strict_float(cast(f32_t, magnitude))) - 0x0c000000;
54
55 Expr exponent_mantissa = cast(u32_t, magnitude) << 13;
56 exponent_mantissa = select(magnitude == 0, 0,
57 magnitude < 0x0400, denorm, // denorms
58 magnitude >= 0x7c00, exponent_mantissa | 0x7f800000, // Map infinity to infinity
59 exponent_mantissa + 0x38000000); // Fix the exponent bias otherwise
60
61 Expr f32 = strict_float(reinterpret(f32_t, (cast(u32_t, sign) << 16) | exponent_mantissa));
62 f32 = common_subexpression_elimination(f32);
63 return f32;
64}

66Expr float32_to_float16(Expr value) {
67 // We're about the sniff the bits of a float, so we should
68 // guard it with strict float to ensure we don't do things
69 // like assume it can't be denormal.
70 value = strict_float(value);
71
72 Type f32_t = Float(32, value.type().lanes());
73 Type f16_t = Float(16, value.type().lanes());
74 Type u32_t = UInt(32, value.type().lanes());
75 Type u16_t = UInt(16, value.type().lanes());
76
77 Expr bits = reinterpret(u32_t, value);
78
79 // Extract the sign bit
80 Expr sign = bits & make_const(u32_t, 0x80000000);
81 bits = bits ^ sign;
82
83 // Test the endpoints
84 Expr is_denorm = (bits < make_const(u32_t, 0x38800000));
85 Expr is_inf = (bits >= make_const(u32_t, 0x47800000));
86 Expr is_nan = (bits > make_const(u32_t, 0x7f800000));
87
88 // Denorms are linearly spaced, so we can handle them
89 // by scaling up the input as a float and using the
90 // existing int-conversion rounding instructions.
91 Expr denorm_bits = cast(u16_t, strict_float(round(strict_float(reinterpret(f32_t, bits + 0x0c000000)))));
92 Expr inf_bits = make_const(u16_t, 0x7c00);
93 Expr nan_bits = make_const(u16_t, 0x7fff);
94
95 // We want to round to nearest even, so we add either
96 // 0.5 if the integer part is odd, or 0.4999999 if the
97 // integer part is even, then truncate.
98 bits += (bits >> 13) & 1;
99 bits += 0xfff;
100 bits = bits >> 13;
101 // Rebias the exponent
102 bits -= 0x1c000;
103 // Truncate the top bits of the exponent
104 bits = bits & 0x7fff;
105 bits = select(is_denorm, denorm_bits,
106 is_inf, inf_bits,
107 is_nan, nan_bits,
108 cast(u16_t, bits));
109 // Recover the sign bit
110 bits = bits | cast(u16_t, sign >> 16);
111 return common_subexpression_elimination(reinterpret(f16_t, bits));
112}

143bool is_float16_transcendental(const Call *op) {
144 return transcendental_remapping.find(op->name) != transcendental_remapping.end();

147Expr lower_float16_transcendental_to_float32_equivalent(const Call *op) {
148 auto it = transcendental_remapping.find(op->name);
149 if (it != transcendental_remapping.end()) {
150 std::vector<Expr> new_args(op->args.size());
151 for (size_t i = 0; i < op->args.size(); i++) {
152 new_args[i] = float16_to_float32(op->args[i]);
153 }
154 Expr e = Call::make(Float(32, op->type.lanes()), it->second, new_args, op->call_type,
155 op->func, op->value_index, op->image, op->param);
156 return float32_to_float16(e);
157 } else {
158 internal_error << "Unknown float16 transcendental: " << Expr(op) << "\n";
159 return Expr();
160 }
161}

163Expr lower_float16_cast(const Cast *op) {
164 Type src = op->value.type();
165 Type dst = op->type;
166 Type f32 = Float(32, dst.lanes());
167 Expr val = op->value;
168
169 if (src.is_bfloat()) {
170 internal_assert(src.bits() == 16);
171 val = bfloat16_to_float32(val);
172 } else if (src.is_float() && src.bits() < 32) {
173 internal_assert(src.bits() == 16);
174 val = float16_to_float32(val);
175 }
176
177 if (dst.is_bfloat()) {
178 internal_assert(dst.bits() == 16);
179 val = float32_to_bfloat16(cast(f32, val));
180 } else if (dst.is_float() && dst.bits() < 32) {
181 internal_assert(dst.bits() == 16);
182 val = float32_to_float16(cast(f32, val));
183 }
184
185 return cast(dst, val);
186}
