17void check(const Expr &pred, std::vector<Expr> &expected) {
18 std::vector<Expr> result;
19 split_into_ands(pred, result);
20 bool is_equal = true;
21
22 if (result.size() != expected.size()) {
23 is_equal = false;
24 } else {
25 for (size_t i = 0; i < expected.size(); ++i) {
26 if (!equal(simplify(result[i]), simplify(expected[i]))) {
27 is_equal = false;
28 break;
29 }
30 }
31 }
32
33 if (!is_equal) {
34 std::cout << "Expect predicate " << pred << " to be split into:\n";
35 for (const auto &e : expected) {
36 std::cout << " " << e << "\n";
37 }
38 std::cout << "Got:\n";
39 for (const auto &e : result) {
40 std::cout << " " << e << "\n";
41 }
42 internal_error << "\n";
43 }
44}

48void split_predicate_test() {
49 Expr x = Var("x"), y = Var("y"), z = Var("z"), w = Var("w");
50
51 {
52 std::vector<Expr> expected;
53 expected.push_back(z < 10);
54 check(z < 10, expected);
55 }
56
57 {
58 std::vector<Expr> expected;
59 expected.push_back((x < y) || (x == 10));
60 check((x < y) || (x == 10), expected);
61 }
62
63 {
64 std::vector<Expr> expected;
65 expected.push_back(x < y);
66 expected.push_back(x == 10);
67 check((x < y) && (x == 10), expected);
68 }
69
70 {
71 std::vector<Expr> expected;
72 expected.push_back(x < y);
73 expected.push_back(x == 10);
74 expected.push_back(y == z);
75 check((x < y) && (x == 10) && (y == z), expected);
76 }
77
78 {
79 std::vector<Expr> expected;
80 expected.push_back((w == 1) || ((x == 10) && (y == z)));
81 check((w == 1) || ((x == 10) && (y == z)), expected);
82 }
83
84 {
85 std::vector<Expr> expected;
86 expected.push_back(x < y);
87 expected.push_back((w == 1) || ((x == 10) && (y == z)));
88 check((x < y) && ((w == 1) || ((x == 10) && (y == z))), expected);
89 }
90
91 std::cout << "Split predicate test passed" << std::endl;
92}

94struct ReductionDomainContents {
95 mutable RefCount ref_count;
96 std::vector<ReductionVariable> domain;
97 Expr predicate;
98 bool frozen = false;
99
100 ReductionDomainContents()
101 : predicate(const_true()) {
102 }
103
104 // Pass an IRVisitor through to all Exprs referenced in the ReductionDomainContents
105 void accept(IRVisitor *visitor) {
106 for (const ReductionVariable &rvar : domain) {
107 if (rvar.min.defined()) {
108 rvar.min.accept(visitor);
109 }
110 if (rvar.extent.defined()) {
111 rvar.extent.accept(visitor);
112 }
113 }
114 if (predicate.defined()) {
115 predicate.accept(visitor);
116 }
117 }
118
119 // Pass an IRMutator through to all Exprs referenced in the ReductionDomainContents
120 void mutate(IRMutator *mutator) {
121 for (ReductionVariable &rvar : domain) {
122 if (rvar.min.defined()) {
123 rvar.min = mutator->mutate(rvar.min);
124 }
125 if (rvar.extent.defined()) {
126 rvar.extent = mutator->mutate(rvar.extent);
127 }
128 }
129 if (predicate.defined()) {
130 predicate = mutator->mutate(predicate);
131 }
132 }
133};

165class DropSelfReferences : public IRMutator {
166 using IRMutator::visit;
167
168 Expr visit(const Variable *op) override {
169 if (op->reduction_domain.defined()) {
170 user_assert(op->reduction_domain.same_as(domain))
171 << "An RDom's predicate may only refer to its own RVars, "
172 << " not the RVars of some other RDom. "
173 << "Cannot set the predicate to : " << predicate << "\n";
174 return Variable::make(op->type, op->name);
175 } else {
176 return op;
177 }
178 }
179
180public:
181 Expr predicate;
182 const ReductionDomain &domain;
183 DropSelfReferences(Expr p, const ReductionDomain &d)
184 : predicate(std::move(p)), domain(d) {
185 }
186};
187} // namespace

189void ReductionDomain::set_predicate(const Expr &p) {
190 // The predicate can refer back to the RDom. We need to break
191 // those cycles to prevent a leak.
192 contents->predicate = DropSelfReferences(p, *this).mutate(p);
193}
