19/** A mutator which eagerly folds no-op stmts */
20class NoOpCollapsingMutator : public IRMutator {
21protected:
22 using IRMutator::visit;
23
24 Stmt visit(const LetStmt *op) override {
25 Stmt body = mutate(op->body);
26 if (is_no_op(body)) {
27 return body;
28 } else {
29 return LetStmt::make(op->name, op->value, body);
30 }
31 }
32
33 Stmt visit(const For *op) override {
34 Stmt body = mutate(op->body);
35 if (is_no_op(body)) {
36 return body;
37 } else {
38 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
39 }
40 }
41
42 Stmt visit(const Block *op) override {
43 Stmt first = mutate(op->first);
44 Stmt rest = mutate(op->rest);
45 if (is_no_op(first)) {
46 return rest;
47 } else if (is_no_op(rest)) {
48 return first;
49 } else {
50 return Block::make(first, rest);
51 }
52 }
53
54 Stmt visit(const Fork *op) override {
55 Stmt first = mutate(op->first);
56 Stmt rest = mutate(op->rest);
57 if (is_no_op(first)) {
58 return rest;
59 } else if (is_no_op(rest)) {
60 return first;
61 } else {
62 return Fork::make(first, rest);
63 }
64 }
65
66 Stmt visit(const Realize *op) override {
67 Stmt body = mutate(op->body);
68 if (is_no_op(body)) {
69 return body;
70 } else {
71 return Realize::make(op->name, op->types, op->memory_type,
72 op->bounds, op->condition, body);
73 }
74 }
75
76 Stmt visit(const Allocate *op) override {
77 Stmt body = mutate(op->body);
78 if (is_no_op(body)) {
79 return body;
80 } else {
81 return Allocate::make(op->name, op->type, op->memory_type,
82 op->extents, op->condition, body,
83 op->new_expr, op->free_function);
84 }
85 }
86
87 Stmt visit(const IfThenElse *op) override {
88 Stmt then_case = mutate(op->then_case);
89 Stmt else_case = mutate(op->else_case);
90 if (is_no_op(then_case) && is_no_op(else_case)) {
91 return then_case;
92 } else {
93 return IfThenElse::make(op->condition, then_case, else_case);
94 }
95 }
96
97 Stmt visit(const Atomic *op) override {
98 Stmt body = mutate(op->body);
99 if (is_no_op(body)) {
100 return body;
101 } else {
102 return Atomic::make(op->producer_name,
103 op->mutex_name,
104 std::move(body));
105 }
106 }

109class GenerateProducerBody : public NoOpCollapsingMutator {
110 const string &func;
111 vector<Expr> sema;
112
113 using NoOpCollapsingMutator::visit;
114
115 // Preserve produce nodes and add synchronization
116 Stmt visit(const ProducerConsumer *op) override {
117 if (op->name == func && op->is_producer) {
118 // Add post-synchronization
119 internal_assert(!sema.empty()) << "Duplicate produce node: " << op->name << "\n";
120 Stmt body = op->body;
121 while (!sema.empty()) {
122 Expr release = Call::make(Int(32), "halide_semaphore_release", {sema.back(), 1}, Call::Extern);
123 body = Block::make(body, Evaluate::make(release));
124 sema.pop_back();
125 }
126 return ProducerConsumer::make_produce(op->name, body);
127 } else {
128 Stmt body = mutate(op->body);
129 if (is_no_op(body) || op->is_producer) {
130 return body;
131 } else {
132 return ProducerConsumer::make(op->name, op->is_producer, body);
133 }
134 }
135 }
136
137 // Other stmt leaves get replaced with no-ops
138 Stmt visit(const Evaluate *) override {
139 return Evaluate::make(0);
140 }
141
142 Stmt visit(const Provide *) override {
143 return Evaluate::make(0);
144 }
145
146 Stmt visit(const Store *op) override {
147 if (starts_with(op->name, func + ".folding_semaphore.") && ends_with(op->name, ".head")) {
148 // This is a counter associated with the producer side of a storage-folding semaphore. Keep it.
149 return op;
150 } else {
151 return Evaluate::make(0);
152 }
153 }
154
155 Stmt visit(const AssertStmt *) override {
156 return Evaluate::make(0);
157 }
158
159 Stmt visit(const Prefetch *) override {
160 return Evaluate::make(0);
161 }
162
163 Stmt visit(const Acquire *op) override {
164 Stmt body = mutate(op->body);
165 const Variable *var = op->semaphore.as<Variable>();
166 internal_assert(var);
167 if (is_no_op(body)) {
168 return body;
169 } else if (starts_with(var->name, func + ".folding_semaphore.")) {
170 // This is a storage-folding semaphore for the func we're producing. Keep it.
171 return Acquire::make(op->semaphore, op->count, body);
172 } else {
173 // This semaphore will end up on both sides of the fork,
174 // so we'd better duplicate it.
175 vector<string> &clones = cloned_acquires[var->name];
176 clones.push_back(var->name + unique_name('_'));
177 return Acquire::make(Variable::make(type_of<halide_semaphore_t *>(), clones.back()), op->count, body);
178 }
179 }
180
181 Stmt visit(const Atomic *op) override {
182 return Evaluate::make(0);
183 }
184
185 Expr visit(const Call *op) override {
186 if (op->name == "halide_semaphore_init") {
187 internal_assert(op->args.size() == 2);
188 const Variable *var = op->args[0].as<Variable>();
189 internal_assert(var);
190 inner_semaphores.insert(var->name);
191 }
192 return op;
193 }
194
195 map<string, vector<string>> &cloned_acquires;
196 set<string> inner_semaphores;
197
198public:
199 GenerateProducerBody(const string &f, const vector<Expr> &s, map<string, vector<string>> &a)
200 : func(f), sema(s), cloned_acquires(a) {
201 }
202};

204class GenerateConsumerBody : public NoOpCollapsingMutator {
205 const string &func;
206 vector<Expr> sema;
207
208 using NoOpCollapsingMutator::visit;
209
210 Stmt visit(const ProducerConsumer *op) override {
211 if (op->name == func) {
212 if (op->is_producer) {
213 // Remove the work entirely
214 return Evaluate::make(0);
215 } else {
216 // Synchronize on the work done by the producer before beginning consumption
217 Expr acquire_sema = sema.back();
218 sema.pop_back();
219 return Acquire::make(acquire_sema, 1, op);
220 }
221 } else {
222 return NoOpCollapsingMutator::visit(op);
223 }
224 }
225
226 Stmt visit(const Allocate *op) override {
227 // Don't want to keep the producer's storage-folding tracker - it's dead code on the consumer side
228 if (starts_with(op->name, func + ".folding_semaphore.") && ends_with(op->name, ".head")) {
229 return mutate(op->body);
230 } else {
231 return NoOpCollapsingMutator::visit(op);
232 }
233 }
234
235 Stmt visit(const Store *op) override {
236 if (starts_with(op->name, func + ".folding_semaphore.") && ends_with(op->name, ".head")) {
237 return Evaluate::make(0);
238 } else {
239 return NoOpCollapsingMutator::visit(op);
240 }
241 }
242
243 Stmt visit(const Acquire *op) override {
244 // Don't want to duplicate any semaphore acquires.
245 // Ones from folding should go to the producer side.
246 const Variable *var = op->semaphore.as<Variable>();
247 internal_assert(var);
248 if (starts_with(var->name, func + ".folding_semaphore.")) {
249 return mutate(op->body);
250 } else {
251 return NoOpCollapsingMutator::visit(op);
252 }
253 }
254
255public:
256 GenerateConsumerBody(const string &f, const vector<Expr> &s)
257 : func(f), sema(s) {
258 }

261class CloneAcquire : public IRMutator {
262 using IRMutator::visit;
263
264 const string &old_name;
265 Expr new_var;
266
267 Stmt visit(const Evaluate *op) override {
268 const Call *call = op->value.as<Call>();
269 const Variable *var = ((call && !call->args.empty()) ? call->args[0].as<Variable>() : nullptr);
270 if (var && var->name == old_name &&
271 (call->name == "halide_semaphore_release" ||
272 call->name == "halide_semaphore_init")) {
273 vector<Expr> args = call->args;
274 args[0] = new_var;
275 Stmt new_stmt =
276 Evaluate::make(Call::make(call->type, call->name, args, call->call_type));
277 return Block::make(op, new_stmt);
278 } else {
279 return op;
280 }
281 }
282
283public:
284 CloneAcquire(const string &o, const string &new_name)
285 : old_name(o) {
286 new_var = Variable::make(type_of<halide_semaphore_t *>(), new_name);
287 }

309class ForkAsyncProducers : public IRMutator {
310 using IRMutator::visit;
311
312 const map<string, Function> &env;
313
314 map<string, vector<string>> cloned_acquires;
315
316 Stmt visit(const Realize *op) override {
317 auto it = env.find(op->name);
318 internal_assert(it != env.end());
319 Function f = it->second;
320 if (f.schedule().async()) {
321 Stmt body = op->body;
322
323 // Make two copies of the body, one which only does the
324 // producer, and one which only does the consumer. Inject
325 // synchronization to preserve dependencies. Put them in a
326 // task-parallel block.
327
328 // Make a semaphore per consume node
329 CountConsumeNodes consumes(op->name);
330 body.accept(&consumes);
331
332 vector<string> sema_names;
333 vector<Expr> sema_vars;
334 for (int i = 0; i < consumes.count; i++) {
335 sema_names.push_back(op->name + ".semaphore_" + std::to_string(i));
336 sema_vars.push_back(Variable::make(Handle(), sema_names.back()));
337 }
338
339 Stmt producer = GenerateProducerBody(op->name, sema_vars, cloned_acquires).mutate(body);
340 Stmt consumer = GenerateConsumerBody(op->name, sema_vars).mutate(body);
341
342 // Recurse on both sides
343 producer = mutate(producer);
344 consumer = mutate(consumer);
345
346 // Run them concurrently
347 body = Fork::make(producer, consumer);
348
349 for (const string &sema_name : sema_names) {
350 // Make a semaphore on the stack
351 Expr sema_space = Call::make(type_of<halide_semaphore_t *>(), "halide_make_semaphore",
352 {0}, Call::Extern);
353
354 // If there's a nested async producer, we may have
355 // recursively cloned this semaphore inside the mutation
356 // of the producer and consumer.
357 const vector<string> &clones = cloned_acquires[sema_name];
358 for (const auto &i : clones) {
359 body = CloneAcquire(sema_name, i).mutate(body);
360 body = LetStmt::make(i, sema_space, body);
361 }
362
363 body = LetStmt::make(sema_name, sema_space, body);
364 }
365
366 return Realize::make(op->name, op->types, op->memory_type,
367 op->bounds, op->condition, body);
368 } else {
369 return IRMutator::visit(op);
370 }
371 }
372
373public:
374 ForkAsyncProducers(const map<string, Function> &e)
375 : env(e) {
376 }

379// Lowers semaphore initialization from a call to
380// "halide_make_semaphore" to an alloca followed by a call into the
381// runtime to initialize. If something crashes before releasing a
382// semaphore, the task system is responsible for propagating the
383// failure to all branches of the fork. This depends on all semaphore
384// acquires happening as part of the halide_do_parallel_tasks logic,
385// not via explicit code in the closure. The current design for this
386// does not propagate failures downward to subtasks of a failed
387// fork. It assumes these will be able to reach completion in spite of
388// the failure, which remains to be proven. (There is a test for the
389// simple failure case, error_async_require_fail. One has not been
390// written for the complex nested case yet.)
391class InitializeSemaphores : public IRMutator {
392 using IRMutator::visit;
393
394 const Type sema_type = type_of<halide_semaphore_t *>();
395
396 Stmt visit(const LetStmt *op) override {
397 vector<const LetStmt *> frames;
398
399 // Find first op that is of sema_type
400 while (op && op->value.type() != sema_type) {
401 frames.push_back(op);
402 op = op->body.as<LetStmt>();
403 }
404
405 Stmt body;
406 if (op) {
407 body = mutate(op->body);
408 // Peel off any enclosing let expressions from the value
409 vector<pair<string, Expr>> lets;
410 Expr value = op->value;
411 while (const Let *l = value.as<Let>()) {
412 lets.emplace_back(l->name, l->value);
413 value = l->body;
414 }
415 const Call *call = value.as<Call>();
416 if (call && call->name == "halide_make_semaphore") {
417 internal_assert(call->args.size() == 1);
418
419 Expr sema_var = Variable::make(sema_type, op->name);
420 Expr sema_init = Call::make(Int(32), "halide_semaphore_init",
421 {sema_var, call->args[0]}, Call::Extern);
422 Expr sema_allocate = Call::make(sema_type, Call::alloca,
423 {(int)sizeof(halide_semaphore_t)}, Call::Intrinsic);
424 body = Block::make(Evaluate::make(sema_init), std::move(body));
425 body = LetStmt::make(op->name, std::move(sema_allocate), std::move(body));
426
427 // Re-wrap any other lets
428 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
429 body = LetStmt::make(it->first, it->second, std::move(body));
430 }
431 }
432 } else {
433 body = mutate(frames.back()->body);
434 }
435
436 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
437 Expr value = mutate((*it)->value);
438 if (value.same_as((*it)->value) && body.same_as((*it)->body)) {
439 body = *it;
440 } else {
441 body = LetStmt::make((*it)->name, std::move(value), std::move(body));
442 }
443 }
444 return body;
445 }
446
447 Expr visit(const Call *op) override {
448 internal_assert(op->name != "halide_make_semaphore")
449 << "Call to halide_make_semaphore in unexpected place\n";
450 return op;
451 }

454// Tighten the scope of consume nodes as much as possible to avoid needless synchronization.
455class TightenProducerConsumerNodes : public IRMutator {
456 using IRMutator::visit;
457
458 Stmt make_producer_consumer(const string &name, bool is_producer, Stmt body, const Scope<int> &scope) {
459 if (const LetStmt *let = body.as<LetStmt>()) {
460 if (expr_uses_vars(let->value, scope)) {
461 return ProducerConsumer::make(name, is_producer, body);
462 } else {
463 return LetStmt::make(let->name, let->value, make_producer_consumer(name, is_producer, let->body, scope));
464 }
465 } else if (const Block *block = body.as<Block>()) {
466 // Check which sides it's used on
467 bool first = stmt_uses_vars(block->first, scope);
468 bool rest = stmt_uses_vars(block->rest, scope);
469 if (is_producer) {
470 return ProducerConsumer::make(name, is_producer, body);
471 } else if (first && rest) {
472 return Block::make(make_producer_consumer(name, is_producer, block->first, scope),
473 make_producer_consumer(name, is_producer, block->rest, scope));
474 } else if (first) {
475 return Block::make(make_producer_consumer(name, is_producer, block->first, scope), block->rest);
476 } else if (rest) {
477 return Block::make(block->first, make_producer_consumer(name, is_producer, block->rest, scope));
478 } else {
479 // Used on neither side?!
480 return body;
481 }
482 } else if (const ProducerConsumer *pc = body.as<ProducerConsumer>()) {
483 return ProducerConsumer::make(pc->name, pc->is_producer, make_producer_consumer(name, is_producer, pc->body, scope));
484 } else if (const Realize *r = body.as<Realize>()) {
485 return Realize::make(r->name, r->types, r->memory_type,
486 r->bounds, r->condition,
487 make_producer_consumer(name, is_producer, r->body, scope));
488 } else {
489 return ProducerConsumer::make(name, is_producer, body);
490 }
491 }
492
493 Stmt visit(const ProducerConsumer *op) override {
494 Stmt body = mutate(op->body);
495 Scope<int> scope;
496 scope.push(op->name, 0);
497 Function f = env.find(op->name)->second;
498 if (f.outputs() == 1) {
499 scope.push(op->name + ".buffer", 0);
500 } else {
501 for (int i = 0; i < f.outputs(); i++) {
502 scope.push(op->name + "." + std::to_string(i) + ".buffer", 0);
503 }
504 }
505 return make_producer_consumer(op->name, op->is_producer, body, scope);
506 }
507
508 const map<string, Function> &env;
509
510public:
511 TightenProducerConsumerNodes(const map<string, Function> &e)
512 : env(e) {
513 }

516// Broaden the scope of acquire nodes to pack trailing work into the
517// same task and to potentially reduce the nesting depth of tasks.
518class ExpandAcquireNodes : public IRMutator {
519 using IRMutator::visit;
520
521 Stmt visit(const Block *op) override {
522 // Do an entire sequence of blocks in a single visit method to conserve stack space.
523 vector<Stmt> stmts;
524 Stmt result;
525 do {
526 stmts.push_back(mutate(op->first));
527 result = op->rest;
528 } while ((op = result.as<Block>()));
529
530 result = mutate(result);
531
532 vector<pair<Expr, Expr>> semaphores;
533 for (auto it = stmts.rbegin(); it != stmts.rend(); it++) {
534 Stmt s = *it;
535 while (const Acquire *a = s.as<Acquire>()) {
536 semaphores.emplace_back(a->semaphore, a->count);
537 s = a->body;
538 }
539 result = Block::make(s, result);
540 while (!semaphores.empty()) {
541 result = Acquire::make(semaphores.back().first, semaphores.back().second, result);
542 semaphores.pop_back();
543 }
544 }
545
546 return result;
547 }
548
549 Stmt visit(const Realize *op) override {
550 Stmt body = mutate(op->body);
551 if (const Acquire *a = body.as<Acquire>()) {
552 // Don't do the allocation until we have the
553 // semaphore. Reduces peak memory use.
554 return Acquire::make(a->semaphore, a->count,
555 mutate(Realize::make(op->name, op->types, op->memory_type,
556 op->bounds, op->condition, a->body)));
557 } else {
558 return Realize::make(op->name, op->types, op->memory_type,
559 op->bounds, op->condition, body);
560 }
561 }
562
563 Stmt visit(const LetStmt *op) override {
564 Stmt body = mutate(op->body);
565 const Acquire *a = body.as<Acquire>();
566 if (a &&
567 !expr_uses_var(a->semaphore, op->name) &&
568 !expr_uses_var(a->count, op->name)) {
569 return Acquire::make(a->semaphore, a->count,
570 LetStmt::make(op->name, op->value, a->body));
571 } else {
572 return LetStmt::make(op->name, op->value, body);
573 }
574 }
575
576 Stmt visit(const ProducerConsumer *op) override {
577 Stmt body = mutate(op->body);
578 if (const Acquire *a = body.as<Acquire>()) {
579 return Acquire::make(a->semaphore, a->count,
580 mutate(ProducerConsumer::make(op->name, op->is_producer, a->body)));
581 } else {
582 return ProducerConsumer::make(op->name, op->is_producer, body);
583 }
584 }

587class TightenForkNodes : public IRMutator {
588 using IRMutator::visit;
589
590 Stmt make_fork(const Stmt &first, const Stmt &rest) {
591 const LetStmt *lf = first.as<LetStmt>();
592 const LetStmt *lr = rest.as<LetStmt>();
593 const Realize *rf = first.as<Realize>();
594 const Realize *rr = rest.as<Realize>();
595 if (lf && lr &&
596 lf->name == lr->name &&
597 equal(lf->value, lr->value)) {
598 return LetStmt::make(lf->name, lf->value, make_fork(lf->body, lr->body));
599 } else if (lf && !stmt_uses_var(rest, lf->name)) {
600 return LetStmt::make(lf->name, lf->value, make_fork(lf->body, rest));
601 } else if (lr && !stmt_uses_var(first, lr->name)) {
602 return LetStmt::make(lr->name, lr->value, make_fork(first, lr->body));
603 } else if (rf && !stmt_uses_var(rest, rf->name)) {
604 return Realize::make(rf->name, rf->types, rf->memory_type,
605 rf->bounds, rf->condition, make_fork(rf->body, rest));
606 } else if (rr && !stmt_uses_var(first, rr->name)) {
607 return Realize::make(rr->name, rr->types, rr->memory_type,
608 rr->bounds, rr->condition, make_fork(first, rr->body));
609 } else {
610 return Fork::make(first, rest);
611 }
612 }
613
614 Stmt visit(const Fork *op) override {
615 Stmt first, rest;
616 {
617 ScopedValue<bool> old_in_fork(in_fork, true);
618 first = mutate(op->first);
619 rest = mutate(op->rest);
620 }
621
622 if (is_no_op(first)) {
623 return rest;
624 } else if (is_no_op(rest)) {
625 return first;
626 } else {
627 return make_fork(first, rest);
628 }
629 }
630
631 // This is also a good time to nuke any dangling allocations and lets in the fork children.
632 Stmt visit(const Realize *op) override {
633 Stmt body = mutate(op->body);
634 if (in_fork && !stmt_uses_var(body, op->name) && !stmt_uses_var(body, op->name + ".buffer")) {
635 return body;
636 } else {
637 return Realize::make(op->name, op->types, op->memory_type,
638 op->bounds, op->condition, body);
639 }
640 }
641
642 Stmt visit(const LetStmt *op) override {
643 Stmt body = mutate(op->body);
644 if (in_fork && !stmt_uses_var(body, op->name)) {
645 return body;
646 } else {
647 return LetStmt::make(op->name, op->value, body);
648 }
649 }
650
651 bool in_fork = false;
652};

658Stmt fork_async_producers(Stmt s, const map<string, Function> &env) {
659 s = TightenProducerConsumerNodes(env).mutate(s);
660 s = ForkAsyncProducers(env).mutate(s);
661 s = ExpandAcquireNodes().mutate(s);
662 s = TightenForkNodes().mutate(s);
663 s = InitializeSemaphores().mutate(s);
664 return s;
