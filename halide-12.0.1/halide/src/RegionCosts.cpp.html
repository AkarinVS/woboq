21// Visitor for keeping track of all input images accessed and their types.
22class FindImageInputs : public IRVisitor {
23 using IRVisitor::visit;
24 set<string> seen_image_param;
25
26 void visit(const Call *call) override {
27 if (call->call_type == Call::Image) {
28 input_type[call->name] = call->type;
29
30 // Call to an ImageParam
31 if (call->param.defined() && (seen_image_param.count(call->name) == 0)) {
32 for (int i = 0; i < call->param.dimensions(); ++i) {
33 const Expr &min = call->param.min_constraint_estimate(i);
34 const Expr &extent = call->param.extent_constraint_estimate(i);
35
36 user_assert(min.defined())
37 << "AutoSchedule: Estimate of the min value of ImageParam \""
38 << call->name << "\" in dimension " << i << " is not specified.\n";
39 user_assert(extent.defined())
40 << "AutoSchedule: Estimate of the extent value of ImageParam \""
41 << call->name << "\" in dimension " << i << " is not specified.\n";
42
43 string min_var = call->param.name() + ".min." + std::to_string(i);
44 string extent_var = call->param.name() + ".extent." + std::to_string(i);
45
46 input_estimates.emplace(min_var, Interval(min, min));
47 input_estimates.emplace(extent_var, Interval(extent, extent));
48 seen_image_param.insert(call->name);
49 }
50 }
51 }
52 for (size_t i = 0; i < call->args.size(); i++) {
53 call->args[i].accept(this);
54 }
55 }
56
57public:
58 map<string, Type> input_type;
59 map<string, Interval> input_estimates;
60};

153 void visit(const Call *call) override {
154 if (call->is_intrinsic(Call::if_then_else)) {
155 internal_assert(call->args.size() == 3);
156
157 int64_t current_arith = arith, current_memory = memory;
158 arith = 0, memory = 0;
159 call->args[2].accept(this);
160
161 // Check if this if_then_else is because of tracing or print_when.
162 // If it is, we should only take into account the cost of computing
163 // the false expr since the true expr is debugging/tracing code.
164 const Call *true_value_call = call->args[1].as<Call>();
165 if (!true_value_call || !true_value_call->is_intrinsic(Call::return_second)) {
166 int64_t false_cost_arith = arith;
167 int64_t false_cost_memory = memory;
168
169 // For if_then_else intrinsic, the cost is the max of true and
170 // false branch costs plus the predicate cost.
171 arith = 0, memory = 0;
172 call->args[0].accept(this);
173 int64_t pred_cost_arith = arith;
174 int64_t pred_cost_memory = memory;
175
176 arith = 0, memory = 0;
177 call->args[1].accept(this);
178 int64_t true_cost_arith = arith;
179 int64_t true_cost_memory = memory;
180
181 arith = pred_cost_arith + std::max(true_cost_arith, false_cost_arith);
182 memory = pred_cost_memory + std::max(true_cost_memory, false_cost_memory);
183 }
184 arith += current_arith;
185 memory += current_memory;
186 return;
187 } else if (call->is_intrinsic(Call::return_second)) {
188 // For return_second, since the first expr would usually either be a
189 // print_when or tracing, we should only take into account the cost
190 // of computing the second expr.
191 internal_assert(call->args.size() == 2);
192 call->args[1].accept(this);
193 return;
194 }
195
196 if (call->call_type == Call::Halide || call->call_type == Call::Image) {
197 // Each call also counts as an op since it results in a load instruction.
198 arith += 1;
199 memory += call->type.bytes();
200 detailed_byte_loads[call->name] += (int64_t)call->type.bytes();
201 } else if (call->is_extern()) {
202 // TODO: Suffix based matching is kind of sketchy; but going ahead with
203 // it for now. Also not all the PureExtern's are accounted for yet.
204 if (ends_with(call->name, "_f64")) {
205 arith += 20;
206 } else if (ends_with(call->name, "_f32")) {
207 arith += 10;
208 } else if (ends_with(call->name, "_f16")) {
209 arith += 5;
210 } else {
211 // There is no visibility into an extern stage so there is no
212 // way to know the cost of the call statically. Modeling the
213 // cost of an extern stage requires profiling or user annotation.
214 user_warning << "Unknown extern call " << call->name << "\n";
215 }
216 } else if (call->is_intrinsic()) {
217 // TODO: Improve the cost model. In some architectures (e.g. ARM or
218 // NEON), count_leading_zeros should be as cheap as bitwise ops.
219 // div_round_to_zero and mod_round_to_zero can also get fairly expensive.
220 if (call->is_intrinsic(Call::reinterpret) || call->is_intrinsic(Call::bitwise_and) ||
221 call->is_intrinsic(Call::bitwise_not) || call->is_intrinsic(Call::bitwise_xor) ||
222 call->is_intrinsic(Call::bitwise_or) || call->is_intrinsic(Call::shift_left) ||
223 call->is_intrinsic(Call::shift_right) || call->is_intrinsic(Call::div_round_to_zero) ||
224 call->is_intrinsic(Call::mod_round_to_zero) || call->is_intrinsic(Call::undef) ||
225 call->is_intrinsic(Call::mux)) {
226 arith += 1;
227 } else if (call->is_intrinsic(Call::abs) || call->is_intrinsic(Call::absd) ||
228 call->is_intrinsic(Call::lerp) || call->is_intrinsic(Call::random) ||
229 call->is_intrinsic(Call::count_leading_zeros) ||
230 call->is_intrinsic(Call::count_trailing_zeros)) {
231 arith += 5;
232 } else if (Call::as_tag(call)) {
233 // Tags do not result in actual operations.
234 } else {
235 // For other intrinsics, use 1 for the arithmetic cost.
236 arith += 1;
237 user_warning << "Unhandled intrinsic call " << call->name << "\n";
238 }
239 }
240
241 for (size_t i = 0; i < call->args.size(); i++) {
242 call->args[i].accept(this);
243 }
244 }

395RegionCosts::RegionCosts(const map<string, Function> &_env,
396 const vector<string> &_order)
397 : env(_env), order(_order) {
398 for (const auto &kv : env) {
399 // Pre-compute the function costs without any inlining.
400 func_cost[kv.first] = get_func_cost(kv.second);
401
402 // Get the types of all the image inputs to the pipeline, including
403 // their estimated min/extent values if applicable (i.e. if they are
404 // ImageParam).
405 FindImageInputs find;
406 kv.second.accept(&find);
407 for (const auto &in : find.input_type) {
408 inputs[in.first] = in.second;
409 }
410 for (const auto &iter : find.input_estimates) {
411 input_estimates.push(iter.first, iter.second);
412 }
413 }
414}

416Cost RegionCosts::stage_region_cost(const string &func, int stage, const DimBounds &bounds,
417 const set<string> &inlines) {
418 Function curr_f = get_element(env, func);
419
420 Box stage_region;
421
422 const vector<Dim> &dims = get_stage_dims(curr_f, stage);
423 for (int d = 0; d < (int)dims.size() - 1; d++) {
424 stage_region.push_back(get_element(bounds, dims[d].var));
425 }
426
427 Expr size = box_size(stage_region);
428 if (!size.defined()) {
429 // Size could not be determined; therefore, it is not possible to
430 // determine the arithmetic and memory costs.
431 return Cost();
432 }
433
434 // If there is nothing to be inlined, use the pre-computed function cost.
435 Cost cost = inlines.empty() ? get_element(func_cost, func)[stage] : get_func_stage_cost(curr_f, stage, inlines);
436 if (!cost.defined()) {
437 return Cost();
438 }
439 return Cost(simplify(size * cost.arith), simplify(size * cost.memory));
440}

457Cost RegionCosts::region_cost(const string &func, const Box &region, const set<string> &inlines) {
458 Function curr_f = get_element(env, func);
459 Cost region_cost(0, 0);
460
461 int num_stages = curr_f.updates().size() + 1;
462 for (int s = 0; s < num_stages; s++) {
463 Cost stage_cost = stage_region_cost(func, s, region, inlines);
464 if (!stage_cost.defined()) {
465 return Cost();
466 } else {
467 region_cost.arith += stage_cost.arith;
468 region_cost.memory += stage_cost.memory;
469 }
470 }
471
472 internal_assert(region_cost.defined());
473 region_cost.simplify();
474 return region_cost;
475}

477Cost RegionCosts::region_cost(const map<string, Box> &regions, const set<string> &inlines) {
478 Cost total_cost(0, 0);
479 for (const auto &f : regions) {
480 // The cost for pure inlined functions will be accounted in the
481 // consumer of the inlined function so they should be skipped.
482 if (inlines.find(f.first) != inlines.end()) {
483 internal_assert(get_element(env, f.first).is_pure());
484 continue;
485 }
486
487 Cost cost = region_cost(f.first, f.second, inlines);
488 if (!cost.defined()) {
489 return Cost();
490 } else {
491 total_cost.arith += cost.arith;
492 total_cost.memory += cost.memory;
493 }
494 }
495
496 internal_assert(total_cost.defined());
497 total_cost.simplify();
498 return total_cost;
499}

501map<string, Expr>
502RegionCosts::stage_detailed_load_costs(const string &func, int stage,
503 const set<string> &inlines) {
504 map<string, Expr> load_costs;
505 Function curr_f = get_element(env, func);
506
507 if (curr_f.has_extern_definition()) {
508 // TODO(psuriana): We need a better cost for extern function
509 //load_costs.emplace(func, Int(64).max());
510 load_costs.emplace(func, Expr());
511 } else {
512 Definition def = get_stage_definition(curr_f, stage);
513 for (const auto &e : def.values()) {
514 Expr inlined_expr = perform_inline(e, env, inlines, order);
515 inlined_expr = simplify(inlined_expr);
516
517 map<string, Expr> expr_load_costs = compute_expr_detailed_byte_loads(inlined_expr);
518 combine_load_costs(load_costs, expr_load_costs);
519
520 auto iter = load_costs.find(func);
521 if (iter != load_costs.end()) {
522 internal_assert(iter->second.defined());
523 iter->second = simplify(iter->second + e.type().bytes());
524 } else {
525 load_costs.emplace(func, make_const(Int(64), e.type().bytes()));
526 }
527 }
528 }
529
530 return load_costs;
531}

533map<string, Expr>
534RegionCosts::stage_detailed_load_costs(const string &func, int stage,
535 DimBounds &bounds,
536 const set<string> &inlines) {
537 Function curr_f = get_element(env, func);
538
539 Box stage_region;
540
541 const vector<Dim> &dims = get_stage_dims(curr_f, stage);
542 for (int d = 0; d < (int)dims.size() - 1; d++) {
543 stage_region.push_back(get_element(bounds, dims[d].var));
544 }
545
546 map<string, Expr> load_costs = stage_detailed_load_costs(func, stage, inlines);
547
548 Expr size = box_size(stage_region);
549 for (auto &kv : load_costs) {
550 if (!kv.second.defined()) {
551 continue;
552 } else if (!size.defined()) {
553 kv.second = Expr();
554 } else {
555 kv.second = simplify(kv.second * size);
556 }
557 }
558
559 return load_costs;
560}

562map<string, Expr>
563RegionCosts::detailed_load_costs(const string &func, const Box &region,
564 const set<string> &inlines) {
565 Function curr_f = get_element(env, func);
566 map<string, Expr> load_costs;
567
568 int num_stages = curr_f.updates().size() + 1;
569
570 DimBounds pure_bounds;
571 const vector<string> &args = curr_f.args();
572 internal_assert(args.size() == region.size());
573 for (size_t d = 0; d < args.size(); d++) {
574 pure_bounds.emplace(args[d], region[d]);
575 }
576
577 vector<DimBounds> stage_bounds = get_stage_bounds(curr_f, pure_bounds);
578
579 for (int s = 0; s < num_stages; s++) {
580 map<string, Expr> stage_load_costs = stage_detailed_load_costs(func, s, inlines);
581
582 Box stage_region;
583
584 const vector<Dim> &dims = get_stage_dims(curr_f, s);
585 for (int d = 0; d < (int)dims.size() - 1; d++) {
586 stage_region.push_back(get_element(stage_bounds[s], dims[d].var));
587 }
588
589 Expr size = box_size(stage_region);
590 for (auto &kv : stage_load_costs) {
591 if (!kv.second.defined()) {
592 continue;
593 } else if (!size.defined()) {
594 kv.second = Expr();
595 } else {
596 kv.second = simplify(kv.second * size);
597 }
598 }
599
600 combine_load_costs(load_costs, stage_load_costs);
601 }
602
603 return load_costs;
604}

606map<string, Expr>
607RegionCosts::detailed_load_costs(const map<string, Box> &regions,
608 const set<string> &inlines) {
609 map<string, Expr> load_costs;
610 for (const auto &r : regions) {
611 // The cost for pure inlined functions will be accounted in the
612 // consumer of the inlined function so they should be skipped.
613 if (inlines.find(r.first) != inlines.end()) {
614 internal_assert(get_element(env, r.first).is_pure());
615 continue;
616 }
617
618 map<string, Expr> partial_load_costs = detailed_load_costs(r.first, r.second, inlines);
619 combine_load_costs(load_costs, partial_load_costs);
620 }
621
622 return load_costs;

625Cost RegionCosts::get_func_stage_cost(const Function &f, int stage,
626 const set<string> &inlines) const {
627 if (f.has_extern_definition()) {
628 return Cost();
629 }
630
631 Definition def = get_stage_definition(f, stage);
632
633 Cost cost(0, 0);
634
635 for (const auto &e : def.values()) {
636 Expr inlined_expr = perform_inline(e, env, inlines, order);
637 inlined_expr = simplify(inlined_expr);
638
639 Cost expr_cost = compute_expr_cost(inlined_expr);
640 internal_assert(expr_cost.defined());
641 cost.arith += expr_cost.arith;
642 cost.memory += expr_cost.memory;
643
644 // Accounting for the store
645 cost.memory += e.type().bytes();
646 cost.arith += 1;
647 }
648
649 if (!f.is_pure()) {
650 for (const auto &arg : def.args()) {
651 Expr inlined_arg = perform_inline(arg, env, inlines, order);
652 inlined_arg = simplify(inlined_arg);
653
654 Cost expr_cost = compute_expr_cost(inlined_arg);
655 internal_assert(expr_cost.defined());
656 cost.arith += expr_cost.arith;
657 cost.memory += expr_cost.memory;
658 }
659 }
660
661 cost.simplify();
662 return cost;
663}

665vector<Cost> RegionCosts::get_func_cost(const Function &f, const set<string> &inlines) {
666 if (f.has_extern_definition()) {
667 return {Cost()};

669

671 size_t num_stages = f.updates().size() + 1;
672 for (size_t s = 0; s < num_stages; s++) {
673 func_costs.push_back(get_func_stage_cost(f, s, inlines));
674 }
675 return func_costs;

678Expr RegionCosts::region_size(const string &func, const Box &region) {
679 const Function &f = get_element(env, func);
680 Expr size = box_size(region);
681 if (!size.defined()) {
682 return Expr();

689Expr RegionCosts::region_footprint(const map<string, Box> &regions,
690 const set<string> &inlined) {
691 map<string, int> num_consumers;
692 for (const auto &f : regions) {
693 num_consumers[f.first] = 0;
694 }
695 for (const auto &f : regions) {
696 map<string, Function> prods = find_direct_calls(get_element(env, f.first));
697 for (const auto &p : prods) {
698 auto iter = num_consumers.find(p.first);
699 if (iter != num_consumers.end()) {
700 iter->second += 1;
701 }
702 }
703 }
704
705 vector<Function> outs;
706 for (const auto &f : num_consumers) {
707 if (f.second == 0) {
708 outs.push_back(get_element(env, f.first));
709 }
710 }
711
712 vector<string> top_order = topological_order(outs, env);
713
714 Expr working_set_size = make_zero(Int(64));
715 Expr curr_size = make_zero(Int(64));
716
717 map<string, Expr> func_sizes;
718
719 for (const auto &f : regions) {
720 // Inlined functions do not have allocations
721 bool is_inlined = inlined.find(f.first) != inlined.end();
722 Expr size = is_inlined ? make_zero(Int(64)) : region_size(f.first, f.second);
723 if (!size.defined()) {
724 return Expr();
725 } else {
726 func_sizes.emplace(f.first, size);
727 }
728 }
729
730 for (const auto &f : top_order) {
731 if (regions.find(f) != regions.end()) {
732 curr_size += get_element(func_sizes, f);
733 }
734 working_set_size = max(curr_size, working_set_size);
735 map<string, Function> prods = find_direct_calls(get_element(env, f));
736 for (const auto &p : prods) {
737 auto iter = num_consumers.find(p.first);
738 if (iter != num_consumers.end()) {
739 iter->second -= 1;
740 if (iter->second == 0) {
741 curr_size -= get_element(func_sizes, p.first);
742 internal_assert(!can_prove(curr_size < 0));
743 }
744 }
745 }
746 }
747
748 return simplify(working_set_size);
749}

751Expr RegionCosts::input_region_size(const string &input, const Box &region) {
752 Expr size = box_size(region);
753 if (!size.defined()) {
754 return Expr();
755 }
756 Expr size_per_ele = make_const(Int(64), get_element(inputs, input).bytes());
757 internal_assert(size_per_ele.defined());
758 return simplify(size * size_per_ele);
759}

797bool is_func_trivial_to_inline(const Function &func) {
798 if (!func.can_be_inlined()) {
799 return false;
800 }
801
802 // For multi-dimensional tuple, we want to take the max over the arithmetic
803 // and memory cost separately for conservative estimate.
804
805 Cost inline_cost(0, 0);
806 for (const auto &val : func.values()) {
807 Cost cost = compute_expr_cost(val);
808 internal_assert(cost.defined());
809 inline_cost.arith = max(cost.arith, inline_cost.arith);
810 inline_cost.memory = max(cost.memory, inline_cost.memory);
811 }
812
813 // Compute the cost if we were to call the function instead of inline it
814 Cost call_cost(1, 0);
815 for (const auto &type : func.output_types()) {
816 call_cost.memory = max(type.bytes(), call_cost.memory);
817 }
818
819 Expr is_trivial = (call_cost.arith + call_cost.memory) >= (inline_cost.arith + inline_cost.memory);
820 return can_prove(is_trivial);
821}
