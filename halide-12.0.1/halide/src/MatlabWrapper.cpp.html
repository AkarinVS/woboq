9// Define the mex wrapper API call (mexFunction) for a func with name pipeline_name.
10llvm::Function *define_matlab_wrapper(llvm::Module *module,
11 llvm::Function *pipeline_argv_wrapper,
12 llvm::Function *metadata_getter) {
13 user_assert(!module->getFunction("mexFunction"))
14 << "Module already contains a mexFunction. Only one pipeline can define a mexFunction.\n";
15
16 LLVMContext &ctx = module->getContext();
17
18 llvm::Function *call_pipeline = module->getFunction("halide_matlab_call_pipeline");
19 internal_assert(call_pipeline) << "Did not find function 'halide_matlab_call_pipeline' in module.\n";
20
21 llvm::Type *void_ty = llvm::Type::getVoidTy(ctx);
22 llvm::Type *i8_ty = llvm::Type::getInt8Ty(ctx);
23 llvm::Type *i32_ty = llvm::Type::getInt32Ty(ctx);
24 Value *user_context = ConstantPointerNull::get(i8_ty->getPointerTo());
25
26 llvm::StructType *mxArray_ty = get_llvm_struct_type_by_name(module, "struct.mxArray");
27 internal_assert(mxArray_ty) << "Did not find mxArray in initial module";
28 llvm::Type *mxArray_ptr_ty = mxArray_ty->getPointerTo();
29 llvm::Type *mxArray_ptr_ptr_ty = mxArray_ptr_ty->getPointerTo();
30
31 // Create the mexFunction function.
32 // (http://www.mathworks.com/help/matlab/apiref/mexfunction.html)
33 llvm::Type *mex_arg_types[] = {
34 i32_ty,
35 mxArray_ptr_ptr_ty,
36 i32_ty,
37 mxArray_ptr_ptr_ty,
38 };
39 FunctionType *mex_ty = FunctionType::get(void_ty, mex_arg_types, false);
40 llvm::Function *mex = llvm::Function::Create(mex_ty, llvm::GlobalValue::ExternalLinkage, "mexFunction", module);
41 BasicBlock *entry = BasicBlock::Create(ctx, "entry", mex);
42
43 IRBuilder<> ir(ctx);
44 ir.SetInsertPoint(entry);
45
46 // Call the metadata_getter function to get the metadata pointer block.
47 llvm::CallInst *metadata_ptr = ir.CreateCall(metadata_getter);
48
49 // Extract the argument values from the mexFunction.
50 llvm::Function::arg_iterator mex_args = mex->arg_begin();
51 Value *nlhs = iterator_to_pointer(mex_args++);
52 Value *plhs = iterator_to_pointer(mex_args++);
53 Value *nrhs = iterator_to_pointer(mex_args++);
54 Value *prhs = iterator_to_pointer(mex_args++);
55
56 Value *call_pipeline_args[] = {
57 user_context,
58 pipeline_argv_wrapper,
59 metadata_ptr,
60 nlhs,
61 plhs,
62 nrhs,
63 prhs,
64 };
65 ir.CreateCall(call_pipeline, call_pipeline_args);
66 ir.CreateRetVoid();
67
68 return mex;
69}
