82 class GatherRVars : public IRGraphVisitor {
83 public:

86 void visit(const Variable *op) override {

115Expr add_let_expression(const Expr &expr,
116 const map<string, Expr> &let_var_mapping,
117 const vector<string> &let_variables) {
118 // TODO: find a faster way to do this
119 Expr ret = StripLets().mutate(expr);
120 bool changed = true;
121 vector<bool> injected(let_variables.size(), false);
122 while (changed) {
123 changed = false;
124 for (size_t i = 0; i < let_variables.size(); i++) {
125 const auto &let_variable = let_variables[i];
126 if (!injected[i] && expr_uses_var(ret, let_variable)) {
127 auto value = let_var_mapping.find(let_variable)->second;
128 ret = Let::make(let_variable, value, ret);
129 injected[i] = true;
130 changed = true;
131 }
132 }
133 }
134 return ret;
135}

214map<string, Box> inference_bounds(const vector<Func> &funcs,
215 const vector<Box> &output_bounds) {
216 internal_assert(funcs.size() == output_bounds.size());
217 // Obtain all dependencies
218 vector<Function> functions;
219 functions.reserve(funcs.size());
220 for (const auto &func : funcs) {
221 functions.push_back(func.function());
222 }
223 map<string, Function> env;
224 for (const auto &func : functions) {
225 map<string, Function> local_env = find_transitive_calls(func);
226 env.insert(local_env.begin(), local_env.end());
227 }
228
229 // Reduction variable scopes
230 Scope<Interval> scope;
231 for (const auto &it : env) {
232 Func func = Func(it.second);
233 for (int i = 0; i < func.num_update_definitions(); i++) {
234 map<string, ReductionVariableInfo> rvars =
235 gather_rvariables(func.update_values(i));
236 for (const auto &it : rvars) {
237 Interval interval(it.second.min, it.second.min + it.second.extent - 1);
238 scope.push(it.first, interval);
239 }
240 }
241 }
242 // Sort functions
243 vector<string> order = realization_order(functions, env).first;
244 FuncValueBounds func_value_bounds = compute_function_value_bounds(order, env);
245
246 map<string, Box> bounds;
247 // Set up bounds for outputs
248 for (int i = 0; i < (int)funcs.size(); i++) {
249 const Func &func = funcs[i];
250 bounds[func.name()] = output_bounds[i];
251 }
252 // Traverse from the consumers to the producers
253 for (auto it = order.rbegin(); it != order.rend(); it++) {
254 Func func = Func(env[*it]);
255 // We should already have the bounds of this function
256 internal_assert(bounds.find(*it) != bounds.end());
257 const Box &current_bounds = bounds[*it];
258 internal_assert(func.args().size() == current_bounds.size());
259 // We know the range for each argument of this function
260 for (int i = 0; i < (int)current_bounds.size(); i++) {
261 string arg = func.args()[i].name();
262 scope.push(arg, current_bounds[i]);
263 }
264 // Propagate the bounds
265 for (int update_id = -1; update_id < func.num_update_definitions(); update_id++) {
266 // For each rhs expression
267 Tuple tuple = update_id == -1 ? func.values() : func.update_values(update_id);
268 for (const auto &expr : tuple.as_vector()) {
269 // For all the immediate dependencies of this expression,
270 // find the required ranges
271 map<string, Box> update_bounds =
272 boxes_required(expr, scope, func_value_bounds);
273 // Loop over the dependencies
274 for (const auto &it : update_bounds) {
275 if (it.first == func.name()) {
276 // Skip self reference
277 continue;
278 }
279 // Update the bounds, if not exists then create a new one
280 auto found = bounds.find(it.first);
281 if (found == bounds.end()) {
282 bounds[it.first] = it.second;
283 } else {
284 Box new_box = box_union(found->second, it.second);
285 bounds[it.first] = new_box;
286 }
287 }
288 }
289 }
290 for (int i = 0; i < (int)current_bounds.size(); i++) {
291 scope.pop(func.args()[i].name());
292 }
293 }
294 for (auto &it : bounds) {
295 auto &bound = it.second;
296 for (int i = 0; i < (int)bound.size(); i++) {
297 bound[i].min = common_subexpression_elimination(simplify(bound[i].min));
298 bound[i].max = common_subexpression_elimination(simplify(bound[i].max));
299 }
300 }
301 return bounds;
302}

310vector<pair<Expr, Expr>> box_to_vector(const Box &bounds) {
311 vector<pair<Expr, Expr>> ret;
312 ret.reserve(bounds.size());
313 for (const auto &b : bounds.bounds) {
314 ret.emplace_back(b.min, b.max - b.min + 1);
315 }
316 return ret;

319bool equal(const RDom &bounds0, const RDom &bounds1) {
320 if (bounds0.domain().domain().size() != bounds1.domain().domain().size()) {
321 return false;
322 }
323 for (int bid = 0; bid < (int)bounds0.domain().domain().size(); bid++) {
324 if (!equal(bounds0[bid].min(), bounds1[bid].min()) ||
325 !equal(bounds0[bid].extent(), bounds1[bid].extent())) {
326 return false;
327 }
328 }
329 return true;
330}

368pair<bool, Expr> solve_inverse(Expr expr,
369 const string &new_var,
370 const string &var) {
371 expr = substitute_in_all_lets(simplify(expr));
372 Interval interval = solve_for_outer_interval(expr, var);
373 if (!interval.is_bounded()) {
374 return {false, Expr()};
375 }
376 Expr rmin = simplify(interval.min);
377 Expr rmax = simplify(interval.max);
378 Expr rextent = simplify(rmax - rmin + 1);
379
380 const int64_t *extent_int = as_const_int(rextent);
381 if (extent_int == nullptr) {
382 return {false, Expr()};
383 }
384
385 // For some reason interval.is_single_point() doesn't work
386 if (extent_int != nullptr && *extent_int == 1) {
387 return {true, rmin};
388 }
389
390 // Create a RDom to loop over the interval
391 RDom r(0, int(*extent_int));
392 Expr cond = substitute(var, rmin + r.x, expr.as<EQ>()->b);
393 cond = substitute(new_var, Var(var), cond) == Var(var);
394 r.where(cond);
395 return {true, rmin + r.x};
396}

400struct BufferDimensionsFinder : public IRGraphVisitor {
401public:
402 using IRGraphVisitor::visit;
403 map<string, BufferInfo> find(const Func &func) {
404 buffer_calls.clear();
405 vector<Expr> vals = func.values().as_vector();
406 for (const Expr &val : vals) {
407 val.accept(this);
408 }
409 for (int update_id = 0; update_id < func.num_update_definitions(); update_id++) {
410 vals = func.update_values(update_id).as_vector();
411 for (const Expr &val : vals) {
412 val.accept(this);
413 }
414 }
415 return buffer_calls;
416 }
417
418 void visit(const Variable *op) override {
419 IRGraphVisitor::visit(op);
420 if (op->param.defined() && op->type.is_float()) {
421 buffer_calls[op->name] = BufferInfo{
422 op->param.dimensions(),
423 op->type};
424 }
425 }
426
427 void visit(const Call *op) override {
428 IRGraphVisitor::visit(op);
429 if (op->call_type == Call::Image && op->type.is_float()) {
430 if (op->image.defined()) {
431 buffer_calls[op->name] = BufferInfo{
432 op->image.dimensions(),
433 op->type};
434 } else {
435 internal_assert(op->param.defined());
436 buffer_calls[op->name] = BufferInfo{
437 op->param.dimensions(),
438 op->type};
439 }
440 }
441 }
442
443 map<string, BufferInfo> buffer_calls;

481Expr substitute_rdom_predicate(
482 const string &name, const Expr &replacement, const Expr &expr) {
483 Expr substituted = substitute(name, replacement, expr);
484 map<string, ReductionVariableInfo> rvars =
485 gather_rvariables(substituted);
486 set<ReductionDomain, ReductionDomain::Compare> rdoms_set;
487 for (const auto &it : rvars) {
488 rdoms_set.insert(it.second.domain);
489 }
490 vector<ReductionDomain> rdoms;
491 copy(rdoms_set.begin(), rdoms_set.end(), std::back_inserter(rdoms));
492 for (auto &r : rdoms) {
493 Expr predicate = r.predicate();
494 predicate = substitute(name, replacement, predicate);
495 r.set_predicate(predicate);
496 }
497 return substituted;
498}

502struct FunctionCallFinder : public IRGraphVisitor {
503public:
504 using IRGraphVisitor::visit;
505 bool find(const string &func_name_,
506 const Expr &expr,
507 const map<string, Expr> &let_var_mapping_) {
508 func_name = func_name_;
509 let_var_mapping = &let_var_mapping_;
510 found = false;
511 expr.accept(this);
512 return found;
513 }
514
515 bool find(const Expr &expr,
516 const map<string, Expr> &let_var_mapping_) {
517 func_name = "";
518 let_var_mapping = &let_var_mapping_;
519 found = false;
520 expr.accept(this);
521 return found;
522 }
523
524 void visit(const Variable *var) override {
525 if (!found) {
526 auto it = let_var_mapping->find(var->name);
527 if (it != let_var_mapping->end()) {
528 found = find(func_name, it->second, *let_var_mapping);
529 }
530 }
531 }
532
533 void visit(const Call *op) override {
534 if (op->call_type == Call::Image || op->call_type == Call::Halide) {
535 if (func_name.empty() || op->name == func_name) {
536 found = true;
537 }
538 }
539 if (!found) {
540 IRGraphVisitor::visit(op);
541 }
542 }
543
544 string func_name;
545 map<string, Expr> const *let_var_mapping;
546 bool found;
547};
