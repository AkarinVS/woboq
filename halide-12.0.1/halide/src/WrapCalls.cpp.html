19void insert_func_wrapper_helper(map<FunctionPtr, SubstitutionMap> &func_wrappers_map,
20 const FunctionPtr &in_func,
21 const FunctionPtr &wrapped_func,
22 const FunctionPtr &wrapper) {
23 internal_assert(in_func.defined() &&
24 wrapped_func.defined() &&
25 wrapper.defined());
26 internal_assert(func_wrappers_map[in_func].count(wrapped_func) == 0)
27 << "Should only have one wrapper for each function call in a Func\n";
28
29 SubstitutionMap &wrappers_map = func_wrappers_map[in_func];
30 for (auto iter = wrappers_map.begin(); iter != wrappers_map.end(); ++iter) {
31 if (iter->second.same_as(wrapped_func)) {
32 debug(4) << "Merging wrapper of " << Function(in_func).name()
33 << " [" << Function(iter->first).name()
34 << ", " << Function(iter->second).name()
35 << "] with [" << Function(wrapped_func).name() << ", "
36 << Function(wrapper).name() << "]\n";
37 iter->second = wrapper;
38 return;
39 } else if (wrapper.same_as(iter->first)) {
40 debug(4) << "Merging wrapper of " << Function(in_func).name()
41 << " [" << Function(wrapped_func).name()
42 << ", " << Function(wrapper).name()
43 << "] with [" << Function(iter->first).name()
44 << ", " << Function(iter->second).name() << "]\n";
45 wrappers_map.emplace(wrapped_func, iter->second);
46 wrappers_map.erase(iter);
47 return;
48 }
49 }
50 wrappers_map[wrapped_func] = wrapper;
51}

53void validate_custom_wrapper(const Function &in_func, const Function &wrapped, const Function &wrapper) {
54 map<string, Function> callees = find_direct_calls(in_func);
55 if (!callees.count(wrapper.name())) {
56 std::ostringstream callees_text;
57 for (const auto &it : callees) {
58 callees_text << " " << it.second.name() << "\n";
59 }
60
61 user_error
62 << "Cannot wrap \"" << wrapped.name() << "\" in \"" << in_func.name()
63 << "\" because \"" << in_func.name() << "\" does not call \""
64 << wrapped.name() << "\"\n"
65 << "Direct callees of \"" << in_func.name() << "\" are:\n"
66 << callees_text.str();
67 }
68}

72map<string, Function> wrap_func_calls(const map<string, Function> &env) {
73 map<string, Function> wrapped_env;
74
75 map<FunctionPtr, SubstitutionMap> func_wrappers_map; // In Func -> [wrapped Func -> wrapper]
76 set<string> global_wrappers;
77
78 for (const auto &iter : env) {
79 wrapped_env.emplace(iter.first, iter.second);
80 func_wrappers_map[iter.second.get_contents()];
81 }
82
83 for (const auto &it : env) {
84 string wrapped_fname = it.first;
85 FunctionPtr wrapped_func = it.second.get_contents();
86 const auto &wrappers = it.second.schedule().wrappers();
87
88 // Put the names of all wrappers of this Function into the set for
89 // faster comparison during the substitution.
90 set<string> all_func_wrappers;
91 for (const auto &iter : wrappers) {
92 all_func_wrappers.insert(Function(iter.second).name());
93 }
94
95 for (const auto &iter : wrappers) {
96 string in_func = iter.first;
97 FunctionPtr wrapper = iter.second;
98
99 if (in_func.empty()) { // Global wrapper
100 global_wrappers.insert(Function(wrapper).name());
101 for (const auto &wrapped_env_iter : wrapped_env) {
102 in_func = wrapped_env_iter.first;
103 if ((wrapped_fname == in_func) ||
104 (all_func_wrappers.find(in_func) != all_func_wrappers.end())) {
105 // The wrapper should still call the original function,
106 // so we don't want to rewrite the calls done by the
107 // wrapper. We also shouldn't rewrite the original
108 // function itself.
109 debug(4) << "Skip over replacing \"" << in_func
110 << "\" with \"" << Function(wrapper).name() << "\"\n";
111 continue;
112 }
113 if (wrappers.count(in_func)) {
114 // If the 'in_func' already has custom wrapper for
115 // 'wrapped_func', don't substitute in the global wrapper.
116 // Custom wrapper always takes precedence over global wrapper
117 continue;
118 }
119 debug(4) << "Global wrapper: replacing reference of \""
120 << wrapped_fname << "\" in \"" << in_func
121 << "\" with \"" << Function(wrapper).name() << "\"\n";
122 insert_func_wrapper_helper(func_wrappers_map,
123 wrapped_env_iter.second.get_contents(),
124 wrapped_func, wrapper);
125 }
126 } else { // Custom wrapper
127 debug(4) << "Custom wrapper: replacing reference of \""
128 << wrapped_fname << "\" in \"" << in_func << "\" with \""
129 << Function(wrapper).name() << "\"\n";
130
131 const auto &in_func_iter = wrapped_env.find(in_func);
132 if (in_func_iter == wrapped_env.end()) {
133 // We find a wrapper definition of 'wrapped_func 'for 'in_func'
134 // which is not in this pipeline. We don't need to perform
135 // the substitution since no function in this pipeline will ever
136 // refer to 'in_func'.
137 //
138 // This situation might arise in the following case below:
139 // f(x) = x;
140 // g(x) = f(x) + 1;
141 // f.in(g);
142 // f.realize(..);
143 debug(4) << " skip custom wrapper for " << in_func << " [" << wrapped_fname
144 << " -> " << Function(wrapper).name() << "] since it's not in the pipeline\n";
145 continue;
146 }
147 insert_func_wrapper_helper(func_wrappers_map,
148 wrapped_env[in_func].get_contents(),
149 wrapped_func,
150 wrapper);
151 }
152 }
153 }
154
155 // Perform the substitution
156 for (auto &iter : wrapped_env) {
157 const auto &substitutions = func_wrappers_map[iter.second.get_contents()];
158 if (!substitutions.empty()) {
159 iter.second.substitute_calls(substitutions);
160 }
161 }
162
163 // Assert that the custom wrappers are actually used, i.e. if f.in(g) is
164 // called, but 'f' is never called inside 'g', this will throw a user error.
165 // Perform the check after the wrapper substitution to handle multi-fold
166 // wrappers, e.g. f.in(g).in(g).
167 for (const auto &iter : wrapped_env) {
168 const auto &substitutions = func_wrappers_map[iter.second.get_contents()];
169 for (const auto &pair : substitutions) {
170 if (global_wrappers.find(Function(pair.second).name()) == global_wrappers.end()) {
171 validate_custom_wrapper(iter.second, Function(pair.first), Function(pair.second));
172 }
173 }
174 }
175
176 return wrapped_env;
177}
