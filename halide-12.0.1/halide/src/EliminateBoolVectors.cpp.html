25 template<typename T>
26 Expr visit_comparison(const T *op) {
27 Expr a = mutate(op->a);
28 Expr b = mutate(op->b);
29 Type t = a.type();
30
31 // Ensure both a and b have the same type (if this is a vector
32 // comparison). This should only be necessary if the operands are
33 // integer vectors (promoted from bool vectors).
34 if (t.lanes() > 1 && t.bits() != b.type().bits()) {
35 internal_assert(t.is_int() && b.type().is_int());
36
37 t = t.with_bits(std::max(t.bits(), b.type().bits()));
38 if (t != a.type()) {
39 a = Call::make(t, Call::cast_mask, {a}, Call::PureIntrinsic);
40 }
41 if (t != b.type()) {
42 b = Call::make(t, Call::cast_mask, {b}, Call::PureIntrinsic);
43 }
44 }
45
46 Expr expr;
47 if (!a.same_as(op->a) || !b.same_as(op->b)) {
48 expr = T::make(a, b);
49 } else {
50 expr = op;
51 }
52
53 if (t.lanes() > 1) {
54 // To represent bool vectors, OpenCL uses vectors of signed
55 // integers with the same width as the types being compared.
56 expr = Call::make(t.with_code(Type::Int), Call::bool_to_mask, {expr}, Call::PureIntrinsic);
57 }
58 return expr;
59 }

80 template<typename T>
81 Expr visit_logical_binop(const T *op, Call::IntrinsicOp bitwise_op) {
82 Expr a = mutate(op->a);
83 Expr b = mutate(op->b);
84
85 Type ta = a.type();
86 Type tb = b.type();
87 if (ta.lanes() > 1) {
88 // Ensure that both a and b have the same type.
89 Type t = ta.with_bits(std::max(ta.bits(), tb.bits()));
90 if (t != a.type()) {
91 a = Call::make(t, Call::cast_mask, {a}, Call::PureIntrinsic);
92 }
93 if (t != b.type()) {
94 b = Call::make(t, Call::cast_mask, {b}, Call::PureIntrinsic);
95 }
96 // Replace logical operation with bitwise operation.
97 return Call::make(t, bitwise_op, {a, b}, Call::PureIntrinsic);
98 } else if (!a.same_as(op->a) || !b.same_as(op->b)) {
99 return T::make(a, b);
100 } else {
101 return op;
102 }
103 }

125 Expr visit(const Cast *op) override {
126 if (op->value.type().is_bool() && op->value.type().is_vector()) {
127 // Casting from bool
128 return mutate(Select::make(op->value,
129 make_one(op->type),
130 make_zero(op->type)));

139 Stmt visit(const Store *op) override {
140 Expr predicate = op->predicate;
141 if (!is_const_one(predicate)) {
142 predicate = mutate(predicate);
143 }
144 Expr value = op->value;
145 if (op->value.type().is_bool()) {
146 Type ty = UInt(8, op->value.type().lanes());
147 value = Select::make(value,
148 make_one(ty),
149 make_zero(ty));
150 }
151 value = mutate(value);
152 Expr index = mutate(op->index);
153
154 if (predicate.same_as(op->predicate) && value.same_as(op->value) && index.same_as(op->index)) {
155 return op;
156 } else {
157 return Store::make(op->name, value, index, op->param, predicate, op->alignment);
158 }
159 }

161 Expr visit(const Load *op) override {
162 Expr predicate = op->predicate;
163 if (!is_const_one(predicate)) {
164 predicate = mutate(predicate);
165 }
166 Expr index = mutate(op->index);
167 if (predicate.same_as(op->predicate) && index.same_as(op->index)) {
168 return op;
169 } else {
170 return Load::make(op->type, op->name, std::move(index),
171 op->image, op->param, std::move(predicate),
172 op->alignment);
173 }

176 // Assuming that a and b should have the same scalar type and they might have
177 // been boolean vectors converted to integer vectors, cast the masks to be the
178 // same type. This is necessary in the case of a select or Call::if_then_else, e.g.:
179 //
180 // Expr a = float_expr1() < float_expr2(); // promoted to int32xN
181 // Expr b = uint8_expr1() < uint8_expr2(); // promoted to int8xN
182 // Expr c = select(a < b, a, b); // whoops
183 static void unify_bool_vector_types(Expr &a, Expr &b) {
184 if (a.type().bits() != b.type().bits() &&
185 a.type().lanes() == b.type().lanes() &&
186 a.type().is_int() && b.type().is_int()) {
187 if (a.type().bits() > b.type().bits()) {
188 b = Call::make(a.type(), Call::cast_mask, {b}, Call::PureIntrinsic);
189 } else {
190 a = Call::make(b.type(), Call::cast_mask, {a}, Call::PureIntrinsic);
191 }
192 }
193 }

195 Expr visit(const Call *op) override {
196 if (op->is_intrinsic(Call::if_then_else)) {
197 internal_assert(op->args.size() == 3);
198 if (op->args[0].type().is_vector()) {
199 Expr cond = mutate(op->args[0]);
200 Expr true_value = mutate(op->args[1]);
201 Expr false_value = mutate(op->args[2]);
202 Type cond_ty = cond.type();
203
204 // If the condition is a vector, it should be a vector of ints.
205 internal_assert(cond_ty.code() == Type::Int);
206
207 // if_then_else_mask requires that all 3 operands have the same
208 // width.
209 unify_bool_vector_types(true_value, false_value);
210 internal_assert(true_value.type().bits() == false_value.type().bits());
211 if (true_value.type().bits() != cond_ty.bits()) {
212 cond_ty = cond_ty.with_bits(true_value.type().bits());
213 cond = Call::make(cond_ty, Call::cast_mask, {cond}, Call::PureIntrinsic);
214 }
215
216 return Call::make(true_value.type(), Call::if_then_else_mask, {cond, true_value, false_value}, Call::PureIntrinsic);
217 }
218 } else if (op->is_intrinsic(Call::require)) {
219 internal_assert(op->args.size() == 3);
220 if (op->args[0].type().is_vector()) {
221 Expr cond = mutate(op->args[0]);
222 Expr value = mutate(op->args[1]);
223 Expr message = mutate(op->args[2]);
224
225 // If the condition is a vector, it should be a vector of ints.
226 internal_assert(cond.type().code() == Type::Int);
227
228 return Call::make(value.type(), Call::require_mask, {cond, value, message}, Call::PureIntrinsic);
229 }
230 }
231
232 return IRMutator::visit(op);

235 Expr visit(const Select *op) override {
236 Expr cond = mutate(op->condition);
237 Expr true_value = mutate(op->true_value);
238 Expr false_value = mutate(op->false_value);
239 Type cond_ty = cond.type();
240 if (cond_ty.is_vector()) {
241 // If the condition is a vector, it should be a vector of
242 // ints.
243 internal_assert(cond_ty.code() == Type::Int);
244
245 // select_mask requires that all 3 operands have the same
246 // width.
247 unify_bool_vector_types(true_value, false_value);
248 internal_assert(true_value.type().bits() == false_value.type().bits());
249 if (true_value.type().bits() != cond_ty.bits()) {
250 cond_ty = cond_ty.with_bits(true_value.type().bits());
251 cond = Call::make(cond_ty, Call::cast_mask, {cond}, Call::PureIntrinsic);
252 }
253
254 return Call::make(true_value.type(), Call::select_mask, {cond, true_value, false_value}, Call::PureIntrinsic);
255 } else if (!cond.same_as(op->condition) ||
256 !true_value.same_as(op->true_value) ||
257 !false_value.same_as(op->false_value)) {
258 return Select::make(cond, true_value, false_value);
259 } else {
260 return op;
261 }
262 }

264 Expr visit(const Broadcast *op) override {
265 Expr value = mutate(op->value);

275 Expr visit(const Shuffle *op) override {
276 Expr expr = IRMutator::visit(op);
277 if (op->is_extract_element() && op->type.is_bool()) {
278 op = expr.as<Shuffle>();
279 internal_assert(op);
280 // This is extracting a scalar element of a bool
281 // vector. Generate a call to extract_mask_element.
282 expr = Call::make(Bool(), Call::extract_mask_element, {Shuffle::make_concat(op->vectors), op->indices[0]}, Call::PureIntrinsic);
283 }
284 return expr;

287 template<typename NodeType, typename LetType>
288 NodeType visit_let(const LetType *op) {
289 Expr value = mutate(op->value);
290
291 // We changed the type of the let, we need to replace the
292 // references to the let in the body. We can't just substitute
293 // them, because the types won't match without running the
294 // other visitors during the substitution, so we save the
295 // types that we changed for later.
296 if (value.type() != op->value.type()) {
297 lets.push(op->name, value.type());
298 }
299 auto body = mutate(op->body);
300 if (value.type() != op->value.type()) {
301 lets.pop(op->name);
302 }
303
304 if (!value.same_as(op->value) || !body.same_as(op->body)) {
305 return LetType::make(op->name, value, body);
306 } else {
307 return op;
308 }
309 }
310
311 Expr visit(const Let *op) override {
312 return visit_let<Expr>(op);
313 }
314 Stmt visit(const LetStmt *op) override {
315 return visit_let<Stmt>(op);
