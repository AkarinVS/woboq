32/** Call device_sync() for all Buffers in the Realization.
33 * If one of the calls returns an error, subsequent Buffers won't have
34 * device_sync called; thus callers should consider a nonzero return
35 * code to mean that potentially all of the Buffers are in an indeterminate
36 * state of sync.
37 * Calling this explicitly should rarely be necessary, except for profiling. */
38int Realization::device_sync(void *ctx) {
39 for (auto &b : images) {
40 int result = b.device_sync(ctx);
41 if (result != 0) {
42 return result;
43 }
44 }
45 return 0;
46}
