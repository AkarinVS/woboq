44ValType convert_halide_type_to_val_type(const Type &halide_t) {
45 internal_assert(halide_t.is_scalar() && !halide_t.is_handle());
46
47 ValType val_t;
48 if (halide_t.is_uint()) {
49 if (halide_t.bits() == 1) { // Bool
50 val_t = ValType::UInt1;
51 } else if (halide_t.bits() == 8) {
52 val_t = ValType::UInt8;
53 } else if (halide_t.bits() == 16) {
54 val_t = ValType::UInt16;
55 } else if (halide_t.bits() == 32) {
56 val_t = ValType::UInt32;
57 } else {
58 internal_assert(halide_t.bits() == 64);
59 val_t = ValType::UInt64;
60 }
61 } else if (halide_t.is_int()) {
62 if (halide_t.bits() == 8) {
63 val_t = ValType::Int8;
64 } else if (halide_t.bits() == 16) {
65 val_t = ValType::Int16;
66 } else if (halide_t.bits() == 32) {
67 val_t = ValType::Int32;
68 } else {
69 internal_assert(halide_t.bits() == 64);
70 val_t = ValType::Int64;
71 }
72 } else {
73 internal_assert(halide_t.is_float());
74 if (halide_t.bits() == 16) {
75 val_t = ValType::Float16;
76 } else if (halide_t.bits() == 32) {
77 val_t = ValType::Float32;
78 } else {
79 internal_assert(halide_t.bits() == 64);
80 val_t = ValType::Float64;
81 }
82 }
83 return val_t;
84}

86vector<ValType> convert_halide_types_to_val_types(const vector<Type> &halide_types) {
87 vector<ValType> val_types(halide_types.size());
88 for (size_t i = 0; i < halide_types.size(); ++i) {
89 val_types[i] = convert_halide_type_to_val_type(halide_types[i]);
90 }
91 return val_types;
92}

94struct TableKey {
95 vector<ValType> types;
96 RootExpr root;
97 size_t dim;
98 TableKey(ValType t, RootExpr r, size_t d)
99 : types({t}), root(r), dim(d) {
100 }
101 TableKey(const vector<ValType> &t, RootExpr r, size_t d)
102 : types(t), root(r), dim(d) {
103 }
104
105 bool operator==(const TableKey &other) const {
106 return (types == other.types) && (root == other.root) && (dim == other.dim);
107 }
108 bool operator<(const TableKey &other) const {
109 if (types < other.types) {
110 return true;
111 } else if (types > other.types) {
112 return false;
113 }
114 if (root < other.root) {
115 return true;
116 } else if (root > other.root) {
117 return false;
118 }
119 return (dim < other.dim);
120 }
121};

125#define declare_vars(t, index) \
126 Expr x##index = Variable::make((t), "x" + std::to_string(index)); \
127 Expr y##index = Variable::make((t), "y" + std::to_string(index)); \

284const vector<AssociativePattern> &get_ops_table_helper(const vector<Type> &types, RootExpr root, size_t dim) {
285 TableKey gen_key(ValType::All, root, dim);
286 TableKey key(convert_halide_types_to_val_types(types), root, dim);
287
288 const auto &table_it = pattern_tables.find(key);
289 if (table_it == pattern_tables.end()) { // Populate the table if we haven't done so previously
290 vector<AssociativePattern> &table = pattern_tables[key];
291
292 // Populate the general associative op LUT
293 const auto &gen_iter = val_type_to_populate_luts_fn.find(gen_key);
294 if (gen_iter != val_type_to_populate_luts_fn.end()) {
295 gen_iter->second(types, table);
296 }
297
298 // Populate the type-specific associative op LUT
299 const auto &iter = val_type_to_populate_luts_fn.find(key);
300 if (iter != val_type_to_populate_luts_fn.end()) {
301 iter->second(types, table);
302 }
303
304 return table;
305 }
306 return table_it->second;
307}

324const vector<AssociativePattern> &get_ops_table(const vector<Expr> &exprs) {
325 internal_assert(!exprs.empty());
326
327 static const vector<AssociativePattern> empty;
328
329 if (exprs.size() > 2) {
330 debug(5) << "Returning empty table since tuple size is larger than 2\n";
331 return empty;
332 }
333
334 vector<Type> types(exprs.size());
335 for (size_t i = 0; i < exprs.size(); ++i) {
336 types[i] = exprs[i].type();
337 }
338
339 RootExpr root = RootExpr::Unknown;
340 if (exprs[0].as<Halide::Internal::Add>()) {
341 debug(5) << "Returning Add root table for type " << print_types(types) << "\n";
342 root = RootExpr::Add;
343 } else if (exprs[0].as<Halide::Internal::Sub>()) {
344 debug(5) << "Returning Sub root table for type " << print_types(types) << "\n";
345 root = RootExpr::Sub;
346 } else if (exprs[0].as<Halide::Internal::Mul>()) {
347 debug(5) << "Returning Mul root table for type " << print_types(types) << "\n";
348 root = RootExpr::Mul;
349 } else if (exprs[0].as<Halide::Internal::Min>()) {
350 debug(5) << "Returning Min root table for type " << print_types(types) << "\n";
351 root = RootExpr::Min;
352 } else if (exprs[0].as<Halide::Internal::Max>()) {
353 debug(5) << "Returning Max root table for type " << print_types(types) << "\n";
354 root = RootExpr::Max;
355 } else if (exprs[0].as<Halide::Internal::Select>()) {
356 debug(5) << "Returning Select root table for type " << print_types(types) << "\n";
357 root = RootExpr::Select;
358 } else if (exprs[0].as<Halide::Internal::And>()) {
359 debug(5) << "Returning And root table for type " << print_types(types) << "\n";
360 root = RootExpr::And;
361 } else if (exprs[0].as<Halide::Internal::Or>()) {
362 debug(5) << "Returning Or root table for type " << print_types(types) << "\n";
363 root = RootExpr::Or;
364 } else if (exprs[0].as<Halide::Internal::Cast>()) {
365 debug(5) << "Returning Cast root table for type " << print_types(types) << "\n";
366 root = RootExpr::Cast;
367 }
368
369 if (root != RootExpr::Unknown) {
370 // get_ops_table_helper() lazily initializes the table, so ensure
371 // that multiple threads can't try to do so at the same time.
372 static std::mutex ops_table_lock;
373 std::lock_guard<std::mutex> lock_guard(ops_table_lock);
374
375 const vector<AssociativePattern> &table = get_ops_table_helper(types, root, exprs.size());
376 debug(7) << "Table size: " << table.size() << "\n";
377 for (const auto &p : table) {
378 debug(7) << p;
379 }
380 return table;
381 }
382 debug(5) << "Returning empty table\n";
383 return empty;
384}
