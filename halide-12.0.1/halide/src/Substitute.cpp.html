14class Substitute : public IRMutator {
15 const map<string, Expr> &replace;
16 Scope<> hidden;
17
18 Expr find_replacement(const string &s) {
19 map<string, Expr>::const_iterator iter = replace.find(s);
20 if (iter != replace.end() && !hidden.contains(s)) {
21 return iter->second;
22 } else {
23 return Expr();
24 }
25 }
26
27public:
28 Substitute(const map<string, Expr> &m)
29 : replace(m) {
30 }
31
32 using IRMutator::visit;
33
34 Expr visit(const Variable *v) override {
35 Expr r = find_replacement(v->name);
36 if (r.defined()) {
37 return r;
38 } else {
39 return v;
40 }
41 }
42
43 Expr visit(const Let *op) override {
44 Expr new_value = mutate(op->value);
45 hidden.push(op->name);
46 Expr new_body = mutate(op->body);
47 hidden.pop(op->name);
48
49 if (new_value.same_as(op->value) &&
50 new_body.same_as(op->body)) {
51 return op;
52 } else {
53 return Let::make(op->name, new_value, new_body);
54 }
55 }
56
57 Stmt visit(const LetStmt *op) override {
58 Expr new_value = mutate(op->value);
59 hidden.push(op->name);
60 Stmt new_body = mutate(op->body);
61 hidden.pop(op->name);
62
63 if (new_value.same_as(op->value) &&
64 new_body.same_as(op->body)) {
65 return op;
66 } else {
67 return LetStmt::make(op->name, new_value, new_body);
68 }
69 }
70
71 Stmt visit(const For *op) override {
72 Expr new_min = mutate(op->min);
73 Expr new_extent = mutate(op->extent);
74 hidden.push(op->name);
75 Stmt new_body = mutate(op->body);
76 hidden.pop(op->name);
77
78 if (new_min.same_as(op->min) &&
79 new_extent.same_as(op->extent) &&
80 new_body.same_as(op->body)) {
81 return op;
82 } else {
83 return For::make(op->name, new_min, new_extent, op->for_type, op->device_api, new_body);
84 }
85 }
86};

149/** Substitute an expr for a var in a graph. */
150class GraphSubstitute : public IRGraphMutator {
151 string var;
152 Expr value;
153
154 using IRGraphMutator::visit;
155
156 Expr visit(const Variable *op) override {
157 if (op->name == var) {
158 return value;
159 } else {
160 return op;
161 }
162 }
163
164 Expr visit(const Let *op) override {
165 Expr new_value = mutate(op->value);
166 if (op->name == var) {
167 return Let::make(op->name, new_value, op->body);
168 } else {
169 return Let::make(op->name, new_value, mutate(op->body));
170 }
171 }
172
173public:
174 GraphSubstitute(const string &var, const Expr &value)
175 : var(var), value(value) {
176 }
177};
178
179/** Substitute an Expr for another Expr in a graph. Unlike substitute,
180 * this only checks for shallow equality. */
181class GraphSubstituteExpr : public IRGraphMutator {
182 Expr find, replace;
183
184public:
185 using IRGraphMutator::mutate;
186
187 Expr mutate(const Expr &e) override {
188 if (e.same_as(find)) {
189 return replace;
190 } else {
191 return IRGraphMutator::mutate(e);
192 }
193 }
194
195 GraphSubstituteExpr(const Expr &find, const Expr &replace)
196 : find(find), replace(replace) {
197 }
198};
