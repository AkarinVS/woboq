27// Collect the bounds of all the externally referenced buffers in a stmt.
28class CollectExternalBufferBounds : public IRVisitor {
29public:
30 map<string, Box> buffers;
31
32 using IRVisitor::visit;
33
34 void add_buffer_bounds(const string &name, const Buffer<> &image, const Parameter &param, int dims) {
35 Box b;
36 for (int i = 0; i < dims; ++i) {
37 string dim_name = std::to_string(i);
38 Expr buf_min_i = Variable::make(Int(32), name + ".min." + dim_name,
39 image, param, ReductionDomain());
40 Expr buf_extent_i = Variable::make(Int(32), name + ".extent." + dim_name,
41 image, param, ReductionDomain());
42 Expr buf_max_i = buf_min_i + buf_extent_i - 1;
43 b.push_back(Interval(buf_min_i, buf_max_i));
44 }
45 buffers.emplace(name, b);
46 }
47
48 void visit(const Call *op) override {
49 IRVisitor::visit(op);
50 add_buffer_bounds(op->name, op->image, op->param, (int)op->args.size());
51 }
52
53 void visit(const Variable *op) override {
54 if (op->param.defined() && op->param.is_buffer()) {
55 add_buffer_bounds(op->name, Buffer<>(), op->param, op->param.dimensions());
56 }
57 }

60class InjectPrefetch : public IRMutator {
61public:
62 InjectPrefetch(const map<string, Function> &e, const map<string, Box> &buffers)
63 : env(e), external_buffers(buffers) {
64 }
65
66private:
67 const map<string, Function> &env;
68 const map<string, Box> &external_buffers;
69 Scope<Box> buffer_bounds;
70
71 using IRMutator::visit;
72
73 Box get_buffer_bounds(const string &name, int dims) {
74 if (buffer_bounds.contains(name)) {
75 const Box &b = buffer_bounds.ref(name);
76 internal_assert((int)b.size() == dims);
77 return b;
78 }
79
80 // It is an external buffer.
81 user_assert(env.find(name) == env.end())
82 << "Prefetch to buffer \"" << name << "\" which has not been allocated\n";
83
84 const auto &iter = external_buffers.find(name);
85 internal_assert(iter != external_buffers.end());
86 return iter->second;
87 }
88
89 Stmt visit(const Realize *op) override {
90 Box b;
91 b.used = op->condition;
92 for (const auto &r : op->bounds) {
93 b.push_back(Interval(r.min, r.min + r.extent - 1));
94 }
95 ScopedBinding<Box> bind(buffer_bounds, op->name, b);
96 return IRMutator::visit(op);
97 }
98
99 Stmt visit(const Prefetch *op) override {
100 Stmt body = mutate(op->body);
101
102 const PrefetchDirective &p = op->prefetch;
103 Expr loop_var = Variable::make(Int(32), p.var);
104
105 // Add loop variable + prefetch offset to interval scope for box computation
106 Expr fetch_at = loop_var + p.offset;
107 map<string, Box> boxes_rw = boxes_touched(LetStmt::make(p.var, fetch_at, body));
108
109 // TODO(psuriana): Only prefetch the newly accessed data. We
110 // should subtract the box accessed during previous iteration
111 // from the one accessed during this iteration.
112
113 // TODO(psuriana): Add a new PrefetchBoundStrategy::ShiftInwards
114 // that shifts the base address of the prefetched box so that
115 // the box is completely within the bounds.
116 const auto &b = boxes_rw.find(p.name);
117 if (b != boxes_rw.end()) {
118 Box prefetch_box = b->second;
119 // Only prefetch the region that is in bounds.
120 Box bounds = get_buffer_bounds(b->first, b->second.size());
121 internal_assert(prefetch_box.size() == bounds.size());
122
123 if (p.strategy == PrefetchBoundStrategy::Clamp) {
124 prefetch_box = box_intersection(prefetch_box, bounds);
125 } else if (p.strategy == PrefetchBoundStrategy::GuardWithIf) {
126 Expr predicate = prefetch_box.used.defined() ? prefetch_box.used : const_true();
127 for (size_t i = 0; i < bounds.size(); ++i) {
128 predicate = predicate && (prefetch_box[i].min >= bounds[i].min) &&
129 (prefetch_box[i].max <= bounds[i].max);
130 }
131 prefetch_box.used = simplify(predicate);
132 } else {
133 internal_assert(p.strategy == PrefetchBoundStrategy::NonFaulting);
134 // Assume the prefetch won't fault when accessing region
135 // outside the bounds.
136 }
137
138 // Construct the region to be prefetched.
139 Region new_bounds;
140 for (size_t i = 0; i < prefetch_box.size(); i++) {
141 Expr extent = prefetch_box[i].max - prefetch_box[i].min + 1;
142 new_bounds.push_back(Range(simplify(prefetch_box[i].min), simplify(extent)));
143 }
144 Expr condition = op->condition;
145 if (prefetch_box.maybe_unused()) {
146 condition = simplify(prefetch_box.used && condition);
147 }
148 internal_assert(!new_bounds.empty());
149 return Prefetch::make(op->name, op->types, new_bounds, op->prefetch, condition, body);
150 }
151
152 if (!body.same_as(op->body)) {
153 return Prefetch::make(op->name, op->types, op->bounds, op->prefetch, op->condition, body);
154 } else if (op->bounds.empty()) {
155 // Remove the Prefetch IR since it is prefetching an empty region
156 user_warning << "Removing prefetch of " << p.name
157 << " within loop nest of " << p.var << " (offset: "
158 << p.offset << ") since it is not used at all.\n";
159 return body;
160 } else {
161 return op;
162 }
163 }

166class InjectPlaceholderPrefetch : public IRMutator {
167public:
168 InjectPlaceholderPrefetch(const map<string, Function> &e, const string &prefix,
169 const vector<PrefetchDirective> &prefetches)
170 : env(e), prefix(prefix), prefetch_list(prefetches) {
171 }
172
173private:
174 const map<string, Function> &env;
175 const string &prefix;
176 const vector<PrefetchDirective> &prefetch_list;
177
178 using IRMutator::visit;
179
180 Stmt add_placeholder_prefetch(const string &loop_var, PrefetchDirective p, const Stmt &body) {
181 debug(5) << "...Injecting placeholder prefetch for " << loop_var << "\n";
182 p.var = loop_var;
183 internal_assert(body.defined());
184 if (p.param.defined()) {
185 return Prefetch::make(p.name, {p.param.type()}, Region(), p, const_true(), body);
186 } else {
187 const auto &it = env.find(p.name);
188 internal_assert(it != env.end());
189 return Prefetch::make(p.name, it->second.output_types(), Region(), p, const_true(), body);
190 }
191 }
192
193 Stmt visit(const For *op) override {
194 Stmt body = mutate(op->body);
195
196 if (!prefetch_list.empty() && starts_with(op->name, prefix)) {
197 // If there are multiple prefetches of the same Func or ImageParam,
198 // use the most recent one
199 set<string> seen;
200 for (int i = prefetch_list.size() - 1; i >= 0; --i) {
201 const PrefetchDirective &p = prefetch_list[i];
202 if (!ends_with(op->name, "." + p.var) || (seen.find(p.name) != seen.end())) {
203 continue;
204 }
205 seen.insert(p.name);
206
207 body = add_placeholder_prefetch(op->name, p, body);
208 }
209 }
210
211 Stmt stmt;
212 if (!body.same_as(op->body)) {
213 stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
214 } else {
215 stmt = op;
216 }
217 return stmt;
218 }

221// Reduce the prefetch dimension if bigger than 'max_dim'. It keeps the 'max_dim'
222// innermost dimensions and replaces the rests with for-loops.
223class ReducePrefetchDimension : public IRMutator {
224 using IRMutator::visit;
225
226 size_t max_dim;
227
228 Stmt visit(const Evaluate *op) override {
229 Stmt stmt = IRMutator::visit(op);
230 op = stmt.as<Evaluate>();
231 internal_assert(op);
232 const Call *prefetch = Call::as_intrinsic(op->value, {Call::prefetch});
233
234 // TODO(psuriana): Ideally, we want to keep the loop size minimal to
235 // minimize the number of prefetch calls. We probably want to lift
236 // the dimensions with larger strides and keep the smaller ones in
237 // the prefetch call.
238
239 size_t max_arg_size = 2 + 2 * max_dim; // Prefetch: {base, offset, extent0, stride0, extent1, stride1, ...}
240 if (prefetch && (prefetch->args.size() > max_arg_size)) {
241 const Variable *base = prefetch->args[0].as<Variable>();
242 internal_assert(base && base->type.is_handle());
243
244 vector<string> index_names;
245 Expr new_offset = prefetch->args[1];
246 for (size_t i = max_arg_size; i < prefetch->args.size(); i += 2) {
247 Expr stride = prefetch->args[i + 1];
248 string index_name = "prefetch_reduce_" + base->name + "." + std::to_string((i - 1) / 2);
249 index_names.push_back(index_name);
250 new_offset += Variable::make(Int(32), index_name) * stride;
251 }
252
253 vector<Expr> args = {base, new_offset};
254 for (size_t i = 2; i < max_arg_size; ++i) {
255 args.push_back(prefetch->args[i]);
256 }
257
258 stmt = Evaluate::make(Call::make(prefetch->type, Call::prefetch, args, Call::Intrinsic));
259 for (size_t i = 0; i < index_names.size(); ++i) {
260 stmt = For::make(index_names[i], 0, prefetch->args[(i + max_dim) * 2 + 2],
261 ForType::Serial, DeviceAPI::None, stmt);
262 }
263 debug(5) << "\nReduce prefetch to " << max_dim << " dim:\n"
264 << "Before:\n"
265 << Expr(prefetch) << "\nAfter:\n"
266 << stmt << "\n";
267 }
268 return stmt;
269 }
270
271public:
272 ReducePrefetchDimension(size_t dim)
273 : max_dim(dim) {
274 }
275};

277// If the prefetched data is larger than 'max_byte_size', we need to tile the
278// prefetch. This will split the prefetch call into multiple calls by adding
279// an outer for-loop around the prefetch.
280class SplitPrefetch : public IRMutator {
281 using IRMutator::visit;
282
283 Expr max_byte_size;
284
285 Stmt visit(const Evaluate *op) override {
286 Stmt stmt = IRMutator::visit(op);
287 op = stmt.as<Evaluate>();
288 internal_assert(op);
289 if (const Call *prefetch = Call::as_intrinsic(op->value, {Call::prefetch})) {
290 const Variable *base = prefetch->args[0].as<Variable>();
291 internal_assert(base && base->type.is_handle());
292
293 int elem_size = prefetch->type.bytes();
294
295 vector<string> index_names;
296 vector<Expr> extents;
297 Expr new_offset = prefetch->args[1];
298 for (size_t i = 2; i < prefetch->args.size(); i += 2) {
299 Expr extent = prefetch->args[i];
300 Expr stride = prefetch->args[i + 1];
301 Expr stride_bytes = stride * elem_size;
302
303 string index_name = "prefetch_split_" + base->name + "." + std::to_string((i - 1) / 2);
304 index_names.push_back(index_name);
305
306 Expr is_negative_stride = (stride < 0);
307 Expr outer_var = Variable::make(Int(32), index_name);
308 Expr outer_extent;
309 if (can_prove(max_byte_size < stride_bytes) || can_prove(max_byte_size < -stride_bytes)) {
310 // If 'max_byte_size' is smaller than the absolute value of the
311 // stride bytes, we can only prefetch one element per iteration.
312 outer_extent = extent;
313 new_offset += outer_var * stride_bytes;
314 } else {
315 // Otherwise, we just prefetch 'max_byte_size' per iteration.
316 Expr abs_stride_bytes = Call::make(stride_bytes.type(), Call::abs, {stride_bytes}, Call::PureIntrinsic);
317 outer_extent = simplify((extent * abs_stride_bytes + max_byte_size - 1) / max_byte_size);
318 new_offset += outer_var * simplify(select(is_negative_stride, -max_byte_size, max_byte_size));
319 }
320 extents.push_back(outer_extent);
321 }
322
323 vector<Expr> args = {base, new_offset, Expr(1), simplify(max_byte_size / elem_size)};
324 stmt = Evaluate::make(Call::make(prefetch->type, Call::prefetch, args, Call::Intrinsic));
325 for (size_t i = 0; i < index_names.size(); ++i) {
326 stmt = For::make(index_names[i], 0, extents[i],
327 ForType::Serial, DeviceAPI::None, stmt);
328 }
329 debug(5) << "\nSplit prefetch to max of " << max_byte_size << " bytes:\n"
330 << "Before:\n"
331 << Expr(prefetch) << "\nAfter:\n"
332 << stmt << "\n";
333 }
334 return stmt;
335 }
336
337public:
338 SplitPrefetch(Expr bytes)
339 : max_byte_size(std::move(bytes)) {
340 }
341};

358Stmt reduce_prefetch_dimension(Stmt stmt, const Target &t) {
359 size_t max_dim = 0;
360 Expr max_byte_size;
361
362 // Hexagon's prefetch takes in a range of address and can be maximum of
363 // two dimension. Other architectures generate one prefetch per cache line.
364 if (t.has_feature(Target::HVX)) {
365 max_dim = 2;
366 } else if (t.arch == Target::ARM) {
367 // ARM's cache line size can be 32 or 64 bytes and it can switch the
368 // size at runtime. To be safe, we just use 32 bytes.
369 max_dim = 1;
370 max_byte_size = 32;
371 } else {
372 max_dim = 1;
373 max_byte_size = 64;
374 }
375 internal_assert(max_dim > 0);
376
377 stmt = ReducePrefetchDimension(max_dim).mutate(stmt);
378 if (max_byte_size.defined()) {
379 // If the max byte size is specified, we may need to tile
380 // the prefetch
381 stmt = SplitPrefetch(max_byte_size).mutate(stmt);
382 }
383 return stmt;
384}
