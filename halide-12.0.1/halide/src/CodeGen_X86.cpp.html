24// Populate feature flags in a target according to those implied by
25// existing flags, so that instruction patterns can just check for the
26// oldest feature flag that supports an instruction.
27Target complete_x86_target(Target t) {
28 if (t.has_feature(Target::AVX512_SapphireRapids)) {
29 t.set_feature(Target::AVX512_Cannonlake);
30 }
31 if (t.has_feature(Target::AVX512_Cannonlake)) {
32 t.set_feature(Target::AVX512_Skylake);
33 }
34 if (t.has_feature(Target::AVX512_Cannonlake) ||
35 t.has_feature(Target::AVX512_Skylake) ||
36 t.has_feature(Target::AVX512_KNL)) {
37 t.set_feature(Target::AVX2);
38 }
39 if (t.has_feature(Target::AVX2)) {
40 t.set_feature(Target::AVX);
41 }
42 if (t.has_feature(Target::AVX)) {
43 t.set_feature(Target::SSE41);
44 }
45 return t;
46}

205void CodeGen_X86::init_module() {
206 CodeGen_Posix::init_module();
207
208 for (const x86Intrinsic &i : intrinsic_defs) {
209 if (i.feature != Target::FeatureEnd && !target.has_feature(i.feature)) {
210 continue;
211 }
212
213 Type ret_type = i.ret_type;
214 vector<Type> arg_types;
215 arg_types.reserve(max_intrinsic_args);
216 for (halide_type_t j : i.arg_types) {
217 if (j.bits == 0) {
218 break;
219 }
220 arg_types.emplace_back(j);
221 }
222
223 auto *fn = declare_intrin_overload(i.name, ret_type, i.intrin_name, std::move(arg_types));
224 fn->addFnAttr(llvm::Attribute::ReadNone);
225 fn->addFnAttr(llvm::Attribute::NoUnwind);
226 }
227}

229// i32(i16_a)*i32(i16_b) +/- i32(i16_c)*i32(i16_d) can be done by
230// interleaving a, c, and b, d, and then using dot_product.
231bool should_use_dot_product(const Expr &a, const Expr &b, vector<Expr> &result) {
232 Type t = a.type();
233 internal_assert(b.type() == t);
234
235 if (!(t.is_int() && t.bits() == 32 && t.lanes() >= 4)) {
236 return false;
237 }
238
239 const Call *ma = Call::as_intrinsic(a, {Call::widening_mul});
240 const Call *mb = Call::as_intrinsic(b, {Call::widening_mul});
241 // dot_product can't handle mixed type widening muls.
242 if (ma && ma->args[0].type() != ma->args[1].type()) {
243 return false;
244 }
245 if (mb && mb->args[0].type() != mb->args[1].type()) {
246 return false;
247 }
248 // If the operands are widening shifts, we might be able to treat these as
249 // multiplies.
250 const Call *sa = Call::as_intrinsic(a, {Call::widening_shift_left});
251 const Call *sb = Call::as_intrinsic(b, {Call::widening_shift_left});
252 if (sa && !is_const(sa->args[1])) {
253 sa = nullptr;
254 }
255 if (sb && !is_const(sb->args[1])) {
256 sb = nullptr;
257 }
258 if ((ma || sa) && (mb || sb)) {
259 Expr a0 = ma ? ma->args[0] : sa->args[0];
260 Expr a1 = ma ? ma->args[1] : lossless_cast(sa->args[0].type(), simplify(make_const(sa->type, 1) << sa->args[1]));
261 Expr b0 = mb ? mb->args[0] : sb->args[0];
262 Expr b1 = mb ? mb->args[1] : lossless_cast(sb->args[0].type(), simplify(make_const(sb->type, 1) << sb->args[1]));
263 if (a1.defined() && b1.defined()) {
264 std::vector<Expr> args = {a0, a1, b0, b1};
265 result.swap(args);
266 return true;
267 }
268 }
269 return false;
270}

272void CodeGen_X86::visit(const Add *op) {
273 vector<Expr> matches;
274 if (should_use_dot_product(op->a, op->b, matches)) {
275 Expr ac = Shuffle::make_interleave({matches[0], matches[2]});
276 Expr bd = Shuffle::make_interleave({matches[1], matches[3]});
277 value = call_overloaded_intrin(op->type, "dot_product", {ac, bd});
278 if (value) {
279 return;
280 }
281 }
282 CodeGen_Posix::visit(op);
283}

285void CodeGen_X86::visit(const Sub *op) {
286 vector<Expr> matches;
287 if (should_use_dot_product(op->a, op->b, matches)) {
288 // Negate one of the factors in the second expression
289 Expr negative_2 = lossless_negate(matches[2]);
290 Expr negative_3 = lossless_negate(matches[3]);
291 if (negative_2.defined() || negative_3.defined()) {
292 if (negative_2.defined()) {
293 matches[2] = negative_2;
294 } else {
295 matches[3] = negative_3;
296 }
297 Expr ac = Shuffle::make_interleave({matches[0], matches[2]});
298 Expr bd = Shuffle::make_interleave({matches[1], matches[3]});
299 value = call_overloaded_intrin(op->type, "dot_product", {ac, bd});
300 if (value) {
301 return;
302 }
303 }
304 }
305 CodeGen_Posix::visit(op);
306}

308void CodeGen_X86::visit(const GT *op) {
309 Type t = op->a.type();
310
311 if (t.is_vector() &&
312 upgrade_type_for_arithmetic(t) == t) {
313 // Non-native vector widths get legalized poorly by llvm. We
314 // split it up ourselves.
315
316 int slice_size = vector_lanes_for_slice(t);
317
318 Value *a = codegen(op->a), *b = codegen(op->b);
319 vector<Value *> result;
320 for (int i = 0; i < op->type.lanes(); i += slice_size) {
321 Value *sa = slice_vector(a, i, slice_size);
322 Value *sb = slice_vector(b, i, slice_size);
323 Value *slice_value;
324 if (t.is_float()) {
325 slice_value = builder->CreateFCmpOGT(sa, sb);
326 } else if (t.is_int()) {
327 slice_value = builder->CreateICmpSGT(sa, sb);
328 } else {
329 slice_value = builder->CreateICmpUGT(sa, sb);
330 }
331 result.push_back(slice_value);
332 }
333
334 value = concat_vectors(result);
335 value = slice_vector(value, 0, t.lanes());
336 } else {
337 CodeGen_Posix::visit(op);
338 }
339}

388void CodeGen_X86::visit(const Select *op) {
389 if (op->condition.type().is_vector()) {
390 // LLVM handles selects on vector conditions much better at native width
391 Value *cond = codegen(op->condition);
392 Value *true_val = codegen(op->true_value);
393 Value *false_val = codegen(op->false_value);
394 Type t = op->true_value.type();
395 int slice_size = vector_lanes_for_slice(t);
396
397 vector<Value *> result;
398 for (int i = 0; i < t.lanes(); i += slice_size) {
399 Value *st = slice_vector(true_val, i, slice_size);
400 Value *sf = slice_vector(false_val, i, slice_size);
401 Value *sc = slice_vector(cond, i, slice_size);
402 Value *slice_value = builder->CreateSelect(sc, st, sf);
403 result.push_back(slice_value);
404 }
405
406 value = concat_vectors(result);
407 value = slice_vector(value, 0, t.lanes());
408 } else {
409 CodeGen_Posix::visit(op);
410 }

413void CodeGen_X86::visit(const Cast *op) {
414
415 if (!op->type.is_vector()) {
416 // We only have peephole optimizations for vectors in here.
417 CodeGen_Posix::visit(op);
418 return;
419 }
420
421 struct Pattern {
422 string intrin;
423 Expr pattern;
424 };
425
426 // clang-format off
427 static Pattern patterns[] = {
428 // This isn't rounding_multiply_quantzied(i16, i16, 15) because it doesn't
429 // saturate the result.
430 {"pmulhrs", i16(rounding_shift_right(widening_mul(wild_i16x_, wild_i16x_), 15))},
431
432 {"saturating_narrow", i16_sat(wild_i32x_)},
433 {"saturating_narrow", u16_sat(wild_i32x_)},
434 {"saturating_narrow", i8_sat(wild_i16x_)},
435 {"saturating_narrow", u8_sat(wild_i16x_)},
436
437 {"f32_to_bf16", bf16(wild_f32x_)},
438 };
439 // clang-format on
440
441 vector<Expr> matches;
442 for (const Pattern &p : patterns) {
443 if (expr_match(p.pattern, op, matches)) {
444 value = call_overloaded_intrin(op->type, p.intrin, matches);
445 if (value) {
446 return;
447 }
448 }
449 }
450
451 if (const Call *mul = Call::as_intrinsic(op->value, {Call::widening_mul})) {
452 if (op->value.type().bits() < op->type.bits() && op->type.bits() <= 32) {
453 // LLVM/x86 really doesn't like 8 -> 16 bit multiplication. If we're
454 // widening to 32-bits after a widening multiply, LLVM prefers to see a
455 // widening multiply directly to 32-bits. This may result in extra
456 // casts, so simplify to remove them.
457 value = codegen(simplify(Mul::make(Cast::make(op->type, mul->args[0]), Cast::make(op->type, mul->args[1]))));
458 return;
459 }
460 }
461
462 CodeGen_Posix::visit(op);
463}

483 // clang-format on

485 vector<Expr> matches;
486 for (size_t i = 0; i < sizeof(patterns) / sizeof(patterns[0]); i++) {
487 const Pattern &pattern = patterns[i];
488 if (expr_match(pattern.pattern, op, matches)) {
489 value = call_overloaded_intrin(op->type, pattern.intrin, matches);
490 if (value) {
491 return;
492 }
493 }
494 }
495
496 CodeGen_Posix::visit(op);
497}

499void CodeGen_X86::codegen_vector_reduce(const VectorReduce *op, const Expr &init) {
500 if (op->op != VectorReduce::Add && op->op != VectorReduce::SaturatingAdd) {
501 CodeGen_Posix::codegen_vector_reduce(op, init);
502 return;
503 }
504 const int factor = op->value.type().lanes() / op->type.lanes();
505
506 struct Pattern {
507 VectorReduce::Operator reduce_op;
508 int factor;
509 Expr pattern;
510 const char *intrin;
511 Type narrow_type;
512 uint32_t flags = 0;
513 enum {
514 CombineInit = 1 << 0,
515 SwapOperands = 1 << 1,
516 };
517 };
518 // clang-format off
519 // These patterns are roughly sorted "best to worst", in case there are two
520 // patterns that match the expression.
521 static const Pattern patterns[] = {
522 // 4-way dot products
523 {VectorReduce::Add, 4, i32(widening_mul(wild_u8x_, wild_i8x_)), "dot_product", {}, Pattern::CombineInit},
524 {VectorReduce::Add, 4, i32(widening_mul(wild_i8x_, wild_u8x_)), "dot_product", {}, Pattern::CombineInit | Pattern::SwapOperands},
525 {VectorReduce::SaturatingAdd, 4, i32(widening_mul(wild_u8x_, wild_i8x_)), "saturating_dot_product", {}, Pattern::CombineInit},
526 {VectorReduce::SaturatingAdd, 4, i32(widening_mul(wild_i8x_, wild_u8x_)), "saturating_dot_product", {}, Pattern::CombineInit | Pattern::SwapOperands},
527
528 // 2-way dot products
529 {VectorReduce::Add, 2, i32(widening_mul(wild_i8x_, wild_i8x_)), "dot_product", Int(16)},
530 {VectorReduce::Add, 2, i32(widening_mul(wild_i8x_, wild_u8x_)), "dot_product", Int(16)},
531 {VectorReduce::Add, 2, i32(widening_mul(wild_u8x_, wild_i8x_)), "dot_product", Int(16)},
532 {VectorReduce::Add, 2, i32(widening_mul(wild_u8x_, wild_u8x_)), "dot_product", Int(16)},
533 {VectorReduce::SaturatingAdd, 2, i32(widening_mul(wild_u8x_, wild_i8x_)), "saturating_dot_product", {}, Pattern::CombineInit},
534 {VectorReduce::SaturatingAdd, 2, i32(widening_mul(wild_i8x_, wild_u8x_)), "saturating_dot_product", {}, Pattern::CombineInit | Pattern::SwapOperands},
535 {VectorReduce::SaturatingAdd, 2, widening_mul(wild_u8x_, wild_i8x_), "saturating_dot_product"},
536 {VectorReduce::SaturatingAdd, 2, widening_mul(wild_i8x_, wild_u8x_), "saturating_dot_product", {}, Pattern::SwapOperands},
537
538 {VectorReduce::Add, 2, i32(widening_mul(wild_i16x_, wild_i16x_)), "dot_product", {}, Pattern::CombineInit},
539 {VectorReduce::Add, 2, i32(widening_mul(wild_i16x_, wild_i16x_)), "dot_product", Int(16)},
540 {VectorReduce::SaturatingAdd, 2, i32(widening_mul(wild_i16x_, wild_i16x_)), "saturating_dot_product", {}, Pattern::CombineInit},
541
542 {VectorReduce::Add, 2, wild_f32x_ * wild_f32x_, "dot_product", BFloat(16), Pattern::CombineInit},
543
544 // One could do a horizontal widening addition with
545 // dot_product against a vector of ones. Currently disabled
546 // because I haven't found case where it's clearly better.
547 };
548 // clang-format on
549
550 std::vector<Expr> matches;
551 for (const Pattern &p : patterns) {
552 if (op->op != p.reduce_op || p.factor != factor) {
553 continue;
554 }
555 if (expr_match(p.pattern, op->value, matches)) {
556 Expr a = matches[0];
557 Expr b = matches[1];
558 if (p.flags & Pattern::SwapOperands) {
559 std::swap(a, b);
560 }
561 if (p.narrow_type.bits() > 0) {
562 a = lossless_cast(p.narrow_type.with_lanes(a.type().lanes()), a);
563 b = lossless_cast(p.narrow_type.with_lanes(b.type().lanes()), b);
564 }
565 if (!a.defined() || !b.defined()) { continue; }
566
567 if (init.defined() && (p.flags & Pattern::CombineInit)) {
568 value = call_overloaded_intrin(op->type, p.intrin, {init, a, b});
569 if (value) { return; }
570 } else {
571 value = call_overloaded_intrin(op->type, p.intrin, {a, b});
572 if (value) {
573 if (init.defined()) {
574 Value *x = value;
575 Value *y = codegen(init);
576 value = builder->CreateAdd(x, y);
577 }
578 return;
579 }
580 }
581 }
582 }
583
584 CodeGen_Posix::codegen_vector_reduce(op, init);
585}

587string CodeGen_X86::mcpu() const {
588 if (target.has_feature(Target::AVX512_SapphireRapids)) {
589#if LLVM_VERSION >= 120
590 return "sapphirerapids";
591#else
592 user_error << "AVX512 SapphireRapids requires LLVM 12 or later.";
593 return "";
594#endif
595 } else if (target.has_feature(Target::AVX512_Cannonlake)) {
596 return "cannonlake";
597 } else if (target.has_feature(Target::AVX512_Skylake)) {
598 return "skylake-avx512";
599 } else if (target.has_feature(Target::AVX512_KNL)) {
600 return "knl";
601 } else if (target.has_feature(Target::AVX2)) {
602 return "haswell";
603 } else if (target.has_feature(Target::AVX)) {
604 return "corei7-avx";
605 } else if (target.has_feature(Target::SSE41)) {
606 // We want SSE4.1 but not SSE4.2, hence "penryn" rather than "corei7"
607 return "penryn";
608 } else {
609 // Default should not include SSSE3, hence "k8" rather than "core2"
610 return "k8";
611 }
612}

674int CodeGen_X86::vector_lanes_for_slice(const Type &t) const {
675 // We don't want to pad all the way out to natural_vector_size,
676 // because llvm generates crappy code. Better to use a smaller
677 // type if we can.
678 int vec_bits = t.lanes() * t.bits();
679 int natural_vec_bits = target.natural_vector_size(t) * t.bits();
680 // clang-format off
681 int slice_bits = ((vec_bits > 256 && natural_vec_bits > 256) ? 512 :
682 (vec_bits > 128 && natural_vec_bits > 128) ? 256 :
683 128);
684 // clang-format on
685 return slice_bits / t.bits();
686}

688llvm::Type *CodeGen_X86::llvm_type_of(const Type &t) const {
689 if (t.is_float() && t.bits() < 32) {
690 // LLVM as of August 2019 has all sorts of issues in the x86
691 // backend for half types. It injects expensive calls to
692 // convert between float and half for seemingly no reason
693 // (e.g. to do a select), and bitcasting to int16 doesn't
694 // help, because it simplifies away the bitcast for you.
695 // See: https://bugs.llvm.org/show_bug.cgi?id=43065
696 // and: https://github.com/halide/Halide/issues/4166
697 return llvm_type_of(t.with_code(halide_type_uint));
698 } else {
699 return CodeGen_Posix::llvm_type_of(t);
700 }
701}
