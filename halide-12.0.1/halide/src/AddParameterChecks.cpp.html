17// Find all the externally referenced scalar parameters
18class FindParameters : public IRGraphVisitor {
19public:
20 map<string, Parameter> params;
21
22 using IRGraphVisitor::visit;
23
24 void visit(const Variable *op) override {
25 if (op->param.defined()) {
26 params[op->name] = op->param;
27 }
28 }
29};
30
31} // namespace

33// Insert checks to make sure that parameters are within their
34// declared range.
35Stmt add_parameter_checks(const vector<Stmt> &preconditions, Stmt s, const Target &t) {
36
37 // First, find all the parameters
38 FindParameters finder;
39 s.accept(&finder);
40
41 map<string, Expr> replace_with_constrained;
42 vector<pair<string, Expr>> lets;
43
44 struct ParamAssert {
45 Expr condition;
46 Expr value, limit_value;
47 string param_name;
48 };
49
50 vector<ParamAssert> asserts;
51
52 // Make constrained versions of the params
53 for (pair<const string, Parameter> &i : finder.params) {
54 Parameter param = i.second;
55
56 if (!param.is_buffer() &&
57 (param.min_value().defined() ||
58 param.max_value().defined())) {
59
60 string constrained_name = i.first + ".constrained";
61
62 Expr constrained_var = Variable::make(param.type(), constrained_name);
63 Expr constrained_value = Variable::make(param.type(), i.first, param);
64 replace_with_constrained[i.first] = constrained_var;
65
66 if (param.min_value().defined()) {
67 ParamAssert p = {
68 constrained_value >= param.min_value(),
69 constrained_value, param.min_value(),
70 param.name()};
71 asserts.push_back(p);
72 constrained_value = max(constrained_value, param.min_value());
73 }
74
75 if (param.max_value().defined()) {
76 ParamAssert p = {
77 constrained_value <= param.max_value(),
78 constrained_value, param.max_value(),
79 param.name()};
80 asserts.push_back(p);
81 constrained_value = min(constrained_value, param.max_value());
82 }
83
84 lets.emplace_back(constrained_name, constrained_value);
85 }
86 }
87
88 // Replace the params with their constrained version in the rest of the pipeline
89 s = substitute(replace_with_constrained, s);
90
91 // Inject the let statements
92 for (size_t i = 0; i < lets.size(); i++) {
93 s = LetStmt::make(lets[i].first, lets[i].second, s);
94 }
95
96 // Inject the assert statements
97 for (size_t i = 0; i < asserts.size(); i++) {
98 ParamAssert p = asserts[i];
99 // Upgrade the types to 64-bit versions for the error call
100 Type wider = p.value.type().with_bits(64);
101 p.limit_value = cast(wider, p.limit_value);
102 p.value = cast(wider, p.value);
103
104 string error_call_name = "halide_error_param";
105
106 if (p.condition.as<LE>()) {
107 error_call_name += "_too_large";
108 } else {
109 internal_assert(p.condition.as<GE>());
110 error_call_name += "_too_small";
111 }
112
113 if (wider.is_int()) {
114 error_call_name += "_i64";
115 } else if (wider.is_uint()) {
116 error_call_name += "_u64";
117 } else {
118 internal_assert(wider.is_float());
119 error_call_name += "_f64";
120 }
121
122 Expr error = Call::make(Int(32), error_call_name,
123 {p.param_name, p.value, p.limit_value},
124 Call::Extern);
125
126 s = Block::make(AssertStmt::make(p.condition, error), s);
127 }
128
129 // The unstructured assertions get checked first (because they
130 // have a custom error message associated with them), so prepend
131 // them last.
132 vector<Stmt> stmts = preconditions;
133 stmts.push_back(s);
134 return Block::make(stmts);
135}
