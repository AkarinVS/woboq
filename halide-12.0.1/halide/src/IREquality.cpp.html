13/** The class that does the work of comparing two IR nodes. */
14class IRComparer : public IRVisitor {
15public:
16 /** Different possible results of a comparison. Unknown should
17 * only occur internally due to a cache miss. */
18 enum CmpResult { Unknown,
19 Equal,
20 LessThan,
21 GreaterThan };
22
23 /** The result of the comparison. Should be Equal, LessThan, or GreaterThan. */
24 CmpResult result = Equal;
25
26 /** Compare two expressions or statements and return the
27 * result. Returns the result immediately if it is already
28 * non-zero. */
29 // @{
30 CmpResult compare_expr(const Expr &a, const Expr &b);
31 CmpResult compare_stmt(const Stmt &a, const Stmt &b);
32 // @}
33
34 /** If the expressions you're comparing may contain many repeated
35 * subexpressions, it's worth passing in a cache to use.
36 * Currently this is only done in common-subexpression
37 * elimination. */
38 IRComparer(IRCompareCache *c = nullptr)
39 : cache(c) {
40 }
41
42private:
43 Expr expr;
44 Stmt stmt;
45 IRCompareCache *cache;

104IRComparer::CmpResult IRComparer::compare_scalar(T a, T b) {
105 if (result != Equal) {
106 return result;
107 }
108
109 if (a < b) {
110 result = LessThan;
111 } else if (a > b) {
112 result = GreaterThan;
113 }
114
115 return result;
116}

118IRComparer::CmpResult IRComparer::compare_expr(const Expr &a, const Expr &b) {
119 if (result != Equal) {
120 return result;
121 }
122
123 if (a.same_as(b)) {
124 result = Equal;
125 return result;
126 }
127
128 if (!a.defined() && !b.defined()) {
129 result = Equal;
130 return result;
131 }
132
133 if (!a.defined()) {
134 result = LessThan;
135 return result;
136 }
137
138 if (!b.defined()) {
139 result = GreaterThan;
140 return result;
141 }
142
143 // If in the future we have hashes for Exprs, this is a good place
144 // to compare the hashes:
145 // if (compare_scalar(a.hash(), b.hash()) != Equal) {
146 // return result;
147 // }
148
149 if (compare_scalar(a->node_type, b->node_type) != Equal) {
150 return result;
151 }
152
153 if (compare_types(a.type(), b.type()) != Equal) {
154 return result;
155 }
156
157 // Check the cache - perhaps these exprs have already been compared and found equal.
158 if (cache && cache->contains(a, b)) {
159 result = Equal;
160 return result;
161 }
162
163 expr = a;
164 b.accept(this);
165
166 if (cache && result == Equal) {
167 cache->insert(a, b);
168 }
169
170 return result;
171}

173IRComparer::CmpResult IRComparer::compare_stmt(const Stmt &a, const Stmt &b) {
174 if (result != Equal) {
175 return result;
176 }
177
178 if (a.same_as(b)) {
179 result = Equal;
180 return result;
181 }
182
183 if (!a.defined() && !b.defined()) {
184 result = Equal;
185 return result;
186 }
187
188 if (!a.defined()) {
189 result = LessThan;
190 return result;
191 }
192
193 if (!b.defined()) {
194 result = GreaterThan;
195 return result;
196 }
197
198 if (compare_scalar(a->node_type, b->node_type) != Equal) {
199 return result;
200 }
201
202 stmt = a;
203 b.accept(this);
204
205 return result;
206}

208IRComparer::CmpResult IRComparer::compare_types(Type a, Type b) {
209 if (result != Equal) {
210 return result;
211 }
212
213 compare_scalar(a.code(), b.code());
214 compare_scalar(a.bits(), b.bits());
215 compare_scalar(a.lanes(), b.lanes());
216
217 if (result != Equal) {
218 return result;
219 }
220
221 const halide_handle_cplusplus_type *ha = a.handle_type;
222 const halide_handle_cplusplus_type *hb = b.handle_type;
223
224 if (ha == hb) {
225 // Same handle type, or both not handles, or both void *
226 return result;
227 }
228
229 if (ha == nullptr) {
230 // void* < T*
231 result = LessThan;
232 return result;
233 }
234
235 if (hb == nullptr) {
236 // T* > void*
237 result = GreaterThan;
238 return result;
239 }
240
241 // They're both non-void handle types with distinct type info
242 // structs. We now need to distinguish between different C++
243 // pointer types (e.g. char * vs const float *). If would be nice
244 // if the structs were unique per C++ type. Then comparing the
245 // pointers above would be sufficient. Unfortunately, different
246 // shared libraries in the same process each create a distinct
247 // struct for the same type. We therefore have to do a deep
248 // comparison of the type info fields.
249
250 compare_scalar(ha->reference_type, hb->reference_type);
251 compare_names(ha->inner_name.name, hb->inner_name.name);
252 compare_scalar(ha->inner_name.cpp_type_type, hb->inner_name.cpp_type_type);
253 compare_scalar(ha->namespaces.size(), hb->namespaces.size());
254 compare_scalar(ha->enclosing_types.size(), hb->enclosing_types.size());
255 compare_scalar(ha->cpp_type_modifiers.size(), hb->cpp_type_modifiers.size());
256
257 if (result != Equal) {
258 return result;
259 }
260
261 for (size_t i = 0; i < ha->namespaces.size(); i++) {
262 compare_names(ha->namespaces[i], hb->namespaces[i]);
263 }
264
265 if (result != Equal) {
266 return result;
267 }
268
269 for (size_t i = 0; i < ha->enclosing_types.size(); i++) {
270 compare_scalar(ha->enclosing_types[i].cpp_type_type,
271 hb->enclosing_types[i].cpp_type_type);
272 compare_names(ha->enclosing_types[i].name,
273 hb->enclosing_types[i].name);
274 }
275
276 if (result != Equal) {
277 return result;
278 }
279
280 for (size_t i = 0; i < ha->cpp_type_modifiers.size(); i++) {
281 compare_scalar(ha->cpp_type_modifiers[i],
282 hb->cpp_type_modifiers[i]);
283 }
284
285 return result;
286}

288IRComparer::CmpResult IRComparer::compare_names(const string &a, const string &b) {
289 if (result != Equal) {
290 return result;
291 }
292
293 int string_cmp = a.compare(b);
294 if (string_cmp < 0) {
295 result = LessThan;
296 } else if (string_cmp > 0) {
297 result = GreaterThan;
298 }
299
300 return result;
301}

303IRComparer::CmpResult IRComparer::compare_expr_vector(const vector<Expr> &a, const vector<Expr> &b) {
304 if (result != Equal) {
305 return result;
306 }
307
308 compare_scalar(a.size(), b.size());
309 for (size_t i = 0; (i < a.size()) && result == Equal; i++) {
310 compare_expr(a[i], b[i]);
311 }
312
313 return result;
314}

660IRComparer::CmpResult flip_result(IRComparer::CmpResult r) {
661 switch (r) {
662 case IRComparer::LessThan:
663 return IRComparer::GreaterThan;
664 case IRComparer::Equal:
665 return IRComparer::Equal;
666 case IRComparer::GreaterThan:
667 return IRComparer::LessThan;
668 case IRComparer::Unknown:
669 return IRComparer::Unknown;
670 }
671 return IRComparer::Unknown;
672}

674void check_equal(const Expr &a, const Expr &b) {
675 IRCompareCache cache(5);
676 IRComparer::CmpResult r = IRComparer(&cache).compare_expr(a, b);
677 internal_assert(r == IRComparer::Equal)
678 << "Error in ir_equality_test: " << r
679 << " instead of " << IRComparer::Equal
680 << " when comparing:\n"
681 << a
682 << "\nand\n"
683 << b << "\n";
684}
685
686void check_not_equal(const Expr &a, const Expr &b) {
687 IRCompareCache cache(5);
688 IRComparer::CmpResult r1 = IRComparer(&cache).compare_expr(a, b);
689 IRComparer::CmpResult r2 = IRComparer(&cache).compare_expr(b, a);
690 internal_assert(r1 != IRComparer::Equal &&
691 r1 != IRComparer::Unknown &&
692 flip_result(r1) == r2)
693 << "Error in ir_equality_test: " << r1
694 << " is not the opposite of " << r2
695 << " when comparing:\n"
696 << a
697 << "\nand\n"
698 << b << "\n";
699}

703void ir_equality_test() {
704 Expr x = Variable::make(Int(32), "x");
705 check_equal(Ramp::make(x, 4, 3), Ramp::make(x, 4, 3));
706 check_not_equal(Ramp::make(x, 2, 3), Ramp::make(x, 4, 3));
707
708 check_equal(x, Variable::make(Int(32), "x"));
709 check_not_equal(x, Variable::make(Int(32), "y"));
710
711 // Something that will hang if IREquality has poor computational
712 // complexity.
713 Expr e1 = x, e2 = x;
714 for (int i = 0; i < 100; i++) {
715 e1 = e1 * e1 + e1;
716 e2 = e2 * e2 + e2;
717 }
718 check_equal(e1, e2);
719 // These are only discovered to be not equal way down the tree:
720 e2 = e2 * e2 + e2;
721 check_not_equal(e1, e2);
722
723 debug(0) << "ir_equality_test passed\n";
724}
