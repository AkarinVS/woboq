18string sanitize_name(const string &name) {
19 ostringstream oss;
20 for (size_t i = 0; i < name.size(); i++) {
21 if (name[i] == '.' || name[i] == '_') {
22 oss << "_";
23 } else if (!isalnum(name[i])) {
24 oss << "_" << (int)name[i];
25 } else {
26 oss << name[i];
27 }
28 }
29 return oss.str();
30}

32string remove_namespaces(const string &name) {
33 size_t i = name.find_last_of(':');
34 if (i == string::npos) {
35 return name;
36 } else {
37 return name.substr(i + 1);
38 }

41bool can_convert(const LoweredArgument *arg) {
42 if (arg->type.is_handle()) {
43 if (arg->name == "__user_context") {
44 /* __user_context is a void* pointer to a user supplied memory region.
45 * We allow the Python callee to pass PyObject* pointers to that. */
46 return true;
47 } else {
48 return false;
49 }
50 }
51 if (arg->type.is_vector()) {
52 return false;
53 }
54 if (arg->type.is_float() && arg->type.bits() != 32 && arg->type.bits() != 64) {
55 return false;
56 }
57 if (arg->is_buffer() && arg->type.bits() == 1) {
58 // The Python buffer API doesn't support bit arrays.
59 return false;
60 }
61 if ((arg->type.is_int() || arg->type.is_uint()) &&
62 arg->type.bits() != 1 &&
63 arg->type.bits() != 8 && arg->type.bits() != 16 &&
64 arg->type.bits() != 32 && arg->type.bits() != 64) {
65 return false;
66 }
67 return true;
68}

70std::pair<string, string> print_type(const LoweredArgument *arg) {
71 // Excluded by can_convert() above:
72 internal_assert(!arg->type.is_vector());
73
74 if (arg->type.is_handle()) {
75 /* Handles can be any pointer. However, from Python, all you can pass to
76 * a function is a PyObject*, so we can restrict to that. */
77 return std::make_pair("O", "PyObject*");
78 } else if (arg->is_buffer()) {
79 return std::make_pair("O", "PyObject*");
80 } else if (arg->type.is_float() && arg->type.bits() == 32) {
81 return std::make_pair("f", "float");
82 } else if (arg->type.is_float() && arg->type.bits() == 64) {
83 return std::make_pair("d", "double");
84 } else if (arg->type.bits() == 1) {
85 // "b" expects an unsigned char, so we assume that bool == uint8.
86 return std::make_pair("b", "bool");
87 } else if (arg->type.is_int() && arg->type.bits() == 64) {
88 return std::make_pair("L", "long long");
89 } else if (arg->type.is_uint() && arg->type.bits() == 64) {
90 return std::make_pair("K", "unsigned long long");
91 } else if (arg->type.is_int()) {
92 return std::make_pair("i", "int");
93 } else if (arg->type.is_uint()) {
94 return std::make_pair("I", "unsigned int");
95 } else {
96 return std::make_pair("E", "unknown type");
97 }
98}

102void PythonExtensionGen::convert_buffer(const string &name, const LoweredArgument *arg) {
103 internal_assert(arg->is_buffer());
104 internal_assert(arg->dimensions);
105 dest << " halide_buffer_t buffer_" << name << ";\n";
106 dest << " halide_dimension_t dimensions_" << name << "[" << (int)arg->dimensions << "];\n";
107 dest << " Py_buffer view_" << name << ";\n";
108 dest << " if (_convert_py_buffer_to_halide(";
109 dest << /*pyobj*/ "py_" << name << ", ";
110 dest << /*dimensions*/ (int)arg->dimensions << ", ";
111 dest << /*flags*/ (arg->is_output() ? "PyBUF_WRITABLE" : "0") << ", ";
112 dest << /*dim*/ "dimensions_" << name << ", ";
113 dest << /*out*/ "&buffer_" << name << ", ";
114 dest << /*buf*/ "view_" << name << ", ";
115 dest << /*name*/ "\"" << name << "\"";
116 dest << ") < 0) {\n";
117 release_buffers(" ");
118 dest << " return nullptr;\n";
119 dest << " }\n";

132void PythonExtensionGen::compile(const Module &module) {
133 dest << "#include \"Python.h\"\n";
134 dest << "#include \"HalideRuntime.h\"\n\n";
135
136 // Emit extern decls of the Halide-generated functions we use directly
137 // into this file, so that we don't have to #include the relevant .h
138 // file directly; this simplifies certain compile/build setups (since
139 // we don't have to build files in tandem and/or get include paths right),
140 // and should be totally safe, since we are using the same codegen logic
141 // that would be in the .h file anyway.
142 {
143 CodeGen_C extern_decl_gen(dest, module.target(), CodeGen_C::CPlusPlusExternDecl);
144 extern_decl_gen.compile(module);
145 }
146
147 dest << "#define MODULE_NAME \"" << module.name() << "\"\n";
148
149 dest << R"INLINE_CODE(
150/* Older Python versions don't set up PyMODINIT_FUNC correctly. */
151#if defined(_MSC_VER)
152# define HALIDE_PYTHON_EXPORT __declspec(dllexport)
153#else
154# define HALIDE_PYTHON_EXPORT __attribute__((visibility("default")))
155#endif
156
157#ifdef __cplusplus
158extern "C" {
159#endif
160
161static
162#if !defined(_MSC_VER)
163__attribute__((unused))
164#endif
165int _convert_py_buffer_to_halide(
166 PyObject* pyobj, int dimensions, int flags,
167 halide_dimension_t* dim, // array of size `dimensions`
168 halide_buffer_t* out, Py_buffer &buf, const char* name) {
169 int ret = PyObject_GetBuffer(
170 pyobj, &buf, PyBUF_FORMAT | PyBUF_STRIDED_RO | PyBUF_ANY_CONTIGUOUS | flags);
171 if (ret < 0) {
172 return ret;
173 }
174 if (dimensions && buf.ndim != dimensions) {
175 PyErr_Format(PyExc_ValueError, "Invalid argument %s: Expected %d dimensions, got %d",
176 name, dimensions, buf.ndim);
177 PyBuffer_Release(&buf);
178 return -1;
179 }
180 /* We'll get a buffer that's either:
181 * C_CONTIGUOUS (last dimension varies the fastest, i.e., has stride=1) or
182 * F_CONTIGUOUS (first dimension varies the fastest, i.e., has stride=1).
183 * The latter is preferred, since it's already in the format that Halide
184 * needs. It can can be achieved in numpy by passing order='F' during array
185 * creation. However, if we do get a C_CONTIGUOUS buffer, flip the dimensions
186 * (transpose) so we can process it without having to reallocate.
187 */
188 int i, j, j_step;
189 if (PyBuffer_IsContiguous(&buf, 'F')) {
190 j = 0;
191 j_step = 1;
192 } else if (PyBuffer_IsContiguous(&buf, 'C')) {
193 j = buf.ndim - 1;
194 j_step = -1;
195 } else {
196 /* Python checks all dimensions and strides, so this typically indicates
197 * a bug in the array's buffer protocol. */
198 PyErr_Format(PyExc_ValueError, "Invalid buffer: neither C nor Fortran contiguous");
199 PyBuffer_Release(&buf);
200 return -1;
201 }
202 for (i = 0; i < buf.ndim; ++i, j += j_step) {
203 dim[i].min = 0;
204 dim[i].stride = (int)(buf.strides[j] / buf.itemsize); // strides is in bytes
205 dim[i].extent = (int)buf.shape[j];
206 dim[i].flags = 0;
207 if (buf.suboffsets && buf.suboffsets[i] >= 0) {
208 // Halide doesn't support arrays of pointers. But we should never see this
209 // anyway, since we specified PyBUF_STRIDED.
210 PyErr_Format(PyExc_ValueError, "Invalid buffer: suboffsets not supported");
211 PyBuffer_Release(&buf);
212 return -1;
213 }
214 }
215 if (dim[buf.ndim - 1].extent * dim[buf.ndim - 1].stride * buf.itemsize != buf.len) {
216 PyErr_Format(PyExc_ValueError, "Invalid buffer: length %ld, but computed length %ld",
217 buf.len, buf.shape[0] * buf.strides[0]);
218 PyBuffer_Release(&buf);
219 return -1;
220 }
221 *out = halide_buffer_t();
222 if (!buf.format) {
223 out->type.code = halide_type_uint;
224 out->type.bits = 8;
225 } else {
226 /* Convert struct type code. See
227 * https://docs.python.org/2/library/struct.html#module-struct */
228 char* p = buf.format;
229 while (strchr("@<>!=", *p)) {
230 p++; // ignore little/bit endian (and alignment)
231 }
232 if (*p == 'f' || *p == 'd') {
233 // 'f' and 'd' are float and double, respectively.
234 out->type.code = halide_type_float;
235 } else if (*p >= 'a' && *p <= 'z') {
236 // lowercase is signed int.
237 out->type.code = halide_type_int;
238 } else {
239 // uppercase is unsigned int.
240 out->type.code = halide_type_uint;
241 }
242 const char* type_codes = "bB?hHiIlLqQfd"; // integers and floats
243 if (strchr(type_codes, *p)) {
244 out->type.bits = (uint8_t)buf.itemsize * 8;
245 } else {
246 // We don't handle 's' and 'p' (char[]) and 'P' (void*)
247 PyErr_Format(PyExc_ValueError, "Invalid data type for %s: %s", name, buf.format);
248 PyBuffer_Release(&buf);
249 return -1;
250 }
251 }
252 out->type.lanes = 1;
253 out->dimensions = buf.ndim;
254 out->dim = dim;
255 out->host = (uint8_t*)buf.buf;
256 return 0;
257}
258
259)INLINE_CODE";
260
261 for (const auto &f : module.functions()) {
262 if (f.linkage == LinkageType::ExternalPlusMetadata) {
263 compile(f);
264 }
265 }
266
267 dest << "\n";
268 dest << "static PyMethodDef _methods[] = {\n";
269 for (const auto &f : module.functions()) {
270 if (f.linkage == LinkageType::ExternalPlusMetadata) {
271 const string basename = remove_namespaces(f.name);
272 dest << " {\"" << basename << "\", (PyCFunction)_f_" << basename
273 << ", METH_VARARGS|METH_KEYWORDS, nullptr},\n";
274 }
275 }
276 dest << " {0, 0, 0, nullptr}, // sentinel\n";
277 dest << "};\n";
278
279 dest << R"INLINE_CODE(
280static_assert(PY_MAJOR_VERSION >= 3, "Python bindings for Halide require Python 3+");
281static struct PyModuleDef _moduledef = {
282 PyModuleDef_HEAD_INIT,
283 MODULE_NAME,
284 nullptr,
285 -1,
286 _methods,
287};
288HALIDE_PYTHON_EXPORT PyObject* PyInit_)INLINE_CODE";
289
290 dest << module.name() << "(void) {";
291
292 dest << R"INLINE_CODE(
293 return PyModule_Create(&_moduledef);
294}
295
296#ifdef __cplusplus
297}
298#endif
299)INLINE_CODE";

302void PythonExtensionGen::compile(const LoweredFunc &f) {
303 const std::vector<LoweredArgument> &args = f.args;
304 const string basename = remove_namespaces(f.name);
305 std::vector<string> arg_names(args.size());
306 dest << "// " << f.name << "\n";
307 dest << "static PyObject* _f_" << basename << "(PyObject* module, PyObject* args, PyObject* kwargs) {\n";
308 for (size_t i = 0; i < args.size(); i++) {
309 arg_names[i] = sanitize_name(args[i].name);
310 if (!can_convert(&args[i])) {
311 /* Some arguments can't be converted to Python yet. In those
312 * cases, just add a dummy function that always throws an
313 * Exception. */
314 // TODO: Add support for handles and vectors.
315 dest << " PyErr_Format(PyExc_NotImplementedError, "
316 << "\"Can't convert argument " << args[i].name << " from Python\");\n";
317 dest << " return nullptr;\n";
318 dest << "}";
319 return;
320 }
321 }
322 dest << " static const char* const kwlist[] = {";
323 for (size_t i = 0; i < args.size(); i++) {
324 dest << "\"" << arg_names[i] << "\", ";
325 }
326 dest << "nullptr};\n";
327 for (size_t i = 0; i < args.size(); i++) {
328 dest << " " << print_type(&args[i]).second << " py_" << arg_names[i] << ";\n";
329 }
330 dest << " if (!PyArg_ParseTupleAndKeywords(args, kwargs, \"";
331 for (size_t i = 0; i < args.size(); i++) {
332 dest << print_type(&args[i]).first;
333 }
334 dest << "\", (char**)kwlist";
335 for (size_t i = 0; i < args.size(); i++) {
336 dest << ", ";
337 dest << "&py_" << arg_names[i];
338 }
339 dest << ")) {\n";
340 dest << " return nullptr;\n";
341 dest << " }\n";
342 for (size_t i = 0; i < args.size(); i++) {
343 if (args[i].is_buffer()) {
344 convert_buffer(arg_names[i], &args[i]);
345 buffer_refs.push_back("view_" + arg_names[i]);
346 } else {
347 // Python already converted this.
348 }
349 }
350 dest << " int result = " << f.name << "(";
351 for (size_t i = 0; i < args.size(); i++) {
352 if (i > 0) {
353 dest << ", ";
354 }
355 if (args[i].is_buffer()) {
356 dest << "&buffer_" << arg_names[i];
357 } else {
358 dest << "py_" << arg_names[i];
359 }
360 }
361 dest << ");\n";
362 release_buffers();
363 dest << R"INLINE_CODE(
364 if (result != 0) {
365 /* In the optimal case, we'd be generating an exception declared
366 * in python_bindings/src, but since we're self-contained,
367 * we don't have access to that API. */
368 PyErr_Format(PyExc_ValueError, "Halide error %d", result);
369 return nullptr;
370 }
371 Py_INCREF(Py_True);
372 return Py_True;
373)INLINE_CODE";
374 dest << "}\n";
375}
