12class StrictifyFloat : public IRMutator {
13 enum Strictness {
14 FastMath,
15 StrictFloat,
16 } strictness;
17
18 using IRMutator::visit;
19
20 Expr visit(const Call *call) override {
21 Strictness new_strictness = strictness;
22
23 if (call->is_intrinsic(Call::strict_float)) {
24 new_strictness = StrictFloat;
25 any_strict_float |= true;
26 }
27
28 ScopedValue<Strictness> save_strictness(strictness, new_strictness);
29
30 return IRMutator::visit(call);
31 }
32
33 using IRMutator::mutate;
34
35 Expr mutate(const Expr &expr) override {
36 if (!expr.defined()) {
37 return expr;
38 }
39 Expr e = IRMutator::mutate(expr);
40 if (e.type().is_float()) {
41 switch (strictness) {
42 case FastMath:
43 return e;
44 case StrictFloat:
45 return strict_float(e);
46 }
47 }
48 return e;
49 }
50
51public:
52 enum StrictnessMode {
53 Allowed,
54 Forced
55 };
56 bool any_strict_float{false};
57
58 StrictifyFloat(StrictnessMode mode)
59 : strictness((mode == Forced) ? StrictFloat : FastMath) {
60 any_strict_float |= (mode == Forced);
61 }
62};
63
64} // namespace
65
66bool strictify_float(std::map<std::string, Function> &env, const Target &t) {
67 bool any_strict_float = false;
68 for (auto &iter : env) {
69 Function &func = iter.second;
70
71 StrictifyFloat::StrictnessMode mode = StrictifyFloat::Allowed;
72 if (t.has_feature(Target::StrictFloat)) {
73 mode = StrictifyFloat::Forced;
74 }
75 // TODO(zalman): Some targets don't allow strict float and we can provide errors for these.
76
77 StrictifyFloat strictify(mode);
78 func.mutate(&strictify);
79 any_strict_float |= strictify.any_strict_float;
80 }
81 return any_strict_float;
82}
