16class FindLastUse : public IRVisitor {
17public:
18 string func;
19 Stmt last_use;
20
21 FindLastUse(string s)
22 : func(std::move(s)) {
23 }
24
25private:
26 bool in_loop = false;
27 Stmt containing_stmt;
28
29 using IRVisitor::visit;
30
31 void visit(const For *loop) override {
32 loop->min.accept(this);
33 loop->extent.accept(this);
34 ScopedValue<bool> old_in_loop(in_loop, true);
35 loop->body.accept(this);
36 }
37
38 void visit(const Fork *fork) override {
39 ScopedValue<bool> old_in_loop(in_loop, true);
40 fork->first.accept(this);
41 fork->rest.accept(this);
42 }
43
44 void visit(const Acquire *acq) override {
45 acq->semaphore.accept(this);
46 acq->count.accept(this);
47 ScopedValue<bool> old_in_loop(in_loop, true);
48 acq->body.accept(this);
49 }
50
51 void visit(const Load *load) override {
52 if (func == load->name) {
53 last_use = containing_stmt;
54 }
55 IRVisitor::visit(load);
56 }
57
58 void visit(const Call *call) override {
59 if (call->name == func) {
60 last_use = containing_stmt;
61 }
62 IRVisitor::visit(call);
63 }
64
65 void visit(const Store *store) override {
66 if (func == store->name) {
67 last_use = containing_stmt;
68 }
69 IRVisitor::visit(store);
70 }
71
72 void visit(const Variable *var) override {
73 if (var->name == func || var->name == func + ".buffer") {
74 // Don't free the allocation while a buffer that may refer
75 // to it is still in use.
76 last_use = containing_stmt;
77 }
78 }
79
80 void visit(const IfThenElse *op) override {
81 // It's a bad idea to inject it in either side of an
82 // ifthenelse, so we treat this as being in a loop.
83 op->condition.accept(this);
84 ScopedValue<bool> old_in_loop(in_loop, true);
85 op->then_case.accept(this);
86 if (op->else_case.defined()) {
87 op->else_case.accept(this);
88 }
89 }
90
91 void visit(const Block *block) override {
92 if (in_loop) {
93 IRVisitor::visit(block);
94 } else {
95 ScopedValue<Stmt> old_containing_stmt(containing_stmt, block->first);
96 block->first.accept(this);
97 if (block->rest.defined()) {
98 containing_stmt = block->rest;
99 block->rest.accept(this);
100 }
101 }
102 }
103
104 void visit(const Atomic *op) override {
105 if (op->mutex_name == func) {
106 last_use = containing_stmt;
107 }
108 IRVisitor::visit(op);
109 }
110};

112class InjectMarker : public IRMutator {
113public:
114 string func;
115 Stmt last_use;
116
117private:
118 bool injected = false;
119
120 using IRMutator::visit;
121
122 Stmt inject_marker(Stmt s) {
123 if (injected) {
124 return s;
125 }
126 if (s.same_as(last_use)) {
127 injected = true;
128 return Block::make(s, Free::make(func));
129 } else {
130 return mutate(s);
131 }
132 }
133
134 Stmt visit(const Block *block) override {
135 Stmt new_rest = inject_marker(block->rest);
136 Stmt new_first = inject_marker(block->first);
137
138 if (new_first.same_as(block->first) &&
139 new_rest.same_as(block->rest)) {
140 return block;
141 } else {
142 return Block::make(new_first, new_rest);
143 }
144 }
145};

147class InjectEarlyFrees : public IRMutator {
148 using IRMutator::visit;
149
150 Stmt visit(const Allocate *alloc) override {
151 Stmt stmt = IRMutator::visit(alloc);
152 alloc = stmt.as<Allocate>();
153 internal_assert(alloc);
154
155 FindLastUse last_use(alloc->name);
156 stmt.accept(&last_use);
157
158 if (last_use.last_use.defined()) {
159 InjectMarker inject_marker;
160 inject_marker.func = alloc->name;
161 inject_marker.last_use = last_use.last_use;
162 stmt = inject_marker.mutate(stmt);
163 } else {
164 stmt = Allocate::make(alloc->name, alloc->type, alloc->memory_type,
165 alloc->extents, alloc->condition,
166 Block::make(alloc->body, Free::make(alloc->name)),
167 alloc->new_expr, alloc->free_function);
168 }
169 return stmt;
170 }
171};
