22struct TraceEventBuilder {
23 string func;
24 Expr trace_tag_expr = Expr("");
25 vector<Expr> value;
26 vector<Expr> coordinates;
27 Type type;
28 enum halide_trace_event_code_t event;
29 Expr parent_id, value_index;
30
31 Expr build() {
32 Expr values = Call::make(type_of<void *>(), Call::make_struct,
33 value, Call::Intrinsic);
34 Expr coords = Call::make(type_of<int32_t *>(), Call::make_struct,
35 coordinates, Call::Intrinsic);
36 Expr idx = value_index;
37 if (!idx.defined()) {
38 idx = 0;
39 }
40
41 // Note: if these arguments are changed in any meaningful way,
42 // VectorizeLoops will likely need attention; it does nontrivial
43 // special-casing of this call to get appropriate results.
44 vector<Expr> args = {Expr(func),
45 values, coords,
46 (int)type.code(), (int)type.bits(), (int)type.lanes(),
47 (int)event,
48 parent_id, idx, (int)coordinates.size(),
49 trace_tag_expr};
50 return Call::make(Int(32), Call::trace, args, Call::Extern);
51 }
52};

54class InjectTracing : public IRMutator {
55public:
56 const map<string, Function> &env;
57 const bool trace_all_loads, trace_all_stores, trace_all_realizations;
58 // We want to preserve the order, so use a vector<pair> rather than a map
59 vector<pair<string, vector<string>>> trace_tags;
60 set<string> trace_tags_added;
61 // The funcs that will have any tracing info emitted (not just trace tags),
62 // and the Type(s) of their elements.
63 map<string, vector<Type>> funcs_touched;
64
65 InjectTracing(const map<string, Function> &e, const Target &t)
66 : env(e),
67 trace_all_loads(t.has_feature(Target::TraceLoads)),
68 trace_all_stores(t.has_feature(Target::TraceStores)),
69 // Set trace_all_realizations to true if either trace_loads or trace_stores is on too:
70 // They don't work without trace_all_realizations being on (and the errors are missing symbol mysterious nonsense).
71 trace_all_realizations(t.features_any_of({Target::TraceLoads, Target::TraceStores, Target::TraceRealizations})) {
72 }
73
74private:
75 void add_trace_tags(const string &name, const vector<string> &t) {
76 if (!t.empty() && !trace_tags_added.count(name)) {
77 trace_tags.emplace_back(name, t);
78 trace_tags_added.insert(name);
79 }
80 }
81
82 void add_func_touched(const string &name, int value_index, const Type &type) {
83 auto it = funcs_touched.find(name);
84 if (it == funcs_touched.end()) {
85 vector<Type> types(value_index + 1);
86 types[value_index] = type;
87 funcs_touched[name] = types;
88 } else {
89 // If the type already present is missing, or "handle0" (aka "we don't know yet",
90 // replace it with the given type. Otherwise, assert the types match.
91 vector<Type> &types = it->second;
92 if ((int)types.size() <= value_index) {
93 types.resize(value_index + 1);
94 types[value_index] = type;
95 } else {
96 internal_assert(type == Type() || type == types[value_index])
97 << "Type was already specified as " << types[value_index]
98 << " but now is " << type;
99 }
100 }
101 }

103 using IRMutator::visit;
104
105 Expr visit(const Call *op) override {
106 Expr expr = IRMutator::visit(op);
107 op = expr.as<Call>();
108 internal_assert(op);
109 bool trace_it = false;
110 Expr trace_parent;
111 if (op->call_type == Call::Halide) {
112 auto it = env.find(op->name);
113 internal_assert(it != env.end()) << op->name << " not in environment\n";
114 Function f = it->second;
115 internal_assert(!f.can_be_inlined() || !f.schedule().compute_level().is_inlined());
116
117 trace_it = f.is_tracing_loads() || trace_all_loads;
118 trace_parent = Variable::make(Int(32), op->name + ".trace_id");
119 if (trace_it) {
120 add_trace_tags(op->name, f.get_trace_tags());
121 }
122 } else if (op->call_type == Call::Image) {
123 trace_it = trace_all_loads;
124 // If there is a Function in the env named "name_im", assume that
125 // this image is an ImageParam, so sniff that Function to see
126 // if we want to trace loads on it. (This allows us to trace
127 // loads on inputs without having to enable them globally.)
128 auto it = env.find(op->name + "_im");
129 if (it != env.end()) {
130 Function f = it->second;
131 // f could be scheduled and have actual loads from it (via ImageParam::in),
132 // so only honor trace the loads if it is inlined.
133 if ((f.is_tracing_loads() || trace_all_loads) &&
134 f.can_be_inlined() &&
135 f.schedule().compute_level().is_inlined()) {
136 trace_it = true;
137 add_trace_tags(op->name, f.get_trace_tags());
138 }
139 }
140
141 trace_parent = Variable::make(Int(32), "pipeline.trace_id");
142 }
143
144 if (trace_it) {
145 add_func_touched(op->name, op->value_index, op->type);
146
147 string value_var_name = unique_name('t');
148 Expr value_var = Variable::make(op->type, value_var_name);
149
150 TraceEventBuilder builder;
151 builder.func = op->name;
152 builder.value = {value_var};
153 builder.coordinates = op->args;
154 builder.type = op->type;
155 builder.event = halide_trace_load;
156 builder.parent_id = trace_parent;
157 builder.value_index = op->value_index;
158 Expr trace = builder.build();
159
160 expr = Let::make(value_var_name, op,
161 Call::make(op->type, Call::return_second,
162 {trace, value_var}, Call::PureIntrinsic));
163 }
164 return expr;
165 }

167 Stmt visit(const Provide *op) override {
168 Stmt stmt = IRMutator::visit(op);
169 op = stmt.as<Provide>();
170 internal_assert(op);
171
172 map<string, Function>::const_iterator iter = env.find(op->name);
173 if (iter == env.end()) {
174 return stmt;
175 }
176 Function f = iter->second;
177 internal_assert(!f.can_be_inlined() || !f.schedule().compute_level().is_inlined());
178
179 if (f.is_tracing_stores() || trace_all_stores) {
180 // Wrap each expr in a tracing call
181
182 const vector<Expr> &values = op->values;
183 vector<Expr> traces(op->values.size());
184
185 TraceEventBuilder builder;
186 builder.func = f.name();
187 builder.coordinates = op->args;
188 builder.event = halide_trace_store;
189 builder.parent_id = Variable::make(Int(32), op->name + ".trace_id");
190 for (size_t i = 0; i < values.size(); i++) {
191 Type t = values[i].type();
192 add_func_touched(f.name(), (int)i, t);
193 string value_var_name = unique_name('t');
194 Expr value_var = Variable::make(t, value_var_name);
195
196 builder.type = t;
197 builder.value_index = (int)i;
198 builder.value = {value_var};
199 Expr trace = builder.build();
200
201 traces[i] = Let::make(value_var_name, values[i],
202 Call::make(t, Call::return_second,
203 {trace, value_var}, Call::PureIntrinsic));
204 }
205
206 // Lift the args out into lets so that the order of
207 // evaluation is right for scatters. Otherwise the store
208 // is traced before any loads in the index.
209 vector<Expr> args = op->args;
210 vector<pair<string, Expr>> lets;
211 for (size_t i = 0; i < args.size(); i++) {
212 if (!args[i].as<Variable>() && !is_const(args[i])) {
213 string name = unique_name('t');
214 lets.emplace_back(name, args[i]);
215 args[i] = Variable::make(args[i].type(), name);
216 }
217 }
218
219 stmt = Provide::make(op->name, traces, args);
220 for (const auto &p : lets) {
221 stmt = LetStmt::make(p.first, p.second, stmt);
222 }
223 }
224 return stmt;
225 }

227 Stmt visit(const Realize *op) override {
228 Stmt stmt = IRMutator::visit(op);
229 op = stmt.as<Realize>();
230 internal_assert(op);
231
232 map<string, Function>::const_iterator iter = env.find(op->name);
233 if (iter == env.end()) {
234 return stmt;
235 }
236 Function f = iter->second;
237 if (f.is_tracing_realizations() || trace_all_realizations) {
238 add_trace_tags(op->name, f.get_trace_tags());
239 for (size_t i = 0; i < op->types.size(); i++) {
240 add_func_touched(op->name, i, op->types[i]);
241 }
242
243 // Throw a tracing call before and after the realize body
244 TraceEventBuilder builder;
245 builder.func = op->name;
246 builder.parent_id = Variable::make(Int(32), "pipeline.trace_id");
247 builder.event = halide_trace_begin_realization;
248 for (size_t i = 0; i < op->bounds.size(); i++) {
249 builder.coordinates.push_back(op->bounds[i].min);
250 builder.coordinates.push_back(op->bounds[i].extent);
251 }
252
253 // Begin realization returns a unique token to pass to further trace calls affecting this buffer.
254 Expr call_before = builder.build();
255
256 builder.event = halide_trace_end_realization;
257 builder.parent_id = Variable::make(Int(32), op->name + ".trace_id");
258 Expr call_after = builder.build();
259
260 Stmt new_body = op->body;
261 new_body = Block::make(new_body, Evaluate::make(call_after));
262 new_body = LetStmt::make(op->name + ".trace_id", call_before, new_body);
263 stmt = Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, new_body);
264 // Warning: 'op' may be invalid at this point
265 } else if (f.is_tracing_stores() || f.is_tracing_loads()) {
266 // We need a trace id defined to pass to the loads and stores
267 Stmt new_body = op->body;
268 new_body = LetStmt::make(op->name + ".trace_id", 0, new_body);
269 stmt = Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, new_body);
270 }
271 return stmt;
272 }

274 Stmt visit(const ProducerConsumer *op) override {
275 Stmt stmt = IRMutator::visit(op);
276 op = stmt.as<ProducerConsumer>();
277 internal_assert(op);
278 map<string, Function>::const_iterator iter = env.find(op->name);
279 if (iter == env.end()) {
280 return stmt;
281 }
282 Function f = iter->second;
283 if (f.is_tracing_realizations() || trace_all_realizations) {
284 // Throw a tracing call around each pipeline event
285 TraceEventBuilder builder;
286 builder.func = op->name;
287 builder.parent_id = Variable::make(Int(32), op->name + ".trace_id");
288
289 // Use the size of the pure step
290 const vector<string> &f_args = f.args();
291 for (int i = 0; i < f.dimensions(); i++) {
292 Expr min = Variable::make(Int(32), f.name() + ".s0." + f_args[i] + ".min");
293 Expr max = Variable::make(Int(32), f.name() + ".s0." + f_args[i] + ".max");
294 Expr extent = (max + 1) - min;
295 builder.coordinates.push_back(min);
296 builder.coordinates.push_back(extent);
297 }
298
299 builder.event = (op->is_producer ? halide_trace_produce : halide_trace_consume);
300 Expr begin_op_call = builder.build();
301
302 builder.event = (op->is_producer ? halide_trace_end_produce : halide_trace_end_consume);
303 Expr end_op_call = builder.build();
304
305 Stmt new_body = Block::make(op->body, Evaluate::make(end_op_call));
306
307 stmt = LetStmt::make(f.name() + ".trace_id", begin_op_call,
308 ProducerConsumer::make(op->name, op->is_producer, new_body));
309 }
310 return stmt;
311 }

314class RemoveRealizeOverOutput : public IRMutator {
315 using IRMutator::visit;

317
318 Stmt visit(const Realize *op) override {
319 for (const Function &f : outputs) {
320 if (op->name == f.name()) {
321 return mutate(op->body);
322 }
323 }

335Stmt inject_tracing(Stmt s, const string &pipeline_name, bool trace_pipeline,
336 const map<string, Function> &env, const vector<Function> &outputs,
337 const Target &t) {
338 Stmt original = s;
339 InjectTracing tracing(env, t);
340
341 // Add a dummy realize block for the output buffers
342 for (const Function &output : outputs) {
343 Region output_region;
344 Parameter output_buf = output.output_buffers()[0];
345 internal_assert(output_buf.is_buffer());
346 for (int i = 0; i < output.dimensions(); i++) {
347 string d = std::to_string(i);
348 Expr min = Variable::make(Int(32), output_buf.name() + ".min." + d);
349 Expr extent = Variable::make(Int(32), output_buf.name() + ".extent." + d);
350 output_region.push_back(Range(min, extent));
351 }
352 s = Realize::make(output.name(), output.output_types(), MemoryType::Auto, output_region, const_true(), s);
353 }
354
355 // Inject tracing calls
356 s = tracing.mutate(s);
357
358 // Strip off the dummy realize blocks
359 s = RemoveRealizeOverOutput(outputs).mutate(s);
360
361 if (!s.same_as(original) || trace_pipeline || t.has_feature(Target::TracePipeline)) {
362 // Add pipeline start and end events
363 TraceEventBuilder builder;
364 builder.func = pipeline_name;
365 builder.event = halide_trace_begin_pipeline;
366 builder.parent_id = 0;
367
368 Expr pipeline_start = builder.build();
369
370 builder.event = halide_trace_end_pipeline;
371 builder.parent_id = Variable::make(Int(32), "pipeline.trace_id");
372 Expr pipeline_end = builder.build();
373
374 s = Block::make(s, Evaluate::make(pipeline_end));
375
376 // All trace_tag events go at the start, immediately after begin_pipeline.
377 // For a given realization/input/output, we output them in the order
378 // we encounter them (which is to say, the order they were added); however,
379 // we don't attempt to preserve a particular order between functions.
380 for (const auto &trace_tags : tracing.trace_tags) {
381 // builder.parent_id is already set correctly
382 builder.func = trace_tags.first; // func name
383 builder.event = halide_trace_tag;
384 // We must reverse-iterate to preserve order
385 for (auto it = trace_tags.second.rbegin(); it != trace_tags.second.rend(); ++it) {
386 user_assert(it->find('\0') == string::npos)
387 << "add_trace_tag() may not contain the null character.";
388 builder.trace_tag_expr = Expr(*it);
389 s = Block::make(Evaluate::make(builder.build()), s);
390 }
391 }
392
393 builder.event = halide_trace_tag;
394
395 vector<string> order = topological_order(outputs, env);
396
397 // Compute boxes_touched and send a func_type_and_dim trace-tag for
398 // everything that we actually touched, in topological order.
399 // We include the type(s) of each Func (could be multiple for Tuple-valued
400 // Funcs), and the dimensions and guess-at-ranges-rouched. Note that the
401 // dimensions should be exact, but the ranges-touched is a conservative estimate;
402 // that's ok, as we just want to send these as rough guesses for a tracing tool to use for
403 // automatic layout. (Note that we deliberately send these
404 // before any user-specified trace-tags.)
405 Expr space = Expr(" ");
406
407 std::map<std::string, Box> bt = boxes_touched(s);
408 for (auto topo_it = order.rbegin(); topo_it != order.rend(); ++topo_it) {
409 const string &o = *topo_it;
410 auto p = tracing.funcs_touched.find(*topo_it);
411 if (p == tracing.funcs_touched.end() && ends_with(o, "_im")) {
412 p = tracing.funcs_touched.find(o.substr(0, o.size() - 3));
413 }
414 if (p == tracing.funcs_touched.end()) {
415 continue;
416 }
417 const string &func_name = p->first;
418 const vector<Type> &func_types = p->second;
419 builder.func = func_name;
420
421 vector<Expr> strings;
422 strings.emplace_back("func_type_and_dim:");
423 strings.push_back(space);
424 strings.emplace_back((int)func_types.size());
425 for (const auto &func_type : func_types) {
426 strings.push_back(space);
427 strings.emplace_back((int)func_type.code());
428 strings.push_back(space);
429 strings.emplace_back(func_type.bits());
430 strings.push_back(space);
431 strings.emplace_back(func_type.lanes());
432 }
433 auto it = bt.find(func_name);
434 internal_assert(it != bt.end());
435 const Box &box = it->second;
436 strings.push_back(space);
437 strings.emplace_back((int)box.bounds.size());
438 for (const Interval &i : box.bounds) {
439 internal_assert(i.min.defined() && i.max.defined());
440 if (i.is_bounded()) {
441 strings.push_back(space);
442 strings.push_back(i.min);
443 strings.push_back(space);
444 // Emit as (min, extent) rather than (min, max)
445 strings.push_back(i.max - i.min + Expr(1));
446 } else {
447 // This should really only happen for unusual cases
448 // that we won't end up realizing, so we can just
449 // use any numeric values.
450 strings.push_back(space);
451 strings.emplace_back(0);
452 strings.push_back(space);
453 strings.emplace_back(0);
454 }
455 }
456 builder.trace_tag_expr =
457 Internal::Call::make(type_of<const char *>(),
458 Internal::Call::stringify,
459 strings,
460 Internal::Call::Intrinsic);
461 s = Block::make(Evaluate::make(builder.build()), s);
462 }
463
464 s = LetStmt::make("pipeline.trace_id", pipeline_start, s);
465 }
466
467 return s;
468}
