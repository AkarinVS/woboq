21class InferArguments : public IRGraphVisitor {
22public:
23 vector<InferredArgument> &args;
24
25 InferArguments(vector<InferredArgument> &a, const vector<Function> &o, const Stmt &body)
26 : args(a), outputs(o) {
27 args.clear();
28 for (const Function &f : outputs) {
29 visit_function(f);
30 }
31 if (body.defined()) {
32 body.accept(this);
33 }
34 }
35
36private:
37 const vector<Function> outputs;
38 set<string> visited_functions;
39
40 struct ParamOrBuffer {
41 Parameter param;
42 Buffer<> buffer;
43 };
44 map<string, ParamOrBuffer> args_by_name;
45
46 using IRGraphVisitor::visit;
47
48 bool is_output_name(const string &name) const {
49 for (const Function &output : outputs) {
50 if (name == output.name() || starts_with(name, output.name() + ".")) {
51 return true;
52 }
53 }
54 return false;
55 }
56
57 static bool dupe_names_error(const string &name) {
58 user_error << "All Params and embedded Buffers must have unique names, but the name '"
59 << name << "' was seen multiple times.\n";
60 return false; // not reached
61 }
62
63 bool already_have(const Parameter &p) {
64 const string &name = p.name();
65
66 // Ignore dependencies on the output buffers
67 if (is_output_name(name)) {
68 return true;
69 }
70
71 auto it = args_by_name.find(name);
72 if (it == args_by_name.end()) {
73 // If the Parameter is already bound to a Buffer, include it here.
74 if (p.is_buffer() && p.buffer().defined()) {
75 args_by_name[name] = {p, p.buffer()};
76 } else {
77 args_by_name[name] = {p, Buffer<>()};
78 }
79 return false;
80 }
81
82 ParamOrBuffer &pob = it->second;
83 if (pob.param.defined()) {
84 // If the name is already in the args, verify that it's the same
85 // Parameter that we've already seen.
86 if (p.same_as(pob.param)) {
87 return true;
88 } else {
89 // Multiple different Parameters with the same name -> illegal
90 return dupe_names_error(name);
91 }
92 } else if (pob.buffer.defined()) {
93 // If the name is in the args, but only as a Buffer,
94 // maybe it's the Buffer that the Parameter is bound to?
95 if (p.is_buffer() && p.buffer().defined() && p.buffer().same_as(pob.buffer)) {
96 // Update this entry to have both the Parameter and Buffer.
97 pob.param = p;
98 return true;
99 } else {
100 // A Parameter and Buffer with the same name (but aren't connected) -> illegal
101 return dupe_names_error(name);
102 }
103 } else {
104 internal_error << "There should be no empty ParamOrBuffers in the map.";
105 return false; // not reached
106 }
107 }
108
109 bool already_have(const Buffer<> &b) {
110 const string &name = b.name();
111
112 // Ignore dependencies on the output buffers
113 if (is_output_name(name)) {
114 return true;
115 }
116
117 auto it = args_by_name.find(name);
118 if (it == args_by_name.end()) {
119 args_by_name[name] = {Parameter(), b};
120 return false;
121 }
122
123 ParamOrBuffer &pob = it->second;
124 if (pob.buffer.defined()) {
125 // If the name is already in the args, verify that it's the same
126 // Buffer that we've already seen.
127 if (b.same_as(pob.buffer)) {
128 return true;
129 } else {
130 // Multiple different Buffers with the same name -> illegal
131 return dupe_names_error(name);
132 }
133 } else if (pob.param.defined()) {
134 // If the name is in the args, but only as a Parameter,
135 // maybe it's the Parameter that this Buffer is bound to?
136 if (pob.param.is_buffer() && pob.param.buffer().same_as(b)) {
137 // Update this entry to have both the Parameter and Buffer.
138 pob.buffer = b;
139 return true;
140 } else {
141 // A Parameter and Buffer with the same name (but aren't connected) -> illegal
142 return dupe_names_error(name);
143 }
144 } else {
145 internal_error << "There should be no empty ParamOrBuffers in the map.";
146 return false; // not reached
147 }
148 }

150 void visit_exprs(const vector<Expr> &v) {
151 for (const Expr &i : v) {
152 visit_expr(i);
153 }
154 }
155
156 void visit_expr(const Expr &e) {
157 if (!e.defined()) {
158 return;
159 }
160 e.accept(this);
161 }
162
163 void visit_function(const Function &func) {
164 if (visited_functions.count(func.name())) {
165 return;
166 }
167 visited_functions.insert(func.name());
168
169 func.accept(this);
170
171 // Function::accept hits all the Expr children of the
172 // Function, but misses the buffers and images that might be
173 // extern arguments.
174 if (func.has_extern_definition()) {
175 for (const ExternFuncArgument &extern_arg : func.extern_arguments()) {
176 if (extern_arg.is_func()) {
177 visit_function(Function(extern_arg.func));
178 } else if (extern_arg.is_buffer()) {
179 include_buffer(extern_arg.buffer);
180 } else if (extern_arg.is_image_param()) {
181 include_parameter(extern_arg.image_param);
182 }
183 }
184 }
185 }
186
187 void include_parameter(const Parameter &p) {
188 if (!p.defined() ||
189 already_have(p)) {
190 return;
191 }
192
193 ArgumentEstimates argument_estimates = p.get_argument_estimates();
194 if (!p.is_buffer()) {
195 argument_estimates.scalar_def = p.scalar_expr();
196 argument_estimates.scalar_min = p.min_value();
197 argument_estimates.scalar_max = p.max_value();
198 argument_estimates.scalar_estimate = p.estimate();
199 }
200
201 InferredArgument a = {
202 Argument(p.name(),
203 p.is_buffer() ? Argument::InputBuffer : Argument::InputScalar,
204 p.type(), p.dimensions(), argument_estimates),
205 p,
206 Buffer<>()};
207 args.push_back(a);
208
209 // Visit child expressions
210 visit_expr(argument_estimates.scalar_def);
211 visit_expr(argument_estimates.scalar_min);
212 visit_expr(argument_estimates.scalar_max);
213 visit_expr(argument_estimates.scalar_estimate);
214 for (const auto &be : argument_estimates.buffer_estimates) {
215 visit_expr(be.min);
216 visit_expr(be.extent);
217 }
218
219 if (p.is_buffer()) {
220 for (int i = 0; i < p.dimensions(); i++) {
221 visit_expr(p.min_constraint(i));
222 visit_expr(p.extent_constraint(i));
223 visit_expr(p.stride_constraint(i));
224 }
225 }
226 }
227
228 void include_buffer(const Buffer<> &b) {
229 if (!b.defined() ||
230 already_have(b)) {
231 return;
232 }
233
234 InferredArgument a = {
235 Argument(b.name(), Argument::InputBuffer, b.type(), b.dimensions(), ArgumentEstimates{}),
236 Parameter(),
237 b};
238 args.push_back(a);
239 }
240
241 void visit(const Load *op) override {
242 IRGraphVisitor::visit(op);
243 include_parameter(op->param);
244 include_buffer(op->image);
245 }
246
247 void visit(const Variable *op) override {
248 IRGraphVisitor::visit(op);
249 include_parameter(op->param);
250 include_buffer(op->image);
251 }
252
253 void visit(const Call *op) override {
254 IRGraphVisitor::visit(op);
255 if (op->func.defined()) {
256 Function fn(op->func);
257 visit_function(fn);
258 }
259 include_buffer(op->image);
260 include_parameter(op->param);
261 }
262};

266vector<InferredArgument> infer_arguments(const Stmt &body, const vector<Function> &outputs) {
267 vector<InferredArgument> inferred_args;
268 // Infer an arguments vector by walking the IR
269 InferArguments infer_args(inferred_args,
270 outputs,
271 body);
272
273 // Sort the Arguments with all buffers first (alphabetical by name),
274 // followed by all non-buffers (alphabetical by name).
275 std::sort(inferred_args.begin(), inferred_args.end());
276
277 return inferred_args;
278}
