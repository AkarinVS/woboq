56// Fold the storage of a function in a particular dimension by a particular factor
57class FoldStorageOfFunction : public IRMutator {
58 string func;
59 int dim;
60 Expr factor;
61 string dynamic_footprint;
62
63 using IRMutator::visit;
64
65 Expr visit(const Call *op) override {
66 Expr expr = IRMutator::visit(op);
67 op = expr.as<Call>();
68 internal_assert(op);
69 if (op->name == func && op->call_type == Call::Halide) {
70 vector<Expr> args = op->args;
71 internal_assert(dim < (int)args.size());
72 args[dim] = is_const_one(factor) ? 0 : (args[dim] % factor);
73 expr = Call::make(op->type, op->name, args, op->call_type,
74 op->func, op->value_index, op->image, op->param);
75 } else if (op->name == Call::buffer_crop) {
76 Expr source = op->args[2];
77 const Variable *buf_var = source.as<Variable>();
78 if (buf_var &&
79 starts_with(buf_var->name, func + ".") &&
80 ends_with(buf_var->name, ".buffer")) {
81 // We are taking a crop of a folded buffer. For now
82 // we'll just assert that the crop doesn't wrap
83 // around, so that the crop doesn't need to be treated
84 // as a folded buffer too. But to take the crop, we
85 // need to use folded coordinates, and then restore
86 // the non-folded min after the crop operation.
87
88 // Pull out the expressions we need
89 internal_assert(op->args.size() >= 5);
90 Expr mins_arg = op->args[3];
91 Expr extents_arg = op->args[4];
92 const Call *mins_call = mins_arg.as<Call>();
93 const Call *extents_call = extents_arg.as<Call>();
94 internal_assert(mins_call && extents_call);
95 vector<Expr> mins = mins_call->args;
96 const vector<Expr> &extents = extents_call->args;
97 internal_assert(dim < (int)mins.size() && dim < (int)extents.size());
98 Expr old_min = mins[dim];
99 Expr old_extent = extents[dim];
100
101 // Rewrite the crop args
102 mins[dim] = old_min % factor;
103 Expr new_mins = Call::make(type_of<int *>(), Call::make_struct, mins, Call::Intrinsic);
104 vector<Expr> new_args = op->args;
105 new_args[3] = new_mins;
106 expr = Call::make(op->type, op->name, new_args, op->call_type);
107
108 // Inject the assertion
109 Expr no_wraparound = mins[dim] + extents[dim] <= factor;
110
111 Expr valid_min = old_min;
112 if (!dynamic_footprint.empty()) {
113 // If the footprint is being tracked dynamically, it's
114 // not enough to just check we don't overlap a
115 // fold. We also need to check the min against the
116 // valid min.
117
118 // TODO: dynamic footprint is no longer the min, and may be tracked separately on producer and consumer sides (head vs tail)
119 valid_min =
120 Load::make(Int(32), dynamic_footprint, 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
121 Expr check = (old_min >= valid_min &&
122 (old_min + old_extent - 1) < valid_min + factor);
123 no_wraparound = no_wraparound && check;
124 }
125
126 Expr error = Call::make(Int(32), "halide_error_bad_extern_fold",
127 {Expr(func), Expr(dim), old_min, old_extent, valid_min, factor},
128 Call::Extern);
129 expr = Call::make(op->type, Call::require,
130 {no_wraparound, expr, error}, Call::Intrinsic);
131
132 // Restore the correct min coordinate
133 expr = Call::make(op->type, Call::buffer_set_bounds,
134 {expr, dim, old_min, old_extent}, Call::Extern);
135 }
136 }
137 return expr;
138 }
139
140 Stmt visit(const Provide *op) override {
141 Stmt stmt = IRMutator::visit(op);
142 op = stmt.as<Provide>();
143 internal_assert(op);
144 if (op->name == func) {
145 vector<Expr> args = op->args;
146 args[dim] = is_const_one(factor) ? 0 : (args[dim] % factor);
147 stmt = Provide::make(op->name, op->values, args);
148 }
149 return stmt;
150 }
151
152public:
153 FoldStorageOfFunction(string f, int d, Expr e, string p)
154 : func(std::move(f)), dim(d), factor(std::move(e)), dynamic_footprint(std::move(p)) {
155 }
156};

158// Inject dynamic folding checks against a tracked live range.
159class InjectFoldingCheck : public IRMutator {
160 Function func;
161 string head, tail, loop_var;
162 Expr sema_var;
163 int dim;
164 bool in_produce;
165 const StorageDim &storage_dim;
166 using IRMutator::visit;
167
168 Stmt visit(const ProducerConsumer *op) override {
169 if (op->name == func.name()) {
170 Stmt body = op->body;
171 if (op->is_producer) {
172 if (func.has_extern_definition()) {
173 // We'll update the valid min at the buffer_crop call.
174 in_produce = true;
175 body = mutate(op->body);
176 } else {
177 // Update valid range based on bounds written to.
178 Box b = box_provided(body, func.name());
179 Expr old_leading_edge =
180 Load::make(Int(32), head + "_next", 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
181
182 internal_assert(!b.empty());
183
184 // Track the logical address range the memory
185 // currently represents.
186 Expr new_leading_edge;
187 if (storage_dim.fold_forward) {
188 new_leading_edge = max(b[dim].max, old_leading_edge);
189 } else {
190 new_leading_edge = min(b[dim].min, old_leading_edge);
191 }
192
193 string new_leading_edge_var_name = unique_name('t');
194 Expr new_leading_edge_var = Variable::make(Int(32), new_leading_edge_var_name);
195
196 Stmt update_leading_edge =
197 Store::make(head, new_leading_edge_var, 0, Parameter(), const_true(), ModulusRemainder());
198 Stmt update_next_leading_edge =
199 Store::make(head + "_next", new_leading_edge_var, 0, Parameter(), const_true(), ModulusRemainder());
200
201 // Check the region being written to in this
202 // iteration lies within the range of coordinates
203 // currently represented.
204 Expr fold_non_monotonic_error =
205 Call::make(Int(32), "halide_error_bad_fold",
206 {func.name(), storage_dim.var, loop_var},
207 Call::Extern);
208
209 Expr in_valid_range;
210 if (storage_dim.fold_forward) {
211 in_valid_range = b[dim].min > new_leading_edge - storage_dim.fold_factor;
212 } else {
213 in_valid_range = b[dim].max < new_leading_edge + storage_dim.fold_factor;
214 }
215 Stmt check_in_valid_range =
216 AssertStmt::make(in_valid_range, fold_non_monotonic_error);
217
218 Expr extent = b[dim].max - b[dim].min + 1;
219
220 // Separately check the extent for *this* loop iteration fits.
221 Expr fold_too_small_error =
222 Call::make(Int(32), "halide_error_fold_factor_too_small",
223 {func.name(), storage_dim.var, storage_dim.fold_factor, loop_var, extent},
224 Call::Extern);
225
226 Stmt check_extent =
227 AssertStmt::make(extent <= storage_dim.fold_factor, fold_too_small_error);
228
229 Stmt checks = Block::make({check_extent, check_in_valid_range,
230 update_leading_edge, update_next_leading_edge});
231 if (func.schedule().async()) {
232 Expr to_acquire;
233 if (storage_dim.fold_forward) {
234 to_acquire = new_leading_edge_var - old_leading_edge;
235 } else {
236 to_acquire = old_leading_edge - new_leading_edge_var;
237 }
238 body = Block::make(checks, body);
239 body = Acquire::make(sema_var, to_acquire, body);
240 body = LetStmt::make(new_leading_edge_var_name, new_leading_edge, body);
241 } else {
242 checks = LetStmt::make(new_leading_edge_var_name, new_leading_edge, checks);
243 body = Block::make(checks, body);
244 }
245 }
246
247 } else {
248 // Check the accessed range against the valid range.
249 Box b = box_required(body, func.name());
250 if (b.empty()) {
251 // Must be used in an extern call (TODO:
252 // assert this, TODO: What if it's used in an
253 // extern call and native Halide). We'll
254 // update the valid min at the buffer_crop
255 // call.
256 in_produce = false;
257 body = mutate(op->body);
258 } else {
259 Expr leading_edge =
260 Load::make(Int(32), tail + "_next", 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
261
262 if (func.schedule().async()) {
263 Expr new_leading_edge;
264 if (storage_dim.fold_forward) {
265 new_leading_edge = b[dim].min - 1 + storage_dim.fold_factor;
266 } else {
267 new_leading_edge = b[dim].max + 1 - storage_dim.fold_factor;
268 }
269 string new_leading_edge_name = unique_name('t');
270 Expr new_leading_edge_var = Variable::make(Int(32), new_leading_edge_name);
271 Expr to_release;
272 if (storage_dim.fold_forward) {
273 to_release = new_leading_edge_var - leading_edge;
274 } else {
275 to_release = leading_edge - new_leading_edge_var;
276 }
277 Expr release_producer =
278 Call::make(Int(32), "halide_semaphore_release", {sema_var, to_release}, Call::Extern);
279 // The consumer is going to get its own forked copy of the footprint, so it needs to update it too.
280 Stmt update_leading_edge = Store::make(tail, new_leading_edge_var, 0, Parameter(), const_true(), ModulusRemainder());
281 update_leading_edge = Block::make(Store::make(tail + "_next", new_leading_edge_var, 0, Parameter(), const_true(), ModulusRemainder()),
282 update_leading_edge);
283 update_leading_edge = Block::make(Evaluate::make(release_producer), update_leading_edge);
284 update_leading_edge = LetStmt::make(new_leading_edge_name, new_leading_edge, update_leading_edge);
285 body = Block::make(update_leading_edge, body);
286 } else {
287 Expr check;
288 if (storage_dim.fold_forward) {
289 check = (b[dim].min > leading_edge - storage_dim.fold_factor && b[dim].max <= leading_edge);
290 } else {
291 check = (b[dim].max < leading_edge + storage_dim.fold_factor && b[dim].min >= leading_edge);
292 }
293 Expr bad_fold_error = Call::make(Int(32), "halide_error_bad_fold",
294 {func.name(), storage_dim.var, loop_var},
295 Call::Extern);
296 body = Block::make(AssertStmt::make(check, bad_fold_error), body);
297 }
298 }
299 }
300
301 return ProducerConsumer::make(op->name, op->is_producer, body);
302 } else {
303 return IRMutator::visit(op);
304 }
305 }
306
307 Stmt visit(const LetStmt *op) override {
308 if (starts_with(op->name, func.name() + ".") &&
309 ends_with(op->name, ".tmp_buffer")) {
310
311 Stmt body = op->body;
312 Expr buf = Variable::make(type_of<halide_buffer_t *>(), op->name);
313
314 if (in_produce) {
315 // We're taking a crop of the buffer to act as an output
316 // to an extern stage. Update the valid min or max
317 // coordinate accordingly.
318
319 Expr leading_edge;
320 if (storage_dim.fold_forward) {
321 leading_edge =
322 Call::make(Int(32), Call::buffer_get_max, {buf, dim}, Call::Extern);
323 } else {
324 leading_edge =
325 Call::make(Int(32), Call::buffer_get_min, {buf, dim}, Call::Extern);
326 }
327
328 Stmt update_leading_edge =
329 Store::make(head, leading_edge, 0, Parameter(), const_true(), ModulusRemainder());
330 body = Block::make(update_leading_edge, body);
331
332 // We don't need to make sure the min is moving
333 // monotonically, because we can't do sliding window on
334 // extern stages, so we don't have to worry about whether
335 // we're preserving valid values from previous loop
336 // iterations.
337
338 if (func.schedule().async()) {
339 Expr old_leading_edge =
340 Load::make(Int(32), head, 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
341 Expr to_acquire;
342 if (storage_dim.fold_forward) {
343 to_acquire = leading_edge - old_leading_edge;
344 } else {
345 to_acquire = old_leading_edge - leading_edge;
346 }
347 body = Acquire::make(sema_var, to_acquire, body);
348 }
349 } else {
350 // We're taking a crop of the buffer to act as an input
351 // to an extern stage. Update the valid min or max
352 // coordinate accordingly.
353
354 Expr leading_edge;
355 if (storage_dim.fold_forward) {
356 leading_edge =
357 Call::make(Int(32), Call::buffer_get_min, {buf, dim}, Call::Extern) - 1 + storage_dim.fold_factor;
358 } else {
359 leading_edge =
360 Call::make(Int(32), Call::buffer_get_max, {buf, dim}, Call::Extern) + 1 - storage_dim.fold_factor;
361 }
362
363 Stmt update_leading_edge =
364 Store::make(tail, leading_edge, 0, Parameter(), const_true(), ModulusRemainder());
365 body = Block::make(update_leading_edge, body);
366
367 if (func.schedule().async()) {
368 Expr old_leading_edge =
369 Load::make(Int(32), tail, 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
370 Expr to_release;
371 if (storage_dim.fold_forward) {
372 to_release = leading_edge - old_leading_edge;
373 } else {
374 to_release = old_leading_edge - leading_edge;
375 }
376 Expr release_producer =
377 Call::make(Int(32), "halide_semaphore_release", {sema_var, to_release}, Call::Extern);
378 body = Block::make(Evaluate::make(release_producer), body);
379 }
380 }
381
382 return LetStmt::make(op->name, op->value, body);
383 } else {
384 return LetStmt::make(op->name, op->value, mutate(op->body));
385 }
386 }
387
388public:
389 InjectFoldingCheck(Function func,
390 string head, string tail,
391 string loop_var, Expr sema_var,
392 int dim, const StorageDim &storage_dim)
393 : func(std::move(func)),
394 head(std::move(head)), tail(std::move(tail)), loop_var(std::move(loop_var)), sema_var(std::move(sema_var)),
395 dim(dim), storage_dim(storage_dim) {
396 }

483// Attempt to fold the storage of a particular function in a statement
484class AttemptStorageFoldingOfFunction : public IRMutator {
485 Function func;
486 bool explicit_only;
487
488 using IRMutator::visit;
489
490 Stmt visit(const ProducerConsumer *op) override {
491 if (op->name == func.name()) {
492 // Can't proceed into the pipeline for this func
493 return op;
494 } else {
495 return IRMutator::visit(op);
496 }
497 }
498
499 Stmt visit(const For *op) override {
500 if (op->for_type != ForType::Serial && op->for_type != ForType::Unrolled) {
501 // We can't proceed into a parallel for loop.
502
503 // TODO: If there's no overlap between the region touched
504 // by the threads as this loop counter varies
505 // (i.e. there's no cross-talk between threads), then it's
506 // safe to proceed.
507 return op;
508 }
509
510 Stmt stmt;
511 Stmt body = op->body;
512
513 Box provided = box_provided(body, func.name());
514 Box required = box_required(body, func.name());
515 // For storage folding, we don't care about conditional reads.
516 required.used = Expr();
517 Box box = box_union(provided, required);
518
519 Expr loop_var = Variable::make(Int(32), op->name);
520 Expr loop_min = Variable::make(Int(32), op->name + ".loop_min");
521 Expr loop_max = Variable::make(Int(32), op->name + ".loop_max");
522
523 string dynamic_footprint;
524
525 Scope<Interval> bounds;
526 bounds.push(op->name, Interval(op->min, simplify(op->min + op->extent - 1)));
527
528 Scope<Interval> steady_bounds;
529 steady_bounds.push(op->name, Interval(simplify(op->min + 1), simplify(op->min + op->extent - 1)));
530
531 HasExternConsumer has_extern_consumer(func.name());
532 body.accept(&has_extern_consumer);
533
534 // Try each dimension in turn from outermost in
535 for (size_t i = box.size(); i > 0; i--) {
536 int dim = (int)(i - 1);
537
538 if (!box[dim].is_bounded()) {
539 continue;
540 }
541
542 Expr min = simplify(common_subexpression_elimination(box[dim].min));
543 Expr max = simplify(common_subexpression_elimination(box[dim].max));
544
545 if (is_const(min) || is_const(max)) {
546 debug(3) << "\nNot considering folding " << func.name()
547 << " over for loop over " << op->name
548 << " dimension " << i - 1 << "\n"
549 << " because the min or max are constants."
550 << "Min: " << min << "\n"
551 << "Max: " << max << "\n";
552 continue;
553 }
554
555 Expr min_provided, max_provided, min_required, max_required;
556 if (func.schedule().async() && !explicit_only) {
557 if (!provided.empty()) {
558 min_provided = simplify(provided[dim].min);
559 max_provided = simplify(provided[dim].max);
560 }
561 if (!required.empty()) {
562 min_required = simplify(required[dim].min);
563 max_required = simplify(required[dim].max);
564 }
565 }
566 string sema_name = func.name() + ".folding_semaphore." + unique_name('_');
567 Expr sema_var = Variable::make(type_of<halide_semaphore_t *>(), sema_name);
568
569 // Consider the initial iteration and steady state
570 // separately for all these proofs.
571 Expr loop_var = Variable::make(Int(32), op->name);
572 Expr steady_state = (op->min < loop_var);
573
574 Expr min_steady = simplify(substitute(steady_state, const_true(), min), true, steady_bounds);
575 Expr max_steady = simplify(substitute(steady_state, const_true(), max), true, steady_bounds);
576 Expr min_initial = simplify(substitute(steady_state, const_false(), min), true, bounds);
577 Expr max_initial = simplify(substitute(steady_state, const_false(), max), true, bounds);
578 Expr extent_initial = simplify(substitute(loop_var, op->min, max_initial - min_initial + 1), true, bounds);
579 Expr extent_steady = simplify(max_steady - min_steady + 1, true, steady_bounds);
580 Expr extent = Max::make(extent_initial, extent_steady);
581 extent = simplify(common_subexpression_elimination(extent), true, bounds);
582
583 // Find the StorageDim corresponding to dim.
584 const std::vector<StorageDim> &storage_dims = func.schedule().storage_dims();
585 auto storage_dim_i = std::find_if(storage_dims.begin(), storage_dims.end(),
586 [&](const StorageDim &i) { return i.var == func.args()[dim]; });
587 internal_assert(storage_dim_i != storage_dims.end());
588 const StorageDim &storage_dim = *storage_dim_i;
589
590 Expr explicit_factor;
591 if (!is_pure(min) ||
592 !is_pure(max) ||
593 has_extern_consumer.result ||
594 expr_uses_var(min, op->name) ||
595 expr_uses_var(max, op->name)) {
596 // We only use the explicit fold factor if the fold is
597 // relevant for this loop. If the fold isn't relevant
598 // for this loop, the added asserts will be too
599 // conservative.
600 explicit_factor = storage_dim.fold_factor;
601 }
602
603 debug(3) << "\nConsidering folding " << func.name()
604 << " over for loop over " << op->name
605 << " dimension " << i - 1 << "\n"
606 << "Min: " << min << "\n"
607 << "Max: " << max << "\n"
608 << "Extent: " << extent << "\n"
609 << "explicit_factor: " << explicit_factor << "\n";
610
611 // First, attempt to detect if the loop is monotonically
612 // increasing or decreasing (if we allow automatic folding).
613 bool can_fold_forwards = false, can_fold_backwards = false;
614
615 if (!explicit_only) {
616 // We can't clobber data that will be read later. If
617 // async, the producer can't un-release slots in the
618 // circular buffer.
619 can_fold_forwards = (is_monotonic(min, op->name) == Monotonic::Increasing);
620 can_fold_backwards = (is_monotonic(max, op->name) == Monotonic::Decreasing);
621 if (func.schedule().async()) {
622 // Our semaphore acquire primitive can't take
623 // negative values, so we can't un-acquire slots
624 // in the circular buffer.
625 can_fold_forwards &= (is_monotonic(max_provided, op->name) == Monotonic::Increasing);
626 can_fold_backwards &= (is_monotonic(min_provided, op->name) == Monotonic::Decreasing);
627 // We need to be able to analyze the required footprint to know how much to release
628 can_fold_forwards &= min_required.defined();
629 can_fold_backwards &= max_required.defined();
630 }
631 }
632
633 // Uncomment to pretend that static analysis always fails (for testing)
634 // can_fold_forwards = can_fold_backwards = false;
635
636 if (!can_fold_forwards && !can_fold_backwards) {
637 if (explicit_factor.defined()) {
638 // If we didn't find a monotonic dimension, and we
639 // have an explicit fold factor, we need to
640 // dynamically check that the min/max do in fact
641 // monotonically increase/decrease. We'll allocate
642 // some stack space to store the valid footprint,
643 // update it outside produce nodes, and check it
644 // outside consume nodes.
645
646 string head, tail;
647 if (func.schedule().async()) {
648 // If we're async, we need to keep a separate
649 // counter for the producer and consumer. They
650 // are coupled by a semaphore. The counter
651 // represents the max index the producer may
652 // write to. The invariant is that the
653 // semaphore count is the difference between
654 // the counters. So...
655 //
656 // when folding forwards, semaphore == head - tail
657 // when folding backwards, semaphore == tail - head
658 //
659 // We'll initialize to head = tail, and
660 // semaphore = 0. Every time the producer or
661 // consumer wants to move the counter, it must
662 // also acquire or release the semaphore to
663 // prevent them from diverging too far.
664 dynamic_footprint = func.name() + ".folding_semaphore." + op->name + unique_name('_');
665 head = dynamic_footprint + ".head";
666 tail = dynamic_footprint + ".tail";
667 } else {
668 dynamic_footprint = func.name() + "." + op->name + unique_name('_') + ".head";
669 head = tail = dynamic_footprint;
670 }
671
672 body = InjectFoldingCheck(func,
673 head, tail,
674 op->name,
675 sema_var,
676 dim,
677 storage_dim)
678 .mutate(body);
679
680 if (storage_dim.fold_forward) {
681 can_fold_forwards = true;
682 } else {
683 can_fold_backwards = true;
684 }
685 } else {
686 // Can't do much with this dimension
687 if (!explicit_only) {
688 debug(3) << "Not folding because loop min or max not monotonic in the loop variable\n"
689 << "min_initial = " << min_initial << "\n"
690 << "min_steady = " << min_steady << "\n"
691 << "max_initial = " << max_initial << "\n"
692 << "max_steady = " << max_steady << "\n";
693 } else {
694 debug(3) << "Not folding because there is no explicit storage folding factor\n";
695 }
696 continue;
697 }
698 }
699
700 internal_assert(can_fold_forwards || can_fold_backwards);
701
702 Expr factor;
703 if (explicit_factor.defined()) {
704 if (dynamic_footprint.empty() && !func.schedule().async()) {
705 // We were able to prove monotonicity
706 // statically, but we may need a runtime
707 // assertion for maximum extent. In many cases
708 // it will simplify away. For async schedules
709 // it gets dynamically tracked anyway.
710 Expr error = Call::make(Int(32), "halide_error_fold_factor_too_small",
711 {func.name(), storage_dim.var, explicit_factor, op->name, extent},
712 Call::Extern);
713 body = Block::make(AssertStmt::make(extent <= explicit_factor, error), body);
714 }
715 factor = explicit_factor;
716 } else {
717 // The max of the extent over all values of the loop variable must be a constant
718 Scope<Interval> scope;
719 scope.push(op->name, Interval(loop_min, loop_max));
720 Expr max_extent = find_constant_bound(extent, Direction::Upper, scope);
721 scope.pop(op->name);
722
723 const int max_fold = 1024;
724 const int64_t *const_max_extent = as_const_int(max_extent);
725 if (const_max_extent && *const_max_extent <= max_fold) {
726 factor = static_cast<int>(next_power_of_two(*const_max_extent));
727 } else {
728 // Try a little harder to find a bounding power of two
729 int e = max_fold * 2;
730 bool success = false;
731 while (e > 0 && can_prove(extent <= e / 2)) {
732 success = true;
733 e /= 2;
734 }
735 if (success) {
736 factor = e;
737 } else {
738 debug(3) << "Not folding because extent not bounded by a constant not greater than " << max_fold << "\n"
739 << "extent = " << extent << "\n"
740 << "max extent = " << max_extent << "\n";
741 // Try the next dimension
742 continue;
743 }
744 }
745 }
746
747 internal_assert(factor.defined());
748
749 if (!explicit_factor.defined()) {
750 VectorAccessOfFoldedDim vector_access_of_folded_dim{func.name(), dim};
751 body.accept(&vector_access_of_folded_dim);
752 if (vector_access_of_folded_dim.result) {
753 user_warning
754 << "Not folding Func " << func.name() << " along dimension " << func.args()[dim]
755 << " because there is vectorized access to that Func in that dimension and "
756 << "storage folding was not explicitly requested in the schedule. In previous "
757 << "versions of Halide this would have folded with factor " << factor
758 << ". To restore the old behavior add " << func.name()
759 << ".fold_storage(" << func.args()[dim] << ", " << factor
760 << ") to your schedule.\n";
761 // Try the next dimension
762 continue;
763 }
764 }
765
766 debug(3) << "Proceeding with factor " << factor << "\n";
767
768 Fold fold = {(int)i - 1, factor};
769 dims_folded.push_back(fold);
770 {
771 string head;
772 if (!dynamic_footprint.empty() && func.schedule().async()) {
773 head = dynamic_footprint + ".head";
774 } else {
775 head = dynamic_footprint;
776 }
777 body = FoldStorageOfFunction(func.name(), (int)i - 1, factor, head).mutate(body);
778 }
779
780 // If the producer is async, it can run ahead by
781 // some amount controlled by a semaphore.
782 if (func.schedule().async()) {
783 Semaphore sema;
784 sema.name = sema_name;
785 sema.var = sema_var;
786 sema.init = 0;
787
788 if (dynamic_footprint.empty()) {
789 // We are going to do the sem acquires and releases using static analysis of the boxes accessed.
790 sema.init = factor;
791 // Do the analysis of how much to acquire and release statically
792 Expr to_acquire, to_release;
793 if (can_fold_forwards) {
794 Expr max_provided_prev = substitute(op->name, loop_var - 1, max_provided);
795 Expr min_required_next = substitute(op->name, loop_var + 1, min_required);
796 to_acquire = max_provided - max_provided_prev; // This is the first time we use these entries
797 to_release = min_required_next - min_required; // This is the last time we use these entries
798 } else {
799 internal_assert(can_fold_backwards);
800 Expr min_provided_prev = substitute(op->name, loop_var - 1, min_provided);
801 Expr max_required_next = substitute(op->name, loop_var + 1, max_required);
802 to_acquire = min_provided_prev - min_provided; // This is the first time we use these entries
803 to_release = max_required - max_required_next; // This is the last time we use these entries
804 }
805
806 if (provided.used.defined()) {
807 to_acquire = select(provided.used, to_acquire, 0);
808 }
809 // We should always release the required region, even if we don't use it.
810
811 // On the first iteration, we need to acquire the extent of the region shared
812 // between the producer and consumer, and we need to release it on the last
813 // iteration.
814 to_acquire = select(loop_var > loop_min, to_acquire, extent);
815 to_release = select(loop_var < loop_max, to_release, extent);
816
817 // We may need dynamic assertions that a positive
818 // amount of the semaphore is acquired/released,
819 // and that the semaphore is initialized to a
820 // positive value. If we are able to prove it,
821 // these checks will simplify away.
822 string to_release_name = unique_name('t');
823 Expr to_release_var = Variable::make(Int(32), to_release_name);
824 string to_acquire_name = unique_name('t');
825 Expr to_acquire_var = Variable::make(Int(32), to_acquire_name);
826
827 Expr bad_fold_error =
828 Call::make(Int(32), "halide_error_bad_fold",
829 {func.name(), storage_dim.var, op->name},
830 Call::Extern);
831
832 Expr release_producer =
833 Call::make(Int(32), "halide_semaphore_release", {sema.var, to_release_var}, Call::Extern);
834 Stmt release = Evaluate::make(release_producer);
835 Stmt check_release = AssertStmt::make(to_release_var >= 0 && to_release <= factor, bad_fold_error);
836 release = Block::make(check_release, release);
837 release = LetStmt::make(to_release_name, to_release, release);
838
839 Stmt check_acquire = AssertStmt::make(to_acquire_var >= 0 && to_acquire_var <= factor, bad_fold_error);
840
841 body = Block::make(body, release);
842 body = Acquire::make(sema.var, to_acquire_var, body);
843 body = Block::make(check_acquire, body);
844 body = LetStmt::make(to_acquire_name, to_acquire, body);
845 } else {
846 // We injected runtime tracking and semaphore logic already
847 }
848 dims_folded.back().semaphore = sema;
849 }
850
851 if (!dynamic_footprint.empty()) {
852 if (func.schedule().async()) {
853 dims_folded.back().head = dynamic_footprint + ".head";
854 dims_folded.back().tail = dynamic_footprint + ".tail";
855 } else {
856 dims_folded.back().head = dynamic_footprint;
857 dims_folded.back().tail.clear();
858 }
859 dims_folded.back().fold_forward = storage_dim.fold_forward;
860 }
861
862 Expr min_next = substitute(op->name, loop_var + 1, min);
863
864 if (can_prove(max < min_next)) {
865 // There's no overlapping usage between loop
866 // iterations, so we can continue to search
867 // for further folding opportunities
868 // recursively.
869 } else if (!body.same_as(op->body)) {
870 stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
871 break;
872 } else {
873 stmt = op;
874 debug(3) << "Not folding because loop min or max not monotonic in the loop variable\n"
875 << "min = " << min << "\n"
876 << "max = " << max << "\n";
877 break;
878 }
879 }
880
881 // If there's no communication of values from one loop
882 // iteration to the next (which may happen due to sliding),
883 // then we're safe to fold an inner loop.
884 if (box_contains(provided, required)) {
885 body = mutate(body);
886 }
887
888 if (body.same_as(op->body)) {
889 stmt = op;
890 } else {
891 stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
892 }
893
894 if (func.schedule().async() && !dynamic_footprint.empty()) {
895 // Step the counters backwards over the entire extent of
896 // the realization, in case we're in an inner loop and are
897 // going to run this loop again with the same
898 // semaphore. Our invariant is that the difference between
899 // the two counters is the semaphore.
900 //
901 // Doing this instead of synchronizing and resetting the
902 // counters and semaphores lets producers advance to the
903 // next scanline while a consumer is still on the last few
904 // pixels of the previous scanline.
905
906 Expr head = Load::make(Int(32), dynamic_footprint + ".head", 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
907 Expr tail = Load::make(Int(32), dynamic_footprint + ".tail", 0, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
908 Expr step = Variable::make(Int(32), func.name() + ".extent." + std::to_string(dims_folded.back().dim)) + dims_folded.back().factor;
909 Stmt reset_head = Store::make(dynamic_footprint + ".head_next", head - step, 0, Parameter(), const_true(), ModulusRemainder());
910 Stmt reset_tail = Store::make(dynamic_footprint + ".tail_next", tail - step, 0, Parameter(), const_true(), ModulusRemainder());
911 stmt = Block::make({stmt, reset_head, reset_tail});
912 }
913 return stmt;
914 }

916public:
917 struct Fold {
918 int dim;
919 Expr factor;
920 Semaphore semaphore;
921 string head, tail;
922 bool fold_forward;
923 };
924 vector<Fold> dims_folded;
925
926 AttemptStorageFoldingOfFunction(Function f, bool explicit_only)
927 : func(std::move(f)), explicit_only(explicit_only) {
928 }
929};

931// Look for opportunities for storage folding in a statement
932class StorageFolding : public IRMutator {
933 const map<string, Function> &env;
934
935 using IRMutator::visit;
936
937 Stmt visit(const Realize *op) override {
938 Stmt body = mutate(op->body);
939
940 // Get the function associated with this realization, which
941 // contains the explicit fold directives from the schedule.
942 auto func_it = env.find(op->name);
943 Function func = func_it != env.end() ? func_it->second : Function();
944
945 // Don't attempt automatic storage folding if there is
946 // more than one produce node for this func.
947 bool explicit_only = count_producers(body, op->name) != 1;
948 AttemptStorageFoldingOfFunction folder(func, explicit_only);
949 if (explicit_only) {
950 debug(3) << "Attempting to fold " << op->name << " explicitly\n";
951 } else {
952 debug(3) << "Attempting to fold " << op->name << " automatically or explicitly\n";
953 }
954 body = folder.mutate(body);
955
956 if (body.same_as(op->body)) {
957 return op;
958 } else if (folder.dims_folded.empty()) {
959 return Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, body);
960 } else {
961 Region bounds = op->bounds;
962
963 // Collapse down the extent in the folded dimension
964 for (size_t i = 0; i < folder.dims_folded.size(); i++) {
965 int d = folder.dims_folded[i].dim;
966 Expr f = folder.dims_folded[i].factor;
967 internal_assert(d >= 0 &&
968 d < (int)bounds.size());
969 bounds[d] = Range(0, f);
970 }
971
972 Stmt stmt = Realize::make(op->name, op->types, op->memory_type, bounds, op->condition, body);
973
974 // Each fold may have an associated semaphore that needs initialization, along with some counters
975 for (const auto &fold : folder.dims_folded) {
976 auto sema = fold.semaphore;
977 if (sema.var.defined()) {
978 Expr sema_space = Call::make(type_of<halide_semaphore_t *>(), "halide_make_semaphore",
979 {sema.init}, Call::Extern);
980 stmt = LetStmt::make(sema.name, sema_space, stmt);
981 }
982 Expr init;
983 if (fold.fold_forward) {
984 init = op->bounds[fold.dim].min;
985 } else {
986 init = op->bounds[fold.dim].min + op->bounds[fold.dim].extent - 1;
987 }
988 if (!fold.head.empty()) {
989 stmt = Block::make(Store::make(fold.head + "_next", init, 0, Parameter(), const_true(), ModulusRemainder()), stmt);
990 stmt = Allocate::make(fold.head + "_next", Int(32), MemoryType::Stack, {}, const_true(), stmt);
991 stmt = Block::make(Store::make(fold.head, init, 0, Parameter(), const_true(), ModulusRemainder()), stmt);
992 stmt = Allocate::make(fold.head, Int(32), MemoryType::Stack, {}, const_true(), stmt);
993 }
994 if (!fold.tail.empty()) {
995 internal_assert(func.schedule().async()) << "Expected a single counter for synchronous folding";
996 stmt = Block::make(Store::make(fold.tail + "_next", init, 0, Parameter(), const_true(), ModulusRemainder()), stmt);
997 stmt = Allocate::make(fold.tail + "_next", Int(32), MemoryType::Stack, {}, const_true(), stmt);
998 stmt = Block::make(Store::make(fold.tail, init, 0, Parameter(), const_true(), ModulusRemainder()), stmt);
999 stmt = Allocate::make(fold.tail, Int(32), MemoryType::Stack, {}, const_true(), stmt);
1000 }
1001 }
1002
1003 return stmt;
1004 }
1005 }

1009 : env(env) {
1010 }
