14/* static */
15void ObjectInstanceRegistry::register_instance(void *this_ptr, size_t size, Kind kind,
16 void *subject_ptr, const void *introspection_helper) {
17 ObjectInstanceRegistry &registry = get_registry();
18 std::lock_guard<std::mutex> lock(registry.mutex);
19 uintptr_t key = (uintptr_t)this_ptr;
20 internal_assert(registry.instances.find(key) == registry.instances.end());
21 if (introspection_helper) {
22 registry.instances[key] = InstanceInfo(size, kind, subject_ptr, true);
23 Introspection::register_heap_object(this_ptr, size, introspection_helper);
24 } else {
25 registry.instances[key] = InstanceInfo(size, kind, subject_ptr, false);
26 }
27}

29/* static */
30void ObjectInstanceRegistry::unregister_instance(void *this_ptr) {
31 ObjectInstanceRegistry &registry = get_registry();
32 std::lock_guard<std::mutex> lock(registry.mutex);
33 uintptr_t key = (uintptr_t)this_ptr;
34 std::map<uintptr_t, InstanceInfo>::iterator it = registry.instances.find(key);
35 internal_assert(it != registry.instances.end());
36 if (it->second.registered_for_introspection) {
37 Introspection::deregister_heap_object(this_ptr, it->second.size);
38 }
39 registry.instances.erase(it);
40}

42/* static */
43std::vector<void *> ObjectInstanceRegistry::instances_in_range(void *start, size_t size,
44 Kind kind) {
45 std::vector<void *> results;
46
47 ObjectInstanceRegistry &registry = get_registry();
48 std::lock_guard<std::mutex> lock(registry.mutex);
49
50 std::map<uintptr_t, InstanceInfo>::const_iterator it =
51 registry.instances.lower_bound((uintptr_t)start);
52
53 uintptr_t limit_ptr = ((uintptr_t)start) + size;
54 while (it != registry.instances.end() && it->first < limit_ptr) {
55 if (it->second.kind == kind) {
56 results.push_back(it->second.subject_ptr);
57 }
58
59 if (it->first > (uintptr_t)start && it->second.size != 0) {
60 // Skip over containers that we enclose
61 it = registry.instances.lower_bound(it->first + it->second.size);
62 } else {
63 it++;
64 }
65 }
66
67 return results;
68}
