12class CountVarUses : public IRVisitor {
13 std::map<std::string, int> &var_uses;
14
15 void visit(const Variable *var) override {
16 var_uses[var->name]++;
17 }
18
19 void visit(const Load *op) override {
20 var_uses[op->name]++;
21 IRVisitor::visit(op);
22 }
23
24 void visit(const Store *op) override {
25 var_uses[op->name]++;
26 IRVisitor::visit(op);
27 }
28
29 using IRVisitor::visit;
30
31public:
32 CountVarUses(std::map<std::string, int> &var_uses)
33 : var_uses(var_uses) {
34 }
35};
36
37template<typename StmtOrExpr>
38void count_var_uses(StmtOrExpr x, std::map<std::string, int> &var_uses) {
39 CountVarUses counter(var_uses);
40 x.accept(&counter);
41}

45template<typename LetOrLetStmt, typename Body>
46Body Simplify::simplify_let(const LetOrLetStmt *op, ExprInfo *bounds) {
47
48 // Lets are often deeply nested. Get the intermediate state off
49 // the call stack where it could overflow onto an explicit stack.
50 struct Frame {
51 const LetOrLetStmt *op;
52 Expr value, new_value;
53 string new_name;
54 bool new_value_alignment_tracked = false, new_value_bounds_tracked = false;
55 bool value_alignment_tracked = false, value_bounds_tracked = false;
56 Frame(const LetOrLetStmt *op)
57 : op(op) {
58 }
59 };
60
61 vector<Frame> frames;
62 Body result;
63
64 while (op) {
65 frames.emplace_back(op);
66 Frame &f = frames.back();
67
68 internal_assert(!var_info.contains(op->name))
69 << "Simplify only works on code where every name is unique. Repeated name: " << op->name << "\n";
70
71 // If the value is trivial, make a note of it in the scope so
72 // we can subs it in later
73 ExprInfo value_bounds;
74 f.value = mutate(op->value, &value_bounds);
75
76 // Iteratively peel off certain operations from the let value and push them inside.
77 f.new_value = f.value;
78 f.new_name = op->name + ".s";
79 Expr new_var = Variable::make(f.new_value.type(), f.new_name);
80 Expr replacement = new_var;
81
82 debug(4) << "simplify let " << op->name << " = " << f.value << " in...\n";
83
84 while (true) {
85 const Variable *var = f.new_value.template as<Variable>();
86 const Add *add = f.new_value.template as<Add>();
87 const Sub *sub = f.new_value.template as<Sub>();
88 const Mul *mul = f.new_value.template as<Mul>();
89 const Div *div = f.new_value.template as<Div>();
90 const Mod *mod = f.new_value.template as<Mod>();
91 const Min *min = f.new_value.template as<Min>();
92 const Max *max = f.new_value.template as<Max>();
93 const Ramp *ramp = f.new_value.template as<Ramp>();
94 const Cast *cast = f.new_value.template as<Cast>();
95 const Broadcast *broadcast = f.new_value.template as<Broadcast>();
96 const Shuffle *shuffle = f.new_value.template as<Shuffle>();
97 const Variable *var_b = nullptr;
98 const Variable *var_a = nullptr;
99 const Call *tag = nullptr;
100
101 if (add) {
102 var_a = add->a.as<Variable>();
103 var_b = add->b.as<Variable>();
104 } else if (sub) {
105 var_b = sub->b.as<Variable>();
106 } else if (mul) {
107 var_b = mul->b.as<Variable>();
108 } else if (shuffle && shuffle->is_concat() && shuffle->vectors.size() == 2) {
109 var_a = shuffle->vectors[0].as<Variable>();
110 var_b = shuffle->vectors[1].as<Variable>();
111 }
112
113 if (is_const(f.new_value)) {
114 replacement = substitute(f.new_name, f.new_value, replacement);
115 f.new_value = Expr();
116 break;
117 } else if (var) {
118 replacement = substitute(f.new_name, var, replacement);
119 f.new_value = Expr();
120 break;
121 } else if (add && (is_const(add->b) || var_b)) {
122 replacement = substitute(f.new_name, Add::make(new_var, add->b), replacement);
123 f.new_value = add->a;
124 } else if (add && var_a) {
125 replacement = substitute(f.new_name, Add::make(add->a, new_var), replacement);
126 f.new_value = add->b;
127 } else if (mul && (is_const(mul->b) || var_b)) {
128 replacement = substitute(f.new_name, Mul::make(new_var, mul->b), replacement);
129 f.new_value = mul->a;
130 } else if (div && is_const(div->b)) {
131 replacement = substitute(f.new_name, Div::make(new_var, div->b), replacement);
132 f.new_value = div->a;
133 } else if (sub && (is_const(sub->b) || var_b)) {
134 replacement = substitute(f.new_name, Sub::make(new_var, sub->b), replacement);
135 f.new_value = sub->a;
136 } else if (sub && is_const(sub->a)) {
137 replacement = substitute(f.new_name, Sub::make(sub->a, new_var), replacement);
138 f.new_value = sub->b;
139 } else if (mod && is_const(mod->b)) {
140 replacement = substitute(f.new_name, Mod::make(new_var, mod->b), replacement);
141 f.new_value = mod->a;
142 } else if (min && is_const(min->b)) {
143 replacement = substitute(f.new_name, Min::make(new_var, min->b), replacement);
144 f.new_value = min->a;
145 } else if (max && is_const(max->b)) {
146 replacement = substitute(f.new_name, Max::make(new_var, max->b), replacement);
147 f.new_value = max->a;
148 } else if (ramp && is_const(ramp->stride)) {
149 f.new_value = ramp->base;
150 new_var = Variable::make(f.new_value.type(), f.new_name);
151 replacement = substitute(f.new_name, Ramp::make(new_var, ramp->stride, ramp->lanes), replacement);
152 } else if (broadcast) {
153 f.new_value = broadcast->value;
154 new_var = Variable::make(f.new_value.type(), f.new_name);
155 replacement = substitute(f.new_name, Broadcast::make(new_var, broadcast->lanes), replacement);
156 } else if (cast && cast->type.bits() > cast->value.type().bits()) {
157 // Widening casts get pushed inwards, narrowing casts
158 // stay outside. This keeps the temporaries small, and
159 // helps with peephole optimizations in codegen that
160 // skip the widening entirely.
161 f.new_value = cast->value;
162 new_var = Variable::make(f.new_value.type(), f.new_name);
163 replacement = substitute(f.new_name, Cast::make(cast->type, new_var), replacement);
164 } else if (shuffle && shuffle->is_slice()) {
165 // Replacing new_value below might free the shuffle
166 // indices vector, so save them now.
167 std::vector<int> slice_indices = shuffle->indices;
168 f.new_value = Shuffle::make_concat(shuffle->vectors);
169 new_var = Variable::make(f.new_value.type(), f.new_name);
170 replacement = substitute(f.new_name, Shuffle::make({new_var}, slice_indices), replacement);
171 } else if (shuffle && shuffle->is_concat() &&
172 ((var_a && !var_b) || (!var_a && var_b))) {
173 new_var = Variable::make(var_a ? shuffle->vectors[1].type() : shuffle->vectors[0].type(), f.new_name);
174 Expr op_a = var_a ? shuffle->vectors[0] : new_var;
175 Expr op_b = var_a ? new_var : shuffle->vectors[1];
176 replacement = substitute(f.new_name, Shuffle::make_concat({op_a, op_b}), replacement);
177 f.new_value = var_a ? shuffle->vectors[1] : shuffle->vectors[0];
178 } else if ((tag = Call::as_tag(f.new_value)) != nullptr && !tag->is_intrinsic(Call::strict_float)) {
179 // Most tags should be stripped here, but not strict_float(); removing it will change the semantics
180 // of the let-expr we are producing.
181 replacement = substitute(f.new_name, Call::make(tag->type, tag->name, {new_var}, Call::PureIntrinsic), replacement);
182 f.new_value = tag->args[0];
183 } else {
184 break;
185 }
186 }
187
188 if (f.new_value.same_as(f.value)) {
189 // Nothing to substitute
190 f.new_value = Expr();
191 replacement = Expr();
192 } else {
193 debug(4) << "new let " << f.new_name << " = " << f.new_value << " in ... " << replacement << " ...\n";
194 }
195
196 VarInfo info;
197 info.old_uses = 0;
198 info.new_uses = 0;
199 info.replacement = replacement;
200
201 var_info.push(op->name, info);
202
203 // Before we enter the body, track the alignment info
204
205 if (f.new_value.defined() && no_overflow_scalar_int(f.new_value.type())) {
206 // Remutate new_value to get updated bounds
207 ExprInfo new_value_bounds;
208 f.new_value = mutate(f.new_value, &new_value_bounds);
209 if (new_value_bounds.min_defined || new_value_bounds.max_defined || new_value_bounds.alignment.modulus != 1) {
210 // There is some useful information
211 bounds_and_alignment_info.push(f.new_name, new_value_bounds);
212 f.new_value_bounds_tracked = true;
213 }
214 }
215
216 if (no_overflow_scalar_int(f.value.type())) {
217 if (value_bounds.min_defined || value_bounds.max_defined || value_bounds.alignment.modulus != 1) {
218 bounds_and_alignment_info.push(op->name, value_bounds);
219 f.value_bounds_tracked = true;
220 }
221 }
222
223 result = op->body;
224 op = result.template as<LetOrLetStmt>();
225 }
226
227 result = mutate_let_body(result, bounds);
228
229 // TODO: var_info and vars_used are pretty redundant; however, at the time
230 // of writing, both cover cases that the other does not:
231 // - var_info prevents duplicate lets from being generated, even
232 // from different Frame objects.
233 // - vars_used avoids dead lets being generated in cases where vars are
234 // seen as used by var_info, and then later removed.
235 std::map<std::string, int> vars_used;
236 count_var_uses(result, vars_used);
237
238 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
239 if (it->value_bounds_tracked) {
240 bounds_and_alignment_info.pop(it->op->name);
241 }
242 if (it->new_value_bounds_tracked) {
243 bounds_and_alignment_info.pop(it->new_name);
244 }
245
246 VarInfo info = var_info.get(it->op->name);
247 var_info.pop(it->op->name);
248
249 if (it->new_value.defined() && (info.new_uses > 0 && vars_used.count(it->new_name) > 0)) {
250 // The new name/value may be used
251 result = LetOrLetStmt::make(it->new_name, it->new_value, result);
252 count_var_uses(it->new_value, vars_used);
253 }
254
255 if ((!remove_dead_code && std::is_same<LetOrLetStmt, LetStmt>::value) ||
256 (info.old_uses > 0 && vars_used.count(it->op->name) > 0)) {
257 // The old name is still in use. We'd better keep it as well.
258 result = LetOrLetStmt::make(it->op->name, it->value, result);
259 count_var_uses(it->value, vars_used);
260 }
261
262 const LetOrLetStmt *new_op = result.template as<LetOrLetStmt>();
263 if (new_op &&
264 new_op->name == it->op->name &&
265 new_op->body.same_as(it->op->body) &&
266 new_op->value.same_as(it->op->value)) {
267 result = it->op;
268 }
269 }
270
271 return result;
272}
