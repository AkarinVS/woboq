10class SelectGPUAPI : public IRMutator {
11 using IRMutator::visit;
12
13 DeviceAPI default_api, parent_api;
14
15 Expr visit(const Call *op) override {
16 if (op->name == "halide_default_device_interface") {
17 return make_device_interface_call(default_api);
18 } else {
19 return IRMutator::visit(op);
20 }
21 }
22
23 Stmt visit(const For *op) override {
24 DeviceAPI selected_api = op->device_api;
25 if (op->device_api == DeviceAPI::Default_GPU) {
26 selected_api = default_api;
27 }
28
29 DeviceAPI old_parent_api = parent_api;
30 parent_api = selected_api;
31 Stmt stmt = IRMutator::visit(op);
32 parent_api = old_parent_api;
33
34 op = stmt.as<For>();
35 internal_assert(op);
36
37 if (op->device_api != selected_api) {
38 return For::make(op->name, op->min, op->extent, op->for_type, selected_api, op->body);
39 }
40 return stmt;
41 }
42
43public:
44 SelectGPUAPI(const Target &t) {
45 default_api = get_default_device_api_for_target(t);
46 parent_api = DeviceAPI::Host;
47 }
48};
