81std::string get_env_variable(char const *env_var_name) {
82 if (!env_var_name) {
83 return "";
84 }
85
86#ifdef _MSC_VER
87 // call getenv_s without a buffer to determine the correct string length:
88 size_t length = 0;
89 if ((getenv_s(&length, nullptr, 0, env_var_name) != 0) || (length == 0)) {
90 return "";
91 }
92 // call it again to retrieve the value of the environment variable;
93 // note that 'length' already accounts for the null-terminator
94 std::string lvl(length - 1, '@');
95 size_t read = 0;
96 if ((getenv_s(&read, &lvl[0], length, env_var_name) != 0) || (read != length)) {
97 return "";
98 }
99 return lvl;
100#else
101 char *lvl = getenv(env_var_name);
102 if (lvl) {
103 return std::string(lvl);
104 }
105 return "";
106#endif
107}

135// We use 64K of memory to store unique counters for the purpose of
136// making names unique. Using less memory increases the likelihood of
137// hash collisions. This wouldn't break anything, but makes stmts
138// slightly confusing to read because names that are actually unique
139// will get suffixes that falsely hint that they are not.
140
141const int num_unique_name_counters = (1 << 14);
142
143// We want to init these to zero, but cannot use = {0} because that
144// would invoke a (deleted) copy ctor. The default initialization for
145// atomics doesn't guarantee any actual initialization. Fortunately
146// this is a global, which is always zero-initialized.
147std::atomic<int> unique_name_counters[num_unique_name_counters] = {};
148
149int unique_count(size_t h) {
150 h = h & (num_unique_name_counters - 1);
151 return unique_name_counters[h]++;
152}
153} // namespace

155// There are three possible families of names returned by the methods below:
156// 1) char pattern: (char that isn't '$') + number (e.g. v234)
157// 2) string pattern: (string without '$') + '$' + number (e.g. fr#nk82$42)
158// 3) a string that does not match the patterns above
159// There are no collisions within each family, due to the unique_count
160// done above, and there can be no collisions across families by
161// construction.
162
163string unique_name(char prefix) {
164 if (prefix == '$') {
165 prefix = '_';
166 }
167 return prefix + std::to_string(unique_count((size_t)(prefix)));
168}
169
170string unique_name(const std::string &prefix) {
171 string sanitized = prefix;
172
173 // Does the input string look like something returned from unique_name(char)?
174 bool matches_char_pattern = true;
175
176 // Does the input string look like something returned from unique_name(string)?
177 bool matches_string_pattern = true;
178
179 // Rewrite '$' to '_'. This is a many-to-one mapping, but that's
180 // OK, we're about to hash anyway. It just means that some names
181 // will share the same counter.
182 int num_dollars = 0;
183 for (size_t i = 0; i < sanitized.size(); i++) {
184 if (sanitized[i] == '$') {
185 num_dollars++;
186 sanitized[i] = '_';
187 }
188 if (i > 0 && !isdigit(sanitized[i])) {
189 // Found a non-digit after the first char
190 matches_char_pattern = false;
191 if (num_dollars) {
192 // Found a non-digit after a '$'
193 matches_string_pattern = false;
194 }
195 }
196 }
197 matches_string_pattern &= num_dollars == 1;
198 matches_char_pattern &= prefix.size() > 1;
199
200 // Then add a suffix that's globally unique relative to the hash
201 // of the sanitized name.
202 int count = unique_count(std::hash<std::string>()(sanitized));
203 if (count == 0) {
204 // We can return the name as-is if there's no risk of it
205 // looking like something unique_name has ever returned in the
206 // past or will ever return in the future.
207 if (!matches_char_pattern && !matches_string_pattern) {
208 return prefix;
209 }
210 }
211
212 return sanitized + "$" + std::to_string(count);
213}

215bool starts_with(const string &str, const string &prefix) {
216 if (str.size() < prefix.size()) {
217 return false;
218 }
219 for (size_t i = 0; i < prefix.size(); i++) {
220 if (str[i] != prefix[i]) {
221 return false;
222 }
223 }
224 return true;
225}

227bool ends_with(const string &str, const string &suffix) {
228 if (str.size() < suffix.size()) {
229 return false;
230 }
231 size_t off = str.size() - suffix.size();
232 for (size_t i = 0; i < suffix.size(); i++) {
233 if (str[off + i] != suffix[i]) {
234 return false;
235 }
236 }
237 return true;
238}

266std::vector<std::string> split_string(const std::string &source, const std::string &delim) {
267 std::vector<std::string> elements;
268 size_t start = 0;
269 size_t found = 0;
270 while ((found = source.find(delim, start)) != std::string::npos) {
271 elements.push_back(source.substr(start, found - start));
272 start = found + delim.size();
273 }
274
275 // If start is exactly source.size(), the last thing in source is a
276 // delimiter, in which case we want to add an empty string to elements.
277 if (start <= source.size()) {
278 elements.push_back(source.substr(start, std::string::npos));
279 }
280 return elements;
281}

359// GetTempPath() will fail rudely if env vars aren't set properly,
360// which is the case when we run under a tool in Bazel. Instead,
361// look for the current user's AppData/Local/Temp path, which
362// should be valid and writable in all versions of Windows that
363// we support for compilation purposes.
364std::string get_windows_tmp_dir() {
365 // Allow overriding of the tmpdir on Windows via an env var;
366 // some Windows configs can (apparently) lock down AppData/Local/Temp
367 // via policy, making various things break. (Note that this is intended
368 // to be a short-lived workaround; we would prefer to be able to avoid
369 // requiring this sort of band-aid if possible.)
370 std::string tmp_dir = get_env_variable("HL_WINDOWS_TMP_DIR");
371 if (!tmp_dir.empty()) {
372 return tmp_dir;
373 }
374
375 PWSTR wlocal_path;
376 HRESULT ret = SHGetKnownFolderPath(FOLDERID_LocalAppData, 0, nullptr, &wlocal_path);
377 internal_assert(ret == S_OK) << "Unable to get Local AppData folder; error " << GetLastError() << "\n";
378
379 std::string tmp = from_utf16(wlocal_path);
380 CoTaskMemFree(wlocal_path);
381
382 tmp = replace_all(tmp, "\\", "/");
383 if (tmp.back() != '/') tmp += '/';
384 tmp += "Temp/";
385 return tmp;
386}

391std::string file_make_temp(const std::string &prefix, const std::string &suffix) {
392 internal_assert(prefix.find('/') == string::npos &&
393 prefix.find('\\') == string::npos &&
394 suffix.find('/') == string::npos &&
395 suffix.find('\\') == string::npos);
396#ifdef _WIN32
397 // Windows implementations of mkstemp() try to create the file in the root
398 // directory Unfortunately, that requires ADMIN privileges, which are not
399 // guaranteed here.
400 std::wstring tmp_dir = from_utf8(get_windows_tmp_dir());
401 std::wstring wprefix = from_utf8(prefix);
402
403 WCHAR tmp_file[MAX_PATH];
404 // Note that GetTempFileNameW() actually creates the file.
405 DWORD ret = GetTempFileNameW(tmp_dir.c_str(), wprefix.c_str(), 0, tmp_file);
406 internal_assert(ret != 0) << "GetTempFileNameW() failed; error " << GetLastError() << "\n";
407 return from_utf16(tmp_file);
408#else
409 std::string templ = "/tmp/" + prefix + "XXXXXX" + suffix;
410 // Copy into a temporary buffer, since mkstemp modifies the buffer in place.
411 std::vector<char> buf(templ.size() + 1);
412 strcpy(&buf[0], templ.c_str());
413 int fd = mkstemps(&buf[0], suffix.size());
414 internal_assert(fd != -1) << "Unable to create temp file for (" << &buf[0] << ")\n";
415 close(fd);
416 return std::string(&buf[0]);
417#endif
418}

420std::string dir_make_temp() {
421#ifdef _WIN32
422 std::string tmp_dir = get_windows_tmp_dir();
423 // There's no direct API to do this in Windows;
424 // our clunky-but-adequate approach here is to use
425 // CoCreateGuid() to create a probably-unique name.
426 // Add a limit on the number of tries just in case.
427 for (int tries = 0; tries < 100; ++tries) {
428 GUID guid;
429 HRESULT hr = CoCreateGuid(&guid);
430 internal_assert(hr == S_OK);
431 std::ostringstream name;
432 name << std::hex
433 << std::setfill('0')
434 << std::setw(8)
435 << guid.Data1
436 << std::setw(4)
437 << guid.Data2
438 << guid.Data3
439 << std::setw(2);
440 for (int i = 0; i < 8; i++) {
441 name << (int)guid.Data4[i];
442 }
443 std::string dir = tmp_dir + name.str();
444 std::wstring wdir = from_utf8(dir);
445 BOOL success = CreateDirectoryW(wdir.c_str(), nullptr);
446 if (success) {
447 debug(1) << "temp dir is: " << dir << "\n";
448 return dir;
449 }
450 // If name already existed, just loop and try again.
451 // Any other error, break from loop and fail.
452 if (GetLastError() != ERROR_ALREADY_EXISTS) {
453 break;
454 }
455 }
456 internal_error << "Unable to create temp directory in " << tmp_dir << "\n";
457 return "";
458#else
459 std::string templ = "/tmp/XXXXXX";
460 // Copy into a temporary buffer, since mkdtemp modifies the buffer in place.
461 std::vector<char> buf(templ.size() + 1);
462 strcpy(&buf[0], templ.c_str());
463 char *result = mkdtemp(&buf[0]);
464 internal_assert(result != nullptr) << "Unable to create temp directory.\n";
465 return std::string(result);
466#endif
467}

492bool add_would_overflow(int bits, int64_t a, int64_t b) {
493 int64_t max_val = 0x7fffffffffffffffLL >> (64 - bits);
494 int64_t min_val = -max_val - 1;
495 return ((b > 0 && a > max_val - b) || // (a + b) > max_val, rewritten to avoid overflow
496 (b < 0 && a < min_val - b)); // (a + b) < min_val, rewritten to avoid overflow
497}
498
499bool sub_would_overflow(int bits, int64_t a, int64_t b) {
500 int64_t max_val = 0x7fffffffffffffffLL >> (64 - bits);
501 int64_t min_val = -max_val - 1;
502 return ((b < 0 && a > max_val + b) || // (a - b) > max_val, rewritten to avoid overflow
503 (b > 0 && a < min_val + b)); // (a - b) < min_val, rewritten to avoid overflow
504}
505
506bool mul_would_overflow(int bits, int64_t a, int64_t b) {
507 int64_t max_val = 0x7fffffffffffffffLL >> (64 - bits);
508 int64_t min_val = -max_val - 1;
509 if (a == 0) {
510 return false;
511 } else if (a == -1) {
512 return b == min_val;
513 } else {
514 // Do the multiplication as a uint64, for which overflow is
515 // well defined, then cast the bits back to int64 to get
516 // multiplication modulo 2^64.
517 int64_t ab = (int64_t)((uint64_t)a) * ((uint64_t)b);
518 // The first two clauses catch overflow mod 2^bits, assuming
519 // no 64-bit overflow occurs, and the third clause catches
520 // 64-bit overflow.
521 return ab < min_val || ab > max_val || (ab / a != b);
522 }

539void halide_toc_impl(const char *file, int line) {
540 auto t1 = tick_stack.back();
541 auto t2 = std::chrono::high_resolution_clock::now();
542 std::chrono::duration<double> diff = t2 - t1.time;
543 tick_stack.pop_back();
544 for (size_t i = 0; i < tick_stack.size(); i++) {
545 debug(1) << " ";
546 }
547 string f = file;
548 f = split_string(f, "/").back();
549 debug(1) << t1.file << ":" << t1.line << " ... " << f << ":" << line << " : " << diff.count() * 1000 << " ms\n";
550}

552std::string c_print_name(const std::string &name) {
553 ostringstream oss;
554
555 // Prefix an underscore to avoid reserved words (e.g. a variable named "while")
556 if (isalpha(name[0])) {
557 oss << "_";
558 }
559
560 for (size_t i = 0; i < name.size(); i++) {
561 if (name[i] == '.') {
562 oss << "_";
563 } else if (name[i] == '$') {
564 oss << "__";
565 } else if (name[i] != '_' && !isalnum(name[i])) {
566 oss << "___";
567 } else {
568 oss << name[i];
569 }
570 }
571 return oss.str();
572}

609void run_with_large_stack(const std::function<void()> &action) {
610#if _WIN32
611 constexpr auto required_stack = 8 * 1024 * 1024;
612
613 // Only exists for its address, which is used to compute remaining stack space.
614 ULONG_PTR approx_stack_pos;
615
616 ULONG_PTR stack_low, stack_high;
617 GetCurrentThreadStackLimits(&stack_low, &stack_high);
618 ptrdiff_t stack_remaining = (char *)&approx_stack_pos - (char *)stack_low;
619
620 if (stack_remaining < required_stack) {
621 debug(1) << "Insufficient stack space (" << stack_remaining << " bytes). Switching to fiber with " << required_stack << "-byte stack.\n";
622
623 auto was_a_fiber = IsThreadAFiber();
624
625 auto *main_fiber = was_a_fiber ? GetCurrentFiber() : ConvertThreadToFiber(nullptr);
626 internal_assert(main_fiber) << "ConvertThreadToFiber failed with code: " << GetLastError() << "\n";
627
628 GenericFiberArgs fiber_args{action, main_fiber};
629 auto *lower_fiber = CreateFiber(required_stack, generic_fiber_entry_point, &fiber_args);
630 internal_assert(lower_fiber) << "CreateFiber failed with code: " << GetLastError() << "\n";
631
632 SwitchToFiber(lower_fiber);
633 DeleteFiber(lower_fiber);
634
635 debug(1) << "Returned from fiber.\n";
636
637#ifdef HALIDE_WITH_EXCEPTIONS
638 if (fiber_args.exception) {
639 debug(1) << "Fiber threw exception. Rethrowing...\n";
640 std::rethrow_exception(fiber_args.exception);
641 }
642#endif
643
644 if (!was_a_fiber) {
645 BOOL success = ConvertFiberToThread();
646 internal_assert(success) << "ConvertFiberToThread failed with code: " << GetLastError() << "\n";
647 }
648
649 return;
650 }
651
652#endif
653
654 action();
655}

659void load_plugin(const std::string &lib_name) {
660#ifdef _WIN32
661 std::string lib_path = lib_name;
662 if (lib_path.find('.') == std::string::npos) {
663 lib_path += ".dll";
664 }
665
666 std::wstring wide_lib = from_utf8(lib_path);
667 HMODULE library = LoadLibraryW(wide_lib.c_str());
668 if (!library) {
669 DWORD error = GetLastError();
670 LPWSTR message = nullptr;
671 FormatMessageW(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
672 nullptr, error, 0, reinterpret_cast<LPWSTR>(&message), 0, nullptr);
673
674 user_assert(message)
675 << "Failed to load: " << lib_path << ".\n"
676 << "FormatMessage failed while processing error in LoadLibraryW (errno "
677 << error << ").\n";
678
679 std::string err_msg = from_utf16(message);
680 LocalFree(message);
681 user_error << "Failed to load: " << lib_path << ";\n"
682 << "LoadLibraryW failed with error " << error << ": "
683 << err_msg << "\n";
684 }
685#else
686 std::string lib_path = lib_name;
