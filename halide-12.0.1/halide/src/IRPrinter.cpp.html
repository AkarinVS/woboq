19ostream &operator<<(ostream &out, const Type &type) {
20 switch (type.code()) {
21 case Type::Int:
22 out << "int";
23 break;
24 case Type::UInt:
25 out << "uint";
26 break;
27 case Type::Float:
28 out << "float";
29 break;
30 case Type::Handle:
31 if (type.handle_type) {
32 out << "(" << type.handle_type->inner_name.name << " *)";
33 } else {
34 out << "(void *)";
35 }
36 break;
37 case Type::BFloat:
38 out << "bfloat";
39 break;
40 }
41 if (!type.is_handle()) {
42 out << type.bits();
43 }
44 if (type.lanes() > 1) {
45 out << "x" << type.lanes();
46 }
47 return out;
48}

175void IRPrinter::test() {
176 Type i32 = Int(32);
177 Type f32 = Float(32);
178 Expr x = Variable::make(Int(32), "x");
179 Expr y = Variable::make(Int(32), "y");
180 ostringstream expr_source;
181 expr_source << (x + 3) * (y / 2 + 17);
182 internal_assert(expr_source.str() == "((x + 3)*((y/2) + 17))");
183
184 Stmt store = Store::make("buf", (x * 17) / (x - 3), y - 1, Parameter(), const_true(), ModulusRemainder());
185 Stmt for_loop = For::make("x", -2, y + 2, ForType::Parallel, DeviceAPI::Host, store);
186 vector<Expr> args(1);
187 args[0] = x % 3;
188 Expr call = Call::make(i32, "buf", args, Call::Extern);
189 Stmt store2 = Store::make("out", call + 1, x, Parameter(), const_true(), ModulusRemainder(3, 5));
190 Stmt for_loop2 = For::make("x", 0, y, ForType::Vectorized, DeviceAPI::Host, store2);
191
192 Stmt producer = ProducerConsumer::make_produce("buf", for_loop);
193 Stmt consumer = ProducerConsumer::make_consume("buf", for_loop2);
194 Stmt pipeline = Block::make(producer, consumer);
195
196 Stmt assertion = AssertStmt::make(y >= 3, Call::make(Int(32), "halide_error_param_too_small_i64",
197 {string("y"), y, 3}, Call::Extern));
198 Stmt block = Block::make(assertion, pipeline);
199 Stmt let_stmt = LetStmt::make("y", 17, block);
200 Stmt allocate = Allocate::make("buf", f32, MemoryType::Stack, {1023}, const_true(), let_stmt);
201
202 ostringstream source;
203 source << allocate;
204 std::string correct_source =
205 "allocate buf[float32 * 1023] in Stack\n"
206 "let y = 17\n"
207 "assert(y >= 3, halide_error_param_too_small_i64(\"y\", y, 3))\n"
208 "produce buf {\n"
209 " parallel (x, -2, y + 2) {\n"
210 " buf[y - 1] = (x*17)/(x - 3)\n"
211 " }\n"
212 "}\n"
213 "consume buf {\n"
214 " vectorized (x, 0, y) {\n"
215 " out[x] = buf(x % 3) + 1\n"
216 " }\n"
217 "}\n";
218
219 if (source.str() != correct_source) {
220 internal_error << "Correct output:\n"
221 << correct_source
222 << "Actual output:\n"
223 << source.str();
224 }
225 std::cout << "IRPrinter test passed\n";
226}

439void IRPrinter::visit(const StringImm *op) {
440 stream << "\"";
441 for (size_t i = 0; i < op->value.size(); i++) {
442 unsigned char c = op->value[i];
443 if (c >= ' ' && c <= '~' && c != '\\' && c != '"') {
444 stream << c;
445 } else {
446 stream << "\\";
447 switch (c) {
448 case '"':
449 stream << "\"";
450 break;
451 case '\\':
452 stream << "\\";
453 break;
454 case '\t':
455 stream << "t";
456 break;
457 case '\r':
458 stream << "r";
459 break;
460 case '\n':
461 stream << "n";
462 break;
463 default:
464 string hex_digits = "0123456789ABCDEF";
465 stream << "x" << hex_digits[c >> 4] << hex_digits[c & 0xf];
466 }
467 }
468 }
469 stream << "\"";
470}

771void IRPrinter::visit(const Store *op) {
772 stream << get_indent();
773 const bool has_pred = !is_const_one(op->predicate);
774 const bool show_alignment = op->value.type().is_vector() && (op->alignment.modulus > 1);
775 if (has_pred) {
776 stream << "predicate (";
777 print_no_parens(op->predicate);
778 stream << ")\n";
779 indent++;
780 stream << get_indent();
781 }
782 stream << op->name << "[";
783 print_no_parens(op->index);
784 if (show_alignment) {
785 stream << " aligned("
786 << op->alignment.modulus
787 << ", "
788 << op->alignment.remainder << ")";
789 }
790 stream << "] = ";
791 if (const Let *let = op->value.as<Let>()) {
792 // Use some nicer line breaks for containing Lets
793 stream << "\n";
794 indent += 2;
795 print_lets(let);
796 indent -= 2;
797 } else {
798 // Just print the value in-line
799 print_no_parens(op->value);
800 }
801 stream << "\n";
802 if (has_pred) {
803 indent--;
804 }
805}

920void IRPrinter::visit(const Fork *op) {
921 vector<Stmt> stmts;
922 stmts.push_back(op->first);
923 Stmt rest = op->rest;
924 while (const Fork *f = rest.as<Fork>()) {
925 stmts.push_back(f->first);
926 rest = f->rest;
927 }
928 stmts.push_back(rest);
929
930 stream << get_indent() << "fork ";
931 for (const Stmt &s : stmts) {
932 stream << "{\n";
933 indent++;
934 print(s);
935 indent--;
936 stream << get_indent() << "} ";
937 }
938 stream << "\n";
939}

941void IRPrinter::visit(const IfThenElse *op) {
942 stream << get_indent();
943 while (true) {
944 stream << "if (";
945 print_no_parens(op->condition);
946 stream << ") {\n";
947 indent++;
948 print(op->then_case);
949 indent--;
950
951 if (!op->else_case.defined()) {
952 break;
953 }
954
955 if (const IfThenElse *nested_if = op->else_case.as<IfThenElse>()) {
956 stream << get_indent() << "} else ";
957 op = nested_if;
958 } else {
959 stream << get_indent() << "} else {\n";
960 indent++;
961 print(op->else_case);
962 indent--;
963 break;
964 }
965 }

967 stream << get_indent() << "}\n";

977 if (op->is_concat()) {

980 stream << ")";
981 } else if (op->is_interleave()) {

985 } else if (op->is_extract_element()) {
986 stream << "extract_element(";

989 } else if (op->is_slice()) {
990 stream << "slice_vectors(";
991 print_list(op->vectors);
992 stream << ", " << op->slice_begin()
993 << ", " << op->slice_stride()
994 << ", " << op->indices.size()
995 << ")";
996 } else if (op->is_broadcast()) {
997 stream << "broadcast(";
998 print_list(op->vectors);
999 stream << ", " << op->broadcast_factor() << ")";
1000 } else {
1001 stream << "shuffle(";
1002 print_list(op->vectors);
1003 stream << ", ";
1004 for (size_t i = 0; i < op->indices.size(); i++) {
1005 print_no_parens(op->indices[i]);
1006 if (i < op->indices.size() - 1) {
1007 stream << ", ";
1008 }
1009 }
1010 stream << ")";
