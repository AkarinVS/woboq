7extern "C" int mul_by_two(
8 halide_buffer_t *input,
9 halide_buffer_t *output) {
10 if (input->is_bounds_query()) {
11 // Bounds query: infer the input dimensions from the output dimensions. In
12 // this example, the dimensions are exactly the same
13 for (int i = 0; i < 2; ++i) {
14 input->dim[i] = output->dim[i];
15 }
16 return 0;
17 }
18
19 // Actual computation: return 2 times x as an example. The first dimension is
20 // the innermost, so iterate over it last to avoid inefficient memory access
21 // patterns.
22 for (int j = 0; j < input->dim[1].extent; ++j) {
23 for (int i = 0; i < input->dim[0].extent; ++i) {
24 float *out = (float *)output->host + i * output->dim[0].stride +
25 j * output->dim[1].stride;
26 float *in = (float *)input->host + i * input->dim[0].stride +
27 j * input->dim[1].stride;
28 (*out) = 2 * (*in);
29 }
30 }
31 return 0;
32}

82extern "C" int matmul(
83 halide_buffer_t *input1,
84 halide_buffer_t *input2,
85 halide_buffer_t *output) {
86 if (input1->is_bounds_query() || input2->is_bounds_query()) {
87 // Bounds query: infer the input dimensions from the output dimensions.
88 // We leave the k dimension alone since we can't infer it from the output dimensions.
89 input1->dim[0].min = output->dim[0].min;
90 input1->dim[0].extent = output->dim[0].extent;
91 input2->dim[1].min = output->dim[1].min;
92 input2->dim[1].extent = output->dim[1].extent;
93 return 0;
94 }
95
96 // Actual computation: return input1 * input2.
97 const int max_i = output->dim[0].min + output->dim[0].extent;
98 const int max_j = output->dim[1].min + output->dim[1].extent;
99 for (int i = output->dim[0].min; i < max_i; ++i) {
100 for (int j = output->dim[1].min; j < max_j; ++j) {
101 int pos[2] = {i, j};
102 float *out = (float *)output->address_of(pos);
103 *out = 0.0f;
104 for (int k = 0; k < input1->dim[1].extent; ++k) {
105 int pos1[2] = {i, k};
106 float *in1 = (float *)input1->address_of(pos1);
107 int pos2[2] = {k, j};
108 float *in2 = (float *)input2->address_of(pos2);
109 (*out) += (*in1) * (*in2);
110 }
111 }
112 }
113 return 0;
114}
