8std::string get_env_variable(char const *env_var_name) {
9 if (!env_var_name) {
10 return "";
11 }
12
13#ifdef _MSC_VER
14 // call getenv_s without a buffer to determine the correct string length:
15 size_t length = 0;
16 if ((getenv_s(&length, nullptr, 0, env_var_name) != 0) || (length == 0)) {
17 return "";
18 }
19 // call it again to retrieve the value of the environment variable;
20 // note that 'length' already accounts for the null-terminator
21 std::string lvl(length - 1, '@');
22 size_t read = 0;
23 if ((getenv_s(&read, &lvl[0], length, env_var_name) != 0) || (read != length)) {
24 return "";
25 }
26 return lvl;
27#else
28 char *lvl = getenv(env_var_name);
29 if (lvl) {
30 return std::string(lvl);
31 }
32#endif
33
34 return "";
35}

39int aslog::aslog_level() {
40 static int cached_aslog_level = ([]() -> int {
41 // If HL_DEBUG_AUTOSCHEDULE is defined, use that value.
42 std::string lvl = get_env_variable("HL_DEBUG_AUTOSCHEDULE");
43 if (!lvl.empty()) {
44 return atoi(lvl.c_str());
45 }
46 // Otherwise, use HL_DEBUG_CODEGEN.
47 lvl = get_env_variable("HL_DEBUG_CODEGEN");
48 return !lvl.empty() ? atoi(lvl.c_str()) : 0;
49 })();
50 return cached_aslog_level;
51}
