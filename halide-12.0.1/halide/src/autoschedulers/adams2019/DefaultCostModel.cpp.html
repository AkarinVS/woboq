34bool ends_with(const std::string &str, const std::string &suffix) {
35 if (str.size() < suffix.size()) {
36 return false;
37 }
38 size_t off = str.size() - suffix.size();
39 for (size_t i = 0; i < suffix.size(); i++) {
40 if (str[off + i] != suffix[i]) {
41 return false;
42 }
43 }
44 return true;
45}

49void DefaultCostModel::set_pipeline_features(const Internal::Autoscheduler::FunctionDAG &dag,
50 const MachineParams &params) {
51
52 const int pipeline_feat_size = head1_w * head1_h;
53 // We ignore the first seven pipeline features in the cost
54 // model. It's just a mask of which types are in use.
55 static_assert(sizeof(PipelineFeatures) - 7 * sizeof(int) ==
56 sizeof(int) * pipeline_feat_size,
57 "Incorrect size for pipeline features");
58 int num_stages = 0;
59 for (const auto &n : dag.nodes) {
60 if (!n.is_input) num_stages += (int)n.stages.size();
61 }
62 Runtime::Buffer<float> pipeline_features(head1_w, head1_h, num_stages);
63 int stage = 0;
64 for (const auto &n : dag.nodes) {
65 if (n.is_input) continue;
66 for (auto it = n.stages.rbegin(); it != n.stages.rend(); it++) {
67 const auto &s = *it;
68 const int *pipeline_feats = (const int *)(&(s.features)) + 7;
69 // skip the first 7 features
70 for (int i = 0; i < pipeline_feat_size; i++) {
71 int x = i / 7;
72 int y = i % 7;
73 pipeline_features(x, y, stage) = pipeline_feats[i];
74 }
75 stage += 1;
76 }
77 }
78 internal_assert(stage == num_stages);
79 pipeline_feat_queue = pipeline_features;
80 internal_assert(params.parallelism > 0);
81 num_cores = params.parallelism;
82}

85 pipeline_feat_queue = pipeline_feats;
86 internal_assert(n > 0);
87 num_cores = n;

90void DefaultCostModel::enqueue(const Internal::Autoscheduler::FunctionDAG &dag,
91 const Halide::Internal::Autoscheduler::StageMapOfScheduleFeatures &schedule_feats,
92 double *cost_ptr) {
93 num_stages = (int)schedule_feats.size();
94
95 Runtime::Buffer<float> schedule_features;
96
97 // Tell the cost model about this state. It won't actually
98 // evaluate it until we call evaluate_costs (or if it runs out
99 // of internal buffer space), so that the evaluations can be
100 // batched.
101 enqueue(num_stages, &schedule_features, cost_ptr);
102
103 // index of current stage whose features we are reading
104 int stage = 0;
105 // load schedule features into input buffer
106 for (const auto &n : dag.nodes) {
107
108 // Inputs are computed outside of the pipeline and don't count.
109 if (n.is_input) continue;
110
111 // The remaining stages are not yet
112 // scheduled. Optimistically assume their internal costs
113 // will not depend on the decisions made already, so
114 // there's no point adding it on to the total because it's
115 // the same across all states. An underestimate of the
116 // cost for loading from these unscheduled stages is
117 // already baked into the scheduled stages that consume
118 // them.
119 if (stage >= num_stages) break;
120
121 // Load up the schedule features for all stages of this Func.
122 for (auto it = n.stages.rbegin(); it != n.stages.rend(); it++) {
123 internal_assert(schedule_feats.contains(&*it)) << n.func.name() << "\n";
124 const auto &feat = schedule_feats.get(&*it);
125 for (size_t i = 0; i < ScheduleFeatures::num_features(); i++) {
126 schedule_features(i, stage) = feat[i];
127 }
128 stage += 1;
129 }
130 }
131 // Check we considered everything we were supposed to.
132 internal_assert(stage == num_stages);
133}

135void DefaultCostModel::enqueue(int ns, Runtime::Buffer<float> *schedule_feats, double *cost_ptr) {
136 num_stages = ns;
137
138 // We know the most stages that will ever be enqueued from the schedule features
139 internal_assert(pipeline_feat_queue.data() && "Call set_schedule_features before calling enqueue\n");
140 const int max_num_stages = pipeline_feat_queue.dim(2).extent();
141 internal_assert(num_stages <= max_num_stages)
142 << "schedule features has more stages (" << num_stages
143 << ") than pipeline features (" << max_num_stages << ")\n";
144
145 const int batch_size = 1024;
146 if (!schedule_feat_queue.data() ||
147 schedule_feat_queue.dim(2).extent() < max_num_stages) {
148 internal_assert(cursor == 0);
149 schedule_feat_queue = Runtime::Buffer<float>(batch_size, head2_w, max_num_stages);
150 if (!costs.data()) {
151 internal_assert(!cost_ptrs.data());
152 costs = Runtime::Buffer<float>(batch_size);
153 cost_ptrs = Runtime::Buffer<double *>(batch_size);
154 }
155 }
156
157 if (cursor == batch_size) {
158 evaluate_costs();
159 }
160
161 *schedule_feats = schedule_feat_queue.sliced(0, cursor);
162 cost_ptrs(cursor) = cost_ptr;
163
164 cursor++;
165} // namespace Halide
166
167// Backprop state. To run ADAM we need a running average of the
168// gradients and gradients squared. We add an outer dimension of
169// size 3 to the new weight outputs to track this state. So buf(_,
170// 0) is the new weight, buf(_, 1) is the ADAM running average of
171// the first moment, and buf(_, 2) is the ADAM running average of
172// the second moment.
173float DefaultCostModel::backprop(const Runtime::Buffer<const float> &true_runtimes, float learning_rate) {
174 internal_assert(cursor != 0);
175 internal_assert(pipeline_feat_queue.data());
176 internal_assert(schedule_feat_queue.data());
177
178 auto loss = Runtime::Buffer<float>::make_scalar();
179
180 if (!head1_filter_update.data()) {
181 auto weight_update_buffer = [](const Runtime::Buffer<float> &w) {
182 std::vector<int> size;
183 for (int i = 0; i < w.dimensions(); i++) {
184 size.push_back(w.dim(i).extent());
185 }
186 size.push_back(4);
187 auto buf = Runtime::Buffer<float>(size);
188 buf.fill(0.0f);
189 return buf;
190 };
191
192 head1_filter_update = weight_update_buffer(weights.head1_filter);
193 head1_bias_update = weight_update_buffer(weights.head1_bias);
194 head2_filter_update = weight_update_buffer(weights.head2_filter);
195 head2_bias_update = weight_update_buffer(weights.head2_bias);
196 conv1_filter_update = weight_update_buffer(weights.conv1_filter);
197 conv1_bias_update = weight_update_buffer(weights.conv1_bias);
198 timestep = 0;
199 }
200
201 Runtime::Buffer<float> dst = costs.cropped(0, 0, cursor);
202
203 int fastest_idx = 0;
204 for (int i = 0; i < cursor; i++) {
205 if (true_runtimes(i) < true_runtimes(fastest_idx)) {
206 fastest_idx = i;
207 }
208 }
209
210 int result = train_cost_model(num_stages,
211 cursor,
212 num_cores,
213 pipeline_feat_queue,
214 schedule_feat_queue,
215 weights.head1_filter, weights.head1_bias,
216 weights.head2_filter, weights.head2_bias,
217 weights.conv1_filter, weights.conv1_bias,
218 learning_rate, timestep++,
219 fastest_idx,
220 true_runtimes.alias(),
221 head1_filter_update, head1_bias_update,
222 head2_filter_update, head2_bias_update,
223 conv1_filter_update, conv1_bias_update,
224 dst,
225 loss);
226 (void)result;
227 internal_assert(result == 0);
228
229 bool any_nans = false;
230 for (int i = 0; i < cursor; i++) {
231 internal_assert(cost_ptrs(i));
232 *(cost_ptrs(i)) = dst(i);
233 if (std::isnan(dst(i))) {
234 any_nans = true;
235 aslog(0) << "Prediction " << i << " is NaN. True runtime is " << true_runtimes(i) << "\n";
236 aslog(0) << "Checking pipeline features for NaNs...\n";
237 pipeline_feat_queue.for_each_value([&](float f) { if (std::isnan(f)) abort(); });
238 aslog(0) << "None found\n";
239 aslog(0) << "Checking schedule features for NaNs...\n";
240 schedule_feat_queue.for_each_value([&](float f) { if (std::isnan(f)) abort(); });
241 aslog(0) << "None found\n";
242 aslog(0) << "Checking network weights for NaNs...\n";
243 weights.for_each_buffer([&](const Runtime::Buffer<float> &buf) {
244 buf.for_each_value([&](float f) { if (std::isnan(f)) abort(); });
245 });
246 aslog(0) << "None found\n";
247 }
248 internal_assert(true_runtimes(i) > 0);
249 }
250 if (any_nans) {
251 abort();
252 }
253
254 // Update weights locally
255 auto update_weight = [](const Runtime::Buffer<float> &src, Runtime::Buffer<float> &dst) {
256 dst.copy_from(src.sliced(src.dimensions() - 1, 0));
257 };
258 update_weight(head1_filter_update, weights.head1_filter);
259 update_weight(head1_bias_update, weights.head1_bias);
260 update_weight(head2_filter_update, weights.head2_filter);
261 update_weight(head2_bias_update, weights.head2_bias);
262 update_weight(conv1_filter_update, weights.conv1_filter);
263 update_weight(conv1_bias_update, weights.conv1_bias);
264
265 internal_assert(cursor != 0);
266
267 return loss();
268}

270void DefaultCostModel::evaluate_costs() {
271 if (cursor == 0 || !schedule_feat_queue.data()) {
272 return;
273 }
274
275 internal_assert(pipeline_feat_queue.data());
276 internal_assert(schedule_feat_queue.data());
277
278 Runtime::Buffer<float> dst = costs.cropped(0, 0, cursor);
279
280 auto loss = Runtime::Buffer<float>::make_scalar();
281
282 int result = cost_model(num_stages,
283 cursor,
284 num_cores,
285 pipeline_feat_queue,
286 schedule_feat_queue,
287 weights.head1_filter, weights.head1_bias,
288 weights.head2_filter, weights.head2_bias,
289 weights.conv1_filter, weights.conv1_bias,
290 0.0f, 0, 0, nullptr,
291 dst, loss);
292 (void)result;
293 internal_assert(result == 0);
294
295 for (int i = 0; i < cursor; i++) {
296 internal_assert(cost_ptrs(i));
297 *(cost_ptrs(i)) = dst(i);
298 }
299
300 cursor = 0;
301}

303void DefaultCostModel::load_weights() {
304 bool need_randomize = randomize_weights;
305
306 if (weights_in_path.empty()) {
307 aslog(1) << "Loading weights from built-in data...\n";
308 // This copy shouldn't be necessary, but std::istream in C++ doesn't seem
309 // to have a convenient wrap-around-constant-data variant... and since
310 // this isn't much data, just copy it.
311 const std::string baseline_weights_data((const char *)&baseline_weights[0], baseline_weights_length);
312 std::istringstream i(baseline_weights_data);
313 if (!weights.load(i)) {
314 std::cerr << "The built-in baseline weights should never fail to load\n";
315 internal_assert(0);
316 }
317 } else if (ends_with(weights_in_path, ".weights")) {
318 aslog(1) << "Loading weights from " << weights_in_path << " ...\n";
319 if (!weights.load_from_file(weights_in_path)) {
320 // Emit to cout (rather than cerr) because the latter is hidden during the autotune loop,
321 // and we want this to be seen.
322 std::cout << "WARNING, error in reading weights from " << weights_in_path << ", randomizing...\n";
323 need_randomize = true;
324 }
325 } else {
326 aslog(1) << "Loading weights from directory " << weights_in_path << " ...\n";
327 std::cerr << "Loading weights from a directory is deprecated; please convert to a .weights file\n";
328 if (!weights.load_from_dir(weights_in_path)) {
329 std::cout << "WARNING, error in reading weights from " << weights_in_path << ", randomizing...\n";
330 need_randomize = true;
331 }
332 }
333
334 if (!need_randomize && weights.pipeline_features_version != PipelineFeatures::version()) {
335 // Emit to cout (rather than cerr) because the latter is hidden during the autotune loop,
336 // and we want this to be seen.
337 std::cout << "WARNING: loaded weights have pipeline_version = "
338 << weights.pipeline_features_version
339 << " but current pipeline_version is " << PipelineFeatures::version()
340 << "; the weights may be invalid. Using anyway.\n";
341 }
342
343 if (!need_randomize && weights.schedule_features_version != ScheduleFeatures::version()) {
344 // Emit to cout (rather than cerr) because the latter is hidden during the autotune loop,
345 // and we want this to be seen.
346 std::cout << "WARNING: loaded weights have schedule_features_version = "
347 << weights.schedule_features_version
348 << " but current schedule_features_version is " << ScheduleFeatures::version()
349 << "; the weights may be invalid. Using anyway.\n";
350 }
351
352 if (need_randomize) {
353 auto seed = time(nullptr);
354 std::cout << "Randomizing weights using seed = " << seed << "\n";
355 weights.randomize((uint32_t)seed);
356 }
357
358 // Update so that any version of this we save will have the current version
359 weights.pipeline_features_version = PipelineFeatures::version();
360 weights.schedule_features_version = ScheduleFeatures::version();
361}

363void DefaultCostModel::save_weights() {
364 internal_assert(!weights_out_path.empty())
365 << "Unable to save weights: no output path specified\n";
366
367 if (ends_with(weights_out_path, ".weights")) {
368 internal_assert(weights.save_to_file(weights_out_path))
369 << "Unable to save weights to file: " << weights_out_path << "\n";
370 } else {
371 std::cerr << "Saving weights to a directory is deprecated; please convert to a .weights file\n";
372 internal_assert(weights.save_to_dir(weights_out_path))
373 << "Unable to save weights to file: " << weights_out_path << "\n";
374 }
