17bool Cache::add_memoized_blocks(const State *state,
18 std::function<void(IntrusivePtr<State> &&)> &accept_child,
19 const FunctionDAG::Node *node, int &num_children,
20 const FunctionDAG &dag,
21 const MachineParams &params,
22 CostModel *cost_model,
23 int64_t memory_limit) const {
24 if (!options.cache_blocks || !memoized_compute_root_blocks.contains(node)) {
25 // either memoization is turned off, or we haven't cached this node yet.
26 return false;
27 }
28
29 // get correct vector dimension.
30 int vector_dims = -1;
31 for (const auto &child : state->root->children) {
32 if (child->node == node && child->stage->index == 0) {
33 vector_dims = child->vector_dim;
34 break;
35 }
36 }
37
38 const auto &vector_dim_map = memoized_compute_root_blocks.get(node);
39
40 if (vector_dim_map.count(vector_dims) == 0) {
41 // Never cached this vector dimension before.
42 return false;
43 }
44
45 auto blocks = vector_dim_map.at(vector_dims);
46
47 size_t num_stages = node->stages.size();
48
49 for (size_t i = 0; i < blocks.size(); i += num_stages) {
50 // Construct child from memoization.
51 IntrusivePtr<State> child = state->make_child();
52 LoopNest *new_root = new LoopNest;
53 new_root->copy_from(*(state->root));
54 child->root = new_root;
55 child->num_decisions_made++;
56
57 int block_index = 0;
58 for (const auto &new_child : new_root->children) {
59 if (new_child->node == node) {
60 break;
61 }
62 block_index++;
63 }
64
65 // Copy all stages into new_root.
66 for (size_t j = 0; j < num_stages; j++) {
67 LoopNest *new_block = new LoopNest;
68 new_block->copy_from_including_features(*blocks[i + j]);
69 new_root->children[block_index++] = new_block;
70 }
71
72 if (child->calculate_cost(dag, params, cost_model, this->options, memory_limit)) {
73 num_children++;
74 accept_child(std::move(child));
75 cache_hits++;
76 }
77 }
78
79 // succesfully added cached items!
80 return true;
81}

83void Cache::memoize_blocks(const FunctionDAG::Node *node, LoopNest *new_root) {
84 if (!options.cache_blocks) {
85 return;
86 }
87
88 int vector_dim = -1;
89 bool loop_nest_found = false;
90
91 for (auto &child : new_root->children) {
92 if (child->node == node && child->stage->index == 0) {
93 vector_dim = child->vector_dim;
94 loop_nest_found = true;
95 break;
96 }
97 }
98
99 internal_assert(loop_nest_found) << "memoize_blocks did not find loop nest!\n";
100
101 auto &blocks = memoized_compute_root_blocks.get_or_create(node)[vector_dim];
102
103 for (auto &child : new_root->children) {
104 if (child->node == node) {
105 LoopNest *new_block = new LoopNest;
106 // Need const reference for copy.
107 const LoopNest *child_ptr = child.get();
108 new_block->copy_from_including_features(*child_ptr);
109 blocks.emplace_back(new_block);
110 cache_misses++;
111 }
112 }
113}
