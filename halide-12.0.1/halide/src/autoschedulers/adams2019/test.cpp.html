17bool test_caching(Pipeline &p1, Pipeline &p2, const Target &target, const MachineParams &params) {
18 static const std::string seed_value = Internal::get_env_variable("HL_SEED");
19 if (seed_value.empty()) {
20 // If HL_SEED is not set, then set seed for both autoscheduling executions.
21 int seed = (int)time(nullptr);
22 set_env_variable("HL_SEED", std::to_string(seed), /* overwrite */ 0);
23 }
24
25 // Turn off caching.
26 set_env_variable("HL_DISABLE_MEMOIZED_FEATURES", "1", /* overwrite */ 1);
27 set_env_variable("HL_DISABLE_MEMOIZED_BLOCKS", "1", /* overwrite */ 1);
28 auto results_without_caching = p1.auto_schedule(target, params);
29
30 // Turn on caching.
31 set_env_variable("HL_DISABLE_MEMOIZED_FEATURES", "0", /* overwrite */ 1);
32 set_env_variable("HL_DISABLE_MEMOIZED_BLOCKS", "0", /* overwrite */ 1);
33 auto results_with_caching = p2.auto_schedule(target, params);
34
35 // Reset environment variables to what they were before (memoization variables are reset in main).
36 if (seed_value.empty()) {
37 // Re-empty seed.
38 set_env_variable("HL_SEED", "", /* overwrite */ 1);
39 }
40
41 // Compare calculated features.
42 if (results_without_caching.featurization.size() != results_with_caching.featurization.size()) {
43 return false;
44 } else {
45 size_t feat_size = results_without_caching.featurization.size();
46 for (size_t i = 0; i < feat_size; i++) {
47 if (results_without_caching.featurization[i] != results_with_caching.featurization[i]) {
48 return false;
49 }
50 }
51 }
52
53 return true;
54}
