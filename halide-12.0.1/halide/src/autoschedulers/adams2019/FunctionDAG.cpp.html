35 PipelineFeatures::ScalarType classify_type(Type t) {
36 if (t.is_float() && t.bits() > 32) {
37 return PipelineFeatures::ScalarType::Double;
38 } else if (t.is_float()) {
39 return PipelineFeatures::ScalarType::Float;
40 } else if (t.bits() == 1) {
41 return PipelineFeatures::ScalarType::Bool;
42 } else if (t.bits() <= 8) {
43 return PipelineFeatures::ScalarType::UInt8;

144 void visit(const Call *op) override {
145 IRVisitor::visit(op);
146 if (op->call_type == Call::Halide) {
147 if (op->name == func.name()) {

163 // Take the derivative of an integer index expression. If it's
164 // a rational constant, return it, otherwise return a sentinel
165 // value.
166
167 // The derivative of each let w.r.t each var. The keys are
168 // just the var names separated by a space.
169 Scope<OptionalRational> dlets;
170
171 OptionalRational differentiate(const Expr &e, const string &v) {
172 if (!expr_uses_var(e, v, lets)) {
173 return {true, 0, 1};
174 } else if (const Variable *var = e.as<Variable>()) {
175 if (var->name == v) {
176 return {true, 1, 1};
177 }
178 for (const auto &l : stage.loop) {
179 if (var->name == l.var) {
180 // Some other loop variable
181 return {true, 0, 1};
182 }
183 }
184 if (var->param.defined()) {
185 // An argument
186 return {true, 0, 1};
187 } else if (lets.contains(var->name)) {
188 string key = v + " " + var->name;
189 if (dlets.contains(key)) {
190 return dlets.get(key);
191 }
192 auto a = differentiate(lets.get(var->name), v);
193 dlets.push(key, a);
194 return a;
195 }
196 // Some mystery variable. Who knows what it depends on.
197 internal_error << "Encountered unbound variable in call args: " << var->name << "\n";
198 return {false, 0, 0};
199 } else if (const Add *op = e.as<Add>()) {
200 auto a = differentiate(op->a, v);
201 a += differentiate(op->b, v);
202 return a;
203 } else if (const Sub *op = e.as<Sub>()) {
204 auto a = differentiate(op->a, v);
205 auto b = differentiate(op->b, v);
206 b.numerator = -b.numerator;
207 a += b;
208 return a;
209 } else if (const Mul *op = e.as<Mul>()) {
210 auto a = differentiate(op->a, v);
211 if (const int64_t *ib = as_const_int(op->b)) {
212 a.numerator *= *ib;
213 return a;
214 } else {
215 return {false, 0, 0};
216 }
217 } else if (const Div *op = e.as<Div>()) {
218 auto a = differentiate(op->a, v);
219 if (const int64_t *ib = as_const_int(op->b)) {
220 if (a.numerator != 0) {
221 a.denominator *= *ib;
222 }
223 return a;
224 } else {
225 return {false, 0, 0};
226 }
227 } else if (const Call *op = e.as<Call>()) {
228 if (op->is_intrinsic(Call::likely)) {
229 // TODO: Should a likely on one side of a min/max dominate?
230 return differentiate(op->args[0], v);
231 }
232 }
233
234 return {false, 0, 0};
235 }

237 void visit_memory_access(const std::string &name, Type t, const vector<Expr> &args, PipelineFeatures::AccessType type) {
238 // Compute matrix of partial derivatives of args w.r.t. loop params
239 vector<vector<OptionalRational>> matrix;
240 vector<size_t> ones_per_row(args.size(), 0),
241 zeros_per_row(args.size(), 0),
242 ones_per_col(stage.loop.size(), 0),
243 zeros_per_col(stage.loop.size(), 0);
244 matrix.resize(args.size());
245 bool is_pointwise = args.size() == stage.loop.size();
246 for (size_t i = 0; i < args.size(); i++) {
247 matrix[i].resize(stage.loop.size());
248 for (size_t j = 0; j < stage.loop.size(); j++) {
249 auto deriv = differentiate(args[i], stage.loop[j].var);
250 zeros_per_row[i] += deriv == 0;
251 ones_per_row[i] += deriv == 1;
252 zeros_per_col[j] += deriv == 0;
253 ones_per_col[j] += deriv == 1;
254 is_pointwise &= (i == j ? deriv == 1 : deriv == 0);
255 matrix[i][j] = deriv;
256 }
257 }
258 bool is_transpose = (args.size() == stage.loop.size());
259 bool is_broadcast = true, is_slice = true;
260 for (size_t i = 0; i < args.size(); i++) {
261 bool single_one = (ones_per_row[i] == 1) && (zeros_per_row[i] == stage.loop.size() - 1);
262 bool all_zero = (zeros_per_row[i] == stage.loop.size());
263 is_transpose &= single_one;
264 is_broadcast &= single_one;
265 is_slice &= single_one || all_zero;
266 }
267 for (size_t j = 0; j < stage.loop.size(); j++) {
268 bool single_one = (ones_per_col[j] == 1) && (zeros_per_col[j] == args.size() - 1);
269 bool all_zero = (zeros_per_col[j] == args.size());
270 is_transpose &= single_one || all_zero;
271 is_broadcast &= single_one;
272 is_slice &= single_one;
273 }
274
275 auto type_class = classify_type(t);
276
277 stage.features.pointwise_accesses[(int)type][(int)type_class] += is_pointwise;
278 stage.features.transpose_accesses[(int)type][(int)type_class] += is_transpose;
279 stage.features.broadcast_accesses[(int)type][(int)type_class] += is_broadcast;
280 stage.features.slice_accesses[(int)type][(int)type_class] += is_slice;
281
282 for (auto *e : stage.incoming_edges) {
283 if (e->producer->func.name() == name) {
284 // The same name can be encountered multiple times
285 // (e.g. a+a, where a is a trivial function),
286 // so we can't use std::move(matrix) here without making a copy
287 vector<vector<OptionalRational>> copy = matrix;
288 e->add_load_jacobian(std::move(copy));
289 }
290 }
291 }

360void BoundContents::Layout::allocate_some_more() const {
361 size_t size_of_one = sizeof(BoundContents) + total_size * sizeof(Span);
362 const size_t number_per_block = std::max((size_t)8, 4096 / size_of_one); // Make a page of them, or 8, whichever is larger.
363 const size_t bytes_to_allocate = std::max(size_of_one * number_per_block, (size_t)4096);
364 unsigned char *mem = (unsigned char *)malloc(bytes_to_allocate);
365
366 blocks.push_back(mem);
367 static_assert((sizeof(BoundContents) & 7) == 0, "BoundContents header is not aligned");
368 for (size_t i = 0; i < number_per_block; i++) {
369 BoundContents *b = (BoundContents *)(mem + i * size_of_one);
370 new (b) BoundContents;
371 b->layout = this;
372 pool.push_back(b);
373 }
374 internal_assert(((unsigned char *)(pool[0]) + size_of_one) == (unsigned char *)(pool[1]));
375}

377BoundContents *BoundContents::Layout::make() const {
378 if (pool.empty()) {
379 allocate_some_more();
380 }
381 BoundContents *b = pool.back();
382 pool.pop_back();

394void FunctionDAG::Node::loop_nest_for_region(int stage_idx, const Span *computed, Span *loop) const {
395 const auto &s = stages[stage_idx];
396 map<string, Expr> computed_map;
397 if (!s.loop_nest_all_common_cases) {
398 for (int i = 0; i < func.dimensions(); i++) {
399 computed_map[region_required[i].min.name()] = (int)computed[i].min();
400 computed_map[region_required[i].max.name()] = (int)computed[i].max();
401 }
402 }
403
404 for (size_t i = 0; i < s.loop.size(); i++) {
405 const auto &l = s.loop[i];
406 if (l.equals_region_computed) {
407 loop[i] = computed[l.region_computed_dim];
408 } else if (l.bounds_are_constant) {
409 loop[i] = Span(l.c_min, l.c_max, true);
410 } else {
411 Expr min = simplify(substitute(computed_map, l.min));
412 Expr max = simplify(substitute(computed_map, l.max));
413 const int64_t *imin = as_const_int(min);
414 const int64_t *imax = as_const_int(max);
415 internal_assert(imin && imax) << min << ", " << max << "\n";
416 loop[i] = Span(*imin, *imax, false);
417 }
418 }

421void FunctionDAG::Node::required_to_computed(const Span *required, Span *computed) const {
422 map<string, Expr> required_map;
423 if (!region_computed_all_common_cases) {
424 // Make a binding for the value of each symbolic variable
425 for (int i = 0; i < func.dimensions(); i++) {
426 required_map[region_required[i].min.name()] = (int)required[i].min();
427 required_map[region_required[i].max.name()] = (int)required[i].max();
428 }
429 }
430 for (int i = 0; i < func.dimensions(); i++) {
431 const auto &comp = region_computed[i];
432 if (comp.equals_required) {
433 computed[i] = required[i];
434 } else if (comp.equals_union_of_required_with_constants) {
435 computed[i] = Span(std::min(required[i].min(), comp.c_min),
436 std::max(required[i].max(), comp.c_max),
437 false);
438 } else {
439 Expr min = simplify(substitute(required_map, comp.in.min));
440 Expr max = simplify(substitute(required_map, comp.in.max));
441 const int64_t *imin = as_const_int(min);
442 const int64_t *imax = as_const_int(max);
443 internal_assert(imin && imax) << min << ", " << max << "\n";
444 computed[i] = Span(*imin, *imax, false);
445 }
446 }

449FunctionDAG::Edge::BoundInfo::BoundInfo(const Expr &e, const Node::Stage &consumer)
450 : expr(e) {
451 // Do the analysis to detect if this is a simple case
452 // that can be evaluated more cheaply. Currently this
453 // acceleration recognises affine expressions. In the
454 // future we may consider quasi-affine, or even
455 // piecewise-quasi-affine. If the bounds are
456 // non-affine, we use the symbolic expression.
457 const Add *add = expr.as<Add>();
458 const Mul *mul = add ? add->a.as<Mul>() : expr.as<Mul>();
459 const IntImm *coeff_imm = mul ? mul->b.as<IntImm>() : nullptr;
460 const IntImm *constant_imm = add ? add->b.as<IntImm>() : nullptr;
461 // clang-format off
462 Expr v = (mul ? mul->a :
463 add ? add->a :
464 expr);
465 // clang-format on
466 const Variable *var = v.as<Variable>();
467
468 if (const IntImm *c = e.as<IntImm>()) {
469 affine = true;
470 coeff = 0;
471 constant = c->value;
472 } else if (var && (!mul || coeff_imm) && (!add || constant_imm)) {
473 affine = true;
474 coeff = mul ? coeff_imm->value : 1;
475 constant = add ? constant_imm->value : 0;
476 consumer_dim = -1;
477 for (int i = 0; i < (int)consumer.loop.size(); i++) {
478 const auto &in = consumer.loop[i];
479 if (var->name == consumer.node->func.name() + "." + in.var + ".min") {
480 consumer_dim = i;
481 uses_max = false;
482 break;
483 } else if (var->name == consumer.node->func.name() + "." + in.var + ".max") {
484 consumer_dim = i;
485 uses_max = true;
486 break;
487 }
488 }
489 internal_assert(consumer_dim >= 0) << "Could not find consumer loop variable: " << var->name << "\n";
490 aslog(2) << "Bound is affine: " << e << " == " << var->name << " * " << coeff << " + " << constant << "\n";
491 } else {
492 affine = false;
493 aslog(2) << "Bound is non-affine: " << e << "\n";
494 }
495}

506void FunctionDAG::Edge::expand_footprint(const Span *consumer_loop, Span *producer_required) const {
507 // Create a map from the symbolic loop variables to the actual loop size
508 const auto &symbolic_loop = consumer->loop;
509 map<string, Expr> s;
510 if (!all_bounds_affine) {
511 for (size_t i = 0; i < symbolic_loop.size(); i++) {
512 auto p = consumer_loop[i];
513 const string &var = symbolic_loop[i].var;
514 s[consumer->node->func.name() + "." + var + ".min"] = (int)p.min();
515 s[consumer->node->func.name() + "." + var + ".max"] = (int)p.max();
516 }
517 }
518 // Apply that map to the bounds relationship encoded
519 // in the edge to expand the bounds of the producer to
520 // satisfy the consumer
521 for (int i = 0; i < producer->func.dimensions(); i++) {
522 // Get bounds required of this dimension of the
523 // producer in terms of a symbolic region of the
524 // consumer.
525 bool bounds_are_constant = true;
526 auto eval_bound = [&](const BoundInfo &b) {
527 if (b.affine) {
528 // Common-case performance optimization
529 if (b.coeff == 0) {
530 return b.constant;
531 } else {
532 const auto &src_pair = consumer_loop[b.consumer_dim];
533 int64_t src = b.uses_max ? src_pair.max() : src_pair.min();
534 bounds_are_constant &= src_pair.constant_extent();
535 return src * b.coeff + b.constant;
536 }
537 } else {
538 Expr substituted = substitute(s, b.expr);
539 Expr e = simplify(substituted);
540 const int64_t *i = as_const_int(e);
541 internal_assert(i) << "Should be constant: " << b.expr << " -> " << substituted << " -> " << e << "\n";
542 bounds_are_constant = false;
543 return *i;
544 }
545 };
546 int64_t a = eval_bound(bounds[i].first);
547 int64_t b = eval_bound(bounds[i].second);
548 producer_required[i].union_with(Span(a, b, bounds_are_constant));
549 }
550}

552FunctionDAG::FunctionDAG(const vector<Function> &outputs, const MachineParams &params, const Target &target) {
553 map<string, Function> env;
554 for (const Function &o : outputs) {
555 populate_environment(o, env);
556 }
557
558 // A mutator to apply parameter estimates to the expressions
559 // we encounter while constructing the graph.
560 class ApplyParamEstimates : public IRMutator {
561 using IRMutator::visit;
562
563 Expr visit(const Variable *op) override {
564 Expr expr;
565 if (op->param.defined()) {
566 if (!op->param.is_buffer()) {
567 expr = op->param.estimate();
568 } else {
569 for (int i = 0; i < op->param.dimensions(); i++) {
570 if (op->name == op->param.name() + ".min." + std::to_string(i)) {
571 expr = op->param.min_constraint_estimate(i);
572 } else if (op->name == op->param.name() + ".extent." + std::to_string(i)) {
573 expr = op->param.extent_constraint_estimate(i);
574 }
575 }
576 }
577 internal_assert(expr.defined()) << "Missing estimate for " << op->name << "\n";
578 return expr;
579 } else {
580 return op;
581 }
582 }
583 } apply_param_estimates;
584
585 // Compute a realization order
586 vector<string> order = topological_order(outputs, env);
587
588 // Construct the mapping from Funcs to Nodes
589 nodes.resize(order.size());
590 map<Function, Node *, Function::Compare> node_map;
591 for (size_t i = 0; i < order.size(); i++) {
592 Function f = env[order[order.size() - i - 1]];
593 nodes[i].func = f;
594 nodes[i].id = (int)i;
595 nodes[i].max_id = (int)order.size();
596 nodes[i].dag = this;
597 node_map[f] = &nodes[i];
598 }
599
600 int stage_count = 0;
601
602 for (size_t i = order.size(); i > 0; i--) {
603 Node &node = nodes[order.size() - i];
604 Function consumer = node.func;
605 Scope<Interval> scope;
606
607 // Create a symbolic region for this Func.
608 for (int j = 0; j < consumer.dimensions(); j++) {
609 Halide::Var min_var(consumer.name() + "." + consumer.args()[j] + ".min");
610 Halide::Var max_var(consumer.name() + "." + consumer.args()[j] + ".max");
611 Interval interval(min_var, max_var);
612 scope.push(consumer.args()[j], interval);
613 node.region_required.emplace_back(SymbolicInterval{min_var, max_var});
614 }
615
616 auto pure_args = node.func.args();
617
618 for (int s = 0; s <= (int)consumer.updates().size(); s++) {
619 stage_count++;
620 if (s == 0) {
621 node.stages.emplace_back(Stage(consumer, consumer.definition(), 0));
622 } else {
623 node.stages.emplace_back(Stage(consumer, consumer.update(s - 1), s));
624 }
625 }
626
627 for (int s = 0; s <= (int)consumer.updates().size(); s++) {
628 auto &stage = node.stages[s];
629 stage.node = &node;
630 stage.name = consumer.name();
631 if (s > 0) {
632 stage.name += ".update(" + std::to_string(s - 1) + ")";
633 }
634
635 const Definition &def = (s == 0) ? consumer.definition() : consumer.update(s - 1);
636 const StageSchedule &sched = def.schedule();
637
638 Scope<Interval> stage_scope_with_concrete_rvar_bounds, stage_scope_with_symbolic_rvar_bounds;
639 stage_scope_with_concrete_rvar_bounds.set_containing_scope(&scope);
640 stage_scope_with_symbolic_rvar_bounds.set_containing_scope(&scope);
641 for (const auto &rv : sched.rvars()) {
642 Expr min = simplify(apply_param_estimates.mutate(rv.min));
643 Expr max = simplify(apply_param_estimates.mutate(rv.min + rv.extent - 1));
644 stage_scope_with_concrete_rvar_bounds.push(rv.var, Interval(min, max));
645 min = Variable::make(Int(32), consumer.name() + "." + rv.var + ".min");
646 max = Variable::make(Int(32), consumer.name() + "." + rv.var + ".max");
647 stage_scope_with_symbolic_rvar_bounds.push(rv.var, Interval(min, max));
648 }
649
650 // Figure out the region computed of the stage by taking bounds of the LHS Exprs
651 if (s == 0) {
652 node.region_computed.resize(consumer.dimensions());
653 }
654
655 FuncValueBounds func_value_bounds = compute_function_value_bounds(order, env);
656 for (int j = 0; j < consumer.dimensions(); j++) {
657 // The region computed always uses the full extent of the rvars
658 Interval in = bounds_of_expr_in_scope(def.args()[j], stage_scope_with_concrete_rvar_bounds, func_value_bounds);
659 internal_assert(in.is_bounded())
660 << "Region computed of " << consumer.name()
661 << " is unbounded: [" << in.min << " " << in.max << "]\n";
662 if (s == 0) {
663 node.region_computed[j].in = in;
664 } else {
665 node.region_computed[j].in.include(in);
666 }
667 }
668 if (s == (int)consumer.updates().size()) {
669 // Simplify region computed and perform additional
670 // special-case analysis to make it faster to evaluate.
671 node.region_computed_all_common_cases = true;
672 for (int j = 0; j < consumer.dimensions(); j++) {
673 const auto &req = node.region_required[j];
674 auto &comp = node.region_computed[j];
675 comp.in.min = simplify(apply_param_estimates.mutate(comp.in.min));
676 comp.in.max = simplify(apply_param_estimates.mutate(comp.in.max));
677 if (equal(comp.in.min, req.min) && equal(comp.in.max, req.max)) {
678 comp.equals_required = true;
679 } else {
680 const Min *min = comp.in.min.as<Min>();
681 const Max *max = comp.in.max.as<Max>();
682 const int64_t *min_b = min ? as_const_int(min->b) : nullptr;
683 const int64_t *max_b = max ? as_const_int(max->b) : nullptr;
684 if (min_b && max_b && equal(min->a, req.min) && equal(max->a, req.max)) {
685 comp.equals_union_of_required_with_constants = true;
686 comp.c_min = *min_b;
687 comp.c_max = *max_b;
688 } else {
689 node.region_computed_all_common_cases = false;
690 }
691 }
692 }
693 }
694
695 // We'll take any existing reordering, but won't handle existing splits
696 user_assert(sched.splits().empty())
697 << "The Func \"" << consumer.name() << "\" has scheduling directive(s) "
698 << "applied to it; you must remove these, or conditionalize them "
699 << "using `if (!auto_schedule)`, to use the autoscheduler on this pipeline.";
700 stage.loop_nest_all_common_cases = true;
701 for (size_t i = 0; i < sched.dims().size(); i++) {
702 const auto &d = sched.dims()[i];
703 // Skip synthetic loops like "__outermost"
704 if (!stage_scope_with_symbolic_rvar_bounds.contains(d.var)) {
705 continue;
706 }
707
708 Node::Loop l;
709 l.var = d.var;
710 l.accessor = stage.name + ".get_schedule().dims()[" + std::to_string(i) + "].var";
711
712 // We already have the right variable names in the stage scope
713 Interval in = stage_scope_with_concrete_rvar_bounds.get(l.var);
714 l.min = in.min;
715 l.max = in.max;
716 l.pure = d.is_pure();
717 l.rvar = d.is_rvar();
718 l.pure_dim = -1;
719
720 // Additional analysis to speed up evaluation of
721 // common cases. Loop bounds that are just one of
722 // the dimensions of the symbolic region computed
723 // are common, as are constant bounds.
724 l.equals_region_computed = false;
725 for (int j = 0; j < consumer.dimensions(); j++) {
726 if (l.var == pure_args[j]) {
727 l.pure_dim = j;
728 }
729 if (equal(l.min, node.region_computed[j].in.min) &&
730 equal(l.max, node.region_computed[j].in.max)) {
731 l.equals_region_computed = true;
732 l.region_computed_dim = j;
733 break;
734 }
735 }
736
737 if (!l.equals_region_computed) {
738 const int64_t *c_min = as_const_int(l.min), *c_max = as_const_int(l.max);
739 if (c_min && c_max) {
740 l.bounds_are_constant = true;
741 l.c_min = *c_min;
742 l.c_max = *c_max;
743 } else {
744 l.bounds_are_constant = false;
745 }
746 }
747
748 stage.loop_nest_all_common_cases &= (l.bounds_are_constant || l.equals_region_computed);
749 stage.loop.emplace_back(std::move(l));
750 }
751
752 // Bundle all expressions associated with the definition into a single dummy call node
753 vector<Expr> exprs_vector = def.args();
754 exprs_vector.insert(exprs_vector.end(), def.values().begin(), def.values().end());
755 if (def.predicate().defined()) {
756 exprs_vector.push_back(def.predicate());
757 }
758 Expr exprs = Call::make(Int(32), "dummy", exprs_vector, Call::Extern);
759
760 // Walk over the expressions involved sniffing types
761 class CheckTypes : public IRVisitor {
762 using IRVisitor::visit;
763
764 void visit(const IntImm *op) override {
765 check_type(op->type);
766 }
767
768 void visit(const UIntImm *op) override {
769 check_type(op->type);
770 }
771
772 void visit(const FloatImm *op) override {
773 check_type(op->type);
774 }
775
776 void visit(const Variable *op) override {
777 check_type(op->type);
778 }
779
780 void visit(const Call *op) override {
781 calls[op->name]++;
782 IRVisitor::visit(op);
783 check_type(op->type);
784 if (op->call_type == Call::Halide || op->call_type == Call::Image) {
785 is_pointwise &= op->args.size() == func.args().size();
786 if (is_pointwise) {
787 for (size_t i = 0; i < op->args.size(); i++) {
788 const Variable *v = op->args[i].as<Variable>();
789 is_pointwise &= (v != nullptr) && (v->name == func.args()[i]);
790 }
791 }
792 }
793 }
794
795 void visit(const Cast *op) override {
796 IRVisitor::visit(op);
797 check_type(op->type);
798 }
799
800 void check_type(Type t) {
801 if (t.bits() > 1 &&
802 (!narrowest_type.bits() ||
803 t.bits() < narrowest_type.bits())) {
804 narrowest_type = t;
805 }
806 }
807 Function func;
808
809 public:
810 bool is_pointwise = true;
811 int leaves = 0;
812 Type narrowest_type;
813 map<string, int> calls;
814 CheckTypes(const Function &f)
815 : func(f) {
816 }
817 };
818 CheckTypes checker(consumer);
819 exprs.accept(&checker);
820
821 Type widest_output_type = def.values()[0].type();
822
823 int bytes_per_point = 0;
824 for (const auto &e : def.values()) {
825 bytes_per_point += e.type().bytes();
826 if (e.type().bytes() > widest_output_type.bytes()) {
827 widest_output_type = e.type();
828 }
829 }
830 if (s == 0) {
831 node.bytes_per_point = bytes_per_point;
832 }
833
834 stage.vector_size = target.natural_vector_size(checker.narrowest_type);
835
836 if (s == 0) {
837 node.vector_size = stage.vector_size;
838 } else {
839 node.vector_size = std::max(node.vector_size, stage.vector_size);
840 }
841
842 node.is_output = false;
843 for (const auto &o : outputs) {
844 node.is_output |= o.same_as(node.func);
845 }
846
847 if (node.is_output) {
848 // Get the bounds estimate
849 map<string, Span> estimates;
850 for (const auto &b : consumer.schedule().estimates()) {
851 int64_t i_min = *as_const_int(b.min);
852 int64_t i_extent = *as_const_int(b.extent);
853
854 if ((false)) { // Intentional dead code. Extra parens to pacify clang-tidy.
855 // Some methods we compare to compile for
856 // statically known input/output sizes. We
857 // don't need to - we take estimates but
858 // the compiled code doesn't enforce
859 // them. If you want to make a comparison
860 // fair and target a fixed size, use this
861 // branch of the if. In practice we don't
862 // see a runtime difference, so we left it
863 // disabled. In theory, Sizes being
864 // constant makes it possible to do things
865 // like unroll across color channels, so
866 // it affects the scheduling space.
867 Func(node.func).bound(b.var, b.min, b.extent);
868 estimates[b.var] = Span(i_min, i_min + i_extent - 1, true);
869 } else {
870 estimates[b.var] = Span(i_min, i_min + i_extent - 1, false);
871 }
872 }
873 for (const auto &b : consumer.schedule().bounds()) {
874 const int64_t *i_min = as_const_int(b.min);
875 const int64_t *i_extent = as_const_int(b.extent);
876 if (i_min && i_extent) {
877 // It's a true bound, not just an estimate
878 estimates[b.var] = Span(*i_min, *i_min + *i_extent - 1, true);
879 }
880 }
881 // Set the bounds using the estimates
882 for (int i = 0; i < consumer.dimensions(); i++) {
883 auto it = estimates.find(consumer.args()[i]);
884 user_assert(it != estimates.end())
885 << "Need an estimate on dimension " << i << " of \"" << consumer.name() << "\"";
886 node.estimated_region_required.push_back(it->second);
887 }
888 }
889
890 stage.index = s;
891
892 exprs = apply_param_estimates.mutate(exprs);
893
894 for (auto &p : func_value_bounds) {
895 p.second.min = apply_param_estimates.mutate(p.second.min);
896 p.second.max = apply_param_estimates.mutate(p.second.max);
897 }
898
899 // For this stage scope we want symbolic bounds for the rvars
900
901 // Now create the edges that lead to this func
902 bool any_incoming_edges = false;
903 node.is_pointwise = !node.func.has_update_definition();
904
905 // TODO: peephole the boundary condition call pattern instead of assuming the user used the builtin
906 node.is_boundary_condition = node.is_pointwise && starts_with(node.func.name(), "repeat_edge");
907
908 auto boxes = boxes_required(exprs, stage_scope_with_symbolic_rvar_bounds, func_value_bounds);
909 for (auto &p : boxes) {
910 auto it = env.find(p.first);
911 if (it != env.end() && p.first != consumer.name()) {
912 // Discard loads from input images and self-loads
913 Edge edge;
914 edge.consumer = &stage;
915 edge.producer = node_map.at(env[p.first]);
916 edge.all_bounds_affine = true;
917
918 for (Interval &in : p.second.bounds) {
919 // Whenever a relationship is unbounded, we must inline
920 internal_assert(in.is_bounded())
921 << "Unbounded producer->consumer relationship: "
922 << edge.producer->func.name() << " -> " << edge.consumer->name << "\n";
923 Edge::BoundInfo min(simplify(in.min), *edge.consumer);
924 Edge::BoundInfo max(simplify(in.max), *edge.consumer);
925 edge.bounds.emplace_back(std::move(min), std::move(max));
926 edge.all_bounds_affine &= edge.bounds.back().first.affine;
927 edge.all_bounds_affine &= edge.bounds.back().second.affine;
928 }
929 edge.calls = checker.calls[edge.producer->func.name()];
930 any_incoming_edges = true;
931 node.is_pointwise &= checker.is_pointwise;
932 edges.emplace_back(std::move(edge));
933 }
934 }
935
936 node.is_wrapper = node.func.is_wrapper();
937 node.is_input = !node.func.has_update_definition() && node.is_wrapper && !any_incoming_edges;
938 node.dimensions = node.func.dimensions();
939 }
940 }
941
942 // Initialize the memory layouts for the bounds structs
943 for (auto &n : nodes) {
944 n.bounds_memory_layout.reset(new BoundContents::Layout);
945 auto &l = *(n.bounds_memory_layout);
946 l.computed_offset = n.func.dimensions();
947 l.total_size = l.computed_offset + n.func.dimensions();
948 for (const auto &s : n.stages) {
949 l.loop_offset.push_back(l.total_size);
950 l.total_size += (int)s.loop.size();
951 }
952 }
953
954 // Give all the stages unique ids to support perfect hashing of them
955 {
956 int i = 0;
957 for (auto &n : nodes) {
958 for (auto &s : n.stages) {
959 s.id = i;
960 s.max_id = stage_count;
961 i++;
962 }
963 }
964 }
965
966 for (size_t i = 0; i < edges.size(); i++) {
967 edges[i].producer->outgoing_edges.push_back(&(edges[i]));
968 edges[i].consumer->incoming_edges.push_back(&(edges[i]));
969 }
970
971 // Compute transitive dependencies
972 for (size_t i = nodes.size(); i > 0; i--) {
973 auto &n = nodes[i - 1];
974 for (auto &s : n.stages) {
975 s.dependencies.resize(nodes.size(), false);
976 for (auto *e : s.incoming_edges) {
977 s.dependencies[e->producer->id] = true;
978 for (auto &s2 : e->producer->stages) {
979 for (size_t j = 0; j < nodes.size(); j++) {
980 s.dependencies[j] = s.dependencies[j] || s2.dependencies[j];
981 }
982 }
983 }
984 }
985 }
986
987 // Compute the algorithm-specific features for the neural net
988 featurize();
989}

991void FunctionDAG::featurize() {
992 for (Node &node : nodes) {
993 for (size_t stage_idx = 0; stage_idx < node.stages.size(); stage_idx++) {
994 Node::Stage &stage = node.stages[stage_idx];
995
996 Featurizer featurizer(node.func, stage);
997
998 if (node.func.extern_definition_proxy_expr().get()) {
999 // Extern function call with a proxy implementation specified: generate the featurization from the proxy
1000 Expr v = simplify(node.func.extern_definition_proxy_expr());
1001 v = common_subexpression_elimination(v);
1002 v.accept(&featurizer);
1003 } else {
1004 Definition def = node.func.definition();
1005 if (stage_idx > 0) {
1006 def = node.func.updates()[stage_idx - 1];
1007 }
1008 stage.features = PipelineFeatures();
1009
1010 for (auto v : def.values()) {
1011 featurizer.visit_store_args(node.func.name(), v.type(), def.args());
1012 v = common_subexpression_elimination(simplify(v)); // Get things into canonical form
1013 v.accept(&featurizer);
1014 }
1015 for (auto v : def.args()) {
1016 v = common_subexpression_elimination(simplify(v)); // Get things into canonical form
1017 v.accept(&featurizer);
1018 }
1019 }
1020 }
1021 }

1025void FunctionDAG::dump_internal(OS &os) const {
1026 for (const Node &n : nodes) {
1027 os << "Node: " << n.func.name() << "\n"
1028 << " Symbolic region required: \n";
1029 for (const SymbolicInterval &i : n.region_required) {
1030 os << " " << i.min << ", " << i.max << "\n";
1031 }
1032 os << " Region computed: \n";
1033 for (const auto &i : n.region_computed) {
1034 os << " " << i.in.min << ", " << i.in.max << "\n";
1035 }
1036 for (size_t i = 0; i < n.stages.size(); i++) {
1037 os << " Stage " << i << ":\n";
1038 for (const auto &l : n.stages[i].loop) {
1039 os << " " << l.var << " " << l.min << " " << l.max << "\n";
1040 }
1041 n.stages[i].features.dump(os);
1042 }
1043 os << " pointwise: " << n.is_pointwise
1044 << " boundary condition: " << n.is_boundary_condition
1045 << " wrapper: " << n.is_wrapper
1046 << " input: " << n.is_input
1047 << " output: " << n.is_output << "\n";
1048 }
1049 for (const Edge &e : edges) {
1050 os << "Edge: " << e.producer->func.name() << " -> " << e.consumer->name << "\n"
1051 << " Footprint: \n";
1052 int j = 0;
1053 for (const auto &i : e.bounds) {
1054 os << " Min " << j << ": " << i.first.expr << "\n";
1055 os << " Max " << j << ": " << i.second.expr << "\n";
1056 j++;
1057 }
1058
1059 os << " Load Jacobians:\n";
1060 for (const auto &jac : e.load_jacobians) {
1061 jac.dump(" ");
1062 }
1063 }
