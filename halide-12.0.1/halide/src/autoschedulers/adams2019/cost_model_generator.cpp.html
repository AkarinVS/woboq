15// A model weight is either just an input, or an input and an output
16// (the updated weights and the ADAM state) depending on whether we're
17// doing inference or training.
18template<bool training>
19struct ModelWeight;

41template<>
42struct ModelWeight<true> : public GeneratorInput<Buffer<float>> {
43 GeneratorOutput<Buffer<float>> grad;
44
45 ModelWeight(const std::string &name, int dim)
46 : GeneratorInput<Buffer<float>>(name, dim), grad("updated_" + name, dim + 1) {
47 }
48 void backprop(const Derivative &d, const Expr &learning_rate, const Expr &timestep) {
49 std::vector<Expr> args(dimensions() + 1);
50 for (auto &e : args) {
51 e = Var();
52 }
53 grad(args) = undef<float>();
54
55 // We'll report back the new weights and the loss gradients,
56 // and update the ADAM state. Depending on the mode the caller
57 // is in, it may use the new weights, or it may just send the
58 // loss gradients up to an ADAM server.
59 args.back() = 0;
60 FuncRef new_weight = grad(args);
61 args.back() = 1;
62 FuncRef smoothed_deriv = grad(args);
63 args.back() = 2;
64 FuncRef smoothed_second_moment = grad(args);
65 args.back() = 3;
66 FuncRef loss_gradient = grad(args);
67
68 args.pop_back();
69 Expr current_weight = (*this)(args);
70
71 loss_gradient = d(*this)(args);
72
73 // Update the first and second moment estimates
74 smoothed_deriv = 0.9f * smoothed_deriv + 0.1f * loss_gradient;
75 smoothed_second_moment = 0.999f * smoothed_second_moment + 0.001f * pow(loss_gradient, 2);
76
77 // Correction to account for the fact that the smoothed_deriv
78 // and smoothed_second_moment start at zero when t == 0
79 Expr smoothed_deriv_correction = 1 / (1 - pow(0.9f, timestep + 1));
80 Expr smoothed_second_moment_correction = 1 / (1 - pow(0.999f, timestep + 1));
81
82 // Update the weights
83 Expr step = learning_rate * smoothed_deriv * smoothed_deriv_correction;
84 step /= sqrt(smoothed_second_moment * smoothed_second_moment_correction) + 1e-5f;
85
86 new_weight = current_weight - step;
87 }
88
89 void set_shape(int s0 = 0, int s1 = 0, int s2 = 0) {
90 if (s0) {
91 dim(0).set_bounds(0, s0);
92 dim(0).set_estimate(0, s0);
93 grad.dim(0).set_bounds(0, s0);
94 grad.dim(0).set_estimate(0, s0);
95 grad.bound(grad.args()[0], 0, s0);
96 grad.set_estimate(grad.args()[0], 0, s0);
97 }
98 if (s1) {
99 dim(1).set_bounds(0, s1);
100 dim(1).set_estimate(0, s1);
101 grad.dim(1).set_bounds(0, s1);
102 grad.dim(1).set_estimate(0, s1);
103 grad.bound(grad.args()[1], 0, s1);
104 grad.set_estimate(grad.args()[1], 0, s1);
105 }
106 if (s2) {
107 dim(2).set_bounds(0, s2);
108 dim(2).set_estimate(0, s2);
109 grad.dim(2).set_bounds(0, s2);
110 grad.dim(2).set_estimate(0, s2);
111 grad.bound(grad.args()[2], 0, s2);
112 grad.set_estimate(grad.args()[2], 0, s2);
113 }
114 grad.dim(dimensions()).set_bounds(0, 4);
115 grad.dim(dimensions()).set_estimate(0, 4);
116 }

119template<bool training>
120class CostModel : public Generator<CostModel<training>> {
121public:
122 template<typename T>
123 using Input = GeneratorInput<T>;
124 template<typename T>
125 using Output = GeneratorOutput<T>;
126 using Generator<CostModel<training>>::auto_schedule;
127 using Generator<CostModel<training>>::get_pipeline;
128
129 // Number of pipeline stages
130 Input<int> num_stages{"num_stages", 1};
131
132 // Batch size. Every item in the batch is a different schedule for
133 // the same algorithm.
134 Input<int> batch_size{"batch_size", 1};
135
136 // Number of cores on the target machine. Used to reason about idle cores.
137 Input<int> num_cores{"num_cores", 1};
138
139 // Algorithm-specific features
140 Input<Buffer<float>> pipeline_features{"pipeline_features", 3};
141
142 // Schedule-specific features
143 Input<Buffer<float>> schedule_features{"schedule_features", 3};
144
145 // Network weights. We use some template-fu so that they are
146 // inputs in inference mode, and inputs and outputs in training
147 // mode.
148 using Weight = ModelWeight<training>;
149 Weight head1_filter{"head1_filter", 3};
150 Weight head1_bias{"head1_bias", 1};
151 Weight head2_filter{"head2_filter", 2};
152 Weight head2_bias{"head2_bias", 1};
153 Weight filter1{"filter1", 2};
154 Weight bias1{"bias1", 1};
155
156 // Some extra inputs for training mode.
157 Input<float> learning_rate{"learning_rate", 1.0f};
158 Input<int> timestep{"timestep", 0}; // Needed by ADAM
159
160 // The index of the fastest schedule in the batch. Used as a
161 // reference point for computing relative throughput.
162 Input<int> reference{"reference", 0};
163
164 // The true runtimes obtained by benchmarking.
165 Input<Buffer<float>> true_runtime{"true_runtime", 1};
166
167 // The predicted runtimes
168 Output<Buffer<float>> prediction_output{"prediction_output", 1};
169
170 // The loss. L2 on relative throughput.
171 Output<Buffer<float>> loss_output{"loss_output", 0};
172
173 // Zero pad alone the last dimension of a Func
174 Func pad_stages(const Func &f, const Expr &stages) {
175 Halide::Region bounds(f.dimensions());
176 bounds[1].min = 0;
177 bounds[1].extent = stages;
178 return BoundaryConditions::constant_exterior(f, cast(f.value().type(), 0), bounds);
179 }
180
181 Expr activation(const Expr &e) {
182 // relu
183 return max(e, 0);
184 }
185
186 Expr sigmoid(const Expr &e) {
187 return 1 / (1 + exp(-e));
188 }
189
190 void generate() {
191 Var c("c"), w("w"), n("n"), j("j"), s("s");
192
193 Func normalized_schedule_features("normalized_schedule_features");
194 normalized_schedule_features(n, c, s) = fast_log(schedule_features(n, c, s) + 1);
195
196 // Force the weights of the algorithm embedding layer to be positive and bounded.
197 Func squashed_head1_filter("squashed_head1_filter");
198 squashed_head1_filter(c, s, n) = sigmoid(head1_filter(c, s, n));
199
200 // Explicitly broadcast the weights across the batch. This
201 // give the autoscheduler some more options in the
202 // reverse-mode pipeline.
203 Func squashed_head1_filter_broadcast("squashed_head1_filter_broadcast");
204 squashed_head1_filter_broadcast(c, w, s, n) = squashed_head1_filter(c, s, n);
205
206 // The conv layer that embeds the algorithm-specific features.
207 Func head1_conv("head1_conv");
208 RDom r_head1(0, head1_w, 0, head1_h);
209 head1_conv(c, w) = head1_bias(c);
210 head1_conv(c, w) += (squashed_head1_filter_broadcast(c, w, r_head1.x, r_head1.y) *
211 pipeline_features(r_head1.x, r_head1.y, w));
212
213 // No point in a relu - the inputs and weights are positive
214
215 // The conv layer that embeds the schedule-specific features.
216 Func head2_conv("head2_conv");
217 RDom r_head2(0, head2_w);
218 head2_conv(c, w, n) = head2_bias(c);
219 head2_conv(c, w, n) += head2_filter(c, r_head2) * normalized_schedule_features(n, r_head2, w);
220
221 Func head2_relu("head2_relu");
222 head2_relu(c, w, n) = activation(head2_conv(c, w, n));
223
224 // The conv layer that computes coefficients, split into two
225 // stages. First we consumer the algorithm embedding.
226 Func conv1_stage1("conv1_stage1");
227 RDom r1_stage1(0, head1_channels);
228 conv1_stage1(c, w) = bias1(c);
229 conv1_stage1(c, w) += filter1(c, r1_stage1.x) * head1_conv(r1_stage1.x, w);
230
231 // Then we consume the schedule embedding.
232 Func conv1_stage2("conv1_stage2");
233 RDom r1_stage2(0, head2_channels);
234 conv1_stage2(c, w, n) = conv1_stage1(c, w);
235 conv1_stage2(c, w, n) += filter1(c, head1_filter.dim(0).extent() + r1_stage2.x) * head2_relu(r1_stage2.x, w, n);
236
237 // The final set of predicted coefficients.
238 Func relu1("relu1");
239 relu1(c, w, n) = activation(conv1_stage2(c, w, n));
240
241 // That's the end of the neural network. Now we will use these
242 // coefficients with a bunch of hand-designed terms.
243
244 // Unpack all of the schedule features. We don't use all of
245 // them, but it's easier to avoid bugs if we just unpack them
246 // all in the same order as Featurization.h
247 int idx = 0;
248 Expr num_realizations = schedule_features(n, idx++, w);
249 Expr num_productions = schedule_features(n, idx++, w);
250 Expr points_computed_per_realization = schedule_features(n, idx++, w);
251 Expr points_computed_per_production = schedule_features(n, idx++, w);
252 Expr points_computed_total = schedule_features(n, idx++, w);
253 Expr points_computed_minimum = schedule_features(n, idx++, w);
254 Expr innermost_loop_extent = schedule_features(n, idx++, w);
255 Expr innermost_pure_loop_extent = schedule_features(n, idx++, w);
256 Expr unrolled_loop_extent = schedule_features(n, idx++, w);
257 Expr inner_parallelism = schedule_features(n, idx++, w);
258 Expr outer_parallelism = schedule_features(n, idx++, w);
259 Expr bytes_at_realization = schedule_features(n, idx++, w);
260 Expr bytes_at_production = schedule_features(n, idx++, w);
261 Expr bytes_at_root = schedule_features(n, idx++, w);
262 Expr innermost_bytes_at_realization = schedule_features(n, idx++, w);
263 Expr innermost_bytes_at_production = schedule_features(n, idx++, w);
264 Expr innermost_bytes_at_root = schedule_features(n, idx++, w);
265 Expr inlined_calls = schedule_features(n, idx++, w);
266 Expr unique_bytes_read_per_realization = schedule_features(n, idx++, w);
267 Expr unique_lines_read_per_realization = schedule_features(n, idx++, w);
268 Expr allocation_bytes_read_per_realization = schedule_features(n, idx++, w);
269 Expr working_set = schedule_features(n, idx++, w);
270 Expr vector_size = schedule_features(n, idx++, w);
271 Expr native_vector_size = schedule_features(n, idx++, w);
272 Expr num_vectors = schedule_features(n, idx++, w);
273 Expr num_scalars = schedule_features(n, idx++, w);
274 Expr scalar_loads_per_vector = schedule_features(n, idx++, w);
275 Expr vector_loads_per_vector = schedule_features(n, idx++, w);
276 Expr scalar_loads_per_scalar = schedule_features(n, idx++, w);
277 Expr bytes_at_task = schedule_features(n, idx++, w);
278 Expr innermost_bytes_at_task = schedule_features(n, idx++, w);
279 Expr unique_bytes_read_per_vector = schedule_features(n, idx++, w);
280 Expr unique_lines_read_per_vector = schedule_features(n, idx++, w);
281 Expr unique_bytes_read_per_task = schedule_features(n, idx++, w);
282 Expr unique_lines_read_per_task = schedule_features(n, idx++, w);
283 Expr working_set_at_task = schedule_features(n, idx++, w);
284 Expr working_set_at_production = schedule_features(n, idx++, w);
285 Expr working_set_at_realization = schedule_features(n, idx++, w);
286 Expr working_set_at_root = schedule_features(n, idx++, w);
287 assert(idx == head2_w);
288
289 // Count up the number of things computed, applying a
290 // different cost to vectors and scalars, and a different cost
291 // depending on whether we were inlined.
292 Expr compute_cost = select(inlined_calls == 0,
293 (vector_size * num_vectors * relu1(0, w, n) +
294 num_scalars * relu1(1, w, n)),
295 (vector_size * num_vectors * relu1(2, w, n) +
296 num_scalars * relu1(3, w, n)));
297
298 // Round up these costs according to how neatly we're using
299 // our cores.
300 Expr num_tasks = max(1, inner_parallelism * outer_parallelism);
301 Expr tasks_per_core = num_tasks / num_cores;
302 Expr idle_core_wastage = ceil(tasks_per_core) / max(1, tasks_per_core);
303 compute_cost *= idle_core_wastage;
304
305 // Next comes a long list of plausible terms to capture the cost of loads.
306 Expr load_cost = (num_realizations * unique_lines_read_per_realization * relu1(5, w, n) +
307 num_realizations * unique_bytes_read_per_realization * relu1(6, w, n) +
308 num_vectors * scalar_loads_per_vector * relu1(7, w, n) +
309 num_scalars * scalar_loads_per_scalar * relu1(8, w, n) +
310 num_vectors * vector_loads_per_vector * relu1(9, w, n) +
311 num_scalars * unique_bytes_read_per_vector * relu1(10, w, n) +
312 num_vectors * unique_bytes_read_per_vector * relu1(11, w, n) +
313 num_scalars * unique_lines_read_per_vector * relu1(12, w, n) +
314 num_vectors * unique_lines_read_per_vector * relu1(13, w, n) +
315 num_tasks * unique_bytes_read_per_task * relu1(14, w, n) +
316 num_tasks * unique_lines_read_per_task * relu1(15, w, n));
317
318 // Next we have the cost of stores.
319 Expr lines_written_per_realization = inner_parallelism * (bytes_at_task / max(1, innermost_bytes_at_task));
320
321 // Use separate coefficients for things with internal
322 // parallelism, because for stages with internal parallelism,
323 // most values of the values being stored will be consumed on
324 // another core, so they will get punted out to L3 no matter
325 // how small. Also use a separate term for the final stage, as
326 // we never pay the cost of loading from it.
327 Expr alpha = select(inner_parallelism > 1, relu1(16, w, n),
328 w == 0, relu1(17, w, n),
329 relu1(18, w, n));
330 Expr beta = select(inner_parallelism > 1, relu1(19, w, n),
331 w == 0, relu1(20, w, n),
332 relu1(21, w, n));
333
334 Expr store_cost = num_realizations * (lines_written_per_realization * alpha +
335 bytes_at_realization * beta);
336
337 // Now account for false sharing of cache lines. The
338 // probability of a store hitting a cache line also hit by
339 // another core is inversely proportional to
340 // innermost_bytes_at_task, and the cost is paid on every
341 // store.
342 Expr cost_of_false_sharing =
343 select(inner_parallelism > 1,
344 relu1(22, w, n) * (num_vectors + num_scalars) / max(1, innermost_bytes_at_task),
345 0.0f);
346
347 store_cost += cost_of_false_sharing;
348
349 // Now add a term for false sharing of pages. The maximum
350 // number of threads that could all fault on the same page at
351 // the same time is:
352 Expr max_threads_hitting_same_page_fault = min(inner_parallelism, 4096 / max(1, innermost_bytes_at_task));
353
354 // The total number of page faults is proportionate to the number of bytes allocated
355 const Expr &num_page_faults = bytes_at_production;
356
357 // And page faults are serviced serially, so the total CPU time gets multiplied by the thread count again!
358 Expr cost_of_page_faults = (num_page_faults * max_threads_hitting_same_page_fault *
359 inner_parallelism * outer_parallelism * relu1(23, w, n));
360
361 store_cost += cost_of_page_faults;
362
363 // Malloc is not free, so add a cost per allocation.
364 Expr cost_of_malloc = relu1(24, w, n) * num_realizations;
365
366 // A cost for launching a parallel task...
367 Expr cost_of_parallel_launches = num_productions * select(inner_parallelism > 1, relu1(25, w, n), 0.0f);
368
369 // ... and an overhead per task.
370 Expr cost_of_parallel_tasks = num_productions * (inner_parallelism - 1) * relu1(26, w, n);
371
372 Expr cost_of_parallelism = cost_of_parallel_tasks + cost_of_parallel_launches;
373
374 // Make it easier for the model to penalize working sets that
375 // start to fall out of cache by giving it a term that gets
376 // multiplied by the working set.
377 Expr cost_of_working_set = working_set * relu1(27, w, n);
378
379 // FIXME: For our best set of trained weights, store_cost was
380 // accidentally in the list below twice, so we double it here
381 // in order to not have to retrain.
382 store_cost *= 2;
383
384 Expr cost = (compute_cost +
385 store_cost +
386 load_cost +
387 cost_of_malloc +
388 cost_of_parallelism +
389 cost_of_working_set);
390
391 for (int i = 0; i < 32; i++) {
392 cost += 0.0f * relu1(i, w, n);
393 }
394
395 Func runtime_per_stage;
396 // Change units so that network weights are in a human-readable range.
397 runtime_per_stage(n, w) = cost * 1e-9f;
398
399 // Sum across the stages.
400 Func prediction;
401 RDom r_reduce(0, num_stages);
402 prediction(n) += runtime_per_stage(n, r_reduce);
403
404 prediction_output(n) = prediction(n);
405
406 Func err;
407
408 if (!training) {
409 loss_output() = 0.0f;
410 } else {
411
412 // The tail end of the reverse-mode pipeline
413 RDom r_batch(0, batch_size);
414
415 // We believe the coefficients on all the various
416 // components of cost should be positive, even before the
417 // relu, and even before schedule-specific features are
418 // taken into account. The network shouldn't be telling us
419 // that things would be cheaper if we would do more
420 // mallocs, or compute more values, or launch more
421 // parallel tasks. So we add a regularization term. This
422 // helps dead relus get unstuck.
423 RDom r_conv1_output(0, conv1_channels, 0, num_stages);
424 Expr regularize = sum(-min(conv1_stage2(r_conv1_output.x, r_conv1_output.y, n), 0));
425
426 // Our loss will be L2 on relative throughput.
427
428 // Get the reference runtime.
429 Expr n2 = clamp(reference, 0, batch_size - 1);
430 Expr scale = 1.0f / true_runtime(n2);
431
432 // Compute the relative true runtime and the relative predicted runtime
433 Expr p1 = prediction(n) * scale;
434 Expr r1 = true_runtime(n) * scale;
435
436 // Invert them to get relative throughput, and compute L2 loss.
437 Expr delta = pow(1.0f / max(p1, 1e-10f) - 1.0f / r1, 2);
438
439 // Add the regulization with a small weight.
440 err(n) = delta + 1e-5f * regularize;
441
442 // Sum the errors over the batch.
443 Expr loss = sum(err(r_batch));
444
445 loss_output() = loss;
446
447 // Compute derivatives of the loss, and backpropagate them
448 // to the model weights.
449 Derivative d_loss_d = propagate_adjoints(loss_output);
450
451 Weight *weights[] = {&head1_filter, &head1_bias,
452 &head2_filter, &head2_bias,
453 &filter1, &bias1};
454
455 for (Weight *w : weights) {
456 w->backprop(d_loss_d, learning_rate, timestep);
457 }
458 }
459
460 // All the model weight shapes are statically known, so we
461 // tell Halide their sizes to simplify the generated code.
462 head1_filter.set_shape(head1_channels, head1_w, head1_h);
463 head1_bias.set_shape(head1_channels);
464 head2_filter.set_shape(head2_channels, head2_w);
465 head2_bias.set_shape(head2_channels);
466 filter1.set_shape(conv1_channels, head1_channels + head2_channels);
467 bias1.set_shape(conv1_channels);
468
469 // Estimates for autoscheduling this pipeline (using
470 // itself!). We do that offline and check in the generated
471 // schedule source, so that bugs in our autoscheduler don't
472 // cause build nightmares due to the circular dependency.
473 num_cores.set_estimate(32);
474 reference.set_estimate(0);
475 batch_size.set_estimate(80);
476 num_stages.set_estimate(13);
477 prediction_output.set_estimates({{0, 80}});
478 learning_rate.set_estimate(0.001f);
479 timestep.set_estimate(37);
480 pipeline_features.set_estimates({{0, head1_w}, {0, head1_h}, {0, 13}});
481 schedule_features.set_estimates({{0, 80}, {0, head2_w}, {0, 13}});
482 true_runtime.set_estimates({{0, 80}});
483
484 // SCHEDULE
485 if (training && !auto_schedule) {
486 do_cost_model_schedule(get_pipeline());
487 } else if (auto_schedule) {
488 // Do nothing.
489 } else {
490 // We just write down a good schedule for
491 // inference. Scheduling a couple of convs is easy.
492 Var no;
493 prediction_output.specialize(batch_size < 8).split(n, no, n, 1);
494 prediction_output.compute_root().split(n, no, n, 8).parallel(no);
495 prediction_output.bound(n, 0, batch_size);
496
497 // schedule for the forwards path
498 const int vec = 8;
499
500 // A helper function for scheduling conv layers
501 auto schedule_conv = [&](Func conv, Func relu, const RVar &r_channels) {
502 Var ci, wi;
503 if (!training) {
504 relu
505 .compute_at(prediction_output, n)
506 .store_at(prediction_output, no)
507 .tile(c, w, ci, wi, vec, 4, TailStrategy::RoundUp)
508 .vectorize(ci);
509 conv.compute_at(relu, c);
510 } else {
511 // In training mode, we need the conv activations pre-relu too
512 conv.in()
513 .compute_root()
514 .tile(c, w, ci, wi, vec, 1, TailStrategy::RoundUp)
515 .vectorize(ci)
516 .unroll(wi)
517 .parallel(n, 8);
518 conv.compute_at(conv.in(), c);
519 relu
520 .compute_root()
521 .reorder_storage(c, w, n)
522 .reorder(c, w, n)
523 .vectorize(c, vec)
524 .parallel(n, 8);
525 }
526 conv
527 .vectorize(c)
528 .unroll(w)
529 .update()
530 .vectorize(c)
531 .unroll(w)
532 .reorder(c, w, r_channels);
533 };
534
535 // Pipeline features processing
536 conv1_stage1.compute_root().vectorize(c);
537 squashed_head1_filter.compute_root().vectorize(c);
538
539 // Schedule features processing. The number of schedule
540 // features is not close to a multiple of 8, so vectorized
541 // across the batch.
542 if (!training) {
543 normalized_schedule_features
544 .compute_at(prediction_output, no)
545 .vectorize(n);
546 } else {
547 normalized_schedule_features
548 .compute_root()
549 .vectorize(n, 8);
550 }
551
552 // conv+relu layers
553 schedule_conv(head2_conv, head2_relu, r_head2.x);
554 schedule_conv(conv1_stage2, relu1, r1_stage2.x);
555 }
556 }
557};
