17void State::compute_featurization(const FunctionDAG &dag, const MachineParams &params,
18 StageMap<ScheduleFeatures> *features, const CachingOptions &cache_options) {
19 StageMap<LoopNest::Sites> sites;
20 sites.make_large(dag.nodes[0].stages[0].max_id);
21 features->make_large(dag.nodes[0].stages[0].max_id);
22 internal_assert(root.defined());
23 root->get_sites(sites);
24
25 // For the input nodes and unscheduled outputs, the compute
26 // and store sites are root, and the produce and innermost
27 // sites are unset (nullptr)
28 for (const auto &n : dag.nodes) {
29 if (n.is_input || n.is_output) {
30 for (const auto &stage : n.stages) {
31 auto &s = sites.get_or_create(&stage);
32 if (s.compute == nullptr) {
33 s.compute = root.get();
34 s.store = root.get();
35 }
36 }
37 }
38 }
39
40 // For the unscheduled nodes, give them sites as deep as they
41 // could possibly be. We'll ignore the possibility of inlining
42 // them for now.
43 map<const LoopNest *, pair<const LoopNest *, int>> parent;
44 compute_loop_nest_parents(parent, root.get(), 0);
45 for (const auto &n : dag.nodes) {
46 if (sites.contains(&(n.stages[0]))) {
47 continue;
48 }
49 const LoopNest *loop = nullptr;
50 for (const auto *e : n.outgoing_edges) {
51 const auto &consumer_site = sites.get(e->consumer);
52 const LoopNest *l = consumer_site.innermost;
53 if (!l) {
54 l = consumer_site.compute;
55 }
56 if (!l) {
57 if (aslog::aslog_level() > 0) {
58 dump();
59 }
60 internal_error << e->producer->func.name() << " -> " << e->consumer->name << "\n";
61 }
62 if (loop) {
63 loop = deepest_common_ancestor(parent, l, loop);
64 } else {
65 loop = l;
66 }
67 }
68 internal_assert(loop)
69 << "Could not compute plausible site for unscheduled Func: "
70 << n.func.name() << "\n";
71 for (const auto &stage : n.stages) {
72 auto &site = sites.get_or_create(&stage);
73 site.compute = loop;
74 site.store = loop;
75 }
76 }
77
78 if (cache_options.cache_features) {
79 // Store unique hashes for each Site, to be used as keys into cache
80 for (const auto &c : root->children) {
81 sites.get(c->stage).hash_of_producers_stored_at_root = c->compute_hash_of_producers_stored_at_root(sites);
82 }
83 }
84
85 root->compute_features(dag, params, sites, 1, 1, nullptr, nullptr, *root, nullptr, features, cache_options.cache_features);
86
87 for (const auto &n : dag.nodes) {
88 if (sites.get(&(n.stages[0])).produce == nullptr) {
89 internal_assert(!features->contains(&(n.stages[0])))
90 << "Somehow an input or unscheduled node ended up in the featurization: "
91 << n.func.name() << "\n";
92 }
93 }
94}

96void State::save_featurization(const FunctionDAG &dag, const MachineParams &params,
97 const CachingOptions &cache_options, std::ostream &out) {
98 StageMap<ScheduleFeatures> features;
99 compute_featurization(dag, params, &features, cache_options);
100
101 for (const auto &n : dag.nodes) {
102 if (n.is_input) {
103 continue;
104 }
105 for (size_t stage_idx = n.stages.size(); stage_idx > 0; stage_idx--) {
106 const auto &s = n.stages[stage_idx - 1];
107 const size_t num_schedule_features = ScheduleFeatures::num_features();
108 const size_t num_pipeline_features = PipelineFeatures::num_features();
109 const auto &sched_feat = features.get(&s);
110
111 float buf[num_schedule_features + num_pipeline_features];
112 // Save them as floats
113 for (size_t i = 0; i < num_schedule_features; i++) {
114 buf[i] = sched_feat[i];
115 }
116
117 for (size_t i = 0; i < num_pipeline_features; i++) {
118 buf[i + num_schedule_features] = s.features[i];
119 }
120
121 out.write((const char *)buf, sizeof(buf));
122 }
123 }

126bool State::calculate_cost(const FunctionDAG &dag, const MachineParams &params,
127 CostModel *cost_model, const CachingOptions &cache_options,
128 int64_t memory_limit, bool verbose) {
129 StageMap<ScheduleFeatures> features;
130 compute_featurization(dag, params, &features, cache_options);
131
132 cost = 0.0f;
133
134 if (verbose) {
135 for (auto it = features.begin(); it != features.end(); it++) {
136 const auto &stage = *(it.key());
137 const auto &feat = it.value();
138 aslog(0) << "Schedule features for " << stage.stage.name() << "\n";
139 feat.dump();
140 }
141 }
142
143 internal_assert(cost_model) << "calculate_cost received nullptr for cost_model\n";
144
145 // Perform some addition pruning before burdening the cost model with silly states
146 for (auto it = features.begin(); it != features.end(); it++) {
147 if (!it.key()->node->is_wrapper) { // It's OK to repeatedly stage data
148 auto &feat = it.value();
149 if (feat.points_computed_total + feat.inlined_calls > 8 * feat.points_computed_minimum) {
150 cost = 1e50;
151 return false;
152 }
153 }
154 }
155
156 // Avoid code size explosion from recursive inlining.
157 if (root->max_inlined_calls() >= 256) {
158 cost = 1e50;
159 return false;
160 }
161
162 // Apply the hard limit on memory use
163 if (memory_limit >= 0) {
164 int64_t mem_used = (int64_t)features.begin().value().working_set_at_root;
165 for (auto it = features.begin(); it != features.end(); it++) {
166 if (it.key()->node->is_output ||
167 it.key()->node->is_input) {
168 // Not allocated by this pipeline
169 mem_used -= it.value().bytes_at_production;
170 }
171 }
172 if (mem_used > memory_limit) {
173 cost = 1e50;
174 return false;
175 }
176 }
177
178 // Tell the cost model about this state. It won't actually
179 // evaluate it until we call evaluate_costs (or if it runs out
180 // of internal buffer space), so that the evaluations can be
181 // batched.
182 cost_model->enqueue(dag, features, &cost);
183
184 cost_calculations++;
185 return true;
186}
187
188// Make a child copy of this state. The loop nest is const (we
189// make mutated copies of it, rather than mutating it), so we can
190// continue to point to the same one and so this is a cheap
191// operation.
192IntrusivePtr<State> State::make_child() const {
193 State *s = new State;
194 s->parent = this;
195 s->root = root;
196 s->cost = cost;
197 s->num_decisions_made = num_decisions_made;
198 return s;
199}

201// Generate the successor states to this state
202void State::generate_children(const FunctionDAG &dag,
203 const MachineParams &params,
204 CostModel *cost_model,
205 int64_t memory_limit,
206 std::function<void(IntrusivePtr<State> &&)> &accept_child,
207 Cache *cache) const {
208
209 internal_assert(root.defined() && root->is_root()) << "generate_children needs defined root\n";
210
211 if (num_decisions_made == 2 * (int)dag.nodes.size()) {
212 return;
213 }
214
215 int next_node = num_decisions_made / 2;
216 int phase = num_decisions_made % 2;
217
218 if (!may_subtile()) {
219 // When emulating the older search space, we do all
220 // parallelizing last, so that it is independent of the
221 // tiling decisions.
222 next_node = num_decisions_made % dag.nodes.size();
223 phase = num_decisions_made / dag.nodes.size();
224 }
225
226 // Enumerate all legal ways to schedule the next Func
227 const FunctionDAG::Node *node = &dag.nodes[next_node];
228 for (const auto *e : node->outgoing_edges) {
229 internal_assert(root->computes(e->consumer->node))
230 << "Partially scheduled code doesn't compute " << e->consumer->name
231 << ", which is one of the consumers of " << node->func.name();
232 }
233
234 if (node->is_input) {
235 // We don't need to schedule nodes that represent inputs,
236 // and there are no other decisions to be made about them
237 // at this time.
238 // aslog(0) << "Skipping over scheduling input node: " << node->func.name() << "\n";
239 auto child = make_child();
240 child->num_decisions_made++;
241 accept_child(std::move(child));
242 return;
243 }
244
245 if (!node->outgoing_edges.empty() && !root->calls(node)) {
246 aslog(0) << "In state:\n";
247 dump();
248 aslog(0) << node->func.name() << " is consumed by:\n";
249 for (const auto *e : node->outgoing_edges) {
250 aslog(0) << e->consumer->name << "\n";
251 aslog(0) << "Which in turn consumes:\n";
252 for (const auto *e2 : e->consumer->incoming_edges) {
253 aslog(0) << " " << e2->producer->func.name() << "\n";
254 }
255 }
256 internal_error << "Pipeline so far doesn't use next Func: " << node->func.name() << "\n";
257 }
258
259 int num_children = 0;
260
261 if (phase == 0) {
262 // Injecting realizations
263 {
264 // 1) Inline it
265 if (node->stages.size() == 1 && !node->is_output) {
266 auto child = make_child();
267 LoopNest *new_root = new LoopNest;
268 new_root->copy_from(*root);
269 new_root->inline_func(node);
270 child->root = new_root;
271 child->num_decisions_made++;
272 if (child->calculate_cost(dag, params, cost_model, cache->options, memory_limit)) {
273 num_children++;
274 accept_child(std::move(child));
275 }
276 }
277 }
278
279 // Some search-space pruning. If a node is pointwise, and
280 // so are all its inputs and so is its sole output, and
281 // inlining it is legal, just inline it. This saves time
282 // on long chains of pointwise things.
283 bool must_inline = (node->is_pointwise &&
284 (num_children > 0) &&
285 (node->outgoing_edges.size() == 1));
286 if (must_inline) {
287 for (const auto *e : node->stages[0].incoming_edges) {
288 must_inline &= e->producer->is_pointwise;
289 }
290 for (const auto *e : node->outgoing_edges) {
291 must_inline &= (e->consumer->node->is_pointwise ||
292 e->consumer->node->is_boundary_condition);
293 }
294 if (must_inline) {
295 return;
296 }
297 }
298
299 // Construct a list of plausible dimensions to vectorize
300 // over. Currently all of them. TODO: Pre-prune the list
301 // of sane dimensions to vectorize a Func over to reduce
302 // branching factor.
303 vector<int> vector_dims;
304 if (!node->is_input && !node->is_output) {
305 for (int v = 0; v < node->dimensions; v++) {
306 const auto &p = root->get_bounds(node)->region_computed(v);
307 if (p.extent() >= node->vector_size) {
308 vector_dims.push_back(v);
309 }
310 }
311 }
312
313 // Outputs must be vectorized over their innermost
314 // dimension, because we don't have control of the
315 // storage. Infer which dimension(s) is(are) the innermost one(s) by
316 // looking at the stride. Note that there can be more than one in
317 // case some dimensions have an extent of 1.
318 if (node->is_output && !node->func.output_buffers().empty()) {
319 const Parameter &output = node->func.output_buffers()[0];
320 int num_dims = output.dimensions();
321 for (int i = 0; i < num_dims; ++i) {
322 const Expr stride = output.stride_constraint(i);
323 const int64_t *s = as_const_int(stride);
324 if (s && *s == 1) {
325 vector_dims.push_back(i);
326 }
327 }
328 }
329
330 if (vector_dims.empty()) {
331 // This can happen if the output strides aren't known, or if all
332 // the dimensions are smaller than the vector size.
333 // TBD: consider extending compute_in_tiles to support -1 as a
334 // vector dim to indicate no vectorization.
335 for (int v = 0; v < node->dimensions; v++) {
336 vector_dims.push_back(v);
337 }
338 // Handle the case of full reductions that generate a scalar.
339 // We need at least one vector dimension to call cmopute_in_tiles
340 // below.
341 // TBD: figure out a better fallback strategy.
342 if (vector_dims.empty()) {
343 vector_dims.push_back(0);
344 }
345 }
346
347 // 2) Realize it somewhere
348 for (int vector_dim : vector_dims) {
349 auto tile_options = root->compute_in_tiles(node, nullptr, params, vector_dim, false);
350 for (IntrusivePtr<const LoopNest> &n : tile_options) {
351 auto child = make_child();
352 child->root = std::move(n);
353 child->num_decisions_made++;
354 if (child->calculate_cost(dag, params, cost_model, cache->options, memory_limit)) {
355 num_children++;
356 accept_child(std::move(child));
357 }
358 }
359 }
360 } else {
361 // We are parallelizing the loops of the func we just injected a realization for.
362
363 bool should_parallelize = false;
364 const vector<int64_t> *pure_size = nullptr;
365 if (params.parallelism > 1) {
366 for (const auto &c : root->children) {
367 if (c->node == node && node->dimensions > 0) {
368 if (c->stage->index == 0) {
369 pure_size = &(c->size);
370 }
371 should_parallelize = true;
372 }
373 }
374 }
375
376 if (!should_parallelize) {
377 // The Func must be scalar, or not compute_root, or
378 // we're not asking to use multiple cores. Just
379 // return a copy of the parent state
380 num_children++;
381 auto child = make_child();
382 child->num_decisions_made++;
383 accept_child(std::move(child));
384 } else {
385 internal_assert(pure_size);
386
387 if (cache->add_memoized_blocks(this, accept_child, node, num_children, dag, params, cost_model, memory_limit)) {
388 return; // successfully added cached states.
389 }
390
391 // Generate some candidate parallel task shapes.
392 auto tilings = generate_tilings(*pure_size, node->dimensions - 1, 2, true);
393
394 // We could also just parallelize the outer loop entirely
395 std::vector<int64_t> ones;
396 ones.resize(pure_size->size(), 1);
397 tilings.emplace_back(std::move(ones));
398
399 // Sort / filter the options
400 struct Option {
401 vector<int64_t> tiling;
402 double idle_core_wastage;
403 bool entire;
404 bool operator<(const Option &other) const {
405 return idle_core_wastage < other.idle_core_wastage;
406 }
407 // Ensure we don't accidentally copy this type
408 Option() = default;
409 Option(Option &&) = default;
410 Option &operator=(Option &&) = default;
411 Option(const Option &) = delete;
412 Option &operator=(const Option &) = delete;
413 };
414
415 vector<Option> options;
416 for (size_t i = 0; i < tilings.size(); i++) {
417 auto &t = tilings[i];
418 Option o;
419 o.entire = (i == tilings.size() - 1);
420
421 for (size_t j = 0; j < pure_size->size(); j++) {
422 t[j] = ((*pure_size)[j] + t[j] - 1) / t[j];
423 }
424 t.swap(o.tiling);
425
426 // Compute max idle cores across the other stages of the Func
427 int64_t min_total = 0, max_total = 0;
428 o.idle_core_wastage = 1;
429 for (const auto &c : root->children) {
430 if (c->node == node) {
431 int64_t total = 1;
432 for (const auto &l : c->stage->loop) {
433 if (!l.rvar) {
434 total *= o.tiling[l.pure_dim];
435 }
436 }
437 if (min_total != 0) {
438 min_total = std::min(min_total, total);
439 } else {
440 min_total = total;
441 }
442 max_total = std::max(max_total, total);
443 const double tasks_per_core = ((double)total) / params.parallelism;
444 o.idle_core_wastage = std::max(o.idle_core_wastage,
445 std::ceil(tasks_per_core) /
446 tasks_per_core);
447 }
448 }
449
450 // Filter out the less useful options
451 bool ok =
452 ((o.entire || min_total >= params.parallelism) &&
453 (max_total <= params.parallelism * 16));
454
455 if (!ok) {
456 continue;
457 }
458
459 options.emplace_back(std::move(o));
460 }
461 std::sort(options.begin(), options.end());
462
463 // If none of the options were acceptable, don't
464 // parallelize. This tends to happen for things like
465 // compute_root color matrices.
466 if (options.empty()) {
467 num_children++;
468 auto child = make_child();
469 child->num_decisions_made++;
470 accept_child(std::move(child));
471 return;
472 }
473
474 for (const auto &o : options) {
475 if (num_children >= 1 && (o.idle_core_wastage > 1.2 || !may_subtile())) {
476 // We have considered several options, and the
477 // remaining ones leave lots of cores idle.
478 break;
479 }
480
481 auto child = make_child();
482 LoopNest *new_root = new LoopNest;
483 new_root->copy_from(*root);
484 for (auto &c : new_root->children) {
485 if (c->node == node) {
486 if (may_subtile()) {
487 c = c->parallelize_in_tiles(params, o.tiling, new_root);
488 } else {
489 // We're emulating the old
490 // autoscheduler for an ablation, so
491 // emulate its parallelism strategy:
492 // just keep parallelizing outer loops
493 // until enough are parallel.
494 vector<int64_t> tiling = c->size;
495 int64_t total = 1;
496 for (size_t i = c->size.size(); i > 0; i--) {
497 if (!c->stage->loop[i - 1].pure || total >= params.parallelism) {
498 tiling[i - 1] = 1;
499 }
500 while (tiling[i - 1] > 1 &&
501 total * tiling[i - 1] > params.parallelism * 8) {
502 tiling[i - 1] /= 2;
503 }
504 total *= tiling[i - 1];
505 }
506 c = c->parallelize_in_tiles(params, tiling, new_root);
507 }
508 }
509 }
510 child->root = new_root;
511 child->num_decisions_made++;
512 if (child->calculate_cost(dag, params, cost_model, cache->options, memory_limit)) {
513 num_children++;
514 accept_child(std::move(child));
515 // Will early return if block caching is not enabled.
516 cache->memoize_blocks(node, new_root);
517 }
518 }
519 }
520 }
521
522 if (num_children == 0) {
523 aslog(0) << "Warning: Found no legal way to schedule "
524 << node->func.name() << " in the following State:\n";
525 dump();
526 // All our children died. Maybe other states have had
527 // children. Carry on.
528 }
529}

537// Apply the schedule represented by this state to a Halide
538// Pipeline. Also generate source code for the schedule for the
539// user to copy-paste to freeze this schedule as permanent artifact.
540void State::apply_schedule(const FunctionDAG &dag, const MachineParams &params) {
541 StageMap<std::unique_ptr<LoopNest::StageScheduleState>> state_map;
542 root->apply(LoopLevel::root(), state_map, params.parallelism, 0, nullptr, nullptr);
543
544 std::ostringstream src;
545
546 // Print handles for all the Funcs
547 int i = (int)(dag.nodes.size() - 1);
548 for (const auto &n : dag.nodes) {
549 if (!n.is_input) {
550 src << "Func " << n.func.name() << " = pipeline.get_func(" << i << ");\n";
551 }
552 i--;
553 }
554
555 // Gather all Vars and RVars so that we can declare them in the emitted source
556 map<string, string> vars, rvars;
557 for (auto &p : state_map) {
558 for (auto &v : p.second->vars) {
559 if (v.exists) {
560 if (v.var.is_rvar) {
561 rvars.emplace(v.var.name(), v.accessor);
562 } else {
563 vars.emplace(v.var.name(), v.accessor);
564 }
565 }
566 }
567 }
568 if (!vars.empty()) {
569 for (const auto &p : vars) {
570 if (p.second.empty()) {
571 src << "Var " << p.first << "(\"" << p.first << "\");\n";
572 } else {
573 src << "Var " << p.first << "(" << p.second << ");\n";
574 }
575 }
576 }
577 if (!rvars.empty()) {
578 for (const auto &p : rvars) {
579 if (p.second.empty()) {
580 src << "RVar " << p.first << "(\"" << p.first << "\");\n";
581 } else {
582 src << "RVar " << p.first << "(" << p.second << ");\n";
583 }
584 }
585 }
586
587 for (auto &p : state_map) {
588 if (p.first->node->is_input) {
589 continue;
590 }
591
592 Stage stage(p.first->stage);
593
594 // Do all the reorders and pick which vars to
595 // parallelize.
596 vector<VarOrRVar> vars;
597 int64_t parallel_tasks = 1;
598 vector<VarOrRVar> parallel_vars;
599 bool any_parallel_vars = false, any_parallel_rvars = false;
600 for (auto it = p.second->vars.rbegin(); it != p.second->vars.rend(); it++) {
601 if (!it->exists || it->extent == 1) {
602 continue;
603 }
604 if (!it->parallel) {
605 break;
606 }
607 any_parallel_rvars |= it->var.is_rvar;
608 any_parallel_vars |= !it->var.is_rvar;
609 parallel_tasks *= it->extent;
610 parallel_vars.push_back(it->var);
611 }
612
613 if (p.second->vars.size() > 1) {
614 p.second->schedule_source << "\n .reorder(";
615 bool first = true;
616 for (auto &v : p.second->vars) {
617 if (v.exists) {
618 vars.push_back(v.var);
619 if (!first) {
620 p.second->schedule_source << ", ";
621 } else {
622 p.second->schedule_source << "{";
623 }
624 first = false;
625 p.second->schedule_source << v.var.name();
626 }
627 }
628 p.second->schedule_source << "})";
629 stage.reorder(vars);
630 }
631
632 // Halide doesn't let you fuse an RVar with a Var, even if
633 // they are both pure.
634 bool can_fuse = !(any_parallel_vars && any_parallel_rvars);
635 if (can_fuse) {
636 for (size_t i = 1; i < parallel_vars.size(); i++) {
637 // Outermost, and next outermost. Preserve the inner
638 // name to not invalidate any compute_ats.
639 p.second->schedule_source << "\n .fuse(" << parallel_vars[i].name()
640 << ", " << parallel_vars[i - 1].name()
641 << ", " << parallel_vars[i].name() << ")";
642 stage.fuse(parallel_vars[i], parallel_vars[i - 1], parallel_vars[i]);
643 }
644 if (!parallel_vars.empty()) {
645 p.second->schedule_source << "\n .parallel(" << parallel_vars.back().name() << ")";
646 stage.parallel(parallel_vars.back());
647 }
648 } else {
649 for (const auto &v : parallel_vars) {
650 p.second->schedule_source << "\n .parallel(" << v.name() << ")";
651 stage.parallel(v);
652 }
653 }
654
655 // Reorder the vector dimension innermost
656 if (p.first->index == 0 && p.second->vector_dim > 0) {
657 vector<Var> storage_vars = Func(p.first->node->func).args();
658 for (int i = p.second->vector_dim; i > 0; i--) {
659 std::swap(storage_vars[i], storage_vars[i - 1]);
660 }
661 p.second->schedule_source << "\n .reorder_storage(";
662 bool first = true;
663 for (const auto &v : storage_vars) {
664 if (!first) {
665 p.second->schedule_source << ", ";
666 }
667 first = false;
668 p.second->schedule_source << v.name();
669 }
670 p.second->schedule_source << ")";
671 Func(p.first->node->func).reorder_storage(storage_vars);
672 }
673
674 // Dump the schedule source string
675 src << p.first->name
676 << p.second->schedule_source.str()
677 << ";\n";
678 }
679 // Sanitize the names of things to make them legal source code.
680 schedule_source = src.str();
681 bool in_quotes = false;
682 for (auto &c : schedule_source) {
683 in_quotes ^= (c == '"');
684 if (!in_quotes && c == '$') {
685 c = '_';
686 }
687 }
688}
