21template<typename T>
22std::vector<int> sort_indices(const std::vector<T> &v) {
23 std::vector<int> idx(v.size());
24 std::iota(idx.begin(), idx.end(), 0);
25 std::sort(idx.begin(), idx.end(),
26 [&v](int i1, int i2) { return v[i1] < v[i2]; });
27 return idx;
28}

30std::vector<int> get_int_bounds(const Box &bounds) {
31 std::vector<int> int_bounds;
32 int_bounds.reserve(bounds.size());
33 for (int i = 0; i < (int)bounds.size(); i++) {
34 Interval interval = bounds[i];
35 Expr extent = simplify(interval.max - interval.min + 1);
36 extent = simplify(substitute_var_estimates(extent));
37 const int64_t *extent_int = as_const_int(extent);
38 user_assert(extent_int != nullptr)
39 << "extent:" << extent << " is not constant.\n";
40 int_bounds.push_back(*extent_int);
41 }
42 return int_bounds;
43}

45std::vector<int> get_rvar_bounds(const std::vector<ReductionVariable> &rvars) {
46 std::vector<int> rvar_bounds;
47 rvar_bounds.reserve(rvars.size());
48 for (int arg_id = 0; arg_id < (int)rvars.size(); arg_id++) {
49 Expr extent = simplify(substitute_var_estimates(rvars[arg_id].extent));
50 const int64_t *extent_int = as_const_int(extent);
51 user_assert(extent_int != nullptr)
52 << "extent:" << extent << " is not constant.\n";
53 rvar_bounds.push_back(*extent_int);
54 }
55 return rvar_bounds;
56}

87template<typename FuncOrStage>
88void parallelize_vars_and_rvars_gpu(
89 const MachineParams &params,
90 FuncOrStage func_or_stage,
91 bool is_pure_def,
92 const std::vector<Var> &vars,
93 const std::vector<int> &var_bounds,
94 const std::vector<RVar> &rvars,
95 const std::vector<int> &rvar_bounds,
96 TailStrategy tail,
97 std::ostringstream &schedule_source) {
98 // Find the first variable that has bounds larger or equal than 64,
99 // this is our GPU thread.
100 // We use 64 since it's twice the warp size, so this launches enough
101 // GPU threads for a block to be work efficient.
102 constexpr int warp_size = 32;
103 constexpr int split_size = 2 * warp_size;
104 std::vector<Var> gpu_blocks;
105 std::string gpu_threads;
106 int gpu_thread_dim = -1;
107 for (int i = 0; i < (int)vars.size(); i++) {
108 if (gpu_threads.empty() && var_bounds[i] >= split_size) {
109 gpu_thread_dim = i;
110 Var outer, inner;
111 func_or_stage.split(vars[i],
112 outer,
113 inner,
114 split_size,
115 tail);
116 schedule_source << " .split("
117 << vars[i].name() << ","
118 << outer.name() << ","
119 << inner.name() << ","
120 << split_size << ","
121 << tail << ")\n";
122 gpu_blocks.push_back(outer);
123 gpu_threads = inner.name();
124 } else {
125 gpu_blocks.push_back(vars[i]);
126 }
127 }
128
129 std::vector<RVar> serial_rvars;
130 std::vector<RVar> r_gpu_blocks;
131 std::string r_gpu_threads;
132 if (gpu_threads.empty()) {
133 // If we can't find any GPU threads, parallelize RVars to find more parallelism
134 for (int i = 0; i < (int)rvars.size(); i++) {
135 if (!r_gpu_threads.empty() && rvar_bounds[i] > split_size) {
136 RVar outer, inner;
137 func_or_stage.split(rvars[i],
138 outer,
139 inner,
140 split_size,
141 tail);
142 schedule_source << " .split("
143 << rvars[i].name() << ","
144 << outer.name() << ","
145 << inner.name() << ","
146 << split_size << ","
147 << tail << ")\n";
148 r_gpu_blocks.push_back(outer);
149 r_gpu_threads = inner.name();
150 } else {
151 r_gpu_blocks.push_back(rvars[i]);
152 }
153 }
154 } else {
155 serial_rvars = rvars;
156 }
157
158 if (gpu_threads.empty() && r_gpu_threads.empty()) {
159 // If we didn't assign any GPU threads in the previous
160 // process, use the largest loop as the GPU thread.
161 int loop_size = 0;
162 int largest_loop_id = -1;
163 for (int i = 0; i < (int)vars.size(); i++) {
164 if (var_bounds[i] > loop_size) {
165 loop_size = var_bounds[i];
166 largest_loop_id = i;
167 }
168 }
169 for (int i = 0; i < (int)rvars.size(); i++) {
170 if (rvar_bounds[i] > loop_size) {
171 loop_size = rvar_bounds[i];
172 largest_loop_id = i + (int)vars.size();
173 }
174 }
175 if (largest_loop_id != -1) {
176 if (largest_loop_id < (int)vars.size()) {
177 // The largest loop is a pure variable
178 const Var &v = vars[largest_loop_id];
179 Var inner;
180 func_or_stage.split(v,
181 v,
182 inner,
183 warp_size,
184 TailStrategy::GuardWithIf);
185 schedule_source << " .split("
186 << v.name() << ","
187 << v.name() << ","
188 << inner.name() << ","
189 << warp_size << ","
190 << TailStrategy::GuardWithIf << ")\n";
191 gpu_threads = inner.name();
192 } else {
193 // The largest loop is a reduction variable
194 const RVar &v = rvars[largest_loop_id - (int)vars.size()];
195 RVar inner;
196 func_or_stage.split(v,
197 v,
198 inner,
199 warp_size,
200 TailStrategy::GuardWithIf);
201 schedule_source << " .split("
202 << v.name() << ","
203 << v.name() << ","
204 << inner.name() << ","
205 << warp_size << ","
206 << TailStrategy::GuardWithIf << ")\n";
207 r_gpu_threads = inner.name();
208 }
209 }
210 }
211 // Fuse all gpu blocks into a single variable
212 std::string fused_var;
213 if (!gpu_blocks.empty()) {
214 fused_var = gpu_blocks[0].name();
215 // inner to outer
216 for (int i = 1; i < (int)gpu_blocks.size(); i++) {
217 func_or_stage.fuse(Var(fused_var), gpu_blocks[i], Var(fused_var));
218 schedule_source << " .fuse("
219 << fused_var << ","
220 << gpu_blocks[i].name() << ","
221 << fused_var << ")\n";
222 }
223 }
224 std::string fused_rvar;
225 if (!r_gpu_blocks.empty()) {
226 fused_rvar = r_gpu_blocks[0].name();
227 // inner to outer
228 for (int i = 1; i < (int)r_gpu_blocks.size(); i++) {
229 func_or_stage.fuse(RVar(fused_rvar), r_gpu_blocks[i], RVar(fused_rvar));
230 schedule_source << " .fuse("
231 << fused_rvar << ","
232 << r_gpu_blocks[i].name() << ","
233 << fused_rvar << ")\n";
234 }
235 }
236 // CUDA places rather strict restriction on the second dimension of the GPU blocks (usually 65536),
237 // so we want to split it if it is too large
238 int rdomain_size = 1;
239 for (int b : rvar_bounds) {
240 rdomain_size *= b;
241 }
242 std::string fused_rvar2;
243 // CUDA supports up to 65536 blocks in the second and third dimensions
244 constexpr int cuda_gpu_block_split = 65536;
245 if (rdomain_size >= cuda_gpu_block_split) {
246 RVar r;
247 fused_rvar2 = r.name();
248 func_or_stage.split(RVar(fused_rvar), RVar(fused_rvar), RVar(fused_rvar2), int(std::sqrt(double(rdomain_size))));
249 }
250
251 // Reorder: the order is rvars -> gpu_threads -> gpu_blocks
252 std::vector<VarOrRVar> all_vars;
253 all_vars.reserve(serial_rvars.size() + 4);
254 for (const RVar &v : serial_rvars) {
255 all_vars.emplace_back(v);
256 }
257 if (!r_gpu_threads.empty()) {
258 all_vars.emplace_back(RVar(r_gpu_threads));
259 }
260 if (!gpu_threads.empty()) {
261 all_vars.emplace_back(Var(gpu_threads));
262 }
263 if (!fused_var.empty()) {
264 all_vars.emplace_back(Var(fused_var));
265 }
266 if (!fused_rvar.empty()) {
267 all_vars.emplace_back(RVar(fused_rvar));
268 }
269 if (!fused_rvar2.empty()) {
270 all_vars.emplace_back(RVar(fused_rvar2));
271 }
272 // Only reorder if there's more than one variables.
273 if (all_vars.size() > 1) {
274 func_or_stage.reorder(all_vars);
275 schedule_source << " .reorder(";
276 for (int i = 0; i < (int)all_vars.size(); i++) {
277 schedule_source << all_vars[i].name();
278 if (i != (int)all_vars.size() - 1) {
279 schedule_source << ",";
280 }
281 }
282 schedule_source << ")\n";
283 if (is_pure_def) {
284 if (gpu_thread_dim > 0) {
285 std::vector<Var> reordered_vars = vars;
286 std::swap(reordered_vars[0], reordered_vars[gpu_thread_dim]);
287 reorder_storage(func_or_stage, reordered_vars, schedule_source);
288 }
289 }
290 }
291
292 if (!gpu_blocks.empty() || !r_gpu_blocks.empty()) {
293 // Assign outer loops to GPU blocks
294 if (!fused_var.empty()) {
295 func_or_stage.gpu_blocks(Var(fused_var));
296 schedule_source << " .gpu_blocks(" << fused_var << ")\n";
297 }
298 if (!fused_rvar.empty()) {
299 func_or_stage.atomic()
300 .gpu_blocks(RVar(fused_rvar));
301 schedule_source << " .atomic()\n";
302 schedule_source << " .gpu_blocks(" << fused_rvar << ")\n";
303 }
304 if (!fused_rvar2.empty()) {
305 internal_assert(!fused_rvar.empty());
306 func_or_stage.gpu_blocks(RVar(fused_rvar2));
307 schedule_source << " .gpu_blocks(" << fused_rvar2 << ")\n";
308 }
309 // Assign inner loops to GPU threads
310 if (!gpu_threads.empty()) {
311 func_or_stage.gpu_threads(Var(gpu_threads));
312 schedule_source << " .gpu_threads(" << gpu_threads << ")\n";
313 }
314 if (!r_gpu_threads.empty()) {
315 func_or_stage.gpu_threads(RVar(r_gpu_threads));
316 schedule_source << " .r_gpu_threads(" << r_gpu_threads << ")\n";
317 }
318 } else {
319 // Not enough parallelism, use a single GPU thread
320 func_or_stage.gpu_single_thread();
321 schedule_source << " .gpu_single_thread()\n";
322 }
323}

325template<typename FuncOrStage>
326void parallelize_vars_and_rvars_cpu(
327 const MachineParams &params,
328 FuncOrStage func_or_stage,
329 int natural_vector_size,
330 bool is_pure_def,
331 const std::vector<Var> &vars,
332 const std::vector<int> &var_bounds,
333 const std::vector<RVar> &rvars,
334 const std::vector<int> &rvar_bounds,
335 TailStrategy tail,
336 std::ostringstream &schedule_source) {
337 // Find the first variable that has bounds larger or equal than natural_vector_size,
338 // this is our vectorized dimension
339 const int split_size = natural_vector_size;
340 std::vector<Var> parallel_vars;
341 std::string vectorized_var;
342 int num_threads_var = 1;
343 int vectorized_dim = -1;
344 for (int i = 0; i < (int)vars.size(); i++) {
345 if (vectorized_var.empty() && var_bounds[i] >= split_size) {
346 vectorized_dim = i;
347 Var outer, inner;
348 func_or_stage.split(vars[i],
349 outer,
350 inner,
351 split_size,
352 tail);
353 schedule_source << " .split("
354 << vars[i].name() << ","
355 << outer.name() << ","
356 << inner.name() << ","
357 << split_size << ","
358 << tail << ")\n";
359 parallel_vars.push_back(outer);
360 vectorized_var = inner.name();
361 int b = var_bounds[i] / split_size;
362 if (var_bounds[i] % split_size == 0) {
363 b++;
364 }
365 num_threads_var *= b;
366 } else {
367 parallel_vars.push_back(vars[i]);
368 num_threads_var *= var_bounds[i];
369 }
370 }
371
372 // If there's not enough parallelism, find in rvars.
373 // Two cases: 1) not enough threads 2) no vectorized dimension
374 std::vector<RVar> serial_rvars;
375 std::vector<RVar> parallel_rvars;
376 std::string vectorized_rvar;
377 int num_threads_rvar = 1;
378 for (int i = 0; i < (int)rvars.size(); i++) {
379 if (vectorized_var.empty() && vectorized_rvar.empty() &&
380 rvar_bounds[i] >= split_size) {
381 RVar outer, inner;
382 func_or_stage.split(rvars[i],
383 outer,
384 inner,
385 split_size,
386 tail);
387 schedule_source << " .split("
388 << rvars[i].name() << ","
389 << outer.name() << ","
390 << inner.name() << ","
391 << split_size << ","
392 << tail << ")\n";
393 int b = rvar_bounds[i] / split_size;
394 if (rvar_bounds[i] % split_size == 0) {
395 b++;
396 }
397 if (num_threads_var * num_threads_rvar < params.parallelism) {
398 parallel_rvars.push_back(outer);
399 num_threads_rvar *= b;
400 } else {
401 serial_rvars.push_back(outer);
402 }
403 vectorized_rvar = inner.name();
404 } else {
405 if (num_threads_var * num_threads_rvar < params.parallelism) {
406 num_threads_rvar *= rvar_bounds[i];
407 parallel_rvars.push_back(rvars[i]);
408 } else {
409 serial_rvars.push_back(rvars[i]);
410 }
411 }
412 }
413
414 // Fuse all parallel vars into a single variable for parallelism
415 std::string fused_var;
416 if (!parallel_vars.empty()) {
417 fused_var = parallel_vars[0].name();
418 // inner to outer
419 for (int i = 1; i < (int)parallel_vars.size(); i++) {
420 func_or_stage.fuse(Var(fused_var), parallel_vars[i], Var(fused_var));
421 schedule_source << " .fuse("
422 << fused_var << ","
423 << parallel_vars[i].name() << ","
424 << fused_var << ")\n";
425 }
426 }
427
428 // Fuse all parallel rvars into a single variable for parallelism
429 std::string fused_rvar;
430 if (!parallel_rvars.empty()) {
431 fused_rvar = parallel_rvars[0].name();
432 // inner to outer
433 for (int i = 1; i < (int)parallel_rvars.size(); i++) {
434 func_or_stage.fuse(RVar(fused_rvar), parallel_rvars[i], RVar(fused_rvar));
435 schedule_source << " .fuse("
436 << fused_rvar << ","
437 << parallel_rvars[i].name() << ","
438 << fused_rvar << ")\n";
439 }
440 }
441
442 // Reorder: the order is serial_rvars -> vectorized_rvar/vectorized_var ->
443 // fused_rvars -> fused_vars
444 std::vector<VarOrRVar> all_vars;
445 all_vars.reserve(serial_rvars.size() + 4);
446 for (const RVar &v : serial_rvars) {
447 all_vars.emplace_back(v);
448 }
449 if (!vectorized_rvar.empty()) {
450 all_vars.emplace_back(RVar(vectorized_rvar));
451 }
452 if (!vectorized_var.empty()) {
453 all_vars.emplace_back(Var(vectorized_var));
454 }
455 if (!fused_rvar.empty()) {
456 all_vars.emplace_back(RVar(fused_rvar));
457 }
458 if (!fused_var.empty()) {
459 all_vars.emplace_back(Var(fused_var));
460 }
461 // Only reorder if there's more than one variables.
462 if (all_vars.size() > 1) {
463 func_or_stage.reorder(all_vars);
464 schedule_source << " .reorder(";
465 for (int i = 0; i < (int)all_vars.size(); i++) {
466 schedule_source << all_vars[i].name();
467 if (i != (int)all_vars.size() - 1) {
468 schedule_source << ",";
469 }
470 }
471 schedule_source << ")\n";
472 if (is_pure_def) {
473 if (vectorized_dim > 0) {
474 std::vector<Var> reordered_vars = vars;
475 std::swap(reordered_vars[0], reordered_vars[vectorized_dim]);
476 reorder_storage(func_or_stage, reordered_vars, schedule_source);
477 }
478 }
479 }
480
481 if (!fused_var.empty()) {
482 // Parallelize vars
483 if (num_threads_var > params.parallelism * 8) {
484 func_or_stage.parallel(Var(fused_var),
485 num_threads_var / (params.parallelism * 8),
486 tail);
487 schedule_source << " .parallel("
488 << fused_var << ","
489 << num_threads_var / (params.parallelism * 8) << ","
490 << tail << ")\n";
491 } else {
492 func_or_stage.parallel(Var(fused_var));
493 schedule_source << " .parallel(" << fused_var << ")\n";
494 }
495 }
496 if (!fused_rvar.empty()) {
497 // Parallelize rvars
498 if (num_threads_rvar > params.parallelism * 8) {
499 func_or_stage.atomic()
500 .parallel(RVar(fused_rvar),
501 num_threads_rvar / (params.parallelism * 8),
502 tail);
503 schedule_source << " .atomic()\n";
504 schedule_source << " .parallel("
505 << fused_rvar << ","
506 << num_threads_rvar / (params.parallelism * 8) << ","
507 << tail << ")\n";
508 } else {
509 func_or_stage.atomic()
510 .parallel(RVar(fused_rvar));
511 schedule_source << " .atomic()\n";
512 schedule_source << " .parallel("
513 << fused_rvar << ")\n";
514 }
515 }
516 if (!vectorized_var.empty()) {
517 func_or_stage.vectorize(Var(vectorized_var));
518 schedule_source << " .vectorize("
519 << vectorized_var << ")\n";
520 }
521 if (!vectorized_rvar.empty()) {
522 func_or_stage.atomic().vectorize(RVar(vectorized_rvar));
523 schedule_source << " .atomic()\n";
524 schedule_source << " .vectorize("
525 << vectorized_rvar << ")\n";
526 }
527}

568void apply_schedule(const MachineParams &params,
569 const Target &target,
570 Func func,
571 int update_id,
572 const std::vector<int> &var_bounds,
573 bool is_gpu,
574 std::ostringstream &schedule_source) {
575 if (update_id == -1) {
576 func.compute_root();
577 schedule_source << func.name() << ".compute_root()\n";
578 if (func.dimensions() > 0) {
579 parallelize_vars_and_rvars(
580 params,
581 func,
582 natural_vector_size(target, func.values()[0].type()),
583 true, // is_pure_def
584 func.args(),
585 var_bounds,
586 {}, // rvars
587 {}, // rvar_bounds
588 TailStrategy::ShiftInwards,
589 is_gpu,
590 schedule_source);
591 }
592 } else {
593 // If the pure domain is smaller than some thresholds,
594 // we try to apply rfactor to increase parallelism:
595 bool checked_associative = false;
596 bool is_associative = false;
597 int domain_size = 1;
598 for (int b : var_bounds) {
599 domain_size *= b;
600 }
601 std::vector<ReductionVariable> reduction_vars =
602 func.update(update_id).get_schedule().rvars();
603 std::vector<int> rvar_bounds = get_rvar_bounds(reduction_vars);
604 std::vector<RVar> rvars;
605 rvars.reserve(reduction_vars.size());
606 for (const ReductionVariable &r : reduction_vars) {
607 rvars.emplace_back(r.var);
608 }
609 int rdomain_size = 1;
610 for (int b : rvar_bounds) {
611 rdomain_size *= b;
612 }
613 // Define the thresholds for the pure domain.
614 // For CPU we want at least params.parallelism number of elements
615 // to launch threads. For GPU we want to launch at least 64 GPU blocks.
616 // We don't use a larger domain size for GPU since we can also use atomic
617 // to increase parallelism and atomics are faster on GPU.
618 // These numbers can be better tuned (issue 4346).
619 const int cpu_max_domain_size = 8 * params.parallelism;
620 constexpr int gpu_max_domain_size = 4096;
621 int max_domain_size = is_gpu ? gpu_max_domain_size : cpu_max_domain_size;
622 if (domain_size < max_domain_size) {
623 if (!rvars.empty()) {
624 // Check associativity
625 std::vector<Expr> values =
626 func.update_values(update_id).as_vector();
627 const auto &prover_result =
628 prove_associativity(func.name(),
629 func.update_args(update_id),
630 values);
631 // Cache the associative check for later use.
632 checked_associative = true;
633 is_associative = prover_result.associative();
634 if (is_associative) {
635 schedule_source << func.name() << ".update(" << update_id << ")\n";
636 // Generate a list of tiled RVars
637 std::vector<RVar> outer_rvars, inner_rvars;
638 std::vector<int> outer_rvar_sizes, inner_rvar_sizes;
639 for (int i = 0; i < (int)rvars.size(); i++) {
640 if (rvar_bounds[i] >= 8) {
641 // Let split_size = 8 * n where n is an integer and
642 // split_size > sqrt(rvar_bounds)
643 float target = std::sqrt(rvar_bounds[i]);
644 int split_size = int(std::ceil(target / 8.f)) * 8;
645 // Split the rvar
646 RVar outer, inner;
647 func.update(update_id)
648 .split(rvars[i], outer, inner, split_size,
649 TailStrategy::GuardWithIf);
650 schedule_source << " .split("
651 << rvars[i].name() << ","
652 << outer.name() << ","
653 << inner.name() << ","
654 << split_size << ","
655 << TailStrategy::GuardWithIf << ")\n";
656 outer_rvars.push_back(outer);
657 inner_rvars.push_back(inner);
658 int outer_size = rvar_bounds[i] % split_size == 0 ? split_size : split_size + 1;
659 outer_rvar_sizes.push_back(outer_size);
660 inner_rvar_sizes.push_back(split_size);
661 } else {
662 inner_rvars.push_back(rvars[i]);
663 inner_rvar_sizes.push_back(rvar_bounds[i]);
664 }
665 }
666 schedule_source << ";\n";
667 if (!outer_rvars.empty() && !inner_rvars.empty()) {
668 // Rfactor all the outer RVars.
669 std::vector<std::pair<RVar, Var>> preserved;
670 std::vector<Var> interim_vars;
671 preserved.reserve(outer_rvars.size());
672 interim_vars.reserve(outer_rvars.size());
673 for (const RVar &r : outer_rvars) {
674 Var v;
675 preserved.emplace_back(r, v);
676 interim_vars.push_back(v);
677 }
678
679 Var factored;
680 Func interim =
681 func.update(update_id)
682 .rfactor(preserved)
683 .compute_root();
684 schedule_source << interim.name() << " = "
685 << func.name() << ".update(" << update_id << ")\n";
686 schedule_source << " .rfactor({";
687 for (int i = 0; i < (int)preserved.size(); i++) {
688 schedule_source << "{" << preserved[i].first.name() << ","
689 << preserved[i].second.name() << "}";
690 if (i != (int)preserved.size() - 1) {
691 schedule_source << ",";
692 }
693 }
694 schedule_source << "})\n";
695 schedule_source << " .compute_root()\n";
696
697 parallelize_vars_and_rvars(
698 params,
699 interim,
700 natural_vector_size(target, interim.values()[0].type()),
701 true,
702 interim_vars,
703 outer_rvar_sizes,
704 {},
705 {},
706 TailStrategy::ShiftInwards,
707 is_gpu,
708 schedule_source);
709 schedule_source << ";\n";
710 schedule_source << interim.name() << ".update()\n";
711 parallelize_vars_and_rvars(
712 params,
713 interim.update(0),
714 natural_vector_size(target, interim.values()[0].type()),
715 false,
716 interim_vars,
717 outer_rvar_sizes,
718 inner_rvars,
719 inner_rvar_sizes,
720 TailStrategy::GuardWithIf,
721 is_gpu,
722 schedule_source);
723 // Update rvars
724 rvars = outer_rvars;
725 rvar_bounds = outer_rvar_sizes;
726 }
727 }
728 }
729 }
730 // Gather pure variables
731 std::vector<Expr> update_args = func.update_args(update_id);
732 std::vector<Var> pure_args;
733 std::vector<int> pure_arg_bounds;
734 pure_args.reserve(update_args.size());
735 pure_arg_bounds.reserve(update_args.size());
736 int parallelism = 1;
737 for (int arg_id = 0; arg_id < (int)update_args.size(); arg_id++) {
738 Expr arg = update_args[arg_id];
739 const Variable *var = arg.as<Variable>();
740 if (var != nullptr &&
741 !var->param.defined() &&
742 !var->image.defined() &&
743 !var->reduction_domain.defined()) {
744 pure_args.emplace_back(var->name);
745 pure_arg_bounds.push_back(var_bounds[arg_id]);
746 parallelism *= pure_arg_bounds.back();
747 }
748 }
749 // For CPU we want at least (8 * cores) * 16 parallelism
750 // for vectorization + threading.
751 // For GPU we want at least 10 * (num SMs) * 32 parallelism
752 // Turing has ~70 SMs
753 // These numbers can be better tuned (issue 4346).
754 int cpu_min_parallelism = 8 * params.parallelism * 16;
755 int gpu_min_parallelism = 10 * 70 * 32;
756 int min_parallelism =
757 is_gpu ? gpu_min_parallelism : cpu_min_parallelism;
758 if (parallelism >= min_parallelism) {
759 schedule_source << func.name() << ".update(" << update_id << ")\n";
760 parallelize_vars_and_rvars(
761 params,
762 func.update(update_id),
763 natural_vector_size(target, func.values()[0].type()),
764 false, // is_pure_def
765 pure_args,
766 pure_arg_bounds,
767 {}, // rvars
768 {}, // rvar_bounds
769 TailStrategy::GuardWithIf,
770 is_gpu,
771 schedule_source);
772 } else {
773 // Not enough parallelism. Find parallelism from RDoms.
774 if (!checked_associative) {
775 // We can only do this for associative RDoms.
776 std::vector<Expr> values =
777 func.update_values(update_id).as_vector();
778 const auto &prover_result =
779 prove_associativity(func.name(),
780 func.update_args(update_id),
781 values);
782 checked_associative = true;
783 is_associative = prover_result.associative();
784 }
785 if (is_associative) {
786 schedule_source << func.name() << ".update(" << update_id << ")\n";
787 parallelize_vars_and_rvars(
788 params,
789 func.update(update_id),
790 natural_vector_size(target, func.values()[0].type()),
791 false, // is_pure_def
792 pure_args,
793 pure_arg_bounds,
794 rvars,
795 rvar_bounds,
796 TailStrategy::GuardWithIf,
797 is_gpu,
798 schedule_source);
799 } else {
800 // Fall back to pure var parallelization
801 schedule_source << func.name() << ".update(" << update_id << ")\n";
802 parallelize_vars_and_rvars(
803 params,
804 func.update(update_id),
805 natural_vector_size(target, func.values()[0].type()),
806 false, // is_pure_def
807 pure_args,
808 pure_arg_bounds,
809 {}, // rvars
810 {}, // rvar_bounds
811 TailStrategy::GuardWithIf,
812 is_gpu,
813 schedule_source);
814 }
815 }
816 }
817 schedule_source << ";\n";
818}

822void generate_schedule(const std::vector<Function> &outputs,
823 const Target &target,
824 const MachineParams &params,
825 AutoSchedulerResults *auto_scheduler_results) {
826 // The first few steps are the same as src/AutoSchedule.cpp
827 // Make an environment map which is used throughout the auto scheduling process.
828 std::map<std::string, Function> env;
829 for (const auto &func : outputs) {
830 std::map<std::string, Function> local_env =
831 find_transitive_calls(func);
832 env.insert(local_env.begin(), local_env.end());
833 }
834
835 // Finalize all the LoopLevels
836 for (auto &iter : env) {
837 iter.second.lock_loop_levels();
838 }
839
840 // Compute the topological order
841 std::vector<std::string> top_order = topological_order(outputs, env);
842 // Run a pre-pass that inlines all trivial Funcs (i.e. the cost of
843 // computing a Func <= calling that Func).
844 // TODO: Note that the cost is estimated using heuristics based on CPU statistics
845 // so this can be bad on GPU. In particular GPU should inline more aggressively.
846 if (inline_all_trivial_functions(outputs, top_order, env)) {
847 // Recompute env map since some functions are inlined.
848 env.clear();
849 for (const Function &f : outputs) {
850 std::map<std::string, Function> more_funcs = find_transitive_calls(f);
851 env.insert(more_funcs.begin(), more_funcs.end());
852 }
853 }
854 std::vector<std::string> order =
855 realization_order(outputs, env).first;
856 // Repeatedly inline the functions that are only used by another function
857 while (inline_all_element_wise_functions(outputs, order, env)) {
858 // Recompute env map since some functions are inlined.
859 env.clear();
860 for (const Function &f : outputs) {
861 std::map<std::string, Function> more_funcs = find_transitive_calls(f);
862 env.insert(more_funcs.begin(), more_funcs.end());
863 }
864 order = realization_order(outputs, env).first;
865 }
866
867 // Bounds inference using the given estimation
868 std::vector<Box> output_bounds_expr;
869 output_bounds_expr.reserve(outputs.size());
870 for (const auto &output : outputs) {
871 const FuncSchedule &schedule = output.schedule();
872 const std::vector<Bound> &estimates = schedule.estimates();
873
874 std::vector<Interval> b;
875 b.reserve(output.args().size());
876 for (const auto &arg : output.args()) {
877 bool found = false;
878 Bound est;
879 for (int i = (int)estimates.size() - 1; i >= 0; --i) {
880 if ((estimates[i].var == arg) && estimates[i].min.defined() &&
881 estimates[i].extent.defined()) {
882 found = true;
883 est = estimates[i];
884 break;
885 }
886 }
887 user_assert(found && est.min.type().is_int() && est.extent.type().is_int())
888 << "Please provide a valid estimate for dimension "
889 << arg << " of output \"" << output.name() << "\"\n";
890 b.emplace_back(est.min, simplify(est.min + est.extent - 1));
891 }
892 output_bounds_expr.emplace_back(b);
893 }
894
895 std::map<std::string, Box> func_bounds = inference_bounds(outputs, output_bounds_expr);
896 for (const auto &it : func_bounds) {
897 const Box &bounds = it.second;
898 for (int d = 0; d < (int)bounds.size(); d++) {
899 user_assert(bounds[d].is_bounded()) << "Access to function or buffer " << it.first << " at dimension " << d << " is not bounded. "
900 << "We can only schedule bounded accesses.\n";
901 }
902 }
903
904 std::set<std::string> output_set;
905 for (const auto &output : outputs) {
906 output_set.insert(output.name());
907 }
908
909 std::ostringstream schedule_source;
910 // Traverse from the consumers to the producers
911 for (auto it = order.rbegin(); it != order.rend(); it++) {
912 Func func(env[*it]);
913 debug(1) << "[gradient_autoscheduler] Processing function:" << *it << "\n";
914 // Get the bounds in integer constant by substitute all the parameters' estimates.
915 Box bounds = func_bounds[*it];
916 std::vector<int> int_bounds = get_int_bounds(bounds);
917 // Scheduling pure definition
918 apply_schedule(params, target, func, -1, int_bounds, target.has_gpu_feature(), schedule_source);
919 // Scheduling the updates
920 for (int update_id = 0;
921 update_id < func.num_update_definitions(); update_id++) {
922 apply_schedule(params, target, func, update_id, int_bounds, target.has_gpu_feature(), schedule_source);
923 }
924 }
925
926 auto_scheduler_results->scheduler_name = "Li2018";
927 auto_scheduler_results->schedule_source = schedule_source.str();
928 debug(1) << schedule_source.str() << "\n";
929}

931struct Li2018 {
932 void operator()(const Pipeline &p, const Target &target, const MachineParams &params, AutoSchedulerResults *results) {
933 std::vector<Function> outputs;
934 for (const Func &f : p.outputs()) {
935 outputs.push_back(f.function());
936 }
937 generate_schedule(outputs, target, params, results);
938 }
939};
