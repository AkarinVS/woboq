80void LoopLevel::check_locked() const {
81 // A LoopLevel can be in one of two states:
82 // - Unlocked (the default state): An unlocked LoopLevel can be mutated freely (via the set() method),
83 // but cannot be inspected (calls to func(), var(), is_inlined(), is_root(), etc.
84 // will assert-fail). This is the only sort of LoopLevel that most user code will ever encounter.
85 // - Locked: Once a LoopLevel is locked, it can be freely inspected, but no longer mutated.
86 // Halide locks all LoopLevels during the lowering process to ensure that no user
87 // code (e.g. custom passes) can interfere with invariants.
88 user_assert(contents->locked)
89 << "Cannot inspect an unlocked LoopLevel: "
90 << contents->func_name << "." << contents->var_name
91 << "\n";
92}
93
94void LoopLevel::check_defined_and_locked() const {
95 check_defined();
96 check_locked();
97}

99void LoopLevel::set(const LoopLevel &other) {
100 // Don't check locked(), since we don't care if it's defined() or not
101 user_assert(!contents->locked)
102 << "Cannot call set() on a locked LoopLevel: "
103 << contents->func_name << "." << contents->var_name
104 << "\n";
105 contents->func_name = other.contents->func_name;
106 contents->stage_index = other.contents->stage_index;
107 contents->var_name = other.contents->var_name;
108 contents->is_rvar = other.contents->is_rvar;
109}
110
111LoopLevel &LoopLevel::lock() {
112 contents->locked = true;
113
114 // If you have an undefined LoopLevel at the point we're
115 // locking it (i.e., start of lowering), you've done something wrong,
116 // so let's give a more useful error message.
117 user_assert(defined())
118 << "There should be no undefined LoopLevels at the start of lowering. "
119 << "(Did you mean to use LoopLevel::inlined() instead of LoopLevel() ?)";
120
121 return *this;
122}

168std::string LoopLevel::to_string() const {
169 check_defined_and_locked();
170 if (contents->stage_index == -1) {
171 return contents->func_name + "." + contents->var_name;
172 } else {
173 return contents->func_name + ".s" + std::to_string(contents->stage_index) + "." + contents->var_name;
174 }
175}

177bool LoopLevel::match(const std::string &loop) const {
178 check_defined_and_locked();
179 if (contents->stage_index == -1) {
180 return Internal::starts_with(loop, contents->func_name + ".") &&
181 Internal::ends_with(loop, "." + contents->var_name);
182 } else {
183 std::string prefix = contents->func_name + ".s" + std::to_string(contents->stage_index) + ".";
184 return Internal::starts_with(loop, prefix) &&
185 Internal::ends_with(loop, "." + contents->var_name);
186 }
187}
188
189bool LoopLevel::match(const LoopLevel &other) const {
190 check_defined_and_locked();
191 other.check_defined_and_locked();
192 return (contents->func_name == other.contents->func_name &&
193 (contents->var_name == other.contents->var_name ||
194 Internal::ends_with(contents->var_name, "." + other.contents->var_name) ||
195 Internal::ends_with(other.contents->var_name, "." + contents->var_name)) &&
196 (contents->stage_index == other.contents->stage_index));
197}

199bool LoopLevel::operator==(const LoopLevel &other) const {
200 check_defined_and_locked();
201 other.check_defined_and_locked();
202 return (contents->func_name == other.contents->func_name) &&
203 (contents->stage_index == other.contents->stage_index) &&
204 (contents->var_name == other.contents->var_name);
205}

211/** A schedule for a halide function, which defines where, when, and
212 * how it should be evaluated. */
213struct FuncScheduleContents {
214 mutable RefCount ref_count;
215
216 LoopLevel store_level, compute_level;
217 std::vector<StorageDim> storage_dims;
218 std::vector<Bound> bounds;
219 std::vector<Bound> estimates;
220 std::map<std::string, Internal::FunctionPtr> wrappers;
221 MemoryType memory_type = MemoryType::Auto;
222 bool memoized = false;
223 bool async = false;
224 Expr memoize_eviction_key;
225
226 FuncScheduleContents()
227 : store_level(LoopLevel::inlined()), compute_level(LoopLevel::inlined()) {
228 }
229
230 // Pass an IRMutator through to all Exprs referenced in the FuncScheduleContents
231 void mutate(IRMutator *mutator) {
232 for (Bound &b : bounds) {
233 if (b.min.defined()) {
234 b.min = mutator->mutate(b.min);
235 }
236 if (b.extent.defined()) {
237 b.extent = mutator->mutate(b.extent);
238 }
239 if (b.modulus.defined()) {
240 b.modulus = mutator->mutate(b.modulus);
241 }
242 if (b.remainder.defined()) {
243 b.remainder = mutator->mutate(b.remainder);
244 }
245 }
246 for (Bound &b : estimates) {
247 if (b.min.defined()) {
248 b.min = mutator->mutate(b.min);
249 }
250 if (b.extent.defined()) {
251 b.extent = mutator->mutate(b.extent);
252 }
253 if (b.modulus.defined()) {
254 b.modulus = mutator->mutate(b.modulus);
255 }
256 if (b.remainder.defined()) {
257 b.remainder = mutator->mutate(b.remainder);
258 }
259 }
260 }
261};

330FuncSchedule FuncSchedule::deep_copy(
331 std::map<FunctionPtr, FunctionPtr> &copied_map) const {
332
333 internal_assert(contents.defined()) << "Cannot deep-copy undefined FuncSchedule\n";
334 FuncSchedule copy;
335 copy.contents->store_level = contents->store_level;
336 copy.contents->compute_level = contents->compute_level;
337 copy.contents->storage_dims = contents->storage_dims;
338 copy.contents->bounds = contents->bounds;
339 copy.contents->estimates = contents->estimates;
340 copy.contents->memory_type = contents->memory_type;
341 copy.contents->memoized = contents->memoized;
342 copy.contents->memoize_eviction_key = contents->memoize_eviction_key;
343 copy.contents->async = contents->async;
344
345 // Deep-copy wrapper functions.
346 for (const auto &iter : contents->wrappers) {
347 FunctionPtr &copied_func = copied_map[iter.second];
348 internal_assert(copied_func.defined()) << Function(iter.second).name() << "\n";
349 copy.contents->wrappers[iter.first] = copied_func;
350 }
351 internal_assert(copy.contents->wrappers.size() == contents->wrappers.size());
352 return copy;
353}

449 for (const Bound &b : bounds()) {
450 if (b.min.defined()) {
451 b.min.accept(visitor);
452 }
453 if (b.extent.defined()) {

459 if (b.remainder.defined()) {

462 }

476 }
