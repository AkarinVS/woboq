6Expr Simplify::visit(const Cast *op, ExprInfo *bounds) {
7 Expr value = mutate(op->value, bounds);
8
9 if (bounds) {
10 if (bounds->min_defined && !op->type.can_represent(bounds->min)) {
11 bounds->min_defined = false;
12 if (!no_overflow(op->type)) {
13 // If the type overflows, this invalidates the max too.
14 bounds->max_defined = false;
15 }
16 }
17 if (bounds->max_defined && !op->type.can_represent(bounds->max)) {
18 if (!no_overflow(op->type)) {
19 bounds->min_defined = false;
20 }
21 bounds->max_defined = false;
22 }
23 if (!op->type.can_represent(bounds->alignment.modulus) ||
24 !op->type.can_represent(bounds->alignment.remainder)) {
25 bounds->alignment = ModulusRemainder();
26 }
27 }
28
29 if (may_simplify(op->type) && may_simplify(op->value.type())) {
30 const Cast *cast = value.as<Cast>();
31 const Broadcast *broadcast_value = value.as<Broadcast>();
32 const Ramp *ramp_value = value.as<Ramp>();
33 double f = 0.0;
34 int64_t i = 0;
35 uint64_t u = 0;
36 if (Call::as_intrinsic(value, {Call::signed_integer_overflow})) {
37 return make_signed_integer_overflow(op->type);
38 } else if (value.type() == op->type) {
39 return value;
40 } else if (op->type.is_int() &&
41 const_float(value, &f) &&
42 std::isfinite(f)) {
43 // float -> int
44 // Recursively call mutate just to set the bounds
45 return mutate(make_const(op->type, safe_numeric_cast<int64_t>(f)), bounds);
46 } else if (op->type.is_uint() &&
47 const_float(value, &f) &&
48 std::isfinite(f)) {
49 // float -> uint
50 return make_const(op->type, safe_numeric_cast<uint64_t>(f));
51 } else if (op->type.is_float() &&
52 const_float(value, &f)) {
53 // float -> float
54 return make_const(op->type, f);
55 } else if (op->type.is_int() &&
56 const_int(value, &i)) {
57 // int -> int
58 // Recursively call mutate just to set the bounds
59 return mutate(make_const(op->type, i), bounds);
60 } else if (op->type.is_uint() &&
61 const_int(value, &i)) {
62 // int -> uint
63 return make_const(op->type, safe_numeric_cast<uint64_t>(i));
64 } else if (op->type.is_float() &&
65 const_int(value, &i)) {
66 // int -> float
67 return make_const(op->type, safe_numeric_cast<double>(i));
68 } else if (op->type.is_int() &&
69 const_uint(value, &u) &&
70 op->type.bits() < value.type().bits()) {
71 // uint -> int
72 // Recursively call mutate just to set the bounds
73 return mutate(make_const(op->type, safe_numeric_cast<int64_t>(u)), bounds);
74 } else if (op->type.is_int() &&
75 const_uint(value, &u) &&
76 op->type.bits() >= value.type().bits()) {
77 // uint -> int with less than or equal to the number of bits
78 if (op->type.can_represent(u)) {
79 // Recursively call mutate just to set the bounds
80 return mutate(make_const(op->type, safe_numeric_cast<int64_t>(u)), bounds);
81 } else {
82 return make_signed_integer_overflow(op->type);
83 }
84 } else if (op->type.is_uint() &&
85 const_uint(value, &u)) {
86 // uint -> uint
87 return make_const(op->type, u);
88 } else if (op->type.is_float() &&
89 const_uint(value, &u)) {
90 // uint -> float
91 return make_const(op->type, safe_numeric_cast<double>(u));
92 } else if (cast &&
93 op->type.code() == cast->type.code() &&
94 op->type.bits() < cast->type.bits()) {
95 // If this is a cast of a cast of the same type, where the
96 // outer cast is narrower, the inner cast can be
97 // eliminated.
98 return mutate(Cast::make(op->type, cast->value), bounds);
99 } else if (cast &&
100 (op->type.is_int() || op->type.is_uint()) &&
101 (cast->type.is_int() || cast->type.is_uint()) &&
102 op->type.bits() <= cast->type.bits() &&
103 op->type.bits() <= op->value.type().bits()) {
104 // If this is a cast between integer types, where the
105 // outer cast is narrower than the inner cast and the
106 // inner cast's argument, the inner cast can be
107 // eliminated. The inner cast is either a sign extend
108 // or a zero extend, and the outer cast truncates the extended bits
109 return mutate(Cast::make(op->type, cast->value), bounds);
110 } else if (broadcast_value) {
111 // cast(broadcast(x)) -> broadcast(cast(x))
112 return mutate(Broadcast::make(Cast::make(op->type.with_lanes(broadcast_value->value.type().lanes()), broadcast_value->value), broadcast_value->lanes), bounds);
113 } else if (ramp_value &&
114 op->type.element_of() == Int(64) &&
115 op->value.type().element_of() == Int(32)) {
116 // cast(ramp(a, b, w)) -> ramp(cast(a), cast(b), w)
117 return mutate(Ramp::make(Cast::make(op->type.with_lanes(ramp_value->base.type().lanes()),
118 ramp_value->base),
119 Cast::make(op->type.with_lanes(ramp_value->stride.type().lanes()),
120 ramp_value->stride),
121 ramp_value->lanes),
122 bounds);
123 }
124 }
125
126 if (value.same_as(op->value)) {
127 return op;
128 } else {
129 return Cast::make(op->type, value);
130 }
131}
