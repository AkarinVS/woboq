33 Expr mutate_binary_operator(const T *op) {
34 Expr a = mutate(op->a);
35 if (!a.defined()) {
36 return Expr();
37 }
38 Expr b = mutate(op->b);
39 if (!b.defined()) {
40 return Expr();
41 }
42 if (a.same_as(op->a) &&
43 b.same_as(op->b)) {
44 return op;

46 return T::make(std::move(a), std::move(b));

120 Expr visit(const Select *op) override {
121 Expr cond = mutate(op->condition);
122 Expr t = mutate(op->true_value);
123 Expr f = mutate(op->false_value);
124
125 if (!cond.defined()) {
126 return Expr();
127 }
128
129 if (!t.defined() && !f.defined()) {
130 return Expr();
131 }
132
133 if (!t.defined()) {
134 // Swap the cases so that we only need to deal with the
135 // case when false is not defined below.
136 cond = Not::make(cond);
137 t = f;
138 f = Expr();
139 }
140
141 if (!f.defined()) {
142 // We need to convert this to an if-then-else
143 if (predicate.defined()) {
144 predicate = predicate && cond;
145 } else {
146 predicate = cond;
147 }
148 return t;
149 } else if (cond.same_as(op->condition) &&
150 t.same_as(op->true_value) &&
151 f.same_as(op->false_value)) {
152 return op;
153 } else {
154 return Select::make(cond, t, f);
155 }
156 }

203 Expr visit(const Call *op) override {
204 if (op->is_intrinsic(Call::undef)) {
205 return Expr();
206 }
207
208 vector<Expr> new_args(op->args.size());
209 bool changed = false;
210
211 // Mutate the args
212 for (size_t i = 0; i < op->args.size(); i++) {
213 Expr old_arg = op->args[i];
214 Expr new_arg = mutate(old_arg);
215 if (!new_arg.defined()) {
216 return Expr();
217 }
218 if (!new_arg.same_as(old_arg)) {
219 changed = true;
220 }
221 new_args[i] = new_arg;
222 }
223
224 if (!changed) {
225 return op;
226 } else {
227 return Call::make(op->type, op->name, new_args, op->call_type,
228 op->func, op->value_index, op->image, op->param);
229 }

232 template<typename T, typename Body>
233 Body visit_let(const T *op) {
234 // Visit an entire chain of lets in a single method to conserve stack space.
235 struct Frame {
236 const T *op;
237 Expr new_value;
238 ScopedBinding<> binding;
239 Frame(const T *op, Expr v, Scope<> &scope)
240 : op(op), new_value(std::move(v)),
241 binding(!new_value.defined(), scope, op->name) {
242 }
243 };
244 vector<Frame> frames;
245
246 Body result;
247 do {
248 frames.emplace_back(op, mutate(op->value), dead_vars);
249 result = op->body;
250 } while ((op = result.template as<T>()));
251
252 result = mutate(result);
253
254 if (result.defined()) {
255 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
256 if (!it->new_value.defined()) {
257 continue;
258 }
259 predicate = substitute(it->op->name, it->new_value, predicate);
260 if (it->new_value.same_as(it->op->value) && result.same_as(it->op->body)) {
261 result = it->op;
262 } else {
263 result = T::make(it->op->name, std::move(it->new_value), result);
264 }
265 }
266 }
267
268 return result;
269 }

332 predicate = Expr();
333
334 Expr pred = mutate(op->predicate);
335 Expr value = mutate(op->value);
336 if (!value.defined()) {
337 return Stmt();
338 }
339
340 Expr index = mutate(op->index);

342 return Stmt();

344
345 if (predicate.defined()) {
346 // This becomes a conditional store
347 Stmt stmt = IfThenElse::make(predicate, Store::make(op->name, value, index, op->param, pred, op->alignment));

359 Stmt visit(const Provide *op) override {
360 predicate = Expr();
361
362 vector<Expr> new_args(op->args.size());
363 vector<Expr> new_values(op->values.size());
364 vector<Expr> args_predicates;
365 vector<Expr> values_predicates;
366 bool changed = false;
367
368 // Mutate the args
369 for (size_t i = 0; i < op->args.size(); i++) {
370 Expr old_arg = op->args[i];
371 predicate = Expr();
372 Expr new_arg = mutate(old_arg);
373 if (!new_arg.defined()) {
374 return Stmt();
375 }
376 args_predicates.push_back(predicate);
377 if (!new_arg.same_as(old_arg)) {
378 changed = true;
379 }
380 new_args[i] = new_arg;
381 }
382
383 for (size_t i = 1; i < args_predicates.size(); i++) {
384 user_assert(equal(args_predicates[i - 1], args_predicates[i]))
385 << "Conditionally-undef args in a Tuple should have the same conditions\n"
386 << " Condition " << i - 1 << ": " << args_predicates[i - 1] << "\n"
387 << " Condition " << i << ": " << args_predicates[i] << "\n";
388 }
389
390 bool all_values_undefined = true;
391 for (size_t i = 0; i < op->values.size(); i++) {
392 Expr old_value = op->values[i];
393 predicate = Expr();
394 Expr new_value = mutate(old_value);
395 if (!new_value.defined()) {
396 new_value = undef(old_value.type());
397 } else {
398 all_values_undefined = false;
399 values_predicates.push_back(predicate);
400 }
401 if (!new_value.same_as(old_value)) {
402 changed = true;
403 }
404 new_values[i] = new_value;
405 }
406
407 if (all_values_undefined) {
408 return Stmt();
409 }
410
411 for (size_t i = 1; i < values_predicates.size(); i++) {
412 user_assert(equal(values_predicates[i - 1], values_predicates[i]))
413 << "Conditionally-undef values in a Tuple should have the same conditions\n"
414 << " Condition " << i - 1 << ": " << values_predicates[i - 1] << "\n"
415 << " Condition " << i << ": " << values_predicates[i] << "\n";
416 }
417
418 if (predicate.defined()) {
419 Stmt stmt = IfThenElse::make(predicate, Provide::make(op->name, new_values, new_args));
420 predicate = Expr();
421 return stmt;
422 } else if (!changed) {
423 return op;
424 } else {
425 return Provide::make(op->name, new_values, new_args);
426 }
427 }

429 Stmt visit(const Allocate *op) override {
430 std::vector<Expr> new_extents;
431 bool all_extents_unmodified = true;
432 for (size_t i = 0; i < op->extents.size(); i++) {
433 new_extents.push_back(mutate(op->extents[i]));
434 if (!new_extents.back().defined()) {
435 return Stmt();
436 }
437 all_extents_unmodified &= new_extents[i].same_as(op->extents[i]);
438 }
439 Stmt body = mutate(op->body);
440 if (!body.defined()) {
441 return Stmt();
442 }
443
444 Expr condition = mutate(op->condition);
445 if (!condition.defined()) {
446 return Stmt();
447 }
448
449 Expr new_expr;
450 if (op->new_expr.defined()) {
451 new_expr = mutate(op->new_expr);
452 }
453
454 if (all_extents_unmodified &&
455 body.same_as(op->body) &&
456 condition.same_as(op->condition) &&
457 new_expr.same_as(op->new_expr)) {
458 return op;
459 } else {
460 return Allocate::make(op->name, op->type, op->memory_type,
461 new_extents, condition, body, new_expr, op->free_function);
462 }

469 Stmt visit(const Realize *op) override {
470 Region new_bounds(op->bounds.size());
471 bool bounds_changed = false;
472
473 // Mutate the bounds
474 for (size_t i = 0; i < op->bounds.size(); i++) {
475 Expr old_min = op->bounds[i].min;
476 Expr old_extent = op->bounds[i].extent;
477 Expr new_min = mutate(old_min);
478 if (!new_min.defined()) {
479 return Stmt();
480 }
481 Expr new_extent = mutate(old_extent);
482 if (!new_extent.defined()) {
483 return Stmt();
484 }
485 if (!new_min.same_as(old_min)) {
486 bounds_changed = true;
487 }
488 if (!new_extent.same_as(old_extent)) {
489 bounds_changed = true;
490 }
491 new_bounds[i] = Range(new_min, new_extent);
492 }
493
494 Stmt body = mutate(op->body);
495 if (!body.defined()) {
496 return Stmt();
497 }
498
499 Expr condition = mutate(op->condition);
500 if (!condition.defined()) {
501 return Stmt();
502 }
503
504 if (!bounds_changed &&
505 body.same_as(op->body) &&
506 condition.same_as(op->condition)) {
507 return op;
508 } else {
509 return Realize::make(op->name, op->types, op->memory_type, new_bounds, condition, body);
510 }

513 Stmt visit(const Block *op) override {
514 // Visit a sequence of blocks in a single method to conserve stack space.
515 Stmt result;
516 vector<std::pair<const Block *, Stmt>> frames;
517
518 do {
519 Stmt next = mutate(op->first);
520 if (next.defined()) {
521 frames.emplace_back(op, std::move(next));
522 }
523 result = op->rest;
524 } while ((op = result.as<Block>()));
525
526 result = mutate(result);
527
528 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
529 op = it->first;
530 Stmt new_first = std::move(it->second);
531 if (!result.defined()) {
532 result = new_first;
533 } else if (new_first.same_as(op->first) && result.same_as(op->rest)) {
534 result = op;
535 } else {
536 result = Block::make(new_first, result);
537 }
538 }
539 return result;
540 }

542 Stmt visit(const IfThenElse *op) override {
543 Expr condition = mutate(op->condition);
544 if (!condition.defined()) {
545 return Stmt();
546 }
547 Stmt then_case = mutate(op->then_case);
548 Stmt else_case = mutate(op->else_case);
549
550 if (!then_case.defined() && !else_case.defined()) {
551 return Stmt();
552 }
553
554 if (!then_case.defined()) {
555 condition = Not::make(condition);
556 then_case = else_case;
557 else_case = Stmt();
558 }
559
560 if (condition.same_as(op->condition) &&
561 then_case.same_as(op->then_case) &&
562 else_case.same_as(op->else_case)) {
563 return op;
564 } else {
565 return IfThenElse::make(condition, then_case, else_case);
566 }
567 }

583Stmt remove_undef(Stmt s) {
584 RemoveUndef r;
585 s = r.mutate(s);
586 internal_assert(!r.predicate.defined())
587 << "Undefined expression leaked outside of a Store node: "
588 << r.predicate << "\n";
589 return s;
590}
