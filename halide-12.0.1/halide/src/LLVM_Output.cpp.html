30// This is a bare-bones Windows .lib file writer, based on inspection
31// of the LLVM ArchiveWriter class and the documentation at
32// https://www.microsoft.com/msj/0498/hood0498.aspx and
33// https://msdn.microsoft.com/en-us/library/windows/desktop/ms680547(v=vs.85).aspx#archive__library__file_format
34//
35// It has been compared with the output of VS2015's lib.exe and appears to be
36// bit-identical (to meaningful bits, anyway) for a sampling of Halide
37// AOT output, but it is quite possible that there are omissions, mistakes,
38// or just plain bugs.
39
40// Emit a field that is 'size' characters wide.
41// If data too small, pad on the right with spaces.
42// If data too large, assert.
43// Return the offset at which 'data' was written.
44template<typename T>
45size_t emit_padded(std::ostream &out, T data, size_t size) {
46 size_t pos = out.tellp();
47 out << data;
48 size_t written = (size_t)out.tellp() - pos;
49 internal_assert(written <= size);
50 while (written < size) {
51 out.put(' ');
52 written++;
53 }
54 return pos;
55}

57using EmitU32 = std::function<void(std::ostream &, uint32_t)>;
58
59void emit_big_endian_u32(std::ostream &out, uint32_t value) {
60 out << static_cast<uint8_t>((value >> 24) & 0xff)
61 << static_cast<uint8_t>((value >> 16) & 0xff)
62 << static_cast<uint8_t>((value >> 8) & 0xff)
63 << static_cast<uint8_t>((value)&0xff);
64}
65
66void emit_little_endian_u32(std::ostream &out, uint32_t value) {
67 out << static_cast<uint8_t>((value)&0xff)
68 << static_cast<uint8_t>((value >> 8) & 0xff)
69 << static_cast<uint8_t>((value >> 16) & 0xff)
70 << static_cast<uint8_t>((value >> 24) & 0xff);
71}
72
73void emit_little_endian_u16(std::ostream &out, uint16_t value) {
74 out << static_cast<uint8_t>((value)&0xff)
75 << static_cast<uint8_t>((value >> 8) & 0xff);
76}

78// Return the offset at which 'size' was written
79size_t finish_member_header(std::ostream &out, size_t size) {
80 // Emit zero for all of these, to mimic the 'deterministic' flag
81 emit_padded(out, 0, 12); // timestamp
82 emit_padded(out, ' ', 6); // UID
83 emit_padded(out, ' ', 6); // GID
84 emit_padded(out, 0, 8); // perm
85 const size_t pos = emit_padded(out, size, 10); // total size of the archive member (not including header)
86 out << "\x60\x0A";
87 return pos;
88}

94std::map<std::string, size_t> write_string_table(std::ostream &out,
95 const std::vector<llvm::NewArchiveMember> &members) {
96 std::map<std::string, size_t> string_to_offset_map;
97 size_t start_offset = 0;
98 for (const llvm::NewArchiveMember &m : members) {
99 std::string name = member_name(m);
100 internal_assert(string_to_offset_map.count(name) == 0);
101 if (name.size() < 16 && name.find('/') == std::string::npos) {
102 // small strings that don't contain '/' can be inlined
103 continue;
104 }
105 if (start_offset == 0) {
106 emit_padded(out, "//", 16);
107 finish_member_header(out, 0);
108 start_offset = out.tellp();
109 }
110 string_to_offset_map[name] = (size_t)out.tellp() - start_offset;
111 out << name;
112 out.put('\0');
113 }
114 // If all strings are short enough, we skip the string table entirely
115 if (start_offset != 0) {
116 size_t member_end = out.tellp();
117 if (out.tellp() % 2) {
118 out.put('\x0A');
119 }
120 size_t final_offset = out.tellp();
121 out.seekp(start_offset - 12);
122 emit_padded(out, member_end - start_offset, 10);
123 out.seekp(final_offset);
124 }
125 return string_to_offset_map;
126}

133void write_symbol_table(std::ostream &out,
134 const std::vector<llvm::NewArchiveMember> &members,
135 bool windows_coff_format,
136 std::map<size_t, std::vector<PatchInfo>> *patchers) {
137 internal_assert(!members.empty());
138
139 EmitU32 emit_u32 = windows_coff_format ? emit_little_endian_u32 : emit_big_endian_u32;
140
141 // Write zero for sizes/offsets that will be patched later.
142 const size_t kPatchLater = 0;
143
144 std::map<std::string, size_t> name_to_member_index;
145
146 const auto kFileMagicUnknown = llvm::file_magic::unknown;
147
148 llvm::LLVMContext context;
149 for (size_t i = 0, n = members.size(); i < n; ++i) {
150 llvm::MemoryBufferRef member_buffer = members[i].Buf->getMemBufferRef();
151 llvm::Expected<std::unique_ptr<llvm::object::SymbolicFile>> obj_or_err =
152 llvm::object::SymbolicFile::createSymbolicFile(
153 member_buffer, kFileMagicUnknown, &context);
154 if (!obj_or_err) {
155 // Don't use internal_assert: the call to new_member.takeError() will be
156 // evaluated even if the assert does not fail, leaving new_member in an
157 // indeterminate state.
158 internal_error << llvm::toString(obj_or_err.takeError()) << "\n";
159 }
160 llvm::object::SymbolicFile &obj = *obj_or_err.get();
161 for (const auto &sym : obj.symbols()) {
162 auto flags = sym.getFlags();
163 if (!flags) {
164 internal_error << llvm::toString(flags.takeError()) << "\n";
165 }
166 const uint32_t sym_flags = flags.get();
167 if (sym_flags & llvm::object::SymbolRef::SF_FormatSpecific) {
168 continue;
169 }
170 if (!(sym_flags & llvm::object::SymbolRef::SF_Global)) {
171 continue;
172 }
173 if ((sym_flags & llvm::object::SymbolRef::SF_Undefined) &&
174 !(sym_flags & llvm::object::SymbolRef::SF_Indirect)) {
175 continue;
176 }
177 // Windows COFF doesn't support weak symbols.
178 if (sym_flags & llvm::object::SymbolRef::SF_Weak) {
179 continue;
180 }
181
182 llvm::SmallString<128> symbols_buf;
183 llvm::raw_svector_ostream symbols(symbols_buf);
184 auto err = sym.printName(symbols);
185 internal_assert(!err);
186 std::string name = symbols.str().str();
187 if (name_to_member_index.find(name) != name_to_member_index.end()) {
188 user_warning << "Warning: symbol '" << name << "' seen multiple times in library.\n";
189 continue;
190 }
191 name_to_member_index[name] = i;
192 }
193 }
194
195 size_t header_start_offset = emit_padded(out, "/", 16);
196 size_t symbol_table_size_offset = finish_member_header(out, kPatchLater); // size of symbol table
197
198 size_t symbol_count_offset = 0;
199 if (windows_coff_format) {
200 emit_u32(out, members.size());
201 for (size_t i = 0, n = members.size(); i < n; ++i) {
202 size_t pos = out.tellp();
203 emit_u32(out, kPatchLater); // offset to this .obj member
204 (*patchers)[i].push_back({emit_u32, pos});
205 }
206 symbol_count_offset = out.tellp();
207 emit_u32(out, kPatchLater); // number of symbols
208 // symbol-to-archive-member-index, but 1-based rather than zero-based.
209 for (auto &it : name_to_member_index) {
210 internal_assert(it.second <= 65534);
211 emit_little_endian_u16(out, (uint16_t)it.second + 1);
212 }
213 } else {
214 symbol_count_offset = out.tellp();
215 emit_u32(out, kPatchLater); // number of symbols
216 for (auto &it : name_to_member_index) {
217 size_t pos = out.tellp();
218 emit_u32(out, kPatchLater); // offset to the .obj member containing this symbol
219 (*patchers)[it.second].push_back({emit_u32, pos});
220 }
221 }
222
223 // Symbol table goes at the end for both variants.
224 for (auto &it : name_to_member_index) {
225 out << it.first;
226 out.put('\0');
227 }
228
229 size_t member_end = out.tellp();
230
231 // lib.exe pads to 2-byte align with 0x0a
232 if (out.tellp() % 2) {
233 out.put('\x0A');
234 }
235 size_t final_offset = out.tellp();
236
237 // Patch the size of the symbol table.
238 const size_t member_header_size = 60;
239 out.seekp(symbol_table_size_offset);
240 emit_padded(out, member_end - member_header_size - header_start_offset, 10);
241
242 // Patch the number of symbols.
243 out.seekp(symbol_count_offset);
244 emit_u32(out, name_to_member_index.size());
245
246 // Seek back to where we left off.
247 out.seekp(final_offset);
248}

250void write_coff_archive(std::ostream &out,
251 const std::vector<llvm::NewArchiveMember> &members) {
252 out << "!<arch>\x0A";
253
254 // First member is named "/" and is the traditional symbol table,
255 // with big-endian offsets.
256 std::map<size_t, std::vector<PatchInfo>> patchers;
257 write_symbol_table(out, members, false, &patchers);
258
259 // Second member (for Windows COFF) is also named "/" and is also a symbol table,
260 // but with little-endian offsets and with symbols sorted by name. (We actually sort
261 // both tables as a side-effect, but the first isn't required to be sorted.)
262 write_symbol_table(out, members, true, &patchers);
263
264 // Third member, named "//", is the optional string table. (MS docs say it is required but
265 // lib.exe only emits as needed, so we will follow its example)
266 std::map<std::string, size_t> string_to_offset_map = write_string_table(out, members);
267
268 // The remaining members are just (header + contents of .obj file).
269 std::vector<size_t> member_offset;
270 for (const llvm::NewArchiveMember &m : members) {
271 size_t pos = out.tellp();
272 member_offset.push_back(pos);
273
274 std::string name = member_name(m);
275 auto it = string_to_offset_map.find(name);
276 if (it != string_to_offset_map.end()) {
277 out.put('/');
278 emit_padded(out, it->second, 15);
279 } else {
280 emit_padded(out, name + "/", 16);
281 }
282 size_t size = m.Buf->getBufferSize();
283 finish_member_header(out, size);
284
285 out << m.Buf->getMemBufferRef().getBuffer().str();
286
287 if (out.tellp() % 2) {
288 out.put('\x0A');
289 }
290 }
291
292 for (auto &it : patchers) {
293 size_t i = it.first;
294 for (auto &patcher : it.second) {
295 out.seekp(patcher.pos);
296 patcher.emit_u32(out, member_offset.at(i));
297 }
298 }
299}

321// llvm::CloneModule has issues with debug info. As a workaround,
322// serialize it to bitcode in memory, and then parse the bitcode back in.
323std::unique_ptr<llvm::Module> clone_module(const llvm::Module &module_in) {
324 Internal::debug(2) << "Cloning module " << module_in.getName().str() << "\n";
325
326 // Write the module to a buffer.
327 llvm::SmallVector<char, 16> clone_buffer;
328 llvm::raw_svector_ostream clone_ostream(clone_buffer);
329 WriteBitcodeToFile(module_in, clone_ostream);
330
331 // Read it back in.
332 llvm::MemoryBufferRef buffer_ref(llvm::StringRef(clone_buffer.data(), clone_buffer.size()), "clone_buffer");
333 auto cloned_module = llvm::parseBitcodeFile(buffer_ref, module_in.getContext());
334 internal_assert(cloned_module);
335
336 return std::move(cloned_module.get());
337}

341void emit_file(const llvm::Module &module_in, Internal::LLVMOStream &out,
342 llvm::CodeGenFileType file_type) {
343 Internal::debug(1) << "emit_file.Compiling to native code...\n";
344 Internal::debug(2) << "Target triple: " << module_in.getTargetTriple() << "\n";
345
346 auto time_start = std::chrono::high_resolution_clock::now();
347
348 // Work on a copy of the module to avoid modifying the original.
349 std::unique_ptr<llvm::Module> module = clone_module(module_in);
350
351 // Get the target specific parser.
352 auto target_machine = Internal::make_target_machine(*module);
353 internal_assert(target_machine.get()) << "Could not allocate target machine!\n";
354
355 llvm::DataLayout target_data_layout(target_machine->createDataLayout());
356 if (!(target_data_layout == module->getDataLayout())) {
357 internal_error << "Warning: module's data layout does not match target machine's\n"
358 << target_data_layout.getStringRepresentation() << "\n"
359 << module->getDataLayout().getStringRepresentation() << "\n";
360 }
361
362 // Build up all of the passes that we want to do to the module.
363 llvm::legacy::PassManager pass_manager;
364
365 pass_manager.add(new llvm::TargetLibraryInfoWrapperPass(llvm::Triple(module->getTargetTriple())));
366
367 // Make sure things marked as always-inline get inlined
368 pass_manager.add(llvm::createAlwaysInlinerLegacyPass());
369
370 // Remove any stale debug info
371 pass_manager.add(llvm::createStripDeadDebugInfoPass());
372
373 // Enable symbol rewriting. This allows code outside libHalide to
374 // use symbol rewriting when compiling Halide code (for example, by
375 // using cl::ParseCommandLineOption and then passing the appropriate
376 // rewrite options via -mllvm flags).
377 pass_manager.add(llvm::createRewriteSymbolsPass());
378
379 // Override default to generate verbose assembly.
380 target_machine->Options.MCOptions.AsmVerbose = true;
381
382 // Ask the target to add backend passes as necessary.
383 target_machine->addPassesToEmitFile(pass_manager, out, nullptr, file_type);
384
385 pass_manager.run(*module);
386
387 auto *logger = Internal::get_compiler_logger();
388 if (logger) {
389 auto time_end = std::chrono::high_resolution_clock::now();
390 std::chrono::duration<double> diff = time_end - time_start;
391 logger->record_compilation_time(Internal::CompilerLogger::Phase::LLVM, diff.count());
392 }
393
394 // If -time-passes is in HL_LLVM_ARGS, this will print llvm passes time statstics otherwise its no-op.
395 llvm::reportAndResetTimings();
396}

424std::string get_current_directory() {
425#ifdef _WIN32
426 DWORD dir_buf_size = GetCurrentDirectoryW(0, nullptr);
427 internal_assert(dir_buf_size) << "GetCurrentDirectoryW() failed; error " << GetLastError() << "\n";
428
429 // GetCurrentDirectoryW returns a _buffer size_, not a character count.
430 // std::wstring null-terminates on its own, so don't count that here.
431 std::wstring wdir(dir_buf_size - 1, 0);
432
433 DWORD ret = GetCurrentDirectoryW(dir_buf_size, &wdir[0]);
434 internal_assert(ret) << "GetCurrentDirectoryW() failed; error " << GetLastError() << "\n";
435
436 int dir_len = WideCharToMultiByte(CP_UTF8, 0, &wdir[0], (int)wdir.size(), nullptr, 0, nullptr, nullptr);
437 internal_assert(dir_len) << "WideCharToMultiByte() failed; error " << GetLastError() << "\n";
438
439 std::string dir(dir_len, 0);
440
441 ret = WideCharToMultiByte(CP_UTF8, 0, &wdir[0], (int)wdir.size(), &dir[0], (int)dir.size(), nullptr, nullptr);
442 internal_assert(ret) << "WideCharToMultiByte() failed; error " << GetLastError() << "\n";
443
444 return dir;
445#else
446 std::string dir;
447 // Note that passing null for the first arg isn't strictly POSIX, but is
448 // supported everywhere we currently build.
449 char *p = getcwd(nullptr, 0);
450 internal_assert(p != nullptr) << "getcwd() failed";
451 dir = p;
452 free(p);
453 return dir;
454#endif
455}

457void set_current_directory(const std::string &d) {
458#ifdef _WIN32

472std::pair<std::string, std::string> dir_and_file(const std::string &path) {
473 std::string dir, file;
474 size_t slash_pos = path.rfind('/');
475#ifdef _WIN32
476 if (slash_pos == std::string::npos) {
477 // Windows is a thing
478 slash_pos = path.rfind('\\');
479 }
480#endif
481 if (slash_pos != std::string::npos) {
482 dir = path.substr(0, slash_pos);
483 file = path.substr(slash_pos + 1);
484 } else {
485 file = path;
486 }
487 return {dir, file};
488}

490std::string make_absolute_path(const std::string &path) {
491 bool is_absolute = !path.empty() && path[0] == '/';
492 char sep = '/';
493#ifdef _WIN32
494 // Allow for C:\whatever or c:/whatever on Windows
495 if (path.size() >= 3 && path[1] == ':' && (path[2] == '\\' || path[2] == '/')) {
496 is_absolute = true;
497 sep = path[2];
498 } else if (path.size() > 2 && path[0] == '\\' && path[1] == '\\') {
499 // Also allow for UNC-style paths beginning with double-backslash
500 is_absolute = true;
501 sep = path[0];
502 }
503#endif
504 if (!is_absolute) {
505 return get_current_directory() + sep + path;
506 }
507 return path;
508}

525void create_static_library(const std::vector<std::string> &src_files_in, const Target &target,
526 const std::string &dst_file_in, bool deterministic) {
527 internal_assert(!src_files_in.empty());
528
529 // Ensure that dst_file is an absolute path, since we're going to change the
530 // working directory temporarily.
531 std::string dst_file = make_absolute_path(dst_file_in);
532
533 // If we give absolute paths to LLVM, it will dutifully embed them in the resulting
534 // .a file; some versions of 'ar x' are unable to deal with the resulting files,
535 // which is inconvenient. So let's doctor the inputs to be simple filenames,
536 // and temporarily change the working directory. (Note that this requires all the
537 // input files be in the same directory; this is currently always the case for
538 // our existing usage.)
539 std::string src_dir = dir_and_file(src_files_in.front()).first;
540 std::vector<std::string> src_files;
541 for (const auto &s_in : src_files_in) {
542 auto df = dir_and_file(s_in);
543 internal_assert(df.first == src_dir) << "All inputs to create_static_library() must be in the same directory";
544 for (auto &s_existing : src_files) {
545 internal_assert(s_existing != df.second) << "create_static_library() does not allow duplicate filenames.";
546 }
547 src_files.push_back(df.second);
548 }
549
550 SetCwd set_cwd(src_dir);
551
552 std::vector<llvm::NewArchiveMember> new_members;
553 for (auto &src : src_files) {
554 llvm::Expected<llvm::NewArchiveMember> new_member =
555 llvm::NewArchiveMember::getFile(src, /*Deterministic=*/true);
556 if (!new_member) {
557 // Don't use internal_assert: the call to new_member.takeError() will be evaluated
558 // even if the assert does not fail, leaving new_member in an indeterminate
559 // state.
560 internal_error << src << ": " << llvm::toString(new_member.takeError()) << "\n";
561 }
562 new_members.push_back(std::move(*new_member));
563 }
564
565 // LLVM can't write MS PE/COFF Lib format, which is almost-but-not-quite
566 // the same as GNU ar format.
567 if (Internal::get_triple_for_target(target).isWindowsMSVCEnvironment()) {
568 std::ofstream f(dst_file, std::ios_base::trunc | std::ios_base::binary);
569 Internal::Archive::write_coff_archive(f, new_members);
570 f.flush();
571 f.close();
572 return;
573 }
574
575 const bool write_symtab = true;
576 const auto kind = Internal::get_triple_for_target(target).isOSDarwin() ? llvm::object::Archive::K_BSD : llvm::object::Archive::K_GNU;
577 const bool thin = false;
578 auto result = llvm::writeArchive(dst_file, new_members,
579 write_symtab, kind,
580 deterministic, thin, nullptr);
581 internal_assert(!result)
582 << "Failed to write archive: " << dst_file
583 << ", reason: " << llvm::toString(std::move(result)) << "\n";
584}
