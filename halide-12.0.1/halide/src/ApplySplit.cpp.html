14vector<ApplySplitResult> apply_split(const Split &split, bool is_update, const string &prefix,
15 map<string, Expr> &dim_extent_alignment) {
16 vector<ApplySplitResult> result;
17
18 Expr outer = Variable::make(Int(32), prefix + split.outer);
19 Expr outer_max = Variable::make(Int(32), prefix + split.outer + ".loop_max");
20 if (split.is_split()) {
21 Expr inner = Variable::make(Int(32), prefix + split.inner);
22 Expr old_max = Variable::make(Int(32), prefix + split.old_var + ".loop_max");
23 Expr old_min = Variable::make(Int(32), prefix + split.old_var + ".loop_min");
24 Expr old_extent = Variable::make(Int(32), prefix + split.old_var + ".loop_extent");
25
26 dim_extent_alignment[split.inner] = split.factor;
27
28 Expr base = outer * split.factor + old_min;
29 string base_name = prefix + split.inner + ".base";
30 Expr base_var = Variable::make(Int(32), base_name);
31 string old_var_name = prefix + split.old_var;
32 Expr old_var = Variable::make(Int(32), old_var_name);
33
34 map<string, Expr>::iterator iter = dim_extent_alignment.find(split.old_var);
35
36 TailStrategy tail = split.tail;
37 internal_assert(tail != TailStrategy::Auto)
38 << "An explicit tail strategy should exist at this point\n";
39
40 if ((iter != dim_extent_alignment.end()) &&
41 is_const_zero(simplify(iter->second % split.factor))) {
42 // We have proved that the split factor divides the
43 // old extent. No need to adjust the base or add an if
44 // statement.
45 dim_extent_alignment[split.outer] = iter->second / split.factor;
46 } else if (is_negative_const(split.factor) || is_const_zero(split.factor)) {
47 user_error << "Can't split " << split.old_var << " by " << split.factor
48 << ". Split factors must be strictly positive\n";
49 } else if (is_const_one(split.factor)) {
50 // The split factor trivially divides the old extent,
51 // but we know nothing new about the outer dimension.
52 } else if (tail == TailStrategy::GuardWithIf ||
53 tail == TailStrategy::Predicate) {
54 // It's an exact split but we failed to prove that the
55 // extent divides the factor. Use predication to avoid
56 // running off the end of the original loop.
57
58 // Bounds inference has trouble exploiting an if
59 // condition. We'll directly tell it that the loop
60 // variable is bounded above by the original loop max by
61 // replacing the variable with a promise-clamped version
62 // of it. We don't also use the original loop min because
63 // it needlessly complicates the expressions and doesn't
64 // actually communicate anything new.
65 Expr guarded = promise_clamped(old_var, old_var, old_max);
66 string guarded_var_name = prefix + split.old_var + ".guarded";
67 Expr guarded_var = Variable::make(Int(32), guarded_var_name);
68 result.emplace_back(prefix + split.old_var, guarded_var, ApplySplitResult::Substitution);
69 result.emplace_back(guarded_var_name, guarded, ApplySplitResult::LetStmt);
70
71 // Inject the if condition *after* doing the substitution
72 // for the guarded version.
73 Expr cond = likely(old_var <= old_max);
74 if (tail == TailStrategy::Predicate) {
75 // Add the hint for predication.
76 cond = predicate(cond);
77 }
78 result.emplace_back(cond);
79
80 } else if (tail == TailStrategy::ShiftInwards) {
81 // Adjust the base downwards to not compute off the
82 // end of the realization.
83
84 // We'll only mark the base as likely (triggering a loop
85 // partition) if we're at or inside the innermost
86 // non-trivial loop.
87 base = likely_if_innermost(base);
88 base = Min::make(base, old_max + (1 - split.factor));
89 } else {
90 internal_assert(tail == TailStrategy::RoundUp);
91 }
92
93 // Define the original variable as the base value computed above plus the inner loop variable.
94 result.emplace_back(old_var_name, base_var + inner, ApplySplitResult::LetStmt);
95 result.emplace_back(base_name, base, ApplySplitResult::LetStmt);
96
97 } else if (split.is_fuse()) {
98 // Define the inner and outer in terms of the fused var
99 Expr fused = Variable::make(Int(32), prefix + split.old_var);
100 Expr inner_min = Variable::make(Int(32), prefix + split.inner + ".loop_min");
101 Expr outer_min = Variable::make(Int(32), prefix + split.outer + ".loop_min");
102 Expr inner_extent = Variable::make(Int(32), prefix + split.inner + ".loop_extent");
103
104 const Expr &factor = inner_extent;
105 Expr inner = fused % factor + inner_min;
106 Expr outer = fused / factor + outer_min;
107
108 result.emplace_back(prefix + split.inner, inner, ApplySplitResult::Substitution);
109 result.emplace_back(prefix + split.outer, outer, ApplySplitResult::Substitution);
110 result.emplace_back(prefix + split.inner, inner, ApplySplitResult::LetStmt);
111 result.emplace_back(prefix + split.outer, outer, ApplySplitResult::LetStmt);
112
113 // Maintain the known size of the fused dim if
114 // possible. This is important for possible later splits.
115 map<string, Expr>::iterator inner_dim = dim_extent_alignment.find(split.inner);
116 map<string, Expr>::iterator outer_dim = dim_extent_alignment.find(split.outer);
117 if (inner_dim != dim_extent_alignment.end() &&
118 outer_dim != dim_extent_alignment.end()) {
119 dim_extent_alignment[split.old_var] = inner_dim->second * outer_dim->second;
120 }
121 } else {
122 // rename or purify
123 result.emplace_back(prefix + split.old_var, outer, ApplySplitResult::Substitution);
124 result.emplace_back(prefix + split.old_var, outer, ApplySplitResult::LetStmt);
125 }
126
127 return result;
128}

130vector<std::pair<string, Expr>> compute_loop_bounds_after_split(const Split &split, const string &prefix) {
131 // Define the bounds on the split dimensions using the bounds
132 // on the function args. If it is a purify, we should use the bounds
133 // from the dims instead.
134
135 vector<std::pair<string, Expr>> let_stmts;
136
137 Expr old_var_extent = Variable::make(Int(32), prefix + split.old_var + ".loop_extent");
138 Expr old_var_max = Variable::make(Int(32), prefix + split.old_var + ".loop_max");
139 Expr old_var_min = Variable::make(Int(32), prefix + split.old_var + ".loop_min");
140 if (split.is_split()) {
141 Expr inner_extent = split.factor;
142 Expr outer_extent = (old_var_max - old_var_min + split.factor) / split.factor;
143 let_stmts.emplace_back(prefix + split.inner + ".loop_min", 0);
144 let_stmts.emplace_back(prefix + split.inner + ".loop_max", inner_extent - 1);
145 let_stmts.emplace_back(prefix + split.inner + ".loop_extent", inner_extent);
146 let_stmts.emplace_back(prefix + split.outer + ".loop_min", 0);
147 let_stmts.emplace_back(prefix + split.outer + ".loop_max", outer_extent - 1);
148 let_stmts.emplace_back(prefix + split.outer + ".loop_extent", outer_extent);
149 } else if (split.is_fuse()) {
150 // Define bounds on the fused var using the bounds on the inner and outer
151 Expr inner_extent = Variable::make(Int(32), prefix + split.inner + ".loop_extent");
152 Expr outer_extent = Variable::make(Int(32), prefix + split.outer + ".loop_extent");
153 Expr fused_extent = inner_extent * outer_extent;
154 let_stmts.emplace_back(prefix + split.old_var + ".loop_min", 0);
155 let_stmts.emplace_back(prefix + split.old_var + ".loop_max", fused_extent - 1);
156 let_stmts.emplace_back(prefix + split.old_var + ".loop_extent", fused_extent);
157 } else if (split.is_rename()) {
158 let_stmts.emplace_back(prefix + split.outer + ".loop_min", old_var_min);
159 let_stmts.emplace_back(prefix + split.outer + ".loop_max", old_var_max);
160 let_stmts.emplace_back(prefix + split.outer + ".loop_extent", old_var_extent);
161 }
162 // Do nothing for purify
163
164 return let_stmts;
165}
