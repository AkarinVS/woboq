15template<typename fn_type>
16bool lookup_runtime_routine(const std::string &name,
17 const Target &target,
18 fn_type &result) {
19 std::vector<JITModule> runtime(
20 JITSharedRuntime::get(nullptr, target.with_feature(Target::JIT)));
21
22 for (size_t i = 0; i < runtime.size(); i++) {
23 std::map<std::string, JITModule::Symbol>::const_iterator f =
24 runtime[i].exports().find(name);
25 if (f != runtime[i].exports().end()) {
26 result = reinterpret_bits<fn_type>(f->second.address);
27 return true;
28 }
29 }
30 return false;
31}

35bool host_supports_target_device(const Target &t) {
36 const DeviceAPI d = t.get_required_device_api();
37 if (d == DeviceAPI::None) {
38 // If the target requires no DeviceAPI, then
39 // the host trivially supports the target device.
40 return true;
41 }
42
43 const struct halide_device_interface_t *i = get_device_interface_for_device_api(d, t);
44 if (!i) {
45 debug(1) << "host_supports_device_api: get_device_interface_for_device_api() failed for d=" << (int)d << " t=" << t << "\n";
46 return false;
47 }
48
49 Halide::Runtime::Buffer<uint8_t> temp(8, 8, 3);
50 temp.fill(0);
51 temp.set_host_dirty();
52
53 Halide::Internal::JITHandlers handlers;
54 handlers.custom_error = [](void *user_context, const char *msg) {
55 debug(1) << "host_supports_device_api: saw error (" << msg << ")\n";
56 };
57 Halide::Internal::JITHandlers old_handlers = Halide::Internal::JITSharedRuntime::set_default_handlers(handlers);
58
59 int result = temp.copy_to_device(i);
60
61 Halide::Internal::JITSharedRuntime::set_default_handlers(old_handlers);
62
63 if (result != 0) {
64 debug(1) << "host_supports_device_api: copy_to_device() failed for with result=" << result << " for d=" << (int)d << " t=" << t << "\n";
65 return false;
66 }
67 return true;
68}

70const halide_device_interface_t *get_device_interface_for_device_api(DeviceAPI d,
71 const Target &t,
72 const char *error_site) {
73
74 if (d == DeviceAPI::Default_GPU) {
75 d = get_default_device_api_for_target(t);
76 if (d == DeviceAPI::Host) {
77 if (error_site) {
78 user_error
79 << "get_device_interface_for_device_api called from "
80 << error_site
81 << " requested a default GPU but no GPU feature is specified in target ("
82 << t.to_string()
83 << ").\n";
84 }
85 return nullptr;
86 }
87 }
88
89 const struct halide_device_interface_t *(*fn)();
90 std::string name;
91 if (d == DeviceAPI::Metal) {
92 name = "metal";
93 } else if (d == DeviceAPI::OpenCL) {
94 name = "opencl";
95 } else if (d == DeviceAPI::CUDA) {
96 name = "cuda";
97 } else if (d == DeviceAPI::OpenGLCompute) {
98 name = "openglcompute";
99 } else if (d == DeviceAPI::HexagonDma) {
100 name = "hexagon_dma";
101 } else if (d == DeviceAPI::D3D12Compute) {
102 name = "d3d12compute";
103 } else {
104 if (error_site) {
105 user_error
106 << "get_device_interface_for_device_api called from "
107 << error_site
108 << " requested unknown DeviceAPI ("
109 << (int)d
110 << ").\n";
111 }
112 return nullptr;
113 }
114
115 if (!t.supports_device_api(d)) {
116 if (error_site) {
117 user_error
118 << "get_device_interface_for_device_api called from "
119 << error_site
120 << " DeviceAPI ("
121 << name
122 << ") is not supported by target ("
123 << t.to_string()
124 << ").\n";
125 }
126 return nullptr;
127 }
128
129 if (lookup_runtime_routine("halide_" + name + "_device_interface", t, fn)) {
130 return (*fn)();
131 } else {
132 if (error_site) {
133 user_error
134 << "get_device_interface_for_device_api called from "
135 << error_site
136 << " cannot find runtime or device interface symbol for "
137 << name
138 << ".\n";
139 }
140 return nullptr;
141 }
142}

144DeviceAPI get_default_device_api_for_target(const Target &target) {
145 if (target.has_feature(Target::Metal)) {
146 return DeviceAPI::Metal;
147 } else if (target.has_feature(Target::OpenCL)) {
148 return DeviceAPI::OpenCL;
149 } else if (target.has_feature(Target::CUDA)) {
150 return DeviceAPI::CUDA;
151 } else if (target.has_feature(Target::OpenGLCompute)) {
152 return DeviceAPI::OpenGLCompute;
153 } else if (target.has_feature(Target::HexagonDma)) {
154 return DeviceAPI::HexagonDma;
155 } else if (target.has_feature(Target::D3D12Compute)) {
156 return DeviceAPI::D3D12Compute;
157 } else {
158 return DeviceAPI::Host;
159 }
160}

162namespace Internal {
163Expr make_device_interface_call(DeviceAPI device_api, MemoryType memory_type) {
164 if (device_api == DeviceAPI::Host) {
165 return make_zero(type_of<const halide_device_interface_t *>());
166 }
167
168 std::string interface_name;
169 switch (device_api) {
170 case DeviceAPI::CUDA:
171 interface_name = "halide_cuda_device_interface";
172 break;
173 case DeviceAPI::OpenCL:
174 if (memory_type == MemoryType::GPUTexture) {
175 interface_name = "halide_opencl_image_device_interface";
176 } else {
177 interface_name = "halide_opencl_device_interface";
178 }
179 break;
180 case DeviceAPI::Metal:
181 interface_name = "halide_metal_device_interface";
182 break;
183 case DeviceAPI::OpenGLCompute:
184 interface_name = "halide_openglcompute_device_interface";
185 break;
186 case DeviceAPI::Hexagon:
187 interface_name = "halide_hexagon_device_interface";
188 break;
189 case DeviceAPI::HexagonDma:
190 interface_name = "halide_hexagon_dma_device_interface";
191 break;
192 case DeviceAPI::D3D12Compute:
193 interface_name = "halide_d3d12compute_device_interface";
194 break;
195 case DeviceAPI::Default_GPU:
196 // Will be resolved later
197 interface_name = "halide_default_device_interface";
198 break;
199 default:
200 internal_error << "Bad DeviceAPI " << static_cast<int>(device_api) << "\n";
201 break;
202 }
203 return Call::make(type_of<const halide_device_interface_t *>(), interface_name, {}, Call::Extern);
204}
