18void CodeGen_PyTorch::compile(const Module &module) {
19 const Target target = module.target();
20
21 if (target.has_feature(Target::CUDA)) {
22 if (!target.has_feature(Target::UserContext)) {
23 user_error << "Compile a PyTorch wrapper for a CUDA op requires the "
24 "UserContext feature to properly manage the GPU memory. "
25 "Please add \"-user_context\" to the generator's target options.\n";
26 }
27 stream << "#include \"ATen/cuda/CUDAContext.h\"\n";
28 stream << "#include \"HalideBuffer.h\"\n";
29 stream << "#include \"HalidePyTorchCudaHelpers.h\"\n";
30 stream << "#include \"HalidePyTorchHelpers.h\"\n";
31 stream << "#include \"torch/extension.h\"\n";
32 } else {
33 stream << "#include \"HalideBuffer.h\"\n";
34 stream << "#include \"HalidePyTorchHelpers.h\"\n";
35 stream << "#include \"torch/extension.h\"\n";
36 }
37
38 stream << "\n";
39
40 // Emit extern decls of the Halide-generated functions we use directly
41 // into this file, so that we don't have to #include the relevant .h
42 // file directly; this simplifies certain compile/build setups (since
43 // we don't have to build files in tandem and/or get include paths right),
44 // and should be totally safe, since we are using the same codegen logic
45 // that would be in the .h file anyway.
46 {
47 CodeGen_C extern_decl_gen(stream, module.target(), CodeGen_C::CPlusPlusExternDecl);
48 extern_decl_gen.compile(module);
49 }
50
51 for (const auto &f : module.functions()) {
52 if (target.has_feature(Target::CUDA)) {
53 compile(f, true);
54 } else {
55 compile(f, false);
56 }
57 }
58}

60void CodeGen_PyTorch::compile(const LoweredFunc &f, bool is_cuda) {
61 // Don't put non-external function declarations in headers.
62 std::vector<std::string> namespaces;
63 std::string simple_name = extract_namespaces(f.name, namespaces);
64
65 if (!namespaces.empty()) {
66 for (const auto &ns : namespaces) {
67 stream << "namespace " << ns << " {\n";
68 }
69 stream << "\n";
70 }
71 const std::vector<LoweredArgument> &args = f.args;
72 std::vector<LoweredArgument> buffer_args;
73
74 stream << "HALIDE_FUNCTION_ATTRS\n";
75 stream << "inline int " << simple_name << "_th_(";
76 for (size_t i = 0; i < args.size(); i++) {
77 if (args[i].name == "__user_context") {
78 continue;
79 } else if (args[i].is_buffer()) {
80 buffer_args.push_back(args[i]);
81 stream
82 << "at::Tensor &"
83 << c_print_name(args[i].name);
84 } else {
85 stream
86 << type_to_c_type(args[i].type, true)
87 << c_print_name(args[i].name);
88 }
89
90 if (i < args.size() - 1) {
91 stream << ", ";
92 }
93 }
94
95 stream << ") {\n";
96 indent += 4;
97
98 if (is_cuda) {
99 stream << get_indent() << "// Setup CUDA\n";
100 stream << get_indent() << "int device_id = at::cuda::current_device();\n";
101 stream << get_indent() << "CUcontext ctx = 0;\n";
102 stream << get_indent() << "CUresult res = cuCtxGetCurrent(&ctx);\n";
103 stream << get_indent() << "AT_ASSERTM(res == 0, \"Could not acquire CUDA context\");\n";
104 stream << get_indent() << "cudaStream_t stream = at::cuda::getCurrentCUDAStream(device_id);\n";
105 stream << get_indent() << "Halide::PyTorch::UserContext user_ctx(device_id, &ctx, &stream);\n";
106 stream << get_indent() << "void* __user_context = (void*) &user_ctx;\n\n";
107 }
108
109 stream << get_indent() << "// Check tensors have contiguous memory and are on the correct device\n";
110 for (size_t i = 0; i < buffer_args.size(); i++) {
111 stream << get_indent();
112 stream
113 << "HLPT_CHECK_CONTIGUOUS("
114 << c_print_name(buffer_args[i].name)
115 << ");\n";
116
117 if (is_cuda) {
118 stream << get_indent();
119 stream
120 << "HLPT_CHECK_DEVICE("
121 << c_print_name(buffer_args[i].name)
122 << ", device_id);\n";
123 }
124 }
125 stream << "\n";
126
127 stream << get_indent() << "// Wrap tensors in Halide buffers\n";
128 for (size_t i = 0; i < buffer_args.size(); i++) {
129 if (!buffer_args[i].is_buffer()) {
130 continue;
131 }
132
133 stream << get_indent();
134 std::string tp = type_to_c_type(buffer_args[i].type, false);
135 stream
136 << "Halide::Runtime::Buffer<" << tp << "> "
137 << c_print_name(buffer_args[i].name)
138 << "_buffer = Halide::PyTorch::wrap<" << tp << ">("
139 << c_print_name(buffer_args[i].name)
140 << ");\n";
141 }
142 stream << "\n";
143
144 stream << get_indent() << "// Run Halide pipeline\n";
145
146 stream << get_indent() << "int err = " << simple_name << "(";
147 for (size_t i = 0; i < args.size(); i++) {
148 if (args[i].is_buffer()) {
149 stream
150 << c_print_name(args[i].name)
151 << "_buffer";
152 } else {
153 stream << c_print_name(args[i].name);
154 }
155 if (i < args.size() - 1) {
156 stream << ", ";
157 }
158 }
159 stream << ");\n";
160
161 stream << "\n";
162
163 stream << get_indent() << "AT_ASSERTM(err == 0, \"Halide call failed\");\n";
164
165 if (is_cuda) {
166 stream << get_indent() << "// Make sure data is on device\n";
167 for (size_t i = 0; i < buffer_args.size(); i++) {
168 if (buffer_args[i].is_buffer()) {
169 stream << get_indent();
170 stream
171 << "AT_ASSERTM(!"
172 << c_print_name(buffer_args[i].name) << "_buffer.host_dirty(),"
173 << "\"device not synchronized for buffer "
174 << c_print_name(buffer_args[i].name)
175 << ", make sure all update stages are explicitly computed on GPU."
176 << "\");\n";
177 stream << get_indent();
178 stream
179 << c_print_name(buffer_args[i].name) << "_buffer"
180 << ".device_detach_native();\n";
181 }
182 }
183 stream << "\n";
184 }
185
186 // TODO(mgharbi): this is not very well documented
187 if (get_env_variable("FLUSH_MEMOIZE_CACHE") == "1") {
188 stream << get_indent() << "// Flush cache\n";
189 if (is_cuda) {
190 stream << get_indent() << "halide_memoization_cache_cleanup(__user_context);\n";
191 } else {
192 stream << get_indent() << "halide_memoization_cache_cleanup(nullptr);\n";
193 }
194 }
195
196 stream << get_indent() << "return 0;\n";
197
198 indent -= 4;
199 stream << "}\n";
200
201 if (!namespaces.empty()) {
202 stream << "\n";
203 for (size_t i = namespaces.size(); i > 0; i--) {
204 stream << "} // namespace " << namespaces[i - 1] << "\n";
205 }
206 stream << "\n";
207 }

211 // Dummy Halide pipeline
212 LoweredArgument buffer_arg("buf", Argument::OutputBuffer, Int(32), 3, ArgumentEstimates{});
213 LoweredArgument float_arg("alpha", Argument::InputScalar, Float(32), 0, ArgumentEstimates{});
214 LoweredArgument int_arg("beta", Argument::InputScalar, Int(32), 0, ArgumentEstimates{});
215 std::vector<LoweredArgument> args = {buffer_arg, float_arg, int_arg};
216 Var x("x");
217 Param<float> alpha("alpha");
218 Param<int> beta("beta");
219 Expr e = Add::make(alpha, Cast::make(Float(32), beta));
220 Stmt s = Store::make("buf", e, x, Parameter(), const_true(), ModulusRemainder());
221 Expr buf = Variable::make(Handle(), "buf.buffer");
222 s = LetStmt::make("buf", Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern), s);
223
224 const auto compare_src = [&](const std::string &src, const std::string &correct_src) {
225 if (src != correct_src) {
226 int diff = 0;
227 while (src[diff] == correct_src[diff]) {
228 diff++;
229 }
230 int diff_end = diff + 1;
231 while (diff > 0 && src[diff] != '\n') {
232 diff--;
233 }
234 while (diff_end < (int)src.size() && src[diff_end] != '\n') {
235 diff_end++;
236 }
237
238 internal_error
239 << "Correct source code:\n"
240 << correct_src
241 << "Actual source code:\n"
242 << src
243 << "Difference starts at:" << diff << "\n"
244 << "Correct: " << correct_src.substr(diff, diff_end - diff) << "\n"
245 << "Actual: " << src.substr(diff, diff_end - diff) << "\n";
246 }
247 };
248
249 {
250 // TODO(mgharbi): test that Target("host-cuda") raises an exception since
251 // we require the "user_context" feature when using CUDA

253 Module m("", Target("host"));

366HALIDE_FUNCTION_ATTRS
367inline int test1_th_(at::Tensor &_buf, float _alpha, int32_t _beta) {
368 // Setup CUDA
369 int device_id = at::cuda::current_device();
370 CUcontext ctx = 0;
371 CUresult res = cuCtxGetCurrent(&ctx);
372 AT_ASSERTM(res == 0, "Could not acquire CUDA context");
373 cudaStream_t stream = at::cuda::getCurrentCUDAStream(device_id);
374 Halide::PyTorch::UserContext user_ctx(device_id, &ctx, &stream);
375 void* __user_context = (void*) &user_ctx;
376
377 // Check tensors have contiguous memory and are on the correct device
378 HLPT_CHECK_CONTIGUOUS(_buf);
379 HLPT_CHECK_DEVICE(_buf, device_id);
380
381 // Wrap tensors in Halide buffers
382 Halide::Runtime::Buffer<int32_t> _buf_buffer = Halide::PyTorch::wrap<int32_t>(_buf);
383
384 // Run Halide pipeline
385 int err = test1(_buf_buffer, _alpha, _beta);
386
387 AT_ASSERTM(err == 0, "Halide call failed");
388 // Make sure data is on device
389 AT_ASSERTM(!_buf_buffer.host_dirty(),"device not synchronized for buffer _buf, make sure all update stages are explicitly computed on GPU.");
390 _buf_buffer.device_detach_native();
391
392 return 0;
393}
394)GOLDEN_CODE";
395
396 compare_src(src.str(), correct_src);
397 } else {
398 user_warning << "Host does not support " << host_cuda << ", skipping part of test";
399 }
400
401 std::cout << "CodeGen_PyTorch test passed\n";
