8Expr Simplify::visit(const Shuffle *op, ExprInfo *bounds) {
9 if (op->is_extract_element() &&
10 (op->vectors[0].as<Ramp>() ||
11 op->vectors[0].as<Broadcast>())) {
12 // Extracting a single lane of a ramp or broadcast
13 if (const Ramp *r = op->vectors[0].as<Ramp>()) {
14 return mutate(r->base + op->indices[0] * r->stride, bounds);
15 } else if (const Broadcast *b = op->vectors[0].as<Broadcast>()) {
16 return mutate(b->value, bounds);
17 } else {
18 internal_error << "Unreachable";
19 return Expr();
20 }
21 }
22
23 // Mutate the vectors
24 vector<Expr> new_vectors;
25 bool changed = false;
26 for (const Expr &vector : op->vectors) {
27 ExprInfo v_bounds;
28 Expr new_vector = mutate(vector, &v_bounds);
29 if (!vector.same_as(new_vector)) {
30 changed = true;
31 }
32 if (bounds) {
33 if (new_vectors.empty()) {
34 *bounds = v_bounds;
35 } else {
36 bounds->min_defined &= v_bounds.min_defined;
37 bounds->max_defined &= v_bounds.max_defined;
38 bounds->min = std::min(bounds->min, v_bounds.min);
39 bounds->max = std::max(bounds->max, v_bounds.max);
40 bounds->alignment = ModulusRemainder::unify(bounds->alignment, v_bounds.alignment);
41 }
42 }
43 new_vectors.push_back(new_vector);
44 }
45
46 // Try to convert a load with shuffled indices into a
47 // shuffle of a dense load.
48 if (const Load *first_load = new_vectors[0].as<Load>()) {
49 vector<Expr> load_predicates;
50 vector<Expr> load_indices;
51 bool unpredicated = true;
52 for (const Expr &e : new_vectors) {
53 const Load *load = e.as<Load>();
54 if (load && load->name == first_load->name) {
55 load_predicates.push_back(load->predicate);
56 load_indices.push_back(load->index);
57 unpredicated = unpredicated && is_const_one(load->predicate);
58 } else {
59 break;
60 }
61 }
62
63 if (load_indices.size() == new_vectors.size()) {
64 Type t = load_indices[0].type().with_lanes(op->indices.size());
65 Expr shuffled_index = Shuffle::make(load_indices, op->indices);
66 ExprInfo shuffled_index_info;
67 shuffled_index = mutate(shuffled_index, &shuffled_index_info);
68 if (shuffled_index.as<Ramp>()) {
69 ExprInfo base_info;
70 if (const Ramp *r = shuffled_index.as<Ramp>()) {
71 mutate(r->base, &base_info);
72 }
73
74 ModulusRemainder alignment =
75 ModulusRemainder::intersect(base_info.alignment, shuffled_index_info.alignment);
76
77 Expr shuffled_predicate;
78 if (unpredicated) {
79 shuffled_predicate = const_true(t.lanes());
80 } else {
81 shuffled_predicate = Shuffle::make(load_predicates, op->indices);
82 shuffled_predicate = mutate(shuffled_predicate, nullptr);
83 }
84 t = first_load->type;
85 t = t.with_lanes(op->indices.size());
86 return Load::make(t, first_load->name, shuffled_index, first_load->image,
87 first_load->param, shuffled_predicate, alignment);
88 }
89 }
90 }
91
92 // Try to collapse a shuffle of broadcasts into a single
93 // broadcast. Note that it doesn't matter what the indices
94 // are.
95 const Broadcast *b1 = new_vectors[0].as<Broadcast>();
96 if (b1) {
97 bool can_collapse = true;
98 for (size_t i = 1; i < new_vectors.size() && can_collapse; i++) {
99 if (const Broadcast *b2 = new_vectors[i].as<Broadcast>()) {
100 Expr check = mutate(b1->value - b2->value, nullptr);
101 can_collapse &= is_const_zero(check);
102 } else {
103 can_collapse = false;
104 }
105 }
106 if (can_collapse) {
107 if (op->indices.size() == 1) {
108 return b1->value;
109 } else {
110 return Broadcast::make(b1->value, op->indices.size());
111 }
112 }
113 }
114
115 if (op->is_interleave()) {
116 int terms = (int)new_vectors.size();
117
118 // Try to collapse an interleave of ramps into a single ramp.
119 const Ramp *r = new_vectors[0].as<Ramp>();
120 if (r) {
121 bool can_collapse = true;
122 for (size_t i = 1; i < new_vectors.size() && can_collapse; i++) {
123 // If we collapse these terms into a single ramp,
124 // the new stride is going to be the old stride
125 // divided by the number of terms, so the
126 // difference between two adjacent terms in the
127 // interleave needs to be a broadcast of the new
128 // stride.
129 Expr diff = mutate(new_vectors[i] - new_vectors[i - 1], nullptr);
130 const Broadcast *b = diff.as<Broadcast>();
131 if (b) {
132 Expr check = mutate(b->value * terms - r->stride, nullptr);
133 can_collapse &= is_const_zero(check);
134 } else {
135 can_collapse = false;
136 }
137 }
138 if (can_collapse) {
139 return mutate(Ramp::make(r->base, r->stride / terms, r->lanes * terms), bounds);
140 }
141 }
142
143 // Try to collapse an interleave of slices of vectors from
144 // the same vector into a single vector.
145 if (const Shuffle *first_shuffle = new_vectors[0].as<Shuffle>()) {
146 if (first_shuffle->is_slice()) {
147 bool can_collapse = true;
148 for (size_t i = 0; i < new_vectors.size() && can_collapse; i++) {
149 const Shuffle *i_shuffle = new_vectors[i].as<Shuffle>();
150
151 // Check that the current shuffle is a slice...
152 if (!i_shuffle || !i_shuffle->is_slice()) {
153 can_collapse = false;
154 break;
155 }
156
157 // ... and that it is a slice in the right place...
158 // If the shuffle is a single element, we don't care what the stride is.
159 if (i_shuffle->slice_begin() != (int)i ||
160 (i_shuffle->indices.size() != 1 && i_shuffle->slice_stride() != terms)) {
161 can_collapse = false;
162 break;
163 }
164
165 if (i > 0) {
166 // ... and that the vectors being sliced are the same.
167 if (first_shuffle->vectors.size() != i_shuffle->vectors.size()) {
168 can_collapse = false;
169 break;
170 }
171
172 for (size_t j = 0; j < first_shuffle->vectors.size() && can_collapse; j++) {
173 if (!equal(first_shuffle->vectors[j], i_shuffle->vectors[j])) {
174 can_collapse = false;
175 }
176 }
177 }
178 }
179
180 if (can_collapse) {
181 // It's possible the slices didn't use all of the vector, in which case we need to slice it.
182 Expr result = Shuffle::make_concat(first_shuffle->vectors);
183 if (result.type().lanes() != op->type.lanes()) {
184 result = Shuffle::make_slice(result, 0, 1, op->type.lanes());
185 }
186 return result;
187 }
188 }
189 }
190 } else if (op->is_concat()) {
191 // Try to collapse a concat of ramps into a single ramp.
192 const Ramp *r = new_vectors[0].as<Ramp>();
193 if (r) {
194 bool can_collapse = true;
195 for (size_t i = 1; i < new_vectors.size() && can_collapse; i++) {
196 Expr diff;
197 if (new_vectors[i].type().lanes() == new_vectors[i - 1].type().lanes()) {
198 diff = mutate(new_vectors[i] - new_vectors[i - 1], nullptr);
199 }
200
201 const Broadcast *b = diff.as<Broadcast>();
202 if (b) {
203 Expr check = mutate(b->value - r->stride * new_vectors[i - 1].type().lanes(), nullptr);
204 can_collapse &= is_const_zero(check);
205 } else {
206 can_collapse = false;
207 }
208 }
209 if (can_collapse) {
210 return Ramp::make(r->base, r->stride, op->indices.size());
211 }
212 }
213
214 // Try to collapse a concat of scalars into a ramp.
215 if (new_vectors[0].type().is_scalar() && new_vectors[1].type().is_scalar()) {
216 bool can_collapse = true;
217 Expr stride = mutate(new_vectors[1] - new_vectors[0], nullptr);
218 for (size_t i = 1; i < new_vectors.size() && can_collapse; i++) {
219 if (!new_vectors[i].type().is_scalar()) {
220 can_collapse = false;
221 break;
222 }
223
224 Expr check = mutate(new_vectors[i] - new_vectors[i - 1] - stride, nullptr);
225 if (!is_const_zero(check)) {
226 can_collapse = false;
227 }
228 }
229
230 if (can_collapse) {
231 return Ramp::make(new_vectors[0], stride, op->indices.size());
232 }
233 }
234 }
235
236 // Pull a widening cast outside of a slice
237 if (new_vectors.size() == 1 &&
238 op->type.lanes() < new_vectors[0].type().lanes()) {
239 if (const Cast *cast = new_vectors[0].as<Cast>()) {
240 if (cast->type.bits() > cast->value.type().bits()) {
241 return mutate(Cast::make(cast->type.with_lanes(op->type.lanes()),
242 Shuffle::make({cast->value}, op->indices)),
243 bounds);
244 }
245 }
246 }
247
248 if (op->is_slice() && (new_vectors.size() == 1)) {
249 if (const Shuffle *inner_shuffle = new_vectors[0].as<Shuffle>()) {
250 // Try to collapse a slice of slice.
251 if (inner_shuffle->is_slice() && (inner_shuffle->vectors.size() == 1)) {
252 // Indices of the slice are ramp, so nested slice is a1 * (a2 * x + b2) + b1 =
253 // = a1 * a2 * x + a1 * b2 + b1.
254 return Shuffle::make_slice(inner_shuffle->vectors[0],
255 op->slice_begin() * inner_shuffle->slice_stride() + inner_shuffle->slice_begin(),
256 op->slice_stride() * inner_shuffle->slice_stride(),
257 op->indices.size());
258 }
259 // Check if we really need to concat all vectors before slicing.
260 if (inner_shuffle->is_concat()) {
261 int slice_min = op->indices.front();
262 int slice_max = op->indices.back();
263 int concat_index = 0;
264 int new_slice_start = -1;
265 vector<Expr> new_concat_vectors;
266 for (const auto &v : inner_shuffle->vectors) {
267 // Check if current concat vector overlaps with slice.
268 if ((concat_index >= slice_min && concat_index <= slice_max) ||
269 ((concat_index + v.type().lanes() - 1) >= slice_min && (concat_index + v.type().lanes() - 1) <= slice_max)) {
270 if (new_slice_start < 0) {
271 new_slice_start = concat_index;
272 }
273 new_concat_vectors.push_back(v);
274 }
275
276 concat_index += v.type().lanes();
277 }
278 if (new_concat_vectors.size() < inner_shuffle->vectors.size()) {
279 return Shuffle::make_slice(Shuffle::make_concat(new_concat_vectors), op->slice_begin() - new_slice_start, op->slice_stride(), op->indices.size());
280 }
281 }
282 }
283 }
284
285 if (!changed) {
286 return op;
287 } else {
288 return Shuffle::make(new_vectors, op->indices);
289 }
290}

292template<typename T>
293Expr Simplify::hoist_slice_vector(Expr e) {
294 const T *op = e.as<T>();
295 internal_assert(op);
296
297 const Shuffle *shuffle_a = op->a.template as<Shuffle>();
298 const Shuffle *shuffle_b = op->b.template as<Shuffle>();
299
300 internal_assert(shuffle_a && shuffle_b &&
301 shuffle_a->is_slice() &&
302 shuffle_b->is_slice());
303
304 if (shuffle_a->indices != shuffle_b->indices) {
305 return e;
306 }
307
308 const std::vector<Expr> &slices_a = shuffle_a->vectors;
309 const std::vector<Expr> &slices_b = shuffle_b->vectors;
310 if (slices_a.size() != slices_b.size()) {
311 return e;
312 }
313
314 for (size_t i = 0; i < slices_a.size(); i++) {
315 if (slices_a[i].type() != slices_b[i].type()) {
316 return e;
317 }
318 }
319
320 vector<Expr> new_slices;
321 for (size_t i = 0; i < slices_a.size(); i++) {
322 new_slices.push_back(T::make(slices_a[i], slices_b[i]));
323 }
324
325 return Shuffle::make(new_slices, shuffle_a->indices);
326}
