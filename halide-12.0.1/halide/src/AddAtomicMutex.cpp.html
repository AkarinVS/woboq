20/** Collect names of all stores matching the producer name inside a statement. */
21class CollectProducerStoreNames : public IRGraphVisitor {
22public:
23 CollectProducerStoreNames(const std::string &producer_name)
24 : producer_name(producer_name) {
25 }
26
27 Scope<void> store_names;
28
29protected:
30 using IRGraphVisitor::visit;
31
32 void visit(const Store *op) override {
33 IRGraphVisitor::visit(op);
34 if (op->name == producer_name || starts_with(op->name, producer_name + ".")) {
35 // This is a Store for the desginated Producer.
36 store_names.push(op->name);
37 }
38 }
39
40 const std::string &producer_name;
41};

43/** Find Store inside of an Atomic node for the designated producer
44 * and return their indices. */
45class FindProducerStoreIndex : public IRGraphVisitor {
46public:
47 FindProducerStoreIndex(const std::string &producer_name)
48 : producer_name(producer_name) {
49 }
50
51 Expr index; // The returned index.
52
53protected:
54 using IRGraphVisitor::visit;
55
56 // Need to also extract the let bindings of a Store index.
57 void visit(const Let *op) override {
58 IRGraphVisitor::visit(op); // Make sure we visit the Store first.
59 if (index.defined()) {
60 if (expr_uses_var(index, op->name)) {
61 index = Let::make(op->name, op->value, index);
62 }
63 }
64 }
65 void visit(const LetStmt *op) override {
66 IRGraphVisitor::visit(op); // Make sure we visit the Store first.
67 if (index.defined()) {
68 if (expr_uses_var(index, op->name)) {
69 index = Let::make(op->name, op->value, index);
70 }
71 }
72 }
73
74 void visit(const Store *op) override {
75 IRGraphVisitor::visit(op);
76 if (op->name == producer_name || starts_with(op->name, producer_name + ".")) {
77 // This is a Store for the designated producer.
78
79 // Ideally we want to insert equal() checks here for different stores,
80 // but the indices of them actually are different in the case of tuples,
81 // since they usually refer to the strides/min/extents of their own tuple
82 // buffers. However, different elements in a tuple would have the same
83 // strides/min/extents so we are fine.
84 if (index.defined()) {
85 return;
86 }
87 index = op->index;
88 }
89 }
90
91 const std::string &producer_name;
92};

94/** Throws an assertion for cases where the indexing on left-hand-side of
95 * an atomic update references to itself.
96 * e.g. f(clamp(f(r), 0, 100)) = f(r) + 1 should be rejected. */
97class CheckAtomicValidity : public IRGraphVisitor {
98protected:
99 using IRGraphVisitor::visit;
100
101 void visit(const Atomic *op) override {
102 // Collect the names of all Store nodes inside.
103 CollectProducerStoreNames collector(op->producer_name);
104 op->body.accept(&collector);
105
106 // Find the indices from the Store nodes inside the body.
107 FindProducerStoreIndex find(op->producer_name);
108 op->body.accept(&find);
109
110 Expr index = find.index;
111 if (index.defined()) {
112 user_assert(!expr_uses_vars(index, collector.store_names))
113 << "Can't use atomic() on an update where the index written "
114 << "to depends on the current value of the Func\n";
115 }
116 op->body.accept(this);
117 }
118};

120/** Search if the value of a Store node has a variable pointing to a let binding,
121 * where the let binding contains the Store location. Use for checking whether
122 * we need a mutex lock for Atomic since some lowering pass before lifted a let
123 * binding from the Store node (currently only SplitTuple would do this). */
124class FindAtomicLetBindings : public IRGraphVisitor {
125public:
126 FindAtomicLetBindings(const Scope<void> &store_names)
127 : store_names(store_names) {
128 }
129
130 bool found = false;
131
132protected:
133 using IRVisitor::visit;
134
135 void visit(const Let *op) override {
136 include(op->value);
137 {
138 ScopedBinding<Expr> bind(let_bindings, op->name, op->value);
139 include(op->body);
140 }
141 }
142
143 void visit(const LetStmt *op) override {
144 include(op->value);
145 {
146 ScopedBinding<Expr> bind(let_bindings, op->name, op->value);
147 include(op->body);
148 }
149 }
150
151 void visit(const Variable *op) override {
152 if (!inside_store.empty()) {
153 // If this Variable inside the store value is an expression
154 // that depends on one of the store_names, we found a lifted let.
155 if (expr_uses_vars(op, store_names, let_bindings)) {
156 found = true;
157 }
158 }
159 }
160
161 void visit(const Store *op) override {
162 include(op->predicate);
163 if (store_names.contains(op->name)) {
164 // If we are in a designated store and op->value has a let binding
165 // that uses one of the store_names, we found a lifted let.
166 ScopedValue<string> old_inside_store(inside_store, op->name);
167 include(op->value);
168 } else {
169 include(op->value);
170 }
171 include(op->index);
172 }
173
174 string inside_store;
175 const Scope<void> &store_names;
176 Scope<Expr> let_bindings;
177};

179/** Clear out the Atomic node's mutex usages if it doesn't need one. */
180class RemoveUnnecessaryMutexUse : public IRMutator {
181public:
182 set<string> remove_mutex_lock_names;
183
184protected:
185 using IRMutator::visit;
186
187 Stmt visit(const Atomic *op) override {
188 // Collect the names of all Store nodes inside.
189 CollectProducerStoreNames collector(op->producer_name);
190 op->body.accept(&collector);
191 // Search for let bindings that access the producers.
192 FindAtomicLetBindings finder(collector.store_names);
193 op->body.accept(&finder);
194 // Each individual Store that remains can be done as a CAS
195 // loop or an actual atomic RMW of some form.
196 if (finder.found) {
197 // Can't remove mutex lock. Leave the Stmt as is.
198 return IRMutator::visit(op);
199 } else {
200 remove_mutex_lock_names.insert(op->mutex_name);
201 Stmt body = mutate(op->body);
202 return Atomic::make(op->producer_name,
203 string(),
204 std::move(body));
205 }
206 }

209/** Find Store inside an Atomic that matches the provided store_names. */
210class FindStoreInAtomicMutex : public IRGraphVisitor {
211public:
212 using IRGraphVisitor::visit;
213
214 FindStoreInAtomicMutex(const std::set<std::string> &store_names)
215 : store_names(store_names) {
216 }
217
218 bool found = false;
219 string producer_name;
220 string mutex_name;
221
222protected:
223 void visit(const Atomic *op) override {
224 if (!found && !op->mutex_name.empty()) {
225 ScopedValue<bool> old_in_atomic_mutex(in_atomic_mutex, true);
226 include(op->body);
227 if (found) {
228 // We found a Store inside Atomic with matching name,
229 // record the mutex information.
230 producer_name = op->producer_name;
231 mutex_name = op->mutex_name;
232 }
233 } else {
234 include(op->body);
235 }
236 }
237
238 void visit(const Store *op) override {
239 if (in_atomic_mutex) {
240 if (store_names.find(op->name) != store_names.end()) {
241 found = true;
242 }
243 }
244 IRGraphVisitor::visit(op);
245 }
246
247 bool in_atomic_mutex = false;
248 const set<string> &store_names;
249};

276/** Add mutex allocation & lock & unlock if required. */
277class AddAtomicMutex : public IRMutator {
278public:
279 AddAtomicMutex(const map<string, Function> &env)
280 : env(env) {
281 }
282
283protected:
284 using IRMutator::visit;
285
286 const map<string, Function> &env;
287 // The set of producers that have allocated a mutex buffer
288 set<string> allocated_mutexes;
289
290 Stmt allocate_mutex(const string &mutex_name, Expr extent, Stmt body) {
291 Expr mutex_array = Call::make(type_of<halide_mutex_array *>(),
292 "halide_mutex_array_create",
293 {std::move(extent)},
294 Call::Extern);
295 // Allocate a scalar of halide_mutex_array.
296 // This generates halide_mutex_array mutex[1];
297 body = Allocate::make(mutex_name,
298 Handle(),
299 MemoryType::Stack,
300 {},
301 const_true(),
302 body,
303 mutex_array,
304 "halide_mutex_array_destroy");
305 return body;
306 }
307
308 Stmt visit(const Allocate *op) override {
309 // If this Allocate node is allocating a buffer for a producer,
310 // and there is a Store node inside of an Atomic node requiring mutex lock
311 // matching the name of the Allocate, allocate a mutex lock.
312 set<string> store_names{op->name};
313 FindStoreInAtomicMutex finder(store_names);
314 op->body.accept(&finder);
315 if (!finder.found) {
316 // No Atomic node that requires mutex lock from this node inside.
317 return IRMutator::visit(op);
318 }
319
320 if (allocated_mutexes.find(finder.mutex_name) != allocated_mutexes.end()) {
321 // We've already allocated a mutex.
322 return IRMutator::visit(op);
323 }
324
325 allocated_mutexes.insert(finder.mutex_name);
326
327 const string &mutex_name = finder.mutex_name;
328 Stmt body = mutate(op->body);
329 Expr extent = Expr(1);
330 for (const Expr &e : op->extents) {
331 extent = extent * e;
332 }
333 body = allocate_mutex(mutex_name, extent, body);
334 return Allocate::make(op->name,
335 op->type,
336 op->memory_type,
337 op->extents,
338 op->condition,
339 std::move(body),
340 op->new_expr,
341 op->free_function);
342 }
343
344 Stmt visit(const ProducerConsumer *op) override {
345 // Usually we allocate the mutex buffer at the Allocate node,
346 // but outputs don't have Allocate. For those we allocate the mutex
347 // buffer at the producer node.
348
349 if (!op->is_producer) {
350 // This is a consumer.
351 return IRMutator::visit(op);
352 }
353
354 // Find the corresponding output.
355 auto func_it = env.find(op->name);
356 if (func_it == env.end()) {
357 // Not an output.
358 return IRMutator::visit(op);
359 }
360 Func f = Func(func_it->second);
361 if (f.output_buffers().empty()) {
362 // Not an output.
363 return IRMutator::visit(op);
364 }
365
366 set<string> store_names;
367 for (const auto &buffer : f.output_buffers()) {
368 store_names.insert(buffer.name());
369 }
370
371 FindStoreInAtomicMutex finder(store_names);
372 op->body.accept(&finder);
373 if (!finder.found) {
374 // No Atomic node that requires mutex lock from this node inside.
375 return IRMutator::visit(op);
376 }
377
378 if (allocated_mutexes.find(finder.mutex_name) != allocated_mutexes.end()) {
379 // We've already allocated a mutex.
380 return IRMutator::visit(op);
381 }
382
383 allocated_mutexes.insert(finder.mutex_name);
384
385 // We assume all output buffers in a Tuple have the same extent.
386 OutputImageParam output_buffer = f.output_buffers()[0];
387 Expr extent = Expr(1);
388 for (int i = 0; i < output_buffer.dimensions(); i++) {
389 extent = extent * output_buffer.dim(i).extent();
390 }
391 Stmt body = mutate(op->body);
392 body = allocate_mutex(finder.mutex_name, extent, body);
393 return ProducerConsumer::make(op->name, op->is_producer, std::move(body));
394 }
395
396 Stmt visit(const Atomic *op) override {
397 if (op->mutex_name.empty()) {
398 return IRMutator::visit(op);
399 }
400
401 // Lock the mutexes using the indices from the Store nodes inside the body.
402 FindProducerStoreIndex find(op->producer_name);
403 op->body.accept(&find);
404
405 Stmt body = op->body;
406
407 Expr index = find.index;
408 Expr index_let; // If defined, represents the value of the lifted let binding.
409 if (!index.defined()) {
410 // Scalar output.
411 index = Expr(0);
412 } else {
413 // Lift the index outside of the atomic node.
414 // This is for avoiding side-effects inside those expressions
415 // being evaluated twice.
416 string name = unique_name('t');
417 index_let = index;
418 index = Variable::make(index.type(), name);
419 body = ReplaceStoreIndexWithVar(op->producer_name, index).mutate(body);
420 }
421 // This generates a pointer to the mutex array
422 Expr mutex_array = Variable::make(
423 type_of<halide_mutex_array *>(), op->mutex_name);
424 // Add mutex locks & unlocks
425 // If a thread locks the mutex and throws an exception,
426 // halide_mutex_array_destroy will be called and cleanup the mutex locks.
427 body = Block::make(
428 Evaluate::make(Call::make(type_of<int>(),
429 "halide_mutex_array_lock",
430 {mutex_array, index},
431 Call::CallType::Extern)),
432 Block::make(std::move(body),
433 Evaluate::make(Call::make(type_of<int>(),
434 "halide_mutex_array_unlock",
435 {mutex_array, index},
436 Call::CallType::Extern))));
437 Stmt ret = Atomic::make(op->producer_name,
438 op->mutex_name,
439 std::move(body));
440
441 if (index_let.defined()) {
442 // Attach the let binding outside of the atomic node.
443 internal_assert(index.as<Variable>() != nullptr);
444 ret = LetStmt::make(index.as<Variable>()->name, index_let, ret);
445 }
446 return ret;
447 }

452Stmt add_atomic_mutex(Stmt s, const map<string, Function> &env) {
453 CheckAtomicValidity check;
454 s.accept(&check);
455 s = RemoveUnnecessaryMutexUse().mutate(s);
456 s = AddAtomicMutex(env).mutate(s);
457 return s;
458}
