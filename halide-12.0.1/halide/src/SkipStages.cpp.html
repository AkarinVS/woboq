25bool extern_call_uses_buffer(const Call *op, const std::string &func) {
26 if (op->is_extern()) {
27 if (starts_with(op->name, "halide_memoization")) {
28 return false;
29 }
30 for (size_t i = 0; i < op->args.size(); i++) {
31 const Variable *var = op->args[i].as<Variable>();
32 if (var &&
33 starts_with(var->name, func + ".") &&
34 ends_with(var->name, ".buffer")) {
35 return true;
36 }
37 }
38 }
39 return false;
40}

42class PredicateFinder : public IRVisitor {
43public:
44 Expr predicate;
45 PredicateFinder(const string &b, bool s)
46 : predicate(const_false()),
47 buffer(b),
48 varies(false),
49 treat_selects_as_guards(s),
50 in_produce(false) {
51 }
52
53private:
54 using IRVisitor::visit;
55 string buffer;
56 bool varies;
57 bool treat_selects_as_guards;
58 bool in_produce;
59 Scope<> varying;
60 Scope<> in_pipeline;
61 Scope<> local_buffers;

63 void visit(const Variable *op) override {
64 bool this_varies = varying.contains(op->name);
65
66 varies |= this_varies;
67 }
68
69 void visit(const For *op) override {
70 op->min.accept(this);
71 bool min_varies = varies;
72 op->extent.accept(this);
73 bool should_pop = false;
74 if (!is_const_one(op->extent) || min_varies) {
75 should_pop = true;
76 varying.push(op->name);
77 }
78 op->body.accept(this);
79 if (should_pop) {
80 varying.pop(op->name);
81 //internal_assert(!expr_uses_var(predicate, op->name));
82 } else if (expr_uses_var(predicate, op->name)) {
83 predicate = Let::make(op->name, op->min, predicate);
84 }

87 template<typename T>
88 void visit_let(const std::string &name, const Expr &value, T body) {
89 bool old_varies = varies;
90 varies = false;
91 value.accept(this);
92 bool value_varies = varies;
93 varies |= old_varies;
94 if (value_varies) {
95 varying.push(name);
96 }
97 body.accept(this);
98 if (value_varies) {
99 varying.pop(name);
100 }
101 if (expr_uses_var(predicate, name)) {
102 predicate = Let::make(name, value, predicate);
103 }
104 }

106 void visit(const LetStmt *op) override {
107 visit_let(op->name, op->value, op->body);

110 void visit(const Let *op) override {
111 visit_let(op->name, op->value, op->body);

114 void visit(const ProducerConsumer *op) override {
115 ScopedBinding<> bind(in_pipeline, op->name);
116 if (op->is_producer && op->name == buffer) {
117 ScopedValue<bool> sv(in_produce, true);

124 // Logical operators with eager constant folding
125 Expr make_and(Expr a, Expr b) {
126 if (is_const_zero(a) || is_const_one(b)) {
127 return a;
128 } else if (is_const_zero(b) || is_const_one(a)) {
129 return b;
130 } else if (equal(a, b)) {
131 return a;
132 } else {
133 return a && b;
134 }
135 }
136
137 Expr make_or(Expr a, Expr b) {
138 if (is_const_zero(a) || is_const_one(b)) {
139 return b;
140 } else if (is_const_zero(b) || is_const_one(a)) {
141 return a;
142 } else if (equal(a, b)) {
143 return a;
144 } else {
145 return a || b;
146 }
147 }
148
149 Expr make_select(const Expr &a, Expr b, Expr c) {
150 if (is_const_one(a)) {
151 return b;
152 } else if (is_const_zero(a)) {
153 return c;
154 } else if (is_const_one(b)) {
155 return make_or(a, c);
156 } else if (is_const_zero(b)) {
157 return make_and(make_not(a), c);
158 } else if (is_const_one(c)) {
159 return make_or(make_not(a), b);
160 } else if (is_const_zero(c)) {
161 return make_and(a, b);
162 } else {
163 return select(a, b, c);
164 }
165 }
166
167 Expr make_not(const Expr &a) {
168 if (is_const_one(a)) {
169 return make_zero(a.type());
170 } else if (is_const_zero(a)) {
171 return make_one(a.type());
172 } else {
173 return !a;
174 }
175 }

177 template<typename T>
178 void visit_conditional(const Expr &condition, T true_case, T false_case) {
179 Expr old_predicate = predicate;
180
181 predicate = const_false();
182 true_case.accept(this);
183 Expr true_predicate = predicate;
184
185 predicate = const_false();
186 if (false_case.defined()) {
187 false_case.accept(this);
188 }
189 Expr false_predicate = predicate;
190
191 bool old_varies = varies;
192 predicate = const_false();
193 varies = false;
194 condition.accept(this);
195
196 predicate = make_or(predicate, old_predicate);
197 if (varies) {
198 predicate = make_or(predicate, make_or(true_predicate, false_predicate));
199 } else {
200 predicate = make_or(predicate, make_select(condition, true_predicate, false_predicate));
201 }
202
203 varies = varies || old_varies;
204 }

206 void visit(const Select *op) override {
207 if (treat_selects_as_guards) {
208 visit_conditional(op->condition, op->true_value, op->false_value);
209 } else {
210 IRVisitor::visit(op);
211 }
212 }

214 void visit(const IfThenElse *op) override {
215 visit_conditional(op->condition, op->then_case, op->else_case);
216 }

218 void visit(const Call *op) override {
219 varies |= in_pipeline.contains(op->name);
220
221 IRVisitor::visit(op);
222
223 if (!in_produce && (op->name == buffer || extern_call_uses_buffer(op, buffer))) {
224 predicate = const_true();
225 }
226 }
227
228 void visit(const Provide *op) override {
229 IRVisitor::visit(op);
230 if (in_produce && op->name != buffer && !local_buffers.contains(op->name)) {
231 predicate = const_true();
232 }
233 }
234
235 void visit(const Realize *op) override {
236 ScopedBinding<> bind(local_buffers, op->name);
237 IRVisitor::visit(op);
238 }

240 void visit(const Allocate *op) override {
241 // This code works to ensure expressions depending on an
242 // allocation don't get moved outside the allocation and are
243 // marked as varying if predicate depends on the value of the
244 // allocation.
245 ScopedBinding<>
246 bind_host_ptr(varying, op->name),
247 bind_buffer(varying, op->name + ".buffer");
248 IRVisitor::visit(op);

252class ProductionGuarder : public IRMutator {
253public:
254 ProductionGuarder(const string &b, Expr compute_p, Expr alloc_p)
255 : buffer(b), compute_predicate(std::move(compute_p)), alloc_predicate(std::move(alloc_p)) {
256 }
257
258private:
259 string buffer;
260 Expr compute_predicate;
261 Expr alloc_predicate;
262
263 using IRMutator::visit;
264
265 bool memoize_call_uses_buffer(const Call *op) {
266 internal_assert(op->call_type == Call::Extern);
267 internal_assert(starts_with(op->name, "halide_memoization"));
268 for (size_t i = 0; i < op->args.size(); i++) {
269 const Variable *var = op->args[i].as<Variable>();
270 if (var &&
271 starts_with(var->name, buffer + ".") &&
272 ends_with(var->name, ".buffer")) {
273 return true;
274 }
275 }
276 return false;
277 }
278
279 Expr visit(const Call *op) override {
280
281 if ((op->name == "halide_memoization_cache_lookup") &&
282 memoize_call_uses_buffer(op)) {
283 // We need to guard call to halide_memoization_cache_lookup to only
284 // be executed if the corresponding buffer is allocated. We ignore
285 // the compute_predicate since in the case that alloc_predicate is
286 // true but compute_predicate is false, the consumer would still load
287 // data from the buffer even if it won't actually use the result,
288 // hence, we need to allocate some scratch memory for the consumer
289 // to load from. For memoized func, the memory might already be in
290 // the cache, so we perform the lookup instead of allocating a new one.
291 return Call::make(op->type, Call::if_then_else,
292 {alloc_predicate, op, 0}, Call::PureIntrinsic);
293 } else if ((op->name == "halide_memoization_cache_store") &&
294 memoize_call_uses_buffer(op)) {
295 // We need to wrap the halide_memoization_cache_store with the
296 // compute_predicate, since the data to be written is only valid if
297 // the producer of the buffer is executed.
298 return Call::make(op->type, Call::if_then_else,
299 {compute_predicate, op, 0}, Call::PureIntrinsic);
300 } else {
301 return IRMutator::visit(op);
302 }
303 }
304
305 Stmt visit(const ProducerConsumer *op) override {
306 // If the compute_predicate at this stage depends on something
307 // vectorized we should bail out.
308 Stmt stmt = IRMutator::visit(op);
309
310 if (op->is_producer) {
311 op = stmt.as<ProducerConsumer>();
312 internal_assert(op);
313 if (op->name == buffer) {
314 Stmt body = IfThenElse::make(compute_predicate, op->body);
315 stmt = ProducerConsumer::make(op->name, op->is_producer, body);
316 }
317 }
318 return stmt;
319 }

322class StageSkipper : public IRMutator {
323public:
324 StageSkipper(const string &f)
325 : func(f), in_vector_loop(false) {
326 }
327
328private:
329 string func;
330 using IRMutator::visit;
331
332 Scope<> vector_vars;
333 bool in_vector_loop;
334
335 Stmt visit(const For *op) override {
336 bool old_in_vector_loop = in_vector_loop;
337
338 // We want to be sure that the predicate doesn't vectorize.
339 if (op->for_type == ForType::Vectorized) {
340 vector_vars.push(op->name);
341 in_vector_loop = true;
342 }
343
344 Stmt stmt = IRMutator::visit(op);
345
346 if (op->for_type == ForType::Vectorized) {
347 vector_vars.pop(op->name);
348 }
349
350 in_vector_loop = old_in_vector_loop;
351
352 return stmt;
353 }
354
355 Stmt visit(const LetStmt *op) override {
356 struct Frame {
357 const LetStmt *op;
358 bool vector_var;
359 };
360 vector<Frame> frames;
361 Stmt result;
362
363 while (op) {
364 bool vector_var = in_vector_loop && expr_uses_vars(op->value, vector_vars);
365 frames.push_back({op, vector_var});
366 if (vector_var) {
367 vector_vars.push(op->name);
368 }
369 result = op->body;
370 op = result.as<LetStmt>();
371 }
372
373 result = mutate(result);
374
375 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
376 if (it->vector_var) {
377 vector_vars.pop(it->op->name);
378 }
379 result = LetStmt::make(it->op->name, it->op->value, result);
380 }
381 return result;
382 }
383
384 Stmt visit(const Realize *op) override {
385 if (op->name == func) {
386 debug(3) << "Finding compute predicate for " << op->name << "\n";
387 PredicateFinder find_compute(op->name, true);
388 op->body.accept(&find_compute);
389
390 debug(3) << "Simplifying compute predicate for " << op->name << ": " << find_compute.predicate << "\n";
391 Expr compute_predicate = simplify(common_subexpression_elimination(find_compute.predicate));
392
393 debug(3) << "Compute predicate for " << op->name << " : " << compute_predicate << "\n";
394
395 if (expr_uses_vars(compute_predicate, vector_vars)) {
396 // Don't try to skip stages if the predicate may vary
397 // per lane. This will just unvectorize the
398 // production, which is probably contrary to the
399 // intent of the user.
400 compute_predicate = const_true();
401 }
402
403 if (!is_const_one(compute_predicate)) {
404
405 debug(3) << "Finding allocate predicate for " << op->name << "\n";
406 PredicateFinder find_alloc(op->name, false);
407 op->body.accept(&find_alloc);
408 debug(3) << "Simplifying allocate predicate for " << op->name << "\n";
409 Expr alloc_predicate = simplify(common_subexpression_elimination(find_alloc.predicate));
410
411 debug(3) << "Allocate predicate for " << op->name << " : " << alloc_predicate << "\n";
412
413 ProductionGuarder g(op->name, compute_predicate, alloc_predicate);
414 Stmt body = g.mutate(op->body);
415
416 debug(3) << "Done guarding computation for " << op->name << "\n";
417
418 return Realize::make(op->name, op->types, op->memory_type, op->bounds,
419 alloc_predicate, body);
420 } else {
421 return IRMutator::visit(op);
422 }
423 } else {
424 return IRMutator::visit(op);
425 }
426 }

429// Find Funcs where at least one of the consume nodes only uses the
430// Func conditionally. We may want to guard the production of these
431// Funcs.
432class MightBeSkippable : public IRVisitor {
433
434 using IRVisitor::visit;
435
436 bool in_conditional_stmt{false};
437
438 void visit(const Call *op) override {
439 IRVisitor::visit(op);
440 if (op->call_type == Call::Halide) {
441 unconditionally_used.insert(op->name);
442 }
443 }
444
445 void visit(const IfThenElse *op) override {
446 op->condition.accept(this);
447
448 std::set<string> old;
449 unconditionally_used.swap(old);
450
451 ScopedValue<bool> old_in_conditional(in_conditional_stmt, true);
452 op->then_case.accept(this);
453
454 std::set<string> used_in_true;
455 used_in_true.swap(unconditionally_used);
456 if (op->else_case.defined()) {
457 op->else_case.accept(this);
458 }
459
460 // Take the set intersection of the true and false paths, and add them to the set.
461 std::set_intersection(used_in_true.begin(), used_in_true.end(),
462 unconditionally_used.begin(), unconditionally_used.end(),
463 std::inserter(old, old.begin()));
464
465 unconditionally_used.swap(old);
466 }
467
468 void visit(const Select *op) override {
469 op->condition.accept(this);
470
471 std::set<string> old;
472 unconditionally_used.swap(old);
473
474 op->true_value.accept(this);
475 std::set<string> used_in_true;
476 used_in_true.swap(unconditionally_used);
477
478 op->false_value.accept(this);
479
480 // Again, take the set intersection
481 std::set_intersection(used_in_true.begin(), used_in_true.end(),
482 unconditionally_used.begin(), unconditionally_used.end(),
483 std::inserter(old, old.begin()));
484
485 unconditionally_used.swap(old);
486 }
487
488 void visit(const ProducerConsumer *op) override {
489 if (!op->is_producer) {
490 op->body.accept(this);
491 if (!unconditionally_used.count(op->name) || in_conditional_stmt) {
492 // This Func has a least one consume clause in which
493 // it is only used conditionally.
494 candidates.insert(op->name);
495 }
496 } else {
497 IRVisitor::visit(op);
498 // Calls inside the produce don't count - that's the block of code we intend to guard.
499 unconditionally_used.erase(op->name);
500 }
501 }
502
503 set<string> unconditionally_used;
504
505public:
506 set<string> candidates;
507};

511Stmt skip_stages(Stmt stmt, const vector<string> &order) {
512 // Don't consider the last stage, because it's the output, so it's
513 // never skippable.
514 MightBeSkippable check;
515 stmt.accept(&check);
516 for (size_t i = order.size() - 1; i > 0; i--) {
517 debug(2) << "skip_stages checking " << order[i - 1] << "\n";
518 if (check.candidates.count(order[i - 1])) {
519 debug(2) << "skip_stages can skip " << order[i - 1] << "\n";
520 StageSkipper skipper(order[i - 1]);
521 Stmt new_stmt = skipper.mutate(stmt);
522 if (!new_stmt.same_as(stmt)) {
523 // Might have made earlier stages skippable too
524 new_stmt.accept(&check);
525 }
526 stmt = new_stmt;
527 }
528 }
529 return stmt;
530}
