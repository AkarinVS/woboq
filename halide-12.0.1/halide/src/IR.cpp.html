414int32_t Allocate::constant_allocation_size(const std::vector<Expr> &extents, const std::string &name) {
415 int64_t result = 1;
416
417 for (size_t i = 0; i < extents.size(); i++) {
418 if (const IntImm *int_size = extents[i].as<IntImm>()) {
419 // Check if the individual dimension is > 2^31 - 1. Not
420 // currently necessary because it's an int32_t, which is
421 // always smaller than 2^31 - 1. If we ever upgrade the
422 // type of IntImm but not the maximum allocation size, we
423 // should re-enable this.
424 /*
425 if ((int64_t)int_size->value > (((int64_t)(1)<<31) - 1)) {
426 user_error
427 << "Dimension " << i << " for allocation " << name << " has size " <<
428 int_size->value << " which is greater than 2^31 - 1.";
429 }
430 */
431 result *= int_size->value;
432 if (result > (static_cast<int64_t>(1) << 31) - 1) {
433 user_error
434 << "Total size for allocation " << name
435 << " is constant but exceeds 2^31 - 1.\n";
436 }
437 } else {
438 return 0;
439 }
440 }
441
442 return static_cast<int32_t>(result);
443}

445int32_t Allocate::constant_allocation_size() const {
446 return Allocate::constant_allocation_size(extents, name);
447}

502Stmt Block::make(Stmt first, Stmt rest) {
503 internal_assert(first.defined()) << "Block of undefined\n";
504 internal_assert(rest.defined()) << "Block of undefined\n";
505
506 Block *node = new Block;
507
508 if (const Block *b = first.as<Block>()) {
509 // Use a canonical block nesting order
510 node->first = b->first;
511 node->rest = Block::make(b->rest, std::move(rest));
512 } else {
513 node->first = std::move(first);
514 node->rest = std::move(rest);
515 }
516
517 return node;
518}

520Stmt Block::make(const std::vector<Stmt> &stmts) {
521 if (stmts.empty()) {
522 return Stmt();
523 }
524 Stmt result = stmts.back();
525 for (size_t i = stmts.size() - 1; i > 0; i--) {
526 result = Block::make(stmts[i - 1], result);
527 }
528 return result;
529}

568Expr Call::make(const Function &func, const std::vector<Expr> &args, int idx) {
569 internal_assert(idx >= 0 &&

575 func.get_contents(), idx, Buffer<>(), Parameter());

672Expr Call::make(Type type, const std::string &name, const std::vector<Expr> &args, CallType call_type,
673 FunctionPtr func, int value_index,
674 Buffer<> image, Parameter param) {
675 if (name == intrinsic_op_names[Call::prefetch] && call_type == Call::Intrinsic) {
676 internal_assert(args.size() % 2 == 0)
677 << "Number of args to a prefetch call should be even: {base, offset, extent0, stride0, extent1, stride1, ...}\n";
678 }
679 for (size_t i = 0; i < args.size(); i++) {
680 internal_assert(args[i].defined()) << "Call of " << name << " with argument " << i << " undefined.\n";

684 internal_assert(args[i].type() == Int(32))
685 << "Args to call to halide function must be type Int(32)\n";
686 }
687 } else if (call_type == Image) {
688 internal_assert((param.defined() || image.defined()))
689 << "Call node to undefined image\n";
690 for (size_t i = 0; i < args.size(); i++) {
691 internal_assert(args[i].type() == Int(32))
692 << "Args to load from image must be type Int(32)\n";
693 }
694 }
695
696 Call *node = new Call;
697 node->type = type;
698 node->name = name;
699 node->args = args;
700 node->call_type = call_type;
701 node->func = std::move(func);
702 node->value_index = value_index;
703 node->image = std::move(image);
704 node->param = std::move(param);
705 return node;
706}

719Expr Shuffle::make(const std::vector<Expr> &vectors,
720 const std::vector<int> &indices) {
721 internal_assert(!vectors.empty()) << "Shuffle of zero vectors.\n";
722 internal_assert(!indices.empty()) << "Shufle with zero indices.\n";
723 Type element_ty = vectors.front().type().element_of();
724 int input_lanes = 0;
725 for (const Expr &i : vectors) {
726 internal_assert(i.type().element_of() == element_ty) << "Shuffle of vectors of mismatched types.\n";
727 input_lanes += i.type().lanes();
728 }
729 for (int i : indices) {
730 internal_assert(0 <= i && i < input_lanes) << "Shuffle vector index out of range: " << i << "\n";
731 }
732
733 Shuffle *node = new Shuffle;
734 node->type = element_ty.with_lanes((int)indices.size());
735 node->vectors = vectors;
736 node->indices = indices;
737 return node;
738}

740Expr Shuffle::make_interleave(const std::vector<Expr> &vectors) {
741 internal_assert(!vectors.empty()) << "Interleave of zero vectors.\n";
742
743 if (vectors.size() == 1) {
744 return vectors.front();
745 }
746
747 int lanes = vectors.front().type().lanes();
748
749 for (const Expr &i : vectors) {
750 internal_assert(i.type().lanes() == lanes)
751 << "Interleave of vectors with different sizes.\n";
752 }
753
754 std::vector<int> indices;
755 for (int i = 0; i < lanes; i++) {
756 for (int j = 0; j < (int)vectors.size(); j++) {
757 indices.push_back(j * lanes + i);
758 }
759 }
760
761 return make(vectors, indices);
762}
763
764Expr Shuffle::make_concat(const std::vector<Expr> &vectors) {
765 internal_assert(!vectors.empty()) << "Concat of zero vectors.\n";
766
767 if (vectors.size() == 1) {
768 return vectors.front();
769 }
770
771 std::vector<int> indices;
772 int lane = 0;
773 for (int i = 0; i < (int)vectors.size(); i++) {
774 for (int j = 0; j < vectors[i].type().lanes(); j++) {
775 indices.push_back(lane++);
776 }
777 }
778
779 return make(vectors, indices);
780}

782Expr Shuffle::make_broadcast(Expr vector, int factor) {
783 std::vector<int> indices(factor * vector.type().lanes());
784 for (int ix = 0; ix < factor; ix++) {
785 std::iota(indices.begin() + ix * vector.type().lanes(),
786 indices.begin() + (ix + 1) * vector.type().lanes(), 0);
787 }
788
789 return make({std::move(vector)}, indices);
790}

792Expr Shuffle::make_slice(Expr vector, int begin, int stride, int size) {
793 if (begin == 0 && size == vector.type().lanes() && stride == 1) {
794 return vector;
795 }
796
797 std::vector<int> indices;
798 for (int i = 0; i < size; i++) {
799 indices.push_back(begin + i * stride);
800 }
801
802 return make({std::move(vector)}, indices);
803}

809bool Shuffle::is_broadcast() const {
810 int lanes = indices.size();
811 int factor = broadcast_factor();
812 if (factor == 0 || factor >= lanes) {
813 return false;
814 }
815 int broadcasted_lanes = lanes / factor;
816
817 if (broadcasted_lanes < 2 || broadcasted_lanes >= lanes || lanes % broadcasted_lanes != 0) {
818 return false;
819 }
820 for (int i = 0; i < lanes; i++) {
821 if (indices[i % broadcasted_lanes] != indices[i]) {
822 return false;
823 }
824 }
825 return true;
826}
827
828int Shuffle::broadcast_factor() const {
829 int lanes = indices.size();
830 int broadcasted_lanes = 0;
831 for (; broadcasted_lanes < lanes; broadcasted_lanes++) {
832 if (indices[broadcasted_lanes] != broadcasted_lanes) {
833 break;
834 }
835 }
836 if (broadcasted_lanes > 0) {
837 return lanes / broadcasted_lanes;
838 } else {
839 return 0;
840 }
841}
842
843bool Shuffle::is_interleave() const {
844 int lanes = vectors.front().type().lanes();
845
846 // Don't consider concat of scalars as an interleave.
847 if (lanes == 1) {
848 return false;
849 }
850
851 for (const Expr &i : vectors) {
852 if (i.type().lanes() != lanes) {
853 return false;
854 }
855 }
856
857 // Require that we are a complete interleaving.
858 if (lanes * vectors.size() != indices.size()) {
859 return false;
860 }
861
862 for (int i = 0; i < (int)vectors.size(); i++) {
863 for (int j = 0; j < lanes; j++) {
864 if (indices[j * (int)vectors.size() + i] != i * lanes + j) {
865 return false;
866 }
867 }
868 }
869
870 return true;
871}

884Expr VectorReduce::make(VectorReduce::Operator op,
885 Expr vec,
886 int lanes) {
887 if (vec.type().is_bool()) {
888 internal_assert(op == VectorReduce::And || op == VectorReduce::Or)
889 << "The only legal operators for VectorReduce on a Bool"
890 << "vector are VectorReduce::And and VectorReduce::Or\n";
891 }
892 internal_assert(!vec.type().is_handle()) << "VectorReduce of handle type";
893 // Check the output lanes is a factor of the input lanes. They can
894 // also both be zero if we're constructing a wildcard expression.
895 internal_assert((lanes == 0 && vec.type().lanes() == 0) ||
896 (lanes != 0 && (vec.type().lanes() % lanes == 0)))
897 << "Vector reduce output lanes must be a divisor of the number of lanes in the argument "
898 << lanes << " " << vec.type().lanes() << "\n";
899 VectorReduce *node = new VectorReduce;
900 node->type = vec.type().with_lanes(lanes);
901 node->op = op;
902 node->value = std::move(vec);
903 return node;
904}

908// Helper function to determine if a sequence of indices is a
909// contiguous ramp.
910bool is_ramp(const std::vector<int> &indices, int stride = 1) {
911 for (size_t i = 0; i + 1 < indices.size(); i++) {
912 if (indices[i + 1] != indices[i] + stride) {
913 return false;
914 }
915 }
916 return true;

921bool Shuffle::is_concat() const {
922 size_t input_lanes = 0;
923 for (const Expr &i : vectors) {
924 input_lanes += i.type().lanes();
925 }
926
927 // A concat is a ramp where the output has the same number of
928 // lanes as the input.
929 return indices.size() == input_lanes && is_ramp(indices);
930}

932bool Shuffle::is_slice() const {
933 size_t input_lanes = 0;
934 for (const Expr &i : vectors) {
935 input_lanes += i.type().lanes();
936 }
937
938 // A slice is a ramp where the output does not contain all of the
939 // lanes of the input.
940 return indices.size() < input_lanes && is_ramp(indices, slice_stride());
