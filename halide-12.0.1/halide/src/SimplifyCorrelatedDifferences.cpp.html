22class SimplifyCorrelatedDifferences : public IRMutator {
23 using IRMutator::visit;
24
25 string loop_var;
26
27 Scope<ConstantInterval> monotonic;
28
29 struct OuterLet {
30 string name;
31 Expr value;
32 bool may_substitute;
33 };
34 vector<OuterLet> lets;
35
36 template<typename LetStmtOrLet, typename StmtOrExpr>
37 StmtOrExpr visit_let(const LetStmtOrLet *op) {
38 // Visit an entire chain of lets in a single method to conserve stack space.
39 struct Frame {
40 const LetStmtOrLet *op;
41 ScopedBinding<ConstantInterval> binding;
42 Expr new_value;
43 Frame(const LetStmtOrLet *op, const string &loop_var, Scope<ConstantInterval> &scope)
44 : op(op),
45 binding(scope, op->name, derivative_bounds(op->value, loop_var, scope)) {
46 }
47 Frame(const LetStmtOrLet *op)
48 : op(op) {
49 }
50 };
51 std::vector<Frame> frames;
52 StmtOrExpr result;
53
54 // Note that we must add *everything* that depends on the loop
55 // var to the monotonic scope and the list of lets, even
56 // things which we can never substitute in (e.g. impure
57 // things). This is for two reasons. First this pass could be
58 // used at a time when we still have nested lets under the
59 // same name. If we decide not to add an inner let, but do add
60 // the outer one, then later references to it will be
61 // incorrect. Second, if we don't add something that happens
62 // to be non-monotonic, then derivative_bounds finds a variable
63 // that references it in a later let, it will think it's a
64 // constant, not an unknown.
65 do {
66 result = op->body;
67 if (loop_var.empty()) {
68 frames.emplace_back(op);
69 continue;
70 }
71
72 bool pure = is_pure(op->value);
73 if (!pure || expr_uses_vars(op->value, monotonic) || monotonic.contains(op->name)) {
74 frames.emplace_back(op, loop_var, monotonic);
75 Expr new_value = mutate(op->value);
76 bool may_substitute_in = new_value.type() == Int(32) && pure;
77 lets.emplace_back(OuterLet{op->name, new_value, may_substitute_in});
78 frames.back().new_value = std::move(new_value);
79 } else {
80 // Pure and constant w.r.t the loop var. Doesn't
81 // shadow any outer thing already in the monotonic
82 // scope.
83 frames.emplace_back(op);
84 }
85 } while ((op = result.template as<LetStmtOrLet>()));
86
87 result = mutate(result);
88
89 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
90 if (it->new_value.defined()) {
91 result = LetStmtOrLet::make(it->op->name, it->new_value, result);
92 } else {
93 result = LetStmtOrLet::make(it->op->name, it->op->value, result);
94 }
95 if (it->binding.bound()) {
96 lets.pop_back();
97 }
98 }
99
100 return result;
101 }
102
103 Expr visit(const Let *op) override {
104 return visit_let<Let, Expr>(op);
105 }
106
107 Stmt visit(const LetStmt *op) override {
108 return visit_let<LetStmt, Stmt>(op);
109 }
110
111 Stmt visit(const For *op) override {
112 Stmt s = op;
113 // This is unfortunately quadratic in maximum loop nesting depth
114 if (loop_var.empty()) {
115 decltype(monotonic) tmp_monotonic;
116 decltype(lets) tmp_lets;
117 tmp_monotonic.swap(monotonic);
118 tmp_lets.swap(lets);
119 loop_var = op->name;
120 {
121 ScopedBinding<ConstantInterval> bind(monotonic, loop_var, ConstantInterval::single_point(1));
122 s = IRMutator::visit(op);
123 }
124 loop_var.clear();
125 tmp_monotonic.swap(monotonic);
126 tmp_lets.swap(lets);
127 }
128 s = IRMutator::visit(s.as<For>());
129 return s;
130 }

132 class PartiallyCancelDifferences : public IRMutator {
133 using IRMutator::visit;
134
135 // Symbols used by rewrite rules
136 IRMatcher::Wild<0> x;
137 IRMatcher::Wild<1> y;
138 IRMatcher::Wild<2> z;
139 IRMatcher::WildConst<0> c0;
140 IRMatcher::WildConst<1> c1;
141
142 Expr visit(const Sub *op) override {
143
144 Expr a = mutate(op->a), b = mutate(op->b);
145
146 // Partially cancel terms in correlated differences of
147 // various kinds to get tighter bounds. We assume any
148 // correlated term has already been pulled leftmost by
149 // solve_expression.
150 if (op->type == Int(32)) {
151 auto rewrite = IRMatcher::rewriter(IRMatcher::sub(a, b), op->type);
152 if (
153 // Differences of quasi-affine functions
154 rewrite((x + y) / c0 - (x + z) / c0, ((x % c0) + y) / c0 - ((x % c0) + z) / c0) ||
155 rewrite(x / c0 - (x + z) / c0, 0 - ((x % c0) + z) / c0) ||
156 rewrite((x + y) / c0 - x / c0, ((x % c0) + y) / c0) ||
157
158 // truncated cones have a constant upper or lower
159 // bound that isn't apparent when expressed in the
160 // form in the LHS below
161 rewrite(min(x, c0) - max(x, c1), min(min(c0 - x, x - c1), fold(min(0, c0 - c1)))) ||
162 rewrite(max(x, c0) - min(x, c1), max(max(c0 - x, x - c1), fold(max(0, c0 - c1)))) ||
163 rewrite(min(x, y) - max(x, z), min(min(x, y) - max(x, z), 0)) ||
164 rewrite(max(x, y) - min(x, z), max(max(x, y) - min(x, z), 0)) ||
165
166 false) {
167 return rewrite.result;
168 }
169 }
170 return a - b;
171 }
172 };

174 Expr cancel_correlated_subexpression(Expr e, const Expr &a, const Expr &b, bool correlated) {
175 auto ma = is_monotonic(a, loop_var, monotonic);
176 auto mb = is_monotonic(b, loop_var, monotonic);
177
178 if ((ma == Monotonic::Increasing && mb == Monotonic::Increasing && correlated) ||
179 (ma == Monotonic::Decreasing && mb == Monotonic::Decreasing && correlated) ||
180 (ma == Monotonic::Increasing && mb == Monotonic::Decreasing && !correlated) ||
181 (ma == Monotonic::Decreasing && mb == Monotonic::Increasing && !correlated)) {
182
183 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
184 if (expr_uses_var(e, it->name)) {
185 if (!it->may_substitute) {
186 // We have to stop here. Can't continue
187 // because there might be an outer let with
188 // the same name that we *can* substitute in,
189 // and then inner uses will get the wrong
190 // value.
191 break;
192 }
193 }
194 e = Let::make(it->name, it->value, e);
195 }
196 e = common_subexpression_elimination(e);
197 e = solve_expression(e, loop_var).result;
198 e = PartiallyCancelDifferences().mutate(e);
199 e = simplify(e);
200
201 const bool check_non_monotonic = debug::debug_level() > 0 || get_compiler_logger() != nullptr;
202 if (check_non_monotonic &&
203 is_monotonic(e, loop_var) == Monotonic::Unknown) {
204 // Might be a missed simplification opportunity. Log to help improve the simplifier.
205 if (get_compiler_logger()) {
206 get_compiler_logger()->record_non_monotonic_loop_var(loop_var, e);
207 }
208 debug(1) << "Warning: expression is non-monotonic in loop variable "
209 << loop_var << ": " << e << "\n";
210 }
211 }
212 return e;
213 }

215 template<typename T>
216 Expr visit_binop(const T *op, bool correlated) {
217 Expr e = IRMutator::visit(op);
218 op = e.as<T>();
219 if (op == nullptr ||
220 op->a.type() != Int(32) ||
221 loop_var.empty()) {
222 return e;
223 } else {
224 // Bury the logic that doesn't depend on the template
225 // parameter in a separate function to save code size and
226 // reduce stack frame size in the recursive path.
227 return cancel_correlated_subexpression(e, op->a, op->b, correlated);
228 }
229 }
230
231 // Binary ops where it pays to cancel a correlated term on both
232 // sides. E.g. consider the x in:
233 //
234 // (x*3 + y)*2 - max(x*6, 0)))
235 //
236 // Both sides increase monotonically with x so interval arithmetic
237 // will overestimate the bounds. If we subtract x*6 from both
238 // sides we get:
239 //
240 // y*2 - max(0, x*-6)
241 //
242 // Now only one side depends on x and interval arithmetic becomes
243 // exact.
244 Expr visit(const Sub *op) override {
245 return visit_binop(op, true);
246 }
247
248 Expr visit(const LT *op) override {
249 return visit_binop(op, true);
250 }
251
252 Expr visit(const LE *op) override {
253 return visit_binop(op, true);
254 }
255
256 Expr visit(const GT *op) override {
257 return visit_binop(op, true);
258 }
259
260 Expr visit(const GE *op) override {
261 return visit_binop(op, true);
262 }
263
264 Expr visit(const EQ *op) override {
265 return visit_binop(op, true);
266 }
267
268 Expr visit(const NE *op) override {
269 return visit_binop(op, true);
270 }
271
272 // For add you actually want to cancel any anti-correlated term
273 // (e.g. x in (x*3 + y)*2 + max(x*-6, 0))
274 Expr visit(const Add *op) override {
275 return visit_binop(op, false);
276 }
