21vector<llvm::Type *> llvm_types(const Closure &closure, llvm::StructType *halide_buffer_t_type, LLVMContext &context) {
22 vector<llvm::Type *> res;
23 for (const auto &v : closure.vars) {
24 res.push_back(llvm_type_of(&context, v.second));
25 }
26 for (const auto &b : closure.buffers) {
27 res.push_back(llvm_type_of(&context, b.second.type)->getPointerTo());
28 res.push_back(halide_buffer_t_type->getPointerTo());
29 }
30 return res;
31}

43void pack_closure(llvm::StructType *type,
44 Value *dst,
45 const Closure &closure,
46 const Scope<Value *> &src,
47 llvm::StructType *halide_buffer_t_type,
48 IRBuilder<> *builder) {
49 // type, type of dst should be a pointer to a struct of the type returned by build_type
50 int idx = 0;
51 for (const auto &v : closure.vars) {
52 llvm::Type *t = type->elements()[idx];
53 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, dst, 0, idx++);
54 Value *val = src.get(v.first);
55 val = builder->CreateBitCast(val, t);
56 builder->CreateStore(val, ptr);
57 }
58 for (const auto &b : closure.buffers) {
59 // For buffers we pass through base address (the symbol with
60 // the same name as the buffer), and the .buffer symbol (GPU
61 // code might implicitly need it).
62 // FIXME: This dependence should be explicitly encoded in the IR.
63 {
64 llvm::Type *t = type->elements()[idx];
65 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, dst, 0, idx++);
66 Value *val = src.get(b.first);
67 val = builder->CreateBitCast(val, t);
68 builder->CreateStore(val, ptr);
69 }
70 {
71 llvm::PointerType *t = halide_buffer_t_type->getPointerTo();
72 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, dst, 0, idx++);
73 Value *val = nullptr;
74 if (src.contains(b.first + ".buffer")) {
75 val = src.get(b.first + ".buffer");
76 val = builder->CreateBitCast(val, t);
77 } else {
78 val = ConstantPointerNull::get(t);
79 }
80 builder->CreateStore(val, ptr);
81 }
82 }

85void unpack_closure(const Closure &closure,
86 Scope<Value *> &dst,
87 llvm::StructType *type,
88 Value *src,
89 IRBuilder<> *builder) {
90 // type, type of src should be a pointer to a struct of the type returned by build_type
91 int idx = 0;
92 for (const auto &v : closure.vars) {
93 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, src, 0, idx++);
94 LoadInst *load = builder->CreateLoad(ptr->getType()->getPointerElementType(), ptr);
95 dst.push(v.first, load);
96 load->setName(v.first);
97 }
98 for (const auto &b : closure.buffers) {
99 {
100 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, src, 0, idx++);
101 LoadInst *load = builder->CreateLoad(ptr->getType()->getPointerElementType(), ptr);
102 dst.push(b.first, load);
103 load->setName(b.first);
104 }
105 {
106 Value *ptr = builder->CreateConstInBoundsGEP2_32(type, src, 0, idx++);
107 LoadInst *load = builder->CreateLoad(ptr->getType()->getPointerElementType(), ptr);
108 dst.push(b.first + ".buffer", load);
109 load->setName(b.first + ".buffer");
110 }
111 }

114llvm::Type *llvm_type_of(LLVMContext *c, Halide::Type t) {
115 if (t.lanes() == 1) {
116 if (t.is_float() && !t.is_bfloat()) {
117 switch (t.bits()) {
118 case 16:
119 return llvm::Type::getHalfTy(*c);
120 case 32:
121 return llvm::Type::getFloatTy(*c);
122 case 64:
123 return llvm::Type::getDoubleTy(*c);
124 default:
125 internal_error << "There is no llvm type matching this floating-point bit width: " << t << "\n";
126 return nullptr;
127 }
128 } else if (t.is_handle()) {
129 return llvm::Type::getInt8PtrTy(*c);
130 } else {
131 return llvm::Type::getIntNTy(*c, t.bits());
132 }
133 } else {
134 llvm::Type *element_type = llvm_type_of(c, t.element_of());
135 return get_vector_type(element_type, t.lanes());
136 }

268Expr lower_int_uint_div(const Expr &a, const Expr &b) {
269 // Detect if it's a small int division
270 internal_assert(a.type() == b.type());
271 const int64_t *const_int_divisor = as_const_int(b);
272 const uint64_t *const_uint_divisor = as_const_uint(b);
273
274 Type t = a.type();
275 internal_assert(!t.is_float())
276 << "lower_int_uint_div is not meant to handle floating-point case.\n";
277
278 int shift_amount;
279 if (is_const_power_of_two_integer(b, &shift_amount) &&
280 (t.is_int() || t.is_uint())) {
281 return a >> make_const(UInt(a.type().bits()), shift_amount);
282 } else if (const_int_divisor &&
283 t.is_int() &&
284 (t.bits() == 8 || t.bits() == 16 || t.bits() == 32) &&
285 *const_int_divisor > 1 &&
286 ((t.bits() > 8 && *const_int_divisor < 256) || *const_int_divisor < 128)) {
287
288 int64_t multiplier;
289 int shift;
290 if (t.bits() == 32) {
291 multiplier = IntegerDivision::table_s32[*const_int_divisor][2];
292 shift = IntegerDivision::table_s32[*const_int_divisor][3];
293 } else if (t.bits() == 16) {
294 multiplier = IntegerDivision::table_s16[*const_int_divisor][2];
295 shift = IntegerDivision::table_s16[*const_int_divisor][3];
296 } else {
297 // 8 bit
298 multiplier = IntegerDivision::table_s8[*const_int_divisor][2];
299 shift = IntegerDivision::table_s8[*const_int_divisor][3];
300 }
301 Expr num = a;
302
303 // Make an all-ones mask if the numerator is negative
304 Type num_as_uint_t = num.type().with_code(Type::UInt);
305 Expr sign = cast(num_as_uint_t, num >> make_const(UInt(t.bits()), t.bits() - 1));
306
307 // Flip the numerator bits if the mask is high.
308 num = cast(num_as_uint_t, num);
309 num = num ^ sign;
310
311 // Multiply and keep the high half of the
312 // result, and then apply the shift.
313 Expr mult = make_const(num.type(), multiplier);
314 num = mul_shift_right(num, mult, shift + num.type().bits());
315
316 // Maybe flip the bits back again.
317 num = cast(a.type(), num ^ sign);
318
319 return num;
320 } else if (const_uint_divisor &&
321 t.is_uint() &&
322 (t.bits() == 8 || t.bits() == 16 || t.bits() == 32) &&
323 *const_uint_divisor > 1 && *const_uint_divisor < 256) {
324
325 int64_t method, multiplier, shift;
326 if (t.bits() == 32) {
327 method = IntegerDivision::table_u32[*const_uint_divisor][1];
328 multiplier = IntegerDivision::table_u32[*const_uint_divisor][2];
329 shift = IntegerDivision::table_u32[*const_uint_divisor][3];
330 } else if (t.bits() == 16) {
331 method = IntegerDivision::table_u16[*const_uint_divisor][1];
332 multiplier = IntegerDivision::table_u16[*const_uint_divisor][2];
333 shift = IntegerDivision::table_u16[*const_uint_divisor][3];
334 } else {
335 method = IntegerDivision::table_u8[*const_uint_divisor][1];
336 multiplier = IntegerDivision::table_u8[*const_uint_divisor][2];
337 shift = IntegerDivision::table_u8[*const_uint_divisor][3];
338 }
339
340 internal_assert(method != 0)
341 << "method 0 division is for powers of two and should have been handled elsewhere\n";
342 const Expr &num = a;
343
344 // Widen, multiply, narrow
345 Expr mult = make_const(num.type(), multiplier);
346 Expr val = mul_shift_right(num, mult, (method == 1 ? shift : 0) + num.type().bits());
347
348 if (method == 2) {
349 // Average with original numerator.
350 val = Call::make(val.type(), Call::sorted_avg, {val, num}, Call::PureIntrinsic);
351
352 // Do the final shift
353 if (shift) {
354 val = val >> make_const(UInt(t.bits()), shift);
355 }
356 }
357
358 return val;
359 } else {
360 return lower_euclidean_div(a, b);
361 }
362}

364Expr lower_int_uint_mod(const Expr &a, const Expr &b) {
365 // Detect if it's a small int modulus
366 const int64_t *const_int_divisor = as_const_int(b);
367 const uint64_t *const_uint_divisor = as_const_uint(b);
368
369 Type t = a.type();
370 internal_assert(!t.is_float())
371 << "lower_int_uint_div is not meant to handle floating-point case.\n";
372
373 int bits;
374 if (is_const_power_of_two_integer(b, &bits)) {
375 return a & simplify(b - 1);
376 } else if (const_int_divisor &&
377 t.is_int() &&
378 (t.bits() == 8 || t.bits() == 16 || t.bits() == 32) &&
379 *const_int_divisor > 1 &&
380 ((t.bits() > 8 && *const_int_divisor < 256) || *const_int_divisor < 128)) {
381 // We can use our fast signed integer division
382 return common_subexpression_elimination(a - (a / b) * b);
383 } else if (const_uint_divisor &&
384 t.is_uint() &&
385 (t.bits() == 8 || t.bits() == 16 || t.bits() == 32) &&
386 *const_uint_divisor > 1 && *const_uint_divisor < 256) {
387 // We can use our fast unsigned integer division
388 return common_subexpression_elimination(a - (a / b) * b);
389 } else {
390 // To match our definition of division, mod should be between 0
391 // and |b|.
392 return lower_euclidean_mod(a, b);
393 }

396std::pair<Expr, Expr> unsigned_long_div_mod_round_to_zero(Expr &num, const Expr &den,
397 const uint64_t *upper_bound) {
398 internal_assert(num.type() == den.type());
399 internal_assert(num.type().is_uint());
400 Type ty = num.type();
401 Expr q = make_zero(ty);
402 Expr leading_zeros = cast(ty, count_leading_zeros(den));
403 // Each iteration of the loop below checks for a bit in the result.
404 const int times = ty.bits();
405 int start = 1;
406 if (upper_bound) {
407 // Set start to times - (index of most significant bit in max_val)
408 // as for each iteration:
409 // (1 << shift) <= upper_bound
410 start = times;
411 uint64_t max_val = *upper_bound;
412 while (max_val >>= 1) {
413 --start;
414 }
415 debug(1) << "Max value for long division: " << *upper_bound
416 << ". Evaluate only first " << 1 + times - start << " bits.\n";
417 }
418 Expr r = num;
419 for (int i = start; i <= times; i++) {
420 // Check if the bit at 'shift' index should be set in the result.
421 int shift = times - i;
422 Expr shift_expr = make_const(ty, shift);
423 Expr new_r = r - (den << shift_expr);
424 // Don't drop any set bits from den after shift. The bit is set if
425 // den << shift is no more than remainder.
426 Expr bit_set = ((shift_expr <= leading_zeros) && r >= (den << shift_expr));
427 // Update the and the quotient.
428 r = select(bit_set, new_r, r);
429 q = select(bit_set, make_const(ty, uint64_t(1) << shift) | q, q);
430 }
431 return {q, r};
432}

434std::pair<Expr, Expr> long_div_mod_round_to_zero(const Expr &num, const Expr &den,
435 const uint64_t *max_abs) {
436 debug(1) << "Using long div: (num: " << num << "); (den: " << den << ")\n";
437 internal_assert(num.type() == den.type());
438 Expr abs_num = (num.type().is_int()) ? abs(num) : num;
439 Expr abs_den = (den.type().is_int()) ? abs(den) : den;
440 std::pair<Expr, Expr> qr = unsigned_long_div_mod_round_to_zero(abs_num, abs_den, max_abs);
441 Expr q = qr.first;
442 Expr r = qr.second;
443 // Correct the signs for quotient and remainder for signed integer division.
444 if (num.type().is_int()) {
445 Expr num_neg = num >> make_const(UInt(num.type().bits()), (num.type().bits() - 1));
446 Expr den_neg = den >> make_const(UInt(num.type().bits()), (num.type().bits() - 1));
447 q = cast(num.type(), q) * ((num_neg ^ den_neg) | 1);
448 r = cast(num.type(), r) * (num_neg | 1);
449 }
450 q = simplify(common_subexpression_elimination(q));
451 r = simplify(common_subexpression_elimination(r));
452 return {q, r};
453}

455Expr lower_euclidean_div(Expr a, Expr b) {
456 internal_assert(a.type() == b.type());
457
458 Expr q;
459
460 if (a.type().is_uint()) {
461 // IROperator's div_round_to_zero will replace this with a / b for
462 // unsigned ops, so create the intrinsic directly.
463 Expr b_is_const_zero = (b == 0);
464 if (!can_prove(!b_is_const_zero)) {
465 b = b | cast(a.type(), b_is_const_zero);
466 }
467 q = Call::make(a.type(), Call::div_round_to_zero, {a, b}, Call::Intrinsic);
468 q = select(b_is_const_zero, 0, q);
469 } else {
470 internal_assert(a.type().is_int());
471
472 // Signed integer division sucks. It should be defined such
473 // that it satisifies (a/b)*b + a%b = a, where 0 <= a%b < |b|,
474 // i.e. Euclidean division.
475 //
476 // We additionally define division by zero to be zero, and
477 // division of the most negative integer by -1 to be the most
478 // negative integer.
479
480 // See div_imp in IROperator.h for the C code we're trying to match.
481
482 Expr zero = make_zero(a.type());
483 Expr minus_one = make_const(a.type(), -1);
484
485 Expr a_neg = a >> make_const(UInt(a.type().bits()), (a.type().bits() - 1));
486 Expr b_neg = b >> make_const(UInt(b.type().bits()), (b.type().bits() - 1));
487 Expr b_zero = select(b == zero, minus_one, zero);
488
489 // Give the simplifier the chance to skip some of this nonsense
490 if (can_prove(b != zero)) {
491 b_zero = zero;
492 }
493 if (can_prove(a >= zero)) {
494 a_neg = zero;
495 } else if (can_prove(a < zero)) {
496 a_neg = minus_one;
497 }
498 if (can_prove(b >= zero)) {
499 b_neg = zero;
500 } else if (can_prove(b < zero)) {
501 b_neg = minus_one;
502 }
503
504 // If b is zero, set it to one instead to avoid faulting
505 b -= b_zero;
506 // If a is negative, add one to it to get the rounding to work out.
507 a -= a_neg;
508 // Do the C-style division
509 q = Call::make(a.type(), Call::div_round_to_zero, {a, b}, Call::Intrinsic);
510 // If a is negative, either add or subtract one, depending on
511 // the sign of b, to fix the rounding. This can't overflow,
512 // because we move the result towards zero in either case (we
513 // add zero or one when q is negative, and subtract zero or
514 // one when it's positive).
515 q += a_neg & (~b_neg - b_neg);
516 // Set the result to zero when b is zero
517 q = q & ~b_zero;
518 }
519
520 q = simplify(common_subexpression_elimination(q));
521
522 return q;
523}

525Expr lower_euclidean_mod(Expr a, Expr b) {
526 Expr q;
527
528 if (a.type().is_uint()) {
529 Expr b_is_const_zero = (b == 0);
530 if (!can_prove(!b_is_const_zero)) {
531 b = b | cast(a.type(), b_is_const_zero);
532 }
533 q = Call::make(a.type(), Call::mod_round_to_zero, {a, b}, Call::Intrinsic);
534 q = select(b_is_const_zero, make_zero(a.type()), q);
535 } else {
536 internal_assert(a.type().is_int());
537
538 Expr zero = make_zero(a.type());
539 Expr minus_one = make_const(a.type(), -1);
540
541 Expr a_neg = a >> make_const(UInt(a.type().bits()), (a.type().bits() - 1));
542 Expr b_neg = b >> make_const(UInt(a.type().bits()), (a.type().bits() - 1));
543 Expr b_zero = select(b == zero, minus_one, zero);
544
545 // Give the simplifier the chance to skip some of this nonsense
546 if (can_prove(b != zero)) {
547 b_zero = zero;
548 }
549 if (can_prove(a >= zero)) {
550 a_neg = zero;
551 } else if (can_prove(a < zero)) {
552 a_neg = minus_one;
553 }
554 if (can_prove(b >= zero)) {
555 b_neg = zero;
556 } else if (can_prove(b < zero)) {
557 b_neg = minus_one;
558 }
559
560 // If a is negative, add one to get the rounding to work out
561 a -= a_neg;
562 // Do the mod, avoiding taking mod by zero
563 q = Call::make(a.type(), Call::mod_round_to_zero, {a, (b | b_zero)}, Call::Intrinsic);
564 // If a is negative, we either need to add b - 1 to the
565 // result, or -b - 1, depending on the sign of b.
566 q += (a_neg & ((b ^ b_neg) + ~b_neg));
567 // If b is zero, return zero by masking off the current result.
568 q = q & ~b_zero;
569 }
570
571 q = simplify(common_subexpression_elimination(q));
572
573 return q;
574}

576Expr lower_signed_shift_left(const Expr &a, const Expr &b) {
577 internal_assert(b.type().is_int());
578 const int64_t *const_int_b = as_const_int(b);
579 if (const_int_b) {
580 Expr val;
581 const uint64_t b_unsigned = std::abs(*const_int_b);
582 if (*const_int_b >= 0) {
583 val = a << make_const(UInt(a.type().bits()), b_unsigned);
584 } else if (*const_int_b < 0) {
585 val = a >> make_const(UInt(a.type().bits()), b_unsigned);
586 }
587 return common_subexpression_elimination(val);
588 } else {
589 // The abs() below uses Halide's abs operator. This eliminates the overflow
590 // case for the most negative value because its result is unsigned.
591 Expr b_unsigned = abs(b);
592 Expr val = select(b >= 0, a << b_unsigned, a >> b_unsigned);
593 return common_subexpression_elimination(val);
594 }
595}

597Expr lower_signed_shift_right(const Expr &a, const Expr &b) {
598 internal_assert(b.type().is_int());
599 const int64_t *const_int_b = as_const_int(b);
600 if (const_int_b) {
601 Expr val;
602 const uint64_t b_unsigned = std::abs(*const_int_b);
603 if (*const_int_b >= 0) {
604 val = a >> make_const(UInt(a.type().bits()), b_unsigned);
605 } else if (*const_int_b < 0) {
606 val = a << make_const(UInt(a.type().bits()), b_unsigned);
607 }
608 return common_subexpression_elimination(val);
609 } else {
610 // The abs() below uses Halide's abs operator. This eliminates the overflow
611 // case for the most negative value because its result is unsigned.
612 Expr b_unsigned = abs(b);
613 Expr val = select(b >= 0, a >> b_unsigned, a << b_unsigned);
614 return common_subexpression_elimination(val);
615 }
616}

618Expr lower_mux(const Call *mux) {
619 internal_assert(mux->args.size() >= 2);
620 Expr equiv = mux->args.back();
621 Expr index = mux->args[0];
622 int num_vals = (int)mux->args.size() - 1;
623 for (int i = num_vals - 1; i >= 0; i--) {
624 equiv = select(index == make_const(index.type(), i), mux->args[i + 1], equiv);
625 }
626 return equiv;
627}

629bool get_md_bool(llvm::Metadata *value, bool &result) {
630 if (!value) {
631 return false;
632 }
633 llvm::ConstantAsMetadata *cam = llvm::cast<llvm::ConstantAsMetadata>(value);
634 if (!cam) {
635 return false;
636 }
637 llvm::ConstantInt *c = llvm::cast<llvm::ConstantInt>(cam->getValue());
638 if (!c) {
639 return false;
640 }
641 result = !c->isZero();
642 return true;
643}

646 if (!value) {
647 result = "";
648 return false;

650 llvm::MDString *c = llvm::dyn_cast<llvm::MDString>(value);
651 if (c) {
652 result = c->getString().str();
653 return true;
654 }
655 return false;

658void get_target_options(const llvm::Module &module, llvm::TargetOptions &options, std::string &mcpu, std::string &mattrs) {
659 bool use_soft_float_abi = false;
660 get_md_bool(module.getModuleFlag("halide_use_soft_float_abi"), use_soft_float_abi);
661 get_md_string(module.getModuleFlag("halide_mcpu"), mcpu);
662 get_md_string(module.getModuleFlag("halide_mattrs"), mattrs);
663 std::string mabi;
664 get_md_string(module.getModuleFlag("halide_mabi"), mabi);
665 bool use_pic = true;
666 get_md_bool(module.getModuleFlag("halide_use_pic"), use_pic);
667
668 bool per_instruction_fast_math_flags = false;
669 get_md_bool(module.getModuleFlag("halide_per_instruction_fast_math_flags"), per_instruction_fast_math_flags);
670
671 options = llvm::TargetOptions();
672 options.AllowFPOpFusion = per_instruction_fast_math_flags ? llvm::FPOpFusion::Strict : llvm::FPOpFusion::Fast;
673 options.UnsafeFPMath = !per_instruction_fast_math_flags;
674 options.NoInfsFPMath = !per_instruction_fast_math_flags;
675 options.NoNaNsFPMath = !per_instruction_fast_math_flags;
676 options.HonorSignDependentRoundingFPMathOption = !per_instruction_fast_math_flags;
677 options.NoZerosInBSS = false;
678 options.GuaranteedTailCallOpt = false;
679 options.StackAlignmentOverride = 0;
680 options.FunctionSections = true;
681 options.UseInitArray = true;
682 options.FloatABIType =
683 use_soft_float_abi ? llvm::FloatABI::Soft : llvm::FloatABI::Hard;
684 options.RelaxELFRelocations = false;
685 options.MCOptions.ABIName = mabi;
686}

714std::unique_ptr<llvm::TargetMachine> make_target_machine(const llvm::Module &module) {
715 std::string error_string;
716
717 const llvm::Target *llvm_target = llvm::TargetRegistry::lookupTarget(module.getTargetTriple(), error_string);
718 if (!llvm_target) {
719 std::cout << error_string << "\n";
720 llvm::TargetRegistry::printRegisteredTargetsForVersion(llvm::outs());
721 }
722 auto triple = llvm::Triple(module.getTargetTriple());
723 internal_assert(llvm_target) << "Could not create LLVM target for " << triple.str() << "\n";
724
725 llvm::TargetOptions options;
726 std::string mcpu = "";
727 std::string mattrs = "";
728 get_target_options(module, options, mcpu, mattrs);
729
730 bool use_pic = true;
731 get_md_bool(module.getModuleFlag("halide_use_pic"), use_pic);
732
733 bool use_large_code_model = false;
734 get_md_bool(module.getModuleFlag("halide_use_large_code_model"), use_large_code_model);
735
736 auto *tm = llvm_target->createTargetMachine(module.getTargetTriple(),
737 mcpu, mattrs,
738 options,
739 use_pic ? llvm::Reloc::PIC_ : llvm::Reloc::Static,
740 use_large_code_model ? llvm::CodeModel::Large : llvm::CodeModel::Small,
741 llvm::CodeGenOpt::Aggressive);
742 return std::unique_ptr<llvm::TargetMachine>(tm);
743}

751void embed_bitcode(llvm::Module *M, const string &halide_command) {
752 // Save llvm.compiler.used and remote it.
753 SmallVector<Constant *, 2> used_array;
754#if LLVM_VERSION >= 130
755 SmallVector<GlobalValue *, 4> used_globals;
756#else
757 SmallPtrSet<GlobalValue *, 4> used_globals;
758#endif
759 llvm::Type *used_element_type = llvm::Type::getInt8Ty(M->getContext())->getPointerTo(0);
760 GlobalVariable *used = collectUsedGlobalVariables(*M, used_globals, true);
761 for (auto *GV : used_globals) {
762 if (GV->getName() != "llvm.embedded.module" &&
763 GV->getName() != "llvm.cmdline") {
764 used_array.push_back(
765 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, used_element_type));
766 }
767 }
768 if (used) {
769 used->eraseFromParent();
770 }
771
772 // Embed the bitcode for the llvm module.
773 std::string data;
774 Triple triple(M->getTargetTriple());
775 // Create a constant that contains the bitcode.
776 llvm::raw_string_ostream OS(data);
777 llvm::WriteBitcodeToFile(*M, OS, /* ShouldPreserveUseListOrder */ true);
778 ArrayRef<uint8_t> module_data =
779 ArrayRef<uint8_t>((const uint8_t *)OS.str().data(), OS.str().size());
780
781 llvm::Constant *module_constant =
782 llvm::ConstantDataArray::get(M->getContext(), module_data);
783 llvm::GlobalVariable *GV = new llvm::GlobalVariable(
784 *M, module_constant->getType(), true, llvm::GlobalValue::PrivateLinkage,
785 module_constant);
786 GV->setSection((triple.getObjectFormat() == Triple::MachO) ? "__LLVM,__bitcode" : ".llvmbc");
787 used_array.push_back(
788 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, used_element_type));
789 if (llvm::GlobalVariable *old =
790 M->getGlobalVariable("llvm.embedded.module", true)) {
791 internal_assert(old->hasOneUse()) << "llvm.embedded.module can only be used once in llvm.compiler.used";
792 GV->takeName(old);
793 old->eraseFromParent();
794 } else {
795 GV->setName("llvm.embedded.module");
796 }
797
798 // Embed command-line options.
799 ArrayRef<uint8_t> command_line_data(const_cast<uint8_t *>(reinterpret_cast<const uint8_t *>(halide_command.data())),
800 halide_command.size());
801 llvm::Constant *command_line_constant =
802 llvm::ConstantDataArray::get(M->getContext(), command_line_data);
803 GV = new llvm::GlobalVariable(*M, command_line_constant->getType(), true,
804 llvm::GlobalValue::PrivateLinkage,
805 command_line_constant);
806
807 GV->setSection((triple.getObjectFormat() == Triple::MachO) ? "__LLVM,__cmdline" : ".llvmcmd");
808 used_array.push_back(
809 ConstantExpr::getPointerBitCastOrAddrSpaceCast(GV, used_element_type));
810 if (llvm::GlobalVariable *old =
811 M->getGlobalVariable("llvm.cmdline", true)) {
812 internal_assert(old->hasOneUse()) << "llvm.cmdline can only be used once in llvm.compiler.used";
813 GV->takeName(old);
814 old->eraseFromParent();
815 } else {
816 GV->setName("llvm.cmdline");
817 }
818
819 if (!used_array.empty()) {
820 // Recreate llvm.compiler.used.
821 ArrayType *ATy = ArrayType::get(used_element_type, used_array.size());
822 auto *new_used = new GlobalVariable(
823 *M, ATy, false, llvm::GlobalValue::AppendingLinkage,
824 llvm::ConstantArray::get(ATy, used_array), "llvm.compiler.used");
825 new_used->setSection("llvm.metadata");
826 }
827}
