13// Find a constant upper bound on the size of each thread-local allocation
14class BoundSmallAllocations : public IRMutator {
15 using IRMutator::visit;
16
17 // Track constant bounds
18 Scope<Interval> scope;
19
20 template<typename T, typename Body>
21 Body visit_let(const T *op) {
22 // Visit an entire chain of lets in a single method to conserve stack space.
23 struct Frame {
24 const T *op;
25 ScopedBinding<Interval> binding;
26 Frame(const T *op, Scope<Interval> &scope)
27 : op(op),
28 binding(scope, op->name, find_constant_bounds(op->value, scope)) {
29 }
30 };
31 std::vector<Frame> frames;
32 Body result;
33
34 do {
35 result = op->body;
36 frames.emplace_back(op, scope);
37 } while ((op = result.template as<T>()));
38
39 result = mutate(result);
40
41 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
42 result = T::make(it->op->name, it->op->value, result);
43 }
44
45 return result;
46 }
47
48 Stmt visit(const LetStmt *op) override {
49 return visit_let<LetStmt, Stmt>(op);
50 }
51
52 Expr visit(const Let *op) override {
53 return visit_let<Let, Expr>(op);
54 }
55
56 bool in_thread_loop = false;
57
58 DeviceAPI device_api = DeviceAPI::None;
59
60 Stmt visit(const For *op) override {
61 Interval min_bounds = find_constant_bounds(op->min, scope);
62 Interval max_bounds = find_constant_bounds(op->min + op->extent - 1, scope);
63 Interval b = Interval::make_union(min_bounds, max_bounds);
64 b.min = simplify(b.min);
65 b.max = simplify(b.max);
66 ScopedBinding<Interval> bind(scope, op->name, b);
67 bool new_in_thread_loop =
68 in_thread_loop || op->for_type == ForType::GPUThread;
69 ScopedValue<bool> old_in_thread_loop(in_thread_loop, new_in_thread_loop);
70 DeviceAPI new_device_api =
71 op->device_api == DeviceAPI::None ? device_api : op->device_api;
72 ScopedValue<DeviceAPI> old_device_api(device_api, new_device_api);
73 return IRMutator::visit(op);
74 }
75
76 bool must_be_constant(MemoryType memory_type) const {
77 return (memory_type == MemoryType::Register ||
78 (device_api == DeviceAPI::OpenGLCompute &&
79 memory_type == MemoryType::GPUShared));
80 }
81
82 Stmt visit(const Realize *op) override {
83 // Called pre-storage-flattening. At this point we just want
84 // to ensure any extents on allocations which *must* be
85 // constant *are* constant.
86 if (must_be_constant(op->memory_type)) {
87 Region region = op->bounds;
88 bool changed = false;
89 for (Range &r : region) {
90 Expr bound = find_constant_bound(r.extent, Direction::Upper, scope);
91 user_assert(bound.defined())
92 << "Was unable to infer constant upper bound on extent of allocation "
93 << op->name << ". Use Func::bound_extent to specify it manually.";
94 if (!bound.same_as(r.extent)) {
95 r.extent = bound;
96 changed = true;
97 }
98 }
99
100 Stmt body = mutate(op->body);
101 if (changed || !body.same_as(op->body)) {
102 return Realize::make(op->name, op->types, op->memory_type, region, op->condition, body);
103 } else {
104 return op;
105 }
106 } else {
107 return IRMutator::visit(op);
108 }
109 }
110
111 Stmt visit(const Allocate *op) override {
112 Expr total_extent = make_const(Int(64), 1);
113 for (const Expr &e : op->extents) {
114 total_extent *= e;
115 }
116 Expr bound = find_constant_bound(total_extent, Direction::Upper, scope);
117
118 if (!bound.defined() && must_be_constant(op->memory_type)) {
119 user_assert(op->memory_type != MemoryType::Register)
120 << "Allocation " << op->name << " has a dynamic size. "
121 << "Only fixed-size allocations can be stored in registers. "
122 << "Try storing on the heap or stack instead.";
123
124 user_assert(!(device_api == DeviceAPI::OpenGLCompute &&
125 op->memory_type == MemoryType::GPUShared))
126 << "Allocation " << op->name << " has a dynamic size. "
127 << "Only fixed-size allocations can be stored in shared memory "
128 << "in OpenGL compute shaders. Try storing in MemoryType::Heap "
129 << "instead.";
130 }
131
132 const int64_t *size_ptr = bound.defined() ? as_const_int(bound) : nullptr;
133 int64_t size = size_ptr ? *size_ptr : 0;
134
135 if (size_ptr && size == 0 && !op->new_expr.defined()) {
136 // This allocation is dead
137 return Allocate::make(op->name, op->type, op->memory_type, {0}, const_false(),
138 mutate(op->body), op->new_expr, op->free_function);
139 }
140
141 // 128 bytes is a typical minimum allocation size in
142 // halide_malloc. For now we are very conservative, and only
143 // round sizes up to a constant if they're smaller than that.
144 int malloc_overhead = 128 / op->type.bytes();
145 if (size_ptr &&
146 (in_thread_loop ||
147 (op->memory_type == MemoryType::Stack && can_allocation_fit_on_stack(size)) ||
148 must_be_constant(op->memory_type) ||
149 (op->memory_type == MemoryType::Auto && size <= malloc_overhead))) {
150 user_assert(size >= 0 && size < (int64_t)1 << 31)
151 << "Allocation " << op->name << " has a size greater than 2^31: " << bound << "\n";
152 return Allocate::make(op->name, op->type, op->memory_type, {(int32_t)size}, op->condition,
153 mutate(op->body), op->new_expr, op->free_function);
154 } else {
155 return IRMutator::visit(op);
156 }
157 }
158};

164}
