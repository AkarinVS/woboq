21Stmt call_extern_and_assert(const string &name, const vector<Expr> &args) {

31class FindBufferUsage : public IRVisitor {
32 using IRVisitor::visit;
33
34 void visit(const Load *op) override {
35 IRVisitor::visit(op);
36 if (op->name == buffer) {
37 devices_touched.insert(current_device_api);
38 }
39 }
40
41 void visit(const Store *op) override {
42 IRVisitor::visit(op);
43 if (op->name == buffer) {
44 devices_touched.insert(current_device_api);
45 devices_writing.insert(current_device_api);
46 }
47 }
48
49 bool is_buffer_var(const Expr &e) {
50 const Variable *var = e.as<Variable>();
51 return var && (var->name == buffer + ".buffer");
52 }
53
54 void visit(const Call *op) override {
55 if (op->is_intrinsic(Call::image_load)) {
56 internal_assert(!op->args.empty());
57 if (is_buffer_var(op->args[1])) {
58 devices_touched.insert(current_device_api);
59 }
60 for (size_t i = 0; i < op->args.size(); i++) {
61 if (i == 1) {
62 continue;
63 }
64 op->args[i].accept(this);
65 }
66 } else if (op->is_intrinsic(Call::image_store)) {
67 internal_assert(!op->args.empty());
68 if (is_buffer_var(op->args[1])) {
69 devices_touched.insert(current_device_api);
70 devices_writing.insert(current_device_api);
71 }
72 for (size_t i = 0; i < op->args.size(); i++) {
73 if (i == 1) {
74 continue;
75 }
76 op->args[i].accept(this);
77 }
78 } else if (op->is_intrinsic(Call::debug_to_file)) {
79 internal_assert(op->args.size() == 3);
80 if (is_buffer_var(op->args[2])) {
81 devices_touched.insert(current_device_api);
82 devices_writing.insert(current_device_api);
83 }
84 } else if (op->is_extern() && op->func.defined()) {
85 // This is a call to an extern stage
86 Function f(op->func);
87
88 internal_assert((f.extern_arguments().size() + f.outputs()) == op->args.size())
89 << "Mismatch between args size and extern_arguments size in call to "
90 << op->name << "\n";
91
92 // Check each buffer arg
93 for (size_t i = 0; i < op->args.size(); i++) {
94 if (is_buffer_var(op->args[i])) {
95 DeviceAPI extern_device_api = f.extern_function_device_api();
96 devices_touched_by_extern.insert(extern_device_api);
97 if (i >= f.extern_arguments().size()) {
98 // An output. The extern stage is responsible
99 // for dealing with any device transitions for
100 // inputs.
101 devices_touched.insert(extern_device_api);
102 devices_writing.insert(extern_device_api);
103 }
104 } else {
105 op->args[i].accept(this);
106 }
107 }
108 } else {
109 IRVisitor::visit(op);
110 }
111 }
112
113 void visit(const For *op) override {
114 internal_assert(op->device_api != DeviceAPI::Default_GPU)
115 << "A GPU API should have been selected by this stage in lowering\n";
116 DeviceAPI old = current_device_api;
117 if (op->device_api != DeviceAPI::None) {
118 current_device_api = op->device_api;
119 }
120 IRVisitor::visit(op);
121 current_device_api = old;
122 }
123
124 string buffer;
125 DeviceAPI current_device_api;
126
127public:
128 std::set<DeviceAPI> devices_writing, devices_touched;
129 // Any buffer passed to an extern stage may have had its dirty
130 // bits and device allocation messed with.
131 std::set<DeviceAPI> devices_touched_by_extern;
132
133 FindBufferUsage(const std::string &buf, DeviceAPI d)
134 : buffer(buf), current_device_api(d) {
135 }

138// Inject the device copies, mallocs, and dirty flag setting for a
139// single allocation. Sticks to the same loop level as the original
140// allocation and treats the stmt as a serial sequence of leaf
141// stmts. We walk this sequence of leaves, tracking what we know about
142// the buffer as we go, sniffing usage within each leaf using
143// FindBufferUsage, and injecting device buffer logic as needed.
144class InjectBufferCopiesForSingleBuffer : public IRMutator {
145 using IRMutator::visit;
146
147 // The buffer being managed
148 const string buffer;
149
150 const bool is_external;
151
152 MemoryType memory_type;
153
154 enum FlagState {
155 Unknown,
156 False,
157 True
158 };
159
160 struct State {
161 // What do we know about the dirty flags and the existence of a device allocation?
162 FlagState device_dirty = Unknown, host_dirty = Unknown, device_allocation_exists = Unknown;
163
164 // If it exists on a known device API, which device does it exist
165 // on? Meaningless if device_allocation_exists is not True.
166 DeviceAPI current_device = DeviceAPI::None;
167
168 void union_with(const State &other) {
169 if (device_dirty != other.device_dirty) {
170 device_dirty = Unknown;
171 }
172 if (host_dirty != other.host_dirty) {
173 host_dirty = Unknown;
174 }
175 if (device_allocation_exists != other.device_allocation_exists ||
176 other.current_device != current_device) {
177 device_allocation_exists = Unknown;
178 current_device = DeviceAPI::None;
179 }
180 }
181 } state;

183 Expr buffer_var() {
184 return Variable::make(type_of<struct halide_buffer_t *>(), buffer + ".buffer");
185 }

187 Stmt make_device_malloc(DeviceAPI target_device_api) {
188 Expr device_interface = make_device_interface_call(target_device_api, memory_type);
189 Stmt device_malloc = call_extern_and_assert("halide_device_malloc",
190 {buffer_var(), device_interface});
191 return device_malloc;
192 }

194 Stmt make_copy_to_host() {
195 return call_extern_and_assert("halide_copy_to_host", {buffer_var()});
196 }
197
198 Stmt make_copy_to_device(DeviceAPI target_device_api) {
199 Expr device_interface = make_device_interface_call(target_device_api, memory_type);
200 return call_extern_and_assert("halide_copy_to_device", {buffer_var(), device_interface});
201 }

203 Stmt make_host_dirty() {
204 return Evaluate::make(Call::make(Int(32), Call::buffer_set_host_dirty,
205 {buffer_var(), const_true()}, Call::Extern));
206 }
207
208 Stmt make_device_dirty() {
209 return Evaluate::make(Call::make(Int(32), Call::buffer_set_device_dirty,
210 {buffer_var(), const_true()}, Call::Extern));
211 }

213 Stmt make_device_free() {
214 return call_extern_and_assert("halide_device_free", {buffer_var()});

217 Stmt do_copies(Stmt s) {
218 // Sniff what happens to the buffer inside the stmt
219 FindBufferUsage finder(buffer, DeviceAPI::Host);
220 s.accept(&finder);
221
222 // Insert any appropriate copies/allocations before, and set
223 // dirty flags after. Do not recurse into the stmt.
224
225 // First figure out what happened
226 bool touched_on_host = finder.devices_touched.count(DeviceAPI::Host);
227 bool touched_on_device = finder.devices_touched.size() > (touched_on_host ? 1 : 0);
228 bool written_on_host = finder.devices_writing.count(DeviceAPI::Host);
229 bool written_on_device = finder.devices_writing.size() > (touched_on_host ? 1 : 0);
230
231 DeviceAPI touching_device = DeviceAPI::None;
232 for (DeviceAPI d : finder.devices_touched) {
233 // TODO: looks dubious, but removing causes crashes in correctness_debug_to_file
234 // with target=host-metal.
235 if (d == DeviceAPI::Host) {
236 continue;
237 }
238 internal_assert(touching_device == DeviceAPI::None)
239 << "Buffer " << buffer << " was touched on multiple devices within a single leaf Stmt!\n";
240 touching_device = d;
241 }
242
243 // Then figure out what to do
244 bool needs_device_malloc = (touched_on_device &&
245 (state.device_allocation_exists != True));
246
247 bool needs_device_flip = (state.device_allocation_exists != False &&
248 state.current_device != touching_device &&
249 state.current_device != DeviceAPI::None &&
250 touching_device != DeviceAPI::None &&
251 !is_external);
252
253 // TODO: If only written on device, and entirely clobbered on
254 // device, a copy-to-device is not actually necessary.
255 bool needs_copy_to_device = (touched_on_device &&
256 ((state.host_dirty != False) ||
257 needs_device_flip));
258
259 if (needs_copy_to_device) {
260 // halide_copy_to_device already does a halide_device_malloc if necessary.
261 needs_device_malloc = false;
262 }
263
264 // Device flips go via host memory
265 bool needs_copy_to_host = ((touched_on_host || needs_device_flip) &&
266 (state.device_dirty != False));
267
268 bool needs_host_dirty = (written_on_host &&
269 (state.host_dirty != True));
270
271 bool needs_device_dirty = (written_on_device &&
272 (state.device_dirty != True));
273
274 vector<Stmt> stmts;
275
276 // Then do it, updating what we know about the buffer
277 if (needs_copy_to_host) {
278 stmts.push_back(make_copy_to_host());
279 state.device_dirty = False;
280 }
281
282 // When flipping a buffer between devices, we need to free the
283 // old device memory before allocating the new one.
284 if (needs_device_flip) {
285 stmts.push_back(make_host_dirty());
286 stmts.push_back(make_device_free());
287 state.device_allocation_exists = False;
288 state.device_dirty = False;
289 }
290
291 if (needs_copy_to_device) {
292 stmts.push_back(make_copy_to_device(touching_device));
293 state.host_dirty = False;
294 state.device_allocation_exists = True;
295 state.current_device = touching_device;
296 }
297
298 if (needs_device_malloc) {
299 stmts.push_back(make_device_malloc(touching_device));
300 state.device_allocation_exists = True;
301 state.current_device = touching_device;
302 }
303
304 stmts.push_back(s);
305
306 if (needs_host_dirty) {
307 stmts.push_back(make_host_dirty());
308 state.host_dirty = True;
309 }
310
311 if (needs_device_dirty) {
312 stmts.push_back(make_device_dirty());
313 state.device_dirty = True;
314 }
315
316 s = Block::make(stmts);
317
318 if (!finder.devices_touched_by_extern.empty()) {
319 // This buffer was passed to an extern stage. Unless we
320 // explicitly marked it after the stmt ran, we no longer
321 // know the state of the dirty bits.
322 if (!needs_host_dirty) {
323 state.host_dirty = Unknown;
324 }
325 if (!needs_device_dirty) {
326 state.device_dirty = Unknown;
327 }
328 // Also, the extern stage may have gifted a host
329 // allocation, or flipped the buffer to another device.
330 state.device_allocation_exists = Unknown;
331 state.current_device = DeviceAPI::None;
332 }
333
334 return s;
335 }
336
337 // We want to break things down into a serial sequence of leaf
338 // stmts, and possibly do copies and update state around each
339 // leaf.
340 Stmt visit(const For *op) override {
341 FindBufferUsage finder(buffer, DeviceAPI::Host);
342 op->accept(&finder);
343 if (finder.devices_touched.size() > 1) {
344 // The state of the buffer going into the loop is the
345 // union of the state before the loop starts and the state
346 // after one iteration. Just forget everything we know.
347 state = State{};
348 Stmt s = IRMutator::visit(op);
349 // The state after analyzing the loop body might not be the
350 // true state if the loop ran for zero iterations. So
351 // forget everything again.
352 state = State{};
353 return s;
354 } else {
355 return do_copies(op);
356 }
357 }
358
359 Stmt visit(const Fork *op) override {
360 return do_copies(op);
361 }
362
363 Stmt visit(const Evaluate *op) override {
364 return do_copies(op);
365 }
366
367 Stmt visit(const LetStmt *op) override {
368 // If op->value uses the buffer, we need to treat this as a
369 // single leaf. Otherwise we can recurse.
370 FindBufferUsage finder(buffer, DeviceAPI::Host);
371 op->value.accept(&finder);
372 if (finder.devices_touched.empty() &&
373 finder.devices_touched_by_extern.empty()) {
374 return IRMutator::visit(op);
375 } else {
376 return do_copies(op);
377 }
378 }

380 Stmt visit(const AssertStmt *op) override {
381 return do_copies(op);

384 // Check if a stmt has any for loops (and hence possible device
385 // transitions).
386 class HasLoops : public IRVisitor {
387 using IRVisitor::visit;
388 void visit(const For *op) override {
389 result = true;
390 }
391
392 public:
393 bool result = false;
394 };
395
396 Stmt visit(const Block *op) override {
397 // If both sides of the block have no loops (and hence no
398 // device transitions), treat it as a single leaf. This stops
399 // host dirties from getting in between blocks of store stmts
400 // that could be interleaved.
401 HasLoops loops;
402 op->accept(&loops);
403 if (loops.result) {
404 return IRMutator::visit(op);
405 } else {
406 return do_copies(op);
407 }
408 }
409
410 Stmt visit(const Store *op) override {
411 return do_copies(op);
412 }
413
414 Stmt visit(const IfThenElse *op) override {
415 State old = state;
416 Stmt then_case = mutate(op->then_case);
417 State then_state = state;
418 state = old;
419 Stmt else_case = mutate(op->else_case);
420 state.union_with(then_state);
421 return IfThenElse::make(op->condition, then_case, else_case);
422 }
423
424public:
425 InjectBufferCopiesForSingleBuffer(const std::string &b, bool e, MemoryType m)
426 : buffer(b), is_external(e), memory_type(m) {
427 if (is_external) {
428 // The state of the buffer is totally unknown, which is
429 // the default constructor for this->state
430 } else {
431 // This is a fresh allocation
432 state.device_allocation_exists = False;
433 state.device_dirty = False;
434 state.host_dirty = False;
435 state.current_device = DeviceAPI::None;
436 }
437 }

511 // Inject the registration of a device destructor just after the
512 // .buffer symbol is defined (which is safely before the first

515 using IRMutator::visit;
516

518 if (op->name == buffer) {

538 // Find the let stmt that defines the .buffer and insert inside of
539 // it a combined host/dev allocation, a destructor registration,
540 // and an Allocate node that takes its host field from the
541 // .buffer.
542 class InjectCombinedAllocation : public IRMutator {
543 using IRMutator::visit;
544
545 Stmt visit(const LetStmt *op) override {
546 if (op->name == buffer + ".buffer") {
547 Expr buf = Variable::make(type_of<struct halide_buffer_t *>(), buffer + ".buffer");
548 Stmt body = op->body;
549
550 // The allocate node is innermost
551 Expr host = Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
552 body = Allocate::make(buffer, type, MemoryType::Heap, extents, condition, body,
553 host, "halide_device_host_nop_free");
554
555 // Then the destructor
556 Stmt destructor =
557 Evaluate::make(Call::make(Handle(), Call::register_destructor,
558 {Expr("halide_device_and_host_free_as_destructor"), buf},
559 Call::Intrinsic));
560 body = Block::make(destructor, body);
561
562 // Then the device_and_host malloc
563 Stmt device_malloc = call_extern_and_assert("halide_device_and_host_malloc",
564 {buf, device_interface});
565 if (!is_const_one(condition)) {
566 device_malloc = IfThenElse::make(condition, device_malloc);
567 }
568 body = Block::make(device_malloc, body);
569
570 // In the value, we want to use null for the initial value of the host field.
571 Expr value = substitute(buffer, reinterpret(Handle(), make_zero(UInt(64))), op->value);
572
573 // Rewrap the letstmt
574 return LetStmt::make(op->name, value, body);
575 } else {
576 return IRMutator::visit(op);
577 }
578 }
579
580 string buffer;
581 Type type;
582 vector<Expr> extents;
583 Expr condition;
584 Expr device_interface;
585
586 public:
587 InjectCombinedAllocation(string b, Type t, vector<Expr> e, Expr c, Expr d)
588 : buffer(std::move(b)), type(t), extents(std::move(e)),
589 condition(std::move(c)), device_interface(std::move(d)) {
590 }
591 };

593 class FreeAfterLastUse : public IRMutator {
594 Stmt last_use;
595 Stmt free_stmt;
596
597 public:
598 bool success = false;
599 using IRMutator::mutate;
600
601 Stmt mutate(const Stmt &s) override {
602 if (s.same_as(last_use)) {
603 internal_assert(!success);
604 success = true;
605 return Block::make(last_use, free_stmt);
606 } else {
607 return IRMutator::mutate(s);
608 }
609 }
610
611 FreeAfterLastUse(Stmt s, Stmt f)
612 : last_use(std::move(s)), free_stmt(std::move(f)) {
613 }
614 };
615
616 Stmt visit(const Allocate *op) override {
617 FindBufferUsage finder(op->name, DeviceAPI::Host);
618 op->body.accept(&finder);
619
620 bool touched_on_host = finder.devices_touched.count(DeviceAPI::Host);
621 bool touched_on_device = finder.devices_touched.size() > (touched_on_host ? 1 : 0);
622
623 if (!touched_on_device && finder.devices_touched_by_extern.empty()) {
624 // Boring.
625 return IRMutator::visit(op);
626 }
627
628 Stmt body = mutate(op->body);
629
630 InjectBufferCopiesForSingleBuffer injector(op->name, false, op->memory_type);
631 body = injector.mutate(body);
632
633 string buffer_name = op->name + ".buffer";
634 Expr buffer = Variable::make(Handle(), buffer_name);
635
636 // Device what type of allocation to make.
637
638 if (touched_on_host && finder.devices_touched.size() == 2) {
639 // Touched on a single device and the host. Use a combined allocation.
640 DeviceAPI touching_device = DeviceAPI::None;
641 for (DeviceAPI d : finder.devices_touched) {
642 if (d != DeviceAPI::Host) {
643 touching_device = d;
644 }
645 }
646
647 // Make a device_and_host_free stmt
648
649 FindLastUse last_use(op->name);
650 body.accept(&last_use);
651 if (last_use.last_use.defined()) {
652 Stmt device_free = call_extern_and_assert("halide_device_and_host_free", {buffer});
653 FreeAfterLastUse free_injecter(last_use.last_use, device_free);
654 body = free_injecter.mutate(body);
655 internal_assert(free_injecter.success);
656 }
657
658 Expr device_interface = make_device_interface_call(touching_device, op->memory_type);
659
660 return InjectCombinedAllocation(op->name, op->type, op->extents,
661 op->condition, device_interface)
662 .mutate(body);
663 } else {
664 // Only touched on host but passed to an extern stage, or
665 // only touched on device, or touched on multiple
666 // devices. Do separate device and host allocations.
667
668 // Add a device destructor
669 body = InjectDeviceDestructor(buffer_name).mutate(body);
670
671 // Make a device_free stmt
672
673 FindLastUse last_use(op->name);
674 body.accept(&last_use);
675 if (last_use.last_use.defined()) {
676 Stmt device_free = call_extern_and_assert("halide_device_free", {buffer});
677 FreeAfterLastUse free_injecter(last_use.last_use, device_free);
678 body = free_injecter.mutate(body);
679 internal_assert(free_injecter.success);
680 }
681
682 Expr condition = op->condition;
683 bool touched_on_one_device = !touched_on_host && finder.devices_touched.size() == 1 &&
684 (finder.devices_touched_by_extern.empty() ||
685 (finder.devices_touched_by_extern.size() == 1 &&
686 *(finder.devices_touched.begin()) == *(finder.devices_touched_by_extern.begin())));
687 if (touched_on_one_device) {
688 condition = const_false();
689 // There's no host allocation, so substitute any
690 // references to it (e.g. the one in the make_buffer
691 // call) with NULL.
692 body = substitute(op->name, reinterpret(Handle(), make_zero(UInt(64))), body);
693 }
694
695 return Allocate::make(op->name, op->type, op->memory_type, op->extents,
696 condition, body, op->new_expr, op->free_function);
697 }
698 }
699
700 Stmt visit(const For *op) override {
701 if (op->device_api != DeviceAPI::Host &&
702 op->device_api != DeviceAPI::None) {
703 // Don't enter device loops
704 return op;
705 } else {
706 return IRMutator::visit(op);
707 }
708 }

734};

736// Inject the buffer handling code for the inputs and outputs at the
737// appropriate site.
738class InjectBufferCopiesForInputsAndOutputs : public IRMutator {
739 Stmt site;
740
741 // Find all references to external buffers.
742 class FindInputsAndOutputs : public IRVisitor {
743 using IRVisitor::visit;
744
745 void include(const Parameter &p) {
746 if (p.defined()) {
747 result.insert(p.name());
748 result_storage[p.name()] = p.memory_type();
749 }
750 }
751
752 void include(const Buffer<> &b) {
753 if (b.defined()) {
754 result.insert(b.name());
755 result_storage[b.name()] = MemoryType::Auto;
756 }
757 }
758
759 void visit(const Variable *op) override {
760 include(op->param);
761 include(op->image);
762 }
763
764 void visit(const Load *op) override {
765 include(op->param);
766 include(op->image);
767 IRVisitor::visit(op);
768 }
769
770 void visit(const Store *op) override {
771 include(op->param);
772 IRVisitor::visit(op);
773 }
774
775 void visit(const Call *op) override {
776 // We shouldn't need to look for Buffers here,
777 // since we expect this to be run after StorageFlattening.
778 // Add an assertion check just in case a change to lowering ever
779 // subverts this ordering expectation.
780 internal_assert(op->call_type != Call::Halide &&
781 op->call_type != Call::Image);
782 IRVisitor::visit(op);
783 }
784
785 public:
786 set<string> result;
787 std::map<string, MemoryType> result_storage;
788 };
789
790public:
791 using IRMutator::mutate;
792
793 Stmt mutate(const Stmt &s) override {
794 if (s.same_as(site)) {
795 FindInputsAndOutputs finder;
796 s.accept(&finder);
797 Stmt new_stmt = s;
798 for (const string &buf : finder.result) {
799 new_stmt = InjectBufferCopiesForSingleBuffer(buf, true, finder.result_storage.at(buf)).mutate(new_stmt);
800 }
801 return new_stmt;
802 } else {
803 return IRMutator::mutate(s);
804 }
805 }
806
807 InjectBufferCopiesForInputsAndOutputs(Stmt s)
808 : site(std::move(s)) {
809 }
810};

812} // namespace

814Stmt inject_host_dev_buffer_copies(Stmt s, const Target &t) {
815 // Hexagon code assumes that the host-based wrapper code
816 // handles all copies to/from device, so this isn't necessary;
817 // furthermore, we would actually generate wrong code by proceeding
818 // here, as this implementation assumes we start from the host (which
819 // isn't true for Hexagon), and that it's safe to inject calls to copy
820 // and/or mark things dirty (which also isn't true for Hexagon).
821 if (t.arch == Target::Hexagon) {
822 return s;
823 }
824
825 // Handle internal allocations
826 s = InjectBufferCopies().mutate(s);
827
828 // Handle inputs and outputs
829 FindOutermostProduce outermost;
830 s.accept(&outermost);
831 if (outermost.result.defined()) {
832 // If the entire pipeline simplified away, or just dispatches
833 // to another pipeline, there may be no outermost produce.
834 s = InjectBufferCopiesForInputsAndOutputs(outermost.result).mutate(s);
835 }
836
837 return s;
838}
