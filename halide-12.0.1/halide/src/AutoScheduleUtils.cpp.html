25 Expr visit(const Variable *var) override {
26 if (var->param.defined() && var->param.is_buffer()) {
27 // This is a var associated with an ImageParam object. This
28 // should be something of the form XXX.min.[dim_index] or
29 // XXX.extent.[dim_index]
30 std::vector<std::string> v = split_string(var->name, ".");
31 user_assert(v.size() >= 3);
32 int d = string_to_int(v[v.size() - 1]);
33 if (v[v.size() - 2] == "min") {
34 Expr est = var->param.min_constraint_estimate(d);
35 return est.defined() ? est : var;
36 } else {
37 internal_assert(v[v.size() - 2] == "extent");
38 Expr est = var->param.extent_constraint_estimate(d);
39 return est.defined() ? est : var;
40 }
41 } else if (var->param.defined() && !var->param.is_buffer() &&
42 var->param.estimate().defined()) {
43 // This is a var from a Param object
44 return var->param.estimate();
45 } else {
46 return var;
47 }
48 }
49};
50} // anonymous namespace

81Expr box_size(const Box &b) {
82 Expr size = make_one(Int(64));
83 for (size_t i = 0; i < b.size(); i++) {
84 Expr extent = get_extent(b[i]);
85 if (extent.defined() && size.defined()) {
86 size *= extent;
87 } else if (is_const_zero(extent)) {
88 return make_zero(Int(64));
89 } else {
90 return Expr();
91 }
92 }
93 return simplify(size);
94}

96void combine_load_costs(map<string, Expr> &result, const map<string, Expr> &partial) {
97 for (const auto &kv : partial) {
98 auto iter = result.find(kv.first);
99 if (iter == result.end()) {
100 result.emplace(kv.first, kv.second);
101 } else {
102 if (!iter->second.defined()) {
103 continue;
104 } else if (!kv.second.defined()) {
105 iter->second = Expr();
106 } else {
107 iter->second = simplify(iter->second + kv.second);
108 }
109 }
110 }
111}

113Definition get_stage_definition(const Function &f, int stage_num) {
114 internal_assert(!f.has_extern_definition());

133DimBounds get_stage_bounds(const Function &f, int stage_num, const DimBounds &pure_bounds) {
134 DimBounds bounds;
135 // Assume that the domain of the pure vars across all the update
136 // definitions is the same. This may not be true and can result in
137 // over estimation of the extent.
138 for (const auto &b : pure_bounds) {
139 bounds[b.first] = b.second;
140 }
141
142 if (!f.has_extern_definition()) {
143 Definition def = get_stage_definition(f, stage_num);
144 for (const auto &rvar : def.schedule().rvars()) {
145 Expr lower = substitute_var_estimates(rvar.min);
146 Expr upper = substitute_var_estimates(rvar.min + rvar.extent - 1);
147 bounds.emplace(rvar.var, Interval(lower, upper));
148 }
149 }
150
151 return bounds;
152}

154vector<DimBounds> get_stage_bounds(const Function &f, const DimBounds &pure_bounds) {
155 vector<DimBounds> stage_bounds;
156 size_t num_stages = f.updates().size() + 1;
157 for (size_t s = 0; s < num_stages; s++) {
158 stage_bounds.push_back(get_stage_bounds(f, s, pure_bounds));
159 }
160 return stage_bounds;
161}

163Expr perform_inline(Expr e, const map<string, Function> &env,
164 const set<string> &inlines,
165 const vector<string> &order) {
166 if (inlines.empty()) {
167 return e;
168 }
169
170 bool funcs_to_inline = false;
171 Expr inlined_expr = e;
172
173 do {
174 funcs_to_inline = false;
175 // Find all the function calls in the current expression.
176 FindAllCalls find;
177 inlined_expr.accept(&find);
178 const set<string> &calls_unsorted = find.funcs_called;
179
180 vector<string> calls(calls_unsorted.begin(), calls_unsorted.end());
181 // Sort 'calls' based on the realization order in descending order
182 // if provided (i.e. last to be realized comes first).
183 if (!order.empty()) {
184 std::sort(calls.begin(), calls.end(),
185 [&order](const string &lhs, const string &rhs) {
186 const auto &iter_lhs = std::find(order.begin(), order.end(), lhs);
187 const auto &iter_rhs = std::find(order.begin(), order.end(), rhs);
188 return iter_lhs > iter_rhs;
189 });
190 }
191
192 // Check if any of the calls are in the set of functions to be inlined.
193 // Inline from the last function to be realized to avoid extra
194 // inlining works.
195 for (const auto &call : calls) {
196 if (inlines.find(call) != inlines.end()) {
197 const Function &prod_func = env.at(call);
198 // Impure functions cannot be inlined.
199 internal_assert(prod_func.is_pure());
200 // Inline the function call and set the flag to check for
201 // further inlining opportunities.
202 inlined_expr = inline_function(inlined_expr, prod_func);
203 funcs_to_inline = true;
204 break;
205 }
206 }
207 } while (funcs_to_inline);
208
209 return inlined_expr;
210}

212set<string> get_parents(Function f, int stage) {
213 set<string> parents;
214 if (f.has_extern_definition()) {
215 internal_assert(stage == 0);
216 for (const ExternFuncArgument &arg : f.extern_arguments()) {
217 if (arg.is_func()) {
218 string prod_name = Function(arg.func).name();
219 parents.insert(prod_name);
220 } else if (arg.is_expr()) {
221 FindAllCalls find;
222 arg.expr.accept(&find);
223 parents.insert(find.funcs_called.begin(), find.funcs_called.end());
224 } else if (arg.is_image_param() || arg.is_buffer()) {
225 Buffer<> buf;
226 if (arg.is_image_param()) {
227 buf = arg.image_param.buffer();
228 } else {
229 buf = arg.buffer;
230 }
231 parents.insert(buf.name());
232 }
233 }
234 } else {
235 FindAllCalls find;
236 Definition def = get_stage_definition(f, stage);
237 def.accept(&find);
238 parents.insert(find.funcs_called.begin(), find.funcs_called.end());
239 }
240 return parents;
241}

252// If the cost of computing a Func is about the same as calling the Func,
253// inline the Func. Return true of any of the Funcs is inlined.
254bool inline_all_trivial_functions(const vector<Function> &outputs,
255 const vector<string> &order,
256 const map<string, Function> &env) {
257 bool inlined = false;
258 // The very last few functions in 'order' are the last to be realized in the
259 // pipeline (the final producers) so there is no point in checking it.
260 for (int i = 0; i < (int)order.size() - (int)outputs.size(); ++i) {
261 bool is_output = false;
262 for (const Function &f : outputs) {
263 if (order[i] == f.name()) {
264 is_output = true;
265 break;
266 }
267 }
268 if (is_output) {
269 // Should not inline output Func
270 debug(5) << "Skip inlining " << order[i] << " since it is an output\n";
271 continue;
272 }
273 const Function &f1 = env.at(order[i]);
274 if (is_func_trivial_to_inline(f1)) {
275 inlined = true;
276 debug(4) << "Function \"" << order[i] << "\" is trivial to inline\n";
277 for (int j = i + 1; j < (int)order.size() - (int)outputs.size(); ++j) {
278 internal_assert(order[i] != order[j]);
279 const Function &f2 = env.at(order[j]);
280
281 if (f2.has_extern_definition() && !f1.is_wrapper()) {
282 debug(5) << "Skip inlining of function \"" << f1.name()
283 << "\" inside \"" << f2.name() << "\", because "
284 << "non-wrapper functions cannot be inlined inside "
285 << "extern functions.\n";
286 } else {
287 debug(5) << "Inline trivial function \"" << f1.name()
288 << "\" inside \"" << f2.name() << "\"\n";
289 inline_function(f2, f1);
290 }
291 }
292 }
293 }
294 return inlined;
295}

297// Determine if a Func (order[index]) is only consumed by another single Func
298// in element-wise manner. If it is, return the name of the consumer Func;
299// otherwise, return an empty string.
300string is_func_called_element_wise(const vector<string> &order, size_t index,
301 const map<string, Function> &env) {
302 const Function &f1 = env.at(order[index]);
303 if (f1.has_extern_definition() || !f1.can_be_inlined()) {
304 return "";
305 }
306 internal_assert(index < order.size());
307
308 string caller = "";
309 for (size_t i = index + 1; i < order.size(); ++i) {
310 const Function &f2 = env.at(order[i]);
311 if (f2.has_extern_definition()) {
312 continue;
313 }
314 int num_stages = f2.updates().size() + 1;
315 for (int s = 0; s < num_stages; ++s) {
316 Definition def = get_stage_definition(f2, s);
317 FindAllCalls find;
318 def.accept(&find);
319
320 if (find.funcs_called.count(f1.name())) {
321 if (caller.empty()) {
322 caller = f2.name();
323 } else {
324 // Found another caller of 'f1'
325 return "";
326 }
327 }
328 for (const auto &iter : find.call_args) {
329 if (iter.first != f1.name()) {
330 continue;
331 }
332 if (def.args().size() != iter.second.size()) {
333 // It's not an element-wise access
334 return "";
335 }
336 for (size_t j = 0; j < iter.second.size(); ++j) {
337 if (!equal(def.args()[j], iter.second[j])) {
338 // It's not an element-wise access
339 return "";
340 }
341 }
342 }
343 }
344 }
345 return caller;
346}

348// Inline a Func if its values are only consumed by another single Func in
349// element-wise manner.
350bool inline_all_element_wise_functions(const vector<Function> &outputs,
351 const vector<string> &order,
352 const map<string, Function> &env) {
353 bool inlined = false;
354 // The very last few functions in 'order' are the last to be realized in the
355 // pipeline (the final producers) so there is no point in checking it.
356 for (int i = 0; i < (int)order.size() - (int)outputs.size(); ++i) {
357 bool is_output = false;
358 for (const Function &f : outputs) {
359 if (order[i] == f.name()) {
360 is_output = true;
361 break;
362 }
363 }
364 if (is_output) {
365 // Should not inline output Func
366 debug(5) << "Skip inlining " << order[i] << " since it is an output\n";
367 continue;
368 }
369 string caller = is_func_called_element_wise(order, i, env);
370 if (!caller.empty()) {
371 inlined = true;
372 debug(4) << "Inline function \"" << order[i] << "\" since it is called only by "
373 << caller << " in element-wise manner\n";
374 internal_assert(order[i] != caller);
375 inline_function(env.at(caller), get_element(env, order[i]));
376 }
377 }
378 return inlined;
379}

382void check(const Expr &input, Expr expected) {
383 Expr result = simplify(substitute_var_estimates(input));
384 expected = simplify(expected);
385 if (!equal(result, expected)) {
386 internal_error
387 << "\nsubstitute_var_estimates() failure:\n"
388 << "Input: " << input << "\n"
389 << "Result: " << result << "\n"
390 << "Expected result: " << expected << "\n";
391 }
