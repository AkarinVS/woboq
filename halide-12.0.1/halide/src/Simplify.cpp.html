22Simplify::Simplify(bool r, const Scope<Interval> *bi, const Scope<ModulusRemainder> *ai)
23 : remove_dead_code(r), no_float_simplify(false) {
24
25 // Only respect the constant bounds from the containing scope.
26 for (auto iter = bi->cbegin(); iter != bi->cend(); ++iter) {
27 ExprInfo bounds;
28 if (const int64_t *i_min = as_const_int(iter.value().min)) {
29 bounds.min_defined = true;
30 bounds.min = *i_min;
31 }
32 if (const int64_t *i_max = as_const_int(iter.value().max)) {
33 bounds.max_defined = true;
34 bounds.max = *i_max;
35 }
36
37 if (ai->contains(iter.name())) {
38 bounds.alignment = ai->get(iter.name());
39 }
40
41 if (bounds.min_defined || bounds.max_defined || bounds.alignment.modulus != 1) {
42 bounds_and_alignment_info.push(iter.name(), bounds);
43 }
44 }
45
46 for (auto iter = ai->cbegin(); iter != ai->cend(); ++iter) {
47 if (bounds_and_alignment_info.contains(iter.name())) {
48 // Already handled
49 continue;
50 }
51 ExprInfo bounds;
52 bounds.alignment = iter.value();
53 bounds_and_alignment_info.push(iter.name(), bounds);
54 }
55}

102void Simplify::ScopedFact::learn_false(const Expr &fact) {
103 Simplify::VarInfo info;
104 info.old_uses = info.new_uses = 0;
105 if (const Variable *v = fact.as<Variable>()) {
106 info.replacement = const_false(fact.type().lanes());
107 simplify->var_info.push(v->name, info);
108 pop_list.push_back(v);
109 } else if (const NE *ne = fact.as<NE>()) {
110 const Variable *v = ne->a.as<Variable>();
111 if (v && is_const(ne->b)) {
112 info.replacement = ne->b;
113 simplify->var_info.push(v->name, info);
114 pop_list.push_back(v);
115 }
116 } else if (const LT *lt = fact.as<LT>()) {
117 const Variable *v = lt->a.as<Variable>();
118 Simplify::ExprInfo i;
119 if (v) {
120 simplify->mutate(lt->b, &i);
121 if (i.min_defined) {
122 // !(v < i)
123 learn_lower_bound(v, i.min);
124 }
125 }
126 v = lt->b.as<Variable>();
127 if (v) {
128 simplify->mutate(lt->a, &i);
129 if (i.max_defined) {
130 // !(i < v)
131 learn_upper_bound(v, i.max);
132 }
133 }
134 } else if (const LE *le = fact.as<LE>()) {
135 const Variable *v = le->a.as<Variable>();
136 Simplify::ExprInfo i;
137 if (v && v->type.is_int() && v->type.bits() >= 32) {
138 simplify->mutate(le->b, &i);
139 if (i.min_defined) {
140 // !(v <= i)
141 learn_lower_bound(v, i.min + 1);
142 }
143 }
144 v = le->b.as<Variable>();
145 if (v && v->type.is_int() && v->type.bits() >= 32) {
146 simplify->mutate(le->a, &i);
147 if (i.max_defined) {
148 // !(i <= v)
149 learn_upper_bound(v, i.max - 1);
150 }
151 }
152 } else if (const Call *c = Call::as_tag(fact)) {
153 learn_false(c->args[0]);
154 return;
155 } else if (const Or *o = fact.as<Or>()) {
156 // Both must be false
157 learn_false(o->a);
158 learn_false(o->b);
159 return;
160 } else if (const Not *n = fact.as<Not>()) {
161 learn_true(n->a);
162 return;
163 }
164 if (simplify->falsehoods.insert(fact).second) {
165 falsehoods.push_back(fact);
166 }
167}

169void Simplify::ScopedFact::learn_upper_bound(const Variable *v, int64_t val) {
170 ExprInfo b;
171 b.max_defined = true;
172 b.max = val;
173 if (simplify->bounds_and_alignment_info.contains(v->name)) {
174 b.intersect(simplify->bounds_and_alignment_info.get(v->name));
175 }
176 simplify->bounds_and_alignment_info.push(v->name, b);
177 bounds_pop_list.push_back(v);
178}

191void Simplify::ScopedFact::learn_true(const Expr &fact) {
192 Simplify::VarInfo info;
193 info.old_uses = info.new_uses = 0;
194 if (const Variable *v = fact.as<Variable>()) {
195 info.replacement = const_true(fact.type().lanes());
196 simplify->var_info.push(v->name, info);
197 pop_list.push_back(v);
198 } else if (const EQ *eq = fact.as<EQ>()) {
199 const Variable *v = eq->a.as<Variable>();
200 const Mod *m = eq->a.as<Mod>();
201 const int64_t *modulus = m ? as_const_int(m->b) : nullptr;
202 const int64_t *remainder = m ? as_const_int(eq->b) : nullptr;
203 if (v) {
204 if (is_const(eq->b) || eq->b.as<Variable>()) {
205 // TODO: consider other cases where we might want to entirely substitute
206 info.replacement = eq->b;
207 simplify->var_info.push(v->name, info);
208 pop_list.push_back(v);
209 } else if (v->type.is_int()) {
210 // Visit the rhs again to get bounds and alignment info to propagate to the LHS
211 // TODO: Visiting it again is inefficient
212 Simplify::ExprInfo expr_info;
213 simplify->mutate(eq->b, &expr_info);
214 if (simplify->bounds_and_alignment_info.contains(v->name)) {
215 // We already know something about this variable and don't want to suppress it.
216 auto existing_knowledge = simplify->bounds_and_alignment_info.get(v->name);
217 expr_info.intersect(existing_knowledge);
218 }
219 simplify->bounds_and_alignment_info.push(v->name, expr_info);
220 bounds_pop_list.push_back(v);
221 }
222 } else if (const Variable *vb = eq->b.as<Variable>()) {
223 // y % 2 == x
224 // We know that LHS is not a const due to
225 // canonicalization, and that the LHS is not a variable or
226 // the case above would have triggered. Learn from the
227 // bounds and alignment of the LHS.
228 // TODO: Visiting it again is inefficient
229 Simplify::ExprInfo expr_info;
230 simplify->mutate(eq->a, &expr_info);
231 if (simplify->bounds_and_alignment_info.contains(vb->name)) {
232 // We already know something about this variable and don't want to suppress it.
233 auto existing_knowledge = simplify->bounds_and_alignment_info.get(vb->name);
234 expr_info.intersect(existing_knowledge);
235 }
236 simplify->bounds_and_alignment_info.push(vb->name, expr_info);
237 bounds_pop_list.push_back(vb);
238 } else if (modulus && remainder && (v = m->a.as<Variable>())) {
239 // Learn from expressions of the form x % 8 == 3
240 Simplify::ExprInfo expr_info;
241 expr_info.alignment.modulus = *modulus;
242 expr_info.alignment.remainder = *remainder;
243 if (simplify->bounds_and_alignment_info.contains(v->name)) {
244 // We already know something about this variable and don't want to suppress it.
245 auto existing_knowledge = simplify->bounds_and_alignment_info.get(v->name);
246 expr_info.intersect(existing_knowledge);
247 }
248 simplify->bounds_and_alignment_info.push(v->name, expr_info);
249 bounds_pop_list.push_back(v);
250 }
251 } else if (const LT *lt = fact.as<LT>()) {
252 const Variable *v = lt->a.as<Variable>();
253 Simplify::ExprInfo i;
254 if (v && v->type.is_int() && v->type.bits() >= 32) {
255 simplify->mutate(lt->b, &i);
256 if (i.max_defined) {
257 // v < i
258 learn_upper_bound(v, i.max - 1);
259 }
260 }
261 v = lt->b.as<Variable>();
262 if (v && v->type.is_int() && v->type.bits() >= 32) {
263 simplify->mutate(lt->a, &i);
264 if (i.min_defined) {
265 // i < v
266 learn_lower_bound(v, i.min + 1);
267 }
268 }
269 } else if (const LE *le = fact.as<LE>()) {
270 const Variable *v = le->a.as<Variable>();
271 Simplify::ExprInfo i;
272 if (v) {
273 simplify->mutate(le->b, &i);
274 if (i.max_defined) {
275 // v <= i
276 learn_upper_bound(v, i.max);
277 }
278 }
279 v = le->b.as<Variable>();
280 if (v) {
281 simplify->mutate(le->a, &i);
282 if (i.min_defined) {
283 // i <= v
284 learn_lower_bound(v, i.min);
285 }
286 }
287 } else if (const Call *c = Call::as_tag(fact)) {
288 learn_true(c->args[0]);
289 return;
290 } else if (const And *a = fact.as<And>()) {
291 // Both must be true
292 learn_true(a->a);
293 learn_true(a->b);
294 return;
295 } else if (const Not *n = fact.as<Not>()) {
296 learn_false(n->a);
297 return;
298 }
299 if (simplify->truths.insert(fact).second) {
300 truths.push_back(fact);
301 }

304template<class T>
305T substitute_facts_impl(T t, const vector<Expr> &truths, const vector<Expr> &falsehoods) {
306 // An std::map<Expr, Expr> version of substitute might be an optimization?
307 for (const auto &i : truths) {
308 t = substitute(i, const_true(i.type().lanes()), t);
309 }
310 for (const auto &i : falsehoods) {
311 t = substitute(i, const_false(i.type().lanes()), t);
312 }
313 return t;
314}

324Simplify::ScopedFact::~ScopedFact() {
325 for (const auto *v : pop_list) {
326 simplify->var_info.pop(v->name);
327 }
328 for (const auto *v : bounds_pop_list) {
329 simplify->bounds_and_alignment_info.pop(v->name);
330 }
331 for (const auto &e : truths) {
332 simplify->truths.erase(e);
333 }
334 for (const auto &e : falsehoods) {
335 simplify->falsehoods.erase(e);
336 }

339Expr simplify(const Expr &e, bool remove_dead_let_stmts,
340 const Scope<Interval> &bounds,
341 const Scope<ModulusRemainder> &alignment) {
342 Simplify m(remove_dead_let_stmts, &bounds, &alignment);
343 Expr result = m.mutate(e, nullptr);
344 if (m.in_unreachable) {
345 return unreachable(e.type());
346 }
347 return result;
348}

373bool can_prove(Expr e, const Scope<Interval> &bounds) {
374 internal_assert(e.type().is_bool())
375 << "Argument to can_prove is not a boolean Expr: " << e << "\n";
376
377 e = remove_likelies(e);
378 e = common_subexpression_elimination(e);
379
380 Expr orig = e;
381
382 e = simplify(e, true, bounds);
383
384 // Take a closer look at all failed proof attempts to hunt for
385 // simplifier weaknesses
386 const bool check_failed_proofs = debug::debug_level() > 0 || get_compiler_logger() != nullptr;
387 if (check_failed_proofs && !is_const(e)) {
388 struct RenameVariables : public IRMutator {
389 using IRMutator::visit;
390
391 Expr visit(const Variable *op) override {
392 auto it = vars.find(op->name);
393 if (lets.contains(op->name)) {
394 return Variable::make(op->type, lets.get(op->name));
395 } else if (it == vars.end()) {
396 std::string name = "v" + std::to_string(count++);
397 vars[op->name] = name;
398 out_vars.emplace_back(op->type, name);
399 return Variable::make(op->type, name);
400 } else {
401 return Variable::make(op->type, it->second);
402 }
403 }
404
405 Expr visit(const Let *op) override {
406 std::string name = "v" + std::to_string(count++);
407 ScopedBinding<string> bind(lets, op->name, name);
408 return Let::make(name, mutate(op->value), mutate(op->body));
409 }
410
411 int count = 0;
412 map<string, string> vars;
413 Scope<string> lets;
414 std::vector<pair<Type, string>> out_vars;
415 } renamer;
416
417 e = renamer.mutate(e);
418
419 // Look for a concrete counter-example with random probing
420 static std::mt19937 rng(0);
421 for (int i = 0; i < 100; i++) {
422 map<string, Expr> s;
423 for (const auto &p : renamer.out_vars) {
424 if (p.first.is_handle()) {
425 // This aint gonna work
426 return false;
427 }
428 s[p.second] = make_const(p.first, (int)(rng() & 0xffff) - 0x7fff);
429 }
430 Expr probe = unwrap_tags(simplify(substitute(s, e)));
431 if (!is_const_one(probe)) {
432 // Found a counter-example, or something that fails to fold
433 return false;
434 }
435 }
436
437 if (get_compiler_logger()) {
438 get_compiler_logger()->record_failed_to_prove(e, orig);
439 }
440
441 debug(1) << "Failed to prove, but could not find a counter-example:\n " << e << "\n";
442 debug(1) << "Original expression:\n"
443 << orig << "\n";
444 return false;
445 }
446
447 return is_const_one(e);
448}
