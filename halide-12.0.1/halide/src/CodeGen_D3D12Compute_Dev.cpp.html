122string CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::print_type_maybe_storage(Type type, bool storage, AppendSpaceIfNeeded space) {
123 ostringstream oss;
124
125 // Storage uses packed vector types.
126 if (storage && type.lanes() != 1) {
127 // NOTE(marcos): HLSL has 'packoffset' but it only applies to constant
128 // buffer fields (constant registers (c)); because constant arguments
129 // are always converted to 32bit values by the runtime prior to kernel
130 // dispatch, there is no need to complicate things with packoffset.
131 }
132
133 if (type.is_float()) {
134 switch (type.bits()) {
135 case 16:
136 // 16-bit floating point value. This data type is provided only for language compatibility.
137 // Direct3D 10 shader targets map all half data types to float data types.
138 // A half data type cannot be used on a uniform global variable (use the /Gec flag if this functionality is desired).
139 oss << "half";
140 break;
141 case 32:
142 oss << "float";
143 break;
144 case 64:
145 // "64-bit floating point value. You cannot use double precision values as inputs and outputs for a stream.
146 // To pass double precision values between shaders, declare each double as a pair of uint data types.
147 // Then, use the asdouble function to pack each double into the pair of uints and the asuint function to
148 // unpack the pair of uints back into the double."
149 user_error << "HLSL (SM 5.1) does not have transparent support for 'double' types.\n";
150 oss << "double";
151 break;
152 default:
153 user_error << "Can't represent a float with this many bits in HLSL (SM 5.1): " << type << "\n";
154 }
155 } else {
156 switch (type.bits()) {
157 case 1:
158 oss << "bool";
159 break;
160 case 8:
161 case 16:
162 case 32:
163 if (type.is_uint()) {
164 oss << "u";
165 }
166 oss << "int";
167#if DEBUG_TYPES
168 oss << type.bits();
169#endif
170 break;
171 case 64:
172 user_error << "HLSL (SM 5.1) does not support 64-bit integers.\n";
173 break;
174 default:
175 user_error << "Can't represent an integer with this many bits in HLSL (SM 5.1): " << type << "\n";
176 }
177 }
178 switch (type.lanes()) {
179 case 1:
180 break;
181 case 2:
182 case 3:
183 case 4:
184#if DEBUG_TYPES
185 oss << "_(";
186#endif
187 oss << type.lanes();
188#if DEBUG_TYPES
189 oss << ")";
190#endif
191 break;
192 case 8:
193 case 16:
194 // TODO(marcos): are there 8-wide and 16-wide types in HLSL?
195 // (CodeGen_GLSLBase seems to happily generate invalid vector types)
196 default:
197 user_error << "Unsupported vector width in HLSL (SM 5.1): " << type << "\n";
198 }
199
200 if (space == AppendSpace) {
201 oss << " ";
202 }
203
204 return oss.str();
205}

224string simt_intrinsic(const string &name) {
225 if (ends_with(name, ".__thread_id_x")) {
226 return "tid_in_tgroup.x";
227 } else if (ends_with(name, ".__thread_id_y")) {
228 return "tid_in_tgroup.y";
229 } else if (ends_with(name, ".__thread_id_z")) {
230 return "tid_in_tgroup.z";
231 } else if (ends_with(name, ".__thread_id_w")) {
232 user_error << "HLSL (SM5.1) does not support more than three dimensions for compute kernel threads.\n";
233 } else if (ends_with(name, ".__block_id_x")) {
234 return "tgroup_index.x";
235 } else if (ends_with(name, ".__block_id_y")) {
236 return "tgroup_index.y";
237 } else if (ends_with(name, ".__block_id_z")) {
238 return "tgroup_index.z";
239 } else if (ends_with(name, ".__block_id_w")) {
240 user_error << "HLSL (SM5.1) does not support more than three dimensions for compute dispatch groups.\n";
241 }
242 internal_error << "simt_intrinsic called on bad variable name: " << name << "\n";
243 return "";
244}

357void CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::visit(const Call *op) {
358 if (op->is_intrinsic(Call::gpu_thread_barrier)) {
359 internal_assert(op->args.size() == 1) << "gpu_thread_barrier() intrinsic must specify memory fence type.\n";
360
361 const auto *fence_type_ptr = as_const_int(op->args[0]);
362 internal_assert(fence_type_ptr) << "gpu_thread_barrier() parameter is not a constant integer.\n";
363 auto fence_type = *fence_type_ptr;
364
365 // By default, we'll just issue a Group (aka Shared) memory barrier,
366 // since it seems there isn't a sync without a memory barrier
367 // available
368 // NOTE(shoaibkamil): should we replace with AllMemoryBarrierWithGroupSync()
369 // if both are required?
370 if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device &&
371 !(fence_type & CodeGen_GPU_Dev::MemoryFenceType::Shared)) {
372 stream << get_indent() << "DeviceMemoryBarrierWithGroupSync();\n";
373 } else if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device) {
374 stream << get_indent() << "DeviceMemoryBarrier();\n";
375 }
376 stream << get_indent() << "GroupMemoryBarrierWithGroupSync();\n";
377 print_assignment(op->type, "0");
378 } else if (op->name == "pow_f32" && can_prove(op->args[0] > 0)) {
379 // If we know pow(x, y) is called with x > 0, we can use HLSL's pow
380 // directly.
381 stream << "pow(" << print_expr(op->args[0]) << ", " << print_expr(op->args[1]) << ")";
382 } else {
383 CodeGen_C::visit(op);
384 }
385}

413struct StoragePackUnpack {
414 using CodeGen = CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C;
415
416 // Shader Model 5.1: threadgroup shared memory is limited 32KB
417 static const size_t ThreadGroupSharedStorageLimit = 32 * 1024;
418
419 void pack_storage(const Allocate *op, size_t elements, size_t size_in_bytes) {
420 // we could try to compact things for smaller types:
421 size_t packing_factor = 1;
422 while (size_in_bytes > ThreadGroupSharedStorageLimit) {
423 // must pack/unpack elements to/from shared memory...
424 elements = (elements + 1) & ~size_t(1);
425 elements /= 2;
426 size_in_bytes = (size_in_bytes + 1) & ~size_t(1);
427 size_in_bytes /= 2;
428 packing_factor *= 2;
429 }
430 // smallest possible pack type is a byte (no nibbles)
431 internal_assert(packing_factor <= 4);
432 }

434 std::ostringstream pack_store(CodeGen &cg, const Store *op) {
435 std::ostringstream lhs;
436 // NOTE(marcos): 8bit and 16bit word packing -- the smallest integer
437 // type granularity available in HLSL SM 5.1 is 32bit (int/uint):
438 Type value_type = op->value.type();
439 if (value_type.bits() == 32) {
440 // storing a 32bit word? great! just reinterpret value to uint32:
441 lhs << cg.print_name(op->name)
442 << "[" << cg.print_expr(op->index) << "]"
443 << " = "
444 << cg.print_reinterpret(UInt(32), op->value)
445 << ";\n";
446 } else {
447 // nightmare:
448 internal_assert(value_type.bits() < 32);
449 // must map element index to uint word array index:
450 ostringstream index;
451 auto bits = value_type.bits();
452 auto divisor = (32 / bits);
453 auto i = cg.print_expr(op->index);
454 index << i << " / " << divisor;
455 ostringstream word;
456 word << cg.print_name(op->name)
457 << "[" + index.str() + "]";
458 // now mask the appropriate bits:
459 ostringstream mask;
460 mask << "("
461 << hex_literal((1 << bits) - 1)
462 << " << "
463 << "(" << bits << "*(" << i << " % " << divisor << " ))"
464 << ")";
465 // apply the mask to the rhs value:
466 ostringstream value;
467 value << "("
468 << mask.str()
469 << " & "
470 << "(" << cg.print_expr(op->value) << ")"
471 << ")";
472
473 // the performance impact of atomic operations on shared memory is
474 // not well documented... here is something:
475 // https://stackoverflow.com/a/19548723
476 lhs << cg.get_indent() << "InterlockedAnd(" << word.str() << ", "
477 << "~" << mask.str() << ");\n";
478 lhs << cg.get_indent() << "InterlockedXor(" << word.str() << ", " << value.str() << ");\n";
479 }
480 return lhs;
481 }

483 std::ostringstream unpack_load(CodeGen &cg, const Load *op) {
484 std::ostringstream rhs;
485 // NOTE(marcos): let's keep this block of code here (disabled) in case
486 // we need to "emulate" byte/short packing in shared memory (recall that
487 // the smallest type granularity in HLSL SM 5.1 allows is 32bit types):
488 if (op->type.bits() == 32) {
489 // loading a 32bit word? great! just reinterpret as float/int/uint
490 rhs << "as" << cg.print_type(op->type.element_of())
491 << "("
492 << cg.print_name(op->name)
493 << "[" << cg.print_expr(op->index) << "]"
494 << ")";
495 } else {
496 // not a 32bit word? hell ensues:
497 internal_assert(op->type.bits() < 32);
498 // must map element index to uint word array index:
499 ostringstream index;
500 auto bits = op->type.bits();
501 auto divisor = (32 / bits);
502 auto i = cg.print_expr(op->index);
503 index << i << " / " << divisor;
504 ostringstream word;
505 word << cg.print_name(op->name)
506 << "[" + index.str() + "]";
507 // now mask the appropriate bits:
508 ostringstream mask;
509 mask << "("
510 << hex_literal((1 << bits) - 1)
511 << " << "
512 << "(" << bits << "*(" << i << " % " << divisor << " ))"
513 << ")";
514 // extract the correct bits from the word
515 ostringstream cut;
516 cut << "("
517 << word.str()
518 << " & "
519 << mask.str()
520 << ")"
521 << " >> "
522 << "(" << bits << "*(" << i << " % " << divisor << " ))";
523 // if it is signed, need to propagate sign... shift-up then down:
524 if (op->type.is_int()) {
525 rhs << "asint"
526 << "("
527 << cut.str()
528 << " << "
529 << (32 - bits)
530 << ")"
531 << " >> "
532 << (32 - bits);
533 } else {
534 rhs << cut.str();
535 }
536 }
537 return rhs;
538 }

541void CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::visit(const Load *op) {
542 user_assert(is_const_one(op->predicate)) << "Predicated load is not supported inside D3D12Compute kernel.\n";
543
544 // elements in a threadgroup shared buffer are always 32bits:
545 // must reinterpret (and maybe unpack) bits.
546 bool shared_promotion_required = false;
547 string promotion_str = "";
548 if (groupshared_allocations.contains(op->name)) {
549 internal_assert(allocations.contains(op->name));
550 Type promoted_type = op->type.with_bits(32).with_lanes(1);
551 if (promoted_type != op->type) {
552 shared_promotion_required = true;
553 // NOTE(marcos): might need to resort to StoragePackUnpack::unpack_load() here
554 promotion_str = "as" + print_type(promoted_type);
555 }
556 }
557
558 // If we're loading a contiguous ramp, "unroll" the ramp into loads:
559 Expr ramp_base = is_ramp_one(op->index);
560 if (ramp_base.defined()) {
561 internal_assert(op->type.is_vector());
562
563 ostringstream rhs;
564 rhs << print_type(op->type)
565 << "(";
566 const int lanes = op->type.lanes();
567 for (int i = 0; i < lanes; ++i) {
568 if (shared_promotion_required) {
569 rhs << promotion_str << "(";
570 }
571 rhs << print_name(op->name)
572 << "["
573 << print_expr(ramp_base)
574 << "+" << i
575 << "]";
576 if (shared_promotion_required) {
577 rhs << ")";
578 }
579 if (i < lanes - 1) {
580 rhs << ", ";
581 }
582 }
583 rhs << ")";
584
585 print_assignment(op->type, rhs.str());
586
587 return;
588 }
589
590 string id_index = print_expr(op->index);
591
592 // Get the rhs just for the cache.
593 bool type_cast_needed = !(allocations.contains(op->name) &&
594 allocations.get(op->name).type == op->type);
595
596 ostringstream rhs;
597 if (type_cast_needed) {
598 ostringstream element;
599 // Vector cases handled below
600 if (shared_promotion_required && !op->index.type().is_vector()) {
601 element << promotion_str << "(";
602 }
603 element << print_name(op->name)
604 << "[" << id_index << "]";
605 if (shared_promotion_required && !op->index.type().is_vector()) {
606 element << ")";
607 }
608 Type target_type = op->type;
609 Type source_type = allocations.get(op->name).type;
610 rhs << print_cast(target_type, source_type, element.str());
611 } else {
612 // Vector cases handled below
613 if (shared_promotion_required && !op->index.type().is_vector()) {
614 rhs << promotion_str << "(";
615 }
616 rhs << print_name(op->name);
617 rhs << "[" << id_index << "]";
618 if (shared_promotion_required && !op->index.type().is_vector()) {
619 rhs << ")";
620 }
621 }
622
623 std::map<string, string>::iterator cached = cache.find(rhs.str());
624 if (cached != cache.end()) {
625 id = cached->second;
626 return;
627 }
628
629 if (op->index.type().is_vector()) {
630 // If index is a vector, gather vector elements.
631 internal_assert(op->type.is_vector());
632
633 // This has to be underscore as print_name prepends an underscore to
634 // names without one and that results in a name mismatch if a Load
635 // appears as the value of a Let
636 id = unique_name('_');
637 cache[rhs.str()] = id;
638
639 stream << get_indent() << print_type(op->type)
640 << " " << id << ";\n";
641
642 for (int i = 0; i < op->type.lanes(); ++i) {
643 stream << get_indent() << id << "[" << i << "] = "
644 << print_type(op->type.element_of())
645 << "(";
646 if (shared_promotion_required) {
647 stream << promotion_str << "(";
648 }
649 stream << print_name(op->name)
650 << "[" << id_index << "[" << i << "]]";
651 if (shared_promotion_required) {
652 stream << ")";
653 }
654 stream << ");\n";
655 }
656 } else {
657 print_assignment(op->type, rhs.str());
658 }
659}

661void CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::visit(const Store *op) {
662 user_assert(is_const_one(op->predicate)) << "Predicated store is not supported inside D3D12Compute kernel.\n";
663
664 Type value_type = op->value.type();
665
666 // elements in a threadgroup shared buffer are always 32bits:
667 // must reinterpret (and maybe pack) bits.
668 bool shared_promotion_required = false;
669 string promotion_str = "";
670 if (groupshared_allocations.contains(op->name)) {
671 internal_assert(allocations.contains(op->name));
672 Type promoted_type = allocations.get(op->name).type;
673 if (promoted_type != op->value.type()) {
674 shared_promotion_required = true;
675 // NOTE(marcos): might need to resort to StoragePackUnpack::pack_store() here
676 promotion_str = "as" + print_type(promoted_type);
677 }
678 }
679
680 // If we're writing a contiguous ramp, "unroll" the ramp into stores:
681 Expr ramp_base = is_ramp_one(op->index);
682 if (ramp_base.defined()) {
683 internal_assert(value_type.is_vector());
684
685 int lanes = value_type.lanes();
686 for (int i = 0; i < lanes; ++i) {
687 // TODO(marcos): this indentation looks funny in the generated code
688 ostringstream rhs;
689 rhs << print_name(op->name)
690 << "["
691 << print_expr(ramp_base)
692 << " + " << i
693 << "]"
694 << " = ";
695 if (shared_promotion_required) {
696 rhs << promotion_str << "(";
697 }
698 rhs << print_expr(op->value)
699 << "["
700 << i
701 << "]";
702 if (shared_promotion_required) {
703 rhs << ")";
704 }
705 rhs << ";\n";
706 stream << get_indent() << rhs.str();
707 }
708 } else if (op->index.type().is_vector()) {
709 // If index is a vector, scatter vector elements.
710 internal_assert(value_type.is_vector());
711
712 for (int i = 0; i < value_type.lanes(); ++i) {
713 ostringstream rhs;
714 rhs << print_name(op->name)
715 << "["
716 << print_expr(op->index) << "[" << i << "]"
717 << "]"
718 << " = ";
719 if (shared_promotion_required) {
720 rhs << promotion_str << "(";
721 }
722 rhs << print_expr(op->value) << "[" << i << "]";
723 if (shared_promotion_required) {
724 rhs << ")";
725 }
726 rhs << ";\n";
727 stream << get_indent() << rhs.str();
728 }
729 } else {
730 ostringstream rhs;
731 rhs << print_name(op->name)
732 << "[" << print_expr(op->index) << "]"
733 << " = ";
734 if (shared_promotion_required) {
735 rhs << promotion_str << "(";
736 }
737 rhs << print_expr(op->value);
738 if (shared_promotion_required) {
739 rhs << ")";
740 }
741 rhs << ";\n";
742 stream << get_indent() << rhs.str();
743 }
744
745 cache.clear();
746}

827string CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::print_reinforced_cast(Type type, const string &value_expr) {
828 if (type.is_float() || type.is_bool() || type.bits() == 32) {
829 return value_expr;
830 }
831
832 // HLSL SM 5.1 only supports 32bit integer types; smaller integer types have
833 // to be placed in 32bit integers, with special attention to signed integers
834 // that require propagation of the sign bit (MSB):
835 // a) for signed types: shift-up then shift-down
836 // b) for unsigned types: simply mask the LSB (but shift-up and down also works)
837 ostringstream sl;
838 sl << "(" << value_expr << ")"
839 << " << "
840 << "(" << (32 - type.bits()) << ")"; // 1. shift-up to MSB
841 ostringstream rsr;
842 rsr << print_reinterpret_cast(type, sl.str()) // 2. reinterpret bits
843 << " >> " << (32 - type.bits()); // 3. shift-down to LSB
844 return rsr.str();
845}

847string CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::print_reinterpret_cast(Type type, const string &value_expr) {
848 type = type.element_of();
849
850 string cast_expr;
851 cast_expr += "as";
852 switch (type.code()) {
853 case halide_type_uint:
854 cast_expr += "uint";
855 break;
856 case halide_type_int:
857 cast_expr += "int";
858 break;
859 case halide_type_float:
860 cast_expr += "float";
861 break;
862 case halide_type_handle:
863 default:
864 cast_expr = "BADCAST";
865 user_error << "Invalid reinterpret cast.\n";
866 break;
867 }
868 cast_expr += "(" + value_expr + ")";
869 return cast_expr;
870}

872string CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::print_cast(Type target_type, Type source_type, const string &value_expr) {
873 // casting to or from a float type? just use the language cast:
874 if (target_type.is_float() || source_type.is_float()) {
875 return print_vanilla_cast(target_type, value_expr);
876 }
877
878 // casting to or from a bool type? just use the language cast:
879 if (target_type.is_bool() || source_type.is_bool()) {
880 return print_vanilla_cast(target_type, value_expr);
881 }
882
883 // let the integer cast zoo begin...
884 internal_assert(!target_type.is_float());
885 internal_assert(!source_type.is_float());
886
887 // HLSL (SM 5.1) only supports 32bit integers (signed and unsigned)...
888 // integer downcasting-to (or upcasting-from) lower bit integers require
889 // some emulation in code...
890 internal_assert(target_type.bits() >= 8);
891 internal_assert(source_type.bits() >= 8);
892 internal_assert(target_type.bits() <= 32);
893 internal_assert(source_type.bits() <= 32);
894 internal_assert(target_type.bits() % 8 == 0);
895 internal_assert(source_type.bits() % 8 == 0);
896
897 // Case 1: source and target both have the same "signess"
898 bool same_signess = false;
899 same_signess |= target_type.is_int() && source_type.is_int();
900 same_signess |= target_type.is_uint() && source_type.is_uint();
901 if (same_signess) {
902 ostringstream ss;
903 if (target_type.bits() >= source_type.bits()) {
904 // target has enough bits to fully accommodate the source:
905 // it's a no-op, but we print a vanilla cast for clarity:
906 ss << "(" << print_vanilla_cast(target_type, value_expr) << ")";
907 } else {
908 // for signed types: shift-up then shift-down
909 // for unsigned types: mask the target LSB (but shift-up and down also works)
910 ss << "("
911 << "(" << value_expr << ")"
912 << " << "
913 << "(" << (32 - target_type.bits()) << ")" // 1. shift-up to MSB
914 << ")"
915 << " >> " << (32 - target_type.bits()); // 2. shift-down to LSB
916 }
917 return ss.str();
918 }
919
920 // Case 2: casting from a signed source to an unsigned target
921 if (source_type.is_int() && target_type.is_uint()) {
922 // reinterpret resulting bits as uint(32):
923 string masked = value_expr;
924 if (target_type.bits() < 32) {
925 masked = "(" + value_expr + ")" + " & " + hex_literal((1 << target_type.bits()) - 1); // mask target LSB
926 }
927 return print_reinterpret_cast(target_type, masked);
928 }
929
930 // Case 3: casting from an unsigned source to a signed target
931 internal_assert(source_type.is_uint());
932 internal_assert(target_type.is_int());
933 ostringstream ss;
934 if (target_type.bits() > source_type.bits()) {
935 // target has enough bits to fully accommodate the source:
936 // it's a no-op, but we print a vanilla cast for clarity:
937 ss << "(" << print_vanilla_cast(target_type, value_expr) << ")";
938 } else {
939 // shift-up, reinterpret as int (target_type), then shift-down
940 ss << print_reinforced_cast(target_type, value_expr);
941 }
942 return ss.str();
943}

995
996void CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C::add_kernel(Stmt s,
997 const string &name,
998 const vector<DeviceArgument> &args) {
999
1000 debug(2) << "Adding D3D12Compute kernel " << name << "\n";
1001
1002 // Figure out which arguments should be passed in constant.
1003 // Such arguments should be:
1004 // - not written to,
1005 // - loads are block-uniform,
1006 // - constant size,
1007 // - and all allocations together should be less than the max constant
1008 // buffer size given by D3D11_REQ_CONSTANT_BUFFER_ELEMENT_COUNT*(4*sizeof(float))
1009 // The last condition is handled via the preprocessor in the kernel
1010 // declaration.
1011 vector<BufferSize> constants;
1012 auto isConstantBuffer = [&s](const DeviceArgument &arg) {
1013 return arg.is_buffer && CodeGen_GPU_Dev::is_buffer_constant(s, arg.name) && arg.size > 0;
1014 };
1015 for (const auto &arg : args) {
1016 if (isConstantBuffer(arg)) {
1017 constants.emplace_back(arg.name, arg.size);
1018 }
1019 }
1020
1021 // Sort the constant candidates from smallest to largest. This will put
1022 // as many of the constant allocations in constant as possible.
1023 // Ideally, we would prioritize constant buffers by how frequently they
1024 // are accessed.
1025 sort(constants.begin(), constants.end());
1026
1027 // Compute the cumulative sum of the constants.
1028 for (size_t i = 1; i < constants.size(); i++) {
1029 constants[i].size += constants[i - 1].size;
1030 }
1031
1032 // Find all the shared allocations, uniquify their names,
1033 // and declare them at global scope.
1034 class FindSharedAllocationsAndUniquify : public IRMutator {
1035 using IRMutator::visit;
1036 Stmt visit(const Allocate *op) override {
1037 if (is_shared_allocation(op)) {
1038 // Because these will go in global scope,
1039 // we need to ensure they have unique names.
1040 std::string new_name = unique_name(op->name);
1041 replacements[op->name] = new_name;
1042
1043 std::vector<Expr> new_extents;
1044 for (size_t i = 0; i < op->extents.size(); i++) {
1045 new_extents.push_back(mutate(op->extents[i]));
1046 }
1047 Stmt new_body = mutate(op->body);
1048 Expr new_condition = mutate(op->condition);
1049 Expr new_new_expr;
1050 if (op->new_expr.defined()) {
1051 new_new_expr = mutate(op->new_expr);
1052 }
1053
1054 Stmt new_alloc = Allocate::make(new_name, op->type, op->memory_type, new_extents,
1055 std::move(new_condition), std::move(new_body),
1056 std::move(new_new_expr), op->free_function);
1057
1058 allocs.push_back(new_alloc);
1059 replacements.erase(op->name);
1060 return new_alloc;
1061 } else {
1062 return IRMutator::visit(op);
1063 }
1064 }
1065
1066 Stmt visit(const Free *op) override {
1067 auto it = replacements.find(op->name);
1068 if (it != replacements.end()) {
1069 return Free::make(it->second);
1070 } else {
1071 return IRMutator::visit(op);
1072 }
1073 }
1074
1075 Expr visit(const Load *op) override {
1076 auto it = replacements.find(op->name);
1077 if (it != replacements.end()) {
1078 return Load::make(op->type, it->second,
1079 mutate(op->index), op->image, op->param,
1080 mutate(op->predicate), op->alignment);
1081 } else {
1082 return IRMutator::visit(op);
1083 }
1084 }
1085
1086 Stmt visit(const Store *op) override {
1087 auto it = replacements.find(op->name);
1088 if (it != replacements.end()) {
1089 return Store::make(it->second, mutate(op->value),
1090 mutate(op->index), op->param,
1091 mutate(op->predicate), op->alignment);
1092 } else {
1093 return IRMutator::visit(op);
1094 }
1095 }
1096
1097 std::map<string, string> replacements;
1098 friend class CodeGen_D3D12Compute_Dev::CodeGen_D3D12Compute_C;
1099 vector<Stmt> allocs;
1100 };
1101
1102 FindSharedAllocationsAndUniquify fsa;
1103 s = fsa.mutate(s);
1104
1105 uint32_t total_shared_bytes = 0;
1106 for (const Stmt &sop : fsa.allocs) {
1107 const Allocate *op = sop.as<Allocate>();
1108 internal_assert(op->extents.size() == 1);
1109 internal_assert(op->type.lanes() == 1);
1110 // In D3D12/HLSL, only 32bit types (int/uint/float) are suppoerted (even
1111 // though things are changing with newer shader models). Since there is
1112 // no uint8 type, we'll have to emulate it with 32bit types...
1113 // This will also require pack/unpack logic with bit-masking and aliased
1114 // type reinterpretation via asfloat()/asuint() in the shader code... :(
1115 Type smem_type = op->type;
1116 smem_type.with_bits(32);
1117 stream << "groupshared"
1118 << " " << print_type(smem_type)
1119 << " " << print_name(op->name);
1120 if (is_const(op->extents[0])) {
1121 std::stringstream ss;
1122 ss << op->extents[0];
1123 size_t elements = 0;
1124 ss >> elements;
1125 size_t bytesize = elements * sizeof(uint32_t);
1126 // NOTE(marcos): might need to resort to StoragePackUnpack::pack_storage() here...
1127 internal_assert(bytesize <= StoragePackUnpack::ThreadGroupSharedStorageLimit);
1128 total_shared_bytes += bytesize;
1129 stream << " [" << elements << "];\n";
1130 } else {
1131 // fill-in __GROUPSHARED_SIZE_IN_BYTES later on when D3DCompile()
1132 // is invoked in halide_d3d12compute_run(); must get divided by 4
1133 // since groupshared memory elements have 32bit granularity:
1134 stream << " [ ( __GROUPSHARED_SIZE_IN_BYTES + 3 ) / 4 ];\n";
1135 }
1136 if (total_shared_bytes > StoragePackUnpack::ThreadGroupSharedStorageLimit) {
1137 debug(1) << "D3D12 CodeGen ERROR: Total thread group shared memory required for kernel '" << name
1138 << "' exceeds the SM 5.1 limit of 32KB: " << total_shared_bytes << " bytes required.\n";
1139 }
1140 Allocation alloc;
1141 alloc.type = op->type;
1142 allocations.push(op->name, alloc);
1143 }
1144
1145 // Emit the kernel function preamble (numtreads):
1146 // must first figure out the thread group size by traversing the stmt:
1147 struct FindThreadGroupSize : public IRVisitor {
1148 using IRVisitor::visit;
1149 void visit(const For *loop) override {
1150 if (!is_gpu_var(loop->name)) {
1151 return loop->body.accept(this);
1152 }
1153 if (loop->for_type != ForType::GPUThread) {
1154 return loop->body.accept(this);
1155 }
1156 internal_assert(is_const_zero(loop->min));
1157 int index = thread_loop_workgroup_index(loop->name);
1158 user_assert(index >= 0) << "Invalid 'numthreads' index for loop variable '" << loop->name << "'.\n";
1159 // if 'numthreads' for a given dimension can't be determined at code
1160 // generation time, emit code such that it can be patched some point
1161 // later when calling D3DCompile() / halide_d3d12compute_run()
1162 numthreads[index] = 0; // <-- 0 indicates 'undetermined'
1163 const IntImm *int_limit = loop->extent.as<IntImm>();
1164 if (nullptr != int_limit) {
1165 numthreads[index] = int_limit->value;
1166 user_assert(numthreads[index] > 0) << "For D3D12Compute, 'numthreads[" << index << "]' values must be greater than zero.\n";
1167 }
1168 debug(4) << "Thread group size for index " << index << " is " << numthreads[index] << "\n";
1169 loop->body.accept(this);
1170 }
1171 int thread_loop_workgroup_index(const string &name) {
1172 string ids[] = {".__thread_id_x",
1173 ".__thread_id_y",
1174 ".__thread_id_z",
1175 ".__thread_id_w"};
1176 for (auto &id : ids) {
1177 if (ends_with(name, id)) {
1178 return (&id - ids);
1179 }
1180 }
1181 return -1;
1182 }
1183 int numthreads[3] = {1, 1, 1};
1184 };
1185 FindThreadGroupSize ftg;
1186 s.accept(&ftg);
1187 // for undetermined 'numthreads' dimensions, insert placeholders to the code
1188 // such as '__NUM_TREADS_X' that will later be patched when D3DCompile() is
1189 // invoked in halide_d3d12compute_run()
1190 stream << "[ numthreads(";
1191 (ftg.numthreads[0] > 0) ? (stream << " " << ftg.numthreads[0]) : (stream << " __NUM_TREADS_X ");
1192 (ftg.numthreads[1] > 0) ? (stream << ", " << ftg.numthreads[1]) : (stream << ", __NUM_TREADS_Y ");
1193 (ftg.numthreads[2] > 0) ? (stream << ", " << ftg.numthreads[2]) : (stream << ", __NUM_TREADS_Z ");
1194 stream << ") ]\n";
1195
1196 // Emit the kernel function prototype:
1197
1198 stream << "void " << name << "(\n";
1199 stream << " "
1200 << "uint3 tgroup_index : SV_GroupID,\n"
1201 << " "
1202 << "uint3 tid_in_tgroup : SV_GroupThreadID";
1203 for (const auto &arg : args) {
1204 stream << ",\n";
1205 stream << " ";
1206 if (arg.is_buffer) {
1207 // NOTE(marcos): Passing all buffers as RWBuffers in order to bind
1208 // all buffers as UAVs since there is no way the runtime can know
1209 // if a given halide_buffer_t is read-only (SRV) or read-write...
1210 stream << "RW"
1211 << "Buffer"
1212 << "<" << print_type(arg.type) << ">"
1213 << " " << print_name(arg.name);
1214 Allocation alloc;
1215 alloc.type = arg.type;
1216 allocations.push(arg.name, alloc);
1217 } else {
1218 stream << "uniform"
1219 << " " << print_type(arg.type)
1220 << " " << print_name(arg.name);
1221 }
1222 }
1223 stream << ")\n";
1224
1225 // Emit the kernel code:
1226
1227 open_scope();
1228 print(s);
1229 close_scope("kernel " + name);
1230
1231 for (size_t i = 0; i < args.size(); i++) {
1232 // Remove buffer arguments from allocation scope
1233 if (args[i].is_buffer) {
1234 allocations.pop(args[i].name);
1235 }
1236 }
1237
1238 stream << "\n";
1239}

1241void CodeGen_D3D12Compute_Dev::init_module() {
1242 debug(2) << "D3D12Compute device codegen init_module\n";
1243
1244 // wipe the internal kernel source
1245 src_stream.str("");
1246 src_stream.clear();
1247
1248 // compiler control pragmas
1249 src_stream
1250 // Disable innocent warnings:
1251 // warning X3078 : loop control variable conflicts with a previous declaration in the outer scope; most recent declaration will be used
1252 << "#pragma warning( disable : 3078 )"
1253 "\n"
1254 // warning X3557: loop only executes for 1 iteration(s), forcing loop to unroll
1255 << "#pragma warning( disable : 3557 )"
1256 "\n"
1257 // Disable more serious warnings:
1258 // TODO(marcos): should revisit the warnings below, as they are likely to impact performance (and possibly correctness too)
1259 // warning X3556 : integer modulus may be much slower, try using uints if possible
1260 // TODO(marcos): can we interchangeably replace ints by uints when we have modulo operations in the generated code?
1261 << "#pragma warning( disable : 3556 )"
1262 "\n"
1263 // warning X3571 : pow(f, e) will not work for negative f, use abs(f) or conditionally handle negative values if you expect them
1264 << "#pragma warning( disable : 3571 )"
1265 "\n"
1266 // warning X4714 : sum of temp registers and indexable temp registers times 256 threads exceeds the recommended total 16384. Performance may be reduced
