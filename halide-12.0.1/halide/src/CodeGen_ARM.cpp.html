201struct ArmIntrinsic {
202 const char *arm32;
203 const char *arm64;
204 halide_type_t ret_type;
205 const char *name;
206 halide_type_t arg_types[max_intrinsic_args];
207 int flags;
208 enum {
209 AllowUnsignedOp1 = 1 << 0, // Generate a second version of the instruction with the second operand unsigned.
210 HalfWidth = 1 << 1, // This is a half-width instruction that should have a full width version generated as well.
211 NoMangle = 1 << 2, // Don't mangle this intrinsic name.
212 MangleArgs = 1 << 3, // Most intrinsics only mangle the return type. Some mangle the arguments instead.
213 MangleRetArgs = 1 << 4, // Most intrinsics only mangle the return type. Some mangle the return type and arguments instead.
214 ScalarsAreVectors = 1 << 5, // Some intrinsics have scalar arguments that are vector parameters :(
215 SplitArg0 = 1 << 6, // This intrinsic requires splitting the argument into the low and high halves.
216 NoPrefix = 1 << 7, // Don't prefix the intrinsic with llvm.*
217 };
218};

572llvm::Function *CodeGen_ARM::define_concat_args_wrapper(llvm::Function *inner, const string &name) {
573 llvm::FunctionType *inner_ty = inner->getFunctionType();
574
575 internal_assert(inner_ty->getNumParams() == 2);
576 llvm::Type *inner_arg0_ty = inner_ty->getParamType(0);
577 llvm::Type *inner_arg1_ty = inner_ty->getParamType(1);
578 int inner_arg0_lanes = get_vector_num_elements(inner_arg0_ty);
579 int inner_arg1_lanes = get_vector_num_elements(inner_arg1_ty);
580
581 llvm::Type *concat_arg_ty =
582 get_vector_type(inner_arg0_ty->getScalarType(), inner_arg0_lanes + inner_arg1_lanes);
583
584 // Make a wrapper.
585 llvm::FunctionType *wrapper_ty =
586 llvm::FunctionType::get(inner_ty->getReturnType(), {concat_arg_ty}, false);
587 llvm::Function *wrapper =
588 llvm::Function::Create(wrapper_ty, llvm::GlobalValue::InternalLinkage, name, module.get());
589 llvm::BasicBlock *block =
590 llvm::BasicBlock::Create(module->getContext(), "entry", wrapper);
591 IRBuilderBase::InsertPoint here = builder->saveIP();
592 builder->SetInsertPoint(block);
593
594 // Call the real intrinsic.
595 Value *low = slice_vector(wrapper->getArg(0), 0, inner_arg0_lanes);
596 Value *high = slice_vector(wrapper->getArg(0), inner_arg0_lanes, inner_arg1_lanes);
597 Value *ret = builder->CreateCall(inner, {low, high});
598 builder->CreateRet(ret);
599
600 // Always inline these wrappers.
601 wrapper->addFnAttr(llvm::Attribute::AlwaysInline);
602
603 builder->restoreIP(here);
604
605 llvm::verifyFunction(*wrapper);
606 return wrapper;
607}

609void CodeGen_ARM::init_module() {
610 CodeGen_Posix::init_module();
611
612 if (neon_intrinsics_disabled()) {
613 return;
614 }
615
616 string prefix = target.bits == 32 ? "llvm.arm.neon." : "llvm.aarch64.neon.";
617 for (const ArmIntrinsic &intrin : intrinsic_defs) {
618 // Get the name of the intrinsic with the appropriate prefix.
619 const char *intrin_name = nullptr;
620 if (target.bits == 32) {
621 intrin_name = intrin.arm32;
622 } else {
623 intrin_name = intrin.arm64;
624 }
625 if (!intrin_name) {
626 continue;
627 }
628 string full_name = intrin_name;
629 if (!starts_with(full_name, "llvm.") && (intrin.flags & ArmIntrinsic::NoPrefix) == 0) {
630 full_name = prefix + full_name;
631 }
632
633 // We might have to generate versions of this intrinsic with multiple widths.
634 vector<int> width_factors = {1};
635 if (intrin.flags & ArmIntrinsic::HalfWidth) {
636 width_factors.push_back(2);
637 }
638
639 for (int width_factor : width_factors) {
640 Type ret_type = intrin.ret_type;
641 ret_type = ret_type.with_lanes(ret_type.lanes() * width_factor);
642 internal_assert(ret_type.bits() * ret_type.lanes() <= 128) << full_name << "\n";
643 vector<Type> arg_types;
644 arg_types.reserve(4);
645 for (halide_type_t i : intrin.arg_types) {
646 if (i.bits == 0) {
647 break;
648 }
649 Type arg_type = i;
650 if (arg_type.is_vector()) {
651 arg_type = arg_type.with_lanes(arg_type.lanes() * width_factor);
652 }
653 arg_types.emplace_back(arg_type);
654 }
655
656 // Generate the LLVM mangled name.
657 std::stringstream mangled_name_builder;
658 mangled_name_builder << full_name;
659 if (starts_with(full_name, "llvm.") && (intrin.flags & ArmIntrinsic::NoMangle) == 0) {
660 // Append LLVM name mangling for either the return type or the arguments, or both.
661 vector<Type> types;
662 if (intrin.flags & ArmIntrinsic::MangleArgs) {
663 types = arg_types;
664 } else if (intrin.flags & ArmIntrinsic::MangleRetArgs) {
665 types = {ret_type};
666 types.insert(types.end(), arg_types.begin(), arg_types.end());
667 } else {
668 types = {ret_type};
669 }
670 for (const Type &t : types) {
671 mangled_name_builder << ".v" << t.lanes();
672 if (t.is_int() || t.is_uint()) {
673 mangled_name_builder << "i";
674 } else if (t.is_float()) {
675 mangled_name_builder << "f";
676 }
677 mangled_name_builder << t.bits();
678 }
679 }
680 string mangled_name = mangled_name_builder.str();
681
682 llvm::Function *intrin_impl = nullptr;
683 if (intrin.flags & ArmIntrinsic::SplitArg0) {
684 // This intrinsic needs a wrapper to split the argument.
685 string wrapper_name = intrin.name + unique_name("_wrapper");
686 Type split_arg_type = arg_types[0].with_lanes(arg_types[0].lanes() / 2);
687 llvm::Function *to_wrap = get_llvm_intrin(ret_type, mangled_name, {split_arg_type, split_arg_type});
688 intrin_impl = define_concat_args_wrapper(to_wrap, wrapper_name);
689 } else {
690 bool scalars_are_vectors = intrin.flags & ArmIntrinsic::ScalarsAreVectors;
691 intrin_impl = get_llvm_intrin(ret_type, mangled_name, arg_types, scalars_are_vectors);
692 }
693
694 intrin_impl->addFnAttr(llvm::Attribute::ReadNone);
695 intrin_impl->addFnAttr(llvm::Attribute::NoUnwind);
696 declare_intrin_overload(intrin.name, ret_type, intrin_impl, arg_types);
697 if (intrin.flags & ArmIntrinsic::AllowUnsignedOp1) {
698 // Also generate a version of this intrinsic where the second operand is unsigned.
699 arg_types[1] = arg_types[1].with_code(halide_type_uint);
700 declare_intrin_overload(intrin.name, ret_type, intrin_impl, arg_types);
701 }
702 }
703 }

706void CodeGen_ARM::visit(const Cast *op) {
707 if (!neon_intrinsics_disabled() && op->type.is_vector()) {
708 vector<Expr> matches;
709 for (const Pattern &pattern : casts) {
710 if (expr_match(pattern.pattern, op, matches)) {
711 if (pattern.intrin.find("shift_right_narrow") != string::npos) {
712 // The shift_right_narrow patterns need the shift to be constant in [1, output_bits].
713 const uint64_t *const_b = as_const_uint(matches[1]);
714 if (!const_b || *const_b == 0 || (int)*const_b > op->type.bits()) {
715 continue;
716 }
717 }
718 if (target.bits == 32 && pattern.intrin.find("shift_right") != string::npos) {
719 // The 32-bit ARM backend wants right shifts as negative values.
720 matches[1] = simplify(-cast(matches[1].type().with_code(halide_type_int), matches[1]));
721 }
722 value = call_overloaded_intrin(op->type, pattern.intrin, matches);
723 if (value) {
724 return;
725 }
726 }
727 }
728
729 // Catch signed widening of absolute difference.
730 // Catch widening of absolute difference
731 Type t = op->type;
732 if ((t.is_int() || t.is_uint()) &&
733 (op->value.type().is_int() || op->value.type().is_uint()) &&
734 t.bits() == op->value.type().bits() * 2) {
735 if (const Call *absd = Call::as_intrinsic(op->value, {Call::absd})) {
736 value = call_overloaded_intrin(t, "widening_absd", absd->args);
737 return;
738 }
739 }
740
741 // If we didn't find a pattern, try rewriting the cast.
742 static const vector<pair<Expr, Expr>> cast_rewrites = {
743 // Double or triple narrowing saturating casts are better expressed as
744 // regular narrowing casts.
745 {u8_sat(wild_u32x_), u8_sat(u16_sat(wild_u32x_))},
746 {u8_sat(wild_i32x_), u8_sat(i16_sat(wild_i32x_))},
747 {i8_sat(wild_u32x_), i8_sat(u16_sat(wild_u32x_))},
748 {i8_sat(wild_i32x_), i8_sat(i16_sat(wild_i32x_))},
749 {u16_sat(wild_u64x_), u16_sat(u32_sat(wild_u64x_))},
750 {u16_sat(wild_i64x_), u16_sat(i32_sat(wild_i64x_))},
751 {i16_sat(wild_u64x_), i16_sat(u32_sat(wild_u64x_))},
752 {i16_sat(wild_i64x_), i16_sat(i32_sat(wild_i64x_))},
753 {u8_sat(wild_u64x_), u8_sat(u16_sat(u32_sat(wild_u64x_)))},
754 {u8_sat(wild_i64x_), u8_sat(i16_sat(i32_sat(wild_i64x_)))},
755 {i8_sat(wild_u64x_), i8_sat(u16_sat(u32_sat(wild_u64x_)))},
756 {i8_sat(wild_i64x_), i8_sat(i16_sat(i32_sat(wild_i64x_)))},
757 };
758 for (const auto &i : cast_rewrites) {
759 if (expr_match(i.first, op, matches)) {
760 Expr replacement = substitute("*", matches[0], with_lanes(i.second, op->type.lanes()));
761 debug(3) << "rewriting cast to: " << replacement << " from " << Expr(op) << "\n";
762 value = codegen(replacement);
763 return;
764 }
765 }
766 }
767
768 CodeGen_Posix::visit(op);
769}

837void CodeGen_ARM::visit(const Store *op) {
838 // Predicated store
839 if (!is_const_one(op->predicate)) {
840 CodeGen_Posix::visit(op);
841 return;
842 }
843
844 if (neon_intrinsics_disabled()) {
845 CodeGen_Posix::visit(op);
846 return;
847 }
848
849 // A dense store of an interleaving can be done using a vst2 intrinsic
850 const Ramp *ramp = op->index.as<Ramp>();
851
852 // We only deal with ramps here
853 if (!ramp) {
854 CodeGen_Posix::visit(op);
855 return;
856 }
857
858 // First dig through let expressions
859 Expr rhs = op->value;
860 vector<pair<string, Expr>> lets;
861 while (const Let *let = rhs.as<Let>()) {
862 rhs = let->body;
863 lets.emplace_back(let->name, let->value);
864 }
865 const Shuffle *shuffle = rhs.as<Shuffle>();
866
867 // Interleaving store instructions only exist for certain types.
868 bool type_ok_for_vst = false;
869 Type intrin_type = Handle();
870 if (shuffle) {
871 Type t = shuffle->vectors[0].type();
872 intrin_type = t;
873 Type elt = t.element_of();
874 int vec_bits = t.bits() * t.lanes();
875 if (elt == Float(32) ||
876 elt == Int(8) || elt == Int(16) || elt == Int(32) ||
877 elt == UInt(8) || elt == UInt(16) || elt == UInt(32)) {
878 if (vec_bits % 128 == 0) {
879 type_ok_for_vst = true;
880 intrin_type = intrin_type.with_lanes(128 / t.bits());
881 } else if (vec_bits % 64 == 0) {
882 type_ok_for_vst = true;
883 intrin_type = intrin_type.with_lanes(64 / t.bits());
884 }
885 }
886 }
887
888 if (is_const_one(ramp->stride) &&
889 shuffle && shuffle->is_interleave() &&
890 type_ok_for_vst &&
891 2 <= shuffle->vectors.size() && shuffle->vectors.size() <= 4) {
892
893 const int num_vecs = shuffle->vectors.size();
894 vector<Value *> args(num_vecs);
895
896 Type t = shuffle->vectors[0].type();
897
898 // Assume element-aligned.
899 int alignment = t.bytes();
900
901 // Codegen the lets
902 for (size_t i = 0; i < lets.size(); i++) {
903 sym_push(lets[i].first, codegen(lets[i].second));
904 }
905
906 // Codegen all the vector args.
907 for (int i = 0; i < num_vecs; ++i) {
908 args[i] = codegen(shuffle->vectors[i]);
909 }
910
911 // Declare the function
912 std::ostringstream instr;
913 vector<llvm::Type *> arg_types;
914 if (target.bits == 32) {
915 instr << "llvm.arm.neon.vst"
916 << num_vecs
917 << ".p0i8"
918 << ".v"
919 << intrin_type.lanes()
920 << (t.is_float() ? 'f' : 'i')
921 << t.bits();
922 arg_types = vector<llvm::Type *>(num_vecs + 2, llvm_type_of(intrin_type));
923 arg_types.front() = i8_t->getPointerTo();
924 arg_types.back() = i32_t;
925 } else {
926 instr << "llvm.aarch64.neon.st"
927 << num_vecs
928 << ".v"
929 << intrin_type.lanes()
930 << (t.is_float() ? 'f' : 'i')
931 << t.bits()
932 << ".p0"
933 << (t.is_float() ? 'f' : 'i')
934 << t.bits();
935 arg_types = vector<llvm::Type *>(num_vecs + 1, llvm_type_of(intrin_type));
936 arg_types.back() = llvm_type_of(intrin_type.element_of())->getPointerTo();
937 }
938 llvm::FunctionType *fn_type = FunctionType::get(llvm::Type::getVoidTy(*context), arg_types, false);
939 llvm::FunctionCallee fn = module->getOrInsertFunction(instr.str(), fn_type);
940 internal_assert(fn);
941
942 // How many vst instructions do we need to generate?
943 int slices = t.lanes() / intrin_type.lanes();
944
945 internal_assert(slices >= 1);
946 for (int i = 0; i < t.lanes(); i += intrin_type.lanes()) {
947 Expr slice_base = simplify(ramp->base + i * num_vecs);
948 Expr slice_ramp = Ramp::make(slice_base, ramp->stride, intrin_type.lanes() * num_vecs);
949 Value *ptr = codegen_buffer_pointer(op->name, shuffle->vectors[0].type().element_of(), slice_base);
950
951 vector<Value *> slice_args = args;
952 // Take a slice of each arg
953 for (int j = 0; j < num_vecs; j++) {
954 slice_args[j] = slice_vector(slice_args[j], i, intrin_type.lanes());
955 }
956
957 if (target.bits == 32) {
958 // The arm32 versions take an i8*, regardless of the type stored.
959 ptr = builder->CreatePointerCast(ptr, i8_t->getPointerTo());
960 // Set the pointer argument
961 slice_args.insert(slice_args.begin(), ptr);
962 // Set the alignment argument
963 slice_args.push_back(ConstantInt::get(i32_t, alignment));
964 } else {
965 // Set the pointer argument
966 slice_args.push_back(ptr);
967 }
968
969 CallInst *store = builder->CreateCall(fn, slice_args);
970 add_tbaa_metadata(store, op->name, slice_ramp);
971 }
972
973 // pop the lets from the symbol table
974 for (size_t i = 0; i < lets.size(); i++) {
975 sym_pop(lets[i].first);
976 }
977
978 return;
979 }
980
981 // If the stride is one or minus one, we can deal with that using vanilla codegen
982 const IntImm *stride = ramp->stride.as<IntImm>();
983 if (stride && (stride->value == 1 || stride->value == -1)) {
984 CodeGen_Posix::visit(op);
985 return;
986 }
987
988 // We have builtins for strided stores with fixed but unknown stride, but they use inline assembly
989 if (target.bits != 64 /* Not yet implemented for aarch64 */) {
990 ostringstream builtin;
991 builtin << "strided_store_"
992 << (op->value.type().is_float() ? "f" : "i")
993 << op->value.type().bits()
994 << "x" << op->value.type().lanes();
995
996 llvm::Function *fn = module->getFunction(builtin.str());
997 if (fn) {
998 Value *base = codegen_buffer_pointer(op->name, op->value.type().element_of(), ramp->base);
999 Value *stride = codegen(ramp->stride * op->value.type().bytes());
1000 Value *val = codegen(op->value);
1001 debug(4) << "Creating call to " << builtin.str() << "\n";
1002 Value *store_args[] = {base, stride, val};
1003 Instruction *store = builder->CreateCall(fn, store_args);
1004 (void)store;
1005 add_tbaa_metadata(store, op->name, op->index);
1006 return;
1007 }
1008 }
1009
1010 CodeGen_Posix::visit(op);
1011}

1064void CodeGen_ARM::visit(const Call *op) {
1065 if (op->is_intrinsic(Call::sorted_avg)) {
1066 value = codegen(halving_add(op->args[0], op->args[1]));
1067 return;
1068 }
1069
1070 if (op->is_intrinsic(Call::rounding_shift_right)) {
1071 // LLVM wants these as rounding_shift_left with a negative b instead.
1072 Expr b = op->args[1];
1073 if (!b.type().is_int()) {
1074 b = Cast::make(b.type().with_code(halide_type_int), b);
1075 }
1076 value = codegen(rounding_shift_left(op->args[0], simplify(-b)));
1077 return;
1078 } else if (op->is_intrinsic(Call::widening_shift_right) && op->args[1].type().is_int()) {
1079 // We want these as left shifts with a negative b instead.
1080 value = codegen(widening_shift_left(op->args[0], simplify(-op->args[1])));
1081 return;
1082 } else if (op->is_intrinsic(Call::shift_right) && op->args[1].type().is_int()) {
1083 // We want these as left shifts with a negative b instead.
1084 value = codegen(op->args[0] << simplify(-op->args[1]));
1085 return;
1086 }
1087
1088 if (op->type.is_vector()) {
1089 vector<Expr> matches;
1090 for (const Pattern &pattern : calls) {
1091 if (expr_match(pattern.pattern, op, matches)) {
1092 value = call_overloaded_intrin(op->type, pattern.intrin, matches);
1093 if (value) {
1094 return;
1095 }
1096 }
1097 }
1098 }
1099
1100 CodeGen_Posix::visit(op);
1101}

1131void CodeGen_ARM::codegen_vector_reduce(const VectorReduce *op, const Expr &init) {
1132 if (neon_intrinsics_disabled() ||
1133 op->op == VectorReduce::Or ||
1134 op->op == VectorReduce::And ||
1135 op->op == VectorReduce::Mul) {
1136 CodeGen_Posix::codegen_vector_reduce(op, init);
1137 return;
1138 }
1139
1140 struct Pattern {
1141 VectorReduce::Operator reduce_op;
1142 int factor;
1143 Expr pattern;
1144 const char *intrin;
1145 Target::Feature required_feature;
1146 std::vector<int> extra_operands;
1147 };
1148 // clang-format off
1149 static const Pattern patterns[] = {
1150 {VectorReduce::Add, 4, i32(widening_mul(wild_i8x_, wild_i8x_)), "dot_product", Target::ARMDotProd},
1151 {VectorReduce::Add, 4, i32(widening_mul(wild_u8x_, wild_u8x_)), "dot_product", Target::ARMDotProd},
1152 {VectorReduce::Add, 4, u32(widening_mul(wild_u8x_, wild_u8x_)), "dot_product", Target::ARMDotProd},
1153 // A sum is the same as a dot product with a vector of ones, and this appears to
1154 // be a bit faster.
1155 {VectorReduce::Add, 4, i32(wild_i8x_), "dot_product", Target::ARMDotProd, {1}},
1156 {VectorReduce::Add, 4, i32(wild_u8x_), "dot_product", Target::ARMDotProd, {1}},
1157 {VectorReduce::Add, 4, u32(wild_u8x_), "dot_product", Target::ARMDotProd, {1}},
1158 };
1159 // clang-format on
1160
1161 int factor = op->value.type().lanes() / op->type.lanes();
1162 vector<Expr> matches;
1163 for (const Pattern &p : patterns) {
1164 if (op->op != p.reduce_op || factor % p.factor != 0) {
1165 continue;
1166 }
1167 if (!target.has_feature(p.required_feature)) {
1168 continue;
1169 }
1170 if (expr_match(p.pattern, op->value, matches)) {
1171 if (factor != p.factor) {
1172 Expr equiv = VectorReduce::make(op->op, op->value, op->value.type().lanes() / p.factor);
1173 equiv = VectorReduce::make(op->op, equiv, op->type.lanes());
1174 codegen_vector_reduce(equiv.as<VectorReduce>(), init);
1175 return;
1176 }
1177
1178 for (int i : p.extra_operands) {
1179 matches.push_back(make_const(matches[0].type(), i));
1180 }
1181
1182 Expr i = init;
1183 if (!i.defined()) {
1184 i = make_zero(op->type);
1185 }
1186 if (const Shuffle *s = matches[0].as<Shuffle>()) {
1187 if (s->is_broadcast()) {
1188 // LLVM wants the broadcast as the second operand for the broadcasting
1189 // variant of udot/sdot.
1190 std::swap(matches[0], matches[1]);
1191 }
1192 }
1193 value = call_overloaded_intrin(op->type, p.intrin, {i, matches[0], matches[1]});
1194 if (value) {
1195 return;
1196 }
1197 }
1198 }
1199
1200 // TODO: Move this to be patterns? The patterns are pretty trivial, but some
1201 // of the other logic is tricky.
1202 const char *intrin = nullptr;
1203 vector<Expr> intrin_args;
1204 Expr accumulator = init;
1205 if (op->op == VectorReduce::Add && factor == 2) {
1206 Type narrow_type = op->type.narrow().with_lanes(op->value.type().lanes());
1207 Expr narrow = lossless_cast(narrow_type, op->value);
1208 if (!narrow.defined() && op->type.is_int()) {
1209 // We can also safely accumulate from a uint into a
1210 // wider int, because the addition uses at most one
1211 // extra bit.
1212 narrow = lossless_cast(narrow_type.with_code(Type::UInt), op->value);
1213 }
1214 if (narrow.defined()) {
1215 if (init.defined() && target.bits == 32) {
1216 // On 32-bit, we have an intrinsic for widening add-accumulate.
1217 intrin = "pairwise_widening_add_accumulate";
1218 intrin_args = {accumulator, narrow};
1219 accumulator = Expr();
1220 } else {
1221 // On 64-bit, LLVM pattern matches widening add-accumulate if
1222 // we give it the widening add.
1223 intrin = "pairwise_widening_add";
1224 intrin_args = {narrow};
1225 }
1226 } else {
1227 intrin = "pairwise_add";
1228 intrin_args = {op->value};
1229 }
1230 } else if (op->op == VectorReduce::Min && factor == 2) {
1231 intrin = "pairwise_min";
1232 intrin_args = {op->value};
1233 } else if (op->op == VectorReduce::Max && factor == 2) {
1234 intrin = "pairwise_max";
1235 intrin_args = {op->value};
1236 }
1237
1238 if (intrin) {
1239 value = call_overloaded_intrin(op->type, intrin, intrin_args);
1240 if (value) {
1241 if (accumulator.defined()) {
1242 // We still have an initial value to take care of
1243 string n = unique_name('t');
1244 sym_push(n, value);
1245 Expr v = Variable::make(accumulator.type(), n);
1246 switch (op->op) {
1247 case VectorReduce::Add:
1248 accumulator += v;
1249 break;
1250 case VectorReduce::Min:
1251 accumulator = min(accumulator, v);
1252 break;
1253 case VectorReduce::Max:
1254 accumulator = max(accumulator, v);
1255 break;
1256 default:
1257 internal_error << "unreachable";
1258 }
1259 codegen(accumulator);
1260 sym_pop(n);
1261 }
1262 return;
1263 }
1264 }
1265
1266 CodeGen_Posix::codegen_vector_reduce(op, init);
1267}
