21/** Find all calls arguments to the given function. Substitutes in
22 * lets, so take care with the combinatorially large results. */
23class FindLoads : public IRVisitor {
24 using IRVisitor::visit;
25
26 const string &func;
27
28 void visit(const Call *op) override {
29 if (op->name == func && op->call_type == Call::Halide) {
30 loads.push_back(op->args);
31 }
32 IRVisitor::visit(op);
33 }
34
35 void visit(const Let *op) override {
36 IRVisitor::visit(op);
37 for (size_t i = 0; i < loads.size(); i++) {
38 for (size_t j = 0; j < loads[i].size(); j++) {
39 loads[i][j] = graph_substitute(op->name, op->value, loads[i][j]);
40 }
41 }
42 }
43
44public:
45 FindLoads(const string &f)
46 : func(f) {
47 }
48
49 vector<vector<Expr>> loads;
50};

52/** Rename all free variables to unique new names. */
53class RenameFreeVars : public IRMutator {
54 using IRMutator::visit;
55
56 map<string, string> new_names;
57
58 Expr visit(const Variable *op) override {
59 if (!op->param.defined() && !op->image.defined()) {
60 return Variable::make(op->type, get_new_name(op->name));
61 } else {
62 return op;
63 }
64 }
65
66public:
67 string get_new_name(const string &s) {
68 map<string, string>::iterator iter = new_names.find(s);
69 if (iter != new_names.end()) {
70 return iter->second;
71 } else {
72 string new_name = s + "$_";
73 new_names[s] = new_name;
74 return new_name;
75 }
76 }
77};
78
79/** Substitute in boolean expressions. */
80class SubstituteInBooleanLets : public IRMutator {
81 using IRMutator::visit;
82
83 Expr visit(const Let *op) override {
84 if (op->value.type() == Bool()) {
85 return substitute(op->name, mutate(op->value), mutate(op->body));
86 } else {
87 return IRMutator::visit(op);
88 }
89 }
90};

93bool can_parallelize_rvar(const string &v,
94 const string &f,
95 const Definition &r) {
96 const vector<Expr> &values = r.values();
97 const vector<Expr> &args = r.args();
98 const vector<ReductionVariable> &rvars = r.schedule().rvars();
99
100 FindLoads find(f);
101 for (size_t i = 0; i < values.size(); i++) {
102 values[i].accept(&find);
103 }
104
105 // Make an expr representing the store done by a different thread.
106 RenameFreeVars renamer;
107 vector<Expr> other_store(args.size());
108 for (size_t i = 0; i < args.size(); i++) {
109 other_store[i] = renamer.mutate(args[i]);
110 }
111
112 // Construct an expression which is true when the two threads are
113 // in fact two different threads. We'll use this liberally in the
114 // following conditions to give the simplifier the best chance.
115 Expr distinct_v = (Variable::make(Int(32), v) !=
116 Variable::make(Int(32), renamer.get_new_name(v)));
117
118 // Construct an expression which is true if there's a collision
119 // between this thread's store and the other thread's store.
120 Expr hazard = const_true();
121 for (size_t i = 0; i < args.size(); i++) {
122 hazard = hazard && (distinct_v && (args[i] == other_store[i]));
123 }
124
125 // Add expressions that are true if there's a collision between
126 // the other thread's store and this thread's loads.
127 for (size_t i = 0; i < find.loads.size(); i++) {
128 internal_assert(find.loads[i].size() == other_store.size());
129 Expr check = const_true();
130 for (size_t j = 0; j < find.loads[i].size(); j++) {
131 check = check && (distinct_v && (find.loads[i][j] == other_store[j]));
132 }
133 hazard = hazard || check;
134 }
135
136 // Make a scope representing the bounds of the reduction domain
137 Scope<Interval> bounds;
138 for (const auto &rv : rvars) {
139 Interval in = Interval(rv.min, simplify(rv.min + rv.extent - 1));
140 bounds.push(rv.var, in);
141 bounds.push(renamer.get_new_name(rv.var), in);
142 }
143
144 // Add the definition's predicate if there is any
145 Expr pred = simplify(r.predicate());
146 if (pred.defined() || !equal(const_true(), pred)) {
147 Expr this_pred = pred;
148 Expr other_pred = renamer.mutate(pred);
149 debug(3) << "......this thread predicate: " << this_pred << "\n";
150 debug(3) << "......other thread predicate: " << other_pred << "\n";
151 hazard = hazard && this_pred && other_pred;
152 }
153
154 debug(3) << "Attempting to falsify: " << hazard << "\n";
155 // Pull out common non-boolean terms
156 hazard = common_subexpression_elimination(hazard);
157 hazard = SubstituteInBooleanLets().mutate(hazard);
158 hazard = simplify(hazard, false, bounds);
159 debug(3) << "Simplified to: " << hazard << "\n";
160
161 // strip lets
162 while (const Let *l = hazard.as<Let>()) {
163 hazard = l->body;
164 }
165
166 return is_const_zero(hazard);
167}
