17class UnrollLoops : public IRMutator {
18 using IRMutator::visit;
19
20 vector<pair<std::string, Expr>> lets;
21
22 Stmt visit(const LetStmt *op) override {
23 if (is_pure(op->value)) {
24 lets.emplace_back(op->name, op->value);
25 Stmt s = IRMutator::visit(op);
26 lets.pop_back();
27 return s;
28 } else {
29 return IRMutator::visit(op);
30 }
31 }
32
33 Stmt visit(const For *for_loop) override {
34 if (for_loop->for_type == ForType::Unrolled) {
35 // Give it one last chance to simplify to an int
36 Expr extent = simplify(for_loop->extent);
37 Stmt body = for_loop->body;
38 const IntImm *e = extent.as<IntImm>();
39
40 if (e == nullptr) {
41 // We're about to hard fail. Get really aggressive
42 // with the simplifier.
43 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
44 extent = Let::make(it->first, it->second, extent);
45 }
46 extent = remove_likelies(extent);
47 extent = substitute_in_all_lets(extent);
48 extent = simplify(extent);
49 e = extent.as<IntImm>();
50 }
51
52 Expr extent_upper;
53 bool use_guard = false;
54 if (e == nullptr) {
55 // Still no luck. Try taking an upper bound and
56 // injecting an if statement around the body.
57 extent_upper = find_constant_bound(extent, Direction::Upper, Scope<Interval>());
58 if (extent_upper.defined()) {
59 e = extent_upper.as<IntImm>();
60 use_guard = true;
61 }
62 }
63
64 if (e == nullptr && permit_failed_unroll) {
65 // Still no luck, but we're allowed to fail. Rewrite
66 // to a serial loop.
67 user_warning << "HL_PERMIT_FAILED_UNROLL is allowing us to unroll a non-constant loop into a serial loop. Did you mean to do this?\n";
68 body = mutate(body);
69 return For::make(for_loop->name, for_loop->min, for_loop->extent,
70 ForType::Serial, for_loop->device_api, std::move(body));
71 }
72
73 user_assert(e)
74 << "Can only unroll for loops over a constant extent.\n"
75 << "Loop over " << for_loop->name << " has extent " << extent << ".\n";
76 body = mutate(body);
77
78 if (e->value == 1) {
79 user_warning << "Warning: Unrolling a for loop of extent 1: " << for_loop->name << "\n";
80 }
81
82 Stmt iters;
83 for (int i = e->value - 1; i >= 0; i--) {
84 Stmt iter = substitute(for_loop->name, for_loop->min + i, body);
85 if (!iters.defined()) {
86 iters = iter;
87 } else {
88 iters = Block::make(iter, iters);
89 }
90 if (use_guard) {
91 iters = IfThenElse::make(likely_if_innermost(i < for_loop->extent), iters);
92 }
93 }
94
95 return simplify(iters);
96
97 } else {
98 return IRMutator::visit(for_loop);
99 }
100 }
101 bool permit_failed_unroll = false;
102
103public:
104 UnrollLoops() {
105 // Experimental autoschedulers may want to unroll without
106 // being totally confident the loop will indeed turn out
107 // to be constant-sized. If this feature continues to be
108 // important, we need to expose it in the scheduling
109 // language somewhere, but how? For now we do something
110 // ugly and expedient.
111
112 // For the tracking issue to fix this, see
113 // https://github.com/halide/Halide/issues/3479
114 permit_failed_unroll = get_env_variable("HL_PERMIT_FAILED_UNROLL") == "1";
115 }
116};
