20class DebugToFile : public IRMutator {
21 const map<string, Function> &env;
22
23 using IRMutator::visit;
24
25 Stmt visit(const Realize *op) override {
26 map<string, Function>::const_iterator iter = env.find(op->name);
27 if (iter != env.end() && !iter->second.debug_file().empty()) {
28 Function f = iter->second;
29 vector<Expr> args;
30
31 user_assert(op->types.size() == 1)
32 << "debug_to_file doesn't handle functions with multiple values yet\n";
33
34 // The name of the file
35 args.emplace_back(f.debug_file());
36
37 // Inject loads to the corners of the function so that any
38 // passes doing further analysis of buffer use understand
39 // what we're doing (e.g. so we trigger a copy-back from a
40 // device pointer).
41 Expr num_elements = 1;
42 for (size_t i = 0; i < op->bounds.size(); i++) {
43 num_elements *= op->bounds[i].extent;
44 }
45
46 int type_code = 0;
47 Type t = op->types[0];
48 if (t == Float(32)) {
49 type_code = 0;
50 } else if (t == Float(64)) {
51 type_code = 1;
52 } else if (t == UInt(8) || t == UInt(1)) {
53 type_code = 2;
54 } else if (t == Int(8)) {
55 type_code = 3;
56 } else if (t == UInt(16)) {
57 type_code = 4;
58 } else if (t == Int(16)) {
59 type_code = 5;
60 } else if (t == UInt(32)) {
61 type_code = 6;
62 } else if (t == Int(32)) {
63 type_code = 7;
64 } else if (t == UInt(64)) {
65 type_code = 8;
66 } else if (t == Int(64)) {
67 type_code = 9;
68 } else {
69 user_error << "Type " << t << " not supported for debug_to_file\n";
70 }
71 args.emplace_back(type_code);
72
73 Expr buf = Variable::make(Handle(), f.name() + ".buffer");
74 args.push_back(buf);
75
76 Expr call = Call::make(Int(32), Call::debug_to_file, args, Call::Intrinsic);
77 string call_result_name = unique_name("debug_to_file_result");
78 Expr call_result_var = Variable::make(Int(32), call_result_name);
79 Stmt body = AssertStmt::make(call_result_var == 0,
80 Call::make(Int(32), "halide_error_debug_to_file_failed",
81 {f.name(), f.debug_file(), call_result_var},
82 Call::Extern));
83 body = LetStmt::make(call_result_name, call, body);
84 body = Block::make(mutate(op->body), body);
85
86 return Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, body);
87 } else {
88 return IRMutator::visit(op);
89 }
90 }
91
92public:
93 DebugToFile(const map<string, Function> &e)
94 : env(e) {
95 }
96};

98class RemoveDummyRealizations : public IRMutator {
99 const vector<Function> &outputs;
100
101 using IRMutator::visit;
102
103 Stmt visit(const Realize *op) override {
104 for (const Function &f : outputs) {
105 if (op->name == f.name()) {
106 return mutate(op->body);
107 }
108 }
109 return IRMutator::visit(op);
110 }
111
112public:
113 RemoveDummyRealizations(const vector<Function> &o)
114 : outputs(o) {
115 }
116};

118class AddDummyRealizations : public IRMutator {
119 const vector<Function> &outputs;
120
121 using IRMutator::visit;
122
123 Stmt visit(const ProducerConsumer *op) override {
124 Stmt s = IRMutator::visit(op);
125 for (const Function &out : outputs) {
126 if (op->name == out.name()) {
127 std::vector<Range> output_bounds;
128 for (int i = 0; i < out.dimensions(); i++) {
129 string dim = std::to_string(i);
130 Expr min = Variable::make(Int(32), out.name() + ".min." + dim);
131 Expr extent = Variable::make(Int(32), out.name() + ".extent." + dim);
132 output_bounds.emplace_back(min, extent);
133 }
134 return Realize::make(out.name(),
135 out.output_types(),
136 MemoryType::Auto,
137 output_bounds,
138 const_true(),
139 s);
140 }
141 }
142 return s;
143 }
144
145public:
146 AddDummyRealizations(const vector<Function> &o)
147 : outputs(o) {
148 }
149};

153Stmt debug_to_file(Stmt s, const vector<Function> &outputs, const map<string, Function> &env) {
154 // Temporarily wrap the produce nodes for the output functions in
155 // realize nodes so that we know when to write the debug outputs.
156 s = AddDummyRealizations(outputs).mutate(s);
157
158 s = DebugToFile(env).mutate(s);
159
160 // Remove the realize node we wrapped around the output
161 s = RemoveDummyRealizations(outputs).mutate(s);
162
163 return s;
164}
