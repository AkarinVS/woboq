10Dimension::Dimension(const Internal::Parameter &p, int d, Func f)
11 : param(p), d(d), f(std::move(f)) {
12 user_assert(param.defined())
13 << "Can't access the dimensions of an undefined Parameter\n";
14 user_assert(param.is_buffer())
15 << "Can't access the dimensions of a scalar Parameter\n";
16 user_assert(d >= 0 && d < param.dimensions())
17 << "Can't access dimension " << d
18 << " of a " << param.dimensions() << "-dimensional Parameter\n";
19}

53Dimension Dimension::set_extent(Expr extent) {
54 // Propagate constant bounds into estimates as well.
55 if (is_const(extent)) {
56 param.set_extent_constraint_estimate(d, extent);
57 }
58 param.set_extent_constraint(d, std::move(extent));
59 return *this;
60}

80Dimension Dimension::set_estimate(Expr min, Expr extent) {
81 // Update the estimates on the linked Func as well.
82 // (This matters mainly for OutputImageParams.)
83 // Note that while it's possible/legal for a Dimension to have an undefined
84 // Func, you shouldn't ever call set_estimate on such an instance.
85 internal_assert(f.defined());
86 f.set_estimate(f.args()[d], min, extent);
87 param.set_min_constraint_estimate(d, std::move(min));
88 param.set_extent_constraint_estimate(d, std::move(extent));
89 return *this;
90}
