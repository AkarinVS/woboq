14Stmt Simplify::visit(const IfThenElse *op) {
15 Expr condition = mutate(op->condition, nullptr);
16 if (in_unreachable) {
17 return op;
18 }
19
20 // Remove tags
21 Expr unwrapped_condition = unwrap_tags(condition);
22
23 // If (true) ...
24 if (is_const_one(unwrapped_condition)) {
25 return mutate(op->then_case);
26 }
27
28 // If (false) ...
29 if (is_const_zero(unwrapped_condition)) {
30 if (op->else_case.defined()) {
31 return mutate(op->else_case);
32 } else {
33 return Evaluate::make(0);
34 }
35 }
36
37 Stmt then_case, else_case;
38 {
39 auto f = scoped_truth(unwrapped_condition);
40 then_case = mutate(op->then_case);
41 Stmt learned_then_case = f.substitute_facts(then_case);
42 if (!learned_then_case.same_as(then_case)) {
43 then_case = mutate(learned_then_case);
44 }
45 }
46 bool then_unreachable = in_unreachable;
47 in_unreachable = false;
48
49 {
50 auto f = scoped_falsehood(unwrapped_condition);
51 else_case = mutate(op->else_case);
52 Stmt learned_else_case = f.substitute_facts(else_case);
53 if (!learned_else_case.same_as(else_case)) {
54 else_case = mutate(learned_else_case);
55 }
56 }
57 bool else_unreachable = in_unreachable;
58
59 if (then_unreachable && else_unreachable) {
60 return then_case;
61 }
62 in_unreachable = false;
63 if (else_unreachable) {
64 return then_case;
65 } else if (then_unreachable) {
66 return else_case;
67 }
68
69 if (is_no_op(else_case)) {
70 // If both sides are no-ops, bail out.
71 if (is_pure(condition) && is_no_op(then_case)) {
72 return then_case;
73 }
74 // Replace no-ops with empty stmts.
75 else_case = Stmt();
76 }
77
78 // Pull out common nodes, but only when the "late in lowering" flag is set. This
79 // avoids simplifying specializations before they have a chance to specialize.
80 if (remove_dead_code && equal(then_case, else_case)) {
81 return then_case;
82 }
83 const Acquire *then_acquire = then_case.as<Acquire>();
84 const Acquire *else_acquire = else_case.as<Acquire>();
85 const ProducerConsumer *then_pc = then_case.as<ProducerConsumer>();
86 const ProducerConsumer *else_pc = else_case.as<ProducerConsumer>();
87 const Block *then_block = then_case.as<Block>();
88 const Block *else_block = else_case.as<Block>();
89 const For *then_for = then_case.as<For>();
90 const IfThenElse *then_if = then_case.as<IfThenElse>();
91 const IfThenElse *else_if = else_case.as<IfThenElse>();
92 if (then_acquire &&
93 else_acquire &&
94 equal(then_acquire->semaphore, else_acquire->semaphore) &&
95 equal(then_acquire->count, else_acquire->count)) {
96 // TODO: This simplification sometimes prevents useful loop partioning/no-op
97 // trimming from happening, e.g. it rewrites:
98 //
99 // for (x, min + -2, extent + 2) {
100 // if (x < min) {
101 // acquire (f24.semaphore_0, 1) {}
102 // } else {
103 // acquire (f24.semaphore_0, 1) { ... }
104 // }
105 // }
106 //
107 // This could be partitioned and simplified, but not after this simplification.
108 return Acquire::make(then_acquire->semaphore, then_acquire->count,
109 mutate(IfThenElse::make(condition, then_acquire->body, else_acquire->body)));
110 } else if (then_pc &&
111 else_pc &&
112 then_pc->name == else_pc->name &&
113 then_pc->is_producer == else_pc->is_producer) {
114 return ProducerConsumer::make(then_pc->name, then_pc->is_producer,
115 mutate(IfThenElse::make(condition, then_pc->body, else_pc->body)));
116 } else if (then_pc &&
117 is_no_op(else_case)) {
118 return ProducerConsumer::make(then_pc->name, then_pc->is_producer,
119 mutate(IfThenElse::make(condition, then_pc->body)));
120 } else if (then_block &&
121 else_block &&
122 equal(then_block->first, else_block->first)) {
123 return Block::make(then_block->first,
124 mutate(IfThenElse::make(condition, then_block->rest, else_block->rest)));
125 } else if (then_block &&
126 else_block &&
127 equal(then_block->rest, else_block->rest)) {
128 return Block::make(mutate(IfThenElse::make(condition, then_block->first, else_block->first)),
129 then_block->rest);
130 } else if (then_block && equal(then_block->first, else_case)) {
131 return Block::make(else_case,
132 mutate(IfThenElse::make(condition, then_block->rest)));
133 } else if (then_block && equal(then_block->rest, else_case)) {
134 return Block::make(mutate(IfThenElse::make(condition, then_block->first)),
135 else_case);
136 } else if (else_block && equal(then_case, else_block->first)) {
137 return Block::make(then_case,
138 mutate(IfThenElse::make(condition, Evaluate::make(0), else_block->rest)));
139 } else if (else_block && equal(then_case, else_block->rest)) {
140 return Block::make(mutate(IfThenElse::make(condition, Evaluate::make(0), else_block->first)),
141 then_case);
142 } else if (then_for &&
143 !else_case.defined() &&
144 equal(unwrapped_condition, 0 < then_for->extent)) {
145 // This guard is redundant
146 return then_case;
147 } else if (then_if &&
148 else_if &&
149 !then_if->else_case.defined() &&
150 !else_if->else_case.defined() &&
151 is_pure(condition) &&
152 is_pure(then_if->condition) &&
153 is_pure(else_if->condition) &&
154 equal(then_if->condition, else_if->condition)) {
155 // Rewrite if(a) { if(b) X } else { if(b) Y }
156 // to if(b) { if(a) X else Y }
157 return mutate(IfThenElse::make(then_if->condition,
158 IfThenElse::make(condition, then_if->then_case, else_if->then_case)));
159 } else if (condition.same_as(op->condition) &&
160 then_case.same_as(op->then_case) &&
161 else_case.same_as(op->else_case)) {
162 return op;
163 } else {
164 return IfThenElse::make(condition, then_case, else_case);
165 }
166}

168Stmt Simplify::visit(const AssertStmt *op) {
169 Expr cond = mutate(op->condition, nullptr);
170
171 // The message is only evaluated when the condition is false
172 Expr message;
173 {
174 auto f = scoped_falsehood(cond);
175 message = mutate(op->message, nullptr);
176 }
177
178 if (is_const_zero(cond)) {
179 // Usually, assert(const-false) should generate a warning;
180 // in at least one case (specialize_fail()), we want to suppress
181 // the warning, because the assertion is generated internally
182 // by Halide and is expected to always fail.
183 const Call *call = message.as<Call>();
184 const bool const_false_conditions_expected =
185 call && call->name == "halide_error_specialize_fail";
186 if (!const_false_conditions_expected) {
187 user_warning << "This pipeline is guaranteed to fail an assertion at runtime: \n"
188 << message << "\n";
189 }
190 } else if (is_const_one(cond)) {
191 return Evaluate::make(0);
192 }
193
194 if (cond.same_as(op->condition) && message.same_as(op->message)) {
195 return op;
196 } else {
197 return AssertStmt::make(cond, message);
198 }

201Stmt Simplify::visit(const For *op) {
202 ExprInfo min_bounds, extent_bounds;
203 Expr new_min = mutate(op->min, &min_bounds);
204 if (in_unreachable) {
205 return Evaluate::make(new_min);
206 }
207 Expr new_extent = mutate(op->extent, &extent_bounds);
208 if (in_unreachable) {
209 return Evaluate::make(new_extent);
210 }
211
212 ScopedValue<bool> old_in_vector_loop(in_vector_loop,
213 (in_vector_loop ||
214 op->for_type == ForType::Vectorized));
215
216 bool bounds_tracked = false;
217 if (min_bounds.min_defined || (min_bounds.max_defined && extent_bounds.max_defined)) {
218 min_bounds.max += extent_bounds.max - 1;
219 min_bounds.max_defined &= extent_bounds.max_defined;
220 min_bounds.alignment = ModulusRemainder{};
221 bounds_tracked = true;
222 bounds_and_alignment_info.push(op->name, min_bounds);
223 }
224
225 Stmt new_body;
226 {
227 // If we're in the loop, the extent must be greater than 0.
228 ScopedFact fact = scoped_truth(0 < new_extent);
229 new_body = mutate(op->body);
230 }
231 if (in_unreachable) {
232 if (extent_bounds.min_defined && extent_bounds.min >= 1) {
233 // If we know the loop executes once, the code that runs this loop is unreachable.
234 return new_body;
235 }
236 in_unreachable = false;
237 return Evaluate::make(0);
238 }
239
240 if (bounds_tracked) {
241 bounds_and_alignment_info.pop(op->name);
242 }
243
244 if (const Acquire *acquire = new_body.as<Acquire>()) {
245 if (is_no_op(acquire->body)) {
246 // Rewrite iterated no-op acquires as a single acquire.
247 return Acquire::make(acquire->semaphore, mutate(acquire->count * new_extent, nullptr), acquire->body);
248 }
249 }
250
251 if (is_no_op(new_body)) {
252 return new_body;
253 } else if (extent_bounds.max_defined &&
254 extent_bounds.max <= 0) {
255 return Evaluate::make(0);
256 } else if (extent_bounds.max_defined &&
257 extent_bounds.max <= 1 &&
258 op->device_api == DeviceAPI::None) {
259 Stmt s = LetStmt::make(op->name, new_min, new_body);
260 if (extent_bounds.min < 1) {
261 s = IfThenElse::make(0 < new_extent, s);
262 }
263 return mutate(s);
264 } else if (!stmt_uses_var(new_body, op->name) && !is_const_zero(op->min)) {
265 return For::make(op->name, make_zero(Int(32)), new_extent, op->for_type, op->device_api, new_body);
266 } else if (op->min.same_as(new_min) &&
267 op->extent.same_as(new_extent) &&
268 op->body.same_as(new_body)) {
269 return op;
270 } else {
271 return For::make(op->name, new_min, new_extent, op->for_type, op->device_api, new_body);
272 }
273}

275Stmt Simplify::visit(const Provide *op) {
276 found_buffer_reference(op->name, op->args.size());
277
278 vector<Expr> new_args(op->args.size());
279 vector<Expr> new_values(op->values.size());
280 bool changed = false;
281
282 // Mutate the args
283 for (size_t i = 0; i < op->args.size(); i++) {
284 const Expr &old_arg = op->args[i];
285 Expr new_arg = mutate(old_arg, nullptr);
286 if (!new_arg.same_as(old_arg)) {
287 changed = true;
288 }
289 new_args[i] = new_arg;
290 }
291
292 for (size_t i = 0; i < op->values.size(); i++) {
293 const Expr &old_value = op->values[i];
294 Expr new_value = mutate(old_value, nullptr);
295 if (!new_value.same_as(old_value)) {
296 changed = true;
297 }
298 new_values[i] = new_value;
299 }
300
301 if (!changed) {
302 return op;
303 } else {
304 return Provide::make(op->name, new_values, new_args);
305 }

308Stmt Simplify::visit(const Store *op) {
309 found_buffer_reference(op->name);
310
311 Expr predicate = mutate(op->predicate, nullptr);
312 Expr value = mutate(op->value, nullptr);
313
314 ExprInfo index_info;
315 Expr index = mutate(op->index, &index_info);
316
317 // If the store is fully out of bounds, drop it.
318 // This should only occur inside branches that make the store unreachable,
319 // but perhaps the branch was hard to prove constant true or false. This
320 // provides an alternative mechanism to simplify these unreachable stores.
321 string alloc_extent_name = op->name + ".total_extent_bytes";
322 if (bounds_and_alignment_info.contains(alloc_extent_name)) {
323 if (index_info.max_defined && index_info.max < 0) {
324 in_unreachable = true;
325 return Evaluate::make(unreachable());
326 }
327 const ExprInfo &alloc_info = bounds_and_alignment_info.get(alloc_extent_name);
328 if (alloc_info.max_defined && index_info.min_defined) {
329 int index_min_bytes = index_info.min * op->value.type().bytes();
330 if (index_min_bytes > alloc_info.max) {
331 in_unreachable = true;
332 return Evaluate::make(unreachable());
333 }
334 }
335 }
336
337 ExprInfo base_info;
338 if (const Ramp *r = index.as<Ramp>()) {
339 mutate(r->base, &base_info);
340 }
341 base_info.alignment = ModulusRemainder::intersect(base_info.alignment, index_info.alignment);
342
343 const Load *load = value.as<Load>();
344 const Broadcast *scalar_pred = predicate.as<Broadcast>();
345
346 ModulusRemainder align = ModulusRemainder::intersect(op->alignment, base_info.alignment);
347
348 if (is_const_zero(predicate)) {
349 // Predicate is always false
350 return Evaluate::make(0);
351 } else if (scalar_pred && !is_const_one(scalar_pred->value)) {
352 return IfThenElse::make(scalar_pred->value,
353 Store::make(op->name, value, index, op->param, const_true(value.type().lanes()), align));
354 } else if (is_undef(value) || (load && load->name == op->name && equal(load->index, index))) {
355 // foo[x] = foo[x] or foo[x] = undef is a no-op
356 return Evaluate::make(0);
357 } else if (predicate.same_as(op->predicate) && value.same_as(op->value) && index.same_as(op->index) && align == op->alignment) {
358 return op;
359 } else {
360 return Store::make(op->name, value, index, op->param, predicate, align);
361 }
362}

364Stmt Simplify::visit(const Allocate *op) {
365 std::vector<Expr> new_extents;
366 bool all_extents_unmodified = true;
367 ExprInfo total_extent_info;
368 total_extent_info.min_defined = true;
369 total_extent_info.max_defined = true;
370 total_extent_info.min = 1;
371 total_extent_info.max = 1;
372 for (size_t i = 0; i < op->extents.size(); i++) {
373 ExprInfo extent_info;
374 new_extents.push_back(mutate(op->extents[i], &extent_info));
375 all_extents_unmodified &= new_extents[i].same_as(op->extents[i]);
376 if (extent_info.min_defined) {
377 total_extent_info.min *= extent_info.min;
378 } else {
379 total_extent_info.min_defined = false;
380 }
381 if (extent_info.max_defined) {
382 total_extent_info.max *= extent_info.max;
383 } else {
384 total_extent_info.max_defined = false;
385 }
386 }
387 if (total_extent_info.min_defined) {
388 total_extent_info.min *= op->type.bytes();
389 total_extent_info.min -= 1;
390 }
391 if (total_extent_info.max_defined) {
392 total_extent_info.max *= op->type.bytes();
393 total_extent_info.max -= 1;
394 }
395
396 ScopedBinding<ExprInfo> b(bounds_and_alignment_info, op->name + ".total_extent_bytes", total_extent_info);
397
398 Stmt body = mutate(op->body);
399 Expr condition = mutate(op->condition, nullptr);
400 Expr new_expr;
401 if (op->new_expr.defined()) {
402 new_expr = mutate(op->new_expr, nullptr);
403 }
404 const IfThenElse *body_if = body.as<IfThenElse>();
405 if (body_if &&
406 op->condition.defined() &&
407 equal(op->condition, body_if->condition)) {
408 // We can move the allocation into the if body case. The
409 // else case must not use it.
410 Stmt stmt = Allocate::make(op->name, op->type, op->memory_type,
411 new_extents, condition, body_if->then_case,
412 new_expr, op->free_function);
413 return IfThenElse::make(body_if->condition, stmt, body_if->else_case);
414 } else if (all_extents_unmodified &&
415 body.same_as(op->body) &&
416 condition.same_as(op->condition) &&
417 new_expr.same_as(op->new_expr)) {
418 return op;
419 } else {
420 return Allocate::make(op->name, op->type, op->memory_type,
421 new_extents, condition, body,
422 new_expr, op->free_function);
423 }
424}

426Stmt Simplify::visit(const Evaluate *op) {
427 Expr value = mutate(op->value, nullptr);
428
429 // Rewrite Lets inside an evaluate as LetStmts outside the Evaluate.
430 vector<pair<string, Expr>> lets;
431 while (const Let *let = value.as<Let>()) {
432 lets.emplace_back(let->name, let->value);
433 value = let->body;
434 }
435
436 if (value.same_as(op->value)) {
437 internal_assert(lets.empty());
438 return op;
439 } else {
440 // Rewrap the lets outside the evaluate node
441 Stmt stmt = Evaluate::make(value);
442 for (size_t i = lets.size(); i > 0; i--) {
443 stmt = LetStmt::make(lets[i - 1].first, lets[i - 1].second, stmt);
444 }
445 return stmt;
446 }
447}

461Stmt Simplify::visit(const Block *op) {
462 Stmt first = mutate(op->first);
463 Stmt rest = op->rest;
464
465 if (const AssertStmt *first_assert = first.as<AssertStmt>()) {
466 bool unchanged = first.same_as(op->first);
467
468 // Handle an entire sequence of asserts here to avoid a deeply
469 // nested stack. We won't be popping any knowledge until
470 // after the end of this chain of asserts, so we can use a
471 // single ScopedFact and progressively add knowledge to it.
472 ScopedFact knowledge(this);
473 vector<Stmt> result;
474 result.push_back(first);
475 knowledge.learn_true(first_assert->condition);
476
477 // Loop invariants: 'first' has already been mutated and is in
478 // the result list. 'first' was an AssertStmt before it was
479 // mutated, and its condition has been captured in
480 // 'knowledge'. 'rest' has not been mutated and is not in the
481 // result list.
482 const Block *rest_block;
483 while ((rest_block = rest.as<Block>()) &&
484 (first_assert = rest_block->first.as<AssertStmt>())) {
485 first = mutate(first_assert);
486 unchanged &= first.same_as(first_assert);
487 rest = rest_block->rest;
488 result.push_back(first);
489 if ((first_assert = first.as<AssertStmt>())) {
490 // If it didn't fold away to trivially true or false,
491 // learn the condition.
492 knowledge.learn_true(first_assert->condition);
493 }
494 }
495
496 Stmt new_rest = mutate(rest);
497 Stmt learned_new_rest = knowledge.substitute_facts(new_rest);
498 if (!learned_new_rest.same_as(new_rest)) {
499 new_rest = mutate(learned_new_rest);
500 }
501 unchanged &= new_rest.same_as(rest);
502
503 if (unchanged) {
504 return op;
505 } else {
506 result.push_back(new_rest);
507 return Block::make(result);
508 }
509
510 } else {
511 rest = mutate(op->rest);
512 }
513
514 // Check if both halves start with a let statement.
515 const LetStmt *let_first = first.as<LetStmt>();
516 const LetStmt *let_rest = rest.as<LetStmt>();
517 const Block *block_rest = rest.as<Block>();
518 const IfThenElse *if_first = first.as<IfThenElse>();
519 const IfThenElse *if_next = block_rest ? block_rest->first.as<IfThenElse>() : rest.as<IfThenElse>();
520 Stmt if_rest = block_rest ? block_rest->rest : Stmt();
521
522 const Store *store_first = first.as<Store>();
523 const Store *store_next = block_rest ? block_rest->first.as<Store>() : rest.as<Store>();
524
525 if (is_no_op(first) &&
526 is_no_op(rest)) {
527 return Evaluate::make(0);
528 } else if (is_no_op(first)) {
529 return rest;
530 } else if (is_no_op(rest)) {
531 return first;
532 } else if (let_first &&
533 let_rest &&
534 equal(let_first->value, let_rest->value) &&
535 is_pure(let_first->value)) {
536
537 // Do both first and rest start with the same let statement (occurs when unrolling).
538 Stmt new_block = mutate(Block::make(let_first->body, let_rest->body));
539
540 // We need to make a new name since we're pulling it out to a
541 // different scope.
542 string var_name = unique_name('t');
543 Expr new_var = Variable::make(let_first->value.type(), var_name);
544 new_block = substitute(let_first->name, new_var, new_block);
545 new_block = substitute(let_rest->name, new_var, new_block);
546
547 return LetStmt::make(var_name, let_first->value, new_block);
548 } else if (store_first &&
549 store_next &&
550 store_first->name == store_next->name &&
551 equal(store_first->index, store_next->index) &&
552 equal(store_first->predicate, store_next->predicate) &&
553 is_pure(store_first->index) &&
554 is_pure(store_first->value) &&
555 is_pure(store_first->predicate) &&
556 !expr_uses_var(store_next->index, store_next->name) &&
557 !expr_uses_var(store_next->value, store_next->name) &&
558 !expr_uses_var(store_next->predicate, store_next->name)) {
559 // Second store clobbers first
560 if (block_rest) {
561 return Block::make(store_next, block_rest->rest);
562 } else {
563 return store_next;
564 }
565 } else if (if_first &&
566 if_next &&
567 equal(if_first->condition, if_next->condition) &&
568 is_pure(if_first->condition)) {
569 // Two ifs with matching conditions.
570 Stmt then_case = mutate(Block::make(if_first->then_case, if_next->then_case));
571 Stmt else_case;
572 if (if_first->else_case.defined() && if_next->else_case.defined()) {
573 else_case = mutate(Block::make(if_first->else_case, if_next->else_case));
574 } else if (if_first->else_case.defined()) {
575 // We already simplified the body of the ifs.
576 else_case = if_first->else_case;
577 } else {
578 else_case = if_next->else_case;
579 }
580 Stmt result = IfThenElse::make(if_first->condition, then_case, else_case);
581 if (if_rest.defined()) {
582 result = Block::make(result, if_rest);
583 }
584 return result;
585 } else if (if_first &&
586 if_next &&
587 !if_next->else_case.defined() &&
588 is_pure(if_first->condition) &&
589 is_pure(if_next->condition) &&
590 is_const_one(mutate((if_first->condition && if_next->condition) == if_next->condition, nullptr))) {
591 // Two ifs where the second condition is tighter than
592 // the first condition. The second if can be nested
593 // inside the first one, because if it's true the
594 // first one must also be true.
595 Stmt then_case = mutate(Block::make(if_first->then_case, if_next));
596 Stmt else_case = if_first->else_case;
597 Stmt result = IfThenElse::make(if_first->condition, then_case, else_case);
598 if (if_rest.defined()) {
599 result = Block::make(result, if_rest);
600 }
601 return result;
602 } else if (if_first &&
603 if_next &&
604 is_pure(if_first->condition) &&
605 is_pure(if_next->condition) &&
606 is_const_one(mutate(!(if_first->condition && if_next->condition), nullptr))) {
607 // Two ifs where the first condition being true implies the
608 // second is false. The second if can be nested inside the
609 // else case of the first one, turning a block of if
610 // statements into an if-else chain.
611 Stmt then_case = if_first->then_case;
612 Stmt else_case = if_next;
613 if (if_first->else_case.defined()) {
614 else_case = Block::make(if_first->else_case, else_case);
615 }
616 Stmt result = IfThenElse::make(if_first->condition, then_case, else_case);
617 if (if_rest.defined()) {
618 result = Block::make(result, if_rest);
619 }
620 return result;
621 } else if (op->first.same_as(first) &&
622 op->rest.same_as(rest)) {
623 return op;
624 } else {
625 return Block::make(first, rest);
626 }
627}

629Stmt Simplify::visit(const Realize *op) {
630 Region new_bounds;
631 bool bounds_changed;
632
633 // Mutate the bounds
634 std::tie(new_bounds, bounds_changed) = mutate_region(this, op->bounds, nullptr);
635
636 Stmt body = mutate(op->body);
637 Expr condition = mutate(op->condition, nullptr);
638 if (!bounds_changed &&
639 body.same_as(op->body) &&
640 condition.same_as(op->condition)) {
641 return op;
642 }
643 return Realize::make(op->name, op->types, op->memory_type, new_bounds,
644 std::move(condition), std::move(body));
645}

649 Expr condition = mutate(op->condition, nullptr);

651 if (is_const_zero(op->condition)) {

653 return body;

656 Region new_bounds;

663 body.same_as(op->body) &&
664 condition.same_as(op->condition)) {

667 return Prefetch::make(op->name, op->types, new_bounds, op->prefetch, std::move(condition), std::move(body));
