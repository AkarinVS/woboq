12// Conversion routines to and from float cribbed from Christian Rau's
13// half library (half.sourceforge.net)
14uint16_t float_to_float16(float value) {
15 // Start by copying over the sign bit
16 uint16_t bits = std::signbit(value) << 15;
17
18 // Check for special values
19 if (value == 0) {
20 return bits;
21 } else if (std::isnan(value)) {
22 return bits | float16_t::exponent_mask | float16_t::mantissa_mask;
23 } else if (std::isinf(value)) {
24 return bits | float16_t::exponent_mask;
25 }
26
27 int exp;
28 // Get exponent, with bias already subtracted.
29 std::frexp(value, &exp);
30 if (exp > 16) {
31 // Too large, return infinity. Per initialization, bits only
32 // contains the sign bit, so this is +/-inf.
33 return bits | float16_t::exponent_mask;
34 } else if (exp < -13) {
35 // Too small, clamp to 2^-24
36 value = std::ldexp(value, 24);
37 } else {
38 // Move the exponent from the float into the half.
39 value = std::ldexp(value, 11 - exp);
40 bits |= ((exp + 13) << float16_t::mantissa_bits);
41 }
42
43 // We've normalized value as much as possible. Put the integer
44 // portion of it into the mantissa.
45 float ival;
46 float frac = std::modf(value, &ival);
47 bits += (uint16_t)(std::abs((int)ival));
48
49 // Now consider the fractional part. We round to nearest with ties
50 // going to even.
51 frac = std::abs(frac);
52 bits += (frac > 0.5f) | ((frac == 0.5f) & bits);
53
54 return bits;
55}

339 uint32_t ret;
