13Buffer<uint8_t> integer_divide_table_u8() {
14 static std::mutex initialize_lock;
15 std::lock_guard<std::mutex> lock_guard(initialize_lock);
16 {
17 static Buffer<uint8_t> im(256, 2);
18 static bool initialized = false;
19 if (!initialized) {
20 initialized = true;
21 for (size_t i = 0; i < 256; i++) {
22 im(i, 0) = table_runtime_u8[i][2];
23 im(i, 1) = table_runtime_u8[i][3];
24 }
25 }
26 return im;
27 }
28}

116Expr fast_integer_divide(Expr numerator, Expr denominator) {
117 if (is_const(denominator)) {
118 // There's code elsewhere for this case.
119 return numerator / cast<uint8_t>(denominator);
120 }
121 user_assert(denominator.type() == UInt(8))
122 << "Fast integer divide requires a UInt(8) denominator\n";
123 Type t = numerator.type();
124 user_assert(t.is_uint() || t.is_int())
125 << "Fast integer divide requires an integer numerator\n";
126 user_assert(t.bits() == 8 || t.bits() == 16 || t.bits() == 32)
127 << "Fast integer divide requires a numerator with 8, 16, or 32 bits\n";
128
129 Type wide = t.widen();
130
131 Expr result;
132 if (t.is_uint()) {
133 Expr mul, shift;
134 switch (t.bits()) {
135 case 8: {
136 Buffer<uint8_t> table = integer_divide_table_u8();
137 mul = table(denominator, 0);
138 shift = table(denominator, 1);
139 break;
140 }
141 case 16: {
142 Buffer<uint16_t> table = integer_divide_table_u16();
143 mul = table(denominator, 0);
144 shift = table(denominator, 1);
145 break;
146 }
147 default: // 32
148 {
149 Buffer<uint32_t> table = integer_divide_table_u32();
150 mul = table(denominator, 0);
151 shift = table(denominator, 1);
152 break;
153 }
154 }
155
156 // Multiply-keep-high-half
157 result = (cast(wide, mul) * numerator);
158
159 if (t.bits() < 32) {
160 result = result / (1 << t.bits());
161 } else {
162 result = result >> Internal::make_const(result.type(), t.bits());
163 }
164
165 result = cast(t, result);
166
167 // Add half the difference between input and output so far
168 result = result + (numerator - result) / 2;
169
170 // Do a final shift
171 result = result >> cast(result.type(), shift);
172
173 } else {
174
175 Expr mul, shift;
176 switch (t.bits()) {
177 case 8: {
178 Buffer<uint8_t> table = integer_divide_table_s8();
179 mul = table(denominator, 0);
180 shift = table(denominator, 1);
181 break;
182 }
183 case 16: {
184 Buffer<uint16_t> table = integer_divide_table_s16();
185 mul = table(denominator, 0);
186 shift = table(denominator, 1);
187 break;
188 }
189 default: // 32
190 {
191 Buffer<uint32_t> table = integer_divide_table_s32();
192 mul = table(denominator, 0);
193 shift = table(denominator, 1);
194 break;
195 }
196 }
197
198 // Extract sign bit
199 //Expr xsign = (t.bits() < 32) ? (numerator / (1 << (t.bits()-1))) : (numerator >> (t.bits()-1));
200 Expr xsign = select(numerator > 0, cast(t, 0), cast(t, -1));
201
202 // If it's negative, flip the bits of the
203 // numerator. Equivalent to:
204 // select(numerator < 0, -(numerator+1), numerator);
205 numerator = xsign ^ numerator;
206
207 // Multiply-keep-high-half
208 result = (cast(wide, mul) * numerator);
209 if (t.bits() < 32) {
210 result = result / (1 << t.bits());
211 } else {
212 result = result >> Internal::make_const(result.type(), t.bits());
213 }
214 result = cast(t, result);
215
216 // Do the final shift
217 result = result >> cast(result.type(), shift);
218
219 // Maybe flip the bits again
220 result = xsign ^ result;
221 }
222
223 // The tables don't work for denominator == 1
224 result = select(std::move(denominator) == 1, std::move(numerator), result);
225
226 internal_assert(result.type() == t);
227
228 return result;
229}

231Expr fast_integer_modulo(Expr numerator, Expr denominator) {
232 Expr ratio = fast_integer_divide(numerator, denominator);
233 return std::move(numerator) - ratio * std::move(denominator);
234}
