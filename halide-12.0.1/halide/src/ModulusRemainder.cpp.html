13// A version of mod where a % 0 == a
14int64_t mod(int64_t a, int64_t b) {
15 if (b == 0) {
16 return a;
17 } else {
18 return mod_imp(a, b);

292bool reduce_expr_modulo(const Expr &expr, int64_t modulus, int64_t *remainder) {
293 ModulusRemainder result = modulus_remainder(expr);
294
295 /* As an example: If we asked for expr mod 8, and the analysis
296 * said that expr = 16*k + 13, then because 16 % 8 == 0, the
297 * result is 13 % 8 == 5. But if the analysis says that expr =
298 * 6*k + 3, then expr mod 8 could be 1, 3, 5, or 7, so we just
299 * return false.
300 */
301
302 if (mod(result.modulus, modulus) == 0) {
303 *remainder = mod(result.remainder, modulus);
304 return true;
305 } else {
306 return false;
307 }
308}

326void check(const Expr &e, int64_t m, int64_t r) {
327 ModulusRemainder result = modulus_remainder(e);
328 if (result.modulus != m || result.remainder != r) {
329 std::cerr << "Test failed for modulus_remainder:\n";
330 std::cerr << "Expression: " << e << "\n";
331 std::cerr << "Correct modulus, remainder = " << m << ", " << r << "\n";
332 std::cerr << "Computed modulus, remainder = "
333 << result.modulus << ", "
334 << result.remainder << "\n";
335 exit(-1);
336 }

357int64_t gcd(int64_t a, int64_t b) {
358 if (a < b) {
359 std::swap(a, b);
360 }
361 while (b != 0) {
362 int64_t tmp = b;
363 b = a % b;
364 a = tmp;
365 }
366 return a;
367}

369int64_t lcm(int64_t a, int64_t b) {
370 // Remove all of the common factors from one of the operands
371 b /= gcd(a, b);
372
373 // Then multiply
374 if (mul_would_overflow(64, a, b)) {
375 return 0;
376 } else {
377 return a * b;
378 }
379}

381ModulusRemainder operator+(const ModulusRemainder &a, const ModulusRemainder &b) {
382 if (add_would_overflow(64, a.remainder, b.remainder)) {
383 return {1, 0};
384 } else {
385 int64_t modulus = gcd(a.modulus, b.modulus);
386 int64_t remainder = mod(a.remainder + b.remainder, modulus);
387 return {modulus, remainder};
388 }
389}

391ModulusRemainder operator-(const ModulusRemainder &a, const ModulusRemainder &b) {
392 if (sub_would_overflow(64, a.remainder, b.remainder)) {
393 return {1, 0};
394 } else {
395 int64_t modulus = gcd(a.modulus, b.modulus);
396 int64_t remainder = mod(a.remainder - b.remainder, modulus);
397 return {modulus, remainder};
398 }
399}

401ModulusRemainder operator*(const ModulusRemainder &a, const ModulusRemainder &b) {
402 if (a.modulus == 0) {
403 // a is constant
404 if (!mul_would_overflow(64, a.remainder, b.modulus) && !mul_would_overflow(64, a.remainder, b.remainder)) {
405 return {a.remainder * b.modulus, a.remainder * b.remainder};
406 }
407 } else if (b.modulus == 0) {
408 // b is constant
409 if (!mul_would_overflow(64, b.remainder, a.modulus) && !mul_would_overflow(64, a.remainder, b.remainder)) {
410 return {b.remainder * a.modulus, a.remainder * b.remainder};
411 }
412 } else if (a.remainder == 0 && b.remainder == 0) {
413 // multiple times multiple
414 if (!mul_would_overflow(64, a.modulus, b.modulus)) {
415 return {a.modulus * b.modulus, 0};
416 }
417 } else if (a.remainder == 0) {
418 int64_t g = gcd(b.modulus, b.remainder);
419 if (!mul_would_overflow(64, a.modulus, g)) {
420 return {a.modulus * g, 0};
421 }
422 } else if (b.remainder == 0) {
423 int64_t g = gcd(a.modulus, a.remainder);
424 if (!mul_would_overflow(64, b.modulus, g)) {
425 return {b.modulus * g, 0};
426 }
427 } else {
428 // Convert them to the same modulus and multiply
429 if (!mul_would_overflow(64, a.remainder, b.remainder)) {
430 int64_t modulus = gcd(a.modulus, b.modulus);
431 int64_t remainder = mod(a.remainder * b.remainder, modulus);
432 return {modulus, remainder};
433 }
434 }
435
436 return ModulusRemainder{};
437}

439ModulusRemainder operator/(const ModulusRemainder &a, const ModulusRemainder &b) {
440 // What can we say about:
441 // floor((m1 * x + r1) / (m2 * y + r2))
442
443 // If m2 is zero and m1 is a multiple of r2, then we can pull the
444 // varying term out of the floor div and the expression simplifies
445 // to:
446 // (m1 / r2) * x + floor(r1 / r2)
447 // E.g. (8x + 3) / 2 -> (4x + 1)
448
449 if (b.modulus == 0 && b.remainder != 0) {
450 if (mod(a.modulus, b.remainder) == 0) {
451 return {a.modulus / b.remainder, div_imp(a.remainder, b.remainder)};
452 }
453 }
454
455 return ModulusRemainder{};
456}

458ModulusRemainder ModulusRemainder::unify(const ModulusRemainder &a, const ModulusRemainder &b) {
459 // We don't know if we're going to get a or b, so we'd better find
460 // a single modulus remainder that works for both.
461
462 // For example:
463 // max(30*_ + 13, 40*_ + 27) ->
464 // max(10*_ + 3, 10*_ + 7) ->
465 // max(2*_ + 1, 2*_ + 1) ->
466 // 2*_ + 1
467
468 if (b.remainder > a.remainder) {
469 return unify(b, a);
470 }
471
472 // Reduce them to the same modulus and the same remainder
473 int64_t modulus = gcd(a.modulus, b.modulus);
474
475 if (sub_would_overflow(64, a.remainder, b.remainder)) {
476 // The modulus is not representable as an int64.
477 return {0, 1};
478 }
479
480 int64_t diff = a.remainder - b.remainder;
481
482 modulus = gcd(diff, modulus);
483
484 int64_t ra = mod(a.remainder, modulus);
485
486 internal_assert(ra == mod(b.remainder, modulus))
487 << "There's a bug inside ModulusRemainder in unify_alternatives:\n"
488 << "a.modulus = " << a.modulus << "\n"
489 << "a.remainder = " << a.remainder << "\n"
490 << "b.modulus = " << b.modulus << "\n"
491 << "b.remainder = " << b.remainder << "\n"
492 << "diff = " << diff << "\n"
493 << "unified modulus = " << modulus << "\n"
494 << "unified remainder = " << ra << "\n";
495
496 return {modulus, ra};
497}

499ModulusRemainder ModulusRemainder::intersect(const ModulusRemainder &a, const ModulusRemainder &b) {
500 // We have x == ma * y + ra == mb * z + rb
501
502 // We want to synthesize these two facts into one modulus
503 // remainder relationship. We are permitted to be
504 // conservatively-large, so it's OK if some elements of the result
505 // only satisfy one of the two constraints.
506
507 // For coprime ma and mb you want to use the Chinese remainder
508 // theorem. In our case, the moduli will almost always be
509 // powers of two, so we should just return the smaller of the two
510 // sets (usually the one with the larger modulus).
511 if (a.modulus == 0) {
512 return a;
513 }
514 if (b.modulus == 0) {
515 return b;
516 }
517 if (a.modulus > b.modulus) {
518 return a;
519 }
520 return b;
521}

527ModulusRemainder operator%(const ModulusRemainder &a, const ModulusRemainder &b) {
528 // For non-zero y, we can treat x mod y as x + z*y, where we know
529 // nothing about z.
530 // (ax + b) + z (cx + d) ->
531 // ax + b + zcx + dz ->
532 // gcd(a, c, d) * w + b
533
534 // E.g:
535 // (8x + 5) mod (6x + 2) ->
536 // (8x + 5) + z (6x + 2) ->
537 // (8x + 6zx + 2x) + 5 ->
538 // 2(4x + 3zx + x) + 5 ->
539 // 2w + 1
540 int64_t modulus = gcd(a.modulus, b.modulus);
541 modulus = gcd(modulus, b.remainder);
542 int64_t remainder = mod(a.remainder, modulus);
543
544 if (b.remainder == 0 && remainder != 0) {
545 // b could be zero, so the result could also just be zero.
546 if (modulus == 0) {
547 remainder = 0;
548 } else {
549 // This can no longer be expressed as ax + b
550 remainder = 0;
551 modulus = 1;
552 }
553 }
554
555 return {modulus, remainder};
556}
