22/** If an integer expression varies linearly with the variables in the
23 * scope, return the linear term. Otherwise return an undefined
24 * Expr. */
25Expr is_linear(const Expr &e, const Scope<Expr> &linear) {
26 if (e.type() != Int(32)) {
27 return Expr();
28 }
29 if (const Variable *v = e.as<Variable>()) {
30 if (linear.contains(v->name)) {
31 return linear.get(v->name);
32 } else {
33 return make_zero(v->type);
34 }
35 } else if (const IntImm *op = e.as<IntImm>()) {
36 return make_zero(op->type);
37 } else if (const Add *add = e.as<Add>()) {
38 Expr la = is_linear(add->a, linear);
39 Expr lb = is_linear(add->b, linear);
40 if (is_const_zero(lb)) {
41 return la;
42 } else if (is_const_zero(la)) {
43 return lb;
44 } else if (la.defined() && lb.defined()) {
45 return la + lb;
46 } else {
47 return Expr();
48 }
49 } else if (const Sub *sub = e.as<Sub>()) {
50 Expr la = is_linear(sub->a, linear);
51 Expr lb = is_linear(sub->b, linear);
52 if (is_const_zero(lb)) {
53 return la;
54 } else if (la.defined() && lb.defined()) {
55 return la - lb;
56 } else {
57 return Expr();
58 }
59 } else if (const Mul *mul = e.as<Mul>()) {
60 Expr la = is_linear(mul->a, linear);
61 Expr lb = is_linear(mul->b, linear);
62 if (is_const_zero(la) && is_const_zero(lb)) {
63 return la;
64 } else if (is_const_zero(la) && lb.defined()) {
65 return mul->a * lb;
66 } else if (la.defined() && is_const_zero(lb)) {
67 return la * mul->b;
68 } else {
69 return Expr();
70 }
71 } else if (const Ramp *r = e.as<Ramp>()) {
72 Expr la = is_linear(r->base, linear);
73 Expr lb = is_linear(r->stride, linear);
74 if (is_const_zero(lb)) {
75 return la;
76 } else {
77 return Expr();
78 }
79 } else if (const Broadcast *b = e.as<Broadcast>()) {
80 return is_linear(b->value, linear);
81 } else {
82 return Expr();
83 }

133/** Given a scope of things that move linearly over time, come up with
134 * the next time step's version of some arbitrary Expr (which may be a
135 * nasty graph). Variables that move non-linearly through time are
136 * undefined Exprs in the scope. */
137class StepForwards : public IRGraphMutator {
138 const Scope<Expr> &linear;
139
140 using IRGraphMutator::visit;
141
142 Expr visit(const Variable *op) override {
143 if (linear.contains(op->name)) {
144 Expr step = linear.get(op->name);
145 if (!step.defined()) {
146 // It's non-linear
147 success = false;
148 return op;
149 } else if (is_const_zero(step)) {
150 // It's a known inner constant
151 return op;
152 } else {
153 // It's linear
154 return Expr(op) + step;
155 }
156 } else {
157 // It's some external constant
158 return op;
159 }
160 }
161
162public:
163 bool success = true;
164 StepForwards(const Scope<Expr> &s)
165 : linear(s) {
166 }
167};
168
169Expr step_forwards(Expr e, const Scope<Expr> &linear) {
170 StepForwards step(linear);
171 e = step.mutate(e);
172 if (!step.success) {
173 return Expr();
174 } else {
175 // We need to simplify it to reduce it to a canonical form,
176 // but it's a full graph, so we'll need to CSE it first.
177 e = common_subexpression_elimination(e);
178 e = simplify(e);
179 e = substitute_in_all_lets(e);
180 return e;
181 }
182}

184/** Carry loads over a single For loop body. */
185class LoopCarryOverLoop : public IRMutator {
186 // Track vars that step linearly with loop iterations
187 Scope<Expr> linear;
188 vector<pair<string, Expr>> containing_lets;
189
190 // Productions we're in a consume node for. They're fixed and safe
191 // to lift out.
192 const Scope<> &in_consume;
193
194 int max_carried_values;
195
196 using IRMutator::visit;
197
198 Stmt visit(const LetStmt *op) override {
199 // Track containing LetStmts and their linearity w.r.t. the
200 // loop variable.
201 Expr value = mutate(op->value);
202 Expr step = is_linear(value, linear);
203 ScopedBinding<Expr> bind(linear, op->name, step);
204
205 containing_lets.emplace_back(op->name, value);
206
207 Stmt stmt;
208 Stmt body = mutate(op->body);
209 if (value.same_as(op->value) &&
210 body.same_as(op->body)) {
211 stmt = op;
212 } else {
213 stmt = LetStmt::make(op->name, value, body);
214 }
215
216 containing_lets.pop_back();
217 return stmt;
218 }
219
220 Stmt visit(const Store *op) override {
221 return lift_carried_values_out_of_stmt(op);
222 }
223
224 Stmt visit(const Block *op) override {
225 vector<Stmt> v = block_to_vector(op);
226
227 vector<Stmt> stores;
228 vector<Stmt> result;
229 for (size_t i = 0; i < v.size(); i++) {
230 if (v[i].as<Store>()) {
231 stores.push_back(v[i]);
232 } else {
233 if (!stores.empty()) {
234 result.push_back(lift_carried_values_out_of_stmt(Block::make(stores)));
235 stores.clear();
236 }
237 result.push_back(mutate(v[i]));
238 }
239 }
240 if (!stores.empty()) {
241 result.push_back(lift_carried_values_out_of_stmt(Block::make(stores)));
242 }
243
244 return Block::make(result);
245 }
246
247 Stmt lift_carried_values_out_of_stmt(const Stmt &orig_stmt) {
248 debug(4) << "About to lift carried values out of stmt: " << orig_stmt << "\n";
249
250 // The stmts, as graphs (lets subtituted in). We must only use
251 // graph-aware methods to touch these, lest we incur
252 // exponential runtime.
253 Stmt graph_stmt = substitute_in_all_lets(orig_stmt);
254
255 // Find all the loads in these stmts.
256 FindLoads find_loads;
257 graph_stmt.accept(&find_loads);
258
259 debug(4) << "Found " << find_loads.result.size() << " loads\n";
260
261 // Group equal loads
262 vector<vector<const Load *>> loads;
263 for (const Load *load : find_loads.result) {
264 // Check if it's safe to lift out.
265 bool safe = (load->image.defined() ||
266 load->param.defined() ||
267 in_consume.contains(load->name));
268 if (!safe) {
269 continue;
270 }
271
272 bool represented = false;
273 for (vector<const Load *> &v : loads) {
274 if (graph_equal(Expr(load), Expr(v[0]))) {
275 v.push_back(load);
276 represented = true;
277 }
278 }
279 if (!represented) {
280 loads.push_back({load});
281 }
282 }
283
284 // For each load, move the load index forwards by one loop iteration
285 vector<Expr> indices, next_indices, predicates, next_predicates;
286 for (const vector<const Load *> &v : loads) {
287 indices.push_back(v[0]->index);
288 next_indices.push_back(step_forwards(v[0]->index, linear));
289 predicates.push_back(v[0]->predicate);
290 next_predicates.push_back(step_forwards(v[0]->predicate, linear));
291 }
292
293 // Find loads done on this loop iteration that will be
294 // reusable as some other Expr on the next loop iteration.
295 vector<vector<int>> chains;
296 for (int i = 0; i < (int)indices.size(); i++) {
297 for (int j = 0; j < (int)indices.size(); j++) {
298 // Don't catch loop invariants here.
299 if (i == j) {
300 continue;
301 }
302 if (loads[i][0]->name == loads[j][0]->name &&
303 next_indices[j].defined() &&
304 graph_equal(indices[i], next_indices[j]) &&
305 next_predicates[j].defined() &&
306 graph_equal(predicates[i], next_predicates[j])) {
307 chains.push_back({j, i});
308 debug(3) << "Found carried value:\n"
309 << i << ": -> " << Expr(loads[i][0]) << "\n"
310 << j << ": -> " << Expr(loads[j][0]) << "\n";
311 }
312 }
313 }
314
315 if (chains.empty()) {
316 return orig_stmt;
317 }
318
319 // Agglomerate chains of carries
320 bool done = false;
321 while (!done) {
322 done = true;
323 for (size_t i = 0; i < chains.size(); i++) {
324 if (chains[i].empty()) {
325 continue;
326 }
327 for (size_t j = 0; j < chains.size(); j++) {
328 if (chains[j].empty()) {
329 continue;
330 }
331 if (chains[i].back() == chains[j].front()) {
332 chains[i].insert(chains[i].end(), chains[j].begin() + 1, chains[j].end());
333 chains[j].clear();
334 done = false;
335 }
336 }
337 }
338
339 for (size_t i = 0; i < chains.size(); i++) {
340 while (i < chains.size() && chains[i].empty()) {
341 chains[i].swap(chains.back());
342 chains.pop_back();
343 }
344 }
345 }
346
347 // Sort the carry chains by decreasing order of size. The
348 // longest ones get the most reuse of each value.
349 //
350 // Use of stable_sort is just so that IR generated by different C++ compilers
351 // is identical; it doesn't appear to make any meaningful difference
352 // in code output, but makes debugging IR output easier to deal with.
353 std::stable_sort(chains.begin(), chains.end(),
354 [&](const vector<int> &c1, const vector<int> &c2) { return c1.size() > c2.size(); });
355
356 for (const vector<int> &c : chains) {
357 debug(3) << "Found chain of carried values:\n";
358 for (int i : c) {
359 debug(3) << i << ": <- " << indices[i] << "\n";
360 }
361 }
362
363 // Only keep the top N carried values. Otherwise we'll just
364 // spray stack spills everywhere. This is ugly, because we're
365 // relying on a heuristic.
366 vector<vector<int>> trimmed;
367 size_t sz = 0;
368 for (const vector<int> &c : chains) {
369 if (sz + c.size() > (size_t)max_carried_values) {
370 if (sz < (size_t)max_carried_values - 1) {
371 // Take a partial chain
372 trimmed.emplace_back(c.begin(), c.begin() + max_carried_values - sz);
373 }
374 break;
375 }
376 trimmed.push_back(c);
377 sz += c.size();
378 }
379 chains.swap(trimmed);
380
381 // We now have chains of the form:
382 // f[x] <- f[x+1] <- ... <- f[x+N-1]
383
384 // There are N values in that chain. N-1 of these can be
385 // reused from the previous loop iteration. We make a scratch
386 // buffer of size N. First we load the leading edge (the end
387 // of the chain) into in the last position. Then we run the
388 // original stmt with the loads replaced with loads from the
389 // scratch buffer, then we move the last N-1 values in the
390 // scratch buffer down one, so they're in the right place for
391 // the next loop iteration. If it's the first loop iteration,
392 // we need to populate the entire scratch buffer.
393
394 vector<Stmt> not_first_iteration_scratch_stores;
395 vector<Stmt> scratch_shuffles;
396 Stmt core = graph_stmt;
397
398 for (const vector<int> &c : chains) {
399 string scratch = unique_name('c');
400 vector<Expr> initial_scratch_values;
401
402 for (size_t i = 0; i < c.size(); i++) {
403 const Load *orig_load = loads[c[i]][0];
404 Expr scratch_idx = scratch_index(i, orig_load->type);
405 // Don't worry about alignment - the load is at a constant address.
406 Expr load_from_scratch = Load::make(orig_load->type, scratch, scratch_idx,
407 Buffer<>(), Parameter(), const_true(orig_load->type.lanes()), ModulusRemainder());
408 for (const Load *l : loads[c[i]]) {
409 core = graph_substitute(l, load_from_scratch, core);
410 }
411
412 if (i == c.size() - 1) {
413 Stmt store_to_scratch = Store::make(scratch, orig_load, scratch_idx,
414 Parameter(), const_true(orig_load->type.lanes()), ModulusRemainder());
415 not_first_iteration_scratch_stores.push_back(store_to_scratch);
416 } else {
417 initial_scratch_values.emplace_back(orig_load);
418 }
419 if (i > 0) {
420 Stmt shuffle = Store::make(scratch, load_from_scratch,
421 scratch_index(i - 1, orig_load->type),
422 Parameter(), const_true(orig_load->type.lanes()), ModulusRemainder());
423 scratch_shuffles.push_back(shuffle);
424 }
425 }
426
427 // Do joint CSE on the initial scratch values instead of
428 // cse'ing each independently. They'll shared common
429 // values and they originated from the same Expr.
430 vector<pair<string, Expr>> initial_lets;
431 // Group them into a single expression with a call node
432 Expr call = Call::make(Int(32), unique_name('b'), initial_scratch_values, Call::PureIntrinsic);
433 // Run CSE
434 call = simplify(common_subexpression_elimination(call));
435 // Peel off lets
436 while (const Let *l = call.as<Let>()) {
437 initial_lets.emplace_back(l->name, l->value);
438 call = l->body;
439 }
440 internal_assert(call.as<Call>());
441 initial_scratch_values = call.as<Call>()->args;
442
443 // Create the initial stores to scratch
444 vector<Stmt> initial_scratch_stores;
445 for (size_t i = 0; i < c.size() - 1; i++) {
446 Expr scratch_idx = scratch_index(i, initial_scratch_values[i].type());
447 Stmt store_to_scratch = Store::make(scratch, initial_scratch_values[i],
448 scratch_idx, Parameter(),
449 const_true(scratch_idx.type().lanes()),
450 ModulusRemainder());
451 initial_scratch_stores.push_back(store_to_scratch);
452 }
453
454 Stmt initial_stores = Block::make(initial_scratch_stores);
455
456 // Wrap them in the appropriate lets
457 for (size_t i = initial_lets.size(); i > 0; i--) {
458 auto l = initial_lets[i - 1];
459 initial_stores = LetStmt::make(l.first, l.second, initial_stores);
460 }
461 // We may be lifting the initial stores out of let stmts,
462 // so rewrap them in the necessary ones.
463 for (size_t i = containing_lets.size(); i > 0; i--) {
464 auto l = containing_lets[i - 1];
465 if (stmt_uses_var(initial_stores, l.first)) {
466 initial_stores = LetStmt::make(l.first, l.second, initial_stores);
467 }
468 }
469
470 allocs.push_back({scratch,
471 loads[c.front()][0]->type.element_of(),
472 (int)c.size() * loads[c.front()][0]->type.lanes(),
473 initial_stores});
474 }
475
476 Stmt s = Block::make(not_first_iteration_scratch_stores);
477 s = Block::make(s, core);
478 s = Block::make(s, Block::make(scratch_shuffles));
479 s = common_subexpression_elimination(s);
480 return s;
481 }

511class LoopCarry : public IRMutator {
512 using IRMutator::visit;
513
514 int max_carried_values;
515 Scope<> in_consume;
516
517 Stmt visit(const ProducerConsumer *op) override {
518 if (op->is_producer) {
519 return IRMutator::visit(op);
520 } else {
521 ScopedBinding<> bind(in_consume, op->name);
522 Stmt body = mutate(op->body);
523 return ProducerConsumer::make(op->name, op->is_producer, body);
524 }
525 }
526
527 Stmt visit(const For *op) override {
528 if (op->for_type == ForType::Serial && !is_const_one(op->extent)) {
529 Stmt stmt;
530 Stmt body = mutate(op->body);
531 LoopCarryOverLoop carry(op->name, in_consume, max_carried_values);
532 body = carry.mutate(body);
533 if (body.same_as(op->body)) {
534 stmt = op;
535 } else {
536 stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
537 }
538
539 // Inject the scratch buffer allocations.
540 for (const auto &alloc : carry.allocs) {
541 stmt = Block::make(substitute(op->name, op->min, alloc.initial_stores), stmt);
542 stmt = Allocate::make(alloc.name, alloc.type, MemoryType::Stack, {alloc.size}, const_true(), stmt);
543 }
544 if (!carry.allocs.empty()) {
545 stmt = IfThenElse::make(op->extent > 0, stmt);
546 }
547 return stmt;
548 } else {
549 return IRMutator::visit(op);
550 }
551 }
552
553public:
554 LoopCarry(int max_carried_values)
555 : max_carried_values(max_carried_values) {
556 }
557};
