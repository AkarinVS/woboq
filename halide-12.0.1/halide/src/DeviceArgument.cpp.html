14std::vector<DeviceArgument> HostClosure::arguments() {
15 std::vector<DeviceArgument> res;
16 for (const auto &v : vars) {
17 debug(2) << "var: " << v.first << "\n";
18 res.emplace_back(v.first, false, MemoryType::Auto, v.second, 0);
19 }
20 for (const auto &b : buffers) {
21 debug(2) << "buffer: " << b.first << " " << b.second.size;
22 if (b.second.read) {
23 debug(2) << " (read)";
24 }
25 if (b.second.write) {
26 debug(2) << " (write)";
27 }
28 if (b.second.memory_type == MemoryType::GPUTexture) {
29 debug(2) << " <texture>";
30 }
31 debug(2) << " dims=" << (int)b.second.dimensions;
32 debug(2) << "\n";
33
34 DeviceArgument arg(b.first, true, b.second.memory_type, b.second.type, b.second.dimensions, b.second.size);
35 arg.read = b.second.read;
36 arg.write = b.second.write;
37 res.push_back(arg);
38 }
39 return res;
40}

42void HostClosure::visit(const Call *op) {
43 if (op->is_intrinsic(Call::image_load) ||
44 op->is_intrinsic(Call::image_store)) {
45
46 // The argument to the call is either a StringImm or a broadcasted
47 // StringImm if this is part of a vectorized expression
48
49 const StringImm *string_imm = op->args[0].as<StringImm>();
50 if (!string_imm) {
51 internal_assert(op->args[0].as<Broadcast>());
52 string_imm = op->args[0].as<Broadcast>()->value.as<StringImm>();
53 }
54
55 internal_assert(string_imm);
56
57 std::string bufname = string_imm->value;
58 Buffer &ref = buffers[bufname];
59 ref.type = op->type;
60 ref.memory_type = op->is_intrinsic(Call::image_load) ||
61 op->is_intrinsic(Call::image_store) ?
62 MemoryType::GPUTexture :
63 MemoryType::Auto;
64
65 if (op->is_intrinsic(Call::image_load)) {
66 ref.read = true;
67 ref.dimensions = (op->args.size() - 2) / 2;
68 } else if (op->is_intrinsic(Call::image_store)) {
69 ref.write = true;
70 ref.dimensions = op->args.size() - 3;
71 }
72
73 // The Func's name and the associated .buffer are mentioned in the
74 // argument lists, but don't treat them as free variables.
75 ScopedBinding<> p1(ignore, bufname);
76 ScopedBinding<> p2(ignore, bufname + ".buffer");
77 Internal::Closure::visit(op);
78 } else {
79 Internal::Closure::visit(op);
80 }
81}

83void HostClosure::visit(const For *loop) {
84 if (CodeGen_GPU_Dev::is_gpu_var(loop->name)) {
85 // The size of the threads and blocks is not part of the closure
86 ScopedBinding<> p(ignore, loop->name);
87 loop->body.accept(this);
88 } else {
89 Internal::Closure::visit(loop);
