101void lower_impl(const vector<Function> &output_funcs,
102 const string &pipeline_name,
103 const Target &t,
104 const vector<Argument> &args,
105 const LinkageType linkage_type,
106 const vector<Stmt> &requirements,
107 bool trace_pipeline,
108 const vector<IRMutator *> &custom_passes,
109 Module &result_module) {
110 auto time_start = std::chrono::high_resolution_clock::now();
111
112 // Compute an environment
113 map<string, Function> env;
114 for (const Function &f : output_funcs) {
115 populate_environment(f, env);
116 }
117
118 // Create a deep-copy of the entire graph of Funcs.
119 vector<Function> outputs;
120 std::tie(outputs, env) = deep_copy(output_funcs, env);
121
122 bool any_strict_float = strictify_float(env, t);
123 result_module.set_any_strict_float(any_strict_float);
124
125 // Output functions should all be computed and stored at root.
126 for (const Function &f : outputs) {
127 Func(f).compute_root().store_root();
128 }
129
130 // Finalize all the LoopLevels
131 for (auto &iter : env) {
132 iter.second.lock_loop_levels();
133 }
134
135 // Substitute in wrapper Funcs
136 env = wrap_func_calls(env);
137
138 // Compute a realization order and determine group of functions which loops
139 // are to be fused together
140 vector<string> order;
141 vector<vector<string>> fused_groups;
142 std::tie(order, fused_groups) = realization_order(outputs, env);
143
144 // Try to simplify the RHS/LHS of a function definition by propagating its
145 // specializations' conditions
146 simplify_specializations(env);
147
148 LoweringLogger log;
149
150 debug(1) << "Creating initial loop nests...\n";
151 bool any_memoized = false;
152 Stmt s = schedule_functions(outputs, fused_groups, env, t, any_memoized);
153 log("Lowering after creating initial loop nests:", s);
154
155 if (any_memoized) {
156 debug(1) << "Injecting memoization...\n";
157 s = inject_memoization(s, env, pipeline_name, outputs);
158 log("Lowering after injecting memoization:", s);
159 } else {
160 debug(1) << "Skipping injecting memoization...\n";
161 }
162 debug(1) << "Injecting tracing...\n";
163 s = inject_tracing(s, pipeline_name, trace_pipeline, env, outputs, t);
164 log("Lowering after injecting tracing:", s);
165
166 debug(1) << "Adding checks for parameters\n";
167 s = add_parameter_checks(requirements, s, t);
168 log("Lowering after injecting parameter checks:", s);
169
170 // Compute the maximum and minimum possible value of each
171 // function. Used in later bounds inference passes.
172 debug(1) << "Computing bounds of each function's value\n";
173 FuncValueBounds func_bounds = compute_function_value_bounds(order, env);
174
175 // This pass injects nested definitions of variable names, so we
176 // can't simplify statements from here until we fix them up. (We
177 // can still simplify Exprs).
178 debug(1) << "Performing computation bounds inference...\n";
179 s = bounds_inference(s, outputs, order, fused_groups, env, func_bounds, t);
180 log("Lowering after computation bounds inference:", s);
181
182 debug(1) << "Removing extern loops...\n";
183 s = remove_extern_loops(s);
184 log("Lowering after removing extern loops:", s);
185
186 debug(1) << "Performing sliding window optimization...\n";
187 s = sliding_window(s, env);
188 log("Lowering after sliding window:", s);
189
190 // This uniquifies the variable names, so we're good to simplify
191 // after this point. This lets later passes assume syntactic
192 // equivalence means semantic equivalence.
193 debug(1) << "Uniquifying variable names...\n";
194 s = uniquify_variable_names(s);
195 log("Lowering after uniquifying variable names:", s);
196
197 debug(1) << "Simplifying...\n";
198 s = simplify(s, false); // Storage folding and allocation bounds inference needs .loop_max symbols
199 log("Lowering after first simplification:", s);
200
201 debug(1) << "Simplifying correlated differences...\n";
202 s = simplify_correlated_differences(s);
203 log("Lowering after simplifying correlated differences:", s);
204
205 debug(1) << "Performing allocation bounds inference...\n";
206 s = allocation_bounds_inference(s, env, func_bounds);
207 log("Lowering after allocation bounds inference:", s);
208
209 bool will_inject_host_copies =
210 (t.has_gpu_feature() ||
211 t.has_feature(Target::OpenGLCompute) ||
212 t.has_feature(Target::HexagonDma) ||
213 (t.arch != Target::Hexagon && (t.has_feature(Target::HVX))));
214
215 debug(1) << "Adding checks for images\n";
216 s = add_image_checks(s, outputs, t, order, env, func_bounds, will_inject_host_copies);
217 log("Lowering after injecting image checks:", s);
218
219 debug(1) << "Removing code that depends on undef values...\n";
220 s = remove_undef(s);
221 log("Lowering after removing code that depends on undef values:", s);
222
223 debug(1) << "Performing storage folding optimization...\n";
224 s = storage_folding(s, env);
225 log("Lowering after storage folding:", s);
226
227 debug(1) << "Injecting debug_to_file calls...\n";
228 s = debug_to_file(s, outputs, env);
229 log("Lowering after injecting debug_to_file calls:", s);
230
231 debug(1) << "Injecting prefetches...\n";
232 s = inject_prefetch(s, env);
233 log("Lowering after injecting prefetches:", s);
234
235 debug(1) << "Discarding safe promises...\n";
236 s = lower_safe_promises(s);
237 log("Lowering after discarding safe promises:", s);
238
239 debug(1) << "Dynamically skipping stages...\n";
240 s = skip_stages(s, order);
241 log("Lowering after dynamically skipping stages:", s);
242
243 debug(1) << "Forking asynchronous producers...\n";
244 s = fork_async_producers(s, env);
245 log("Lowering after forking asynchronous producers:", s);
246
247 debug(1) << "Destructuring tuple-valued realizations...\n";
248 s = split_tuples(s, env);
249 log("Lowering after destructuring tuple-valued realizations:", s);
250
251 // OpenGL relies on GPU var canonicalization occurring before
252 // storage flattening.
253 if (t.has_gpu_feature() ||
254 t.has_feature(Target::OpenGLCompute)) {
255 debug(1) << "Canonicalizing GPU var names...\n";
256 s = canonicalize_gpu_vars(s);
257 log("Lowering after canonicalizing GPU var names:", s);
258 }
259
260 debug(1) << "Bounding small realizations...\n";
261 s = simplify_correlated_differences(s);
262 s = bound_small_allocations(s);
263 log("Lowering after bounding small realizations:", s);
264
265 debug(1) << "Performing storage flattening...\n";
266 s = storage_flattening(s, outputs, env, t);
267 log("Lowering after storage flattening:", s);
268
269 debug(1) << "Adding atomic mutex allocation...\n";
270 s = add_atomic_mutex(s, env);
271 log("Lowering after adding atomic mutex allocation:", s);
272
273 debug(1) << "Unpacking buffer arguments...\n";
274 s = unpack_buffers(s);
275 log("Lowering after unpacking buffer arguments:", s);
276
277 if (any_memoized) {
278 debug(1) << "Rewriting memoized allocations...\n";
279 s = rewrite_memoized_allocations(s, env);
280 log("Lowering after rewriting memoized allocations:", s);
281 } else {
282 debug(1) << "Skipping rewriting memoized allocations...\n";
283 }
284
285 if (will_inject_host_copies) {
286 debug(1) << "Selecting a GPU API for GPU loops...\n";
287 s = select_gpu_api(s, t);
288 log("Lowering after selecting a GPU API:", s);
289
290 debug(1) << "Injecting host <-> dev buffer copies...\n";
291 s = inject_host_dev_buffer_copies(s, t);
292 log("Lowering after injecting host <-> dev buffer copies:", s);
293
294 debug(1) << "Selecting a GPU API for extern stages...\n";
295 s = select_gpu_api(s, t);
296 log("Lowering after selecting a GPU API for extern stages:", s);
297 }
298
299 debug(1) << "Simplifying...\n";
300 s = simplify(s);
301 s = unify_duplicate_lets(s);
302 log("Lowering after second simplifcation:", s);
303
304 debug(1) << "Reduce prefetch dimension...\n";
305 s = reduce_prefetch_dimension(s, t);
306 log("Lowering after reduce prefetch dimension:", s);
307
308 debug(1) << "Simplifying correlated differences...\n";
309 s = simplify_correlated_differences(s);
310 log("Lowering after simplifying correlated differences:", s);
311
312 debug(1) << "Unrolling...\n";
313 s = unroll_loops(s);
314 log("Lowering after unrolling:", s);
315
316 debug(1) << "Vectorizing...\n";
317 s = vectorize_loops(s, env, t);
318 s = simplify(s);
319 log("Lowering after vectorizing:", s);
320
321 if (t.has_gpu_feature() ||
322 t.has_feature(Target::OpenGLCompute)) {
323 debug(1) << "Injecting per-block gpu synchronization...\n";
324 s = fuse_gpu_thread_loops(s);
325 log("Lowering after injecting per-block gpu synchronization:", s);
326 }
327
328 debug(1) << "Detecting vector interleavings...\n";
329 s = rewrite_interleavings(s);
330 s = simplify(s);
331 log("Lowering after rewriting vector interleavings:", s);
332
333 debug(1) << "Partitioning loops to simplify boundary conditions...\n";
334 s = partition_loops(s);
335 s = simplify(s);
336 log("Lowering after partitioning loops:", s);
337
338 debug(1) << "Trimming loops to the region over which they do something...\n";
339 s = trim_no_ops(s);
340 log("Lowering after loop trimming:", s);
341
342 debug(1) << "Rebasing loops to zero...\n";
343 s = rebase_loops_to_zero(s);
344 debug(2) << "Lowering after rebasing loops to zero:\n"
345 << s << "\n\n";
346
347 debug(1) << "Hoisting loop invariant if statements...\n";
348 s = hoist_loop_invariant_if_statements(s);
349 log("Lowering after hoisting loop invariant if statements:", s);
350
351 debug(1) << "Injecting early frees...\n";
352 s = inject_early_frees(s);
353 log("Lowering after injecting early frees:", s);
354
355 if (t.has_feature(Target::FuzzFloatStores)) {
356 debug(1) << "Fuzzing floating point stores...\n";
357 s = fuzz_float_stores(s);
358 log("Lowering after fuzzing floating point stores:", s);
359 }
360
361 debug(1) << "Simplifying correlated differences...\n";
362 s = simplify_correlated_differences(s);
363 log("Lowering after simplifying correlated differences:", s);
364
365 debug(1) << "Bounding small allocations...\n";
366 s = bound_small_allocations(s);
367 log("Lowering after bounding small allocations:", s);
368
369 if (t.has_feature(Target::Profile)) {
370 debug(1) << "Injecting profiling...\n";
371 s = inject_profiling(s, pipeline_name);
372 log("Lowering after injecting profiling:", s);
373 }
374
375 if (t.has_feature(Target::CUDA)) {
376 debug(1) << "Injecting warp shuffles...\n";
377 s = lower_warp_shuffles(s);
378 log("Lowering after injecting warp shuffles:", s);
379 }
380
381 debug(1) << "Simplifying...\n";
382 s = common_subexpression_elimination(s);
383
384 debug(1) << "Lowering unsafe promises...\n";
385 s = lower_unsafe_promises(s, t);
386 log("Lowering after lowering unsafe promises:", s);
387
388 debug(1) << "Flattening nested ramps...\n";
389 s = flatten_nested_ramps(s);
390 log("Lowering after flattening nested ramps:", s);
391
392 debug(1) << "Removing dead allocations and moving loop invariant code...\n";
393 s = remove_dead_allocations(s);
394 s = simplify(s);
395 s = hoist_loop_invariant_values(s);
396 s = hoist_loop_invariant_if_statements(s);
397 log("Lowering after removing dead allocations and hoisting loop invariants:", s);
398
399 debug(1) << "Finding intrinsics...\n";
400 s = find_intrinsics(s);
401 log("Lowering after finding intrinsics:", s);
402
403 debug(1) << "Lowering after final simplification:\n"
404 << s << "\n\n";
405
406 if (!custom_passes.empty()) {
407 for (size_t i = 0; i < custom_passes.size(); i++) {
408 debug(1) << "Running custom lowering pass " << i << "...\n";
409 s = custom_passes[i]->mutate(s);
410 debug(1) << "Lowering after custom pass " << i << ":\n"
411 << s << "\n\n";
412 }
413 }
414
415 if (t.arch != Target::Hexagon && t.has_feature(Target::HVX)) {
416 debug(1) << "Splitting off Hexagon offload...\n";
417 s = inject_hexagon_rpc(s, t, result_module);
418 debug(2) << "Lowering after splitting off Hexagon offload:\n"
419 << s << "\n";
420 } else {
421 debug(1) << "Skipping Hexagon offload...\n";
422 }
423
424 if (t.has_gpu_feature()) {
425 debug(1) << "Offloading GPU loops...\n";
426 s = inject_gpu_offload(s, t);
427 debug(2) << "Lowering after splitting off GPU loops:\n"
428 << s << "\n\n";
429 } else {
430 debug(1) << "Skipping GPU offload...\n";
431 }
432
433 vector<Argument> public_args = args;
434 for (const auto &out : outputs) {
435 for (const Parameter &buf : out.output_buffers()) {
436 public_args.emplace_back(buf.name(),
437 Argument::OutputBuffer,
438 buf.type(), buf.dimensions(), buf.get_argument_estimates());
439 }
440 }
441
442 vector<InferredArgument> inferred_args = infer_arguments(s, outputs);
443 for (const InferredArgument &arg : inferred_args) {
444 if (arg.param.defined() && arg.param.name() == "__user_context") {
445 // The user context is always in the inferred args, but is
446 // not required to be in the args list.
447 continue;
448 }
449
450 internal_assert(arg.arg.is_input()) << "Expected only input Arguments here";
451
452 bool found = false;
453 for (const Argument &a : args) {
454 found |= (a.name == arg.arg.name);
455 }
456
457 if (arg.buffer.defined() && !found) {
458 // It's a raw Buffer used that isn't in the args
459 // list. Embed it in the output instead.
460 debug(1) << "Embedding image " << arg.buffer.name() << "\n";
461 result_module.append(arg.buffer);
462 } else if (!found) {
463 std::ostringstream err;
464 err << "Generated code refers to ";
465 if (arg.arg.is_buffer()) {
466 err << "image ";
467 }
468 err << "parameter " << arg.arg.name
469 << ", which was not found in the argument list.\n";
470
471 err << "\nArgument list specified: ";
472 for (size_t i = 0; i < args.size(); i++) {
473 err << args[i].name << " ";
474 }
475 err << "\n\nParameters referenced in generated code: ";
476 for (const InferredArgument &ia : inferred_args) {
477 if (ia.arg.name != "__user_context") {
478 err << ia.arg.name << " ";
479 }
480 }
481 err << "\n\n";
482 user_error << err.str();
483 }
484 }

486 // We're about to drop the environment and outputs vector, which
487 // contain the only strong refs to Functions that may still be
488 // pointed to by the IR. So make those refs strong.
489 class StrengthenRefs : public IRMutator {
490 using IRMutator::visit;
491 Expr visit(const Call *c) override {
492 Expr expr = IRMutator::visit(c);
493 c = expr.as<Call>();
494 internal_assert(c);
495 if (c->func.defined()) {
496 FunctionPtr ptr = c->func;
497 ptr.strengthen();
498 expr = Call::make(c->type, c->name, c->args, c->call_type,
499 ptr, c->value_index,
500 c->image, c->param);
501 }
502 return expr;
503 }
504 };
505 s = StrengthenRefs().mutate(s);
506
507 LoweredFunc main_func(pipeline_name, public_args, s, linkage_type);
508
509 // If we're in debug mode, add code that prints the args.
510 if (t.has_feature(Target::Debug)) {
511 debug_arguments(&main_func, t);
512 }
513
514 result_module.append(main_func);
515
516 auto *logger = get_compiler_logger();
517 if (logger) {
518 auto time_end = std::chrono::high_resolution_clock::now();
519 std::chrono::duration<double> diff = time_end - time_start;
520 logger->record_compilation_time(CompilerLogger::Phase::HalideLowering, diff.count());
521 }
522}

526Module lower(const vector<Function> &output_funcs,
527 const string &pipeline_name,
528 const Target &t,
529 const vector<Argument> &args,
530 const LinkageType linkage_type,
531 const vector<Stmt> &requirements,
532 bool trace_pipeline,
533 const vector<IRMutator *> &custom_passes) {
534 Module result_module{extract_namespaces(pipeline_name), t};
535 run_with_large_stack([&]() {
536 lower_impl(output_funcs, pipeline_name, t, args, linkage_type, requirements, trace_pipeline, custom_passes, result_module);
537 });
538 return result_module;
539}

541Stmt lower_main_stmt(const std::vector<Function> &output_funcs,
542 const std::string &pipeline_name,
543 const Target &t,
544 const std::vector<Stmt> &requirements,
545 bool trace_pipeline,
546 const std::vector<IRMutator *> &custom_passes) {
547 // We really ought to start applying for appellation d'origine contrôlée
548 // status on types representing arguments in the Halide compiler.
549 vector<InferredArgument> inferred_args = infer_arguments(Stmt(), output_funcs);
550 vector<Argument> args;
551 for (const auto &ia : inferred_args) {
552 if (!ia.arg.name.empty() && ia.arg.is_input()) {
553 args.push_back(ia.arg);
554 }
555 }
556
557 Module module = lower(output_funcs, pipeline_name, t, args, LinkageType::External, requirements, trace_pipeline, custom_passes);
558
559 return module.functions().front().body;
560}
