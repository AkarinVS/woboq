114string CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::print_type(Type type, AppendSpaceIfNeeded space) {
115 ostringstream oss;
116 if (type.is_float()) {
117 if (type.bits() == 16) {
118 user_assert(target.has_feature(Target::CLHalf))
119 << "OpenCL kernel uses half type, but CLHalf target flag not enabled\n";
120 oss << "half";
121 } else if (type.bits() == 32) {
122 oss << "float";
123 } else if (type.bits() == 64) {
124 user_assert(target.has_feature(Target::CLDoubles))
125 << "OpenCL kernel uses double type, but CLDoubles target flag not enabled\n";
126 oss << "double";
127 } else {
128 user_error << "Can't represent a float with this many bits in OpenCL C: " << type << "\n";
129 }
130
131 } else {
132 if (type.is_uint() && type.bits() > 1) {
133 oss << "u";
134 }
135 switch (type.bits()) {
136 case 1:
137 internal_assert(type.lanes() == 1) << "Encountered vector of bool\n";
138 oss << "bool";
139 break;
140 case 8:
141 oss << "char";
142 break;
143 case 16:
144 oss << "short";
145 break;
146 case 32:
147 oss << "int";
148 break;
149 case 64:
150 oss << "long";
151 break;
152 default:
153 user_error << "Can't represent an integer with this many bits in OpenCL C: " << type << "\n";
154 }
155 }
156 if (type.lanes() != 1) {
157 switch (type.lanes()) {
158 case 2:
159 case 3:
160 case 4:
161 case 8:
162 case 16:
163 oss << type.lanes();
164 break;
165 default:
166 user_error << "Unsupported vector width in OpenCL C: " << type << "\n";
167 }
168 }
169 if (space == AppendSpace) {
170 oss << " ";
171 }
172 return oss.str();
173}

186string simt_intrinsic(const string &name) {
187 if (ends_with(name, ".__thread_id_x")) {
188 return "get_local_id(0)";
189 } else if (ends_with(name, ".__thread_id_y")) {
190 return "get_local_id(1)";
191 } else if (ends_with(name, ".__thread_id_z")) {
192 return "get_local_id(2)";
193 } else if (ends_with(name, ".__thread_id_w")) {
194 return "get_local_id(3)";
195 } else if (ends_with(name, ".__block_id_x")) {
196 return "get_group_id(0)";
197 } else if (ends_with(name, ".__block_id_y")) {
198 return "get_group_id(1)";
199 } else if (ends_with(name, ".__block_id_z")) {
200 return "get_group_id(2)";
201 } else if (ends_with(name, ".__block_id_w")) {
202 return "get_group_id(3)";
203 }
204 internal_error << "simt_intrinsic called on bad variable name: " << name << "\n";
205 return "";
206}

209void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const For *loop) {
210 user_assert(loop->for_type != ForType::GPULane)
211 << "The OpenCL backend does not support the gpu_lanes() scheduling directive.";
212
213 if (is_gpu_var(loop->name)) {
214 internal_assert((loop->for_type == ForType::GPUBlock) ||
215 (loop->for_type == ForType::GPUThread))
216 << "kernel loop must be either gpu block or gpu thread\n";
217 internal_assert(is_const_zero(loop->min));
218
219 stream << get_indent() << print_type(Int(32)) << " " << print_name(loop->name)
220 << " = " << simt_intrinsic(loop->name) << ";\n";
221
222 loop->body.accept(this);
223
224 } else {
225 user_assert(loop->for_type != ForType::Parallel) << "Cannot use parallel loops inside OpenCL kernel\n";
226 CodeGen_C::visit(loop);
227 }

280void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const Call *op) {
281 if (op->is_intrinsic(Call::bool_to_mask)) {
282 if (op->args[0].type().is_vector()) {
283 // The argument is already a mask of the right width. Just
284 // sign-extend to the expected type.
285 op->args[0].accept(this);
286 } else {
287 // The argument is a scalar bool. Casting it to an int
288 // produces zero or one. Convert it to -1 of the requested
289 // type.
290 Expr equiv = -Cast::make(op->type, op->args[0]);
291 equiv.accept(this);
292 }
293 } else if (op->is_intrinsic(Call::cast_mask)) {
294 // Sign-extension is fine
295 Expr equiv = Cast::make(op->type, op->args[0]);
296 equiv.accept(this);
297 } else if (op->is_intrinsic(Call::select_mask)) {
298 internal_assert(op->args.size() == 3);
299 string cond = print_expr(op->args[0]);
300 string true_val = print_expr(op->args[1]);
301 string false_val = print_expr(op->args[2]);
302
303 // Yes, you read this right. OpenCL's select function is declared
304 // 'select(false_case, true_case, condition)'.
305 ostringstream rhs;
306 rhs << "select(" << false_val << ", " << true_val << ", " << cond << ")";
307 print_assignment(op->type, rhs.str());
308 } else if (op->is_intrinsic(Call::abs)) {
309 if (op->type.is_float()) {
310 ostringstream rhs;
311 rhs << "abs_f" << op->type.bits() << "(" << print_expr(op->args[0]) << ")";
312 print_assignment(op->type, rhs.str());
313 } else {
314 ostringstream rhs;
315 rhs << "abs(" << print_expr(op->args[0]) << ")";
316 print_assignment(op->type, rhs.str());
317 }
318 } else if (op->is_intrinsic(Call::absd)) {
319 ostringstream rhs;
320 rhs << "abs_diff(" << print_expr(op->args[0]) << ", " << print_expr(op->args[1]) << ")";
321 print_assignment(op->type, rhs.str());
322 } else if (op->is_intrinsic(Call::gpu_thread_barrier)) {
323 internal_assert(op->args.size() == 1) << "gpu_thread_barrier() intrinsic must specify memory fence type.\n";
324
325 const auto *fence_type_ptr = as_const_int(op->args[0]);
326 internal_assert(fence_type_ptr) << "gpu_thread_barrier() parameter is not a constant integer.\n";
327 auto fence_type = *fence_type_ptr;
328
329 stream << get_indent() << "barrier(0";
330 if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device) {
331 stream << " | CLK_GLOBAL_MEM_FENCE";
332 }
333 if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Shared) {
334 stream << " | CLK_LOCAL_MEM_FENCE";
335 }
336 stream << ");\n";
337 print_assignment(op->type, "0");
338 } else if (op->is_intrinsic(Call::shift_left) || op->is_intrinsic(Call::shift_right)) {
339 // Some OpenCL implementations forbid mixing signed-and-unsigned shift values;
340 // if the RHS is uint, quietly cast it back to int if the LHS is int
341 if (op->args[0].type().is_int() && op->args[1].type().is_uint()) {
342 Type t = op->args[0].type().with_code(halide_type_int);
343 Expr e = Call::make(op->type, op->name, {op->args[0], cast(t, op->args[1])}, op->call_type);
344 e.accept(this);
345 } else {
346 CodeGen_C::visit(op);
347 }
348 } else if (op->is_intrinsic(Call::image_load)) {
349 // image_load(<image name>, <buffer>, <x>, <x-extent>, <y>,
350 // <y-extent>, <z>, <z-extent>)
351 int dims = (op->args.size() - 2) / 2;
352 internal_assert(dims >= 1 && dims <= 3);
353 const StringImm *string_imm = op->args[0].as<StringImm>();
354 if (!string_imm) {
355 internal_assert(op->args[0].as<Broadcast>());
356 string_imm = op->args[0].as<Broadcast>()->value.as<StringImm>();
357 }
358 internal_assert(string_imm);
359 Type arg_type = op->args[2].type();
360 internal_assert(arg_type.lanes() <= 16);
361 internal_assert(arg_type.lanes() == op->type.lanes());
362
363 std::array<string, 3> coord;
364 for (int i = 0; i < dims; i++) {
365 coord[i] = print_expr(op->args[i * 2 + 2]);
366 }
367 vector<string> results(arg_type.lanes());
368 // For vectorized reads, codegen as a sequence of read_image calls
369 for (int i = 0; i < arg_type.lanes(); i++) {
370 ostringstream rhs;
371 rhs << "read_image" << image_type_suffix(op->type) << "(" << print_name(string_imm->value) << ", ";
372 string idx = arg_type.is_vector() ? string(".s") + vector_elements[i] : "";
373 switch (dims) {
374 case 1:
375 rhs << coord[0] << idx << ").s0";
376 break;
377 case 2:
378 rhs << "(int2)(" << coord[0] << idx << ", " << coord[1] << idx << ")).s0";
379 break;
380 case 3:
381 rhs << "(int4)(" << coord[0] << idx << ", " << coord[1] << idx
382 << ", " << coord[2] << idx << ", 0)).s0";
383 break;
384 }
385 print_assignment(op->type.with_bits(32).with_lanes(1), rhs.str());
386 results[i] = id;
387 }
388
389 if (op->type.is_vector()) {
390 // Combine all results into a single vector
391 ostringstream rhs;
392 rhs << "(" << print_type(op->type) << ")(";
393 for (int i = 0; i < op->type.lanes(); i++) {
394 rhs << results[i];
395 if (i < op->type.lanes() - 1) {
396 rhs << ", ";
397 }
398 }
399 rhs << ")";
400 print_assignment(op->type, rhs.str());
401 }
402 if (op->type.bits() != 32) {
403 // Widen to the correct type
404 print_assignment(op->type, "convert_" + print_type(op->type) + "(" + id + ")");
405 }
406 } else if (op->is_intrinsic(Call::image_store)) {
407 // image_store(<image name>, <buffer>, <x>, <y>, <z>, <value>)
408 const StringImm *string_imm = op->args[0].as<StringImm>();
409 if (!string_imm) {
410 internal_assert(op->args[0].as<Broadcast>());
411 string_imm = op->args[0].as<Broadcast>()->value.as<StringImm>();
412 }
413 internal_assert(string_imm);
414 int dims = op->args.size() - 3;
415 internal_assert(dims >= 1 && dims <= 3);
416 Type arg_type = op->args[2].type();
417 internal_assert(arg_type.lanes() <= 16);
418 Type value_type = op->args.back().type();
419 internal_assert(arg_type.lanes() == value_type.lanes());
420
421 std::array<string, 3> coord;
422 for (int i = 0; i < dims; i++) {
423 coord[i] = print_expr(op->args[i + 2]);
424 }
425 string value = print_expr(op->args.back());
426 // For vectorized writes, codegen as a sequence of write_image calls
427 for (int i = 0; i < arg_type.lanes(); i++) {
428 ostringstream write_image;
429 write_image << "write_image" << image_type_suffix(op->type)
430 << "(" << print_name(string_imm->value) << ", ";
431 string idx = arg_type.is_vector() ? string(".s") + vector_elements[i] : "";
432 switch (dims) {
433 case 1:
434 write_image << coord[0] << idx;
435 break;
436 case 2:
437 write_image << "(int2)(" << coord[0] << idx << ", " << coord[1] << idx << ")";
438 break;
439 case 3:
440 write_image << "(int4)(" << coord[0] << idx << ", " << coord[1] << idx
441 << ", " << coord[2] << idx << ", 0)";
442 break;
443 }
444 write_image << ", (" << print_type(value_type.with_bits(32).with_lanes(4))
445 << ")(" << value << idx << ", 0, 0, 0));\n";
446 // do_indent();
447 stream << write_image.str();
448 }
449 } else {
450 CodeGen_C::visit(op);
451 }
452}

485void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const Load *op) {
486 user_assert(is_const_one(op->predicate)) << "Predicated load is not supported inside OpenCL kernel.\n";
487
488 // If we're loading a contiguous ramp into a vector, use vload instead.
489 Expr ramp_base = strided_ramp_base(op->index);
490 if (ramp_base.defined()) {
491 internal_assert(op->type.is_vector());
492
493 ostringstream rhs;
494 if ((op->alignment.modulus % op->type.lanes() == 0) &&
495 (op->alignment.remainder % op->type.lanes() == 0)) {
496 // Get the rhs just for the cache.
497 string id_ramp_base = print_expr(ramp_base / op->type.lanes());
498 string array_indexing = print_array_access(op->name, op->type, id_ramp_base);
499
500 rhs << array_indexing;
501 } else {
502 string id_ramp_base = print_expr(ramp_base);
503 rhs << "vload" << op->type.lanes()
504 << "(0, (" << get_memory_space(op->name) << " "
505 << print_type(op->type.element_of()) << "*)"
506 << print_name(op->name) << " + " << id_ramp_base << ")";
507 }
508 print_assignment(op->type, rhs.str());
509 return;
510 }
511
512 string id_index = print_expr(op->index);
513
514 // Get the rhs just for the cache.
515 string array_indexing = print_array_access(op->name, op->type, id_index);
516
517 std::map<string, string>::iterator cached = cache.find(array_indexing);
518 if (cached != cache.end()) {
519 id = cached->second;
520 return;
521 }
522
523 if (op->index.type().is_vector()) {
524 // If index is a vector, gather vector elements.
525 internal_assert(op->type.is_vector());
526
527 id = "_" + unique_name('V');
528 cache[array_indexing] = id;
529
530 stream << get_indent() << print_type(op->type)
531 << " " << id << ";\n";
532
533 for (int i = 0; i < op->type.lanes(); ++i) {
534 stream << get_indent();
535 stream
536 << id << ".s" << vector_elements[i]
537 << " = ((" << get_memory_space(op->name) << " "
538 << print_type(op->type.element_of()) << "*)"
539 << print_name(op->name) << ")"
540 << "[" << id_index << ".s" << vector_elements[i] << "];\n";
541 }
542 } else {
543 print_assignment(op->type, array_indexing);
544 }
545}

547void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const Store *op) {
548 user_assert(is_const_one(op->predicate)) << "Predicated store is not supported inside OpenCL kernel.\n";
549
550 if (emit_atomic_stores) {
551 // Currently only support scalar atomics.
552 user_assert(op->value.type().is_scalar()) << "OpenCL atomic store does not support vectorization.\n";
553 user_assert(op->value.type().bits() >= 32) << "OpenCL only support 32 and 64 bit atomics.\n";
554 if (op->value.type().bits() == 64) {
555 user_assert(target.has_feature(Target::CLAtomics64))
556 << "Enable feature CLAtomics64 for 64-bit atomics in OpenCL.\n";
557 }
558 // Detect whether we can describe this as an atomic-read-modify-write,
559 // otherwise fallback to a compare-and-swap loop.
560 // Current only test for atomic add.
561 Expr val_expr = op->value;
562 Type t = val_expr.type();
563 Expr equiv_load = Load::make(t, op->name, op->index, Buffer<>(), op->param, op->predicate, op->alignment);
564 Expr delta = simplify(common_subexpression_elimination(op->value - equiv_load));
565 // For atomicAdd, we check if op->value - store[index] is independent of store.
566 // The atomicAdd operations in OpenCL only supports integers so we also check that.
567 bool is_atomic_add = t.is_int_or_uint() && !expr_uses_var(delta, op->name);
568 bool type_cast_needed = !(allocations.contains(op->name) &&
569 allocations.get(op->name).type == t);
570 auto print_store_var = [&]() {
571 if (type_cast_needed) {
572 stream << "(("
573 << get_memory_space(op->name) << " "
574 << print_type(t)
575 << " *)"
576 << print_name(op->name)
577 << ")";
578 } else {
579 stream << print_name(op->name);
580 }
581 };
582 if (is_atomic_add) {
583 string id_index = print_expr(op->index);
584 string id_delta = print_expr(delta);
585 stream << get_indent();
586 // atomic_add(&x[i], delta);
587 if (t.bits() == 32) {
588 stream << "atomic_add(&";
589 } else {
590 stream << "atom_add(&";
591 }
592
593 print_store_var();
594 stream << "[" << id_index << "]";
595 stream << "," << id_delta << ");\n";
596 } else {
597 // CmpXchg loop
598 // {
599 // union {unsigned int i; float f;} old_val;
600 // union {unsigned int i; float f;} new_val;
601 // do {
602 // old_val.f = x[id_index];
603 // new_val.f = ...
604 // } while(atomic_cmpxchg((volatile address_space unsigned int*)&x[id_index], old_val.i, new_val.i) != old_val.i);
605 // }
606 stream << get_indent() << "{\n";
607 indent += 2;
608 string id_index = print_expr(op->index);
609 std::string int_type = t.bits() == 32 ? "int" : "long";
610 if (t.is_float() || t.is_uint()) {
611 int_type = "unsigned " + int_type;
612 }
613 if (t.is_float()) {
614 stream << get_indent() << "union {" << int_type << " i; " << print_type(t) << " f;} old_val;\n";
615 stream << get_indent() << "union {" << int_type << " i; " << print_type(t) << " f;} new_val;\n";
616 } else {
617 stream << get_indent() << int_type << " old_val;\n";
618 stream << get_indent() << int_type << " new_val;\n";
619 }
620 stream << get_indent() << "do {\n";
621 indent += 2;
622 stream << get_indent();
623 if (t.is_float()) {
624 stream << "old_val.f = ";
625 } else {
626 stream << "old_val = ";
627 }
628 print_store_var();
629 stream << "[" << id_index << "];\n";
630 string id_value = print_expr(op->value);
631 stream << get_indent();
632 if (t.is_float()) {
633 stream << "new_val.f = ";
634 } else {
635 stream << "new_val = ";
636 }
637 stream << id_value << ";\n";
638 indent -= 2;
639 std::string old_val = t.is_float() ? "old_val.i" : "old_val";
640 std::string new_val = t.is_float() ? "new_val.i" : "new_val";
641 stream << get_indent()
642 << "} while(atomic_cmpxchg((volatile "
643 << get_memory_space(op->name) << " " << int_type << "*)&"
644 << print_name(op->name) << "[" << id_index << "], "
645 << old_val << ", " << new_val << ") != " << old_val << ");\n"
646 << get_indent() << "}\n";
647 indent -= 2;
648 }
649 cache.clear();
650 return;
651 }
652
653 string id_value = print_expr(op->value);
654 Type t = op->value.type();
655
656 // If we're writing a contiguous ramp, use vstore instead.
657 Expr ramp_base = strided_ramp_base(op->index);
658 if (ramp_base.defined()) {
659 internal_assert(op->value.type().is_vector());
660
661 if ((op->alignment.modulus % op->value.type().lanes() == 0) &&
662 (op->alignment.remainder % op->value.type().lanes() == 0)) {
663 string id_ramp_base = print_expr(ramp_base / op->value.type().lanes());
664 string array_indexing = print_array_access(op->name, t, id_ramp_base);
665 stream << get_indent() << array_indexing << " = " << id_value << ";\n";
666 } else {
667 string id_ramp_base = print_expr(ramp_base);
668 stream << get_indent() << "vstore" << t.lanes() << "("
669 << id_value << ","
670 << 0 << ", (" << get_memory_space(op->name) << " "
671 << print_type(t.element_of()) << "*)"
672 << print_name(op->name) << " + " << id_ramp_base
673 << ");\n";
674 }
675 } else if (op->index.type().is_vector()) {
676 // If index is a vector, scatter vector elements.
677 internal_assert(t.is_vector());
678
679 string id_index = print_expr(op->index);
680
681 for (int i = 0; i < t.lanes(); ++i) {
682 stream << get_indent() << "((" << get_memory_space(op->name) << " "
683 << print_type(t.element_of()) << " *)"
684 << print_name(op->name)
685 << ")["
686 << id_index << ".s" << vector_elements[i] << "] = "
687 << id_value << ".s" << vector_elements[i] << ";\n";
688 }
689 } else {
690 string id_index = print_expr(op->index);
691 stream << get_indent();
692 std::string array_indexing = print_array_access(op->name, t, id_index);
693 stream << array_indexing << " = " << id_value << ";\n";
694 }
695
696 cache.clear();
697}

753void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const Allocate *op) {
754 user_assert(!op->new_expr.defined()) << "Allocate node inside OpenCL kernel has custom new expression.\n"
755 << "(Memoization is not supported inside GPU kernels at present.)\n";
756
757 if (op->memory_type == MemoryType::GPUShared) {
758 // Already handled
759 op->body.accept(this);
760 } else {
761 open_scope();
762
763 debug(2) << "Allocate " << op->name << " on device\n";
764
765 debug(3) << "Pushing allocation called " << op->name << " onto the symbol table\n";
766
767 // Allocation is not a shared memory allocation, just make a local declaration.
768 // It must have a constant size.
769 int32_t size = op->constant_allocation_size();
770 user_assert(size > 0)
771 << "Allocation " << op->name << " has a dynamic size. "
772 << "Only fixed-size allocations are supported on the gpu. "
773 << "Try storing into shared memory instead.";

776 << print_name(op->name) << "[" << size << "];\n";
777 stream << get_indent() << "#define " << get_memory_space(op->name) << " __private\n";
778
779 Allocation alloc;
780 alloc.type = op->type;
781 allocations.push(op->name, alloc);
782
783 op->body.accept(this);
784
785 // Should have been freed internally
786 internal_assert(!allocations.contains(op->name));
787
788 close_scope("alloc " + print_name(op->name));
789 }

807void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::visit(const Shuffle *op) {
808 if (op->is_interleave()) {
809 int op_lanes = op->type.lanes();
810 internal_assert(!op->vectors.empty());
811 int arg_lanes = op->vectors[0].type().lanes();
812 if (op->vectors.size() == 1) {
813 // 1 argument, just do a simple assignment
814 internal_assert(op_lanes == arg_lanes);
815 print_assignment(op->type, print_expr(op->vectors[0]));
816 } else if (op->vectors.size() == 2) {
817 // 2 arguments, set the .even to the first arg and the
818 // .odd to the second arg
819 internal_assert(op->vectors[1].type().lanes() == arg_lanes);
820 internal_assert(op_lanes / 2 == arg_lanes);
821 string a1 = print_expr(op->vectors[0]);
822 string a2 = print_expr(op->vectors[1]);
823 id = unique_name('_');
824 stream << get_indent() << print_type(op->type) << " " << id << ";\n";
825 stream << get_indent() << id << ".even = " << a1 << ";\n";
826 stream << get_indent() << id << ".odd = " << a2 << ";\n";
827 } else {
828 // 3+ arguments, interleave via a vector literal
829 // selecting the appropriate elements of the vectors
830 int dest_lanes = op->type.lanes();
831 internal_assert(dest_lanes <= 16);
832 int num_vectors = op->vectors.size();
833 vector<string> arg_exprs(num_vectors);
834 for (int i = 0; i < num_vectors; i++) {
835 internal_assert(op->vectors[i].type().lanes() == arg_lanes);
836 arg_exprs[i] = print_expr(op->vectors[i]);
837 }
838 internal_assert(num_vectors * arg_lanes >= dest_lanes);
839 id = unique_name('_');
840 stream << get_indent() << print_type(op->type) << " " << id;
841 stream << " = (" << print_type(op->type) << ")(";
842 for (int i = 0; i < dest_lanes; i++) {
843 int arg = i % num_vectors;
844 int arg_idx = i / num_vectors;
845 internal_assert(arg_idx <= arg_lanes);
846 stream << arg_exprs[arg] << ".s" << vector_elements[arg_idx];
847 if (i != dest_lanes - 1) {
848 stream << ", ";
849 }
850 }
851 stream << ");\n";
852 }
853 } else {
854 internal_error << "Shuffle not implemented.\n";
855 }
856}

888struct BufferSize {
889 string name;
890 size_t size = 0;
891
892 BufferSize() = default;
893 BufferSize(string name, size_t size)
894 : name(std::move(name)), size(size) {
895 }
896
897 bool operator<(const BufferSize &r) const {
898 return size < r.size;
899 }

903void CodeGen_OpenCL_Dev::CodeGen_OpenCL_C::add_kernel(Stmt s,
904 const string &name,
905 const vector<DeviceArgument> &args) {
906
907 debug(2) << "Adding OpenCL kernel " << name << "\n";
908
909 debug(2) << "Eliminating bool vectors\n";
910 s = eliminate_bool_vectors(s);
911 debug(2) << "After eliminating bool vectors:\n"
912 << s << "\n";
913
914 // Figure out which arguments should be passed in __constant.
915 // Such arguments should be:
916 // - not written to,
917 // - loads are block-uniform,
918 // - constant size,
919 // - and all allocations together should be less than the max constant
920 // buffer size given by CL_DEVICE_MAX_CONSTANT_BUFFER_SIZE.
921 // The last condition is handled via the preprocessor in the kernel
922 // declaration.
923 vector<BufferSize> constants;
924 for (size_t i = 0; i < args.size(); i++) {
925 if (args[i].is_buffer &&
926 CodeGen_GPU_Dev::is_buffer_constant(s, args[i].name) &&
927 args[i].size > 0) {
928 constants.emplace_back(args[i].name, args[i].size);
929 }
930 }
931
932 // Sort the constant candidates from smallest to largest. This will put
933 // as many of the constant allocations in __constant as possible.
934 // Ideally, we would prioritize constant buffers by how frequently they
935 // are accessed.
936 sort(constants.begin(), constants.end());
937
938 // Compute the cumulative sum of the constants.
939 for (size_t i = 1; i < constants.size(); i++) {
940 constants[i].size += constants[i - 1].size;
941 }
942
943 // Create preprocessor replacements for the address spaces of all our buffers.
944 stream << "// Address spaces for " << name << "\n";
945 for (size_t i = 0; i < args.size(); i++) {
946 if (args[i].is_buffer) {
947 vector<BufferSize>::iterator constant = constants.begin();
948 while (constant != constants.end() &&
949 constant->name != args[i].name) {
950 constant++;
951 }
952
953 if (constant != constants.end()) {
954 stream << "#if " << constant->size << " <= MAX_CONSTANT_BUFFER_SIZE && "
955 << constant - constants.begin() << " < MAX_CONSTANT_ARGS\n";
956 stream << "#define " << get_memory_space(args[i].name) << " __constant\n";
957 stream << "#else\n";
958 stream << "#define " << get_memory_space(args[i].name) << " __global\n";
959 stream << "#endif\n";
960 } else {
961 stream << "#define " << get_memory_space(args[i].name) << " __global\n";
962 }
963 }
964 }
965
966 // Emit the function prototype.
967 stream << "__kernel void " << name << "(\n";
968 for (size_t i = 0; i < args.size(); i++) {
969 if (args[i].is_buffer) {
970 if (args[i].memory_type == MemoryType::GPUTexture) {
971 int dims = args[i].dimensions;
972 internal_assert(dims >= 1 && dims <= 3) << "dims = " << dims << "\n";
973 if (args[i].read && args[i].write) {
974 stream << " __read_write ";
975 } else if (args[i].read) {
976 stream << " __read_only ";
977 } else if (args[i].write) {
978 stream << " __write_only ";
979 } else {
980 internal_error << "CL Image argument " << args[i].name
981 << " is neither read nor write";
982 }
983 stream << "image" << dims << "d_t ";
984 } else {
985 stream << " " << get_memory_space(args[i].name) << " ";
986 if (!args[i].write) {
987 stream << "const ";
988 }
989 stream << print_type(args[i].type) << " *"
990 << "restrict ";
991 }
992 stream << print_name(args[i].name);
993 Allocation alloc;
994 alloc.type = args[i].type;
995 allocations.push(args[i].name, alloc);
996 } else {
997 Type t = args[i].type;
998 string name = args[i].name;
999 // Bools are passed as a uint8.
1000 t = t.with_bits(t.bytes() * 8);
1001 // float16 are passed as uints
1002 if (t.is_float() && t.bits() < 32) {
1003 t = t.with_code(halide_type_uint);
1004 name += "_bits";
1005 }
1006 stream << " const "
1007 << print_type(t)
1008 << " "
1009 << print_name(name);
1010 }
1011
1012 if (i < args.size() - 1) {
1013 stream << ",\n";
1014 }
1015 }
1016
1017 class FindShared : public IRVisitor {
1018 using IRVisitor::visit;
1019 void visit(const Allocate *op) override {
1020 if (op->memory_type == MemoryType::GPUShared) {
1021 internal_assert(alloc == nullptr)
1022 << "Found multiple shared allocations in opencl kernel\n";
1023 alloc = op;
1024 }
1025 }
1026
1027 public:
1028 const Allocate *alloc = nullptr;
1029 } find_shared;
1030 s.accept(&find_shared);
1031
1032 if (find_shared.alloc) {
1033 shared_name = find_shared.alloc->name;
1034 } else {
1035 shared_name = "__shared";
1036 }
1037 // Note that int16 below is an int32x16, not an int16_t. The type
1038 // is chosen to be large to maximize alignment.
1039 stream << ",\n"
1040 << " __local int16* "
1041 << print_name(shared_name)
1042 << ")\n";
1043
1044 open_scope();
1045
1046 // Reinterpret half args passed as uint16 back to half
1047 for (size_t i = 0; i < args.size(); i++) {
1048 if (!args[i].is_buffer &&
1049 args[i].type.is_float() &&
1050 args[i].type.bits() < 32) {
1051 stream << " const " << print_type(args[i].type)
1052 << " " << print_name(args[i].name)
1053 << " = half_from_bits(" << print_name(args[i].name + "_bits") << ");\n";
1054 }
1055 }
1056
1057 print(s);
1058 close_scope("kernel " + name);
1059
1060 for (size_t i = 0; i < args.size(); i++) {
1061 // Remove buffer arguments from allocation scope
1062 if (args[i].is_buffer) {
1063 allocations.pop(args[i].name);
1064 }
1065 }
1066
1067 // Undef all the buffer address spaces, in case they're different in another kernel.
1068 for (size_t i = 0; i < args.size(); i++) {
1069 if (args[i].is_buffer) {
1070 stream << "#undef " << get_memory_space(args[i].name) << "\n";
1071 }
1072 }
1073}
