11// Prefix all names in an expression with some string.
12class QualifyExpr : public IRMutator {
13 using IRMutator::visit;
14
15 const string &prefix;
16
17 Expr visit(const Variable *v) override {
18 if (v->param.defined()) {
19 return v;
20 } else {
21 return Variable::make(v->type, prefix + v->name, v->reduction_domain);
22 }
23 }
24 Expr visit(const Let *op) override {
25 Expr value = mutate(op->value);
26 Expr body = mutate(op->body);
27 return Let::make(prefix + op->name, value, body);
28 }
29
30public:
31 QualifyExpr(const string &p)
32 : prefix(p) {
33 }

38Expr qualify(const string &prefix, const Expr &value) {

40 return q.mutate(value);
