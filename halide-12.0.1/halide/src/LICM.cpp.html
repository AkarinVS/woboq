54// Lift pure loop invariants to the top level. Applied independently
55// to each loop.
56class LiftLoopInvariants : public IRMutator {
57 using IRMutator::visit;
58
59 Scope<> varying;
60
61 bool can_lift(const Expr &e) {
62 CanLift check(varying);
63 e.accept(&check);
64 return check.result;
65 }
66
67 bool should_lift(const Expr &e) {
68 if (!can_lift(e)) {
69 return false;
70 }
71 if (e.as<Variable>()) {
72 return false;
73 }
74 if (e.as<Broadcast>()) {
75 return false;
76 }
77 if (is_const(e)) {
78 return false;
79 }
80 // bool vectors are buggy enough in LLVM that lifting them is a bad idea.
81 // (We just skip all vectors on the principle that we don't want them
82 // on the stack anyway.)
83 if (e.type().is_vector()) {
84 return false;
85 }
86 if (const Cast *cast = e.as<Cast>()) {
87 if (cast->type.bytes() > cast->value.type().bytes()) {
88 // Don't lift widening casts.
89 return false;
90 }
91 }
92 if (const Add *add = e.as<Add>()) {
93 if (add->type == Int(32) &&
94 is_const(add->b)) {
95 // Don't lift constant integer offsets. They're often free.
96 return false;
97 }
98 }
99 if (const Call *call = e.as<Call>()) {
100 if (Call::as_tag(call) ||
101 call->is_intrinsic(Call::reinterpret)) {
102 // Don't lift these intrinsics. They're free.
103 return should_lift(call->args[0]);
104 }
105 if (call->is_intrinsic(Call::size_of_halide_buffer_t)) {
106 return true;
107 }
108 }
109 return true;
110 }

112 template<typename T, typename Body>
113 Body visit_let(const T *op) {
114 // Visit an entire chain of lets in a single method to conserve stack space.
115 struct Frame {
116 const T *op;
117 Expr new_value;
118 ScopedBinding<> binding;
119 Frame(const T *op, Expr v, Scope<> &scope)
120 : op(op), new_value(std::move(v)), binding(scope, op->name) {
121 }
122 };
123 vector<Frame> frames;
124 Body result;
125 do {
126 frames.emplace_back(op, mutate(op->value), varying);
127 result = op->body;
128 } while ((op = result.template as<T>()));
129
130 result = mutate(result);
131
132 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
133 if (it->new_value.same_as(it->op->value) && result.same_as(it->op->body)) {
134 result = it->op;
135 } else {
136 result = T::make(it->op->name, std::move(it->new_value), result);
137 }
138 }
139
140 return result;
141 }
142
143 Expr visit(const Let *op) override {
144 return visit_let<Let, Expr>(op);
145 }
146
147 Stmt visit(const LetStmt *op) override {
148 return visit_let<LetStmt, Stmt>(op);
149 }
150
151 Stmt visit(const For *op) override {
152 ScopedBinding<> p(varying, op->name);
153 return IRMutator::visit(op);
154 }
155
156public:
157 using IRMutator::mutate;
158
159 Expr mutate(const Expr &e) override {
160 if (should_lift(e)) {
161 // Lift it in canonical form
162 Expr lifted_expr = simplify(e);
163 auto it = lifted.find(lifted_expr);
164 if (it == lifted.end()) {
165 string name = unique_name('t');
166 lifted[lifted_expr] = name;
167 return Variable::make(e.type(), name);
168 } else {
169 return Variable::make(e.type(), it->second);
170 }
171 } else {
172 return IRMutator::mutate(e);
173 }
174 }
175
176 map<Expr, string, IRDeepCompare> lifted;
177};

179// The pass above can lift out the value of lets entirely, leaving
180// them as just renamings of other variables. Easier to substitute
181// them in as a post-pass rather than make the pass above more clever.
182class SubstituteTrivialLets : public IRMutator {
183 using IRMutator::visit;
184
185 Expr visit(const Let *op) override {
186 if (op->value.as<Variable>()) {
187 return mutate(substitute(op->name, op->value, op->body));
188 } else {
189 return IRMutator::visit(op);
190 }
191 }
192
193 Stmt visit(const LetStmt *op) override {
194 if (op->value.as<Variable>()) {
195 return mutate(substitute(op->name, op->value, op->body));
196 } else {
197 return IRMutator::visit(op);
198 }

202class LICM : public IRMutator {
203 using IRMutator::visit;
204
205 bool in_gpu_loop{false};
206
207 // Compute the cost of computing an expression inside the inner
208 // loop, compared to just loading it as a parameter.
209 int cost(const Expr &e, const set<string> &vars) {
210 if (is_const(e)) {
211 return 0;
212 } else if (const Variable *var = e.as<Variable>()) {
213 if (vars.count(var->name)) {
214 // We're loading this already
215 return 0;
216 } else {
217 // Would have to load this
218 return 1;
219 }
220 } else if (const Add *add = e.as<Add>()) {
221 return cost(add->a, vars) + cost(add->b, vars) + 1;
222 } else if (const Sub *sub = e.as<Sub>()) {
223 return cost(sub->a, vars) + cost(sub->b, vars) + 1;
224 } else if (const Mul *mul = e.as<Mul>()) {
225 return cost(mul->a, vars) + cost(mul->b, vars) + 1;
226 } else if (const Call *call = e.as<Call>()) {
227 if (call->is_intrinsic(Call::reinterpret)) {
228 internal_assert(call->args.size() == 1);
229 return cost(call->args[0], vars);
230 } else {
231 return 100;
232 }
233 } else {
234 return 100;
235 }
236 }
237
238 Stmt visit(const For *op) override {
239 ScopedValue<bool> old_in_gpu_loop(in_gpu_loop);
240 in_gpu_loop =
241 (op->for_type == ForType::GPUBlock ||
242 op->for_type == ForType::GPUThread);
243
244 if (old_in_gpu_loop && in_gpu_loop) {
245 // Don't lift lets to in-between gpu blocks/threads
246 return IRMutator::visit(op);
247 } else {
248
249 // Lift invariants
250 LiftLoopInvariants lifter;
251 Stmt new_stmt = lifter.mutate(op);
252 new_stmt = SubstituteTrivialLets().mutate(new_stmt);
253
254 // As an optimization to reduce register pressure, take
255 // the set of expressions to lift and check if any can
256 // cheaply be computed from others. If so it's better to
257 // do that than to load multiple related values off the
258 // stack. We currently only consider expressions that are
259 // the sum, difference, or product of two variables
260 // already used in the kernel, or a variable plus a
261 // constant.
262
263 // Linearize all the exprs and names
264 vector<Expr> exprs;
265 vector<string> names;
266 for (const auto &p : lifter.lifted) {
267 exprs.push_back(p.first);
268 names.push_back(p.second);
269 }
270
271 // Jointly CSE the lifted exprs put putting them together into a dummy Expr
272 Expr dummy_call = Call::make(Int(32), Call::bundle, exprs, Call::PureIntrinsic);
273 dummy_call = common_subexpression_elimination(dummy_call, true);
274
275 // Peel off containing lets. These will be lifted.
276 vector<pair<string, Expr>> lets;
277 while (const Let *let = dummy_call.as<Let>()) {
278 lets.emplace_back(let->name, let->value);
279 dummy_call = let->body;
280 }

282 // Track the set of variables used by the inner loop
283 class CollectVars : public IRVisitor {
284 using IRVisitor::visit;
285 void visit(const Variable *op) override {
286 vars.insert(op->name);
287 }
288
289 public:
290 set<string> vars;
291 } vars;
292 new_stmt.accept(&vars);
293
294 // Now consider substituting back in each use
295 const Call *call = dummy_call.as<Call>();
296 internal_assert(call->is_intrinsic(Call::bundle));
297 bool converged;
298 do {
299 converged = true;
300 for (size_t i = 0; i < exprs.size(); i++) {
301 if (!exprs[i].defined()) {
302 continue;
303 }
304 Expr e = call->args[i];
305 if (cost(e, vars.vars) <= 1) {
306 // Just subs it back in - computing it is as cheap
307 // as loading it.
308 e.accept(&vars);
309 new_stmt = substitute(names[i], e, new_stmt);
310 names[i].clear();
311 exprs[i] = Expr();
312 converged = false;
313 } else {
314 exprs[i] = e;
315 }
316 }
317 } while (!converged);
318
319 // Recurse
320 const For *loop = new_stmt.as<For>();
321 internal_assert(loop);
322
323 new_stmt = For::make(loop->name, loop->min, loop->extent,
324 loop->for_type, loop->device_api, mutate(loop->body));
325
326 // Wrap lets for the lifted invariants
327 for (size_t i = 0; i < exprs.size(); i++) {
328 if (exprs[i].defined()) {
329 new_stmt = LetStmt::make(names[i], exprs[i], new_stmt);
330 }
331 }
332
333 // Wrap the lets pulled out by CSE
334 while (!lets.empty()) {
335 new_stmt = LetStmt::make(lets.back().first, lets.back().second, new_stmt);
336 lets.pop_back();
337 }
338
339 return new_stmt;
340 }
341 }

344// Reassociate summations to group together the loop invariants. Useful to run before LICM.
345class GroupLoopInvariants : public IRMutator {
346 using IRMutator::visit;
347
348 Scope<int> var_depth;
349
350 class ExprDepth : public IRVisitor {
351 using IRVisitor::visit;
352 const Scope<int> &depth;
353
354 void visit(const Variable *op) override {
355 if (depth.contains(op->name)) {
356 result = std::max(result, depth.get(op->name));
357 }
358 }
359
360 public:
361 int result = 0;
362 ExprDepth(const Scope<int> &var_depth)
363 : depth(var_depth) {
364 }
365 };
366
367 int expr_depth(const Expr &e) {
368 ExprDepth depth(var_depth);
369 e.accept(&depth);
370 return depth.result;
371 }
372
373 struct Term {
374 Expr expr;
375 bool positive;
376 int depth;
377 };
378
379 vector<Term> extract_summation(const Expr &e) {
380 vector<Term> pending, terms;
381 pending.push_back({e, true, 0});
382 while (!pending.empty()) {
383 Term next = pending.back();
384 pending.pop_back();
385 const Add *add = next.expr.as<Add>();
386 const Sub *sub = next.expr.as<Sub>();
387 if (add) {
388 pending.push_back({add->a, next.positive, 0});
389 pending.push_back({add->b, next.positive, 0});
390 } else if (sub) {
391 pending.push_back({sub->a, next.positive, 0});
392 pending.push_back({sub->b, !next.positive, 0});
393 } else {
394 next.expr = mutate(next.expr);
395 if (next.expr.as<Add>() || next.expr.as<Sub>()) {
396 // After mutation it became an add or sub, throw it back on the pending queue.
397 pending.push_back(next);
398 } else {
399 next.depth = expr_depth(next.expr);
400 terms.push_back(next);
401 }
402 }
403 }
404
405 // Sort the terms by loop depth. Terms of equal depth are
406 // likely already in a good order, so don't mess with them.
407 std::stable_sort(terms.begin(), terms.end(),
408 [](const Term &a, const Term &b) {
409 return a.depth > b.depth;
410 });
411
412 return terms;
413 }
414
415 Expr reassociate_summation(const Expr &e) {
416 vector<Term> terms = extract_summation(e);
417
418 Expr result;
419 bool positive = true;
420 while (!terms.empty()) {
421 Term next = terms.back();
422 terms.pop_back();
423 if (result.defined()) {
424 if (next.positive == positive) {
425 result += next.expr;
426 } else if (next.positive) {
427 result = next.expr - result;
428 positive = true;
429 } else {
430 result -= next.expr;
431 }
432 } else {
433 result = next.expr;
434 positive = next.positive;
435 }
436 }
437
438 if (!positive) {
439 result = make_zero(result.type()) - result;
440 }
441
442 return result;
443 }
444
445 Expr visit(const Add *op) override {
446 if (op->type.is_float() || (op->type == Int(32) && is_const(op->b))) {
447 // Don't reassociate float exprs. (If strict_float is
448 // off, we're allowed to reassociate, and we do
449 // reassociate elsewhere, but there's no benefit to it
450 // here and it's friendlier not to.)
451 //
452 // Also don't reassociate trailing integer constants. They're the
453 // ultimate loop invariant, but doing this to stencils
454 // causes inner loops to track N different pointers
455 // instead of one pointer with constant offsets, and that
456 // complicates aliasing analysis.
457 return IRMutator::visit(op);
458 }
459
460 return reassociate_summation(op);
461 }
462
463 Expr visit(const Sub *op) override {
464 if (op->type.is_float() || (op->type == Int(32) && is_const(op->b))) {
465 return IRMutator::visit(op);
466 }
467
468 return reassociate_summation(op);
469 }
470
471 int depth = 0;
472
473 Stmt visit(const For *op) override {
474 depth++;
475 ScopedBinding<int> bind(var_depth, op->name, depth);
476 Stmt stmt = IRMutator::visit(op);
477 depth--;
478 return stmt;
479 }

481 template<typename T, typename Body>
482 Body visit_let(const T *op) {
483 struct Frame {
484 const T *op;
485 Expr new_value;
486 ScopedBinding<int> binding;
487 Frame(const T *op, Expr v, int depth, Scope<int> &scope)
488 : op(op),
489 new_value(std::move(v)),
490 binding(scope, op->name, depth) {
491 }
492 };
493 std::vector<Frame> frames;
494 Body result;
495
496 do {
497 result = op->body;
498 int d = 0;
499 if (depth > 0) {
500 d = expr_depth(op->value);
501 }
502 frames.emplace_back(op, mutate(op->value), d, var_depth);
503 } while ((op = result.template as<T>()));
504
505 result = mutate(result);
506
507 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
508 if (it->new_value.same_as(it->op->value) && result.same_as(it->op->body)) {
509 result = it->op;
510 } else {
511 result = T::make(it->op->name, it->new_value, result);
512 }
513 }
514
515 return result;
516 }
517
518 Expr visit(const Let *op) override {
519 return visit_let<Let, Expr>(op);
520 }
521
522 Stmt visit(const LetStmt *op) override {
523 return visit_let<LetStmt, Stmt>(op);
524 }

539// Move IfThenElse nodes from the inside of a piece of Stmt IR to the
540// outside when legal.
541class HoistIfStatements : public IRMutator {
542 using IRMutator::visit;
543
544 Stmt visit(const LetStmt *op) override {
545 Stmt body = mutate(op->body);
546 if (const IfThenElse *i = body.as<IfThenElse>()) {
547 if (!i->else_case.defined() &&
548 is_pure(op->value) &&
549 is_pure(i->condition) &&
550 !expr_uses_var(i->condition, op->name)) {
551 Stmt s = LetStmt::make(op->name, op->value, i->then_case);
552 return IfThenElse::make(i->condition, s);
553 }
554 }
555 if (body.same_as(op->body)) {
556 return op;
557 } else {
558 return LetStmt::make(op->name, op->value, body);
559 }
560 }
561
562 Stmt visit(const For *op) override {
563 Stmt body = mutate(op->body);
564 if (const IfThenElse *i = body.as<IfThenElse>()) {
565 if (!i->else_case.defined() &&
566 is_pure(i->condition) &&
567 !expr_uses_var(i->condition, op->name)) {
568 Stmt s = For::make(op->name, op->min, op->extent,
569 op->for_type, op->device_api, i->then_case);
570 return IfThenElse::make(i->condition, s);
571 }
572 }
573 if (body.same_as(op->body)) {
574 return op;
575 } else {
576 return For::make(op->name, op->min, op->extent,
577 op->for_type, op->device_api, body);
578 }
579 }
580
581 Stmt visit(const ProducerConsumer *op) override {
582 Stmt body = mutate(op->body);
583 if (const IfThenElse *i = body.as<IfThenElse>()) {
584 if (!i->else_case.defined() &&
585 is_pure(i->condition)) {
586 Stmt s = ProducerConsumer::make(op->name, op->is_producer, i->then_case);
587 return IfThenElse::make(i->condition, s);
588 }
589 }
590 if (body.same_as(op->body)) {
591 return op;
592 } else {
593 return ProducerConsumer::make(op->name, op->is_producer, body);
594 }
595 }
596
597 Stmt visit(const IfThenElse *op) override {
598 Stmt then_case = mutate(op->then_case);
599 if (!op->else_case.defined() &&
600 is_pure(op->condition)) {
601 if (const IfThenElse *i = then_case.as<IfThenElse>()) {
602 if (!i->else_case.defined() &&
603 is_pure(i->condition)) {
604 return IfThenElse::make(op->condition && i->condition, then_case);
605 }
606 }
607 }
608 Stmt else_case = mutate(op->else_case);
609 if (then_case.same_as(op->then_case) && else_case.same_as(op->else_case)) {
610 return op;
611 } else {
612 return IfThenElse::make(op->condition, then_case, else_case);
613 }
614 }
615
616 Stmt visit(const Allocate *op) override {
617 Stmt body = mutate(op->body);
618 if (const IfThenElse *i = body.as<IfThenElse>()) {
619 if (!i->else_case.defined() &&
620 is_pure(i->condition)) {
621 Stmt s = Allocate::make(op->name, op->type, op->memory_type,
622 op->extents, op->condition, i->then_case,
623 op->new_expr, op->free_function);
624 return IfThenElse::make(i->condition, s);
625 }
626 }
627 if (body.same_as(op->body)) {
628 return op;
629 } else {
630 return Allocate::make(op->name, op->type, op->memory_type,
631 op->extents, op->condition, body,
632 op->new_expr, op->free_function);
633 }
634 }
635
636 Stmt visit(const Block *op) override {
637 Stmt first = mutate(op->first);
638 Stmt rest = mutate(op->rest);
639
640 const IfThenElse *i1 = first.as<IfThenElse>();
641 const Block *b = rest.as<Block>();
642 const IfThenElse *i2 = b ? b->first.as<IfThenElse>() : rest.as<IfThenElse>();
643
644 if (i1 &&
645 i2 &&
646 !i1->else_case.defined() &&
647 !i2->else_case.defined() &&
648 is_pure(i1->condition) &&
649 can_prove(i1->condition == i2->condition)) {
650 Stmt s = Block::make(i1->then_case, i2->then_case);
651 s = IfThenElse::make(i1->condition, s);
652 if (b) {
653 s = Block::make(s, b->rest);
654 }
655 return s;
656 } else if (first.same_as(op->first) && rest.same_as(op->rest)) {
657 return op;
658 } else {
659 return Block::make(first, rest);
660 }
661 }
662};
