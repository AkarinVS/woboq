15/* Find all the internal halide calls in an expr */
16class FindCalls : public IRVisitor {
17public:
18 map<string, Function> calls;
19
20 using IRVisitor::visit;
21
22 void include_function(const Function &f) {
23 map<string, Function>::iterator iter = calls.find(f.name());
24 if (iter == calls.end()) {
25 calls[f.name()] = f;
26 } else {
27 user_assert(iter->second.same_as(f))
28 << "Can't compile a pipeline using multiple functions with same name: "
29 << f.name() << "\n";
30 }
31 }
32
33 void visit(const Call *call) override {
34 IRVisitor::visit(call);
35
36 if (call->call_type == Call::Halide && call->func.defined()) {
37 Function f(call->func);
38 include_function(f);
39 }
40 }
41};

43void populate_environment_helper(Function f, map<string, Function> &env,
44 bool recursive = true, bool include_wrappers = false) {
45 map<string, Function>::const_iterator iter = env.find(f.name());
46 if (iter != env.end()) {
47 user_assert(iter->second.same_as(f))
48 << "Can't compile a pipeline using multiple functions with same name: "
49 << f.name() << "\n";
50 return;
51 }
52
53 FindCalls calls;
54 f.accept(&calls);
55 if (f.has_extern_definition()) {
56 for (const ExternFuncArgument &arg : f.extern_arguments()) {
57 if (arg.is_func()) {
58 Function g(arg.func);
59 calls.calls[g.name()] = g;
60 }
61 }
62 }
63
64 if (include_wrappers) {
65 for (const auto &it : f.schedule().wrappers()) {
66 Function g(it.second);
67 calls.calls[g.name()] = g;
68 }
69 }
70
71 if (!recursive) {
72 env.insert(calls.calls.begin(), calls.calls.end());
73 } else {
74 env[f.name()] = f;
75
76 for (const auto &i : calls.calls) {
77 populate_environment_helper(i.second, env, recursive, include_wrappers);
78 }
79 }
80}
81
82} // namespace

84void populate_environment(Function f, map<string, Function> &env) {
85 populate_environment_helper(std::move(f), env, true, true);
86}

88map<string, Function> find_transitive_calls(Function f) {
89 map<string, Function> res;
