136void CodeGen_PTX_Dev::add_kernel(Stmt stmt,
137 const std::string &name,
138 const std::vector<DeviceArgument> &args) {
139 internal_assert(module != nullptr);
140
141 debug(2) << "In CodeGen_PTX_Dev::add_kernel\n";
142
143 // Now deduce the types of the arguments to our function
144 vector<llvm::Type *> arg_types(args.size());
145 for (size_t i = 0; i < args.size(); i++) {
146 if (args[i].is_buffer) {
147 arg_types[i] = llvm_type_of(UInt(8))->getPointerTo();
148 } else {
149 arg_types[i] = llvm_type_of(args[i].type);
150 }
151 }
152
153 // Make our function
154 FunctionType *func_t = FunctionType::get(void_t, arg_types, false);
155 function = llvm::Function::Create(func_t, llvm::Function::ExternalLinkage, name, module.get());
156 set_function_attributes_for_target(function, target);
157
158 // Mark the buffer args as no alias
159 for (size_t i = 0; i < args.size(); i++) {
160 if (args[i].is_buffer) {
161 function->addParamAttr(i, Attribute::NoAlias);
162 }
163 }
164
165 // Make the initial basic block
166 entry_block = BasicBlock::Create(*context, "entry", function);
167 builder->SetInsertPoint(entry_block);
168
169 // Put the arguments in the symbol table
170 vector<string> arg_sym_names;
171 {
172 size_t i = 0;
173 for (auto &fn_arg : function->args()) {
174
175 string arg_sym_name = args[i].name;
176 sym_push(arg_sym_name, &fn_arg);
177 fn_arg.setName(arg_sym_name);
178 arg_sym_names.push_back(arg_sym_name);
179
180 i++;
181 }
182 }
183
184 // We won't end the entry block yet, because we'll want to add
185 // some allocas to it later if there are local allocations. Start
186 // a new block to put all the code.
187 BasicBlock *body_block = BasicBlock::Create(*context, "body", function);
188 builder->SetInsertPoint(body_block);
189
190 debug(1) << "Generating llvm bitcode for kernel...\n";
191 // Ok, we have a module, function, context, and a builder
192 // pointing at a brand new basic block. We're good to go.
193 stmt.accept(this);
194
195 // Now we need to end the function
196 builder->CreateRetVoid();
197
198 // Make the entry block point to the body block
199 builder->SetInsertPoint(entry_block);
200 builder->CreateBr(body_block);
201
202 // Add the nvvm annotation that it is a kernel function.
203 llvm::Metadata *md_args[] = {
204 llvm::ValueAsMetadata::get(function),
205 MDString::get(*context, "kernel"),
206 llvm::ValueAsMetadata::get(ConstantInt::get(i32_t, 1))};
207
208 MDNode *md_node = MDNode::get(*context, md_args);
209
210 module->getOrInsertNamedMetadata("nvvm.annotations")->addOperand(md_node);
211
212 // Now verify the function is ok
213 verifyFunction(*function);
214
215 // Finally, verify the module is ok
216 verifyModule(*module);
217
218 debug(2) << "Done generating llvm bitcode for PTX\n";
219
220 // Clear the symbol table
221 for (size_t i = 0; i < arg_sym_names.size(); i++) {
222 sym_pop(arg_sym_names[i]);
223 }
224}

256void CodeGen_PTX_Dev::visit(const Call *op) {
257 if (op->is_intrinsic(Call::gpu_thread_barrier)) {
258 // Even though we always insert a __syncthreads equivalent
259 // (which has both a device and shared memory fence)
260 // check to make sure the intrinsic has the right number of
261 // arguments
262 internal_assert(op->args.size() == 1) << "gpu_thread_barrier() intrinsic must specify memory fence type.\n";
263
264 const auto *fence_type_ptr = as_const_int(op->args[0]);
265 internal_assert(fence_type_ptr) << "gpu_thread_barrier() parameter is not a constant integer.\n";
266
267 llvm::Function *barrier0 = module->getFunction("llvm.nvvm.barrier0");
268 internal_assert(barrier0) << "Could not find PTX barrier intrinsic (llvm.nvvm.barrier0)\n";
269 builder->CreateCall(barrier0);
270 value = ConstantInt::get(i32_t, 0);
271 } else if (op->name == "dp2a" || op->name == "dp4a") {
272 // TODO: It would be better if CodeGen_LLVM could handle overloaded intrin calls by default.
273 value = call_overloaded_intrin(op->type, op->name, op->args);
274 internal_assert(value) << Expr(op) << "\n";
275 } else {
276 CodeGen_LLVM::visit(op);
277 }
278}

303 if (is_gpu_var(loop->name)) {

314void CodeGen_PTX_Dev::visit(const Allocate *alloc) {
315 user_assert(!alloc->new_expr.defined()) << "Allocate node inside PTX kernel has custom new expression.\n"
316 << "(Memoization is not supported inside GPU kernels at present.)\n";
317 if (alloc->memory_type == MemoryType::GPUShared) {
318 // PTX uses zero in address space 3 as the base address for shared memory
319 Value *shared_base = Constant::getNullValue(PointerType::get(i8_t, 3));
320 sym_push(alloc->name, shared_base);
321 } else {
322 debug(2) << "Allocate " << alloc->name << " on device\n";
323
324 string allocation_name = alloc->name;
325 debug(3) << "Pushing allocation called " << allocation_name << " onto the symbol table\n";
326
327 // Jump back to the entry and generate an alloca. Note that by
328 // jumping back we're rendering any expression we carry back
329 // meaningless, so we had better only be dealing with
330 // constants here.
331 int32_t size = alloc->constant_allocation_size();
332 internal_assert(size > 0)
333 << "Allocation " << alloc->name << " has a dynamic size. "
334 << "This should have been moved to the heap by the "
335 << "fuse_gpu_thread_loops lowering pass.\n";
336
337 BasicBlock *here = builder->GetInsertBlock();
338
339 builder->SetInsertPoint(entry_block);
340 Value *ptr = builder->CreateAlloca(llvm_type_of(alloc->type), ConstantInt::get(i32_t, size));
341 builder->SetInsertPoint(here);
342 sym_push(allocation_name, ptr);
343 }
344 codegen(alloc->body);
345}

357void CodeGen_PTX_Dev::visit(const Load *op) {
358
359 // Do aligned 4-wide 32-bit loads as a single i128 load.
360 const Ramp *r = op->index.as<Ramp>();
361 // TODO: lanes >= 4, not lanes == 4
362 if (is_const_one(op->predicate) && r && is_const_one(r->stride) && r->lanes == 4 && op->type.bits() == 32) {
363 ModulusRemainder align = op->alignment;
364 if (align.modulus % 4 == 0 && align.remainder % 4 == 0) {
365 Expr index = simplify(r->base / 4);
366 Expr equiv = Load::make(UInt(128), op->name, index,
367 op->image, op->param, const_true(), align / 4);
368 equiv = reinterpret(op->type, equiv);
369 codegen(equiv);
370 return;
371 }
372 }
373
374 CodeGen_LLVM::visit(op);
375}

377void CodeGen_PTX_Dev::visit(const Store *op) {
378 // Issue atomic store if we are inside an Atomic node.
379 if (emit_atomic_stores) {
380 user_assert(is_const_one(op->predicate)) << "Atomic update does not support predicated store.\n";
381 user_assert(op->value.type().bits() >= 32) << "CUDA: 8-bit or 16-bit atomics are not supported.\n";
382 }
383
384 // Do aligned 4-wide 32-bit stores as a single i128 store.
385 const Ramp *r = op->index.as<Ramp>();
386 // TODO: lanes >= 4, not lanes == 4
387 if (is_const_one(op->predicate) && r && is_const_one(r->stride) && r->lanes == 4 && op->value.type().bits() == 32) {
388 ModulusRemainder align = op->alignment;
389 if (align.modulus % 4 == 0 && align.remainder % 4 == 0) {
390 Expr index = simplify(r->base / 4);
391 Expr value = reinterpret(UInt(128), op->value);
392 Stmt equiv = Store::make(op->name, value, index, op->param, const_true(), align / 4);
393 codegen(equiv);
394 return;
395 }
396 }

398 CodeGen_LLVM::visit(op);

412// The NVPTX backend generates really terrible code if loads aren't 32-bit. This
413// mutator replaces 8- or 16-bit loads aligned to 32-bits with 32-bit loads of fewer
414// lanes instead.
415class RewriteLoadsAs32Bit : public IRMutator {
416 using IRMutator::visit;
417
418 Expr visit(const Load *op) override {
419 if (op->type.is_scalar() || op->type.bits() * op->type.lanes() < 32) {
420 return IRMutator::visit(op);
421 }
422
423 Expr index = mutate(op->index);
424 int sub_lanes = 32 / op->type.bits();
425 const Ramp *idx = index.as<Ramp>();
426 if (idx &&
427 is_const_one(op->predicate) &&
428 is_const_one(idx->stride) &&
429 op->alignment.modulus % sub_lanes == 0 &&
430 op->alignment.remainder % sub_lanes == 0) {
431 Expr new_idx = simplify(idx->base / sub_lanes);
432 int load_lanes = op->type.lanes() / sub_lanes;
433 if (op->type.lanes() > sub_lanes) {
434 new_idx = Ramp::make(new_idx, 1, load_lanes);
435 }
436 Expr new_load = Load::make(Int(32, load_lanes), op->name, new_idx, op->image, op->param, const_true(load_lanes), op->alignment / sub_lanes);
437 return reinterpret(op->type, new_load);
438 } else if (index.same_as(op->index)) {
439 return op;
440 } else {
441 return Load::make(op->type, op->name, op->index, op->image, op->param, op->predicate, op->alignment);
442 }
443 }

446void CodeGen_PTX_Dev::codegen_vector_reduce(const VectorReduce *op, const Expr &init) {
447 // Pattern match 8/16-bit dot products
448 struct Pattern {
449 VectorReduce::Operator op;
450 int factor;
451 Expr pattern;
452 const char *name;
453 int flags;
454 enum {
455 SwapOps = 1 << 0, // This happens before narrowing op 1 below.
456 NarrowOp1 = 1 << 1,
457 };
458 };
459 static Expr wild_i8x = Variable::make(Int(8, 0), "*");
460 static Expr wild_u8x = Variable::make(UInt(8, 0), "*");
461 static Expr wild_i16x = Variable::make(Int(16, 0), "*");
462 static Expr wild_u16x = Variable::make(UInt(16, 0), "*");
463 // TODO: Support rewriting to arbitrary calls in IRMatch and use that instead
464 // of expr_match here. That would probably allow avoiding the redundant swapping
465 // operands logic.
466 // clang-format off
467 static const Pattern patterns[] = {
468 {VectorReduce::Add, 4, i32(widening_mul(wild_i8x, wild_i8x)), "dp4a"},
469 {VectorReduce::Add, 4, i32(widening_mul(wild_i8x, wild_u8x)), "dp4a"},
470 {VectorReduce::Add, 4, i32(widening_mul(wild_u8x, wild_i8x)), "dp4a"},
471 {VectorReduce::Add, 4, u32(widening_mul(wild_u8x, wild_u8x)), "dp4a"},
472 {VectorReduce::Add, 4, widening_mul(wild_i16x, wild_i16x), "dp2a", Pattern::NarrowOp1},
473 {VectorReduce::Add, 4, widening_mul(wild_i16x, wild_u16x), "dp2a", Pattern::NarrowOp1},
474 {VectorReduce::Add, 4, widening_mul(wild_u16x, wild_i16x), "dp2a", Pattern::NarrowOp1},
475 {VectorReduce::Add, 4, widening_mul(wild_u16x, wild_u16x), "dp2a", Pattern::NarrowOp1},
476 {VectorReduce::Add, 4, widening_mul(wild_i16x, wild_i16x), "dp2a", Pattern::SwapOps | Pattern::NarrowOp1},
477 {VectorReduce::Add, 4, widening_mul(wild_u16x, wild_i16x), "dp2a", Pattern::SwapOps | Pattern::NarrowOp1},
478 {VectorReduce::Add, 4, widening_mul(wild_i16x, wild_u16x), "dp2a", Pattern::SwapOps | Pattern::NarrowOp1},
479 {VectorReduce::Add, 4, widening_mul(wild_u16x, wild_u16x), "dp2a", Pattern::SwapOps | Pattern::NarrowOp1},
480 };
481 // clang-format on
482
483 const int input_lanes = op->value.type().lanes();
484 const int factor = input_lanes / op->type.lanes();
485
486 std::vector<Expr> matches;
487 for (const Pattern &p : patterns) {
488 if (p.op != op->op || factor % p.factor != 0) {
489 continue;
490 }
491 if (!expr_match(p.pattern, op->value, matches)) {
492 continue;
493 }
494 Expr a = matches[0];
495 Expr b = matches[1];
496 if (p.flags & Pattern::SwapOps) {
497 std::swap(a, b);
498 }
499 if (p.flags & Pattern::NarrowOp1) {
500 // This pattern needs the second operand to be narrowed further.
501 Expr b_narrow = lossless_cast(b.type().narrow(), b);
502 if (!b_narrow.defined()) {
503 b_narrow = lossless_cast(b.type().narrow().with_code(halide_type_uint), b);
504 if (!b_narrow.defined()) {
505 continue;
506 }
507 }
508 b = b_narrow;
509 }
510 Expr i = init;
511 if (!i.defined()) {
512 i = cast(op->value.type(), 0);
513 }
514
515 vector<Expr> result;
516 for (int l = 0; l < op->type.lanes(); l++) {
517 // To compute a single lane of the output, we'll
518 // extract the appropriate slice of the args, which
519 // have been reinterpreted as 32-bit vectors, then
520 // call either dp4a or dp2a the appropriate number of
521 // times, and finally sum the result.
522 Expr i_slice = Shuffle::make_extract_element(i, l);
523 for (int i = 0; i < factor; i += p.factor) {
524 Expr a_slice = Shuffle::make_slice(a, i + l * factor, 1, p.factor);
525 Expr b_slice = Shuffle::make_slice(b, i + l * factor, 1, p.factor);
526 i_slice = Call::make(i_slice.type(), p.name, {a_slice, b_slice, i_slice}, Call::PureExtern);
527 }
528 i_slice = RewriteLoadsAs32Bit().mutate(i_slice);
529 i_slice = simplify(i_slice);
530 i_slice = common_subexpression_elimination(i_slice);
531 result.push_back(i_slice);
532 }
533 // Concatenate the per-lane results to get the full vector result
534 Expr equiv = Shuffle::make_concat(result);
535 equiv.accept(this);
536 return;
537 }
538 CodeGen_LLVM::codegen_vector_reduce(op, init);
539}

590 debug(2) << "In CodeGen_PTX_Dev::compile_to_src";
591
592 // DISABLED - hooked in here to force PrintBeforeAll option - seems to be the only way?
593 /*char* argv[] = { "llc", "-print-before-all" };*/
594 /*int argc = sizeof(argv)/sizeof(char*);*/
595 /*cl::ParseCommandLineOptions(argc, argv, "Halide PTX internal compiler\n");*/
596
597 llvm::Triple triple(module->getTargetTriple());
598
599 // Allocate target machine
600
601 std::string err_str;
602 const llvm::Target *llvm_target = TargetRegistry::lookupTarget(triple.str(), err_str);
603 internal_assert(llvm_target) << err_str << "\n";
604
605 TargetOptions options;
606#if LLVM_VERSION < 120
607 options.PrintMachineCode = false;
608#endif
609 options.AllowFPOpFusion = FPOpFusion::Fast;
610 options.UnsafeFPMath = true;
611 options.NoInfsFPMath = true;
612 options.NoNaNsFPMath = true;
613 options.HonorSignDependentRoundingFPMathOption = false;
614 options.NoZerosInBSS = false;
615 options.GuaranteedTailCallOpt = false;
616 options.StackAlignmentOverride = 0;
617
618 std::unique_ptr<TargetMachine>
619 target_machine(llvm_target->createTargetMachine(triple.str(),
620 mcpu(), mattrs(), options,
621 llvm::Reloc::PIC_,
622 llvm::CodeModel::Small,
623 CodeGenOpt::Aggressive));
624
625 internal_assert(target_machine.get()) << "Could not allocate target machine!";
626
627 module->setDataLayout(target_machine->createDataLayout());
628
629 // Set up passes
630 llvm::SmallString<8> outstr;
631 raw_svector_ostream ostream(outstr);
632 ostream.SetUnbuffered();
633
634 legacy::FunctionPassManager function_pass_manager(module.get());
635 legacy::PassManager module_pass_manager;
636
637 module_pass_manager.add(createTargetTransformInfoWrapperPass(target_machine->getTargetIRAnalysis()));
638 function_pass_manager.add(createTargetTransformInfoWrapperPass(target_machine->getTargetIRAnalysis()));
639
640 // NVidia's libdevice library uses a __nvvm_reflect to choose
641 // how to handle denormalized numbers. (The pass replaces calls
642 // to __nvvm_reflect with a constant via a map lookup. The inliner
643 // pass then resolves these situations to fast code, often a single
644 // instruction per decision point.)
645 //
646 // The default is (more) IEEE like handling. FTZ mode flushes them
647 // to zero. (This may only apply to single-precision.)
648 //
649 // The libdevice documentation covers other options for math accuracy
650 // such as replacing division with multiply by the reciprocal and
651 // use of fused-multiply-add, but they do not seem to be controlled
652 // by this __nvvvm_reflect mechanism and may be flags to earlier compiler
653 // passes.
654 const int kFTZDenorms = 1;
655
656 // Insert a module flag for the FTZ handling.
657 module->addModuleFlag(llvm::Module::Override, "nvvm-reflect-ftz",
658 kFTZDenorms);
659
660 if (kFTZDenorms) {
661 for (llvm::Function &fn : *module) {
662 fn.addFnAttr("nvptx-f32ftz", "true");
663 }
664 }
665
666 // At present, we default to *enabling* LLVM loop optimization,
667 // unless DisableLLVMLoopOpt is set; we're going to flip this to defaulting
668 // to *not* enabling these optimizations (and removing the DisableLLVMLoopOpt feature).
669 // See https://github.com/halide/Halide/issues/4113 for more info.
670 // (Note that setting EnableLLVMLoopOpt always enables loop opt, regardless
671 // of the setting of DisableLLVMLoopOpt.)
672 const bool do_loop_opt = !target.has_feature(Target::DisableLLVMLoopOpt) ||
673 target.has_feature(Target::EnableLLVMLoopOpt);
674
675 PassManagerBuilder b;
676 b.OptLevel = 3;
677 b.Inliner = createFunctionInliningPass(b.OptLevel, 0, false);
678 b.LoopVectorize = do_loop_opt;
679 b.SLPVectorize = true;
680 b.DisableUnrollLoops = !do_loop_opt;
681
682 target_machine->adjustPassManager(b);
683
684 b.populateFunctionPassManager(function_pass_manager);
685 b.populateModulePassManager(module_pass_manager);
686
687 // Override default to generate verbose assembly.
688 target_machine->Options.MCOptions.AsmVerbose = true;
689
690 // Output string stream
691
692 // Ask the target to add backend passes as necessary.
693 bool fail = target_machine->addPassesToEmitFile(module_pass_manager, ostream, nullptr,
694 ::llvm::CGFT_AssemblyFile,
695 true);
696 if (fail) {
697 internal_error << "Failed to set up passes to emit PTX source\n";
698 }
699
700 // Run optimization passes
701 function_pass_manager.doInitialization();
702 for (llvm::Module::iterator i = module->begin(); i != module->end(); i++) {
703 function_pass_manager.run(*i);
704 }
705 function_pass_manager.doFinalization();
706 module_pass_manager.run(*module);
707
708 if (debug::debug_level() >= 2) {
709 dump();
710 }
711 debug(2) << "Done with CodeGen_PTX_Dev::compile_to_src";
712
713 debug(1) << "PTX kernel:\n"
714 << outstr.c_str() << "\n";
715
716 vector<char> buffer(outstr.begin(), outstr.end());
717
718 // Dump the SASS too if the cuda SDK is in the path
719 if (debug::debug_level() >= 2) {
720 debug(2) << "Compiling PTX to SASS. Will fail if CUDA SDK is not installed (and in the path).\n";
721
722 TemporaryFile ptx(get_current_kernel_name(), ".ptx");
723 TemporaryFile sass(get_current_kernel_name(), ".sass");
724
725 std::ofstream f(ptx.pathname());
726 f.write(buffer.data(), buffer.size());
727 f.close();
728
729 string cmd = "ptxas --gpu-name " + mcpu() + " " + ptx.pathname() + " -o " + sass.pathname();
730 if (system(cmd.c_str()) == 0) {
731 cmd = "nvdisasm " + sass.pathname();
732 int ret = system(cmd.c_str());
733 (void)ret; // Don't care if it fails
734 }
735
736 // Note: It works to embed the contents of the .sass file in
737 // the buffer instead of the ptx source, and this could help
738 // with app startup times. Expose via the target?
739 /*
740 {
741 std::ifstream f(sass.pathname());
742 buffer.clear();
743 f.seekg(0, std::ios_base::end);
744 std::streampos sz = f.tellg();
745 buffer.resize(sz);
746 f.seekg(0, std::ios_base::beg);
747 f.read(buffer.data(), sz);
748 }
749 */
750 }
751
752 // Null-terminate the ptx source
753 buffer.push_back(0);
754 return buffer;
755}

774bool CodeGen_PTX_Dev::supports_atomic_add(const Type &t) const {
775 if (t.bits() < 32) {
776 // TODO: Half atomics are supported by compute capability 7.x or higher.
777 return false;
778 }
779 if (t.is_int_or_uint()) {
780 return true;
781 }
782 if (t.is_float() && t.bits() == 32) {
783 return true;
784 }
785 if (t.is_float() && t.bits() == 64) {
786 // double atomics are supported since CC6.1
787 return target.get_cuda_capability_lower_bound() >= 61;
788 }
789 return false;
790}
