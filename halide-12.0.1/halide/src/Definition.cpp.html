18struct DefinitionContents {
19 mutable RefCount ref_count;
20 bool is_init = true;
21 Expr predicate;
22 std::vector<Expr> values, args;
23 StageSchedule stage_schedule;
24 std::vector<Specialization> specializations;
25 std::string source_location;
26
27 DefinitionContents()
28 : predicate(const_true()) {
29 }
30
31 void accept(IRVisitor *visitor) const {
32 if (predicate.defined()) {
33 predicate.accept(visitor);
34 }
35
36 for (const Expr &val : values) {
37 val.accept(visitor);
38 }
39 for (const Expr &arg : args) {
40 arg.accept(visitor);
41 }
42
43 stage_schedule.accept(visitor);
44
45 for (const Specialization &s : specializations) {
46 if (s.condition.defined()) {
47 s.condition.accept(visitor);
48 }
49 s.definition.accept(visitor);
50 }
51 }
52
53 void mutate(IRMutator *mutator) {
54 if (predicate.defined()) {
55 predicate = mutator->mutate(predicate);
56 }
57
58 for (size_t i = 0; i < values.size(); ++i) {
59 values[i] = mutator->mutate(values[i]);
60 }
61 for (size_t i = 0; i < args.size(); ++i) {
62 args[i] = mutator->mutate(args[i]);
63 }
64
65 stage_schedule.mutate(mutator);
66
67 for (Specialization &s : specializations) {
68 if (s.condition.defined()) {
69 s.condition = mutator->mutate(s.condition);
70 }
71 s.definition.mutate(mutator);
72 }
73 }
74};

96Definition::Definition(const std::vector<Expr> &args, const std::vector<Expr> &values,
97 const ReductionDomain &rdom, bool is_init)
98 : contents(new DefinitionContents) {
99 contents->is_init = is_init;
100 contents->values = values;
101 contents->args = args;
102 contents->source_location = Introspection::get_source_location();
103 if (rdom.defined()) {
104 contents->predicate = rdom.predicate();
105 for (size_t i = 0; i < rdom.domain().size(); i++) {
106 contents->stage_schedule.rvars().push_back(rdom.domain()[i]);
107 }
108 }
109}

111Definition Definition::get_copy() const {
112 internal_assert(contents.defined()) << "Cannot copy undefined Definition\n";
113
114 Definition copy(new DefinitionContents);
115 copy.contents->is_init = contents->is_init;
116 copy.contents->predicate = contents->predicate;
117 copy.contents->values = contents->values;
118 copy.contents->args = contents->args;
119 copy.contents->stage_schedule = contents->stage_schedule.get_copy();
120 copy.contents->source_location = contents->source_location;
121
122 // Deep-copy specializations
123 for (const Specialization &s : contents->specializations) {
124 Specialization s_copy;
125 s_copy.condition = s.condition;
126 s_copy.definition = s.definition.get_copy();
127 s_copy.failure_message = s.failure_message;
128 copy.contents->specializations.push_back(std::move(s_copy));
129 }
130 return copy;
131}

199const Specialization &Definition::add_specialization(Expr condition) {
200 Specialization s;
201 s.condition = std::move(condition);
202 s.definition.contents = new DefinitionContents;
203 s.definition.contents->is_init = contents->is_init;
204 s.definition.contents->predicate = contents->predicate;
205 s.definition.contents->values = contents->values;
206 s.definition.contents->args = contents->args;
207 s.definition.contents->source_location = contents->source_location;
208
209 // The sub-schedule inherits everything about its parent except for its specializations.
210 s.definition.contents->stage_schedule = contents->stage_schedule.get_copy();
211
212 contents->specializations.push_back(s);
213 return contents->specializations.back();
214}
