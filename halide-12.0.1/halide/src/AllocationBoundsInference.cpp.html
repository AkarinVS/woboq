21// Figure out the region touched of each buffer, and deposit them as
22// let statements outside of each realize node, or at the top level if
23// they're not internal allocations.
24
25class AllocationInference : public IRMutator {
26 using IRMutator::visit;
27
28 const map<string, Function> &env;
29 const FuncValueBounds &func_bounds;
30 set<string> touched_by_extern;
31
32 Stmt visit(const Realize *op) override {
33 map<string, Function>::const_iterator iter = env.find(op->name);
34 internal_assert(iter != env.end());
35 Function f = iter->second;
36 const vector<string> f_args = f.args();
37
38 Scope<Interval> empty_scope;
39 Box b = box_touched(op->body, op->name, empty_scope, func_bounds);
40
41 Stmt new_body = mutate(op->body);
42 Stmt stmt = Realize::make(op->name, op->types, op->memory_type, op->bounds, op->condition, new_body);
43
44 // If the realization is dead and there's is no access to the
45 // buffer (e.g. because we're in a specialization), then
46 // b.size() may be zero. In this case just drop the realize
47 // node.
48 if (b.empty() && !op->bounds.empty()) {
49 return new_body;
50 }
51
52 for (size_t i = 0; i < b.size(); i++) {
53 // Get any applicable bound on this dimension
54 Bound bound;
55 for (size_t j = 0; j < f.schedule().bounds().size(); j++) {
56 Bound b = f.schedule().bounds()[j];
57 if (f_args[i] == b.var) {
58 bound = b;
59 }
60 }
61
62 string prefix = op->name + "." + f_args[i];
63 string min_name = prefix + ".min_realized";
64 string max_name = prefix + ".max_realized";
65 string extent_name = prefix + ".extent_realized";
66 if (!b[i].is_bounded()) {
67 user_error << op->name << " is accessed over an unbounded domain in dimension "
68 << f_args[i] << "\n";
69 }
70 Expr min, max, extent;
71 b[i].min = simplify(b[i].min);
72 b[i].max = simplify(b[i].max);
73 if (bound.min.defined()) {
74 min = bound.min;
75 } else {
76 min = b[i].min;
77 }
78 if (bound.extent.defined()) {
79 extent = bound.extent;
80 max = simplify(min + extent - 1);
81 } else {
82 max = b[i].max;
83 extent = simplify((max - min) + 1);
84 }
85 if (bound.modulus.defined()) {
86 if (bound.remainder.defined()) {
87 min -= bound.remainder;
88 min = (min / bound.modulus) * bound.modulus;
89 min += bound.remainder;
90 Expr max_plus_one = max + 1;
91 max_plus_one -= bound.remainder;
92 max_plus_one = ((max_plus_one + bound.modulus - 1) / bound.modulus) * bound.modulus;
93 max_plus_one += bound.remainder;
94 extent = simplify(max_plus_one - min);
95 max = max_plus_one - 1;
96 } else {
97 extent = simplify(((extent + bound.modulus - 1) / bound.modulus) * bound.modulus);
98 max = simplify(min + extent - 1);
99 }
100 }
101
102 Expr min_var = Variable::make(Int(32), min_name);
103 Expr max_var = Variable::make(Int(32), max_name);
104
105 internal_assert(min_var.type() == min.type());
106 internal_assert(max_var.type() == max.type());
107
108 Expr error_msg = Call::make(Int(32), "halide_error_explicit_bounds_too_small",
109 {f_args[i], f.name(), min_var, max_var, b[i].min, b[i].max},
110 Call::Extern);
111
112 if (bound.min.defined()) {
113 stmt = Block::make(AssertStmt::make(min_var <= b[i].min, error_msg), stmt);
114 }
115 if (bound.extent.defined()) {
116 stmt = Block::make(AssertStmt::make(max_var >= b[i].max, error_msg), stmt);
117 }
118
119 stmt = LetStmt::make(extent_name, extent, stmt);
120 stmt = LetStmt::make(min_name, min, stmt);
121 stmt = LetStmt::make(max_name, max, stmt);
122 }
123 return stmt;
124 }
125
126public:
127 AllocationInference(const map<string, Function> &e, const FuncValueBounds &fb)
128 : env(e), func_bounds(fb) {
129 // Figure out which buffers are touched by extern stages
130 for (map<string, Function>::const_iterator iter = e.begin();
131 iter != e.end(); ++iter) {
132 Function f = iter->second;
133 if (f.has_extern_definition()) {
134 touched_by_extern.insert(f.name());
135 for (size_t i = 0; i < f.extern_arguments().size(); i++) {
136 ExternFuncArgument arg = f.extern_arguments()[i];
137 if (!arg.is_func()) {
138 continue;
139 }
140 Function input(arg.func);
141 touched_by_extern.insert(input.name());
142 }
143 }
144 }
145 }
146};
147
148// We can strip box_touched declarations here. We're done with
149// them. Reconsider this decision if we want to use
150// box_touched on extern stages later in lowering. Storage
151// folding currently does box_touched too, but it handles extern
152// stages specially already.
153class StripDeclareBoxTouched : public IRMutator {
154 using IRMutator::visit;
155
156 Expr visit(const Call *op) override {
157 if (op->is_intrinsic(Call::declare_box_touched)) {
158 return 0;
159 } else {
160 return IRMutator::visit(op);
161 }
162 }
163};

165} // namespace

167Stmt allocation_bounds_inference(Stmt s,
168 const map<string, Function> &env,
169 const FuncValueBounds &fb) {
170 s = AllocationInference(env, fb).mutate(s);
171 s = StripDeclareBoxTouched().mutate(s);
172 return s;
173}
