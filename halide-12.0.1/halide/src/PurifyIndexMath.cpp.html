11class PurifyIndexMath : public IRMutator {
12 using IRMutator::visit;
13
14 Expr visit(const Call *op) override {
15 if (op->is_intrinsic(Call::signed_integer_overflow)) {
16 // This should only occur for values that are evaluated
17 // but never actually used (e.g. on select branches that
18 // are unreachable). Just replace it with zero.
19 return make_zero(op->type);
20 } else {
21 return IRMutator::visit(op);
22 }
23 }
24};

26} // namespace
