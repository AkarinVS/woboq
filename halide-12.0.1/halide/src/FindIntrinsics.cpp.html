48// Returns true iff t is an integral type where overflow is undefined
49bool no_overflow_int(Type t) {
50 return t.is_int() && t.bits() >= 32;

53// Returns true iff t does not have a well defined overflow behavior.
54bool no_overflow(Type t) {
55 return t.is_float() || no_overflow_int(t);

58// If there's a widening add or subtract in the first e.type().bits() / 2 - 1
59// levels down a tree of adds or subtracts, we know there's enough headroom for
60// another add without overflow. For example, it is safe to add to
61// (widening_add(x, y) - z) without overflow.
62bool is_safe_for_add(const Expr &e, int max_depth) {
63 if (max_depth-- <= 0) {
64 return false;
65 }
66 if (const Add *add = e.as<Add>()) {
67 return is_safe_for_add(add->a, max_depth) || is_safe_for_add(add->b, max_depth);
68 } else if (const Sub *sub = e.as<Sub>()) {
69 return is_safe_for_add(sub->a, max_depth) || is_safe_for_add(sub->b, max_depth);
70 } else if (const Cast *cast = e.as<Cast>()) {
71 if (cast->type.bits() > cast->value.type().bits()) {
72 return true;
73 } else if (cast->type.bits() == cast->value.type().bits()) {
74 return is_safe_for_add(cast->value, max_depth);
75 }
76 } else if (Call::as_intrinsic(e, {Call::widening_add, Call::widening_sub})) {
77 return true;
78 }
79 return false;
80}
81
82bool is_safe_for_add(const Expr &e) {
83 return is_safe_for_add(e, e.type().bits() / 2 - 1);
84}

86// We want to find and remove an add of 'round' from e. This is not
87// the same thing as just subtracting round, we specifically want
88// to remove an addition of exactly round.
89Expr find_and_subtract(const Expr &e, const Expr &round) {
90 if (const Add *add = e.as<Add>()) {
91 Expr a = find_and_subtract(add->a, round);
92 if (a.defined()) {
93 return Add::make(a, add->b);
94 }
95 Expr b = find_and_subtract(add->b, round);
96 if (b.defined()) {
97 return Add::make(add->a, b);
98 }
99 } else if (const Sub *sub = e.as<Sub>()) {
100 Expr a = find_and_subtract(sub->a, round);
101 if (a.defined()) {
102 return Sub::make(a, sub->b);
103 }
104 // We can't recurse into the negatve part of a subtract.
105 } else if (can_prove(e == round)) {
106 return make_zero(e.type());
107 }
108 return Expr();
109}

111Expr to_rounding_shift(const Call *c) {
112 if (c->is_intrinsic(Call::shift_left) || c->is_intrinsic(Call::shift_right)) {
113 internal_assert(c->args.size() == 2);
114 Expr a = c->args[0];
115 Expr b = c->args[1];
116
117 // Helper to make the appropriate shift.
118 auto rounding_shift = [&](const Expr &a, const Expr &b) {
119 if (c->is_intrinsic(Call::shift_right)) {
120 return rounding_shift_right(a, b);
121 } else {
122 return rounding_shift_left(a, b);
123 }
124 };
125
126 // The rounding offset for the shift we have.
127 Type round_type = a.type().with_lanes(1);
128 if (Call::as_intrinsic(a, {Call::widening_add})) {
129 round_type = round_type.narrow();
130 }
131 Expr round;
132 if (c->is_intrinsic(Call::shift_right)) {
133 round = simplify((make_one(round_type) << max(cast(b.type().with_bits(round_type.bits()), b), 0)) / 2);
134 } else {
135 round = simplify((make_one(round_type) >> min(cast(b.type().with_bits(round_type.bits()), b), 0)) / 2);
136 }
137
138 // We can always handle widening adds.
139 if (const Call *add = Call::as_intrinsic(a, {Call::widening_add})) {
140 if (can_prove(lower_intrinsics(add->args[0]) == round)) {
141 return rounding_shift(cast(add->type, add->args[1]), b);
142 } else if (can_prove(lower_intrinsics(add->args[1]) == round)) {
143 return rounding_shift(cast(add->type, add->args[0]), b);
144 }
145 }
146
147 // If it wasn't a widening or saturating add, we might still
148 // be able to safely accept the rounding.
149 Expr a_less_round = find_and_subtract(a, round);
150 if (a_less_round.defined()) {
151 // We found and removed the rounding. However, we may have just changed
152 // behavior due to overflow. This is still safe if the type is not
153 // overflowing, or we can find a widening add or subtract in the tree
154 // of adds/subtracts. This is a common pattern, e.g.
155 // rounding_halving_add(a, b) = shift_round(widening_add(a, b) + 1, 1).
156 // TODO: This could be done with bounds inference instead of this hack
157 // if it supported intrinsics like widening_add and tracked bounds for
158 // types other than int32.
159 if (no_overflow(a.type()) || is_safe_for_add(a_less_round)) {
160 return rounding_shift(simplify(a_less_round), b);
161 }
162 }
163 }
164
165 return Expr();
166}

177 Expr visit(const Add *op) override {
178 if (!find_intrinsics_for_type(op->type)) {
179 return IRMutator::visit(op);
180 }
181
182 Expr a = mutate(op->a);
183 Expr b = mutate(op->b);
184
185 // Try widening both from the same signedness as the result, and from uint.
186 for (halide_type_code_t code : {op->type.code(), halide_type_uint}) {
187 Type narrow = op->type.narrow().with_code(code);
188 Expr narrow_a = lossless_cast(narrow, a);
189 Expr narrow_b = lossless_cast(narrow, b);
190
191 if (narrow_a.defined() && narrow_b.defined()) {
192 Expr result = widening_add(narrow_a, narrow_b);
193 if (result.type() != op->type) {
194 result = Cast::make(op->type, result);
195 }
196 return mutate(result);
197 }
198 }
199
200 if (a.same_as(op->a) && b.same_as(op->b)) {
201 return op;
202 } else {
203 return Add::make(a, b);

248 Expr visit(const Mul *op) override {
249 if (!find_intrinsics_for_type(op->type)) {
250 return IRMutator::visit(op);
251 }
252
253 if (as_const_int(op->b) || as_const_uint(op->b)) {
254 // Distribute constants through add/sub. Do this before we muck everything up with widening
255 // intrinsics.
256 // TODO: Only do this for widening?
257 // TODO: Try to do this with IRMatcher::rewriter. The challenge is managing the narrowing/widening casts,
258 // and doing constant folding without the simplifier undoing the work.
259 if (const Add *add_a = op->a.as<Add>()) {
260 return mutate(Add::make(simplify(Mul::make(add_a->a, op->b)), simplify(Mul::make(add_a->b, op->b))));
261 } else if (const Sub *sub_a = op->a.as<Sub>()) {
262 return mutate(Sub::make(simplify(Mul::make(sub_a->a, op->b)), simplify(Mul::make(sub_a->b, op->b))));
263 }
264 }
265
266 Expr a = mutate(op->a);
267 Expr b = mutate(op->b);
268
269 // Rewrite multiplies to shifts if possible.
270 if (op->type.is_int() || op->type.is_uint()) {
271 int pow2 = 0;
272 if (is_const_power_of_two_integer(a, &pow2)) {
273 return mutate(b << cast(UInt(b.type().bits()), pow2));
274 } else if (is_const_power_of_two_integer(b, &pow2)) {
275 return mutate(a << cast(UInt(a.type().bits()), pow2));
276 }
277 }
278
279 // We're applying this to float, which seems OK? float16 * float16 -> float32 is a widening multiply?
280 // This uses strip_widening_cast to ignore the signedness of the narrow value.
281 Expr narrow_a = strip_widening_cast(a);
282 Expr narrow_b = strip_widening_cast(b);
283 if (narrow_a.defined() && narrow_b.defined() &&
284 (narrow_a.type().is_int_or_uint() == narrow_b.type().is_int_or_uint() ||
285 narrow_a.type().is_float() == narrow_b.type().is_float())) {
286 Expr result = widening_mul(narrow_a, narrow_b);
287 if (result.type() != op->type) {
288 result = Cast::make(op->type, result);
289 }
290 return mutate(result);
291 }
292
293 if (a.same_as(op->a) && b.same_as(op->b)) {
294 return op;
295 } else {
296 return Mul::make(a, b);
297 }

355 if (!find_intrinsics_for_type(op->type)) {
356 return IRMutator::visit(op);
357 }
358
359 Expr value = mutate(op->value);
360
361 // This mutator can generate redundant casts. We can't use the simplifier because it
362 // undoes some of the intrinsic lowering here, and it causes some problems due to
363 // factoring (instead of distributing) constants.
364 if (const Cast *cast = value.as<Cast>()) {
365 if (cast->type.can_represent(cast->value.type()) || cast->type.can_represent(op->type)) {
366 // The intermediate cast is redundant.
367 value = cast->value;
368 }
369 }
370
371 if (op->type.is_int() || op->type.is_uint()) {
372 Expr lower = cast(value.type(), op->type.min());
373 Expr upper = cast(value.type(), op->type.max());
374
375 auto rewrite = IRMatcher::rewriter(value, op->type);
376
377 Type op_type_wide = op->type.widen();
378 Type signed_type_wide = op_type_wide.with_code(halide_type_int);
379 Type unsigned_type = op->type.with_code(halide_type_uint);

381 int bits = op->type.bits();
382 auto is_x_same_int = op->type.is_int() && is_int(x, bits);

384 auto is_x_same_int_or_uint = is_x_same_int || is_x_same_uint;
385 // clang-format off
386 if (rewrite(max(min(widening_add(x, y), upper), lower), saturating_add(x, y), is_x_same_int_or_uint) ||

388 rewrite(min(cast(signed_type_wide, widening_add(x, y)), upper), saturating_add(x, y), is_x_same_uint) ||

403 rewrite(min(shift_right(widening_mul(x, y), z), upper), mul_shift_right(x, y, cast(unsigned_type, z)), is_x_same_uint && is_uint(z)) ||
404 rewrite(min(rounding_shift_right(widening_mul(x, y), z), upper), rounding_mul_shift_right(x, y, cast(unsigned_type, z)), is_x_same_uint && is_uint(z)) ||
405 // We don't need saturation for the full upper half of a multiply.
406 // For signed integers, this is almost true, except for when x and y
407 // are both the most negative value. For these, we only need saturation
408 // at the upper bound.
409 rewrite(min(shift_right(widening_mul(x, y), c0), upper), mul_shift_right(x, y, cast(unsigned_type, c0)), is_x_same_int && c0 >= bits - 1) ||
410 rewrite(min(rounding_shift_right(widening_mul(x, y), c0), upper), rounding_mul_shift_right(x, y, cast(unsigned_type, c0)), is_x_same_int && c0 >= bits - 1) ||

424 // When there is saturation, we can only avoid the widening if we know the shift is

441 Expr b = lossless_cast(op->type.with_code(shift->args[1].type().code()), shift->args[1]);

443 if (!is_saturated ||
444 (shift->is_intrinsic(Call::rounding_shift_right) && can_prove(b >= 0)) ||
445 (shift->is_intrinsic(Call::rounding_shift_left) && can_prove(b <= 0))) {
446 return mutate(Call::make(op->type, shift->name, {a, b}, Call::PureIntrinsic));
447 }
448 }
449 }
450 // clang-format on
451 }
452
453 if (value.same_as(op->value)) {
454 return op;
455 } else if (op->type != value.type()) {
456 return Cast::make(op->type, value);
457 } else {
458 return value;

462 Expr visit(const Call *op) override {
463 if (!find_intrinsics_for_type(op->type)) {
464 return IRMutator::visit(op);
465 }
466
467 Expr mutated = IRMutator::visit(op);
468 op = mutated.as<Call>();
469 if (!op) {
470 return mutated;
471 }
472
473 auto rewrite = IRMatcher::rewriter(op, op->type);
474 if (rewrite(intrin(Call::abs, widening_sub(x, y)), cast(op->type, intrin(Call::absd, x, y))) ||
475 false) {
476 return rewrite.result;
477 }
478
479 if (no_overflow(op->type)) {
480 // clang-format off
481 if (rewrite(halving_add(x + y, 1), rounding_halving_add(x, y)) ||
482 rewrite(halving_add(x, y + 1), rounding_halving_add(x, y)) ||
483 rewrite(halving_add(x + 1, y), rounding_halving_add(x, y)) ||
484 rewrite(halving_add(x - y, 1), rounding_halving_sub(x, y)) ||
485 rewrite(halving_sub(x + 1, y), rounding_halving_sub(x, y)) ||
486 rewrite(halving_add(x, 1), rounding_shift_right(x, 1)) ||
487 rewrite(shift_right(x + y, 1), halving_add(x, y)) ||
488 rewrite(shift_right(x - y, 1), halving_sub(x, y)) ||
489 rewrite(rounding_shift_right(x + y, 1), rounding_halving_add(x, y)) ||
490 rewrite(rounding_shift_right(x - y, 1), rounding_halving_sub(x, y)) ||
491 false) {
492 return mutate(rewrite.result);
493 }
494 // clang-format on
495 }
496
497 // Move widening casts inside widening arithmetic outside the arithmetic,
498 // e.g. widening_mul(widen(u8), widen(i8)) -> widen(widening_mul(u8, i8)).
499 if (op->is_intrinsic(Call::widening_mul)) {
500 internal_assert(op->args.size() == 2);
501 Expr narrow_a = strip_widening_cast(op->args[0]);
502 Expr narrow_b = strip_widening_cast(op->args[1]);
503 if (narrow_a.defined() && narrow_b.defined()) {
504 return mutate(Cast::make(op->type, widening_mul(narrow_a, narrow_b)));
505 }
506 } else if (op->is_intrinsic(Call::widening_add)) {
507 internal_assert(op->args.size() == 2);
508 for (halide_type_code_t t : {op->type.code(), halide_type_uint}) {
509 Type narrow_t = op->type.narrow().narrow().with_code(t);
510 Expr narrow_a = lossless_cast(narrow_t, op->args[0]);
511 Expr narrow_b = lossless_cast(narrow_t, op->args[1]);
512 if (narrow_a.defined() && narrow_b.defined()) {
513 return mutate(Cast::make(op->type, widening_add(narrow_a, narrow_b)));
514 }
515 }
516 } else if (op->is_intrinsic(Call::widening_sub)) {
517 internal_assert(op->args.size() == 2);
518 for (halide_type_code_t t : {op->type.code(), halide_type_uint}) {
519 Type narrow_t = op->type.narrow().narrow().with_code(t);
520 Expr narrow_a = lossless_cast(narrow_t, op->args[0]);
521 Expr narrow_b = lossless_cast(narrow_t, op->args[1]);
522 if (narrow_a.defined() && narrow_b.defined()) {
523 return mutate(Cast::make(op->type, widening_sub(narrow_a, narrow_b)));
524 }
525 }
526 }
527
528 if (op->is_intrinsic(Call::shift_right) || op->is_intrinsic(Call::shift_left)) {
529 // Try to turn this into a widening shift.
530 internal_assert(op->args.size() == 2);
531 Expr a_narrow = lossless_narrow(op->args[0]);
532 Expr b_narrow = lossless_narrow(op->args[1]);
533 if (a_narrow.defined() && b_narrow.defined()) {
534 Expr result = op->is_intrinsic(Call::shift_left) ? widening_shift_left(a_narrow, b_narrow) : widening_shift_right(a_narrow, b_narrow);
535 if (result.type() != op->type) {
536 result = Cast::make(op->type, result);
537 }
538 return mutate(result);
539 }
540
541 // Try to turn this into a rounding shift.
542 Expr rounding_shift = to_rounding_shift(op);
543 if (rounding_shift.defined()) {
544 return mutate(rounding_shift);
545 }
546 }
547
548 if (op->is_intrinsic(Call::rounding_shift_left) || op->is_intrinsic(Call::rounding_shift_right)) {
549 // Try to turn this into a widening shift.
550 internal_assert(op->args.size() == 2);
551 Expr a_narrow = lossless_narrow(op->args[0]);
552 Expr b_narrow = lossless_narrow(op->args[1]);
553 if (a_narrow.defined() && b_narrow.defined()) {
554 Expr result;
555 if (op->is_intrinsic(Call::rounding_shift_right) && can_prove(b_narrow > 0)) {
556 result = rounding_shift_right(a_narrow, b_narrow);
557 } else if (op->is_intrinsic(Call::rounding_shift_left) && can_prove(b_narrow < 0)) {
558 result = rounding_shift_left(a_narrow, b_narrow);
559 } else {
560 return op;
561 }
562 if (result.type() != op->type) {
563 result = Cast::make(op->type, result);
564 }
565 return mutate(result);
566 }
567 }
568 return op;
569 }
570};

590Expr lower_widening_sub(const Expr &a, const Expr &b) {
591 Type wide = a.type().widen();
592 if (wide.is_uint()) {
593 wide = wide.with_code(halide_type_int);
594 }
595 return Cast::make(wide, a) - Cast::make(wide, b);
596}

598Expr lower_widening_shift_left(const Expr &a, const Expr &b) {
599 return widen(a) << b;
600}

602Expr lower_widening_shift_right(const Expr &a, const Expr &b) {
603 return widen(a) >> b;
604}

606Expr lower_rounding_shift_left(const Expr &a, const Expr &b) {
607 // Shift left, then add one to the result if bits were dropped
608 // (because b < 0) and the most significant dropped bit was a one.
609 Expr b_negative = select(b < 0, make_one(a.type()), make_zero(a.type()));
610 return simplify((a << b) + (b_negative & (a << (b + 1))));
611}
612
613Expr lower_rounding_shift_right(const Expr &a, const Expr &b) {
614 // Shift right, then add one to the result if bits were dropped
615 // (because b > 0) and the most significant dropped bit was a one.
616 Expr b_positive = select(b > 0, make_one(a.type()), make_zero(a.type()));
617 return simplify((a >> b) + (b_positive & (a >> (b - 1))));
618}

620Expr lower_saturating_add(const Expr &a, const Expr &b) {
621 internal_assert(a.type() == b.type());
622 // Lower saturating add without using widening arithmetic, which may require
623 // types that aren't supported.
624 return simplify(clamp(a, a.type().min() - min(b, 0), a.type().max() - max(b, 0))) + b;
625}
626
627Expr lower_saturating_sub(const Expr &a, const Expr &b) {
628 internal_assert(a.type() == b.type());
629 // Lower saturating add without using widening arithmetic, which may require
630 // types that aren't supported.
631 return simplify(clamp(a, a.type().min() + max(b, 0), a.type().max() + min(b, 0))) - b;
632}

634Expr lower_halving_add(const Expr &a, const Expr &b) {
635 internal_assert(a.type() == b.type());
636 // Borrowed from http://aggregate.org/MAGIC/#Average%20of%20Integers
637 return (a & b) + ((a ^ b) >> 1);
638}
639
640Expr lower_halving_sub(const Expr &a, const Expr &b) {
641 internal_assert(a.type() == b.type());
642 return (a >> 1) - (b >> 1) - (((b & 1) - (a & 1) + 1) >> 1);
643}
644
645// TODO: These should using rounding_shift_right, but lowering that
646// results in double widening and the simplifier doesn't fix it.
647Expr lower_rounding_halving_add(const Expr &a, const Expr &b) {
648 internal_assert(a.type() == b.type());
649 return (a >> 1) + (b >> 1) + (((a & 1) + (b & 1) + 1) >> 1);
650}
651
652Expr lower_rounding_halving_sub(const Expr &a, const Expr &b) {
653 internal_assert(a.type() == b.type());
654 return (a >> 1) - (b >> 1) + (((a & 1) - (b & 1) + 1) >> 1);
655}

657Expr lower_sorted_avg(const Expr &a, const Expr &b) {
658 // b > a, so the following works without widening.
659 return a + ((b - a) >> 1);
660}

662Expr lower_mul_shift_right(const Expr &a, const Expr &b, const Expr &q) {
663 internal_assert(a.type() == b.type());
664 int full_q = a.type().bits();
665 if (a.type().is_int()) {
666 full_q -= 1;
667 }
668 if (can_prove(q < full_q)) {
669 // Try to rewrite this to a "full precision" multiply by multiplying
670 // one of the operands and the denominator by a constant. We only do this
671 // if it isn't already full precision. This avoids infinite loops despite
672 // "lowering" this to another mul_shift_right operation.
673 Expr missing_q = full_q - q;
674 internal_assert(missing_q.type().bits() == b.type().bits());
675 Expr new_b = simplify(b << missing_q);
676 if (is_const(new_b) && can_prove(new_b >> missing_q == b)) {
677 return mul_shift_right(a, new_b, full_q);
678 }
679 Expr new_a = simplify(a << missing_q);
680 if (is_const(new_a) && can_prove(new_a >> missing_q == a)) {
681 return mul_shift_right(new_a, b, full_q);
682 }
683 }
684
685 if (can_prove(q > a.type().bits())) {
686 // If q is bigger than the narrow type, write it as an exact upper
687 // half multiply, followed by an extra shift.
688 Expr result = mul_shift_right(a, b, a.type().bits());
689 result = result >> simplify(q - a.type().bits());
690 return result;
691 }
692
693 // If all else fails, just widen, shift, and narrow.
694 Expr result = widening_mul(a, b) >> q;
695 if (!can_prove(q >= a.type().bits())) {
696 result = saturating_narrow(result);
697 } else {
698 result = narrow(result);
699 }
700 return result;
701}
702
703Expr lower_rounding_mul_shift_right(const Expr &a, const Expr &b, const Expr &q) {
704 internal_assert(a.type() == b.type());
705 int full_q = a.type().bits();
706 if (a.type().is_int()) {
707 full_q -= 1;
708 }
709 // Try to rewrite this to a "full precision" multiply by multiplying
710 // one of the operands and the denominator by a constant. We only do this
711 // if it isn't already full precision. This avoids infinite loops despite
712 // "lowering" this to another mul_shift_right operation.
713 if (can_prove(q < full_q)) {
714 Expr missing_q = full_q - q;
715 internal_assert(missing_q.type().bits() == b.type().bits());
716 Expr new_b = simplify(b << missing_q);
717 if (is_const(new_b) && can_prove(new_b >> missing_q == b)) {
718 return rounding_mul_shift_right(a, new_b, full_q);
719 }
720 Expr new_a = simplify(a << missing_q);
721 if (is_const(new_a) && can_prove(new_a >> missing_q == a)) {
722 return rounding_mul_shift_right(new_a, b, full_q);
723 }
724 }
725
726 // If all else fails, just widen, shift, and narrow.
727 Expr result = rounding_shift_right(widening_mul(a, b), q);
728 if (!can_prove(q >= a.type().bits())) {
729 result = saturating_narrow(result);
730 } else {
731 result = narrow(result);
732 }
733 return result;
734}

736Expr lower_intrinsic(const Call *op) {
737 if (op->is_intrinsic(Call::widening_add)) {
738 internal_assert(op->args.size() == 2);
739 return lower_widening_add(op->args[0], op->args[1]);
740 } else if (op->is_intrinsic(Call::widening_mul)) {
741 internal_assert(op->args.size() == 2);
742 return lower_widening_mul(op->args[0], op->args[1]);
743 } else if (op->is_intrinsic(Call::widening_sub)) {
744 internal_assert(op->args.size() == 2);
745 return lower_widening_sub(op->args[0], op->args[1]);
746 } else if (op->is_intrinsic(Call::saturating_add)) {
747 internal_assert(op->args.size() == 2);
748 return lower_saturating_add(op->args[0], op->args[1]);
749 } else if (op->is_intrinsic(Call::saturating_sub)) {
750 internal_assert(op->args.size() == 2);
751 return lower_saturating_sub(op->args[0], op->args[1]);
752 } else if (op->is_intrinsic(Call::widening_shift_left)) {
753 internal_assert(op->args.size() == 2);
754 return lower_widening_shift_left(op->args[0], op->args[1]);
755 } else if (op->is_intrinsic(Call::widening_shift_right)) {
756 internal_assert(op->args.size() == 2);
757 return lower_widening_shift_right(op->args[0], op->args[1]);
758 } else if (op->is_intrinsic(Call::rounding_shift_right)) {
759 internal_assert(op->args.size() == 2);
760 return lower_rounding_shift_right(op->args[0], op->args[1]);
761 } else if (op->is_intrinsic(Call::rounding_shift_left)) {
762 internal_assert(op->args.size() == 2);
763 return lower_rounding_shift_left(op->args[0], op->args[1]);
764 } else if (op->is_intrinsic(Call::halving_add)) {
765 internal_assert(op->args.size() == 2);
766 return lower_halving_add(op->args[0], op->args[1]);
767 } else if (op->is_intrinsic(Call::halving_sub)) {
768 internal_assert(op->args.size() == 2);
769 return lower_halving_sub(op->args[0], op->args[1]);
770 } else if (op->is_intrinsic(Call::rounding_halving_add)) {
771 internal_assert(op->args.size() == 2);
772 return lower_rounding_halving_add(op->args[0], op->args[1]);
773 } else if (op->is_intrinsic(Call::rounding_halving_sub)) {
774 internal_assert(op->args.size() == 2);
775 return lower_rounding_halving_sub(op->args[0], op->args[1]);
776 } else if (op->is_intrinsic(Call::rounding_mul_shift_right)) {
777 internal_assert(op->args.size() == 3);
778 return lower_rounding_mul_shift_right(op->args[0], op->args[1], op->args[2]);
779 } else if (op->is_intrinsic(Call::mul_shift_right)) {
780 internal_assert(op->args.size() == 3);
781 return lower_mul_shift_right(op->args[0], op->args[1], op->args[2]);
782 } else if (op->is_intrinsic(Call::sorted_avg)) {
783 internal_assert(op->args.size() == 2);
784 return lower_sorted_avg(op->args[0], op->args[1]);
785 } else {
786 return Expr();
787 }
