108// Combine type-and-code into a single integer to avoid nested switches.
109// Must be constexpr to allow use in case clauses.
110inline constexpr int halide_type_code(halide_type_code_t code, int bits) {
111 return (((int)code) << 8) | bits;
112}
113
114template<typename T>
115T value_as(const halide_type_t &type, const halide_scalar_value_t &value) {
116 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
117 case halide_type_code(halide_type_int, 8):
118 return (T)value.u.i8;
119 case halide_type_code(halide_type_int, 16):
120 return (T)value.u.i16;
121 case halide_type_code(halide_type_int, 32):
122 return (T)value.u.i32;
123 case halide_type_code(halide_type_int, 64):
124 return (T)value.u.i64;
125 case halide_type_code(halide_type_uint, 1):
126 return (T)value.u.b;
127 case halide_type_code(halide_type_uint, 8):
128 return (T)value.u.u8;
129 case halide_type_code(halide_type_uint, 16):
130 return (T)value.u.u16;
131 case halide_type_code(halide_type_uint, 32):
132 return (T)value.u.u32;
133 case halide_type_code(halide_type_uint, 64):
134 return (T)value.u.u64;
135 case halide_type_code(halide_type_float, 32):
136 return (T)value.u.f32;
137 case halide_type_code(halide_type_float, 64):
138 return (T)value.u.f64;
139 default:
140 fail() << "Can't convert packet with type: " << (int)type.code << "bits: " << type.bits;
141 return (T)0;
142 }
143}

145template<typename T>
146T get_value_as(const halide_trace_packet_t &p, int idx) {
147 const uint8_t *val = (const uint8_t *)(p.value()) + idx * p.type.bytes();
148 // 'val' may not be aligned: memcpy it to an aligned local
149 // so that value_as<>() won't complain under sanitizers.
150 halide_scalar_value_t aligned_value;
151 // Only copy the number of bytes in the type: the stream isn't guaranteed
152 // to be padded to sizeof(halide_scalar_value_t).
153 memcpy(&aligned_value, val, p.type.bits / 8);
154 return value_as<double>(p.type, aligned_value);
155}

157struct PacketAndPayload : public halide_trace_packet_t {
158 uint8_t payload[4096];
159
160 static bool read_or_die(void *buf, size_t count) {
161 char *p = (char *)buf;
162 char *p_end = p + count;
163 while (p < p_end) {
164 int64_t bytes_read = ::read(STDIN_FILENO, p, p_end - p);
165 if (bytes_read == 0) {
166 return false; // EOF
167 } else if (bytes_read < 0) {
168 fail() << "Unable to read packet";
169 }
170 p += bytes_read;
171 }
172 assert(p == p_end);
173 return true;
174 }
175
176 bool read() {
177 constexpr size_t header_size = sizeof(halide_trace_packet_t);
178 if (!read_or_die(this, header_size)) {
179 return false; // EOF
180 }
181
182 const size_t payload_size = this->size - header_size;
183 if (payload_size > sizeof(this->payload) || !read_or_die(this->payload, payload_size)) {
184 // Shouldn't ever get EOF here
185 fail() << "Unable to read packet payload of size " << payload_size;
186 }
187 return true;
188 }
189};

194struct FuncInfo {
195
196 // Info about Funcs type and touched-extent, emitted
197 // by the tracing code.
198 FuncTypeAndDim type_and_dim;

200 int layout_order = -1;

206 // Information about actual observed values gathered while parsing the trace
207 struct Observed {
208 std::string qualified_name;
209 int first_draw_time = -1, first_packet_idx = -1;
210 double min_value = 0.0, max_value = 0.0;
211 int min_coord[16];
212 int max_coord[16];
213 int num_realizations = 0, num_productions = 0;
214 uint64_t stores = 0, loads = 0;
215
216 Observed() {
217 memset(min_coord, 0, sizeof(min_coord));
218 memset(max_coord, 0, sizeof(max_coord));
219 }
220
221 void observe_load(const halide_trace_packet_t &p) {
222 observe_load_or_store(p);
223 loads += p.type.lanes;
224 }
225
226 void observe_store(const halide_trace_packet_t &p) {
227 observe_load_or_store(p);
228 stores += p.type.lanes;
229 }
230
231 void observe_load_or_store(const halide_trace_packet_t &p) {
232 const int *coords = p.coordinates();
233 for (int i = 0; i < std::min(16, p.dimensions / p.type.lanes); i++) {
234 for (int lane = 0; lane < p.type.lanes; lane++) {
235 int coord = coords[i * p.type.lanes + lane];
236 if (loads + stores == 0 && lane == 0) {
237 min_coord[i] = coord;
238 max_coord[i] = coord + 1;
239 } else {
240 min_coord[i] = std::min(min_coord[i], coord);
241 max_coord[i] = std::max(max_coord[i], coord + 1);
242 }
243 }
244 }
245
246 for (int i = 0; i < p.type.lanes; i++) {
247 double value = get_value_as<double>(p, i);
248 if (stores + loads == 0) {
249 min_value = value;
250 max_value = value;
251 } else {
252 min_value = std::min(min_value, value);
253 max_value = std::max(max_value, value);
254 }
255 }
256 }
257
258 void report() {
259 std::ostringstream o;
260 for (int i = 0; i < 16; i++) {
261 if (min_coord[i] == 0 && max_coord[i] == 0) {
262 break;
263 }
264 if (i > 0) {
265 o << " x ";
266 }
267 o << "[" << min_coord[i] << ", " << max_coord[i] << ")";
268 }
269 info()
270 << "Func " << qualified_name << ":\n"
271 << o.str() << "\n"
272 << " range of values: [" << min_value << ", " << max_value << "]\n"
273 << " number of realizations: " << num_realizations << "\n"
274 << " number of productions: " << num_productions << "\n"
275 << " number of loads: " << loads << "\n"
276 << " number of stores: " << stores << "\n";
277 }
278
279 } stats;
280};

429// Calculate the maximum 2d rendered size for a given Box and stride, assuming
430// a zoom factor of 1. This uses the same recursive approach as fill_realization()
431// for simplicity.
432void calc_2d_size(const std::vector<Range> &dims, const std::vector<Point> &strides, Range *x, Range *y,
433 int current_dimension = 0, int x_off = 0, int y_off = 0) {
434 if (current_dimension == 0) {
435 x->min = 2147483647;
436 x->extent = -2147483647;
437 y->min = 2147483647;
438 y->extent = -2147483647;
439 }
440 if (current_dimension == (int)dims.size()) {
441 x->min = std::min(x->min, x_off);
442 x->extent = std::max(x->extent, x_off);
443 y->min = std::min(y->min, y_off);
444 y->extent = std::max(y->extent, y_off);
445 } else {
446 const auto &m = dims.at(current_dimension);
447 const Point &stride = strides.at(current_dimension);
448 x_off += stride.x * m.min;
449 y_off += stride.y * m.min;
450 for (int i = 0; i < m.extent; i++) {
451 calc_2d_size(dims, strides, x, y, current_dimension + 1, x_off, y_off);
452 x_off += stride.x;
453 y_off += stride.y;
454 }
455 }
456 if (current_dimension == 0) {
457 x->extent = std::max(1, x->extent - x->min + 1);
458 y->extent = std::max(1, y->extent - y->min + 1);
459 }
460}

464// Given a FuncConfig, check each field for "use some reasonable default"
465// value and fill in something reasonable.
466void finalize_func_config_values(const GlobalConfig &globals, FuncInfo &fi) {
467 // Make a FuncConfig with 'safe' defaults for everything,
468 // then merge the existing cfg into it.
469 FuncConfig safe;
470 safe.zoom = 1.f;
471 safe.load_cost = 0;
472 safe.store_cost = 1;
473 safe.pos = {0, 0};
474 safe.strides = {{1, 0}, {0, 1}};
475 safe.color_dim = -1;
476 safe.min = 0.0;
477 safe.max = 1.0;
478 safe.labels = {};
479 safe.blank_on_end_realization = 0;
480 safe.uninitialized_memory_color = globals.default_uninitialized_memory_color;
481
482 if (fi.type_and_dim_valid) {
483 // Try to choose better values for min and max based on type.
484 // TODO: only considers the first type given; in general,
485 // HTV doesn't deal with Tuple-valued Funcs very well.
486 const halide_type_t &type = fi.type_and_dim.types.at(0);
487 if (type.code == halide_type_uint) {
488 safe.max = (double)((1 << type.bits) - 1);
489 } else if (type.code == halide_type_int) {
490 double d = (double)(1 << (type.bits - 1));
491 safe.max = d - 1;
492 // safe.min = -d;
493 // In practice, assuming a min of zero (rather then -INT_MIN)
494 // for signed types produces less-weird results.
495 safe.min = 0.0;
496 }
497 }
498
499 safe.merge_from(fi.config);
500 safe.uninitialized_memory_color |= 0xff000000;
501 fi.config = safe;
502}

514void do_auto_layout(const GlobalConfig &globals, const std::string &func_name, FuncInfo &fi) {
515 assert(fi.type_and_dim_valid);
516
517 const Point &pad = globals.auto_layout_pad;
518 Point cell_size = {
519 globals.frame_size.x / globals.auto_layout_grid.x,
520 globals.frame_size.y / globals.auto_layout_grid.y};
521 info() << "cell_size is " << cell_size << "\n";
522 info() << "auto_layout_pad is " << pad << "\n";
523
524 int row = fi.layout_order / globals.auto_layout_grid.x;
525 int col = fi.layout_order % globals.auto_layout_grid.x;
526
527 if (fi.config.color_dim < -1) {
528 // If color_dim is unspecified and it looks like a 2d RGB Func, make it one
529 const auto &dims = fi.type_and_dim.dims;
530 if (dims.size() == 3) {
531 if ((dims[2].extent == 3 || dims[2].extent == 4)) {
532 fi.config.color_dim = 2;
533 } else if ((dims[0].extent == 3 || dims[0].extent == 4)) {
534 fi.config.color_dim = 0;
535 if (fi.config.strides.empty()) {
536 fi.config.strides = {{0, 0}, {1, 0}, {0, 1}};
537 }
538 }
539 } else if (dims.size() == 4) {
540 // 4D, maybe a Tensor? Treat as grayscale with x = dim(1), y = dim(2)
541 fi.config.strides = {{0, 0}, {1, 0}, {0, 1}, {0, 0}};
542 }
543 }
544
545 if (fi.config.zoom < 0.f) {
546 // Ensure that all of the FuncInfos have strides that match
547 // the number of dimensions expected by FuncTypeAndDim, adding
548 // zero-stride pairs as needed (this simplifies rendering checks
549 // later on)
550 if (fi.config.strides.empty()) {
551 fi.config.strides = {{1, 0}, {0, 1}};
552 }
553 while (fi.config.strides.size() < fi.type_and_dim.dims.size()) {
554 fi.config.strides.emplace_back(0, 0);
555 }
556
557 // Calc the 2d size that this would render at (including stride-stretching) for zoom=1
558 Range xr, yr;
559 calc_2d_size(fi.type_and_dim.dims, fi.config.strides, &xr, &yr);
560 info() << "calc_2d_size for " << func_name << " is " << xr << ", " << yr << "\n";
561
562 // Use that size to calculate the zoom we need -- this chooses
563 // a zoom that maximizes the size within the cell.
564 float zoom_x = (float)(cell_size.x - pad.x) / (float)xr.extent;
565 float zoom_y = (float)(cell_size.y - pad.y) / (float)yr.extent;
566 fi.config.zoom = std::min(zoom_x, zoom_y);
567
568 // Try to choose an even-multiple zoom for better display
569 // and just less weirdness.
570 if (fi.config.zoom > 100.f) {
571 // Zooms this large are usually for things like input matrices.
572 // Perhaps clamp at something smaller?
573 fi.config.zoom = std::floor(fi.config.zoom / 100.f) * 100.f;
574 } else if (fi.config.zoom > 10.f) {
575 fi.config.zoom = std::floor(fi.config.zoom / 10.f) * 10.f;
576 } else if (fi.config.zoom > 1.f) {
577 fi.config.zoom = std::floor(fi.config.zoom * 2.f) / 2.f;
578 } else if (fi.config.zoom < 1.f) {
579 fi.config.zoom = std::ceil(fi.config.zoom * 20.f) / 20.f;
580 }
581 info() << "zoom for " << func_name << " is " << zoom_x << " " << zoom_y << " -> " << fi.config.zoom << "\n";
582 }
583
584 // Put the image at the top-left of the cell. (Should we try to
585 // center within the cell?)
586 if (fi.config.pos.x < 0 && fi.config.pos.y < 0) {
587 fi.config.pos.x = col * cell_size.x + pad.x;
588 fi.config.pos.y = row * cell_size.y + pad.y;
589 }
590 info() << "pos for " << func_name << " is " << fi.config.pos.x << " " << fi.config.pos.y << "\n";
591
592 if (fi.config.labels.empty()) {
593 std::string label_suffix = " (" + std::to_string((int)(fi.config.zoom * 100)) + "%)";
594 std::string label = func_name + label_suffix;
595 const int label_space = cell_size.x - pad.x * 2;
596 float h_scale = 1.f;
597 int label_width = label.size() * inconsolata_char_width;
598 if (label_width > label_space) {
599 // "minimum" depends on lots of things but for 1080p output, 70% seems fair
600 const float min_readable_h_scale = 0.7f;
601 h_scale = std::max(min_readable_h_scale, std::min(1.f, (float)label_space / (float)label_width));
602 info() << "h_scale for label (" << label << ") is " << h_scale << "\n";
603 // Still too wide? Discard the suffix.
604 if (label_width * h_scale > label_space) {
605 label = func_name;
606 label_width = label.size() * inconsolata_char_width;
607 }
608 // Still too wide? Try lopping off characters to shorten it rather
609 // than squishing it into oblivion. Let's lop off the *beginning*
610 // rather than the end, on the assumption that long names are more unique at the end.
611 if (label_width * h_scale > label_space) {
612 while (label.size() > 1 && (label.size() + 1) * inconsolata_char_width * h_scale > label_space) {
613 label = label.substr(1);
614 }
615 // prepend "~" to hint it's squished
616 label = "~" + label;
617 info() << "label squished to (" << label << ")\n";
618 }
619 }
620 fi.config.labels.push_back({label, {0, 0}, 10, h_scale});
621 }
622
623 fi.config_valid = true;
624}

638float calc_side_length(int min_cells, int width, int height) {
639 const float aspect_ratio = (float)width / (float)height;
640 const float p = std::ceil(std::sqrt(min_cells * aspect_ratio));
641 const float par = p / aspect_ratio;
642 const float s = std::floor(par) * p < min_cells ?
643 height / std::ceil(par) :
644 width / p;
645 return s;
646}
647
648// Calculate the 'best' cell size such that we can fit at least min_cells
649// into the given width x height. Currently this calculates perfectly
650// square cells, which is OK but a little wasteful (eg for min_cells=20
651// and size 1920x1080, it calculates a grid of 7x4 which wastes 8 cells).
652// We could probably do better if we just tried to keep the cells 'nearly'
653// square (aspect ratio <= 1.25).
654Point best_cell_size(int min_cells, int width, int height) {
655 const float sx = calc_side_length(min_cells, width, height);
656 const float sy = calc_side_length(min_cells, height, width);
657 const int edge = floor(std::max(sx, sy));
658 return {edge, edge};
659}

663void process_args(int argc, char **argv, VizState *state) {
664 GlobalConfig &globals = state->globals;
665 std::map<std::string, FuncInfo> &funcs = state->funcs;
666
667 // The struct's default values are what we want
668 FuncConfig config;
669 std::vector<Point> pos_stack;
670 std::set<std::string> labels_seen;
671
672 // If the condition is false, print usage and exit with error.
673 const auto expect = [](bool cond, int i) {
674 if (!cond) {
675 if (i) {
676 fail() << "Argument parsing failed at argument " << i << "\n"
677 << usage();
678 } else {
679 fail() << usage();
680 }
681 }
682 };
683
684 const auto parse_int = [](const char *str) -> int {
685 char *endptr = nullptr;
686 errno = 0;
687 long result = strtol(str, &endptr, 0);
688 if (errno == ERANGE || str == endptr) {
689 fail() << "Unable to parse '" << str << "' as an int\n"
690 << usage();
691 }
692 return (int)result;
693 };
694
695 const auto parse_float = [](const char *str) -> float {
696 char *endptr = nullptr;
697 errno = 0;
698 float result = strtof(str, &endptr);
699 if (errno == ERANGE || str == endptr) {
700 fail() << "Unable to parse '" << str << "' as a float\n"
701 << usage();
702 }
703 return result;
704 };
705
706 const auto parse_double = [](const char *str) -> double {
707 char *endptr = nullptr;
708 errno = 0;
709 double result = strtod(str, &endptr);
710 if (errno == ERANGE || str == endptr) {
711 fail() << "Unable to parse '" << str << "' as a double\n"
712 << usage();
713 }
714 return result;
715 };
716
717 // Parse command line args
718 int i = 1;
719 while (i < argc) {
720 std::string next = argv[i];
721 if (next == "--size") {
722 expect(i + 2 < argc, i);
723 globals.frame_size.x = parse_int(argv[++i]);
724 globals.frame_size.y = parse_int(argv[++i]);
725 } else if (next == "--func") {
726 expect(i + 1 < argc, i);
727 const char *func = argv[++i];
728 FuncInfo &fi = funcs[func];
729 fi.config.merge_from(config);
730 fi.config_valid = true;
731 } else if (next == "--min") {
732 expect(i + 1 < argc, i);
733 config.min = parse_double(argv[++i]);
734 } else if (next == "--max") {
735 expect(i + 1 < argc, i);
736 config.max = parse_double(argv[++i]);
737 } else if (next == "--move") {
738 expect(i + 2 < argc, i);
739 config.pos.x = parse_int(argv[++i]);
740 config.pos.y = parse_int(argv[++i]);
741 } else if (next == "--left") {
742 expect(i + 1 < argc, i);
743 config.pos.x -= parse_int(argv[++i]);
744 } else if (next == "--right") {
745 expect(i + 1 < argc, i);
746 config.pos.x += parse_int(argv[++i]);
747 } else if (next == "--up") {
748 expect(i + 1 < argc, i);
749 config.pos.y -= parse_int(argv[++i]);
750 } else if (next == "--down") {
751 expect(i + 1 < argc, i);
752 config.pos.y += parse_int(argv[++i]);
753 } else if (next == "--push") {
754 pos_stack.push_back(config.pos);
755 } else if (next == "--pop") {
756 expect(!pos_stack.empty(), i);
757 config.pos = pos_stack.back();
758 pos_stack.pop_back();
759 } else if (next == "--rgb") {
760 expect(i + 1 < argc, i);
761 config.color_dim = parse_int(argv[++i]);
762 } else if (next == "--gray") {
763 config.color_dim = -1;
764 } else if (next == "--blank") {
765 config.blank_on_end_realization = 1;
766 } else if (next == "--no-blank") {
767 config.blank_on_end_realization = 0;
768 } else if (next == "--zoom") {
769 expect(i + 1 < argc, i);
770 config.zoom = parse_float(argv[++i]);
771 } else if (next == "--load") {
772 expect(i + 1 < argc, i);
773 config.load_cost = parse_int(argv[++i]);
774 } else if (next == "--store") {
775 expect(i + 1 < argc, i);
776 config.store_cost = parse_int(argv[++i]);
777 } else if (next == "--strides") {
778 config.strides.clear();
779 while (i + 1 < argc) {
780 const char *next_arg = argv[i + 1];
781 if (next_arg[0] == '-' &&
782 next_arg[1] == '-') {
783 break;
784 }
785 expect(i + 2 < argc, i);
786 int x = parse_int(argv[++i]);
787 int y = parse_int(argv[++i]);
788 config.strides.emplace_back(x, y);
789 }
790 } else if (next == "--label") {
791 expect(i + 3 < argc, i);
792 char *func = argv[++i];
793 char *text = argv[++i];
794 int n = parse_int(argv[++i]);
795 FuncInfo &fi = funcs[func];
796 // A Label's position is relative to its Func's position;
797 // the --label flag has always expected an absolute position,
798 // so convert it to an offset.
799 Point offset = {config.pos.x - fi.config.pos.x, config.pos.y - fi.config.pos.y};
800 if (!labels_seen.count(func)) {
801 // If there is at least one --label specified for a Func,
802 // it overrides the entire previous std::set of labels, rather
803 // than simply appending.
804 fi.config.labels.clear();
805 labels_seen.insert(func);
806 }
807 fi.config.labels.emplace_back(text, offset, n);
808 } else if (next == "--rlabel") {
809 expect(i + 5 < argc, i);
810 char *func = argv[++i];
811 char *text = argv[++i];
812 int dx = parse_int(argv[++i]);
813 int dy = parse_int(argv[++i]);
814 int n = parse_int(argv[++i]);
815 FuncInfo &fi = funcs[func];
816 Point offset = {dx, dy};
817 if (!labels_seen.count(func)) {
818 // If there is at least one --label specified for a Func,
819 // it overrides the entire previous std::set of labels, rather
820 // than simply appending.
821 fi.config.labels.clear();
822 labels_seen.insert(func);
823 }
824 fi.config.labels.emplace_back(text, offset, n);
825 } else if (next == "--timestep") {
826 expect(i + 1 < argc, i);
827 globals.timestep = parse_int(argv[++i]);
828 } else if (next == "--decay") {
829 expect(i + 2 < argc, i);
830 globals.decay_factor_during_compute = parse_int(argv[++i]);
831 globals.decay_factor_after_compute = parse_int(argv[++i]);
832 } else if (next == "--hold") {
833 expect(i + 1 < argc, i);
834 globals.hold_frames = parse_int(argv[++i]);
835 } else if (next == "--uninit") {
836 expect(i + 3 < argc, i);
837 int r = parse_int(argv[++i]);
838 int g = parse_int(argv[++i]);
839 int b = parse_int(argv[++i]);
840 config.uninitialized_memory_color = ((b & 255) << 16) | ((g & 255) << 8) | (r & 255);
841 } else if (next == "--auto_layout") {
842 globals.auto_layout = true;
843 } else if (next == "--no-auto_layout") {
844 globals.auto_layout = false;
845 } else if (next == "--auto_layout_grid") {
846 expect(i + 2 < argc, i);
847 globals.auto_layout_grid.x = parse_int(argv[++i]);
848 globals.auto_layout_grid.y = parse_int(argv[++i]);
849 } else if (next == "--uninit_default") {
850 expect(i + 3 < argc, i);
851 int r = parse_int(argv[++i]);
852 int g = parse_int(argv[++i]);
853 int b = parse_int(argv[++i]);
854 globals.default_uninitialized_memory_color = ((b & 255) << 16) | ((g & 255) << 8) | (r & 255);
855 } else if (next == "--ignore_tags" || next == "--no-ignore_tags") {
856 // Already processed, just continue
857 } else if (next == "--verbose" || next == "--no-verbose") {
858 // Already processed, just continue
859 } else {
860 expect(false, i);
861 }
862 i++;
863 }

866// There are three layers - image data, an animation on top of
867// it, and text labels. These layers get composited.
868struct Surface {
869 const Point frame_size;
870 std::vector<uint32_t> image, anim, anim_decay, text_buf, blend;
871
872 // Composite a single pixel of 'over' over a single pixel of 'under', writing the result into dst.
873 // Note that under or over might be dst.
874 static void composite_one(const uint32_t *under, const uint32_t *over, uint32_t *dst) {
875 const uint32_t o = *over;
876 const uint8_t alpha = o >> 24;
877 // alpha is almost always 0 or 255.
878 if (alpha == 0) {
879 *dst = *under;
880 } else if (alpha == 255) {
881 *dst = o;
882 } else {
883 // TODO: this could be done using 64-bit ops more simply
884 const uint8_t *a = (const uint8_t *)under;
885 const uint8_t *b = (const uint8_t *)over;
886 uint8_t *d = (uint8_t *)dst;
887 d[0] = (alpha * b[0] + (255 - alpha) * a[0]) / 255;
888 d[1] = (alpha * b[1] + (255 - alpha) * a[1]) / 255;
889 d[2] = (alpha * b[2] + (255 - alpha) * a[2]) / 255;
890 d[3] = 255 - (((255 - a[3]) * (255 - alpha)) / 255);
891 }
892 }
893
894 void do_decay(int decay_factor, uint32_t *dst) {
895 if (decay_factor != 1) {
896 const uint32_t inv_d1 = (1 << 24) / std::max(1, decay_factor);
897 for (uint32_t *dst_end = dst + frame_elems(); dst < dst_end; ++dst) {
898 uint32_t color = *dst;
899 uint32_t rgb = color & 0x00ffffff;
900 uint32_t alpha = (color >> 24);
901 alpha *= inv_d1;
902 alpha &= 0xff000000;
903 *dst = alpha | rgb;
904 }
905 }
906 }
907
908 // TODO this doesn't bounds-check against frame_size
909 void do_draw_pixel(const float zoom, const int x, const int y, const uint32_t color, uint32_t *dst) {
910 const int izoom = (int)std::ceil(zoom);
911 const int y_advance = frame_size.x - izoom;
912 dst += frame_size.x * y + x;
913 for (int dy = 0; dy < izoom; dy++) {
914 for (int dx = 0; dx < izoom; dx++) {
915 *dst++ = color;
916 }
917 dst += y_advance;
918 }
919 }
920
921 // Fill a rectangle in dst with color.
922 // opaque RGB(1,1,1) is a "magic" color that means "fill with checkerboard".
923 // dst is assumed to point to the start of a frame_size buffer.
924 void fill_rect(int left, int top, int width, int height, uint32_t color, uint32_t *dst) {
925 const int x_min = std::max(left, 0);
926 const int x_end = std::min(left + width, frame_size.x);
927 const int y_min = std::max(top, 0);
928 const int y_end = std::min(top + height, frame_size.y);
929 const int y_stride = frame_size.x - (x_end - x_min);
930 dst += y_min * frame_size.x + x_min;
931 if (color == 0xff010101) {
932 for (int y = y_min; y < y_end; y++) {
933 for (int x = x_min; x < x_end; x++) {
934 const int check = ((x / 16) % 2) ^ ((y / 16) % 2);
935 *dst++ = check ? 0xff808080 : 0xffffffff;
936 }
937 dst += y_stride;
938 }
939 } else {
940 for (int y = y_min; y < y_end; y++) {
941 for (int x = x_min; x < x_end; x++) {
942 *dst++ = color;
943 }
944 dst += y_stride;
945 }
946 }
947 }
948
949 // Set all boxes corresponding to positions in a Func's allocation to
950 // the given color. Recursive to handle arbitrary
951 // dimensionalities. Used by begin and end realization events.
952 void do_fill_realization(uint32_t *dst, uint32_t color,
953 const FuncInfo &fi, const halide_trace_packet_t &p,
954 int current_dimension = 0, int x_off = 0, int y_off = 0) {
955 if (2 * current_dimension == p.dimensions) {
956 const int x_min = x_off * fi.config.zoom + fi.config.pos.x;
957 const int y_min = y_off * fi.config.zoom + fi.config.pos.y;
958 const int izoom = (int)std::ceil(fi.config.zoom);
959 fill_rect(x_min, y_min, izoom, izoom, color, dst);
960 } else {
961 const int *coords = p.coordinates();
962 const int min = coords[current_dimension * 2 + 0];
963 const int extent = coords[current_dimension * 2 + 1];
964 // If we don't have enough strides, assume subsequent dimensions have stride (0, 0)
965 const Point pt = current_dimension < (int)fi.config.strides.size() ? fi.config.strides.at(current_dimension) : Point{0, 0};
966 x_off += pt.x * min;
967 y_off += pt.y * min;
968 for (int i = 0; i < extent; i++) {
969 do_fill_realization(dst, color, fi, p, current_dimension + 1, x_off, y_off);
970 x_off += pt.x;
971 y_off += pt.y;
972 }
973 }
974 }

1001 void draw_text(const std::string &text, const Point &pos, uint32_t color, float h_scale = 1.0f) {
1002 uint32_t *dst = text_buf.data();
1003
1004 // Drop any alpha component of color
1005 color &= 0xffffff;
1006
1007 int c = -1;
1008 for (int chr : text) {
1009 ++c;
1010
1011 // We only handle a subset of ascii
1012 if (chr < 32 || chr >= 32 + inconsolata_char_count) {
1013 chr = 32;
1014 }
1015 chr -= 32;
1016
1017 const uint8_t *font_ptr = inconsolata_raw + chr * (inconsolata_char_width * inconsolata_char_height);
1018 const int h_scale_numerator = std::ceil(std::min(1.f, h_scale) * 256);
1019 for (int fy = 0; fy < inconsolata_char_height; fy++) {
1020 for (int fx = 0; fx < inconsolata_char_width; fx++) {
1021 int px = pos.x + (((inconsolata_char_width * c + fx) * h_scale_numerator) >> 8);
1022 int py = pos.y - inconsolata_char_height + fy + 1;
1023 if (px < 0 || px >= frame_size.x ||
1024 py < 0 || py >= frame_size.y) {
1025 continue;
1026 }
1027 dst[py * frame_size.x + px] = (font_ptr[fy * inconsolata_char_width + fx] << 24) | color;
1028 }
1029 }
1030 }
1031 }

1034 do_draw_pixel(zoom, x, y, color, anim.data());
1035 }

1038 do_draw_pixel(zoom, x, y, color, image.data());
1039 }

1042 do_fill_realization(image.data(), color, fi, p);
1043 }

1045 void composite() {
1046 // Composite text over anim over image
1047 uint32_t *anim_decay_px = anim_decay.data();
1048 uint32_t *anim_px = anim.data();
1049 uint32_t *image_px = image.data();
1050 uint32_t *text_px = text_buf.data();
1051 uint32_t *blend_px = blend.data();
1052 for (size_t i = 0; i < image.size(); i++) {
1053 // anim over anim_decay -> anim_decay
1054 composite_one(anim_decay_px, anim_px, anim_decay_px);
1055 // anim_decay over image -> blend
1056 composite_one(image_px, anim_decay_px, blend_px);
1057 // text over blend -> blend
1058 composite_one(blend_px, text_px, blend_px);
1059 anim_decay_px++;
1060 anim_px++;
1061 image_px++;
1062 text_px++;
1063 blend_px++;
1064 }
1065 }
1066
1067 void decay_animations(int decay_factor_after_compute, int decay_factor_during_compute) {
1068 // Decay the anim_decay
1069 do_decay(decay_factor_after_compute, anim_decay.data());
1070
1071 // Also decay the anim
1072 do_decay(decay_factor_during_compute, anim.data());
1073 }
1074
1075 void clear_animations() {
1076 std::fill(anim.begin(), anim.end(), 0);
1077 }

1080using FlagProcessor = std::function<void(VizState *state)>;
1081
1082int run(bool ignore_trace_tags, FlagProcessor flag_processor) {
1083 // State that determines how different funcs get drawn
1084 VizState state;
1085
1086 // halide_clock counts halide events. video_clock counts how many
1087 // of these events have been output. When halide_clock gets ahead
1088 // of video_clock, we emit a new frame.
1089 size_t halide_clock = 0, video_clock = 0;
1090 bool is_state_finalized = false;
1091 bool seen_global_config_tag = false;
1092
1093 std::unique_ptr<Surface> surface;
1094
1095 const std::function<void()> finalize_state = [&]() -> void {
1096 if (is_state_finalized) {
1097 return;
1098 }
1099
1100 is_state_finalized = true;
1101
1102 if (verbose) {
1103 std::ostringstream dumps;
1104 for (const auto &p : state.funcs) {
1105 const auto &fi = p.second;
1106 assert(fi.type_and_dim_valid);
1107 fi.type_and_dim.dump(dumps, p.first);
1108 }
1109 info() << dumps.str();
1110 }
1111
1112 // We wait until now to process the cmd-line args;
1113 // this allows us to override trace-tag specifications
1114 // via the commandline, which is handy for experimentations.
1115 flag_processor(&state);
1116
1117 // allocate the surface after all tags and flags are processed
1118 surface = std::unique_ptr<Surface>(new Surface(state.globals.frame_size));
1119
1120 if (state.globals.auto_layout_grid.x < 0 || state.globals.auto_layout_grid.y < 0) {
1121 int cells_needed = 0;
1122 for (const auto &p : state.funcs) {
1123 if (p.second.type_and_dim_valid) {
1124 cells_needed++;
1125 }
1126 }
1127 Point cell_size = best_cell_size(cells_needed, state.globals.frame_size.x, state.globals.frame_size.y);
1128 state.globals.auto_layout_grid.x = state.globals.frame_size.x / cell_size.x;
1129 state.globals.auto_layout_grid.y = state.globals.frame_size.y / cell_size.y;
1130 assert(state.globals.auto_layout_grid.x * state.globals.auto_layout_grid.y >= cells_needed);
1131 info() << "For cells_needed = " << cells_needed
1132 << " using " << state.globals.auto_layout_grid.x << "x" << state.globals.auto_layout_grid.y << " grid"
1133 << " with cells of size " << cell_size.x << "x" << cell_size.y;
1134 }
1135
1136 // If globals.default_uninitialized_memory_color was never set, init to black or checkerboard.
1137 if (state.globals.default_uninitialized_memory_color & 0xff000000) {
1138 if (state.globals.auto_layout) {
1139 // auto-layout defaults to checkerboard.
1140 state.globals.default_uninitialized_memory_color = 0x00010101;
1141 } else {
1142 // non-auto-layout defaults to black, to preserve existing look.
1143 state.globals.default_uninitialized_memory_color = 0x00000000;
1144 }
1145 }
1146
1147 do_auto_layout(state);
1148 finalize_func_config_values(state.globals, state.funcs);
1149 };
1150
1151 struct PipelineInfo {
1152 std::string name;
1153 int32_t id;
1154 };
1155 std::map<uint32_t, PipelineInfo> pipeline_info;
1156
1157 int layout_order = 0;
1158 std::list<std::pair<Label, int>> labels_being_drawn;
1159 size_t end_counter = 0;
1160 size_t packet_clock = 0;
1161 for (;;) {
1162 // Hold for some number of frames once the trace has finished.
1163 if (end_counter) {
1164 halide_clock += state.globals.timestep;
1165 if (end_counter >= (size_t)state.globals.hold_frames) {
1166 break;
1167 }
1168 }
1169
1170 if (halide_clock > video_clock) {
1171 assert(is_state_finalized);
1172
1173 const int64_t frame_bytes = surface->frame_elems() * sizeof(uint32_t);
1174
1175 while (halide_clock > video_clock) {
1176 // Always render text last, since it's on top of everything
1177 // and there's no need to re-render for every packet.
1178 for (auto it = labels_being_drawn.begin(); it != labels_being_drawn.end();) {
1179 const Label &label = it->first;
1180 int first_draw_clock = it->second;
1181 int frames_since_first_draw = (halide_clock - first_draw_clock) / state.globals.timestep;
1182 if (frames_since_first_draw < label.fade_in_frames) {
1183 uint32_t color = ((1 + frames_since_first_draw) * 255) / std::max(1, label.fade_in_frames);
1184 if (color > 255) {
1185 color = 255;
1186 }
1187 color *= 0x10101;
1188 surface->draw_text(label.text, label.pos, color, label.h_scale);
1189 ++it;
1190 } else {
1191 // Once we reach or exceed the final frame, draw at 100% opacity, then remove
1192 surface->draw_text(label.text, label.pos, 0xffffff, label.h_scale);
1193 it = labels_being_drawn.erase(it);
1194 }
1195 }
1196
1197 // Composite text over anim over image
1198 surface->composite();
1199
1200 // Dump the frame
1201 int64_t bytes_written = write(STDOUT_FILENO, surface->frame_data(), frame_bytes);
1202 if (bytes_written < frame_bytes) {
1203 fail() << "Could not write frame to stdout.";
1204 }
1205
1206 video_clock += state.globals.timestep;
1207
1208 surface->decay_animations(state.globals.decay_factor_after_compute, state.globals.decay_factor_during_compute);
1209 }
1210
1211 // Blank anim
1212 surface->clear_animations();
1213 }
1214
1215 // Read a tracing packet
1216 PacketAndPayload p;
1217 if (!p.read()) {
1218 end_counter++;
1219 continue;
1220 }
1221 packet_clock++;
1222
1223 // It's a pipeline begin/end event
1224 if (p.event == halide_trace_begin_pipeline) {
1225 pipeline_info[p.id] = {p.func(), p.id};
1226 continue;
1227 } else if (p.event == halide_trace_end_pipeline) {
1228 assert(pipeline_info.count(p.parent_id));
1229 pipeline_info.erase(p.parent_id);
1230 continue;
1231 } else if (p.event == halide_trace_tag) {
1232 // If there are trace tags, they will come immediately after the pipeline's
1233 // halide_trace_begin_pipeline but before any realizations.
1234 if (halide_clock != 0 || video_clock != 0) {
1235 // Messing with timestamp, framesize, etc partway thru
1236 // a visualization would be bad, but let's just warn
1237 // rather than fail.
1238 // TODO: May need to check parent_id here, as nested
1239 // pipelines called via define_extern could emit these.
1240 warn() << "trace_tags are only expected at the start of a visualization:"
1241 << " (" << p.trace_tag() << ") for func (" << p.func() << ")";
1242 }
1243 if (FuncConfig::match(p.trace_tag())) {
1244 if (ignore_trace_tags) {
1245 continue;
1246 }
1247 FuncConfig cfg(p.trace_tag());
1248 auto &fi = state.funcs[p.func()];
1249 fi.config = cfg;
1250 fi.config_valid = true;
1251 } else if (GlobalConfig::match(p.trace_tag())) {
1252 if (ignore_trace_tags) {
1253 continue;
1254 }
1255 if (seen_global_config_tag) {
1256 warn() << "saw multiple GlobalConfig trace_tags, some will be ignored.";
1257 }
1258 state.globals = GlobalConfig(p.trace_tag());
1259 seen_global_config_tag = true;
1260 } else if (FuncTypeAndDim::match(p.trace_tag())) {
1261 auto &fi = state.funcs[p.func()];
1262 fi.type_and_dim = FuncTypeAndDim(p.trace_tag());
1263 fi.type_and_dim_valid = true;
1264 fi.layout_order = layout_order++;
1265 } else {
1266 warn() << "Ignoring trace_tag: (" << p.trace_tag() << ") for func (" << p.func() << ")";
1267 }
1268 continue;
1269 }
1270
1271 finalize_state();
1272
1273 const PipelineInfo pipeline = pipeline_info[p.parent_id];
1274
1275 if (p.event == halide_trace_begin_realization ||
1276 p.event == halide_trace_produce ||
1277 p.event == halide_trace_consume) {
1278 assert(!pipeline_info.count(p.id));
1279 pipeline_info[p.id] = pipeline;
1280 } else if (p.event == halide_trace_end_realization ||
1281 p.event == halide_trace_end_produce ||
1282 p.event == halide_trace_end_consume) {
1283 assert(pipeline_info.count(p.parent_id));
1284 pipeline_info.erase(p.parent_id);
1285 }
1286
1287 std::string qualified_name = pipeline.name + ":" + p.func();
1288 if (state.funcs.find(qualified_name) == state.funcs.end()) {
1289 if (state.funcs.find(p.func()) != state.funcs.end()) {
1290 state.funcs[qualified_name] = state.funcs[p.func()];
1291 state.funcs.erase(p.func());
1292 } else {
1293 warn() << "ignoring func " << qualified_name << " event " << p.event << "; parent event " << p.parent_id << " " << pipeline.name;
1294 }
1295 }
1296
1297 // Draw the event
1298 FuncInfo &fi = state.funcs[qualified_name];
1299 if (!fi.config_valid) {
1300 continue;
1301 }
1302
1303 if (fi.stats.first_draw_time < 0) {
1304 fi.stats.first_draw_time = halide_clock;
1305
1306 for (const auto &label : fi.config.labels) {
1307 // Convert offset to absolute position before enqueuing
1308 Label l = label;
1309 l.pos.x += fi.config.pos.x;
1310 l.pos.y += fi.config.pos.y;
1311 labels_being_drawn.emplace_back(l, halide_clock);
1312 }
1313 }
1314
1315 if (fi.stats.first_packet_idx < 0) {
1316 fi.stats.first_packet_idx = packet_clock;
1317 fi.stats.qualified_name = qualified_name;
1318 }
1319
1320 switch (p.event) {
1321 case halide_trace_load:
1322 case halide_trace_store: {
1323 if (p.event == halide_trace_store) {
1324 // Stores take time proportional to the number of
1325 // items stored times the cost of the func.
1326 halide_clock += fi.config.store_cost * p.type.lanes;
1327 fi.stats.observe_store(p);
1328 } else {
1329 halide_clock += fi.config.load_cost * p.type.lanes;
1330 fi.stats.observe_load(p);
1331 }
1332
1333 // zero- or one-dimensional Funcs can have dimensions < strides.size().
1334 // This may seem confusing, so keep in mind:
1335 // fi.config.strides are provided by the --stride flag, so it can contain anything; i
1336 // if you don't specify them at all, they default to {{1,0},{0,1} (aka size=2).
1337 // So if we have excess strides, just ignore them.
1338 const int dims = std::min(p.dimensions / p.type.lanes, (int)fi.config.strides.size());
1339 const int *coords = p.coordinates();
1340 for (int lane = 0; lane < p.type.lanes; lane++) {
1341 // Compute the screen-space x, y coord to draw this.
1342 int x = fi.config.pos.x;
1343 int y = fi.config.pos.y;
1344 const float z = fi.config.zoom;
1345 for (int d = 0; d < dims; d++) {
1346 const int coord = d * p.type.lanes + lane;
1347 assert(coord < p.dimensions);
1348 const int a = coords[coord];
1349 const auto &stride = fi.config.strides[d];
1350 x += z * stride.x * a;
1351 y += z * stride.y * a;
1352 }
1353
1354 // The box to draw must be entirely on-screen
1355 if (y < 0 || y >= state.globals.frame_size.y ||
1356 x < 0 || x >= state.globals.frame_size.x ||
1357 y + z - 1 < 0 || y + z - 1 >= state.globals.frame_size.y ||
1358 x + z - 1 < 0 || x + z - 1 >= state.globals.frame_size.x) {
1359 continue;
1360 }
1361
1362 // Update one or more of the color channels of the
1363 // image layer in case it's a store or a load from
1364 // the input.
1365 if (p.event == halide_trace_store || fi.stats.num_realizations == 0 /* load from an input */) {
1366 // Get the old color, in case we're only
1367 // updating one of the color channels.
1368 uint32_t image_color = surface->get_image_pixel(x, y);
1369 double value = get_value_as<double>(p, lane);
1370
1371 // Normalize it.
1372 value = std::max(0.0, std::min(255.0, 255.0 * (value - fi.config.min) /
1373 (fi.config.max - fi.config.min)));
1374
1375 // Convert to 8-bit color.
1376 uint8_t int_value = (uint8_t)value;
1377
1378 if (fi.config.color_dim < 0) {
1379 // Grayscale
1380 image_color = (int_value * 0x00010101) | 0xff000000;
1381 } else {
1382 // Color
1383 uint32_t channel = coords[fi.config.color_dim * p.type.lanes + lane];
1384 uint32_t mask = ~(255 << (channel * 8));
1385 image_color &= mask;
1386 image_color |= int_value << (channel * 8);
1387 }
1388 surface->draw_image_pixel(fi.config.zoom, x, y, image_color);
1389 }
1390
1391 // Stores are orange, loads are blue.
1392 uint32_t color = p.event == halide_trace_load ? 0xffffdd44 : 0xff44ddff;
1393 surface->draw_anim_pixel(fi.config.zoom, x, y, color);
1394 }
1395 break;
1396 }
1397 case halide_trace_begin_realization:
1398 fi.stats.num_realizations++;
1399 surface->fill_realization(0xff000000 | fi.config.uninitialized_memory_color, fi, p);
1400 break;
1401 case halide_trace_end_realization:
1402 if (fi.config.blank_on_end_realization > 0) {
1403 surface->fill_realization(0, fi, p);
1404 }
1405 break;
1406 case halide_trace_produce:
1407 fi.stats.num_productions++;
1408 break;
1409 case halide_trace_end_produce:
1410 case halide_trace_consume:
1411 case halide_trace_end_consume:
1412 // Note that you can get nested pipeline begin/end events when you trace
1413 // something that has extern stages that are also Halide-being-traced;
1414 // these should just be ignored.
1415 case halide_trace_begin_pipeline:
1416 case halide_trace_end_pipeline:
1417 case halide_trace_tag:
1418 break;
1419 default:
1420 fail() << "Unknown tracing event code: " << p.event;
1421 }
1422 }
1423
1424 if (verbose) {
1425 info() << "Total number of Funcs: " << state.funcs.size();
1426
1427 // Dump this info at the end, since some is determined as we go
1428 std::ostringstream dumps;
1429 state.globals.dump(dumps);
1430 for (const auto &p : state.funcs) {
1431 const auto &fi = p.second;
1432 if (fi.config_valid) {
1433 fi.config.dump(dumps, p.first);
1434 }
1435 }
1436 info() << dumps.str();
1437
1438 // Print stats about the Func gleaned from the trace.
1439 std::vector<std::pair<std::string, FuncInfo>> funcs;
1440 for (const auto &p : state.funcs) {
1441 funcs.emplace_back(p);
1442 }
1443 struct by_first_packet_idx {
1444 bool operator()(const std::pair<std::string, FuncInfo> &a,
1445 const std::pair<std::string, FuncInfo> &b) const {
1446 return a.second.stats.first_packet_idx < b.second.stats.first_packet_idx;
1447 }
1448 };
1449 std::sort(funcs.begin(), funcs.end(), by_first_packet_idx());
1450 for (std::pair<std::string, FuncInfo> p : funcs) {
1451 p.second.stats.report();
1452 }
1453 }
1454
1455 return 0;
1456}
