14bool Packet::read_from_filedesc(FILE *fdesc) {
15 size_t header_size = sizeof(halide_trace_packet_t);
16 if (!Packet::read(this, header_size, fdesc)) {
17 return false;
18 }
19 size_t payload_size = size - header_size;
20 if (payload_size > sizeof(payload)) {
21 fprintf(stderr, "Payload larger than %d bytes in trace stream (%d)\n", (int)sizeof(payload), (int)payload_size);
22 abort();
23 return false;
24 }
25 if (!Packet::read(payload, payload_size, fdesc)) {
26 fprintf(stderr, "Unexpected EOF mid-packet");
27 return false;
28 }
29 return true;
30}

32bool Packet::read(void *d, size_t size, FILE *fdesc) {
33 uint8_t *dst = (uint8_t *)d;
34 if (!size) {
35 return true;
36 }
37 size_t s = fread(dst, 1, size, fdesc);
38 if (s != size) {
39 if (ferror(fdesc) || !feof(fdesc)) {
40 perror("Failed during read");
41 exit(-1);
42 }
43 return false; //EOF
44 }
45
46 return true;
47}
