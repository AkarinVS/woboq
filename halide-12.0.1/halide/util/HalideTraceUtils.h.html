13// Simple conversion. Note: assume that 'value' is aligned properly.
14template<typename T>
15T value_as(halide_type_t type, const halide_scalar_value_t &value) {
16 switch (type.code) {
17 case halide_type_int:
18 switch (type.bits) {
19 case 8:
20 return (T)value.u.i8;
21 case 16:
22 return (T)value.u.i16;
23 case 32:
24 return (T)value.u.i32;
25 case 64:
26 return (T)value.u.i64;
27 default:
28 bad_type_error(type);
29 }
30 break;
31 case halide_type_uint:
32 switch (type.bits) {
33 case 1:
34 return (T)value.u.b;
35 case 8:
36 return (T)value.u.u8;
37 case 16:
38 return (T)value.u.u16;
39 case 32:
40 return (T)value.u.u32;
41 case 64:
42 return (T)value.u.u64;
43 default:
44 bad_type_error(type);
45 }
46 break;
47 case halide_type_float:
48 switch (type.bits) {
49 case 32:
50 return (T)value.u.f32;
51 case 64:
52 return (T)value.u.f64;
53 default:
54 bad_type_error(type);
55 }
56 break;
57 default:
58 bad_type_error(type);
59 }
60 return (T)0;
61}

63// A struct representing a single Halide tracing packet.
64struct Packet : public halide_trace_packet_t {
65 // Not all of this will be used, but this
66 // is the max possible packet size we
67 // consider here.
68 uint8_t payload[4096];
69
70 int get_coord(int idx) const {
71 return coordinates()[idx];
72 }
73
74 template<typename T>
75 T get_value_as(int idx) const {
76 const uint8_t *val = (const uint8_t *)(value()) + idx * type.bytes();
77 // 'val' may not be aligned: memcpy it to an aligned local
78 // so that value_as<>() won't complain under sanitizers.
79 halide_scalar_value_t aligned_value;
80 // Only copy the number of bytes in the type: the stream isn't guaranteed
81 // to be padded to sizeof(halide_scalar_value_t).
82 memcpy(&aligned_value, val, type.bits / 8);
83 return value_as<T>(type, aligned_value);
84 }

95};
