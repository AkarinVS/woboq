5namespace {
6class StringParam : public Halide::Generator<StringParam> {
7public:
8 GeneratorParam<std::string> rpn{"rpn_expr", ""};
9
10 Output<Buffer<int>> output{"output", 2};
11
12 void generate() {
13 // Remove cmake extra skip characters if any exist.
14 const std::string value = Halide::Internal::replace_all(rpn.value(), "\\ ", " ");
15 std::vector<std::string> tokens = Halide::Internal::split_string(value, " ");
16 std::stack<Halide::Expr> exprs;
17 // Assume input is a valid RPN expression no checks for simplicity.
18 for (const std::string &token : tokens) {
19 bool is_op = (token == "+" || token == "-" || token == "*" || token == "/");
20 bool is_var = (token == "x" || token == "y");
21 if (is_var) {
22 if (token == "x") {
23 exprs.push(x);
24 } else {
25 exprs.push(y);
26 }
27 } else if (is_op) {
28 Halide::Expr a = exprs.top();
29 exprs.pop();
30 Halide::Expr b = exprs.top();
31 exprs.pop();
32 if (token == "+") {
33 exprs.push(a + b);
34 } else if (token == "-") {
35 exprs.push(a - b);
36 } else if (token == "*") {
37 exprs.push(a * b);
38 } else {
39 exprs.push(a / b);
40 }
41 } else {
42 // Numerical constant.
43 exprs.push(std::stoi(token));
44 }
45 }
46
47 output(x, y) = exprs.top();
48 }
49
50 Var x, y;
51};
52} // namespace
53
