5class Configure : public Halide::Generator<Configure> {
6public:
7 GeneratorParam<int> num_extra_buffer_inputs{"num_extra_buffer_inputs", 3};
8
9 Input<Buffer<>> input{"input"};
10 Input<int> bias{"bias"};
11
12 Output<Buffer<>> output{"output"};
13
14 void configure() {
15 configure_calls++;
16
17 // It's fine to examine GeneratorParams in the configure() method.
18 assert(num_extra_buffer_inputs == 3);
19
20 // Pointers returned by add_input() are managed by the Generator;
21 // user code must not free them. We can stash them in member variables
22 // as-is or in containers, like so:
23 for (int i = 0; i < num_extra_buffer_inputs; ++i) {
24 auto *extra = add_input<Buffer<>>("extra_" + std::to_string(i), UInt(8), 2);
25 extra_buffer_inputs.push_back(extra);
26 }
27
28 typed_extra_buffer_input = add_input<Buffer<int16_t>>("typed_extra_buffer_input", 2);
29
30 extra_func_input = add_input<Func>("extra_func_input", UInt(16), 3);
31
32 extra_scalar_input = add_input<int>("extra_scalar_input");
33
34 extra_dynamic_scalar_input = add_input<Expr>("extra_dynamic_scalar_input", Int(8));
35
36 extra_buffer_output = add_output<Buffer<>>("extra_buffer_output", Float(32), 3);
37
38 extra_func_output = add_output<Func>("extra_func_output", Float(64), 2);
39
40 // This is ok: you can't *examine* an Input or Output here, but you can call
41 // set_type() iff the type is unspecified. (This allows you to base the type on,
42 // e.g., the value in get_target(), or the value of any GeneratorParam.)
43 input.set_type(Int(32));
44 output.set_type(Int(32));
45
46 // Ditto for set_dimensions.
47 input.set_dimensions(3);
48 output.set_dimensions(3);
49
50 // Will fail: it is not legal to call set_type on an Input or Output that
51 // already has a type specified.
52 // bias.set_type(Int(32));
53
54 // Will fail: it is not legal to examine Inputs in the configure() method
55 // assert(input.dimensions() == 3);
56
57 // Will fail: it is not legal to examine Inputs in the configure() method
58 // Expr b = bias;
59 // assert(b.defined());
60
61 // Will fail: it is not legal to examine Outputs in the configure() method
62 // Func o = output;
63 // assert(output.defined());
64 }
65
66 void generate() {
67 assert(configure_calls == 1);
68
69 // Will fail: it is not legal to call set_type(), etc from anywhere but configure().
70 // input.set_type(Int(32));
71 // input.set_dimensions(3);
72
73 // Attempting to call add_input() outside of the configure method will fail.
74 // auto *this_will_fail = add_input<Buffer<>>("untyped_uint8", UInt(8), 2);
75
76 assert((*extra_dynamic_scalar_input).type() == Int(8));
77
78 Var x, y, c;
79
80 Expr extra_sum = cast<int>(0);
81 for (int i = 0; i < num_extra_buffer_inputs; ++i) {
82 extra_sum += cast<int>((*extra_buffer_inputs[i])(x, y));
83 }
84 extra_sum += cast<int>((*typed_extra_buffer_input)(x, y));
85 extra_sum += cast<int>((*extra_func_input)(x, y, c));
86 extra_sum += *extra_scalar_input + *extra_dynamic_scalar_input;
87
88 output(x, y, c) = input(x, y, c) + bias + extra_sum;
89
90 (*extra_buffer_output)(x, y, c) = cast<float>(output(x, y, c));
91 (*extra_func_output)(x, y) = cast<double>(output(x, y, 0));
92 }
93
94private:
95 int configure_calls = 0;
96
97 std::vector<Input<Buffer<>> *> extra_buffer_inputs;
98 Input<Buffer<int16_t>> *typed_extra_buffer_input;
99 Input<Func> *extra_func_input;
100 Input<int> *extra_scalar_input;
101 Input<Expr> *extra_dynamic_scalar_input;
102
103 Output<Buffer<>> *extra_buffer_output;
104 Output<Func> *extra_func_output;
105};
