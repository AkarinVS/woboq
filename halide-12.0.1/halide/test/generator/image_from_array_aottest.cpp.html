13// Returns the dimension sizes of a statically sized array from inner to outer.
14// E.g. ary[2][3][4] returns (4, 3, 2).
15
16template<typename T>
17vector<int> dimension_sizes(T const &, vector<int> sizes = vector<int>()) {
18 return sizes;
19}

21template<typename Array, size_t N>
22vector<int> dimension_sizes(Array (&vals)[N], vector<int> sizes = vector<int>()) {
23 sizes = dimension_sizes(vals[0], sizes);
24 sizes.push_back((int)N);
25 return sizes;
26}

55void compare_vectors(vector<int> const &under_test, vector<int> const &reference) {

91template<typename Image>
92void compare_extents(const Image &img, int reference, int dimension) {
93 if (img.dim(dimension).extent() == reference)
94 return;
95 cout << "Extent of dimension " << dimension << " of " << img.dimensions()
96 << " is " << img.dim(dimension).extent() << " instead of " << reference << "\n";
97 exit(-1);
98}

100template<typename Array, typename T = typename remove_all_extents<Array>::type>
101void verify_image_construction_from_array(Array &vals) {
102 Buffer<T> img(vals);
103 vector<int> sizes(dimension_sizes(vals));
104 int dims = (int)sizes.size();
105 int n = 1;
106 for (int i = 0; i < dims; ++i) {
107 compare_extents(img, sizes[i], i);
108 n *= sizes[i];
109 }
110 const void *reference = (const void *)first_of_array(vals);
111 const void *under_test = (const void *)(&img());
112 if (reference != under_test) {
113 cerr << "Start of array: " << reference
114 << "Start of image: " << under_test << "\n";
115 exit(-1);
116 }
117}
