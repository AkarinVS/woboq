10void my_error_handler(void *user_context, const char *message) {
11 // Don't use the word "error": if CMake sees it in the output
12 // from an add_custom_command() on Windows, it can decide that
13 // the command failed, regardless of error code.
14 printf("Saw: (%s)\n", message);
15}

17std::pair<std::string, bool> get_env_variable(char const *env_var_name) {
18 if (env_var_name) {
19 size_t read = 0;
20#ifdef _MSC_VER
21 char lvl[32];
22 if (getenv_s(&read, lvl, env_var_name) != 0) read = 0;
23#else
24 char *lvl = getenv(env_var_name);
25 read = (lvl) ? 1 : 0;
26#endif
27 if (read) {
28 return {std::string(lvl), true};
29 }
30 }
31 return {"", false};
32}

34bool use_noboundsquery_feature() {
35 std::string value;
36 bool read;
37 std::tie(value, read) = get_env_variable("HL_MULTITARGET_TEST_USE_NOBOUNDSQUERY_FEATURE");
38 if (!read) {
39 return false;
40 }
41 return std::stoi(value) != 0;
42}

44static std::atomic<int> can_use_count{0};
45
46int my_can_use_target_features(int count, const uint64_t *features) {
47 can_use_count += 1;
48 const int word = halide_target_feature_no_bounds_query / 64;
49 const int bit = halide_target_feature_no_bounds_query % 64;
50 if (features[word] & (1ULL << bit)) {
51 if (use_noboundsquery_feature()) {
52 return 1;
53 } else {
54 return 0;
55 }
56 }
57 return 1;
58}

60int main(int argc, char **argv) {
61 const int W = 32, H = 32;
62 Buffer<uint32_t> output(W, H);
63
64 halide_set_error_handler(my_error_handler);
65 halide_set_custom_can_use_target_features(my_can_use_target_features);
66
67 if (HalideTest::multitarget(output) != 0) {
68 printf("Error at multitarget\n");
69 return -1;
70 }
71
72 // Verify output.
73 for (int y = 0; y < H; y++) {
74 for (int x = 0; x < W; x++) {
75 const uint32_t expected = use_noboundsquery_feature() ? 0xdeadbeef : 0xf00dcafe;
76 const uint32_t actual = output(x, y);
77 if (actual != expected) {
78 printf("Error at %d, %d: expected %x, got %x\n", x, y, expected, actual);
79 return -1;
80 }
81 }
82 }
83
84 // halide_can_use_target_features() should be called exactly once, with the
85 // result cached; call this a few more times to verify.
86 for (int i = 0; i < 10; ++i) {
87 if (HalideTest::multitarget(output) != 0) {
88 printf("Error at multitarget\n");
89 return -1;
90 }
91 }
92 if (can_use_count != 1) {
93 printf("Error: halide_can_use_target_features was called %d times!\n", (int)can_use_count);
94 return -1;
95 }
96
97 {
98 // Verify that the multitarget wrapper code propagates nonzero error
99 // results back to the caller properly.
100 Buffer<uint8_t> bad_type(W, H);
101 int result = HalideTest::multitarget(bad_type);
102 if (result != halide_error_code_bad_type) {
103 printf("Error: expected to fail with halide_error_code_bad_type (%d) but actually got %d!\n", (int)halide_error_code_bad_type, result);
104 return -1;
105 }
106 }
107
108 printf("Saw %x for no_bounds_query=%d\n", output(0, 0), use_noboundsquery_feature());
109 printf("Success!\n");
110 return 0;
111}
