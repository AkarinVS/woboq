21int main(int argc, char **argv) {
22#if defined(TEST_OPENCL)
23 const auto *interface = halide_opencl_device_interface();
24 assert(interface->compute_capability != nullptr);
25 int major, minor;
26 int err = interface->compute_capability(nullptr, &major, &minor);
27 if (err != 0 || (major == 1 && minor < 2)) {
28 printf("[SKIP] OpenCl %d.%d is less than required 1.2.\n", major, minor);
29 return 0;
30 }
31
32 const int W = 32, H = 32;
33 Buffer<int> input(W, H);
34 for (int y = 0; y < input.height(); y++) {
35 for (int x = 0; x < input.width(); x++) {
36 input(x, y) = x + y;
37 }
38 }
39
40 // Explicitly copy data to the GPU.
41 input.set_host_dirty();
42
43 Buffer<int> output(W, H);
44
45 gpu_texture(input, output);
46
47 if (input.raw_buffer()->device_interface != halide_opencl_image_device_interface()) {
48 printf("Expected input to be copied to texture storage");
49 return -1;
50 }
51 if (output.raw_buffer()->device_interface != halide_opencl_image_device_interface()) {
52 printf("Expected output to be copied to texture storage");
53 return -1;
54 }
55
56 output.copy_to_host();
57
58 // Verify output.
59 for (int y = 0; y < H; y++) {
60 for (int x = 0; x < W; x++) {
61 if (input(x, y) * 2 != output(x, y)) {
62 printf("Error at %d, %d: %d != %d\n", x, y, input(x, y), output(x, y));
63 return -1;
64 }
65 }
66 }
67
68 printf("Success!\n");
69#else
70 printf("[SKIP] No OpenCL target enabled.\n");
71#endif
72 return 0;
73}
