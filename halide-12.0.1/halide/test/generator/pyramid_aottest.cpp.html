4#include "HalideBuffer.h"

7#include <vector>
8using std::vector;
9using namespace Halide::Runtime;
10
11int main(int argc, char **argv) {
12 Buffer<float> input(1024, 1024);
13
14 // Put some junk in the input. Keep it to small integers so the float averaging stays exact.
15 for (int y = 0; y < input.height(); y++) {
16 for (int x = 0; x < input.width(); x++) {
17 input(x, y) = (float)(((x * 17 + y) / 8) % 32);
18 }
19 }
20
21 vector<Buffer<float>> levels(10);
22
23 for (int l = 0; l < 10; l++) {
24 levels[l] = Buffer<float>(1024 >> l, 1024 >> l);
25 }
26
27 // Will throw a compiler error if we didn't compile the generator with 10 levels.
28 pyramid(input,
29 levels[0], levels[1], levels[2], levels[3], levels[4],
30 levels[5], levels[6], levels[7], levels[8], levels[9]);
31
32 // The bottom level should be the input
33 for (int y = 0; y < input.height(); y++) {
34 for (int x = 0; x < input.width(); x++) {
35 if (input(x, y) != levels[0](x, y)) {
36 printf("input(%d, %d) = %f, but levels[0](%d, %d) = %f\n",
37 x, y, input(x, y), x, y, levels[0](x, y));
38 return -1;
39 }
40 }
41 }
42
43 // The remaining levels should be averaging of the levels above them.
44 for (int l = 1; l < 10; l++) {
45 for (int y = 0; y < (input.height() >> l); y++) {
46 for (int x = 0; x < (input.width() >> l); x++) {
47 float correct = (levels[l - 1](2 * x, 2 * y) +
48 levels[l - 1](2 * x + 1, 2 * y) +
49 levels[l - 1](2 * x, 2 * y + 1) +
50 levels[l - 1](2 * x + 1, 2 * y + 1)) /
51 4;
52 float actual = levels[l](x, y);
53 if (correct != actual) {
54 printf("levels[%d](%d, %d) = %f instead of %f\n",
55 l, x, y, actual, correct);
56 return -1;
57 }
58 }
59 }
60 }
61
62 printf("Success!\n");
63 return 0;
64}
