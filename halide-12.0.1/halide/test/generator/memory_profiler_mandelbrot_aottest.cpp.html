19const int num_launcher_tasks = 10000;
20
21const int width = 100;
22const int height = 30;
23const int iters = 20;
24
25// Mandelbrot tiles by 8x8 and vectorizes x by 4
26const int tile_x = 8, tile_y = 8, vectorize = 4;
27
28// Expected stack size for argmin
29const int argmin_stack_peak = vectorize * sizeof(uint8_t) + vectorize * sizeof(int32_t);
30
31// Expected heap size for mandelbrot
32const int y_niters = (height + tile_y - 1) / tile_y;
33const int x_niters = (width + tile_x - 1) / tile_x;
34const int mandelbrot_n_mallocs = 2 * y_niters * x_niters * num_launcher_tasks;
35const uint64_t mandelbrot_heap_per_iter = 2 * tile_x * tile_y * 4 * (iters + 1); // Heap per iter for one task
36const uint64_t mandelbrot_heap_total = mandelbrot_heap_per_iter * y_niters * x_niters * num_launcher_tasks;
37
38void validate(halide_profiler_state *s) {
39 for (halide_profiler_pipeline_stats *p = s->pipelines; p;
40 p = (halide_profiler_pipeline_stats *)(p->next)) {
41 assert(p->num_allocs == mandelbrot_n_mallocs);
42 assert(p->memory_total == mandelbrot_heap_total);
43
44 assert(mandelbrot_heap_per_iter <= p->memory_peak);
45 assert(p->memory_peak <= mandelbrot_heap_total);
46
47 for (int i = 0; i < p->num_funcs; i++) {
48 halide_profiler_func_stats *fs = p->funcs + i;
49 if (strncmp(fs->name, "argmin", 6) == 0) {
50 assert(fs->stack_peak == argmin_stack_peak);
51 } else if (strncmp(fs->name, "mandelbrot", 10) == 0) {
52 assert(mandelbrot_heap_per_iter <= fs->memory_peak);
53 assert(fs->memory_peak <= mandelbrot_heap_total);
54
55 assert(fs->num_allocs == mandelbrot_n_mallocs);
56 assert(fs->memory_total == mandelbrot_heap_total);
57 }
58 }
59 }
60}

62int launcher_task(void *user_context, int index, uint8_t *closure) {
63 Buffer<int> output(width, height);
64 float fx = cos(index / 10.0f), fy = sin(index / 10.0f);
65 memory_profiler_mandelbrot(-2.0f, 2.0f, -1.4f, 1.4f, fx, fy, iters,
66 output.width(), output.height(), output);
67
68 return 0;
69}

73int main(int argc, char **argv) {
74 // Hijack halide's runtime to run a bunch of instances of this function
75 // in parallel.
76 printf("Running memory profiler comparison test\n");
77 printf("mandelbrot expected value\n nmalocs (all tasks): %d, heap/iter "
78 "(per task): %d K, heap total (all tasks): %d K\n",
79 mandelbrot_n_mallocs,
80 (int)(mandelbrot_heap_per_iter / 1024),
81 (int)(mandelbrot_heap_total / 1024));
82 printf("argmin expected value\n stack peak: %d\n", argmin_stack_peak);
83 printf("\n");
84
85 halide_do_par_for(nullptr, launcher_task, 0, num_launcher_tasks, nullptr);
86
87 halide_profiler_state *state = halide_profiler_get_state();
88 assert(state != nullptr);
89
90 validate(state);
91
92 printf("Success!\n");
93 return 0;
94}
