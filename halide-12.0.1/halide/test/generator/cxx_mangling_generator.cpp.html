36class CPlusPlusNameManglingGenerator : public Halide::Generator<CPlusPlusNameManglingGenerator> {
37public:
38 // Use all the parameter types to make sure mangling works for each of them.
39 // TODO: verify this provides full coverage.
40 Input<Buffer<uint8_t>> input{"input", 1};
41 Input<int8_t> offset_i8{"offset_i8", 0};
42 Input<uint8_t> offset_u8{"offset_u8", 0};
43 Input<int16_t> offset_i16{"offset_i16", 0};
44 Input<uint16_t> offset_u16{"offset_u16", 0};
45 Input<int32_t> offset_i32{"offset_i32", 0};
46 Input<uint32_t> offset_u32{"offset_u32", 0};
47 Input<int64_t> offset_i64{"offset_i64", 0};
48 Input<uint64_t> offset_u64{"offset_u64", 0};
49
50 Input<bool> scale_direction{"scale_direction", 1};
51 Input<float> scale_f{"scale_f", 0};
52 Input<double> scale_d{"scale_d", 0};
53 Input<int32_t *> ptr{"ptr", 0};
54 Input<int32_t const *> const_ptr{"const_ptr", 0};
55 Input<void *> void_ptr{"void_ptr", 0};
56 Input<void const *> const_void_ptr{"const_void_ptr", 0};
57 // 'string' is just a convenient struct-like thing that isn't special
58 // cased by Halide; it will be generated as a void* (but const-ness
59 // should be preserved).
60 Input<std::string *> string_ptr{"string_ptr", 0};
61 Input<std::string const *> const_string_ptr{"const_string_ptr", 0};
62
63 // Test some manually-registered types. These won't be void *.
64 Input<const my_namespace::my_class *> const_my_class_ptr{"const_my_class_ptr", 0};
65 Input<const my_namespace::my_subnamespace::my_struct *> const_my_struct_ptr{"const_my_struct_ptr", 0};
66 Input<const my_union *> const_my_union_ptr{"const_my_union_ptr", 0};
67
68 Output<Buffer<double>> output{"output", 1};
69
70 void generate() {
71 assert(get_target().has_feature(Target::CPlusPlusMangling));
72 Var x("x");
73
74 Expr offset = offset_i8 + offset_u8 + offset_i16 + offset_u16 +
75 offset_i32 + offset_u32 + offset_i64 + offset_u64 +
76 extract_value_global(ptr) + extract_value_ns(const_ptr);
77
78 // No significance to the calculation here.
79 output(x) = select(scale_direction, (input(x) * scale_f + offset) / scale_d,
80 (input(x) * scale_d + offset) / scale_f);
81 }
82
83 void schedule() {
84 input.set_estimates({{0, 100}});
85 offset_i8.set_estimate(0);
86 offset_u8.set_estimate(0);
87 offset_i16.set_estimate(0);
88 offset_u16.set_estimate(0);
89 offset_i32.set_estimate(0);
90 offset_u32.set_estimate(0);
91 offset_i64.set_estimate(0);
92 offset_u64.set_estimate(0);
93 scale_direction.set_estimate(1);
94 scale_f.set_estimate(0);
95 scale_d.set_estimate(0);
96 ptr.set_estimate(nullptr);
97 const_ptr.set_estimate(nullptr);
98 void_ptr.set_estimate(nullptr);
99 const_void_ptr.set_estimate(nullptr);
100 string_ptr.set_estimate(nullptr);
101 const_string_ptr.set_estimate(nullptr);
102 const_my_class_ptr.set_estimate(nullptr);
103 const_my_struct_ptr.set_estimate(nullptr);
104 const_my_union_ptr.set_estimate(nullptr);
105 output.set_estimates({{0, 100}});
106 }
107};
