24 halide_buffer_t in = {0}, out = {0};
25 halide_dimension_t shape[] = {{0, 64, 1},
26 {0, 123, 64}};
27
28 in.host = (uint8_t *)malloc(64 * 123 * 4);
29 in.type = halide_type_of<int>();
30 in.dim = shape;
31 in.dimensions = 2;
32
33 out.host = (uint8_t *)malloc(64 * 123 * 4);
34 out.type = halide_type_of<int>();
35 out.dim = shape;
36 out.dimensions = 2;
37
38 // First, a successful run.
39 int result = error_codes(&in, 64, &out);
40 int correct = halide_error_code_success;
41 check(result, correct);
42
43 // Passing 50 as the second arg violates the call to Func::bound
44 // in the generator
45 result = error_codes(&in, 50, &out);
46 correct = halide_error_code_explicit_bounds_too_small;
47 check(result, correct);
48
49 // Would read out of bounds on the input
50 halide_dimension_t smaller[] = {{0, 50, 1},
51 {0, 123, 64}};
52 in.dim = smaller;
53 result = error_codes(&in, 64, &out);
54 correct = halide_error_code_access_out_of_bounds;
55 check(result, correct);
56 in.dim = shape;
57
58 // buffer extent negative, but in a way that doesn't trigger oob checks
59 {
60 halide_dimension_t bad_shape[] = {{0, 64, 1},
61 {0, -123, 64}};
62 halide_buffer_t i = in, o = out;
63 i.dim = bad_shape;
64 o.dim = bad_shape;
65
66 result = error_codes(&i, 0, &o);
67 correct = halide_error_code_buffer_extents_negative;
68 check(result, correct);
69 }
70
71 // Input buffer larger than 2GB
72 halide_dimension_t huge[] = {{0, 10000000, 1},
73 {0, 10000000, 64}};
74 in.dim = huge;
75 result = error_codes(&in, 64, &out);
76 correct = halide_error_code_buffer_extents_too_large;
77 check(result, correct);
78 in.dim = shape;
79
80 // Input buffer requires addressing math that would overflow 32 bits.
81 halide_dimension_t huge_stride[] = {{0, 64, 1},
82 {0, 123, 0x7fffffff}};
83 in.dim = huge_stride;
84 result = error_codes(&in, 64, &out);
85 correct = halide_error_code_buffer_allocation_too_large;
86 check(result, correct);
87 in.dim = shape;
88
89 // strides and extents are 32-bit signed integers. It's
90 // therefore impossible to make a halide_buffer_t that can address
91 // more than 2^31 * 2^31 * dimensions elements, which is less
92 // than 2^63, so there's no way a Halide pipeline can return
93 // the above two error codes in 64-bit code.
94
95 // stride[0] is constrained to be 1
96 halide_dimension_t wrong_stride[] = {{0, 64, 2},
97 {0, 123, 64}};
98 in.dim = wrong_stride;
99 result = error_codes(&in, 64, &out);
100 correct = halide_error_code_constraint_violated;
101 check(result, correct);
102 in.dim = shape;
103
104 // The second argument is supposed to be between 0 and 64.
105 result = error_codes(&in, -23, &out);
106 correct = halide_error_code_param_too_small;
107 check(result, correct);
108
109 shape[0].extent = 108;
110 result = error_codes(&in, 108, &out);
111 correct = halide_error_code_param_too_large;
112 check(result, correct);
113 shape[0].extent = 64;
114
115 // You can't pass nullptr as a halide_buffer_t argument.
116 result = error_codes(nullptr, 64, &out);
117 correct = halide_error_code_buffer_argument_is_null;
118 check(result, correct);
119
120 // Violate the custom requirement that the height of the input is 123
121 halide_dimension_t too_tall[] = {{0, 64, 1},
122 {0, 200, 64}};
123 in.dim = too_tall;
124 result = error_codes(&in, 64, &out);
125 correct = halide_error_code_requirement_failed;
126 check(result, correct);
127 in.dim = shape;
128
129 printf("Success!\n");
130 return 0;
131}
