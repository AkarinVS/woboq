26void test(Buffer<float> (*factory)(int w, int h, int c)) {
27 Buffer<float> input = factory(W, H, 3);
28 input.for_each_element([&](int x, int y, int c) {
29 // Just an arbitrary color pattern with enough variation to notice the blur
30 if (c == 0) {
31 input(x, y, c) = ((x % 7) + (y % 3)) / 255.f;
32 } else if (c == 1) {
33 input(x, y, c) = (x + y) / 255.f;
34 } else {
35 input(x, y, c) = ((x * 5) + (y * 2)) / 255.f;
36 }
37 });
38 Buffer<float> output = factory(W, H, 3);
39
40 printf("Evaluating output over %d x %d\n", W, H);
41 blur2x2(input, W, H, output);
42
43#if RUN_BENCHMARKS
44 double t = Halide::Tools::benchmark(10, 100, [&]() {
45 blur2x2(input, W, H, output);
46 });
47 const float megapixels = (W * H) / (1024.f * 1024.f);
48 printf("Benchmark: %d %d -> %f mpix/s\n", W, H, megapixels / t);
49#endif
50}
