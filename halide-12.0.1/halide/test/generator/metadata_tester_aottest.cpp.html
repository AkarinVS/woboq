16inline std::ostream &operator<<(std::ostream &o, const halide_type_t &type) {
17 if (type.code == halide_type_uint && type.bits == 1) {
18 o << "bool";
19 } else {
20 assert(type.code >= 0 && type.code <= 3);
21 static const char *const names[4] = {"int", "uint", "float", "handle"};
22 o << names[type.code] << (int)type.bits;
23 }
24 if (type.lanes > 1) {
25 o << "x" << (int)type.lanes;
26 }
27 return o;

30inline constexpr int halide_type_code(halide_type_code_t code, int bits) {
31 return (((int)code) << 8) | bits;

34struct typed_scalar {
35 halide_type_t type;
36 halide_scalar_value_t value;

38 typed_scalar(const halide_type_t &t, const halide_scalar_value_t &v)
39 : type(t), value(v) {
40 }
41
42 bool operator==(const typed_scalar &that) const {
43 if (this->type != that.type) {
44 std::cerr << "Mismatched types\n";
45 exit(-1);
46 }
47 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
48 case halide_type_code(halide_type_float, 32):
49 return value.u.f32 == that.value.u.f32;
50 case halide_type_code(halide_type_float, 64):
51 return value.u.f64 == that.value.u.f64;
52 case halide_type_code(halide_type_int, 8):
53 return value.u.i8 == that.value.u.i8;
54 case halide_type_code(halide_type_int, 16):
55 return value.u.i16 == that.value.u.i16;
56 case halide_type_code(halide_type_int, 32):
57 return value.u.i32 == that.value.u.i32;
58 case halide_type_code(halide_type_int, 64):
59 return value.u.i64 == that.value.u.i64;
60 case halide_type_code(halide_type_uint, 1):
61 return value.u.b == that.value.u.b;
62 case halide_type_code(halide_type_uint, 8):
63 return value.u.u8 == that.value.u.u8;
64 case halide_type_code(halide_type_uint, 16):
65 return value.u.u16 == that.value.u.u16;
66 case halide_type_code(halide_type_uint, 32):
67 return value.u.u32 == that.value.u.u32;
68 case halide_type_code(halide_type_uint, 64):
69 return value.u.u64 == that.value.u.u64;
70 case halide_type_code(halide_type_handle, 64):
71 return value.u.handle == that.value.u.handle;
72 default:
73 std::cerr << "Unsupported type\n";
74 exit(-1);
75 break;
76 }
77 }
78
79 bool operator!=(const typed_scalar &that) const {
80 return !(*this == that);
81 }

130#define EXPECT_EQ(exp, act) \
131 do { \
132 if ((exp) != (act)) { \
133 std::cerr << #exp << " == " << #act << ": Expected " << exp << ", Actual " << act << "\n"; \
134 exit(-1); \
135 } \
136 } while (0);
137
138#define EXPECT_STREQ(exp, act) \
139 EXPECT_EQ(std::string(exp), std::string(act))
140
141#define EXPECT_TYPE_AND_SCALAR_EQ(etype, exp, atype, act) \
142 EXPECT_EQ(typed_scalar((etype), (exp)), typed_scalar((atype), (act)))
143
144#define EXPECT_TYPE_AND_SCALAR_PTR_EQ(etype, exp_ptr, atype, act_ptr) \
145 do { \
146 if ((exp_ptr) && (act_ptr)) { \
147 EXPECT_TYPE_AND_SCALAR_EQ((etype), *(exp_ptr), (atype), *(act_ptr)); \
148 } else if (!(exp_ptr) && !(act_ptr)) { \
149 EXPECT_EQ(etype, atype); \
150 } else { \
151 std::cerr << "One null, one non-null\n"; \
152 exit(-1); \
153 } \
154 } while (0);

156void match_argument(const halide_filter_argument_t &e, const halide_filter_argument_t &a) {
157 EXPECT_STREQ(e.name, a.name);
158 EXPECT_EQ(e.dimensions, a.dimensions);
159 EXPECT_EQ(e.kind, a.kind);
160 EXPECT_EQ(e.type.code, a.type.code);
161 EXPECT_EQ(e.type.bits, a.type.bits);
162 EXPECT_EQ(e.dimensions, a.dimensions);
163 EXPECT_TYPE_AND_SCALAR_PTR_EQ(e.type, e.scalar_def, a.type, a.scalar_def);
164 EXPECT_TYPE_AND_SCALAR_PTR_EQ(e.type, e.scalar_min, a.type, a.scalar_min);
165 EXPECT_TYPE_AND_SCALAR_PTR_EQ(e.type, e.scalar_max, a.type, a.scalar_max);
166 EXPECT_TYPE_AND_SCALAR_PTR_EQ(e.type, e.scalar_estimate, a.type, a.scalar_estimate);
167
168 // we want to treat
169 // buffer_estimates == nullptr
170 // as equivalent to
171 // buffer_estimates == { nullptr, nullptr, ... nullptr },
172 // so do a little pre-flighting to simplify things.
173
174 int64_t const *const *eb = e.buffer_estimates;
175 int64_t const *const *ab = a.buffer_estimates;
176 int e_null = 0, a_null = 0;
177 for (int i = 0; i < e.dimensions * 2; ++i) {
178 if (eb && !eb[i]) e_null++;
179 if (ab && !ab[i]) a_null++;
180 }
181 if (e_null == e.dimensions * 2) eb = nullptr;
182 if (a_null == a.dimensions * 2) ab = nullptr;
183
184 EXPECT_EQ((eb != nullptr), (ab != nullptr));
185 if (eb && ab) {
186 for (int i = 0; i < e.dimensions * 2; ++i) {
187 EXPECT_TYPE_AND_SCALAR_PTR_EQ(
188 halide_type_t(halide_type_int, 64),
189 (const halide_scalar_value_t *)eb[i],
190 halide_type_t(halide_type_int, 64),
191 (const halide_scalar_value_t *)ab[i]);
192 }
193 }
194}

196template<typename Type>
197Buffer<Type> make_image() {
198 Buffer<Type> im(kSize, kSize, 3);
199 for (int x = 0; x < kSize; x++) {
200 for (int y = 0; y < kSize; y++) {
201 for (int c = 0; c < 3; c++) {
202 im(x, y, c) = static_cast<Type>(x + y + c);
203 }
204 }
205 }
206 return im;
207}

209void verify(const Buffer<uint8_t> &input,
210 const Buffer<float> &output0,
211 const Buffer<float> &output1,
212 const Buffer<float> &output_scalar,
213 const Buffer<float> &output_array0,
214 const Buffer<float> &output_array1,
215 const Buffer<float> &untyped_output_buffer,
216 const Buffer<float> &tupled_output_buffer0,
217 const Buffer<int32_t> &tupled_output_buffer1) {
218 if (output_scalar.dimensions() != 0) {
219 std::cerr << "output_scalar should be zero-dimensional\n";
220 exit(-1);
221 }
222 if (output_scalar() != 1234.25f) {
223 std::cerr << "output_scalar value is wrong (" << output_scalar() << "\n";
224 exit(-1);
225 }
226 for (int x = 0; x < kSize; x++) {
227 for (int y = 0; y < kSize; y++) {
228 for (int c = 0; c < 3; c++) {
229 const float expected0 = static_cast<float>(input(x, y, c) + 1);
230 const float expected1 = expected0 + 1;
231 const float actual0 = output0(x, y, c);
232 const float actual1 = output1(x, y, c);
233 if (expected0 != actual0) {
234 std::cerr << "img0[" << x << "," << y << "," << c << "] = " << actual0 << ", expected " << expected0 << "\n";
235 exit(-1);
236 }
237 if (expected1 != actual1) {
238 std::cerr << "img1[" << x << "," << y << "," << c << "] = " << actual1 << ", expected " << expected1 << "\n";
239 exit(-1);
240 }
241 if (output_array0(x, y, c) != 1.5f) {
242 std::cerr << "output_array0[" << x << "," << y << "," << c << "] = " << output_array0(x, y, c) << ", expected " << 1.5f << "\n";
243 exit(-1);
244 }
245 if (output_array1(x, y, c) != 3.0f) {
246 std::cerr << "output_array1[" << x << "," << y << "," << c << "] = " << output_array1(x, y, c) << ", expected " << 2.0f << "\n";
247 exit(-1);
248 }
249 if (untyped_output_buffer(x, y, c) != expected1) {
250 std::cerr << "untyped_output_buffer[" << x << "," << y << "," << c << "] = " << untyped_output_buffer(x, y, c) << ", expected " << expected1 << "\n";
251 exit(-1);
252 }
253 if (tupled_output_buffer0(x, y, c) != expected1) {
254 std::cerr << "tupled_output_buffer0[" << x << "," << y << "," << c << "] = " << tupled_output_buffer0(x, y, c) << ", expected " << expected1 << "\n";
255 exit(-1);
256 }
257 }
258 }

349constexpr int64_t NO_VALUE = (int64_t)0xFFFFFFFFFFFFFFFF;
350
351int64_t const *const *make_int64_array(const std::vector<int64_t> &v) {
352 using int64_t_ptr = int64_t *;
353 int64_t **s = new int64_t_ptr[v.size()]();
354 for (size_t i = 0; i < v.size(); ++i) {
355 s[i] = (v[i] == NO_VALUE) ? nullptr : new int64_t(v[i]);
356 }
357 return s;
358}
