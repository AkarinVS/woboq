14float float_from_bits(uint32_t bits) {

32int main() {
33 uint16_t inputs[] = {
34 0x0000, // +ve zero
35 0x8000, // -ve zero
36 0x7c00, // +ve infinity
37 0xfc00, // -ve infinity
38 0x7e00, // quiet NaN
39 0x7bff, // Largest +ve normal number
40 0xfbff, // Smallest -ve normal number
41 0x0001, // Smallest +ve subnormal number
42 0x8001, // Largest -ve subnormal number
43 0x0002, // 2nd smallest +ve subnormal number
44 0x8002, // 2nd largest -ve subnormal number
45 0x0003, // 3rd smallest subnormal number
46 0x03ff, // Largest subnormal
47 0x03fe, // 2nd largest subnormal
48 0x3c00, // 1.0
49 0xbc00 // -1.0
50 };
51
52 // Unfortunately we can't use the C99 hex float format
53 // here because MSVC doesn't support them
54 float expectedF[] = {
55 0.0f,
56 -0.0f,
57 std::numeric_limits<float>::infinity(),
58 -std::numeric_limits<float>::infinity(),
59 std::numeric_limits<float>::quiet_NaN(),
60 65504.0f,
61 -65504.0f,
62 (1.0f) / (1 << 24),
63 (-1.0f) / (1 << 24),
64 (1.0f) / (1 << 23), // 0x1.000000p-23
65 (-1.0f) / (1 << 23), // -0x1.000000p-23
66 (1.5f) / (1 << 23), // 0x1.800000p-23
67 float_from_bits(0x387fc000), //0x1.ff8000p-15,
68 float_from_bits(0x387f8000), // 0x1.ff0000p-15,
69 1.0f,
70 -1.0f};
71
72 double expectedD[] = {
73 0.0,
74 -0.0,
75 std::numeric_limits<double>::infinity(),
76 -std::numeric_limits<double>::infinity(),
77 std::numeric_limits<double>::quiet_NaN(),
78 65504.0,
79 -65504.0,
80 (1.0) / (1 << 24),
81 (-1.0) / (1 << 24),
82 (1.0) / (1 << 23), // 0x1.000000000000p-23
83 (-1.0) / (1 << 23), // -0x1.000000000000p-23
84 (1.5) / (1 << 23), // 0x1.800000000000p-23
85 double_from_bits(0x3f0ff80000000000), // 0x1.ff8000000000p-15,
86 double_from_bits(0x3f0ff00000000000), // 0x1.ff0000000000p-15,
87 1.0,
88 -1.0};
89
90 h_assert(sizeof(inputs) / sizeof(uint16_t) == sizeof(expectedF) / sizeof(float),
91 "size of half array doesn't match float array");
92 h_assert(sizeof(inputs) / sizeof(uint16_t) == sizeof(expectedD) / sizeof(double),
93 "size of half array doesn't match double array");
94
95 for (unsigned int index = 0; index < sizeof(inputs) / sizeof(uint16_t); ++index) {
96 uint16_t in = inputs[index];
97 union {
98 float asFloat;
99 uint32_t asUInt;
100 } outF;
101 outF.asFloat = halide_float16_bits_to_float(in);
102 union {
103 double asDouble;
104 uint64_t asUInt;
105 } outD;
106 outD.asDouble = halide_float16_bits_to_double(in);
107
108 union {
109 float asFloat;
110 uint32_t asUInt;
111 } expectedFValue;
112 expectedFValue.asFloat = expectedF[index];
113
114 union {
115 double asDouble;
116 uint64_t asUInt;
117 } expectedDValue;
118 expectedDValue.asDouble = expectedD[index];
119
120 // Compare bits because NaN in not comparable
121 h_assert(outF.asUInt == expectedFValue.asUInt, "Failed to match on convert to float");
122 h_assert(outD.asUInt == expectedDValue.asUInt, "Failed to match on convert to double");
123 }
124 printf("Success!\n");
125 return 0;
126}
