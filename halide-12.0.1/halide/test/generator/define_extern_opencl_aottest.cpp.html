62cl_int init_extern_program() {
63 cl_int error;
64
65 cl_context ocl_ctx = nullptr;
66 cl_command_queue ocl_q = nullptr;
67 int halide_error = halide_acquire_cl_context(nullptr, &ocl_ctx, &ocl_q);
68 if (halide_error != 0) {
69 printf("halide_acquire_cl_context failed (%d).\n", halide_error);
70 return halide_error;
71 }
72
73 const char *ocl_source = "__kernel void add42(__global const int *in, __global int *out) { out[get_global_id(0)] = in[get_global_id(0)] + 42; }";
74 const char *sources[] = {ocl_source};
75 ocl_program = clCreateProgramWithSource(ocl_ctx, 1, &sources[0], nullptr, &error);
76 if (error != CL_SUCCESS) {
77 halide_release_cl_context(nullptr);
78 printf("clCreateProgramWithSource failed (%d).\n", error);
79 return error;
80 }
81
82 const cl_uint maxDevices = 4;
83 cl_device_id devices[maxDevices];
84 size_t actual_size = 0;
85 error = clGetContextInfo(ocl_ctx, CL_CONTEXT_DEVICES, sizeof(devices), devices, &actual_size);
86 if (error != CL_SUCCESS) {
87 printf("clGetContextInfo failed (%d).\n", error);
88 halide_release_cl_context(nullptr);
89 return error;
90 }
91
92 error = clBuildProgram(ocl_program, actual_size / sizeof(devices[0]), devices, nullptr, nullptr, nullptr);
93
94 halide_release_cl_context(nullptr);
95
96 if (error != CL_SUCCESS) {
97 if (error == CL_BUILD_PROGRAM_FAILURE) {
98 size_t msg_size;
99 clGetProgramBuildInfo(ocl_program, devices[0], CL_PROGRAM_BUILD_LOG, 0, nullptr, &msg_size);
100 char *msg = (char *)malloc(msg_size);
101 clGetProgramBuildInfo(ocl_program, devices[0], CL_PROGRAM_BUILD_LOG, msg_size, msg, nullptr);
102 printf("clBuildProgram failed. Error message %s\n", msg);
103 } else {
104 printf("clBuildProgram failed (%d).\n", error);
105 }
106 clReleaseProgram(ocl_program);
107 return error;
108 }
109
110 return error;
111}

114 clReleaseProgram(ocl_program);

119extern "C" int32_t gpu_input(halide_buffer_t *input, halide_buffer_t *output) {
120 if (input->is_bounds_query()) {
121 printf("gpu_input: Bounds query for output size %d\n", output->dim[0].extent);
122 input->type = output->type;
123 input->dimensions = 1;
124 input->dim[0] = output->dim[0];
125 return 0;
126 }
127
128 printf("gpu_input: Called to compute on size %d\n", input->dim[0].extent);
129 assert(output->device != 0);
130
131 halide_copy_to_device(nullptr, input, halide_opencl_device_interface());
132
133 cl_int error;
134 cl_context ocl_ctx = nullptr;
135 cl_command_queue ocl_q = nullptr;
136 int halide_result = halide_acquire_cl_context(nullptr, &ocl_ctx, &ocl_q);
137 assert(halide_result == 0);
138
139 cl_kernel kernel = clCreateKernel(ocl_program, "add42", &error);
140 if (error != CL_SUCCESS) {
141 printf("clCreateKernel failed (%d).\n", error);
142 }
143
144 size_t global_dim[1] = {(size_t)input->dim[0].extent};
145 size_t local_dim[1] = {16};
146
147 // Set args
148 void *in = reinterpret_cast<void *>(halide_opencl_get_cl_mem(nullptr, input));
149 error = clSetKernelArg(kernel, 0, sizeof(void *), &in);
150 assert(error == CL_SUCCESS);
151 void *out = reinterpret_cast<void *>(halide_opencl_get_cl_mem(nullptr, output));
152 error = clSetKernelArg(kernel, 1, sizeof(void *), &out);
153 assert(error == CL_SUCCESS);
154
155 printf("gpu_input: Calling clEnqueueNDRangeKernel.\n");
156 error = clEnqueueNDRangeKernel(ocl_q, kernel, 1, nullptr, global_dim, local_dim, 0, nullptr, nullptr);
157 printf("gpu_input: Returned from clEnqueueNDRangeKernel with result %d.\n", error);
158 assert(error == CL_SUCCESS);
159
160 clReleaseKernel(kernel);
161
162 halide_release_cl_context(nullptr);
163
164 // Return with the kernel queued. Halide guarantees any use of the
165 // device buffer will happen on the same queue or there will be a
166 // sync on the queue first.
167
168 return 0;
169}
