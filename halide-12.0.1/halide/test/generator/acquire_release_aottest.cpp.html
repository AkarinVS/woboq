117bool run_test() {
118 // Initialize the runtime specific GPU context.
119 if (!init_context()) {
120 return false;
121 }
122
123 // Everything else is a normal Halide program. The GPU runtime will call
124 // the above acquire/release functions to get the context instead of using
125 // its own internal context.
126 Buffer<float> input(W, H);
127 for (int y = 0; y < input.height(); y++) {
128 for (int x = 0; x < input.width(); x++) {
129 input(x, y) = (float)(x * y);
130 }
131 }
132
133 input.set_host_dirty(true);
134
135 Buffer<float> output(W, H);
136
137 acquire_release(input, output);
138
139 output.copy_to_host();
140
141 for (int y = 0; y < output.height(); y++) {
142 for (int x = 0; x < output.width(); x++) {
143 if (input(x, y) * 2.0f + 1.0f != output(x, y)) {
144 printf("Error at (%d, %d): %f != %f\n", x, y, input(x, y) * 2.0f + 1.0f,
145 output(x, y));
146 return false;
147 }
148 }
149 }
150
151 const halide_device_interface_t *interface = output.raw_buffer()->device_interface;
152
153 // We need to free our GPU buffers before destroying the context.
154 input.device_free();
155 output.device_free();
156
157 if (interface != nullptr) {
158 halide_device_release(nullptr, interface);
159
160 // Free the context we created.
161 destroy_context();
162 } else {
163 printf("Device interface is nullptr.\n");
164 }
165
166 printf("Success!\n");
167 return true;
168}
