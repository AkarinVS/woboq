15extern "C" DLLEXPORT int extern_stage(halide_buffer_t *input, int addend, halide_buffer_t *output) {
16 // Note the final output buffer argument is unused.
17 if (input->is_bounds_query()) {
18 for (int d = 0; d < 2; d++) {
19 // Request some range of the input buffer
20 input->dim[d].min = output->dim[d].min;
21 input->dim[d].extent = output->dim[d].extent;
22 }
23 } else if (!output->is_bounds_query()) {
24 int min_x = output->dim[0].min;
25 int max_x = min_x + output->dim[0].extent - 1;
26 int min_y = output->dim[1].min;
27 int max_y = min_y + output->dim[1].extent - 1;
28 for (int y = min_y; y <= max_y; y++) {
29 for (int x = min_x; x <= max_x; x++) {
30 int coords[2] = {x, y};
31 *(int *)output->address_of(coords) = *(int *)input->address_of(coords) + addend;
32 }
33 }
34 }
35
36 return 0;
37}
