26int my_halide_trace(void *user_context, const halide_trace_event_t *ev) {
27 if (ev->event == halide_trace_begin_realization) {
28 assert(ev->dimensions == 6);
29 int min_x = ev->coordinates[0], width = ev->coordinates[1];
30 int min_y = ev->coordinates[2], height = ev->coordinates[3];
31 int max_x = min_x + width - 1;
32 int max_y = min_y + height - 1;
33#if !RUN_BENCHMARKS
34 printf("Using %d x %d input tile over [%d - %d] x [%d - %d]\n", width, height, min_x, max_x,
35 min_y, max_y);
36#endif
37 assert(min_x >= 0 && min_y >= 0 && max_x < W && max_y < H);
38
39 // The input is large enough that the boundary condition could
40 // only ever apply on one side.
41 assert(width == 33 || width == 34);
42 assert(height == 33 || height == 34);
43 }
44 return 0;
45}

55void test(Buffer<uint8_t> (*factory)(int w, int h, int c)) {
56 Buffer<uint8_t> input = factory(W, H, 3);
57 input.for_each_element([&](int x, int y, int c) {
58 // Just an arbitrary color pattern with enough variation to notice the brighten + blur
59 if (c == 0) {
60 input(x, y, c) = (uint8_t)((x % 7) + (y % 3));
61 } else if (c == 1) {
62 input(x, y, c) = (uint8_t)(x + y);
63 } else {
64 input(x, y, c) = (uint8_t)((x * 5) + (y * 2));
65 }
66 });
67 Buffer<uint8_t> output = factory(W, H, 3);
68
69 printf("Evaluating output over %d x %d in tiles of size 32 x 32\n", W, H);
70 tiled_blur(input, output);
71
72#if RUN_BENCHMARKS
73 double t = Halide::Tools::benchmark(10, 100, [&]() {
74 tiled_blur(input, output);
75 });
76 const float megapixels = (W * H) / (1024.f * 1024.f);
77 printf("Benchmark: %d %d -> %f mpix/s\n", W, H, megapixels / t);
78#endif
79
80#if SAVE_IMAGES
81 static int x = 0;
82 Halide::Tools::save_image(input, "/tmp/tiled_input" + std::to_string(x) + ".png");
83 Halide::Tools::save_image(output, "/tmp/tiled_output" + std::to_string(x) + ".png");
84 ++x;
85#endif
86}
