50class Mandelbrot : public Generator<Mandelbrot> {
51public:
52 Input<float> x_min{"x_min"};
53 Input<float> x_max{"x_max"};
54 Input<float> y_min{"y_min"};
55 Input<float> y_max{"y_max"};
56 Input<float> c_real{"c_real"};
57 Input<float> c_imag{"c_imag"};
58 Input<int> iters{"iters"};
59 Input<int> w{"w"};
60 Input<int> h{"h"};
61
62 Output<Buffer<int32_t>> count{"count", 2};
63
64 void generate() {
65 assert(get_target().has_feature(Target::Profile));
66
67 Var x, y, z;
68
69 Complex initial(lerp(x_min, x_max, cast<float>(x) / w),
70 lerp(y_min, y_max, cast<float>(y) / h));
71 Complex c(c_real, c_imag);
72
73 Func mandelbrot;
74 mandelbrot(x, y, z) = initial;
75 RDom t(1, iters);
76 Complex current = mandelbrot(x, y, t - 1);
77 mandelbrot(x, y, t) = current * current + c;
78
79 // How many iterations until something escapes a circle of radius 2?
80 Tuple escape = argmin(magnitude(mandelbrot(x, y, t)) < 4);
81
82 // If it never escapes, use the value 0
83 count(x, y) = select(escape[1], 0, escape[0]);
84
85 Var xi, yi, xo, yo;
86 mandelbrot.compute_at(count, xo);
87
88 count.tile(x, y, xo, yo, xi, yi, 8, 8).parallel(yo).vectorize(xi, 4).unroll(xi).unroll(yi, 2);
89 }
90};
