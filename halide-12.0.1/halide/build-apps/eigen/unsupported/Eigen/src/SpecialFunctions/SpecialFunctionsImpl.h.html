102 * Polynomial evaluation helper for the Psi (digamma) function.
103 *
104 * digamma_impl_maybe_poly::run(s) evaluates the asymptotic Psi expansion for
105 * input Scalar s, assuming s is above 10.0.
106 *
107 * If s is above a certain threshold for the given Scalar type, zero
108 * is returned. Otherwise the polynomial is evaluated with enough
109 * coefficients for results matching Scalar machine precision.

111 *

118 THIS_TYPE_IS_NOT_SUPPORTED);
119 return Scalar(0);
120 }
121};

124template <>
125struct digamma_impl_maybe_poly<float> {
126 EIGEN_DEVICE_FUNC
127 static EIGEN_STRONG_INLINE float run(const float s) {
128 const float A[] = {
129 -4.16666666666666666667E-3f,
130 3.96825396825396825397E-3f,
131 -8.33333333333333333333E-3f,
132 8.33333333333333333333E-2f
133 };
134
135 float z;
136 if (s < 1.0e8f) {
137 z = 1.0f / (s * s);
138 return z * internal::ppolevl<float, 3>::run(z, A);
139 } else return 0.0f;
140 }

166template <typename Scalar>
167struct digamma_impl {
168 EIGEN_DEVICE_FUNC
169 static Scalar run(Scalar x) {
170 /*
171 *
172 * Psi (digamma) function (modified for Eigen)
173 *
174 *
175 * SYNOPSIS:
176 *
177 * double x, y, psi();
178 *
179 * y = psi( x );
180 *
181 *
182 * DESCRIPTION:
183 *
184 * d -
185 * psi(x) = -- ln | (x)
186 * dx
187 *
188 * is the logarithmic derivative of the gamma function.
189 * For integer x,
190 * n-1
191 * -
192 * psi(n) = -EUL + > 1/k.
193 * -
194 * k=1
195 *
196 * If x is negative, it is transformed to a positive argument by the
197 * reflection formula psi(1-x) = psi(x) + pi cot(pi x).
198 * For general positive x, the argument is made greater than 10
199 * using the recurrence psi(x+1) = psi(x) + 1/x.
200 * Then the following asymptotic expansion is applied:
201 *
202 * inf. B
203 * - 2k
204 * psi(x) = log(x) - 1/2x - > -------
205 * - 2k
206 * k=1 2k x
207 *
208 * where the B2k are Bernoulli numbers.
209 *
210 * ACCURACY (float):
211 * Relative error (except absolute when |psi| < 1):
212 * arithmetic domain # trials peak rms
213 * IEEE 0,30 30000 1.3e-15 1.4e-16
214 * IEEE -30,0 40000 1.5e-15 2.2e-16
215 *
216 * ACCURACY (double):
217 * Absolute error, relative when |psi| > 1 :
218 * arithmetic domain # trials peak rms
219 * IEEE -33,0 30000 8.2e-7 1.2e-7
220 * IEEE 0,33 100000 7.3e-7 7.7e-8
221 *
222 * ERROR MESSAGES:
223 * message condition value returned
224 * psi singularity x integer <=0 INFINITY
225 */
226
227 Scalar p, q, nz, s, w, y;
228 bool negative = false;
229
230 const Scalar maxnum = NumTraits<Scalar>::infinity();
231 const Scalar m_pi = Scalar(EIGEN_PI);
232
233 const Scalar zero = Scalar(0);
234 const Scalar one = Scalar(1);
235 const Scalar half = Scalar(0.5);
236 nz = zero;
237
238 if (x <= zero) {
239 negative = true;
240 q = x;
241 p = numext::floor(q);
242 if (p == q) {
243 return maxnum;
244 }
245 /* Remove the zeros of tan(m_pi x)
246 * by subtracting the nearest integer from x
247 */
248 nz = q - p;
249 if (nz != half) {
250 if (nz > half) {
251 p += one;
252 nz = q - p;
253 }
254 nz = m_pi / numext::tan(m_pi * nz);
255 }
256 else {
257 nz = zero;
258 }
259 x = one - x;
260 }
261
262 /* use the recurrence psi(x+1) = psi(x) + 1/x. */
263 s = x;
264 w = zero;
265 while (s < Scalar(10)) {
266 w += one / s;
267 s += one;
268 }
269
270 y = digamma_impl_maybe_poly<Scalar>::run(s);
271
272 y = numext::log(s) - (half / s) - y - w;
273
274 return (negative) ? y - nz : y;
275 }
276};

282/** \internal \returns the error function of \a a (coeff-wise)
283 Doesn't do anything fancy, just a 13/8-degree rational interpolant which
284 is accurate up to a couple of ulp in the range [-4, 4], outside of which
285 fl(erf(x)) = +/-1.
286
287 This implementation works on both scalars and Ts.
288*/
289template <typename T>
290EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE T generic_fast_erf_float(const T& a_x) {
291 // Clamp the inputs to the range [-4, 4] since anything outside
292 // this range is +/-1.0f in single-precision.
293 const T plus_4 = pset1<T>(4.f);
294 const T minus_4 = pset1<T>(-4.f);
295 const T x = pmax(pmin(a_x, plus_4), minus_4);
296 // The monomial coefficients of the numerator polynomial (odd).
297 const T alpha_1 = pset1<T>(-1.60960333262415e-02f);
298 const T alpha_3 = pset1<T>(-2.95459980854025e-03f);
299 const T alpha_5 = pset1<T>(-7.34990630326855e-04f);
300 const T alpha_7 = pset1<T>(-5.69250639462346e-05f);
301 const T alpha_9 = pset1<T>(-2.10102402082508e-06f);
302 const T alpha_11 = pset1<T>(2.77068142495902e-08f);
303 const T alpha_13 = pset1<T>(-2.72614225801306e-10f);
304
305 // The monomial coefficients of the denominator polynomial (even).
306 const T beta_0 = pset1<T>(-1.42647390514189e-02f);
307 const T beta_2 = pset1<T>(-7.37332916720468e-03f);
308 const T beta_4 = pset1<T>(-1.68282697438203e-03f);
309 const T beta_6 = pset1<T>(-2.13374055278905e-04f);
310 const T beta_8 = pset1<T>(-1.45660718464996e-05f);
311
312 // Since the polynomials are odd/even, we need x^2.
313 const T x2 = pmul(x, x);
314
315 // Evaluate the numerator polynomial p.
316 T p = pmadd(x2, alpha_13, alpha_11);
317 p = pmadd(x2, p, alpha_9);
318 p = pmadd(x2, p, alpha_7);
319 p = pmadd(x2, p, alpha_5);
320 p = pmadd(x2, p, alpha_3);
321 p = pmadd(x2, p, alpha_1);
322 p = pmul(x, p);
323
324 // Evaluate the denominator polynomial p.
325 T q = pmadd(x2, beta_8, beta_6);
326 q = pmadd(x2, q, beta_4);
327 q = pmadd(x2, q, beta_2);
328 q = pmadd(x2, q, beta_0);
329
330 // Divide the numerator by the denominator.
331 return pdiv(p, q);
332}

423/* Inverse of Normal distribution function (modified for Eigen).
424 *
425 *
426 * SYNOPSIS:
427 *
428 * double x, y, ndtri();
429 *
430 * x = ndtri( y );
431 *
432 *
433 *
434 * DESCRIPTION:
435 *
436 * Returns the argument, x, for which the area under the
437 * Gaussian probability density function (integrated from
438 * minus infinity to x) is equal to y.
439 *
440 *
441 * For small arguments 0 < y < exp(-2), the program computes
442 * z = sqrt( -2.0 * log(y) ); then the approximation is
443 * x = z - log(z)/z - (1/z) P(1/z) / Q(1/z).
444 * There are two rational functions P/Q, one for 0 < y < exp(-32)
445 * and the other for y up to exp(-2). For larger arguments,
446 * w = y - 0.5, and x/sqrt(2pi) = w + w**3 R(w**2)/S(w**2)).
447 *
448 *
449 * ACCURACY:
450 *
451 * Relative error:
452 * arithmetic domain # trials peak rms
453 * DEC 0.125, 1 5500 9.5e-17 2.1e-17
454 * DEC 6e-39, 0.135 3500 5.7e-17 1.3e-17
455 * IEEE 0.125, 1 20000 7.2e-16 1.3e-16
456 * IEEE 3e-308, 0.135 50000 4.6e-16 9.8e-17
457 *
458 *
459 * ERROR MESSAGES:
460 *
461 * message condition value returned
462 * ndtri domain x <= 0 -MAXNUM
463 * ndtri domain x >= 1 MAXNUM
464 *
465 */
466 /*
467 Cephes Math Library Release 2.2: June, 1992
468 Copyright 1985, 1987, 1992 by Stephen L. Moshier
469 Direct inquiries to 30 Frost Street, Cambridge, MA 02140
470 */

496// We split this computation in to two so that in the scalar path
497// only one branch is evaluated (due to our template specialization of pselect
498// being an if statement.)

500template <typename T, typename ScalarType>
501EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE T generic_ndtri_gt_exp_neg_two(const T& b) {
502 const ScalarType p0[] = {
503 ScalarType(-5.99633501014107895267e1),
504 ScalarType(9.80010754185999661536e1),
505 ScalarType(-5.66762857469070293439e1),
506 ScalarType(1.39312609387279679503e1),
507 ScalarType(-1.23916583867381258016e0)
508 };
509 const ScalarType q0[] = {
510 ScalarType(1.0),
511 ScalarType(1.95448858338141759834e0),
512 ScalarType(4.67627912898881538453e0),
513 ScalarType(8.63602421390890590575e1),
514 ScalarType(-2.25462687854119370527e2),
515 ScalarType(2.00260212380060660359e2),
516 ScalarType(-8.20372256168333339912e1),
517 ScalarType(1.59056225126211695515e1),
518 ScalarType(-1.18331621121330003142e0)
519 };
520 const T sqrt2pi = pset1<T>(ScalarType(2.50662827463100050242e0));
521 const T half = pset1<T>(ScalarType(0.5));
522 T c, c2, ndtri_gt_exp_neg_two;
523
524 c = psub(b, half);
525 c2 = pmul(c, c);
526 ndtri_gt_exp_neg_two = pmadd(c, pmul(
527 c2, pdiv(
528 internal::ppolevl<T, 4>::run(c2, p0),
529 internal::ppolevl<T, 8>::run(c2, q0))), c);
530 return pmul(ndtri_gt_exp_neg_two, sqrt2pi);
531}

533template <typename T, typename ScalarType>
534EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE T generic_ndtri_lt_exp_neg_two(
535 const T& b, const T& should_flipsign) {
536 /* Approximation for interval z = sqrt(-2 log a ) between 2 and 8
537 * i.e., a between exp(-2) = .135 and exp(-32) = 1.27e-14.
538 */
539 const ScalarType p1[] = {
540 ScalarType(4.05544892305962419923e0),
541 ScalarType(3.15251094599893866154e1),
542 ScalarType(5.71628192246421288162e1),
543 ScalarType(4.40805073893200834700e1),
544 ScalarType(1.46849561928858024014e1),
545 ScalarType(2.18663306850790267539e0),
546 ScalarType(-1.40256079171354495875e-1),
547 ScalarType(-3.50424626827848203418e-2),
548 ScalarType(-8.57456785154685413611e-4)
549 };
550 const ScalarType q1[] = {
551 ScalarType(1.0),
552 ScalarType(1.57799883256466749731e1),
553 ScalarType(4.53907635128879210584e1),
554 ScalarType(4.13172038254672030440e1),
555 ScalarType(1.50425385692907503408e1),
556 ScalarType(2.50464946208309415979e0),
557 ScalarType(-1.42182922854787788574e-1),
558 ScalarType(-3.80806407691578277194e-2),
559 ScalarType(-9.33259480895457427372e-4)
560 };
561 /* Approximation for interval z = sqrt(-2 log a ) between 8 and 64
562 * i.e., a between exp(-32) = 1.27e-14 and exp(-2048) = 3.67e-890.
563 */
564 const ScalarType p2[] = {
565 ScalarType(3.23774891776946035970e0),
566 ScalarType(6.91522889068984211695e0),
567 ScalarType(3.93881025292474443415e0),
568 ScalarType(1.33303460815807542389e0),
569 ScalarType(2.01485389549179081538e-1),
570 ScalarType(1.23716634817820021358e-2),
571 ScalarType(3.01581553508235416007e-4),
572 ScalarType(2.65806974686737550832e-6),
573 ScalarType(6.23974539184983293730e-9)
574 };
575 const ScalarType q2[] = {
576 ScalarType(1.0),
577 ScalarType(6.02427039364742014255e0),
578 ScalarType(3.67983563856160859403e0),
579 ScalarType(1.37702099489081330271e0),
580 ScalarType(2.16236993594496635890e-1),
581 ScalarType(1.34204006088543189037e-2),
582 ScalarType(3.28014464682127739104e-4),
583 ScalarType(2.89247864745380683936e-6),
584 ScalarType(6.79019408009981274425e-9)
585 };
586 const T eight = pset1<T>(ScalarType(8.0));
587 const T one = pset1<T>(ScalarType(1));
588 const T neg_two = pset1<T>(ScalarType(-2));
589 T x, x0, x1, z;
590
591 x = psqrt(pmul(neg_two, plog(b)));
592 x0 = psub(x, pdiv(plog(x), x));
593 z = pdiv(one, x);
594 x1 = pmul(
595 z, pselect(
596 pcmp_lt(x, eight),
597 pdiv(internal::ppolevl<T, 8>::run(z, p1),
598 internal::ppolevl<T, 8>::run(z, q1)),
599 pdiv(internal::ppolevl<T, 8>::run(z, p2),
600 internal::ppolevl<T, 8>::run(z, q2))));
601 return flipsign(should_flipsign, psub(x0, x1));
602}

604template <typename T, typename ScalarType>
605EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE
606T generic_ndtri(const T& a) {
607 const T maxnum = pset1<T>(NumTraits<ScalarType>::infinity());
608 const T neg_maxnum = pset1<T>(-NumTraits<ScalarType>::infinity());
609
610 const T zero = pset1<T>(ScalarType(0));
611 const T one = pset1<T>(ScalarType(1));
612 // exp(-2)
613 const T exp_neg_two = pset1<T>(ScalarType(0.13533528323661269189));
614 T b, ndtri, should_flipsign;
615
616 should_flipsign = pcmp_le(a, psub(one, exp_neg_two));
617 b = pselect(should_flipsign, a, psub(one, a));
618
619 ndtri = pselect(
620 pcmp_lt(exp_neg_two, b),
621 generic_ndtri_gt_exp_neg_two<T, ScalarType>(b),
622 generic_ndtri_lt_exp_neg_two<T, ScalarType>(b, should_flipsign));
623
624 return pselect(
625 pcmp_le(a, zero), neg_maxnum,
626 pselect(pcmp_le(one, a), maxnum, ndtri));
627}

747template <typename Scalar, IgammaComputationMode mode>
748struct igammac_cf_impl {
749 /* Computes igamc(a, x) or derivative (depending on the mode)
750 * using the continued fraction expansion of the complementary
751 * incomplete Gamma function.
752 *
753 * Preconditions:
754 * a > 0
755 * x >= 1
756 * x >= a
757 */
758 EIGEN_DEVICE_FUNC
759 static Scalar run(Scalar a, Scalar x) {
760 const Scalar zero = 0;
761 const Scalar one = 1;
762 const Scalar two = 2;
763 const Scalar machep = cephes_helper<Scalar>::machep();
764 const Scalar big = cephes_helper<Scalar>::big();
765 const Scalar biginv = cephes_helper<Scalar>::biginv();
766
767 if ((numext::isinf)(x)) {
768 return zero;
769 }
770
771 Scalar ax = main_igamma_term<Scalar>(a, x);
772 // This is independent of mode. If this value is zero,
773 // then the function value is zero. If the function value is zero,
774 // then we are in a neighborhood where the function value evalutes to zero,
775 // so the derivative is zero.
776 if (ax == zero) {
777 return zero;
778 }
779
780 // continued fraction
781 Scalar y = one - a;
782 Scalar z = x + y + one;
783 Scalar c = zero;
784 Scalar pkm2 = one;
785 Scalar qkm2 = x;
786 Scalar pkm1 = x + one;
787 Scalar qkm1 = z * x;
788 Scalar ans = pkm1 / qkm1;
789
790 Scalar dpkm2_da = zero;
791 Scalar dqkm2_da = zero;
792 Scalar dpkm1_da = zero;
793 Scalar dqkm1_da = -x;
794 Scalar dans_da = (dpkm1_da - ans * dqkm1_da) / qkm1;
795
796 for (int i = 0; i < igamma_num_iterations<Scalar, mode>(); i++) {
797 c += one;
798 y += one;
799 z += two;
800
801 Scalar yc = y * c;
802 Scalar pk = pkm1 * z - pkm2 * yc;
803 Scalar qk = qkm1 * z - qkm2 * yc;
804
805 Scalar dpk_da = dpkm1_da * z - pkm1 - dpkm2_da * yc + pkm2 * c;
806 Scalar dqk_da = dqkm1_da * z - qkm1 - dqkm2_da * yc + qkm2 * c;
807
808 if (qk != zero) {
809 Scalar ans_prev = ans;
810 ans = pk / qk;
811
812 Scalar dans_da_prev = dans_da;
813 dans_da = (dpk_da - ans * dqk_da) / qk;
814
815 if (mode == VALUE) {
816 if (numext::abs(ans_prev - ans) <= machep * numext::abs(ans)) {
817 break;
818 }
819 } else {
820 if (numext::abs(dans_da - dans_da_prev) <= machep) {
821 break;
822 }
823 }
824 }
825
826 pkm2 = pkm1;
827 pkm1 = pk;
828 qkm2 = qkm1;
829 qkm1 = qk;
830
831 dpkm2_da = dpkm1_da;
832 dpkm1_da = dpk_da;
833 dqkm2_da = dqkm1_da;
834 dqkm1_da = dqk_da;
835
836 if (numext::abs(pk) > big) {
837 pkm2 *= biginv;
838 pkm1 *= biginv;
839 qkm2 *= biginv;
840 qkm1 *= biginv;
841
842 dpkm2_da *= biginv;
843 dpkm1_da *= biginv;
844 dqkm2_da *= biginv;
845 dqkm1_da *= biginv;
846 }
847 }
848
849 /* Compute x**a * exp(-x) / gamma(a) */
850 Scalar dlogax_da = numext::log(x) - digamma_impl<Scalar>::run(a);
851 Scalar dax_da = ax * dlogax_da;
852
853 switch (mode) {
854 case VALUE:
855 return ans * ax;
856 case DERIVATIVE:
857 return ans * dax_da + dans_da * ax;
858 case SAMPLE_DERIVATIVE:
859 default: // this is needed to suppress clang warning
860 return -(dans_da + ans * dlogax_da) * x;
861 }
862 }
863};

865template <typename Scalar, IgammaComputationMode mode>
866struct igamma_series_impl {
867 /* Computes igam(a, x) or its derivative (depending on the mode)
868 * using the series expansion of the incomplete Gamma function.
869 *
870 * Preconditions:
871 * x > 0
872 * a > 0
873 * !(x > 1 && x > a)
874 */
875 EIGEN_DEVICE_FUNC
876 static Scalar run(Scalar a, Scalar x) {
877 const Scalar zero = 0;
878 const Scalar one = 1;
879 const Scalar machep = cephes_helper<Scalar>::machep();
880
881 Scalar ax = main_igamma_term<Scalar>(a, x);
882
883 // This is independent of mode. If this value is zero,
884 // then the function value is zero. If the function value is zero,
885 // then we are in a neighborhood where the function value evalutes to zero,
886 // so the derivative is zero.
887 if (ax == zero) {
888 return zero;
889 }
890
891 ax /= a;
892
893 /* power series */
894 Scalar r = a;
895 Scalar c = one;
896 Scalar ans = one;
897
898 Scalar dc_da = zero;
899 Scalar dans_da = zero;
900
901 for (int i = 0; i < igamma_num_iterations<Scalar, mode>(); i++) {
902 r += one;
903 Scalar term = x / r;
904 Scalar dterm_da = -x / (r * r);
905 dc_da = term * dc_da + dterm_da * c;
906 dans_da += dc_da;
907 c *= term;
908 ans += c;
909
910 if (mode == VALUE) {
911 if (c <= machep * ans) {
912 break;
913 }
914 } else {
915 if (numext::abs(dc_da) <= machep * numext::abs(dans_da)) {
916 break;
917 }
918 }
919 }
920
921 Scalar dlogax_da = numext::log(x) - digamma_impl<Scalar>::run(a + one);
922 Scalar dax_da = ax * dlogax_da;
923
924 switch (mode) {
925 case VALUE:
926 return ans * ax;
927 case DERIVATIVE:
928 return ans * dax_da + dans_da * ax;
929 case SAMPLE_DERIVATIVE:
930 default: // this is needed to suppress clang warning
931 return -(dans_da + ans * dlogax_da) * x / a;
932 }
933 }
934};

1587/* Continued fraction expansion #1 for incomplete beta integral (small_branch = True)
1588 * Continued fraction expansion #2 for incomplete beta integral (small_branch = False)
1589 */
1590template <typename Scalar>
1591struct incbeta_cfe {
1592 EIGEN_DEVICE_FUNC
1593 static EIGEN_STRONG_INLINE Scalar run(Scalar a, Scalar b, Scalar x, bool small_branch) {
1594 EIGEN_STATIC_ASSERT((internal::is_same<Scalar, float>::value ||
1595 internal::is_same<Scalar, double>::value),
1596 THIS_TYPE_IS_NOT_SUPPORTED);
1597 const Scalar big = cephes_helper<Scalar>::big();
1598 const Scalar machep = cephes_helper<Scalar>::machep();
1599 const Scalar biginv = cephes_helper<Scalar>::biginv();
1600
1601 const Scalar zero = 0;
1602 const Scalar one = 1;
1603 const Scalar two = 2;
1604
1605 Scalar xk, pk, pkm1, pkm2, qk, qkm1, qkm2;
1606 Scalar k1, k2, k3, k4, k5, k6, k7, k8, k26update;
1607 Scalar ans;
1608 int n;
1609
1610 const int num_iters = (internal::is_same<Scalar, float>::value) ? 100 : 300;
1611 const Scalar thresh =
1612 (internal::is_same<Scalar, float>::value) ? machep : Scalar(3) * machep;
1613 Scalar r = (internal::is_same<Scalar, float>::value) ? zero : one;
1614
1615 if (small_branch) {
1616 k1 = a;
1617 k2 = a + b;
1618 k3 = a;
1619 k4 = a + one;
1620 k5 = one;
1621 k6 = b - one;
1622 k7 = k4;
1623 k8 = a + two;
1624 k26update = one;
1625 } else {
1626 k1 = a;
1627 k2 = b - one;
1628 k3 = a;
1629 k4 = a + one;
1630 k5 = one;
1631 k6 = a + b;
1632 k7 = a + one;
1633 k8 = a + two;
1634 k26update = -one;
1635 x = x / (one - x);
1636 }
1637
1638 pkm2 = zero;
1639 qkm2 = one;
1640 pkm1 = one;
1641 qkm1 = one;
1642 ans = one;
1643 n = 0;
1644
1645 do {
1646 xk = -(x * k1 * k2) / (k3 * k4);
1647 pk = pkm1 + pkm2 * xk;
1648 qk = qkm1 + qkm2 * xk;
1649 pkm2 = pkm1;
1650 pkm1 = pk;
1651 qkm2 = qkm1;
1652 qkm1 = qk;
1653
1654 xk = (x * k5 * k6) / (k7 * k8);
1655 pk = pkm1 + pkm2 * xk;
1656 qk = qkm1 + qkm2 * xk;
1657 pkm2 = pkm1;
1658 pkm1 = pk;
1659 qkm2 = qkm1;
1660 qkm1 = qk;
1661
1662 if (qk != zero) {
1663 r = pk / qk;
1664 if (numext::abs(ans - r) < numext::abs(r) * thresh) {
1665 return r;
1666 }
1667 ans = r;
1668 }
1669
1670 k1 += one;
1671 k2 += k26update;
1672 k3 += two;
1673 k4 += two;
1674 k5 += one;
1675 k6 -= k26update;
1676 k7 += two;
1677 k8 += two;
1678
1679 if ((numext::abs(qk) + numext::abs(pk)) > big) {
1680 pkm2 *= biginv;
1681 pkm1 *= biginv;
1682 qkm2 *= biginv;
1683 qkm1 *= biginv;
1684 }
1685 if ((numext::abs(qk) < biginv) || (numext::abs(pk) < biginv)) {
1686 pkm2 *= big;
1687 pkm1 *= big;
1688 qkm2 *= big;
1689 qkm1 *= big;
1690 }
1691 } while (++n < num_iters);
1692
1693 return ans;
1694 }
1695};
