69// Define a macro for catching SYCL exceptions if exceptions are enabled
70#define EIGEN_SYCL_TRY_CATCH(X) \
71 do { \
72 EIGEN_TRY {X;} \
73 EIGEN_CATCH(const cl::sycl::exception& e) { \
74 EIGEN_THROW_X(std::runtime_error("SYCL exception at " + \
75 std::string(__FILE__) + ":" + \
76 std::to_string(__LINE__) + "\n" + \
77 e.what())); \
78 } \
79 } while (false)
