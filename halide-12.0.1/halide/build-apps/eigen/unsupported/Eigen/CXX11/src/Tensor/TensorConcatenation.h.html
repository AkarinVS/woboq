111template<typename Axis, typename LeftArgType, typename RightArgType, typename Device>
112struct TensorEvaluator<const TensorConcatenationOp<Axis, LeftArgType, RightArgType>, Device>
113{
114 typedef TensorConcatenationOp<Axis, LeftArgType, RightArgType> XprType;
115 typedef typename XprType::Index Index;
116 static const int NumDims = internal::array_size<typename TensorEvaluator<LeftArgType, Device>::Dimensions>::value;
117 static const int RightNumDims = internal::array_size<typename TensorEvaluator<RightArgType, Device>::Dimensions>::value;
118 typedef DSizes<Index, NumDims> Dimensions;
119 typedef typename XprType::Scalar Scalar;
120 typedef typename XprType::CoeffReturnType CoeffReturnType;
121 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
122 typedef StorageMemory<CoeffReturnType, Device> Storage;
123 typedef typename Storage::Type EvaluatorPointerType;
124 enum {
125 IsAligned = false,
126 PacketAccess = TensorEvaluator<LeftArgType, Device>::PacketAccess &&
127 TensorEvaluator<RightArgType, Device>::PacketAccess,
128 BlockAccess = false,
129 PreferBlockAccess = TensorEvaluator<LeftArgType, Device>::PreferBlockAccess ||
130 TensorEvaluator<RightArgType, Device>::PreferBlockAccess,
131 Layout = TensorEvaluator<LeftArgType, Device>::Layout,
132 RawAccess = false
133 };

135 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
136 typedef internal::TensorBlockNotImplemented TensorBlock;

139 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
140 : m_leftImpl(op.lhsExpression(), device), m_rightImpl(op.rhsExpression(), device), m_axis(op.axis())
141 {
142 EIGEN_STATIC_ASSERT((static_cast<int>(TensorEvaluator<LeftArgType, Device>::Layout) == static_cast<int>(TensorEvaluator<RightArgType, Device>::Layout) || NumDims == 1), YOU_MADE_A_PROGRAMMING_MISTAKE);
143 EIGEN_STATIC_ASSERT((NumDims == RightNumDims), YOU_MADE_A_PROGRAMMING_MISTAKE);
144 EIGEN_STATIC_ASSERT((NumDims > 0), YOU_MADE_A_PROGRAMMING_MISTAKE);
145
146 eigen_assert(0 <= m_axis && m_axis < NumDims);
147 const Dimensions& lhs_dims = m_leftImpl.dimensions();
148 const Dimensions& rhs_dims = m_rightImpl.dimensions();
149 {
150 int i = 0;
151 for (; i < m_axis; ++i) {
152 eigen_assert(lhs_dims[i] > 0);
153 eigen_assert(lhs_dims[i] == rhs_dims[i]);
154 m_dimensions[i] = lhs_dims[i];
155 }
156 eigen_assert(lhs_dims[i] > 0); // Now i == m_axis.
157 eigen_assert(rhs_dims[i] > 0);
158 m_dimensions[i] = lhs_dims[i] + rhs_dims[i];
159 for (++i; i < NumDims; ++i) {
160 eigen_assert(lhs_dims[i] > 0);
161 eigen_assert(lhs_dims[i] == rhs_dims[i]);
162 m_dimensions[i] = lhs_dims[i];
163 }
164 }
165
166 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
167 m_leftStrides[0] = 1;
168 m_rightStrides[0] = 1;
169 m_outputStrides[0] = 1;
170
171 for (int j = 1; j < NumDims; ++j) {
172 m_leftStrides[j] = m_leftStrides[j-1] * lhs_dims[j-1];
173 m_rightStrides[j] = m_rightStrides[j-1] * rhs_dims[j-1];
174 m_outputStrides[j] = m_outputStrides[j-1] * m_dimensions[j-1];
175 }
176 } else {
177 m_leftStrides[NumDims - 1] = 1;
178 m_rightStrides[NumDims - 1] = 1;
179 m_outputStrides[NumDims - 1] = 1;
180
181 for (int j = NumDims - 2; j >= 0; --j) {
182 m_leftStrides[j] = m_leftStrides[j+1] * lhs_dims[j+1];
183 m_rightStrides[j] = m_rightStrides[j+1] * rhs_dims[j+1];
184 m_outputStrides[j] = m_outputStrides[j+1] * m_dimensions[j+1];
185 }
186 }
187 }

189 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Dimensions& dimensions() const { return m_dimensions; }

205 // TODO(phli): attempt to speed this up. The integer divisions and modulo are slow.
206 // See CL/76180724 comments for more ideas.
207 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE CoeffReturnType coeff(Index index) const
208 {
209 // Collect dimension-wise indices (subs).
210 array<Index, NumDims> subs;
211 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
212 for (int i = NumDims - 1; i > 0; --i) {
213 subs[i] = index / m_outputStrides[i];
214 index -= subs[i] * m_outputStrides[i];
215 }
216 subs[0] = index;
217 } else {
218 for (int i = 0; i < NumDims - 1; ++i) {
219 subs[i] = index / m_outputStrides[i];
220 index -= subs[i] * m_outputStrides[i];
221 }
222 subs[NumDims - 1] = index;
223 }
224
225 const Dimensions& left_dims = m_leftImpl.dimensions();
226 if (subs[m_axis] < left_dims[m_axis]) {
227 Index left_index;
228 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
229 left_index = subs[0];
230 EIGEN_UNROLL_LOOP
231 for (int i = 1; i < NumDims; ++i) {
232 left_index += (subs[i] % left_dims[i]) * m_leftStrides[i];
233 }
234 } else {
235 left_index = subs[NumDims - 1];
236 EIGEN_UNROLL_LOOP
237 for (int i = NumDims - 2; i >= 0; --i) {
238 left_index += (subs[i] % left_dims[i]) * m_leftStrides[i];
239 }
240 }
241 return m_leftImpl.coeff(left_index);
242 } else {
243 subs[m_axis] -= left_dims[m_axis];
244 const Dimensions& right_dims = m_rightImpl.dimensions();
245 Index right_index;
246 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
247 right_index = subs[0];
248 EIGEN_UNROLL_LOOP
249 for (int i = 1; i < NumDims; ++i) {
250 right_index += (subs[i] % right_dims[i]) * m_rightStrides[i];
251 }
252 } else {
253 right_index = subs[NumDims - 1];
254 EIGEN_UNROLL_LOOP
255 for (int i = NumDims - 2; i >= 0; --i) {
256 right_index += (subs[i] % right_dims[i]) * m_rightStrides[i];
257 }
258 }
259 return m_rightImpl.coeff(right_index);
260 }
261 }

263 // TODO(phli): Add a real vectorization.
264 template<int LoadMode>
265 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
266 {
267 const int packetSize = PacketType<CoeffReturnType, Device>::size;
268 EIGEN_STATIC_ASSERT((packetSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
269 eigen_assert(index + packetSize - 1 < dimensions().TotalSize());
270
271 EIGEN_ALIGN_MAX CoeffReturnType values[packetSize];
272 EIGEN_UNROLL_LOOP
273 for (int i = 0; i < packetSize; ++i) {
274 values[i] = coeff(index+i);
275 }
276 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
277 return rslt;
278 }

280 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost
281 costPerCoeff(bool vectorized) const {
282 const double compute_cost = NumDims * (2 * TensorOpCost::AddCost<Index>() +
283 2 * TensorOpCost::MulCost<Index>() +
284 TensorOpCost::DivCost<Index>() +
285 TensorOpCost::ModCost<Index>());
286 const double lhs_size = m_leftImpl.dimensions().TotalSize();
287 const double rhs_size = m_rightImpl.dimensions().TotalSize();
288 return (lhs_size / (lhs_size + rhs_size)) *
289 m_leftImpl.costPerCoeff(vectorized) +
290 (rhs_size / (lhs_size + rhs_size)) *
291 m_rightImpl.costPerCoeff(vectorized) +
292 TensorOpCost(0, 0, compute_cost);
293 }

338 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(XprType& op, const Device& device)
339 : Base(op, device)
340 {
341 EIGEN_STATIC_ASSERT((static_cast<int>(Layout) == static_cast<int>(ColMajor)), YOU_MADE_A_PROGRAMMING_MISTAKE);
342 }
343
344 typedef typename XprType::Index Index;
345 typedef typename XprType::Scalar Scalar;
346 typedef typename XprType::CoeffReturnType CoeffReturnType;
347 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
348
349 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE CoeffReturnType& coeffRef(Index index)
350 {
351 // Collect dimension-wise indices (subs).
352 array<Index, Base::NumDims> subs;
353 for (int i = Base::NumDims - 1; i > 0; --i) {
354 subs[i] = index / this->m_outputStrides[i];
355 index -= subs[i] * this->m_outputStrides[i];
356 }
357 subs[0] = index;
358
359 const Dimensions& left_dims = this->m_leftImpl.dimensions();
360 if (subs[this->m_axis] < left_dims[this->m_axis]) {
361 Index left_index = subs[0];
362 for (int i = 1; i < Base::NumDims; ++i) {
363 left_index += (subs[i] % left_dims[i]) * this->m_leftStrides[i];
364 }
365 return this->m_leftImpl.coeffRef(left_index);
366 } else {
367 subs[this->m_axis] -= left_dims[this->m_axis];
368 const Dimensions& right_dims = this->m_rightImpl.dimensions();
369 Index right_index = subs[0];
370 for (int i = 1; i < Base::NumDims; ++i) {
371 right_index += (subs[i] % right_dims[i]) * this->m_rightStrides[i];
372 }
373 return this->m_rightImpl.coeffRef(right_index);
374 }
375 }

377 template <int StoreMode> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
378 void writePacket(Index index, const PacketReturnType& x)
379 {
380 const int packetSize = PacketType<CoeffReturnType, Device>::size;
381 EIGEN_STATIC_ASSERT((packetSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
382 eigen_assert(index + packetSize - 1 < this->dimensions().TotalSize());
383
384 EIGEN_ALIGN_MAX CoeffReturnType values[packetSize];
385 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
386 for (int i = 0; i < packetSize; ++i) {
387 coeffRef(index+i) = values[i];
388 }
389 }
