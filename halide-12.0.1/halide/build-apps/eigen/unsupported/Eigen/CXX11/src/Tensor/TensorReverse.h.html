139 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op,
140 const Device& device)
141 : m_impl(op.expression(), device),
142 m_reverse(op.reverse()),
143 m_device(device)
144 {
145 // Reversing a scalar isn't supported yet. It would be a no-op anyway.
146 EIGEN_STATIC_ASSERT((NumDims > 0), YOU_MADE_A_PROGRAMMING_MISTAKE);
147
148 // Compute strides
149 m_dimensions = m_impl.dimensions();
150 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
151 m_strides[0] = 1;
152 for (int i = 1; i < NumDims; ++i) {
153 m_strides[i] = m_strides[i-1] * m_dimensions[i-1];
154 if (m_strides[i] > 0) m_fastStrides[i] = IndexDivisor(m_strides[i]);
155 }
156 } else {
157 m_strides[NumDims-1] = 1;
158 for (int i = NumDims - 2; i >= 0; --i) {
159 m_strides[i] = m_strides[i+1] * m_dimensions[i+1];
160 if (m_strides[i] > 0) m_fastStrides[i] = IndexDivisor(m_strides[i]);
161 }
162 }
163 }

185 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index reverseIndex(
186 Index index) const {
187 eigen_assert(index < dimensions().TotalSize());
188 Index inputIndex = 0;
189 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
190 EIGEN_UNROLL_LOOP
191 for (int i = NumDims - 1; i > 0; --i) {
192 Index idx = index / m_fastStrides[i];
193 index -= idx * m_strides[i];
194 if (m_reverse[i]) {
195 idx = m_dimensions[i] - idx - 1;
196 }
197 inputIndex += idx * m_strides[i] ;
198 }
199 if (m_reverse[0]) {
200 inputIndex += (m_dimensions[0] - index - 1);
201 } else {
202 inputIndex += index;
203 }
204 } else {
205 EIGEN_UNROLL_LOOP
206 for (int i = 0; i < NumDims - 1; ++i) {
207 Index idx = index / m_fastStrides[i];
208 index -= idx * m_strides[i];
209 if (m_reverse[i]) {
210 idx = m_dimensions[i] - idx - 1;
211 }
212 inputIndex += idx * m_strides[i] ;
213 }
214 if (m_reverse[NumDims-1]) {
215 inputIndex += (m_dimensions[NumDims-1] - index - 1);
216 } else {
217 inputIndex += index;
218 }
219 }
220 return inputIndex;
221 }

228 template<int LoadMode>
229 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
230 PacketReturnType packet(Index index) const
231 {
232 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
233 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
234
235 // TODO(ndjaitly): write a better packing routine that uses
236 // local structure.
237 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type
238 values[PacketSize];
239 EIGEN_UNROLL_LOOP
240 for (int i = 0; i < PacketSize; ++i) {
241 values[i] = coeff(index+i);
242 }
243 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
244 return rslt;
245 }

247 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
248 internal::TensorBlockResourceRequirements getResourceRequirements() const {
249 const size_t target_size = m_device.lastLevelCacheSize();
250 // Block evaluation reads underlying memory in reverse order, and default
251 // cost model does not properly catch this in bytes stored/loaded.
252 return internal::TensorBlockResourceRequirements::skewed<Scalar>(
253 target_size)
254 .addCostPerCoeff({0, 0, 24});
255 }

257 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorBlock
258 block(TensorBlockDesc& desc, TensorBlockScratch& scratch,
259 bool /*root_of_expr_ast*/ = false) const {
260 // TODO(ezhulenev): If underlying tensor expression supports and prefers
261 // block evaluation we must use it. Currently we use coeff and packet
262 // access into the underlying tensor expression.
263 // static const bool useBlockAccessForArgType =
264 // TensorEvaluator<ArgType, Device>::BlockAccess &&
265 // TensorEvaluator<ArgType, Device>::PreferBlockAccess;
266
267 static const bool isColMajor =
268 static_cast<int>(Layout) == static_cast<int>(ColMajor);
269
270 static const Index inner_dim_idx = isColMajor ? 0 : NumDims - 1;
271 const bool inner_dim_reversed = m_reverse[inner_dim_idx];
272
273 // Offset in the output block.
274 Index block_offset = 0;
275
276 // Offset in the input Tensor.
277 Index input_offset = reverseIndex(desc.offset());
278
279 // Initialize output block iterator state. Dimension in this array are
280 // always in inner_most -> outer_most order (col major layout).
281 array<BlockIteratorState, NumDims> it;
282 for (int i = 0; i < NumDims; ++i) {
283 const int dim = isColMajor ? i : NumDims - 1 - i;
284 it[i].size = desc.dimension(dim);
285 it[i].count = 0;
286 it[i].reverse = m_reverse[dim];
287
288 it[i].block_stride =
289 i == 0 ? 1 : (it[i - 1].size * it[i - 1].block_stride);
290 it[i].block_span = it[i].block_stride * (it[i].size - 1);
291
292 it[i].input_stride = m_strides[dim];
293 it[i].input_span = it[i].input_stride * (it[i].size - 1);
294
295 if (it[i].reverse) {
296 it[i].input_stride = -1 * it[i].input_stride;
297 it[i].input_span = -1 * it[i].input_span;
298 }
299 }
300
301 // If multiple inner dimensions have the same reverse flag, check if we can
302 // merge them into a single virtual inner dimension.
303 int effective_inner_dim = 0;
304 for (int i = 1; i < NumDims; ++i) {
305 if (it[i].reverse != it[effective_inner_dim].reverse) break;
306 if (it[i].block_stride != it[effective_inner_dim].size) break;
307 if (it[i].block_stride != numext::abs(it[i].input_stride)) break;
308
309 it[i].size = it[effective_inner_dim].size * it[i].size;
310
311 it[i].block_stride = 1;
312 it[i].input_stride = (inner_dim_reversed ? -1 : 1);
313
314 it[i].block_span = it[i].block_stride * (it[i].size - 1);
315 it[i].input_span = it[i].input_stride * (it[i].size - 1);
316
317 effective_inner_dim = i;
318 }
319
320 eigen_assert(it[effective_inner_dim].block_stride == 1);
321 eigen_assert(it[effective_inner_dim].input_stride ==
322 (inner_dim_reversed ? -1 : 1));
323
324 const Index inner_dim_size = it[effective_inner_dim].size;
325
326 // Prepare storage for the materialized reverse result.
327 const typename TensorBlock::Storage block_storage =
328 TensorBlock::prepareStorage(desc, scratch);
329 CoeffReturnType* block_buffer = block_storage.data();
330
331 while (it[NumDims - 1].count < it[NumDims - 1].size) {
332 // Copy inner-most dimension data from reversed location in input.
333 Index dst = block_offset;
334 Index src = input_offset;
335
336 // NOTE(ezhulenev): Adding vectorized path with internal::preverse showed
337 // worse results in benchmarks than a simple coefficient loop.
338 if (inner_dim_reversed) {
339 for (Index i = 0; i < inner_dim_size; ++i) {
340 block_buffer[dst] = m_impl.coeff(src);
341 ++dst;
342 --src;
343 }
344 } else {
345 for (Index i = 0; i < inner_dim_size; ++i) {
346 block_buffer[dst] = m_impl.coeff(src);
347 ++dst;
348 ++src;
349 }
350 }
351
352 // For the 1d tensor we need to generate only one inner-most dimension.
353 if ((NumDims - effective_inner_dim) == 1) break;
354
355 // Update offset.
356 for (Index i = effective_inner_dim + 1; i < NumDims; ++i) {
357 if (++it[i].count < it[i].size) {
358 block_offset += it[i].block_stride;
359 input_offset += it[i].input_stride;
360 break;
361 }
362 if (i != NumDims - 1) it[i].count = 0;
363 block_offset -= it[i].block_span;
364 input_offset -= it[i].input_span;
365 }
366 }
367
368 return block_storage.AsTensorMaterializedBlock();
369 }

371 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost costPerCoeff(bool vectorized) const {
372 double compute_cost = NumDims * (2 * TensorOpCost::AddCost<Index>() +
373 2 * TensorOpCost::MulCost<Index>() +
374 TensorOpCost::DivCost<Index>());
375 for (int i = 0; i < NumDims; ++i) {
376 if (m_reverse[i]) {
377 compute_cost += 2 * TensorOpCost::AddCost<Index>();
378 }
379 }
380 return m_impl.costPerCoeff(vectorized) +
381 TensorOpCost(0, 0, compute_cost, false /* vectorized */, PacketSize);
382 }

419 };

435 enum {
436 IsAligned = false,
437 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
438 BlockAccess = false,
439 PreferBlockAccess = false,

444 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op,

450 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
451 static const int PacketSize = PacketType<CoeffReturnType, Device>::size;

458 const Dimensions& dimensions() const { return this->m_dimensions; }

463
464 template <int StoreMode> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
465 void writePacket(Index index, const PacketReturnType& x) {
466 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
467 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
468
469 // This code is pilfered from TensorMorphing.h
470 EIGEN_ALIGN_MAX CoeffReturnType values[PacketSize];
471 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
472 EIGEN_UNROLL_LOOP
473 for (int i = 0; i < PacketSize; ++i) {
474 this->coeffRef(index+i) = values[i];
475 }
476 }
