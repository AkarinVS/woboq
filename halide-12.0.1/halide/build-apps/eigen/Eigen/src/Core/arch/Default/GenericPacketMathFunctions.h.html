22template<typename Packet> EIGEN_STRONG_INLINE Packet
23pfrexp_float(const Packet& a, Packet& exponent) {
24 typedef typename unpacket_traits<Packet>::integer_packet PacketI;
25 const Packet cst_126f = pset1<Packet>(126.0f);
26 const Packet cst_half = pset1<Packet>(0.5f);
27 const Packet cst_inv_mant_mask = pset1frombits<Packet>(~0x7f800000u);
28 exponent = psub(pcast<PacketI,Packet>(plogical_shift_right<23>(preinterpret<PacketI>(a))), cst_126f);
29 return por(pand(a, cst_inv_mant_mask), cst_half);
30}

32template<typename Packet> EIGEN_STRONG_INLINE Packet
33pldexp_float(Packet a, Packet exponent)
34{
35 typedef typename unpacket_traits<Packet>::integer_packet PacketI;
36 const Packet cst_127 = pset1<Packet>(127.f);
37 // return a * 2^exponent
38 PacketI ei = pcast<Packet,PacketI>(padd(exponent, cst_127));
39 return pmul(a, preinterpret<Packet>(plogical_shift_left<23>(ei)));
40}

42// Natural logarithm
43// Computes log(x) as log(2^e * m) = C*e + log(m), where the constant C =log(2)
44// and m is in the range [sqrt(1/2),sqrt(2)). In this range, the logarithm can
45// be easily approximated by a polynomial centered on m=1 for stability.
46// TODO(gonnet): Further reduce the interval allowing for lower-degree
47// polynomial interpolants -> ... -> profit!
48template <typename Packet>
49EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
50EIGEN_UNUSED
51Packet plog_float(const Packet _x)
52{
53 Packet x = _x;
54
55 const Packet cst_1 = pset1<Packet>(1.0f);
56 const Packet cst_half = pset1<Packet>(0.5f);
57 // The smallest non denormalized float number.
58 const Packet cst_min_norm_pos = pset1frombits<Packet>( 0x00800000u);
59 const Packet cst_minus_inf = pset1frombits<Packet>( 0xff800000u);
60 const Packet cst_pos_inf = pset1frombits<Packet>( 0x7f800000u);
61
62 // Polynomial coefficients.
63 const Packet cst_cephes_SQRTHF = pset1<Packet>(0.707106781186547524f);
64 const Packet cst_cephes_log_p0 = pset1<Packet>(7.0376836292E-2f);
65 const Packet cst_cephes_log_p1 = pset1<Packet>(-1.1514610310E-1f);
66 const Packet cst_cephes_log_p2 = pset1<Packet>(1.1676998740E-1f);
67 const Packet cst_cephes_log_p3 = pset1<Packet>(-1.2420140846E-1f);
68 const Packet cst_cephes_log_p4 = pset1<Packet>(+1.4249322787E-1f);
69 const Packet cst_cephes_log_p5 = pset1<Packet>(-1.6668057665E-1f);
70 const Packet cst_cephes_log_p6 = pset1<Packet>(+2.0000714765E-1f);
71 const Packet cst_cephes_log_p7 = pset1<Packet>(-2.4999993993E-1f);
72 const Packet cst_cephes_log_p8 = pset1<Packet>(+3.3333331174E-1f);
73 const Packet cst_cephes_log_q1 = pset1<Packet>(-2.12194440e-4f);
74 const Packet cst_cephes_log_q2 = pset1<Packet>(0.693359375f);
75
76 // Truncate input values to the minimum positive normal.
77 x = pmax(x, cst_min_norm_pos);
78
79 Packet e;
80 // extract significant in the range [0.5,1) and exponent
81 x = pfrexp(x,e);
82
83 // part2: Shift the inputs from the range [0.5,1) to [sqrt(1/2),sqrt(2))
84 // and shift by -1. The values are then centered around 0, which improves
85 // the stability of the polynomial evaluation.
86 // if( x < SQRTHF ) {
87 // e -= 1;
88 // x = x + x - 1.0;
89 // } else { x = x - 1.0; }
90 Packet mask = pcmp_lt(x, cst_cephes_SQRTHF);
91 Packet tmp = pand(x, mask);
92 x = psub(x, cst_1);
93 e = psub(e, pand(cst_1, mask));
94 x = padd(x, tmp);
95
96 Packet x2 = pmul(x, x);
97 Packet x3 = pmul(x2, x);
98
99 // Evaluate the polynomial approximant of degree 8 in three parts, probably
100 // to improve instruction-level parallelism.
101 Packet y, y1, y2;
102 y = pmadd(cst_cephes_log_p0, x, cst_cephes_log_p1);
103 y1 = pmadd(cst_cephes_log_p3, x, cst_cephes_log_p4);
104 y2 = pmadd(cst_cephes_log_p6, x, cst_cephes_log_p7);
105 y = pmadd(y, x, cst_cephes_log_p2);
106 y1 = pmadd(y1, x, cst_cephes_log_p5);
107 y2 = pmadd(y2, x, cst_cephes_log_p8);
108 y = pmadd(y, x3, y1);
109 y = pmadd(y, x3, y2);
110 y = pmul(y, x3);
111
112 // Add the logarithm of the exponent back to the result of the interpolation.
113 y1 = pmul(e, cst_cephes_log_q1);
114 tmp = pmul(x2, cst_half);
115 y = padd(y, y1);
116 x = psub(x, tmp);
117 y2 = pmul(e, cst_cephes_log_q2);
118 x = padd(x, y);
119 x = padd(x, y2);
120
121 Packet invalid_mask = pcmp_lt_or_nan(_x, pzero(_x));
122 Packet iszero_mask = pcmp_eq(_x,pzero(_x));
123 Packet pos_inf_mask = pcmp_eq(_x,cst_pos_inf);
124 // Filter out invalid inputs, i.e.:
125 // - negative arg will be NAN
126 // - 0 will be -INF
127 // - +INF will be +INF
128 return pselect(iszero_mask, cst_minus_inf,
129 por(pselect(pos_inf_mask,cst_pos_inf,x), invalid_mask));
130}

132/** \internal \returns log(1 + x) computed using W. Kahan's formula.
133 See: http://www.plunk.org/~hatch/rightway.php
134 */
135template<typename Packet>
136Packet generic_plog1p(const Packet& x)
137{
138 typedef typename unpacket_traits<Packet>::type ScalarType;
139 const Packet one = pset1<Packet>(ScalarType(1));
140 Packet xp1 = padd(x, one);
141 Packet small_mask = pcmp_eq(xp1, one);
142 Packet log1 = plog(xp1);
143 Packet inf_mask = pcmp_eq(xp1, log1);
144 Packet log_large = pmul(x, pdiv(log1, psub(xp1, one)));
145 return pselect(por(small_mask, inf_mask), x, log_large);
146}

148/** \internal \returns exp(x)-1 computed using W. Kahan's formula.
149 See: http://www.plunk.org/~hatch/rightway.php
150 */
151template<typename Packet>
152Packet generic_expm1(const Packet& x)
153{
154 typedef typename unpacket_traits<Packet>::type ScalarType;
155 const Packet one = pset1<Packet>(ScalarType(1));
156 const Packet neg_one = pset1<Packet>(ScalarType(-1));
157 Packet u = pexp(x);
158 Packet one_mask = pcmp_eq(u, one);
159 Packet u_minus_one = psub(u, one);
160 Packet neg_one_mask = pcmp_eq(u_minus_one, neg_one);
161 Packet logu = plog(u);
162 // The following comparison is to catch the case where
163 // exp(x) = +inf. It is written in this way to avoid having
164 // to form the constant +inf, which depends on the packet
165 // type.
166 Packet pos_inf_mask = pcmp_eq(logu, u);
167 Packet expm1 = pmul(u_minus_one, pdiv(x, logu));
168 expm1 = pselect(pos_inf_mask, u, expm1);
169 return pselect(one_mask,
170 x,
171 pselect(neg_one_mask,
172 neg_one,
173 expm1));
174}

177// Exponential function. Works by writing "x = m*log(2) + r" where
178// "m = floor(x/log(2)+1/2)" and "r" is the remainder. The result is then
179// "exp(x) = 2^m*exp(r)" where exp(r) is in the range [-1,1).
180template <typename Packet>
181EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
182EIGEN_UNUSED
183Packet pexp_float(const Packet _x)
184{
185 const Packet cst_1 = pset1<Packet>(1.0f);
186 const Packet cst_half = pset1<Packet>(0.5f);
187 const Packet cst_exp_hi = pset1<Packet>( 88.3762626647950f);
188 const Packet cst_exp_lo = pset1<Packet>(-88.3762626647949f);
189
190 const Packet cst_cephes_LOG2EF = pset1<Packet>(1.44269504088896341f);
191 const Packet cst_cephes_exp_p0 = pset1<Packet>(1.9875691500E-4f);
192 const Packet cst_cephes_exp_p1 = pset1<Packet>(1.3981999507E-3f);
193 const Packet cst_cephes_exp_p2 = pset1<Packet>(8.3334519073E-3f);
194 const Packet cst_cephes_exp_p3 = pset1<Packet>(4.1665795894E-2f);
195 const Packet cst_cephes_exp_p4 = pset1<Packet>(1.6666665459E-1f);
196 const Packet cst_cephes_exp_p5 = pset1<Packet>(5.0000001201E-1f);
197
198 // Clamp x.
199 Packet x = pmax(pmin(_x, cst_exp_hi), cst_exp_lo);
200
201 // Express exp(x) as exp(m*ln(2) + r), start by extracting
202 // m = floor(x/ln(2) + 0.5).
203 Packet m = pfloor(pmadd(x, cst_cephes_LOG2EF, cst_half));
204
205 // Get r = x - m*ln(2). If no FMA instructions are available, m*ln(2) is
206 // subtracted out in two parts, m*C1+m*C2 = m*ln(2), to avoid accumulating
207 // truncation errors.
208 Packet r;
209#ifdef EIGEN_HAS_SINGLE_INSTRUCTION_MADD
210 const Packet cst_nln2 = pset1<Packet>(-0.6931471805599453f);
211 r = pmadd(m, cst_nln2, x);
212#else
213 const Packet cst_cephes_exp_C1 = pset1<Packet>(0.693359375f);
214 const Packet cst_cephes_exp_C2 = pset1<Packet>(-2.12194440e-4f);
215 r = psub(x, pmul(m, cst_cephes_exp_C1));
216 r = psub(r, pmul(m, cst_cephes_exp_C2));
217#endif
218
219 Packet r2 = pmul(r, r);
220
221 // TODO(gonnet): Split into odd/even polynomials and try to exploit
222 // instruction-level parallelism.
223 Packet y = cst_cephes_exp_p0;
224 y = pmadd(y, r, cst_cephes_exp_p1);
225 y = pmadd(y, r, cst_cephes_exp_p2);
226 y = pmadd(y, r, cst_cephes_exp_p3);
227 y = pmadd(y, r, cst_cephes_exp_p4);
228 y = pmadd(y, r, cst_cephes_exp_p5);
229 y = pmadd(y, r2, r);
230 y = padd(y, cst_1);
231
232 // Return 2^m * exp(r).
233 return pmax(pldexp(y,m), _x);
234}

237template<>
238EIGEN_DEVICE_FUNC inline float pexp(const float& a) { return pexp_float(a); }

240template <typename Packet>
241EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
242EIGEN_UNUSED
243Packet pexp_double(const Packet _x)
244{
245 Packet x = _x;
246
247 const Packet cst_1 = pset1<Packet>(1.0);
248 const Packet cst_2 = pset1<Packet>(2.0);
249 const Packet cst_half = pset1<Packet>(0.5);
250
251 const Packet cst_exp_hi = pset1<Packet>(709.437);
252 const Packet cst_exp_lo = pset1<Packet>(-709.436139303);
253
254 const Packet cst_cephes_LOG2EF = pset1<Packet>(1.4426950408889634073599);
255 const Packet cst_cephes_exp_p0 = pset1<Packet>(1.26177193074810590878e-4);
256 const Packet cst_cephes_exp_p1 = pset1<Packet>(3.02994407707441961300e-2);
257 const Packet cst_cephes_exp_p2 = pset1<Packet>(9.99999999999999999910e-1);
258 const Packet cst_cephes_exp_q0 = pset1<Packet>(3.00198505138664455042e-6);
259 const Packet cst_cephes_exp_q1 = pset1<Packet>(2.52448340349684104192e-3);
260 const Packet cst_cephes_exp_q2 = pset1<Packet>(2.27265548208155028766e-1);
261 const Packet cst_cephes_exp_q3 = pset1<Packet>(2.00000000000000000009e0);
262 const Packet cst_cephes_exp_C1 = pset1<Packet>(0.693145751953125);
263 const Packet cst_cephes_exp_C2 = pset1<Packet>(1.42860682030941723212e-6);
264
265 Packet tmp, fx;
266
267 // clamp x
268 x = pmax(pmin(x, cst_exp_hi), cst_exp_lo);
269 // Express exp(x) as exp(g + n*log(2)).
270 fx = pmadd(cst_cephes_LOG2EF, x, cst_half);
271
272 // Get the integer modulus of log(2), i.e. the "n" described above.
273 fx = pfloor(fx);
274
275 // Get the remainder modulo log(2), i.e. the "g" described above. Subtract
276 // n*log(2) out in two steps, i.e. n*C1 + n*C2, C1+C2=log2 to get the last
277 // digits right.
278 tmp = pmul(fx, cst_cephes_exp_C1);
279 Packet z = pmul(fx, cst_cephes_exp_C2);
280 x = psub(x, tmp);
281 x = psub(x, z);
282
283 Packet x2 = pmul(x, x);
284
285 // Evaluate the numerator polynomial of the rational interpolant.
286 Packet px = cst_cephes_exp_p0;
287 px = pmadd(px, x2, cst_cephes_exp_p1);
288 px = pmadd(px, x2, cst_cephes_exp_p2);
289 px = pmul(px, x);
290
291 // Evaluate the denominator polynomial of the rational interpolant.
292 Packet qx = cst_cephes_exp_q0;
293 qx = pmadd(qx, x2, cst_cephes_exp_q1);
294 qx = pmadd(qx, x2, cst_cephes_exp_q2);
295 qx = pmadd(qx, x2, cst_cephes_exp_q3);
296
297 // I don't really get this bit, copied from the SSE2 routines, so...
298 // TODO(gonnet): Figure out what is going on here, perhaps find a better
299 // rational interpolant?
300 x = pdiv(px, psub(qx, px));
301 x = pmadd(cst_2, x, cst_1);
302
303 // Construct the result 2^n * exp(g) = e * x. The max is used to catch
304 // non-finite values in the input.
305 return pmax(pldexp(x,fx), _x);
306}

312// The following code is inspired by the following stack-overflow answer:
313// https://stackoverflow.com/questions/30463616/payne-hanek-algorithm-implementation-in-c/30465751#30465751
314// It has been largely optimized:
315// - By-pass calls to frexp.
316// - Aligned loads of required 96 bits of 2/pi. This is accomplished by
317// (1) balancing the mantissa and exponent to the required bits of 2/pi are
318// aligned on 8-bits, and (2) replicating the storage of the bits of 2/pi.
319// - Avoid a branch in rounding and extraction of the remaining fractional part.
320// Overall, I measured a speed up higher than x2 on x86-64.
321inline float trig_reduce_huge (float xf, int *quadrant)
322{
323 using Eigen::numext::int32_t;
324 using Eigen::numext::uint32_t;
325 using Eigen::numext::int64_t;
326 using Eigen::numext::uint64_t;
327
328 const double pio2_62 = 3.4061215800865545e-19; // pi/2 * 2^-62
329 const uint64_t zero_dot_five = uint64_t(1) << 61; // 0.5 in 2.62-bit fixed-point foramt
330
331 // 192 bits of 2/pi for Payne-Hanek reduction
332 // Bits are introduced by packet of 8 to enable aligned reads.
333 static const uint32_t two_over_pi [] =
334 {
335 0x00000028, 0x000028be, 0x0028be60, 0x28be60db,
336 0xbe60db93, 0x60db9391, 0xdb939105, 0x9391054a,
337 0x91054a7f, 0x054a7f09, 0x4a7f09d5, 0x7f09d5f4,
338 0x09d5f47d, 0xd5f47d4d, 0xf47d4d37, 0x7d4d3770,
339 0x4d377036, 0x377036d8, 0x7036d8a5, 0x36d8a566,
340 0xd8a5664f, 0xa5664f10, 0x664f10e4, 0x4f10e410,
341 0x10e41000, 0xe4100000
342 };
343
344 uint32_t xi = numext::as_uint(xf);
345 // Below, -118 = -126 + 8.
346 // -126 is to get the exponent,
347 // +8 is to enable alignment of 2/pi's bits on 8 bits.
348 // This is possible because the fractional part of x as only 24 meaningful bits.
349 uint32_t e = (xi >> 23) - 118;
350 // Extract the mantissa and shift it to align it wrt the exponent
351 xi = ((xi & 0x007fffffu)| 0x00800000u) << (e & 0x7);
352
353 uint32_t i = e >> 3;
354 uint32_t twoopi_1 = two_over_pi[i-1];
355 uint32_t twoopi_2 = two_over_pi[i+3];
356 uint32_t twoopi_3 = two_over_pi[i+7];
357
358 // Compute x * 2/pi in 2.62-bit fixed-point format.
359 uint64_t p;
360 p = uint64_t(xi) * twoopi_3;
361 p = uint64_t(xi) * twoopi_2 + (p >> 32);
362 p = (uint64_t(xi * twoopi_1) << 32) + p;
363
364 // Round to nearest: add 0.5 and extract integral part.
365 uint64_t q = (p + zero_dot_five) >> 62;
366 *quadrant = int(q);
367 // Now it remains to compute "r = x - q*pi/2" with high accuracy,
368 // since we have p=x/(pi/2) with high accuracy, we can more efficiently compute r as:
369 // r = (p-q)*pi/2,
370 // where the product can be be carried out with sufficient accuracy using double precision.
371 p -= q<<62;
372 return float(double(int64_t(p)) * pio2_62);
373}

375template<bool ComputeSine,typename Packet>
376EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
377EIGEN_UNUSED
378#if EIGEN_GNUC_AT_LEAST(4,4) && EIGEN_COMP_GNUC_STRICT
379__attribute__((optimize("-fno-unsafe-math-optimizations")))
380#endif
381Packet psincos_float(const Packet& _x)
382{
383// Workaround -ffast-math aggressive optimizations
384// See bug 1674
385#if EIGEN_COMP_CLANG && defined(EIGEN_VECTORIZE_SSE)
386#define EIGEN_SINCOS_DONT_OPT(X) __asm__ ("" : "+x" (X));
387#else
388#define EIGEN_SINCOS_DONT_OPT(X)
389#endif
390
391 typedef typename unpacket_traits<Packet>::integer_packet PacketI;
392
393 const Packet cst_2oPI = pset1<Packet>(0.636619746685028076171875f); // 2/PI
394 const Packet cst_rounding_magic = pset1<Packet>(12582912); // 2^23 for rounding
395 const PacketI csti_1 = pset1<PacketI>(1);
396 const Packet cst_sign_mask = pset1frombits<Packet>(0x80000000u);
397
398 Packet x = pabs(_x);
399
400 // Scale x by 2/Pi to find x's octant.
401 Packet y = pmul(x, cst_2oPI);
402
403 // Rounding trick:
404 Packet y_round = padd(y, cst_rounding_magic);
405 EIGEN_SINCOS_DONT_OPT(y_round)
406 PacketI y_int = preinterpret<PacketI>(y_round); // last 23 digits represent integer (if abs(x)<2^24)
407 y = psub(y_round, cst_rounding_magic); // nearest integer to x*4/pi
408
409 // Reduce x by y octants to get: -Pi/4 <= x <= +Pi/4
410 // using "Extended precision modular arithmetic"
411 #if defined(EIGEN_HAS_SINGLE_INSTRUCTION_MADD)
412 // This version requires true FMA for high accuracy
413 // It provides a max error of 1ULP up to (with absolute_error < 5.9605e-08):
414 const float huge_th = ComputeSine ? 117435.992f : 71476.0625f;
415 x = pmadd(y, pset1<Packet>(-1.57079601287841796875f), x);
416 x = pmadd(y, pset1<Packet>(-3.1391647326017846353352069854736328125e-07f), x);
417 x = pmadd(y, pset1<Packet>(-5.390302529957764765544681040410068817436695098876953125e-15f), x);
418 #else
419 // Without true FMA, the previous set of coefficients maintain 1ULP accuracy
420 // up to x<15.7 (for sin), but accuracy is immediately lost for x>15.7.
421 // We thus use one more iteration to maintain 2ULPs up to reasonably large inputs.
422
423 // The following set of coefficients maintain 1ULP up to 9.43 and 14.16 for sin and cos respectively.
424 // and 2 ULP up to:
425 const float huge_th = ComputeSine ? 25966.f : 18838.f;
426 x = pmadd(y, pset1<Packet>(-1.5703125), x); // = 0xbfc90000
427 EIGEN_SINCOS_DONT_OPT(x)
428 x = pmadd(y, pset1<Packet>(-0.000483989715576171875), x); // = 0xb9fdc000
429 EIGEN_SINCOS_DONT_OPT(x)
430 x = pmadd(y, pset1<Packet>(1.62865035235881805419921875e-07), x); // = 0x342ee000
431 x = pmadd(y, pset1<Packet>(5.5644315544167710640977020375430583953857421875e-11), x); // = 0x2e74b9ee
432
433 // For the record, the following set of coefficients maintain 2ULP up
434 // to a slightly larger range:
435 // const float huge_th = ComputeSine ? 51981.f : 39086.125f;
436 // but it slightly fails to maintain 1ULP for two values of sin below pi.
437 // x = pmadd(y, pset1<Packet>(-3.140625/2.), x);
438 // x = pmadd(y, pset1<Packet>(-0.00048351287841796875), x);
439 // x = pmadd(y, pset1<Packet>(-3.13855707645416259765625e-07), x);
440 // x = pmadd(y, pset1<Packet>(-6.0771006282767103812147979624569416046142578125e-11), x);
441
442 // For the record, with only 3 iterations it is possible to maintain
443 // 1 ULP up to 3PI (maybe more) and 2ULP up to 255.
444 // The coefficients are: 0xbfc90f80, 0xb7354480, 0x2e74b9ee
445 #endif
446
447 if(predux_any(pcmp_le(pset1<Packet>(huge_th),pabs(_x))))
448 {
449 const int PacketSize = unpacket_traits<Packet>::size;
450 EIGEN_ALIGN_TO_BOUNDARY(sizeof(Packet)) float vals[PacketSize];
451 EIGEN_ALIGN_TO_BOUNDARY(sizeof(Packet)) float x_cpy[PacketSize];
452 EIGEN_ALIGN_TO_BOUNDARY(sizeof(Packet)) int y_int2[PacketSize];
453 pstoreu(vals, pabs(_x));
454 pstoreu(x_cpy, x);
455 pstoreu(y_int2, y_int);
456 for(int k=0; k<PacketSize;++k)
457 {
458 float val = vals[k];
459 if(val>=huge_th && (numext::isfinite)(val))
460 x_cpy[k] = trig_reduce_huge(val,&y_int2[k]);
461 }
462 x = ploadu<Packet>(x_cpy);
463 y_int = ploadu<PacketI>(y_int2);
464 }
465
466 // Compute the sign to apply to the polynomial.
467 // sin: sign = second_bit(y_int) xor signbit(_x)
468 // cos: sign = second_bit(y_int+1)
469 Packet sign_bit = ComputeSine ? pxor(_x, preinterpret<Packet>(plogical_shift_left<30>(y_int)))
470 : preinterpret<Packet>(plogical_shift_left<30>(padd(y_int,csti_1)));
471 sign_bit = pand(sign_bit, cst_sign_mask); // clear all but left most bit
472
473 // Get the polynomial selection mask from the second bit of y_int
474 // We'll calculate both (sin and cos) polynomials and then select from the two.
475 Packet poly_mask = preinterpret<Packet>(pcmp_eq(pand(y_int, csti_1), pzero(y_int)));
476
477 Packet x2 = pmul(x,x);
478
479 // Evaluate the cos(x) polynomial. (-Pi/4 <= x <= Pi/4)
480 Packet y1 = pset1<Packet>(2.4372266125283204019069671630859375e-05f);
481 y1 = pmadd(y1, x2, pset1<Packet>(-0.00138865201734006404876708984375f ));
482 y1 = pmadd(y1, x2, pset1<Packet>(0.041666619479656219482421875f ));
483 y1 = pmadd(y1, x2, pset1<Packet>(-0.5f));
484 y1 = pmadd(y1, x2, pset1<Packet>(1.f));
485
486 // Evaluate the sin(x) polynomial. (Pi/4 <= x <= Pi/4)
487 // octave/matlab code to compute those coefficients:
488 // x = (0:0.0001:pi/4)';
489 // A = [x.^3 x.^5 x.^7];
490 // w = ((1.-(x/(pi/4)).^2).^5)*2000+1; # weights trading relative accuracy
491 // c = (A'*diag(w)*A)\(A'*diag(w)*(sin(x)-x)); # weighted LS, linear coeff forced to 1
492 // printf('%.64f\n %.64f\n%.64f\n', c(3), c(2), c(1))
493 //
494 Packet y2 = pset1<Packet>(-0.0001959234114083702898469196984621021329076029360294342041015625f);
495 y2 = pmadd(y2, x2, pset1<Packet>( 0.0083326873655616851693794799871284340042620897293090820312500000f));
496 y2 = pmadd(y2, x2, pset1<Packet>(-0.1666666203982298255503735617821803316473960876464843750000000000f));
497 y2 = pmul(y2, x2);
498 y2 = pmadd(y2, x, x);
499
500 // Select the correct result from the two polynomials.
501 y = ComputeSine ? pselect(poly_mask,y2,y1)
502 : pselect(poly_mask,y1,y2);
503
504 // Update the sign and filter huge inputs
505 return pxor(y, sign_bit);
506
507#undef EIGEN_SINCOS_DONT_OPT
508}

526/* polevl (modified for Eigen)
527 *
528 * Evaluate polynomial
529 *
530 *
531 *
532 * SYNOPSIS:
533 *
534 * int N;
535 * Scalar x, y, coef[N+1];
536 *
537 * y = polevl<decltype(x), N>( x, coef);
538 *
539 *
540 *
541 * DESCRIPTION:
542 *
543 * Evaluates polynomial of degree N:
544 *
545 * 2 N
546 * y = C + C x + C x +...+ C x
547 * 0 1 2 N
548 *
549 * Coefficients are stored in reverse order:
550 *
551 * coef[0] = C , ..., coef[N] = C .
552 * N 0
553 *
554 * The function p1evl() assumes that coef[N] = 1.0 and is
555 * omitted from the array. Its calling arguments are
556 * otherwise the same as polevl().
557 *
558 *
559 * The Eigen implementation is templatized. For best speed, store
560 * coef as a const array (constexpr), e.g.
561 *
562 * const double coef[] = {1.0, 2.0, 3.0, ...};
563 *
564 */
565template <typename Packet, int N>
566struct ppolevl {
567 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Packet run(const Packet& x, const typename unpacket_traits<Packet>::type coeff[]) {
568 EIGEN_STATIC_ASSERT((N > 0), YOU_MADE_A_PROGRAMMING_MISTAKE);
569 return pmadd(ppolevl<Packet, N-1>::run(x, coeff), x, pset1<Packet>(coeff[N]));
570 }
571};
572
573template <typename Packet>
574struct ppolevl<Packet, 0> {
575 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Packet run(const Packet& x, const typename unpacket_traits<Packet>::type coeff[]) {
576 EIGEN_UNUSED_VARIABLE(x);
577 return pset1<Packet>(coeff[0]);
578 }
579};

581/* chbevl (modified for Eigen)
582 *
583 * Evaluate Chebyshev series
584 *
585 *
586 *
587 * SYNOPSIS:
588 *
589 * int N;
590 * Scalar x, y, coef[N], chebevl();
591 *
592 * y = chbevl( x, coef, N );
593 *
594 *
595 *
596 * DESCRIPTION:
597 *
598 * Evaluates the series
599 *
600 * N-1
601 * - '
602 * y = > coef[i] T (x/2)
603 * - i
604 * i=0
605 *
606 * of Chebyshev polynomials Ti at argument x/2.
607 *
608 * Coefficients are stored in reverse order, i.e. the zero
609 * order term is last in the array. Note N is the number of
610 * coefficients, not the order.
611 *
612 * If coefficients are for the interval a to b, x must
613 * have been transformed to x -> 2(2x - b - a)/(b-a) before
614 * entering the routine. This maps x from (a, b) to (-1, 1),
615 * over which the Chebyshev polynomials are defined.
616 *
617 * If the coefficients are for the inverted interval, in
618 * which (a, b) is mapped to (1/b, 1/a), the transformation
619 * required is x -> 2(2ab/x - b - a)/(b-a). If b is infinity,
620 * this becomes x -> 4a/x - 1.
621 *
622 *
623 *
624 * SPEED:
625 *
626 * Taking advantage of the recurrence properties of the
627 * Chebyshev polynomials, the routine requires one more
628 * addition per loop than evaluating a nested polynomial of
629 * the same degree.
630 *
631 */
632
633template <typename Packet, int N>
634struct pchebevl {
635 EIGEN_DEVICE_FUNC
636 static EIGEN_STRONG_INLINE Packet run(Packet x, const typename unpacket_traits<Packet>::type coef[]) {
637 typedef typename unpacket_traits<Packet>::type Scalar;
638 Packet b0 = pset1<Packet>(coef[0]);
639 Packet b1 = pset1<Packet>(static_cast<Scalar>(0.f));
640 Packet b2;
641
642 for (int i = 1; i < N; i++) {
643 b2 = b1;
644 b1 = b0;
645 b0 = psub(pmadd(x, b1, pset1<Packet>(coef[i])), b2);
646 }
647
648 return pmul(pset1<Packet>(static_cast<Scalar>(0.5f)), psub(b0, b2));
649 }
650};
