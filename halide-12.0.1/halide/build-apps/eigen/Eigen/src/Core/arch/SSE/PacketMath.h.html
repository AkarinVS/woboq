34// With GCC's default ABI version, a __m128 or __m256 are the same types and therefore we cannot
35// have overloads for both types without linking error.
36// One solution is to increase ABI version using -fabi-version=4 (or greater).
37// Otherwise, we workaround this inconvenience by wrapping 128bit types into the following helper
38// structure:
39typedef eigen_packet_wrapper<__m128> Packet4f;
40typedef eigen_packet_wrapper<__m128d> Packet2d;
41#else
42typedef __m128 Packet4f;
43typedef __m128d Packet2d;
44#endif

46typedef eigen_packet_wrapper<__m128i, 0> Packet4i;
47typedef eigen_packet_wrapper<__m128i, 1> Packet16b;

54#define EIGEN_SSE_SHUFFLE_MASK(p,q,r,s) ((s)<<6|(r)<<4|(q)<<2|(p))
55
56#define vec4f_swizzle1(v,p,q,r,s) \
57 (_mm_castsi128_ps(_mm_shuffle_epi32( _mm_castps_si128(v), EIGEN_SSE_SHUFFLE_MASK(p,q,r,s))))
58
59#define vec4i_swizzle1(v,p,q,r,s) \
60 (_mm_shuffle_epi32( v, EIGEN_SSE_SHUFFLE_MASK(p,q,r,s)))
61
62#define vec2d_swizzle1(v,p,q) \
63 (_mm_castsi128_pd(_mm_shuffle_epi32( _mm_castpd_si128(v), EIGEN_SSE_SHUFFLE_MASK(2*p,2*p+1,2*q,2*q+1))))
64
65#define vec4f_swizzle2(a,b,p,q,r,s) \
66 (_mm_shuffle_ps( (a), (b), EIGEN_SSE_SHUFFLE_MASK(p,q,r,s)))
67
68#define vec4i_swizzle2(a,b,p,q,r,s) \
69 (_mm_castps_si128( (_mm_shuffle_ps( _mm_castsi128_ps(a), _mm_castsi128_ps(b), EIGEN_SSE_SHUFFLE_MASK(p,q,r,s)))))

233// GCC generates a shufps instruction for _mm_set1_ps/_mm_load1_ps instead of the more efficient pshufd instruction.
234// However, using inrinsics for pset1 makes gcc to generate crappy code in some cases (see bug 203)
235// Using inline assembly is also not an option because then gcc fails to reorder properly the instructions.
236// Therefore, we introduced the pload1 functions to be used in product kernels for which bug 203 does not apply.
237// Also note that with AVX, we want it to generate a vbroadcastss.
238#if EIGEN_COMP_GNUC_STRICT && (!defined __AVX__)
239template<> EIGEN_STRONG_INLINE Packet4f pload1<Packet4f>(const float *from) {
240 return vec4f_swizzle1(_mm_load_ss(from),0,0,0,0);
241}
242#endif

337template<> EIGEN_STRONG_INLINE Packet4f pmin<Packet4f>(const Packet4f& a, const Packet4f& b) {
338#if EIGEN_COMP_GNUC && EIGEN_COMP_GNUC < 63
339 // There appears to be a bug in GCC, by which the optimizer may
340 // flip the argument order in calls to _mm_min_ps, so we have to
341 // resort to inline ASM here. This is supposed to be fixed in gcc6.3,
342 // see also: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=72867
343 #ifdef EIGEN_VECTORIZE_AVX
344 Packet4f res;
345 asm("vminps %[a], %[b], %[res]" : [res] "=x" (res) : [a] "x" (a), [b] "x" (b));
346 #else
347 Packet4f res = b;
348 asm("minps %[a], %[res]" : [res] "+x" (res) : [a] "x" (a));
349 #endif
350 return res;
351#else
352 // Arguments are reversed to match NaN propagation behavior of std::min.
353 return _mm_min_ps(b, a);
354#endif
355}
