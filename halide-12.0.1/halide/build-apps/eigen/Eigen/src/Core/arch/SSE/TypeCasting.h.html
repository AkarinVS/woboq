55template<> EIGEN_STRONG_INLINE Packet4i pcast<Packet4f, Packet4i>(const Packet4f& a) {
56 return _mm_cvttps_epi32(a);

59template<> EIGEN_STRONG_INLINE Packet4f pcast<Packet4i, Packet4f>(const Packet4i& a) {

63template<> EIGEN_STRONG_INLINE Packet4f pcast<Packet2d, Packet4f>(const Packet2d& a, const Packet2d& b) {
64 return _mm_shuffle_ps(_mm_cvtpd_ps(a), _mm_cvtpd_ps(b), (1 << 2) | (1 << 6));
65}

67template<> EIGEN_STRONG_INLINE Packet2d pcast<Packet4f, Packet2d>(const Packet4f& a) {

94template<> EIGEN_STRONG_INLINE Packet4f pcast<Packet4h, Packet4f>(const Packet4h& a) {
95 __int64_t a64 = _mm_cvtm64_si64(a.x);
96 Eigen::half h = raw_uint16_to_half(static_cast<unsigned short>(a64));
97 float f1 = static_cast<float>(h);
98 h = raw_uint16_to_half(static_cast<unsigned short>(a64 >> 16));
99 float f2 = static_cast<float>(h);
100 h = raw_uint16_to_half(static_cast<unsigned short>(a64 >> 32));
101 float f3 = static_cast<float>(h);
102 h = raw_uint16_to_half(static_cast<unsigned short>(a64 >> 48));
103 float f4 = static_cast<float>(h);
104 return _mm_set_ps(f4, f3, f2, f1);
105}

116template<> EIGEN_STRONG_INLINE Packet4h pcast<Packet4f, Packet4h>(const Packet4f& a) {
117 EIGEN_ALIGN16 float aux[4];
118 pstore(aux, a);
119 Eigen::half h0(aux[0]);
120 Eigen::half h1(aux[1]);
121 Eigen::half h2(aux[2]);
122 Eigen::half h3(aux[3]);
123
124 Packet4h result;
125 result.x = _mm_set_pi16(h3.x, h2.x, h1.x, h0.x);
126 return result;
127}
