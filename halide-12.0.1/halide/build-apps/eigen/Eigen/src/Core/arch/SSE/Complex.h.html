95template<> EIGEN_STRONG_INLINE Packet2cf pset1<Packet2cf>(const std::complex<float>& from)
96{
97 Packet2cf res;
98#if EIGEN_GNUC_AT_MOST(4,2)
99 // Workaround annoying "may be used uninitialized in this function" warning with gcc 4.2
100 res.v = _mm_loadl_pi(_mm_set1_ps(0.0f), reinterpret_cast<const __m64*>(&from));
101#elif EIGEN_GNUC_AT_LEAST(4,6)
102 // Suppress annoying "may be used uninitialized in this function" warning with gcc >= 4.6
103 #pragma GCC diagnostic push
104 #pragma GCC diagnostic ignored "-Wuninitialized"
105 res.v = _mm_loadl_pi(res.v, (const __m64*)&from);
106 #pragma GCC diagnostic pop
107#else
108 res.v = _mm_loadl_pi(res.v, (const __m64*)&from);
109#endif
110 return Packet2cf(_mm_movelh_ps(res.v,res.v));
111}

135template<> EIGEN_STRONG_INLINE std::complex<float> pfirst<Packet2cf>(const Packet2cf& a)
136{
137 #if EIGEN_GNUC_AT_MOST(4,3)
138 // Workaround gcc 4.2 ICE - this is not performance wise ideal, but who cares...
139 // This workaround also fix invalid code generation with gcc 4.3
140 EIGEN_ALIGN16 std::complex<float> res[2];
141 _mm_store_ps((float*)res, a.v);
142 return res[0];
143 #else
144 std::complex<float> res;
145 _mm_storel_pi((__m64*)&res, a.v);
146 return res;
147 #endif
148}

402EIGEN_DEVICE_FUNC inline void
403ptranspose(PacketBlock<Packet2cf,2>& kernel) {
404 __m128d w1 = _mm_castps_pd(kernel.packet[0].v);
405 __m128d w2 = _mm_castps_pd(kernel.packet[1].v);
406
407 __m128 tmp = _mm_castpd_ps(_mm_unpackhi_pd(w1, w2));
408 kernel.packet[0].v = _mm_castpd_ps(_mm_unpacklo_pd(w1, w2));
409 kernel.packet[1].v = tmp;
410}
