25// copy_using_evaluator_traits is based on assign_traits
26
27template <typename DstEvaluator, typename SrcEvaluator, typename AssignFunc, int MaxPacketSize = -1>
28struct copy_using_evaluator_traits
29{
30 typedef typename DstEvaluator::XprType Dst;
31 typedef typename Dst::Scalar DstScalar;
32
33 enum {
34 DstFlags = DstEvaluator::Flags,
35 SrcFlags = SrcEvaluator::Flags
36 };
37
38public:
39 enum {
40 DstAlignment = DstEvaluator::Alignment,
41 SrcAlignment = SrcEvaluator::Alignment,
42 DstHasDirectAccess = (DstFlags & DirectAccessBit) == DirectAccessBit,
43 JointAlignment = EIGEN_PLAIN_ENUM_MIN(DstAlignment,SrcAlignment)
44 };
45
46private:
47 enum {
48 InnerSize = int(Dst::IsVectorAtCompileTime) ? int(Dst::SizeAtCompileTime)
49 : int(DstFlags)&RowMajorBit ? int(Dst::ColsAtCompileTime)
50 : int(Dst::RowsAtCompileTime),
51 InnerMaxSize = int(Dst::IsVectorAtCompileTime) ? int(Dst::MaxSizeAtCompileTime)
52 : int(DstFlags)&RowMajorBit ? int(Dst::MaxColsAtCompileTime)
53 : int(Dst::MaxRowsAtCompileTime),
54 RestrictedInnerSize = EIGEN_SIZE_MIN_PREFER_FIXED(InnerSize,MaxPacketSize),
55 RestrictedLinearSize = EIGEN_SIZE_MIN_PREFER_FIXED(Dst::SizeAtCompileTime,MaxPacketSize),
56 OuterStride = int(outer_stride_at_compile_time<Dst>::ret),
57 MaxSizeAtCompileTime = Dst::SizeAtCompileTime
58 };
59
60 // TODO distinguish between linear traversal and inner-traversals
61 typedef typename find_best_packet<DstScalar,RestrictedLinearSize>::type LinearPacketType;
62 typedef typename find_best_packet<DstScalar,RestrictedInnerSize>::type InnerPacketType;
63
64 enum {
65 LinearPacketSize = unpacket_traits<LinearPacketType>::size,
66 InnerPacketSize = unpacket_traits<InnerPacketType>::size
67 };
68
69public:
70 enum {
71 LinearRequiredAlignment = unpacket_traits<LinearPacketType>::alignment,
72 InnerRequiredAlignment = unpacket_traits<InnerPacketType>::alignment
73 };
74
75private:
76 enum {
77 DstIsRowMajor = DstFlags&RowMajorBit,
78 SrcIsRowMajor = SrcFlags&RowMajorBit,
79 StorageOrdersAgree = (int(DstIsRowMajor) == int(SrcIsRowMajor)),
80 MightVectorize = bool(StorageOrdersAgree)
81 && (int(DstFlags) & int(SrcFlags) & ActualPacketAccessBit)
82 && bool(functor_traits<AssignFunc>::PacketAccess),
83 MayInnerVectorize = MightVectorize
84 && int(InnerSize)!=Dynamic && int(InnerSize)%int(InnerPacketSize)==0
85 && int(OuterStride)!=Dynamic && int(OuterStride)%int(InnerPacketSize)==0
86 && (EIGEN_UNALIGNED_VECTORIZE || int(JointAlignment)>=int(InnerRequiredAlignment)),
87 MayLinearize = bool(StorageOrdersAgree) && (int(DstFlags) & int(SrcFlags) & LinearAccessBit),
88 MayLinearVectorize = bool(MightVectorize) && bool(MayLinearize) && bool(DstHasDirectAccess)
89 && (EIGEN_UNALIGNED_VECTORIZE || (int(DstAlignment)>=int(LinearRequiredAlignment)) || MaxSizeAtCompileTime == Dynamic),
90 /* If the destination isn't aligned, we have to do runtime checks and we don't unroll,
91 so it's only good for large enough sizes. */
92 MaySliceVectorize = bool(MightVectorize) && bool(DstHasDirectAccess)
93 && (int(InnerMaxSize)==Dynamic || int(InnerMaxSize)>=(EIGEN_UNALIGNED_VECTORIZE?InnerPacketSize:(3*InnerPacketSize)))
94 /* slice vectorization can be slow, so we only want it if the slices are big, which is
95 indicated by InnerMaxSize rather than InnerSize, think of the case of a dynamic block
96 in a fixed-size matrix
97 However, with EIGEN_UNALIGNED_VECTORIZE and unrolling, slice vectorization is still worth it */
98 };
99
100public:
101 enum {
102 Traversal = (int(MayLinearVectorize) && (LinearPacketSize>InnerPacketSize)) ? int(LinearVectorizedTraversal)
103 : int(MayInnerVectorize) ? int(InnerVectorizedTraversal)
104 : int(MayLinearVectorize) ? int(LinearVectorizedTraversal)
105 : int(MaySliceVectorize) ? int(SliceVectorizedTraversal)
106 : int(MayLinearize) ? int(LinearTraversal)
107 : int(DefaultTraversal),
108 Vectorized = int(Traversal) == InnerVectorizedTraversal
109 || int(Traversal) == LinearVectorizedTraversal
110 || int(Traversal) == SliceVectorizedTraversal
111 };
112
113 typedef typename conditional<int(Traversal)==LinearVectorizedTraversal, LinearPacketType, InnerPacketType>::type PacketType;
114
115private:
116 enum {
117 ActualPacketSize = int(Traversal)==LinearVectorizedTraversal ? LinearPacketSize
118 : Vectorized ? InnerPacketSize
119 : 1,
120 UnrollingLimit = EIGEN_UNROLLING_LIMIT * ActualPacketSize,
121 MayUnrollCompletely = int(Dst::SizeAtCompileTime) != Dynamic
122 && int(Dst::SizeAtCompileTime) * (int(DstEvaluator::CoeffReadCost)+int(SrcEvaluator::CoeffReadCost)) <= int(UnrollingLimit),
123 MayUnrollInner = int(InnerSize) != Dynamic
124 && int(InnerSize) * (int(DstEvaluator::CoeffReadCost)+int(SrcEvaluator::CoeffReadCost)) <= int(UnrollingLimit)
125 };
126
127public:
128 enum {
129 Unrolling = (int(Traversal) == int(InnerVectorizedTraversal) || int(Traversal) == int(DefaultTraversal))
130 ? (
131 int(MayUnrollCompletely) ? int(CompleteUnrolling)
132 : int(MayUnrollInner) ? int(InnerUnrolling)
133 : int(NoUnrolling)
134 )
135 : int(Traversal) == int(LinearVectorizedTraversal)
136 ? ( bool(MayUnrollCompletely) && ( EIGEN_UNALIGNED_VECTORIZE || (int(DstAlignment)>=int(LinearRequiredAlignment)))
137 ? int(CompleteUnrolling)
138 : int(NoUnrolling) )
139 : int(Traversal) == int(LinearTraversal)
140 ? ( bool(MayUnrollCompletely) ? int(CompleteUnrolling)
141 : int(NoUnrolling) )
142#if EIGEN_UNALIGNED_VECTORIZE
143 : int(Traversal) == int(SliceVectorizedTraversal)
144 ? ( bool(MayUnrollInner) ? int(InnerUnrolling)
145 : int(NoUnrolling) )
146#endif
147 : int(NoUnrolling)
148 };

367template <bool IsAligned = false>
368struct unaligned_dense_assignment_loop

375template <>
376struct unaligned_dense_assignment_loop<false>
377{
378 // MSVC must not inline this functions. If it does, it fails to optimize the
379 // packet access path.
380 // FIXME check which version exhibits this issue
381#if EIGEN_COMP_MSVC
382 template <typename Kernel>
383 static EIGEN_DONT_INLINE void run(Kernel &kernel,
384 Index start,
385 Index end)
386#else
387 template <typename Kernel>
388 EIGEN_DEVICE_FUNC static EIGEN_STRONG_INLINE void run(Kernel &kernel,
389 Index start,
390 Index end)
391#endif
392 {
393 for (Index index = start; index < end; ++index)
394 kernel.assignCoeff(index);
395 }

398template<typename Kernel>
399struct dense_assignment_loop<Kernel, LinearVectorizedTraversal, NoUnrolling>
400{
401 EIGEN_DEVICE_FUNC static EIGEN_STRONG_INLINE void run(Kernel &kernel)
402 {
403 const Index size = kernel.size();
404 typedef typename Kernel::Scalar Scalar;
405 typedef typename Kernel::PacketType PacketType;
406 enum {
407 requestedAlignment = Kernel::AssignmentTraits::LinearRequiredAlignment,
408 packetSize = unpacket_traits<PacketType>::size,
409 dstIsAligned = int(Kernel::AssignmentTraits::DstAlignment)>=int(requestedAlignment),
410 dstAlignment = packet_traits<Scalar>::AlignedOnScalar ? int(requestedAlignment)
411 : int(Kernel::AssignmentTraits::DstAlignment),
412 srcAlignment = Kernel::AssignmentTraits::JointAlignment
413 };
414 const Index alignedStart = dstIsAligned ? 0 : internal::first_aligned<requestedAlignment>(kernel.dstDataPtr(), size);
415 const Index alignedEnd = alignedStart + ((size-alignedStart)/packetSize)*packetSize;
416
417 unaligned_dense_assignment_loop<dstIsAligned!=0>::run(kernel, 0, alignedStart);
418
419 for(Index index = alignedStart; index < alignedEnd; index += packetSize)
420 kernel.template assignPacket<dstAlignment, srcAlignment, PacketType>(index);
421
422 unaligned_dense_assignment_loop<>::run(kernel, alignedEnd, size);
423 }
424};
425
426template<typename Kernel>
427struct dense_assignment_loop<Kernel, LinearVectorizedTraversal, CompleteUnrolling>
428{
429 EIGEN_DEVICE_FUNC static EIGEN_STRONG_INLINE void run(Kernel &kernel)
430 {
431 typedef typename Kernel::DstEvaluatorType::XprType DstXprType;
432 typedef typename Kernel::PacketType PacketType;
433
434 enum { size = DstXprType::SizeAtCompileTime,
435 packetSize =unpacket_traits<PacketType>::size,
436 alignedSize = (size/packetSize)*packetSize };
437
438 copy_using_evaluator_innervec_CompleteUnrolling<Kernel, 0, alignedSize>::run(kernel);
439 copy_using_evaluator_DefaultTraversal_CompleteUnrolling<Kernel, alignedSize, size>::run(kernel);
440 }
441};

515/**************************
516*** Slice vectorization ***
517***************************/
518
519template<typename Kernel>
520struct dense_assignment_loop<Kernel, SliceVectorizedTraversal, NoUnrolling>
521{
522 EIGEN_DEVICE_FUNC static EIGEN_STRONG_INLINE void run(Kernel &kernel)
523 {
524 typedef typename Kernel::Scalar Scalar;
525 typedef typename Kernel::PacketType PacketType;
526 enum {
527 packetSize = unpacket_traits<PacketType>::size,
528 requestedAlignment = int(Kernel::AssignmentTraits::InnerRequiredAlignment),
529 alignable = packet_traits<Scalar>::AlignedOnScalar || int(Kernel::AssignmentTraits::DstAlignment)>=sizeof(Scalar),
530 dstIsAligned = int(Kernel::AssignmentTraits::DstAlignment)>=int(requestedAlignment),
531 dstAlignment = alignable ? int(requestedAlignment)
532 : int(Kernel::AssignmentTraits::DstAlignment)
533 };
534 const Scalar *dst_ptr = kernel.dstDataPtr();
535 if((!bool(dstIsAligned)) && (UIntPtr(dst_ptr) % sizeof(Scalar))>0)
536 {
537 // the pointer is not aligned-on scalar, so alignment is not possible
538 return dense_assignment_loop<Kernel,DefaultTraversal,NoUnrolling>::run(kernel);
539 }
540 const Index packetAlignedMask = packetSize - 1;
541 const Index innerSize = kernel.innerSize();
542 const Index outerSize = kernel.outerSize();
543 const Index alignedStep = alignable ? (packetSize - kernel.outerStride() % packetSize) & packetAlignedMask : 0;
544 Index alignedStart = ((!alignable) || bool(dstIsAligned)) ? 0 : internal::first_aligned<requestedAlignment>(dst_ptr, innerSize);
545
546 for(Index outer = 0; outer < outerSize; ++outer)
547 {
548 const Index alignedEnd = alignedStart + ((innerSize-alignedStart) & ~packetAlignedMask);
549 // do the non-vectorizable part of the assignment
550 for(Index inner = 0; inner<alignedStart ; ++inner)
551 kernel.assignCoeffByOuterInner(outer, inner);
552
553 // do the vectorizable part of the assignment
554 for(Index inner = alignedStart; inner<alignedEnd; inner+=packetSize)
555 kernel.template assignPacketByOuterInner<dstAlignment, Unaligned, PacketType>(outer, inner);
556
557 // do the non-vectorizable part of the assignment
558 for(Index inner = alignedEnd; inner<innerSize ; ++inner)
559 kernel.assignCoeffByOuterInner(outer, inner);
560
561 alignedStart = numext::mini((alignedStart+alignedStep)%packetSize, innerSize);
562 }
563 }
564};

589/***************************************************************************
590* Part 4 : Generic dense assignment kernel
591***************************************************************************/
592
593// This class generalize the assignment of a coefficient (or packet) from one dense evaluator
594// to another dense writable evaluator.
595// It is parametrized by the two evaluators, and the actual assignment functor.
596// This abstraction level permits to keep the evaluation loops as simple and as generic as possible.
597// One can customize the assignment using this generic dense_assignment_kernel with different
598// functors, or by completely overloading it, by-passing a functor.
599template<typename DstEvaluatorTypeT, typename SrcEvaluatorTypeT, typename Functor, int Version = Specialized>
600class generic_dense_assignment_kernel
601{
602protected:
603 typedef typename DstEvaluatorTypeT::XprType DstXprType;
604 typedef typename SrcEvaluatorTypeT::XprType SrcXprType;
605public:
606
607 typedef DstEvaluatorTypeT DstEvaluatorType;
608 typedef SrcEvaluatorTypeT SrcEvaluatorType;
609 typedef typename DstEvaluatorType::Scalar Scalar;
610 typedef copy_using_evaluator_traits<DstEvaluatorTypeT, SrcEvaluatorTypeT, Functor> AssignmentTraits;
611 typedef typename AssignmentTraits::PacketType PacketType;
612
613
614 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE

687 : int(Traits::RowsAtCompileTime) == 1 ? inner
688 : int(DstEvaluatorType::Flags)&RowMajorBit ? inner
689 : outer;
690 }

751EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void call_dense_assignment_loop(DstXprType& dst, const SrcXprType& src, const Functor &func)
752{
753 typedef evaluator<DstXprType> DstEvaluatorType;
754 typedef evaluator<SrcXprType> SrcEvaluatorType;
755
756 SrcEvaluatorType srcEvaluator(src);
757
758 // NOTE To properly handle A = (A*A.transpose())/s with A rectangular,
759 // we need to resize the destination after the source evaluator has been created.
760 resize_if_allowed(dst, src, func);
761
762 DstEvaluatorType dstEvaluator(dst);
763
764 typedef generic_dense_assignment_kernel<DstEvaluatorType,SrcEvaluatorType,Functor> Kernel;
765 Kernel kernel(dstEvaluator, srcEvaluator, func, dst.const_cast_derived());
766
767 dense_assignment_loop<Kernel>::run(kernel);
768}

843template<typename Dst, typename Src, typename Func>
844EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
845void call_assignment_no_alias(Dst& dst, const Src& src, const Func& func)
846{
847 enum {
848 NeedToTranspose = ( (int(Dst::RowsAtCompileTime) == 1 && int(Src::ColsAtCompileTime) == 1)
849 || (int(Dst::ColsAtCompileTime) == 1 && int(Src::RowsAtCompileTime) == 1)
850 ) && int(Dst::SizeAtCompileTime) != 1
851 };
852
853 typedef typename internal::conditional<NeedToTranspose, Transpose<Dst>, Dst>::type ActualDstTypeCleaned;
854 typedef typename internal::conditional<NeedToTranspose, Transpose<Dst>, Dst&>::type ActualDstType;
855 ActualDstType actualDst(dst);
856
857 // TODO check whether this is the right place to perform these checks:
858 EIGEN_STATIC_ASSERT_LVALUE(Dst)
859 EIGEN_STATIC_ASSERT_SAME_MATRIX_SIZE(ActualDstTypeCleaned,Src)
860 EIGEN_CHECK_BINARY_COMPATIBILIY(Func,typename ActualDstTypeCleaned::Scalar,typename Src::Scalar);
861
862 Assignment<ActualDstTypeCleaned,Src,Func>::run(actualDst, src, func);
863}
