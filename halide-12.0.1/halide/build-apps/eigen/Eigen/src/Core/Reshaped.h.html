17/** \class Reshaped
18 * \ingroup Core_Module
19 *
20 * \brief Expression of a fixed-size or dynamic-size reshape
21 *
22 * \tparam XprType the type of the expression in which we are taking a reshape
23 * \tparam Rows the number of rows of the reshape we are taking at compile time (optional)
24 * \tparam Cols the number of columns of the reshape we are taking at compile time (optional)
25 * \tparam Order can be ColMajor or RowMajor, default is ColMajor.
26 *
27 * This class represents an expression of either a fixed-size or dynamic-size reshape.
28 * It is the return type of DenseBase::reshaped(NRowsType,NColsType) and
29 * most of the time this is the only way it is used.
30 *
31 * However, in C++98, if you want to directly maniputate reshaped expressions,
32 * for instance if you want to write a function returning such an expression, you
33 * will need to use this class. In C++11, it is advised to use the \em auto
34 * keyword for such use cases.
35 *
36 * Here is an example illustrating the dynamic case:
37 * \include class_Reshaped.cpp
38 * Output: \verbinclude class_Reshaped.out
39 *
40 * Here is an example illustrating the fixed-size case:
41 * \include class_FixedReshaped.cpp
42 * Output: \verbinclude class_FixedReshaped.out
43 *
44 * \sa DenseBase::reshaped(NRowsType,NColsType)
45 */
46
47template<typename XprType, int Rows, int Cols, int Order>
48struct traits<Reshaped<XprType, Rows, Cols, Order> > : traits<XprType>
49{
50 typedef typename traits<XprType>::Scalar Scalar;
51 typedef typename traits<XprType>::StorageKind StorageKind;
52 typedef typename traits<XprType>::XprKind XprKind;
53 enum{
54 MatrixRows = traits<XprType>::RowsAtCompileTime,
55 MatrixCols = traits<XprType>::ColsAtCompileTime,
56 RowsAtCompileTime = Rows,
57 ColsAtCompileTime = Cols,
58 MaxRowsAtCompileTime = Rows,
59 MaxColsAtCompileTime = Cols,
60 XpxStorageOrder = ((int(traits<XprType>::Flags) & RowMajorBit) == RowMajorBit) ? RowMajor : ColMajor,
61 ReshapedStorageOrder = (RowsAtCompileTime == 1 && ColsAtCompileTime != 1) ? RowMajor
62 : (ColsAtCompileTime == 1 && RowsAtCompileTime != 1) ? ColMajor
63 : XpxStorageOrder,
64 HasSameStorageOrderAsXprType = (ReshapedStorageOrder == XpxStorageOrder),
65 InnerSize = (ReshapedStorageOrder==int(RowMajor)) ? int(ColsAtCompileTime) : int(RowsAtCompileTime),
66 InnerStrideAtCompileTime = HasSameStorageOrderAsXprType
67 ? int(inner_stride_at_compile_time<XprType>::ret)
68 : Dynamic,
69 OuterStrideAtCompileTime = Dynamic,
70
71 HasDirectAccess = internal::has_direct_access<XprType>::ret
72 && (Order==int(XpxStorageOrder))
73 && ((evaluator<XprType>::Flags&LinearAccessBit)==LinearAccessBit),
74
75 MaskPacketAccessBit = (InnerSize == Dynamic || (InnerSize % packet_traits<Scalar>::size) == 0)
76 && (InnerStrideAtCompileTime == 1)
77 ? PacketAccessBit : 0,
78 //MaskAlignedBit = ((OuterStrideAtCompileTime!=Dynamic) && (((OuterStrideAtCompileTime * int(sizeof(Scalar))) % 16) == 0)) ? AlignedBit : 0,
79 FlagsLinearAccessBit = (RowsAtCompileTime == 1 || ColsAtCompileTime == 1) ? LinearAccessBit : 0,
80 FlagsLvalueBit = is_lvalue<XprType>::value ? LvalueBit : 0,
81 FlagsRowMajorBit = (ReshapedStorageOrder==int(RowMajor)) ? RowMajorBit : 0,
82 FlagsDirectAccessBit = HasDirectAccess ? DirectAccessBit : 0,
83 Flags0 = traits<XprType>::Flags & ( (HereditaryBits & ~RowMajorBit) | MaskPacketAccessBit),
84
85 Flags = (Flags0 | FlagsLinearAccessBit | FlagsLvalueBit | FlagsRowMajorBit | FlagsDirectAccessBit)
86 };
87};

93template<typename XprType, int Rows, int Cols, int Order, typename StorageKind> class ReshapedImpl;

95template<typename XprType, int Rows, int Cols, int Order> class Reshaped
96 : public ReshapedImpl<XprType, Rows, Cols, Order, typename internal::traits<XprType>::StorageKind>
97{
98 typedef ReshapedImpl<XprType, Rows, Cols, Order, typename internal::traits<XprType>::StorageKind> Impl;
99 public:
100 //typedef typename Impl::Base Base;
101 typedef Impl Base;
102 EIGEN_GENERIC_PUBLIC_INTERFACE(Reshaped)
103 EIGEN_INHERIT_ASSIGNMENT_OPERATORS(Reshaped)
104
105 /** Fixed-size constructor
106 */
107 EIGEN_DEVICE_FUNC
108 inline Reshaped(XprType& xpr)
109 : Impl(xpr)
110 {
111 EIGEN_STATIC_ASSERT(RowsAtCompileTime!=Dynamic && ColsAtCompileTime!=Dynamic,THIS_METHOD_IS_ONLY_FOR_FIXED_SIZE)
112 eigen_assert(Rows * Cols == xpr.rows() * xpr.cols());
113 }
114
115 /** Dynamic-size constructor
116 */
117 EIGEN_DEVICE_FUNC
118 inline Reshaped(XprType& xpr,
119 Index reshapeRows, Index reshapeCols)
120 : Impl(xpr, reshapeRows, reshapeCols)
121 {
122 eigen_assert((RowsAtCompileTime==Dynamic || RowsAtCompileTime==reshapeRows)
123 && (ColsAtCompileTime==Dynamic || ColsAtCompileTime==reshapeCols));
124 eigen_assert(reshapeRows * reshapeCols == xpr.rows() * xpr.cols());
125 }
126};

217 /** Fixed-size constructor
218 */
219 EIGEN_DEVICE_FUNC
220 inline ReshapedImpl_dense(XprType& xpr)
221 : Base(xpr.data()), m_xpr(xpr)
222 {}
223
224 /** Dynamic-size constructor
225 */
226 EIGEN_DEVICE_FUNC

228 : Base(xpr.data(), nRows, nCols),
229 m_xpr(xpr)
230 {}
231
232 EIGEN_DEVICE_FUNC
233 const typename internal::remove_all<XprTypeNested>::type& nestedExpression() const
234 {
235 return m_xpr;
236 }
237
238 EIGEN_DEVICE_FUNC
239 XprType& nestedExpression() { return m_xpr; }
240
241 /** \sa MapBase::innerStride() */
242 EIGEN_DEVICE_FUNC
243 inline Index innerStride() const
244 {
245 return m_xpr.innerStride();
246 }
247
248 /** \sa MapBase::outerStride() */
249 EIGEN_DEVICE_FUNC
250 inline Index outerStride() const
251 {
252 return ((Flags&RowMajorBit)==RowMajorBit) ? this->cols() : this->rows();
253 }
254
255 protected:
256
257 XprTypeNested m_xpr;
258};

263template<typename ArgType, int Rows, int Cols, int Order>
264struct evaluator<Reshaped<ArgType, Rows, Cols, Order> >
265 : reshaped_evaluator<ArgType, Rows, Cols, Order, traits<Reshaped<ArgType,Rows,Cols,Order> >::HasDirectAccess>
266{
267 typedef Reshaped<ArgType, Rows, Cols, Order> XprType;
268 typedef typename XprType::Scalar Scalar;
269 // TODO: should check for smaller packet types
270 typedef typename packet_traits<Scalar>::type PacketScalar;
271
272 enum {
273 CoeffReadCost = evaluator<ArgType>::CoeffReadCost,
274 HasDirectAccess = traits<XprType>::HasDirectAccess,
275
276// RowsAtCompileTime = traits<XprType>::RowsAtCompileTime,
277// ColsAtCompileTime = traits<XprType>::ColsAtCompileTime,
278// MaxRowsAtCompileTime = traits<XprType>::MaxRowsAtCompileTime,
279// MaxColsAtCompileTime = traits<XprType>::MaxColsAtCompileTime,
280//
281// InnerStrideAtCompileTime = traits<XprType>::HasSameStorageOrderAsXprType
282// ? int(inner_stride_at_compile_time<ArgType>::ret)
283// : Dynamic,
284// OuterStrideAtCompileTime = Dynamic,
285
286 FlagsLinearAccessBit = (traits<XprType>::RowsAtCompileTime == 1 || traits<XprType>::ColsAtCompileTime == 1 || HasDirectAccess) ? LinearAccessBit : 0,
287 FlagsRowMajorBit = (traits<XprType>::ReshapedStorageOrder==int(RowMajor)) ? RowMajorBit : 0,
288 FlagsDirectAccessBit = HasDirectAccess ? DirectAccessBit : 0,
289 Flags0 = evaluator<ArgType>::Flags & (HereditaryBits & ~RowMajorBit),
290 Flags = Flags0 | FlagsLinearAccessBit | FlagsRowMajorBit | FlagsDirectAccessBit,
291
292 PacketAlignment = unpacket_traits<PacketScalar>::alignment,
293 Alignment = evaluator<ArgType>::Alignment
294 };
295 typedef reshaped_evaluator<ArgType, Rows, Cols, Order, HasDirectAccess> reshaped_evaluator_type;
296 EIGEN_DEVICE_FUNC explicit evaluator(const XprType& xpr) : reshaped_evaluator_type(xpr)
297 {
298 EIGEN_INTERNAL_CHECK_COST_VALUE(CoeffReadCost);
299 }
300};

302template<typename ArgType, int Rows, int Cols, int Order>
303struct reshaped_evaluator<ArgType, Rows, Cols, Order, /* HasDirectAccess */ false>
304 : evaluator_base<Reshaped<ArgType, Rows, Cols, Order> >
305{
306 typedef Reshaped<ArgType, Rows, Cols, Order> XprType;
307
308 enum {
309 CoeffReadCost = evaluator<ArgType>::CoeffReadCost /* TODO + cost of index computations */,
310
311 Flags = (evaluator<ArgType>::Flags & (HereditaryBits /*| LinearAccessBit | DirectAccessBit*/)),
312
313 Alignment = 0
314 };
315
316 EIGEN_DEVICE_FUNC explicit reshaped_evaluator(const XprType& xpr) : m_argImpl(xpr.nestedExpression()), m_xpr(xpr)
317 {
318 EIGEN_INTERNAL_CHECK_COST_VALUE(CoeffReadCost);
319 }
320
321 typedef typename XprType::Scalar Scalar;
322 typedef typename XprType::CoeffReturnType CoeffReturnType;
323
324 typedef std::pair<Index, Index> RowCol;
325
326 inline RowCol index_remap(Index rowId, Index colId) const
327 {
328 if(Order==ColMajor)
329 {
330 const Index nth_elem_idx = colId * m_xpr.rows() + rowId;
331 return RowCol(nth_elem_idx % m_xpr.nestedExpression().rows(),
332 nth_elem_idx / m_xpr.nestedExpression().rows());
333 }
334 else
335 {
336 const Index nth_elem_idx = colId + rowId * m_xpr.cols();
337 return RowCol(nth_elem_idx / m_xpr.nestedExpression().cols(),
338 nth_elem_idx % m_xpr.nestedExpression().cols());
339 }
340 }
341
342 EIGEN_DEVICE_FUNC
343 inline Scalar& coeffRef(Index rowId, Index colId)
344 {
345 EIGEN_STATIC_ASSERT_LVALUE(XprType)
346 const RowCol row_col = index_remap(rowId, colId);
347 return m_argImpl.coeffRef(row_col.first, row_col.second);
348 }
349
350 EIGEN_DEVICE_FUNC
351 inline const Scalar& coeffRef(Index rowId, Index colId) const
352 {
353 const RowCol row_col = index_remap(rowId, colId);
354 return m_argImpl.coeffRef(row_col.first, row_col.second);
355 }
356
357 EIGEN_DEVICE_FUNC
358 EIGEN_STRONG_INLINE const CoeffReturnType coeff(Index rowId, Index colId) const
359 {
360 const RowCol row_col = index_remap(rowId, colId);
361 return m_argImpl.coeff(row_col.first, row_col.second);
362 }
363
364 EIGEN_DEVICE_FUNC
365 inline Scalar& coeffRef(Index index)
366 {
367 EIGEN_STATIC_ASSERT_LVALUE(XprType)
368 const RowCol row_col = index_remap(Rows == 1 ? 0 : index,
369 Rows == 1 ? index : 0);
370 return m_argImpl.coeffRef(row_col.first, row_col.second);
371
372 }
373
374 EIGEN_DEVICE_FUNC
375 inline const Scalar& coeffRef(Index index) const
376 {
377 const RowCol row_col = index_remap(Rows == 1 ? 0 : index,
378 Rows == 1 ? index : 0);
379 return m_argImpl.coeffRef(row_col.first, row_col.second);
380 }
381
382 EIGEN_DEVICE_FUNC
383 inline const CoeffReturnType coeff(Index index) const
384 {
385 const RowCol row_col = index_remap(Rows == 1 ? 0 : index,
386 Rows == 1 ? index : 0);
387 return m_argImpl.coeff(row_col.first, row_col.second);
388 }
389#if 0

391 template<int LoadMode>
392 inline PacketScalar packet(Index rowId, Index colId) const
393 {
394 const RowCol row_col = index_remap(rowId, colId);
395 return m_argImpl.template packet<Unaligned>(row_col.first, row_col.second);
396
397 }
398
399 template<int LoadMode>
400 EIGEN_DEVICE_FUNC
401 inline void writePacket(Index rowId, Index colId, const PacketScalar& val)
402 {
403 const RowCol row_col = index_remap(rowId, colId);
404 m_argImpl.const_cast_derived().template writePacket<Unaligned>
405 (row_col.first, row_col.second, val);
406 }
407
408 template<int LoadMode>
409 EIGEN_DEVICE_FUNC
410 inline PacketScalar packet(Index index) const
411 {
412 const RowCol row_col = index_remap(RowsAtCompileTime == 1 ? 0 : index,
413 RowsAtCompileTime == 1 ? index : 0);
414 return m_argImpl.template packet<Unaligned>(row_col.first, row_col.second);
415 }
416
417 template<int LoadMode>
418 EIGEN_DEVICE_FUNC
419 inline void writePacket(Index index, const PacketScalar& val)
420 {
421 const RowCol row_col = index_remap(RowsAtCompileTime == 1 ? 0 : index,
422 RowsAtCompileTime == 1 ? index : 0);
423 return m_argImpl.template packet<Unaligned>(row_col.first, row_col.second, val);
424 }
425#endif
426protected:
427
428 evaluator<ArgType> m_argImpl;
429 const XprType& m_xpr;

431};
