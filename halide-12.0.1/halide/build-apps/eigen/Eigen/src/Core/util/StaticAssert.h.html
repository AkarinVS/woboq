14/* Some notes on Eigen's static assertion mechanism:
15 *
16 * - in EIGEN_STATIC_ASSERT(CONDITION,MSG) the parameter CONDITION must be a compile time boolean
17 * expression, and MSG an enum listed in struct internal::static_assertion<true>
18 *
19 * - define EIGEN_NO_STATIC_ASSERT to disable them (and save compilation time)
20 * in that case, the static assertion is converted to the following runtime assert:
21 * eigen_assert(CONDITION && "MSG")
22 *
23 * - currently EIGEN_STATIC_ASSERT can only be used in function scope
24 *
25 */
26
27#ifndef EIGEN_STATIC_ASSERT
28#ifndef EIGEN_NO_STATIC_ASSERT
29
30 #if EIGEN_MAX_CPP_VER>=11 && (__has_feature(cxx_static_assert) || (defined(__cplusplus) && __cplusplus >= 201103L) || (EIGEN_COMP_MSVC >= 1600))
31
32 // if native static_assert is enabled, let's use it
33 #define EIGEN_STATIC_ASSERT(X,MSG) static_assert(X,#MSG);
34
35 #else // not CXX0X
36

116 // Specialized implementation for MSVC to avoid "conditional
117 // expression is constant" warnings. This implementation doesn't
118 // appear to work under GCC, hence the multiple implementations.
119 #if EIGEN_COMP_MSVC
120
121 #define EIGEN_STATIC_ASSERT(CONDITION,MSG) \
122 {Eigen::internal::static_assertion<bool(CONDITION)>::MSG;}
123
124 #else
125 // In some cases clang interprets bool(CONDITION) as function declaration
126 #define EIGEN_STATIC_ASSERT(CONDITION,MSG) \
127 if (Eigen::internal::static_assertion<static_cast<bool>(CONDITION)>::MSG) {}
128
129 #endif
130
131 #endif // not CXX0X
132
133#else // EIGEN_NO_STATIC_ASSERT
134
135 #define EIGEN_STATIC_ASSERT(CONDITION,MSG) eigen_assert((CONDITION) && #MSG);
136
137#endif // EIGEN_NO_STATIC_ASSERT
138#endif // EIGEN_STATIC_ASSERT
139
140// static assertion failing if the type \a TYPE is not a vector type
141#define EIGEN_STATIC_ASSERT_VECTOR_ONLY(TYPE) \
142 EIGEN_STATIC_ASSERT(TYPE::IsVectorAtCompileTime, \
143 YOU_TRIED_CALLING_A_VECTOR_METHOD_ON_A_MATRIX)
144
145// static assertion failing if the type \a TYPE is not fixed-size
146#define EIGEN_STATIC_ASSERT_FIXED_SIZE(TYPE) \
147 EIGEN_STATIC_ASSERT(TYPE::SizeAtCompileTime!=Eigen::Dynamic, \
148 YOU_CALLED_A_FIXED_SIZE_METHOD_ON_A_DYNAMIC_SIZE_MATRIX_OR_VECTOR)
149
150// static assertion failing if the type \a TYPE is not dynamic-size
151#define EIGEN_STATIC_ASSERT_DYNAMIC_SIZE(TYPE) \
152 EIGEN_STATIC_ASSERT(TYPE::SizeAtCompileTime==Eigen::Dynamic, \
153 YOU_CALLED_A_DYNAMIC_SIZE_METHOD_ON_A_FIXED_SIZE_MATRIX_OR_VECTOR)
154
155// static assertion failing if the type \a TYPE is not a vector type of the given size
156#define EIGEN_STATIC_ASSERT_VECTOR_SPECIFIC_SIZE(TYPE, SIZE) \
157 EIGEN_STATIC_ASSERT(TYPE::IsVectorAtCompileTime && TYPE::SizeAtCompileTime==SIZE, \
158 THIS_METHOD_IS_ONLY_FOR_VECTORS_OF_A_SPECIFIC_SIZE)
159
160// static assertion failing if the type \a TYPE is not a vector type of the given size
161#define EIGEN_STATIC_ASSERT_MATRIX_SPECIFIC_SIZE(TYPE, ROWS, COLS) \
162 EIGEN_STATIC_ASSERT(TYPE::RowsAtCompileTime==ROWS && TYPE::ColsAtCompileTime==COLS, \
163 THIS_METHOD_IS_ONLY_FOR_MATRICES_OF_A_SPECIFIC_SIZE)
164
165// static assertion failing if the two vector expression types are not compatible (same fixed-size or dynamic size)
166#define EIGEN_STATIC_ASSERT_SAME_VECTOR_SIZE(TYPE0,TYPE1) \
167 EIGEN_STATIC_ASSERT( \
168 (int(TYPE0::SizeAtCompileTime)==Eigen::Dynamic \
169 || int(TYPE1::SizeAtCompileTime)==Eigen::Dynamic \
170 || int(TYPE0::SizeAtCompileTime)==int(TYPE1::SizeAtCompileTime)),\
171 YOU_MIXED_VECTORS_OF_DIFFERENT_SIZES)
172
173#define EIGEN_PREDICATE_SAME_MATRIX_SIZE(TYPE0,TYPE1) \
174 ( \
175 (int(Eigen::internal::size_of_xpr_at_compile_time<TYPE0>::ret)==0 && int(Eigen::internal::size_of_xpr_at_compile_time<TYPE1>::ret)==0) \
176 || (\
177 (int(TYPE0::RowsAtCompileTime)==Eigen::Dynamic \
178 || int(TYPE1::RowsAtCompileTime)==Eigen::Dynamic \
179 || int(TYPE0::RowsAtCompileTime)==int(TYPE1::RowsAtCompileTime)) \
180 && (int(TYPE0::ColsAtCompileTime)==Eigen::Dynamic \
181 || int(TYPE1::ColsAtCompileTime)==Eigen::Dynamic \
182 || int(TYPE0::ColsAtCompileTime)==int(TYPE1::ColsAtCompileTime))\
183 ) \
184 )
185
186#define EIGEN_STATIC_ASSERT_NON_INTEGER(TYPE) \
187 EIGEN_STATIC_ASSERT(!Eigen::NumTraits<TYPE>::IsInteger, THIS_FUNCTION_IS_NOT_FOR_INTEGER_NUMERIC_TYPES)
188
189
190// static assertion failing if it is guaranteed at compile-time that the two matrix expression types have different sizes
191#define EIGEN_STATIC_ASSERT_SAME_MATRIX_SIZE(TYPE0,TYPE1) \
192 EIGEN_STATIC_ASSERT( \
193 EIGEN_PREDICATE_SAME_MATRIX_SIZE(TYPE0,TYPE1),\
194 YOU_MIXED_MATRICES_OF_DIFFERENT_SIZES)
195
196#define EIGEN_STATIC_ASSERT_SIZE_1x1(TYPE) \
197 EIGEN_STATIC_ASSERT((TYPE::RowsAtCompileTime == 1 || TYPE::RowsAtCompileTime == Eigen::Dynamic) && \
198 (TYPE::ColsAtCompileTime == 1 || TYPE::ColsAtCompileTime == Eigen::Dynamic), \
199 THIS_METHOD_IS_ONLY_FOR_1x1_EXPRESSIONS)
200
201#define EIGEN_STATIC_ASSERT_LVALUE(Derived) \
202 EIGEN_STATIC_ASSERT(Eigen::internal::is_lvalue<Derived>::value, \
203 THIS_EXPRESSION_IS_NOT_A_LVALUE__IT_IS_READ_ONLY)
204
205#define EIGEN_STATIC_ASSERT_ARRAYXPR(Derived) \
206 EIGEN_STATIC_ASSERT((Eigen::internal::is_same<typename Eigen::internal::traits<Derived>::XprKind, ArrayXpr>::value), \
207 THIS_METHOD_IS_ONLY_FOR_ARRAYS_NOT_MATRICES)
208
209#define EIGEN_STATIC_ASSERT_SAME_XPR_KIND(Derived1, Derived2) \
210 EIGEN_STATIC_ASSERT((Eigen::internal::is_same<typename Eigen::internal::traits<Derived1>::XprKind, \
211 typename Eigen::internal::traits<Derived2>::XprKind \
212 >::value), \
213 YOU_CANNOT_MIX_ARRAYS_AND_MATRICES)
214
215// Check that a cost value is positive, and that is stay within a reasonable range
216// TODO this check could be enabled for internal debugging only
217#define EIGEN_INTERNAL_CHECK_COST_VALUE(C) \
218 EIGEN_STATIC_ASSERT((C)>=0 && (C)<=HugeCost*HugeCost, EIGEN_INTERNAL_ERROR_PLEASE_FILE_A_BUG_REPORT__INVALID_COST_VALUE);

220#endif // EIGEN_STATIC_ASSERT_H
