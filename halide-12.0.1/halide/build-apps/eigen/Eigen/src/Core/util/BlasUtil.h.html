59// Generic implementation for custom complex types.
60template<typename LhsScalar, typename RhsScalar, bool ConjLhs, bool ConjRhs>
61struct conj_helper
62{
63 typedef typename ScalarBinaryOpTraits<LhsScalar,RhsScalar>::ReturnType Scalar;
64
65 EIGEN_STRONG_INLINE Scalar pmadd(const LhsScalar& x, const RhsScalar& y, const Scalar& c) const
66 { return padd(c, pmul(x,y)); }
67
68 EIGEN_STRONG_INLINE Scalar pmul(const LhsScalar& x, const RhsScalar& y) const
69 { return conj_if<ConjLhs>()(x) * conj_if<ConjRhs>()(y); }
70};

198// TMP to help PacketBlock store implementation.
199// There's currently no known use case for PacketBlock load.
200// The default implementation assumes ColMajor order.
201// It always store each packet sequentially one `stride` apart.
202template<typename Index, typename Scalar, typename Packet, int n, int idx, int StorageOrder>
203struct PacketBlockManagement
204{
205 PacketBlockManagement<Index, Scalar, Packet, n, idx - 1, StorageOrder> pbm;
206 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE void store(Scalar *to, const Index stride, Index i, Index j, const PacketBlock<Packet, n> &block) const {
207 pbm.store(to, stride, i, j, block);
208 pstoreu<Scalar>(to + i + (j + idx)*stride, block.packet[idx]);
209 }
210};
211
212// PacketBlockManagement specialization to take care of RowMajor order without ifs.
213template<typename Index, typename Scalar, typename Packet, int n, int idx>
214struct PacketBlockManagement<Index, Scalar, Packet, n, idx, RowMajor>
215{
216 PacketBlockManagement<Index, Scalar, Packet, n, idx - 1, RowMajor> pbm;
217 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE void store(Scalar *to, const Index stride, Index i, Index j, const PacketBlock<Packet, n> &block) const {
218 pbm.store(to, stride, i, j, block);
219 pstoreu<Scalar>(to + j + (i + idx)*stride, block.packet[idx]);
220 }
221};

257 EIGEN_ONLY_USED_FOR_DEBUG(incr);

271 return VectorMapper(&operator()(i, j));
272 }
273
274
275 EIGEN_DEVICE_FUNC
276 EIGEN_ALWAYS_INLINE Scalar& operator()(Index i, Index j) const {
277 return m_data[StorageOrder==RowMajor ? j + i*m_stride : i + j*m_stride];
278 }
279
280 template<typename PacketType>
281 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE PacketType loadPacket(Index i, Index j) const {
282 return ploadt<PacketType, AlignmentType>(&operator()(i, j));
283 }
284
285 template <typename PacketT, int AlignmentT>
286 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE PacketT load(Index i, Index j) const {
287 return ploadt<PacketT, AlignmentT>(&operator()(i, j));

289
290 template<typename SubPacket>
291 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE void scatterPacket(Index i, Index j, const SubPacket &p) const {
292 pscatter<Scalar, SubPacket>(&operator()(i, j), p, m_stride);
293 }
294
295 template<typename SubPacket>
296 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE SubPacket gatherPacket(Index i, Index j) const {
297 return pgather<Scalar, SubPacket>(&operator()(i, j), m_stride);

299
300 EIGEN_DEVICE_FUNC const Index stride() const { return m_stride; }
301 EIGEN_DEVICE_FUNC const Scalar* data() const { return m_data; }
302

311 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE void storePacketBlock(Index i, Index j, const PacketBlock<SubPacket, n> &block) const {

313 pbm.store(m_data, m_stride, i, j, block);
314 }
315protected:
316 Scalar* EIGEN_RESTRICT m_data;
317 const Index m_stride;

412/* Helper class to analyze the factors of a Product expression.
413 * In particular it allows to pop out operator-, scalar multiples,
414 * and conjugate */
415template<typename XprType> struct blas_traits
416{
417 typedef typename traits<XprType>::Scalar Scalar;
418 typedef const XprType& ExtractType;
419 typedef XprType _ExtractType;
420 enum {
421 IsComplex = NumTraits<Scalar>::IsComplex,
422 IsTransposed = false,
423 NeedToConjugate = false,
424 HasUsableDirectAccess = ( (int(XprType::Flags)&DirectAccessBit)
425 && ( bool(XprType::IsVectorAtCompileTime)
426 || int(inner_stride_at_compile_time<XprType>::ret) == 1)
427 ) ? 1 : 0,
428 HasScalarFactor = false
429 };
430 typedef typename conditional<bool(HasUsableDirectAccess),
431 ExtractType,
432 typename _ExtractType::PlainObject
433 >::type DirectLinearAccessType;
434 static inline EIGEN_DEVICE_FUNC ExtractType extract(const XprType& x) { return x; }
435 static inline EIGEN_DEVICE_FUNC const Scalar extractScalarFactor(const XprType&) { return Scalar(1); }
436};

439template<typename Scalar, typename NestedXpr>
440struct blas_traits<CwiseUnaryOp<scalar_conjugate_op<Scalar>, NestedXpr> >
441 : blas_traits<NestedXpr>
442{
443 typedef blas_traits<NestedXpr> Base;
444 typedef CwiseUnaryOp<scalar_conjugate_op<Scalar>, NestedXpr> XprType;
445 typedef typename Base::ExtractType ExtractType;
446
447 enum {
448 IsComplex = NumTraits<Scalar>::IsComplex,
449 NeedToConjugate = Base::NeedToConjugate ? 0 : IsComplex
450 };
451 static inline ExtractType extract(const XprType& x) { return Base::extract(x.nestedExpression()); }
452 static inline Scalar extractScalarFactor(const XprType& x) { return conj(Base::extractScalarFactor(x.nestedExpression())); }
453};

456template<typename Scalar, typename NestedXpr, typename Plain>
457struct blas_traits<CwiseBinaryOp<scalar_product_op<Scalar>, const CwiseNullaryOp<scalar_constant_op<Scalar>,Plain>, NestedXpr> >
458 : blas_traits<NestedXpr>
459{
460 enum {
461 HasScalarFactor = true
462 };
463 typedef blas_traits<NestedXpr> Base;
464 typedef CwiseBinaryOp<scalar_product_op<Scalar>, const CwiseNullaryOp<scalar_constant_op<Scalar>,Plain>, NestedXpr> XprType;
465 typedef typename Base::ExtractType ExtractType;
466 static inline EIGEN_DEVICE_FUNC ExtractType extract(const XprType& x) { return Base::extract(x.rhs()); }
467 static inline EIGEN_DEVICE_FUNC Scalar extractScalarFactor(const XprType& x)
468 { return x.lhs().functor().m_other * Base::extractScalarFactor(x.rhs()); }
469};
