17// default implementation of digits10(), based on numeric_limits if specialized,
18// 0 for integer types, and log10(epsilon()) otherwise.
19template< typename T,
20 bool use_numeric_limits = std::numeric_limits<T>::is_specialized,
21 bool is_integer = NumTraits<T>::IsInteger>
22struct default_digits10_impl
23{
24 EIGEN_DEVICE_FUNC
25 static int run() { return std::numeric_limits<T>::digits10; }
26};
27
28template<typename T>
29struct default_digits10_impl<T,false,false> // Floating point
30{
31 EIGEN_DEVICE_FUNC
32 static int run() {
33 using std::log10;
34 using std::ceil;
35 typedef typename NumTraits<T>::Real Real;
36 return int(ceil(-log10(NumTraits<Real>::epsilon())));
37 }
38};
39
40template<typename T>
41struct default_digits10_impl<T,false,true> // Integer
42{
43 EIGEN_DEVICE_FUNC
44 static int run() { return 0; }
45};

48// default implementation of digits(), based on numeric_limits if specialized,
49// 0 for integer types, and log2(epsilon()) otherwise.
50template< typename T,
51 bool use_numeric_limits = std::numeric_limits<T>::is_specialized,
52 bool is_integer = NumTraits<T>::IsInteger>
53struct default_digits_impl
54{
55 EIGEN_DEVICE_FUNC
56 static int run() { return std::numeric_limits<T>::digits; }
57};
58
59template<typename T>
60struct default_digits_impl<T,false,false> // Floating point
61{
62 EIGEN_DEVICE_FUNC
63 static int run() {
64 using std::log;
65 using std::ceil;
66 typedef typename NumTraits<T>::Real Real;
67 return int(ceil(-log(NumTraits<Real>::epsilon())/log(static_cast<Real>(2))));
68 }
69};
70
71template<typename T>
72struct default_digits_impl<T,false,true> // Integer
73{
74 EIGEN_DEVICE_FUNC
75 static int run() { return 0; }
76};

80/** \class NumTraits
81 * \ingroup Core_Module
82 *
83 * \brief Holds information about the various numeric (i.e. scalar) types allowed by Eigen.
84 *
85 * \tparam T the numeric type at hand
86 *
87 * This class stores enums, typedefs and static methods giving information about a numeric type.
88 *
89 * The provided data consists of:
90 * \li A typedef \c Real, giving the "real part" type of \a T. If \a T is already real,
91 * then \c Real is just a typedef to \a T. If \a T is \c std::complex<U> then \c Real
92 * is a typedef to \a U.
93 * \li A typedef \c NonInteger, giving the type that should be used for operations producing non-integral values,
94 * such as quotients, square roots, etc. If \a T is a floating-point type, then this typedef just gives
95 * \a T again. Note however that many Eigen functions such as internal::sqrt simply refuse to
96 * take integers. Outside of a few cases, Eigen doesn't do automatic type promotion. Thus, this typedef is
97 * only intended as a helper for code that needs to explicitly promote types.
98 * \li A typedef \c Literal giving the type to use for numeric literals such as "2" or "0.5". For instance, for \c std::complex<U>, Literal is defined as \c U.
99 * Of course, this type must be fully compatible with \a T. In doubt, just use \a T here.
100 * \li A typedef \a Nested giving the type to use to nest a value inside of the expression tree. If you don't know what
101 * this means, just use \a T here.
102 * \li An enum value \a IsComplex. It is equal to 1 if \a T is a \c std::complex
103 * type, and to 0 otherwise.
104 * \li An enum value \a IsInteger. It is equal to \c 1 if \a T is an integer type such as \c int,
105 * and to \c 0 otherwise.
106 * \li Enum values ReadCost, AddCost and MulCost representing a rough estimate of the number of CPU cycles needed
107 * to by move / add / mul instructions respectively, assuming the data is already stored in CPU registers.
108 * Stay vague here. No need to do architecture-specific stuff. If you don't know what this means, just use \c Eigen::HugeCost.
109 * \li An enum value \a IsSigned. It is equal to \c 1 if \a T is a signed type and to 0 if \a T is unsigned.
110 * \li An enum value \a RequireInitialization. It is equal to \c 1 if the constructor of the numeric type \a T must
111 * be called, and to 0 if it is safe not to call it. Default is 0 if \a T is an arithmetic type, and 1 otherwise.
112 * \li An epsilon() function which, unlike <a href="http://en.cppreference.com/w/cpp/types/numeric_limits/epsilon">std::numeric_limits::epsilon()</a>,
113 * it returns a \a Real instead of a \a T.
114 * \li A dummy_precision() function returning a weak epsilon value. It is mainly used as a default
115 * value by the fuzzy comparison operators.
116 * \li highest() and lowest() functions returning the highest and lowest possible values respectively.
117 * \li digits10() function returning the number of decimal digits that can be represented without change. This is
118 * the analogue of <a href="http://en.cppreference.com/w/cpp/types/numeric_limits/digits10">std::numeric_limits<T>::digits10</a>
119 * which is used as the default implementation if specialized.
120 */
121
122template<typename T> struct GenericNumTraits
123{
124 enum {
125 IsInteger = std::numeric_limits<T>::is_integer,
126 IsSigned = std::numeric_limits<T>::is_signed,
127 IsComplex = 0,
128 RequireInitialization = internal::is_arithmetic<T>::value ? 0 : 1,
129 ReadCost = 1,
130 AddCost = 1,
131 MulCost = 1
132 };
133
134 typedef T Real;
135 typedef typename internal::conditional<
136 IsInteger,
137 typename internal::conditional<sizeof(T)<=2, float, double>::type,
138 T
139 >::type NonInteger;
140 typedef T Nested;
141 typedef T Literal;
142
143 EIGEN_DEVICE_FUNC
144 static inline Real epsilon()
145 {
146 return numext::numeric_limits<T>::epsilon();
147 }
148
149 EIGEN_DEVICE_FUNC
150 static inline int digits10()
151 {
152 return internal::default_digits10_impl<T>::run();
153 }
154
155 EIGEN_DEVICE_FUNC
156 static inline int digits()
157 {
158 return internal::default_digits_impl<T>::run();
159 }
160
161 EIGEN_DEVICE_FUNC
162 static inline Real dummy_precision()
163 {
164 // make sure to override this for floating-point types
165 return Real(0);
166 }
167
168
169 EIGEN_DEVICE_FUNC
170 static inline T highest() {
171 return (numext::numeric_limits<T>::max)();
172 }
173
174 EIGEN_DEVICE_FUNC
175 static inline T lowest() {
176 return IsInteger ? (numext::numeric_limits<T>::min)()
177 : static_cast<T>(-(numext::numeric_limits<T>::max)());
178 }
179
180 EIGEN_DEVICE_FUNC
181 static inline T infinity() {
182 return numext::numeric_limits<T>::infinity();
183 }
184
185 EIGEN_DEVICE_FUNC
186 static inline T quiet_NaN() {
187 return numext::numeric_limits<T>::quiet_NaN();
188 }
189};
190
191template<typename T> struct NumTraits : GenericNumTraits<T>
192{};

213template<typename _Real> struct NumTraits<std::complex<_Real> >
214 : GenericNumTraits<std::complex<_Real> >
215{
216 typedef _Real Real;
217 typedef typename NumTraits<_Real>::Literal Literal;
218 enum {
219 IsComplex = 1,
220 RequireInitialization = NumTraits<_Real>::RequireInitialization,
221 ReadCost = 2 * NumTraits<_Real>::ReadCost,
222 AddCost = 2 * NumTraits<Real>::AddCost,
223 MulCost = 4 * NumTraits<Real>::MulCost + 2 * NumTraits<Real>::AddCost
224 };
225
226 EIGEN_DEVICE_FUNC
227 static inline Real epsilon() { return NumTraits<Real>::epsilon(); }
228 EIGEN_DEVICE_FUNC
229 static inline Real dummy_precision() { return NumTraits<Real>::dummy_precision(); }
230 EIGEN_DEVICE_FUNC
231 static inline int digits10() { return NumTraits<Real>::digits10(); }
232};

234template<typename Scalar, int Rows, int Cols, int Options, int MaxRows, int MaxCols>
235struct NumTraits<Array<Scalar, Rows, Cols, Options, MaxRows, MaxCols> >
236{
237 typedef Array<Scalar, Rows, Cols, Options, MaxRows, MaxCols> ArrayType;
238 typedef typename NumTraits<Scalar>::Real RealScalar;
239 typedef Array<RealScalar, Rows, Cols, Options, MaxRows, MaxCols> Real;
240 typedef typename NumTraits<Scalar>::NonInteger NonIntegerScalar;
241 typedef Array<NonIntegerScalar, Rows, Cols, Options, MaxRows, MaxCols> NonInteger;
242 typedef ArrayType & Nested;
243 typedef typename NumTraits<Scalar>::Literal Literal;
244
245 enum {
246 IsComplex = NumTraits<Scalar>::IsComplex,
247 IsInteger = NumTraits<Scalar>::IsInteger,
248 IsSigned = NumTraits<Scalar>::IsSigned,
249 RequireInitialization = 1,
250 ReadCost = ArrayType::SizeAtCompileTime==Dynamic ? HugeCost : ArrayType::SizeAtCompileTime * NumTraits<Scalar>::ReadCost,
251 AddCost = ArrayType::SizeAtCompileTime==Dynamic ? HugeCost : ArrayType::SizeAtCompileTime * NumTraits<Scalar>::AddCost,
252 MulCost = ArrayType::SizeAtCompileTime==Dynamic ? HugeCost : ArrayType::SizeAtCompileTime * NumTraits<Scalar>::MulCost
253 };
254
255 EIGEN_DEVICE_FUNC
256 static inline RealScalar epsilon() { return NumTraits<RealScalar>::epsilon(); }
257 EIGEN_DEVICE_FUNC
258 static inline RealScalar dummy_precision() { return NumTraits<RealScalar>::dummy_precision(); }
259
260 static inline int digits10() { return NumTraits<Scalar>::digits10(); }
261};

263template<> struct NumTraits<std::string>
264 : GenericNumTraits<std::string>
265{
266 enum {
267 RequireInitialization = 1,
268 ReadCost = HugeCost,
269 AddCost = HugeCost,
270 MulCost = HugeCost
271 };
272
273 static inline int digits10() { return 0; }

275private:
276 static inline std::string epsilon();
277 static inline std::string dummy_precision();
278 static inline std::string lowest();
279 static inline std::string highest();
280 static inline std::string infinity();
281 static inline std::string quiet_NaN();
282};
