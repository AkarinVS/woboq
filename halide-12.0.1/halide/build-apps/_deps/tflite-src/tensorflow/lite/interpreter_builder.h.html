38/// Build an interpreter capable of interpreting `model`.
39///
40/// `model`: A model whose lifetime must be at least as long as any
41/// interpreter(s) created by the builder. In principle multiple interpreters
42/// can be made from a single model.
43/// `op_resolver`: An instance that implements the `OpResolver` interface, which
44/// maps custom op names and builtin op codes to op registrations. The
45/// lifetime of the provided `op_resolver` object must be at least as long as
46/// the `InterpreterBuilder`; unlike `model` and `error_reporter`, the
47/// `op_resolver` does not need to exist for the duration of any created
48/// `Interpreter` objects.
49/// `error_reporter`: a functor that is called to report errors that handles
50/// printf var arg semantics. The lifetime of the `error_reporter` object must
51/// be greater than or equal to the `Interpreter` created by `operator()`.
52///
53/// Returns a kTfLiteOk when successful and sets interpreter to a valid
54/// Interpreter. Note: The user must ensure the lifetime of the model (and error
55/// reporter, if provided) is at least as long as interpreter's lifetime.
56class InterpreterBuilder {
57 public:
58 /// For this constructor, the ErrorReporter will be extracted from the
59 /// FlatBufferModel.
60 InterpreterBuilder(const FlatBufferModel& model,
61 const OpResolver& op_resolver);
62 /// Builds an interpreter given only the raw flatbuffer Model object (instead
63 /// of a FlatBufferModel). Mostly used for testing.
64 /// If `error_reporter` is null, then DefaultErrorReporter() is used.
65 InterpreterBuilder(const ::tflite::Model* model,
66 const OpResolver& op_resolver,
67 ErrorReporter* error_reporter = DefaultErrorReporter());
68 ~InterpreterBuilder();
69 InterpreterBuilder(const InterpreterBuilder&) = delete;
70 InterpreterBuilder& operator=(const InterpreterBuilder&) = delete;
71
72 TfLiteStatus operator()(std::unique_ptr<Interpreter>* interpreter);
73 TfLiteStatus operator()(std::unique_ptr<Interpreter>* interpreter,
74 int num_threads);
75
76 /// Enables preserving intermediates for debugging. Otherwise, by default
77 /// intermediates are undefined due to memory planning and reuse.
78 InterpreterBuilder& PreserveAllTensorsExperimental();
79
80 /// Any delegates added with AddDelegate will be applied to the Interpreter
81 /// generated by operator(), in the order that they were added. (The delegate
82 /// parameter passed to AddDelegate should be non-null, otherwise an error
83 /// will be reported, and the call to AddDelegate will have no other effect.)
84 /// The lifetime of the delegate must be at least as long as the lifetime of
85 /// any Interpreter generated by this InterpreterBuilder.
86 /// WARNING: This is an experimental API and subject to change.
87 void AddDelegate(TfLiteDelegate* delegate);
88
89 private:
90 TfLiteStatus BuildLocalIndexToRegistrationMapping();
91 TfLiteStatus ParseNodes(
92 const flatbuffers::Vector<flatbuffers::Offset<Operator>>* operators,
93 Subgraph* subgraph);
94 TfLiteStatus ParseTensors(
95 const flatbuffers::Vector<flatbuffers::Offset<Buffer>>* buffers,
96 const flatbuffers::Vector<flatbuffers::Offset<Tensor>>* tensors,
97 Subgraph* subgraph);
98 TfLiteStatus ApplyDelegates(Interpreter* interpreter, int num_threads);
99 TfLiteStatus ParseQuantization(const QuantizationParameters* src_quantization,
100 TfLiteQuantization* quantization,
101 const std::vector<int>& dims);
102 TfLiteStatus ParseSparsity(const SparsityParameters* src_sparsity,
103 TfLiteSparsity** sparsity);
104 TfLiteStatus ParseSignatureDefs(
105 const flatbuffers::Vector<flatbuffers::Offset<SignatureDef>>*
106 signature_def_list,
107 Interpreter* interpreter);
108
109 const ::tflite::Model* model_;
110 const OpResolver& op_resolver_;
111 ErrorReporter* error_reporter_;
112 std::vector<TfLiteDelegate*> delegates_;
113
114 std::vector<const TfLiteRegistration*> flatbuffer_op_index_to_registration_;
115 std::vector<TfLiteRegistration> unresolved_custom_ops_;
116 std::vector<BuiltinOperator> flatbuffer_op_index_to_registration_types_;
117 const Allocation* allocation_ = nullptr;
118
119 bool has_flex_op_ = false;
120 int num_fp32_tensors_ = 0;
121 bool preserve_all_tensors_ = false;
122};
