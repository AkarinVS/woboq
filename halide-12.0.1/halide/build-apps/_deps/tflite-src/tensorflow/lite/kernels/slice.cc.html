37enum KernelType {

39 kGenericOptimized,

42constexpr int kInputTensor = 0;
43constexpr int kBeginTensor = 1;
44constexpr int kSizeTensor = 2;
45constexpr int kOutputTensor = 0;
46
47// This Op only supports 1-5D cases and since we use the optimized ops 5D
48// implementation, the 1-4D tensors are mapped to 5D.
49const int kMaxDim = 5;
50
51template <typename T>
52TfLiteStatus CalculateOutputShapeVector(TfLiteContext* context,
53 const TfLiteTensor* input,
54 const TfLiteTensor* begin,
55 const TfLiteTensor* size,
56 std::vector<int>* output_shape_vector) {
57 for (int idx = 0; idx < NumDimensions(input); ++idx) {
58 T size_value = GetTensorData<T>(size)[idx];
59 if (size_value < 0) {
60 if (size_value != -1) {
61 context->ReportError(context, "Invalid size.");
62 return kTfLiteError;
63 }
64 size_value = SizeOfDimension(input, idx) - GetTensorData<T>(begin)[idx];
65 } else {
66 if (SizeOfDimension(input, idx) <
67 GetTensorData<T>(begin)[idx] + size_value) {
68 context->ReportError(context, "Invalid begin and size.");
69 return kTfLiteError;
70 }
71 }
72 output_shape_vector->push_back(static_cast<int>(size_value));
73 }
74 return kTfLiteOk;
75}
76
77template <typename T>
78void GetBeginAndSizeVectors(int dimensions, const TfLiteTensor* begin,
79 const TfLiteTensor* size, std::vector<int>* begins,
80 std::vector<int>* sizes) {
81 for (int idx = 0; idx < dimensions; ++idx) {
82 begins->push_back(GetTensorData<T>(begin)[idx]);
83 sizes->push_back(GetTensorData<T>(size)[idx]);
84 }
85}
86
87TfLiteStatus ResizeOutputShape(TfLiteContext* context,
88 const TfLiteTensor* input,
89 const TfLiteTensor* begin,
90 const TfLiteTensor* size, TfLiteTensor* output) {
91 std::vector<int> output_shape_vector;
92
93 if (begin->type == kTfLiteInt32) {
94 TF_LITE_ENSURE_STATUS(CalculateOutputShapeVector<int32_t>(
95 context, input, begin, size, &output_shape_vector));
96 } else if (begin->type == kTfLiteInt64) {
97 TF_LITE_ENSURE_STATUS(CalculateOutputShapeVector<int64_t>(
98 context, input, begin, size, &output_shape_vector));
99 } else {
100 context->ReportError(
101 context, "Type %d is currently not supported by Slice.", begin->type);
102 return kTfLiteError;
103 }
104
105 TfLiteIntArray* output_shape =
106 TfLiteIntArrayCreate(output_shape_vector.size());
107 std::copy(output_shape_vector.begin(), output_shape_vector.end(),
108 output_shape->data);
109 return context->ResizeTensor(context, output, output_shape);
110}

112TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
113 TF_LITE_ENSURE_EQ(context, NumInputs(node), 3);
114 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
115
116 const TfLiteTensor* input;
117 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
118 const TfLiteTensor* begin;
119 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kBeginTensor, &begin));
120 const TfLiteTensor* size;
121 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kSizeTensor, &size));
122 TfLiteTensor* output;
123 TF_LITE_ENSURE_OK(context,
124 GetOutputSafe(context, node, kOutputTensor, &output));
125
126 // Ensure validity of input tensor and its dimension.
127 TF_LITE_ENSURE_TYPES_EQ(context, input->type, output->type);
128 TF_LITE_ENSURE(context,
129 begin->type == kTfLiteInt32 || begin->type == kTfLiteInt64);
130 TF_LITE_ENSURE(context,
131 size->type == kTfLiteInt32 || size->type == kTfLiteInt64);
132 TF_LITE_ENSURE_EQ(context, NumDimensions(begin), 1);
133 TF_LITE_ENSURE_EQ(context, NumDimensions(size), 1);
134 TF_LITE_ENSURE_EQ(context, NumElements(begin), NumElements(size));
135 TF_LITE_ENSURE_MSG(context, NumDimensions(input) <= kMaxDim,
136 "Slice op only supports 1D-5D input arrays.");
137
138 // Postpone allocation of output if any of the indexing tensors is not
139 // constant
140 if (!(IsConstantTensor(begin) && IsConstantTensor(size))) {
141 SetTensorToDynamic(output);
142 return kTfLiteOk;
143 }
144
145 return ResizeOutputShape(context, input, begin, size, output);
146}

148template <KernelType kernel_type>
149TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
150 const TfLiteTensor* input;
151 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
152 const TfLiteTensor* begin;
153 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kBeginTensor, &begin));
154 const TfLiteTensor* size;
155 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kSizeTensor, &size));
156 TfLiteTensor* output;
157 TF_LITE_ENSURE_OK(context,
158 GetOutputSafe(context, node, kOutputTensor, &output));
159
160 if (IsDynamicTensor(output)) {
161 TF_LITE_ENSURE_OK(context,
162 ResizeOutputShape(context, input, begin, size, output));
163 }
164
165 std::vector<int> begins;
166 begins.reserve(kMaxDim);
167 std::vector<int> sizes;
168 sizes.reserve(kMaxDim);
169
170 for (int i = NumDimensions(input); i < kMaxDim; ++i) {
171 begins.push_back(0);
172 sizes.push_back(1);
173 }
174
175 if (begin->type == kTfLiteInt32) {
176 GetBeginAndSizeVectors<int32_t>(NumDimensions(input), begin, size, &begins,
177 &sizes);
178 } else if (begin->type == kTfLiteInt64) {
179 GetBeginAndSizeVectors<int64_t>(NumDimensions(input), begin, size, &begins,
180 &sizes);
181 } else {
182 context->ReportError(
183 context, "Type %d is currently not supported by Slice.", begin->type);
184 return kTfLiteError;
185 }
186
187 // The Slice op implementation only accepts 5-D sizes. That constraint is, for
188 // the present, maintained here.
189 //
190 // The dimensions in the kernel used to be in reverse-order, and TFLite
191 // arranged the begins and sizes vectors accordingly. This macro incorporates
192 // the needed reversing.
193#define TF_LITE_SLICE(data_type) \
194 { \
195 TF_LITE_ENSURE_EQ(context, begins.size(), kMaxDim); \
196 TF_LITE_ENSURE_EQ(context, sizes.size(), kMaxDim); \
197 tflite::SliceParams op_params; \
198 op_params.begin_count = kMaxDim; \
199 op_params.size_count = kMaxDim; \
200 for (int i = 0; i < kMaxDim; ++i) { \
201 op_params.begin[i] = begins[i]; \
202 op_params.size[i] = sizes[i]; \
203 } \
204 \
205 if (kernel_type == kGenericOptimized) { \
206 optimized_ops::Slice<data_type>(op_params, GetTensorShape(input), input, \
207 GetTensorShape(output), output); \
208 } else { \
209 reference_ops::Slice<data_type>(op_params, GetTensorShape(input), input, \
210 GetTensorShape(output), output); \
211 } \
212 }
213
214 switch (input->type) {
215 case kTfLiteFloat32:
216 TF_LITE_SLICE(float);
217 break;
218 case kTfLiteInt32:
219 TF_LITE_SLICE(int32_t);
220 break;
221 case kTfLiteInt64:
222 TF_LITE_SLICE(int64_t);
223 break;
224 case kTfLiteInt8:
225 TF_LITE_SLICE(int8_t);
226 break;
227 case kTfLiteInt16:
228 TF_LITE_SLICE(int16_t);
229 break;
230 case kTfLiteUInt8:
231 TF_LITE_SLICE(uint8_t);
232 break;
233 case kTfLiteBool:
234 TF_LITE_SLICE(bool);
235 break;
236 case kTfLiteString:
237 TF_LITE_SLICE(string);
238 break;
239 default:
240 context->ReportError(
241 context, "Type %d is currently not supported by Slice.", input->type);
242 return kTfLiteError;
243 }
244#undef TF_LITE_SLICE
245 return kTfLiteOk;
246}
