35// This file has two implementation of Quantize.
36enum KernelType {
37 kReference,
38 kGenericOptimized,
39};
40
41struct OpData {
42 int32_t output_multiplier;
43 int output_shift;
44};

46namespace {
47template <KernelType kernel_type, typename output_type>
48static inline void AffineQuantize(const tflite::QuantizationParams& op_params,
49 const RuntimeShape& input_shape,
50 const float* input_data,
51 const RuntimeShape& output_shape,
52 output_type* output_data) {
53 if (kernel_type == kReference) {
54 reference_ops::AffineQuantize(op_params, input_shape, input_data,
55 output_shape, output_data);
56 } else {
57 optimized_ops::AffineQuantize(op_params, input_shape, input_data,
58 output_shape, output_data);
59 }
60}
61
62template <KernelType kernel_type, typename input_type, typename output_type>
63static inline void Requantize(const input_type* input_data, int32_t size,
64 int32_t effective_scale_multiplier,
65 int32_t effective_scale_shift,
66 int32_t input_zeropoint, int32_t output_zeropoint,
67 output_type* output_data) {
68 if (kernel_type == kReference) {
69 reference_ops::Requantize(input_data, size, effective_scale_multiplier,
70 effective_scale_shift, input_zeropoint,
71 output_zeropoint, output_data);
72 } else {
73 optimized_ops::Requantize(input_data, size, effective_scale_multiplier,
74 effective_scale_shift, input_zeropoint,
75 output_zeropoint, output_data);
76 }
77}

95TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
96 OpData* data = static_cast<OpData*>(node->user_data);
97 TF_LITE_ENSURE_EQ(context, NumInputs(node), 1);
98 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
99
100 const TfLiteTensor* input;
101 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
102 TfLiteTensor* output;
103 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
104
105 // TODO(b/128934713): Add support for fixed-point per-channel quantization.
106 // Currently this only support affine per-layer quantization.
107 TF_LITE_ENSURE_EQ(context, output->quantization.type,
108 kTfLiteAffineQuantization);
109 const auto* affine_quantization =
110 static_cast<TfLiteAffineQuantization*>(output->quantization.params);
111 TF_LITE_ENSURE(context, affine_quantization);
112 TF_LITE_ENSURE(context, affine_quantization->scale);
113 TF_LITE_ENSURE(context, affine_quantization->scale->size == 1);
114
115 if (input->type == kTfLiteFloat32) {
116 // Quantize use case.
117 TF_LITE_ENSURE(context, output->type == kTfLiteUInt8 ||
118 output->type == kTfLiteInt8 ||
119 output->type == kTfLiteInt16);
120 } else {
121 // Requantize use case.
122 if (input->type == kTfLiteInt16) {
123 TF_LITE_ENSURE(context, output->type == kTfLiteInt8 ||
124 output->type == kTfLiteInt16 ||
125 output->type == kTfLiteInt32);
126 } else {
127 TF_LITE_ENSURE(context,
128 input->type == kTfLiteInt8 || input->type == kTfLiteUInt8);
129 TF_LITE_ENSURE(
130 context, output->type == kTfLiteUInt8 || output->type == kTfLiteInt8);
131 }
132 const double effective_output_scale =
133 static_cast<double>(input->params.scale) /
134 static_cast<double>(output->params.scale);
135 QuantizeMultiplier(effective_output_scale, &data->output_multiplier,
136 &data->output_shift);
137 }
138
139 if (input->type == kTfLiteInt16 && output->type == kTfLiteInt16) {
140 TF_LITE_ENSURE_EQ(context, input->params.zero_point, 0);
141 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
142 }
143
144 return context->ResizeTensor(context, output,
145 TfLiteIntArrayCopy(input->dims));
146}

148template <KernelType kernel_type>
149TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
150 OpData* data = static_cast<OpData*>(node->user_data);
151
152 const TfLiteTensor* input;
153 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
154 TfLiteTensor* output;
155 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
156
157 const RuntimeShape input_shape = GetTensorShape(input);
158 const RuntimeShape output_shape = GetTensorShape(output);
159
160 switch (input->type) {
161 case kTfLiteFloat32: {
162 // Float to int8, uint8, int16.
163 tflite::QuantizationParams op_params;
164 op_params.zero_point = output->params.zero_point;
165 op_params.scale = output->params.scale;
166 const float* input_data = GetTensorData<float>(input);
167 switch (output->type) {
168 case kTfLiteInt8:
169 AffineQuantize<kernel_type>(op_params, input_shape, input_data,
170 output_shape,
171 GetTensorData<int8_t>(output));
172 return kTfLiteOk;
173 case kTfLiteUInt8:
174 AffineQuantize<kernel_type>(op_params, input_shape, input_data,
175 output_shape,
176 GetTensorData<uint8_t>(output));
177 return kTfLiteOk;
178 case kTfLiteInt16:
179 AffineQuantize<kernel_type>(op_params, input_shape, input_data,
180 output_shape,
181 GetTensorData<int16_t>(output));
182 return kTfLiteOk;
183 default:
184 ReportError(context, input->type, output->type);
185 return kTfLiteError;
186 }
187 }
188 case kTfLiteInt16: {
189 // int16 to int8 or int16.
190 switch (output->type) {
191 case kTfLiteInt8:
192 Requantize<kernel_type>(GetTensorData<int16_t>(input),
193 MatchingFlatSize(input_shape, output_shape),
194 data->output_multiplier, data->output_shift,
195 input->params.zero_point,
196 output->params.zero_point,
197 GetTensorData<int8_t>(output));
198 return kTfLiteOk;
199 case kTfLiteInt16:
200 Requantize<kernel_type>(GetTensorData<int16_t>(input),
201 MatchingFlatSize(input_shape, output_shape),
202 data->output_multiplier, data->output_shift,
203 input->params.zero_point,
204 output->params.zero_point,
205 GetTensorData<int16_t>(output));
206 return kTfLiteOk;
207 case kTfLiteInt32:
208 // This case is not supported by the converter or other TFLite tools.
209 // The only use case is for applications that take quantized int32
210 // inference outputs.
211 Requantize<kernel_type>(GetTensorData<int16_t>(input),
212 MatchingFlatSize(input_shape, output_shape),
213 data->output_multiplier, data->output_shift,
214 input->params.zero_point,
215 output->params.zero_point,
216 GetTensorData<int32_t>(output));
217 return kTfLiteOk;
218 default:
219 ReportError(context, input->type, output->type);
220 return kTfLiteError;
221 }
222 }
223 case kTfLiteInt8: {
224 // int8 to int8, uint8.
225 const int32_t size = MatchingFlatSize(input_shape, output_shape);
226 const int8_t* input_data = GetTensorData<int8_t>(input);
227 switch (output->type) {
228 case kTfLiteInt8:
229 Requantize<kernel_type>(input_data, size, data->output_multiplier,
230 data->output_shift, input->params.zero_point,
231 output->params.zero_point,
232 GetTensorData<int8_t>(output));
233 return kTfLiteOk;
234 case kTfLiteUInt8:
235 Requantize<kernel_type>(input_data, size, data->output_multiplier,
236 data->output_shift, input->params.zero_point,
237 output->params.zero_point,
238 GetTensorData<uint8_t>(output));
239 return kTfLiteOk;
240 default:
241 ReportError(context, input->type, output->type);
242 return kTfLiteError;
243 }
244 }
245 case kTfLiteUInt8: {
246 // uint8 to int8, uint8.
247 const int32_t size = MatchingFlatSize(input_shape, output_shape);
248 const uint8_t* input_data = GetTensorData<uint8_t>(input);
249 switch (output->type) {
250 case kTfLiteInt8:
251 Requantize<kernel_type>(input_data, size, data->output_multiplier,
252 data->output_shift, input->params.zero_point,
253 output->params.zero_point,
254 GetTensorData<int8_t>(output));
255 return kTfLiteOk;
256 case kTfLiteUInt8:
257 Requantize<kernel_type>(input_data, size, data->output_multiplier,
258 data->output_shift, input->params.zero_point,
259 output->params.zero_point,
260 GetTensorData<uint8_t>(output));
261 return kTfLiteOk;
262 default:
263 ReportError(context, input->type, output->type);
264 return kTfLiteError;
265 }
266 }
267 default:
268 ReportError(context, input->type, output->type);
269 return kTfLiteError;
270 }
271}

275// This Op (QUANTIZE) quantizes the input and produces quantized output.
276// The input can be either float or quantized. If the input is float,
277// AffineQuantize takes scale and zero point and quantize the float value to
278// quantized output, in int8 or uint8 format. If the input is quantized value,
279// the op requantize the input (of a certain type, with a given scale and zero
280// point) to the output of the same or different type with a same or different
281// scale and zero point.
282TfLiteRegistration* Register_QUANTIZE_OPT() {
283 static TfLiteRegistration r = {quantize::Init, quantize::Free,
284 quantize::Prepare,
285 quantize::Eval<quantize::kGenericOptimized>};
286 return &r;
287}
288
289TfLiteRegistration* Register_QUANTIZE_REF() {
290 static TfLiteRegistration r = {quantize::Init, quantize::Free,
291 quantize::Prepare,
292 quantize::Eval<quantize::kReference>};
293 return &r;
294}
295
296TfLiteRegistration* Register_QUANTIZE() { return Register_QUANTIZE_OPT(); }
