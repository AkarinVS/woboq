60// Extract upper 8 bits from each 16-bit integer in vector registers. This is
61// performed for a pair, because instructions often work on pairs.
62inline void PairExtractUpper(const uint16x8_t accum_0, const uint16x8_t accum_1,
63 uint8x8_t* res_0, uint8x8_t* res_1) {
64 uint8x16x2_t unzipped =
65 vuzpq_u8(vreinterpretq_u8_u16(accum_0), vreinterpretq_u8_u16(accum_1));
66 *res_0 = vget_low_u8(unzipped.val[1]);
67 *res_1 = vget_high_u8(unzipped.val[1]);
68}

70// This is an exceptional definition.
71//
72// Modify int16x8_t, adding operators.
73//
74// There are exceptional circumstances that make it reasonable to write code
75// on vector types for quantized resize bilinear in *some cases*.
76//
77// (a) In exact quant resize bilinear, it should be possible to guarantee that
78// arithmetic never overflows.
79// (b) When the resize scaling is 2 or 4 or 8 it is possible to guarantee
80// exact accumulation and exact incrementation.
81// (c) In quant resize bilinear the choice of unsigned vs signed accumulation
82// and saturated vs unsaturated arithmetic is often unimportant.
83//
84// This pattern simplifies the code considerably. This pattern should not be
85// used more widely in code since it can hide important numerical detail.
86//
87// DO NOT add to this any "class-like" methods: only those that do no more than
88// redirecting operators to specific intrinsics functions.
89struct op_int16x8_t {
90 inline op_int16x8_t() = default;
91 inline explicit op_int16x8_t(const int16x8_t& initial_val) {
92 val = initial_val;
93 }
94 inline op_int16x8_t& operator=(const int16x8_t& new_val) {
95 val = new_val;
96 return *this;
97 }
98 inline op_int16x8_t operator+=(const op_int16x8_t& add_val) {
99 val = vaddq_s16(val, add_val.val);
100 return *this;
101 }
102 inline op_int16x8_t operator-=(const op_int16x8_t& sub_val) {
103 val = vsubq_s16(val, sub_val.val);
104 return *this;
105 }
106 // This really selects vshlq_n_s16, but requires a longer implementation to
107 // convert the shift argument back to a constant. In some compiles are macros
108 // requiring constant args.
109 inline op_int16x8_t operator<<=(int32 left_shift) {
110 switch (left_shift) {
111 case 1:
112 val = vshlq_n_s16(val, 1);
113 break;
114 case 4:
115 val = vshlq_n_s16(val, 4);
116 break;
117 case 8:
118 val = vshlq_n_s16(val, 8);
119 break;
120 default:
121 TFLITE_CHECK(false);
122 break;
123 }
124 return *this;
125 }
126 // This really selects vshrq_n_u16, but requires a longer implementation to
127 // convert the shift argument back to a constant. In some compiles are macros
128 // requiring constant args.
129 inline op_int16x8_t operator>>=(int32 right_shift) {
130 switch (right_shift) {
131 case 1:
132 val = vshrq_n_s16(val, 1);
133 break;
134 case 4:
135 val = vshrq_n_s16(val, 4);
136 break;
137 case 8:
138 val = vshrq_n_s16(val, 8);
139 break;
140 default:
141 TFLITE_CHECK(false);
142 break;
143 }
144 return *this;
145 }
146 friend inline op_int16x8_t operator+(op_int16x8_t lhs,
147 const op_int16x8_t& rhs) {
148 lhs += rhs;
149 return lhs;
150 }
151 friend inline op_int16x8_t operator-(op_int16x8_t lhs,
152 const op_int16x8_t& rhs) {
153 lhs -= rhs;
154 return lhs;
155 }
156 friend inline op_int16x8_t operator<<(op_int16x8_t lhs, int32 left_shift) {
157 lhs <<= left_shift;
158 return lhs;
159 }
160 friend inline op_int16x8_t operator>>(op_int16x8_t lhs, int32 right_shift) {
161 lhs >>= right_shift;
162 return lhs;
163 }
164
165 int16x8_t val;
166};

258// Optimized resize-bilinear for the special case where the scaling is x8 in
259// width and height, and where we can operate on depth-8 blocks at a time. So
260// the output blocks are 8x8x8 in width-height-depth.
261//
262// This optimization is for the half_pixel_centers == true version, for uint8.
263// There are versions for NEON and non-NEON compilation.
264inline void ResizeBilinear888Uint8(int32 batches, int32 input_height,
265 int32 input_width, int32 depth,
266 const uint8* input_data,
267 uint8* output_data) {
268 TFLITE_DCHECK_GE(input_height, 1);
269 TFLITE_DCHECK_GE(input_width, 1);
270 TFLITE_DCHECK_EQ(depth % 8, 0);
271
272 const int32 input_row_stride = input_width * depth;
273 const int32 output_row_stride = input_row_stride * 8;
274 for (int b = 0; b < batches; ++b) {
275 const uint8* input_base_ptr =
276 input_data + b * input_row_stride * input_height;
277 uint8* output_base_ptr =
278 output_data + b * output_row_stride * input_height * 8;
279
280#ifdef USE_NEON
281 for (int c_block = 0; c_block < depth; c_block += 8) {
282 op_uint16x8_t accum_c_v;
283 // Top-left margin corner.
284 {
285 uint8x8_t output_data = vld1_u8(&input_base_ptr[c_block]);
286 vst1_u8(&output_base_ptr[c_block], output_data);
287 vst1_u8(&output_base_ptr[c_block + depth], output_data);
288 vst1_u8(&output_base_ptr[c_block + depth * 2], output_data);
289 vst1_u8(&output_base_ptr[c_block + depth * 3], output_data);
290
291 // Accumulate in 8.8 representation, pre-adding 0.5 for later rounding.
292 accum_c_v = vaddq_u16(Move8IntoUpperU16(output_data), vdupq_n_u16(128));
293 }
294
295 // Top-centre margin.
296 op_int16x8_t wdelta_c_v;
297 op_int16x8_t wdelta_twice_c_v;
298 for (int j = 0; j < (input_width - 1); ++j) {
299 {
300 uint8x8_t output_data_alt;
301 uint8x8_t output_data;
302
303 const op_int16x8_t tl_val(
304 Load8IntoLowerS16(&input_base_ptr[c_block + depth * j]));
305 const op_int16x8_t tr_val(
306 Load8IntoLowerS16(&input_base_ptr[c_block + depth * (j + 1)]));
307 wdelta_c_v = (tr_val - tl_val) << 4;
308 wdelta_twice_c_v = wdelta_c_v << 1;
309
310 op_uint16x8_t accum_c_v_alt = accum_c_v + wdelta_c_v;
311 accum_c_v = accum_c_v_alt + wdelta_twice_c_v;
312 PairExtractUpper(accum_c_v_alt.val, accum_c_v.val, &output_data_alt,
313 &output_data);
314
315 vst1_u8(&output_base_ptr[c_block + depth * j * 8 + depth * 4],
316 output_data_alt);
317 vst1_u8(&output_base_ptr[c_block + depth * j * 8 + depth + depth * 4],
318 output_data);
319
320 for (int p = 2; p < 8; p += 2) {
321 accum_c_v_alt = accum_c_v + wdelta_twice_c_v;
322 accum_c_v = accum_c_v_alt + wdelta_twice_c_v;
323 PairExtractUpper(accum_c_v_alt.val, accum_c_v.val, &output_data_alt,
324 &output_data);
325
326 vst1_u8(&output_base_ptr[c_block + depth * j * 8 + depth * p +
327 depth * 4],
328 output_data_alt);
329 vst1_u8(&output_base_ptr[c_block + depth * j * 8 + depth * (p + 1) +
330 depth * 4],
331 output_data);
332 }
333 accum_c_v += wdelta_c_v;
334 }
335 }
336
337 // Top-right margin corner.
338 {
339 uint8x8_t output_data_discard;
340 uint8x8_t output_data;
341
342 // Accumulations have pre-added 0.5 for rounding, but that is just
343 // discarded and this just avoids re-loading.
344 PairExtractUpper(accum_c_v.val, accum_c_v.val, &output_data,
345 &output_data_discard);
346
347 vst1_u8(&output_base_ptr[c_block + depth * (input_width - 1) * 8 +
348 depth * 4],
349 output_data);
350 vst1_u8(&output_base_ptr[c_block + depth * (input_width - 1) * 8 +
351 depth * 4 + depth],
352 output_data);
353 vst1_u8(&output_base_ptr[c_block + depth * (input_width - 1) * 8 +
354 depth * 4 + depth * 2],
355 output_data);
356 vst1_u8(&output_base_ptr[c_block + depth * (input_width - 1) * 8 +
357 depth * 4 + depth * 3],
358 output_data);
359 }
360 }
361 // Fill out remainder of top margin.
362 std::memcpy(output_base_ptr + output_row_stride, output_base_ptr,
363 output_row_stride * sizeof(uint8));
364 std::memcpy(output_base_ptr + output_row_stride * 2, output_base_ptr,
365 output_row_stride * sizeof(uint8));
366 std::memcpy(output_base_ptr + output_row_stride * 3, output_base_ptr,
367 output_row_stride * sizeof(uint8));
368 output_base_ptr += output_row_stride * 4;
369
370 // Main rows.
371 for (int k = 0; k < (input_height - 1); ++k) {
372 for (int c_block = 0; c_block < depth; c_block += 8) {
373 uint8* output_base_ptr_0 = output_base_ptr;
374 uint8* output_base_ptr_1;
375 uint8* output_base_ptr_2;
376 uint8* output_base_ptr_3;
377 uint8* output_base_ptr_4;
378 uint8* output_base_ptr_5;
379 uint8* output_base_ptr_6;
380 uint8* output_base_ptr_7;
381
382 op_uint16x8_t accum_0_c_v;
383 op_uint16x8_t accum_1_c_v;
384 op_uint16x8_t accum_2_c_v;
385 op_uint16x8_t accum_3_c_v;
386 op_uint16x8_t accum_4_c_v;
387 op_uint16x8_t accum_5_c_v;
388 op_uint16x8_t accum_6_c_v;
389 op_uint16x8_t accum_7_c_v;
390
391 op_int16x8_t hdelta_c_v;
392 op_int16x8_t hdelta_twice_c_v;
393
394 // Left margin for 8 rows.
395 {
396 uint8x8_t output_data_0_c;
397 uint8x8_t output_data_1_c;
398 uint8x8_t output_data_2_c;
399 uint8x8_t output_data_3_c;
400 uint8x8_t output_data_4_c;
401 uint8x8_t output_data_5_c;
402 uint8x8_t output_data_6_c;
403 uint8x8_t output_data_7_c;
404
405 const op_int16x8_t tl_val(
406 Load8IntoLowerS16(&input_base_ptr[c_block]));
407 const op_int16x8_t bl_val(
408 Load8IntoLowerS16(&input_base_ptr[c_block + input_row_stride]));
409 hdelta_c_v = (bl_val - tl_val) << 4;
410
411 // Accumulate in 8.8 representation, pre-adding 0.5 for later
412 // rounding.
413 accum_0_c_v = VReinterpretQU16S16(tl_val << 8);
414 accum_0_c_v = vaddq_u16(accum_0_c_v.val, vdupq_n_u16(128));
415
416 hdelta_twice_c_v = hdelta_c_v << 1;
417
418 accum_0_c_v += hdelta_c_v;
419 accum_1_c_v = accum_0_c_v + hdelta_twice_c_v;
420 PairExtractUpper(accum_0_c_v.val, accum_1_c_v.val, &output_data_0_c,
421 &output_data_1_c);
422
423 vst1_u8(&output_base_ptr_0[c_block], output_data_0_c);
424 vst1_u8(&output_base_ptr_0[c_block + depth], output_data_0_c);
425 vst1_u8(&output_base_ptr_0[c_block + depth * 2], output_data_0_c);
426 vst1_u8(&output_base_ptr_0[c_block + depth * 3], output_data_0_c);
427
428 output_base_ptr_1 = output_base_ptr_0 + output_row_stride;
429 vst1_u8(&output_base_ptr_1[c_block], output_data_1_c);
430 vst1_u8(&output_base_ptr_1[c_block + depth], output_data_1_c);
431 vst1_u8(&output_base_ptr_1[c_block + depth * 2], output_data_1_c);
432 vst1_u8(&output_base_ptr_1[c_block + depth * 3], output_data_1_c);
433
434 //
435
436 output_base_ptr_2 = output_base_ptr_1 + output_row_stride;
437 accum_2_c_v = accum_1_c_v + hdelta_twice_c_v;
438 accum_3_c_v = accum_2_c_v + hdelta_twice_c_v;
439 PairExtractUpper(accum_2_c_v.val, accum_3_c_v.val, &output_data_2_c,
440 &output_data_3_c);
441
442 vst1_u8(&output_base_ptr_2[c_block], output_data_2_c);
443 vst1_u8(&output_base_ptr_2[c_block + depth], output_data_2_c);
444 vst1_u8(&output_base_ptr_2[c_block + depth * 2], output_data_2_c);
445 vst1_u8(&output_base_ptr_2[c_block + depth * 3], output_data_2_c);
446
447 output_base_ptr_3 = output_base_ptr_2 + output_row_stride;
448 vst1_u8(&output_base_ptr_3[c_block], output_data_3_c);
449 vst1_u8(&output_base_ptr_3[c_block + depth], output_data_3_c);
450 vst1_u8(&output_base_ptr_3[c_block + depth * 2], output_data_3_c);
451 vst1_u8(&output_base_ptr_3[c_block + depth * 3], output_data_3_c);
452
453 //
454
455 output_base_ptr_4 = output_base_ptr_3 + output_row_stride;
456 accum_4_c_v = accum_3_c_v + hdelta_twice_c_v;
457 accum_5_c_v = accum_4_c_v + hdelta_twice_c_v;
458 PairExtractUpper(accum_4_c_v.val, accum_5_c_v.val, &output_data_4_c,
459 &output_data_5_c);
460
461 vst1_u8(&output_base_ptr_4[c_block], output_data_4_c);
462 vst1_u8(&output_base_ptr_4[c_block + depth], output_data_4_c);
463 vst1_u8(&output_base_ptr_4[c_block + depth * 2], output_data_4_c);
464 vst1_u8(&output_base_ptr_4[c_block + depth * 3], output_data_4_c);
465
466 output_base_ptr_5 = output_base_ptr_4 + output_row_stride;
467 vst1_u8(&output_base_ptr_5[c_block], output_data_5_c);
468 vst1_u8(&output_base_ptr_5[c_block + depth], output_data_5_c);
469 vst1_u8(&output_base_ptr_5[c_block + depth * 2], output_data_5_c);
470 vst1_u8(&output_base_ptr_5[c_block + depth * 3], output_data_5_c);
471
472 //
473
474 output_base_ptr_6 = output_base_ptr_5 + output_row_stride;
475 accum_6_c_v = accum_5_c_v + hdelta_twice_c_v;
476 accum_7_c_v = accum_6_c_v + hdelta_twice_c_v;
477 PairExtractUpper(accum_6_c_v.val, accum_7_c_v.val, &output_data_6_c,
478 &output_data_7_c);
479
480 vst1_u8(&output_base_ptr_6[c_block], output_data_6_c);
481 vst1_u8(&output_base_ptr_6[c_block + depth], output_data_6_c);
482 vst1_u8(&output_base_ptr_6[c_block + depth * 2], output_data_6_c);
483 vst1_u8(&output_base_ptr_6[c_block + depth * 3], output_data_6_c);
484
485 output_base_ptr_7 = output_base_ptr_6 + output_row_stride;
486 vst1_u8(&output_base_ptr_7[c_block], output_data_7_c);
487 vst1_u8(&output_base_ptr_7[c_block + depth], output_data_7_c);
488 vst1_u8(&output_base_ptr_7[c_block + depth * 2], output_data_7_c);
489 vst1_u8(&output_base_ptr_7[c_block + depth * 3], output_data_7_c);
490 }

492 // Main central body.
493 op_int16x8_t wdelta_c;
494 op_int16x8_t wdelta_twice_c;
495 op_int16x8_t hwdelta_c;
496 op_int16x8_t hwdelta_twice_c;
497
498 op_int16x8_t incr_0_c;
499 op_int16x8_t incr_1_c;
500 op_int16x8_t incr_2_c;
501 op_int16x8_t incr_3_c;
502 op_int16x8_t incr_4_c;
503 op_int16x8_t incr_5_c;
504 op_int16x8_t incr_6_c;
505 op_int16x8_t incr_7_c;
506
507 uint8x8_t output_data_0_c;
508 uint8x8_t output_data_1_c;
509 uint8x8_t output_data_2_c;
510 uint8x8_t output_data_3_c;
511 uint8x8_t output_data_4_c;
512 uint8x8_t output_data_5_c;
513 uint8x8_t output_data_6_c;
514 uint8x8_t output_data_7_c;
515 for (int j = 0; j < (input_width - 1); ++j) {
516 // output_base_ptr_0 = output_base_ptr;
517 // output_base_ptr_1 = output_base_ptr_0 + output_row_stride; ETC
518 {
519 const op_int16x8_t tl_val(
520 Load8IntoLowerS16(&input_base_ptr[c_block + depth * j]));
521 const op_int16x8_t bl_val(Load8IntoLowerS16(
522 &input_base_ptr[c_block + depth * j + input_row_stride]));
523 const op_int16x8_t tr_val(
524 Load8IntoLowerS16(&input_base_ptr[c_block + depth * (j + 1)]));
525 const op_int16x8_t br_val(Load8IntoLowerS16(
526 &input_base_ptr[c_block + depth * (j + 1) + input_row_stride]));
527
528 const op_int16x8_t tmp_diff = tr_val - tl_val;
529 wdelta_c = tmp_diff << 4;

1510inline void ResizeBilinearGeneric(
1511 int32 batches, int32 input_height, int32 input_width, int32 depth,
1512 int32 output_height, int32 output_width, float height_scale,
1513 float width_scale, const RuntimeShape& input_shape, const float* input_data,
1514 const RuntimeShape& output_shape, float* output_data,
1515 const bool half_pixel_centers) {
1516 memset(output_data, 0,
1517 batches * output_height * output_width * depth * sizeof(float));
1518
1519 int32 output_offset = 0;
1520 for (int b = 0; b < batches; ++b) {
1521 for (int y = 0; y < output_height; ++y) {
1522 float input_y;
1523 int32 y0, y1;
1524 reference_ops::ComputeInterpolationValues(
1525 y, height_scale, half_pixel_centers, input_height, &input_y, &y0,
1526 &y1);
1527 for (int x = 0; x < output_width; ++x) {
1528 float input_x;
1529 int32 x0, x1;
1530 reference_ops::ComputeInterpolationValues(
1531 x, width_scale, half_pixel_centers, input_width, &input_x, &x0,
1532 &x1);
1533 float* output_ptr = &output_data[output_offset];
1534
1535 // Run kernel on the 4 corners of the bilinear resize algorithm.
1536 int32 input_offset = Offset(input_shape, b, y0, x0, 0);
1537 float scale = (1 - (input_y - y0)) * (1 - (input_x - x0));
1538 const float* input_ptr = &input_data[input_offset];
1539 ResizeBilinearKernel(input_ptr, depth, scale, output_ptr);
1540
1541 input_offset = Offset(input_shape, b, y0, x1, 0);
1542 scale = (1 - (input_y - y0)) * (input_x - x0);
1543 input_ptr = &input_data[input_offset];
1544 ResizeBilinearKernel(input_ptr, depth, scale, output_ptr);
1545
1546 input_offset = Offset(input_shape, b, y1, x0, 0);
1547 scale = (input_y - y0) * (1 - (input_x - x0));
1548 input_ptr = &input_data[input_offset];
1549 ResizeBilinearKernel(input_ptr, depth, scale, output_ptr);
1550
1551 input_offset = Offset(input_shape, b, y1, x1, 0);
1552 scale = (input_y - y0) * (input_x - x0);
1553 input_ptr = &input_data[input_offset];
1554 ResizeBilinearKernel(input_ptr, depth, scale, output_ptr);
1555
1556 output_offset += depth;
1557 }
1558 }
1559 }

1563inline void ResizeBilinearGenericSmallChannel(
1564 int32 batches, int32 input_height, int32 input_width, int32 depth,
1565 int32 output_height, int32 output_width, float height_scale,
1566 float width_scale, const RuntimeShape& input_shape, const T* input_data,
1567 const RuntimeShape& output_shape, T* output_data,
1568 const bool half_pixel_centers) {
1569 T* output_ptr = &output_data[0];
1570 const float rounding_offset = std::numeric_limits<T>::is_integer ? .5f : .0f;
1571
1572 for (int b = 0; b < batches; ++b) {
1573 for (int y = 0; y < output_height; ++y) {
1574 float input_y;
1575 int32 y0, y1;

1580 float input_x;

1583 x, width_scale, half_pixel_centers, input_width, &input_x, &x0,
1584 &x1);
1585
1586 int32 input_offset[4] = {Offset(input_shape, b, y0, x0, 0),
1587 Offset(input_shape, b, y0, x1, 0),

1590 float scale[4] = {(1 - (input_y - y0)) * (1 - (input_x - x0)),

1592 (input_y - y0) * (1 - (input_x - x0)),
1593 (input_y - y0) * (input_x - x0)};

1597 *output_ptr++ = static_cast<T>(input_ptr[input_offset[0]] * scale[0] +
1598 input_ptr[input_offset[1]] * scale[1] +
1599 input_ptr[input_offset[2]] * scale[2] +
1600 input_ptr[input_offset[3]] * scale[3] +
1601 rounding_offset);
1602 }
1603 }

1659inline void ResizeBilinear(const tflite::ResizeBilinearParams& op_params,
1660 const RuntimeShape& unextended_input_shape,
1661 const uint8* input_data,
1662 const RuntimeShape& output_size_shape,
1663 const int32* output_size_data,
1664 const RuntimeShape& unextended_output_shape,
1665 uint8* output_data) {
1666 ruy::profiler::ScopeLabel label("ResizeBilinearUint8");
1667 // If half_pixel_centers is True, align_corners must be False.
1668 TFLITE_DCHECK(!op_params.half_pixel_centers || !op_params.align_corners);
1669 TFLITE_DCHECK_LE(unextended_input_shape.DimensionsCount(), 4);
1670 TFLITE_DCHECK_LE(unextended_output_shape.DimensionsCount(), 4);
1671 const RuntimeShape input_shape =
1672 RuntimeShape::ExtendedShape(4, unextended_input_shape);
1673 const RuntimeShape output_shape =
1674 RuntimeShape::ExtendedShape(4, unextended_output_shape);
1675
1676 int32 batches = MatchingDim(input_shape, 0, output_shape, 0);
1677 int32 input_height = input_shape.Dims(1);
1678 int32 input_width = input_shape.Dims(2);
1679 int32 depth = MatchingDim(input_shape, 3, output_shape, 3);
1680
1681 TFLITE_DCHECK_EQ(output_size_shape.FlatSize(), 2);
1682 int32 output_height = output_size_data[0];
1683 int32 output_width = output_size_data[1];
1684
1685 if (!op_params.align_corners && op_params.half_pixel_centers &&
1686 ((depth % 8) == 0)) {
1687 const int32 scale = output_height / input_height;
1688 // Restricting the minimum output dimensions may not be necessary, but
1689 // ensures that kernels can use unrolling with minimal code size.
1690 if ((output_height >= 8) && (output_width >= 8) &&
1691 ((input_height * scale) == output_height) &&
1692 ((input_width * scale) == output_width)) {
1693 if (scale == 8) {
1694 resize_bilinear::ResizeBilinear888Uint8(
1695 batches, input_height, input_width, depth, input_data, output_data);

1697 }
1698 }
1699 }
1700
1701 float height_scale =
1702 (op_params.align_corners && output_height > 1)
1703 ? (static_cast<float>(input_height - 1) / (output_height - 1))
1704 : (static_cast<float>(input_height) / output_height);
1705
1706 float width_scale =
1707 (op_params.align_corners && output_width > 1)
1708 ? (static_cast<float>(input_width - 1) / (output_width - 1))
1709 : (static_cast<float>(input_width) / output_width);

1711 ResizeBilinearGenericSmallChannel<uint8>(
1712 batches, input_height, input_width, depth, output_height, output_width,
1713 height_scale, width_scale, input_shape, input_data, output_shape,
1714 output_data, op_params.half_pixel_centers);
