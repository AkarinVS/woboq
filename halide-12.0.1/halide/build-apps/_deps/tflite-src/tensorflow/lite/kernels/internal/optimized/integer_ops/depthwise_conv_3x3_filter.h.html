108template <>
109struct DepthwiseConvWindowPerChannel<DepthwiseConvOutputRounding::kUpward, 8, 1,
110 1> {
111 public:
112 static inline void Run(const int32* output_multiplier_ptr,
113 const int32* output_shift_ptr, const int8* input_ptr,
114 const int8* filter_ptr, const int32* bias_ptr,
115 int8* output_ptr, int64_t input_depth,
116 int64_t input_row_size, int32 output_window_height,
117 int32 output_window_width,
118 const DepthwiseConvParams* params_ptr) {
119 const int64_t input_width_increment = 2 * input_depth;
120 const int64_t input_height_increment = 2 * input_row_size;
121 const int64_t output_height_increment = 2 * params_ptr->output_row_size;
122 TFLITE_DCHECK_EQ(params_ptr->filter_offset, 0);

124#define DEPTHWISECONV_LABEL_HEIGHT_2_LOOP "1"
125#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP "2"
126#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_1_LEFTOVER "3"
127#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LEFTOVER "4"
128#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_AFTER_LOOP "5"
129#define DEPTHWISECONV_LABEL_HEIGHT_2_AFTER_LOOP "6"
130#define DEPTHWISECONV_LABEL_HEIGHT_1 "7"
131#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_2_LOOP "8"
132#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_1_LEFTOVER "9"
133#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_2_LEFTOVER "10"
134#define DEPTHWISECONV_LABEL_HEIGHT_1_END "11"

136 asm volatile(
137 // Performs depthwise convolutions for a window specified by
138 // |output_window_height| and |output_window_width|. The inner-most loop
139 // processes 2x2 outputs, and any leftovers at the end.
140 //
141 // Algorithm works as follows:
142 //
143 // 1. Load filters of 8 depth (8x3x3). Registers v0--v8 hold filter
144 // values.
145 // 2. For 2 output heights at a time:
146 // i. For 2 output widths at a time, load inputs for a 2x1 (2
147 // height, 1 width) output window (4x3 input window).
148 // Registers v9--v20 hold input values. Mul-add with
149 // accumulators v21--v24. Then run activation, downquantize
150 // and store. Repeat for the next 2x1 output window,
151 // leveraging overlapping inputs.
152 // ii. Handle single leftover width if exists.
153 // 3. Handle single leftover height if exists.
154 // i. For 2 output widths at a time, load inputs for a 1x2 (1
155 // height, 2 width) output window (3x4 input window).
156 // Registers v9--v20 hold input values. Mul-add with
157 // accumulators v21--v24. Then run activation, downquantize
158 // and store. Repeat for the next 1x2 output window,
159 // leveraging overlapping inputs.
160 // ii. Handle single leftover width if exists.
161 //
162 // Loads are placed as soon as the register is no longer needed and
163 // interleaved with arithmetic operations to take advantage of
164 // dual-issue pipelines. We also add input offsets as far from the loads
165 // as possible to give loads enough cycles to fetch data from memory.
166 //
167 // This logic is copied and modified from the non-per-channel quantized
168 // part.
169 // However, the challenges are how to plan the registers allocation
170 // wisely: 25 NEON registers are already reserved for inputs, filters,
171 // and outputs; also, 2 registers (v30, v31) are used for output
172 // min/max, while another 2 registers (v26, v29) are used for input
173 // offset & output offset, so that's total 25 + 2 + 2 = 29 already.
174 // But we need 4 more registers to hold the output multiplier & output
175 // right shift (we only have 3).
176 //
177 // So here's the plan:
178 // v27 (which held duplicated output multiplier previously) will hold
179 // the first 4 values of the output_multiplier_ptr (we have 8 in total);
180 // v30 (which held duplicated output right shift previously) will hold
181 // the first 4 values of the output_shift_ptr (we have 8 in total);
182 // lastly, v28 will hold the last 4 values of output_multiplier and v31
183 // (previously occupied by activations) will hold the last 4 values of
184 // output_shift. Then v25 will be used for output activation min while
185 // output activation max will just reuse other registers, like v24.
186 //
187 // Set "constant" registers. These registers may be replaced with temp
188 // values from time to time when there are not enough NEON registers.
189 // We use x9--x15 general purpose registers as they are caller-saved
190 // temporary registers (see
191 // http://infocenter.arm.com/help/topic/com.arm.doc.ihi0055b/IHI0055B_aapcs64.pdf). // NOLINT
192 "ldr w9, [%[params_ptr], #" STR(OFFSET_INPUT_OFFSET) "]\n"
193 "ldr x3, [%[params_ptr], #" STR(OFFSET_OUTPUT_DEPTH) "]\n"
194 "cmp %w[output_window_height], #2\n"
195 "dup v26.8h, w9\n"
196 "ldr w2, [%[params_ptr], #" STR(OFFSET_OUTPUT_OFFSET) "]\n"
197 "dup v29.8h, w2\n"
198 "ldr w4, [%[params_ptr], #" STR(OFFSET_OUTPUT_ACTIVATION_MIN) "]\n"
199 "ldr w0, [%[params_ptr], #" STR(OFFSET_OUTPUT_ACTIVATION_MAX) "]\n"
200 "add x10, %[bias_ptr], #16\n"
201 "ldr x1, [%[params_ptr], #" STR(OFFSET_OUTPUT_ROW_SIZE) "]\n"
202 "dup v25.16b, w4\n"
203
204 // Deal with output multiplier & output shift.
205 "ld1 {v27.4s, v28.4s}, [%[output_multiplier_ptr]]\n"
206 "ld1 {v30.4s, v31.4s}, [%[output_shift_ptr]]\n"
207
208 // Load filters and add offsets.
209 "ld1 {v0.8b}, [%[filter_ptr]], x3\n"
210 "ld1 {v1.8b}, [%[filter_ptr]], x3\n"
211 "sshll v0.8h, v0.8b, #0\n"
212 "ld1 {v2.8b}, [%[filter_ptr]], x3\n"
213 "sshll v1.8h, v1.8b, #0\n"
214 "ld1 {v3.8b}, [%[filter_ptr]], x3\n"
215 "sshll v2.8h, v2.8b, #0\n"
216 "ld1 {v4.8b}, [%[filter_ptr]], x3\n"
217 "sshll v3.8h, v3.8b, #0\n"
218 "ld1 {v5.8b}, [%[filter_ptr]], x3\n"
219 "sshll v4.8h, v4.8b, #0\n"
220 "ld1 {v6.8b}, [%[filter_ptr]], x3\n"
221 "sshll v5.8h, v5.8b, #0\n"
222 "ld1 {v7.8b}, [%[filter_ptr]], x3\n"
223 "sshll v6.8h, v6.8b, #0\n"
224 "ld1 {v8.8b}, [%[filter_ptr]], x3\n"
225 "sshll v7.8h, v7.8b, #0\n"
226 "sshll v8.8h, v8.8b, #0\n"
227
228 "blt " DEPTHWISECONV_LABEL_HEIGHT_2_AFTER_LOOP "f\n"
229
230 //"loop_%=:\n"
231 DEPTHWISECONV_LABEL_HEIGHT_2_LOOP ":\n"
232 // This loop processes 2x2 outputs. To avoid register exhaustion,
233 // inputs for the left 2 outputs are loaded first, then the right
234 // two outputs.
235 "mov x11, %[input_ptr]\n"
236 "mov x12, x11\n"
237 "ld1 {v9.8b}, [x12], %[input_depth]\n"
238 "add x13, x11, %[input_row_size]\n"
239 "ld1 {v10.8b}, [x12], %[input_depth]\n"
240 "add x14, x13, %[input_row_size]\n"
241 "ld1 {v11.8b}, [x12], %[input_depth]\n"
242 "add x15, x14, %[input_row_size]\n"
243 "ld1 {v12.8b}, [x13], %[input_depth]\n"
244 "mov w5, %w[output_window_width]\n"
245 "ld1 {v13.8b}, [x13], %[input_depth]\n"
246 "mov x6, %[output_ptr]\n"
247 "ld1 {v14.8b}, [x13], %[input_depth]\n"
248 "add x7, %[output_ptr], x1\n"
249 "ld1 {v15.8b}, [x14], %[input_depth]\n"
250 // The height 2 / width 2 loop loads an extra 2x1 outputs (2 height,
251 // 1 width) in anticipation for the next iteration. Make sure
252 // |output_window_width| is large enough to handle the additional
253 // loads, otherwise jump to specific the appropriate label to handle
254 // smaller widths.
255 "cmp w5, #2\n"
256 "saddw v9.8h, v26.8h, v9.8b\n"
257 "ld1 {v16.8b}, [x14], %[input_depth]\n"
258 "saddw v10.8h, v26.8h, v10.8b\n"
259 "ld1 {v17.8b}, [x14], %[input_depth]\n"
260 "saddw v11.8h, v26.8h, v11.8b\n"
261 "ld1 {v18.8b}, [x15], %[input_depth]\n"
262 "saddw v12.8h, v26.8h, v12.8b\n"
263 "ld1 {v19.8b}, [x15], %[input_depth]\n"
264 "saddw v13.8h, v26.8h, v13.8b\n"
265 "ld1 {v20.8b}, [x15], %[input_depth]\n"
266 "saddw v14.8h, v26.8h, v14.8b\n"
267 "ld1 {v21.4s}, [%[bias_ptr]]\n"
268 "saddw v15.8h, v26.8h, v15.8b\n"
269 "ld1 {v22.4s}, [x10]\n"
270 "saddw v16.8h, v26.8h, v16.8b\n"
271 "ld1 {v23.4s}, [%[bias_ptr]]\n"
272 "saddw v17.8h, v26.8h, v17.8b\n"
273 "ld1 {v24.4s}, [x10]\n"
274 "saddw v18.8h, v26.8h, v18.8b\n"
275 "saddw v19.8h, v26.8h, v19.8b\n"
276 "saddw v20.8h, v26.8h, v20.8b\n"
277
278 "beq " DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LEFTOVER "f\n"
279 "cmp w5, #1\n"
280 "beq " DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_1_LEFTOVER "f\n"
281
282 //"loop_%=:\n"
283 DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP ":\n"
284 // Mul-add left outputs.
285 "smlal v21.4s, v0.4h, v9.4h\n"
286 "subs w5, w5, #2\n"
287 "smlal2 v22.4s, v0.8h, v9.8h\n"
288 "cmp w5, #3\n"
289 "smlal v23.4s, v0.4h, v12.4h\n"
290 "ld1 {v9.8b}, [x12]\n"
291 "smlal2 v24.4s, v0.8h, v12.8h\n"
292 "smlal v21.4s, v1.4h, v10.4h\n"
293 "smlal2 v22.4s, v1.8h, v10.8h\n"
294 "smlal v23.4s, v1.4h, v13.4h\n"
295 "smlal2 v24.4s, v1.8h, v13.8h\n"
296 "smlal v21.4s, v2.4h, v11.4h\n"
297 "smlal2 v22.4s, v2.8h, v11.8h\n"
298 "smlal v23.4s, v2.4h, v14.4h\n"
299 "smlal2 v24.4s, v2.8h, v14.8h\n"
300 "smlal v21.4s, v3.4h, v12.4h\n"
301 "smlal2 v22.4s, v3.8h, v12.8h\n"
302 "ld1 {v12.8b}, [x13]\n"
303 "smlal v23.4s, v3.4h, v15.4h\n"
304 "smlal2 v24.4s, v3.8h, v15.8h\n"
305 "smlal v21.4s, v4.4h, v13.4h\n"
306 "smlal2 v22.4s, v4.8h, v13.8h\n"
307 "smlal v23.4s, v4.4h, v16.4h\n"
308 "smlal2 v24.4s, v4.8h, v16.8h\n"
309 "smlal v21.4s, v5.4h, v14.4h\n"
310 "smlal2 v22.4s, v5.8h, v14.8h\n"
311 "smlal v23.4s, v5.4h, v17.4h\n"
312 "smlal2 v24.4s, v5.8h, v17.8h\n"
313 "smlal v21.4s, v6.4h, v15.4h\n"

317 "smlal2 v24.4s, v6.8h, v18.8h\n"
318 "ld1 {v18.8b}, [x15]\n"
319 "smlal v21.4s, v7.4h, v16.4h\n"

353 "st1 {v23.8b}, [x7], x3\n"
354 "saddw v15.8h, v26.8h, v15.8b\n"
355 "ld1 {v21.4s}, [%[bias_ptr]]\n"
356 "saddw v18.8h, v26.8h, v18.8b\n"
357 "ld1 {v23.4s}, [%[bias_ptr]]\n"
358
359 // Mul-add right outputs.
360 "smlal v21.4s, v0.4h, v10.4h\n"
361 "add x11, x11, %[input_width_increment]\n"
362 "smlal2 v22.4s, v0.8h, v10.8h\n"
363 "mov x12, x11\n"
364 "smlal v23.4s, v0.4h, v13.4h\n"
365 "add x13, x11, %[input_row_size]\n"
366 "smlal2 v24.4s, v0.8h, v13.8h\n"
367 "add x14, x13, %[input_row_size]\n"
368 "smlal v21.4s, v1.4h, v11.4h\n"
369 "add x15, x14, %[input_row_size]\n"
370 "smlal2 v22.4s, v1.8h, v11.8h\n"
371 "smlal v23.4s, v1.4h, v14.4h\n"

974 const int64_t input_width_increment = 4 * input_depth;
975 const int64_t input_height_increment = 4 * input_row_size;
976 const int64_t output_height_increment = 2 * params_ptr->output_row_size;
977 TFLITE_DCHECK_EQ(params_ptr->filter_offset, 0);

980#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP "2"

989#define DEPTHWISECONV_LABEL_HEIGHT_1_END "11"

992 // Performs depthwise convolutions for a window specified by
993 // |output_window_height| and |output_window_width|. The inner-most loop
994 // processes 2x2 outputs, and any leftovers at the end.
995 //
996 // Algorithm works as follows:
997 //
998 // 1. Load filters of 8 depth (8x3x3). Registers v0--v8 hold filter
999 // values.
1000 // 2. For 2 output heights at a time:
1001 // i. For 2 output widths at a time at stride 2, a 5x5 input
1002 // window is required. To avoid register exhaustion, we load
1003 // the first 2 rows of the 5x5 input window into registers
1004 // v9--v18, and use the same registers to load the next 2
1005 // rows, and finally v9--v13 to load the last row.
1006 // Accumulators for all 2x2 outputs are reserved by registers
1007 // v21-v22 (top left output), v23-v24 (top right output),
1008 // v19-v20 (bottom left output), v25-v26 (bottom right
1009 // output).
1010 // ii. Handle single leftover width if exists.
1011 // 3. Handle single leftover height if exists.
1012 // i. For 2 output widths at a time at stride 2, load inputs for
1013 // a 1x2 (1 height, 2 width) output window (3x5 input
1014 // window). Registers v9--v24 hold input values. Mul-add with
1015 // accumulators v24--v27.
1016 // ii. Handle single leftover width if exists.
1017 //
1018 // Loads are placed as soon as the register is no longer needed and
1019 // interleaved with arithmetic operations to take advantage of
1020 // dual-issue pipelines. We also add input offsets as far from the loads
1021 // as possible to give loads enough cycles to fetch data from memory.
1022 //
1023 // This logic is copied and modified from the non-per-channel quantized
1024 // part.
1025 // The register planning here is really tricky:
1026 // v0-v29 are all used at least once for either filter/input/output,
1027 // some of them are used for output shift and output multiplier, or
1028 // input/output offset.
1029 // Only v30 & v31 are only used for output activation min/max.
1030 // For per-channel case, we need 4 registers to hold output shift &
1031 // output multiplier. However, given the reality, we simply cannot do
1032 // that without reloading.
1033 //
1034 // So here's the plan:
1035 // We hold output_multiplier in v30 & v31, and we will load output_shift
1036 // into two consecutive registers each time before use.
1037 // We will duplicate output min & max before needed.
1038 // Sometimes we may borrow registers from input offset or bias, we will
1039 // dup them back after use.
1040 //
1041
1042 // Set "constant" registers. These registers may be replaced with temp
1043 // values from time to time when there are not enough NEON registers.
1044 // We use x9--x15 general purpose registers as they are caller-saved
1045 // temporary registers (see http://infocenter.arm.com/help/topic/com.arm.doc.ihi0055b/IHI0055B_aapcs64.pdf). // NOLINT
1046 "ldr w0, [%[params_ptr], #" STR(OFFSET_INPUT_OFFSET) "]\n"
1047 "cmp %w[output_window_height], #2\n"
1048 "dup v28.8h, w0\n"
1049 "ldr w2, [%[params_ptr], #" STR(OFFSET_OUTPUT_OFFSET) "]\n"
1050 "ldr w3, [%[params_ptr], #" STR(OFFSET_OUTPUT_ACTIVATION_MIN) "]\n"
1051 "dup v29.8h, w2\n"
1052 "ldr w4, [%[params_ptr], #" STR(OFFSET_OUTPUT_ACTIVATION_MAX) "]\n"
1053 "ldr x5, [%[params_ptr], #" STR(OFFSET_OUTPUT_DEPTH) "]\n"
1054 "ldr x19, [%[params_ptr], #" STR(OFFSET_OUTPUT_ROW_SIZE) "]\n"
1055
1056 // Deal with output multiplier.
1057 "ld1 {v30.4s, v31.4s}, [%[output_multiplier_ptr]]\n"

1059 // Load filters and add offsets.
1060 "add x10, %[bias_ptr], #16\n"
1061 "ld1 {v0.8b}, [%[filter_ptr]], x5\n"
1062 "ld1 {v1.8b}, [%[filter_ptr]], x5\n"

1064 "ld1 {v2.8b}, [%[filter_ptr]], x5\n"
1065 "sshll v1.8h, v1.8b, #0\n"

1067 "sshll v2.8h, v2.8b, #0\n"
1068 "ld1 {v4.8b}, [%[filter_ptr]], x5\n"
1069 "sshll v3.8h, v3.8b, #0\n"
1070 "ld1 {v5.8b}, [%[filter_ptr]], x5\n"
1071 "sshll v4.8h, v4.8b, #0\n"
1072 "ld1 {v6.8b}, [%[filter_ptr]], x5\n"
1073 "sshll v5.8h, v5.8b, #0\n"
1074 "ld1 {v7.8b}, [%[filter_ptr]], x5\n"
1075 "sshll v6.8h, v6.8b, #0\n"
1076 "ld1 {v8.8b}, [%[filter_ptr]]\n"
1077 "sshll v7.8h, v7.8b, #0\n"
1078 "sshll v8.8h, v8.8b, #0\n"
1079
1080 "blt " DEPTHWISECONV_LABEL_HEIGHT_2_AFTER_LOOP "f\n"

1082 //"loop_%=:\n"
1083 DEPTHWISECONV_LABEL_HEIGHT_2_LOOP ":\n"
1084 // Load the first two rows of the 5x5 input window, then reuse the
1085 // same registers to load subsequent rows as they become available.
1086 "mov x11, %[input_ptr]\n"
1087 "mov x12, x11\n"
1088 "add x13, x12, %[input_row_size]\n"
1089 "ld1 {v9.8b}, [x12], %[input_depth]\n"
1090 "mov w14, %w[output_window_width]\n"
1091 "ld1 {v10.8b}, [x12], %[input_depth]\n"
1092 // The height 2 / width 2 loop loads an extra 1 output horizontally in
1093 // anticipation for the next iteration. Make sure
1094 // |output_window_width| is large enough to handle the additional
1095 // load, otherwise jump to the appropriate label to handle smaller
1096 // widths.
1097 "cmp w14, #2\n"

1102 "ld1 {v15.8b}, [x13], %[input_depth]\n"

1104 "ld1 {v16.8b}, [x13], %[input_depth]\n"

2659 const int32* output_shift, const int8* input_data,

2665 TFLITE_DCHECK(
2666 shuffle_params.input_height ==

2668 TFLITE_DCHECK(
2669 shuffle_params.input_width ==
2670 get_shuffle_input_size(kStrideWidth, shuffle_params.output_width));
2671 TFLITE_DCHECK_LE(
2672 64 * shuffle_params.input_width * shuffle_params.input_height,
2673 kDepthwiseConvScratchWorkspaceSize);

2677 // Run shuffling on inputs with sufficiently large depth and width. When
2678 // these parameters are large enough, more time is taken to load inputs
2679 // from memory. At this point, it becomes useful to prefetch and
2680 // preshuffle the input data to maximize locality.
2681 if (params.output_depth > 64 ||
2682 (params.output_depth <= 64 && params.input_width > 150)) {
2683 for (; out_x <= (end_x - shuffle_params.output_width);
2684 out_x += shuffle_params.output_width) {
2685 const int8* input_ptr = input_data;

2688 const int32* output_shift_ptr = output_shift;

2690 int8* output_ptr = output_data;

2692 const int64_t shuffle_row_size = 64 * shuffle_params.input_width;
2693
2694 for (; depth <= params.output_depth - 64; depth += 64) {
2695 // Preload.
2696 const int8* h_ptr = input_ptr;
2697 for (int32 i = 0; i < shuffle_params.input_height; i++) {
2698 const int8* ptr = h_ptr;
2699 for (int32 j = 0; j < shuffle_params.input_width; j++) {
2700 optimized_ops_preload_l1_keep(ptr);
2701 ptr += params.input_depth;
2702 }
2703 h_ptr += params.input_row_size;
2704 }
2705
2706 // For a large enough input, shuffle into buckets.
2707 ShuffleInput(input_ptr, params.input_depth, params.input_width,
2708 params.input_height, 64, shuffle_params.input_width,
2709 shuffle_params.input_height, shuffle_workspace);
2710 ConvKernel::Run(output_multiplier_ptr, output_shift_ptr,
2711 shuffle_workspace, filter_ptr, bias_ptr, output_ptr,
2712 0, 64, 64, shuffle_row_size,
2713 shuffle_params.output_height,
2714 shuffle_params.output_width, params);
2715 input_ptr += 64;
2716 output_ptr += 64;
2717 filter_ptr += 64;
2718 bias_ptr += 64;
2719 output_multiplier_ptr += 64;
2720 output_shift_ptr += 64;
2721 }
2722
2723 // Preload.
2724 const int8* h_ptr = input_ptr;
2725 for (int32 i = 0; i < shuffle_params.input_height; i++) {
2726 const int8* ptr = h_ptr;
2727 for (int32 j = 0; j < shuffle_params.input_width; j++) {
2728 optimized_ops_preload_l1_keep(ptr);
2729 ptr += params.input_depth;
2730 }
2731 h_ptr += params.input_row_size;
2732 }
2733
2734 // Handle leftover depth.
2735 ConvKernel::Run(output_multiplier_ptr, output_shift_ptr, input_ptr,
2736 filter_ptr, bias_ptr, output_ptr, depth,
2737 params.output_depth, params.input_depth,
2738 params.input_row_size, shuffle_params.output_height,
2739 shuffle_params.output_width, params);

2741 input_data +=
2742 shuffle_params.output_width * kStrideWidth * params.input_depth;
2743 output_data += shuffle_params.output_width * params.output_depth;
2744 }
2745 }

2748 if (output_leftover_width > 0) {

2750 bias_data, output_data, 0, params.output_depth,
2751 params.input_depth, params.input_row_size,
2752 shuffle_params.output_height, output_leftover_width,
2753 params);
2754 }
2755 }
