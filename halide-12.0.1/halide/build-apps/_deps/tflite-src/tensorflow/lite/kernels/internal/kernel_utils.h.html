23// Performs an RNN batch inference step for inputs specified by input_ptr_batch.
24// The RNN cell is specified by the pointers to its input and recurrent weights,
25// and biases, along with the input size, number of units, activation.
26//
27// The pointers to the hidden state and the output are updated as a result.
28//
29// The pointers with the suffix "_batch" point to data aligned in batch_major
30// order, and each step processes batch_size many inputs from input_ptr_batch,
31// and updates batch_size many outputs and hidden states.
32//
33// The output_batch_dim is output.shape[-1], i.e. the outermost dimension of the
34// output tensor, and in most cases will be equal to num_units. It is usually
35// not when we want to store the RNN output into a slice of the output tensor,
36// e.g. for bidirectional RNNs with merge_outputs. In this case, the batched
37// operations cannot be used since they assume that the batched outputs are
38// contiguous, and we manually loop over the batched outputs.
39void RnnBatchStep(const float* input_ptr_batch, const float* input_weights_ptr,
40 const float* recurrent_weights_ptr, const float* bias_ptr,
41 int input_size, int num_units, int batch_size,
42 int output_batch_leading_dim,
43 TfLiteFusedActivation activation,
44 float* hidden_state_ptr_batch, float* output_ptr_batch);
45
46// Same as above but includes an auxiliary input with the corresponding weights.
47void RnnBatchStep(const float* input_ptr_batch, const float* input_weights_ptr,
48 const float* aux_input_ptr_batch,
49 const float* aux_input_weights_ptr,
50 const float* recurrent_weights_ptr, const float* bias_ptr,
51 int input_size, int aux_input_size, int num_units,
52 int batch_size, int output_batch_leading_dim,
53 TfLiteFusedActivation activation,
54 float* hidden_state_ptr_batch, float* output_ptr_batch);
55
56// Performs a quantized RNN batch inference step. Same as above, but for
57// quantization purposes, we also pass in quantized_hidden_state_ptr_batch and
58// quantized_input_ptr_batch pointers for temporary storage of the quantized
59// values of hidden_state_ptr_batch and input_ptr_batch, respectively.
60// These temporary storages are expected to be preallocated to the same size as
61// the respective pointers.
62// An additional preallocated temporary storage 'scaling_factors' (of size
63// batch_size) is used to store the scaling factors of the quantization (used
64// for recovery).
65// {input,recurrent}_weights_scale params are used for dequantization/recovery.
66void RnnBatchStep(
67 const float* input_ptr_batch, const int8_t* input_weights_ptr,
68 float input_weights_scale, const int8_t* recurrent_weights_ptr,
69 float recurrent_weights_scale, const float* bias_ptr, int input_size,
70 int num_units, int batch_size, int output_batch_leading_dim,
71 TfLiteFusedActivation activation, int8_t* quantized_input_ptr_batch,
72 int8_t* quantized_hidden_state_ptr_batch, float* scaling_factors,
73 float* hidden_state_ptr_batch, float* output_ptr_batch,
74 bool asymmetric_quantize_inputs, int32_t* zero_points,
75 int32_t* accum_scratch, int32_t* row_sums, bool* compute_row_sums);
76
77void RnnBatchStep(
78 const float* input_ptr_batch, const int8_t* input_weights_ptr,
79 float input_weights_scale, const float* aux_input_ptr_batch,
80 const int8_t* aux_input_weights_ptr, float aux_input_weights_scale,
81 const int8_t* recurrent_weights_ptr, float recurrent_weights_scale,
82 const float* bias_ptr, int input_size, int aux_input_size, int num_units,
83 int batch_size, int output_batch_leading_dim,
84 TfLiteFusedActivation activation, int8_t* quantized_input_ptr_batch,
85 int8_t* aux_quantized_input_ptr_batch,

88 bool asymmetric_quantize_inputs, int32_t* zero_points,
89 int32_t* accum_scratch, int32_t* row_sums, bool* compute_row_sums);
