30constexpr int kInputTensor = 0;
31constexpr int kShapeTensor = 1;
32constexpr int kOutputTensor = 0;
33
34TfLiteIntArray* GetOutputShape(TfLiteContext*, TfLiteNode*);
35
36TfLiteStatus ResizeOutput(TfLiteContext* context, TfLiteNode* node) {
37 TfLiteIntArray* output_shape = GetOutputShape(context, node);
38 std::unique_ptr<TfLiteIntArray, void (*)(TfLiteIntArray*)>
39 scoped_output_shape(output_shape, TfLiteIntArrayFree);
40
41 const TfLiteTensor* input;
42 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
43 TfLiteTensor* output;
44 TF_LITE_ENSURE_OK(context,
45 GetOutputSafe(context, node, kOutputTensor, &output));
46
47 // Tensorflow's Reshape allows one of the shape components to have the
48 // special -1 value, meaning it will be calculated automatically based on the
49 // input. Here we calculate what that dimension should be so that the number
50 // of output elements is the same as the number of input elements.
51 int num_input_elements = NumElements(input);
52
53 int num_output_elements = 1;
54 int stretch_dim = -1;
55 for (int i = 0; i < output_shape->size; ++i) {
56 int value = output_shape->data[i];
57 if (value == -1) {
58 TF_LITE_ENSURE_EQ(context, stretch_dim, -1);
59 stretch_dim = i;
60 } else {
61 num_output_elements *= value;
62 }
63 }
64 if (stretch_dim != -1) {
65 output_shape->data[stretch_dim] = num_input_elements / num_output_elements;
66 num_output_elements *= output_shape->data[stretch_dim];
67 }
68
69 TF_LITE_ENSURE_EQ(context, num_input_elements, num_output_elements);
70 return context->ResizeTensor(context, output, scoped_output_shape.release());
71}

73inline TfLiteIntArray* GetOutputShapeFromTensor(TfLiteContext* context,
74 TfLiteNode* node) {
75 const TfLiteTensor* shape = GetInput(context, node, kShapeTensor);
76 if (shape == nullptr) return nullptr;
77
78 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(shape->dims->data[0]);
79 for (int i = 0; i < output_shape->size; ++i) {
80 output_shape->data[i] = shape->data.i32[i];
81 }
82
83 return output_shape;
84}
85
86inline TfLiteIntArray* GetOutputShapeFromParam(TfLiteContext* context,
87 TfLiteNode* node) {
88 auto* params = reinterpret_cast<TfLiteReshapeParams*>(node->builtin_data);
89
90 // The function is returned above this line if the shape tensor is usable.
91 // Now fallback to the shape parameter in `TfLiteReshapeParams`.
92 int num_dimensions = params->num_dimensions;
93 if (num_dimensions == 1 && params->shape[0] == 0) {
94 // Legacy tflite models use a shape parameter of [0] to indicate scalars,
95 // so adjust accordingly. TODO(b/111614235): Allow zero-sized buffers during
96 // toco conversion.
97 num_dimensions = 0;
98 }
99 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(num_dimensions);
100 for (int i = 0; i < num_dimensions; ++i) {
101 output_shape->data[i] = params->shape[i];
102 }
103
104 return output_shape;
105}

107// Check if the shape tensor is valid. Shapes should be int32 vectors.
108inline bool ShapeIsVector(TfLiteContext* context, TfLiteNode* node) {
109 const TfLiteTensor* shape = GetInput(context, node, kShapeTensor);
110 return (shape != nullptr && shape->dims->size == 1 &&
111 shape->type == kTfLiteInt32);
112}
113
114TfLiteIntArray* GetOutputShape(TfLiteContext* context, TfLiteNode* node) {
115 if (NumInputs(node) == 2 && ShapeIsVector(context, node)) {
116 return GetOutputShapeFromTensor(context, node);
117 } else {
118 return GetOutputShapeFromParam(context, node);
119 }
120}

122TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
123 TF_LITE_ENSURE(context, NumInputs(node) == 1 || NumInputs(node) == 2);
124 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
125
126 // Always postpone sizing string tensors, even if we could in principle
127 // calculate their shapes now. String tensors don't benefit from having their
128 // shapes precalculated because the actual memory can only be allocated after
129 // we know all the content.
130 TfLiteTensor* output;
131 TF_LITE_ENSURE_OK(context,
132 GetOutputSafe(context, node, kOutputTensor, &output));
133 if (output->type != kTfLiteString) {
134 if (NumInputs(node) == 1 ||
135 IsConstantTensor(GetInput(context, node, kShapeTensor))) {
136 TF_LITE_ENSURE_OK(context, ResizeOutput(context, node));
137 } else {
138 SetTensorToDynamic(output);
139 }
140 }
141 return kTfLiteOk;
142}
143
144TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
145 const TfLiteTensor* input;
146 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
147 TfLiteTensor* output;
148 TF_LITE_ENSURE_OK(context,
149 GetOutputSafe(context, node, kOutputTensor, &output));
150
151 // There are two ways in which the 'output' can be made dynamic: it could be
152 // a string tensor, or its shape cannot be calculated during Prepare(). In
153 // either case, we now have all the information to calculate its shape.
154 if (IsDynamicTensor(output)) {
155 TF_LITE_ENSURE_OK(context, ResizeOutput(context, node));
156 }
157
158 // Note that string tensors are always "dynamic" in the sense that their size
159 // is not known until we have all the content. This applies even when their
160 // shape is known ahead of time. As a result, a string tensor is never given
161 // any memory by ResizeOutput(), and we need to do it manually here. Since
162 // reshape doesn't change the data, the output tensor needs exactly as many
163 // bytes as the input tensor.
164 if (output->type == kTfLiteString) {
165 auto bytes_required = input->bytes;
166 TfLiteTensorRealloc(bytes_required, output);
167 output->bytes = bytes_required;
168 }
169
170 memcpy(output->data.raw, input->data.raw, input->bytes);
171
172 return kTfLiteOk;
173}

179 reshape::Eval};
