27inline int32_t GetNearestNeighbor(const int input_value,
28 const int32_t input_size,
29 const int32_t output_size,
30 const bool align_corners,
31 const bool half_pixel_centers) {
32 const float scale =
33 (align_corners && output_size > 1)
34 ? (input_size - 1) / static_cast<float>(output_size - 1)
35 : input_size / static_cast<float>(output_size);
36 const float offset = half_pixel_centers ? 0.5f : 0.0f;
37 int32_t output_value = std::min(
38 align_corners
39 ? static_cast<int32_t>(TfLiteRound((input_value + offset) * scale))
40 : static_cast<int32_t>(std::floor((input_value + offset) * scale)),
41 input_size - 1);
42 if (half_pixel_centers) {
43 output_value = std::max(static_cast<int32_t>(0), output_value);
44 }
45 return output_value;
46}

48template <typename T>
49inline void ResizeNearestNeighbor(
50 const tflite::ResizeNearestNeighborParams& op_params,
51 const RuntimeShape& unextended_input_shape, const T* input_data,
52 const RuntimeShape& output_size_shape, const int32_t* output_size_data,
53 const RuntimeShape& unextended_output_shape, T* output_data) {
54 TFLITE_DCHECK_LE(unextended_input_shape.DimensionsCount(), 4);
55 TFLITE_DCHECK_LE(unextended_output_shape.DimensionsCount(), 4);
56
57 const RuntimeShape input_shape =
58 RuntimeShape::ExtendedShape(4, unextended_input_shape);
59 const RuntimeShape output_shape =
60 RuntimeShape::ExtendedShape(4, unextended_output_shape);
61
62 int32_t batches = MatchingDim(input_shape, 0, output_shape, 0);
63 int32_t input_height = input_shape.Dims(1);
64 int32_t input_width = input_shape.Dims(2);
65 int32_t depth = MatchingDim(input_shape, 3, output_shape, 3);
66
67 // The Tensorflow version of this op allows resize on the width and height
68 // axis only.
69 TFLITE_DCHECK_EQ(output_size_shape.FlatSize(), 2);
70 int32_t output_height = output_size_data[0];
71 int32_t output_width = output_size_data[1];
72
73 const int col_offset = input_shape.Dims(3);
74 const int row_offset = input_shape.Dims(2) * col_offset;
75 const int batch_offset = input_shape.Dims(1) * row_offset;
76
77 const T* input_ptr = input_data;
78 T* output_ptr = output_data;
79 for (int b = 0; b < batches; ++b) {
80 for (int y = 0; y < output_height; ++y) {
81 int32_t in_y = GetNearestNeighbor(y, input_height, output_height,
82 op_params.align_corners,
83 op_params.half_pixel_centers);
84 const T* y_input_ptr = input_ptr + in_y * row_offset;
85 for (int x = 0; x < output_width; ++x) {
86 int32_t in_x = GetNearestNeighbor(x, input_width, output_width,
87 op_params.align_corners,
88 op_params.half_pixel_centers);
89 const T* x_input_ptr = y_input_ptr + in_x * col_offset;
90 memcpy(output_ptr, x_input_ptr, depth * sizeof(T));
91 output_ptr += depth;
92 }
93 }
94 input_ptr += batch_offset;
95 }
96}
