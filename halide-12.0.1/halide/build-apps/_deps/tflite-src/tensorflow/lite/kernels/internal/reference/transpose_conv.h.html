25inline void TransposeConv(
26 const ConvParams& params, const RuntimeShape& input_shape,
27 const float* input_data, const RuntimeShape& filter_shape,
28 const float* filter_data, const RuntimeShape& bias_shape,
29 const float* bias_data, const RuntimeShape& output_shape,
30 float* output_data, const RuntimeShape& im2col_shape, float* im2col_data) {
31 const int stride_width = params.stride_width;
32 const int stride_height = params.stride_height;
33 const int pad_width = params.padding_values.width;
34 const int pad_height = params.padding_values.height;
35 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
36 TFLITE_DCHECK_EQ(filter_shape.DimensionsCount(), 4);
37 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
38 (void)im2col_data; // only used in optimized code.
39 (void)im2col_shape; // only used in optimized code.
40
41 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
42 const int input_depth = MatchingDim(input_shape, 3, filter_shape, 3);
43 const int output_depth = MatchingDim(filter_shape, 0, output_shape, 3);
44 const int input_height = input_shape.Dims(1);
45 const int input_width = input_shape.Dims(2);
46 const int filter_height = filter_shape.Dims(1);
47 const int filter_width = filter_shape.Dims(2);
48 const int output_height = output_shape.Dims(1);
49 const int output_width = output_shape.Dims(2);
50 if (bias_data) {
51 TFLITE_DCHECK_EQ(bias_shape.FlatSize(), output_depth);
52 }
53
54 // Although transpose convolution simplifies to convolution with transposed
55 // weights for strides of 1, non-unitary striding complicates matters. To
56 // keep this reference implementation as clear as possible, we use a
57 // "scatter" access pattern, where we loop through all the input elements,
58 // computing their influence on the output, rather than looping through the
59 // output elements in the typical "gather" access pattern of a conv. We
60 // therefore must initialize the output array to zero.
61 const int num_elements = output_shape.FlatSize();
62 for (int i = 0; i < num_elements; i++) {
63 output_data[i] = 0.0f;
64 }
65
66 // Loop through input elements one at a time.
67 for (int batch = 0; batch < batches; ++batch) {
68 for (int in_y = 0; in_y < input_height; ++in_y) {
69 for (int in_x = 0; in_x < input_width; ++in_x) {
70 for (int in_channel = 0; in_channel < input_depth; ++in_channel) {
71 // Loop through the output elements it will influence
72 const int out_x_origin = (in_x * stride_width) - pad_width;
73 const int out_y_origin = (in_y * stride_height) - pad_height;
74 for (int filter_y = 0; filter_y < filter_height; ++filter_y) {
75 for (int filter_x = 0; filter_x < filter_width; ++filter_x) {
76 for (int out_channel = 0; out_channel < output_depth;
77 ++out_channel) {
78 // Compute output element location
79 const int out_x = out_x_origin + filter_x;
80 const int out_y = out_y_origin + filter_y;
81 // We cannot accumulate out of bounds
82 if ((out_x >= 0) && (out_x < output_width) && (out_y >= 0) &&
83 (out_y < output_height)) {
84 float input_value = input_data[Offset(
85 input_shape, batch, in_y, in_x, in_channel)];
86 float filter_value =
87 filter_data[Offset(filter_shape, out_channel, filter_y,
88 filter_x, in_channel)];
89 output_data[Offset(output_shape, batch, out_y, out_x,
90 out_channel)] +=
91 input_value * filter_value;
92 }
93 }
94 }
95 }
96 }
97 }
98 }
99 }
100 if (bias_data) {
101 for (int batch = 0; batch < batches; ++batch) {
102 for (int out_y = 0; out_y < output_height; ++out_y) {
103 for (int out_x = 0; out_x < output_width; ++out_x) {
104 for (int out_channel = 0; out_channel < output_depth; ++out_channel) {
105 output_data[Offset(output_shape, batch, out_y, out_x,
106 out_channel)] += bias_data[out_channel];
107 }
108 }
109 }
110 }
111 }
112}

114inline void TransposeConv(
115 const ConvParams& params, const RuntimeShape& input_shape,
116 const uint8_t* input_data, const RuntimeShape& filter_shape,
117 const uint8_t* filter_data, const RuntimeShape& bias_shape,
118 const int32_t* bias_data, const RuntimeShape& output_shape,
119 uint8_t* output_data, const RuntimeShape& im2col_shape,
120 uint8_t* im2col_data, int32_t* scratch_buffer) {
121 const int stride_width = params.stride_width;
122 const int stride_height = params.stride_height;
123 const int pad_width = params.padding_values.width;
124 const int pad_height = params.padding_values.height;
125 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
126 TFLITE_DCHECK_EQ(filter_shape.DimensionsCount(), 4);
127 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
128 (void)im2col_data; // only used in optimized code.
129 (void)im2col_shape; // only used in optimized code.
130
131 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
132 const int input_depth = MatchingDim(input_shape, 3, filter_shape, 3);
133 const int output_depth = MatchingDim(filter_shape, 0, output_shape, 3);
134 const int input_height = input_shape.Dims(1);
135 const int input_width = input_shape.Dims(2);
136 const int filter_height = filter_shape.Dims(1);
137 const int filter_width = filter_shape.Dims(2);
138 const int output_height = output_shape.Dims(1);
139 const int output_width = output_shape.Dims(2);
140 const int32_t input_offset = params.input_offset;
141 const int32_t filter_offset = params.weights_offset;
142 const int32_t output_offset = params.output_offset;
143 const int32_t output_multiplier = params.output_multiplier;
144 const int output_shift = params.output_shift;
145 const int32_t output_activation_min = params.quantized_activation_min;
146 const int32_t output_activation_max = params.quantized_activation_max;
147 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
148 if (bias_data) {
149 TFLITE_DCHECK_EQ(bias_shape.FlatSize(), output_depth);
150 }
151
152 const int num_elements = output_shape.FlatSize();
153 // We need to initialize scratch_buffer to all 0s, as we apply the same
154 // 'scatter' based trick as in float version.
155 memset(scratch_buffer, 0, num_elements * sizeof(int32_t));
156
157 // Loop through input elements one at a time.
158 for (int batch = 0; batch < batches; ++batch) {
159 for (int in_y = 0; in_y < input_height; ++in_y) {
160 for (int in_x = 0; in_x < input_width; ++in_x) {
161 for (int in_channel = 0; in_channel < input_depth; ++in_channel) {
162 // Loop through the output elements it will influence.
163 const int out_x_origin = (in_x * stride_width) - pad_width;
164 const int out_y_origin = (in_y * stride_height) - pad_height;
165 for (int filter_y = 0; filter_y < filter_height; ++filter_y) {
166 for (int filter_x = 0; filter_x < filter_width; ++filter_x) {
167 for (int out_channel = 0; out_channel < output_depth;
168 ++out_channel) {
169 // Compute output element location.
170 const int out_x = out_x_origin + filter_x;
171 const int out_y = out_y_origin + filter_y;
172 // We cannot accumulate out of bounds.
173 if ((out_x >= 0) && (out_x < output_width) && (out_y >= 0) &&
174 (out_y < output_height)) {
175 uint8_t input_value = input_data[Offset(
176 input_shape, batch, in_y, in_x, in_channel)];
177 uint8_t filter_value =
178 filter_data[Offset(filter_shape, out_channel, filter_y,
179 filter_x, in_channel)];
180 scratch_buffer[Offset(output_shape, batch, out_y, out_x,
181 out_channel)] +=
182 (input_value + input_offset) *
183 (filter_value + filter_offset);
184 }
185 }
186 }
187 }
188 }
189 }
190 }
191 }
192 for (int batch = 0; batch < batches; ++batch) {
193 for (int out_y = 0; out_y < output_height; ++out_y) {
194 for (int out_x = 0; out_x < output_width; ++out_x) {
195 for (int out_channel = 0; out_channel < output_depth; ++out_channel) {
196 int32_t acc = scratch_buffer[Offset(output_shape, batch, out_y, out_x,
197 out_channel)];
198 if (bias_data) {
199 acc += bias_data[out_channel];
200 }
201 int32_t scaled_acc = MultiplyByQuantizedMultiplier(
202 acc, output_multiplier, output_shift);
203 scaled_acc += output_offset;
204 scaled_acc = std::max(scaled_acc, output_activation_min);
205 scaled_acc = std::min(scaled_acc, output_activation_max);
206 output_data[Offset(output_shape, batch, out_y, out_x, out_channel)] =
207 static_cast<uint8_t>(scaled_acc);
208 }
209 }
210 }
211 }
