26inline void Tanh(int32_t input_zero_point, int32_t input_range_radius,
27 int32_t input_multiplier, int32_t input_shift,
28 const RuntimeShape& input_shape, const int8_t* input_data,
29 const RuntimeShape& output_shape, int8_t* output_data) {
30 // Integer bits must be in sync with Prepare() function.
31 static constexpr int32_t kInputIntegerBits = 4;
32 static constexpr int32_t kOutputScale = 7;
33 static constexpr int32_t kMinInt8 = std::numeric_limits<int8_t>::min();
34 static constexpr int32_t kMaxInt8 = std::numeric_limits<int8_t>::max();
35 using F4 = gemmlowp::FixedPoint<int32_t, kInputIntegerBits>;
36
37 const int flat_size = MatchingFlatSize(input_shape, output_shape);
38
39 for (int i = 0; i < flat_size; ++i) {
40 const int32_t input =
41 static_cast<int32_t>(input_data[i]) - input_zero_point;
42 if (input <= -input_range_radius) {
43 output_data[i] = kMinInt8;
44 } else if (input >= input_range_radius) {
45 output_data[i] = kMaxInt8;
46 } else {
47 const int32_t input_in_q4 =
48 MultiplyByQuantizedMultiplier(input, input_multiplier, input_shift);
49 const int32_t output_in_q0 =
50 gemmlowp::tanh(F4::FromRaw(input_in_q4)).raw();
51
52 // Rescale and downcast.
53 using gemmlowp::RoundingDivideByPOT;
54 int32_t output_in_q24 =
55 RoundingDivideByPOT(output_in_q0, 31 - kOutputScale);
56 output_in_q24 = std::min(std::max(output_in_q24, kMinInt8), kMaxInt8);
57 output_data[i] = static_cast<int8_t>(output_in_q24);
58 }
59 }
60}

62inline void Tanh(int32_t input_multiplier, int32_t input_left_shift,
63 const RuntimeShape& input_shape, const int16_t* ptr_input_data,
64 const RuntimeShape& output_shape, int16_t* ptr_output_data) {
65 // We use the LUT for sigmoid and take into account, that
66 // tanh(x) = 2*sigmoid(2*x) - 1
67
68 // We scale by 3/4 to expand range [-8,8]->[-10.7,10.7].
69 // In case of general parameter scale, multiplier 3 is taken into account
70 // in TanhPrepare function and it is included in
71 // input_multiplier already.
72
73 if (input_multiplier == 0) { // power of two case
74 input_multiplier = 3 << input_left_shift;
75 input_left_shift = 0;
76 }
77
78 int32_t round = (input_left_shift > 0) ? 1 << (input_left_shift - 1) : 0;
79
80 int flat_size = MatchingFlatSize(input_shape, output_shape);
81
82 for (int i = 0; i < flat_size; ++i, ptr_input_data++, ptr_output_data++) {
83 int32_t input_data =
84 ((*ptr_input_data) * input_multiplier + round) >> input_left_shift;
85
86 uint32_t abs_input_data = abs(input_data);
87 uint32_t uh = abs_input_data >> 8;
88 int32_t result;
89
90 if (uh >= 255) {
91 // Saturate to maximum.
92 result = 0xFFFF << 8;
93 } else {
94 uint32_t ua = sigmoid_table_uint16[uh];
95 uint32_t ub = sigmoid_table_uint16[uh + 1];
96
97 uint8_t ut = abs_input_data & 0xFF;
98
99 result = (ua << 8) + ut * (ub - ua);
100 }
101
102 result = (input_data >= 0)
103 ? (result - (1 << (14 + 9)) + (1 << (9 - 2)))
104 : (-result + (1 << (14 + 9)) + (1 << (9 - 2)) - 1);
105
106 // Convert back to 16-bit.
107 result >>= (9 - 1);
108
109 *ptr_output_data = result;
110 }
111}
