32// CallOnce operator is a control flow op to invoke other subgraph in the graph
33// in order to conduct the given graph's initialization tasks, for example, hash
34// table initialization and variable initialization.
35//
36// This operator will invoke the subgraph for initialization in the first run
37// and become no-op after the first run in an interpreter's life cycle.
38
39struct OpData {
40 // Subgraph index to be invoked once in a life cycle by this CallOnce op.
41 int init_subgraph_index;
42 // Boolean storage to store whether the subgraph for initialization is invoked
43 // successfully once in an interpreter's life cycle.
44 bool init_subgraph_invoked;
45};
46
47void* Init(TfLiteContext* context, const char* buffer, size_t length) {
48 auto* op_data = new OpData;
49 const auto* params = reinterpret_cast<const TfLiteCallOnceParams*>(buffer);
50 op_data->init_subgraph_index = params->init_subgraph_index;
51 op_data->init_subgraph_invoked = false;
52 return op_data;
53}
54
55void Free(TfLiteContext* context, void* buffer) {
56 delete reinterpret_cast<OpData*>(buffer);
57}
58
59TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
60 const OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
61
62 // Return early if the initialization graph is already invoked.
63 if (op_data->init_subgraph_invoked) return kTfLiteOk;
64
65 TF_LITE_ENSURE_EQ(context, node->inputs->size, 0);
66 TF_LITE_ENSURE_EQ(context, node->outputs->size, 0);
67
68 Subgraph* this_subgraph = reinterpret_cast<Subgraph*>(context->impl_);
69 auto* subgraphs = this_subgraph->GetSubgraphs();
70 TF_LITE_ENSURE(context, op_data->init_subgraph_index < subgraphs->size());
71
72 // Ensures that there are no input and output tensors in the subgraph.
73 Subgraph* init_subgraph = (*subgraphs)[op_data->init_subgraph_index].get();
74 TF_LITE_ENSURE_EQ(context, init_subgraph->inputs().size(), 0);
75 TF_LITE_ENSURE_EQ(context, init_subgraph->outputs().size(), 0);
76 return kTfLiteOk;
77}
78
79TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
80 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
81
82 // The initialization graph should be invoked once in a life cycle.
83 if (op_data->init_subgraph_invoked) return kTfLiteOk;
84
85 Subgraph* this_subgraph = reinterpret_cast<Subgraph*>(context->impl_);
86 auto* subgraphs = this_subgraph->GetSubgraphs();
87 Subgraph& init_subgraph = *(*subgraphs)[op_data->init_subgraph_index];
88
89 TF_LITE_ENSURE_OK(context, init_subgraph.AllocateTensors());
90 TF_LITE_ENSURE_OK(context, init_subgraph.Invoke());
91 TF_LITE_ENSURE_OK(context, init_subgraph.ReleaseNonPersistentMemory());
92
93 // Mark the invocation completed.
94 op_data->init_subgraph_invoked = true;
95 return kTfLiteOk;
96}
97
98} // namespace call_once_kernel
99
100TfLiteRegistration* Register_CALL_ONCE() {
101 static TfLiteRegistration r = {call_once_kernel::Init, call_once_kernel::Free,
102 call_once_kernel::Prepare,
103 call_once_kernel::Eval};
104 return &r;
105}
