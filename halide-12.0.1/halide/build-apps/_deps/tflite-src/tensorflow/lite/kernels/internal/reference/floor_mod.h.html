15#ifndef TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_FLOOR_MOD_H_
16#define TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_FLOOR_MOD_H_
17
18#include <functional>
19
20namespace tflite {
21
22namespace reference_ops {
23
24template <typename T>
25T FloorMod(T input1, T input2) {
26 struct FloatMod {
27 float operator()(const float lhs, const float rhs) const {
28 return std::fmod(lhs, rhs);
29 }
30 };
31 using ModFunc = typename std::conditional<std::is_integral<T>::value,
32 std::modulus<T>, FloatMod>::type;
33 ModFunc mod_func;
34 T trunc_mod = mod_func(input1, input2);
35 return (trunc_mod != 0) && ((input2 < 0) != (trunc_mod < 0))
36 ? (trunc_mod + input2)
37 : trunc_mod;
38}
39
40} // namespace reference_ops
41} // namespace tflite
42
43#endif // TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_FLOOR_MOD_H_
