26inline void AveragePool(const PoolParams& params,
27 const RuntimeShape& input_shape,
28 const float* input_data,
29 const RuntimeShape& output_shape, float* output_data) {
30 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
31 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
32 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
33 const int depth = MatchingDim(input_shape, 3, output_shape, 3);
34 const int input_height = input_shape.Dims(1);
35 const int input_width = input_shape.Dims(2);
36 const int output_height = output_shape.Dims(1);
37 const int output_width = output_shape.Dims(2);
38 const int stride_height = params.stride_height;
39 const int stride_width = params.stride_width;
40 for (int batch = 0; batch < batches; ++batch) {
41 for (int out_y = 0; out_y < output_height; ++out_y) {
42 for (int out_x = 0; out_x < output_width; ++out_x) {
43 for (int channel = 0; channel < depth; ++channel) {
44 const int in_x_origin =
45 (out_x * stride_width) - params.padding_values.width;
46 const int in_y_origin =
47 (out_y * stride_height) - params.padding_values.height;
48 // Compute the boundaries of the filter region clamped so as to
49 // ensure that the filter window fits in the input array.
50 const int filter_x_start = std::max(0, -in_x_origin);
51 const int filter_x_end =
52 std::min(params.filter_width, input_width - in_x_origin);
53 const int filter_y_start = std::max(0, -in_y_origin);
54 const int filter_y_end =
55 std::min(params.filter_height, input_height - in_y_origin);
56 float total = 0.f;
57 float filter_count = 0;
58 for (int filter_y = filter_y_start; filter_y < filter_y_end;
59 ++filter_y) {
60 for (int filter_x = filter_x_start; filter_x < filter_x_end;
61 ++filter_x) {
62 const int in_x = in_x_origin + filter_x;
63 const int in_y = in_y_origin + filter_y;
64 total +=
65 input_data[Offset(input_shape, batch, in_y, in_x, channel)];
66 filter_count++;
67 }
68 }
69 const float average = total / filter_count;
70 output_data[Offset(output_shape, batch, out_y, out_x, channel)] =
71 ActivationFunctionWithMinMax(average, params.float_activation_min,
72 params.float_activation_max);
73 }
74 }
75 }
76 }
77}

79inline void AveragePool(const PoolParams& params,
80 const RuntimeShape& input_shape,
81 const uint8_t* input_data,
82 const RuntimeShape& output_shape,
83 uint8_t* output_data) {
84 TFLITE_DCHECK_LE(params.quantized_activation_min,
85 params.quantized_activation_max);
86 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
87 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
88 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
89 const int depth = MatchingDim(input_shape, 3, output_shape, 3);
90 const int input_height = input_shape.Dims(1);
91 const int input_width = input_shape.Dims(2);
92 const int output_height = output_shape.Dims(1);
93 const int output_width = output_shape.Dims(2);
94 const int stride_height = params.stride_height;
95 const int stride_width = params.stride_width;
96 for (int batch = 0; batch < batches; ++batch) {
97 for (int out_y = 0; out_y < output_height; ++out_y) {
98 for (int out_x = 0; out_x < output_width; ++out_x) {
99 for (int channel = 0; channel < depth; ++channel) {
100 const int in_x_origin =
101 (out_x * stride_width) - params.padding_values.width;
102 const int in_y_origin =
103 (out_y * stride_height) - params.padding_values.height;
104 // Compute the boundaries of the filter region clamped so as to
105 // ensure that the filter window fits in the input array.
106 const int filter_x_start = std::max(0, -in_x_origin);
107 const int filter_x_end =
108 std::min(params.filter_width, input_width - in_x_origin);
109 const int filter_y_start = std::max(0, -in_y_origin);
110 const int filter_y_end =
111 std::min(params.filter_height, input_height - in_y_origin);
112 int32_t acc = 0;
113 int filter_count = 0;
114 for (int filter_y = filter_y_start; filter_y < filter_y_end;
115 ++filter_y) {
116 for (int filter_x = filter_x_start; filter_x < filter_x_end;
117 ++filter_x) {
118 const int in_x = in_x_origin + filter_x;
119 const int in_y = in_y_origin + filter_y;
120 acc +=
121 input_data[Offset(input_shape, batch, in_y, in_x, channel)];
122 filter_count++;
123 }
124 }
125 acc = (acc + filter_count / 2) / filter_count;
126 acc = std::max(acc, params.quantized_activation_min);
127 acc = std::min(acc, params.quantized_activation_max);
128 output_data[Offset(output_shape, batch, out_y, out_x, channel)] =
129 static_cast<uint8_t>(acc);
130 }
131 }
132 }
133 }

136inline void L2Pool(const PoolParams& params, const RuntimeShape& input_shape,
137 const float* input_data, const RuntimeShape& output_shape,
138 float* output_data) {
139 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
140 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
141 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
142 const int depth = MatchingDim(input_shape, 3, output_shape, 3);
143 const int input_height = input_shape.Dims(1);
144 const int input_width = input_shape.Dims(2);
145 const int output_height = output_shape.Dims(1);
146 const int output_width = output_shape.Dims(2);
147 const int stride_height = params.stride_height;
148 const int stride_width = params.stride_width;
149 for (int batch = 0; batch < batches; ++batch) {
150 for (int out_y = 0; out_y < output_height; ++out_y) {
151 for (int out_x = 0; out_x < output_width; ++out_x) {
152 for (int channel = 0; channel < depth; ++channel) {
153 const int in_x_origin =
154 (out_x * stride_width) - params.padding_values.width;
155 const int in_y_origin =
156 (out_y * stride_height) - params.padding_values.height;
157 // Compute the boundaries of the filter region clamped so as to
158 // ensure that the filter window fits in the input array.
159 const int filter_x_start = std::max(0, -in_x_origin);
160 const int filter_x_end =
161 std::min(params.filter_width, input_width - in_x_origin);
162 const int filter_y_start = std::max(0, -in_y_origin);
163 const int filter_y_end =
164 std::min(params.filter_height, input_height - in_y_origin);
165 float sum_squares = 0.f;
166 int filter_count = 0;
167 for (int filter_y = filter_y_start; filter_y < filter_y_end;
168 ++filter_y) {
169 for (int filter_x = filter_x_start; filter_x < filter_x_end;
170 ++filter_x) {
171 const int in_x = in_x_origin + filter_x;
172 const int in_y = in_y_origin + filter_y;
173 const float val =
174 input_data[Offset(input_shape, batch, in_y, in_x, channel)];
175 sum_squares += val * val;
176 filter_count++;
177 }
178 }
179 const float l2pool_result = std::sqrt(sum_squares / filter_count);
180 output_data[Offset(output_shape, batch, out_y, out_x, channel)] =
181 ActivationFunctionWithMinMax(l2pool_result,
182 params.float_activation_min,
183 params.float_activation_max);
184 }
185 }
186 }
187 }

190inline void MaxPool(const PoolParams& params, const RuntimeShape& input_shape,
191 const float* input_data, const RuntimeShape& output_shape,
192 float* output_data) {
193 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
194 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
195 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
196 const int depth = MatchingDim(input_shape, 3, output_shape, 3);
197 const int input_height = input_shape.Dims(1);
198 const int input_width = input_shape.Dims(2);
199 const int output_height = output_shape.Dims(1);
200 const int output_width = output_shape.Dims(2);
201 const int stride_height = params.stride_height;
202 const int stride_width = params.stride_width;
203 for (int batch = 0; batch < batches; ++batch) {
204 for (int out_y = 0; out_y < output_height; ++out_y) {
205 for (int out_x = 0; out_x < output_width; ++out_x) {
206 for (int channel = 0; channel < depth; ++channel) {
207 const int in_x_origin =
208 (out_x * stride_width) - params.padding_values.width;
209 const int in_y_origin =
210 (out_y * stride_height) - params.padding_values.height;
211 // Compute the boundaries of the filter region clamped so as to
212 // ensure that the filter window fits in the input array.
213 const int filter_x_start = std::max(0, -in_x_origin);
214 const int filter_x_end =
215 std::min(params.filter_width, input_width - in_x_origin);
216 const int filter_y_start = std::max(0, -in_y_origin);
217 const int filter_y_end =
218 std::min(params.filter_height, input_height - in_y_origin);
219 float max = std::numeric_limits<float>::lowest();
220 for (int filter_y = filter_y_start; filter_y < filter_y_end;
221 ++filter_y) {
222 for (int filter_x = filter_x_start; filter_x < filter_x_end;
223 ++filter_x) {
224 const int in_x = in_x_origin + filter_x;
225 const int in_y = in_y_origin + filter_y;
226 max = std::max(
227 max,
228 input_data[Offset(input_shape, batch, in_y, in_x, channel)]);
229 }
230 }
231 output_data[Offset(output_shape, batch, out_y, out_x, channel)] =
232 ActivationFunctionWithMinMax(max, params.float_activation_min,
233 params.float_activation_max);
234 }
235 }
236 }

240inline void MaxPool(const PoolParams& params, const RuntimeShape& input_shape,
241 const uint8_t* input_data, const RuntimeShape& output_shape,
242 uint8_t* output_data) {
243 TFLITE_DCHECK_LE(params.quantized_activation_min,
244 params.quantized_activation_max);
245 TFLITE_DCHECK_GE(params.quantized_activation_min, 0);
246 TFLITE_DCHECK_LE(params.quantized_activation_max, 255);
247 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
248 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
249 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
250 const int depth = MatchingDim(input_shape, 3, output_shape, 3);
251 const int input_height = input_shape.Dims(1);
252 const int input_width = input_shape.Dims(2);
253 const int output_height = output_shape.Dims(1);
254 const int output_width = output_shape.Dims(2);
255 const int stride_height = params.stride_height;
256 const int stride_width = params.stride_width;
257 for (int batch = 0; batch < batches; ++batch) {
258 for (int out_y = 0; out_y < output_height; ++out_y) {
259 for (int out_x = 0; out_x < output_width; ++out_x) {
260 for (int channel = 0; channel < depth; ++channel) {
261 const int in_x_origin =
262 (out_x * stride_width) - params.padding_values.width;
263 const int in_y_origin =
264 (out_y * stride_height) - params.padding_values.height;
265 // Compute the boundaries of the filter region clamped so as to
266 // ensure that the filter window fits in the input array.
267 const int filter_x_start = std::max(0, -in_x_origin);
268 const int filter_x_end =
269 std::min(params.filter_width, input_width - in_x_origin);
270 const int filter_y_start = std::max(0, -in_y_origin);
271 const int filter_y_end =
272 std::min(params.filter_height, input_height - in_y_origin);
273 uint8_t max = 0;
274 for (int filter_y = filter_y_start; filter_y < filter_y_end;
275 ++filter_y) {
276 for (int filter_x = filter_x_start; filter_x < filter_x_end;
277 ++filter_x) {
278 const int in_x = in_x_origin + filter_x;
279 const int in_y = in_y_origin + filter_y;
280 max = std::max(
281 max,
282 input_data[Offset(input_shape, batch, in_y, in_x, channel)]);
283 }
284 }
285 max = std::max<uint8_t>(max, params.quantized_activation_min);
286 max = std::min<uint8_t>(max, params.quantized_activation_max);
287 output_data[Offset(output_shape, batch, out_y, out_x, channel)] =
288 static_cast<uint8_t>(max);
289 }
290 }
291 }
