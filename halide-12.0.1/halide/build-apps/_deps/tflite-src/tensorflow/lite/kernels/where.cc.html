29constexpr int kInputConditionTensor = 0;
30constexpr int kOutputTensor = 0;
31
32TfLiteStatus ResizeOutputTensor(TfLiteContext* context,
33 const TfLiteTensor* cond_tensor,
34 TfLiteTensor* output_tensor) {
35 // Output tensor should have shape:
36 // (num_true, cond_rank), where num_true denotes the number of true values
37 // in condition.
38 const RuntimeShape& cond_shape = GetTensorShape(cond_tensor);
39 const int size = cond_shape.FlatSize();
40 const int cond_rank = cond_shape.DimensionsCount();
41 const bool* cond_data = GetTensorData<bool>(cond_tensor);
42
43 int true_count = 0;
44 for (int i = 0; i < size; ++i) {
45 if (cond_data[i]) {
46 true_count++;
47 }
48 }
49 TfLiteIntArray* output_dims = TfLiteIntArrayCreate(2);
50 output_dims->data[0] = true_count;
51 output_dims->data[1] = cond_rank;
52 return context->ResizeTensor(context, output_tensor, output_dims);
53}
54
55TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
56 TF_LITE_ENSURE_EQ(context, NumInputs(node), 1);
57 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
58
59 const TfLiteTensor* cond_tensor;
60 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputConditionTensor,
61 &cond_tensor));
62 TfLiteTensor* output;
63 TF_LITE_ENSURE_OK(context,
64 GetOutputSafe(context, node, kOutputTensor, &output));
65
66 if (cond_tensor->type != kTfLiteBool) {
67 context->ReportError(context,
68 "Condition tensor must be of type bool, but saw '%s'.",
69 TfLiteTypeGetName(cond_tensor->type));
70 return kTfLiteError;
71 }
72
73 // As output will be a 2D tensor of indices, use int64 to be consistent with
74 // tensorflow.
75 output->type = kTfLiteInt64;
76
77 // Exit early if cond is a non-const tensor. Set output tensor to dynamic so
78 // output size can be determined in Eval.
79 if (!IsConstantTensor(cond_tensor)) {
80 SetTensorToDynamic(output);
81 return kTfLiteOk;
82 }
83 return ResizeOutputTensor(context, cond_tensor, output);
84}
85
86TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
87 const TfLiteTensor* cond_tensor;
88 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputConditionTensor,
89 &cond_tensor));
90 TfLiteTensor* output;
91 TF_LITE_ENSURE_OK(context,
92 GetOutputSafe(context, node, kOutputTensor, &output));
93
94 if (IsDynamicTensor(output)) {
95 TF_LITE_ENSURE_OK(context,
96 ResizeOutputTensor(context, cond_tensor, output));
97 }
98
99 TfLiteIntArray* dims = cond_tensor->dims;
100 if (dims->size == 0) {
101 // Scalar tensors are not supported.
102 TF_LITE_KERNEL_LOG(context, "Where op requires condition w/ rank > 0");
103 return kTfLiteError;
104 }
105
106 reference_ops::SelectTrueCoords(GetTensorShape(cond_tensor),
107 GetTensorData<bool>(cond_tensor),
108 GetTensorData<int64_t>(output));
109 return kTfLiteOk;
110}
111} // namespace where

113TfLiteRegistration* Register_WHERE() {
114 static TfLiteRegistration r = {/*init*/ nullptr, /*free*/ nullptr,
115 where::Prepare, where::Eval};
116 return &r;
117}
