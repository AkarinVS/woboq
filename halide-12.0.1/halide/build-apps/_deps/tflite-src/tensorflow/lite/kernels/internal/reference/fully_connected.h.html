26inline void FullyConnected(
27 const FullyConnectedParams& params, const RuntimeShape& input_shape,
28 const float* input_data, const RuntimeShape& weights_shape,
29 const float* weights_data, const RuntimeShape& bias_shape,
30 const float* bias_data, const RuntimeShape& output_shape,
31 float* output_data) {
32 const float output_activation_min = params.float_activation_min;
33 const float output_activation_max = params.float_activation_max;
34 // TODO(b/62193649): This really should be:
35 // const int batches = ArraySize(output_dims, 1);
36 // but the current --variable_batch hack consists in overwriting the 3rd
37 // dimension with the runtime batch size, as we don't keep track for each
38 // array of which dimension is the batch dimension in it.
39 const int output_dims_count = output_shape.DimensionsCount();
40 const int weights_dims_count = weights_shape.DimensionsCount();
41 const int batches = FlatSizeSkipDim(output_shape, output_dims_count - 1);
42 const int output_depth = MatchingDim(weights_shape, weights_dims_count - 2,
43 output_shape, output_dims_count - 1);
44 const int accum_depth = weights_shape.Dims(weights_dims_count - 1);
45 for (int b = 0; b < batches; ++b) {
46 for (int out_c = 0; out_c < output_depth; ++out_c) {
47 float total = 0.f;
48 for (int d = 0; d < accum_depth; ++d) {
49 total += input_data[b * accum_depth + d] *
50 weights_data[out_c * accum_depth + d];
51 }
52 float bias_value = 0.0f;
53 if (bias_data) {
54 bias_value = bias_data[out_c];
55 }
56 output_data[out_c + output_depth * b] = ActivationFunctionWithMinMax(
57 total + bias_value, output_activation_min, output_activation_max);
58 }
59 }
60}

62inline void FullyConnected(
63 const FullyConnectedParams& params, const RuntimeShape& input_shape,
64 const uint8_t* input_data, const RuntimeShape& filter_shape,
65 const uint8_t* filter_data, const RuntimeShape& bias_shape,
66 const int32_t* bias_data, const RuntimeShape& output_shape,
67 uint8_t* output_data) {
68 const int32_t input_offset = params.input_offset;
69 const int32_t filter_offset = params.weights_offset;
70 const int32_t output_offset = params.output_offset;
71 const int32_t output_multiplier = params.output_multiplier;
72 const int output_shift = params.output_shift;
73 const int32_t output_activation_min = params.quantized_activation_min;
74 const int32_t output_activation_max = params.quantized_activation_max;
75 TFLITE_DCHECK_GE(filter_shape.DimensionsCount(), 2);
76 TFLITE_DCHECK_GE(output_shape.DimensionsCount(), 1);
77
78 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
79 // TODO(b/62193649): This really should be:
80 // const int batches = ArraySize(output_dims, 1);
81 // but the current --variable_batch hack consists in overwriting the 3rd
82 // dimension with the runtime batch size, as we don't keep track for each
83 // array of which dimension is the batch dimension in it.
84 const int output_dim_count = output_shape.DimensionsCount();
85 const int filter_dim_count = filter_shape.DimensionsCount();
86 const int batches = FlatSizeSkipDim(output_shape, output_dim_count - 1);
87 const int output_depth = MatchingDim(filter_shape, filter_dim_count - 2,
88 output_shape, output_dim_count - 1);
89 const int accum_depth = filter_shape.Dims(filter_dim_count - 1);
90 for (int b = 0; b < batches; ++b) {
91 for (int out_c = 0; out_c < output_depth; ++out_c) {
92 int32_t acc = 0;
93 for (int d = 0; d < accum_depth; ++d) {
94 int32_t input_val = input_data[b * accum_depth + d];
95 int32_t filter_val = filter_data[out_c * accum_depth + d];
96 acc += (filter_val + filter_offset) * (input_val + input_offset);
97 }
98 if (bias_data) {
99 acc += bias_data[out_c];
100 }
101 acc = MultiplyByQuantizedMultiplier(acc, output_multiplier, output_shift);
102 acc += output_offset;
103 acc = std::max(acc, output_activation_min);
104 acc = std::min(acc, output_activation_max);
105 output_data[out_c + output_depth * b] = static_cast<uint8_t>(acc);
106 }
107 }
108}

110inline void FullyConnected(
111 const FullyConnectedParams& params, const RuntimeShape& input_shape,
112 const uint8_t* input_data, const RuntimeShape& filter_shape,
113 const uint8_t* filter_data, const RuntimeShape& bias_shape,
114 const int32_t* bias_data, const RuntimeShape& output_shape,
115 int16_t* output_data) {
116 const int32_t input_offset = params.input_offset;
117 const int32_t filter_offset = params.weights_offset;
118 const int32_t output_offset = params.output_offset;
119 const int32_t output_multiplier = params.output_multiplier;
120 const int output_shift = params.output_shift;
121 const int32_t output_activation_min = params.quantized_activation_min;
122 const int32_t output_activation_max = params.quantized_activation_max;
123
124 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
125 TFLITE_DCHECK_EQ(output_offset, 0);
126 // TODO(b/62193649): This really should be:
127 // const int batches = ArraySize(output_dims, 1);
128 // but the current --variable_batch hack consists in overwriting the 3rd
129 // dimension with the runtime batch size, as we don't keep track for each
130 // array of which dimension is the batch dimension in it.
131 const int output_dim_count = output_shape.DimensionsCount();
132 const int filter_dim_count = filter_shape.DimensionsCount();
133 const int batches = FlatSizeSkipDim(output_shape, output_dim_count - 1);
134 const int output_depth = MatchingDim(filter_shape, filter_dim_count - 2,
135 output_shape, output_dim_count - 1);
136 const int accum_depth = filter_shape.Dims(filter_dim_count - 1);
137 for (int b = 0; b < batches; ++b) {
138 for (int out_c = 0; out_c < output_depth; ++out_c) {
139 // Internal accumulation.
140 // Initialize accumulator with the bias-value.
141 int32_t accum = bias_data[out_c];
142 // Accumulation loop.
143 for (int d = 0; d < accum_depth; ++d) {
144 int16_t input_val = input_data[b * accum_depth + d] + input_offset;
145 int16_t filter_val =
146 filter_data[out_c * accum_depth + d] + filter_offset;
147 accum += filter_val * input_val;
148 }
149 // Down-scale the final int32_t accumulator to the scale used by our
150 // (16-bit, typically 3 integer bits) fixed-point format. The quantized
151 // multiplier and shift here have been pre-computed offline
152 // (e.g. by toco).
153 accum =
154 MultiplyByQuantizedMultiplier(accum, output_multiplier, output_shift);
155 // Saturate, cast to int16_t, and store to output array.
156 accum = std::max(accum, output_activation_min - output_offset);
157 accum = std::min(accum, output_activation_max - output_offset);
158 accum += output_offset;
159 output_data[out_c + output_depth * b] = accum;
160 }
161 }

164inline void ShuffledFullyConnected(
165 const FullyConnectedParams& params, const RuntimeShape& input_shape,
166 const uint8_t* input_data, const RuntimeShape& weights_shape,
167 const uint8_t* shuffled_weights_data, const RuntimeShape& bias_shape,
168 const int32_t* bias_data, const RuntimeShape& output_shape,
169 int16_t* output_data, uint8_t* shuffled_input_workspace_data) {
170 const int32_t output_multiplier = params.output_multiplier;
171 const int output_shift = params.output_shift;
172 const int32_t output_activation_min = params.quantized_activation_min;
173 const int32_t output_activation_max = params.quantized_activation_max;
174 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
175
176 TFLITE_DCHECK_GE(input_shape.DimensionsCount(), 1);
177 TFLITE_DCHECK_GE(weights_shape.DimensionsCount(), 2);
178 TFLITE_DCHECK_GE(output_shape.DimensionsCount(), 1);
179 // TODO(b/62193649): This really should be:
180 // const int batches = ArraySize(output_dims, 1);
181 // but the current --variable_batch hack consists in overwriting the 3rd
182 // dimension with the runtime batch size, as we don't keep track for each
183 // array of which dimension is the batch dimension in it.
184 const int output_dim_count = output_shape.DimensionsCount();
185 const int weights_dim_count = weights_shape.DimensionsCount();
186 const int batches = FlatSizeSkipDim(output_shape, output_dim_count - 1);
187 const int output_depth = MatchingDim(weights_shape, weights_dim_count - 2,
188 output_shape, output_dim_count - 1);
189 const int accum_depth = weights_shape.Dims(weights_dim_count - 1);
190 TFLITE_DCHECK((accum_depth % 16) == 0);
191 TFLITE_DCHECK((output_depth % 4) == 0);
192
193 // Shuffling and xoring of input activations into the workspace buffer
194 uint8_t* shuffled_input_workspace_ptr = shuffled_input_workspace_data;
195 if (batches == 1) {
196 for (int i = 0; i < accum_depth; i++) {
197 shuffled_input_workspace_data[i] = input_data[i] ^ 0x80;
198 }
199 } else if (batches == 4) {
200 for (int c = 0; c < accum_depth; c += 16) {
201 for (int b = 0; b < 4; b++) {
202 const uint8_t* src_data_ptr = input_data + b * accum_depth + c;
203 for (int j = 0; j < 16; j++) {
204 uint8_t src_val = *src_data_ptr++;
205 // Flip the sign bit, so that the kernel will only need to
206 // reinterpret these uint8_t values as int8_t, getting for free the
207 // subtraction of the zero_point value 128.
208 uint8_t dst_val = src_val ^ 0x80;
209 *shuffled_input_workspace_ptr++ = dst_val;
210 }
211 }
212 }
213 } else {
214 TFLITE_DCHECK(false);
215 return;
216 }
217
218 // Actual computation
219 if (batches == 1) {
220 int16_t* output_ptr = output_data;
221 // Shuffled weights have had their sign bit (0x80) pre-flipped (xor'd)
222 // so that just reinterpreting them as int8_t values is equivalent to
223 // subtracting 128 from them, thus implementing for free the subtraction of
224 // the zero_point value 128.
225 const int8_t* shuffled_weights_ptr =
226 reinterpret_cast<const int8_t*>(shuffled_weights_data);
227 // Likewise, we preshuffled and pre-xored the input data above.
228 const int8_t* shuffled_input_data =
229 reinterpret_cast<const int8_t*>(shuffled_input_workspace_data);
230 for (int c = 0; c < output_depth; c += 4) {
231 // Internal accumulation.
232 // Initialize accumulator with the bias-value.
233 int32_t accum[4] = {0};
234 // Accumulation loop.
235 for (int d = 0; d < accum_depth; d += 16) {
236 for (int i = 0; i < 4; i++) {
237 for (int j = 0; j < 16; j++) {
238 int8_t input_val = shuffled_input_data[d + j];
239 int8_t weights_val = *shuffled_weights_ptr++;
240 accum[i] += weights_val * input_val;
241 }
242 }
243 }
244 for (int i = 0; i < 4; i++) {
245 // Add bias value
246 int32_t acc = accum[i] + bias_data[c + i];
247 // Down-scale the final int32_t accumulator to the scale used by our
248 // (16-bit, typically 3 integer bits) fixed-point format. The quantized
249 // multiplier and shift here have been pre-computed offline
250 // (e.g. by toco).
251 acc =
252 MultiplyByQuantizedMultiplier(acc, output_multiplier, output_shift);
253 // Saturate, cast to int16_t, and store to output array.
254 acc = std::max(acc, output_activation_min);
255 acc = std::min(acc, output_activation_max);
256 output_ptr[c + i] = acc;
257 }
258 }
259 } else if (batches == 4) {
260 int16_t* output_ptr = output_data;
261 // Shuffled weights have had their sign bit (0x80) pre-flipped (xor'd)
262 // so that just reinterpreting them as int8_t values is equivalent to
263 // subtracting 128 from them, thus implementing for free the subtraction of
264 // the zero_point value 128.
265 const int8_t* shuffled_weights_ptr =
266 reinterpret_cast<const int8_t*>(shuffled_weights_data);
267 // Likewise, we preshuffled and pre-xored the input data above.
268 const int8_t* shuffled_input_data =
269 reinterpret_cast<const int8_t*>(shuffled_input_workspace_data);
270 for (int c = 0; c < output_depth; c += 4) {
271 const int8_t* shuffled_input_ptr = shuffled_input_data;
272 // Accumulation loop.
273 // Internal accumulation.
274 // Initialize accumulator with the bias-value.
275 int32_t accum[4][4];
276 for (int i = 0; i < 4; i++) {
277 for (int b = 0; b < 4; b++) {
278 accum[i][b] = 0;
279 }
280 }
281 for (int d = 0; d < accum_depth; d += 16) {
282 for (int i = 0; i < 4; i++) {
283 for (int b = 0; b < 4; b++) {
284 for (int j = 0; j < 16; j++) {
285 int8_t input_val = shuffled_input_ptr[16 * b + j];
286 int8_t weights_val = shuffled_weights_ptr[16 * i + j];
287 accum[i][b] += weights_val * input_val;
288 }
289 }
290 }
291 shuffled_input_ptr += 64;
292 shuffled_weights_ptr += 64;
293 }
294 for (int i = 0; i < 4; i++) {
295 for (int b = 0; b < 4; b++) {
296 // Add bias value
297 int32_t acc = accum[i][b] + bias_data[c + i];
298 // Down-scale the final int32_t accumulator to the scale used by our
299 // (16-bit, typically 3 integer bits) fixed-point format. The
300 // quantized multiplier and shift here have been pre-computed offline
301 // (e.g. by toco).
302 acc = MultiplyByQuantizedMultiplier(acc, output_multiplier,
303 output_shift);
304 // Saturate, cast to int16_t, and store to output array.
305 acc = std::max(acc, output_activation_min);
306 acc = std::min(acc, output_activation_max);
307 output_ptr[b * output_depth + c + i] = acc;
308 }
309 }
310 }
311 } else {
312 TFLITE_DCHECK(false);
313 return;
314 }
315}
