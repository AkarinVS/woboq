26// Broadcast prelu to output_shape for quantized uint8_t/int8_t data.
27template <typename T>
28inline void BroadcastPrelu4DSlow(
29 const PreluParams& params, const RuntimeShape& input_shape,
30 const T* input_data, const RuntimeShape& alpha_shape, const T* alpha_data,
31 const RuntimeShape& output_shape, T* output_data) {
32 TFLITE_DCHECK_LE(input_shape.DimensionsCount(), 4);
33 TFLITE_DCHECK_LE(alpha_shape.DimensionsCount(), 4);
34 TFLITE_DCHECK_LE(output_shape.DimensionsCount(), 4);
35 const RuntimeShape extended_output_shape =
36 RuntimeShape::ExtendedShape(4, output_shape);
37 NdArrayDesc<4> desc1;
38 NdArrayDesc<4> desc2;
39 NdArrayDescsForElementwiseBroadcast(input_shape, alpha_shape, &desc1, &desc2);
40
41 for (int b = 0; b < extended_output_shape.Dims(0); ++b) {
42 for (int y = 0; y < extended_output_shape.Dims(1); ++y) {
43 for (int x = 0; x < extended_output_shape.Dims(2); ++x) {
44 for (int c = 0; c < extended_output_shape.Dims(3); ++c) {
45 int output_index = Offset(extended_output_shape, b, y, x, c);
46 int input_index = SubscriptToIndex(desc1, b, y, x, c);
47 const int32_t input_value =
48 params.input_offset + input_data[input_index];
49 int32_t output_value;
50 if (input_value >= 0) {
51 output_value = MultiplyByQuantizedMultiplier(
52 input_value, params.output_multiplier_1, params.output_shift_1);
53 } else {
54 auto alpha_index = SubscriptToIndex(desc2, b, y, x, c);
55 const int32_t alpha_value =
56 params.alpha_offset + alpha_data[alpha_index];
57
58 output_value = MultiplyByQuantizedMultiplier(
59 input_value * alpha_value, params.output_multiplier_2,
60 params.output_shift_2);
61 }
62 output_value += params.output_offset;
63
64 const int32_t quantized_min = std::numeric_limits<T>::min();
65 const int32_t quantized_max = std::numeric_limits<T>::max();
66 const int32_t clamped_output =
67 std::min(quantized_max, std::max(quantized_min, output_value));
68 output_data[output_index] = static_cast<T>(clamped_output);
69 }
70 }
71 }
72 }
73}

75template <typename T>
76inline void Prelu(const PreluParams& params, const RuntimeShape& input_shape,
77 const T* input_data, const RuntimeShape& alpha_shape,
78 const T* alpha_data, const RuntimeShape& output_shape,
79 T* output_data) {
80 const int32_t quantized_min = std::numeric_limits<T>::min();
81 const int32_t quantized_max = std::numeric_limits<T>::max();
82
83 const int flat_size =
84 MatchingElementsSize(input_shape, alpha_shape, output_shape);
85 for (int i = 0; i < flat_size; ++i) {
86 const int32_t input_value = params.input_offset + input_data[i];
87 int32_t output_value;
88 if (input_value >= 0) {
89 output_value = MultiplyByQuantizedMultiplier(
90 input_value, params.output_multiplier_1, params.output_shift_1);
91 } else {
92 const int32_t alpha_value = params.alpha_offset + alpha_data[i];
93
94 output_value = MultiplyByQuantizedMultiplier(input_value * alpha_value,
95 params.output_multiplier_2,
96 params.output_shift_2);
97 }
98 output_value += params.output_offset;
99
100 const int32_t clamped_output =
101 std::min(quantized_max, std::max(quantized_min, output_value));
102 output_data[i] = static_cast<T>(clamped_output);
103 }
