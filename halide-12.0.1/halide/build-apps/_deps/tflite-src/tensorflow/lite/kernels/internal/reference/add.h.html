25template <typename T>
26inline void Add(const ArithmeticParams& params,
27 const RuntimeShape& input1_shape, const T* input1_data,
28 const RuntimeShape& input2_shape, const T* input2_data,
29 const RuntimeShape& output_shape, T* output_data) {
30 const int flat_size =
31 MatchingElementsSize(input1_shape, input2_shape, output_shape);
32 for (int i = 0; i < flat_size; ++i) {
33 output_data[i] = ActivationFunctionWithMinMax(
34 input1_data[i] + input2_data[i], params.quantized_activation_min,
35 params.quantized_activation_max);
36 }
37}

52// Element-wise add that can often be used for inner loop of broadcast add as
53// well as the non-broadcast add.
54
55// This function is used for 8-bit as well as for 16-bit, but the accumulator
56// is 32-bit for both cases. The overflow does not happen due to the
57// choice of the shift (20 or 15, accordingly - see add.cc for more comments).
58template <typename T>
59inline void AddElementwise(int size, const ArithmeticParams& params,
60 const T* input1_data, const T* input2_data,
61 T* output_data) {
62 TFLITE_DCHECK_GT(params.input1_offset, -std::numeric_limits<T>::max());
63 TFLITE_DCHECK_GT(params.input2_offset, -std::numeric_limits<T>::max());
64 TFLITE_DCHECK_LT(params.input1_offset, std::numeric_limits<T>::max());
65 TFLITE_DCHECK_LT(params.input2_offset, std::numeric_limits<T>::max());
66
67 for (int i = 0; i < size; ++i) {
68 const int32_t input1_val = params.input1_offset + input1_data[i];
69 const int32_t input2_val = params.input2_offset + input2_data[i];
70 const int32_t shifted_input1_val = input1_val * (1 << params.left_shift);
71 const int32_t shifted_input2_val = input2_val * (1 << params.left_shift);
72 const int32_t scaled_input1_val =
73 MultiplyByQuantizedMultiplierSmallerThanOneExp(
74 shifted_input1_val, params.input1_multiplier, params.input1_shift);
75 const int32_t scaled_input2_val =
76 MultiplyByQuantizedMultiplierSmallerThanOneExp(
77 shifted_input2_val, params.input2_multiplier, params.input2_shift);
78 const int32_t raw_sum = scaled_input1_val + scaled_input2_val;
79 const int32_t raw_output =
80 MultiplyByQuantizedMultiplierSmallerThanOneExp(
81 raw_sum, params.output_multiplier, params.output_shift) +
82 params.output_offset;
83 const int32_t clamped_output =
84 std::min(params.quantized_activation_max,
85 std::max(params.quantized_activation_min, raw_output));
86 output_data[i] = static_cast<T>(clamped_output);
87 }
88}

90// Scalar-broadcast add that can be used for inner loop of more general
91// broadcast add, so that, for example, scalar-broadcast with batch will still
92// be fast.
93inline void AddScalarBroadcast(int size, const ArithmeticParams& params,
94 uint8_t input1_data, const uint8_t* input2_data,
95 uint8_t* output_data) {
96 TFLITE_DCHECK_GT(params.input1_offset, -256);
97 TFLITE_DCHECK_GT(params.input2_offset, -256);
98 TFLITE_DCHECK_LT(params.input1_offset, 256);
99 TFLITE_DCHECK_LT(params.input2_offset, 256);
100
101 const int32_t input1_val = params.input1_offset + input1_data;
102 const int32_t shifted_input1_val = input1_val * (1 << params.left_shift);
103 const int32_t scaled_input1_val =
104 MultiplyByQuantizedMultiplierSmallerThanOneExp(
105 shifted_input1_val, params.input1_multiplier, params.input1_shift);
106 for (int i = 0; i < size; ++i) {
107 const int32_t input2_val = params.input2_offset + input2_data[i];
108 const int32_t shifted_input2_val = input2_val * (1 << params.left_shift);
109 const int32_t scaled_input2_val =
110 MultiplyByQuantizedMultiplierSmallerThanOneExp(
111 shifted_input2_val, params.input2_multiplier, params.input2_shift);
112 const int32_t raw_sum = scaled_input1_val + scaled_input2_val;
113 const int32_t raw_output =
114 MultiplyByQuantizedMultiplierSmallerThanOneExp(
115 raw_sum, params.output_multiplier, params.output_shift) +
116 params.output_offset;
117 const int32_t clamped_output =
118 std::min(params.quantized_activation_max,
119 std::max(params.quantized_activation_min, raw_output));
120 output_data[i] = static_cast<uint8_t>(clamped_output);
121 }
122}

124inline void Add(const ArithmeticParams& params,
125 const RuntimeShape& input1_shape, const uint8_t* input1_data,
126 const RuntimeShape& input2_shape, const uint8_t* input2_data,
127 const RuntimeShape& output_shape, uint8_t* output_data) {
128 TFLITE_DCHECK_LE(params.quantized_activation_min,
129 params.quantized_activation_max);
130 const int flat_size =
131 MatchingElementsSize(input1_shape, input2_shape, output_shape);
132
133 TFLITE_DCHECK_GT(params.input1_offset, -256);
134 TFLITE_DCHECK_GT(params.input2_offset, -256);
135 TFLITE_DCHECK_LT(params.input1_offset, 256);
136 TFLITE_DCHECK_LT(params.input2_offset, 256);
137 AddElementwise(flat_size, params, input1_data, input2_data, output_data);
138}

140inline void AddGeneralParamScale(const ArithmeticParams& params,
141 const RuntimeShape& input1_shape,
142 const int16_t* input1_data,
143 const RuntimeShape& input2_shape,
144 const int16_t* input2_data,
145 const RuntimeShape& output_shape,
146 int16_t* output_data) {
147 TFLITE_DCHECK_LE(params.quantized_activation_min,
148 params.quantized_activation_max);
149 const int flat_size =
150 MatchingElementsSize(input1_shape, input2_shape, output_shape);
151
152 int max_value = std::numeric_limits<int16_t>::max();
153
154 TFLITE_DCHECK_GT(params.input1_offset, -max_value);
155 TFLITE_DCHECK_GT(params.input2_offset, -max_value);
156 TFLITE_DCHECK_LT(params.input1_offset, max_value);
157 TFLITE_DCHECK_LT(params.input2_offset, max_value);
158 AddElementwise(flat_size, params, input1_data, input2_data, output_data);

161inline void Add(const ArithmeticParams& params,
162 const RuntimeShape& input1_shape, const int16_t* input1_data,
163 const RuntimeShape& input2_shape, const int16_t* input2_data,
164 const RuntimeShape& output_shape, int16_t* output_data,
165 bool pot_scale = true) {
166 if (!pot_scale) {
167 AddGeneralParamScale(params, input1_shape, input1_data, input2_shape,
168 input2_data, output_shape, output_data);
169 return;
170 }
171
172 TFLITE_DCHECK_LE(params.quantized_activation_min,
173 params.quantized_activation_max);
174
175 const int input1_shift = params.input1_shift;
176 const int flat_size =
177 MatchingElementsSize(input1_shape, input2_shape, output_shape);
178 const int16_t output_activation_min = params.quantized_activation_min;
179 const int16_t output_activation_max = params.quantized_activation_max;
180
181 TFLITE_DCHECK(input1_shift == 0 || params.input2_shift == 0);
182 TFLITE_DCHECK_LE(input1_shift, 0);
183 TFLITE_DCHECK_LE(params.input2_shift, 0);
184 const int16_t* not_shift_input =
185 input1_shift == 0 ? input1_data : input2_data;
186 const int16_t* shift_input = input1_shift == 0 ? input2_data : input1_data;
187 const int input_right_shift =
188 input1_shift == 0 ? -params.input2_shift : -input1_shift;
189
190 for (int i = 0; i < flat_size; i++) {
191 // F0 uses 0 integer bits, range [-1, 1].
192 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
193
194 F0 input_ready_scaled = F0::FromRaw(not_shift_input[i]);
195 F0 scaled_input = F0::FromRaw(
196 gemmlowp::RoundingDivideByPOT(shift_input[i], input_right_shift));
197 F0 result = gemmlowp::SaturatingAdd(scaled_input, input_ready_scaled);
198 const int16_t raw_output = result.raw();
199 const int16_t clamped_output = std::min(
200 output_activation_max, std::max(output_activation_min, raw_output));
201 output_data[i] = clamped_output;
202 }
203}

205inline void BroadcastAdd4DSlow(const ArithmeticParams& params,
206 const RuntimeShape& input1_shape,
207 const float* input1_data,
208 const RuntimeShape& input2_shape,
209 const float* input2_data,
210 const RuntimeShape& output_shape,
211 float* output_data) {
212 NdArrayDesc<4> desc1;
213 NdArrayDesc<4> desc2;
214 NdArrayDescsForElementwiseBroadcast(input1_shape, input2_shape, &desc1,
215 &desc2);
216 const RuntimeShape extended_output_shape =
217 RuntimeShape::ExtendedShape(4, output_shape);
218
219 // In Tensorflow, the dimensions are canonically named (batch_number, row,
220 // col, channel), with extents (batches, height, width, depth), with the
221 // trailing dimension changing most rapidly (channels has the smallest stride,
222 // typically 1 element).
223 //
224 // In generated C code, we store arrays with the dimensions reversed. The
225 // first dimension has smallest stride.
226 //
227 // We name our variables by their Tensorflow convention, but generate C code
228 // nesting loops such that the innermost loop has the smallest stride for the
229 // best cache behavior.
230 for (int b = 0; b < extended_output_shape.Dims(0); ++b) {
231 for (int y = 0; y < extended_output_shape.Dims(1); ++y) {
232 for (int x = 0; x < extended_output_shape.Dims(2); ++x) {
233 for (int c = 0; c < extended_output_shape.Dims(3); ++c) {
234 output_data[Offset(extended_output_shape, b, y, x, c)] =
235 ActivationFunctionWithMinMax(
236 input1_data[SubscriptToIndex(desc1, b, y, x, c)] +
237 input2_data[SubscriptToIndex(desc2, b, y, x, c)],
238 params.float_activation_min, params.float_activation_max);
239 }
240 }
241 }
242 }

350inline void BroadcastAddFivefold(const ArithmeticParams& unswitched_params,
351 const RuntimeShape& unswitched_input1_shape,
352 const uint8_t* unswitched_input1_data,
353 const RuntimeShape& unswitched_input2_shape,
354 const uint8_t* unswitched_input2_data,
355 const RuntimeShape& output_shape,
356 uint8_t* output_data) {
357 ArithmeticParams switched_params = unswitched_params;
358 switched_params.input1_offset = unswitched_params.input2_offset;
359 switched_params.input1_multiplier = unswitched_params.input2_multiplier;
360 switched_params.input1_shift = unswitched_params.input2_shift;
361 switched_params.input2_offset = unswitched_params.input1_offset;
362 switched_params.input2_multiplier = unswitched_params.input1_multiplier;
363 switched_params.input2_shift = unswitched_params.input1_shift;
364
365 const bool use_unswitched =
366 unswitched_params.broadcast_category ==
367 tflite::BroadcastableOpCategory::kFirstInputBroadcastsFast;
368
369 const ArithmeticParams& params =
370 use_unswitched ? unswitched_params : switched_params;
371 const uint8_t* input1_data =
372 use_unswitched ? unswitched_input1_data : unswitched_input2_data;
373 const uint8_t* input2_data =
374 use_unswitched ? unswitched_input2_data : unswitched_input1_data;
375
376 // Fivefold nested loops. The second input resets its position for each
377 // iteration of the second loop. The first input resets its position at the
378 // beginning of the fourth loop. The innermost loop is an elementwise add of
379 // sections of the arrays.
380 uint8_t* output_data_ptr = output_data;
381 const uint8_t* input1_data_ptr = input1_data;
382 const uint8_t* input2_data_reset = input2_data;
383 // In the fivefold pattern, y0, y2 and y4 are not broadcast, and so shared
384 // between input shapes. y3 for input 1 is always broadcast, and so the
385 // dimension there is 1, whereas optionally y1 might be broadcast for input 2.
386 // Put another way,
387 // input1.shape.FlatSize = y0 * y1 * y2 * y4,
388 // input2.shape.FlatSize = y0 * y2 * y3 * y4.
389 int y0 = params.broadcast_shape[0];
390 int y1 = params.broadcast_shape[1];
391 int y2 = params.broadcast_shape[2];
392 int y3 = params.broadcast_shape[3];
393 int y4 = params.broadcast_shape[4];
394 if (y4 > 1) {
395 // General fivefold pattern, with y4 > 1 so there is a non-broadcast inner
396 // dimension.
397 for (int i0 = 0; i0 < y0; ++i0) {
398 const uint8_t* input2_data_ptr;
399 for (int i1 = 0; i1 < y1; ++i1) {
400 input2_data_ptr = input2_data_reset;
401 for (int i2 = 0; i2 < y2; ++i2) {
402 for (int i3 = 0; i3 < y3; ++i3) {
403 AddElementwise(y4, params, input1_data_ptr, input2_data_ptr,
404 output_data_ptr);
405 input2_data_ptr += y4;
406 output_data_ptr += y4;
407 }
408 // We have broadcast y4 of input1 data y3 times, and now move on.
409 input1_data_ptr += y4;
410 }
411 }
412 // We have broadcast y2*y3*y4 of input2 data y1 times, and now move on.
413 input2_data_reset = input2_data_ptr;
414 }
415 } else {
416 // Special case of y4 == 1, in which the innermost loop is a single element
417 // and can be combined with the next (y3) as an inner broadcast.
418 //
419 // Note that this handles the case of pure scalar broadcast when
420 // y0 == y1 == y2 == 1. With low overhead it handles cases such as scalar
421 // broadcast with batch (as y2 > 1).
422 //
423 // NOTE The process is the same as the above general case except simplified
424 // for y4 == 1 and the loop over y3 is contained within the
425 // AddScalarBroadcast function.
426 for (int i0 = 0; i0 < y0; ++i0) {
427 const uint8_t* input2_data_ptr;
428 for (int i1 = 0; i1 < y1; ++i1) {
429 input2_data_ptr = input2_data_reset;
430 for (int i2 = 0; i2 < y2; ++i2) {
431 AddScalarBroadcast(y3, params, *input1_data_ptr, input2_data_ptr,
432 output_data_ptr);
433 input2_data_ptr += y3;
434 output_data_ptr += y3;
435 input1_data_ptr += 1;
436 }
437 }
438 input2_data_reset = input2_data_ptr;
439 }
440 }
