55// This file has 4 implementation of Conv.
56enum KernelType {
57 kReference,
58 kGenericOptimized, // Neon-free
59 // kMultithreadOptimized is a mixture of an Eigen-based kernel when threads
60 // are available and kGenericOptimized when we must use only one thread.
61 kMultithreadOptimized,
62 // The kernel uses use CBLAS interface for matrix multiplication.
63 // It's fast when an optimized CBLAS implementation is available (e.g. Apple
64 // Accelerate Framework), and it's slow when falling back to naive
65 // implementation.
66 kCblasOptimized,
67};

71static constexpr size_t kMaxIm2colBufferSizeMobile = 1024 * 1024 * 1024; // 1GB
72
73struct OpData {
74 // IDs are the arbitrary identifiers used by TF Lite to identify and access
75 // memory buffers.
76 int im2col_id = kTensorNotAllocated;
77 int hwcn_weights_id = kTensorNotAllocated;
78 int input_quantized_id = kTensorNotAllocated;
79 int scaling_factors_id = kTensorNotAllocated;
80 int input_offset_id = kTensorNotAllocated;
81 int accum_scratch_id = kTensorNotAllocated;
82 // Row sums are used to cache filter sums for hybrid zero-point calculations.
83 int row_sums_id = kTensorNotAllocated;
84
85 TfLitePaddingValues padding;
86 // The scaling factor from input to output (aka the 'real multiplier') can
87 // be represented as a fixed point multiplier plus a left shift.
88 int32_t output_multiplier;
89 int output_shift;
90
91 // Per channel output multiplier and shift.
92 std::vector<int32_t> per_channel_output_multiplier;
93 std::vector<int> per_channel_output_shift;
94
95 // The range of the fused activation layer. For example for kNone and
96 // uint8_t these would be 0 and 255.
97 int32_t output_activation_min;
98 int32_t output_activation_max;
99 // Indexes are the offset to the memory buffer in the array used to keep track
100 // of the allocated temporaries.
101 int32_t im2col_index;
102 int32_t hwcn_weights_index;
103 int32_t input_quantized_index;
104 int32_t scaling_factors_index;
105 int32_t accum_scratch_index;
106 int32_t input_offset_index;
107 int32_t row_sums_index;
108
109 bool need_hwcn_weights = false;
110 bool have_weights_been_transposed = false;
111 bool need_im2col = false;
112 // If it's true, it means im2col is needed but gets disabled because the
113 // temporary im2col tensor requires too much memory (i.e.
114 // >= kMaxIm2colBufferSize);
115 bool im2col_oversized = false;
116
117 bool supports_multithreaded_kernel = false;
118 bool is_hybrid_per_channel = false;
119 bool compute_hybrid_row_sums = true;
120};

122inline PaddingType RuntimePaddingType(TfLitePadding padding) {
123 switch (padding) {
124 case TfLitePadding::kTfLitePaddingSame:
125 return PaddingType::kSame;
126 case TfLitePadding::kTfLitePaddingValid:
127 return PaddingType::kValid;
128 case TfLitePadding::kTfLitePaddingUnknown:
129 default:
130 return PaddingType::kNone;
131 }

134void* Init(TfLiteContext* context, const char* buffer, size_t length) {
135 // This is a builtin op, so we don't use the contents in 'buffer', if any.
136 // Instead, we allocate a new object to use as scratch space for im2col, and
137 // to carry information from Prepare() to Eval().
138 auto* data = new OpData;
139#if defined(TFLITE_WITH_MULTITHREADED_EIGEN)
140 eigen_support::IncrementUsageCounter(context);
141#endif
142 return data;
143}
144
145void Free(TfLiteContext* context, void* buffer) {
146#if defined(TFLITE_WITH_MULTITHREADED_EIGEN)
147 eigen_support::DecrementUsageCounter(context);
148#endif
149 delete reinterpret_cast<OpData*>(buffer);
150}

152// Naive implementation of transpose for floats. Could be optimized to be more
153// cache friendly, but for now it's a one-time cost on first run, and we would
154// prefer to remove the need to do this at all eventually.
155void TransposeFloatTensor(const TfLiteTensor* input, TfLiteTensor* output) {
156 const int rows = output->dims->data[1];
157 const int cols = output->dims->data[0];
158 const float* input_data = GetTensorData<float>(input);
159 float* output_data = GetTensorData<float>(output);
160 for (int i = 0; i < rows; ++i) {
161 for (int j = 0; j < cols; ++j) {
162 const float in_value = input_data[i * cols + j];
163 output_data[j * rows + i] = in_value;
164 }
165 }
166}

168// Check if im2col needs to be allocated, as some version of optimized Conv dont
169// use it. If any change is supporting im2col in any of the Conv versions, then
170// it should be updated here as well
171bool IsIm2ColRequired(const TfLiteTensor* input, TfLiteConvParams* params,
172 const TfLiteTensor* filter, OpData* data, bool is_hybrid,
173 KernelType kernel_type) {
174 // If HWCN weights are required, Im2Col not required
175 if (data->need_hwcn_weights) return false;
176
177 // segregate based on dilated conv & non-dialated conv
178 const bool need_dilated_im2col =
179 params->dilation_width_factor != 1 || params->dilation_height_factor != 1;
180 const bool need_non_dilated_im2col =
181 params->stride_width != 1 || params->stride_height != 1 ||
182 filter->dims->data[2] != 1 || filter->dims->data[1] != 1;
183
184 const bool need_im2col = need_dilated_im2col || need_non_dilated_im2col;
185
186 // Return early as basic requirement is not met
187 if (!need_im2col) return false;
188
189 // Special case for Hybrid, as it supports only non-dilated im2col currently
190 const bool is_hybrid_non_dilated = is_hybrid && need_non_dilated_im2col;
191 const bool is_quantized =
192 input->type == kTfLiteUInt8 || input->type == kTfLiteInt8;
193
194 switch (kernel_type) {
195 case kReference:
196 if (is_hybrid) {
197 return true;
198 } else {
199 return false;
200 }
201 case kGenericOptimized:
202 case kCblasOptimized:
203 if (is_hybrid && !need_non_dilated_im2col) {
204 return false;
205 } else {
206 return true;
207 }
208 case kMultithreadOptimized:
209 if (is_hybrid_non_dilated || is_quantized ||
210 !data->supports_multithreaded_kernel) {
211 return true;
212 } else {
213 return false;
214 }
215 default:
216 return false;
217 }
218}

220// Allocate temporary tensors (`im2col`, `hwcn_weights` if necessary).
221// Note: `context->AddTensors` might invalidate pointers to existing tensors.
222// Therefore the logic to add tensors are isolated into this function.
223static TfLiteStatus AllocateTemporaryTensorsIfRequired(
224 TfLiteContext* context, TfLiteNode* node, bool is_hybrid,
225 bool is_per_channel, KernelType kernel_type, size_t im2col_bytes) {
226 auto* params = reinterpret_cast<TfLiteConvParams*>(node->builtin_data);
227 OpData* data = reinterpret_cast<OpData*>(node->user_data);
228
229 TF_LITE_ENSURE(context, node->inputs->size >= 2);
230 const TfLiteTensor* input;
231 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
232 const TfLiteTensor* filter;
233 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 1, &filter));
234
235 // If we're using the optimized multithreaded EigenTensor implementation of
236 // convolution, it expects the filter weights to be transposed compared to
237 // the normal TF Lite buffer format. Typical TF Lite weights are
238 // [filter_count, filter_height, filter_width, input_depth], but for the float
239 // implementation we need them as [filter_height, filter_width, input_depth,
240 // filter_count]. We get to that format by transposing, and create a temporary
241 // buffer to store the results.
242 // This path is only used for float processing, so only create the buffer if
243 // we're running with that data type.
244 data->need_hwcn_weights =
245 input->type == kTfLiteFloat32 && data->supports_multithreaded_kernel;
246
247 // We don't always need to allocate im2col. It is only used in some versions
248 // of the optimized Conv. This test just mimics something that happens inside
249 // optimized_ops.h, in order to avoid a DCHECK(!im2col_data).
250 data->need_im2col =
251 IsIm2ColRequired(input, params, filter, data, is_hybrid, kernel_type);
252
253 // If im2col_oversized is found to be true, we have to fallback to an
254 // execution path (like kReference in float/quantized cases) that doesn't
255 // require im2col operation. Therefore, we have to skip checking the hybrid
256 // case (but not the hybrid-per-channel one) where there's no such a fallback
257 // execution path.
258 // TODO(b/178743262): Consider making this check conditioned on the available
259 // memory of the system, rather than coupling to the mobile platform check.
260 if (IsMobilePlatform() && !(is_hybrid && !is_per_channel) &&
261 data->need_im2col && im2col_bytes >= kMaxIm2colBufferSizeMobile) {
262 data->need_im2col = false;
263 data->im2col_oversized = true;
264 }
265 int temporaries_count = 0;
266 if (data->need_im2col) {
267 data->im2col_index = temporaries_count;
268 if (data->im2col_id == kTensorNotAllocated) {
269 context->AddTensors(context, 1, &data->im2col_id);
270 }
271 ++temporaries_count;
272 }
273 if (data->need_hwcn_weights) {
274 data->hwcn_weights_index = temporaries_count;
275 if (data->hwcn_weights_id == kTensorNotAllocated) {
276 context->AddTensors(context, 1, &data->hwcn_weights_id);
277 }
278 ++temporaries_count;
279 }
280
281 if (is_hybrid) {
282 // Allocate tensor to store the on-the-fly quantized inputs.
283 data->input_quantized_index = temporaries_count;
284 if (data->input_quantized_id == kTensorNotAllocated) {
285 TF_LITE_ENSURE_OK(
286 context, context->AddTensors(context, 1, &data->input_quantized_id));
287 }
288 ++temporaries_count;
289
290 // Allocate tensor to store the quantization params computed during
291 // on-the-fly input quantization.
292 data->scaling_factors_index = temporaries_count;
293 if (data->scaling_factors_id == kTensorNotAllocated) {
294 TF_LITE_ENSURE_OK(
295 context, context->AddTensors(context, 1, &data->scaling_factors_id));
296 }
297 ++temporaries_count;
298
299 // Allocate tensor to store the accumulators for the matrix multiply.
300 data->accum_scratch_index = temporaries_count;
301 if (data->accum_scratch_id == kTensorNotAllocated) {
302 TF_LITE_ENSURE_OK(
303 context, context->AddTensors(context, 1, &data->accum_scratch_id));
304 }
305 ++temporaries_count;
306 if (is_per_channel) {
307 data->input_offset_index = temporaries_count;
308 if (data->input_offset_id == kTensorNotAllocated) {
309 TF_LITE_ENSURE_OK(
310 context, context->AddTensors(context, 1, &data->input_offset_id));
311 }
312 ++temporaries_count;
313
314 data->row_sums_index = temporaries_count;
315 if (data->row_sums_id == kTensorNotAllocated) {
316 TF_LITE_ENSURE_OK(context,
317 context->AddTensors(context, 1, &data->row_sums_id));
318 }
319 ++temporaries_count;
320 }
321 }
322
323 TfLiteIntArrayFree(node->temporaries);
324 node->temporaries = TfLiteIntArrayCreate(temporaries_count);
325
326 return kTfLiteOk;
327}

329TfLiteStatus Prepare(KernelType kernel_type, TfLiteContext* context,
330 TfLiteNode* node) {
331 auto* params = reinterpret_cast<TfLiteConvParams*>(node->builtin_data);
332 OpData* data = reinterpret_cast<OpData*>(node->user_data);
333
334 bool has_bias = node->inputs->size == 3;
335 // Check number of inputs/outputs
336 TF_LITE_ENSURE(context, has_bias || node->inputs->size == 2);
337 TF_LITE_ENSURE_EQ(context, node->outputs->size, 1);
338 TfLiteTensor* output;
339 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
340 const TfLiteTensor* input;
341 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
342 const TfLiteTensor* filter;
343 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 1, &filter));
344
345 // Check dimensionality of input, filter
346 TF_LITE_ENSURE_EQ(context, input->dims->size, 4);
347 TF_LITE_ENSURE_EQ(context, filter->dims->size, 4);
348 // Check input channels matching filter
349 TF_LITE_ENSURE_EQ(context, input->dims->data[3], filter->dims->data[3]);
350
351 // Check types. (We assume that UINT8 refers to quantized tensors)
352 TfLiteType input_type = input->type;
353 TF_LITE_ENSURE(context,
354 input_type == kTfLiteFloat32 || input_type == kTfLiteUInt8 ||
355 input_type == kTfLiteInt8 || input_type == kTfLiteInt16);
356 TF_LITE_ENSURE_TYPES_EQ(context, output->type, input_type);
357
358 if (input_type == kTfLiteInt16) {
359 TF_LITE_ENSURE_EQ(context, input->params.zero_point, 0);
360 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
361 }
362
363 const TfLiteTensor* bias = nullptr;
364
365 // TODO(ahentz): At this point the optimized versions require 'bias'. We can
366 // either change that or document that convolution requires it.
367 TF_LITE_ENSURE(context, has_bias);
368
369 if (has_bias) {
370 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 2, &bias));
371 if (input_type == kTfLiteUInt8 || input_type == kTfLiteInt8) {
372 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, kTfLiteInt32);
373 TF_LITE_ENSURE_EQ(context, bias->params.zero_point, 0);
374 } else if (input_type == kTfLiteInt16) {
375 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, kTfLiteInt64);
376 TF_LITE_ENSURE_EQ(context, bias->params.zero_point, 0);
377 } else {
378 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, input_type);
379 }
380 TF_LITE_ENSURE_EQ(context, NumElements(bias), SizeOfDimension(filter, 0));
381 }
382
383 const bool is_hybrid =
384 (input->type == kTfLiteFloat32 &&
385 (filter->type == kTfLiteUInt8 || filter->type == kTfLiteInt8));
386
387 if (is_hybrid && filter->type == kTfLiteInt8 &&
388 filter->quantization.type == kTfLiteAffineQuantization &&
389 filter->quantization.params &&
390 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params)
391 ->scale &&
392 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params)
393 ->scale->size > 1) {
394 const auto* affine_quantization =
395 reinterpret_cast<TfLiteAffineQuantization*>(
396 filter->quantization.params);
397 const float scale = affine_quantization->scale->data[0];
398 for (int i = 1; i < affine_quantization->scale->size; i++) {
399 if (affine_quantization->scale->data[i] != scale) {
400 data->is_hybrid_per_channel = true;
401 break;
402 }
403 }
404 }
405
406 // The multi-threaded kernel supports neither dilation nor hybrid kernels, and
407 // is incompatible with mutable input filters that might change between evals.
408 data->supports_multithreaded_kernel =
409 (kernel_type == kMultithreadOptimized) &&
410 (context->recommended_num_threads != 1) && !is_hybrid &&
411 (params->dilation_width_factor == 1) &&
412 (params->dilation_height_factor == 1) &&
413 (filter->allocation_type != kTfLiteArenaRw) && !IsDynamicTensor(filter);
414
415 int channels_in = filter->dims->data[3];
416 int channels_out = filter->dims->data[0];
417 int width = input->dims->data[2];
418 int height = input->dims->data[1];
419 int filter_width = filter->dims->data[2];
420 int filter_height = filter->dims->data[1];
421 int batches = input->dims->data[0];
422
423 // Matching GetWindowedOutputSize in TensorFlow.
424 auto padding = params->padding;
425 int out_width, out_height;
426 data->padding = ComputePaddingHeightWidth(
427 params->stride_height, params->stride_width,
428 params->dilation_height_factor, params->dilation_width_factor, height,
429 width, filter_height, filter_width, padding, &out_height, &out_width);
430
431 size_t im2col_type_size;
432 TF_LITE_ENSURE_STATUS(GetSizeOfType(context, input->type, &im2col_type_size));
433 const size_t im2col_bytes = batches * out_height * out_width * channels_in *
434 filter_height * filter_width * im2col_type_size;
435 TF_LITE_ENSURE_STATUS(AllocateTemporaryTensorsIfRequired(
436 context, node, is_hybrid, data->is_hybrid_per_channel, kernel_type,
437 im2col_bytes));
438
439 TF_LITE_ENSURE(context, has_bias);
440
441 // Note that full fixed-point inference requires that all tensors have their
442 // parameters set. This is usually done during quantized training or
443 // calibration.
444 if (input_type != kTfLiteFloat32) {
445 TF_LITE_ENSURE_EQ(context, filter->quantization.type,
446 kTfLiteAffineQuantization);
447 const auto* affine_quantization =
448 reinterpret_cast<TfLiteAffineQuantization*>(
449 filter->quantization.params);
450 TF_LITE_ENSURE(context, affine_quantization);
451 TF_LITE_ENSURE(context, affine_quantization->scale);
452 TF_LITE_ENSURE(context, (affine_quantization->scale->size == 1 ||
453 affine_quantization->scale->size == channels_out));
454
455 data->per_channel_output_multiplier.resize(channels_out);
456 data->per_channel_output_shift.resize(channels_out);
457 TF_LITE_ENSURE_STATUS(tflite::PopulateConvolutionQuantizationParams(
458 context, input, filter, bias, output, params->activation,
459 &data->output_multiplier, &data->output_shift,
460 &data->output_activation_min, &data->output_activation_max,
461 data->per_channel_output_multiplier.data(),
462 data->per_channel_output_shift.data(), channels_out));
463 }
464
465 TfLiteIntArray* output_size = TfLiteIntArrayCreate(4);
466 output_size->data[0] = batches;
467 output_size->data[1] = out_height;
468 output_size->data[2] = out_width;
469 output_size->data[3] = channels_out;
470 auto output_status = context->ResizeTensor(context, output, output_size);
471
472 if (output_status != kTfLiteOk) return output_status;
473
474 if (data->need_im2col) {
475 node->temporaries->data[data->im2col_index] = data->im2col_id;
476
477 TfLiteIntArray* im2col_size = TfLiteIntArrayCreate(4);
478
479 int input_depth = input->dims->data[3];
480 im2col_size->data[0] = output_size->data[0];
481 im2col_size->data[1] = output_size->data[1];
482 im2col_size->data[2] = output_size->data[2];
483 im2col_size->data[3] = input_depth * filter_height * filter_width;
484
485 TfLiteTensor* im2col =
486 &context->tensors[node->temporaries->data[data->im2col_index]];
487 im2col->type = input->type;
488 if (is_hybrid) {
489 im2col->type = filter->type;
490 }
491 im2col->allocation_type = kTfLiteArenaRw;
492 auto im2col_status = context->ResizeTensor(context, im2col, im2col_size);
493 if (im2col_status != kTfLiteOk) return im2col_status;
494 }
495
496 if (data->need_hwcn_weights) {
497 node->temporaries->data[data->hwcn_weights_index] = data->hwcn_weights_id;
498 TfLiteIntArray* hwcn_weights_size = TfLiteIntArrayCreate(2);
499
500 // Because we're treating the filter weights as a matrix when we do the
501 // transpose, we allocate the buffer with a two-dimensional shape, where one
502 // dimension is the number of elements in each filter, and the second is the
503 // total number of filters.
504 int input_depth = input->dims->data[3];
505 hwcn_weights_size->data[0] = (filter_height * filter_width * input_depth);
506 hwcn_weights_size->data[1] = channels_out;
507
508 TfLiteTensor* hwcn_weights =
509 &context->tensors[node->temporaries->data[data->hwcn_weights_index]];
510 hwcn_weights->type = input_type;
511 hwcn_weights->allocation_type = kTfLiteArenaRwPersistent;
512
513 auto hwcn_weights_status =
514 context->ResizeTensor(context, hwcn_weights, hwcn_weights_size);
515 if (hwcn_weights_status != kTfLiteOk) return hwcn_weights_status;
516
517 // TODO(petewarden): If Resize() is called when the size hasn't actually
518 // changed, this will do extra redundant work.
519 data->have_weights_been_transposed = false;
520 }
521
522 if (is_hybrid) {
523 node->temporaries->data[data->input_quantized_index] =
524 data->input_quantized_id;
525 TfLiteTensor* input_quantized;
526 TF_LITE_ENSURE_OK(
527 context, GetTemporarySafe(context, node, data->input_quantized_index,
528 &input_quantized));
529 input_quantized->type = kTfLiteInt8;
530 input_quantized->allocation_type = kTfLiteArenaRw;
531 if (!TfLiteIntArrayEqual(input_quantized->dims, input->dims)) {
532 TfLiteIntArray* input_quantized_size = TfLiteIntArrayCopy(input->dims);
533 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
534 input_quantized_size));
535 }
536
537 node->temporaries->data[data->scaling_factors_index] =
538 data->scaling_factors_id;
539 TfLiteTensor* scaling_factors;
540 TF_LITE_ENSURE_OK(
541 context, GetTemporarySafe(context, node, data->scaling_factors_index,
542 &scaling_factors));
543 scaling_factors->type = kTfLiteFloat32;
544 scaling_factors->allocation_type = kTfLiteArenaRw;
545 // Only one scale factor per batch is typically necessary. See optimized
546 // implementation for why we need to allocate for the height of the inputs
547 // flattened to 2D.
548 TF_LITE_ENSURE(context, channels_in != 0);
549 const int height = NumElements(input) / channels_in;
550 int scaling_dims[1] = {height};
551 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
552 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
553 scaling_factors_size->data[0] = height;
554 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
555 scaling_factors_size));
556 }
557
558 node->temporaries->data[data->accum_scratch_index] = data->accum_scratch_id;
559 TfLiteTensor* accum_scratch;
560 TF_LITE_ENSURE_OK(context,
561 GetTemporarySafe(context, node, data->accum_scratch_index,
562 &accum_scratch));
563 accum_scratch->type = kTfLiteInt32;
564 accum_scratch->allocation_type = kTfLiteArenaRw;
565 const int scratch_width = batches * out_height * out_width;
566 int accum_scratch_dims[2] = {channels_out, scratch_width};
567 if (!TfLiteIntArrayEqualsArray(accum_scratch->dims, 2,
568 accum_scratch_dims)) {
569 TfLiteIntArray* accum_scratch_size = TfLiteIntArrayCreate(2);
570 accum_scratch_size->data[0] = channels_out;
571 accum_scratch_size->data[1] = scratch_width;
572 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, accum_scratch,
573 accum_scratch_size));
574 }
575
576 if (data->is_hybrid_per_channel) {
577 const auto* affine_quantization =
578 reinterpret_cast<TfLiteAffineQuantization*>(
579 filter->quantization.params);
580 TF_LITE_ENSURE_EQ(
581 context, affine_quantization->scale->size,
582 filter->dims->data[affine_quantization->quantized_dimension]);
583 node->temporaries->data[data->input_offset_index] = data->input_offset_id;
584 TfLiteTensor* input_offsets;
585 TF_LITE_ENSURE_OK(
586 context, GetTemporarySafe(context, node, data->input_offset_index,
587 &input_offsets));
588 input_offsets->type = kTfLiteInt32;
589 input_offsets->allocation_type = kTfLiteArenaRw;
590 // See above comment for the need to allocate for height of inputs.
591 TF_LITE_ENSURE(context, channels_in != 0);
592 const int height = NumElements(input) / channels_in;
593 const int input_offset_dims[1] = {height};
594 if (!TfLiteIntArrayEqualsArray(input_offsets->dims, 1,
595 input_offset_dims)) {
596 TfLiteIntArray* input_offsets_size = TfLiteIntArrayCreate(1);
597 input_offsets_size->data[0] = input_offset_dims[0];
598 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_offsets,
599 input_offsets_size));
600 }
601 node->temporaries->data[data->row_sums_index] = data->row_sums_id;
602 TfLiteTensor* row_sums;
603 TF_LITE_ENSURE_OK(
604 context,
605 GetTemporarySafe(context, node, data->row_sums_index, &row_sums));
606 row_sums->type = kTfLiteInt32;
607 row_sums->allocation_type = kTfLiteArenaRwPersistent;
608 // See above comment for the need to allocate for height of inputs.
609 const int row_sums_dims[1] = {channels_out};
610 if (!TfLiteIntArrayEqualsArray(row_sums->dims, 1, row_sums_dims)) {
611 TfLiteIntArray* row_sums_size = TfLiteIntArrayCreate(1);
612 row_sums_size->data[0] = row_sums_dims[0];
613 TF_LITE_ENSURE_OK(
614 context, context->ResizeTensor(context, row_sums, row_sums_size));
615 }
616 }
617 }
618 return kTfLiteOk;
619}

626template <KernelType kernel_type>
627void EvalQuantized(TfLiteContext* context, TfLiteNode* node,
628 TfLiteConvParams* params, OpData* data,
629 const TfLiteTensor* input, const TfLiteTensor* filter,
630 const TfLiteTensor* bias, TfLiteTensor* im2col,
631 TfLiteTensor* output) {
632 auto input_offset = -input->params.zero_point;
633 auto filter_offset = -filter->params.zero_point;
634 auto output_offset = output->params.zero_point;
635
636 KernelType effective_kernel_type;
637 if ((kernel_type == kMultithreadOptimized ||
638 kernel_type == kCblasOptimized) &&
639 (params->dilation_width_factor != 1 ||
640 params->dilation_height_factor != 1)) {
641 // kMultithreadOptimized and kCblasOptimized do not support dilation.
642 // Therefore, fallback to optimized.
643 effective_kernel_type = kGenericOptimized;
644 } else {
645 effective_kernel_type = kernel_type;
646 }
647
648 // We have to fallback to reference execution path when im2col is needed but
649 // disabled because to-be-allocated temporary im2col tensor is too large.
650 // See b/178743262 for the detailed motivation.
651 if (data->im2col_oversized) {
652 effective_kernel_type = kReference;
653 }
654
655 ConvParams op_params;
656 op_params.padding_type = PaddingType::kSame;
657 op_params.padding_values.width = data->padding.width;
658 op_params.padding_values.height = data->padding.height;
659 op_params.stride_width = params->stride_width;
660 op_params.stride_height = params->stride_height;
661 op_params.dilation_width_factor = params->dilation_width_factor;
662 op_params.dilation_height_factor = params->dilation_height_factor;
663 op_params.input_offset = input_offset;
664 op_params.weights_offset = filter_offset;
665 op_params.output_offset = output_offset;
666 op_params.output_multiplier = data->output_multiplier;
667 op_params.output_shift = -data->output_shift;
668 op_params.quantized_activation_min = data->output_activation_min;
669 op_params.quantized_activation_max = data->output_activation_max;
670 switch (effective_kernel_type) {
671 case kReference: {
672 reference_ops::Conv(
673 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
674 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
675 GetTensorShape(bias), GetTensorData<int32_t>(bias),
676 GetTensorShape(output), GetTensorData<uint8_t>(output),
677 GetTensorShape(im2col), GetTensorData<uint8_t>(im2col),
678 /* cpu_backend_context = */ nullptr);
679 break;
680 }
681 case kGenericOptimized:
682 case kMultithreadOptimized:
683 case kCblasOptimized: {
684 // There is only one optimized implementation for Quantized Conv.
685 optimized_ops::Conv(
686 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
687 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
688 GetTensorShape(bias), GetTensorData<int32_t>(bias),
689 GetTensorShape(output), GetTensorData<uint8_t>(output),
690 GetTensorShape(im2col), GetTensorData<uint8_t>(im2col),

692 break;
693 }

788template <KernelType kernel_type>
789void EvalFloat(TfLiteContext* context, TfLiteNode* node,
790 TfLiteConvParams* params, OpData* data,
791 const TfLiteTensor* input, const TfLiteTensor* filter,
792 const TfLiteTensor* bias, TfLiteTensor* im2col,
793 TfLiteTensor* hwcn_weights, TfLiteTensor* output) {
794 float output_activation_min, output_activation_max;
795 CalculateActivationRange(params->activation, &output_activation_min,
796 &output_activation_max);
797 KernelType effective_kernel_type = kernel_type;
798 // Fall back to the optimized path if multi-threaded conv is unsupported.
799 if ((kernel_type == kMultithreadOptimized) &&
800 !data->supports_multithreaded_kernel) {
801 effective_kernel_type = kGenericOptimized;
802 }
803
804 // When im2col is needed (which is implied when 'im2col_oversized' is true),
805 // the GEMMM-based optimized path requires im2col data be allocated to ensure
806 // the correctness. Therefore, when im2col is disabled because of the
807 // oversized temporary im2col tensor, fallback to a non-optimized path is
808 // needed.
809 // See b/178743262 for the detailed motivation.
810 if (data->im2col_oversized) {
811 effective_kernel_type = kReference;
812#if defined(TFLITE_WITH_MULTITHREADED_EIGEN)
813 // As detailed by tflite::multithreaded_ops::Conv implementation in
814 // multithreaded_conv.h, the Eigen-based execution doesn't need im2col data.
815 // Therefore, we could rely on it as a better-optimized fallback than the
816 // reference one.
817 if (data->supports_multithreaded_kernel) {
818 effective_kernel_type = kMultithreadOptimized;
819 }
820#endif
821 }
822
823 ConvParams op_params;
824 op_params.padding_type = RuntimePaddingType(params->padding);
825 op_params.padding_values.width = data->padding.width;
826 op_params.padding_values.height = data->padding.height;
827 op_params.stride_width = params->stride_width;
828 op_params.stride_height = params->stride_height;
829 op_params.dilation_width_factor = params->dilation_width_factor;
830 op_params.dilation_height_factor = params->dilation_height_factor;
831 op_params.float_activation_min = output_activation_min;
832 op_params.float_activation_max = output_activation_max;
833 switch (effective_kernel_type) {
834 case kReference: {
835 reference_ops::Conv(op_params, GetTensorShape(input),
836 GetTensorData<float>(input), GetTensorShape(filter),
837 GetTensorData<float>(filter), GetTensorShape(bias),
838 GetTensorData<float>(bias), GetTensorShape(output),
839 GetTensorData<float>(output), GetTensorShape(im2col),
840 GetTensorData<float>(im2col));
841 break;
842 }
843 case kCblasOptimized:
844 case kGenericOptimized: {
845 optimized_ops::Conv(op_params, GetTensorShape(input),
846 GetTensorData<float>(input), GetTensorShape(filter),
847 GetTensorData<float>(filter), GetTensorShape(bias),
848 GetTensorData<float>(bias), GetTensorShape(output),
849 GetTensorData<float>(output), GetTensorShape(im2col),
850 GetTensorData<float>(im2col),
851 CpuBackendContext::GetFromContext(context));
852 break;
853 }
854 case kMultithreadOptimized: {
855#if defined(TFLITE_WITH_MULTITHREADED_EIGEN)
856 const float* filter_data;
857 if (data->need_hwcn_weights) {
858 filter_data = GetTensorData<float>(hwcn_weights);
859 } else {
860 filter_data = GetTensorData<float>(filter);
861 }
862 multithreaded_ops::Conv(
863 *eigen_support::GetThreadPoolDevice(context), op_params,
864 GetTensorShape(input), GetTensorData<float>(input),
865 GetTensorShape(filter), filter_data, GetTensorShape(bias),
866 GetTensorData<float>(bias), GetTensorShape(output),
867 GetTensorData<float>(output), GetTensorShape(im2col),
868 GetTensorData<float>(im2col));
869 break;
870#else // !defined(TFLITE_WITH_MULTITHREADED_EIGEN)
871 // See Register_CONV_2D: we should never be here when TFLITE_WITH_RUY
872 // was enabled. We #if out this code in order to get the corresponding
873 // binary size benefits.
874 TFLITE_DCHECK(false);
875#endif // defined(TFLITE_WITH_MULTITHREADED_EIGEN)
876 }
877 }

880template <KernelType kernel_type>
881TfLiteStatus EvalHybridPerChannel(TfLiteContext* context, TfLiteNode* node,
882 TfLiteConvParams* params, OpData* data,
883 const TfLiteTensor* input,
884 const TfLiteTensor* filter,
885 const TfLiteTensor* bias,
886 TfLiteTensor* im2col, TfLiteTensor* output) {
887 float output_activation_min, output_activation_max;
888 CalculateActivationRange(params->activation, &output_activation_min,
889 &output_activation_max);
890
891 const int batch_size = SizeOfDimension(input, 0);
892 TF_LITE_ENSURE(context, batch_size != 0);
893 const int input_size = NumElements(input) / batch_size;
894 TfLiteTensor* quantized_input_tensor;
895 TF_LITE_ENSURE_OK(context,
896 GetTemporarySafe(context, node, data->input_quantized_index,
897 &quantized_input_tensor));
898 int8_t* quantized_input_ptr_batch =
899 GetTensorData<int8_t>(quantized_input_tensor);
900 TfLiteTensor* scaling_factors_tensor;
901 TF_LITE_ENSURE_OK(context,
902 GetTemporarySafe(context, node, data->scaling_factors_index,
903 &scaling_factors_tensor));
904 float* scaling_factors_ptr = GetTensorData<float>(scaling_factors_tensor);
905 TfLiteTensor* input_offset_tensor;
906 TF_LITE_ENSURE_OK(context,
907 GetTemporarySafe(context, node, data->input_offset_index,
908 &input_offset_tensor));
909 int32_t* input_offset_ptr = GetTensorData<int32_t>(input_offset_tensor);
910
911 for (int b = 0; b < batch_size; ++b) {
912 const int offset = b * input_size;
913 tensor_utils::AsymmetricQuantizeFloats(
914 GetTensorData<float>(input) + offset, input_size,
915 quantized_input_ptr_batch + offset, &scaling_factors_ptr[b],
916 &input_offset_ptr[b]);
917 }
918
919 int8_t* im2col_ptr = nullptr;
920 int8_t* filter_ptr = nullptr;
921 if (im2col != nullptr) {
922 im2col_ptr = im2col->data.int8;
923 }
924 filter_ptr = filter->data.int8;
925 const auto* affine_quantization =
926 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params);
927
928 KernelType effective_kernel_type = kernel_type;
929 // We have to fallback to reference execution path when im2col is needed but
930 // disabled because to-be-allocated temporary im2col tensor is too large.
931 // See b/178743262 for the detailed motivation.
932 if (data->im2col_oversized) {
933 effective_kernel_type = kReference;
934 }
935
936 ConvParams op_params;
937 op_params.padding_type = PaddingType::kSame;
938 op_params.padding_values.width = data->padding.width;
939 op_params.padding_values.height = data->padding.height;
940 op_params.stride_width = params->stride_width;
941 op_params.stride_height = params->stride_height;
942 op_params.dilation_width_factor = 1;
943 op_params.dilation_height_factor = 1;
944 op_params.float_activation_min = output_activation_min;
945 op_params.float_activation_max = output_activation_max;
946 switch (effective_kernel_type) {
947 case kReference:
948 reference_ops::HybridConvPerChannel(
949 op_params, scaling_factors_ptr, GetTensorShape(input),
950 quantized_input_ptr_batch, GetTensorShape(filter), filter_ptr,
951 GetTensorShape(bias), GetTensorData<float>(bias),
952 GetTensorShape(output), GetTensorData<float>(output),
953 GetTensorShape(im2col), im2col_ptr, affine_quantization->scale->data,
954 input_offset_ptr);
955 break;
956 case kGenericOptimized:
957 case kMultithreadOptimized:
958 case kCblasOptimized: {
959 TfLiteTensor* row_sums;
960 TF_LITE_ENSURE_OK(
961 context,
962 GetTemporarySafe(context, node, data->row_sums_index, &row_sums));
963 TfLiteTensor* scratch;
964 TF_LITE_ENSURE_OK(
965 context,
966 GetTemporarySafe(context, node, data->accum_scratch_index, &scratch));
967 optimized_ops::HybridConvPerChannel(
968 op_params, scaling_factors_ptr, GetTensorShape(input),
969 quantized_input_ptr_batch, GetTensorShape(filter), filter_ptr,
970 GetTensorShape(bias), GetTensorData<float>(bias),
971 GetTensorShape(output), GetTensorData<float>(output),
972 GetTensorShape(im2col), im2col_ptr, affine_quantization->scale->data,
973 input_offset_ptr, GetTensorShape(scratch),
974 GetTensorData<int32>(scratch), GetTensorData<int32_t>(row_sums),
975 &data->compute_hybrid_row_sums,
976 CpuBackendContext::GetFromContext(context));
977 data->compute_hybrid_row_sums = false;
978 break;
979 }
980 }
981
982 return kTfLiteOk;
983}

985template <KernelType kernel_type>
986TfLiteStatus EvalHybrid(TfLiteContext* context, TfLiteNode* node,
987 TfLiteConvParams* params, OpData* data,
988 const TfLiteTensor* input, const TfLiteTensor* filter,
989 const TfLiteTensor* bias, TfLiteTensor* im2col,
990 TfLiteTensor* accum_scratch, TfLiteTensor* output) {

992 CalculateActivationRange(params->activation, &output_activation_min,
993 &output_activation_max);
994
995 const int batch_size = SizeOfDimension(input, 0);
996 TF_LITE_ENSURE(context, batch_size != 0);
997 const int input_size = NumElements(input) / batch_size;
998
999 const float* input_ptr = GetTensorData<float>(input);
1000 TfLiteTensor* quantized_input_tensor;
1001 TF_LITE_ENSURE_OK(context,
1002 GetTemporarySafe(context, node, data->input_quantized_index,

1013 {
1014 ruy::profiler::ScopeLabel label("ConvHybridQuantizeInputs");
1015 for (int b = 0; b < batch_size; ++b) {
1016 float unused_min, unused_max;
1017 const int offset = b * input_size;
1018 tensor_utils::SymmetricQuantizeFloats(
1019 input_ptr + offset, input_size, quantized_input_ptr_batch + offset,
1020 &unused_min, &unused_max, &scaling_factors_ptr[b]);
1021 scaling_factors_ptr[b] *= filter->params.scale;

1023 }

1025 switch (kernel_type) {

1038 op_params.dilation_height_factor = params->dilation_height_factor;

1042 op_params, scaling_factors_ptr, GetTensorShape(input),
1043 quantized_input_ptr_batch, GetTensorShape(filter),
1044 GetTensorData<int8_t>(filter), GetTensorShape(bias),
1045 GetTensorData<float>(bias), GetTensorShape(accum_scratch),
1046 GetTensorData<int32_t>(accum_scratch), GetTensorShape(output),
1047 GetTensorData<float>(output), GetTensorShape(im2col),
1048 GetTensorData<int8_t>(im2col),
1049 CpuBackendContext::GetFromContext(context));
1050 break;
1051 }
1052 }

1054 return kTfLiteOk;

1057template <KernelType kernel_type, TfLiteType input_type>
1058TfLiteStatus EvalImpl(TfLiteContext* context, TfLiteNode* node) {
1059 auto* params = reinterpret_cast<TfLiteConvParams*>(node->builtin_data);
1060 OpData* data = reinterpret_cast<OpData*>(node->user_data);
1061
1062 TfLiteTensor* output;
1063 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
1064 const TfLiteTensor* input;
1065 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
1066 const TfLiteTensor* filter;
1067 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 1, &filter));
1068 bool has_bias = node->inputs->size == 3;
1069 const TfLiteTensor* bias = has_bias ? GetInput(context, node, 2) : nullptr;
1070 TfLiteTensor* im2col =
1071 data->need_im2col
1072 ? &context->tensors[node->temporaries->data[data->im2col_index]]
1073 : nullptr;
1074 TfLiteTensor* hwcn_weights =
1075 data->need_hwcn_weights
1076 ? &context->tensors[node->temporaries->data[data->hwcn_weights_index]]
1077 : nullptr;
1078
1079 if (data->need_hwcn_weights && !data->have_weights_been_transposed) {
1080 TransposeFloatTensor(filter, hwcn_weights);
1081 data->have_weights_been_transposed = true;
1082 }
1083
1084 TFLITE_DCHECK_EQ(input_type, input->type);
1085 switch (input_type) { // Already know in/outtypes are same.
1086 case kTfLiteFloat32:
1087 if (filter->type == kTfLiteUInt8 || filter->type == kTfLiteInt8) {
1088 if (data->is_hybrid_per_channel) {
1089 TF_LITE_ENSURE_OK(context, EvalHybridPerChannel<kernel_type>(
1090 context, node, params, data, input,
1091 filter, bias, im2col, output));
1092 } else {
1093 TfLiteTensor* accum_scratch =
1094 &context->tensors[node->temporaries
1095 ->data[data->accum_scratch_index]];
1096 TF_LITE_ENSURE_OK(context,
1097 EvalHybrid<kernel_type>(context, node, params, data,
1098 input, filter, bias, im2col,
1099 accum_scratch, output));
1100 }
1101 } else {
1102 EvalFloat<kernel_type>(context, node, params, data, input, filter, bias,
1103 im2col, hwcn_weights, output);
1104 }
1105 break;
1106 case kTfLiteUInt8:
1107 EvalQuantized<kernel_type>(context, node, params, data, input, filter,
1108 bias, im2col, output);
1109 break;
1110 case kTfLiteInt8:
1111 EvalQuantizedPerChannel<kernel_type>(context, node, params, data, input,
1112 filter, bias, output, im2col);
1113 break;
1114 case kTfLiteInt16:
1115 EvalQuantizedPerChannel16x8<kernel_type>(
1116 context, node, params, data, input, filter, bias, output, im2col);
1117 break;
1118 default:
1119 TF_LITE_KERNEL_LOG(context, "Type %s currently not supported.",
1120 TfLiteTypeGetName(input->type));
1121 return kTfLiteError;
1122 }
1123 return kTfLiteOk;
1124}

1127TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
1128 const TfLiteTensor* input;
1129 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
1130
1131 switch (input->type) {
1132 case kTfLiteFloat32:
1133 return EvalImpl<kernel_type, kTfLiteFloat32>(context, node);
1134 case kTfLiteUInt8:
1135 return EvalImpl<kernel_type, kTfLiteUInt8>(context, node);
1136 case kTfLiteInt8:
1137 return EvalImpl<kernel_type, kTfLiteInt8>(context, node);
1138 case kTfLiteInt16:
1139 return EvalImpl<kernel_type, kTfLiteInt16>(context, node);
1140 default:
1141 TF_LITE_KERNEL_LOG(context, "Type %s not currently supported.",
1142 TfLiteTypeGetName(input->type));
1143 return kTfLiteError;
1144 }
