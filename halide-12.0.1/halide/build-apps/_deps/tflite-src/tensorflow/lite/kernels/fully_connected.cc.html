44bool SupportedSparsityFormat(const TfLiteSparsity& sparsity) {
45 if (sparsity.dim_metadata[0].format == kTfLiteDimDense &&
46 sparsity.dim_metadata[1].format == kTfLiteDimSparseCSR) {
47 return true;
48 }
49
50 return false;
51}

54static const int kDimMetadataSizeBlockSparse = 3;

56TfLiteStatus CreateLedgerTensor(const TfLiteSparsity* sparsity,
57 TfLiteContext* context, TfLiteTensor* ledger) {
58 TF_LITE_ENSURE(context, sparsity != nullptr);
59 ledger->type = kTfLiteUInt8;
60 ledger->allocation_type = kTfLiteArenaRwPersistent;
61 TfLiteIntArray* ledger_size = TfLiteIntArrayCreate(1);
62 ledger_size->data[0] = sparsity->dim_metadata[1].array_indices->size +
63 sparsity->dim_metadata[1].array_segments->size - 1;
64 return context->ResizeTensor(context, ledger, ledger_size);
65}
66
67TfLiteStatus PopulateLedgerData(const TfLiteSparsity* sparsity,
68 TfLiteContext* context, uint8_t* ledger_data) {
69 TF_LITE_ENSURE(context, sparsity != nullptr);
70 const auto* array_segments = sparsity->dim_metadata[1].array_segments;
71 const auto* array_indices = sparsity->dim_metadata[1].array_indices;
72 int output_data_ptr = 0;
73
74 for (int i = 0; i < array_segments->size - 1; i++) {
75 int row_start = array_segments->data[i];
76 int row_end = array_segments->data[i + 1];
77 if (row_end - row_start > UINT8_MAX) {
78 return kTfLiteError;
79 }
80 // Copy num of non-zero blocks in row i.
81 ledger_data[output_data_ptr] = static_cast<uint8_t>(row_end - row_start);
82 output_data_ptr++;
83
84 for (int j = row_start; j < row_end; j++) {
85 if (array_indices->data[j] > UINT8_MAX) {
86 return kTfLiteError;
87 }
88 // Copy indices of non-zero blocks in row i.
89 ledger_data[output_data_ptr] =
90 static_cast<uint8_t>(array_indices->data[j]);
91 output_data_ptr++;
92 }
93 }
94 return kTfLiteOk;
95}

106struct OpData {
107 // The scaling factor from input to output (aka the 'real multiplier') can
108 // be represented as a fixed point multiplier plus a left shift.
109 int32_t output_multiplier;
110 int output_shift;
111 // The range of the fused activation layer. For example for kNone and
112 // uint8_t these would be 0 and 255.
113 int32_t output_activation_min;
114 int32_t output_activation_max;
115 // The index of the temporary tensor where the quantized inputs are cached.
116 int scratch_tensor_index;
117 bool compute_row_sums = false;
118 // Only used for sparse hybrid fully connected kernels.
119 bool ledger_initialized;
120};

122constexpr int kInputTensor = 0;
123constexpr int kWeightsTensor = 1;
124constexpr int kBiasTensor = 2;
125constexpr int kOutputTensor = 0;
126constexpr int kShuffledInputWorkspaceTensor = 1;

128inline TfLiteStatus CheckTypes(TfLiteContext* context,
129 const TfLiteTensor* input,
130 const TfLiteTensor* filter,
131 const TfLiteTensor* bias, TfLiteTensor* output,
132 TfLiteFullyConnectedParams* params) {
133 const bool is_quantized =
134 ((filter->type == kTfLiteUInt8) || (filter->type == kTfLiteInt8));
135 const bool is_hybrid = is_quantized && (input->type == kTfLiteFloat32);
136 const bool is_shuffled =
137 is_quantized && (params->weights_format ==
138 kTfLiteFullyConnectedWeightsFormatShuffled4x16Int8);
139
140 // optional bias tensor.
141 const bool is_optional_bias_float = !bias || (bias->type == kTfLiteFloat32);
142 const bool is_optional_bias_int =
143 !bias || (bias->type == kTfLiteInt32) || (bias->type == kTfLiteInt64);
144
145 if (is_quantized) {
146 if (is_shuffled) {
147 TF_LITE_ENSURE_TYPES_EQ(context, input->type, kTfLiteUInt8);
148 TF_LITE_ENSURE_TYPES_EQ(context, filter->type, kTfLiteUInt8);
149 TF_LITE_ENSURE_TYPES_EQ(context, output->type, kTfLiteInt16);
150 TF_LITE_ENSURE_EQ(context, is_optional_bias_int, true);
151 } else if (is_hybrid) {
152 TF_LITE_ENSURE_TYPES_EQ(context, input->type, kTfLiteFloat32);
153 TF_LITE_ENSURE_TYPES_EQ(context, output->type, kTfLiteFloat32);
154 TF_LITE_ENSURE_EQ(context, is_optional_bias_float, true);
155 } else {
156 TF_LITE_ENSURE(context, input->type == kTfLiteUInt8 ||
157 input->type == kTfLiteInt8 ||
158 input->type == kTfLiteInt16);
159 TF_LITE_ENSURE(context, output->type == kTfLiteUInt8 ||
160 output->type == kTfLiteInt8 ||
161 output->type == kTfLiteInt16);
162 TF_LITE_ENSURE_EQ(context, is_optional_bias_int, true);
163 }
164 } else {
165 // Only float32 is supported currently
166 TF_LITE_ENSURE_TYPES_EQ(context, input->type, kTfLiteFloat32);
167 TF_LITE_ENSURE_TYPES_EQ(context, output->type, kTfLiteFloat32);
168 TF_LITE_ENSURE_TYPES_EQ(context, filter->type, kTfLiteFloat32);
169 TF_LITE_ENSURE_EQ(context, is_optional_bias_float, true);
170 }
171
172 return kTfLiteOk;
173}

175void* Init(TfLiteContext* context, const char* buffer, size_t length) {
176 // This is a builtin op, so we don't use the contents in 'buffer', if any.
177 // Instead, we allocate a new object to carry information from Prepare() to
178 // Eval().
179 auto* op_data = new OpData();
180 context->AddTensors(context, /*tensors_to_add=*/6,
181 &op_data->scratch_tensor_index);
182 return op_data;
183}

186 delete reinterpret_cast<OpData*>(buffer);

189TfLiteStatus PrepareImpl(TfLiteContext* context, TfLiteNode* node) {
190 auto* params =
191 reinterpret_cast<TfLiteFullyConnectedParams*>(node->builtin_data);
192 OpData* data = reinterpret_cast<OpData*>(node->user_data);
193 // Check we have all the inputs and outputs we need.
194 TF_LITE_ENSURE(context, node->inputs->size == 2 || node->inputs->size == 3);
195 // Shuffled formats need a workspace to store the shuffled input activations.
196 const int expected_outputs_count =
197 params->weights_format == kTfLiteFullyConnectedWeightsFormatDefault ? 1
198 : 2;
199 TF_LITE_ENSURE_EQ(context, node->outputs->size, expected_outputs_count);
200
201 const TfLiteTensor* input;
202 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
203 const TfLiteTensor* filter;
204 TF_LITE_ENSURE_OK(context,
205 GetInputSafe(context, node, kWeightsTensor, &filter));
206 const TfLiteTensor* bias =
207 (node->inputs->size == 3)
208 ? GetOptionalInputTensor(context, node, kBiasTensor)
209 : nullptr;
210 TfLiteTensor* output;
211 TF_LITE_ENSURE_OK(context,
212 GetOutputSafe(context, node, kOutputTensor, &output));
213
214 // Check proper datatype match among all Input Tensors
215 TF_LITE_ENSURE_STATUS(
216 CheckTypes(context, input, filter, bias, output, params));
217
218 // Check all the parameters of tensor match within themselves and match the
219 // input configuration.
220 int input_size = 1;
221 for (int i = 0; i < input->dims->size; i++) {
222 input_size *= input->dims->data[i];
223 }
224
225 TF_LITE_ENSURE_EQ(context, NumDimensions(filter), 2);
226 const int batch_size = input_size / filter->dims->data[1];
227 const int num_units = filter->dims->data[0];
228
229 if (bias) {
230 TF_LITE_ENSURE_EQ(context, NumElements(bias), SizeOfDimension(filter, 0));
231 }
232
233 // Note that quantized inference requires that all tensors have their
234 // parameters set. This is usually done during quantized training.
235 if (input->type == kTfLiteUInt8 || input->type == kTfLiteInt8 ||
236 input->type == kTfLiteInt16) {
237 double real_multiplier = 0.0;
238 TF_LITE_ENSURE_STATUS(GetQuantizedConvolutionMultipler(
239 context, input, filter, bias, output, &real_multiplier));
240 int exponent;
241 QuantizeMultiplier(real_multiplier, &data->output_multiplier, &exponent);
242 data->output_shift = exponent;
243 TF_LITE_ENSURE_STATUS(CalculateActivationRangeQuantized(
244 context, params->activation, output, &data->output_activation_min,
245 &data->output_activation_max));
246 }
247
248 if (input->type == kTfLiteInt16 && output->type == kTfLiteInt16) {
249 TF_LITE_ENSURE_EQ(context, input->params.zero_point, 0);
250 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
251 }
252
253 // If we have to perform on-the-fly quantization (with quantized weights and
254 // float inputs) first we need to quantize the inputs. Allocate a temporary
255 // buffer to store the intermediate quantized values.
256 // Additionally, we allocate a temporary buffer to store the accumulated
257 // quantized values prior to multiplication by the scaling factor.
258 const bool is_hybrid =
259 (input->type == kTfLiteFloat32 &&
260 (filter->type == kTfLiteUInt8 || filter->type == kTfLiteInt8));
261 const bool is_sparse = filter->sparsity != nullptr;
262 if (is_hybrid) {
263 TfLiteIntArrayFree(node->temporaries);
264 data->compute_row_sums = true;
265 if (is_sparse) {
266 node->temporaries = TfLiteIntArrayCreate(6);
267 } else {
268 node->temporaries = TfLiteIntArrayCreate(5);
269 }
270 node->temporaries->data[0] = data->scratch_tensor_index;
271
272 TfLiteTensor* input_quantized;
273 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/0,
274 &input_quantized));
275 input_quantized->type = filter->type;
276 input_quantized->allocation_type = kTfLiteArenaRw;
277
278 TfLiteIntArray* input_quantized_size = TfLiteIntArrayCopy(input->dims);
279 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
280 input_quantized_size));
281
282 node->temporaries->data[1] = data->scratch_tensor_index + 1;
283 TfLiteTensor* scaling_factors;
284 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
285 &scaling_factors));
286 scaling_factors->type = kTfLiteFloat32;
287 scaling_factors->allocation_type = kTfLiteArenaRw;
288
289 int scaling_dims[1] = {batch_size};
290 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
291 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
292 scaling_factors_size->data[0] = batch_size;
293 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
294 scaling_factors_size));
295 }
296
297 node->temporaries->data[2] = data->scratch_tensor_index + 2;
298 TfLiteTensor* accum_scratch;
299 TF_LITE_ENSURE_OK(
300 context, GetTemporarySafe(context, node, /*index=*/2, &accum_scratch));
301 accum_scratch->type = kTfLiteInt32;
302 accum_scratch->allocation_type = kTfLiteArenaRw;
303 int accum_scratch_dims[2] = {num_units, batch_size};
304 if (!TfLiteIntArrayEqualsArray(accum_scratch->dims, 2,
305 accum_scratch_dims)) {
306 TfLiteIntArray* accum_size = TfLiteIntArrayCreate(2);
307 accum_size->data[0] = num_units;
308 accum_size->data[1] = batch_size;
309 TF_LITE_ENSURE_OK(
310 context, context->ResizeTensor(context, accum_scratch, accum_size));
311 }
312
313 node->temporaries->data[3] = data->scratch_tensor_index + 3;
314 TfLiteTensor* input_offsets;
315 TF_LITE_ENSURE_OK(
316 context, GetTemporarySafe(context, node, /*index=*/3, &input_offsets));
317 input_offsets->type = kTfLiteInt32;
318 input_offsets->allocation_type = kTfLiteArenaRw;
319 if (!TfLiteIntArrayEqualsArray(input_offsets->dims, 1, scaling_dims)) {
320 TfLiteIntArray* input_offsets_size = TfLiteIntArrayCreate(1);
321 input_offsets_size->data[0] = batch_size;
322 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_offsets,
323 input_offsets_size));
324 }
325 node->temporaries->data[4] = data->scratch_tensor_index + 4;
326 TfLiteTensor* row_sums;
327 TF_LITE_ENSURE_OK(context,
328 GetTemporarySafe(context, node, /*index=*/4, &row_sums));
329 row_sums->type = kTfLiteInt32;
330 row_sums->allocation_type = kTfLiteArenaRwPersistent;
331 int row_sums_dims[1] = {num_units};
332 if (!TfLiteIntArrayEqualsArray(row_sums->dims, 1, row_sums_dims)) {
333 TfLiteIntArray* row_sums_size = TfLiteIntArrayCreate(1);
334 row_sums_size->data[0] = row_sums_dims[0];
335 TF_LITE_ENSURE_OK(
336 context, context->ResizeTensor(context, row_sums, row_sums_size));
337 }
338
339 if (is_sparse) {
340 data->ledger_initialized = false;
341 node->temporaries->data[5] = data->scratch_tensor_index + 5;
342 TfLiteTensor* filter_ledger =
343 &context->tensors[node->temporaries->data[5]];
344 auto status =
345 CreateLedgerTensor(filter->sparsity, context, filter_ledger);
346 if (status != kTfLiteOk) return status;
347 }
348 }
349
350 // Resize output.
351 TfLiteIntArray* output_size_array = nullptr;
352 if (params->keep_num_dims) {
353 // When number of dimensions are kept the filter operates along the last
354 // dimensions. In other words, for an input tensor with shape
355 // [batch_size, ..., n_inputs] and a filter of shape [n_inputs, n_units]
356 // this Op produces an output of shape [batch_size, ..., n_units].
357 TF_LITE_ENSURE_EQ(context, input->dims->data[input->dims->size - 1],
358 SizeOfDimension(filter, 1));
359 output_size_array = TfLiteIntArrayCopy(input->dims);
360 output_size_array->data[output_size_array->size - 1] = num_units;
361 } else {
362 // Otherwise, the output is (potentially flattened to) a 2-D matrix.
363 output_size_array = TfLiteIntArrayCreate(2);
364 output_size_array->data[0] = batch_size;
365 output_size_array->data[1] = num_units;
366 }
367 TF_LITE_ENSURE_OK(context,
368 context->ResizeTensor(context, output, output_size_array));
369
370 return kTfLiteOk;
371}

373template <KernelType kernel_type>
374TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
375 // Check for supported activation types.
376 auto* params =
377 reinterpret_cast<TfLiteFullyConnectedParams*>(node->builtin_data);
378 const TfLiteTensor* filter;
379 TF_LITE_ENSURE_OK(context,
380 GetInputSafe(context, node, kWeightsTensor, &filter));
381 const TfLiteTensor* input;
382 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
383 const bool is_quantized =
384 ((filter->type == kTfLiteUInt8) || (filter->type == kTfLiteInt8));
385 const bool is_hybrid = is_quantized && (input->type == kTfLiteFloat32);
386 const bool is_pie = kernel_type == kLegacyPie;
387
388 // Pie and hybrid path supports all kinds of fused activations, otherwise only
389 // clipping activations are supported.
390 if (!is_pie && !is_hybrid) {
391 TF_LITE_ENSURE(context, params->activation == kTfLiteActNone ||
392 params->activation == kTfLiteActRelu ||
393 params->activation == kTfLiteActReluN1To1 ||
394 params->activation == kTfLiteActRelu6);
395 }
396 return PrepareImpl(context, node);
397}

399TfLiteStatus EvalPie(TfLiteContext* context, TfLiteNode* node,
400 TfLiteFullyConnectedParams* params, OpData* data,
401 const TfLiteTensor* input, const TfLiteTensor* filter,
402 const TfLiteTensor* bias, TfLiteTensor* output) {
403 int total_input_size = 1;
404 for (int i = 0; i < input->dims->size; i++) {
405 total_input_size *= input->dims->data[i];
406 }
407
408 int input_size = filter->dims->data[1];
409 const int batch_size = total_input_size / filter->dims->data[1];
410 const int num_units = filter->dims->data[0];
411
412 // Output = bias if bias tensor exists.
413 if (bias) {
414 tensor_utils::VectorBatchVectorAssign(GetTensorData<float>(bias), num_units,
415 batch_size,
416 GetTensorData<float>(output));
417 } else {
418 std::fill_n(GetTensorData<float>(output), batch_size * num_units, 0.0f);
419 }
420
421 // Compute output += weight * input
422 tensor_utils::MatrixBatchVectorMultiplyAccumulate(
423 GetTensorData<float>(filter), num_units, input_size,
424 GetTensorData<float>(input), batch_size, GetTensorData<float>(output));
425
426 // Apply activation function
427 tensor_utils::ApplyActivationToVector(
428 GetTensorData<float>(output), batch_size * num_units, params->activation,
429 GetTensorData<float>(output));
430
431 return kTfLiteOk;
432}

434TfLiteStatus EvalHybridDense(
435 TfLiteContext* context, TfLiteNode* node,
436 TfLiteFullyConnectedParams* params, OpData* data, const TfLiteTensor* input,
437 const TfLiteTensor* filter, const TfLiteTensor* bias,
438 TfLiteTensor* input_quantized, TfLiteTensor* scaling_factors,
439 TfLiteTensor* accum_scratch, TfLiteTensor* row_sums,
440 TfLiteTensor* input_offsets, TfLiteTensor* output) {
441 int total_input_size = 1;
442 for (int i = 0; i < input->dims->size; i++) {
443 total_input_size *= input->dims->data[i];
444 }
445
446 const int input_size = filter->dims->data[1];
447 const int batch_size = total_input_size / filter->dims->data[1];
448 const int num_units = filter->dims->data[0];
449
450 // Output = bias if bias tensor exists.
451 if (bias) {
452 tensor_utils::VectorBatchVectorAssign(GetTensorData<float>(bias), num_units,
453 batch_size,
454 GetTensorData<float>(output));
455 } else {
456 std::fill_n(GetTensorData<float>(output), batch_size * num_units, 0.0f);
457 }
458
459 // Save matrix multiplication computation for all zero input.
460 if (tensor_utils::IsZeroVector(GetTensorData<float>(input),
461 total_input_size)) {
462 tensor_utils::ApplyActivationToVector(
463 GetTensorData<float>(output), batch_size * num_units,
464 params->activation, GetTensorData<float>(output));
465 return kTfLiteOk;
466 }
467
468 // Quantize input from float to uint8 + quantization params (scaling factor).
469 float* scaling_factors_ptr = GetTensorData<float>(scaling_factors);
470 int32_t* input_offset_ptr = nullptr;
471 int32_t* row_sums_ptr = nullptr;
472 if (params->asymmetric_quantize_inputs) {
473 input_offset_ptr = GetTensorData<int32_t>(input_offsets);
474 row_sums_ptr = GetTensorData<int32_t>(row_sums);
475 }
476 int8_t* quant_data = GetTensorData<int8_t>(input_quantized);
477 const int8_t* filter_data = GetTensorData<int8_t>(filter);
478 const float* input_ptr = GetTensorData<float>(input);
479 tensor_utils::BatchQuantizeFloats(
480 input_ptr, batch_size, input_size, quant_data, scaling_factors_ptr,
481 input_offset_ptr, params->asymmetric_quantize_inputs);
482 for (int b = 0; b < batch_size; ++b) {
483 // Incorporate scaling of the filter.
484 scaling_factors_ptr[b] *= filter->params.scale;
485 }
486
487 // Compute output += weight * quantized_input
488 int32_t* scratch = GetTensorData<int32_t>(accum_scratch);
489 tensor_utils::MatrixBatchVectorMultiplyAccumulate(
490 filter_data, num_units, input_size, quant_data, scaling_factors_ptr,
491 batch_size, GetTensorData<float>(output), /*per_channel_scale=*/nullptr,
492 input_offset_ptr, scratch, row_sums_ptr, &data->compute_row_sums,
493 CpuBackendContext::GetFromContext(context));
494
495 // Apply activation function to floats.
496 tensor_utils::ApplyActivationToVector(
497 GetTensorData<float>(output), batch_size * num_units, params->activation,
498 GetTensorData<float>(output));
499 return kTfLiteOk;
500}

502void EvalSparseHybridImpl(TfLiteContext* context, TfLiteNode* node,
503 TfLiteFullyConnectedParams* params, OpData* data,
504 const TfLiteTensor* input, const TfLiteTensor* filter,
505 const TfLiteTensor* bias, int thread_start,
506 int thread_end, TfLiteTensor* input_quantized,
507 TfLiteTensor* scaling_factors,
508 TfLiteTensor* accum_scratch, TfLiteTensor* row_sums,
509 TfLiteTensor* input_offsets, TfLiteTensor* output) {
510 ruy::profiler::ScopeLabel label("FullyConnected");
511 ruy::profiler::ScopeLabel inner_label("Sparse Hybrid Kernel");
512 const auto& input_shape = GetTensorShape(input);
513 const auto& output_shape = GetTensorShape(output);
514 const auto& filter_shape = GetTensorShape(filter);
515 const int input_dims_count = input_shape.DimensionsCount();
516 const int output_dims_count = output_shape.DimensionsCount();
517 const int filter_dims_count = filter_shape.DimensionsCount();
518 const int batch_size = thread_end - thread_start;
519 const int input_depth = MatchingDim(filter_shape, filter_dims_count - 1,
520 input_shape, input_dims_count - 1);
521 const int output_depth = MatchingDim(filter_shape, filter_dims_count - 2,
522 output_shape, output_dims_count - 1);
523 const int per_thread_input_size = batch_size * input_depth;
524
525 const float* per_thread_input =
526 GetTensorData<float>(input) + thread_start * input_depth;
527 float* per_thread_output =
528 GetTensorData<float>(output) + thread_start * output_depth;
529
530 // Output = bias if bias tensor exists.
531 if (bias) {
532 tensor_utils::VectorBatchVectorAssign(GetTensorData<float>(bias),
533 output_depth, batch_size,
534 per_thread_output);
535 } else {
536 std::fill_n(per_thread_output, batch_size * output_depth, 0.0f);
537 }
538
539 // Save matrix multiplication computation for all zero input.
540 if (tensor_utils::IsZeroVector(per_thread_input, per_thread_input_size)) {
541 tensor_utils::ApplyActivationToVector(
542 per_thread_output, batch_size * output_depth, params->activation,
543 per_thread_output);
544 return;
545 }
546
547 // Quantize input from float to uint8 + quantization params (scaling factor).
548 float* scaling_factors_ptr =
549 GetTensorData<float>(scaling_factors) + thread_start;
550 int32_t* input_offset_ptr = nullptr;
551 int32_t* row_sums_ptr = nullptr;
552 if (params->asymmetric_quantize_inputs) {
553 input_offset_ptr = GetTensorData<int32_t>(input_offsets) + thread_start;
554 row_sums_ptr = GetTensorData<int32_t>(row_sums);
555 }
556 int8_t* quant_data =
557 GetTensorData<int8_t>(input_quantized) + thread_start * input_depth;
558 tensor_utils::BatchQuantizeFloats(per_thread_input, batch_size, input_depth,
559 quant_data, scaling_factors_ptr,
560 input_offset_ptr,
561 params->asymmetric_quantize_inputs);
562 for (int b = 0; b < batch_size; ++b) {
563 // Incorporate scaling of the filter.
564 scaling_factors_ptr[b] *= filter->params.scale;
565 }
566
567 // Compute output += weight * quantized_input
568 TfLiteTensor* filter_ledger = &context->tensors[node->temporaries->data[5]];
569 tensor_utils::SparseMatrixBatchVectorMultiplyAccumulate(
570 GetTensorData<int8_t>(filter), GetTensorData<uint8_t>(filter_ledger),
571 output_depth, input_depth, quant_data, scaling_factors_ptr, batch_size,
572 per_thread_output);
573
574 // Apply activation function to floats.
575 tensor_utils::ApplyActivationToVector(per_thread_output,
576 batch_size * output_depth,
577 params->activation, per_thread_output);
578}

581 SparseHybridFullyConnectedTask(
582 TfLiteContext* context, TfLiteNode* node,
583 TfLiteFullyConnectedParams* params, OpData* data,
584 const TfLiteTensor* input, const TfLiteTensor* filter,
585 const TfLiteTensor* bias, const int thread_start, const int thread_end,
586 TfLiteTensor* input_quantized, TfLiteTensor* scaling_factors,

591 params(params),

593 input(input),

605 void Run() override {
606 EvalSparseHybridImpl(context, node, params, data, input, filter, bias,
607 thread_start, thread_end, input_quantized,
608 scaling_factors, accum_scratch, row_sums,
609 input_offsets, output);
610 }

613 TfLiteContext* context;
614 TfLiteNode* node;
615 TfLiteFullyConnectedParams* params;
616 OpData* data;
617 const TfLiteTensor* input;
618 const TfLiteTensor* filter;
619 const TfLiteTensor* bias;
620 const int thread_start;
621 const int thread_end;
622 TfLiteTensor* input_quantized;
623 TfLiteTensor* scaling_factors;

630TfLiteStatus EvalHybrid(TfLiteContext* context, TfLiteNode* node,
631 TfLiteFullyConnectedParams* params, OpData* data,
632 const TfLiteTensor* input, const TfLiteTensor* filter,
633 const TfLiteTensor* bias, TfLiteTensor* input_quantized,
634 TfLiteTensor* scaling_factors,
635 TfLiteTensor* accum_scratch, TfLiteTensor* row_sums,
636 TfLiteTensor* input_offsets, TfLiteTensor* output) {
637 const auto& output_shape = GetTensorShape(output);
638 CpuBackendContext* cpu_backend_context =
639 CpuBackendContext::GetFromContext(context);
640 const bool is_dense = filter->sparsity == nullptr;
641 if (is_dense) {
642 return EvalHybridDense(context, node, params, data, input, filter, bias,
643 input_quantized, scaling_factors, accum_scratch,
644 row_sums, input_offsets, output);
645 }
646
647 TfLiteTensor* filter_ledger = &context->tensors[node->temporaries->data[5]];
648 if (!data->ledger_initialized) {
649 PopulateLedgerData(filter->sparsity, context,
650 GetTensorData<uint8_t>(filter_ledger));
651 data->ledger_initialized = true;
652 }
653
654 // The multi-threaded kernel slices the workload along the batch dimension. If
655 // there's not enough batches of data, the number of threads used is equal to
656 // the batch size.
657 // TODO(b/173442777): If needed, we can improve this later with slicing along
658 // the row dimension of the weight.
659 const int max_threads = cpu_backend_context->max_num_threads();
660 const int batches =
661 FlatSizeSkipDim(output_shape, output_shape.DimensionsCount() - 1);
662 const int thread_count = std::max(1, std::min(batches, max_threads));
663
664 std::vector<SparseHybridFullyConnectedTask> tasks;
665 tasks.reserve(thread_count);
666 int thread_start = 0;
667 for (int i = 0; i < thread_count; ++i) {
668 // This makes sure the workload is relatively balanced when batches is not
669 // a multiple of thread_count. The first mod(batches, thread_count) tasks
670 // need to process one more batch than the rest.
671 int thread_end = thread_start + batches / thread_count;
672 if (i < batches % thread_count) thread_end++;
673
674 tasks.emplace_back(context, node, params, data, input, filter, bias,
675 thread_start, thread_end, input_quantized,
676 scaling_factors, accum_scratch, row_sums, input_offsets,
677 output);
678 thread_start = thread_end;
679 }
680 cpu_backend_threadpool::Execute(tasks.size(), tasks.data(),
681 cpu_backend_context);
682 return kTfLiteOk;
683}

737template <KernelType kernel_type>
738TfLiteStatus EvalQuantized(TfLiteContext* context, TfLiteNode* node,
739 TfLiteFullyConnectedParams* params, OpData* data,
740 const TfLiteTensor* input,
741 const TfLiteTensor* filter, const TfLiteTensor* bias,
742 TfLiteTensor* output) {
743 int32_t input_offset = -input->params.zero_point;
744 int32_t filter_offset = -filter->params.zero_point;
745 int32_t output_offset = output->params.zero_point;
746 // Only the Pie path supports quantized models and float inputs/outputs.
747 if (input->type == kTfLiteFloat32) {
748 TfLiteTensor* input_quantized;
749 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/0,
750 &input_quantized));
751 TfLiteTensor* scaling_factors;
752 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
753 &scaling_factors));
754 TfLiteTensor* accum_scratch;
755 TF_LITE_ENSURE_OK(
756 context, GetTemporarySafe(context, node, /*index=*/2, &accum_scratch));
757 TfLiteTensor* input_offsets;
758 TF_LITE_ENSURE_OK(
759 context, GetTemporarySafe(context, node, /*index=*/3, &input_offsets));
760 TfLiteTensor* row_sums;
761 TF_LITE_ENSURE_OK(context,
762 GetTemporarySafe(context, node, /*index=*/4, &row_sums));
763 return EvalHybrid(context, node, params, data, input, filter, bias,
764 input_quantized, scaling_factors, accum_scratch, row_sums,
765 input_offsets, output);
766 } else {
767 FullyConnectedParams op_params;
768 op_params.input_offset = input_offset;
769 op_params.weights_offset = filter_offset;
770 op_params.output_offset = output_offset;
771 op_params.output_multiplier = data->output_multiplier;
772 op_params.output_shift = data->output_shift;
773 op_params.quantized_activation_min = data->output_activation_min;
774 op_params.quantized_activation_max = data->output_activation_max;
775 op_params.lhs_cacheable = IsConstantTensor(filter);
776 op_params.rhs_cacheable = IsConstantTensor(input);
777 switch (output->type) {
778 case kTfLiteUInt8:
779 if (kernel_type == kReference) {
780 reference_ops::FullyConnected(
781 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
782 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
783 GetTensorShape(bias), GetTensorData<int32_t>(bias),
784 GetTensorShape(output), GetTensorData<uint8_t>(output));
785 } else {
786 optimized_ops::FullyConnected(
787 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
788 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
789 GetTensorShape(bias), GetTensorData<int32_t>(bias),
790 GetTensorShape(output), GetTensorData<uint8_t>(output),
791 CpuBackendContext::GetFromContext(context));
792 }
793 break;
794 case kTfLiteInt8:
795 FullyConnectedInt8<kernel_type>(
796 data, input, filter, bias, output,
797 CpuBackendContext::GetFromContext(context));
798 break;
799 case kTfLiteInt16:
800 if (input->type == kTfLiteInt16) {
801 FullyConnectedInt16<kernel_type>(data, input, filter, bias, output);
802 } else if (kernel_type == kReference) {
803 reference_ops::FullyConnected(
804 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
805 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
806 GetTensorShape(bias), GetTensorData<int32_t>(bias),
807 GetTensorShape(output), GetTensorData<int16_t>(output));
808 } else {
809 optimized_ops::FullyConnected(
810 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
811 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
812 GetTensorShape(bias), GetTensorData<int32_t>(bias),
813 GetTensorShape(output), GetTensorData<int16_t>(output),
814 CpuBackendContext::GetFromContext(context));
815 }
816 break;
817 default:
818 context->ReportError(context,
819 "Quantized FullyConnected expects output data "
820 "type uint8, int8 or int16");
821 return kTfLiteError;
822 }
823 }
824
825 return kTfLiteOk;
826}

881template <KernelType kernel_type>
882TfLiteStatus EvalFloat(TfLiteContext* context, TfLiteNode* node,
883 TfLiteFullyConnectedParams* params, OpData* data,
884 const TfLiteTensor* input, const TfLiteTensor* filter,
885 const TfLiteTensor* bias, TfLiteTensor* output) {
886 float output_activation_min, output_activation_max;
887 CalculateActivationRange(params->activation, &output_activation_min,
888 &output_activation_max);
889 if (kernel_type == kReference) {
890 FullyConnectedParams op_params;
891 op_params.float_activation_min = output_activation_min;
892 op_params.float_activation_max = output_activation_max;
893 if (filter->sparsity != nullptr) {
894 const auto& sparsity = *filter->sparsity;
895 reference_ops::FullyConnectedSparseWeight(
896 sparsity, op_params, GetTensorShape(input),
897 GetTensorData<float>(input), GetTensorShape(filter),
898 GetTensorData<float>(filter), GetTensorShape(bias),
899 GetTensorData<float>(bias), GetTensorShape(output),
900 GetTensorData<float>(output));
901 } else {
902 reference_ops::FullyConnected(
903 op_params, GetTensorShape(input), GetTensorData<float>(input),
904 GetTensorShape(filter), GetTensorData<float>(filter),
905 GetTensorShape(bias), GetTensorData<float>(bias),
906 GetTensorShape(output), GetTensorData<float>(output));
907 }
908 } else if (kernel_type == kLegacyPie) {
909 return EvalPie(context, node, params, data, input, filter, bias, output);
910 } else {
911 FullyConnectedParams op_params;
912 op_params.float_activation_min = output_activation_min;
913 op_params.float_activation_max = output_activation_max;
914 if (filter->sparsity != nullptr) {
915 const auto& sparsity = *filter->sparsity;
916 if (!SupportedSparsityFormat(sparsity)) {
917 TF_LITE_KERNEL_LOG(context,
918 "Unsupported sparse fully-connected weight format.");
919 return kTfLiteError;
920 }
921
922 if (sparsity.dim_metadata_size == kDimMetadataSizeRandomSparse) {
923 // Random sparse.
924 optimized_ops::FullyConnectedSparseWeight(
925 sparsity, op_params, GetTensorShape(input),
926 GetTensorData<float>(input), GetTensorShape(filter),
927 GetTensorData<float>(filter), GetTensorShape(bias),
928 GetTensorData<float>(bias), GetTensorShape(output),
929 GetTensorData<float>(output));
930 } else if (sparsity.dim_metadata_size == kDimMetadataSizeBlockSparse &&
931 sparsity.dim_metadata[2].dense_size == 4) {
932 // Block sparse with block size of 1x4.
933 optimized_ops::FullyConnectedSparseWeight1x4(
934 sparsity, op_params, GetTensorShape(input),
935 GetTensorData<float>(input), GetTensorShape(filter),
936 GetTensorData<float>(filter), GetTensorShape(bias),
937 GetTensorData<float>(bias), GetTensorShape(output),
938 GetTensorData<float>(output),
939 CpuBackendContext::GetFromContext(context));
940 } else {
941 TF_LITE_KERNEL_LOG(context,
942 "Unsupported sparse fully-connected weight format.");
943 return kTfLiteError;
944 }
945
946 } else {
947 op_params.lhs_cacheable = IsConstantTensor(filter);
948 op_params.rhs_cacheable = IsConstantTensor(input);
949 optimized_ops::FullyConnected(
950 op_params, GetTensorShape(input), GetTensorData<float>(input),
951 GetTensorShape(filter), GetTensorData<float>(filter),
952 GetTensorShape(bias), GetTensorData<float>(bias),
953 GetTensorShape(output), GetTensorData<float>(output),
954 CpuBackendContext::GetFromContext(context));
955 }
956 }
957
958 return kTfLiteOk;
959}

961template <KernelType kernel_type>
962TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
963 auto* params =
964 reinterpret_cast<TfLiteFullyConnectedParams*>(node->builtin_data);
965 OpData* data = reinterpret_cast<OpData*>(node->user_data);
966
967 const TfLiteTensor* input;
968 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
969 const TfLiteTensor* filter;
970 TF_LITE_ENSURE_OK(context,
971 GetInputSafe(context, node, kWeightsTensor, &filter));
972 const TfLiteTensor* bias =
973 (node->inputs->size == 3)
974 ? GetOptionalInputTensor(context, node, kBiasTensor)
975 : nullptr;
976 TfLiteTensor* output;
977 TF_LITE_ENSURE_OK(context,
978 GetOutputSafe(context, node, kOutputTensor, &output));
979
980 switch (filter->type) {
981 case kTfLiteFloat32:
982 return EvalFloat<kernel_type>(context, node, params, data, input, filter,
983 bias, output);
984 case kTfLiteUInt8:
985 if (params->weights_format ==
986 kTfLiteFullyConnectedWeightsFormatShuffled4x16Int8) {
987 TfLiteTensor* shuffled_input_workspace;
988 TF_LITE_ENSURE_OK(
989 context, GetOutputSafe(context, node, kShuffledInputWorkspaceTensor,
990 &shuffled_input_workspace));
991 return EvalShuffledQuantized<kernel_type>(context, node, params, data,
992 input, filter, bias, output,
993 shuffled_input_workspace);
994 } else if (params->weights_format ==
995 kTfLiteFullyConnectedWeightsFormatDefault) {
996 return EvalQuantized<kernel_type>(context, node, params, data, input,
997 filter, bias, output);
998 } else {
999 context->ReportError(context,
1000 "Unhandled fully-connected weights format");
1001 return kTfLiteError;
1002 }
1003 case kTfLiteInt8:
1004 if (params->weights_format == kTfLiteFullyConnectedWeightsFormatDefault) {
1005 return EvalQuantized<kernel_type>(context, node, params, data, input,
1006 filter, bias, output);
1007 } else {
1008 context->ReportError(context,
1009 "Unhandled fully-connected weights format");
1010 return kTfLiteError;
1011 }
1012 default:
1013 context->ReportError(context,
1014 "Filter data type %s currently not supported.",
1015 TfLiteTypeGetName(filter->type));
1016 return kTfLiteError;
1017 }
1018 return kTfLiteOk;
1019}
