47constexpr int kInputTensor = 0;
48constexpr int kFilterTensor = 1;
49constexpr int kBiasTensor = 2;
50constexpr int kOutputTensor = 0;
51
52// This file has three implementation of DepthwiseConv.
53enum KernelType {
54 kReference,
55 kGenericOptimized, // Neon-free
56 kNeonOptimized,
57};
58
59const int kTensorNotAllocated = -1;
60
61struct OpData {
62 TfLitePaddingValues padding;
63 // The scaling factor from input to output (aka the 'real multiplier') can
64 // be represented as a fixed point multiplier plus a left shift.
65 int32_t output_multiplier;
66 int output_shift;
67 // The range of the fused activation layer. For example for kNone and
68 // uint8_t these would be 0 and 255.
69 int32_t output_activation_min;
70 int32_t output_activation_max;
71
72 // Per channel output multiplier and shift.
73 std::vector<int32_t> per_channel_output_multiplier;
74 std::vector<int> per_channel_output_shift;
75
76 // Hybrid per channel temporary tensors.
77 int input_quantized_id = kTensorNotAllocated;
78 int scaling_factors_id = kTensorNotAllocated;
79 int input_offset_id = kTensorNotAllocated;
80 int32_t input_quantized_index;
81 int32_t scaling_factors_index;
82 int32_t input_offset_index;
83};

86 // This is a builtin op, so we don't use the contents in 'buffer', if any.

88 // Eval().
89 return new OpData;
90}

93 delete reinterpret_cast<OpData*>(buffer);

96TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
97 auto* params =
98 reinterpret_cast<TfLiteDepthwiseConvParams*>(node->builtin_data);
99 OpData* data = reinterpret_cast<OpData*>(node->user_data);
100
101 bool hasBias = NumInputs(node) == 3;
102
103 TF_LITE_ENSURE(context, hasBias || NumInputs(node) == 2);
104 const TfLiteTensor* input;
105 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
106 const TfLiteTensor* filter;
107 TF_LITE_ENSURE_OK(context,
108 GetInputSafe(context, node, kFilterTensor, &filter));
109 const TfLiteTensor* bias = nullptr;
110
111 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
112 TfLiteTensor* output;
113 TF_LITE_ENSURE_OK(context,
114 GetOutputSafe(context, node, kOutputTensor, &output));
115
116 TF_LITE_ENSURE_EQ(context, NumDimensions(input), 4);
117 TF_LITE_ENSURE_EQ(context, NumDimensions(filter), 4);
118
119 const TfLiteType data_type = input->type;
120
121 const TfLiteType filter_type = filter->type;
122 const bool is_hybrid =
123 data_type == kTfLiteFloat32 && filter_type == kTfLiteInt8;
124 TF_LITE_ENSURE(context,
125 data_type == kTfLiteFloat32 || data_type == kTfLiteUInt8 ||
126 data_type == kTfLiteInt8 || data_type == kTfLiteInt16);
127 TF_LITE_ENSURE_TYPES_EQ(context, output->type, data_type);
128 if (!is_hybrid) {
129 TF_LITE_ENSURE(context,
130 filter->type == data_type || data_type == kTfLiteInt16);
131 }
132
133 if (data_type == kTfLiteInt16) {
134 TF_LITE_ENSURE_EQ(context, input->params.zero_point, 0);
135 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
136 }
137
138 // Filter in DepthwiseConv is expected to be [1, H, W, O].
139 TF_LITE_ENSURE_EQ(context, SizeOfDimension(filter, 0), 1);
140
141 if (hasBias) {
142 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kBiasTensor, &bias));
143 if (data_type == kTfLiteUInt8 || data_type == kTfLiteInt8) {
144 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, kTfLiteInt32);
145 TF_LITE_ENSURE_EQ(context, bias->params.zero_point, 0);
146 } else if (data_type == kTfLiteInt16) {
147 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, kTfLiteInt64);
148 TF_LITE_ENSURE_EQ(context, bias->params.zero_point, 0);
149 } else {
150 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, data_type);
151 }
152 TF_LITE_ENSURE_EQ(context, NumDimensions(bias), 1);
153 TF_LITE_ENSURE_EQ(context, SizeOfDimension(filter, 3),
154 SizeOfDimension(bias, 0));
155 }
156
157 int channels_out = SizeOfDimension(filter, 3);
158 int width = SizeOfDimension(input, 2);
159 int height = SizeOfDimension(input, 1);
160 int filter_width = SizeOfDimension(filter, 2);
161 int filter_height = SizeOfDimension(filter, 1);
162 int batches = SizeOfDimension(input, 0);
163
164 // Matching GetWindowedOutputSize in TensorFlow.
165 auto padding = params->padding;
166 int out_width, out_height;
167
168 data->padding = ComputePaddingHeightWidth(
169 params->stride_height, params->stride_width,
170 params->dilation_height_factor, params->dilation_width_factor, height,
171 width, filter_height, filter_width, padding, &out_height, &out_width);
172
173 // Note that quantized inference requires that all tensors have their
174 // parameters set. This is usually done during quantized training or
175 // calibration.
176 if (data_type != kTfLiteFloat32) {
177 TF_LITE_ENSURE_EQ(context, filter->quantization.type,
178 kTfLiteAffineQuantization);
179 const auto* affine_quantization =
180 reinterpret_cast<TfLiteAffineQuantization*>(
181 filter->quantization.params);
182 TF_LITE_ENSURE(context, affine_quantization);
183 TF_LITE_ENSURE(context, affine_quantization->scale);
184 TF_LITE_ENSURE(context, (affine_quantization->scale->size == 1 ||
185 affine_quantization->scale->size == channels_out));
186
187 data->per_channel_output_multiplier.resize(channels_out);
188 data->per_channel_output_shift.resize(channels_out);
189 TF_LITE_ENSURE_STATUS(tflite::PopulateConvolutionQuantizationParams(
190 context, input, filter, bias, output, params->activation,
191 &data->output_multiplier, &data->output_shift,
192 &data->output_activation_min, &data->output_activation_max,
193 data->per_channel_output_multiplier.data(),
194 data->per_channel_output_shift.data(), channels_out));
195 }
196
197 if (is_hybrid) {
198 const auto* affine_quantization =
199 reinterpret_cast<TfLiteAffineQuantization*>(
200 filter->quantization.params);
201 TF_LITE_ENSURE(context, affine_quantization);
202 TF_LITE_ENSURE(context, affine_quantization->scale);
203 TF_LITE_ENSURE_EQ(
204 context, affine_quantization->scale->size,
205 filter->dims->data[affine_quantization->quantized_dimension]);
206
207 int temporaries_count = 0;
208 data->input_quantized_index = temporaries_count;
209 if (data->input_quantized_id == kTensorNotAllocated) {
210 TF_LITE_ENSURE_OK(
211 context, context->AddTensors(context, 1, &data->input_quantized_id));
212 }
213 ++temporaries_count;
214 data->scaling_factors_index = temporaries_count;
215 if (data->scaling_factors_id == kTensorNotAllocated) {
216 TF_LITE_ENSURE_OK(
217 context, context->AddTensors(context, 1, &data->scaling_factors_id));
218 }
219 ++temporaries_count;
220 data->input_offset_index = temporaries_count;
221 if (data->input_offset_id == kTensorNotAllocated) {
222 TF_LITE_ENSURE_OK(
223 context, context->AddTensors(context, 1, &data->input_offset_id));
224 }
225 ++temporaries_count;
226
227 TfLiteIntArrayFree(node->temporaries);
228 node->temporaries = TfLiteIntArrayCreate(temporaries_count);
229
230 node->temporaries->data[data->input_quantized_index] =
231 data->input_quantized_id;
232 TfLiteTensor* input_quantized;
233 TF_LITE_ENSURE_OK(
234 context, GetTemporarySafe(context, node, data->input_quantized_index,
235 &input_quantized));
236 input_quantized->type = kTfLiteInt8;
237 input_quantized->allocation_type = kTfLiteArenaRw;
238 if (!TfLiteIntArrayEqual(input_quantized->dims, input->dims)) {
239 TfLiteIntArray* input_quantized_size = TfLiteIntArrayCopy(input->dims);
240 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
241 input_quantized_size));
242 }
243 node->temporaries->data[data->scaling_factors_index] =
244 data->scaling_factors_id;
245 TfLiteTensor* scaling_factors;
246 TF_LITE_ENSURE_OK(
247 context, GetTemporarySafe(context, node, data->scaling_factors_index,
248 &scaling_factors));
249 scaling_factors->type = kTfLiteFloat32;
250 scaling_factors->allocation_type = kTfLiteArenaRw;
251 const int batch_size = SizeOfDimension(input, 0);
252 int scaling_dims[1] = {batch_size};
253 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
254 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
255 scaling_factors_size->data[0] = batch_size;
256 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
257 scaling_factors_size));
258 }
259 node->temporaries->data[data->input_offset_index] = data->input_offset_id;
260 TfLiteTensor* input_offsets;
261 TF_LITE_ENSURE_OK(context,
262 GetTemporarySafe(context, node, data->input_offset_index,
263 &input_offsets));
264 input_offsets->type = kTfLiteInt32;
265 input_offsets->allocation_type = kTfLiteArenaRw;
266 if (!TfLiteIntArrayEqualsArray(input_offsets->dims, 1, scaling_dims)) {
267 TfLiteIntArray* input_offsets_size = TfLiteIntArrayCreate(1);
268 input_offsets_size->data[0] = batch_size;
269 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_offsets,
270 input_offsets_size));
271 }
272 }
273
274 TfLiteIntArray* outputSize = TfLiteIntArrayCreate(4);
275 outputSize->data[0] = batches;
276 outputSize->data[1] = out_height;
277 outputSize->data[2] = out_width;
278 outputSize->data[3] = channels_out;
279 return context->ResizeTensor(context, output, outputSize);
280}

282TfLiteStatus ComputeDepthMultiplier(TfLiteContext* context,
283 const TfLiteTensor* input,
284 const TfLiteTensor* filter,
285 int16* depth_multiplier) {
286 int num_filter_channels = SizeOfDimension(filter, 3);
287 int num_input_channels = SizeOfDimension(input, 3);
288 TF_LITE_ENSURE(context, num_input_channels != 0);
289 TF_LITE_ENSURE_EQ(context, num_filter_channels % num_input_channels, 0);
290 *depth_multiplier = num_filter_channels / num_input_channels;
291 return kTfLiteOk;
292}

294template <KernelType kernel_type>
295TfLiteStatus EvalFloat(TfLiteContext* context, TfLiteNode* node,
296 TfLiteDepthwiseConvParams* params, OpData* data,
297 const TfLiteTensor* input, const TfLiteTensor* filter,
298 const TfLiteTensor* bias, TfLiteTensor* output) {
299 float output_activation_min, output_activation_max;
300 CalculateActivationRange(params->activation, &output_activation_min,
301 &output_activation_max);
302
303 DepthwiseParams op_params;
304 op_params.padding_type = PaddingType::kSame;
305 op_params.padding_values.width = data->padding.width;
306 op_params.padding_values.height = data->padding.height;
307 op_params.stride_width = params->stride_width;
308 op_params.stride_height = params->stride_height;
309 op_params.dilation_width_factor = params->dilation_width_factor;
310 op_params.dilation_height_factor = params->dilation_height_factor;
311 op_params.float_activation_min = output_activation_min;
312 op_params.float_activation_max = output_activation_max;
313 TF_LITE_ENSURE_STATUS(ComputeDepthMultiplier(context, input, filter,
314 &op_params.depth_multiplier));
315 if (kernel_type == kReference) {
316 reference_ops::DepthwiseConv(
317 op_params, GetTensorShape(input), GetTensorData<float>(input),
318 GetTensorShape(filter), GetTensorData<float>(filter),
319 GetTensorShape(bias), GetTensorData<float>(bias),
320 GetTensorShape(output), GetTensorData<float>(output));
321 } else {
322 optimized_ops::DepthwiseConv<float, float>(
323 op_params, GetTensorShape(input), GetTensorData<float>(input),
324 GetTensorShape(filter), GetTensorData<float>(filter),
325 GetTensorShape(bias), GetTensorData<float>(bias),
326 GetTensorShape(output), GetTensorData<float>(output),
327 CpuBackendContext::GetFromContext(context));
328 }
329 return kTfLiteOk;
330}

332template <KernelType kernel_type>
333TfLiteStatus EvalQuantized(TfLiteContext* context, TfLiteNode* node,
334 TfLiteDepthwiseConvParams* params, OpData* data,
335 const TfLiteTensor* input,
336 const TfLiteTensor* filter, const TfLiteTensor* bias,
337 TfLiteTensor* output) {
338 auto input_offset = -input->params.zero_point;
339 auto filter_offset = -filter->params.zero_point;
340 auto output_offset = output->params.zero_point;
341
342 DepthwiseParams op_params;
343 op_params.padding_type = PaddingType::kSame;
344 op_params.padding_values.width = data->padding.width;
345 op_params.padding_values.height = data->padding.height;
346 op_params.stride_width = params->stride_width;
347 op_params.stride_height = params->stride_height;
348 op_params.dilation_width_factor = params->dilation_width_factor;
349 op_params.dilation_height_factor = params->dilation_height_factor;
350 op_params.input_offset = input_offset;
351 op_params.weights_offset = filter_offset;
352 op_params.output_offset = output_offset;
353 op_params.output_multiplier = data->output_multiplier;
354 op_params.output_shift = -data->output_shift;
355 op_params.quantized_activation_min = data->output_activation_min;
356 op_params.quantized_activation_max = data->output_activation_max;
357 TF_LITE_ENSURE_STATUS(ComputeDepthMultiplier(context, input, filter,
358 &op_params.depth_multiplier));
359 if (kernel_type == kReference) {
360 reference_ops::DepthwiseConv(
361 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
362 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
363 GetTensorShape(bias), GetTensorData<int32_t>(bias),
364 GetTensorShape(output), GetTensorData<uint8_t>(output));
365 } else {
366 optimized_ops::DepthwiseConv<uint8, int32>(
367 op_params, GetTensorShape(input), GetTensorData<uint8_t>(input),
368 GetTensorShape(filter), GetTensorData<uint8_t>(filter),
369 GetTensorShape(bias), GetTensorData<int32_t>(bias),
370 GetTensorShape(output), GetTensorData<uint8_t>(output),
371 CpuBackendContext::GetFromContext(context));
372 }
373 return kTfLiteOk;
374}

448template <KernelType kernel_type>
449TfLiteStatus EvalHybridPerChannel(TfLiteContext* context, TfLiteNode* node,
450 TfLiteDepthwiseConvParams* params,
451 OpData* data, const TfLiteTensor* input,
452 const TfLiteTensor* filter,
453 const TfLiteTensor* bias,
454 TfLiteTensor* output) {
455 float output_activation_min, output_activation_max;
456 CalculateActivationRange(params->activation, &output_activation_min,
457 &output_activation_max);
458 const int batch_size = SizeOfDimension(input, 0);
459 TF_LITE_ENSURE(context, batch_size != 0);
460 const int input_size = NumElements(input) / batch_size;
461 TfLiteTensor* input_quantized;
462 TF_LITE_ENSURE_OK(context,
463 GetTemporarySafe(context, node, data->input_quantized_index,
464 &input_quantized));
465 int8_t* quantized_input_ptr_batch = input_quantized->data.int8;
466 TfLiteTensor* scaling_factors_tensor;
467 TF_LITE_ENSURE_OK(context,
468 GetTemporarySafe(context, node, data->scaling_factors_index,
469 &scaling_factors_tensor));
470 float* scaling_factors_ptr = GetTensorData<float>(scaling_factors_tensor);
471 TfLiteTensor* input_offset_tensor;
472 TF_LITE_ENSURE_OK(context,
473 GetTemporarySafe(context, node, data->input_offset_index,
474 &input_offset_tensor));
475 int32_t* input_offset_ptr = GetTensorData<int32_t>(input_offset_tensor);
476
477 for (int b = 0; b < batch_size; ++b) {
478 const int offset = b * input_size;
479 tensor_utils::AsymmetricQuantizeFloats(
480 GetTensorData<float>(input) + offset, input_size,
481 quantized_input_ptr_batch + offset, &scaling_factors_ptr[b],
482 &input_offset_ptr[b]);
483 }
484
485 DepthwiseParams op_params;
486 op_params.padding_type = PaddingType::kSame;
487 op_params.padding_values.width = data->padding.width;
488 op_params.padding_values.height = data->padding.height;
489 op_params.stride_width = params->stride_width;
490 op_params.stride_height = params->stride_height;
491 op_params.dilation_width_factor = params->dilation_width_factor;
492 op_params.dilation_height_factor = params->dilation_height_factor;
493 op_params.depth_multiplier = params->depth_multiplier;
494
495 op_params.weights_offset = 0;
496 op_params.float_activation_min = output_activation_min;
497 op_params.float_activation_max = output_activation_max;
498 const auto* affine_quantization =
499 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params);
500 if (kernel_type == kReference) {
501 reference_integer_ops::DepthwiseConvHybridPerChannel(
502 op_params, scaling_factors_ptr, GetTensorShape(input),
503 quantized_input_ptr_batch, GetTensorShape(filter),
504 GetTensorData<int8>(filter), GetTensorShape(bias),
505 GetTensorData<float>(bias), GetTensorShape(output),
506 GetTensorData<float>(output), affine_quantization->scale->data,
507 input_offset_ptr);
508 } else {
509 optimized_integer_ops::DepthwiseConvHybridPerChannel(
510 op_params, scaling_factors_ptr, GetTensorShape(input),
511 quantized_input_ptr_batch, GetTensorShape(filter),
512 GetTensorData<int8>(filter), GetTensorShape(bias),
513 GetTensorData<float>(bias), GetTensorShape(output),
514 GetTensorData<float>(output), affine_quantization->scale->data,
515 input_offset_ptr, CpuBackendContext::GetFromContext(context));
516 }
517
518 return kTfLiteOk;
519}

521template <KernelType kernel_type, TfLiteType input_type>
522TfLiteStatus EvalImpl(TfLiteContext* context, TfLiteNode* node) {
523 auto* params =
524 reinterpret_cast<TfLiteDepthwiseConvParams*>(node->builtin_data);
525 OpData* data = reinterpret_cast<OpData*>(node->user_data);
526
527 TfLiteTensor* output;
528 TF_LITE_ENSURE_OK(context,
529 GetOutputSafe(context, node, kOutputTensor, &output));
530 const TfLiteTensor* input;
531 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
532 const TfLiteTensor* filter;
533 TF_LITE_ENSURE_OK(context,
534 GetInputSafe(context, node, kFilterTensor, &filter));
535 const TfLiteTensor* bias =
536 (NumInputs(node) == 3) ? GetInput(context, node, kBiasTensor) : nullptr;
537 TFLITE_DCHECK_EQ(input_type, input->type);
538
539 switch (input_type) { // Already know in/out types are same.
540 case kTfLiteFloat32:
541 if (filter->type == kTfLiteFloat32) {
542 return EvalFloat<kernel_type>(context, node, params, data, input,
543 filter, bias, output);
544 } else if (filter->type == kTfLiteInt8) {
545 return EvalHybridPerChannel<kernel_type>(context, node, params, data,
546 input, filter, bias, output);
547 } else {
548 TF_LITE_KERNEL_LOG(
549 context, "Type %s with filter type %s not currently supported.",
550 TfLiteTypeGetName(input->type), TfLiteTypeGetName(filter->type));
551 return kTfLiteError;
552 }
553 break;
554 case kTfLiteUInt8:
555 return EvalQuantized<kernel_type>(context, node, params, data, input,
556 filter, bias, output);
557 break;
558 case kTfLiteInt8:
559 return EvalQuantizedPerChannel<kernel_type>(context, node, params, data,
560 input, filter, bias, output);
561 break;
562 case kTfLiteInt16:
563 return EvalQuantizedPerChannel16x8(params, data, input, filter, bias,
564 output);
565 break;
566 default:
567 context->ReportError(context, "Type %d not currently supported.",
568 input->type);
569 return kTfLiteError;
570 }
571}
572
573template <KernelType kernel_type>
574TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
575 const TfLiteTensor* input;
576 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
577
578 switch (input->type) { // Already know in/out types are same.
579 case kTfLiteFloat32:
580 return EvalImpl<kernel_type, kTfLiteFloat32>(context, node);
581 case kTfLiteUInt8:
582 return EvalImpl<kernel_type, kTfLiteUInt8>(context, node);
583 case kTfLiteInt8:
584 return EvalImpl<kernel_type, kTfLiteInt8>(context, node);
585 case kTfLiteInt16:
586 return EvalImpl<kernel_type, kTfLiteInt16>(context, node);
587 default:
588 context->ReportError(context, "Type %d not currently supported.",
589 input->type);
590 return kTfLiteError;
591 }
592}
