41// Assumes tensor_index is a valid index (in bounds)
42inline TfLiteTensor* GetTensorAtIndex(const TfLiteContext* context,
43 int tensor_index) {
44 if (context->tensors != nullptr) {
45 return &context->tensors[tensor_index];
46 } else {
47 return context->GetTensor(context, tensor_index);
48 }
49}

51// Validate in a single place to reduce binary size
52inline TfLiteStatus ValidateTensorIndexingSafe(const TfLiteContext* context,
53 int index, int max_size,
54 const int* tensor_indices,
55 int* tensor_index) {
56 if (index < 0 || index >= max_size) {
57 TF_LITE_KERNEL_LOG(const_cast<TfLiteContext*>(context),
58 "Invalid tensor index %d (not in [0, %d))\n", index,
59 max_size);
60 return kTfLiteError;
61 }
62 if (tensor_indices[index] == kTfLiteOptionalTensor) {
63 TF_LITE_KERNEL_LOG(const_cast<TfLiteContext*>(context),
64 "Tensor at index %d was optional but was expected\n",
65 index);
66 return kTfLiteError;
67 }
68
69 *tensor_index = tensor_indices[index];
70 return kTfLiteOk;
71}
72
73// Same as above but returns -1 for invalid inputs instead of status + logging
74// error.
75inline int ValidateTensorIndexing(const TfLiteContext* context, int index,
76 int max_size, const int* tensor_indices) {
77 if (index >= 0 && index < max_size) {
78 const int tensor_index = tensor_indices[index];
79 if (tensor_index != kTfLiteOptionalTensor) {
80 return tensor_index;
81 }
82 }
83 return -1;
84}

86inline TfLiteTensor* GetMutableInput(const TfLiteContext* context,
87 const TfLiteNode* node, int index) {
88 const int tensor_index = ValidateTensorIndexing(
89 context, index, node->inputs->size, node->inputs->data);
90 if (tensor_index < 0) {
91 return nullptr;
92 }
93 return GetTensorAtIndex(context, tensor_index);
94}
95
96inline TfLiteStatus GetMutableInputSafe(const TfLiteContext* context,
97 const TfLiteNode* node, int index,
98 const TfLiteTensor** tensor) {
99 int tensor_index;
100 TF_LITE_ENSURE_OK(
101 context, ValidateTensorIndexingSafe(context, index, node->inputs->size,
102 node->inputs->data, &tensor_index));
103 *tensor = GetTensorAtIndex(context, tensor_index);
104 return kTfLiteOk;
105}

194// Per-axis
195TfLiteStatus PopulateConvolutionQuantizationParams(
196 TfLiteContext* context, const TfLiteTensor* input,
197 const TfLiteTensor* filter, const TfLiteTensor* bias, TfLiteTensor* output,
198 const TfLiteFusedActivation& activation, int32_t* multiplier, int* shift,
199 int32_t* output_activation_min, int32_t* output_activation_max,
200 int32_t* per_channel_multiplier, int* per_channel_shift) {
201 const auto* affine_quantization =
202 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params);
203 return PopulateConvolutionQuantizationParams(
204 context, input, filter, bias, output, activation, multiplier, shift,
205 output_activation_min, output_activation_max, per_channel_multiplier,
206 per_channel_shift, affine_quantization->scale->size);
207}

209// Per-axis & per-tensor
210TfLiteStatus PopulateConvolutionQuantizationParams(
211 TfLiteContext* context, const TfLiteTensor* input,
212 const TfLiteTensor* filter, const TfLiteTensor* bias, TfLiteTensor* output,
213 const TfLiteFusedActivation& activation, int32_t* multiplier, int* shift,
214 int32_t* output_activation_min, int32_t* output_activation_max,
215 int32_t* per_channel_multiplier, int* per_channel_shift, int num_channels) {
216 TF_LITE_ENSURE_EQ(context, input->quantization.type,
217 kTfLiteAffineQuantization);
218 TF_LITE_ENSURE_EQ(context, filter->quantization.type,
219 kTfLiteAffineQuantization);
220 // TODO(jianlijianli): Enable bias type check and bias scale == input scale
221 // * filter scale for each channel in affine quantization once bias
222 // quantization is properly populated.
223 // TF_LITE_ENSURE_EQ(context, bias->quantization.type,
224 // kTfLiteAffineQuantization);
225
226 // Check data type.
227 const auto* affine_quantization =
228 reinterpret_cast<TfLiteAffineQuantization*>(filter->quantization.params);
229 TF_LITE_ENSURE(context, affine_quantization);
230 TF_LITE_ENSURE(context, affine_quantization->scale);
231 const bool is_per_channel = affine_quantization->scale->size > 1;
232 if (is_per_channel) {
233 // Currently only Int8/Int16 is supported for per channel quantization.
234 TF_LITE_ENSURE(context,
235 input->type == kTfLiteInt8 || input->type == kTfLiteInt16);
236 TF_LITE_ENSURE_EQ(context, filter->type, kTfLiteInt8);
237 TF_LITE_ENSURE_EQ(context, affine_quantization->scale->size, num_channels);
238 TF_LITE_ENSURE_EQ(
239 context, num_channels,
240 filter->dims->data[affine_quantization->quantized_dimension]);
241 }
242
243 // Populate multiplier and shift using affine quantization.
244 const float input_scale = input->params.scale;
245 const float output_scale = output->params.scale;
246 const float* filter_scales = affine_quantization->scale->data;
247 for (int i = 0; i < num_channels; ++i) {
248 // If per-tensor quantization parameter is specified, broadcast it along the
249 // quantization dimension (channels_out).
250 const float scale = is_per_channel ? filter_scales[i] : filter_scales[0];
251 const double filter_scale = static_cast<double>(scale);
252 const double effective_output_scale = static_cast<double>(input_scale) *
253 filter_scale /
254 static_cast<double>(output_scale);
255 int32_t significand;
256 int channel_shift;
257 QuantizeMultiplier(effective_output_scale, &significand, &channel_shift);
258 per_channel_multiplier[i] = significand;
259 per_channel_shift[i] = channel_shift;
260 }
261
262 // Populate scalar quantization parameters.
263 // This check on legacy quantization parameters is kept only for backward
264 // compatibility.
265 if (input->type == kTfLiteUInt8) {
266 // Check bias scale == input scale * filter scale.
267 double real_multiplier = 0.0;
268 TF_LITE_ENSURE_STATUS(GetQuantizedConvolutionMultipler(
269 context, input, filter, bias, output, &real_multiplier));
270 int exponent;
271
272 // Populate quantization parameters with multiplier and shift.
273 QuantizeMultiplier(real_multiplier, multiplier, &exponent);
274 *shift = -exponent;
275 }
276 if (input->type == kTfLiteInt8 || input->type == kTfLiteUInt8 ||
277 input->type == kTfLiteInt16) {
278 TF_LITE_ENSURE_STATUS(CalculateActivationRangeQuantized(
279 context, activation, output, output_activation_min,
280 output_activation_max));
281 }
282 return kTfLiteOk;
283}

285TfLiteStatus GetQuantizedConvolutionMultipler(TfLiteContext* context,
286 const TfLiteTensor* input,
287 const TfLiteTensor* filter,
288 const TfLiteTensor* bias,
289 TfLiteTensor* output,
290 double* multiplier) {
291 const double input_product_scale = static_cast<double>(input->params.scale) *
292 static_cast<double>(filter->params.scale);
293 // The following conditions must be guaranteed by the training pipeline.
294 if (bias) {
295 const double bias_scale = static_cast<double>(bias->params.scale);
296 // Here we're making sure the input_product_scale & bias_scale are about the
297 // same. Since we have:
298 // (output - output_zp) * output_scale =
299 // input_product_scale * input_product + bias * bias_scale ---- (0)
300 //
301 // (0) equals:
302 // (input_product + bias) * input_product_scale ----- (1)
303 // +
304 // bias * (bias_scale - input_product_scale) ------ (2)
305 //
306 // For the real kernel computation, we're doing (1), so we really need to
307 // make sure (2) has minimum impact on the output, so:
308 // bias * (bias_scale - input_product_scale) / output_scale should be
309 // a small number for an integer.
310 // Since normally bias should be within a small range.
311 // We should expect (bias_scale - input_product_scale) / output_scale to
312 // be a small number like 0.02.
313 const double scale_diff = std::abs(input_product_scale - bias_scale);
314 const double output_scale = static_cast<double>(output->params.scale);
315
316 TF_LITE_ENSURE(context, scale_diff / output_scale <= 0.02);
317 }
318 return GetQuantizedConvolutionMultipler(context, input, filter, output,
319 multiplier);
320}
321
322TfLiteStatus GetQuantizedConvolutionMultipler(TfLiteContext* context,
323 const TfLiteTensor* input,
324 const TfLiteTensor* filter,
325 TfLiteTensor* output,
326 double* multiplier) {
327 const double input_product_scale =
328 static_cast<double>(input->params.scale * filter->params.scale);
329 TF_LITE_ENSURE(context, input_product_scale >= 0);
330 *multiplier = input_product_scale / static_cast<double>(output->params.scale);
331
332 return kTfLiteOk;
333}

336void CalculateActivationRangeQuantizedImpl(TfLiteFusedActivation activation,
337 int32_t qmin, int32_t qmax,
338 TfLiteTensor* output,
339 int32_t* act_min, int32_t* act_max) {
340 const auto scale = output->params.scale;
341 const auto zero_point = output->params.zero_point;
342
343 auto quantize = [scale, zero_point](float f) {
344 return zero_point + static_cast<int32_t>(TfLiteRound(f / scale));
345 };
346
347 if (activation == kTfLiteActRelu) {
348 *act_min = std::max(qmin, quantize(0.0));
349 *act_max = qmax;
350 } else if (activation == kTfLiteActRelu6) {
351 *act_min = std::max(qmin, quantize(0.0));
352 *act_max = std::min(qmax, quantize(6.0));
353 } else if (activation == kTfLiteActReluN1To1) {
354 *act_min = std::max(qmin, quantize(-1.0));
355 *act_max = std::min(qmax, quantize(1.0));
356 } else {
357 *act_min = qmin;
358 *act_max = qmax;
359 }
360}
361} // namespace
362
363TfLiteStatus CalculateActivationRangeQuantized(TfLiteContext* context,
364 TfLiteFusedActivation activation,
365 TfLiteTensor* output,
366 int32_t* act_min,
367 int32_t* act_max) {
368 int32_t qmin = 0;
369 int32_t qmax = 0;
370 if (output->type == kTfLiteUInt8) {
371 qmin = std::numeric_limits<uint8_t>::min();
372 qmax = std::numeric_limits<uint8_t>::max();
373 } else if (output->type == kTfLiteInt8) {
374 qmin = std::numeric_limits<int8_t>::min();
375 qmax = std::numeric_limits<int8_t>::max();
376 } else if (output->type == kTfLiteInt16) {
377 qmin = std::numeric_limits<int16_t>::min();
378 qmax = std::numeric_limits<int16_t>::max();
379 } else {
380 TF_LITE_ENSURE(context, false);
381 }
382
383 CalculateActivationRangeQuantizedImpl(activation, qmin, qmax, output, act_min,
384 act_max);
385 return kTfLiteOk;
386}

399std::string GetShapeDebugString(const TfLiteIntArray* shape) {
400 std::string str;
401 for (int d = 0; d < shape->size; ++d) {
402 if (str.empty())
403 str = "[" + std::to_string(shape->data[d]);

405 str += ", " + std::to_string(shape->data[d]);

411TfLiteStatus CalculateShapeForBroadcast(TfLiteContext* context,
412 const TfLiteTensor* input1,
413 const TfLiteTensor* input2,
414 TfLiteIntArray** output_shape) {
415 int dims1 = NumDimensions(input1);
416 int dims2 = NumDimensions(input2);
417 int out_dims = std::max(dims1, dims2);
418 if (NumElements(input1) == 0) {
419 *output_shape = TfLiteIntArrayCopy(input1->dims);
420 return kTfLiteOk;
421 }
422 std::unique_ptr<TfLiteIntArray, void (*)(TfLiteIntArray*)> shape(
423 TfLiteIntArrayCreate(out_dims), TfLiteIntArrayFree);
424 for (int i = 0; i < out_dims; ++i) {
425 int d1 = i >= dims1 ? 1 : SizeOfDimension(input1, dims1 - i - 1);
426 int d2 = i >= dims2 ? 1 : SizeOfDimension(input2, dims2 - i - 1);
427 if (!(d1 == d2 || d1 == 1 || d2 == 1)) {
428 context->ReportError(context,
429 "Given shapes, %s and %s, are not broadcastable.",
430 GetShapeDebugString(input1->dims).c_str(),
431 GetShapeDebugString(input2->dims).c_str());
432 return kTfLiteError;
433 }
434 shape->data[out_dims - i - 1] = std::max(d1, d2);
435 }
436 *output_shape = shape.release();
437 return kTfLiteOk;
438}

440TfLiteStatus CalculateShapeForBroadcast(TfLiteContext* context,
441 const TfLiteTensor* input1,
442 const TfLiteTensor* input2,
443 const TfLiteTensor* input3,
444 TfLiteIntArray** output_shape) {
445 int dims1 = NumDimensions(input1);
446 int dims2 = NumDimensions(input2);
447 int dims3 = NumDimensions(input3);
448 int out_dims = std::max(std::max(dims1, dims2), dims3);
449 std::unique_ptr<TfLiteIntArray, void (*)(TfLiteIntArray*)> shape(
450 TfLiteIntArrayCreate(out_dims), TfLiteIntArrayFree);
451 for (int i = 0; i < out_dims; ++i) {
452 int d1 = i >= dims1 ? 1 : SizeOfDimension(input1, dims1 - i - 1);
453 int d2 = i >= dims2 ? 1 : SizeOfDimension(input2, dims2 - i - 1);
454 int d3 = i >= dims3 ? 1 : SizeOfDimension(input3, dims3 - i - 1);
455 int max_value = std::max(std::max(d1, d2), d3);
456 if (!(d1 == 1 || d1 == max_value) || !(d2 == 1 || d2 == max_value) ||
457 !(d3 == 1 || d3 == max_value)) {
458 context->ReportError(
459 context, "Given shapes, %s, %s and %s, are not broadcastable.",
460 GetShapeDebugString(input1->dims).c_str(),
461 GetShapeDebugString(input2->dims).c_str(),
462 GetShapeDebugString(input3->dims).c_str());
463 return kTfLiteError;
464 }
465 shape->data[out_dims - i - 1] = max_value;
466 }
467 *output_shape = shape.release();
468 return kTfLiteOk;
469}
470#endif // TF_LITE_STATIC_MEMORY

472// Size of string is not constant, return 0 in such case.
473int TfLiteTypeGetSize(TfLiteType type) {
474 switch (type) {
475 case kTfLiteUInt8:
476 TF_LITE_ASSERT_EQ(sizeof(uint8_t), 1);
477 return 1;
478 case kTfLiteInt8:
479 TF_LITE_ASSERT_EQ(sizeof(int8_t), 1);
480 return 1;
481 case kTfLiteBool:
482 return sizeof(bool);
483 case kTfLiteInt16:
484 TF_LITE_ASSERT_EQ(sizeof(int16_t), 2);
485 return 2;
486 case kTfLiteFloat16:
487 TF_LITE_ASSERT_EQ(sizeof(int16_t), 2);
488 return 2;
489 case kTfLiteFloat32:
490 TF_LITE_ASSERT_EQ(sizeof(float), 4);
491 return 4;
492 case kTfLiteInt32:
493 TF_LITE_ASSERT_EQ(sizeof(int32_t), 4);
494 return 4;
495 case kTfLiteUInt32:
496 TF_LITE_ASSERT_EQ(sizeof(uint32_t), 4);
497 return 4;
498 case kTfLiteInt64:
499 TF_LITE_ASSERT_EQ(sizeof(int64_t), 8);
500 return 8;
501 case kTfLiteUInt64:
502 TF_LITE_ASSERT_EQ(sizeof(uint64_t), 8);
503 return 8;
504 case kTfLiteFloat64:
505 TF_LITE_ASSERT_EQ(sizeof(double), 8);
506 return 8;
507 case kTfLiteComplex64:
508 TF_LITE_ASSERT_EQ(sizeof(std::complex<float>), 8);
509 return 8;
510 case kTfLiteComplex128:
511 TF_LITE_ASSERT_EQ(sizeof(std::complex<double>), 16);
512 return 16;
513 default:
514 return 0;
515 }
