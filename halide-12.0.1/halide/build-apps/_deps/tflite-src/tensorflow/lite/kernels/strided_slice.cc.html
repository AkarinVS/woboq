44constexpr int kInputTensor = 0;
45constexpr int kBeginTensor = 1;
46constexpr int kEndTensor = 2;
47constexpr int kStridesTensor = 3;
48constexpr int kOutputTensor = 0;
49
50struct StridedSliceContext {
51 StridedSliceContext(TfLiteContext* context, TfLiteNode* node) {
52 params = reinterpret_cast<TfLiteStridedSliceParams*>(node->builtin_data);
53 input = GetInput(context, node, kInputTensor);
54 begin = GetInput(context, node, kBeginTensor);
55 end = GetInput(context, node, kEndTensor);
56 strides = GetInput(context, node, kStridesTensor);
57 output = GetOutput(context, node, kOutputTensor);
58 dims = NumDimensions(input);
59 }
60 const TfLiteStridedSliceParams* params;
61 const TfLiteTensor* input;
62 const TfLiteTensor* begin;
63 const TfLiteTensor* end;
64 const TfLiteTensor* strides;
65 TfLiteTensor* output;
66 int dims;
67};
68
69StridedSliceParams BuildStridedSliceParams(StridedSliceContext* op_context) {
70 StridedSliceParams op_params;
71 op_params.start_indices_count = op_context->dims;
72 op_params.stop_indices_count = op_context->dims;
73 op_params.strides_count = op_context->dims;
74
75 op_params.begin_mask = op_context->params->begin_mask;
76 op_params.ellipsis_mask = 0;
77 op_params.end_mask = op_context->params->end_mask;
78 op_params.new_axis_mask = 0;
79 op_params.shrink_axis_mask = op_context->params->shrink_axis_mask;
80
81 int begin_count = GetTensorShape(op_context->begin).Dims(0);
82 for (int i = 0; i < begin_count; ++i) {
83 op_params.start_indices[i] = GetTensorData<int32_t>(op_context->begin)[i];
84 op_params.stop_indices[i] = GetTensorData<int32_t>(op_context->end)[i];
85 op_params.strides[i] = GetTensorData<int32_t>(op_context->strides)[i];
86 }
87
88 // If the length of begin and end smaller than number of input dims, set the
89 // mask bit of begin and end for that index.
90 for (int i = begin_count; i < op_context->dims; ++i) {
91 op_params.start_indices[i] = op_params.stop_indices[i] = 0;
92 op_params.strides[i] = 1;
93 op_params.begin_mask |= (1 << i);
94 op_params.end_mask |= (1 << i);
95 }
96 return op_params;
97}

99// Processes the indexing tensors (begin, end and strides) to resize the
100// output tensor. This function is callable from both Prepare() and Eval() as
101// long as the caller ensures the indexing tensors are present.
102TfLiteStatus ResizeOutputTensor(TfLiteContext* context,
103 StridedSliceContext* op_context) {
104 std::vector<int> output_shape_vector;
105 StridedSliceParams op_params = BuildStridedSliceParams(op_context);
106 RuntimeShape input_shape = GetTensorShape(op_context->input);
107
108 for (int idx = op_context->dims - 1; idx >= 0; --idx) {
109 int32_t stride = op_params.strides[idx];
110 TF_LITE_ENSURE_MSG(context, stride != 0, "stride value has to be non-zero");
111
112 int32_t begin =
113 ::tflite::strided_slice::StartForAxis(op_params, input_shape, idx);
114 int32_t end = ::tflite::strided_slice::StopForAxis(op_params, input_shape,
115 idx, begin);
116
117 // When shrinking an axis, the end position does not matter (and can be
118 // incorrect when negative indexing is used, see Issue #19260). Always use
119 // begin + 1 to generate a length 1 slice, since begin has
120 // already been adjusted for negative indices by GetBeginValueAtIndex.
121 const bool shrink_axis = op_context->params->shrink_axis_mask & (1 << idx);
122 if (shrink_axis) {
123 end = begin + 1;
124 }
125
126 // This is valid for both positive and negative strides
127 int32_t dim_shape = std::ceil((end - begin) / static_cast<float>(stride));
128 dim_shape = dim_shape < 0 ? 0 : dim_shape;
129 if (!shrink_axis) {
130 output_shape_vector.push_back(dim_shape);
131 }
132 }
133
134 TfLiteIntArray* output_shape =
135 TfLiteIntArrayCreate(output_shape_vector.size());
136
137 std::reverse_copy(output_shape_vector.begin(), output_shape_vector.end(),
138 output_shape->data);
139
140 TF_LITE_ENSURE_STATUS(
141 context->ResizeTensor(context, op_context->output, output_shape));
142
143 return kTfLiteOk;
144}

146TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
147 TF_LITE_ENSURE_EQ(context, NumInputs(node), 4);
148 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
149
150 StridedSliceContext op_context(context, node);
151
152 // Ensure validity of input tensor and its dimension
153 TF_LITE_ENSURE_EQ(context, NumDimensions(op_context.begin), 1);
154 TF_LITE_ENSURE_EQ(context, NumDimensions(op_context.end), 1);
155 TF_LITE_ENSURE_EQ(context, NumDimensions(op_context.strides), 1);
156 TF_LITE_ENSURE_EQ(context, op_context.input->type, op_context.output->type);
157 // Only INT32 begin/end/strides are supported
158 // TODO(b/175642009): add support for INT64
159 TF_LITE_ENSURE_TYPES_EQ(context, op_context.begin->type, kTfLiteInt32);
160 TF_LITE_ENSURE_TYPES_EQ(context, op_context.end->type, kTfLiteInt32);
161 TF_LITE_ENSURE_TYPES_EQ(context, op_context.strides->type, kTfLiteInt32);
162 TF_LITE_ENSURE_MSG(context, op_context.dims <= 5,
163 "StridedSlice op only supports 1D-5D input arrays.");
164
165 // TODO(b/138098220): Remove when bug is resolved.
166 // Currently, working on using the compiler to cannonize strided_slice,
167 // so ellipis_mask will become part of begin/end mask, new_axis_mask will
168 // involve in a reshape to pad the dimensions.
169 TF_LITE_ENSURE_MSG(context, op_context.params->ellipsis_mask == 0,
170 "ellipsis_mask is not implemented yet.");
171 TF_LITE_ENSURE_MSG(context, op_context.params->new_axis_mask == 0,
172 "new_axis_mask is not implemented yet.");
173
174 // Postpone allocation of output if any of the indexing tensors is not
175 // constant
176 if (!(IsConstantTensor(op_context.begin) &&
177 IsConstantTensor(op_context.end) &&
178 IsConstantTensor(op_context.strides))) {
179 SetTensorToDynamic(op_context.output);
180 return kTfLiteOk;
181 }
182 return ResizeOutputTensor(context, &op_context);
183}

185template <KernelType kernel_type>
186TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
187 StridedSliceContext op_context(context, node);
188
189 if (IsDynamicTensor(op_context.output)) {
190 TF_LITE_ENSURE_OK(context, ResizeOutputTensor(context, &op_context));
191 }
192 StridedSliceParams op_params = BuildStridedSliceParams(&op_context);
193
194#define TF_LITE_STRIDED_SLICE(data_type) \
195 { \
196 if (kernel_type == kGenericOptimized) { \
197 optimized_ops::StridedSlice<data_type>( \
198 op_params, GetTensorShape(op_context.input), op_context.input, \
199 GetTensorShape(op_context.output), op_context.output); \
200 } else { \
201 reference_ops::StridedSlice<data_type>( \
202 op_params, GetTensorShape(op_context.input), op_context.input, \
203 GetTensorShape(op_context.output), op_context.output); \
204 } \
205 }
206
207 switch (op_context.input->type) {
208 case kTfLiteFloat32:
209 TF_LITE_STRIDED_SLICE(float);
210 break;
211 case kTfLiteInt32:
212 TF_LITE_STRIDED_SLICE(int32_t);
213 break;
214 case kTfLiteInt64:
215 TF_LITE_STRIDED_SLICE(int64_t);
216 break;
217 case kTfLiteUInt8:
218 TF_LITE_STRIDED_SLICE(uint8_t);
219 break;
220 case kTfLiteInt8:
221 TF_LITE_STRIDED_SLICE(int8_t);
222 break;
223 case kTfLiteInt16:
224 TF_LITE_STRIDED_SLICE(int16_t);
225 break;
226 case kTfLiteBool:
227 TF_LITE_STRIDED_SLICE(bool);
228 break;
229 case kTfLiteString:
230 TF_LITE_STRIDED_SLICE(string);
231 break;
232 default:
233 TF_LITE_KERNEL_LOG(context,
234 "Type %s is currently not supported "
235 "by StridedSlice.",
236 TfLiteTypeGetName(op_context.input->type));
237 return kTfLiteError;
238 }
239#undef TF_LITE_STRIDED_SLICE
240 return kTfLiteOk;
241}
