36// This file has two implementation of Concatenation.
37enum KernelType {
38 kReference,
39 kGenericOptimized,
40};

42TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
43 auto* params =
44 reinterpret_cast<TfLiteConcatenationParams*>(node->builtin_data);
45 int axis = params->axis;
46 int num_inputs = node->inputs->size;
47
48 // The number of dimensions of the input tensors must match, and all
49 // dimensions except 'axis' must be equal.
50 const TfLiteTensor* t0;
51 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &t0));
52 TfLiteType input_type = t0->type;
53 if (axis < 0) axis += t0->dims->size;
54 TF_LITE_ENSURE(context, axis >= 0);
55 TF_LITE_ENSURE(context, axis < t0->dims->size);
56
57 TF_LITE_ENSURE_EQ(context, params->activation, kTfLiteActNone);
58 TF_LITE_ENSURE(context,
59 input_type == kTfLiteFloat32 || input_type == kTfLiteUInt8 ||
60 input_type == kTfLiteInt8 || input_type == kTfLiteInt16 ||
61 input_type == kTfLiteInt32 || input_type == kTfLiteInt64 ||
62 input_type == kTfLiteBool);
63
64 // Output dimensions will match input dimensions, except 'axis', which
65 // will be the sum of inputs
66 int sum_axis = t0->dims->data[axis];
67 for (int i = 1; i < num_inputs; ++i) {
68 const TfLiteTensor* t;
69 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, i, &t));
70 TF_LITE_ENSURE_EQ(context, t->dims->size, t0->dims->size);
71 TF_LITE_ENSURE_EQ(context, t->type, input_type);
72 for (int d = 0; d < t0->dims->size; ++d) {
73 if (d == axis) {
74 // Avoid integer overflow in sum_axis below
75 TF_LITE_ENSURE(context, t->dims->data[axis] >= 0);
76 TF_LITE_ENSURE(context, t->dims->data[axis] <=
77 std::numeric_limits<int>::max() - sum_axis);
78 sum_axis += t->dims->data[axis];
79 } else {
80 TF_LITE_ENSURE_EQ(context, t->dims->data[d], t0->dims->data[d]);
81 }
82 }
83 }
84
85 TfLiteIntArray* output_size = TfLiteIntArrayCreate(t0->dims->size);
86 for (int d = 0; d < t0->dims->size; ++d) {
87 output_size->data[d] = (d == axis) ? sum_axis : t0->dims->data[d];
88 }
89
90 TfLiteTensor* output;
91 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
92 TF_LITE_ENSURE_TYPES_EQ(context, output->type, input_type);
93
94 if (input_type == kTfLiteInt8) {
95 // Make sure there is no re-scaling needed for Int8 quantized kernel. This
96 // is a restriction we introduced to Int8 kernels.
97 VectorOfTensors<int8_t> all_inputs(*context, *node->inputs);
98 for (int i = 0; i < node->inputs->size; ++i) {
99 const TfLiteTensor* t;
100 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, i, &t));
101 TF_LITE_ENSURE_EQ(context, t->params.scale, output->params.scale);
102 TF_LITE_ENSURE_EQ(context, t->params.zero_point,
103 output->params.zero_point);
104 }
105 }
106
107 if (input_type == kTfLiteInt16) {
108 // Make sure that all Int16 inputs have a null zero-point.
109 for (int i = 0; i < node->inputs->size; ++i) {
110 const TfLiteTensor* t = GetInput(context, node, i);
111 TF_LITE_ENSURE_EQ(context, t->params.zero_point, 0);
112 }
113 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
114 }
115
116 return context->ResizeTensor(context, output, output_size);
117}
118
119template <KernelType kernel_type>
120TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
121 auto* params =
122 reinterpret_cast<TfLiteConcatenationParams*>(node->builtin_data);
123 int axis = params->axis;
124 TfLiteTensor* output;
125 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
126 if (axis < 0) axis += output->dims->size;
127
128// TODO(ahentz): Creating 'all_inputs' below is not very efficient. We should
129// allocate and populate these during Prepare().
130// TODO(ycling): Activation function parameter is ignored. For now we don't have
131// a model with a Concatenation with fused activation function.
132#define TF_LITE_CONCATENATION(scalar) \
133 { \
134 VectorOfTensors<scalar> all_inputs(*context, *node->inputs); \
135 tflite::ConcatenationParams op_params; \
136 op_params.axis = axis; \
137 op_params.inputs_count = node->inputs->size; \
138 if (kernel_type == kReference) { \
139 reference_ops::Concatenation(op_params, all_inputs.shapes(), \
140 all_inputs.data(), GetTensorShape(output), \
141 GetTensorData<scalar>(output)); \
142 } else { \
143 optimized_ops::Concatenation(op_params, all_inputs.shapes(), \
144 all_inputs.data(), GetTensorShape(output), \
145 GetTensorData<scalar>(output)); \
146 } \
147 }
148
149#define TF_LITE_CONCATENATION_QUANTIZED() \
150 { \
151 VectorOfQuantizedTensors all_inputs(*context, *node->inputs); \
152 tflite::ConcatenationParams op_params; \
153 op_params.axis = axis; \
154 op_params.input_zeropoint = all_inputs.zero_point(); \
155 op_params.input_scale = all_inputs.scale(); \
156 op_params.inputs_count = node->inputs->size; \
157 op_params.output_zeropoint = output->params.zero_point; \
158 op_params.output_scale = output->params.scale; \
159 if (kernel_type == kReference) { \
160 reference_ops::ConcatenationWithScaling( \
161 op_params, all_inputs.shapes(), all_inputs.data(), \
162 GetTensorShape(output), GetTensorData<uint8>(output)); \
163 } else { \
164 optimized_ops::ConcatenationWithScaling( \
165 op_params, all_inputs.shapes(), all_inputs.data(), \
166 GetTensorShape(output), GetTensorData<uint8>(output)); \
167 } \
168 }
169
170 switch (output->type) { // Already know in/outtypes are same.
171 case kTfLiteFloat32:
172 TF_LITE_CONCATENATION(float);
173 break;
174 case kTfLiteInt32:
175 TF_LITE_CONCATENATION(int32);
176 break;
177 case kTfLiteUInt8:
178 TF_LITE_CONCATENATION_QUANTIZED();
179 break;
180 case kTfLiteInt8:
181 TF_LITE_CONCATENATION(int8_t);
182 break;
183 case kTfLiteInt64:
184 TF_LITE_CONCATENATION(int64_t);
185 break;
186 case kTfLiteInt16:
187 TF_LITE_CONCATENATION(int16_t);
188 break;
189 case kTfLiteBool:
190 TF_LITE_CONCATENATION(bool);
191 break;
192 default:
193 context->ReportError(context, "Type '%s' is not supported currently.",
194 TfLiteTypeGetName(output->type));
195 return kTfLiteError;
196 }
197
198#undef TF_LITE_CONCATENATION_QUANTIZED
199#undef TF_LITE_CONCATENATION
200
201 return kTfLiteOk;
202}

208TfLiteRegistration* Register_CONCATENATION_REF() {
209 static TfLiteRegistration r = {
210 nullptr, nullptr, concatenation::Prepare,
211 concatenation::Eval<concatenation::kReference>};
212 return &r;
213}

215TfLiteRegistration* Register_CONCATENATION_GENERIC_OPT() {
216 static TfLiteRegistration r = {
217 nullptr, nullptr, concatenation::Prepare,
218 concatenation::Eval<concatenation::kGenericOptimized>};
219 return &r;
220}

222TfLiteRegistration* Register_CONCATENATION() {
223 // TODO(ahentz): It turns out the two versions of Concatenation are almost
224 // identical, so we should consider removing one.
225 return Register_CONCATENATION_GENERIC_OPT();
226}
