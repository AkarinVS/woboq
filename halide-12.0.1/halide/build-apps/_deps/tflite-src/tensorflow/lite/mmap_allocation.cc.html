27MMAPAllocation::MMAPAllocation(const char* filename,
28 ErrorReporter* error_reporter)
29 : MMAPAllocation(error_reporter, open(filename, O_RDONLY)) {
30 if (mmap_fd_ == -1) {
31 TF_LITE_REPORT_ERROR(error_reporter, "Could not open '%s'.", filename);
32 }
33}
34
35MMAPAllocation::MMAPAllocation(int fd, ErrorReporter* error_reporter)
36 : MMAPAllocation(error_reporter, dup(fd)) {
37 if (mmap_fd_ == -1) {
38 TF_LITE_REPORT_ERROR(error_reporter, "Failed to dup '%d' file descriptor.",
39 fd);
40 }
41}
42
43MMAPAllocation::MMAPAllocation(ErrorReporter* error_reporter, int owned_fd)
44 : Allocation(error_reporter, Allocation::Type::kMMap),
45 mmap_fd_(owned_fd),
46 mmapped_buffer_(MAP_FAILED),
47 buffer_size_bytes_(0) {
48 if (mmap_fd_ == -1) {
49 return;
50 }
51 struct stat sb;
52 fstat(mmap_fd_, &sb);
53 buffer_size_bytes_ = sb.st_size;
54 mmapped_buffer_ =
55 mmap(nullptr, buffer_size_bytes_, PROT_READ, MAP_SHARED, mmap_fd_, 0);
56 if (mmapped_buffer_ == MAP_FAILED) {
57 TF_LITE_REPORT_ERROR(error_reporter, "Mmap of '%d' failed.", mmap_fd_);
58 return;
59 }
60}
61
62MMAPAllocation::~MMAPAllocation() {
63 if (valid()) {
64 munmap(const_cast<void*>(mmapped_buffer_), buffer_size_bytes_);
65 }
66 if (mmap_fd_ != -1) close(mmap_fd_);
67}
68
69const void* MMAPAllocation::base() const { return mmapped_buffer_; }
70
71size_t MMAPAllocation::bytes() const { return buffer_size_bytes_; }
72
73bool MMAPAllocation::valid() const { return mmapped_buffer_ != MAP_FAILED; }
74
75bool MMAPAllocation::IsSupported() { return true; }
76
