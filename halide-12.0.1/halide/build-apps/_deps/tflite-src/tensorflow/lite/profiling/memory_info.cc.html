27const int MemoryUsage::kValueNotSet = 0;
28
29bool MemoryUsage::IsSupported() {
30#ifdef __linux__
31 return true;
32#endif
33 return false;
34}
35
36MemoryUsage GetMemoryUsage() {
37 MemoryUsage result;
38#ifdef __linux__
39 rusage res;
40 if (getrusage(RUSAGE_SELF, &res) == 0) {
41 result.max_rss_kb = res.ru_maxrss;
42 }
43 const auto mem = mallinfo();
44 result.total_allocated_bytes = mem.arena;
45 result.in_use_allocated_bytes = mem.uordblks;
46#endif
47 return result;
48}
49
50void MemoryUsage::AllStatsToStream(std::ostream* stream) const {
51 *stream << "max resident set size = " << max_rss_kb / 1024.0
52 << " MB, total malloc-ed size = "
53 << total_allocated_bytes / 1024.0 / 1024.0
54 << " MB, in-use allocated/mmapped size = "
55 << in_use_allocated_bytes / 1024.0 / 1024.0 << " MB";
56}
