26ResourceVariable::ResourceVariable() {
27 memset(&tensor_, 0, sizeof(TfLiteTensor));
28}
29
30ResourceVariable::ResourceVariable(ResourceVariable&& other) {
31 tensor_ = other.tensor_;
32 is_initialized_ = other.is_initialized_;
33
34 memset(&other.tensor_, 0, sizeof(TfLiteTensor));
35 other.is_initialized_ = false;
36}
37
38ResourceVariable::~ResourceVariable() {
39 if (is_initialized_) {
40 free(tensor_.data.raw);
41 if (tensor_.dims) {
42 TfLiteIntArrayFree(tensor_.dims);
43 }
44 }
45}
46
47TfLiteStatus ResourceVariable::AssignFrom(const TfLiteTensor* tensor) {
48 // Save the old allocated resources and attributes that we might use.
49 char* old_raw = tensor_.data.raw;
50 size_t old_bytes = tensor_.bytes;
51 TfLiteIntArray* old_dims = tensor_.dims;
52
53 // Copy primitive parameters.
54 memset(&tensor_, 0, sizeof(tensor_));
55 tensor_.allocation_type = kTfLiteDynamic;
56 tensor_.type = tensor->type;
57 tensor_.params = tensor->params;
58 tensor_.quantization = tensor->quantization;
59
60 // Copy old shape if possible otherwise create a new one.
61 if (TfLiteIntArrayEqual(old_dims, tensor->dims)) {
62 tensor_.dims = old_dims;
63 } else {
64 TfLiteIntArrayFree(old_dims);
65 tensor_.dims = TfLiteIntArrayCopy(tensor->dims);
66 }
67
68 // Reuse the same buffer if possible otherwise allocate a new one.
69 tensor_.data.raw = old_raw;
70 if (old_bytes != tensor->bytes) {
71 TfLiteTensorRealloc(tensor->bytes, &tensor_);
72 } else {
73 tensor_.bytes = old_bytes;
74 }
75
76 memcpy(tensor_.data.raw, tensor->data.raw, tensor_.bytes);
77 is_initialized_ = true;
78
79 return kTfLiteOk;
80}

82void CreateResourceVariableIfNotAvailable(ResourceMap* resources,
83 int resource_id) {
84 if (resources->count(resource_id) != 0) {
85 return;
86 }
87 resources->emplace(resource_id,
88 std::unique_ptr<ResourceVariable>(new ResourceVariable()));
89}
90
91ResourceVariable* GetResourceVariable(ResourceMap* resources, int resource_id) {
92 auto it = resources->find(resource_id);
93 if (it != resources->end()) {
94 return static_cast<ResourceVariable*>(it->second.get());
95 }
96 return nullptr;
97}
