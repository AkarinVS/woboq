25template <typename KeyType, typename ValueType>
26TfLiteStatus StaticHashtable<KeyType, ValueType>::Lookup(
27 TfLiteContext* context, const TfLiteTensor* keys, TfLiteTensor* values,
28 const TfLiteTensor* default_value) {
29 if (!is_initialized_) {
30 context->ReportError(context,
31 "hashtable need to be initialized before using");
32 return kTfLiteError;
33 }
34 const int size =
35 MatchingFlatSize(GetTensorShape(keys), GetTensorShape(values));
36
37 auto key_tensor_reader = TensorReader<KeyType>(keys);
38 auto value_tensor_writer = TensorWriter<ValueType>(values);
39 auto default_value_tensor_reader = TensorReader<ValueType>(default_value);
40 ValueType first_default_value = default_value_tensor_reader.GetData(0);
41
42 for (int i = 0; i < size; ++i) {
43 auto result = map_.find(key_tensor_reader.GetData(i));
44 if (result != map_.end()) {
45 value_tensor_writer.SetData(i, result->second);
46 } else {
47 value_tensor_writer.SetData(i, first_default_value);
48 }
49 }
50
51 // This is for a string tensor case in order to write buffer back to the
52 // actual tensor destination. Otherwise, it does nothing since the scalar data
53 // will be written into the tensor storage directly.
54 value_tensor_writer.Commit();
55
56 return kTfLiteOk;
57}
58
59template <typename KeyType, typename ValueType>
60TfLiteStatus StaticHashtable<KeyType, ValueType>::Import(
61 TfLiteContext* context, const TfLiteTensor* keys,
62 const TfLiteTensor* values) {
63 // Import nodes can be invoked twice because the converter will not extract
64 // the initializer graph separately from the original graph. The invocations
65 // after the first call will be ignored.
66 if (is_initialized_) {
67 return kTfLiteOk;
68 }
69
70 const int size =
71 MatchingFlatSize(GetTensorShape(keys), GetTensorShape(values));
72
73 auto key_tensor_reader = TensorReader<KeyType>(keys);
74 auto value_tensor_writer = TensorReader<ValueType>(values);
75 for (int i = 0; i < size; ++i) {
76 map_.insert({key_tensor_reader.GetData(i), value_tensor_writer.GetData(i)});
77 }
78
79 is_initialized_ = true;
80 return kTfLiteOk;
81}

83LookupInterface* CreateStaticHashtable(TfLiteType key_type,
84 TfLiteType value_type) {
85 if (key_type == kTfLiteInt64 && value_type == kTfLiteString) {
86 return new StaticHashtable<std::int64_t, std::string>(key_type, value_type);
87 } else if (key_type == kTfLiteString && value_type == kTfLiteInt64) {
88 return new StaticHashtable<std::string, std::int64_t>(key_type, value_type);
89 }
90 return nullptr;
91}

95void CreateHashtableResourceIfNotAvailable(ResourceMap* resources,
96 int resource_id,
97 TfLiteType key_dtype,
98 TfLiteType value_dtype) {
99 if (resources->count(resource_id) != 0) {
100 return;
101 }
102 auto* hashtable = internal::CreateStaticHashtable(key_dtype, value_dtype);
103 resources->emplace(resource_id, std::unique_ptr<LookupInterface>(hashtable));
104}
105
106LookupInterface* GetHashtableResource(ResourceMap* resources, int resource_id) {
107 auto it = resources->find(resource_id);
108 if (it != resources->end()) {
109 return static_cast<LookupInterface*>(it->second.get());
110 }
111 return nullptr;
112}
