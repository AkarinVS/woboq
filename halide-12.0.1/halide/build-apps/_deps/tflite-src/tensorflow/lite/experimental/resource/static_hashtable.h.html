31// A static hash table class. This hash table allows initialization one time in
32// its life cycle. This hash table implements Tensorflow core's HashTableV2 op.
33template <typename KeyType, typename ValueType>
34class StaticHashtable : public tflite::resource::LookupInterface {
35 public:
36 explicit StaticHashtable(TfLiteType key_type, TfLiteType value_type)
37 : key_type_(key_type), value_type_(value_type) {}
38 ~StaticHashtable() override {}
39
40 // Finds the corresponding value of the given keys tensor in the map and
41 // copies the result data to the given values tensor. If there is no matching
42 // value, it will write the default value into the matched position instead.
43 TfLiteStatus Lookup(TfLiteContext* context, const TfLiteTensor* keys,
44 TfLiteTensor* values,
45 const TfLiteTensor* default_value) override;
46
47 // Inserts the given key and value tensor data into the hash table.
48 TfLiteStatus Import(TfLiteContext* context, const TfLiteTensor* keys,
49 const TfLiteTensor* values) override;
50
51 // Returns the item size of the hash table.
52 size_t Size() override { return map_.size(); }
53
54 TfLiteType GetKeyType() const override { return key_type_; }
55 TfLiteType GetValueType() const override { return value_type_; }
56
57 TfLiteStatus CheckKeyAndValueTypes(TfLiteContext* context,
58 const TfLiteTensor* keys,
59 const TfLiteTensor* values) override {
60 TF_LITE_ENSURE_EQ(context, keys->type, key_type_);
61 TF_LITE_ENSURE_EQ(context, values->type, value_type_);
62 return kTfLiteOk;
63 }
64
65 // Returns true if the hash table is initialized.
66 bool IsInitialized() override { return is_initialized_; }
67
68 private:
69 TfLiteType key_type_;
70 TfLiteType value_type_;
71
72 std::unordered_map<KeyType, ValueType> map_;
73 bool is_initialized_ = false;
74};

77 TfLiteType key_type, TfLiteType value_type);
