15#ifndef TENSORFLOW_LITE_PYTHON_INTERPRETER_WRAPPER_INTERPRETER_WRAPPER_H_
16#define TENSORFLOW_LITE_PYTHON_INTERPRETER_WRAPPER_INTERPRETER_WRAPPER_H_
17
18#include <functional>
19#include <memory>
20#include <string>
21#include <vector>
22
23// Place `<locale>` before <Python.h> to avoid build failures in macOS.
24#include <locale>
25
26// The empty line above is on purpose as otherwise clang-format will
27// automatically move <Python.h> before <locale>.
28#include <Python.h>
29
30#include "tensorflow/lite/interpreter.h"
31
32struct TfLiteDelegate;
33
34// We forward declare TFLite classes here to avoid exposing them to SWIG.
35namespace tflite {
36class MutableOpResolver;
37class FlatBufferModel;
38
39namespace interpreter_wrapper {
40
41class PythonErrorReporter;
42
43class InterpreterWrapper {
44 public:
45 using Model = FlatBufferModel;
46
47 // SWIG caller takes ownership of pointer.
48 static InterpreterWrapper* CreateWrapperCPPFromFile(
49 const char* model_path, int op_resolver_id,
50 const std::vector<std::string>& registerers, std::string* error_msg,
51 bool preserve_all_tensors);
52 static InterpreterWrapper* CreateWrapperCPPFromFile(
53 const char* model_path, int op_resolver_id,
54 const std::vector<std::string>& registerers_by_name,
55 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
56 std::string* error_msg, bool preserve_all_tensors);
57
58 // SWIG caller takes ownership of pointer.
59 static InterpreterWrapper* CreateWrapperCPPFromBuffer(
60 PyObject* data, int op_resolver_id,
61 const std::vector<std::string>& registerers, std::string* error_msg,
62 bool preserve_all_tensors);
63 static InterpreterWrapper* CreateWrapperCPPFromBuffer(
64 PyObject* data, int op_resolver_id,
65 const std::vector<std::string>& registerers_by_name,
66 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
67 std::string* error_msg, bool preserve_all_tensors);
68
69 ~InterpreterWrapper();
70 PyObject* AllocateTensors();
71 PyObject* Invoke();
72
73 PyObject* InputIndices() const;
74 PyObject* OutputIndices() const;
75 PyObject* ResizeInputTensor(int i, PyObject* value, bool strict);
76
77 int NumTensors() const;
78 std::string TensorName(int i) const;
79 PyObject* TensorType(int i) const;
80 PyObject* TensorSize(int i) const;
81 PyObject* TensorSizeSignature(int i) const;
82 PyObject* TensorSparsityParameters(int i) const;
83 // Deprecated in favor of TensorQuantizationScales, below.
84 PyObject* TensorQuantization(int i) const;
85 PyObject* TensorQuantizationParameters(int i) const;
86 PyObject* SetTensor(int i, PyObject* value);
87 PyObject* GetTensor(int i) const;
88 PyObject* SetInputTensorFromSignatureDefName(const char* input_name,
89 const char* method_name,
90 PyObject* value);
91 PyObject* GetOutputTensorFromSignatureDefName(const char* output_name,
92 const char* method_name) const;
93 PyObject* GetSignatureDefs() const;
94 PyObject* ResetVariableTensors();
95
96 int NumNodes() const;
97 std::string NodeName(int i) const;
98 PyObject* NodeInputs(int i) const;
99 PyObject* NodeOutputs(int i) const;
100
101 // Returns a reference to tensor index i as a numpy array. The base_object
102 // should be the interpreter object providing the memory.
103 PyObject* tensor(PyObject* base_object, int i);
104
105 PyObject* SetNumThreads(int num_threads);
106
107 // Adds a delegate to the interpreter.
108 PyObject* ModifyGraphWithDelegate(TfLiteDelegate* delegate);
109
110 // Experimental and subject to change.
111 //
112 // Returns a pointer to the underlying interpreter.
113 Interpreter* interpreter() { return interpreter_.get(); }
114
115 private:
116 // Helper function to construct an `InterpreterWrapper` object.
117 // It only returns InterpreterWrapper if it can construct an `Interpreter`.
118 // Otherwise it returns `nullptr`.
119 static InterpreterWrapper* CreateInterpreterWrapper(
120 std::unique_ptr<Model> model, int op_resolver_id,
121 std::unique_ptr<PythonErrorReporter> error_reporter,
122 const std::vector<std::string>& registerers_by_name,
123 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
124 std::string* error_msg, bool preserve_all_tensors);
125
126 InterpreterWrapper(std::unique_ptr<Model> model,
127 std::unique_ptr<PythonErrorReporter> error_reporter,
128 std::unique_ptr<tflite::MutableOpResolver> resolver,
129 std::unique_ptr<Interpreter> interpreter);
130
131 // InterpreterWrapper is not copyable or assignable. We avoid the use of
132 // InterpreterWrapper() = delete here for SWIG compatibility.
133 InterpreterWrapper();
134 InterpreterWrapper(const InterpreterWrapper& rhs);
135
136 // Helper function to resize an input tensor.
137 PyObject* ResizeInputTensorImpl(int i, PyObject* value);
138
139 // The public functions which creates `InterpreterWrapper` should ensure all
140 // these member variables are initialized successfully. Otherwise it should
141 // report the error and return `nullptr`.
142 const std::unique_ptr<Model> model_;
143 const std::unique_ptr<PythonErrorReporter> error_reporter_;
144 const std::unique_ptr<tflite::MutableOpResolver> resolver_;
145 const std::unique_ptr<Interpreter> interpreter_;
146};
147
148} // namespace interpreter_wrapper
149} // namespace tflite
150
151#endif // TENSORFLOW_LITE_PYTHON_INTERPRETER_WRAPPER_INTERPRETER_WRAPPER_H_
