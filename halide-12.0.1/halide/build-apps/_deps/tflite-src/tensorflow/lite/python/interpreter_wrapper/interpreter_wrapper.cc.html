42#define TFLITE_PY_CHECK(x) \
43 if ((x) != kTfLiteOk) { \
44 return error_reporter_->exception(); \
45 }
46
47#define TFLITE_PY_TENSOR_BOUNDS_CHECK(i) \
48 if (i >= interpreter_->tensors_size() || i < 0) { \
49 PyErr_Format(PyExc_ValueError, \
50 "Invalid tensor index %d exceeds max tensor index %lu", i, \
51 interpreter_->tensors_size()); \
52 return nullptr; \
53 }
54
55#define TFLITE_PY_NODES_BOUNDS_CHECK(i) \
56 if (i >= interpreter_->nodes_size() || i < 0) { \
57 PyErr_Format(PyExc_ValueError, "Invalid node index"); \
58 return nullptr; \
59 }
60
61#define TFLITE_PY_ENSURE_VALID_INTERPRETER() \
62 if (!interpreter_) { \
63 PyErr_SetString(PyExc_ValueError, "Interpreter was not initialized."); \
64 return nullptr; \
65 }

74std::unique_ptr<Interpreter> CreateInterpreter(
75 const InterpreterWrapper::Model* model,
76 const tflite::MutableOpResolver& resolver, bool preserve_all_tensors) {
77 if (!model) {
78 return nullptr;
79 }
80
81 ::tflite::python::ImportNumpy();
82
83 std::unique_ptr<Interpreter> interpreter;
84 InterpreterBuilder builder(*model, resolver);
85 if (preserve_all_tensors) builder.PreserveAllTensorsExperimental();
86 if (builder(&interpreter) != kTfLiteOk) {
87 return nullptr;
88 }
89 return interpreter;
90}

92PyObject* PyArrayFromFloatVector(const float* data, npy_intp size) {
93 void* pydata = malloc(size * sizeof(float));
94 memcpy(pydata, data, size * sizeof(float));
95 PyObject* obj = PyArray_SimpleNewFromData(1, &size, NPY_FLOAT32, pydata);
96 PyArray_ENABLEFLAGS(reinterpret_cast<PyArrayObject*>(obj), NPY_ARRAY_OWNDATA);
97 return obj;

115PyObject* PyDictFromSparsityParam(const TfLiteSparsity& param) {
116 PyObject* result = PyDict_New();
117 PyDict_SetItemString(result, "traversal_order",
118 PyArrayFromIntVector(param.traversal_order->data,
119 param.traversal_order->size));
120 PyDict_SetItemString(
121 result, "block_map",
122 PyArrayFromIntVector(param.block_map->data, param.block_map->size));
123 PyObject* dim_metadata = PyList_New(param.dim_metadata_size);
124 for (int i = 0; i < param.dim_metadata_size; i++) {
125 PyObject* dim_metadata_i = PyDict_New();
126 if (param.dim_metadata[i].format == kTfLiteDimDense) {
127 PyDict_SetItemString(dim_metadata_i, "format", PyLong_FromSize_t(0));
128 PyDict_SetItemString(dim_metadata_i, "dense_size",
129 PyLong_FromSize_t(param.dim_metadata[i].dense_size));
130 } else {
131 PyDict_SetItemString(dim_metadata_i, "format", PyLong_FromSize_t(1));
132 const auto* array_segments = param.dim_metadata[i].array_segments;
133 const auto* array_indices = param.dim_metadata[i].array_indices;
134 PyDict_SetItemString(
135 dim_metadata_i, "array_segments",
136 PyArrayFromIntVector(array_segments->data, array_segments->size));
137 PyDict_SetItemString(
138 dim_metadata_i, "array_indices",
139 PyArrayFromIntVector(array_indices->data, array_indices->size));
140 }
141 PyList_SetItem(dim_metadata, i, dim_metadata_i);
142 }
143 PyDict_SetItemString(result, "dim_metadata", dim_metadata);
144 return result;
145}

147bool RegisterCustomOpByName(const char* registerer_name,
148 tflite::MutableOpResolver* resolver,
149 std::string* error_msg) {
150 // Registerer functions take a pointer to a BuiltinOpResolver as an input
151 // parameter and return void.
152 // TODO(b/137576229): We should implement this functionality in a more
153 // principled way.
154 typedef void (*RegistererFunctionType)(tflite::MutableOpResolver*);
155
156 // Look for the Registerer function by name.
157 RegistererFunctionType registerer = reinterpret_cast<RegistererFunctionType>(
158 SharedLibrary::GetSymbol(registerer_name));
159
160 // Fail in an informative way if the function was not found.
161 if (registerer == nullptr) {
162 *error_msg =
163 absl::StrFormat("Looking up symbol '%s' failed with error '%s'.",
164 registerer_name, SharedLibrary::GetError());
165 return false;
166 }
167
168 // Call the registerer with the resolver.
169 registerer(resolver);
170 return true;
171}

175static constexpr int kBuiltinOpResolver = 1;
176static constexpr int kBuiltinRefOpResolver = 2;
177static constexpr int kBuiltinOpResolverWithoutDefaultDelegates = 3;

179InterpreterWrapper* InterpreterWrapper::CreateInterpreterWrapper(
180 std::unique_ptr<InterpreterWrapper::Model> model, int op_resolver_id,
181 std::unique_ptr<PythonErrorReporter> error_reporter,
182 const std::vector<std::string>& registerers_by_name,
183 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
184 std::string* error_msg, bool preserve_all_tensors) {
185 if (!model) {
186 *error_msg = error_reporter->message();
187 return nullptr;
188 }
189
190 std::unique_ptr<tflite::MutableOpResolver> resolver;
191 switch (op_resolver_id) {
192 case kBuiltinOpResolver:
193 resolver = absl::make_unique<tflite::ops::builtin::BuiltinOpResolver>();
194 break;
195 case kBuiltinRefOpResolver:
196 resolver =
197 absl::make_unique<tflite::ops::builtin::BuiltinRefOpResolver>();
198 break;
199 case kBuiltinOpResolverWithoutDefaultDelegates:
200 resolver = absl::make_unique<
201 tflite::ops::builtin::BuiltinOpResolverWithoutDefaultDelegates>();
202 break;
203 default:
204 // This should not never happen because the eventual caller in
205 // interpreter.py should have passed a valid id here.
206 TFLITE_DCHECK(false);
207 return nullptr;
208 }
209
210 for (const auto& registerer : registerers_by_name) {
211 if (!RegisterCustomOpByName(registerer.c_str(), resolver.get(), error_msg))
212 return nullptr;
213 }
214 for (const auto& registerer : registerers_by_func) {
215 registerer(reinterpret_cast<uintptr_t>(resolver.get()));
216 }
217 auto interpreter =
218 CreateInterpreter(model.get(), *resolver, preserve_all_tensors);
219 if (!interpreter) {
220 *error_msg = error_reporter->message();
221 return nullptr;
222 }
223
224 InterpreterWrapper* wrapper =
225 new InterpreterWrapper(std::move(model), std::move(error_reporter),
226 std::move(resolver), std::move(interpreter));
227 return wrapper;
228}

248PyObject* InterpreterWrapper::Invoke() {
249 TFLITE_PY_ENSURE_VALID_INTERPRETER();
250
251 // Release the GIL so that we can run multiple interpreters in parallel
252 TfLiteStatus status_code = kTfLiteOk;
253 Py_BEGIN_ALLOW_THREADS; // To return can happen between this and end!
254 status_code = interpreter_->Invoke();
255 Py_END_ALLOW_THREADS;
256
257 TFLITE_PY_CHECK(
258 status_code); // don't move this into the Py_BEGIN/Py_End block
259
260 Py_RETURN_NONE;
261}

278PyObject* InterpreterWrapper::ResizeInputTensorImpl(int i, PyObject* value) {
279 TFLITE_PY_ENSURE_VALID_INTERPRETER();
280
281 std::unique_ptr<PyObject, PyDecrefDeleter> array_safe(
282 PyArray_FromAny(value, nullptr, 0, 0, NPY_ARRAY_CARRAY, nullptr));
283 if (!array_safe) {
284 PyErr_SetString(PyExc_ValueError,
285 "Failed to convert numpy value into readable tensor.");
286 return nullptr;
287 }
288
289 PyArrayObject* array = reinterpret_cast<PyArrayObject*>(array_safe.get());
290
291 if (PyArray_NDIM(array) != 1) {
292 PyErr_Format(PyExc_ValueError, "Shape should be 1D instead of %d.",
293 PyArray_NDIM(array));
294 return nullptr;
295 }
296
297 if (PyArray_TYPE(array) != NPY_INT32) {
298 PyErr_Format(PyExc_ValueError, "Shape must be type int32 (was %d).",
299 PyArray_TYPE(array));
300 return nullptr;
301 }
302
303 PyArray_ENABLEFLAGS(reinterpret_cast<PyArrayObject*>(array),
304 NPY_ARRAY_OWNDATA);
305 return PyArray_Return(reinterpret_cast<PyArrayObject*>(array));
306}

308PyObject* InterpreterWrapper::ResizeInputTensor(int i, PyObject* value,
309 bool strict) {
310 PyArrayObject* array =
311 reinterpret_cast<PyArrayObject*>(ResizeInputTensorImpl(i, value));
312 if (array == nullptr) {
313 return nullptr;
314 }
315
316 std::vector<int> dims(PyArray_SHAPE(array)[0]);
317 memcpy(dims.data(), PyArray_BYTES(array), dims.size() * sizeof(int));
318
319 if (strict) {
320 TFLITE_PY_CHECK(interpreter_->ResizeInputTensorStrict(i, dims));
321 } else {
322 TFLITE_PY_CHECK(interpreter_->ResizeInputTensor(i, dims));
323 }
324 Py_RETURN_NONE;
325}

343PyObject* InterpreterWrapper::TensorType(int i) const {
344 TFLITE_PY_ENSURE_VALID_INTERPRETER();
345 TFLITE_PY_TENSOR_BOUNDS_CHECK(i);
346
347 const TfLiteTensor* tensor = interpreter_->tensor(i);
348 if (tensor->type == kTfLiteNoType) {
349 PyErr_Format(PyExc_ValueError, "Tensor with no type found.");
350 return nullptr;
351 }
352
353 int code = python_utils::TfLiteTypeToPyArrayType(tensor->type);
354 if (code == -1) {
355 PyErr_Format(PyExc_ValueError, "Invalid tflite type code %d", code);
356 return nullptr;
357 }
358 return PyArray_TypeObjectFromType(code);
359}

414PyObject* InterpreterWrapper::TensorQuantizationParameters(int i) const {
415 TFLITE_PY_ENSURE_VALID_INTERPRETER();
416 TFLITE_PY_TENSOR_BOUNDS_CHECK(i);
417 const TfLiteTensor* tensor = interpreter_->tensor(i);
418 const TfLiteQuantization quantization = tensor->quantization;
419 float* scales_data = nullptr;
420 int32_t* zero_points_data = nullptr;
421 int32_t scales_size = 0;
422 int32_t zero_points_size = 0;
423 int32_t quantized_dimension = 0;
424 if (quantization.type == kTfLiteAffineQuantization) {
425 const TfLiteAffineQuantization* q_params =
426 reinterpret_cast<const TfLiteAffineQuantization*>(quantization.params);
427 if (q_params->scale) {
428 scales_data = q_params->scale->data;
429 scales_size = q_params->scale->size;
430 }
431 if (q_params->zero_point) {
432 zero_points_data = q_params->zero_point->data;
433 zero_points_size = q_params->zero_point->size;
434 }
435 quantized_dimension = q_params->quantized_dimension;
436 }
437 PyObject* scales_array = PyArrayFromFloatVector(scales_data, scales_size);
438 PyObject* zero_points_array =
439 PyArrayFromIntVector(zero_points_data, zero_points_size);
440
441 PyObject* result = PyTuple_New(3);
442 PyTuple_SET_ITEM(result, 0, scales_array);
443 PyTuple_SET_ITEM(result, 1, zero_points_array);
444 PyTuple_SET_ITEM(result, 2, PyLong_FromLong(quantized_dimension));
445 return result;
446}

448PyObject* InterpreterWrapper::SetTensor(int i, PyObject* value) {
449 TFLITE_PY_ENSURE_VALID_INTERPRETER();
450 TFLITE_PY_TENSOR_BOUNDS_CHECK(i);
451
452 std::unique_ptr<PyObject, PyDecrefDeleter> array_safe(
453 PyArray_FromAny(value, nullptr, 0, 0, NPY_ARRAY_CARRAY, nullptr));
454 if (!array_safe) {
455 PyErr_SetString(PyExc_ValueError,
456 "Failed to convert value into readable tensor.");
457 return nullptr;
458 }
459
460 PyArrayObject* array = reinterpret_cast<PyArrayObject*>(array_safe.get());
461 TfLiteTensor* tensor = interpreter_->tensor(i);
462
463 if (python_utils::TfLiteTypeFromPyArray(array) != tensor->type) {
464 PyErr_Format(PyExc_ValueError,
465 "Cannot set tensor:"
466 " Got value of type %s"
467 " but expected type %s for input %d, name: %s ",
468 TfLiteTypeGetName(python_utils::TfLiteTypeFromPyArray(array)),
469 TfLiteTypeGetName(tensor->type), i, tensor->name);
470 return nullptr;
471 }
472
473 if (PyArray_NDIM(array) != tensor->dims->size) {
474 PyErr_Format(PyExc_ValueError,
475 "Cannot set tensor: Dimension mismatch."
476 " Got %d"
477 " but expected %d for input %d.",
478 PyArray_NDIM(array), tensor->dims->size, i);
479 return nullptr;
480 }
481
482 for (int j = 0; j < PyArray_NDIM(array); j++) {
483 if (tensor->dims->data[j] != PyArray_SHAPE(array)[j]) {
484 PyErr_Format(PyExc_ValueError,
485 "Cannot set tensor: Dimension mismatch."
486 " Got %ld"
487 " but expected %d for dimension %d of input %d.",
488 PyArray_SHAPE(array)[j], tensor->dims->data[j], j, i);
489 return nullptr;
490 }
491 }
492
493 if (tensor->type != kTfLiteString) {
494 if (tensor->data.raw == nullptr) {
495 PyErr_Format(PyExc_ValueError,
496 "Cannot set tensor:"
497 " Tensor is unallocated. Try calling allocate_tensors()"
498 " first");
499 return nullptr;
500 }
501
502 size_t size = PyArray_NBYTES(array);
503 if (size != tensor->bytes) {
504 PyErr_Format(PyExc_ValueError,
505 "numpy array had %zu bytes but expected %zu bytes.", size,
506 tensor->bytes);
507 return nullptr;
508 }
509 memcpy(tensor->data.raw, PyArray_DATA(array), size);
510 } else {
511 DynamicBuffer dynamic_buffer;
512 if (!python_utils::FillStringBufferWithPyArray(value, &dynamic_buffer)) {
513 return nullptr;
514 }
515 dynamic_buffer.WriteToTensor(tensor, nullptr);
516 }
517 Py_RETURN_NONE;
518}

547std::string InterpreterWrapper::NodeName(int i) const {
548 if (!interpreter_ || i >= interpreter_->nodes_size() || i < 0) {
549 return "";
550 }
551 // Get op name from registration
552 const TfLiteRegistration* node_registration =
553 &(interpreter_->node_and_registration(i)->second);
554 int32_t op_code = node_registration->builtin_code;
555 std::string op_name;
556 if (op_code == tflite::BuiltinOperator_CUSTOM) {
557 const char* custom_name = node_registration->custom_name;
558 op_name = custom_name ? custom_name : "UnknownCustomOp";
559 } else {
560 op_name = tflite::EnumNamesBuiltinOperator()[op_code];
561 }
562 std::string op_name_str(op_name);
563 return op_name_str;
564}

568// Checks to see if a tensor access can succeed (returns nullptr on error).
569// Otherwise returns Py_None.
570PyObject* CheckGetTensorArgs(Interpreter* interpreter_, int tensor_index,
571 TfLiteTensor** tensor, int* type_num) {
572 TFLITE_PY_ENSURE_VALID_INTERPRETER();
573 TFLITE_PY_TENSOR_BOUNDS_CHECK(tensor_index);
574
575 *tensor = interpreter_->tensor(tensor_index);
576 if ((*tensor)->bytes == 0) {
577 PyErr_SetString(PyExc_ValueError, "Invalid tensor size.");
578 return nullptr;
579 }
580
581 *type_num = python_utils::TfLiteTypeToPyArrayType((*tensor)->type);
582 if (*type_num == -1) {
583 PyErr_SetString(PyExc_ValueError, "Unknown tensor type.");
584 return nullptr;
585 }
586
587 if (!(*tensor)->data.raw) {
588 PyErr_SetString(PyExc_ValueError,
589 "Tensor data is null."
590 " Run allocate_tensors() first");
591 return nullptr;
592 }
593
594 Py_RETURN_NONE;
595}

599PyObject* InterpreterWrapper::GetSignatureDefs() const {
600 PyObject* result = PyDict_New();
601 for (const auto& sig_def_name : interpreter_->signature_def_names()) {
602 PyObject* signature_def = PyDict_New();
603 PyObject* inputs = PyDict_New();
604 PyObject* outputs = PyDict_New();
605 const auto& signature_def_inputs =
606 interpreter_->signature_inputs(sig_def_name->c_str());
607 const auto& signature_def_outputs =
608 interpreter_->signature_outputs(sig_def_name->c_str());
609 for (const auto& input : signature_def_inputs) {
610 PyDict_SetItemString(inputs, input.first.c_str(),
611 PyLong_FromLong(input.second));
612 }
613 for (const auto& output : signature_def_outputs) {
614 PyDict_SetItemString(outputs, output.first.c_str(),
615 PyLong_FromLong(output.second));
616 }
617
618 PyDict_SetItemString(signature_def, "inputs", inputs);
619 PyDict_SetItemString(signature_def, "outputs", outputs);
620 PyDict_SetItemString(result, sig_def_name->c_str(), signature_def);
621 }
622 return result;
623}

633PyObject* InterpreterWrapper::SetInputTensorFromSignatureDefName(
634 const char* input_name, const char* method_name, PyObject* value) {
635 const auto& inputs = interpreter_->signature_inputs(method_name);
636 const auto& input = inputs.find(input_name);
637 if (input == inputs.end()) return nullptr;
638 return SetTensor(input->second, value);
639}

641PyObject* InterpreterWrapper::GetTensor(int i) const {
642 // Sanity check accessor
643 TfLiteTensor* tensor = nullptr;
644 int type_num = 0;
645
646 PyObject* check_result =
647 CheckGetTensorArgs(interpreter_.get(), i, &tensor, &type_num);
648 if (check_result == nullptr) return check_result;
649 Py_XDECREF(check_result);
650
651 std::vector<npy_intp> dims(tensor->dims->data,
652 tensor->dims->data + tensor->dims->size);
653 if (tensor->type != kTfLiteString && tensor->type != kTfLiteResource &&
654 tensor->type != kTfLiteVariant) {
655 // Make a buffer copy but we must tell Numpy It owns that data or else
656 // it will leak.
657 void* data = malloc(tensor->bytes);
658 if (!data) {
659 PyErr_SetString(PyExc_ValueError, "Malloc to copy tensor failed.");
660 return nullptr;
661 }
662 memcpy(data, tensor->data.raw, tensor->bytes);
663 PyObject* np_array;
664 if (tensor->sparsity == nullptr) {
665 np_array =
666 PyArray_SimpleNewFromData(dims.size(), dims.data(), type_num, data);
667 } else {
668 std::vector<npy_intp> sparse_buffer_dims(1);
669 size_t size_of_type;
670 if (GetSizeOfType(nullptr, tensor->type, &size_of_type) != kTfLiteOk) {
671 PyErr_SetString(PyExc_ValueError, "Unknown tensor type.");
672 free(data);
673 return nullptr;
674 }
675 sparse_buffer_dims[0] = tensor->bytes / size_of_type;
676 np_array = PyArray_SimpleNewFromData(
677 sparse_buffer_dims.size(), sparse_buffer_dims.data(), type_num, data);
678 }
679 PyArray_ENABLEFLAGS(reinterpret_cast<PyArrayObject*>(np_array),
680 NPY_ARRAY_OWNDATA);
681 return PyArray_Return(reinterpret_cast<PyArrayObject*>(np_array));
682 } else {
683 // Create a C-order array so the data is contiguous in memory.
684 const int32_t kCOrder = 0;
685 PyObject* py_object =
686 PyArray_EMPTY(dims.size(), dims.data(), NPY_OBJECT, kCOrder);
687
688 if (py_object == nullptr) {
689 PyErr_SetString(PyExc_MemoryError, "Failed to allocate PyArray.");
690 return nullptr;
691 }
692
693 PyArrayObject* py_array = reinterpret_cast<PyArrayObject*>(py_object);
694 PyObject** data = reinterpret_cast<PyObject**>(PyArray_DATA(py_array));
695 auto num_strings = GetStringCount(tensor);
696 for (int j = 0; j < num_strings; ++j) {
697 auto ref = GetString(tensor, j);
698
699 PyObject* bytes = PyBytes_FromStringAndSize(ref.str, ref.len);
700 if (bytes == nullptr) {
701 Py_DECREF(py_object);
702 PyErr_Format(PyExc_ValueError,
703 "Could not create PyBytes from string %d of input %d.", j,
704 i);
705 return nullptr;
706 }
707 // PyArray_EMPTY produces an array full of Py_None, which we must decref.
708 Py_DECREF(data[j]);
709 data[j] = bytes;
710 }
711 return py_object;
712 }
713}

715PyObject* InterpreterWrapper::tensor(PyObject* base_object, int i) {
716 // Sanity check accessor
717 TfLiteTensor* tensor = nullptr;
718 int type_num = 0;
719
720 PyObject* check_result =
721 CheckGetTensorArgs(interpreter_.get(), i, &tensor, &type_num);
722 if (check_result == nullptr) return check_result;
723 Py_XDECREF(check_result);
724
725 std::vector<npy_intp> dims(tensor->dims->data,
726 tensor->dims->data + tensor->dims->size);
727 PyArrayObject* np_array =
728 reinterpret_cast<PyArrayObject*>(PyArray_SimpleNewFromData(
729 dims.size(), dims.data(), type_num, tensor->data.raw));
730 Py_INCREF(base_object); // SetBaseObject steals, so we need to add.
731 PyArray_SetBaseObject(np_array, base_object);
732 return PyArray_Return(np_array);
733}
