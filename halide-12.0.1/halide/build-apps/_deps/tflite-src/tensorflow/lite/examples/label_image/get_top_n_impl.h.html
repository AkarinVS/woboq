30// Returns the top N confidence values over threshold in the provided vector,
31// sorted by confidence in descending order.
32template <class T>
33void get_top_n(T* prediction, int prediction_size, size_t num_results,
34 float threshold, std::vector<std::pair<float, int>>* top_results,
35 TfLiteType input_type) {
36 // Will contain top N results in ascending order.
37 std::priority_queue<std::pair<float, int>, std::vector<std::pair<float, int>>,
38 std::greater<std::pair<float, int>>>
39 top_result_pq;
40
41 const long count = prediction_size; // NOLINT(runtime/int)
42 float value = 0.0;
43
44 for (int i = 0; i < count; ++i) {
45 switch (input_type) {
46 case kTfLiteFloat32:
47 value = prediction[i];
48 break;
49 case kTfLiteInt8:
50 value = (prediction[i] + 128) / 256.0;
51 break;
52 case kTfLiteUInt8:
53 value = prediction[i] / 255.0;
54 break;
55 default:
56 break;
57 }
58 // Only add it if it beats the threshold and has a chance at being in
59 // the top N.
60 if (value < threshold) {
61 continue;
62 }
63
64 top_result_pq.push(std::pair<float, int>(value, i));
65
66 // If at capacity, kick the smallest value out.
67 if (top_result_pq.size() > num_results) {
68 top_result_pq.pop();
69 }
70 }
71
72 // Copy to output vector and reverse into descending order.
73 while (!top_result_pq.empty()) {
74 top_results->push_back(top_result_pq.top());
75 top_result_pq.pop();
76 }
77 std::reverse(top_results->begin(), top_results->end());
78}
