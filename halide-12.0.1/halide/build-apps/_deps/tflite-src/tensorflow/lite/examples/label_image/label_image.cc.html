62using TfLiteDelegatePtr = tflite::Interpreter::TfLiteDelegatePtr;
63using TfLiteDelegatePtrMap = std::map<std::string, TfLiteDelegatePtr>;

65class DelegateProviders {
66 public:
67 DelegateProviders()
68 : delegates_list_(tflite::tools::GetRegisteredDelegateProviders()) {
69 for (const auto& delegate : delegates_list_) {
70 params_.Merge(delegate->DefaultParams());
71 }
72 }
73
74 // Initialize delegate-related parameters from parsing command line arguments,
75 // and remove the matching arguments from (*argc, argv). Returns true if all
76 // recognized arg values are parsed correctly.
77 bool InitFromCmdlineArgs(int* argc, const char** argv) {
78 std::vector<tflite::Flag> flags;
79 for (const auto& delegate : delegates_list_) {
80 auto delegate_flags = delegate->CreateFlags(&params_);
81 flags.insert(flags.end(), delegate_flags.begin(), delegate_flags.end());
82 }
83
84 const bool parse_result = Flags::Parse(argc, argv, flags);
85 if (!parse_result) {
86 std::string usage = Flags::Usage(argv[0], flags);
87 LOG(ERROR) << usage;
88 }
89 return parse_result;
90 }
91
92 // Create a list of TfLite delegates based on what have been initialized (i.e.
93 // 'params_').
94 TfLiteDelegatePtrMap CreateAllDelegates() const {
95 TfLiteDelegatePtrMap delegates_map;
96 for (const auto& delegate : delegates_list_) {
97 auto ptr = delegate->CreateTfLiteDelegate(params_);
98 // It's possible that a delegate of certain type won't be created as
99 // user-specified benchmark params tells not to.
100 if (ptr == nullptr) continue;
101 LOG(INFO) << delegate->GetName() << " delegate created.";
102 delegates_map.emplace(delegate->GetName(), std::move(ptr));
103 }
104 return delegates_map;
105 }
106
107 private:
108 // Contain delegate-related parameters that are initialized from command-line
109 // flags.
110 tflite::tools::ToolParams params_;
111
112 const tflite::tools::DelegateProviderList& delegates_list_;
113};

115TfLiteDelegatePtr CreateGPUDelegate(Settings* s) {
116#if defined(__ANDROID__)
117 TfLiteGpuDelegateOptionsV2 gpu_opts = TfLiteGpuDelegateOptionsV2Default();
118 gpu_opts.inference_preference =
119 TFLITE_GPU_INFERENCE_PREFERENCE_SUSTAINED_SPEED;
120 gpu_opts.inference_priority1 =
121 s->allow_fp16 ? TFLITE_GPU_INFERENCE_PRIORITY_MIN_LATENCY
122 : TFLITE_GPU_INFERENCE_PRIORITY_MAX_PRECISION;
123 return evaluation::CreateGPUDelegate(&gpu_opts);
124#else
125 return evaluation::CreateGPUDelegate();
126#endif
127}

129TfLiteDelegatePtrMap GetDelegates(Settings* s,
130 const DelegateProviders& delegate_providers) {
131 // TODO(b/169681115): deprecate delegate creation path based on "Settings" by
132 // mapping settings to DelegateProvider's parameters.
133 TfLiteDelegatePtrMap delegates;
134 if (s->gl_backend) {
135 auto delegate = CreateGPUDelegate(s);
136 if (!delegate) {
137 LOG(INFO) << "GPU acceleration is unsupported on this platform.";
138 } else {
139 LOG(INFO) << "Use GPU acceleration.";
140 delegates.emplace("GPU", std::move(delegate));
141 }
142 }
143
144 if (s->accel) {
145 StatefulNnApiDelegate::Options options;
146 options.allow_fp16 = s->allow_fp16;
147 auto delegate = evaluation::CreateNNAPIDelegate(options);
148 if (!delegate) {
149 LOG(INFO) << "NNAPI acceleration is unsupported on this platform.";
150 } else {
151 LOG(INFO) << "Use NNAPI acceleration.";
152 delegates.emplace("NNAPI", std::move(delegate));
153 }
154 }
155
156 if (s->hexagon_delegate) {
157 const std::string libhexagon_path("/data/local/tmp");
158 auto delegate =
159 evaluation::CreateHexagonDelegate(libhexagon_path, s->profiling);
160
161 if (!delegate) {
162 LOG(INFO) << "Hexagon acceleration is unsupported on this platform.";
163 } else {
164 LOG(INFO) << "Use Hexagon acceleration.";
165 delegates.emplace("Hexagon", std::move(delegate));
166 }
167 }
168
169 if (s->xnnpack_delegate) {
170 auto delegate = evaluation::CreateXNNPACKDelegate(s->number_of_threads);
171 if (!delegate) {
172 LOG(INFO) << "XNNPACK acceleration is unsupported on this platform.";
173 } else {
174 LOG(INFO) << "Use XNNPACK acceleration.";
175 delegates.emplace("XNNPACK", std::move(delegate));
176 }
177 }
178
179 // Independent of above delegate creation options that are specific to this
180 // binary, we use delegate providers to create TFLite delegates. Delegate
181 // providers have been used in TFLite benchmark/evaluation tools and testing
182 // so that we have a single and more comprehensive set of command line
183 // arguments for delegate creation.
184 TfLiteDelegatePtrMap delegates_from_providers =
185 delegate_providers.CreateAllDelegates();
186 for (auto& name_and_delegate : delegates_from_providers) {
187 delegates.emplace("Delegate_Provider_" + name_and_delegate.first,
188 std::move(name_and_delegate.second));
189 }
190
191 return delegates;
192}

194// Takes a file name, and loads a list of labels from it, one per line, and
195// returns a vector of the strings. It pads with empty strings so the length
196// of the result is a multiple of 16, because our model expects that.
197TfLiteStatus ReadLabelsFile(const string& file_name,
198 std::vector<string>* result,
199 size_t* found_label_count) {
200 std::ifstream file(file_name);
201 if (!file) {
202 LOG(ERROR) << "Labels file " << file_name << " not found";
203 return kTfLiteError;
204 }
205 result->clear();
206 string line;
207 while (std::getline(file, line)) {
208 result->push_back(line);
209 }
210 *found_label_count = result->size();
211 const int padding = 16;
212 while (result->size() % padding) {
213 result->emplace_back();
214 }
215 return kTfLiteOk;
216}

218void PrintProfilingInfo(const profiling::ProfileEvent* e,
219 uint32_t subgraph_index, uint32_t op_index,
220 TfLiteRegistration registration) {
221 // output something like
222 // time (ms) , Node xxx, OpCode xxx, symbolic name
223 // 5.352, Node 5, OpCode 4, DEPTHWISE_CONV_2D
224
225 LOG(INFO) << std::fixed << std::setw(10) << std::setprecision(3)
226 << (e->end_timestamp_us - e->begin_timestamp_us) / 1000.0
227 << ", Subgraph " << std::setw(3) << std::setprecision(3)
228 << subgraph_index << ", Node " << std::setw(3)
229 << std::setprecision(3) << op_index << ", OpCode " << std::setw(3)
230 << std::setprecision(3) << registration.builtin_code << ", "
231 << EnumNameBuiltinOperator(
232 static_cast<BuiltinOperator>(registration.builtin_code));
233}

235void RunInference(Settings* settings,
236 const DelegateProviders& delegate_providers) {
237 if (!settings->model_name.c_str()) {
238 LOG(ERROR) << "no model file name";
239 exit(-1);
240 }
241
242 std::unique_ptr<tflite::FlatBufferModel> model;
243 std::unique_ptr<tflite::Interpreter> interpreter;
244 model = tflite::FlatBufferModel::BuildFromFile(settings->model_name.c_str());
245 if (!model) {
246 LOG(ERROR) << "Failed to mmap model " << settings->model_name;
247 exit(-1);
248 }
249 settings->model = model.get();
250 LOG(INFO) << "Loaded model " << settings->model_name;
251 model->error_reporter();
252 LOG(INFO) << "resolved reporter";
253
254 tflite::ops::builtin::BuiltinOpResolver resolver;
255
256 tflite::InterpreterBuilder(*model, resolver)(&interpreter);
257 if (!interpreter) {
258 LOG(ERROR) << "Failed to construct interpreter";
259 exit(-1);
260 }
261
262 interpreter->SetAllowFp16PrecisionForFp32(settings->allow_fp16);
263
264 if (settings->verbose) {
265 LOG(INFO) << "tensors size: " << interpreter->tensors_size();
266 LOG(INFO) << "nodes size: " << interpreter->nodes_size();
267 LOG(INFO) << "inputs: " << interpreter->inputs().size();
268 LOG(INFO) << "input(0) name: " << interpreter->GetInputName(0);
269
270 int t_size = interpreter->tensors_size();
271 for (int i = 0; i < t_size; i++) {
272 if (interpreter->tensor(i)->name)
273 LOG(INFO) << i << ": " << interpreter->tensor(i)->name << ", "
274 << interpreter->tensor(i)->bytes << ", "
275 << interpreter->tensor(i)->type << ", "
276 << interpreter->tensor(i)->params.scale << ", "
277 << interpreter->tensor(i)->params.zero_point;
278 }
279 }
280
281 if (settings->number_of_threads != -1) {
282 interpreter->SetNumThreads(settings->number_of_threads);
283 }
284
285 int image_width = 224;
286 int image_height = 224;
287 int image_channels = 3;
288 std::vector<uint8_t> in = read_bmp(settings->input_bmp_name, &image_width,
289 &image_height, &image_channels, settings);
290
291 int input = interpreter->inputs()[0];
292 if (settings->verbose) LOG(INFO) << "input: " << input;
293
294 const std::vector<int> inputs = interpreter->inputs();
295 const std::vector<int> outputs = interpreter->outputs();
296
297 if (settings->verbose) {
298 LOG(INFO) << "number of inputs: " << inputs.size();
299 LOG(INFO) << "number of outputs: " << outputs.size();
300 }
301
302 auto delegates_ = GetDelegates(settings, delegate_providers);
303 for (const auto& delegate : delegates_) {
304 if (interpreter->ModifyGraphWithDelegate(delegate.second.get()) !=
305 kTfLiteOk) {
306 LOG(ERROR) << "Failed to apply " << delegate.first << " delegate.";
307 exit(-1);
308 } else {
309 LOG(INFO) << "Applied " << delegate.first << " delegate.";
310 }
311 }
312
313 if (interpreter->AllocateTensors() != kTfLiteOk) {
314 LOG(ERROR) << "Failed to allocate tensors!";
315 exit(-1);
316 }
317
318 if (settings->verbose) PrintInterpreterState(interpreter.get());
319
320 // get input dimension from the input tensor metadata
321 // assuming one input only
322 TfLiteIntArray* dims = interpreter->tensor(input)->dims;
323 int wanted_height = dims->data[1];
324 int wanted_width = dims->data[2];
325 int wanted_channels = dims->data[3];
326
327 settings->input_type = interpreter->tensor(input)->type;
328 switch (settings->input_type) {
329 case kTfLiteFloat32:
330 resize<float>(interpreter->typed_tensor<float>(input), in.data(),
331 image_height, image_width, image_channels, wanted_height,
332 wanted_width, wanted_channels, settings);
333 break;
334 case kTfLiteInt8:
335 resize<int8_t>(interpreter->typed_tensor<int8_t>(input), in.data(),
336 image_height, image_width, image_channels, wanted_height,
337 wanted_width, wanted_channels, settings);
338 break;
339 case kTfLiteUInt8:
340 resize<uint8_t>(interpreter->typed_tensor<uint8_t>(input), in.data(),
341 image_height, image_width, image_channels, wanted_height,
342 wanted_width, wanted_channels, settings);
343 break;
344 default:
345 LOG(ERROR) << "cannot handle input type "
346 << interpreter->tensor(input)->type << " yet";
347 exit(-1);
348 }
349 auto profiler = absl::make_unique<profiling::Profiler>(
350 settings->max_profiling_buffer_entries);
351 interpreter->SetProfiler(profiler.get());
352
353 if (settings->profiling) profiler->StartProfiling();
354 if (settings->loop_count > 1) {
355 for (int i = 0; i < settings->number_of_warmup_runs; i++) {
356 if (interpreter->Invoke() != kTfLiteOk) {
357 LOG(ERROR) << "Failed to invoke tflite!";
358 exit(-1);
359 }
360 }
361 }
362
363 struct timeval start_time, stop_time;
364 gettimeofday(&start_time, nullptr);
365 for (int i = 0; i < settings->loop_count; i++) {
366 if (interpreter->Invoke() != kTfLiteOk) {
367 LOG(ERROR) << "Failed to invoke tflite!";
368 exit(-1);
369 }
370 }
371 gettimeofday(&stop_time, nullptr);
372 LOG(INFO) << "invoked";
373 LOG(INFO) << "average time: "
374 << (get_us(stop_time) - get_us(start_time)) /
375 (settings->loop_count * 1000)
376 << " ms";
377
378 if (settings->profiling) {
379 profiler->StopProfiling();
380 auto profile_events = profiler->GetProfileEvents();
381 for (int i = 0; i < profile_events.size(); i++) {
382 auto subgraph_index = profile_events[i]->extra_event_metadata;
383 auto op_index = profile_events[i]->event_metadata;
384 const auto subgraph = interpreter->subgraph(subgraph_index);
385 const auto node_and_registration =
386 subgraph->node_and_registration(op_index);
387 const TfLiteRegistration registration = node_and_registration->second;
388 PrintProfilingInfo(profile_events[i], subgraph_index, op_index,
389 registration);
390 }
391 }
392
393 const float threshold = 0.001f;
394
395 std::vector<std::pair<float, int>> top_results;
396
397 int output = interpreter->outputs()[0];
398 TfLiteIntArray* output_dims = interpreter->tensor(output)->dims;
399 // assume output dims to be something like (1, 1, ... ,size)
400 auto output_size = output_dims->data[output_dims->size - 1];
401 switch (interpreter->tensor(output)->type) {
402 case kTfLiteFloat32:
403 get_top_n<float>(interpreter->typed_output_tensor<float>(0), output_size,
404 settings->number_of_results, threshold, &top_results,
405 settings->input_type);
406 break;
407 case kTfLiteInt8:
408 get_top_n<int8_t>(interpreter->typed_output_tensor<int8_t>(0),
409 output_size, settings->number_of_results, threshold,
410 &top_results, settings->input_type);
411 break;
412 case kTfLiteUInt8:
413 get_top_n<uint8_t>(interpreter->typed_output_tensor<uint8_t>(0),
414 output_size, settings->number_of_results, threshold,
415 &top_results, settings->input_type);
416 break;
417 default:
418 LOG(ERROR) << "cannot handle output type "
419 << interpreter->tensor(output)->type << " yet";
420 exit(-1);
421 }
422
423 std::vector<string> labels;
424 size_t label_count;
425
426 if (ReadLabelsFile(settings->labels_file_name, &labels, &label_count) !=
427 kTfLiteOk)
428 exit(-1);
429
430 for (const auto& result : top_results) {
431 const float confidence = result.first;
432 const int index = result.second;
433 LOG(INFO) << confidence << ": " << index << " " << labels[index];
434 }
435}

437void display_usage() {
438 LOG(INFO)
439 << "label_image\n"
440 << "--accelerated, -a: [0|1], use Android NNAPI or not\n"
441 << "--allow_fp16, -f: [0|1], allow running fp32 models with fp16 or not\n"
442 << "--count, -c: loop interpreter->Invoke() for certain times\n"
443 << "--gl_backend, -g: [0|1]: use GL GPU Delegate on Android\n"
444 << "--hexagon_delegate, -j: [0|1]: use Hexagon Delegate on Android\n"
445 << "--input_mean, -b: input mean\n"
446 << "--input_std, -s: input standard deviation\n"
447 << "--image, -i: image_name.bmp\n"
448 << "--labels, -l: labels for the model\n"
449 << "--tflite_model, -m: model_name.tflite\n"
450 << "--profiling, -p: [0|1], profiling or not\n"
451 << "--num_results, -r: number of results to show\n"
452 << "--threads, -t: number of threads\n"
453 << "--verbose, -v: [0|1] print more information\n"
454 << "--warmup_runs, -w: number of warmup runs\n"
455 << "--xnnpack_delegate, -x [0:1]: xnnpack delegate\n";
456}

458int Main(int argc, char** argv) {
459 DelegateProviders delegate_providers;
460 bool parse_result = delegate_providers.InitFromCmdlineArgs(
461 &argc, const_cast<const char**>(argv));
462 if (!parse_result) {
463 return EXIT_FAILURE;
464 }
465
466 Settings s;
467
468 int c;
469 while (true) {
470 static struct option long_options[] = {
471 {"accelerated", required_argument, nullptr, 'a'},
472 {"allow_fp16", required_argument, nullptr, 'f'},
473 {"count", required_argument, nullptr, 'c'},
474 {"verbose", required_argument, nullptr, 'v'},
475 {"image", required_argument, nullptr, 'i'},
476 {"labels", required_argument, nullptr, 'l'},
477 {"tflite_model", required_argument, nullptr, 'm'},
478 {"profiling", required_argument, nullptr, 'p'},
479 {"threads", required_argument, nullptr, 't'},
480 {"input_mean", required_argument, nullptr, 'b'},
481 {"input_std", required_argument, nullptr, 's'},
482 {"num_results", required_argument, nullptr, 'r'},
483 {"max_profiling_buffer_entries", required_argument, nullptr, 'e'},
484 {"warmup_runs", required_argument, nullptr, 'w'},
485 {"gl_backend", required_argument, nullptr, 'g'},
486 {"hexagon_delegate", required_argument, nullptr, 'j'},
487 {"xnnpack_delegate", required_argument, nullptr, 'x'},
488 {nullptr, 0, nullptr, 0}};
489
490 /* getopt_long stores the option index here. */
491 int option_index = 0;
492
493 c = getopt_long(argc, argv,
494 "a:b:c:d:e:f:g:i:j:l:m:p:r:s:t:v:w:x:", long_options,
495 &option_index);
496
497 /* Detect the end of the options. */
498 if (c == -1) break;
499
500 switch (c) {
501 case 'a':
502 s.accel = strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
503 break;
504 case 'b':
505 s.input_mean = strtod(optarg, nullptr);
506 break;
507 case 'c':
508 s.loop_count =
509 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
510 break;
511 case 'e':
512 s.max_profiling_buffer_entries =
513 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
514 break;
515 case 'f':
516 s.allow_fp16 =
517 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
518 break;
519 case 'g':
520 s.gl_backend =
521 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
522 break;
523 case 'i':
524 s.input_bmp_name = optarg;
525 break;
526 case 'j':
527 s.hexagon_delegate = optarg;
528 break;
529 case 'l':
530 s.labels_file_name = optarg;
531 break;
532 case 'm':
533 s.model_name = optarg;
534 break;
535 case 'p':
536 s.profiling =
537 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
538 break;
539 case 'r':
540 s.number_of_results =
541 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
542 break;
543 case 's':
544 s.input_std = strtod(optarg, nullptr);
545 break;
546 case 't':
547 s.number_of_threads = strtol( // NOLINT(runtime/deprecated_fn)
548 optarg, nullptr, 10);
549 break;
550 case 'v':
551 s.verbose =
552 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
553 break;
554 case 'w':
555 s.number_of_warmup_runs =
556 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
557 break;
558 case 'x':
559 s.xnnpack_delegate =
560 strtol(optarg, nullptr, 10); // NOLINT(runtime/deprecated_fn)
561 break;
562 case 'h':
563 case '?':
564 /* getopt_long already printed an error message. */
565 display_usage();
566 exit(-1);
567 default:
568 exit(-1);
569 }
570 }
571 RunInference(&s, delegate_providers);
572 return 0;
573}
