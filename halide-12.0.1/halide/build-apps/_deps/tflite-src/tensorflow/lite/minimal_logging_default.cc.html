25void MinimalLogger::LogFormatted(LogSeverity severity, const char* format,
26 va_list args) {
27 fprintf(stderr, "%s: ", GetSeverityName(severity));
28#pragma clang diagnostic push
29#pragma clang diagnostic ignored "-Wformat-nonliteral"
30 vfprintf(stderr, format, args);
31#pragma clang diagnostic pop
32 fputc('\n', stderr);
33}
