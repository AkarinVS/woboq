33// Utility class for safely allocating POD data. This is useful for avoiding
34// leaks in cases where op params are allocated but fail to propagate to the
35// parsed op data (e.g., when model parameters are invalid).
36class SafeBuiltinDataAllocator {
37 public:
38 class BuiltinDataDeleter {
39 public:
40 explicit BuiltinDataDeleter(BuiltinDataAllocator* allocator)
41 : allocator_(allocator) {}
42
43 void operator()(void* data) { allocator_->Deallocate(data); }
44
45 private:
46 BuiltinDataAllocator* allocator_;
47 };
48
49 template <typename T>
50 using BuiltinDataPtr = std::unique_ptr<T, BuiltinDataDeleter>;
51
52 explicit SafeBuiltinDataAllocator(BuiltinDataAllocator* allocator)
53 : allocator_(allocator) {}
54
55 template <typename T>
56 BuiltinDataPtr<T> Allocate() {
57 return BuiltinDataPtr<T>(allocator_->AllocatePOD<T>(),
58 BuiltinDataDeleter(allocator_));
59 }
60
61 private:
62 BuiltinDataAllocator* allocator_;
63};

76// Copies the contents from the flatbuffer int vector `flatbuffer` into the
77// int array `buffer`. `flat_vector` and `buffer` represent the same
78// configuration operation for a given operation.
79TfLiteStatus FlatBufferIntVectorToArray(
80 int max_size_of_buffer, const flatbuffers::Vector<int32_t>* flat_vector,
81 int* buffer, ErrorReporter* error_reporter, const char* op_name) {
82 if (!flat_vector) {
83 TF_LITE_REPORT_ERROR(error_reporter,
84 "Input array not provided for operation '%s'.\n",
85 op_name);
86 return kTfLiteError;
87 } else {
88 size_t num_dimensions = flat_vector->size();
89 if (num_dimensions > max_size_of_buffer / sizeof(int)) {
90 TF_LITE_REPORT_ERROR(
91 error_reporter,
92 "Found too many dimensions in the input array of operation '%s'.\n",
93 op_name);
94 return kTfLiteError;
95 } else {
96 for (size_t i = 0; i < num_dimensions; ++i) {
97 buffer[i] = flat_vector->Get(i);
98 }
99 }
100 }
101 return kTfLiteOk;
102}

104// Converts the flatbuffer activation to what is used at runtime.
105TfLiteFusedActivation ConvertActivation(ActivationFunctionType activation) {
106 switch (activation) {
107 case ActivationFunctionType_NONE:
108 return kTfLiteActNone;
109 case ActivationFunctionType_RELU:
110 return kTfLiteActRelu;
111 case ActivationFunctionType_RELU_N1_TO_1:
112 return kTfLiteActReluN1To1;
113 case ActivationFunctionType_RELU6:
114 return kTfLiteActRelu6;
115 case ActivationFunctionType_TANH:
116 return kTfLiteActTanh;
117 case ActivationFunctionType_SIGN_BIT:
118 return kTfLiteActSignBit;
119 }
120 return kTfLiteActNone;
121}

2130// Micro not have the ParseOpData function at all. This would require splitting
2131// the current file into two separate files, one of which defines the
2132// ParseOpData function and the other that defines the operator specific parse
2133// functions (e.g. ParseAdd).
2134//
2135// Such a split was attempted but was not worth the effort at the time because
2136// of the following reasons:
2137// * We could either duplicate the functions and the SafeBuiltinDataAllocator
2138// class in the anonymous namespace of this file, or attempt to make a common
2139// library with these helper functions and class.
2140// * Making a common library with a separate build target was not feasible as
2141// it introduced circular dependencies due to the ErrorReporter and a common
2142// .cc and .h within the same api build target the also cause circular
2143// dependencies due to the BuiltinDataAllocator class.
2144// * If all the builtin operators were to have their own parse functions, or we
2145// were ok with some amount of code duplication, then this split of the .cc
2146// files would be a lot more feasible.
2147#ifdef TF_LITE_STATIC_MEMORY
2148 TF_LITE_REPORT_ERROR(
2149 error_reporter,
2150 "ParseOpData is unsupported on TfLiteMicro, please use the operator "
2151 "specific parse functions (e.g. ParseAdd etc.).\n");
