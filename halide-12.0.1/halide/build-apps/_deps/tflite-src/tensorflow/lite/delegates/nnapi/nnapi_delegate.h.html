39// TFliteDelegate to interface with NNAPI.
40class StatefulNnApiDelegate : public TfLiteDelegate {
41 public:
42 // Encapsulates all options that are specific to NNAPI delegate.
43 struct Options {
44 // Preferred Power/perf trade-off. For more details please see
45 // ANeuralNetworksCompilation_setPreference documentation in :
46 // https://developer.android.com/ndk/reference/group/neural-networks.html
47 enum ExecutionPreference {
48 kUndefined = -1,
49 kLowPower = 0,
50 kFastSingleAnswer = 1,
51 kSustainedSpeed = 2,
52 };
53
54 // Preferred Power/perf trade-off.
55 ExecutionPreference execution_preference = kUndefined;
56
57 // Selected NNAPI accelerator with nul-terminated name.
58 // Default to nullptr, which implies the NNAPI default behavior: NNAPI
59 // runtime is allowed to use all available accelerators. If the selected
60 // accelerator cannot be found, NNAPI will not be used.
61 // It is the caller's responsibility to ensure the string is valid for the
62 // duration of the Options object lifetime.
63 const char* accelerator_name = nullptr;
64
65 // The nul-terminated cache dir for NNAPI model.
66 // Default to nullptr, which implies the NNAPI will not try caching the
67 // compilation.
68 const char* cache_dir = nullptr;
69
70 // The unique nul-terminated token string for NNAPI model.
71 // Default to nullptr, which implies the NNAPI will not try caching the
72 // compilation. It is the caller's responsibility to ensure there is no
73 // clash of the tokens.
74 // NOTE: when using compilation caching, it is not recommended to use the
75 // same delegate instance for multiple models.
76 const char* model_token = nullptr;
77
78 // Whether to disallow NNAPI CPU usage. Only effective on Android 10 and
79 // above. The NNAPI CPU typically performs less well than built-in TfLite
80 // kernels, but allowing CPU allows partial acceleration of models. If this
81 // is set to true, NNAPI is only used if the whole model is accelerated.
82 bool disallow_nnapi_cpu = true;
83
84 // Specifies the max number of partitions to delegate. A value <= 0 means
85 // no limit.
86 // If the delegation of the full set of supported nodes would generate a
87 // number of partition greater than this parameter, only
88 // <max_number_delegated_partitions> of them will be actually accelerated.
89 // The selection is currently done sorting partitions in decreasing order
90 // of number of nodes and selecting them until the limit is reached.
91 int max_number_delegated_partitions = 3;
92
93 // allow fp32 compuation to be run in fp16.
94 bool allow_fp16 = false;
95
96 // Specifies the relative priority for executions of the model.
97 // Available values are {ANEURALNETWORKS_PRIORITY_LOW,
98 // ANEURALNETWORKS_PRIORITY_MEDIUM, ANEURALNETWORKS_PRIORITY_HIGH,
99 // ANEURALNETWORKS_PRIORITY_DEFAULT}.
100 int execution_priority = ANEURALNETWORKS_PRIORITY_DEFAULT;
101
102 // Specifies the maximum expected duration in nanosecond for compiling the
103 // model. If the device is not able to complete the compilation within the
104 // specified duration, the compilation may be aborted. If set to 0, the
105 // timeout duration is considered infinite.
106 uint64_t max_compilation_timeout_duration_ns = 0;
107
108 // Specifies the maximum expected duration in nanosecond for executing the
109 // model. If the device is not able to complete the execution within the
110 // specified duration, the execution may be aborted. If set to 0, the
111 // timeout duration is considered infinite.
112 uint64_t max_execution_timeout_duration_ns = 0;
113
114 // Specifies the maximum expected duration in nanosecond for WHILE loops in
115 // the execution. If a WHILE loop condition model does not output false
116 // within the specified duration, the execution will be aborted. If set to
117 // 0, the default timeout for loops will be used.
118 uint64_t max_execution_loop_timeout_duration_ns = 0;
119
120 // Whether to allow dynamic dimension sizes without re-compilation.
121 // A tensor of with dynamic dimension must have a valid dim_signature
122 // defined.
123 // Only supported in NNAPI 1.1 and newer versions.
124 // WARNING: Setting this flag to true may result in model being rejected by
125 // accelerator. This should only be enabled if the target device supports
126 // dynamic dimensions of the model.
127 bool allow_dynamic_dimensions = false;
128
129 // Use NNAPI Burst mode if supported.
130 // Burst mode allows accelerators to efficiently manage resources, which
131 // would significantly reduce overhead especially if the same delegate
132 // instance is to be used for multiple inferences.
133 // Default: Disabled.
134 bool use_burst_computation = false;
135 };
136
137 // Uses default options.
138 StatefulNnApiDelegate();
139
140 // The ownership of the NnApi instance is left to the caller of the
141 // StatefulNnApiDelegate constructor; the caller must ensure that the lifetime
142 // of the NnApi instance exceeds the lifetime of the StatefulNnApiDelegate.
143 explicit StatefulNnApiDelegate(const NnApi* nnapi);
144
145 // The constructor that accepts options from user.
146 // This makes a copy of any data that it needs from Options, so
147 // the caller can safely deallocate any storage pointed to by
148 // the 'const char *' members of Options immediately after calling this.
149 explicit StatefulNnApiDelegate(Options options);
150
151 // Constructor that accepts both an NnApi instance and options.
152 // The ownership of the NnApi instance is left to the caller of the
153 // StatefulNnApiDelegate constructor; the caller must ensure that the lifetime
154 // of the NnApi instance exceeds the lifetime of the StatefulNnApiDelegate.
155 // This constructor makes a copy of any data that it needs from Options, so
156 // the caller can safely deallocate any storage pointed to by
157 // the 'const char *' members of Options immediately after calling this.
158 StatefulNnApiDelegate(const NnApi* nnapi, Options options);
159
160 ~StatefulNnApiDelegate() = default;
161
162 // Returns the delegate options.
163 // The lifetime of the storage pointed to by the 'const char *' members of the
164 // returned Options object is the same as the lifetime of the supplied
165 // TfLiteDelegate instance.
166 static const Options GetOptions(TfLiteDelegate* delegate);
167
168 // Callback function which copies data from ANeuralNetworksMemory to host
169 // tensor CPU buffer. It is the users responsibility to implement these
170 // callbacks for the specific types of shared memory they intend to use.
171 // WARNING: This is an experimental interface that is subject to change.
172 typedef TfLiteStatus (*CopyToHostTensorFnPtr)(TfLiteTensor* tensor,
173 ANeuralNetworksMemory* memory,
174 size_t memory_offset,
175 size_t byte_size,
176 void* callback_context);
177
178 // Encapsulates all fields related to memory registration for internal
179 // bookkeeping only.
180 struct MemoryRegistration {
181 ANeuralNetworksMemory* memory;
182 CopyToHostTensorFnPtr callback;
183 void* callback_context;
184 };
185
186 // Register the ANeuralNetworksMemory handle with the delegate. A
187 // TfLiteBufferHandle will be returned to be used with
188 // Interpreter::SetBufferHandle. The callback_context will be passed to the
189 // callback function when invoked.
190 // Note: the returned TfLiteBufferHandle can only be used with a single
191 // Interpreter instance. However, the caller can register the same memory
192 // multiple times to get different handles to use with difference Interpreter
193 // instances
194 // WARNING: This is an experimental interface that is subject to change.
195 TfLiteBufferHandle RegisterNnapiMemory(ANeuralNetworksMemory* memory,
196 CopyToHostTensorFnPtr callback,
197 void* callback_context);
198
199 // Returns the vector of known ANeuralNetworksMemory handles.
200 // Note: this function is not intended to be called by developers.
201 // WARNING: This is an experimental interface that is subject to change.
202 static const std::vector<MemoryRegistration>& GetTensorMemoryMap(
203 TfLiteDelegate* delegate);
204
205 // Returns the int value of the ResultCode returned by the latest
206 // failed call to NNAPI, if any. Zero only in case of NO failed calls since
207 // the construction of this instance of StatefulNnApiDelegate.
208 // The error code is reset when the delegate is re-initialized
209 // (i.e. when calling interpreter.ModifyGraphWithDelegate(delegate)).
210 int GetNnApiErrno() const;
211
212 private:
213 // Encapsulates all delegate data.
214 struct Data {
215 // Pointer to NNAPI implementation to be used by this delegate as
216 // set when building the StatefulNnApiDelegate instance.
217 // Will generally be the NnApiInstance() singleton but can be overridden
218 // for testing or for users needing to wrap or stub parts of NNAPI.
219 // The ownership of the nnapi instance is left to the caller of
220 // the StatefulNnApiDelegate constructor.
221 const NnApi* nnapi;
222 // Preferred Power/perf trade-off.
223 Options::ExecutionPreference execution_preference;
224 // Selected NNAPI accelerator name.
225 std::string accelerator_name;
226 // The cache dir for NNAPI model.
227 std::string cache_dir;
228 // The unique token string for NNAPI model.
229 std::string model_token;
230 // Whether to disallow NNAPI CPU.
231 bool disallow_nnapi_cpu;
232 // Tensor to ANeuralNetworksMemory mapping.
233 std::vector<MemoryRegistration> tensor_memory_map;
234 // Contains a non zero value if any NNAPI method call
235 // operation returned a non zero result code.
236 int nnapi_errno = ANEURALNETWORKS_NO_ERROR;
237 // Cache of kernels already built in StatefulNnApiDelegate::DoPrepare
238 // when trying to understand if all nodes are supported by the target
239 // accelerators.
240 // The key is the index of the first node in the partition.
241 // Couldn't use unique_ptr because of problems building on gcc
242 std::unordered_map<int, NNAPIDelegateKernel*> delegate_state_cache;
243 // Maximum number of NNAPI partition to delegate. Zero or negative means
244 // no limit. Copied from StatefulNnApiDelegate::Options
245 int max_number_delegated_partitions;
246 // allow fp32 computation to be run in fp16.
247 bool allow_fp16;
248 // Specifies the relative priority for executions of the model.
249 int execution_priority = ANEURALNETWORKS_PRIORITY_DEFAULT;
250 // Specifies the maximum expected duration in nanosecond for compiling the
251 // model.
252 uint64_t max_compilation_timeout_duration_ns = 0;
253 // Specifies the maximum expected duration in nanosecond for executing the
254 // model.
255 uint64_t max_execution_timeout_duration_ns = 0;
256 // Specifies the maximum expected duration in nanosecond for WHILE loops in
257 // the execution
258 uint64_t max_execution_loop_timeout_duration_ns = 0;
259 // Whether to allow dynamic dimension sizes without re-compilation.
260 bool allow_dynamic_dimensions = false;
261 // Whether to use NNAPI Burst mode.
262 bool use_burst_computation = false;
263
264 explicit Data(const NnApi* nnapi);
265 ~Data();
266
267 // Caches an initialised NNAPIDelegateKernel.
268 void CacheDelegateKernel(const TfLiteDelegateParams* delegate_params,
269 NNAPIDelegateKernel* delegate_state);
270 // Returns a cached NNAPIDelegateKernel if available and removes it
271 // from the cache transferring the ownership to the caller.
272 NNAPIDelegateKernel* MaybeGetCachedDelegateKernel(
273 const TfLiteDelegateParams* delegate_params);
274 };
275
276 // Implements TfLiteDelegate::Prepare. Please refer to TFLiteDelegate
277 // documentation for more info.
278 static TfLiteStatus DoPrepare(TfLiteContext* context,
279 TfLiteDelegate* delegate);
280
281 // Copy the data from delegate buffer handle into raw memory of the given
282 // 'tensor'. The delegate is allowed to allocate the raw
283 // bytes as long as it follows the rules for kTfLiteDynamic tensors.
284 static TfLiteStatus DoCopyFromBufferHandle(TfLiteContext* context,
285 TfLiteDelegate* delegate,
286 TfLiteBufferHandle buffer_handle,
287 TfLiteTensor* tensor);
288
289 // Copy the data from raw memory of the given 'tensor' to delegate buffer
290 // handle. Currently this function is not supported, and calling the function
291 // will result in an error.
292 static TfLiteStatus DoCopyToBufferHandle(TfLiteContext* context,
293 TfLiteDelegate* delegate,
294 TfLiteBufferHandle buffer_handle,
295 TfLiteTensor* tensor);
296
297 // Free the Delegate Buffer Handle. Note: This only frees the handle, but
298 // this doesn't release the underlying resource (e.g. textures). The
299 // resources are either owned by application layer or the delegate.
300 static void DoFreeBufferHandle(TfLiteContext* context,
301 TfLiteDelegate* delegate,
302 TfLiteBufferHandle* handle);
303
304 // Returns the nodes that can be delegated via NNAPI to the accelerator
305 // specified in the delegate options and information about the way the
306 // graph will be partitioned if the supported nodes will be delegated.
307 // Partition information is composed by the number of partitions and
308 // the delegate parameters associated to each partition.
309 // The method also caches in delegate->data the NNApiDelegateKernel instances
310 // that have been created during the device evaluation.
311 // All arguments are expected to be non-null.
312 static TfLiteStatus GetNodesSupportedByAccelerator(
313 TfLiteContext* context, TfLiteDelegate* delegate, const NnApi* nnapi,
314 const std::vector<int>& supported_nodes,
315 std::vector<int>* device_supported_nodes, int* num_partitions,
316 TfLiteDelegateParams** params_array, int* nnapi_errno);
317
318 // Alters the given array of nodes_to_delegate to limit the number of NNAPI
319 // owned partition to be less or equal than num_partitions. If num_partitions
320 // is less or equal to zero the input is left unaltered.
321 // The nodes_to_delegate array is expected to contain at element 0 the number
322 // of nodes to delegate and in remaining elements the set of nodes
323 // that would be delegated to NNAPI if this function wouldn't be
324 // called. It will be altered storing in the first element the count of
325 // nodes to actually delegate and in the remainder of the array the indexes.
326 // The params_array params might be altered during the functions execution.
327 static TfLiteStatus LimitDelegatedPartitions(
328 int max_partitions,
329 std::vector<TfLiteDelegateParams> partition_params_array,
330 std::vector<int>* nodes_to_delegate);
331
332 // Delegate data presented through TfLiteDelegate::data_.
333 Data delegate_data_;
334};
335
336// DEPRECATED: Please use StatefulNnApiDelegate class instead.
337//
338// Returns a singleton delegate that can be used to use the NN API.
339// e.g.
340// NnApiDelegate* delegate = NnApiDelegate();
341// interpreter->ModifyGraphWithDelegate(&delegate);
342// NnApiDelegate() returns a singleton, so you should not free this
343// pointer or worry about its lifetime.
344TfLiteDelegate* NnApiDelegate();
