40/// An RAII object that represents a read-only tflite model, copied from disk,
41/// or mmapped. This uses flatbuffers as the serialization format.
42///
43/// NOTE: The current API requires that a FlatBufferModel instance be kept alive
44/// by the client as long as it is in use by any dependent Interpreter
45/// instances.
46/// <pre><code>
47/// using namespace tflite;
48/// StderrReporter error_reporter;
49/// auto model = FlatBufferModel::BuildFromFile("interesting_model.tflite",
50/// &error_reporter);
51/// MyOpResolver resolver; // You need to subclass OpResolver to provide
52/// // implementations.
53/// InterpreterBuilder builder(*model, resolver);
54/// std::unique_ptr<Interpreter> interpreter;
55/// if(builder(&interpreter) == kTfLiteOk) {
56/// .. run model inference with interpreter
57/// }
58/// </code></pre>
59///
60/// OpResolver must be defined to provide your kernel implementations to the
61/// interpreter. This is environment specific and may consist of just the
62/// builtin ops, or some custom operators you defined to extend tflite.
63class FlatBufferModel {
64 public:
65 /// Builds a model based on a file.
66 /// Caller retains ownership of `error_reporter` and must ensure its lifetime
67 /// is longer than the FlatBufferModel instance.
68 /// Returns a nullptr in case of failure.
69 static std::unique_ptr<FlatBufferModel> BuildFromFile(
70 const char* filename,
71 ErrorReporter* error_reporter = DefaultErrorReporter());
72
73 /// Verifies whether the content of the file is legit, then builds a model
74 /// based on the file.
75 /// The extra_verifier argument is an additional optional verifier for the
76 /// file contents. By default, we always check with tflite::VerifyModelBuffer.
77 /// If extra_verifier is supplied, the file contents is also checked against
78 /// the extra_verifier after the check against tflite::VerifyModelBuilder.
79 /// Caller retains ownership of `error_reporter` and must ensure its lifetime
80 /// is longer than the FlatBufferModel instance.
81 /// Returns a nullptr in case of failure.
82 static std::unique_ptr<FlatBufferModel> VerifyAndBuildFromFile(
83 const char* filename, TfLiteVerifier* extra_verifier = nullptr,
84 ErrorReporter* error_reporter = DefaultErrorReporter());
85
86 /// Builds a model based on a pre-loaded flatbuffer.
87 /// Caller retains ownership of the buffer and should keep it alive until
88 /// the returned object is destroyed. Caller also retains ownership of
89 /// `error_reporter` and must ensure its lifetime is longer than the
90 /// FlatBufferModel instance.
91 /// Returns a nullptr in case of failure.
92 /// NOTE: this does NOT validate the buffer so it should NOT be called on
93 /// invalid/untrusted input. Use VerifyAndBuildFromBuffer in that case
94 static std::unique_ptr<FlatBufferModel> BuildFromBuffer(
95 const char* caller_owned_buffer, size_t buffer_size,
96 ErrorReporter* error_reporter = DefaultErrorReporter());
97
98 /// Verifies whether the content of the buffer is legit, then builds a model
99 /// based on the pre-loaded flatbuffer.
100 /// The extra_verifier argument is an additional optional verifier for the
101 /// buffer. By default, we always check with tflite::VerifyModelBuffer. If
102 /// extra_verifier is supplied, the buffer is checked against the
103 /// extra_verifier after the check against tflite::VerifyModelBuilder. The
104 /// caller retains ownership of the buffer and should keep it alive until the
105 /// returned object is destroyed. Caller retains ownership of `error_reporter`
106 /// and must ensure its lifetime is longer than the FlatBufferModel instance.
107 /// Returns a nullptr in case of failure.
108 static std::unique_ptr<FlatBufferModel> VerifyAndBuildFromBuffer(
109 const char* caller_owned_buffer, size_t buffer_size,
110 TfLiteVerifier* extra_verifier = nullptr,
111 ErrorReporter* error_reporter = DefaultErrorReporter());
112
113 /// Builds a model directly from an allocation.
114 /// Ownership of the allocation is passed to the model, but the caller
115 /// retains ownership of `error_reporter` and must ensure its lifetime is
116 /// longer than the FlatBufferModel instance.
117 /// Returns a nullptr in case of failure (e.g., the allocation is invalid).
118 static std::unique_ptr<FlatBufferModel> BuildFromAllocation(
119 std::unique_ptr<Allocation> allocation,
120 ErrorReporter* error_reporter = DefaultErrorReporter());
121
122 /// Verifies whether the content of the allocation is legit, then builds a
123 /// model based on the provided allocation.
124 /// The extra_verifier argument is an additional optional verifier for the
125 /// buffer. By default, we always check with tflite::VerifyModelBuffer. If
126 /// extra_verifier is supplied, the buffer is checked against the
127 /// extra_verifier after the check against tflite::VerifyModelBuilder.
128 /// Ownership of the allocation is passed to the model, but the caller
129 /// retains ownership of `error_reporter` and must ensure its lifetime is
130 /// longer than the FlatBufferModel instance.
131 /// Returns a nullptr in case of failure.
132 static std::unique_ptr<FlatBufferModel> VerifyAndBuildFromAllocation(
133 std::unique_ptr<Allocation> allocation,
134 TfLiteVerifier* extra_verifier = nullptr,
135 ErrorReporter* error_reporter = DefaultErrorReporter());
136
137 /// Builds a model directly from a flatbuffer pointer
138 /// Caller retains ownership of the buffer and should keep it alive until the
139 /// returned object is destroyed. Caller retains ownership of `error_reporter`
140 /// and must ensure its lifetime is longer than the FlatBufferModel instance.
141 /// Returns a nullptr in case of failure.
142 static std::unique_ptr<FlatBufferModel> BuildFromModel(
143 const tflite::Model* caller_owned_model_spec,
144 ErrorReporter* error_reporter = DefaultErrorReporter());
145
146 // Releases memory or unmaps mmaped memory.
147 ~FlatBufferModel();
148
149 // Copying or assignment is disallowed to simplify ownership semantics.
150 FlatBufferModel(const FlatBufferModel&) = delete;
151 FlatBufferModel& operator=(const FlatBufferModel&) = delete;
152
153 bool initialized() const { return model_ != nullptr; }
154 const tflite::Model* operator->() const { return model_; }
155 const tflite::Model* GetModel() const { return model_; }
156 ErrorReporter* error_reporter() const { return error_reporter_; }
157 const Allocation* allocation() const { return allocation_.get(); }
158
159 // Returns the minimum runtime version from the flatbuffer. This runtime
160 // version encodes the minimum required interpreter version to run the
161 // flatbuffer model. If the minimum version can't be determined, an empty
162 // string will be returned.
163 // Note that the returned minimum version is a lower-bound but not a strict
164 // lower-bound; ops in the graph may not have an associated runtime version,
165 // in which case the actual required runtime might be greater than the
166 // reported minimum.
167 std::string GetMinimumRuntime() const;
168
169 /// Returns true if the model identifier is correct (otherwise false and
170 /// reports an error).
171 bool CheckModelIdentifier() const;
172
173 private:
174 /// Loads a model from a given allocation. FlatBufferModel will take over the
175 /// ownership of `allocation`, and delete it in destructor. The ownership of
176 /// `error_reporter`remains with the caller and must have lifetime at least
177 /// as much as FlatBufferModel. This is to allow multiple models to use the
178 /// same ErrorReporter instance.
179 FlatBufferModel(std::unique_ptr<Allocation> allocation,
180 ErrorReporter* error_reporter = DefaultErrorReporter());
181
182 /// Loads a model from Model flatbuffer. The `model` has to remain alive and
183 /// unchanged until the end of this flatbuffermodel's lifetime.
184 FlatBufferModel(const Model* model, ErrorReporter* error_reporter);
185
186 /// Flatbuffer traverser pointer. (Model* is a pointer that is within the
187 /// allocated memory of the data allocated by allocation's internals.
188 const tflite::Model* model_ = nullptr;
189 /// The error reporter to use for model errors and subsequent errors when
190 /// the interpreter is created
191 ErrorReporter* error_reporter_;
192 /// The allocator used for holding memory of the model. Note that this will
193 /// be null if the client provides a tflite::Model directly.
194 std::unique_ptr<Allocation> allocation_;
195};
196
197} // namespace tflite
198
199#endif // TENSORFLOW_LITE_MODEL_BUILDER_H_
