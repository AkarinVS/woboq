15#ifndef ABSL_BASE_INTERNAL_ERRNO_SAVER_H_
16#define ABSL_BASE_INTERNAL_ERRNO_SAVER_H_
17
18#include <cerrno>
19
20#include "absl/base/config.h"
21
22namespace absl {
23ABSL_NAMESPACE_BEGIN
24namespace base_internal {
25
26// `ErrnoSaver` captures the value of `errno` upon construction and restores it
27// upon deletion. It is used in low-level code and must be super fast. Do not
28// add instrumentation, even in debug modes.
29class ErrnoSaver {
30 public:
31 ErrnoSaver() : saved_errno_(errno) {}
32 ~ErrnoSaver() { errno = saved_errno_; }
33 int operator()() const { return saved_errno_; }
34
35 private:
36 const int saved_errno_;
37};
38
39} // namespace base_internal
40ABSL_NAMESPACE_END
41} // namespace absl
42
43#endif // ABSL_BASE_INTERNAL_ERRNO_SAVER_H_
