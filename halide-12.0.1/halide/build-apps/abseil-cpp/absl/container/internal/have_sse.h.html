15// Shared config probing for SSE instructions used in Swiss tables.
16#ifndef ABSL_CONTAINER_INTERNAL_HAVE_SSE_H_
17#define ABSL_CONTAINER_INTERNAL_HAVE_SSE_H_
18
19#ifndef SWISSTABLE_HAVE_SSE2
20#if defined(__SSE2__) || \
21 (defined(_MSC_VER) && \
22 (defined(_M_X64) || (defined(_M_IX86) && _M_IX86_FP >= 2)))
23#define SWISSTABLE_HAVE_SSE2 1
24#else
25#define SWISSTABLE_HAVE_SSE2 0
26#endif
27#endif
28
29#ifndef SWISSTABLE_HAVE_SSSE3
30#ifdef __SSSE3__
31#define SWISSTABLE_HAVE_SSSE3 1
32#else
33#define SWISSTABLE_HAVE_SSSE3 0
34#endif
35#endif
36
37#if SWISSTABLE_HAVE_SSSE3 && !SWISSTABLE_HAVE_SSE2
38#error "Bad configuration!"
39#endif
40
41#if SWISSTABLE_HAVE_SSE2
42#include <emmintrin.h>
43#endif
44
45#if SWISSTABLE_HAVE_SSSE3
46#include <tmmintrin.h>
47#endif
48
49#endif // ABSL_CONTAINER_INTERNAL_HAVE_SSE_H_
