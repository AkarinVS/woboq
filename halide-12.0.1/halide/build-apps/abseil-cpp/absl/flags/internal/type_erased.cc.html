33bool GetCommandLineOption(absl::string_view name, std::string* value) {
34 if (name.empty()) return false;
35 assert(value);
36
37 CommandLineFlag* flag = flags_internal::FindCommandLineFlag(name);
38 if (flag == nullptr || flag->IsRetired()) {
39 return false;
40 }
41
42 *value = flag->CurrentValue();
43 return true;
44}
45
46bool SetCommandLineOption(absl::string_view name, absl::string_view value) {
47 return SetCommandLineOptionWithMode(name, value,
48 flags_internal::SET_FLAGS_VALUE);
49}
50
51bool SetCommandLineOptionWithMode(absl::string_view name,
52 absl::string_view value,
53 FlagSettingMode set_mode) {
54 CommandLineFlag* flag = flags_internal::FindCommandLineFlag(name);
55
56 if (!flag || flag->IsRetired()) return false;
57
58 std::string error;
59 if (!flag->SetFromString(value, set_mode, kProgrammaticChange, &error)) {
60 // Errors here are all of the form: the provided name was a recognized
61 // flag, but the value was invalid (bad type, or validation failed).
62 flags_internal::ReportUsageError(error, false);
63 return false;
64 }
65
66 return true;
67}

71bool IsValidFlagValue(absl::string_view name, absl::string_view value) {
72 CommandLineFlag* flag = flags_internal::FindCommandLineFlag(name);
73
74 return flag != nullptr &&
75 (flag->IsRetired() || flag->ValidateInputValue(value));
76}

79
80bool SpecifiedOnCommandLine(absl::string_view name) {
81 CommandLineFlag* flag = flags_internal::FindCommandLineFlag(name);
82 if (flag != nullptr && !flag->IsRetired()) {
83 return flag->IsSpecifiedOnCommandLine();
84 }
85 return false;
86}
