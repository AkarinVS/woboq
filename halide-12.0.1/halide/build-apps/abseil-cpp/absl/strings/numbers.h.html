57// SimpleAtoi()
58//
59// Converts the given string (optionally followed or preceded by ASCII
60// whitespace) into an integer value, returning `true` if successful. The string
61// must reflect a base-10 integer whose value falls within the range of the
62// integer type (optionally preceded by a `+` or `-`). If any errors are
63// encountered, this function returns `false`, leaving `out` in an unspecified
64// state.
65template <typename int_type>
66ABSL_MUST_USE_RESULT bool SimpleAtoi(absl::string_view str, int_type* out);
67
68// SimpleAtof()
69//
70// Converts the given string (optionally followed or preceded by ASCII
71// whitespace) into a float, which may be rounded on overflow or underflow,
72// returning `true` if successful.
73// See https://en.cppreference.com/w/c/string/byte/strtof for details about the
74// allowed formats for `str`, except SimpleAtof() is locale-independent and will
75// always use the "C" locale. If any errors are encountered, this function
76// returns `false`, leaving `out` in an unspecified state.
77ABSL_MUST_USE_RESULT bool SimpleAtof(absl::string_view str, float* out);
78
79// SimpleAtod()
80//
81// Converts the given string (optionally followed or preceded by ASCII
82// whitespace) into a double, which may be rounded on overflow or underflow,
83// returning `true` if successful.
84// See https://en.cppreference.com/w/c/string/byte/strtof for details about the
85// allowed formats for `str`, except SimpleAtod is locale-independent and will
86// always use the "C" locale. If any errors are encountered, this function
87// returns `false`, leaving `out` in an unspecified state.
88ABSL_MUST_USE_RESULT bool SimpleAtod(absl::string_view str, double* out);
89
90// SimpleAtob()
91//
92// Converts the given string into a boolean, returning `true` if successful.
93// The following case-insensitive strings are interpreted as boolean `true`:
94// "true", "t", "yes", "y", "1". The following case-insensitive strings
95// are interpreted as boolean `false`: "false", "f", "no", "n", "0". If any
96// errors are encountered, this function returns `false`, leaving `out` in an
97// unspecified state.
98ABSL_MUST_USE_RESULT bool SimpleAtob(absl::string_view str, bool* out);

109// Digit conversion.
110ABSL_DLL extern const char kHexChar[17]; // 0123456789abcdef
111ABSL_DLL extern const char
112 kHexTable[513]; // 000102030405060708090a0b0c0d0e0f1011...
113ABSL_DLL extern const char
114 two_ASCII_digits[100][2]; // 00, 01, 02, 03...
115
116// Writes a two-character representation of 'i' to 'buf'. 'i' must be in the
117// range 0 <= i < 100, and buf must have space for two characters. Example:
118// char buf[2];
119// PutTwoDigits(42, buf);
120// // buf[0] == '4'
121// // buf[1] == '2'
122inline void PutTwoDigits(size_t i, char* buf) {
123 assert(i < 100);
124 memcpy(buf, two_ASCII_digits[i], 2);
125}

138// Helper function for fast formatting of floating-point values.
139// The result is the same as printf's "%g", a.k.a. "%.6g"; that is, six
140// significant digits are returned, trailing zeros are removed, and numbers
141// outside the range 0.0001-999999 are output using scientific notation
142// (1.23456e+06). This routine is heavily optimized.
143// Required buffer size is `kSixDigitsToBufferSize`.
144size_t SixDigitsToBuffer(double d, char* buffer);
145
146// These functions are intended for speed. All functions take an output buffer
147// as an argument and return a pointer to the last byte they wrote, which is the
148// terminating '\0'. At most `kFastToBufferSize` bytes are written.
149char* FastIntToBuffer(int32_t, char*);
150char* FastIntToBuffer(uint32_t, char*);
151char* FastIntToBuffer(int64_t, char*);
152char* FastIntToBuffer(uint64_t, char*);
153
154// For enums and integer types that are not an exact match for the types above,
155// use templates to call the appropriate one of the four overloads above.
156template <typename int_type>
157char* FastIntToBuffer(int_type i, char* buffer) {
158 static_assert(sizeof(i) <= 64 / 8,
159 "FastIntToBuffer works only with 64-bit-or-less integers.");
160 // TODO(jorg): This signed-ness check is used because it works correctly
161 // with enums, and it also serves to check that int_type is not a pointer.
162 // If one day something like std::is_signed<enum E> works, switch to it.
163 if (static_cast<int_type>(1) - 2 < 0) { // Signed
164 if (sizeof(i) > 32 / 8) { // 33-bit to 64-bit
165 return FastIntToBuffer(static_cast<int64_t>(i), buffer);
166 } else { // 32-bit or less
167 return FastIntToBuffer(static_cast<int32_t>(i), buffer);
168 }
169 } else { // Unsigned
170 if (sizeof(i) > 32 / 8) { // 33-bit to 64-bit
171 return FastIntToBuffer(static_cast<uint64_t>(i), buffer);
172 } else { // 32-bit or less
173 return FastIntToBuffer(static_cast<uint32_t>(i), buffer);
174 }
175 }
176}

178// Implementation of SimpleAtoi, generalized to support arbitrary base (used
179// with base different from 10 elsewhere in Abseil implementation).
180template <typename int_type>
181ABSL_MUST_USE_RESULT bool safe_strtoi_base(absl::string_view s, int_type* out,
182 int base) {
183 static_assert(sizeof(*out) == 4 || sizeof(*out) == 8,
184 "SimpleAtoi works only with 32-bit or 64-bit integers.");
185 static_assert(!std::is_floating_point<int_type>::value,
186 "Use SimpleAtof or SimpleAtod instead.");
187 bool parsed;
188 // TODO(jorg): This signed-ness check is used because it works correctly
189 // with enums, and it also serves to check that int_type is not a pointer.
190 // If one day something like std::is_signed<enum E> works, switch to it.
191 if (static_cast<int_type>(1) - 2 < 0) { // Signed
192 if (sizeof(*out) == 64 / 8) { // 64-bit
193 int64_t val;
194 parsed = numbers_internal::safe_strto64_base(s, &val, base);
195 *out = static_cast<int_type>(val);
196 } else { // 32-bit
197 int32_t val;
198 parsed = numbers_internal::safe_strto32_base(s, &val, base);
199 *out = static_cast<int_type>(val);
200 }
201 } else { // Unsigned
202 if (sizeof(*out) == 64 / 8) { // 64-bit
203 uint64_t val;
204 parsed = numbers_internal::safe_strtou64_base(s, &val, base);
205 *out = static_cast<int_type>(val);
206 } else { // 32-bit
207 uint32_t val;
208 parsed = numbers_internal::safe_strtou32_base(s, &val, base);
209 *out = static_cast<int_type>(val);
210 }
211 }
212 return parsed;
213}

215// FastHexToBufferZeroPad16()
216//
217// Outputs `val` into `out` as if by `snprintf(out, 17, "%016x", val)` but
218// without the terminating null character. Thus `out` must be of length >= 16.
219// Returns the number of non-pad digits of the output (it can never be zero
220// since 0 has one digit).
221inline size_t FastHexToBufferZeroPad16(uint64_t val, char* out) {
222#ifdef __SSE4_2__
223 uint64_t be = absl::big_endian::FromHost64(val);
224 const auto kNibbleMask = _mm_set1_epi8(0xf);
225 const auto kHexDigits = _mm_setr_epi8('0', '1', '2', '3', '4', '5', '6', '7',
226 '8', '9', 'a', 'b', 'c', 'd', 'e', 'f');
227 auto v = _mm_loadl_epi64(reinterpret_cast<__m128i*>(&be)); // load lo dword
228 auto v4 = _mm_srli_epi64(v, 4); // shift 4 right
229 auto il = _mm_unpacklo_epi8(v4, v); // interleave bytes
230 auto m = _mm_and_si128(il, kNibbleMask); // mask out nibbles
231 auto hexchars = _mm_shuffle_epi8(kHexDigits, m); // hex chars
232 _mm_storeu_si128(reinterpret_cast<__m128i*>(out), hexchars);
233#else
234 for (int i = 0; i < 8; ++i) {
235 auto byte = (val >> (56 - 8 * i)) & 0xFF;
236 auto* hex = &absl::numbers_internal::kHexTable[byte * 2];
237 std::memcpy(out + 2 * i, hex, 2);
238 }
239#endif
240 // | 0x1 so that even 0 has 1 digit.
241 return 16 - absl::base_internal::CountLeadingZeros64(val | 0x1) / 4;
242}

246// SimpleAtoi()
247//
248// Converts a string to an integer, using `safe_strto?()` functions for actual
249// parsing, returning `true` if successful. The `safe_strto?()` functions apply
250// strict checking; the string must be a base-10 integer, optionally followed or
251// preceded by ASCII whitespace, with a value in the range of the corresponding
252// integer type.
253template <typename int_type>
254ABSL_MUST_USE_RESULT bool SimpleAtoi(absl::string_view str, int_type* out) {
255 return numbers_internal::safe_strtoi_base(str, out, 10);
256}
257
258ABSL_MUST_USE_RESULT inline bool SimpleAtoi(absl::string_view str,
259 absl::uint128* out) {
260 return numbers_internal::safe_strtou128_base(str, out, 10);
261}
