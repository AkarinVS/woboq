25// Buffer<> uses "shape" to mean "array of halide_dimension_t", but doesn't
26// provide a typedef for it (and doesn't use a vector for it in any event).
27using Shape = std::vector<halide_dimension_t>;

29// A ShapePromise is a function that returns a Shape. If the Promise can't
30// return a valid Shape, it may fail. This allows us to defer error reporting
31// for situations until the Shape is actually needed; in particular, it allows
32// us to attempt doing bounds-query for the shape of input buffers early,
33// but to ignore the error unless we actually need it... which we won't if an
34// estimate is provided for the input in question.
35using ShapePromise = std::function<Shape()>;

37// Standard stream output for halide_type_t
38inline std::ostream &operator<<(std::ostream &stream, const halide_type_t &type) {
39 if (type.code == halide_type_uint && type.bits == 1) {
40 stream << "bool";
41 } else {
42 assert(type.code >= 0 && type.code <= 3);
43 static const char *const names[4] = {"int", "uint", "float", "handle"};
44 stream << names[type.code] << (int)type.bits;
45 }
46 if (type.lanes > 1) {
47 stream << "x" << (int)type.lanes;
48 }
49 return stream;
50}

73// Bottleneck all our logging so that client code can override any/all of them.
74struct Logger {
75 using LogFn = std::function<void(const std::string &)>;
76
77 const LogFn out, info, warn, fail;
78
79 Logger()
80 : out(log_out), info(log_cerr), warn(log_cerr), fail(log_fail) {
81 }
82 Logger(LogFn o, LogFn i, LogFn w, LogFn f)
83 : out(std::move(o)), info(std::move(i)), warn(std::move(w)), fail(std::move(f)) {
84 }
85
86private:
87 static void log_out(const std::string &s) {
88 std::cout << s;
89 }
90
91 static void log_cerr(const std::string &s) {
92 std::cerr << s;
93 }
94
95 static void log_fail(const std::string &s) {
96 log_cerr(s);
97 abort();
98 }
99};

167inline std::vector<std::string> split_string(const std::string &source,
168 const std::string &delim) {
169 std::vector<std::string> elements;
170 size_t start = 0;
171 size_t found = 0;
172 while ((found = source.find(delim, start)) != std::string::npos) {
173 elements.push_back(source.substr(start, found - start));
174 start = found + delim.size();
175 }
176
177 // If start is exactly source.size(), the last thing in source is a
178 // delimiter, in which case we want to add an empty std::string to elements.
179 if (start <= source.size()) {
180 elements.push_back(source.substr(start, std::string::npos));
181 }
182 return elements;
183}

190// dynamic_type_dispatch is a utility for functors that want to be able
191// to dynamically dispatch a halide_type_t to type-specialized code.
192// To use it, a functor must be a *templated* class, e.g.
193//
194// template<typename T> class MyFunctor { int operator()(arg1, arg2...); };
195//
196// dynamic_type_dispatch() is called with a halide_type_t as the first argument,
197// followed by the arguments to the Functor's operator():
198//
199// auto result = dynamic_type_dispatch<MyFunctor>(some_halide_type, arg1, arg2);
200//
201// Note that this means that the functor must be able to instantiate its
202// operator() for all the Halide scalar types; it also means that all those
203// variants *will* be instantiated (increasing code size), so this approach
204// should only be used when strictly necessary.
205template<template<typename> class Functor, typename... Args>
206auto dynamic_type_dispatch(const halide_type_t &type, Args &&...args) -> decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...)) {
207
208#define HANDLE_CASE(CODE, BITS, TYPE) \
209 case halide_type_code(CODE, BITS): \
210 return Functor<TYPE>()(std::forward<Args>(args)...);
211 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
212 HANDLE_CASE(halide_type_float, 32, float)
213 HANDLE_CASE(halide_type_float, 64, double)
214 HANDLE_CASE(halide_type_int, 8, int8_t)
215 HANDLE_CASE(halide_type_int, 16, int16_t)
216 HANDLE_CASE(halide_type_int, 32, int32_t)
217 HANDLE_CASE(halide_type_int, 64, int64_t)
218 HANDLE_CASE(halide_type_uint, 1, bool)
219 HANDLE_CASE(halide_type_uint, 8, uint8_t)
220 HANDLE_CASE(halide_type_uint, 16, uint16_t)
221 HANDLE_CASE(halide_type_uint, 32, uint32_t)
222 HANDLE_CASE(halide_type_uint, 64, uint64_t)
223 HANDLE_CASE(halide_type_handle, 64, void *)
224 default:
225 fail() << "Unsupported type: " << type << "\n";
226 using ReturnType = decltype(std::declval<Functor<uint8_t>>()(std::forward<Args>(args)...));
227 return ReturnType();
228 }
229#undef HANDLE_CASE
230}

232// Functor to parse a string into one of the known Halide scalar types.
233template<typename T>
234struct ScalarParser {
235 bool operator()(const std::string &str, halide_scalar_value_t *v) {
236 std::istringstream iss(str);
237 // std::setbase(0) means "infer base from input", and allows hex and octal constants
238 iss >> std::setbase(0) >> *(T *)v;
239 return !iss.fail() && iss.get() == EOF;
240 }
241};
242
243// Override for int8 and uint8, to avoid parsing as char variants
244template<>
245inline bool ScalarParser<int8_t>::operator()(const std::string &str, halide_scalar_value_t *v) {
246 std::istringstream iss(str);
247 int i;
248 iss >> std::setbase(0) >> i;
249 if (!(!iss.fail() && iss.get() == EOF) || i < -128 || i > 127) {
250 return false;
251 }
252 v->u.i8 = (int8_t)i;
253 return true;
254}
255
256template<>
257inline bool ScalarParser<uint8_t>::operator()(const std::string &str, halide_scalar_value_t *v) {
258 std::istringstream iss(str);
259 unsigned int u;
260 iss >> std::setbase(0) >> u;
261 if (!(!iss.fail() && iss.get() == EOF) || u > 255) {
262 return false;
263 }
264 v->u.u8 = (uint8_t)u;
265 return true;
266}
267
268// Override for bool, since istream just expects '1' or '0'.
269template<>
270inline bool ScalarParser<bool>::operator()(const std::string &str, halide_scalar_value_t *v) {
271 if (str == "true") {
272 v->u.b = true;
273 return true;
274 }
275 if (str == "false") {
276 v->u.b = false;
277 return true;
278 }
279 return false;
280}

282// Override for handle, since we only accept "nullptr".
283template<>
284inline bool ScalarParser<void *>::operator()(const std::string &str, halide_scalar_value_t *v) {
285 if (str == "nullptr") {
286 v->u.handle = nullptr;
287 return true;
288 }
289 return false;
290}

305// Parse an extent list, which should be of the form
306//
307// [extent0, extent1...]
308//
309// Return a vector<halide_dimension_t> (aka a "shape") with the extents filled in,
310// but with the min of each dimension set to zero and the stride set to the
311// planar-default value.
312inline Shape parse_extents(const std::string &extent_list) {
313 if (extent_list.empty() || extent_list[0] != '[' || extent_list.back() != ']') {
314 fail() << "Invalid format for extents: " << extent_list;
315 }
316 Shape result;
317 if (extent_list == "[]") {
318 return result;
319 }
320 std::vector<std::string> extents = split_string(extent_list.substr(1, extent_list.size() - 2), ",");
321 for (size_t i = 0; i < extents.size(); i++) {
322 const std::string &s = extents[i];
323 const int stride = (i == 0) ? 1 : result[i - 1].stride * result[i - 1].extent;
324 halide_dimension_t d = {0, 0, stride};
325 if (!parse_scalar(s, &d.extent)) {
326 fail() << "Invalid value for extents: " << s << " (" << extent_list << ")";
327 }
328 result.push_back(d);
329 }
330 return result;
331}

333// Parse the buffer_estimate list from a given argument's metadata into a Shape.
334// If no valid buffer_estimate exists, return false.
335inline bool try_parse_metadata_buffer_estimates(const halide_filter_argument_t *md, Shape *shape) {
336 if (!md->buffer_estimates) {
337 // zero-dimensional buffers don't have (or need) estimates, so don't fail.
338 if (md->dimensions == 0) {
339 *shape = Shape();
340 return true;
341 }
342 return false;
343 }
344 Shape result(md->dimensions);
345 int32_t stride = 1;
346 for (int i = 0; i < md->dimensions; i++) {
347 const int64_t *min = md->buffer_estimates[i * 2];
348 const int64_t *extent = md->buffer_estimates[i * 2 + 1];
349 if (!min || !extent) {
350 return false;
351 fail() << "Argument " << md->name << " was specified as 'estimate', but no estimate was provided for dimension " << i << " of " << md->dimensions;
352 }
353 result[i] = halide_dimension_t{(int32_t)*min, (int32_t)*extent, stride};
354 stride *= result[i].extent;
355 }
356 *shape = result;
357 return true;
358};

360// Parse the buffer_estimate list from a given argument's metadata into a Shape.
361// If no valid buffer_estimate exists, fail.
362inline Shape parse_metadata_buffer_estimates(const halide_filter_argument_t *md) {
363 Shape shape;
364 if (!try_parse_metadata_buffer_estimates(md, &shape)) {
365 fail() << "Argument " << md->name << " was specified as 'estimate', but no valid estimates were provided.";
366 }
367 return shape;

380// Given a type and shape, create a new Buffer<> but *don't* allocate allocate storage for it.
381inline Buffer<> make_with_shape(const halide_type_t &type, const Shape &shape) {
382 return Buffer<>(type, nullptr, (int)shape.size(), &shape[0]);

385// Given a type and shape, create a new Buffer<> and allocate storage for it.
386// (Oddly, Buffer<> has an API to do this with vector-of-extent, but not vector-of-halide_dimension_t.)
387inline Buffer<> allocate_buffer(const halide_type_t &type, const Shape &shape) {
388 Buffer<> b = make_with_shape(type, shape);
389 if (b.number_of_elements() > 0) {
390 b.check_overflow();
391 b.allocate();
392 b.set_host_dirty();
393 }
394 return b;

397inline Shape choose_output_extents(int dimensions, const Shape &defaults) {
398 Shape s(dimensions);
399 for (int i = 0; i < dimensions; ++i) {
400 if ((size_t)i < defaults.size()) {
401 s[i] = defaults[i];
402 } else {
403 // If the defaults don't provide enough dimensions, make a guess.
404 s[i].extent = (i < 2 ? 1000 : 4);
405 }
406 }
407 return s;

410inline void fix_chunky_strides(const Shape &constrained_shape, Shape *new_shape) {
411 // Special-case Chunky: most "chunky" generators tend to constrain stride[0]
412 // and stride[2] to exact values, leaving stride[1] unconstrained;
413 // in practice, we must ensure that stride[1] == stride[0] * extent[0]
414 // and stride[0] = extent[2] to get results that are not garbled.
415 // This is unpleasantly hacky and will likely need aditional enhancements.
416 // (Note that there are, theoretically, other stride combinations that might
417 // need fixing; in practice, ~all generators that aren't planar tend
418 // to be classically chunky.)
419 if (new_shape->size() >= 3 &&
420 (*new_shape)[0].extent > 1 &&
421 (*new_shape)[1].extent > 1) {
422 if (constrained_shape[2].stride == 1) {
423 if (constrained_shape[0].stride >= 1) {
424 // If we have stride[0] and stride[2] set to obviously-chunky,
425 // then force extent[2] to match stride[0].
426 (*new_shape)[2].extent = constrained_shape[0].stride;
427 } else {
428 // If we have stride[2] == 1 but stride[0] < 1,
429 // force stride[0] = extent[2]
430 (*new_shape)[0].stride = (*new_shape)[2].extent;
431 }
432 // Ensure stride[1] is reasonable.
433 (*new_shape)[1].stride = (*new_shape)[0].extent * (*new_shape)[0].stride;
434 }
435 }

438// Return true iff all of the dimensions in the range [first, last] have an extent of <= 1.
439inline bool dims_in_range_are_trivial(const Buffer<> &b, int first, int last) {
440 for (int d = first; d <= last; ++d) {
441 if (b.dim(d).extent() > 1) {
442 return false;
443 }
444 }
445 return true;
446}

448// Add or subtract dimensions to the given buffer to match dims_needed,
449// emitting warnings if we do so.
450inline Buffer<> adjust_buffer_dims(const std::string &title, const std::string &name,
451 const int dims_needed, Buffer<> b) {
452 const int dims_actual = b.dimensions();
453 if (dims_actual > dims_needed) {
454 // Warn that we are ignoring dimensions, but only if at least one of the
455 // ignored dimensions has extent > 1
456 if (!dims_in_range_are_trivial(b, dims_needed, dims_actual - 1)) {
457 warn() << "Image for " << title << " \"" << name << "\" has "
458 << dims_actual << " dimensions, but only the first "
459 << dims_needed << " were used; data loss may have occurred.";
460 }
461 auto old_shape = get_shape(b);
462 while (b.dimensions() > dims_needed) {
463 b = b.sliced(dims_needed);
464 }
465 info() << "Shape for " << name << " changed: " << old_shape << " -> " << get_shape(b);
466 } else if (dims_actual < dims_needed) {
467 warn() << "Image for " << title << " \"" << name << "\" has "
468 << dims_actual << " dimensions, but this argument requires at least "
469 << dims_needed << " dimensions: adding dummy dimensions of extent 1.";
470 auto old_shape = get_shape(b);
471 while (b.dimensions() < dims_needed) {
472 b = b.embedded(b.dimensions(), 0);
473 }
474 info() << "Shape for " << name << " changed: " << old_shape << " -> " << get_shape(b);
475 }
476 return b;
477}

479// Load a buffer from a pathname, adjusting the type and dimensions to
480// fit the metadata's requirements as needed.
481inline Buffer<> load_input_from_file(const std::string &pathname,
482 const halide_filter_argument_t &metadata) {
483 Buffer<> b = Buffer<>(metadata.type, 0);
484 info() << "Loading input " << metadata.name << " from " << pathname << " ...";
485 if (!Halide::Tools::load<Buffer<>, IOCheckFail>(pathname, &b)) {
486 fail() << "Unable to load input: " << pathname;
487 }
488 if (b.dimensions() != metadata.dimensions) {
489 b = adjust_buffer_dims("Input", metadata.name, metadata.dimensions, b);
490 }
491 if (b.type() != metadata.type) {
492 warn() << "Image loaded for argument \"" << metadata.name << "\" is type "
493 << b.type() << " but this argument expects type "
494 << metadata.type << "; data loss may have occurred.";
495 b = Halide::Tools::ImageTypeConversion::convert_image(b, metadata.type);
496 }
497 return b;
498}

574template<typename T>
575struct FillWithScalar {
576public:
577 void operator()(Buffer<> &b_dynamic, const halide_scalar_value_t &value) {
578 Buffer<T> b = b_dynamic;
579 b.fill(as_T(value));
580 }
581
582private:
583 // Segregate into pointer and non-pointer clauses to avoid compiler warnings
584 // about casting from (e.g.) int8 to void*
585 template<typename T2 = T, typename std::enable_if<!std::is_pointer<T2>::value>::type * = nullptr>
586 T as_T(const halide_scalar_value_t &value) {
587 const halide_type_t type = halide_type_of<T>();
588 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
589 case halide_type_code(halide_type_int, 8):
590 return (T)value.u.i8;
591 case halide_type_code(halide_type_int, 16):
592 return (T)value.u.i16;
593 case halide_type_code(halide_type_int, 32):
594 return (T)value.u.i32;
595 case halide_type_code(halide_type_int, 64):
596 return (T)value.u.i64;
597 case halide_type_code(halide_type_uint, 1):
598 return (T)value.u.b;
599 case halide_type_code(halide_type_uint, 8):
600 return (T)value.u.u8;
601 case halide_type_code(halide_type_uint, 16):
602 return (T)value.u.u16;
603 case halide_type_code(halide_type_uint, 32):
604 return (T)value.u.u32;
605 case halide_type_code(halide_type_uint, 64):
606 return (T)value.u.u64;
607 case halide_type_code(halide_type_float, 32):
608 return (T)value.u.f32;
609 case halide_type_code(halide_type_float, 64):
610 return (T)value.u.f64;
611 default:
612 fail() << "Can't convert value with type: " << (int)type.code << "bits: " << type.bits;
613 return (T)0;
614 }
615 }

617 template<typename T2 = T, typename std::enable_if<std::is_pointer<T2>::value>::type * = nullptr>
618 T as_T(const halide_scalar_value_t &value) {
619 const halide_type_t type = halide_type_of<T>();
620 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
621 case halide_type_code(halide_type_handle, 64):
622 return (T)value.u.handle;
623 default:
624 fail() << "Can't convert value with type: " << (int)type.code << "bits: " << type.bits;
625 return (T)0;
626 }
627 }
628};
629
630// This logic exists in Halide::Tools, but is Internal; we're going to replicate
631// it here for now since we may want slightly different logic in some cases
632// for this tool.
633inline Halide::Tools::FormatInfo best_save_format(const Buffer<> &b, const std::set<Halide::Tools::FormatInfo> &info) {
634 // Perfect score is zero (exact match).
635 // The larger the score, the worse the match.
636 int best_score = 0x7fffffff;
637 Halide::Tools::FormatInfo best{};
638 const halide_type_t type = b.type();
639 const int dimensions = b.dimensions();
640 for (auto &f : info) {
641 int score = 0;
642 // If format has too-few dimensions, that's very bad.
643 score += std::abs(f.dimensions - dimensions) * 128;
644 // If format has too-few bits, that's pretty bad.
645 score += std::abs(f.type.bits - type.bits);
646 // If format has different code, that's a little bad.
647 score += (f.type.code != type.code) ? 1 : 0;
648 if (score < best_score) {
649 best_score = score;
650 best = f;
651 }
652 }
653
654 return best;
655}

657inline std::string scalar_to_string(const halide_type_t &type,
658 const halide_scalar_value_t &value) {
659 std::ostringstream o;
660 switch (halide_type_code((halide_type_code_t)type.code, type.bits)) {
661 case halide_type_code(halide_type_float, 32):
662 o << value.u.f32;
663 break;
664 case halide_type_code(halide_type_float, 64):
665 o << value.u.f64;
666 break;
667 case halide_type_code(halide_type_int, 8):
668 o << (int)value.u.i8;
669 break;
670 case halide_type_code(halide_type_int, 16):
671 o << value.u.i16;
672 break;
673 case halide_type_code(halide_type_int, 32):
674 o << value.u.i32;
675 break;
676 case halide_type_code(halide_type_int, 64):
677 o << value.u.i64;
678 break;
679 case halide_type_code(halide_type_uint, 1):
680 o << (value.u.b ? "true" : "false");
681 break;
682 case halide_type_code(halide_type_uint, 8):
683 o << (int)value.u.u8;
684 break;
685 case halide_type_code(halide_type_uint, 16):
686 o << value.u.u16;
687 break;
688 case halide_type_code(halide_type_uint, 32):
689 o << value.u.u32;
690 break;
691 case halide_type_code(halide_type_uint, 64):
692 o << value.u.u64;
693 break;
694 case halide_type_code(halide_type_handle, 64):
695 o << (uint64_t)value.u.handle;
696 break;
697 default:
698 fail() << "Unsupported type: " << type << "\n";
699 break;
700 }
701 return o.str();
702}

718 Buffer<> load_buffer(ShapePromise shape_promise, const halide_filter_argument_t *argument_metadata) {
719 const auto parse_optional_extents = [&](const std::string &s) -> Shape {
720 if (s == "auto") {
721 return shape_promise();
722 }
723 if (s == "estimate") {
724 return parse_metadata_buffer_estimates(argument_metadata);
725 }
726 if (s == "estimate_then_auto") {
727 Shape shape;
728 if (!try_parse_metadata_buffer_estimates(argument_metadata, &shape)) {
729 info() << "Input " << argument_metadata->name << " has no estimates; using bounds-query result instead.";
730 shape = shape_promise();
731 }
732 return shape;
733 }
734 return parse_extents(s);
735 };
736
737 std::vector<std::string> v = split_string(raw_string, ":");
738 if (v[0] == "zero") {
739 if (v.size() != 2) {
740 fail() << "Invalid syntax: " << raw_string;
741 }
742 auto shape = parse_optional_extents(v[1]);
743 Buffer<> b = allocate_buffer(metadata->type, shape);
744 memset(b.data(), 0, b.size_in_bytes());
745 return b;
746 } else if (v[0] == "constant") {
747 if (v.size() != 3) {
748 fail() << "Invalid syntax: " << raw_string;
749 }
750 halide_scalar_value_t value;
751 if (!parse_scalar(metadata->type, v[1], &value)) {
752 fail() << "Invalid value for constant value";
753 }
754 auto shape = parse_optional_extents(v[2]);
755 Buffer<> b = allocate_buffer(metadata->type, shape);
756 dynamic_type_dispatch<FillWithScalar>(metadata->type, b, value);
757 return b;
758 } else if (v[0] == "identity") {
759 if (v.size() != 2) {
760 fail() << "Invalid syntax: " << raw_string;
761 }
762 auto shape = parse_optional_extents(v[1]);
763 // Make a binary buffer with diagonal elements set to true. Diagonal
764 // elements are those whose first two dimensions are equal.
765 Buffer<bool> b = allocate_buffer(halide_type_of<bool>(), shape);
766 b.for_each_element([&b](const int *pos) {
767 b(pos) = (b.dimensions() >= 2) ? (pos[0] == pos[1]) : (pos[0] == 0);
768 });
769 // Convert the binary buffer to the required type, so true becomes 1.
770 return Halide::Tools::ImageTypeConversion::convert_image(b, metadata->type);
771 } else if (v[0] == "random") {
772 if (v.size() != 3) {
773 fail() << "Invalid syntax: " << raw_string;
774 }
775 int seed;
776 if (!parse_scalar(v[1], &seed)) {
777 fail() << "Invalid value for seed";
778 }
779 auto shape = parse_optional_extents(v[2]);
780 Buffer<> b = allocate_buffer(metadata->type, shape);
781 dynamic_type_dispatch<FillWithRandom>(metadata->type, b, seed);
782 return b;
783 } else {
784 return load_input_from_file(v[0], *metadata);
785 }
786 }

788 Buffer<> load_buffer(const Shape &shape, const halide_filter_argument_t *argument_metadata) {
789 ShapePromise promise = [shape]() -> Shape { return shape; };
790 return load_buffer(promise, argument_metadata);

793 void adapt_input_buffer(const Shape &constrained_shape) {
794 if (metadata->kind != halide_argument_kind_input_buffer) {
795 return;
796 }
797
798 // Ensure that the input Buffer meets our constraints; if it doesn't, allcoate
799 // and copy into a new Buffer.
800 bool updated = false;
801 Shape new_shape = get_shape(buffer_value);
802 info() << "Input " << name << ": Shape is " << new_shape;
803 if (new_shape.size() != constrained_shape.size()) {
804 fail() << "Dimension mismatch; expected " << constrained_shape.size() << "dimensions";
805 }
806 for (size_t i = 0; i < constrained_shape.size(); ++i) {
807 // If the constrained shape is not in bounds of the
808 // buffer's current shape we need to use the constrained
809 // shape.
810 int current_min = new_shape[i].min;
811 int current_max = new_shape[i].min + new_shape[i].extent - 1;
812 int constrained_min = constrained_shape[i].min;
813 int constrained_max = constrained_shape[i].min + constrained_shape[i].extent - 1;
814 if (constrained_min < current_min || constrained_max > current_max) {
815 new_shape[i].min = constrained_shape[i].min;
816 new_shape[i].extent = constrained_shape[i].extent;
817 updated = true;
818 }
819 // stride of nonzero means "required stride", stride of zero means "no constraints"
820 if (constrained_shape[i].stride != 0 && new_shape[i].stride != constrained_shape[i].stride) {
821 new_shape[i].stride = constrained_shape[i].stride;
822 updated = true;
823 }
824 }
825 if (updated) {
826 fix_chunky_strides(constrained_shape, &new_shape);
827 Buffer<> new_buf = allocate_buffer(buffer_value.type(), new_shape);
828 new_buf.copy_from(buffer_value);
829 buffer_value = new_buf;
830 }
831
832 info() << "Input " << name << ": BoundsQuery result is " << constrained_shape;
833 if (updated) {
834 info() << "Input " << name << ": Updated Shape is " << get_shape(buffer_value);
835 }
836 }
837
838 void allocate_output_buffer(const Shape &constrained_shape) {
839 if (metadata->kind != halide_argument_kind_output_buffer) {
840 return;
841 }
842
843 // Given a constraint Shape (generally produced by a bounds query), create a new
844 // Shape that can legally be used to create and allocate a new Buffer:
845 // ensure that extents/strides aren't zero, do some reality checking
846 // on planar vs interleaved, and generally try to guess at a reasonable result.
847 Shape new_shape = constrained_shape;
848
849 // Make sure that the extents and strides for these are nonzero.
850 for (size_t i = 0; i < new_shape.size(); ++i) {
851 if (!new_shape[i].extent) {
852 // A bit of a hack: fill in unconstrained dimensions to 1... except
853 // for probably-the-channels dimension, which we'll special-case to
854 // fill in to 4 when possible (unless it appears to be chunky).
855 // Stride will be fixed below.
856 if (i == 2) {
857 if (constrained_shape[0].stride >= 1 && constrained_shape[2].stride == 1) {
858 // Definitely chunky, so make extent[2] match the chunk size
859 new_shape[i].extent = constrained_shape[0].stride;
860 } else {
861 // Not obviously chunky; let's go with 4 channels.
862 new_shape[i].extent = 4;
863 }
864 } else {
865 new_shape[i].extent = 1;
866 }
867 }
868 }
869
870 fix_chunky_strides(constrained_shape, &new_shape);
871
872 // If anything else is zero, just set strides to planar and hope for the best.
873 bool any_strides_zero = false;
874 for (size_t i = 0; i < new_shape.size(); ++i) {
875 if (!new_shape[i].stride) {
876 any_strides_zero = true;
877 }
878 }
879 if (any_strides_zero) {
880 // Planar
881 new_shape[0].stride = 1;
882 for (size_t i = 1; i < new_shape.size(); ++i) {
883 new_shape[i].stride = new_shape[i - 1].stride * new_shape[i - 1].extent;
884 }
885 }
886
887 buffer_value = allocate_buffer(metadata->type, new_shape);
888
889 // allocate_buffer conservatively sets host dirty. Don't waste
890 // time copying output buffers to device.
891 buffer_value.set_host_dirty(false);
892
893 info() << "Output " << name << ": BoundsQuery result is " << constrained_shape;
894 info() << "Output " << name << ": Shape is " << get_shape(buffer_value);
895 }
896};

990 // Parse all the input arguments, loading images as necessary.
991 // (Don't handle outputs yet.)
992 void load_inputs(const std::string &user_specified_output_shape_string) {
993 assert(output_shapes.empty());
994
995 Shape first_input_shape;
996 std::map<std::string, ShapePromise> auto_input_shape_promises;
997
998 // First, set all the scalar inputs: we need those to be correct
999 // in order to get useful values from the bound-query for input buffers.
1000 for (auto &arg_pair : args) {
1001 auto &arg_name = arg_pair.first;
1002 auto &arg = arg_pair.second;
1003 switch (arg.metadata->kind) {
1004 case halide_argument_kind_input_scalar: {
1005 if (!strcmp(arg.metadata->name, "__user_context")) {
1006 arg.scalar_value.u.handle = nullptr;
1007 info() << "Argument value for: __user_context is special-cased as: nullptr";
1008 break;
1009 }
1010 std::vector<std::pair<const halide_scalar_value_t *, const char *>> values;
1011 // If this gets any more complex, smarten it up, but for now,
1012 // simpleminded code is fine.
1013 if (arg.raw_string == "default") {
1014 values.emplace_back(arg.metadata->scalar_def, "default");
1015 } else if (arg.raw_string == "estimate") {
1016 values.emplace_back(arg.metadata->scalar_estimate, "estimate");
1017 } else if (arg.raw_string == "default,estimate") {
1018 values.emplace_back(arg.metadata->scalar_def, "default");
1019 values.emplace_back(arg.metadata->scalar_estimate, "estimate");
1020 } else if (arg.raw_string == "estimate,default") {
1021 values.emplace_back(arg.metadata->scalar_estimate, "estimate");
1022 values.emplace_back(arg.metadata->scalar_def, "default");
1023 }
1024 if (!values.empty()) {
1025 bool set = false;
1026 for (auto &v : values) {
1027 if (!v.first) {
1028 continue;
1029 }
1030 info() << "Argument value for: " << arg.metadata->name << " is parsed from metadata (" << v.second << ") as: "
1031 << scalar_to_string(arg.metadata->type, *v.first);
1032 arg.scalar_value = *v.first;
1033 set = true;
1034 break;
1035 }
1036 if (!set) {
1037 fail() << "Argument value for: " << arg.metadata->name << " was specified as '" << arg.raw_string << "', "
1038 << "but no default and/or estimate was found in the metadata.";
1039 }
1040 } else {
1041 if (!parse_scalar(arg.metadata->type, arg.raw_string, &arg.scalar_value)) {
1042 fail() << "Argument value for: " << arg_name << " could not be parsed as type "
1043 << arg.metadata->type << ": "
1044 << arg.raw_string;
1045 }
1046 }
1047 break;
1048 }
1049 case halide_argument_kind_input_buffer:
1050 case halide_argument_kind_output_buffer:
1051 // Nothing yet
1052 break;
1053 }
1054 }
1055
1056 if (!user_specified_output_shape_string.empty()) {
1057 // For now, we set all output shapes to be identical -- there's no
1058 // way on the command line to specify different shapes for each
1059 // output. Would be nice to try?
1060 for (auto &arg_pair : args) {
1061 auto &arg = arg_pair.second;
1062 if (arg.metadata->kind == halide_argument_kind_output_buffer) {
1063 auto &arg_name = arg_pair.first;
1064 if (user_specified_output_shape_string == "estimate") {
1065 output_shapes[arg_name] = parse_metadata_buffer_estimates(arg.metadata);
1066 info() << "Output " << arg_name << " is parsed from metadata as: " << output_shapes[arg_name];
1067 } else {
1068 output_shapes[arg_name] = parse_extents(user_specified_output_shape_string);
1069 info() << "Output " << arg_name << " has user-specified Shape: " << output_shapes[arg_name];
1070 }
1071 }
1072 }
1073 auto_input_shape_promises = bounds_query_input_shapes();
1074 }
1075
1076 for (auto &arg_pair : args) {
1077 auto &arg_name = arg_pair.first;
1078 auto &arg = arg_pair.second;
1079 switch (arg.metadata->kind) {
1080 case halide_argument_kind_input_buffer:
1081 arg.buffer_value = arg.load_buffer(auto_input_shape_promises[arg_name], arg.metadata);
1082 info() << "Input " << arg_name << ": Shape is " << get_shape(arg.buffer_value);
1083 if (first_input_shape.empty()) {
1084 first_input_shape = get_shape(arg.buffer_value);
1085 }
1086 break;
1087 case halide_argument_kind_input_scalar:
1088 // Already handled.
1089 break;
1090 case halide_argument_kind_output_buffer:
1091 // Nothing yet
1092 break;
1093 }
1094 }

1098 // the first input buffer (if any). (This is a better-than-nothing guess
1099 // that is definitely not always correct, but is convenient and useful enough
1100 // to be worth doing.)
1101 for (auto &arg_pair : args) {
1102 auto &arg = arg_pair.second;
1103 if (arg.metadata->kind == halide_argument_kind_output_buffer) {
1104 auto &arg_name = arg_pair.first;

1226 // Run a bounds-query call with the given args, and return the shapes
1227 // to which we are constrained.
1228 std::vector<Shape> run_bounds_query() const {
1229 std::vector<void *> filter_argv(args.size(), nullptr);

1400 std::map<std::string, ShapePromise> bounds_query_input_shapes() const {
1401 assert(!output_shapes.empty());
1402 std::vector<void *> filter_argv(args.size(), nullptr);
1403 std::vector<Buffer<>> bounds_query_buffers(args.size());
1404 for (const auto &arg_pair : args) {
1405 auto &arg_name = arg_pair.first;
1406 auto &arg = arg_pair.second;
1407 switch (arg.metadata->kind) {
1408 case halide_argument_kind_input_scalar:
1409 filter_argv[arg.index] = const_cast<halide_scalar_value_t *>(&arg.scalar_value);
1410 break;
1411 case halide_argument_kind_input_buffer:
1412 // Make a Buffer<> that has the right dimension count and extent=0 for all of them
1413 bounds_query_buffers[arg.index] = Buffer<>(arg.metadata->type, std::vector<int>(arg.metadata->dimensions, 0));
1414 filter_argv[arg.index] = bounds_query_buffers[arg.index].raw_buffer();
1415 break;
1416 case halide_argument_kind_output_buffer:
1417 bounds_query_buffers[arg.index] = make_with_shape(arg.metadata->type, output_shapes.at(arg_name));
1418 filter_argv[arg.index] = bounds_query_buffers[arg.index].raw_buffer();
1419 break;
1420 }
1421 }
1422
1423 auto previous_error_handler = halide_set_error_handler(rungen_ignore_error);
1424 int result = halide_argv_call(&filter_argv[0]);
1425 halide_set_error_handler(previous_error_handler);
1426
1427 std::map<std::string, ShapePromise> input_shape_promises;
1428 for (const auto &arg_pair : args) {
1429 auto &arg_name = arg_pair.first;
1430 auto &arg = arg_pair.second;
1431 if (arg.metadata->kind == halide_argument_kind_input_buffer) {
1432 if (result == 0) {
1433 Shape shape = get_shape(bounds_query_buffers[arg.index]);
1434 input_shape_promises[arg_name] = [shape]() -> Shape { return shape; };
1435 info() << "Input " << arg_name << " has a bounds-query shape of " << shape;
1436 } else {
1437 input_shape_promises[arg_name] = [arg_name]() -> Shape {
1438 fail() << "Input " << arg_name << " could not calculate a shape satisfying bounds-query constraints.\n"
1439 << "Try relaxing the constraints, or providing an explicit estimate for the input.\n";
1440 return Shape();
1441 };
1442 info() << "Input " << arg_name << " failed bounds-query\n";
1443 }
1444 }
1445 }
1446 return input_shape_promises;
1447 }
