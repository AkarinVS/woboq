39struct FormatInfo {
40 halide_type_t type;
41 int dimensions;
42
43 bool operator<(const FormatInfo &other) const {
44 if (type.code < other.type.code) {
45 return true;
46 } else if (type.code > other.type.code) {
47 return false;
48 }
49 if (type.bits < other.type.bits) {
50 return true;
51 } else if (type.bits > other.type.bits) {
52 return false;
53 }
54 if (type.lanes < other.type.lanes) {
55 return true;
56 } else if (type.lanes > other.type.lanes) {
57 return false;
58 }
59 return (dimensions < other.dimensions);
60 }
61};

660 char buf[1024];

702// Read a row of ElemTypes from a byte buffer and copy them into a specific image row.
703// Multibyte elements are assumed to be big-endian.
704template<typename ElemType, typename ImageType>
705void read_big_endian_row(const uint8_t *src, int y, ImageType *im) {
706 auto im_typed = im->template as<ElemType>();
707 const int xmin = im_typed.dim(0).min();
708 const int xmax = im_typed.dim(0).max();
709 if (im_typed.dimensions() > 2) {
710 const int cmin = im_typed.dim(2).min();
711 const int cmax = im_typed.dim(2).max();
712 for (int x = xmin; x <= xmax; x++) {
713 for (int c = cmin; c <= cmax; c++) {
714 im_typed(x, y, c + cmin) = read_big_endian<ElemType>(src);
715 src += sizeof(ElemType);
716 }
717 }
718 } else {
719 for (int x = xmin; x <= xmax; x++) {
720 im_typed(x, y) = read_big_endian<ElemType>(src);
721 src += sizeof(ElemType);
722 }
723 }
724}

726// Copy a row from an image into a byte buffer.
727// Multibyte elements are written in big-endian layout.
728template<typename ElemType, typename ImageType>
729void write_big_endian_row(const ImageType &im, int y, uint8_t *dst) {
730 auto im_typed = im.template as<typename std::add_const<ElemType>::type>();
731 const int xmin = im_typed.dim(0).min();
732 const int xmax = im_typed.dim(0).max();
733 if (im_typed.dimensions() > 2) {
734 const int cmin = im_typed.dim(2).min();
735 const int cmax = im_typed.dim(2).max();
736 for (int x = xmin; x <= xmax; x++) {
737 for (int c = cmin; c <= cmax; c++) {
738 write_big_endian<ElemType>(im_typed(x, y, c), dst);
739 dst += sizeof(ElemType);
740 }
741 }
742 } else {
743 for (int x = xmin; x <= xmax; x++) {
744 write_big_endian<ElemType>(im_typed(x, y), dst);
745 dst += sizeof(ElemType);
746 }
747 }

752template<typename ImageType, Internal::CheckFunc check = Internal::CheckReturn>
753bool load_png(const std::string &filename, ImageType *im) {
754 static_assert(!ImageType::has_static_halide_type, "");
755
756 /* open file and test for it being a png */
757 Internal::FileOpener f(filename, "rb");
758 if (!check(f.f != nullptr, "File could not be opened for reading")) {
759 return false;
760 }
761 png_byte header[8];
762 if (!check(f.read_array(header), "File ended before end of header")) {
763 return false;
764 }
765 if (!check(!png_sig_cmp(header, 0, 8), "File is not recognized as a PNG file")) {
766 return false;
767 }
768
769 /* initialize stuff */
770 png_structp png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING, nullptr, nullptr, nullptr);
771 if (!check(png_ptr != nullptr, "png_create_read_struct failed")) {
772 return false;
773 }
774
775 png_infop info_ptr = png_create_info_struct(png_ptr);
776 if (!check(info_ptr != nullptr, "png_create_info_struct failed")) {
777 return false;

905bool read_pnm_header(Internal::FileOpener &f, const std::string &hdr_fmt, int *width, int *height, int *bit_depth) {
906 if (!check(f.f != nullptr, "File could not be opened for reading")) {
907 return false;
908 }
909
910 char header[256];
911 if (!check(f.scan_line("%255s", header) == 1, "Could not read header")) {
912 return false;

915 if (!check(to_lowercase(hdr_fmt) == to_lowercase(header), "Unexpected file header")) {
916 return false;
917 }

920 return false;

924 if (!check(f.scan_line("%d", &maxval) == 1, "Could not read max value")) {

929 } else if (maxval == 65535) {

934 }

1175// return true iff the buffer storage has no padding between
1176// any elements, and is in strictly planar order.
1177template<typename ImageType>
1178bool buffer_is_compact_planar(ImageType &im) {
1179 const halide_type_t im_type = im.type();
1180 const size_t elem_size = (im_type.bits / 8);
1181 if (((const uint8_t *)im.begin() + (im.number_of_elements() * elem_size)) != (const uint8_t *)im.end()) {
1182 return false;
1183 }
1184 for (int d = 1; d < im.dimensions(); ++d) {
1185 if (im.dim(d - 1).stride() > im.dim(d).stride()) {
1186 return false;
1187 }
1188 // Strides can only match if the previous dimension has extent 1
1189 // (this can happen when artificially adding dimension(s), e.g.
1190 // to write a .tmp file)
1191 if (im.dim(d - 1).stride() == im.dim(d).stride() && im.dim(d - 1).extent() != 1) {
1192 return false;
1193 }
1194 }
1195 return true;
1196}

1204 if (!check(f.f != nullptr, "File could not be opened for reading")) {
1205 return false;
1206 }
1207
1208 int32_t header[5];
1209 if (!check(f.read_array(header), "Count not read .tmp header")) {

1214 header[4] >= 0 && header[4] < kNumTmpCodes,

1230 }

1253template<typename ImageType, CheckFunc check = CheckReturn>
1254bool write_planar_payload(ImageType &im, FileOpener &f) {
1255 if (im.dimensions() == 0 || buffer_is_compact_planar(im)) {
1256 // Contiguous buffer! Write it all in one swell foop.
1257 if (!check(f.write_bytes(im.begin(), im.size_in_bytes()), "Count not write .tmp payload")) {
1258 return false;
1259 }
1260 } else {
1261 // We have to do this the hard way.
1262 int d = im.dimensions() - 1;
1263 for (int i = im.dim(d).min(); i <= im.dim(d).max(); i++) {
1264 auto slice = im.sliced(d, i);
1265 if (!write_planar_payload(slice, f)) {
1266 return false;
1267 }
1268 }
1269 }
1270 return true;
1271}

1345template<typename ImageType, CheckFunc check = CheckReturn>
1346bool load_mat(const std::string &filename, ImageType *im) {
1347 static_assert(!ImageType::has_static_halide_type, "");
1348
1349 FileOpener f(filename, "rb");
1350 if (!check(f.f != nullptr, "File could not be opened for reading")) {
1351 return false;
1352 }
1353
1354 uint8_t header[128];
1355 if (!check(f.read_array(header), "Could not read .mat header\n")) {
1356 return false;
1357 }
1358
1359 // Matrix header
1360 uint32_t matrix_header[2];
1361 if (!check(f.read_array(matrix_header), "Could not read .mat header\n")) {
1362 return false;
1363 }
1364 if (!check(matrix_header[0] == miMATRIX, "Could not parse this .mat file: bad matrix header\n")) {
1365 return false;
1366 }
1367
1368 // Array flags
1369 uint32_t flags[4];
1370 if (!check(f.read_array(flags), "Could not read .mat header\n")) {
1371 return false;
1372 }
1373 if (!check(flags[0] == miUINT32 && flags[1] == 8, "Could not parse this .mat file: bad flags\n")) {
1374 return false;
1375 }
1376
1377 // Shape
1378 uint32_t shape_header[2];
1379 if (!check(f.read_array(shape_header), "Could not read .mat header\n")) {
1380 return false;
1381 }
1382 if (!check(shape_header[0] == miINT32, "Could not parse this .mat file: bad shape header\n")) {
1383 return false;
1384 }
1385 int dims = shape_header[1] / 4;
1386 std::vector<int> extents(dims);
1387 if (!check(f.read_vector(&extents), "Could not read .mat header\n")) {
1388 return false;
1389 }
1390 if (dims & 1) {
1391 uint32_t padding;
1392 if (!check(f.read_bytes(&padding, 4), "Could not read .mat header\n")) {
1393 return false;
1394 }
1395 }
1396
1397 // Skip over the name
1398 uint32_t name_header[2];
1399 if (!check(f.read_array(name_header), "Could not read .mat header\n")) {
1400 return false;
1401 }
1402
1403 if (name_header[0] >> 16) {
1404 // Name must be fewer than 4 chars, and so the whole name
1405 // field was stored packed into 8 bytes
1406 } else {
1407 if (!check(name_header[0] == miINT8, "Could not parse this .mat file: bad name header\n")) {
1408 return false;
1409 }
1410 std::vector<uint64_t> scratch((name_header[1] + 7) / 8);
1411 if (!check(f.read_vector(&scratch), "Could not read .mat header\n")) {
1412 return false;
1413 }
1414 }
1415
1416 // Payload header
1417 uint32_t payload_header[2];
1418 if (!check(f.read_array(payload_header), "Could not read .mat header\n")) {
1419 return false;
1420 }
1421 halide_type_t type;
1422 switch (payload_header[0]) {
1423 case miINT8:
1424 type = halide_type_of<int8_t>();
1425 break;
1426 case miINT16:
1427 type = halide_type_of<int16_t>();
1428 break;
1429 case miINT32:
1430 type = halide_type_of<int32_t>();
1431 break;
1432 case miINT64:
1433 type = halide_type_of<int64_t>();
1434 break;
1435 case miUINT8:
1436 type = halide_type_of<uint8_t>();
1437 break;
1438 case miUINT16:
1439 type = halide_type_of<uint16_t>();
1440 break;
1441 case miUINT32:
1442 type = halide_type_of<uint32_t>();
1443 break;
1444 case miUINT64:
1445 type = halide_type_of<uint64_t>();
1446 break;
1447 case miSINGLE:
1448 type = halide_type_of<float>();
1449 break;
1450 case miDOUBLE:
1451 type = halide_type_of<double>();
1452 break;
1453 }
1454
1455 *im = ImageType(type, extents);
1456
1457 // This should never fail unless the default Buffer<> constructor behavior changes.
1458 if (!check(buffer_is_compact_planar(*im), "load_mat() requires compact planar images")) {
1459 return false;
1460 }
1461
1462 if (!check(f.read_bytes(im->begin(), im->size_in_bytes()), "Could not read .tmp payload")) {
1463 return false;
1464 }
1465
1466 im->set_host_dirty();
1467 return true;
1468}

1492template<typename ImageType, CheckFunc check = CheckReturn>
1493bool save_mat(ImageType &im, const std::string &filename) {
1494 static_assert(!ImageType::has_static_halide_type, "");
1495
1496 im.copy_to_host();
1497
1498 uint32_t class_code = 0, type_code = 0;
1499 switch (im.raw_buffer()->type.code) {
1500 case halide_type_int:
1501 switch (im.raw_buffer()->type.bits) {
1502 case 8:
1503 class_code = mxINT8_CLASS;
1504 type_code = miINT8;
1505 break;
1506 case 16:
1507 class_code = mxINT16_CLASS;
1508 type_code = miINT16;
1509 break;
1510 case 32:
1511 class_code = mxINT32_CLASS;
1512 type_code = miINT32;
1513 break;
1514 case 64:
1515 class_code = mxINT64_CLASS;
1516 type_code = miINT64;
1517 break;
1518 default:
1519 check(false, "unreachable");
1520 };
1521 break;
1522 case halide_type_uint:
1523 switch (im.raw_buffer()->type.bits) {
1524 case 8:
1525 class_code = mxUINT8_CLASS;
1526 type_code = miUINT8;
1527 break;
1528 case 16:
1529 class_code = mxUINT16_CLASS;
1530 type_code = miUINT16;
1531 break;
1532 case 32:
1533 class_code = mxUINT32_CLASS;
1534 type_code = miUINT32;
1535 break;
1536 case 64:
1537 class_code = mxUINT64_CLASS;
1538 type_code = miUINT64;
1539 break;
1540 default:
1541 check(false, "unreachable");
1542 };
1543 break;
1544 case halide_type_float:
1545 switch (im.raw_buffer()->type.bits) {
1546 case 16:
1547 check(false, "float16 not supported by .mat");
1548 break;
1549 case 32:
1550 class_code = mxSINGLE_CLASS;
1551 type_code = miSINGLE;
1552 break;
1553 case 64:
1554 class_code = mxDOUBLE_CLASS;
1555 type_code = miDOUBLE;
1556 break;
1557 default:
1558 check(false, "unreachable");
1559 };
1560 break;
1561 case halide_type_bfloat:
1562 check(false, "bfloat not supported by .mat");
1563 break;
1564 default:
1565 check(false, "unreachable");
1566 }
1567
1568 FileOpener f(filename, "wb");
1569 if (!check(f.f != nullptr, "File could not be opened for writing")) {
1570 return false;
1571 }
1572
1573 // Pick a name for the array
1574 size_t idx = filename.rfind('.');
1575 std::string name = filename.substr(0, idx);
1576 idx = filename.rfind('/');
1577 if (idx != std::string::npos) {
1578 name = name.substr(idx + 1);
1579 }
1580
1581 // Matlab variable names conform to similar rules as C
1582 if (name.empty() || !std::isalpha(name[0])) {
1583 name = "v" + name;
1584 }
1585 for (size_t i = 0; i < name.size(); i++) {
1586 if (!std::isalnum(name[i])) {
1587 name[i] = '_';
1588 }
1589 }
1590
1591 uint32_t name_size = (int)name.size();
1592 while (name.size() & 0x7) {
1593 name += '\0';
1594 }
1595
1596 char header[128] = "MATLAB 5.0 MAT-file, produced by Halide";
1597 int len = strlen(header);
1598 memset(header + len, ' ', sizeof(header) - len);
1599
1600 // Version
1601 *((uint16_t *)(header + 124)) = 0x0100;
1602
1603 // Endianness check
1604 header[126] = 'I';
1605 header[127] = 'M';
1606
1607 uint64_t payload_bytes = im.size_in_bytes();
1608
1609 if (!check((payload_bytes >> 32) == 0, "Buffer too large to save as .mat")) {
1610 return false;
1611 }
1612
1613 int dims = im.dimensions();
1614 if (dims < 2) {
1615 dims = 2;
1616 }
1617 int padded_dims = dims + (dims & 1);
1618
1619 uint32_t padding_bytes = 7 - ((payload_bytes - 1) & 7);
1620
1621 // Matrix header
1622 uint32_t matrix_header[2] = {
1623 miMATRIX, 40 + padded_dims * 4 + (uint32_t)name.size() + (uint32_t)payload_bytes + padding_bytes};
1624
1625 // Array flags
1626 uint32_t flags[4] = {
1627 miUINT32, 8, class_code, 1};
1628
1629 // Shape
1630 int32_t shape[2] = {
1631 miINT32,
1632 im.dimensions() * 4,
1633 };
1634 std::vector<int> extents(im.dimensions());
1635 for (int d = 0; d < im.dimensions(); d++) {
1636 extents[d] = im.dim(d).extent();
1637 }
1638 while ((int)extents.size() < dims) {
1639 extents.push_back(1);
1640 }
1641 while ((int)extents.size() < padded_dims) {
1642 extents.push_back(0);
1643 }
1644
1645 // Name
1646 uint32_t name_header[2] = {
1647 miINT8, name_size};
1648
1649 // Payload header
1650 uint32_t payload_header[2] = {
1651 type_code, (uint32_t)payload_bytes};
1652
1653 bool success =
1654 f.write_array(header) &&
1655 f.write_array(matrix_header) &&
1656 f.write_array(flags) &&
1657 f.write_array(shape) &&
1658 f.write_vector(extents) &&
1659 f.write_array(name_header) &&
1660 f.write_bytes(&name[0], name.size()) &&
1661 f.write_array(payload_header);
1662
1663 if (!check(success, "Could not write .mat header")) {
1664 return false;
1665 }
1666
1667 if (!write_planar_payload<ImageType, check>(im, f)) {
1668 return false;
1669 }
1670
1671 // Padding
1672 if (!check(padding_bytes < 8, "Too much padding!\n")) {
1673 return false;
1674 }
1675 uint64_t padding = 0;
1676 if (!f.write_bytes(&padding, padding_bytes)) {
1677 return false;
1678 }
1679
1680 return true;
1681}

1758template<typename ElemType, int BUFFER_SIZE = 1024>
1759struct ElemWriter {
1760 ElemWriter(FileOpener *f)
1761 : f(f), next(&buf[0]), ok(true) {
1762 }
1763 ~ElemWriter() {
1764 flush();
1765 }
1766
1767 void operator()(const ElemType &elem) {
1768 if (!ok) {
1769 return;
1770 }
1771
1772 *next++ = elem;
1773 if (next == &buf[BUFFER_SIZE]) {
1774 flush();
1775 }
1776 }
1777
1778 void flush() {
1779 if (!ok) {
1780 return;
1781 }
1782
1783 if (next > buf) {
1784 if (!f->write_bytes(buf, (next - buf) * sizeof(ElemType))) {
1785 ok = false;
1786 }
1787 next = buf;
1788 }
1789 }
1790
1791 FileOpener *const f;
1792 ElemType buf[BUFFER_SIZE];
1793 ElemType *next;
1794 bool ok;
1795};

1797// Note that this is a fairly simpleminded TIFF writer that doesn't
1798// do any compression. It would be desirable to (optionally) support using libtiff
1799// here instead, which would also allow us to provide a useful implementation
1800// for TIFF reading.
1801template<typename ImageType, Internal::CheckFunc check = Internal::CheckReturn>
1802bool save_tiff(ImageType &im, const std::string &filename) {
1803 static_assert(!ImageType::has_static_halide_type, "");
1804
1805 im.copy_to_host();
1806
1807 if (!check(im.dimensions() <= 4, "Can only save TIFF files with <= 4 dimensions")) {
1808 return false;
1809 }
1810
1811 FileOpener f(filename, "wb");
1812 if (!check(f.f != nullptr, "File could not be opened for writing")) {
1813 return false;
1814 }
1815
1816 const size_t elements = im.number_of_elements();
1817 halide_dimension_t shape[4];
1818 for (int i = 0; i < im.dimensions() && i < 4; i++) {
1819 const auto &d = im.dim(i);
1820 shape[i].min = d.min();
1821 shape[i].extent = d.extent();
1822 shape[i].stride = d.stride();
1823 }
1824 for (int i = im.dimensions(); i < 4; i++) {
1825 shape[i].min = 0;
1826 shape[i].extent = 1;
1827 shape[i].stride = 0;
1828 }
1829 const halide_type_t im_type = im.type();
1830 if (!check(im_type.code >= 0 && im_type.code < 3, "Unsupported image type")) {
1831 return false;
1832 }
1833 const int32_t bytes_per_element = im_type.bytes();
1834 const int32_t width = shape[0].extent;
1835 const int32_t height = shape[1].extent;
1836 int32_t depth = shape[2].extent;
1837 int32_t channels = shape[3].extent;
1838
1839 if ((channels == 0 || channels == 1) && (depth < 5)) {
1840 channels = depth;
1841 depth = 1;
1842 }
1843
1844 // TIFF sample type values are:
1845 // 0 => Signed int
1846 // 1 => Unsigned int
1847 // 2 => Floating-point
1848 static const int16_t type_code_to_tiff_sample_type[] = {
1849 2, 1, 3};
1850
1851 struct halide_tiff_header header;
1852 memset(&header, 0, sizeof(header));
1853
1854 const int32_t MMII = 0x4d4d4949;
1855 // Select the appropriate two bytes signaling byte order automatically
1856 const char *c = (const char *)&MMII;
1857 header.byte_order_marker = (c[0] << 8) | c[1];
1858 header.version = 42;
1859 header.ifd0_offset = offsetof(halide_tiff_header, entry_count);
1860 header.entry_count = sizeof(header.entries) / sizeof(header.entries[0]);
1861
1862 static_assert(sizeof(halide_tiff_tag) == 12, "Unexpected halide_tiff_tag packing");
1863 halide_tiff_tag *tag = &header.entries[0];
1864 tag++->assign32(256, 1, width); // ImageWidth
1865 tag++->assign32(257, 1, height); // ImageLength
1866 tag++->assign16(258, 1, int16_t(bytes_per_element * 8)); // BitsPerSample
1867 tag++->assign16(259, 1, 1); // Compression -- none
1868 tag++->assign16(262, 1, channels >= 3 ? 2 : 1); // PhotometricInterpretation -- black is zero or RGB
1869 tag++->assign32(273, channels, sizeof(header)); // StripOffsets
1870 tag++->assign16(277, 1, int16_t(channels)); // SamplesPerPixel
1871 tag++->assign32(278, 1, height); // RowsPerStrip
1872 tag++->assign32(279, channels, // StripByteCounts
1873 (channels == 1) ?
1874 elements * bytes_per_element :
1875 sizeof(header) + channels * sizeof(int32_t)); // for channels > 1, this is an offset
1876 tag++->assign32(282, 5, 1,
1877 offsetof(halide_tiff_header, width_resolution)); // XResolution
1878 tag++->assign32(283, 5, 1,
1879 offsetof(halide_tiff_header, height_resolution)); // YResolution
1880 tag++->assign16(284, 1, 2); // PlanarConfiguration -- planar
1881 tag++->assign16(296, 1, 1); // ResolutionUnit -- none
1882 tag++->assign16(339, 1, type_code_to_tiff_sample_type[im_type.code]); // SampleFormat
1883 tag++->assign32(32997, 1, depth); // Image depth
1884
1885 // Verify we used exactly the number we declared
1886 assert(tag == &header.entries[header.entry_count]);
1887
1888 header.ifd0_end = 0;
1889 header.width_resolution[0] = 1;
1890 header.width_resolution[1] = 1;
1891 header.height_resolution[0] = 1;
1892 header.height_resolution[1] = 1;
1893
1894 if (!check(f.write_bytes(&header, sizeof(header)), "TIFF write failed")) {
1895 return false;
1896 }
1897
1898 if (channels > 1) {
1899 // Fill in the values for StripOffsets
1900 int32_t offset = sizeof(header) + channels * sizeof(int32_t) * 2;
1901 for (int32_t i = 0; i < channels; i++) {
1902 if (!check(f.write_bytes(&offset, sizeof(offset)), "TIFF write failed")) {
1903 return false;
1904 }
1905 offset += width * height * depth * bytes_per_element;
1906 }
1907 // Fill in the values for StripByteCounts
1908 int32_t count = width * height * depth * bytes_per_element;
1909 for (int32_t i = 0; i < channels; i++) {
1910 if (!check(f.write_bytes(&count, sizeof(count)), "TIFF write failed")) {
1911 return false;
1912 }
1913 }
1914 }
1915
1916 // If image is dense, we can write it in one fell swoop
1917 if (elements * bytes_per_element == im.size_in_bytes()) {
1918 if (!check(f.write_bytes(im.data(), im.size_in_bytes()), "TIFF write failed")) {
1919 return false;
1920 }
1921 return true;
1922 }
1923
1924 // Otherwise, write it out via manual traversal.
1925#define HANDLE_CASE(CODE, BITS, TYPE) \
1926 case halide_type_code(CODE, BITS): { \
1927 ElemWriter<TYPE> ew(&f); \
1928 im.template as<const TYPE>().for_each_value(ew); \
1929 if (!check(ew.ok, "TIFF write failed")) { \
1930 return false; \
1931 } \
1932 break; \
1933 }
1934
1935 switch (halide_type_code((halide_type_code_t)im_type.code, im_type.bits)) {
1936 HANDLE_CASE(halide_type_float, 32, float)
1937 HANDLE_CASE(halide_type_float, 64, double)
1938 HANDLE_CASE(halide_type_int, 8, int8_t)
1939 HANDLE_CASE(halide_type_int, 16, int16_t)
1940 HANDLE_CASE(halide_type_int, 32, int32_t)
1941 HANDLE_CASE(halide_type_int, 64, int64_t)
1942 HANDLE_CASE(halide_type_uint, 1, bool)
1943 HANDLE_CASE(halide_type_uint, 8, uint8_t)
1944 HANDLE_CASE(halide_type_uint, 16, uint16_t)
1945 HANDLE_CASE(halide_type_uint, 32, uint32_t)
1946 HANDLE_CASE(halide_type_uint, 64, uint64_t)
1947 // Note that we don't attempt to handle halide_type_handle here.
1948 default:
1949 assert(false && "Unsupported type");
1950 return false;
1951 }
1952#undef HANDLE_CASE
1953
1954 return true;
1955}

2012template<typename ImageType>
2013FormatInfo best_save_format(const ImageType &im, const std::set<FormatInfo> &info) {
2014 // A bit ad hoc, but will do for now:
2015 // Perfect score is zero (exact match).
2016 // The larger the score, the worse the match.
2017 int best_score = 0x7fffffff;
2018 FormatInfo best{};
2019 const halide_type_t im_type = im.type();
2020 const int im_dimensions = im.dimensions();
2021 for (const auto &f : info) {
2022 int score = 0;
2023 // If format has too-few dimensions, that's very bad.
2024 score += std::max(0, im_dimensions - f.dimensions) * 1024;
2025 // If format has too-few bits, that's pretty bad.
2026 score += std::max(0, im_type.bits - f.type.bits) * 8;
2027 // If format has too-many bits, that's a little bad.
2028 score += std::max(0, f.type.bits - im_type.bits);
2029 // If format has different code, that's a little bad.
2030 score += (f.type.code != im_type.code) ? 1 : 0;
2031 if (score < best_score) {
2032 best_score = score;
2033 best = f;
2034 }
2035 }
2036
2037 return best;
2038}
