97template<typename other_t, typename PythonClass>
98void add_binary_operators_with(PythonClass &class_instance) {
99 using self_t = typename PythonClass::type;
100 // If 'other_t' is double, we want to wrap it as an Expr() prior to calling the binary op
101 // (so that double literals that lose precision when converted to float issue warnings).
102 // For any other type, we just want to leave it as-is.
103 using Promote = typename std::conditional<
104 std::is_same<other_t, double>::value,
105 DoubleToExprCheck,
106 other_t>::type;
107
108#define BINARY_OP(op, method) \
109 do { \
110 class_instance.def( \
111 "__" #method "__", \
112 [](const self_t &self, const other_t &other) -> decltype(self op Promote(other)) { \
113 auto result = self op Promote(other); \
114 LOG_PY_BINARY_OP(self, #method, other, result); \
115 return result; \
116 }, \
117 py::is_operator()); \
118 class_instance.def( \
119 "__r" #method "__", \
120 [](const self_t &self, const other_t &other) -> decltype(Promote(other) op self) { \
121 auto result = Promote(other) op self; \
122 LOG_PY_BINARY_OP(self, "r" #method, other, result); \
123 return result; \
124 }, \
125 py::is_operator()); \
126 } while (0)
127
128 BINARY_OP(+, add);
129 BINARY_OP(-, sub);
130 BINARY_OP(*, mul);
131 BINARY_OP(/, div); // TODO: verify only needed for python 2.x (harmless for Python 3.x)
132 BINARY_OP(/, truediv);
133 BINARY_OP(%, mod);
134 BINARY_OP(<<, lshift);
135 BINARY_OP(>>, rshift);
136 BINARY_OP(&, and);
137 BINARY_OP(|, or);
138 BINARY_OP(^, xor);
139 BINARY_OP(<, lt);
140 BINARY_OP(<=, le);
141 BINARY_OP(==, eq);
142 BINARY_OP(!=, ne);
143 BINARY_OP(>=, ge);
144 BINARY_OP(>, gt);
145
146#undef BINARY_OP

148 const auto floordiv_wrap = [](const self_t &self, const other_t &other) -> decltype(self / Promote(other)) {
149 static_assert(std::is_same<decltype(self / Promote(other)), Expr>::value, "We expect all operator// overloads to produce Expr");
150 Expr e = self / Promote(other);
151 if (e.type().is_float()) {
152 e = Halide::floor(e);
153 }
154 return e;
155 };
156
157 class_instance
158 .def("__floordiv__", floordiv_wrap, py::is_operator())
159 .def("__rfloordiv__", floordiv_wrap, py::is_operator());
160} // namespace PythonBindings

162template<typename PythonClass>
163void add_binary_operators(PythonClass &class_instance) {
164 using self_t = typename PythonClass::type;
165
166 // The order of definitions matters.
167 // Python first will try input value as int, then double, then self_t
168 // (note that we skip 'float' because we should never encounter that in python;
169 // all floating-point literals should be double)
170 add_binary_operators_with<self_t>(class_instance);
171 add_binary_operators_with<Expr>(class_instance);
172 add_binary_operators_with<double>(class_instance);
173 add_binary_operators_with<int>(class_instance);
174
175 // Halide::pow() has only an Expr, Expr variant
176 const auto pow_wrap = [](const Expr &self, const Expr &other) -> decltype(Halide::pow(self, other)) {
177 return Halide::pow(self, other);
178 };
179 class_instance
180 .def("__pow__", pow_wrap, py::is_operator())
181 .def("__rpow__", pow_wrap, py::is_operator());
182
183 const auto logical_not_wrap = [](const self_t &self) -> decltype(!self) {
184 return !self;
185 };
186
187 // Define unary operators
188 class_instance
189 .def(-py::self) // neg
190 .def(~py::self) // invert
191 .def("logical_not", logical_not_wrap);
192}
