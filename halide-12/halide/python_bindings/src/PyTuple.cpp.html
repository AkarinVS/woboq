6void define_tuple(py::module &m) {
7 // Halide::Tuple isn't surfaced to the user in Python;
8 // we define it here to allow PyBind to do some automatic
9 // conversion from Python's built-in tuple type.
10 //
11 // TODO: AFAICT, there isn't a good way to auto-convert Halide::Tuple-as-return-type
12 // *TO* a py::tuple; call sites that return Halide::Tuple currently must manually
13 // do this (typically via a C++ lambda that calls to_python_tuple()).
14 auto tuple_class =
15 py::class_<Tuple>(m, "Tuple")
16 // for implicitly_convertible
17 .def(py::init([](const py::tuple &t) -> Tuple {
18 std::vector<Expr> v;
19 v.reserve(t.size());
20 for (const auto o : t) {
21 v.push_back(o.cast<Expr>());
22 }
23 return Tuple(v);
24 }))
25 .def(py::init([](const FuncRef &f) -> Tuple {
26 std::vector<Expr> v;
27 v.reserve(f.size());
28 if (f.size() == 1) {
29 v.push_back(f);
30 } else {
31 for (size_t i = 0; i < f.size(); ++i) {
32 v.push_back(f[(int)i]);
33 }
34 }
35 return Tuple(v);
36 }))
37 .def(py::init([](const std::vector<Expr> &v) -> Tuple {
38 return Tuple(v);
39 }))
40 .def("__repr__", [](const Tuple &t) -> std::string {
41 std::ostringstream o;
42 o << "<halide.Tuple of size " << t.size() << ">";
43 return o.str();
44 });
45 py::implicitly_convertible<py::tuple, Tuple>();
46
47 // If we autoconvert from vector<Expr>, we must also special-case FuncRef, alas
48 py::implicitly_convertible<FuncRef, Tuple>();
49
50 // We want to allow vector<Expr>->Tuple implicitly, so that
51 // custom classes that are tuple-like will autoconvert (see tutorial/lesson_13).
52 py::implicitly_convertible<std::vector<Expr>, Tuple>();
53}
