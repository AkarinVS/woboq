20std::string halide_type_to_string(const Type &type) {
21 std::ostringstream stream;
22 if (type.code() == halide_type_uint && type.bits() == 1) {
23 stream << "bool";
24 } else {
25 switch (type.code()) {
26 case halide_type_int:
27 stream << "int";
28 break;
29 case halide_type_uint:
30 stream << "uint";
31 break;
32 case halide_type_float:
33 stream << "float";
34 break;
35 case halide_type_handle:
36 stream << "handle";
37 break;
38 default:
39 stream << "#unknown";
40 break;
41 }
42 stream << std::to_string(type.bits());
43 }
44 if (type.lanes() > 1) {
45 stream << "x" + std::to_string(type.lanes());
46 }
47 return stream.str();
48}

50void define_type(py::module &m) {
51 py::class_<Type>(m, "Type")
52 .def(py::init<>())
53 .def(py::init<halide_type_code_t, int, int>(), py::arg("code"), py::arg("bits"), py::arg("lanes"))
54 .def("bytes", &Type::bytes)
55 .def("code", &Type::code)
56 .def("bits", &Type::bits)
57 .def("lanes", &Type::lanes)
58 .def("with_code", &Type::with_code, py::arg("code"))
59 .def("with_bits", &Type::with_bits, py::arg("bits"))
60 .def("with_lanes", &Type::with_lanes, py::arg("lanes"))
61
62 .def("is_bool", &Type::is_bool)
63 .def("is_vector", &Type::is_vector)
64 .def("is_scalar", &Type::is_scalar)
65 .def("is_float", &Type::is_float)
66 .def("is_int", &Type::is_int)
67 .def("is_uint", &Type::is_uint)
68 .def("is_handle", &Type::is_handle)
69 .def("same_handle_type", &Type::same_handle_type, py::arg("other"))
70
71 .def("__eq__", [](const Type &value, Type *value2) -> bool { return value2 && value == *value2; })
72 .def("__ne__", [](const Type &value, Type *value2) -> bool { return !value2 || value != *value2; })
73
74 // This is defined in C++ so that Types can live in std::map, but it's not clear that
75 // it's useful for the Python bindings; leave it out for now.
76 // .def("__lt__", [](const Type &value, Type *value2) -> bool { return value2 && value < *value2; })
77
78 .def("element_of", &Type::element_of)
79 .def("can_represent", (bool (Type::*)(Type) const) & Type::can_represent, py::arg("other"))
80 // Python doesn't have unsigned integers -- all integers are signed --
81 // so we'll never see anything that can usefully be routed to the uint64_t
82 // overloads of these methods.
83 .def("is_max", (bool (Type::*)(int64_t) const) & Type::is_max, py::arg("value"))
84 .def("is_min", (bool (Type::*)(int64_t) const) & Type::is_min, py::arg("value"))
85 .def("max", &Type::max)
86 .def("min", &Type::min)
87 .def("__repr__", &type_repr)
88 .def("__str__", &halide_type_to_string);
89
90 m.def("Int", Int, py::arg("bits"), py::arg("lanes") = 1);
91 m.def("UInt", UInt, py::arg("bits"), py::arg("lanes") = 1);
92 m.def("Float", Float, py::arg("bits"), py::arg("lanes") = 1);
93 m.def("Bool", Bool, py::arg("lanes") = 1);
94 m.def("Handle", make_handle, py::arg("lanes") = 1);
95}
