5template<typename Type, int size = 4, int dim = 1>
6Halide::Buffer<Type> make_image(int extra) {
7 Halide::Buffer<Type> im(size, size, dim);
8 for (int x = 0; x < size; x++) {
9 for (int y = 0; y < size; y++) {
10 for (int c = 0; c < dim; c++) {
11 im(x, y, c) = static_cast<Type>(x + y + c + extra);
12 }
13 }
14 }
15 return im;
16}

18class ComplexStub : public Halide::Generator<ComplexStub> {
19public:
20 GeneratorParam<Type> untyped_buffer_output_type{"untyped_buffer_output_type", Float(32)};
21 GeneratorParam<bool> vectorize{"vectorize", true};
22 GeneratorParam<LoopLevel> intermediate_level{"intermediate_level", LoopLevel::root()};
23
24 Input<Buffer<uint8_t>> typed_buffer_input{"typed_buffer_input", 3};
25 Input<Buffer<>> untyped_buffer_input{"untyped_buffer_input"};
26 Input<Func> simple_input{"simple_input", 3}; // require a 3-dimensional Func but leave Type unspecified
27 Input<Func[]> array_input{"array_input", 3}; // require a 3-dimensional Func but leave Type and ArraySize unspecified
28 // Note that Input<Func> does not (yet) support Tuples
29 Input<float> float_arg{"float_arg", 1.0f, 0.0f, 100.0f};
30 Input<int32_t[]> int_arg{"int_arg", 1}; // leave ArraySize unspecified
31
32 Output<Func> simple_output{"simple_output", Float(32), 3};
33 Output<Func> tuple_output{"tuple_output", 3}; // require a 3-dimensional Func but leave Type(s) unspecified
34 Output<Func[]> array_output{"array_output", Int(16), 2}; // leave ArraySize unspecified
35 Output<Buffer<float>> typed_buffer_output{"typed_buffer_output"};
36 Output<Buffer<>> untyped_buffer_output{"untyped_buffer_output"};
37 Output<Buffer<uint8_t>> static_compiled_buffer_output{"static_compiled_buffer_output", 3};
38
39 void configure() {
40 // Pointers returned by add_input() are managed by the Generator;
41 // user code must not free them. We can stash them in member variables
42 // as-is or in containers, like so:
43 extra_func_input = add_input<Func>("extra_func_input", UInt(16), 3);
44 extra_func_output = add_output<Func>("extra_func_output", Float(64), 2);
45 }
46
47 void generate() {
48 simple_output(x, y, c) = cast<float>(simple_input(x, y, c));
49 typed_buffer_output(x, y, c) = cast<float>(typed_buffer_input(x, y, c));
50 // Note that if we are being invoked via a Stub, "untyped_buffer_output.type()" will
51 // assert-fail, because there is no type constraint set: the type
52 // will end up as whatever we infer from the values put into it. We'll use an
53 // explicit GeneratorParam to allow us to set it.
54 untyped_buffer_output(x, y, c) = cast(untyped_buffer_output_type, untyped_buffer_input(x, y, c));
55
56 // Gratuitous intermediate for the purpose of exercising
57 // GeneratorParam<LoopLevel>
58 intermediate(x, y, c) = simple_input(x, y, c) * float_arg;
59
60 tuple_output(x, y, c) = Tuple(
61 intermediate(x, y, c),
62 intermediate(x, y, c) + int_arg[0]);
63
64 array_output.resize(array_input.size());
65 for (size_t i = 0; i < array_input.size(); ++i) {
66 array_output[i](x, y) = cast<int16_t>(array_input[i](x, y, 0) + int_arg[i]);
67 }
68
69 // This should be compiled into the Generator product itself,
70 // and not produce another input for the Stub or AOT filter.
71 Buffer<uint8_t> static_compiled_buffer = make_image<uint8_t>(42);
72 static_compiled_buffer_output = static_compiled_buffer;
73
74 (*extra_func_output)(x, y) = cast<double>((*extra_func_input)(x, y, 0) + 1);
75 }
76
77 void schedule() {
78 intermediate.compute_at(intermediate_level);
79 intermediate.specialize(vectorize).vectorize(x, natural_vector_size<float>());
80 }
81
82private:
83 Var x{"x"}, y{"y"}, c{"c"};
84
85 Func intermediate{"intermediate"};
86
87 Input<Func> *extra_func_input;
88 Output<Func> *extra_func_output;
89};
