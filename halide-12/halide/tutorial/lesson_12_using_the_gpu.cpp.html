34// Define some Vars to use.
35Var x, y, c, i, ii, xo, yo, xi, yi;
36
37// We're going to want to schedule a pipeline in several ways, so we
38// define the pipeline in a class so that we can recreate it several
39// times with different schedules.
40class MyPipeline {
41public:
42 Func lut, padded, padded16, sharpen, curved;
43 Buffer<uint8_t> input;
44
45 MyPipeline(Buffer<uint8_t> in)
46 : input(in) {
47 // For this lesson, we'll use a two-stage pipeline that sharpens
48 // and then applies a look-up-table (LUT).
49
50 // First we'll define the LUT. It will be a gamma curve.
51
52 lut(i) = cast<uint8_t>(clamp(pow(i / 255.0f, 1.2f) * 255.0f, 0, 255));
53
54 // Augment the input with a boundary condition.
55 padded(x, y, c) = input(clamp(x, 0, input.width() - 1),
56 clamp(y, 0, input.height() - 1), c);
57
58 // Cast it to 16-bit to do the math.
59 padded16(x, y, c) = cast<uint16_t>(padded(x, y, c));
60
61 // Next we sharpen it with a five-tap filter.
62 sharpen(x, y, c) = (padded16(x, y, c) * 2 -
63 (padded16(x - 1, y, c) +
64 padded16(x, y - 1, c) +
65 padded16(x + 1, y, c) +
66 padded16(x, y + 1, c)) /
67 4);
68
69 // Then apply the LUT.
70 curved(x, y, c) = lut(sharpen(x, y, c));
71 }
72
73 // Now we define methods that give our pipeline several different
74 // schedules.
75 void schedule_for_cpu() {
76 // Compute the look-up-table ahead of time.
77 lut.compute_root();
78
79 // Compute color channels innermost. Promise that there will
80 // be three of them and unroll across them.
81 curved.reorder(c, x, y)
82 .bound(c, 0, 3)
83 .unroll(c);
84
85 // Look-up-tables don't vectorize well, so just parallelize
86 // curved in slices of 16 scanlines.
87 Var yo, yi;
88 curved.split(y, yo, yi, 16)
89 .parallel(yo);
90
91 // Compute sharpen as needed per scanline of curved.
92 sharpen.compute_at(curved, yi);
93
94 // Vectorize the sharpen. It's 16-bit so we'll vectorize it 8-wide.
95 sharpen.vectorize(x, 8);
96
97 // Compute the padded input as needed per scanline of curved,
98 // reusing previous values computed within the same strip of
99 // 16 scanlines.
100 padded.store_at(curved, yo)
101 .compute_at(curved, yi);
102
103 // Also vectorize the padding. It's 8-bit, so we'll vectorize
104 // 16-wide.
105 padded.vectorize(x, 16);
106
107 // JIT-compile the pipeline for the CPU.
108 Target target = get_host_target();
109 curved.compile_jit(target);
110 }
111
112 // Now a schedule that uses CUDA or OpenCL.
113 bool schedule_for_gpu() {
114 Target target = find_gpu_target();
115 if (!target.has_gpu_feature()) {
116 return false;
117 }
118
119 // If you want to see all of the OpenCL, Metal, CUDA or D3D 12 API
120 // calls done by the pipeline, you can also enable the Debug flag.
121 // This is helpful for figuring out which stages are slow, or when
122 // CPU -> GPU copies happen. It hurts performance though, so we'll
123 // leave it commented out.
124 //target.set_feature(Target::Debug);
125
126 // We make the decision about whether to use the GPU for each
127 // Func independently. If you have one Func computed on the
128 // CPU, and the next computed on the GPU, Halide will do the
129 // copy-to-gpu under the hood. For this pipeline, there's no
130 // reason to use the CPU for any of the stages. Halide will
131 // copy the input image to the GPU the first time we run the
132 // pipeline, and leave it there to reuse on subsequent runs.
133
134 // As before, we'll compute the LUT once at the start of the
135 // pipeline.
136 lut.compute_root();
137
138 // Let's compute the look-up-table using the GPU in 16-wide
139 // one-dimensional thread blocks. First we split the index
140 // into blocks of size 16:
141 Var block, thread;
142 lut.split(i, block, thread, 16);
143 // Then we tell cuda that our Vars 'block' and 'thread'
144 // correspond to CUDA's notions of blocks and threads, or
145 // OpenCL's notions of thread groups and threads.
146 lut.gpu_blocks(block)
147 .gpu_threads(thread);
148
149 // This is a very common scheduling pattern on the GPU, so
150 // there's a shorthand for it:
151
152 // lut.gpu_tile(i, block, thread, 16);
153
154 // Func::gpu_tile behaves the same as Func::tile, except that
155 // it also specifies that the tile coordinates correspond to
156 // GPU blocks, and the coordinates within each tile correspond
157 // to GPU threads.
158
159 // Compute color channels innermost. Promise that there will
160 // be three of them and unroll across them.
161 curved.reorder(c, x, y)
162 .bound(c, 0, 3)
163 .unroll(c);
164
165 // Compute curved in 2D 8x8 tiles using the GPU.
166 curved.gpu_tile(x, y, xo, yo, xi, yi, 8, 8);
167
168 // This is equivalent to:
169 // curved.tile(x, y, xo, yo, xi, yi, 8, 8)
170 // .gpu_blocks(xo, yo)
171 // .gpu_threads(xi, yi);
172
173 // We'll leave sharpen as inlined into curved.
174
175 // Compute the padded input as needed per GPU block, storing
176 // the intermediate result in shared memory. In the schedule
177 // above xo corresponds to GPU blocks.
178 padded.compute_at(curved, xo);
179
180 // Use the GPU threads for the x and y coordinates of the
181 // padded input.
182 padded.gpu_threads(x, y);
183
184 // JIT-compile the pipeline for the GPU. CUDA, OpenCL, or
185 // Metal are not enabled by default. We have to construct a
186 // Target object, enable one of them, and then pass that
187 // target object to compile_jit. Otherwise your CPU will very
188 // slowly pretend it's a GPU, and use one thread per output
189 // pixel.
190 printf("Target: %s\n", target.to_string().c_str());
191 curved.compile_jit(target);
192
193 return true;
194 }

196 void test_performance() {
197 // Test the performance of the scheduled MyPipeline.
198
199 Buffer<uint8_t> output(input.width(), input.height(), input.channels());
200
201 // Run the filter once to initialize any GPU runtime state.
202 curved.realize(output);
203
204 // Now take the best of 3 runs for timing.
205 double best_time = 0.0;
206 for (int i = 0; i < 3; i++) {
207
208 double t1 = current_time();
209
210 // Run the filter 100 times.
211 for (int j = 0; j < 100; j++) {
212 curved.realize(output);
213 }
214
215 // Force any GPU code to finish by copying the buffer back to the CPU.
216 output.copy_to_host();
217
218 double t2 = current_time();
219
220 double elapsed = (t2 - t1) / 100;
221 if (i == 0 || elapsed < best_time) {
222 best_time = elapsed;
223 }
224 }
225
226 printf("%1.4f milliseconds\n", best_time);
227 }
228
229 void test_correctness(Buffer<uint8_t> reference_output) {
230 Buffer<uint8_t> output =
231 curved.realize({input.width(), input.height(), input.channels()});
232
233 // Check against the reference output.
234 for (int c = 0; c < input.channels(); c++) {
235 for (int y = 0; y < input.height(); y++) {
236 for (int x = 0; x < input.width(); x++) {
237 if (output(x, y, c) != reference_output(x, y, c)) {
238 printf("Mismatch between output (%d) and "
239 "reference output (%d) at %d, %d, %d\n",
240 output(x, y, c),
241 reference_output(x, y, c),
242 x, y, c);
243 exit(-1);
244 }
245 }
246 }
247 }
248 }

251int main(int argc, char **argv) {
252 // Load an input image.
253 Buffer<uint8_t> input = load_image("images/rgb.png");
254
255 // Allocated an image that will store the correct output
256 Buffer<uint8_t> reference_output(input.width(), input.height(), input.channels());
257
258 printf("Running pipeline on CPU:\n");
259 MyPipeline p1(input);
260 p1.schedule_for_cpu();
261 p1.curved.realize(reference_output);
262
263 printf("Running pipeline on GPU:\n");
264 MyPipeline p2(input);
265 bool has_gpu_target = p2.schedule_for_gpu();
266 if (has_gpu_target) {
267 printf("Testing GPU correctness:\n");
268 p2.test_correctness(reference_output);
269 } else {
270 printf("No GPU target available on the host\n");
271 }
272
273 printf("Testing performance on CPU:\n");
274 p1.test_performance();
275
276 if (has_gpu_target) {
277 printf("Testing performance on GPU:\n");
278 p2.test_performance();
279 }
280
281 return 0;
282}

284// A helper function to check if OpenCL, Metal or D3D12 is present on the host machine.
285
286Target find_gpu_target() {
287 // Start with a target suitable for the machine you're running this on.
288 Target target = get_host_target();
289
290 std::vector<Target::Feature> features_to_try;
291 if (target.os == Target::Windows) {
292 // Try D3D12 first; if that fails, try OpenCL.
293 if (sizeof(void*) == 8) {
294 // D3D12Compute support is only available on 64-bit systems at present.
295 features_to_try.push_back(Target::D3D12Compute);
296 }
297 features_to_try.push_back(Target::OpenCL);
298 } else if (target.os == Target::OSX) {
299 // OS X doesn't update its OpenCL drivers, so they tend to be broken.
300 // CUDA would also be a fine choice on machines with NVidia GPUs.
301 features_to_try.push_back(Target::Metal);
302 } else {
303 features_to_try.push_back(Target::OpenCL);
304 }
305 // Uncomment the following lines to also try CUDA:
306 // features_to_try.push_back(Target::CUDA);
307
308 for (Target::Feature f : features_to_try) {
309 Target new_target = target.with_feature(f);
310 if (host_supports_target_device(new_target)) {
311 return new_target;
312 }
313 }
314
315 printf("Requested GPU(s) are not supported. (Do you have the proper hardware and/or driver installed?)\n");
316 return target;
317}
