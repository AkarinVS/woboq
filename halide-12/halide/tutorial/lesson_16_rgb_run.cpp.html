1// Halide tutorial lesson 16: RGB images and memory layouts part 2
2
3// Before reading this file, see lesson_16_rgb_generate.cpp

5// This is the code that actually uses the Halide pipeline we've
6// compiled. It does not depend on libHalide, so we won't be including
7// Halide.h.
8//
9// Instead, it depends on the header files that lesson_16_rgb_generator produced.
10#include "brighten_either.h"
11#include "brighten_interleaved.h"
12#include "brighten_planar.h"
13#include "brighten_specialized.h"
14
15// We'll use the Halide::Runtime::Buffer class for passing data into and out of
16// the pipeline.
17#include "HalideBuffer.h"
18
19#include <assert.h>
20#include <stdio.h>
21#include <stdlib.h>
22#include <string.h>
23
24#include "halide_benchmark.h"
25
26void check_timing(double faster, double slower) {
27 if (faster > slower) {
28 fprintf(stderr, "Warning: performance was worse than expected. %f should be less than %f\n", faster, slower);
29 }
30}
31
32int main(int argc, char **argv) {
33
34 // Let's make some images stored with interleaved and planar
35 // memory. Halide::Runtime::Buffer is planar by default.
36 Halide::Runtime::Buffer<uint8_t> planar_input(1024, 768, 3);
37 Halide::Runtime::Buffer<uint8_t> planar_output(1024, 768, 3);
38 Halide::Runtime::Buffer<uint8_t> interleaved_input =
39 Halide::Runtime::Buffer<uint8_t>::make_interleaved(1024, 768, 3);
40 Halide::Runtime::Buffer<uint8_t> interleaved_output =
41 Halide::Runtime::Buffer<uint8_t>::make_interleaved(1024, 768, 3);
42
43 // Let's check the strides are what we expect, given the
44 // constraints we set up in the generator.
45 assert(planar_input.dim(0).stride() == 1);
46 assert(planar_output.dim(0).stride() == 1);
47 assert(interleaved_input.dim(0).stride() == 3);
48 assert(interleaved_output.dim(0).stride() == 3);
49 assert(interleaved_input.dim(2).stride() == 1);
50 assert(interleaved_output.dim(2).stride() == 1);
51
52 // We'll now call the various functions we compiled and check the
53 // performance of each.
54
55 constexpr int samples = 1;
56 constexpr int iterations = 1000;
57
58 // Run the planar version of the code on the planar images and the
59 // interleaved version of the code on the interleaved
60 // images. We'll use Halide's benchmarking utility, which takes a function
61 // to run, the number of batches to run (1 in this case), and the number
62 // of iterations per batch (1000 in this case). It returns the best
63 // average-iteration time, in seconds. (See halide_benchmark.h for more
64 // information.)
65
66 double planar_time = Halide::Tools::benchmark(samples, iterations, [&]() {
67 brighten_planar(planar_input, 1, planar_output);
68 });
69 printf("brighten_planar: %f msec\n", planar_time * 1000.f);
70
71 double interleaved_time = Halide::Tools::benchmark(samples, iterations, [&]() {
72 brighten_interleaved(interleaved_input, 1, interleaved_output);
73 });
74 printf("brighten_interleaved: %f msec\n", interleaved_time * 1000.f);
75
76 // Planar is generally faster than interleaved for most imaging
77 // operations.
78 check_timing(planar_time, interleaved_time);
79
80 // Either of these next two commented-out calls would throw an
81 // error, because the stride is not what we promised it would be
82 // in the generator.
83
84 // brighten_planar(interleaved_input, 1, interleaved_output);
85 // Error: Constraint violated: brighter.stride.0 (3) == 1 (1)
86
87 // brighten_interleaved(planar_input, 1, planar_output);
88 // Error: Constraint violated: brighter.stride.0 (1) == 3 (3)
89
90 // Run the flexible version of the code and check performance. It
91 // should work, but it'll be slower than the versions above.
92 double either_planar_time = Halide::Tools::benchmark(samples, iterations, [&]() {
93 brighten_either(planar_input, 1, planar_output);
94 });
95 printf("brighten_either on planar images: %f msec\n", either_planar_time * 1000.f);
96 check_timing(planar_time, either_planar_time);
97
98 double either_interleaved_time = Halide::Tools::benchmark(samples, iterations, [&]() {
99 brighten_either(interleaved_input, 1, interleaved_output);
100 });
101 printf("brighten_either on interleaved images: %f msec\n", either_interleaved_time * 1000.f);
102 check_timing(interleaved_time, either_interleaved_time);
103
104 // Run the specialized version of the code on each layout. It
105 // should match the performance of the code compiled specifically
106 // for each case above by branching internally to equivalent
107 // code.
108 double specialized_planar_time = Halide::Tools::benchmark(samples, iterations, [&]() {
109 brighten_specialized(planar_input, 1, planar_output);
110 });
111 printf("brighten_specialized on planar images: %f msec\n", specialized_planar_time * 1000.f);
112
113 // The cost of the if statement should be negligible, but we'll
114 // allow a tolerance of 50% for this test to account for
115 // measurement noise.
116 check_timing(specialized_planar_time, 1.5 * planar_time);
117
118 double specialized_interleaved_time = Halide::Tools::benchmark(samples, iterations, [&]() {
119 brighten_specialized(interleaved_input, 1, interleaved_output);
120 });
121 printf("brighten_specialized on interleaved images: %f msec\n", specialized_interleaved_time * 1000.f);
122 check_timing(specialized_interleaved_time, 2.0 * interleaved_time);
123
124 return 0;
125}
