1// Halide tutorial lesson 16: RGB images and memory layouts part 1
2
3// This lesson demonstrates how to feed Halide RGB images in
4// interleaved or planar format, and how to write code optimized for
5// each case.
6
7// On linux or os x, you can compile and run it like so:

25#include "Halide.h"
26#include <stdio.h>
27
28using namespace Halide;
29
30// We will define a generator that brightens an RGB image.
31class Brighten : public Halide::Generator<Brighten> {
32public:
33 // We declare a three-dimensional input image. The first two
34 // dimensions will be x, and y, and the third dimension will be
35 // the color channel.
36 Input<Buffer<uint8_t>> input{"input", 3};
37
38 // We will compile this generator in several ways to accept
39 // several different memory layouts for the input and output. This
40 // is a good use of a GeneratorParam (see lesson 15).
41 enum class Layout { Planar,
42 Interleaved,
43 Either,
44 Specialized };
45 GeneratorParam<Layout> layout{"layout",
46 // default value
47 Layout::Planar,
48 // map from names to values
49 {{"planar", Layout::Planar},
50 {"interleaved", Layout::Interleaved},
51 {"either", Layout::Either},
52 {"specialized", Layout::Specialized}}};
53
54 // We also declare a scalar input to control the amount of
55 // brightening.
56 Input<uint8_t> offset{"offset"};
57
58 // Declare our outputs
59 Output<Buffer<uint8_t>> brighter{"brighter", 3};
60
61 // Declare our Vars
62 Var x, y, c;
63
64 void generate() {
65 // Define the Func.
66 brighter(x, y, c) = input(x, y, c) + offset;
67
68 // Schedule it.
69 brighter.vectorize(x, 16);
70
71 // We will compile this pipeline to handle memory layouts in
72 // several different ways, depending on the 'layout' generator
73 // param.
74 if (layout == Layout::Planar) {
75 // This pipeline as written will only work with images in
76 // which each scanline is densely-packed single color
77 // channel. In terms of the strides described in lesson
78 // 10, Halide assumes and asserts that the stride in x is
79 // one.
80
81 // This constraint permits planar images, where the red,
82 // green, and blue channels are laid out in memory like
83 // this:
84
85 // RRRRRRRR
86 // RRRRRRRR
87 // RRRRRRRR
88 // RRRRRRRR
89 // GGGGGGGG
90 // GGGGGGGG
91 // GGGGGGGG
92 // GGGGGGGG
93 // BBBBBBBB
94 // BBBBBBBB
95 // BBBBBBBB
96 // BBBBBBBB
97
98 // It also works with the less-commonly used line-by-line
99 // layout, in which scanlines of red, green, and blue
100 // alternate.
101
102 // RRRRRRRR
103 // GGGGGGGG
104 // BBBBBBBB
105 // RRRRRRRR
106 // GGGGGGGG
107 // BBBBBBBB
108 // RRRRRRRR
109 // GGGGGGGG
110 // BBBBBBBB
111 // RRRRRRRR
112 // GGGGGGGG
113 // BBBBBBBB
114
115 } else if (layout == Layout::Interleaved) {
116 // Another common format is 'interleaved', in which the
117 // red, green, and blue values for each pixel occur next
118 // to each other in memory:
119
120 // RGBRGBRGBRGBRGBRGBRGBRGB
121 // RGBRGBRGBRGBRGBRGBRGBRGB
122 // RGBRGBRGBRGBRGBRGBRGBRGB
123 // RGBRGBRGBRGBRGBRGBRGBRGB
124
125 // In this case the stride in x is three, the stride in y
126 // is three times the width of the image, and the stride
127 // in c is one. We can tell Halide to assume (and assert)
128 // that this is the case for the input and output like so:
129
130 input.dim(0).set_stride(3); // stride in dimension 0 (x) is three
131 input.dim(2).set_stride(1); // stride in dimension 2 (c) is one
132
133 brighter.dim(0).set_stride(3);
134 brighter.dim(2).set_stride(1);
135
136 // For interleaved layout, you may want to use a different
137 // schedule. We'll tell Halide to additionally assume and
138 // assert that there are three color channels, then
139 // exploit this fact to make the loop over 'c' innermost
140 // and unrolled.
141
142 input.dim(2).set_bounds(0, 3); // Dimension 2 (c) starts at 0 and has extent 3.
143 brighter.dim(2).set_bounds(0, 3);
144
145 // Move the loop over color channels innermost and unroll
146 // it.
147 brighter.reorder(c, x, y).unroll(c);
148
149 // Note that if we were dealing with an image with an
150 // alpha channel (RGBA), then the stride in x and the
151 // bounds of the channels dimension would both be four
152 // instead of three.
153
154 } else if (layout == Layout::Either) {
155 // We can also remove all constraints and compile a
156 // pipeline that will work with any memory layout. It will
157 // probably be slow, because all vector loads become
158 // gathers, and all vector stores become scatters.
159 input.dim(0).set_stride(Expr()); // Use a default-constructed
160 // undefined Expr to mean
161 // there is no constraint.
162
163 brighter.dim(0).set_stride(Expr());
164
165 } else if (layout == Layout::Specialized) {
166 // We can accept any memory layout with good performance
167 // by telling Halide to inspect the memory layout at
168 // runtime, and branch to different code depending on the
169 // strides it find. First we relax the default constraint
170 // that dim(0).stride() == 1:
171
172 input.dim(0).set_stride(Expr()); // Use an undefined Expr to
173 // mean there is no
174 // constraint.
175
176 brighter.dim(0).set_stride(Expr());
177
178 // The we construct boolean Exprs that detect at runtime
179 // whether we're planar or interleaved. The conditions
180 // should check for all the facts we want to exploit in
181 // each case.
182 Expr input_is_planar =
183 (input.dim(0).stride() == 1);
184 Expr input_is_interleaved =
185 (input.dim(0).stride() == 3 &&
186 input.dim(2).stride() == 1 &&
187 input.dim(2).extent() == 3);
188
189 Expr output_is_planar =
190 (brighter.dim(0).stride() == 1);
191 Expr output_is_interleaved =
192 (brighter.dim(0).stride() == 3 &&
193 brighter.dim(2).stride() == 1 &&
194 brighter.dim(2).extent() == 3);
195
196 // We can then use Func::specialize to write a schedule
197 // that switches at runtime to specialized code based on a
198 // boolean Expr. That code will exploit the fact that the
199 // Expr is known to be true.
200 brighter.specialize(input_is_planar && output_is_planar);
201
202 // We've already vectorized and parallelized brighter, and
203 // our two specializations will inherit those scheduling
204 // directives. We can also add additional scheduling
205 // directives that apply to a single specialization
206 // only. We'll tell Halide to make a specialized version
207 // of the code for interleaved layouts, and to reorder and
208 // unroll that specialized code.
209 brighter.specialize(input_is_interleaved && output_is_interleaved)
210 .reorder(c, x, y)
211 .unroll(c);
212
213 // We could also add specializations for if the input is
214 // interleaved and the output is planar, and vice versa,
215 // but two specializations is enough to demonstrate the
216 // feature. A later tutorial will explore more creative
217 // uses of Func::specialize.
218
219 // Adding specializations can improve performance
220 // substantially for the cases they apply to, but it also
221 // increases the amount of code to compile and ship. If
222 // binary sizes are a concern and the input and output
223 // memory layouts are known, you probably want to use
224 // set_stride and set_extent instead.
225 }
226 }
227};
228
229// As in lesson 15, we register our generator and then compile this
230// file along with tools/GenGen.cpp.
231HALIDE_REGISTER_GENERATOR(Brighten, brighten)
