1// Halide tutorial lesson 21: Auto-Scheduler
2
3// So far we have written Halide schedules by hand, but it is also possible to
4// ask Halide to suggest a reasonable schedule. We call this auto-scheduling.
5// This lesson demonstrates how to use the auto-scheduler to generate a
6// copy-pasteable CPU schedule that can be subsequently improved upon.
7

24#include "Halide.h"
25#include <stdio.h>
26
27using namespace Halide;
28
29// We will define a generator to auto-schedule.
30class AutoScheduled : public Halide::Generator<AutoScheduled> {
31public:
32 Input<Buffer<float>> input{"input", 3};
33 Input<float> factor{"factor"};
34
35 Output<Buffer<float>> output1{"output1", 2};
36 Output<Buffer<float>> output2{"output2", 2};
37
38 Expr sum3x3(Func f, Var x, Var y) {
39 return f(x - 1, y - 1) + f(x - 1, y) + f(x - 1, y + 1) +
40 f(x, y - 1) + f(x, y) + f(x, y + 1) +
41 f(x + 1, y - 1) + f(x + 1, y) + f(x + 1, y + 1);
42 }
43
44 void generate() {
45 // For our algorithm, we'll use Harris corner detection.
46 Func in_b = BoundaryConditions::repeat_edge(input);
47
48 gray(x, y) = 0.299f * in_b(x, y, 0) + 0.587f * in_b(x, y, 1) + 0.114f * in_b(x, y, 2);
49
50 Iy(x, y) = gray(x - 1, y - 1) * (-1.0f / 12) + gray(x - 1, y + 1) * (1.0f / 12) +
51 gray(x, y - 1) * (-2.0f / 12) + gray(x, y + 1) * (2.0f / 12) +
52 gray(x + 1, y - 1) * (-1.0f / 12) + gray(x + 1, y + 1) * (1.0f / 12);
53
54 Ix(x, y) = gray(x - 1, y - 1) * (-1.0f / 12) + gray(x + 1, y - 1) * (1.0f / 12) +
55 gray(x - 1, y) * (-2.0f / 12) + gray(x + 1, y) * (2.0f / 12) +
56 gray(x - 1, y + 1) * (-1.0f / 12) + gray(x + 1, y + 1) * (1.0f / 12);
57
58 Ixx(x, y) = Ix(x, y) * Ix(x, y);
59 Iyy(x, y) = Iy(x, y) * Iy(x, y);
60 Ixy(x, y) = Ix(x, y) * Iy(x, y);
61 Sxx(x, y) = sum3x3(Ixx, x, y);
62 Syy(x, y) = sum3x3(Iyy, x, y);
63 Sxy(x, y) = sum3x3(Ixy, x, y);
64 det(x, y) = Sxx(x, y) * Syy(x, y) - Sxy(x, y) * Sxy(x, y);
65 trace(x, y) = Sxx(x, y) + Syy(x, y);
66 harris(x, y) = det(x, y) - 0.04f * trace(x, y) * trace(x, y);
67 output1(x, y) = harris(x + 2, y + 2);
68 output2(x, y) = factor * harris(x + 2, y + 2);
69 }
70
71 void schedule() {
72 if (auto_schedule) {
73 // The auto-scheduler requires estimates on all the input/output
74 // sizes and parameter values in order to compare different
75 // alternatives and decide on a good schedule.
76
77 // To provide estimates (min and extent values) for each dimension
78 // of the input images ('input', 'filter', and 'bias'), we use the
79 // set_estimates() method. set_estimates() takes in a list of
80 // (min, extent) of the corresponding dimension as arguments.
81 input.set_estimates({{0, 1024}, {0, 1024}, {0, 3}});
82
83 // To provide estimates on the parameter values, we use the
84 // set_estimate() method.
85 factor.set_estimate(2.0f);
86
87 // To provide estimates (min and extent values) for each dimension
88 // of pipeline outputs, we use the set_estimates() method. set_estimates()
89 // takes in a list of (min, extent) for each dimension.
90 output1.set_estimates({{0, 1024}, {0, 1024}});
91 output2.set_estimates({{0, 1024}, {0, 1024}});
92
93 // Technically, the estimate values can be anything, but the closer
94 // they are to the actual use-case values, the better the generated
95 // schedule will be.
96
97 // To auto-schedule the pipeline, we don't have to do anything else:
98 // every Generator implicitly has a GeneratorParam named "auto_schedule";
99 // if this is set to true, Halide will call auto_schedule() on all of
100 // our pipeline's outputs automatically.
101
102 // Every Generator also implicitly has a GeneratorParams named "machine_params",
103 // which allows you to specify characteristics of the machine architecture
104 // for the auto-scheduler; it's generally specified in your Makefile.
105 // If none is specified, the default machine parameters for a generic CPU
106 // architecture will be used by the auto-scheduler.
107
108 // Let's see some arbitrary but plausible values for the machine parameters.
109 //
110 // const int kParallelism = 32;
111 // const int kLastLevelCacheSize = 16 * 1024 * 1024;
112 // const int kBalance = 40;
113 // MachineParams machine_params(kParallelism, kLastLevelCacheSize, kBalance);
114 //
115 // The arguments to MachineParams are the maximum level of parallelism
116 // available, the size of the last-level cache (in KB), and the ratio
117 // between the cost of a miss at the last level cache and the cost
118 // of arithmetic on the target architecture, in that order.
119
120 // Note that when using the auto-scheduler, no schedule should have
121 // been applied to the pipeline; otherwise, the auto-scheduler will
122 // throw an error. The current auto-scheduler cannot handle a
123 // partially-scheduled pipeline.
124
125 // If HL_DEBUG_CODEGEN is set to 3 or greater, the schedule will be dumped
126 // to stdout (along with much other information); a more useful way is
127 // to add "schedule" to the -e flag to the Generator. (In CMake and Bazel,
128 // this is done using the "extra_outputs" flag.)
129
130 // The generated schedule that is dumped to file is an actual
131 // Halide C++ source, which is readily copy-pasteable back into
132 // this very same source file with few modifications. Programmers
133 // can use this as a starting schedule and iteratively improve the
134 // schedule. Note that the current auto-scheduler is only able to
135 // generate CPU schedules and only does tiling, simple vectorization
136 // and parallelization. It doesn't deal with line buffering, storage
137 // reordering, or factoring reductions.
138
139 // At the time of writing, the auto-scheduler will produce the
140 // following schedule for the estimates and machine parameters
141 // declared above when run on this pipeline:
142 //
143 // Var x_i("x_i");
144 // Var x_i_vi("x_i_vi");
145 // Var x_i_vo("x_i_vo");
146 // Var x_o("x_o");
147 // Var x_vi("x_vi");
148 // Var x_vo("x_vo");
149 // Var y_i("y_i");
150 // Var y_o("y_o");
151 //
152 // Func Ix = pipeline.get_func(4);
153 // Func Iy = pipeline.get_func(7);
154 // Func gray = pipeline.get_func(3);
155 // Func harris = pipeline.get_func(14);
156 // Func output1 = pipeline.get_func(15);
157 // Func output2 = pipeline.get_func(16);
158 //
159 // {
160 // Var x = Ix.args()[0];
161 // Ix
162 // .compute_at(harris, x_o)
163 // .split(x, x_vo, x_vi, 8)
164 // .vectorize(x_vi);
165 // }
166 // {
167 // Var x = Iy.args()[0];
168 // Iy
169 // .compute_at(harris, x_o)
170 // .split(x, x_vo, x_vi, 8)
171 // .vectorize(x_vi);
172 // }
173 // {
174 // Var x = gray.args()[0];
175 // gray
176 // .compute_at(harris, x_o)
177 // .split(x, x_vo, x_vi, 8)
178 // .vectorize(x_vi);
179 // }
180 // {
181 // Var x = harris.args()[0];
182 // Var y = harris.args()[1];
183 // harris
184 // .compute_root()
185 // .split(x, x_o, x_i, 256)
186 // .split(y, y_o, y_i, 128)
187 // .reorder(x_i, y_i, x_o, y_o)
188 // .split(x_i, x_i_vo, x_i_vi, 8)
189 // .vectorize(x_i_vi)
190 // .parallel(y_o)
191 // .parallel(x_o);
192 // }
193 // {
194 // Var x = output1.args()[0];
195 // Var y = output1.args()[1];
196 // output1
197 // .compute_root()
198 // .split(x, x_vo, x_vi, 8)
199 // .vectorize(x_vi)
200 // .parallel(y);
201 // }
202 // {
203 // Var x = output2.args()[0];
204 // Var y = output2.args()[1];
205 // output2
206 // .compute_root()
207 // .split(x, x_vo, x_vi, 8)
208 // .vectorize(x_vi)
209 // .parallel(y);
210 // }
211
212 } else {
213 // This is where you would declare the schedule you have written by
214 // hand or paste the schedule generated by the auto-scheduler.
215 // We will use a naive schedule here to compare the performance of
216 // the autoschedule with a basic schedule.
217 gray.compute_root();
218 Iy.compute_root();
219 Ix.compute_root();
220 }
221 }
222
223private:
224 Var x{"x"}, y{"y"}, c{"c"};
225 Func gray, Iy, Ix, Ixx, Iyy, Ixy, Sxx, Syy, Sxy, det, trace, harris;
226};
