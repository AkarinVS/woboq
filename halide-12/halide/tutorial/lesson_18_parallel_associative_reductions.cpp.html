1// Halide tutorial lesson 18: Factoring an associative reduction using rfactor
2
3// This lesson demonstrates how to parallelize or vectorize an associative
4// reduction using the scheduling directive 'rfactor'.
5
6// On linux, you can compile and run it like so:
7// g++ lesson_18*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -lpthread -ldl -o lesson_18 -std=c++11
8// LD_LIBRARY_PATH=<path/to/libHalide.so> ./lesson_18
9
10// On os x:
11// g++ lesson_18*.cpp -g -I <path/to/Halide.h> -L <path/to/libHalide.so> -lHalide -o lesson_18 -std=c++11
12// DYLD_LIBRARY_PATH=<path/to/libHalide.dylib> ./lesson_18
13
14// If you have the entire Halide source tree, you can also build it by
15// running:
16// make tutorial_lesson_18_parallel_associative_reductions
17// in a shell with the current directory at the top of the halide
18// source tree.
19
20#include "Halide.h"
21#include <stdio.h>
22
23using namespace Halide;
24
25int main(int argc, char **argv) {
26 // Declare some Vars to use below.
27 Var x("x"), y("y"), i("i"), u("u"), v("v");
28
29 // Create an input with random values.
30 Buffer<uint8_t> input(8, 8, "input");
31 for (int y = 0; y < 8; ++y) {
32 for (int x = 0; x < 8; ++x) {
33 input(x, y) = (rand() % 256);
34 }
35 }
36
37 {
38 // As mentioned previously in lesson 9, parallelizing variables that
39 // are part of a reduction domain is tricky, since there may be data
40 // dependencies across those variables.
41
42 // Consider the histogram example in lesson 9:
43 Func histogram("hist_serial");
44 histogram(i) = 0;
45 RDom r(0, input.width(), 0, input.height());
46 histogram(input(r.x, r.y) / 32) += 1;
47
48 histogram.vectorize(i, 8);
49 histogram.realize({8});
50
51 // See figures/lesson_18_hist_serial.mp4 for a visualization of
52 // what this does.
53
54 // We can vectorize the initialization of the histogram
55 // buckets, but since there are data dependencies across r.x
56 // and r.y in the update definition (i.e. the update refers to
57 // value computed in the previous iteration), we can't
58 // parallelize or vectorize r.x or r.y without introducing a
59 // race condition. The following code would produce an error:
60 // histogram.update().parallel(r.y);
61 }
62
63 {
64 // Note, however, that the histogram operation (which is a
65 // kind of sum reduction) is associative. A common trick to
66 // speed-up associative reductions is to slice up the
67 // reduction domain into smaller slices, compute a partial
68 // result over each slice, and then merge the results. Since
69 // the computation of each slice is independent, we can
70 // parallelize over slices.
71
72 // Going back to the histogram example, we slice the reduction
73 // domain into rows by defining an intermediate function that
74 // computes the histogram of each row independently:
75 Func intermediate("intm_par_manual");
76 intermediate(i, y) = 0;
77 RDom rx(0, input.width());
78 intermediate(input(rx, y) / 32, y) += 1;
79
80 // We then define a second stage which sums those partial
81 // results:
82 Func histogram("merge_par_manual");
83 histogram(i) = 0;
84 RDom ry(0, input.height());
85 histogram(i) += intermediate(i, ry);
86
87 // Since the intermediate no longer has data dependencies
88 // across the y dimension, we can parallelize it over y:
89 intermediate.compute_root().update().parallel(y);
90
91 // We can also vectorize the initializations.
92 intermediate.vectorize(i, 8);
93 histogram.vectorize(i, 8);
94
95 histogram.realize({8});
96
97 // See figures/lesson_18_hist_manual_par.mp4 for a visualization of
98 // what this does.
99 }
100
101 {
102 // This manual factorization of an associative reduction can
103 // be tedious and bug-prone. Although it's fairly easy to do
104 // manually for the histogram, it can get complex pretty fast,
105 // especially if the RDom may has a predicate (RDom::where),
106 // or when the function reduces onto a multi-dimensional
107 // tuple.
108
109 // Halide provides a way to do this type of factorization
110 // through the scheduling directive 'rfactor'. rfactor splits
111 // an associative update definition into an intermediate which
112 // computes the partial results over slices of a reduction
113 // domain and replaces the current update definition with a
114 // new definition which merges those partial results.
115
116 // Using rfactor, we don't need to change the algorithm at all:
117 Func histogram("hist_rfactor_par");
118 histogram(x) = 0;
119 RDom r(0, input.width(), 0, input.height());
120 histogram(input(r.x, r.y) / 32) += 1;
121
122 // The task of factoring of associative reduction is moved
123 // into the schedule, via rfactor. rfactor takes as input a
124 // list of <RVar, Var> pairs, which contains list of reduction
125 // variables (RVars) to be made "parallelizable". In the
126 // generated intermediate Func, all references to this
127 // reduction variables are replaced with references to "pure"
128 // variables (the Vars). Since, by construction, Vars are
129 // race-condition free, the intermediate reduction is now
130 // parallelizable across those dimensions. All reduction
131 // variables not in the list are removed from the original
132 // function and "lifted" to the intermediate.
133
134 // To generate the same code as the manually-factored version,
135 // we do the following:
136 Func intermediate = histogram.update().rfactor({{r.y, y}});
137 // We pass {r.y, y} as the argument to rfactor to make the
138 // histogram parallelizable across the y dimension, similar to
139 // the manually-factored version.
140 intermediate.compute_root().update().parallel(y);
141
142 // In the case where you are only slicing up the domain across
143 // a single variable, you can actually drop the braces and
144 // write the rfactor the following way.
145 // Func intermediate = histogram.update().rfactor(r.y, y);
146
147 // Vectorize the initializations, as we did above.
148 intermediate.vectorize(x, 8);
149 histogram.vectorize(x, 8);
150
151 // It is important to note that rfactor (or reduction
152 // factorization in general) only works for associative
153 // reductions. Associative reductions have the nice property
154 // that their results are the same no matter how the
155 // computation is grouped (i.e. split into chunks). If rfactor
156 // can't prove the associativity of a reduction, it will throw
157 // an error.
158
159 Buffer<int> halide_result = histogram.realize({8});
160
161 // See figures/lesson_18_hist_rfactor_par.mp4 for a
162 // visualization of what this does.
163
164 // The equivalent C is:
165 int c_intm[8][8];
166 for (int y = 0; y < input.height(); y++) {
167 for (int x = 0; x < 8; x++) {
168 c_intm[y][x] = 0;
169 }
170 }
171 /* parallel */ for (int y = 0; y < input.height(); y++) {
172 for (int r_x = 0; r_x < input.width(); r_x++) {
173 c_intm[y][input(r_x, y) / 32] += 1;
174 }
175 }
176
177 int c_result[8];
178 for (int x = 0; x < 8; x++) {
179 c_result[x] = 0;
180 }
181 for (int x = 0; x < 8; x++) {
182 for (int r_y = 0; r_y < input.height(); r_y++) {
183 c_result[x] += c_intm[r_y][x];
184 }
185 }
186
187 // Check the answers agree:
188 for (int x = 0; x < 8; x++) {
189 if (c_result[x] != halide_result(x)) {
190 printf("halide_result(%d) = %d instead of %d\n",
191 x, halide_result(x), c_result[x]);
192 return -1;
193 }
194 }
195 }
196
197 {
198 // Now that we can factor associative reductions with the
199 // scheduling directive 'rfactor', we can explore various
200 // factorization strategies using the schedule alone. Given
201 // the same serial histogram code:
202 Func histogram("hist_rfactor_vec");
203 histogram(x) = 0;
204 RDom r(0, input.width(), 0, input.height());
205 histogram(input(r.x, r.y) / 32) += 1;
206
207 // Instead of r.y, we rfactor on r.x this time to slice the
208 // domain into columns.
209 Func intermediate = histogram.update().rfactor(r.x, u);
210
211 // Now that we're computing an independent histogram
212 // per-column, we can vectorize over columns.
213 intermediate.compute_root().update().vectorize(u, 8);
214
215 // Note that since vectorizing the inner dimension changes the
216 // order in which values are added to the final histogram
217 // buckets computations, so this trick only works if the
218 // associative reduction is associative *and*
219 // commutative. rfactor will attempt to prove these properties
220 // hold and will throw an error if it can't.
221
222 // Vectorize the initializations.
223 intermediate.vectorize(x, 8);
224 histogram.vectorize(x, 8);
225
226 Buffer<int> halide_result = histogram.realize({8});
227
228 // See figures/lesson_18_hist_rfactor_vec.mp4 for a
229 // visualization of what this does.
230
231 // The equivalent C is:
232 int c_intm[8][8];
233 for (int u = 0; u < input.width(); u++) {
234 for (int x = 0; x < 8; x++) {
235 c_intm[u][x] = 0;
236 }
237 }
238 for (int r_y = 0; r_y < input.height(); r_y++) {
239 for (int u = 0; u < input.width() / 8; u++) {
240 /* vectorize */ for (int u_i = 0; u_i < 8; u_i++) {
241 c_intm[u * 8 + u_i][input(u * 8 + u_i, r_y) / 32] += 1;
242 }
243 }
244 }
245
246 int c_result[8];
247 for (int x = 0; x < 8; x++) {
248 c_result[x] = 0;
249 }
250 for (int x = 0; x < 8; x++) {
251 for (int r_x = 0; r_x < input.width(); r_x++) {
252 c_result[x] += c_intm[r_x][x];
253 }
254 }
255
256 // Check the answers agree:
257 for (int x = 0; x < 8; x++) {
258 if (c_result[x] != halide_result(x)) {
259 printf("halide_result(%d) = %d instead of %d\n",
260 x, halide_result(x), c_result[x]);
261 return -1;
262 }
263 }
264 }
265
266 {
267 // We can also slice a reduction domain up over multiple
268 // dimensions at once. This time, we'll compute partial
269 // histograms over tiles of the domain.
270 Func histogram("hist_rfactor_tile");
271 histogram(x) = 0;
272 RDom r(0, input.width(), 0, input.height());
273 histogram(input(r.x, r.y) / 32) += 1;
274
275 // We first split both r.x and r.y by a factor of four.
276 RVar rx_outer("rx_outer"), rx_inner("rx_inner");
277 RVar ry_outer("ry_outer"), ry_inner("ry_inner");
278 histogram.update()
279 .split(r.x, rx_outer, rx_inner, 4)
280 .split(r.y, ry_outer, ry_inner, 4);
281
282 // We now call rfactor to make an intermediate function that
283 // independently computes a histogram of each tile.
284 Func intermediate = histogram.update().rfactor({{rx_outer, u}, {ry_outer, v}});
285
286 // We can now parallelize the intermediate over tiles.
287 intermediate.compute_root().update().parallel(u).parallel(v);
288
289 // We also reorder the tile indices outermost to give the
290 // classic tiled traversal.
291 intermediate.update().reorder(rx_inner, ry_inner, u, v);
292
293 // Vectorize the initializations.
294 intermediate.vectorize(x, 8);
295 histogram.vectorize(x, 8);
296
297 Buffer<int> halide_result = histogram.realize({8});
298
299 // See figures/lesson_18_hist_rfactor_tile.mp4 for a visualization of
300 // what this does.
301
302 // The equivalent C is:
303 int c_intm[4][4][8];
304 for (int v = 0; v < input.height() / 2; v++) {
305 for (int u = 0; u < input.width() / 2; u++) {
306 for (int x = 0; x < 8; x++) {
307 c_intm[v][u][x] = 0;
308 }
309 }
310 }
311 /* parallel */ for (int v = 0; v < input.height() / 2; v++) {
312 /* parallel */ for (int u = 0; u < input.width() / 2; u++) {
313 for (int ry_inner = 0; ry_inner < 2; ry_inner++) {
314 for (int rx_inner = 0; rx_inner < 2; rx_inner++) {
315 c_intm[v][u][input(u * 2 + rx_inner, v * 2 + ry_inner) / 32] += 1;
316 }
317 }
318 }
319 }
320
321 int c_result[8];
322 for (int x = 0; x < 8; x++) {
323 c_result[x] = 0;
324 }
325 for (int x = 0; x < 8; x++) {
326 for (int ry_outer = 0; ry_outer < input.height() / 2; ry_outer++) {
327 for (int rx_outer = 0; rx_outer < input.width() / 2; rx_outer++) {
328 c_result[x] += c_intm[ry_outer][rx_outer][x];
329 }
330 }
331 }
332
333 // Check the answers agree:
334 for (int x = 0; x < 8; x++) {
335 if (c_result[x] != halide_result(x)) {
336 printf("halide_result(%d) = %d instead of %d\n",
337 x, halide_result(x), c_result[x]);
338 return -1;
339 }
340 }
341 }
342
343 printf("Success!\n");
344
345 return 0;
346}
