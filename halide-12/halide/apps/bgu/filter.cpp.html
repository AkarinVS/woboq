14int main(int argc, char **argv) {
15 if (argc != 3) {
16 printf("Usage: %s in out\n", argv[0]);
17 return 1;
18 }
19
20 const float r_sigma = 1.f / 8.f;
21 const int s_sigma = 16;
22
23 // Bilateral-guided upsampling (BGU) is used for cheaply
24 // transferring some effect applied to a low-res image to a
25 // high-res input. We'll make a low-res version of the input,
26 // sharpen and contrast-enhance it naively, and then call into the
27 // Halide BGU implementation to apply the same effect to the
28 // original high-res input. We're only interested in benchmarking
29 // the last part.
30
31 // BGU will be good at capturing the contrast enhancement and
32 // vignette, and bad at capturing the high-frequency sharpening.
33
34 Halide::Runtime::Buffer<float> high_res_in = load_and_convert_image(argv[1]);
35 const int W = high_res_in.width();
36 const int H = high_res_in.height();
37 const int C = high_res_in.channels();
38
39 Halide::Runtime::Buffer<float> high_res_out(W, H, C);
40 Halide::Runtime::Buffer<float> low_res_in(W / 8, H / 8, C);
41 Halide::Runtime::Buffer<float> low_res_out(W / 8, H / 8, C);
42
43 // Downsample the input with a box filter
44 low_res_in.fill(0.0f);
45 low_res_in.for_each_element([&](int x, int y, int c) {
46 float sum = 0.0f;
47 for (int sy = y * 8; sy < y * 8 + 8; sy++) {
48 for (int sx = x * 8; sx < x * 8 + 8; sx++) {
49 sum += high_res_in(sx, sy, c);
50 }
51 }
52 low_res_in(x, y, c) = sum / 64;
53 });
54
55 // Some straw-man black-box image processing algorithm we're
56 // trying to accelerate.
57 low_res_in.for_each_element([&](int x, int y, int c) {
58 float val;
59 // Sharpen, ignoring edges
60 if (x == 0 || x == W - 1 || y == 0 || y == H - 1) {
61 val = low_res_in(x, y, c);
62 } else {
63 val =
64 (2 * low_res_in(x, y, c) -
65 (low_res_in(x - 1, y, c) +
66 low_res_in(x + 1, y, c) +
67 low_res_in(x, y - 1, c) +
68 low_res_in(x, y + 1, c)) /
69 4);
70 }
71 // Smoothstep for contrast enhancement
72 float boosted = val * val * (3 - 2 * val);
73
74 // We'll only boost the center of the image. The center of
75 // the low-res image is at W/16, H/16
76 float r = std::min(W / 16, H / 16);
77 float mask_x = (x - W / 16) / r;
78 float mask_y = (y - H / 16) / r;
79 float mask = std::sqrt(mask_x * mask_x + mask_y * mask_y);
80 val = val * mask + boosted * (1 - mask);
81
82 // Also apply a vignette.
83 val *= (2 - mask) / 2;
84
85 // Clamp to range
86 val = std::max(std::min(val, 1.0f), 0.0f);
87 low_res_out(x, y, c) = val;
88 });
89
90 // To view the low res output for debugging the algorithm above
91 // convert_and_save_image(low_res_out, "test.png");
92
93 double best_manual = benchmark([&]() {
94 bgu(r_sigma, s_sigma, low_res_in, low_res_out, high_res_in, high_res_out);
95 high_res_out.device_sync();
96 });
97 printf("Manually-tuned time: %gms\n", best_manual * 1e3);
98
99 double best_auto = benchmark([&]() {
100 bgu_auto_schedule(r_sigma, s_sigma, low_res_in, low_res_out, high_res_in, high_res_out);
101 high_res_out.device_sync();
102 });
103 printf("Auto-scheduled time: %gms\n", best_auto * 1e3);
104
105 convert_and_save_image(high_res_out, argv[2]);
106
107 printf("Success!\n");
108 return 0;
109}
