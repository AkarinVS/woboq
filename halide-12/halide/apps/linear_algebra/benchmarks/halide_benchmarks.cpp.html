1// USAGE: halide_benchmarks <subroutine> <size>
2//
3// Benchmarks BLAS subroutines using Halide's implementation. Will
4// construct random size x size matrices and/or size x 1 vectors
5// to test the subroutine with.
6//
7// Accepted values for subroutine are:
8// L1: scal, copy, axpy, dot, nrm2
9// L2: gemv_notrans, gemv_trans
10// L3: gemm_notrans, gemm_trans_A, gemm_trans_B, gemm_trans_AB
11//
12
13#include "HalideBuffer.h"
14#include "clock.h"
15#include "halide_blas.h"
16#include "macros.h"
17#include <iomanip>
18#include <iostream>
19#include <random>
20#include <string>

22template<class T>
23struct BenchmarksBase {
24 typedef T Scalar;
25 typedef Halide::Runtime::Buffer<T> Vector;
26 typedef Halide::Runtime::Buffer<T> Matrix;
27
28 std::random_device rand_dev;
29 std::default_random_engine rand_eng{rand_dev()};
30
31 std::string name;
32
33 Scalar random_scalar() {
34 std::uniform_real_distribution<T> uniform_dist(0.0, 1.0);
35 return uniform_dist(rand_eng);
36 }
37
38 Vector random_vector(int N) {
39 Vector buff(N);
40 Scalar *x = (Scalar *)buff.data();
41 for (int i = 0; i < N; ++i) {
42 x[i] = random_scalar();
43 }
44 return buff;
45 }
46
47 Matrix random_matrix(int N) {
48 Matrix buff(N, N);
49 Scalar *A = (Scalar *)buff.data();
50 for (int i = 0; i < N * N; ++i) {
51 A[i] = random_scalar();
52 }
53 return buff;
54 }
55
56 BenchmarksBase(std::string n)
57 : name(n) {
58 }
59
60 void run(std::string benchmark, int size) {
61 if (benchmark == "copy") {
62 bench_copy(size);
63 } else if (benchmark == "scal") {
64 bench_scal(size);
65 } else if (benchmark == "axpy") {
66 bench_axpy(size);
67 } else if (benchmark == "dot") {
68 bench_dot(size);
69 } else if (benchmark == "asum") {
70 bench_asum(size);
71 } else if (benchmark == "gemv_notrans") {
72 bench_gemv_notrans(size);
73 } else if (benchmark == "gemv_trans") {
74 bench_gemv_trans(size);
75 } else if (benchmark == "ger") {
76 bench_ger(size);
77 } else if (benchmark == "gemm_notrans") {
78 bench_gemm_notrans(size);
79 } else if (benchmark == "gemm_transA") {
80 bench_gemm_transA(size);
81 } else if (benchmark == "gemm_transB") {
82 bench_gemm_transB(size);
83 } else if (benchmark == "gemm_transAB") {
84 bench_gemm_transAB(size);
85 }
86 }
87
88 virtual void bench_copy(int N) = 0;
89 virtual void bench_scal(int N) = 0;
90 virtual void bench_axpy(int N) = 0;
91 virtual void bench_dot(int N) = 0;
92 virtual void bench_asum(int N) = 0;
93 virtual void bench_gemv_notrans(int N) = 0;
94 virtual void bench_gemv_trans(int N) = 0;
95 virtual void bench_ger(int N) = 0;
96 virtual void bench_gemm_notrans(int N) = 0;
97 virtual void bench_gemm_transA(int N) = 0;
98 virtual void bench_gemm_transB(int N) = 0;
99 virtual void bench_gemm_transAB(int N) = 0;
100};
101
102struct BenchmarksFloat : public BenchmarksBase<float> {
103 BenchmarksFloat(std::string n)
104 : BenchmarksBase(n),
105 result(Halide::Runtime::Buffer<float>::make_scalar()) {
106 }
107
108 Halide::Runtime::Buffer<float> result;
109
110 L1Benchmark(copy, "s", halide_scopy(x.raw_buffer(), y.raw_buffer()));
111 L1Benchmark(scal, "s", halide_sscal(alpha, x.raw_buffer()));
112 L1Benchmark(axpy, "s", halide_saxpy(alpha, x.raw_buffer(), y.raw_buffer()));
113 L1Benchmark(dot, "s", halide_sdot(x.raw_buffer(), y.raw_buffer(), result.raw_buffer()));
114 L1Benchmark(asum, "s", halide_sasum(x.raw_buffer(), result.raw_buffer()));
115
116 L2Benchmark(gemv_notrans, "s", halide_sgemv(false, alpha, A.raw_buffer(), x.raw_buffer(), beta, y.raw_buffer()));
117
118 L2Benchmark(gemv_trans, "s", halide_sgemv(true, alpha, A.raw_buffer(), x.raw_buffer(), beta, y.raw_buffer()));
119
120 L2Benchmark(ger, "s", halide_sger(alpha, x.raw_buffer(), y.raw_buffer(), A.raw_buffer()));
121
122 L3Benchmark(gemm_notrans, "s", halide_sgemm(false, false, alpha, A.raw_buffer(), B.raw_buffer(), beta, C.raw_buffer()));
123
124 L3Benchmark(gemm_transA, "s", halide_sgemm(true, false, alpha, A.raw_buffer(), B.raw_buffer(), beta, C.raw_buffer()));
125
126 L3Benchmark(gemm_transB, "s", halide_sgemm(false, true, alpha, A.raw_buffer(), B.raw_buffer(), beta, C.raw_buffer()));
127
128 L3Benchmark(gemm_transAB, "s", halide_sgemm(true, true, alpha, A.raw_buffer(), B.raw_buffer(), beta, C.raw_buffer()));
129};

160int main(int argc, char *argv[]) {
161 if (argc != 3) {
162 std::cout << "USAGE: halide_benchmarks <subroutine> <size>\n";
163 return 0;
164 }
165
166 std::string subroutine = argv[1];
167 char type = subroutine[0];
168 int size = std::stoi(argv[2]);
169
170 subroutine = subroutine.substr(1);
171 if (type == 's') {
172 BenchmarksFloat("Halide").run(subroutine, size);
173 } else if (type == 'd') {
174 BenchmarksDouble("Halide").run(subroutine, size);
175 }
176
177 return 0;
178}
