116template<class T>
117struct BLASTestBase {
118 typedef T Scalar;
119 typedef std::vector<T> Vector;
120 typedef std::vector<T> Matrix;
121
122 std::random_device rand_dev;
123 std::default_random_engine rand_eng;
124
125 BLASTestBase()
126 : rand_eng(rand_dev()) {
127 }
128
129 Scalar random_scalar() {
130 std::uniform_real_distribution<T> uniform_dist(0.0, 1.0);
131 return uniform_dist(rand_eng);
132 }
133
134 Vector random_vector(int N) {
135 Vector buff(N);
136 for (int i = 0; i < N; ++i) {
137 buff[i] = random_scalar();
138 }
139 return buff;
140 }
141
142 Matrix random_matrix(int N) {
143 Matrix buff(N * N);
144 for (int i = 0; i < N * N; ++i) {
145 buff[i] = random_scalar();
146 }
147 return buff;
148 }
149
150 bool compareScalars(Scalar x, Scalar y, Scalar epsilon = 32 * std::numeric_limits<Scalar>::epsilon()) {
151 if (x == y) {
152 return true;
153 } else {
154 const Scalar min_normal = std::numeric_limits<Scalar>::min();
155
156 Scalar ax = std::abs(x);
157 Scalar ay = std::abs(y);
158 Scalar diff = std::abs(x - y);
159
160 bool equal = false;
161 if (x == 0.0 || y == 0.0 || diff < min_normal) {
162 equal = diff < (epsilon * min_normal);
163 } else {
164 equal = diff / (ax + ay) < epsilon;
165 }
166
167 if (!equal) {
168 std::cerr << "FAIL! expected = " << x << ", actual = " << y << "\n";
169 }
170
171 return equal;
172 }
173 }
174
175 bool compareVectors(int N, const Vector &x, const Vector &y,
176 Scalar epsilon = 16 * std::numeric_limits<Scalar>::epsilon()) {
177 bool equal = true;
178 for (int i = 0; i < N; ++i) {
179 if (!compareScalars(x[i], y[i], epsilon)) {
180 std::cerr << "Vectors differ at index: " << i << "\n";
181 equal = false;
182 break;
183 }
184 }
185 return equal;
186 }
187
188 bool compareMatrices(int N, const Matrix &A, const Matrix &B,
189 Scalar epsilon = 16 * std::numeric_limits<Scalar>::epsilon()) {
190 bool equal = true;
191 for (int i = 0; i < N * N; ++i) {
192 if (!compareScalars(A[i], B[i], epsilon)) {
193 std::cerr << "Matrices differ at coords: (" << i % N << ", " << i / N << ")\n";
194 equal = false;
195 break;
196 }
197 }
198 return equal;
199 }
200};
