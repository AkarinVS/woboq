80inline int halide_sgemm(bool transA, bool transB, float a, halide_buffer_t *A, halide_buffer_t *B, float b, halide_buffer_t *C) {
81 if (transA && transB) {
82 return halide_sgemm_transAB(a, A, B, b, C, C);
83 } else if (transA) {
84 return halide_sgemm_transA(a, A, B, b, C, C);
85 } else if (transB) {
86 return halide_sgemm_transB(a, A, B, b, C, C);
87 } else {
88 return halide_sgemm_notrans(a, A, B, b, C, C);
89 }
90 return -1;
91}
