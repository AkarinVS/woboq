8// Generator class for BLAS gemv (GEneralized Matrix-Vector product) operations.
9template<class T>
10class GEMVGenerator : public Generator<GEMVGenerator<T>> {
11public:
12 typedef Generator<GEMVGenerator<T>> Base;
13 using Base::get_target;
14 using Base::natural_vector_size;
15 using Base::target;
16 template<typename T2>
17 using Input = typename Base::template Input<T2>;
18 template<typename T2>
19 using Output = typename Base::template Output<T2>;
20
21 GeneratorParam<bool> vectorize_ = {"vectorize", true};
22 GeneratorParam<bool> parallel_ = {"parallel", true};
23 GeneratorParam<int> block_size_ = {"block_size", 1 << 8};
24 GeneratorParam<bool> transpose_ = {"transpose", false};
25
26 // Standard ordering of parameters in GEMV functions.
27 Input<T> a_ = {"a", 1};
28 Input<Buffer<T>> A_ = {"A", 2};
29 Input<Buffer<T>> x_ = {"x", 1};
30 Input<T> b_ = {"b", 1};
31 Input<Buffer<T>> y_ = {"y", 1};
32
33 Output<Buffer<T>> output_ = {"output", 1};
34
35 void generate() {
36 assert(get_target().has_feature(Target::NoBoundsQuery));
37
38 const int vec_size = vectorize_ ? natural_vector_size(type_of<T>()) : 1;
39 const int unroll_size = std::min(vec_size, 4);
40
41 Var i("i"), j("j");
42 Func result("result");
43
44 if (transpose_) {
45 const Expr size = A_.height();
46 const Expr sum_size = A_.width();
47 const Expr sum_size_vecs = sum_size / vec_size;
48
49 Func prod("prod");
50 prod(j, i) = A_(j, i) * x_(j);
51
52 RDom k(0, sum_size_vecs, "k");
53 Func accum_vecs("accum_vecs");
54 accum_vecs(j, i) += prod(k * vec_size + j, i);
55
56 Func accum_vecs_transpose("accum_vecs_transpose");
57 accum_vecs_transpose(i, j) = accum_vecs(j, i);
58
59 RDom lanes(0, vec_size);
60 Func sum_lanes("sum_lanes");
61 sum_lanes(i) += accum_vecs_transpose(i, lanes);
62
63 RDom tail(sum_size_vecs * vec_size, sum_size - sum_size_vecs * vec_size);
64 Func sum_tail("sum_tail");
65 sum_tail(i) = sum_lanes(i);
66 sum_tail(i) += prod(tail, i);
67
68 Func Ax("Ax");
69 Ax(i) = sum_tail(i);
70 result(i) = b_ * y_(i) + a_ * Ax(i);
71
72 Var ii("ii"), t("t");
73 result.specialize((sum_size / vec_size) * vec_size == sum_size)
74 .specialize(size >= unroll_size)
75 .vectorize(i, unroll_size)
76 .specialize(size >= block_size_)
77 .split(i, t, i, block_size_ / unroll_size)
78 .parallel(t);
79
80 result
81 .specialize(size >= unroll_size)
82 .vectorize(i, unroll_size)
83 .specialize(size >= block_size_)
84 .split(i, t, i, block_size_ / unroll_size)
85 .parallel(t);
86
87 accum_vecs
88 .compute_at(result, i)
89 .unroll(i)
90 .unroll(j)
91 .update()
92 .reorder(i, j, k)
93 .unroll(i)
94 .unroll(j);
95 accum_vecs_transpose
96 .compute_at(result, i)
97 .unroll(i)
98 .unroll(j);
99 sum_lanes
100 .compute_at(result, i)
101 .update()
102 .unroll(lanes);
103 sum_tail
104 .compute_at(result, i)
105 .update()
106 .reorder(i, tail); //.unroll(i);
107
108 if (vectorize_) {
109 accum_vecs.vectorize(j)
110 .update()
111 .vectorize(j);
112 accum_vecs_transpose.vectorize(j);
113
114 sum_lanes.specialize(size >= vec_size).vectorize(i, vec_size);
115 sum_lanes.update().specialize(size >= vec_size).vectorize(i, vec_size);
116
117 sum_tail.specialize(size >= vec_size).vectorize(i, vec_size);
118 sum_tail.update().specialize(size >= vec_size).vectorize(i, vec_size);
119 }
120
121 A_.dim(0).set_min(0).dim(1).set_min(0);
122 x_.dim(0).set_bounds(0, A_.width());
123 y_.dim(0).set_bounds(0, A_.height());
124 } else {
125 const Expr size = A_.width();
126 const Expr sum_size = A_.height();
127 const Expr sum_size_cols = (sum_size / unroll_size) * unroll_size;
128 const Expr tail_size = sum_size - sum_size_cols;
129
130 RDom k(0, sum_size_cols, "k");
131 RDom tail(sum_size_cols, tail_size, "tail");
132 Func block("block");
133 block(i) = b_ * y_(i);
134 block(i) += a_ * A_(i, k) * x_(k);
135 block(i) += a_ * A_(i, tail) * x_(tail);
136 result(i) = block(i);
137
138 RVar ki("ki");
139 Var ii("ii");
140 result.specialize(tail_size == 0)
141 .specialize(size >= vec_size)
142 .vectorize(i, vec_size)
143 .specialize(size >= unroll_size * vec_size)
144 .unroll(i, unroll_size)
145 .specialize(size >= block_size_)
146 .split(i, i, ii, block_size_ / (unroll_size * vec_size))
147 .parallel(i);
148
149 result.specialize(size >= vec_size)
150 .vectorize(i, vec_size)
151 .specialize(size >= unroll_size * vec_size)
152 .unroll(i, unroll_size)
153 .specialize(size >= block_size_)
154 .split(i, i, ii, block_size_ / (unroll_size * vec_size))
155 .parallel(i);
156
157 block.compute_at(result, i);
158 block.specialize(size >= vec_size)
159 .vectorize(i, vec_size);
160 block.update()
161 .specialize(size >= vec_size && sum_size >= unroll_size)
162 .split(i, i, ii, vec_size)
163 .split(k, k, ki, unroll_size)
164 .reorder(ii, ki, i, k)
165 .vectorize(ii)
166 .unroll(ki);
167 block.update()
168 .specialize(size >= vec_size)
169 .vectorize(i, vec_size);
170 block.update(1)
171 .reorder(i, tail)
172 .specialize(size >= vec_size)
173 .vectorize(i, vec_size)
174 .specialize(sum_size >= unroll_size)
175 .unroll(i, unroll_size);
176
177 A_.dim(0).set_min(0).dim(1).set_min(0);
178 x_.dim(0).set_bounds(0, A_.height());
179 y_.dim(0).set_bounds(0, A_.width());
180 }
181
182 // TODO: delete this pointless memcpy, as we probably have the tools to deal with this now.
183 // see https://github.com/halide/Halide/commit/cf999bf71939261bdcbb92d87fc4d07db5770732
184 output_(i) = result(i);
185 result.compute_root();
186
187 const Expr size = x_.width();
188 Var ii("ii");
189 output_.specialize(size >= vec_size)
190 .vectorize(i, vec_size)
191 .specialize(size >= unroll_size * vec_size)
192 .unroll(i, unroll_size)
193 .specialize(size >= block_size_)
194 .split(i, i, ii, block_size_ / (unroll_size * vec_size))
195 .parallel(i);
196 }
197};

199// Generator class for BLAS ger (GEneralized Rank-1 update) operations.
200template<class T>
201class GERGenerator : public Generator<GERGenerator<T>> {
202public:
203 typedef Generator<GERGenerator<T>> Base;
204 using Base::get_target;
205 using Base::natural_vector_size;
206 using Base::target;
207 template<typename T2>
208 using Input = typename Base::template Input<T2>;
209 template<typename T2>
210 using Output = typename Base::template Output<T2>;
211
212 GeneratorParam<bool> vectorize_ = {"vectorize", true};
213 GeneratorParam<bool> parallel_ = {"parallel", true};
214 GeneratorParam<int> block_size_ = {"block_size", 1 << 5};
215
216 // Standard ordering of parameters in GEMV functions.
217 Input<T> a_ = {"a", 1};
218 Input<Buffer<T>> x_ = {"x", 1};
219 Input<Buffer<T>> y_ = {"y", 1};
220
221 Output<Buffer<T>> result_ = {"result", 2};
222
223 void generate() {
224 const int vec_size = vectorize_ ? natural_vector_size(type_of<T>()) : 1;
225
226 Var i("i"), j("j");
227 result_(i, j) = undef<T>(); // in-place operation on the output
228
229 result_(i, j) += (a_ * y_(j)) * x_(i);
230
231 if (vectorize_) {
232 result_.update().vectorize(i, vec_size * 4, TailStrategy::GuardWithIf);
233 }
234 if (parallel_) {
235 result_.update().parallel(j, 8, TailStrategy::GuardWithIf);
236 }
237
238 x_.dim(0).set_min(0);
239 y_.dim(0).set_min(0);
240 result_.dim(0).set_bounds(0, x_.dim(0).extent());
241 result_.dim(1).set_bounds(0, y_.dim(0).extent());
242 }
243};
