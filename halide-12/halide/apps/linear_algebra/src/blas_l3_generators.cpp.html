8// Generator class for BLAS gemm operations.
9template<class T>
10class GEMMGenerator : public Generator<GEMMGenerator<T>> {
11public:
12 typedef Generator<GEMMGenerator<T>> Base;
13 using Base::get_target;
14 using Base::natural_vector_size;
15 using Base::target;
16 template<typename T2>
17 using Input = typename Base::template Input<T2>;
18 template<typename T2>
19 using Output = typename Base::template Output<T2>;
20
21 GeneratorParam<bool> transpose_A_ = {"transpose_A", false};
22 GeneratorParam<bool> transpose_B_ = {"transpose_B", false};
23
24 // Standard ordering of parameters in GEMM functions.
25 Input<T> a_ = {"a_", 1};
26 Input<Buffer<T>> A_ = {"A_", 2};
27 Input<Buffer<T>> B_ = {"B_", 2};
28 Input<T> b_ = {"b_", 1};
29 Input<Buffer<T>> C_ = {"C_", 2};
30
31 Output<Buffer<T>> result_ = {"result", 2};
32
33 void generate() {
34 // Matrices are interpreted as column-major by default. The
35 // transpose GeneratorParams are used to handle cases where
36 // one or both is actually row major.
37 const Expr num_rows = A_.width();
38 const Expr num_cols = B_.height();
39 const Expr sum_size = A_.height();
40
41 const int vec = std::max(4, natural_vector_size(a_.type()));
42 const int s = vec * 2;
43
44 Input<Buffer<T>> *A_in = &A_;
45 Input<Buffer<T>> *B_in = &B_;
46
47 // If they're both transposed, then reverse the order and transpose the result instead.
48 const bool transpose_AB = (bool)transpose_A_ && (bool)transpose_B_;
49 const bool transpose_A = !transpose_AB && (bool)transpose_A_;
50 const bool transpose_B = !transpose_AB && (bool)transpose_B_;
51 if (transpose_AB) {
52 std::swap(A_in, B_in);
53 }
54
55 Var i, j, ii, ji, jii, iii, io, jo, t;
56 Var ti[3], tj[3];
57
58 // Swizzle A for better memory order in the inner loop.
59 Func A("A"), B("B"), Btmp("Btmp"), As("As"), Atmp("Atmp");
60 Atmp(i, j) = BoundaryConditions::constant_exterior(*A_in, cast<T>(0))(i, j);
61
62 if (transpose_A) {
63 As(i, j, io) = Atmp(j, io * s + i);
64 } else {
65 As(i, j, io) = Atmp(io * s + i, j);
66 }
67
68 A(i, j) = As(i % s, j, i / s);
69
70 Btmp(i, j) = (*B_in)(i, j);
71 if (transpose_B) {
72 B(i, j) = Btmp(j, i);
73 } else {
74 B(i, j) = Btmp(i, j);
75 }
76
77 Var k("k");
78 Func prod;
79 // Express all the products we need to do a matrix multiply as a 3D Func.
80 prod(k, i, j) = A(i, k) * B(k, j);
81
82 // Reduce the products along k.
83 Func AB("AB");
84 RDom rv(0, sum_size);
85 AB(i, j) += prod(rv, i, j);
86
87 Func ABt("ABt");
88 if (transpose_AB) {
89 // Transpose A*B if necessary.
90 ABt(i, j) = AB(j, i);
91 } else {
92 ABt(i, j) = AB(i, j);
93 }
94
95 // Do the part that makes it a 'general' matrix multiply.
96 result_(i, j) = (a_ * ABt(i, j) + b_ * C_(i, j));
97
98 result_.tile(i, j, ti[1], tj[1], i, j, 2 * s, 2 * s, TailStrategy::GuardWithIf);
99 if (transpose_AB) {
100 result_
101 .tile(i, j, ii, ji, 4, s)
102 .tile(i, j, ti[0], tj[0], i, j, s / 4, 1);
103
104 } else {
105 result_
106 .tile(i, j, ii, ji, s, 4)
107 .tile(i, j, ti[0], tj[0], i, j, 1, s / 4);
108 }
109
110 // If we have enough work per task, parallelize over these tiles.
111 result_.specialize(num_rows >= 512 && num_cols >= 512)
112 .fuse(tj[1], ti[1], t)
113 .parallel(t);
114
115 // Otherwise tile one more time before parallelizing, or don't
116 // parallelize at all.
117 result_.specialize(num_rows >= 128 && num_cols >= 128)
118 .tile(ti[1], tj[1], ti[2], tj[2], ti[1], tj[1], 2, 2)
119 .fuse(tj[2], ti[2], t)
120 .parallel(t);
121
122 result_.rename(tj[0], t);
123
124 result_.bound(i, 0, num_rows).bound(j, 0, num_cols);
125
126 As.compute_root()
127 .split(j, jo, ji, s)
128 .reorder(i, ji, io, jo)
129 .unroll(i)
130 .vectorize(ji)
131 .specialize(A_.width() >= 256 && A_.height() >= 256)
132 .parallel(jo, 4);
133
134 Atmp.compute_at(As, io)
135 .vectorize(i)
136 .unroll(j);
137
138 if (transpose_B) {
139 B.compute_at(result_, t)
140 .tile(i, j, ii, ji, 8, 8)
141 .vectorize(ii)
142 .unroll(ji);
143 Btmp.reorder_storage(j, i)
144 .compute_at(B, i)
145 .vectorize(i)
146 .unroll(j);
147 }
148
149 AB.compute_at(result_, i)
150 .bound_extent(j, 4)
151 .unroll(j)
152 .bound_extent(i, s)
153 .vectorize(i)
154 .update()
155 .reorder(i, j, rv)
156 .unroll(j)
157 .unroll(rv, 2)
158 .vectorize(i);
159 if (transpose_AB) {
160 ABt.compute_at(result_, i)
161 .bound_extent(i, 4)
162 .unroll(i)
163 .bound_extent(j, s)
164 .vectorize(j);
165 }
166
167 A_.dim(0).set_min(0).dim(1).set_min(0);
168 B_.dim(0).set_bounds(0, sum_size).dim(1).set_min(0);
169 C_.dim(0).set_bounds(0, num_rows);
170 C_.dim(1).set_bounds(0, num_cols);
171 result_.dim(0).set_bounds(0, num_rows).dim(1).set_bounds(0, num_cols);
172 }
173};
