18Buffer<uint16_t> blur(Buffer<uint16_t> in) {
19 Buffer<uint16_t> tmp(in.width() - 8, in.height());
20 Buffer<uint16_t> out(in.width() - 8, in.height() - 2);
21
22 t = benchmark(10, 1, [&]() {
23 for (int y = 0; y < tmp.height(); y++)
24 for (int x = 0; x < tmp.width(); x++)
25 tmp(x, y) = (in(x, y) + in(x + 1, y) + in(x + 2, y)) / 3;
26
27 for (int y = 0; y < out.height(); y++)
28 for (int x = 0; x < out.width(); x++)
29 out(x, y) = (tmp(x, y) + tmp(x, y + 1) + tmp(x, y + 2)) / 3;
30 });
31
32 return out;
33}

35Buffer<uint16_t> blur_fast(Buffer<uint16_t> in) {
36 Buffer<uint16_t> out(in.width() - 8, in.height() - 2);
37
38 t = benchmark(10, 1, [&]() {
39#ifdef __SSE2__
40 __m128i one_third = _mm_set1_epi16(21846);
41#pragma omp parallel for
42 for (int yTile = 0; yTile < out.height(); yTile += 32) {
43 __m128i tmp[(128 / 8) * (32 + 2)];
44 for (int xTile = 0; xTile < out.width(); xTile += 128) {
45 __m128i *tmpPtr = tmp;
46 for (int y = 0; y < 32 + 2; y++) {
47 const uint16_t *inPtr = &(in(xTile, yTile + y));
48 for (int x = 0; x < 128; x += 8) {
49 __m128i a = _mm_load_si128((const __m128i *)(inPtr));
50 __m128i b = _mm_loadu_si128((const __m128i *)(inPtr + 1));
51 __m128i c = _mm_loadu_si128((const __m128i *)(inPtr + 2));
52 __m128i sum = _mm_add_epi16(_mm_add_epi16(a, b), c);
53 __m128i avg = _mm_mulhi_epi16(sum, one_third);
54 _mm_store_si128(tmpPtr++, avg);
55 inPtr += 8;
56 }
57 }
58 tmpPtr = tmp;
59 for (int y = 0; y < 32; y++) {
60 __m128i *outPtr = (__m128i *)(&(out(xTile, yTile + y)));
61 for (int x = 0; x < 128; x += 8) {
62 __m128i a = _mm_load_si128(tmpPtr + (2 * 128) / 8);
63 __m128i b = _mm_load_si128(tmpPtr + 128 / 8);
64 __m128i c = _mm_load_si128(tmpPtr++);
65 __m128i sum = _mm_add_epi16(_mm_add_epi16(a, b), c);
66 __m128i avg = _mm_mulhi_epi16(sum, one_third);
67 _mm_store_si128(outPtr++, avg);
68 }
69 }
70 }
71 }
72#elif __ARM_NEON
73 uint16x4_t one_third = vdup_n_u16(21846);
74#pragma omp parallel for
75 for (int yTile = 0; yTile < out.height(); yTile += 32) {
76 uint16x8_t tmp[(128 / 8) * (32 + 2)];
77 for (int xTile = 0; xTile < out.width(); xTile += 128) {
78 uint16_t *tmpPtr = (uint16_t *)tmp;
79 for (int y = 0; y < 32 + 2; y++) {
80 const uint16_t *inPtr = &(in(xTile, yTile + y));
81 for (int x = 0; x < 128; x += 8) {
82 uint16x8_t a = vld1q_u16(inPtr);
83 uint16x8_t b = vld1q_u16(inPtr + 1);
84 uint16x8_t c = vld1q_u16(inPtr + 2);
85 uint16x8_t sum = vaddq_u16(vaddq_u16(a, b), c);
86 uint16x4_t sumlo = vget_low_u16(sum);
87 uint16x4_t sumhi = vget_high_u16(sum);
88 uint16x4_t avglo = vshrn_n_u32(vmull_u16(sumlo, one_third), 16);
89 uint16x4_t avghi = vshrn_n_u32(vmull_u16(sumhi, one_third), 16);
90 uint16x8_t avg = vcombine_u16(avglo, avghi);
91 vst1q_u16(tmpPtr, avg);
92 tmpPtr += 8;
93 inPtr += 8;
94 }
95 }
96 tmpPtr = (uint16_t *)tmp;
97 for (int y = 0; y < 32; y++) {
98 uint16_t *outPtr = &(out(xTile, yTile + y));
99 for (int x = 0; x < 128; x += 8) {
100 uint16x8_t a = vld1q_u16(tmpPtr + (2 * 128));
101 uint16x8_t b = vld1q_u16(tmpPtr + 128);
102 uint16x8_t c = vld1q_u16(tmpPtr);
103 uint16x8_t sum = vaddq_u16(vaddq_u16(a, b), c);
104 uint16x4_t sumlo = vget_low_u16(sum);
105 uint16x4_t sumhi = vget_high_u16(sum);
106 uint16x4_t avglo = vshrn_n_u32(vmull_u16(sumlo, one_third), 16);
107 uint16x4_t avghi = vshrn_n_u32(vmull_u16(sumhi, one_third), 16);
108 uint16x8_t avg = vcombine_u16(avglo, avghi);
109 vst1q_u16(outPtr, avg);
110 tmpPtr += 8;
111 outPtr += 8;
112 }
113 }
114 }
115 }
116#else
117 // No intrinsics enabled, do a naive thing.
118 for (int y = 0; y < out.height(); y++) {
119 for (int x = 0; x < out.width(); x++) {
120 int tmp[3] = {
121 (in(x, y) + in(x + 1, y) + in(x + 2, y)) / 3,
122 (in(x, y + 1) + in(x + 1, y + 1) + in(x + 2, y + 1)) / 3,
123 (in(x, y + 2) + in(x + 1, y + 2) + in(x + 2, y + 2)) / 3,
124 };
125 out(x, y) = (tmp[0] + tmp[1] + tmp[2]) / 3;
126 }
127 }
128#endif
129 });
130
131 return out;
132}

136Buffer<uint16_t> blur_halide(Buffer<uint16_t> in) {
137 Buffer<uint16_t> out(in.width() - 8, in.height() - 2);
138
139 // Call it once to initialize the halide runtime stuff
140 halide_blur(in, out);
141 // Copy-out result if it's device buffer and dirty.
142 out.copy_to_host();
143
144 t = benchmark(10, 1, [&]() {
145 // Compute the same region of the output as blur_fast (i.e., we're
146 // still being sloppy with boundary conditions)
147 halide_blur(in, out);
148 // Sync device execution if any.
149 out.device_sync();
150 });
151
152 out.copy_to_host();
153
154 return out;
155}

157int main(int argc, char **argv) {
158 const auto *md = halide_blur_metadata();
159 const bool is_hexagon = strstr(md->target, "hvx_128") || strstr(md->target, "hvx_64");
160
161 // The Hexagon simulator can't allocate as much memory as the above wants.
162 const int width = is_hexagon ? 648 : 2568;
163 const int height = is_hexagon ? 482 : 1922;
164
165 Buffer<uint16_t> input(width, height);
166
167 for (int y = 0; y < input.height(); y++) {
168 for (int x = 0; x < input.width(); x++) {
169 input(x, y) = rand() & 0xfff;
170 }
171 }
172
173 Buffer<uint16_t> blurry = blur(input);
174 double slow_time = t;
175
176 Buffer<uint16_t> speedy = blur_fast(input);
177 double fast_time = t;
178
179 Buffer<uint16_t> halide = blur_halide(input);
180 double halide_time = t;
181
182 printf("times: %f %f %f\n", slow_time, fast_time, halide_time);
183
184 for (int y = 64; y < input.height() - 64; y++) {
185 for (int x = 64; x < input.width() - 64; x++) {
186 if (blurry(x, y) != speedy(x, y) || blurry(x, y) != halide(x, y)) {
187 printf("difference at (%d,%d): %d %d %d\n", x, y, blurry(x, y), speedy(x, y), halide(x, y));
188 abort();
189 }
190 }
191 }
192
193 printf("Success!\n");
194 return 0;
195}
