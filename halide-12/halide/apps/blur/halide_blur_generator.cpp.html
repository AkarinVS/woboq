5enum class BlurGPUSchedule {
6 Inline, // Fully inlining schedule.
7 Cache, // Schedule caching intermedia result of blur_x.
8 Slide, // Schedule enabling sliding window opt within each
9 // work-item or cuda thread.
10 SlideVectorize, // The same as above plus vectorization per work-item.
11};
12
13std::map<std::string, BlurGPUSchedule> blurGPUScheduleEnumMap() {
14 return {
15 {"inline", BlurGPUSchedule::Inline},
16 {"cache", BlurGPUSchedule::Cache},
17 {"slide", BlurGPUSchedule::Slide},
18 {"slide_vector", BlurGPUSchedule::SlideVectorize},
19 };
20};
21
22class HalideBlur : public Halide::Generator<HalideBlur> {
23public:
24 GeneratorParam<BlurGPUSchedule> schedule{
25 "schedule",
26 BlurGPUSchedule::SlideVectorize,
27 blurGPUScheduleEnumMap()};
28 GeneratorParam<int> tile_x{"tile_x", 32}; // X tile.
29 GeneratorParam<int> tile_y{"tile_y", 8}; // Y tile.
30
31 Input<Buffer<uint16_t>> input{"input", 2};
32 Output<Buffer<uint16_t>> blur_y{"blur_y", 2};
33
34 void generate() {
35 Func blur_x("blur_x");
36 Var x("x"), y("y"), xi("xi"), yi("yi");
37
38 // The algorithm
39 blur_x(x, y) = (input(x, y) + input(x + 1, y) + input(x + 2, y)) / 3;
40 blur_y(x, y) = (blur_x(x, y) + blur_x(x, y + 1) + blur_x(x, y + 2)) / 3;
41
42 // How to schedule it
43 if (get_target().has_gpu_feature()) {
44 // GPU schedule.
45 switch (schedule) {
46 case BlurGPUSchedule::Inline:
47 // - Fully inlining.
48 blur_y.gpu_tile(x, y, xi, yi, tile_x, tile_y);
49 break;
50 case BlurGPUSchedule::Cache:
51 // - Cache blur_x calculation.
52 blur_y.gpu_tile(x, y, xi, yi, tile_x, tile_y);
53 blur_x.compute_at(blur_y, x).gpu_threads(x, y);
54 break;
55 case BlurGPUSchedule::Slide: {
56 // - Instead caching blur_x calculation explicitly, the
57 // alternative is to allow each work-item in OpenCL or thread
58 // in CUDA to calculate more rows of blur_y so that temporary
59 // blur_x calculation is re-used implicitly. This achieves
60 // the similar schedule of sliding window.
61 Var y_inner("y_inner");
62 blur_y
63 .split(y, y, y_inner, tile_y)
64 .reorder(y_inner, x)
65 .unroll(y_inner)
66 .gpu_tile(x, y, xi, yi, tile_x, 1);
67 break;
68 }
69 case BlurGPUSchedule::SlideVectorize: {
70 // Vectorization factor.
71 int factor = sizeof(int) / sizeof(short);
72 Var y_inner("y_inner");
73 blur_y.vectorize(x, factor)
74 .split(y, y, y_inner, tile_y)
75 .reorder(y_inner, x)
76 .unroll(y_inner)
77 .gpu_tile(x, y, xi, yi, tile_x, 1);
78 break;
79 }
80 default:
81 break;
82 }
83 } else if (get_target().has_feature(Target::HVX)) {
84 // Hexagon schedule.
85 // TODO: Try using a schedule like the CPU one below.
86 const int vector_size = 128;
87
88 blur_y.compute_root()
89 .hexagon()
90 .prefetch(input, y, 2)
91 .split(y, y, yi, 128)
92 .parallel(y)
93 .vectorize(x, vector_size * 2);
94 blur_x
95 .store_at(blur_y, y)
96 .compute_at(blur_y, yi)
97 .vectorize(x, vector_size);
98 } else {
99 // CPU schedule.
100 // Compute blur_x as needed at each vector of the output.
101 // Halide will store blur_x in a circular buffer so its
102 // results can be re-used.
103 blur_y
104 .split(y, y, yi, 32)
105 .parallel(y)
106 .vectorize(x, 16);
107 blur_x
108 .store_at(blur_y, y)
109 .compute_at(blur_y, x)
110 .vectorize(x, 16);
111 }
112 }
113};
