15// Average two positive values rounding up
16Expr avg(Expr a, Expr b) {
17 Type wider = a.type().with_bits(a.type().bits() * 2);
18 return cast(a.type(), (cast(wider, a) + b + 1) / 2);
19}
20
21Expr blur121(Expr a, Expr b, Expr c) {
22 return avg(avg(a, c), b);
23}
24
25Func interleave_x(Func a, Func b) {
26 Func out;
27 out(x, y) = select((x % 2) == 0, a(x / 2, y), b(x / 2, y));
28 return out;
29}
30
31Func interleave_y(Func a, Func b) {
32 Func out;
33 out(x, y) = select((y % 2) == 0, a(x, y / 2), b(x, y / 2));
34 return out;
35}

37class Demosaic : public Halide::Generator<Demosaic> {
38public:
39 GeneratorParam<LoopLevel> intermed_compute_at{"intermed_compute_at", LoopLevel::inlined()};
40 GeneratorParam<LoopLevel> intermed_store_at{"intermed_store_at", LoopLevel::inlined()};
41 GeneratorParam<LoopLevel> output_compute_at{"output_compute_at", LoopLevel::inlined()};
42
43 // Inputs and outputs
44 Input<Func> deinterleaved{"deinterleaved", Int(16), 3};
45 Output<Func> output{"output", Int(16), 3};
46
47 // Defines outputs using inputs
48 void generate() {
49 // These are the values we already know from the input
50 // x_y = the value of channel x at a site in the input of channel y
51 // gb refers to green sites in the blue rows
52 // gr refers to green sites in the red rows
53
54 // Give more convenient names to the four channels we know
55 Func r_r, g_gr, g_gb, b_b;
56
57 g_gr(x, y) = deinterleaved(x, y, 0);
58 r_r(x, y) = deinterleaved(x, y, 1);
59 b_b(x, y) = deinterleaved(x, y, 2);
60 g_gb(x, y) = deinterleaved(x, y, 3);
61
62 // These are the ones we need to interpolate
63 Func b_r, g_r, b_gr, r_gr, b_gb, r_gb, r_b, g_b;
64
65 // First calculate green at the red and blue sites
66
67 // Try interpolating vertically and horizontally. Also compute
68 // differences vertically and horizontally. Use interpolation in
69 // whichever direction had the smallest difference.
70 Expr gv_r = avg(g_gb(x, y - 1), g_gb(x, y));
71 Expr gvd_r = absd(g_gb(x, y - 1), g_gb(x, y));
72 Expr gh_r = avg(g_gr(x + 1, y), g_gr(x, y));
73 Expr ghd_r = absd(g_gr(x + 1, y), g_gr(x, y));
74
75 g_r(x, y) = select(ghd_r < gvd_r, gh_r, gv_r);
76
77 Expr gv_b = avg(g_gr(x, y + 1), g_gr(x, y));
78 Expr gvd_b = absd(g_gr(x, y + 1), g_gr(x, y));
79 Expr gh_b = avg(g_gb(x - 1, y), g_gb(x, y));
80 Expr ghd_b = absd(g_gb(x - 1, y), g_gb(x, y));
81
82 g_b(x, y) = select(ghd_b < gvd_b, gh_b, gv_b);
83
84 // Next interpolate red at gr by first interpolating, then
85 // correcting using the error green would have had if we had
86 // interpolated it in the same way (i.e. add the second derivative
87 // of the green channel at the same place).
88 Expr correction;
89 correction = g_gr(x, y) - avg(g_r(x, y), g_r(x - 1, y));
90 r_gr(x, y) = correction + avg(r_r(x - 1, y), r_r(x, y));
91
92 // Do the same for other reds and blues at green sites
93 correction = g_gr(x, y) - avg(g_b(x, y), g_b(x, y - 1));
94 b_gr(x, y) = correction + avg(b_b(x, y), b_b(x, y - 1));
95
96 correction = g_gb(x, y) - avg(g_r(x, y), g_r(x, y + 1));
97 r_gb(x, y) = correction + avg(r_r(x, y), r_r(x, y + 1));
98
99 correction = g_gb(x, y) - avg(g_b(x, y), g_b(x + 1, y));
100 b_gb(x, y) = correction + avg(b_b(x, y), b_b(x + 1, y));
101
102 // Now interpolate diagonally to get red at blue and blue at
103 // red. Hold onto your hats; this gets really fancy. We do the
104 // same thing as for interpolating green where we try both
105 // directions (in this case the positive and negative diagonals),
106 // and use the one with the lowest absolute difference. But we
107 // also use the same trick as interpolating red and blue at green
108 // sites - we correct our interpolations using the second
109 // derivative of green at the same sites.
110
111 correction = g_b(x, y) - avg(g_r(x, y), g_r(x - 1, y + 1));
112 Expr rp_b = correction + avg(r_r(x, y), r_r(x - 1, y + 1));
113 Expr rpd_b = absd(r_r(x, y), r_r(x - 1, y + 1));
114
115 correction = g_b(x, y) - avg(g_r(x - 1, y), g_r(x, y + 1));
116 Expr rn_b = correction + avg(r_r(x - 1, y), r_r(x, y + 1));
117 Expr rnd_b = absd(r_r(x - 1, y), r_r(x, y + 1));
118
119 r_b(x, y) = select(rpd_b < rnd_b, rp_b, rn_b);
120
121 // Same thing for blue at red
122 correction = g_r(x, y) - avg(g_b(x, y), g_b(x + 1, y - 1));
123 Expr bp_r = correction + avg(b_b(x, y), b_b(x + 1, y - 1));
124 Expr bpd_r = absd(b_b(x, y), b_b(x + 1, y - 1));
125
126 correction = g_r(x, y) - avg(g_b(x + 1, y), g_b(x, y - 1));
127 Expr bn_r = correction + avg(b_b(x + 1, y), b_b(x, y - 1));
128 Expr bnd_r = absd(b_b(x + 1, y), b_b(x, y - 1));
129
130 b_r(x, y) = select(bpd_r < bnd_r, bp_r, bn_r);
131
132 // Resulting color channels
133 Func r, g, b;
134
135 // Interleave the resulting channels
136 r = interleave_y(interleave_x(r_gr, r_r),
137 interleave_x(r_b, r_gb));
138 g = interleave_y(interleave_x(g_gr, g_r),
139 interleave_x(g_b, g_gb));
140 b = interleave_y(interleave_x(b_gr, b_r),
141 interleave_x(b_b, b_gb));
142
143 output(x, y, c) = mux(c, {r(x, y), g(x, y), b(x, y)});
144
145 // These are the stencil stages we want to schedule
146 // separately. Everything else we'll just inline.
147 intermediates.push_back(g_r);
148 intermediates.push_back(g_b);
149 }
150
151 void schedule() {
152 Pipeline p(output);
153
154 if (auto_schedule) {
155 // blank
156 } else if (get_target().has_gpu_feature()) {
157 Var xi, yi;
158 for (Func f : intermediates) {
159 f.compute_at(intermed_compute_at).gpu_threads(x, y);
160 }
161 output.compute_at(output_compute_at)
162 .unroll(x, 2)
163 .gpu_threads(x, y)
164 .reorder(c, x, y)
165 .unroll(c);
166 } else {
167 int vec = get_target().natural_vector_size(UInt(16));
168 bool use_hexagon = get_target().has_feature(Target::HVX);
169
170 for (Func f : intermediates) {
171 f.compute_at(intermed_compute_at)
172 .store_at(intermed_store_at)
173 .vectorize(x, 2 * vec, TailStrategy::RoundUp)
174 .fold_storage(y, 4);
175 }
176 intermediates[1].compute_with(
177 intermediates[0], x,
178 {{x, LoopAlignStrategy::AlignStart}, {y, LoopAlignStrategy::AlignStart}});
179 output.compute_at(output_compute_at)
180 .vectorize(x)
181 .unroll(y)
182 .reorder(c, x, y)
183 .unroll(c);
184 if (use_hexagon) {
185 output.hexagon();
186 for (Func f : intermediates) {
187 f.align_storage(x, vec);
188 }
189 }
190 }
191
192 /* Optional tags to specify layout for HalideTraceViz */
193 Halide::Trace::FuncConfig cfg;
194 cfg.pos = {860, 340 - 220};
195 cfg.max = 1024;
196 for (Func f : intermediates) {
197 std::string label = f.name();
198 std::replace(label.begin(), label.end(), '_', '@');
199 cfg.pos.y += 220;
200 cfg.labels = {{label}};
201 f.add_trace_tag(cfg.to_trace_tag());
202 }
203 }
204
205private:
206 // Intermediate stencil stages to schedule
207 vector<Func> intermediates;
208};

261Func CameraPipe::color_correct(Func input) {
262 // Get a color matrix by linearly interpolating between two
263 // calibrated matrices using inverse kelvin.
264 Expr kelvin = color_temp;
265
266 Func matrix;
267 Expr alpha = (1.0f / kelvin - 1.0f / 3200) / (1.0f / 7000 - 1.0f / 3200);
268 Expr val = (matrix_3200(x, y) * alpha + matrix_7000(x, y) * (1 - alpha));
269 matrix(x, y) = cast<int16_t>(val * 256.0f); // Q8.8 fixed point
270
271 if (!auto_schedule) {
272 matrix.compute_root();
273 if (get_target().has_gpu_feature()) {
274 matrix.gpu_single_thread();
275 }
276 }
277
278 Func corrected;
279 Expr ir = cast<int32_t>(input(x, y, 0));
280 Expr ig = cast<int32_t>(input(x, y, 1));
281 Expr ib = cast<int32_t>(input(x, y, 2));
282
283 Expr r = matrix(3, 0) + matrix(0, 0) * ir + matrix(1, 0) * ig + matrix(2, 0) * ib;
284 Expr g = matrix(3, 1) + matrix(0, 1) * ir + matrix(1, 1) * ig + matrix(2, 1) * ib;
285 Expr b = matrix(3, 2) + matrix(0, 2) * ir + matrix(1, 2) * ig + matrix(2, 2) * ib;
286
287 r = cast<int16_t>(r / 256);
288 g = cast<int16_t>(g / 256);
289 b = cast<int16_t>(b / 256);
290 corrected(x, y, c) = mux(c, {r, g, b});
291
292 return corrected;
293}

295Func CameraPipe::apply_curve(Func input) {
296 // copied from FCam
297 Func curve("curve");
298
299 Expr minRaw = 0 + blackLevel;
300 Expr maxRaw = whiteLevel;
301
302 // How much to upsample the LUT by when sampling it.
303 int lutResample = 1;
304 if (get_target().has_feature(Target::HVX)) {
305 // On HVX, LUT lookups are much faster if they are to LUTs not
306 // greater than 256 elements, so we reduce the tonemap to 256
307 // elements and use linear interpolation to upsample it.
308 lutResample = 8;
309 }
310
311 minRaw /= lutResample;
312 maxRaw /= lutResample;
313
314 Expr invRange = 1.0f / (maxRaw - minRaw);
315 Expr b = 2.0f - pow(2.0f, contrast / 100.0f);
316 Expr a = 2.0f - 2.0f * b;
317
318 // Get a linear luminance in the range 0-1
319 Expr xf = clamp(cast<float>(x - minRaw) * invRange, 0.0f, 1.0f);
320 // Gamma correct it
321 Expr g = pow(xf, 1.0f / gamma);
322 // Apply a piecewise quadratic contrast curve
323 Expr z = select(g > 0.5f,
324 1.0f - (a * (1.0f - g) * (1.0f - g) + b * (1.0f - g)),
325 a * g * g + b * g);
326
327 // Convert to 8 bit and save
328 Expr val = cast(result_type, clamp(z * 255.0f + 0.5f, 0.0f, 255.0f));
329 // makeLUT add guard band outside of (minRaw, maxRaw]:
330 curve(x) = select(x <= minRaw, 0, select(x > maxRaw, 255, val));
331
332 if (!auto_schedule) {
333 // It's a LUT, compute it once ahead of time.
334 curve.compute_root();
335 if (get_target().has_gpu_feature()) {
336 Var xi;
337 curve.gpu_tile(x, xi, 32);
338 }
339 }
340
341 /* Optional tags to specify layout for HalideTraceViz */
342 {
343 Halide::Trace::FuncConfig cfg;
344 cfg.labels = {{"tone curve"}};
345 cfg.pos = {580, 1000};
346 curve.add_trace_tag(cfg.to_trace_tag());
347 }
348
349 Func curved;
350
351 if (lutResample == 1) {
352 // Use clamp to restrict size of LUT as allocated by compute_root
353 curved(x, y, c) = curve(clamp(input(x, y, c), 0, 1023));
354 } else {
355 // Use linear interpolation to sample the LUT.
356 Expr in = input(x, y, c);
357 Expr u0 = in / lutResample;
358 Expr u = in % lutResample;
359 Expr y0 = curve(clamp(u0, 0, 127));
360 Expr y1 = curve(clamp(u0 + 1, 0, 127));
361 curved(x, y, c) = cast<uint8_t>((cast<uint16_t>(y0) * lutResample + (y1 - y0) * u) / lutResample);
362 }
363
364 return curved;
365}

367Func CameraPipe::sharpen(Func input) {
368 // Convert the sharpening strength to 2.5 fixed point. This allows sharpening in the range [0, 4].
369 Func sharpen_strength_x32("sharpen_strength_x32");
370 sharpen_strength_x32() = u8_sat(sharpen_strength * 32);
371 if (!auto_schedule) {
372 sharpen_strength_x32.compute_root();
373 if (get_target().has_gpu_feature()) {
374 sharpen_strength_x32.gpu_single_thread();
375 }
376 }
377
378 /* Optional tags to specify layout for HalideTraceViz */
379 {
380 Halide::Trace::FuncConfig cfg;
381 cfg.labels = {{"sharpen strength"}};
382 cfg.pos = {10, 1000};
383 sharpen_strength_x32.add_trace_tag(cfg.to_trace_tag());
384 }
385
386 // Make an unsharp mask by blurring in y, then in x.
387 Func unsharp_y("unsharp_y");
388 unsharp_y(x, y, c) = blur121(input(x, y - 1, c), input(x, y, c), input(x, y + 1, c));
389
390 Func unsharp("unsharp");
391 unsharp(x, y, c) = blur121(unsharp_y(x - 1, y, c), unsharp_y(x, y, c), unsharp_y(x + 1, y, c));
392
393 Func mask("mask");
394 mask(x, y, c) = cast<int16_t>(input(x, y, c)) - cast<int16_t>(unsharp(x, y, c));
395
396 // Weight the mask with the sharpening strength, and add it to the
397 // input to get the sharpened result.
398 Func sharpened("sharpened");
399 sharpened(x, y, c) = u8_sat(input(x, y, c) + (mask(x, y, c) * sharpen_strength_x32()) / 32);
400
401 return sharpened;
402}

404void CameraPipe::generate() {
405 // shift things inwards to give us enough padding on the
406 // boundaries so that we don't need to check bounds. We're going
407 // to make a 2560x1920 output image, just like the FCam pipe, so
408 // shift by 16, 12. We also convert it to be signed, so we can deal
409 // with values that fall below 0 during processing.
410 Func shifted;
411 shifted(x, y) = cast<int16_t>(input(x + 16, y + 12));
412
413 Func denoised = hot_pixel_suppression(shifted);
414
415 Func deinterleaved = deinterleave(denoised);
416
417 auto demosaiced = create<Demosaic>();
418 demosaiced->apply(deinterleaved);
419
420 Func corrected = color_correct(demosaiced->output);
421
422 Func curved = apply_curve(corrected);
423
424 processed(x, y, c) = sharpen(curved)(x, y, c);
425
426 /* ESTIMATES */
427 // (This can be useful in conjunction with RunGen and benchmarks as well
428 // as auto-schedule, so we do it in all cases.)
429 input.set_estimates({{0, 2592}, {0, 1968}});
430 matrix_3200.set_estimates({{0, 4}, {0, 3}});
431 matrix_7000.set_estimates({{0, 4}, {0, 3}});
432 color_temp.set_estimate(3700);
433 gamma.set_estimate(2.0);
434 contrast.set_estimate(50);
435 sharpen_strength.set_estimate(1.0);
436 blackLevel.set_estimate(25);
437 whiteLevel.set_estimate(1023);
438 processed.set_estimates({{0, 2592}, {0, 1968}, {0, 3}});
439
440 // Schedule
441 if (auto_schedule) {
442 // nothing
443 } else if (get_target().has_gpu_feature()) {
444
445 // We can generate slightly better code if we know the output is even-sized
446 if (!auto_schedule) {
447 // TODO: The autoscheduler really ought to be able to
448 // accommodate bounds on the output Func.
449 Expr out_width = processed.width();
450 Expr out_height = processed.height();
451 processed.bound(c, 0, 3)
452 .bound(x, 0, (out_width / 2) * 2)
453 .bound(y, 0, (out_height / 2) * 2);
454 }
455
456 Var xi, yi, xii, xio;
457
458 /* These tile factors obtain 1391us on a gtx 980. */
459 int tile_x = 28;
460 int tile_y = 12;
461
462 if (get_target().has_feature(Target::D3D12Compute)) {
463 // D3D12 SM 5.1 can only utilize a limited amount of
464 // shared memory, so we use a slightly smaller
465 // tile size.
466 tile_x = 20;
467 tile_y = 12;
468 }
469
470 processed.compute_root()
471 .reorder(c, x, y)
472 .unroll(x, 2)
473 .gpu_tile(x, y, xi, yi, tile_x, tile_y);
474
475 curved.compute_at(processed, x)
476 .unroll(x, 2)
477 .gpu_threads(x, y);
478
479 corrected.compute_at(processed, x)
480 .unroll(x, 2)
481 .gpu_threads(x, y);
482
483 demosaiced->output_compute_at.set({processed, x});
484 demosaiced->intermed_compute_at.set({processed, x});
485
486 denoised.compute_at(processed, x)
487 .tile(x, y, xi, yi, 2, 2)
488 .unroll(xi)
489 .unroll(yi)
490 .gpu_threads(x, y);
491
492 deinterleaved.compute_at(processed, x)
493 .unroll(x, 2)
494 .gpu_threads(x, y)
495 .reorder(c, x, y)
496 .unroll(c);
497
498 } else {
499
500 Expr out_width = processed.width();
501 Expr out_height = processed.height();
502
503 // Depending on the HVX generation, we need 2 or 4 threads
504 // to saturate HVX with work. For simplicity, we'll just
505 // stick to 4 threads. On balance, the overhead should
506 // not be much for the 2 extra threads that we create
507 // on cores that have only two HVX contexts.
508 Expr strip_size;
509 if (get_target().has_feature(Target::HVX)) {
510 strip_size = processed.dim(1).extent() / 4;
511 } else {
512 strip_size = 32;
513 }
514 strip_size = (strip_size / 2) * 2;
515
516 int vec = get_target().natural_vector_size(UInt(16));
517 if (get_target().has_feature(Target::HVX)) {
518 vec = 64;
519 }
520 processed
521 .compute_root()
522 .reorder(c, x, y)
523 .split(y, yi, yii, 2, TailStrategy::RoundUp)
524 .split(yi, yo, yi, strip_size / 2)
525 .vectorize(x, 2 * vec, TailStrategy::RoundUp)
526 .unroll(c)
527 .parallel(yo);
528
529 denoised
530 .compute_at(processed, yi)
531 .store_at(processed, yo)
532 .prefetch(input, y, 2)
533 .fold_storage(y, 4)
534 .tile(x, y, x, y, xi, yi, 2 * vec, 2)
535 .vectorize(xi)
536 .unroll(yi);
537
538 deinterleaved
539 .compute_at(processed, yi)
540 .store_at(processed, yo)
541 .fold_storage(y, 4)
542 .reorder(c, x, y)
543 .vectorize(x, 2 * vec, TailStrategy::RoundUp)
544 .unroll(c);
545
546 curved
547 .compute_at(processed, yi)
548 .store_at(processed, yo)
549 .reorder(c, x, y)
550 .tile(x, y, x, y, xi, yi, 2 * vec, 2, TailStrategy::RoundUp)
551 .vectorize(xi)
552 .unroll(yi)
553 .unroll(c);
554
555 corrected
556 .compute_at(curved, x)
557 .reorder(c, x, y)
558 .vectorize(x)
559 .unroll(c);
560
561 demosaiced->intermed_compute_at.set({processed, yi});
562 demosaiced->intermed_store_at.set({processed, yo});
563 demosaiced->output_compute_at.set({curved, x});
564
565 if (get_target().has_feature(Target::HVX)) {
566 processed.hexagon();
567 denoised.align_storage(x, vec);
568 deinterleaved.align_storage(x, vec);
569 corrected.align_storage(x, vec);
570 }
571
572 // We can generate slightly better code if we know the splits divide the extent.
573 processed
574 .bound(c, 0, 3)
575 .bound(x, 0, ((out_width) / (2 * vec)) * (2 * vec))
576 .bound(y, 0, (out_height / strip_size) * strip_size);
577
578 /* Optional tags to specify layout for HalideTraceViz */
579 {
580 Halide::Trace::FuncConfig cfg;
581 cfg.max = 1024;
582 cfg.pos = {10, 348};
583 cfg.labels = {{"input"}};
584 input.add_trace_tag(cfg.to_trace_tag());
585
586 cfg.pos = {305, 360};
587 cfg.labels = {{"denoised"}};
588 denoised.add_trace_tag(cfg.to_trace_tag());
589
590 cfg.pos = {580, 120};
591 const int y_offset = 220;
592 cfg.strides = {{1, 0}, {0, 1}, {0, y_offset}};
593 cfg.labels = {
594 {"gr", {0, 0 * y_offset}},
595 {"r", {0, 1 * y_offset}},
596 {"b", {0, 2 * y_offset}},
597 {"gb", {0, 3 * y_offset}},
598 };
599 deinterleaved.add_trace_tag(cfg.to_trace_tag());
600
601 cfg.color_dim = 2;
602 cfg.strides = {{1, 0}, {0, 1}, {0, 0}};
