20int main(int argc, char **argv) {

31 fprintf(stderr, "input: %s\n", argv[1]);
32 Buffer<uint16_t> input = load_and_convert_image(argv[1]);
33 fprintf(stderr, " %d %d\n", input.width(), input.height());
34 Buffer<uint8_t> output(((input.width() - 32) / 32) * 32, ((input.height() - 24) / 32) * 32, 3);
35
36#ifdef HL_MEMINFO
37 info(input, "input");
38 stats(input, "input");
39 // dump(input, "input");

42 // These color matrices are for the sensor in the Nokia N900 and are
43 // taken from the FCam source.
44 float _matrix_3200[][4] = {{1.6697f, -0.2693f, -0.4004f, -42.4346f},
45 {-0.3576f, 1.0615f, 1.5949f, -37.1158f},
46 {-0.2175f, -1.8751f, 6.9640f, -26.6970f}};
47
48 float _matrix_7000[][4] = {{2.2997f, -0.4478f, 0.1706f, -39.0923f},
49 {-0.3826f, 1.5906f, -0.2080f, -25.4311f},
50 {-0.0888f, -0.7344f, 2.2832f, -20.0826f}};
51 Buffer<float> matrix_3200(4, 3), matrix_7000(4, 3);
52 for (int i = 0; i < 3; i++) {
53 for (int j = 0; j < 4; j++) {
54 matrix_3200(j, i) = _matrix_3200[i][j];
55 matrix_7000(j, i) = _matrix_7000[i][j];
56 }
57 }
58
59 float color_temp = (float)atof(argv[2]);
60 float gamma = (float)atof(argv[3]);
61 float contrast = (float)atof(argv[4]);
62 float sharpen = (float)atof(argv[5]);
63 int timing_iterations = atoi(argv[6]);
64 int blackLevel = 25;
65 int whiteLevel = 1023;
66
67 double best;
68
69 best = benchmark(timing_iterations, 1, [&]() {
70 camera_pipe(input, matrix_3200, matrix_7000,
71 color_temp, gamma, contrast, sharpen, blackLevel, whiteLevel,
72 output);
73 output.device_sync();
74 });
75 fprintf(stderr, "Halide (manual):\t%gus\n", best * 1e6);
76
77#ifndef NO_AUTO_SCHEDULE
78 best = benchmark(timing_iterations, 1, [&]() {
79 camera_pipe_auto_schedule(input, matrix_3200, matrix_7000,
80 color_temp, gamma, contrast, sharpen, blackLevel, whiteLevel,
81 output);
82 output.device_sync();
83 });
84 fprintf(stderr, "Halide (auto):\t%gus\n", best * 1e6);
85#endif
86
87 fprintf(stderr, "output: %s\n", argv[7]);
88 convert_and_save_image(output, argv[7]);
89 fprintf(stderr, " %d %d\n", output.width(), output.height());
90
91 printf("Success!\n");
92 return 0;
93}
