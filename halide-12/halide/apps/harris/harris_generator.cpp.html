7Expr sum3x3(Func f, Var x, Var y) {
8 return f(x - 1, y - 1) + f(x - 1, y) + f(x - 1, y + 1) +
9 f(x, y - 1) + f(x, y) + f(x, y + 1) +
10 f(x + 1, y - 1) + f(x + 1, y) + f(x + 1, y + 1);
11}
12
13class Harris : public Halide::Generator<Harris> {
14public:
15 Input<Buffer<float>> input{"input", 3};
16 Output<Buffer<float>> output{"output", 2};
17
18 void generate() {
19 Var x("x"), y("y"), c("c");
20
21 // Algorithm
22 Func gray("gray");
23 gray(x, y) = (0.299f * input(x, y, 0) +
24 0.587f * input(x, y, 1) +
25 0.114f * input(x, y, 2));
26
27 Func Iy("Iy");
28 Iy(x, y) = gray(x - 1, y - 1) * (-1.0f / 12) + gray(x - 1, y + 1) * (1.0f / 12) +
29 gray(x, y - 1) * (-2.0f / 12) + gray(x, y + 1) * (2.0f / 12) +
30 gray(x + 1, y - 1) * (-1.0f / 12) + gray(x + 1, y + 1) * (1.0f / 12);
31
32 Func Ix("Ix");
33 Ix(x, y) = gray(x - 1, y - 1) * (-1.0f / 12) + gray(x + 1, y - 1) * (1.0f / 12) +
34 gray(x - 1, y) * (-2.0f / 12) + gray(x + 1, y) * (2.0f / 12) +
35 gray(x - 1, y + 1) * (-1.0f / 12) + gray(x + 1, y + 1) * (1.0f / 12);
36
37 Func Ixx("Ixx");
38 Ixx(x, y) = Ix(x, y) * Ix(x, y);
39
40 Func Iyy("Iyy");
41 Iyy(x, y) = Iy(x, y) * Iy(x, y);
42
43 Func Ixy("Ixy");
44 Ixy(x, y) = Ix(x, y) * Iy(x, y);
45
46 Func Sxx("Sxx");
47 Sxx(x, y) = sum3x3(Ixx, x, y);
48
49 Func Syy("Syy");
50 Syy(x, y) = sum3x3(Iyy, x, y);
51
52 Func Sxy("Sxy");
53 Sxy(x, y) = sum3x3(Ixy, x, y);
54
55 Func det("det");
56 det(x, y) = Sxx(x, y) * Syy(x, y) - Sxy(x, y) * Sxy(x, y);
57
58 Func trace("trace");
59 trace(x, y) = Sxx(x, y) + Syy(x, y);
60
61 output(x, y) = det(x, y) - 0.04f * trace(x, y) * trace(x, y);
62
63 // Estimates (for autoscheduler; ignored otherwise)
64 {
65 const int kWidth = 1536;
66 const int kHeight = 2560;
67 input.dim(0).set_estimate(0, kWidth);
68 input.dim(1).set_estimate(0, kHeight);
69 input.dim(2).set_estimate(0, 3);
70 output.dim(0).set_estimate(3, kWidth - 6);
71 output.dim(1).set_estimate(3, kHeight - 6);
72 }
73
74 // Schedule
75 if (!auto_schedule) {
76 Var xi("xi"), yi("yi");
77 if (get_target().has_gpu_feature()) {
78 // 0.253ms on a 2060 RTX
79 output.gpu_tile(x, y, xi, yi, 62, 14)
80 .unroll(xi, 2);
81 gray.compute_at(output, x)
82 .gpu_threads(x, y)
83 .tile(x, y, xi, yi, 3, 2)
84 .unroll(xi)
85 .unroll(yi);
86 gray.in()
87 .compute_at(Iy, x)
88 .vectorize(x, 2)
89 .unroll(x)
90 .unroll(y);
91 Ix.compute_at(output, x)
92 .gpu_threads(x, y)
93 .unroll(x, 2);
94 Iy.compute_at(output, x)
95 .gpu_threads(x, y)
96 .unroll(x, 2);
97 Ix.compute_with(Iy, x);
98 Ix.in()
99 .compute_at(output, xi)
100 .vectorize(x, 2)
101 .unroll(x)
102 .unroll(y);
103 Iy.in()
104 .compute_at(output, xi)
105 .vectorize(x, 2)
106 .unroll(x)
107 .unroll(y);
108 } else {
109 // 0.92ms on an Intel i9-9960X using 16 threads
110 const int vec = natural_vector_size<float>();
111 output.split(y, y, yi, 32)
112 .parallel(y)
113 .vectorize(x, vec);
114 gray.store_at(output, y)
115 .compute_at(output, yi)
116 .vectorize(x, vec);
117 Ix.store_at(output, y)
118 .compute_at(output, yi)
119 .vectorize(x, vec);
120 Iy.store_at(output, y)
121 .compute_at(output, yi)
122 .vectorize(x, vec);
123 Ix.compute_with(Iy, x);
124 }
125 }
126 }
127};
