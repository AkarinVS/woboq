9template<typename T>
10bool ParseValue(const char *in, T &result) {
11 std::istringstream stream(in);
12 stream >> result;
13 if (!stream.eof() && !stream.good()) {
14 return false;
15 }
16 return true;
17}

19bool ParseOptions(char **options_keys,
20 char **options_values,
21 size_t num_options,
22 HannkDelegateOptions *options) {
23 HannkDelegateOptionsDefault(options);
24
25 for (size_t i = 0; i < num_options; ++i) {
26 if (!strcmp(options_keys[i], "verbosity")) {
27 if (!ParseValue(options_values[i], options->verbosity)) {
28 HLOG(WARNING) << "ParseOptions: malformed option " << options_keys[i] << "\n";
29 return false;
30 }
31 } else {
32 HLOG(WARNING) << "ParseOptions: unknown option " << options_keys[i] << "\n";
33 return false;
34 }
35 }
36
37 return true;
38}

46// Defines two symbols that need to be exported to use the TFLite external
47// delegate. See tensorflow/lite/delegates/external for details.
48TFL_CAPI_EXPORT TfLiteDelegate *tflite_plugin_create_delegate(char **options_keys,
49 char **options_values,
50 size_t num_options,
51 void (*report_error)(const char *)) {
52 HannkDelegateOptions options;
53 if (!ParseOptions(options_keys, options_values, num_options, &options)) {
54 return nullptr;
55 }
56
57 if (options.verbosity >= 1) {
58 HLOG(INFO) << "External HannkDelegate: verbosity set to "
59 << options.verbosity << ".";
60 }
61
62 return HannkDelegateCreate(&options);
63}
