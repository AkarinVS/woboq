1// A collection of utility functions shared by the halide generators.
2
3#ifndef HANNK_COMMON_HALIDE_H
4#define HANNK_COMMON_HALIDE_H
5
6#include "Halide.h"
7
8namespace hannk {
9
10using Halide::Internal::rounding_shift_right;
11using Halide::Internal::saturating_add;
12using Halide::Internal::saturating_sub;
13using Halide::Internal::widening_add;
14using Halide::Internal::widening_mul;
15using Halide::Internal::widening_sub;
16
17// Get the number of vector registers available on the target.
18int get_register_count(const Halide::Target &target);
19
20// Get the vector reduction factor that is convenient for this target
21// for the given type.
22int get_vector_reduction_factor(const Halide::Target &target, Halide::Type t);
23
24// A tensor has the same requirements as a buffer in Halide by default, except
25// the min of the innermost dimension must also be 0.
26void interpret_as_tensor(Halide::OutputImageParam p);
27
28// Require dimension dim have the same min and extent.
29void require_same_min_extent(int dim, Halide::OutputImageParam first, Halide::OutputImageParam second);
30
31Halide::Expr is_interleaved(Halide::OutputImageParam p, int channels);
32
33// Round x down or up to the nearest multiple of n.
34Halide::Expr align_down(const Halide::Expr &x, const Halide::Expr &n);
35Halide::Expr align_up(const Halide::Expr &x, const Halide::Expr &n);
36Halide::Expr align(const Halide::Expr &x, const Halide::Expr &n);
37
38// Compute saturating_narrow(rounding_shift_right(widening_mul(a, b), N))
39// where N is the number of bits of the narrowed result minus one.
40Halide::Expr multiply_2x_high(const Halide::Expr &a, const Halide::Expr &b);
41
42// Approximate log2(x/2^q_x)*2^q.
43// q must be less than 16.
44Halide::Expr approx_log2(int q, const Halide::Expr &x, int q_x, const Halide::Type &type = Halide::Int(32));
45
46// Approximate 2^(x/2^q_x)/2^q.
47// q_x must be less than 16.
48Halide::Expr approx_exp2(int q, const Halide::Expr &x, const Halide::Expr &q_x, const Halide::Type &type = Halide::Int(32));
49
50// Approximate 2^q/x
51Halide::Expr approx_reciprocal(int q, const Halide::Expr &x, const Halide::Type &type = Halide::Int(32));
52
53// Approximate 2^q/sqrt(x)
54Halide::Expr approx_reciprocal_sqrt(int q, const Halide::Expr &x, const Halide::Type &type = Halide::Int(32));
55
56// Approximate 2^q*log2(2^(x/2^q_x) +/- 1)
57Halide::Expr approx_log2p1_exp2(int q, const Halide::Expr &x, const Halide::Expr &q_x, const Halide::Type &type = Halide::Int(32));
58Halide::Expr approx_log2m1_exp2(int q, const Halide::Expr &x, const Halide::Expr &q_x, const Halide::Type &type = Halide::Int(32));
59
60// Approximate 2^q/(1 + 2^(-x/2^q_x))
61Halide::Expr approx_logistic(int q, const Halide::Expr &x, const Halide::Expr &q_x, const Halide::Type &type = Halide::Int(32));
62// Approximate 2^q*tanh(x/2^q_x)
63Halide::Expr approx_tanh(int q, const Halide::Expr &x, const Halide::Expr &q_x, const Halide::Type &type = Halide::Int(32));
64
65} // namespace hannk
