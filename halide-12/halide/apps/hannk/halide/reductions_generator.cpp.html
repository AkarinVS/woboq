7namespace hannk {
8
9class Mean : public Generator<Mean> {
10public:
11 Input<Buffer<uint8_t>> input_{"input", 4};
12
13 // The bounds of the region to reduce. This pipeline is
14 // implemented as a stencil over this reach at each output.
15 // However, the expected usage is to have either the output
16 // extent be 1, or the reduction extent be 1.
17 Input<int> c_min_{"c_min"};
18 Input<int> c_extent_{"c_extent"};
19 Input<int> x_min_{"x_min"};
20 Input<int> x_extent_{"x_extent"};
21 Input<int> y_min_{"y_min"};
22 Input<int> y_extent_{"y_extent"};
23 Input<int> b_min_{"b_min"};
24 Input<int> b_extent_{"b_extent"};
25
26 Output<Buffer<uint8_t>> output_{"output", 4};
27
28 void generate() {
29 // The algorithm.
30 Var c("c"), x("x"), y("y"), b("b");
31
32 Func sum("sum");
33 RDom r(c_min_, c_extent_, x_min_, x_extent_, y_min_, y_extent_, b_min_, b_extent_);
34 sum(c, x, y, b) += u32(input_(c + r.x, x + r.y, y + r.z, b + r.w));
35
36 Expr extent = c_extent_ * x_extent_ * y_extent_ * b_extent_;
37 output_(c, x, y, b) = u8((sum(c, x, y, b) + extent / 2) / extent);
38
39 // Schedule.
40 output_.compute_root()
41 .vectorize(c, natural_vector_size<uint8_t>(), TailStrategy::GuardWithIf);
42
43 // If we aren't reducing c, just moving it out of the inner loop and
44 // vectorizing c is pretty good.
45 sum.compute_at(output_, x)
46 .update()
47 .reorder(r.y, r.z, r.x, r.w)
48 .vectorize(c, natural_vector_size<uint8_t>(), TailStrategy::GuardWithIf);
49
50 // TODO: If we want to reduce c, we should horizontally vectorize r.x.
51 }
52};
53
54} // namespace hannk
