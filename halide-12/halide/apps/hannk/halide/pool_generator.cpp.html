10class AveragePool : public Generator<AveragePool> {
11public:
12 // Unsigned 8-bit input tensor, indexed by c, x, y, b.
13 Input<Buffer<uint8_t>> input_{"input", 4};
14
15 // The stride specifies how the input [x, y] are sub-subsampled. For every
16 // spatial location [x, y] in the output buffer, the input buffer is sampled
17 // spatially at [x * stride, y * stride].
18 Input<int> stride_x_{"stride_x"};
19 Input<int> stride_y_{"stride_y"};
20 Input<int> filter_width_{"filter_width"};
21 Input<int> filter_height_{"filter_height"};
22
23 Input<uint8_t> output_min_{"output_min"};
24 Input<uint8_t> output_max_{"output_max"};
25
26 Output<Buffer<uint8_t>> output_{"output", 4};
27
28 void generate() {
29 // The algorithm.
30 Var c("c"), x("x"), y("y"), b("b");
31
32 Expr min_x = input_.dim(1).min();
33 Expr max_x = input_.dim(1).max();
34 Expr min_y = input_.dim(2).min();
35 Expr max_y = input_.dim(2).max();
36
37 // This pipeline conceptually requires a zero padding boundary condition.
38 // However, zero padding is messy. To avoid this, we'll just use a clamp to
39 // avoid out of bounds reads, and then use 'where' on the RDom to avoid
40 // including these values in the reduction.
41 Func input_bounded("input_bounded");
42 input_bounded(c, x, y, b) =
43 input_(c, clamp(x, min_x, max_x), clamp(y, min_y, max_y), b);
44
45 RDom r(0, filter_width_, 0, filter_height_);
46 Expr x_rx = x * stride_x_ + r.x;
47 Expr y_ry = y * stride_y_ + r.y;
48 r.where(min_x <= x_rx && x_rx <= max_x && min_y <= y_ry && y_ry <= max_y);
49
50 // Accumulating in 16 bits limits filter_width * filter_height <= 256.
51 Func sum("sum");
52 sum(c, x, y, b) += u16(input_bounded(c, x_rx, y_ry, b));
53
54 // TODO: We should probably specialize/optimize for the case
55 // where filter_count = filter_width * filter_height.
56 Expr x_start = max(x * stride_x_, min_x);
57 Expr x_end = min(x * stride_x_ + filter_width_, max_x + 1);
58 Expr y_start = max(y * stride_y_, min_y);
59 Expr y_end = min(y * stride_y_ + filter_height_, max_y + 1);
60 Expr filter_count = (x_end - x_start) * (y_end - y_start);
61 // We assume here that filter_count is not greater than 256 above.
62 // This means that we can compute the result to within 1 bit by using an
63 // integer reciprocal of 16 bits. This reciprocal can be computed once for
64 // each value of (x, y).
65 const int log2_numerator = 16;
66 // Compute (2*2^log2_numerator + filter_count) / (2 * filter_count) to avoid
67 // error in the rounding term.
68 Expr inv_filter_count =
69 u16_sat(((2 << log2_numerator) + filter_count) / (2 * filter_count));
70 Expr average =
71 rounding_mul_shift_right(sum(c, x, y, b), inv_filter_count, log2_numerator);
72
73 output_(c, x, y, b) = clamp(u8_sat(average), output_min_, output_max_);
74
75 // Schedule.
76 require_same_min_extent(0, input_, output_);
77 require_same_min_extent(3, input_, output_);
78
79 // Reorder b inside x so inv_filter_count can be computed outside
80 // that loop.
81 output_.compute_root()
82 .reorder(c, b, x, y);
83
84 // TODO: Figure out how to vectorize this efficiently without this
85 // code duplication. We should be able to just vectorize and predicate
86 // somehow.
87 const int vector_size = natural_vector_size<uint8_t>();
88 Expr output_channels = output_.dim(0).extent();
89 for (int i : {4, 2, 1}) {
90 output_.specialize(output_channels >= vector_size * i)
91 .vectorize(c, vector_size * i, TailStrategy::ShiftInwards);
92 }
93 }
94};

96class MaxPool : public Generator<MaxPool> {
97public:
98 // Unsigned 8-bit input tensor, indexed by c, x, y, b.
99 Input<Buffer<uint8_t>> input_{"input", 4};
100
101 // The stride specifies how the input [x, y] are sub-subsampled. For every
102 // spatial location [x, y] in the output buffer, the input buffer is sampled
103 // spatially at [x * stride, y * stride].
104 Input<int> stride_x_{"stride_x"};
105 Input<int> stride_y_{"stride_y"};
106 Input<int> filter_width_{"filter_width"};
107 Input<int> filter_height_{"filter_height"};
108
109 Input<uint8_t> output_min_{"output_min"};
110 Input<uint8_t> output_max_{"output_max"};
111
112 Output<Buffer<uint8_t>> output_{"output", 4};
113
114 void generate() {
115 // The algorithm.
116 Var c("c"), x("x"), y("y"), b("b");
117
118 Expr min_x = input_.dim(1).min();
119 Expr max_x = input_.dim(1).max();
120 Expr min_y = input_.dim(2).min();
121 Expr max_y = input_.dim(2).max();
122
123 Func input_bounded("input_bounded");
124 input_bounded(c, x, y, b) =
125 input_(c, clamp(x, min_x, max_x), clamp(y, min_y, max_y), b);
126
127 Func maximum("maximum");
128 RDom r(0, filter_width_, 0, filter_height_);
129 Expr x_rx = x * stride_x_ + r.x;
130 Expr y_ry = y * stride_y_ + r.y;
131 // Unlike pools that sum the input, we can use a clamp boundary condition
132 // here. However, it still seems faster to include this where clause.
133 r.where(min_x <= x_rx && x_rx <= max_x && min_y <= y_ry && y_ry <= max_y);
134 maximum(c, x, y, b) = output_min_;
135 maximum(c, x, y, b) = max(maximum(c, x, y, b), input_bounded(c, x_rx, y_ry, b));
136
137 output_(c, x, y, b) = min(maximum(c, x, y, b), output_max_);
138
139 // Schedule.
140 require_same_min_extent(0, input_, output_);
141 require_same_min_extent(3, input_, output_);
142
143 output_.compute_root();
144
145 // TODO: Figure out how to vectorize this efficiently without this
146 // code duplication. We should be able to just vectorize and predicate
147 // somehow.
148 const int vector_size = natural_vector_size<uint8_t>();
149 Expr output_channels = output_.dim(0).extent();
150 for (int i : {4, 2, 1}) {
151 output_.specialize(output_channels >= vector_size * i)
152 .vectorize(c, vector_size * i, TailStrategy::ShiftInwards);
153 }
154 }
155};
