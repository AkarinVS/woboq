8int get_register_count(const Target &target) {
9 switch (target.arch) {
10 case Target::X86:
11 return target.features_any_of({Target::AVX512_Skylake, Target::AVX512_Cannonlake, Target::AVX512_SapphireRapids}) ? 32 : 16;
12 case Target::ARM:
13 return target.bits == 64 ? 32 : 16;
14 case Target::Hexagon:
15 return 32;
16 default:
17 return 16;
18 }
19}

21int get_vector_reduction_factor(const Target &target, Type t) {

23 target.has_feature(Target::ARMDotProd) ||

25 return 32 / t.bits();

29 return 2;

63Expr floor_log2(const Expr &x) {
64 // floor(log2(x)) = B - clz(x) => log2(x) ~ B - clz(x)
65 // B = sizeof(x)*8 - 1
66 // clz(x) = count_leading_zeros(x)
67 int log2_max_x = x.type().bits() - 1;
68 return log2_max_x - i16(count_leading_zeros(x));
69}

71Expr approx_log2(int q, const Expr &x, int q_x, const Type &type) {
72 Expr floor_log2_x = floor_log2(x);
73
74 // Use a cubic polynomial to interpolate the fractional part of the result.
75 // TODO: A cubic might be overkill for our needs.
76 // Coefficients produced by the following numpy snippet:
77 //
78 // points = 6
79 // poly_x = np.arange(points, 2 * points + 1) / points
80 // poly_y = np.log2(poly_x)
81 // p = np.polyfit(poly_x - 1, poly_y, 3)
82 //
83 // Quantize to 14 bits so the polynomial evaluation fits in 15 bits.
84 const int poly_bits = 14;
85 const int p3 = std::lround(1.55971251e-01 * (1 << poly_bits));
86 const int p2 = std::lround(-5.75039427e-01 * (1 << poly_bits));
87 const int p1 = std::lround(1.41903642e+00 * (1 << poly_bits));
88 const int p0 = std::lround(3.32891346e-04 * (1 << poly_bits));
89
90 Expr frac1 = i16(x << (15 - floor_log2_x)) & 0x7fff;
91 Expr frac2 = multiply_2x_high(frac1, frac1);
92 Expr frac3 = multiply_2x_high(frac2, frac1);
93
94 // TODO: On ARM, these polynomial coefficients each get broadcasted into their
95 // own register. But, we could be using the "lane" version of the qrdmulh
96 // instruction, and put all of the coefficients into one vector register. This
97 // would reduce register pressure, which is very high in code using this helper.
98 Expr poly =
99 multiply_2x_high(i16(p3), frac3) +
100 multiply_2x_high(i16(p2), frac2) +
101 multiply_2x_high(i16(p1), frac1) +
102 p0;
103 Expr frac_result;
104 if (q < poly_bits) {
105 frac_result = cast(type, rounding_shift_right(poly, poly_bits - q));
106 } else {
107 frac_result = cast(type, poly) << (q - poly_bits);
108 }
109
110 // We've computed log2(x*2^q_x) = log2(x) + q_x. Subtract
111 // that offset now, before we scale up the output.
112 Expr floor_result = cast(type, floor_log2_x - q_x) << q;
113
114 return saturating_add(floor_result, frac_result);
115}

117Expr approx_exp2(int q, const Expr &x, const Expr &q_x, const Type &type) {
118 // Compute floor(x / precision_x) and frac(x / precision_x)
119 Expr floor_x = cast(type, x >> q_x);
120
121 Expr exp2_floor_x = saturating_cast(type, 1 << (floor_x + q));
122
123 // Use a cubic polynomial to interpolate the fractional part of the argument.
124 // TODO: A cubic might be overkill for our needs.
125 // Coefficients produced by the following numpy snippet:
126 //
127 // points = 6
128 // poly_x = np.arange(points, 2 * points + 1) / points
129 // poly_y = np.exp2(poly_x - 1) - 1
130 // p = np.polyfit(poly_x - 1, poly_y, 3)
131 //
132 // We ignore the constant term from the polynomial.
133 const int p3 = std::lround(7.91076597e-02 * (1 << 15));
134 const int p2 = std::lround(2.24701130e-01 * (1 << 15));
135 const int p1 = std::lround(6.96189819e-01 * (1 << 15)) - 1; // Hack to avoid overflow below.
136
137 Expr frac1 = i16(x - (floor_x << q_x)) << (15 - i16(q_x));
138 Expr frac2 = multiply_2x_high(frac1, frac1);
139 Expr frac3 = multiply_2x_high(frac2, frac1);
140
141 // TODO: On ARM, these polynomial coefficients each get broadcasted into their
142 // own register. But, we could be using the "lane" version of the qrdmulh
143 // instruction, and put all of the coefficients into one vector register. This
144 // would reduce register pressure, which is very high in code using this helper.
145 assert(p1 + p2 + p3 < (1 << 15));
146 Expr poly =
147 multiply_2x_high(i16(p3), frac3) +
148 multiply_2x_high(i16(p2), frac2) +
149 multiply_2x_high(i16(p1), frac1);
150 poly = cast(type, poly) << (type.bits() - 16);
151
152 return saturating_add(exp2_floor_x, multiply_2x_high(exp2_floor_x, poly));
153}

155Expr approx_reciprocal(int q, const Expr &x, const Type &type) {
156 // precision / x
157 // = precision / 2^log2(x)
158 // = precision * 2^(-log2(x))
159 Expr log2_x = approx_log2(15, x, 0);
160 return approx_exp2(q, -log2_x, 15, type);
161}

163Expr approx_reciprocal_sqrt(int q, const Expr &x, const Type &type) {
164 // precision / sqrt(x)
165 // = precision / 2^log2(x^(1/2))
166 // = precision * 2^(-log2(x)/2)
167 Expr log2_x = approx_log2(14, x, 0);
168 return approx_exp2(q, -log2_x, 15, type);
169}

171// TODO: These implementations are pretty slow, at least on x86. However:
172// - They are readily implementable on every target
173// - Produce identical results on every target
174// - Avoid the use of lookup tables, which can be annoying on some targets
175// - Negligibly impact overall performance in most realistic workloads
176
177// Approximate log2(2^(x/2^q) +/- 1)*2^q
178Expr approx_log2_exp2_plus_or_minus_one(int q, Expr x, int sign, Expr q_x, Type type) {
179 // TODO: Try to make this intermediate fit in 16 bits.
180 const int q_exp = 16;
181 int one = sign << q_exp;
182 Expr one_plus_exp2_x = one + approx_exp2(q_exp, x, q_x, Int(32));
183 Expr raw = approx_log2(q, one_plus_exp2_x, q_exp, type);
184
185 // For large x, the intermediate overflows. But log2(1 + 2^x) when x is large is just x.
186 const int threshold = 30 - q_exp;
187 Expr line = saturating_cast(type, rounding_shift_right(cast(type.widen(), x), i16(q_x) - q));
188 return select((x >> q_x) < threshold, raw, line);
189}

191Expr approx_log2p1_exp2(int q, const Expr &x, const Expr &q_x, const Type &type) {
192 return approx_log2_exp2_plus_or_minus_one(q, x, 1, q_x, type);
193}

195Expr approx_log2m1_exp2(int q, const Expr &x, const Expr &q_x, const Type &type) {

197}

201Expr approx_logistic(int q, const Expr &x, const Expr &q_x, const Type &type) {
202 // log2(e) is ~1.5, so to implement this, we quantize log2(e)/2, and adjust
203 // q_x to compensate.
204 const int log2_e_q = std::lround(log2_e * (1 << (x.type().bits() - 2)));
205 Expr x2 = multiply_2x_high(x, cast(x.type(), -log2_e_q));
206
207 const int log_q = 11;
208 Expr log2_d = approx_log2p1_exp2(log_q, x2, q_x - 1, Int(16));
209 return approx_exp2(q, -log2_d, log_q, type);
210}

212Expr approx_tanh(int q, const Expr &x, const Expr &q_x, const Type &type) {
213 // log2(e) is ~1.5, so to implement this, we quantize log2(e)/2, and adjust
214 // q_x to compensate.
215 const int log2_e_q = std::lround(log2_e * (1 << (x.type().bits() - 2)));
216 Expr x2 = multiply_2x_high(x, cast(x.type(), log2_e_q));
217
218 const int log_q = 11;
219 Expr log2_n = approx_log2m1_exp2(log_q, i16(abs(x2)), q_x - 2, Int(16));
220 Expr log2_d = approx_log2p1_exp2(log_q, i16(abs(x2)), q_x - 2, Int(16));
221 Expr abs_output = approx_exp2(q, log2_n - log2_d, log_q, type);
222 return select(x2 < 0, -abs_output, x2 == 0, 0, abs_output);
223

225 // might be faster.
