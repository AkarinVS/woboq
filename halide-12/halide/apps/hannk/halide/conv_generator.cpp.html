13// There are two codepaths in this generator. On targets with widening
14// 8-bit multiplies, we implement the reduction by expanding the subtraction
15// of the offsets into 4 reductions involving 8-bit multiplies. On targets
16// without widening 8-bit multiplication, it's faster to just subtract the
17// offsets and use 16-bit multiplications.
18bool use_8bit_multiply(const Target &target) {
19 return target.arch != Target::X86 || target.has_feature(Target::AVX512_SapphireRapids);
20}
21
22// How many registers to use as accumulators, as a function of the target.
23int get_accumulator_count(const Target &target) {
24 if (target.has_feature(Target::HVX)) {
25 // Hexagon has dot products between vector and scalar registers, so
26 // we don't need to use any vector registers for the input, so we
27 // can use a lot of registers as accumulators without spilling to
28 // the stack.
29 return 24;
30 } else if (get_register_count(target) >= 32) {
31 return 20;
32 } else {
33 return 8;
34 }
35}

37class Conv : public Generator<Conv> {
38public:
39 // How much to unroll the reduction loop over channels. On some targets,
40 // loading a few scalars for one of the reduction inputs is fine, and avoids
41 // a large alignment requirement. However, on other targets, it is beneficial
42 // to load vectors, so making this value larger helps for big reductions.
43 GeneratorParam<int> unroll_reduction_{"unroll_reduction", 4};
44
45 // Unsigned 8-bit input tensor, indexed by c, x, y, b.
46 Input<Buffer<uint8_t>> input_{"input", 4};
47 Input<uint8_t> input_zero_{"input_zero"};
48
49 // A 6D array of filter coefficients indexed by ci % n, co % k, ci / n, co / k, x, y,
50 // where n = vector_reduction and k = accum_vector_size (below).
51 Input<Buffer<>> filter_{"filter", 6};
52 Input<uint8_t> filter_zero_{"filter_zero"};
53
54 // A 1D array of 32-bit biases. The bias should be added to the c
55 // dimension of the output.
56 Input<Buffer<int32_t>> bias_{"bias", 1};
57
58 // The stride specifies how the input [x, y] is sub-subsampled. For every
59 // spatial location [x, y] in the output buffer, the input buffer is sampled
60 // spatially at [x * stride, y * stride]. The caller is responsible for
61 // allocating the correct output memory.
62 Input<int> stride_x_{"stride_x"};
63 Input<int> stride_y_{"stride_y"};
64 Input<int> dilation_x_{"dilation_x"};
65 Input<int> dilation_y_{"dilation_y"};
66
67 Input<int32_t> output_multiplier_{"output_multiplier"};
68 Input<uint32_t> output_shift_{"output_shift"};
69 Input<uint8_t> output_zero_{"output_zero"};
70 Input<uint8_t> output_min_{"output_min"};
71 Input<uint8_t> output_max_{"output_max"};
72
73 Output<Buffer<uint8_t>> output_{"output", 4};
74
75 void configure() {
76 if (use_8bit_multiply(target)) {
77 filter_.set_type(UInt(8));
78 } else {
79 filter_.set_type(Int(16));
80 }
81 }
82
83 void generate() {
84 // The algorithm.
85 Func input("input_wrapper");
86 Expr input_cxyb = input_(c, x, y, b);
87 if (!use_8bit_multiply(target)) {
88 input_cxyb = i16(input_cxyb) - i16(input_zero_);
89 }
90 input(c, x, y, b) = input_cxyb;
91
92 // Align the reduction loop of filter.
93 const int vector_reduction = get_vector_reduction_factor(target, UInt(8));
94 const int unroll_reduction = std::max<int>(vector_reduction, unroll_reduction_);
95 const int accum_vector_size = natural_vector_size<int32_t>();
96
97 // Set up the reduction loop and inputs.
98 Expr filter_depth = filter_.dim(0).extent() * filter_.dim(2).extent();
99 Expr filter_width = filter_.dim(4).extent();
100 Expr filter_height = filter_.dim(5).extent();
101 // Align the filter depth, which requires padding the input.
102 filter_depth = align_up(filter_depth, unroll_reduction);
103 RDom r(0, filter_width, 0, filter_height, 0, filter_depth);
104 Expr filter_rdxyc =
105 filter_(r.z % vector_reduction, c % accum_vector_size, r.z / vector_reduction, c / accum_vector_size, r.x, r.y);
106 Expr input_rdxyc =
107 input(r.z, x * stride_x_ + r.x * dilation_x_, y * stride_y_ + r.y * dilation_y_, b);
108
109 Func offset_c("offset_c");
110 Func sum_input("sum_input");
111 Func convolved("convolved");
112 if (use_8bit_multiply(target)) {
113 // We want to compute the reduction:
114 // convolved(c, x, y, b) = bias_(c)
115 // convolved(c, x, y, b) +=
116 // (i32(input_rdxyc) - i32(input_zero_)) *
117 // (i32(filter_rdxyc) - i32(filter_zero_))
118 //
119 // However, this precludes using efficient dot product instructions. To
120 // fix this, expand the expression:
121 //
122 // convolved(c, x, y, b) = bias_(c)
123 // convolved(c, x, y, b) +=
124 // i32(filter_rdxyc) * i32(input_rdxyc) -
125 // i32(filter_rdxyc) * i32(input_zero_) -
126 // i32(filter_zero_) * i32(input_rdxyc) +
127 // i32(filter_zero_) * i32(input_zero_)
128 //
129 // We can then separate this into several reductions. First, the terms that
130 // depend only on c.
131 Expr r_size = filter_width * filter_height * filter_depth;
132 // We need the negative of this reduction, so compute the sum first, and then
133 // subtract it after.
134 offset_c(c) += i32(u16(filter_rdxyc) * u16(input_zero_));
135 offset_c(c) =
136 bias_(c) + i32(u16(filter_zero_) * u16(input_zero_)) * r_size - offset_c(c);
137
138 // The sum of the input is used to compute the filter_zero * input term.
139 // TODO: This is separable, but a bit messy to optimize this way.
140 sum_input(x, y, b) += i32(input_rdxyc);
141
142 // Finally, the terms that depend on all of c, x, y, b.
143 convolved(c, x, y, b) = offset_c(c) - i32(filter_zero_) * sum_input(x, y, b);
144 } else {
145 // Without 8-bit widening multiplies, we already subtracted the offsets,
146 // and just have a single reduction of 16-bit multiplies to compute.
147 convolved(c, x, y, b) = bias_(c);
148 }
149 convolved(c, x, y, b) += i32(input_rdxyc) * i32(filter_rdxyc);
150
151 // Saturate and narrow the output.
152 Expr output = multiply_2x_high(convolved(c, x, y, b), output_multiplier_);
153 output = i16_sat(rounding_shift_right(output, output_shift_));
154 output = u8_sat(saturating_add(output, output_zero_));
155 output_(c, x, y, b) = clamp(output, output_min_, output_max_);
156
157 // Schedule
158 interpret_as_tensor(input_);
159 interpret_as_tensor(bias_);
160 interpret_as_tensor(output_);
161 require_same_min_extent(3, input_, output_);
162 require_same_min_extent(0, bias_, output_);
163
164 const int filter_alignment = vector_reduction * accum_vector_size;
165 filter_.set_host_alignment(filter_alignment * filter_.type().bytes());
166 filter_.dim(0).set_min(0).set_extent(vector_reduction).set_stride(1);
167 filter_.dim(1).set_min(0).set_extent(accum_vector_size).set_stride(vector_reduction);
168 filter_.dim(2).set_min(0).set_stride(filter_alignment);
169 for (int d = 3; d < filter_.dimensions(); d++) {
170 filter_.dim(d).set_min(0).set_stride(align(filter_.dim(d).stride(), filter_alignment));
171 }
172
173 const int input_alignment = unroll_reduction;
174 input_.set_host_alignment(input_alignment);
175 input_.dim(0).set_min(0).set_extent(filter_depth);
176 for (int d = 1; d < input_.dimensions(); d++) {
177 input_.dim(d).set_stride(align(input_.dim(d).stride(), input_alignment));
178 }
179
180 output_.compute_root();
181
182 // Figure out how big the tiles we should optimize for should be by getting
183 // the total number of accumulators best for this target and figuring out
184 // tile sizes.
185 const int accumulators = get_accumulator_count(target);
186 std::vector<std::pair<int, int>> tile_sizes;
187 const int min_tile_c = 1;
188 const int max_tile_c = 4;
189 for (int tile_c = max_tile_c; tile_c >= min_tile_c; tile_c /= 2) {
190 int tile_x = std::min(8, accumulators / tile_c);
191 tile_sizes.emplace_back(tile_c, tile_x);
192 }
193 tile_sizes.emplace_back(max_tile_c, 1);
194
195 // We need to tile the output, but we can't use GuardWithIf because we need
196 // things computed at the tile to have constant size. We can't assume the
197 // output is bigger than a minimum size. So, we specialize for decreasing
198 // tile sizes, and have a degenerate tile case to handle the rest.
199 Var xo("xo");
200 Expr output_channels = output_.dim(0).extent();
201 Expr output_width = output_.dim(1).extent();
202 for (auto i : tile_sizes) {
203 const int tile_c = i.first;
204 const int tile_x = i.second;
205 output_
206 .specialize(output_channels % (tile_c * accum_vector_size) == 0 && output_width >= tile_x)
207 .split(c, co, c, tile_c * accum_vector_size, TailStrategy::RoundUp)
208 .split(x, xo, x, tile_x, TailStrategy::ShiftInwards)
209 .reorder(x, c, co, xo, y, b)
210 .vectorize(c)
211 .unroll(x);
212 }
213
214 // In case there are no suitable tile sizes, just make a dummy split so the
215 // rest of the schedule still works.
216 output_
217 .split(c, co, c, accum_vector_size * min_tile_c, TailStrategy::Predicate)
218 .split(x, xo, x, 1)
219 .reorder(c, x, co, xo, y, b)
220 .vectorize(c);
221
222 // These GuardWithIf splits simplify for the constant-tile specializations,
223 // but probably generate poor code for the general case.
224 convolved.compute_at(output_, co)
225 .store_in(MemoryType::Stack)
226 .reorder(x, c)
227 .vectorize(c, accum_vector_size * min_tile_c, TailStrategy::RoundUp)
228 .unroll(c, max_tile_c, TailStrategy::GuardWithIf)
229 .unroll(x);
230
231 if (use_8bit_multiply(target)) {
232 // Specialize this to avoid computing sum_input when it isn't needed.
233 convolved.specialize(filter_zero_ == 0);
234 }
235
236 RVar rco, rci;
237 convolved.update()
238 .split(r.z, rco, rci, unroll_reduction)
239 .reorder(rci, c, x, rco, r.x, r.y)
240 .vectorize(c, accum_vector_size, TailStrategy::RoundUp)
241 .unroll(c, max_tile_c, TailStrategy::GuardWithIf)
242 .atomic()
243 .vectorize(rci, vector_reduction)
244 .unroll(rci)
245 .unroll(x);
246
247 if (!use_8bit_multiply(target) && get_target().arch == Target::X86) {
248 // On x86, widening subtracts eat up a lot of the already scarce
249 // registers, so precomputing this outside the inner loop helps
250 // a lot.
251 // TODO: Maybe we should do this in a separate op. We already pad it
252 // separately, we just don't dequantize it to 16-bit.
253 input.compute_at(output_, y)
254 .reorder(c, x);
255
256 input.specialize(is_interleaved(input_, 4))
257 .vectorize(c, 4, TailStrategy::RoundUp)
258 .vectorize(x, natural_vector_size<int32_t>(), TailStrategy::GuardWithIf);
259
260 Expr input_channels = input_.dim(0).extent();
261 for (int i = natural_vector_size<int16_t>(); i >= unroll_reduction; i /= 2) {
262 // Use GuardWithIf here to avoid growing the bounds.
263 input.specialize(input_channels >= i)
264 .vectorize(c, i, TailStrategy::GuardWithIf);
265 }
266 } else if (unroll_reduction >= natural_vector_size<uint8_t>()) {
267 // If we're unrolling a full vector's worth of reduction from the
268 // input, explicitly load a vector of it first. This enables targeting
269 // broadcasting dot products, like ARM's udot.
270 input.in(convolved)
271 .compute_at(convolved, c)
272 .bound_extent(c, unroll_reduction)
273 .vectorize(c);
274 }
275
276 if (use_8bit_multiply(target)) {
277 // Precompute the channel offset at root.
278 // TODO: This gets recomputed often when the op is split up into small
279 // pieces.
280 offset_c.compute_root()
281 .vectorize(c, accum_vector_size, TailStrategy::RoundUp);
282 offset_c.update(0)
283 .specialize(input_zero_ != 0)
284 .split(r.z, rco, rci, unroll_reduction)
285 .split(c, co, c, accum_vector_size, TailStrategy::RoundUp)
286 .reorder(rci, c, rco, r.x, r.y, co)
287 .atomic()
288 .vectorize(rci, vector_reduction)
289 .unroll(rci)
290 .vectorize(c);
291 offset_c.update(1)
292 .vectorize(c, accum_vector_size, TailStrategy::RoundUp);
293
294 // Compute the sum of the input outside the loops over channels.
295 sum_input.compute_at(output_, xo)
296 .vectorize(x)
297 .update()
298 .split(r.z, rco, rci, unroll_reduction)
299 .reorder(rci, x, rco, r.x, r.y)
300 .atomic()
301 .vectorize(rci)
302 .vectorize(x)
303 .specialize(stride_x_ == 1 && filter_depth == unroll_reduction && is_interleaved(input_, unroll_reduction));
304 }
305
306 // TODO: Pad this outside and let it constant fold.
307 bias_.in().compute_root().store_in(MemoryType::Stack);
308 }
309};

312class TileConvFilter : public Generator<TileConvFilter> {
313public:
314 Input<Buffer<uint8_t>> input_{"input", 4};
315 Input<uint8_t> input_zero_{"input_zero"};
316 Input<uint8_t> output_zero_{"output_zero"};
317
318 // 6D array of filter coefficients indexed by ci % n, co % k, ci / n, co / k, x, y,
319 // where n = vector_reduction and k = accum_vector_size (below).
320 Output<Buffer<>> output_{"output", 6};
321
322 void configure() {
323 if (use_8bit_multiply(target)) {
324 output_.set_type(UInt(8));
325 } else {
326 output_.set_type(Int(16));
327 }
328 }
329
330 void generate() {
331 Func input_bounded = constant_exterior(input_, input_zero_);
332
333 const int vector_reduction = get_vector_reduction_factor(target, UInt(8));
334 const int vector_tile = natural_vector_size<int32_t>();
335
336 Var bi("bi"), bo("bo");
337
338 Expr filter_cxyb =
339 i16(input_bounded(co * vector_reduction + ci, x, y, bo * vector_tile + bi)) - i16(input_zero_);
340 output_(ci, bi, co, bo, x, y) = cast(output_.type(), filter_cxyb + output_zero_);
341
342 // Schedule.
343 output_.dim(0).set_min(0).set_extent(vector_reduction);
344 output_.dim(1).set_min(0).set_extent(vector_tile).set_stride(vector_reduction);
345 output_.dim(2).set_min(0).set_stride(vector_tile * vector_reduction);
346
347 // TODO: We probably don't care about the performance of this, but if we do,
348 // we could optimize this more.
349 output_
350 .compute_root()
351 .reorder(ci, bi, bo, x, y, co)
352 .vectorize(ci);
353 }
