11class Add : public Generator<Add> {
12public:
13 // Input buffers and quantization parameters.
14 Input<Buffer<uint8_t>> input1_{"input1", 2};
15 Input<uint8_t> input1_zero_{"input1_zero"};
16 Input<int16_t> input1_multiplier_{"input1_multiplier"};
17
18 Input<Buffer<uint8_t>> input2_{"input2", 2};
19 Input<uint8_t> input2_zero_{"input2_zero"};
20 Input<int16_t> input2_multiplier_{"input2_multiplier"};
21
22 Input<uint8_t> output_zero_{"output_zero"};
23 Input<uint8_t> output_min_{"output_min"};
24 Input<uint8_t> output_max_{"output_max"};
25
26 Output<Buffer<uint8_t>> output_{"output", 2};
27
28 void generate() {
29 Var x("x"), y("y");
30
31 Expr input1 = (i16(input1_(x, y)) - i16(input1_zero_)) << add_input_shift;
32 Expr input2 = (i16(input2_(x, y)) - i16(input2_zero_)) << add_input_shift;
33
34 input1 = widening_mul(input1, input1_multiplier_);
35 input2 = widening_mul(input2, input2_multiplier_);
36 Expr output = i16_sat(rounding_shift_right(input1 + input2, add_output_shift));
37
38 output = u8_sat(saturating_add(output, output_zero_));
39 output_(x, y) = clamp(output, output_min_, output_max_);
40
41 // Schedule.
42 const int vector_size = natural_vector_size<uint8_t>();
43
44 output_.compute_root()
45 .vectorize(x, vector_size * 2, TailStrategy::Predicate);
46
47 // Support broadcasting in the c dimension for input2.
48 input2_.dim(0).set_stride(Expr());
49 output_.specialize(input2_.dim(0).stride() == 1);
50 output_.specialize(input2_.dim(0).stride() == 0);
51 output_.specialize_fail("input2 dimension 0 must have a stride of 0 or 1.");
52 }
53};

96// This is a generator that interprets programs to implement sequences of
97// elementwise operations dynamically.
98class Elementwise : public Generator<Elementwise> {
99public:
100 // This is the type used for all intermediate storage and computation.
101 GeneratorParam<Type> intermediate_type_{"intermediate_type", Int(16)};
102
103 // This is the type used for each output. The output of this pipeline
104 // will be a 1D func that is a tuple of these types with non-zero bits.
105 // Outputs are saturating casts of the intermediate type.
106 GeneratorParam<Type> output1_type_{"output1_type", Int(0)};
107 GeneratorParam<Type> output2_type_{"output2_type", Int(0)};
108 GeneratorParam<Type> output3_type_{"output3_type", Int(0)};
109
110 // An array of inputs.
111 Input<Buffer<>[]> inputs_ { "inputs", 2 };
112 // The program to run. See elementwise_program.h for a description of
113 // this buffer.
114 Input<Buffer<int16_t>> program_{"program", 2};
115
116 Func build() {
117 Var x("x"), y("y"), u("u");
118
119 Type intermediate_type = intermediate_type_;
120 Type unsigned_intermediate = intermediate_type.with_code(halide_type_uint);
121 const int q = intermediate_type.bits() - (intermediate_type.is_int() ? 1 : 0);
122
123 Func scratch("scratch");
124 scratch(x, y, u) = undef(intermediate_type_);
125
126 // Load the inputs into the scratch memory.
127 const int input_count = inputs_.size();
128 for (int i = 0; i < input_count; i++) {
129 scratch(x, y, -i - 1) = cast(intermediate_type, inputs_[i](x, y));
130 }
131
132 // scratch slot 0 is a constant 0.
133 scratch(x, y, 0) = cast(intermediate_type, 0);
134
135 RDom r(0, ElementwiseAssembler::OpCodeCount, 0, program_.dim(1).extent());
136 Expr op = program_(0, r.y);
137 Expr arg1 = program_(1, r.y);
138 Expr arg2 = program_(2, r.y);
139 Expr arg3 = cast(intermediate_type, program_(3, r.y));
140 Expr arg4 = cast(intermediate_type, program_(4, r.y));
141
142 Expr slot = r.y + 1;
143
144 const int max_input = input_count - 1;
145 Expr input1 = scratch(x, y, unsafe_promise_clamped(i32(arg1), -max_input - 1, slot));
146 Expr input2 = scratch(x, y, unsafe_promise_clamped(i32(arg2), -max_input - 1, slot));
147
148 std::vector<Expr> instructions = {
149 scratch(x, y, slot),
150 saturating_add(input1, input2 + arg3),
151 saturating_sub(input1, input2 + arg3),
152 saturating_add(multiply_2x_high(input1, input2 + arg3), arg4),
153 rounding_mul_shift_right(input1, input2 + arg3, cast(unsigned_intermediate, arg4)),
154 rounding_shift_right(input1, input2 + arg3),
155 min(input1, input2 + arg3),
156 max(input1, input2 + arg3),
157 clamp(input1, arg3, arg4),
158 rounding_shift_right(approx_logistic(q, input1, input2 + arg3, intermediate_type), q - arg4),
159 rounding_shift_right(approx_tanh(q, input1, input2 + arg3, intermediate_type), q - arg4),
160 };
161 r.where(r.x == op);
162 scratch(x, y, slot) = mux(r.x, instructions);
163
164 Func output("output");
165 std::vector<Type> output_types;
166 if (((Type)output1_type_).bits() > 0) {
167 output_types.push_back(output1_type_);
168 }
169 if (((Type)output2_type_).bits() > 0) {
170 output_types.push_back(output2_type_);
171 }
172 if (((Type)output3_type_).bits() > 0) {
173 output_types.push_back(output3_type_);
174 }
175 int output_count = output_types.size();
176
177 // Grab the last output_count values from scratch and write them to each output.
178 std::vector<Expr> outputs;
179 for (int i = 0; i < output_count; i++) {
180 Expr output_i = scratch(x, y, program_.dim(1).extent() - output_count + i + 1);
181 output_i = saturating_cast(output_types[i], output_i);
182 outputs.push_back(output_i);
183 }
184 output(x, y) = Tuple(outputs);
185
186 // Schedule.
187 output.compute_root()
188 .vectorize(x, natural_vector_size<uint8_t>(), TailStrategy::Predicate);
189
190 // Only allow this many instructions per input, so we can store scratch
191 // on the real stack. This is a lame heuristic.
192 const int max_instructions_per_input = 4;
193
194 // Support broadcasting of dimension 0 of any input.
195 for (int i = 0; i < input_count; i++) {
196 inputs_[i].dim(0).set_stride(Expr());
197 scratch.update(i).specialize(inputs_[i].dim(0).stride() == 1);
198 scratch.update(i).specialize(inputs_[i].dim(0).stride() == 0);
199 scratch.update(i).specialize_fail("Input dimension 0 must have stride 0 or 1.");
200 }
201
202 const int slots = input_count * max_instructions_per_input;
203 scratch
204 .bound_extent(u, input_count + slots + 1)
205 .store_in(MemoryType::Register)
206 .update(input_count + 1)
207 .unroll(r.x);
208
209 program_.dim(0).set_min(0).set_extent(ElementwiseAssembler::InstructionSize).set_stride(1);
210 program_.dim(1).set_min(0).set_stride(ElementwiseAssembler::InstructionSize);
211
212 return output;
213 }
