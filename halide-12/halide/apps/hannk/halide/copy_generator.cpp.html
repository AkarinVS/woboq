7namespace hannk {

9// TODO: It might be better to implement this in C++ and not Halide. It's a trivial pipeline.
10class Copy : public Generator<Copy> {
11public:
12 Input<Buffer<>> input_{"input", 4};
13
14 Input<int> pad_value_{"pad_value"};
15
16 Output<Buffer<>> output_{"output", 4};
17
18 void generate() {
19 Var c("c"), x("x"), y("y"), b("b");
20
21 // This pipeline only supports padding dimension 0.
22 Expr pad_value = cast(input_.type(), pad_value_);
23 Func input_bounded =
24 constant_exterior(input_, pad_value, {{input_.dim(0).min(), input_.dim(0).extent()}});
25
26 output_(c, x, y, b) = cast(output_.type(), input_bounded(c, x, y, b));
27
28 // Schedule.
29 const int vector_size =
30 std::max(natural_vector_size(output_.type()), natural_vector_size(input_.type()));
31
32 if (input_.type() == UInt(8) && output_.type() == UInt(8)) {
33 // Handle 3 channel -> 4 channel padding as a special case.
34 // TODO: vectorize c instead of unroll c.
35 output_.specialize(is_interleaved(input_, 3) && is_interleaved(output_, 4))
36 .vectorize(x, vector_size, TailStrategy::GuardWithIf)
37 .unroll(c);
38 }
39
40 Expr input_channels = input_.dim(0).extent();
41 Expr output_channels = output_.dim(0).extent();
42 Expr channels_in_bounds =
43 input_.dim(0).min() == output_.dim(0).min() && input_channels == output_channels;
44
45 // Handle cases with a small number of channels.
46 for (int i = vector_size; i >= 2; i /= 2) {
47 output_.specialize(output_channels >= i)
48 .vectorize(c, i, TailStrategy::ShiftInwards)
49 .reorder(c, x, y, b)
50 .specialize(channels_in_bounds);
51 }
52
53 // In the general case, use GuardWithIf and reorder c
54 // away from the inner loop to reduce the if overhead.
55 output_
56 .reorder(x, y, c, b)
57 .vectorize(c, vector_size, TailStrategy::GuardWithIf)
58 .specialize(channels_in_bounds);
59 }
60};
61
62} // namespace hannk
