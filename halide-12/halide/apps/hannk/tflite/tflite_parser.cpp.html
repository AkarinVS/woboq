31 static ActivationFunction parse_activation_function(
32 tflite::ActivationFunctionType f) {
33 switch (f) {
34 case tflite::ActivationFunctionType_NONE:
35 return ActivationFunction::None;
36 case tflite::ActivationFunctionType_RELU:
37 return ActivationFunction::Relu;
38 case tflite::ActivationFunctionType_RELU_N1_TO_1:
39 return ActivationFunction::ReluN1To1;
40 case tflite::ActivationFunctionType_RELU6:
41 return ActivationFunction::Relu6;
42 case tflite::ActivationFunctionType_TANH:
43 return ActivationFunction::Tanh;
44 case tflite::ActivationFunctionType_SIGN_BIT:
45 return ActivationFunction::SignBit;
46 default:
47 HCHECK(0) << "Unknown tflite::ActivationFunctionType";
48 }

92 TensorPtr parse_tensor(const tflite::Tensor *t) {
93 std::vector<int> shape;
94 if (t->shape()) {
95 const int shape_size = t->shape()->size();
96 shape.reserve(shape_size);
97 for (int i = 0; i < shape_size; i++) {
98 shape.push_back(t->shape()->Get(shape_size - 1 - i));
99 }
100 }
101 //HCHECK(t->shape()) << "Dynamic shapes not supported.";
102
103 halide_type_t type = parse_type(t->type());
104
105 QuantizationInfo quantization;
106 if (t->quantization()) {
107 quantization.dimension =
108 shape.size() - t->quantization()->quantized_dimension();
109 if (t->quantization()->scale()) {
110 quantization.scale.assign(t->quantization()->scale()->cbegin(),
111 t->quantization()->scale()->cend());
112 }
113 if (t->quantization()->zero_point()) {
114 quantization.zero.assign(t->quantization()->zero_point()->cbegin(),
115 t->quantization()->zero_point()->cend());
116 }
117 }
118
119 if (t->buffer() != 0) {
120 const auto *tflite_buffer = model_->buffers()->Get(t->buffer())->data();
121 if (tflite_buffer) {
122 // tflite_buffer->Data() points at read-only data in the flatbuffer.
123 // Construct a HalideBuffer that points to it (but does not copy or own it).
124 const void *data = static_cast<const void *>(tflite_buffer->Data());
125 assert(data);
126 HalideBuffer<void> buffer(type, const_cast<void *>(data), shape);
127 assert(tflite_buffer->size() == buffer.size_in_bytes());
128
129 return make_op<Tensor>(t->name()->str(), std::move(buffer), std::move(quantization));
130 }
131 }
132
133 // Create an "unallocated" Buffer, which points to null.
134 HalideBuffer<void> buffer(type, nullptr, shape);
135 return make_op<Tensor>(t->name()->str(), std::move(buffer), std::move(quantization));
136 }

138 OpPtr parse_binary(const tflite::Operator *op, BinaryOp::Operator type, bool swap_operands = false) {
139 TensorPtr a = tensors_[op->inputs()->Get(0)];
140 TensorPtr b = tensors_[op->inputs()->Get(1)];
141 TensorPtr output = tensors_[op->outputs()->Get(0)];
142 if (swap_operands) {
143 std::swap(a, b);
144 }
145 return make_op<BinaryOp>(a, b, output, type, ActivationFunction::None);
146 }

149#define PARSE_BINARY_WITH_ACTIVATION(op, Op) \

157 OpPtr parse_pool2D(const tflite::Operator *op, Pool2DOp::Operator reduce_op) {
158 const auto options = op->builtin_options_as_Pool2DOptions();
159 Padding padding = parse_padding(options->padding());
160 std::array<int, 2> stride = {{
161 options->stride_w(),
162 options->stride_h(),
163 }};
164 std::array<int, 2> filter_size = {{
165 options->filter_width(),
166 options->filter_height(),
167 }};
168 ActivationFunction activation =
169 parse_activation_function(options->fused_activation_function());
170 TensorPtr input = tensors_[op->inputs()->Get(0)];
171 TensorPtr output = tensors_[op->outputs()->Get(0)];
172 return make_op<Pool2DOp>(
173 input, output, stride, filter_size, padding, reduce_op, activation);
174 }

176 OpPtr parse_concatenation(const tflite::Operator *op) {
177 const tflite::ConcatenationOptions *options =
178 op->builtin_options_as_ConcatenationOptions();
179 ActivationFunction activation =
180 parse_activation_function(options->fused_activation_function());
181 HCHECK(activation == ActivationFunction::None);
182 std::vector<TensorPtr> inputs;
183 for (auto i = op->inputs()->cbegin(); i != op->inputs()->cend(); ++i) {
184 inputs.push_back(tensors_[*i]);
185 }
186 TensorPtr output = tensors_[op->outputs()->Get(0)];
187 int axis = options->axis();
188 // Handle negative values, which are legal
189 if (axis < 0) {
190 axis = (int)output->rank() + axis;
191 }
192 // Now 'flip' the axis so that it refers to the right dimension in
193 // the Tensor (since we reverse the dimension order)
194 axis = (int)output->rank() - axis - 1;
195 return make_op<ConcatenationOp>(inputs, output, axis);
196 }

198 OpPtr parse_conv2D(const tflite::Operator *op) {
199 const tflite::Conv2DOptions *options =
200 op->builtin_options_as_Conv2DOptions();
201 std::array<int, 2> dilation_factor = {{
202 options->dilation_w_factor(),
203 options->dilation_h_factor(),
204 }};
205 ActivationFunction activation =
206 parse_activation_function(options->fused_activation_function());
207 Padding padding = parse_padding(options->padding());
208 std::array<int, 2> stride = {{
209 options->stride_w(),
210 options->stride_h(),
211 }};
212 TensorPtr input = tensors_[op->inputs()->Get(0)];
213 TensorPtr filter = tensors_[op->inputs()->Get(1)];
214 TensorPtr bias = tensors_[op->inputs()->Get(2)];
215 TensorPtr output = tensors_[op->outputs()->Get(0)];
216 return make_op<Conv2DOp>(input, filter, bias, output, stride,
217 dilation_factor, padding, activation);
218 }

263 OpPtr parse_reshape(const tflite::Operator *op) {
264 const tflite::ReshapeOptions *options =
265 op->builtin_options_as_ReshapeOptions();
266 TensorPtr input = tensors_[op->inputs()->Get(0)];
267 TensorPtr output = tensors_[op->outputs()->Get(0)];
268 // If there are two inputs, and the second is an int32 vector, it should
269 // be used to specify the new shape (instead of ReshapeOptions).
270 TensorPtr shape_tensor = nullptr;
271 if (op->inputs()->size() == 2) {
272 shape_tensor = tensors_[op->inputs()->Get(1)];
273 } else if (options) {
274 size_t size = options->new_shape()->size();
275 HalideBuffer<int32_t> shape_data(size);
276 for (size_t i = 0; i < size; i++) {
277 shape_data(i) = options->new_shape()->Get(i);
278 }
279 shape_tensor = std::make_shared<Tensor>(input->name() + "_shape", shape_data);
280 }
281 return make_op<ReshapeOp>(input, shape_tensor, output);
282 }

322 OpPtr parse_split(const tflite::Operator *op, int axis_tensor_index, int input_tensor_index) {
323 assert(axis_tensor_index < (int)op->inputs()->size());
324 TensorPtr axis_tensor = tensors_[op->inputs()->Get(axis_tensor_index)];
325 HCHECK(axis_tensor->is_allocated()) << "Can't handle dynamic axis for Split.\n";
326 int axis = axis_tensor->buffer<int32_t>()();
327
328 assert(input_tensor_index < (int)op->inputs()->size());
329 TensorPtr input = tensors_[op->inputs()->Get(input_tensor_index)];
330 std::vector<TensorPtr> outputs;
331 for (auto i = op->outputs()->cbegin(); i != op->outputs()->cend(); ++i) {
332 outputs.push_back(tensors_[*i]);
333 }
334 // Handle negative values, which are legal
335 if (axis < 0) {
336 axis = (int)input->rank() + axis;
337 }
338 // Now 'flip' the axis so that it refers to the right dimension in
339 // the Tensor (since we reverse the dimension order)
340 axis = (int)input->rank() - axis - 1;
341 return make_op<SplitOp>(input, outputs, axis);
342 }

411 OpPtr parse_op(const tflite::Operator *op) {
412 const auto *opcodes = model_->operator_codes();
413
414 int opcode_index = op->opcode_index();
415 const auto *opcode = opcodes->Get(opcode_index);
416
417 auto builtin_code = get_builtin_code(opcode);
418 switch (builtin_code) {
419 case tflite::BuiltinOperator_ADD:
420 return PARSE_BINARY_WITH_ACTIVATION(op, Add);
421 case tflite::BuiltinOperator_AVERAGE_POOL_2D:
422 return parse_pool2D(op, Pool2DOp::Average);
423 case tflite::BuiltinOperator_CONCATENATION:
424 return parse_concatenation(op);
425 case tflite::BuiltinOperator_CONV_2D:
426 return parse_conv2D(op);
427 case tflite::BuiltinOperator_DEPTH_TO_SPACE:
428 return parse_depth_to_space(op);
429 case tflite::BuiltinOperator_DEPTHWISE_CONV_2D:
430 return parse_depthwise_conv2D(op);
431 case tflite::BuiltinOperator_EQUAL:
432 return parse_binary(op, BinaryOp::Equal);
433 case tflite::BuiltinOperator_FULLY_CONNECTED:
434 return parse_fully_connected(op);
435 case tflite::BuiltinOperator_GATHER:
436 return parse_gather(op);
437 case tflite::BuiltinOperator_GREATER:
438 return parse_binary(op, BinaryOp::Less, true);
439 case tflite::BuiltinOperator_GREATER_EQUAL:
440 return parse_binary(op, BinaryOp::LessEqual, true);
441 case tflite::BuiltinOperator_L2_NORMALIZATION:
442 return parse_l2_normalization(op);
443 case tflite::BuiltinOperator_LESS:
444 return parse_binary(op, BinaryOp::Less);
445 case tflite::BuiltinOperator_LESS_EQUAL:
446 return parse_binary(op, BinaryOp::LessEqual);
447 case tflite::BuiltinOperator_LOGISTIC:
448 return parse_unary(op, UnaryOp::Logistic);
449 case tflite::BuiltinOperator_LSTM:
450 return parse_lstm(op);
451 case tflite::BuiltinOperator_MAX_POOL_2D:
452 return parse_pool2D(op, Pool2DOp::Max);
453 case tflite::BuiltinOperator_MEAN:
454 return parse_reduction(op, ReductionOp::Mean);
455 case tflite::BuiltinOperator_MUL:
456 return PARSE_BINARY_WITH_ACTIVATION(op, Mul);
457 case tflite::BuiltinOperator_NEG:
458 return parse_unary(op, UnaryOp::Negate);
459 case tflite::BuiltinOperator_NOT_EQUAL:
460 return parse_binary(op, BinaryOp::NotEqual);
461 case tflite::BuiltinOperator_PAD:
462 return parse_pad(op);
463 case tflite::BuiltinOperator_RELU:
464 return parse_unary(op, UnaryOp::Relu);
465 case tflite::BuiltinOperator_RELU6:
466 return parse_unary(op, UnaryOp::Relu6);
467 case tflite::BuiltinOperator_RELU_N1_TO_1:
468 return parse_unary(op, UnaryOp::ReluN1To1);
469 case tflite::BuiltinOperator_RESHAPE:
470 return parse_reshape(op);
471 case tflite::BuiltinOperator_SHAPE:
472 return parse_shape(op);
473 case tflite::BuiltinOperator_SOFTMAX:
474 return parse_softmax(op);
475 case tflite::BuiltinOperator_SPACE_TO_DEPTH:
476 return parse_space_to_depth(op);
477 case tflite::BuiltinOperator_SPLIT:
478 return parse_split(op);
479 case tflite::BuiltinOperator_SPLIT_V:
480 return parse_split_v(op);
481 case tflite::BuiltinOperator_SQUARE:
482 return parse_unary(op, UnaryOp::Square);
483 case tflite::BuiltinOperator_SUB:
484 return PARSE_BINARY_WITH_ACTIVATION(op, Sub);
485 case tflite::BuiltinOperator_TANH:
486 return parse_unary(op, UnaryOp::Tanh);
487 case tflite::BuiltinOperator_TRANSPOSE:
488 return parse_transpose(op);
489
490 default:
491 HCHECK(0) << "Unsupported op "
492 << tflite::EnumNameBuiltinOperator(builtin_code);
493 }

496 std::unique_ptr<OpGroup> parse_subgraph(const tflite::SubGraph *subgraph) {
497 std::vector<TensorPtr> old_tensors;
498 std::swap(old_tensors, tensors_);
499
500 for (const tflite::Tensor *t : *subgraph->tensors()) {
501 tensors_.emplace_back(parse_tensor(t));
502 }
503
504 std::vector<OpPtr> ops;
505 for (const tflite::Operator *i : *subgraph->operators()) {
506 ops.emplace_back(parse_op(i));
507 }
508
509 std::vector<TensorPtr> inputs;
510 for (int i : *subgraph->inputs()) {
511 tensors_[i]->set_input(true);
512 inputs.push_back(tensors_[i]);
513 }
514 std::vector<TensorPtr> outputs;
515 for (int i : *subgraph->outputs()) {
516 tensors_[i]->set_output(true);
517 outputs.push_back(tensors_[i]);
518 }
519
520 std::swap(tensors_, old_tensors);
521
522 return make_op<OpGroup>(std::move(inputs), std::move(outputs), std::move(ops));
523 }
524
525 std::unique_ptr<OpGroup> parse() {
526 for (const tflite::SubGraph *s : *model_->subgraphs()) {
527 subgraphs_.push_back(parse_subgraph(s));
528 }
529
530 HCHECK(subgraphs_.size() == 1) << "Zero or multiple entry points found.";
531 return std::move(subgraphs_.front());
532 }
