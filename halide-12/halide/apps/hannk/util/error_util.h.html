23inline std::ostream &operator<<(std::ostream &s, const std::vector<T> &v) {
24 s << "{";
25 for (size_t i = 0; i < v.size(); ++i) {
26 if (i > 0) {
27 s << ", ";
28 }
29 s << v[i];
30 }

85// This uses operator precedence as a trick to avoid argument evaluation if
86// an assertion is true: it is intended to be used as part of the
87// _halide_internal_assertion macro, to coerce the result of the stream
88// expression to void (to match the condition-is-false case).
89class Voidifier final {
90public:
91 Voidifier() = default;
92 Voidifier(const Voidifier &) = delete;
93 Voidifier &operator=(const Voidifier &) = delete;
94 Voidifier(Voidifier &&) = delete;
95 Voidifier &operator=(Voidifier &&) = delete;
96
97 // This has to be an operator with a precedence lower than << but
98 // higher than ?:
99 void operator&(Checker &) {
100 }
101};

115/**
116 * HCHECK() is used to implement our assertion macros
117 * in such a way that the messages output for the assertion are only
118 * evaluated if the assertion's value is false.
119 *
120 * Note that this macro intentionally has no parens internally; in actual
121 * use, the implicit grouping will end up being
122 *
123 * condition ? (void) : (Voidifier() & (Checker << arg1 << arg2 ... << argN))
124 *
125 * This (regrettably) requires a macro to work, but has the highly desirable
126 * effect that all assertion parameters are totally skipped (not ever evaluated)
127 * when the assertion is true.
128 */
129#ifndef NDEBUG
130// In debug builds, include file-and-line
131#define HCHECK(condition) \
132 (condition) ? (void)0 : ::hannk::internal::Voidifier() & ::hannk::internal::Checker(__FILE__, __LINE__, #condition).ref()
133#else
134// In nondebug builds, don't include file-and-line
135#define HCHECK(condition) \
136 (condition) ? (void)0 : ::hannk::internal::Voidifier() & ::hannk::internal::Checker(#condition).ref()
137#endif
