12inline std::vector<char> read_entire_file(const std::string &filename) {
13 std::ifstream f(filename, std::ios::in | std::ios::binary);
14 HCHECK(f.is_open()) << "Unable to open file: " << filename;
15
16 std::vector<char> result;
17
18 f.seekg(0, std::ifstream::end);
19 size_t size = f.tellg();
20 result.resize(size);
21 f.seekg(0, std::ifstream::beg);
22 f.read(result.data(), result.size());
23 HCHECK(f.good()) << "Unable to read file: " << filename;
24 f.close();
25 return result;
26}
