4namespace hannk {
5
6// Note: all severity values output to stderr, not stdout.
7// Note: ERROR does *not* trigger an exit()/abort() call. FATAL does.
8enum LogSeverity {
9 INFO = 0,
10 WARNING = 1,
11 ERROR = 2,
12 FATAL = 3,
13};
14
15namespace internal {
16
17// All logging in hannk is done via this bottleneck;
18// this is deliberately put in its own file to allow
19// build systems to replace it without weak-linkage
20// tricks or other shenanigans. Note that code shouldn't
21// call this directly (it's meant for internal use by Logger and Checker).
22//
23// Note that calling with severity doesn't require the implementation to
24// call abort(), although it's fine to do so if you like.
25//
26// Note that in the default implementation, all severity values output to stderr, not stdout.
27void hannk_log(LogSeverity severity, const char *msg);
28
29} // namespace internal
30} // namespace hannk
