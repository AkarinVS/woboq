28halide_type_t tf_lite_type_to_halide_type(TfLiteType t) {
29 switch (t) {
30 case kTfLiteBool:
31 return halide_type_t(halide_type_uint, 1);
32 case kTfLiteFloat16:
33 return halide_type_t(halide_type_float, 16);
34 case kTfLiteFloat32:
35 return halide_type_t(halide_type_float, 32);
36 case kTfLiteFloat64:
37 return halide_type_t(halide_type_float, 64);
38 case kTfLiteInt16:
39 return halide_type_t(halide_type_int, 16);
40 case kTfLiteInt32:
41 return halide_type_t(halide_type_int, 32);
42 case kTfLiteInt64:
43 return halide_type_t(halide_type_int, 64);
44 case kTfLiteInt8:
45 return halide_type_t(halide_type_int, 8);
46 case kTfLiteUInt8:
47 return halide_type_t(halide_type_uint, 8);
48
49 case kTfLiteString:
50 case kTfLiteNoType:
51 case kTfLiteComplex64:
52 case kTfLiteComplex128:
53 default:
54 HCHECK(0) << "Unsupported TfLiteType: " << TfLiteTypeGetName(t);
55 return halide_type_t();
56 }
57}

59HalideBuffer<void> wrap_tf_lite_tensor_with_halide_buffer(const TfLiteTensor *t) {
60 // Wrap a Halide buffer around it.
61 std::vector<halide_dimension_t> shape(t->dims->size);
62 size_t shape_size = 1;
63 for (int i = 0; i < (int)shape.size(); i++) {
64 shape[i].min = 0;
65 shape[i].extent = t->dims->data[shape.size() - 1 - i];
66 shape[i].stride = shape_size;
67 shape_size *= shape[i].extent;
68 }
69 void *buffer_data = t->data.data;
70
71 halide_type_t type = tf_lite_type_to_halide_type(t->type);
72 HalideBuffer<void> b(type, buffer_data, shape.size(), shape.data());
73 assert(b.size_in_bytes() == t->bytes);
74 return b;
75}

77class DelegatePtr {
78 void *delegate_lib_ = nullptr;
79 TfLiteDelegate *delegate_ = nullptr;
80 TfLiteDelegate *(*create_delegate_)(char **options_keys,
81 char **options_values,
82 size_t num_options,
83 void (*report_error)(const char *)) = nullptr;
84 void (*destroy_delegate_)(TfLiteDelegate *delegate) = nullptr;
85
86public:
87 DelegatePtr() = default;
88
89 bool init(const std::string &external_delegate_path, int verbosity) {
90 HCHECK(delegate_lib_ == nullptr);
91 // Look for it in the normal library path if no explicit path specified
92 std::string path = external_delegate_path.empty() ? "libHannkDelegate.so" : external_delegate_path;
93 delegate_lib_ = dlopen(path.c_str(), RTLD_NOW | RTLD_LOCAL);
94 if (!delegate_lib_) {
95 std::cerr << "Unable to dlopen(" << path << "): " << dlerror() << "\n";
96 return false;
97 }
98
99 create_delegate_ = (decltype(create_delegate_))dlsym(delegate_lib_, "tflite_plugin_create_delegate");
100 if (!create_delegate_) {
101 std::cerr << "Unable to find tflite_plugin_create_delegate: " << dlerror() << "\n";
102 return false;
103 }
104
105 destroy_delegate_ = (decltype(destroy_delegate_))dlsym(delegate_lib_, "tflite_plugin_destroy_delegate");
106 if (!destroy_delegate_) {
107 std::cerr << "Unable to find tflite_plugin_destroy_delegate: " << dlerror() << "\n";
108 return false;
109 }
110
111 constexpr size_t num_options = 1;
112 std::pair<std::string, std::string> options_strs[num_options] = {
113 {"verbosity", std::to_string(verbosity)},
114 };
115 char *keys[num_options];
116 char *values[num_options];
117 for (size_t i = 0; i < num_options; i++) {
118 keys[i] = const_cast<char *>(options_strs[i].first.c_str());
119 values[i] = const_cast<char *>(options_strs[i].second.c_str());
120 }
121
122 delegate_ = (*create_delegate_)(keys, values, num_options, nullptr);
123 if (!delegate_) {
124 std::cerr << "tflite_plugin_create_delegate returned nullptr\n";
125 return false;
126 }
127 return true;
128 }
129
130 TfLiteDelegate *get() {
131 return delegate_;
132 }
133
134 ~DelegatePtr() {
135 if (destroy_delegate_) {
136 (*destroy_delegate_)(delegate_);
137 }
138 }
139};

184TfLiteModelRunner::TfLiteModelRunner(const std::vector<char> &buffer,
185 int threads,
186 SeedTracker &seed_tracker,
187 std::ostream *verbose_output,
188 TfLiteDelegate *delegate)
189 : verbose_output_(verbose_output) {
190 tf_model_ = TfLiteModelCreate(buffer.data(), buffer.size());
191 HCHECK(tf_model_);
192
193 tf_options_ = TfLiteInterpreterOptionsCreate();
194 HCHECK(tf_options_);
195 TfLiteInterpreterOptionsSetNumThreads(tf_options_, threads);
196 TfLiteInterpreterOptionsSetErrorReporter(tf_options_, ErrorReporter, (void *)this);
197 if (delegate) {
198 TfLiteInterpreterOptionsAddDelegate(tf_options_, delegate);
199 }
200
201 tf_interpreter_ = TfLiteInterpreterCreate(tf_model_, tf_options_);
202 HCHECK(tf_interpreter_);
203
204 // The options/model can be deleted immediately after interpreter creation.
205 TfLiteInterpreterOptionsDelete(tf_options_);
206 tf_options_ = nullptr;
207 TfLiteModelDelete(tf_model_);
208 tf_model_ = nullptr;
209
210 TfLiteStatus status;
211 HCHECK((status = TfLiteInterpreterAllocateTensors(tf_interpreter_)) ==
212 kTfLiteOk)
213 << status;
214
215 const int inputs = TfLiteInterpreterGetInputTensorCount(tf_interpreter_);
216
217 // Fill in the inputs with predictable pseudorandom data as before.
218 for (int i = 0; i < inputs; i++) {
219 TfLiteTensor *t = TfLiteInterpreterGetInputTensor(tf_interpreter_, i);
220 if (t->allocation_type == kTfLiteMmapRo) {
221 // The Tensor references data from the flatbuffer and is read-only;
222 // presumably it is data we want to keep as-is
223 if (verbose_output_) {
224 *verbose_output_ << "TFLITE input " << t->name
225 << " is being used as-is\n";
226 }
227 continue;
228 }
229 const int seed_here = seed_tracker.seed_for_name(t->name);
230 auto input_buf = wrap_tf_lite_tensor_with_halide_buffer(t);
231 dynamic_type_dispatch<FillWithRandom>(input_buf.type(), input_buf,
232 seed_here);
233 if (verbose_output_) {
234 *verbose_output_ << "TFLITE input " << t->name
235 << " inited with seed = " << seed_here << " type "
236 << input_buf.type() << " from "
237 << TfLiteTypeGetName(t->type) << "\n";
238 }
239 }

247std::vector<HalideBuffer<const void>> TfLiteModelRunner::copy_outputs() {
248 std::vector<HalideBuffer<const void>> results;
249 const int outputs = TfLiteInterpreterGetOutputTensorCount(tf_interpreter_);
250 for (int i = 0; i < outputs; i++) {
251 const TfLiteTensor *t =
252 TfLiteInterpreterGetOutputTensor(tf_interpreter_, i);
253 if (verbose_output_) {
254 *verbose_output_ << "TFLITE output is " << t->name << " type "
255 << TfLiteTypeGetName(t->type) << "\n";
256 }
257 // Make a copy since the Buffer might reference memory owned by the
258 // tf_interpreter_
259 results.emplace_back(wrap_tf_lite_tensor_with_halide_buffer(t).copy());
260 }
261 return results;
262}

276ModelRunner::ModelRunner() {
277 for (int i = 0; i < kNumRuns; i++) {
278 do_run[i] = true;
279 }
280#if defined(__arm__) || defined(__aarch64__)
281 // TFLite on Arm devices generally uses the rounding-shift instructions,
282 // which should match our results exactly (since we mimic the same result,
283 // whether or not we actually generate those specific instructions).
284 // So leave the options at their default.
285 tolerance = 1.0 / 256.0;
286#else
287 // TFLite on x86 (on desktop platforms, at least) appears to mostly
288 // use the reference implementations, which don't have the same
289 // rounding-shift behavior. We'll bump up the 'close' value for these.
290 // This is a lttle hand-wavy but is a decent proxy for now.
291 tolerance = 1.0 / 100.0;
292#endif
293}

314ModelRunner::RunResult ModelRunner::run_in_hannk(const std::vector<char> &buffer) {
315 RunResult result;
316
317 std::unique_ptr<OpGroup> model = parse_tflite_model_from_buffer(buffer.data());
318 if (verbosity) {
319 model->dump(std::cout);
320 }
321
322 Interpreter interpreter(std::move(model));
323
324 // Fill in the inputs with pseudorandom data (save the seeds for later).
325 for (TensorPtr t : interpreter.inputs()) {
326 if (t->is_constant()) {
327 // Skip constant buffers, just like TFlite does later on.
328 continue;
329 }
330 const int seed_here = seed_tracker_.seed_for_name(t->name());
331 auto input_buf = t->buffer();
332 dynamic_type_dispatch<FillWithRandom>(input_buf.type(), input_buf, seed_here);
333 if (verbosity) {
334 std::cout << "HALIDE input " << t->name() << " inited with seed = " << seed_here << " type " << input_buf.type() << "\n";
335 }
336 }
337
338 // No: we won't be parallelizing within Halide code, that will be done within
339 // our interpreter. Leaving this here as an example of what *not* to do.
340 // halide_set_num_threads(threads);
341
342 // Execute once, to prime the pump
343 interpreter.execute();
344
345 // Save the outputs from that execution (before benchmarking)
346 for (TensorPtr t : interpreter.outputs()) {
347 if (verbosity) {
348 std::cout << "HALIDE output is " << t->name() << " type " << t->type() << "\n";
349 }
350 // Make a copy since the Buffer might reference memory owned by the interpreter
351 result.outputs.emplace_back(t->buffer().copy());
352 }
353
354 // Now benchmark it
355 if (do_benchmark) {
356 result.time = bench([&interpreter]() {
357 interpreter.execute();
358 });
359 }
360
361 return result;
362}

385bool ModelRunner::compare_results(const std::string &msg, const RunResult &a, const RunResult &b) {
386 bool all_matched = true;
387 HCHECK(a.outputs.size() == b.outputs.size());
388 for (size_t i = 0; i < a.outputs.size(); ++i) {
389 const HalideBuffer<const void> &tflite_buf = a.outputs[i];
390 const HalideBuffer<const void> &halide_buf = b.outputs[i];
391 HCHECK(tflite_buf.type() == halide_buf.type()) << "Expected type " << tflite_buf.type() << "; saw type " << halide_buf.type();
392 HCHECK(tflite_buf.dimensions() == halide_buf.dimensions());
393 for (int d = 0; d < tflite_buf.dimensions(); d++) {
394 HCHECK(tflite_buf.dim(d).min() == halide_buf.dim(d).min());
395 HCHECK(tflite_buf.dim(d).extent() == halide_buf.dim(d).extent());
396 HCHECK(tflite_buf.dim(d).stride() == halide_buf.dim(d).stride()); // TODO: must the strides match?
397 }
398 CompareBuffersOptions options;
399 options.close_thresh = std::ceil((1ull << tflite_buf.type().bits) * tolerance);
400 options.max_diffs_to_log = 8;
401 std::cout << msg;
402 CompareBuffersResult r = dynamic_type_dispatch<CompareBuffers>(tflite_buf.type(), tflite_buf, halide_buf, options);
403 if (r.ok) {
404 if (verbosity >= 2) {
405 std::cout << "MATCHING output " << i << " is:\n";
406 dynamic_type_dispatch<DumpBuffer>(halide_buf.type(), halide_buf);
407 } else {
408 std::cout << "OK!\n";
409 }
410 } else {
411 all_matched = false;
412 }
413 }
414 return all_matched;
415};

417void ModelRunner::run(const std::string &filename) {
418 std::cout << "Processing " << filename << " ...\n";
419
420 const std::vector<char> buffer = read_entire_file(filename);
421
422 std::map<WhichRun, RunResult> results;
423
424 std::vector<WhichRun> active_runs;
425 for (int i = 0; i < kNumRuns; i++) {
426 if (do_run[i]) {
427 active_runs.push_back((WhichRun)i);
428 }
429 }
430
431 const auto exec_tflite = [this, &buffer]() {
432 return run_in_tflite(buffer);
433 };
434 const auto exec_hannk = [this, &buffer]() {
435 return run_in_hannk(buffer);
436 };
437 const auto exec_hannk_external_delegate = [this, &buffer]() {
438 DelegatePtr delegate_ptr;
439 HCHECK(delegate_ptr.init(external_delegate_path, verbosity));
440 return run_in_tflite(buffer, delegate_ptr.get());
441 };
442 const auto exec_hannk_internal_delegate = [this, &buffer]() {
443 HannkDelegateOptions options;
444 options.verbosity = verbosity;
445 TfLiteDelegate *delegate = HannkDelegateCreate(&options);
446 auto result = run_in_tflite(buffer, delegate);
447 HannkDelegateDelete(delegate);
448 return result;
449 };
450 const std::map<WhichRun, std::function<RunResult()>> execs = {
451 {kTfLite, exec_tflite},
452 {kHannk, exec_hannk},
453 {kExternalDelegate, exec_hannk_external_delegate},
454 {kInternalDelegate, exec_hannk_internal_delegate},
455 };
456
457 std::cout << '\n';
458 for (WhichRun i : active_runs) {
459 std::cout << "Executing in " << RunNames[i] << " ...\n";
460 results[i] = execs.at(i)();
461 }
462
463 // ----- Log benchmark times
464 if (do_benchmark) {
465
466 std::cout << '\n';
467 for (WhichRun i : active_runs) {
468 std::cout << RunNames[i] << " Time: " << std::chrono::duration_cast<std::chrono::microseconds>(results[i].time).count() << " us"
469 << "\n";
470 }
471
472 std::cout << '\n';
473 for (WhichRun i : active_runs) {
474 if (i == kTfLite) {
475 continue;
476 }
477 double ratio = (results[i].time / results[kTfLite].time);
478 std::cout << RunNames[i] << " = " << ratio * 100.0 << "% of " << RunNames[kTfLite];
479 if (ratio > 1.0) {
480 std::cout << " *** " << RunNames[i] << " IS SLOWER";
481 }
482 std::cout << "\n";
483 }
484 }
485
486 // ----- Now compare the outputs
487 if (do_compare_results && do_run[kTfLite]) {
488 std::cout << '\n';
489
490 bool all_matched = true;
491 for (WhichRun i : active_runs) {
492 if (i == kTfLite) {
493 continue;
494 }
495 std::ostringstream msg;
496 msg << "Comparing " << RunNames[kTfLite] << " vs " << RunNames[i] << ":";
497 if (!compare_results(msg.str(), results[kTfLite], results[i])) {
498 all_matched = false;
499 }
500 }
501
502 if (!all_matched) {
503 std::cerr << "Some runs exceeded the error threshold!\n";
504 exit(1);
505 }
506 }
