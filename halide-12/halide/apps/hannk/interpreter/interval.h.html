15// This class mimics std::vector, but never dynamically allocates memory.
16// It can only grow to Capacity elements.
17template<typename T, size_t Capacity>
18class SmallVector {
19 alignas(alignof(T)) char buf_[Capacity * sizeof(T)];
20 size_t size_ = 0;
21
22public:
23 SmallVector() = default;
24 SmallVector(size_t size) {
25 resize(size);
26 }
27 SmallVector(const SmallVector &copy) {
28 for (const T &i : copy) {
29 push_back(i);
30 }
31 }
32 SmallVector(SmallVector &&move) {
33 for (T &i : move) {
34 push_back(std::move(i));
35 }
36 }
37 SmallVector(std::initializer_list<T> values) {
38 for (const T &i : values) {
39 push_back(i);
40 }
41 }
42 ~SmallVector() {
43 resize(0);
44 }
45
46 SmallVector &operator=(const SmallVector &assign) {
47 resize(0);
48 for (const T &i : assign) {
49 push_back(i);
50 }
51 return *this;
52 }
53
54 SmallVector &operator=(SmallVector &&move) {
55 resize(0);
56 for (T &i : move) {
57 push_back(std::move(i));
58 }
59 return *this;
60 }

62 template<typename Iterator>
63 void assign(Iterator begin, Iterator end) {
64 clear();
65 for (Iterator i = begin; i != end; ++i) {
66 push_back(*i);
67 }
68 }

70 void resize(size_t size) {
71 assert(size <= Capacity);
72 // Default construct the new elements.
73 for (size_t i = size_; i < size; ++i) {
74 new (&data()[i]) T();
75 }
76 // Destroy the removed elements.
77 for (size_t i = size; i < size_; ++i) {
78 data()[i].~T();
79 }
80 size_ = size;
81 }
82
83 void clear() {
84 resize(0);
85 }
86
87 void push_back(T x) {
88 assert(size_ < Capacity);
89 new (&data()[size_++]) T(std::move(x));
90 }
91
92 template<typename... Args>
93 void emplace_back(Args &&...args) {
94 assert(size_ < Capacity);
95 new (&data()[size_++]) T(std::forward<Args>(args)...);
96 }
97
98 void pop_back() {
99 assert(size_ > 0);
100 resize(size_ - 1);
101 }

103 using iterator = T *;
104 using const_iterator = const T *;
105
106 size_t size() const {
107 return size_;
108 }
109
110 bool empty() const {
111 return size_ == 0;
112 }
113
114 T *data() {
115 return (T *)&buf_[0];
116 }
117
118 const T *data() const {
119 return (const T *)&buf_[0];
120 }
121
122 iterator begin() {
123 return data();
124 }
125
126 iterator end() {
127 return begin() + size_;
128 }
129
130 const_iterator begin() const {
131 return data();
132 }
133
134 const_iterator end() const {
135 return begin() + size_;
136 }
137
138 T &at(size_t i) {
139 assert(i < size_);
140 return data()[i];
141 }
142
143 const T &at(size_t i) const {
144 assert(i < size_);
145 return data()[i];
146 }
147
148 T &operator[](size_t i) {
149 return data()[i];
150 }
151
152 const T &operator[](size_t i) const {
153 return data()[i];
154 }

157// Compute a / b, rounding down.
158inline int floor_div(int a, int b) {
159 assert(a >= 0 && b >= 0);
160 int q = a / b;
161 if (q * b != a && ((a < 0) != (b < 0))) {
162 q -= 1;
163 }
164 return q;
165}
166
167// Compute a / b, rounding to nearest.
168inline int round_div(int a, int b) {
169 return floor_div(a + b / 2, b);
170}
171
172// Compute a / b, rounding upwards.
173inline int ceil_div(int a, int b) {
174 return floor_div(a + b - 1, b);
175}
176
177// Align x up to the next multiplie of n.
178inline int align_up(int x, int n) {
179 return ceil_div(x, n) * n;
180}
181
182// Align x down to the next multiplie of n.
183inline int align_down(int x, int n) {
184 return floor_div(x, n) * n;
185}

187// This type (and Box below) mirrors Halide::Interval, but is not symbolic.
188struct Interval {
189 int min, max;
190
191 Interval()
192 : min(0), max(0) {
193 }
194 explicit Interval(int point)
195 : min(point), max(point) {
196 }
197 Interval(int min, int max)
198 : min(min), max(max) {
199 }
200
201 bool empty() const {
202 return max < min;
203 }
204 int extent() const {
205 return max - min + 1;
206 }
207 void set_extent(int extent) {
208 max = min + extent - 1;
209 }
210
211 Interval &operator*=(int scale) {
212 min *= scale;
213 max *= scale;
214 return *this;
215 }
216
217 Interval &operator/=(int scale) {
218 assert(min >= 0 && max >= 0);
219 min = floor_div(min, scale);
220 max = floor_div(max, scale);
221 return *this;
222 }
223
224 Interval &operator+=(int offset) {
225 min += offset;
226 max += offset;
227 return *this;
228 }
229
230 Interval &operator-=(int offset) {
231 min -= offset;
232 max -= offset;
233 return *this;
234 }
235
236 Interval &operator+=(const Interval &x) {
237 min += x.min;
238 max += x.max;
239 return *this;
240 }
241
242 Interval operator*(int scale) const {
243 Interval result(*this);
244 result *= scale;
245 return result;
246 }
247
248 Interval operator/(int scale) const {
249 Interval result(*this);
250 result /= scale;
251 return result;
252 }
253
254 Interval operator+(int offset) const {
255 Interval result(*this);
256 result += offset;
257 return result;
258 }
259
260 Interval operator-(int offset) const {
261 Interval result(*this);
262 result -= offset;
263 return result;
264 }
265
266 Interval operator+(const Interval &x) const {
267 Interval result(*this);
268 result += x;
269 return result;
270 }
271
272 bool operator==(const Interval &i) const {
273 return min == i.min && max == i.max;
274 }
275 bool operator!=(const Interval &i) const {
276 return min != i.min || max != i.max;
277 }
278};
