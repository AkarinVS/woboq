12class ElementwiseAssembler {
13public:
14 // Every instruction can use two memory locations op1 and op2, and immediates op3 and op4.
15 // Memory location 0 is the constant 0.
16 enum OpCode {
17 // Nothing
18 Noop,
19 // saturating_add(load(op1), load(op2) + op3)
20 Add,
21 // saturating_sub(load(op1), load(op2) + op3)
22 Sub,
23 // multiply_2x_high(load(op1), load(op2) + op3) + op4
24 MulAdd,
25 // rounding_mul_shift_right(load(op1), load(op2) + op3, op4)
26 MulShift,
27 // rounding_shift_right(load(op1), load(op2) + op3)
28 Shift,
29 // min(load(op1), load(op2) + op3)
30 Min,
31 // max(load(op1), load(op2) + op3)
32 Max,
33 // clamp(load(op1), op3, op4)
34 Clamp,
35 // logistic(load(op1) / 2^(load(op2) + op3)) * 2^op4
36 Logistic,
37 // tanh(load(op1) / 2^(load(op2) + op3)) * 2^op4
38 Tanh,
39 OpCodeCount,
40 };
41 static const char *to_string(OpCode op);
42
43 enum {
44 // The "width" of each instruction.
45 InstructionSize = 5,
46 };
47
48 // Represents a scratch slot. Can't be implicitly converted to an integer to
49 // avoid confusion between immediates and scratch references.
50 struct Slot {
51 int16_t index;
52 };
53
54private:
55 Halide::Runtime::Buffer<int16_t> instructions;
56 int size = 0;
57
58 Slot add_instruction(OpCode op, Slot op1, Slot op2, int16_t op3, int16_t op4 = 0);
59
60public:
61 // Create an assembler that builds programs in the given buffer.
62 ElementwiseAssembler(int16_t *buffer, int buffer_size);
63 template<size_t BufferSize>
64 ElementwiseAssembler(std::array<int16_t, BufferSize> &buffer)
65 : ElementwiseAssembler(buffer.data(), buffer.size()) {
66 }
67
68 // Assemble the current program. The return value is the buffer
69 // fromt his assembler cropped to the region needed for the program.
70 Halide::Runtime::Buffer<int16_t> assemble(std::initializer_list<Slot> outputs);
71
72 // Write the current program to the given stream.
73 void disassemble(std::ostream &output);
74
75 // Generate instructions in the program to implement the given operation.
76 Slot constant(int16_t value);
77 Slot input(int index);
78 Slot add(Slot a, Slot b, int16_t add_b = 0);
79 Slot add(Slot a, int16_t b);
80 Slot sub(Slot a, Slot b, int16_t add_b = 0);
81 Slot sub(Slot a, int16_t b);
82 Slot mul(Slot a, Slot b, int16_t add_b = 0);
83 Slot mul(Slot a, int16_t b);
84 Slot mul_add(Slot a, Slot b, int16_t add);
85 Slot mul_add(Slot a, int16_t b, int16_t add);
86 Slot mul_shift(Slot a, Slot b, int16_t shift);
87 Slot mul_shift(Slot a, int16_t b, int16_t shift);
88 Slot shift(Slot a, Slot shift, int16_t extra_shift = 0);
89 Slot shift(Slot a, int16_t shift);
90 Slot min(Slot a, Slot b, int16_t add_b = 0);
91 Slot min(Slot a, int16_t b);
92 Slot max(Slot a, Slot b, int16_t add_b = 0);
93 Slot max(Slot a, int16_t b);
94 Slot clamp(Slot x, int16_t min, int16_t max);
95 Slot logistic(int16_t q, Slot a, Slot q_a);
96 Slot logistic(int16_t q, Slot a, int16_t q_a);
97 Slot tanh(int16_t q, Slot a, Slot q_a);
98 Slot tanh(int16_t q, Slot a, int16_t q_a);
99};
