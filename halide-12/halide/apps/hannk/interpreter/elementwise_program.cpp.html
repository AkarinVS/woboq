41// Returns a mask with bit (1 << i) set to 1 if operand i is relevant for the op.
42int get_opcode_operand_mask(ElementwiseAssembler::OpCode op) {
43 switch (op) {
44 case ElementwiseAssembler::Noop:
45 return 0x0;
46 case ElementwiseAssembler::Add:
47 return 0x7;
48 case ElementwiseAssembler::Sub:
49 return 0x7;
50 case ElementwiseAssembler::MulAdd:
51 return 0xf;
52 case ElementwiseAssembler::MulShift:
53 return 0xf;
54 case ElementwiseAssembler::Shift:
55 return 0x7;
56 case ElementwiseAssembler::Min:
57 return 0x7;
58 case ElementwiseAssembler::Max:
59 return 0x7;
60 case ElementwiseAssembler::Clamp:
61 return 0xd;
62 case ElementwiseAssembler::Logistic:
63 return 0xf;
64 case ElementwiseAssembler::Tanh:
65 return 0xf;
66 default:
67 return 0;
68 }
69}

73Slot ElementwiseAssembler::add_instruction(OpCode op, Slot op1, Slot op2, int16_t op3, int16_t op4) {
74 assert(size < instructions.dim(1).extent());
75 instructions(0, size) = op;
76 instructions(1, size) = op1.index;
77 instructions(2, size) = op2.index;
78 instructions(3, size) = op3;
79 instructions(4, size) = op4;
80 // Slot 0 is the constant 0, instructions start after that.
81 size++;
82 return {(int16_t)size};
83}

89Halide::Runtime::Buffer<int16_t> ElementwiseAssembler::assemble(std::initializer_list<Slot> outputs) {
90 // Check if the outputs are in the right place already.
91 bool in_order = true;
92 int needed_index = size - (int)outputs.size() + 1;
93 for (Slot i : outputs) {
94 in_order = in_order && i.index == needed_index++;
95 }
96 // If not, we need to add some dummy instructions that
97 // load the value into the right place. We do this by
98 // adding 0 to the value.
99 if (!in_order) {
100 for (Slot i : outputs) {
101 add(i, 0);
102 }
103 }
104 return instructions.cropped(1, 0, size);
105}

107void ElementwiseAssembler::disassemble(std::ostream &output) {
108 for (int i = 0; i < size; i++) {
109 OpCode op = (OpCode)instructions(0, i);
110 output
111 << std::setw(3) << std::right << i + 1 << " "
112 << std::setw(12) << std::left << to_string(op);
113
114 int mask = get_opcode_operand_mask(op);
115 for (int j = 0; j < 4; j++) {
116 if (mask & (1 << j)) {
117 int16_t operand = instructions(j + 1, i);
118 if (j < 2) {
119 if (operand < 0) {
120 output << "input[" << -operand - 1 << "] ";
121 } else if (operand > 0) {
122 output << "scratch[" << operand << "] ";
123 } else {
124 output << "0 ";
125 }
126 } else {
127 output << operand << " ";
128 }
129 }
130 }
131 output << "\n";
132 }
