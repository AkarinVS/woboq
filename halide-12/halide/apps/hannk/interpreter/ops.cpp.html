35// Split a dimension d into two new dimensions. Dim d will have min 0
36// and extent factor, while the new dim d + 1 will have the outer split dimension.
37template<typename T>
38void split(int d, int factor, HalideBuffer<T> &buf) {
39 buf.embed(d, 0);
40 halide_dimension_t &dim0 = buf.raw_buffer()->dim[d];
41 halide_dimension_t &dim1 = buf.raw_buffer()->dim[d + 1];
42 dim0.min = 0;
43 dim0.extent = factor;
44 dim0.stride = dim1.stride;
45 assert(dim1.min % factor == 0);
46 assert(dim1.extent % factor == 0);
47 dim1.min /= factor;
48 dim1.extent /= factor;
49 dim1.stride *= factor;
50}

54 Delete,
55 // Replace the second fused dimension with a dimension of extent 1

58 // Put a new dimension of extent 1 and min 0 at the end of the buffer.
59 Pad,

62// Check if dimension 0 and dimension 1 of buf can be fused, and that the
63// operation is not a no-op.
64// We avoid the use of Halide::Runtime::Buffer where possible in these helpers
65// to reduce template instantiation and runtime overhead.
66bool can_fuse(int d0, int d1, FuseType type, const halide_buffer_t *buf) {
67 assert(d0 != d1);
68 assert(d0 < buf->dimensions);
69 assert(d1 < buf->dimensions);
70 if (buf->dim[d0].min != 0 ||
71 buf->dim[d1].stride != buf->dim[d0].extent * buf->dim[d0].stride) {
72 return false;
73 }
74 if (type == FuseType::Delete) {
75 // This is never a no-op.
76 return true;
77 }
78 if (buf->dim[d1].extent != 1) {
79 // This is not a no-op if the second dimension is not extent 1.
80 return true;
81 }
82 if (type == FuseType::Pad) {
83 // This is a not no-op if any dimension after d1 is not extent 1.
84 for (int d = d1 + 1; d < buf->dimensions; d++) {
85 if (buf->dim[d].extent != 1) {
86 return true;
87 }
88 }
89 }
90 return false;
91}
92bool can_fuse_cx(FuseType type, const halide_buffer_t *buf) {
93 return can_fuse(0, 1, type, buf);
94}
95bool can_fuse_xy(FuseType type, const halide_buffer_t *buf) {
96 return can_fuse(1, 2, type, buf);

99// Fuse dimensions d0 and d1 of buf.
100// If pad is true, add a new dimension of extent 1 and min 0 at the end.
101// If pad is false, d1 is deleted from the buffer.
102void fuse(int d0, int d1, FuseType type, halide_buffer_t *buf) {
103 assert(can_fuse(d0, d1, type, buf));
104 halide_dimension_t &dim0 = buf->dim[d0];
105 halide_dimension_t &dim1 = buf->dim[d1];
106 dim0.extent *= dim1.extent;
107 if (type == FuseType::Delete || type == FuseType::Pad) {
108 for (int d = d1; d + 1 < buf->dimensions; d++) {
109 buf->dim[d] = buf->dim[d + 1];
110 }
111 if (type == FuseType::Pad) {
112 halide_dimension_t &padded = buf->dim[buf->dimensions - 1];
113 halide_dimension_t &prev = buf->dim[buf->dimensions - 2];
114 padded.min = 0;
115 padded.extent = 1;
116 padded.stride = prev.extent * prev.stride;
117 } else {
118 buf->dimensions--;
119 }
120 } else {
121 dim1.min = 0;
122 dim1.extent = 1;
123 dim1.stride = dim0.stride * dim0.extent;
124 }
125}

127 fuse(0, 1, type, buf);
128}
129void fuse_xy(FuseType type, halide_buffer_t *buf) {
130 fuse(1, 2, type, buf);

162// Fuse the innermost (stride 1) dimension with other dimensions as much as possible.
163// This may enable the buffers to be processed with fewer instances of the "tail" of
164// a vectorization loop, and fewer levels of recursion in the loop nest helpers below.
165template<typename... Bufs>
166void optimize_elementwise_shapes(halide_buffer_t *a, Bufs *...rest) {
167 assert(all(a->dimensions == rest->dimensions...));
168 for (int d = 0; d + 1 < a->dimensions; d++) {
169 while (can_fuse(d, d + 1, FuseType::Pad, a) &&
170 all(can_fuse(d, d + 1, FuseType::Pad, rest)...) &&
171 all(a->dim[d].extent == rest->dim[d].extent...)) {
172 fuse(d, d + 1, FuseType::Pad, a, rest...);
173 }
174 }
175}

196template<int FnRank, typename Fn, typename... Bufs>
197void loop_nest_impl(Fn &&fn, halide_buffer_t op0, Bufs... ops) {
198 assert(all(op0.dimensions == ops.dimensions...));
199 if (op0.dimensions == FnRank) {
200 fn(&op0, &ops...);
201 } else {
202 const int last_dim = op0.dimensions - 1;
203 const int min = op0.dim[last_dim].min;
204 const int extent = op0.dim[last_dim].extent;
205 const int max = min + extent - 1;
206 for (int i = min; i <= max; i++) {
207 loop_nest_impl<FnRank>(fn, slice_last_dim(op0, i), slice_last_dim(ops, i)...);
208 }
209 }
210}

246// Broadcast the extent 1 dimensions of one shape to match the extent of the
247// other shape.
248template<typename... Bufs>
249void broadcast_shapes(int rank, halide_buffer_t *a, Bufs *...rest) {
250 for (int d = 0; d < rank; d++) {
251 int min = std::min({a->dim[d].min, rest->dim[d].min...});
252 int extent = std::max({max(a->dim[d]), max(rest->dim[d])...}) - min + 1;
253 broadcast_dims(min, extent, &a->dim[d], &rest->dim[d]...);
254 }
255}

257// This helper implements all of the logic necessary for elementwise operations:
258// 1. Broadcasting any extents of 1 to match the rest of the dimensions.
259// 2. Optimizing the shapes by fusing dimensions where possible.
260// 3. Padding shapes to the required rank of `fn`.
261// 4. Iterating and slicing the extra dimensions of the shapes before calling `fn`.
262template<int FnRank, typename Fn, typename T, typename... Ts>
263void elementwise_loop_nest(Fn &&fn, HalideBuffer<T> op0, HalideBuffer<Ts>... ops) {
264 const int rank = std::max({FnRank, op0.dimensions(), ops.dimensions()...});
265 pad_to_rank(rank, op0, ops...);
266 broadcast_shapes(rank, op0.raw_buffer(), ops.raw_buffer()...);
267 optimize_elementwise_shapes(op0.raw_buffer(), ops.raw_buffer()...);
268 loop_nest_impl<FnRank>(fn, *op0.raw_buffer(), *ops.raw_buffer()...);
269}

271// This is the same as the above, except it calls fn with scalar values at each
272// element of the buffer.
273template<typename Fn, typename T, typename... Ts>
274void scalar_elementwise_loop_nest(Fn &&fn, HalideBuffer<T> op0, HalideBuffer<Ts>... ops) {
275 const int rank = std::max({op0.dimensions(), ops.dimensions()...});
276 pad_to_rank(rank, op0, ops...);
277 broadcast_shapes(rank, op0.raw_buffer(), ops.raw_buffer()...);
278 optimize_elementwise_shapes(op0.raw_buffer(), ops.raw_buffer()...);
279 scalar_loop_nest_impl(fn, *(TypedBufferT<T> *)op0.raw_buffer(), *(TypedBufferT<Ts> *)ops.raw_buffer()...);
280}

295template<typename T, typename U>
296void crop_to_union(HalideBuffer<T> &a, HalideBuffer<U> &b) {
297 assert(a.dimensions() == b.dimensions());
298 for (int d = 0; d < a.dimensions(); d++) {
299 int min = std::max(a.dim(d).min(), b.dim(d).min());
300 int max = std::min(a.dim(d).max(), b.dim(d).max());

302 b.crop(d, min, max - min + 1);

315// Represents a number as mantissa*2^exponent/2^(bits - 1), where bits = 8*sizeof(T)
316// This is very similar to a float.
317template<typename T>
318class IntFloat {
319 T mantissa_;
320 T exponent_;
321
322 static constexpr int log2_one = sizeof(T) * 8 - 1;
323 static constexpr int64_t one = 1LL << log2_one;
324
325public:
326 IntFloat()
327 : mantissa_(0), exponent_(0) {
328 }
329
330 IntFloat(float x) {
331 int exponent;
332 float mantissa_float = std::frexp(x, &exponent);
333 int64_t mantissa_long = (int64_t)std::round(mantissa_float * one);
334 assert(std::abs(mantissa_long) <= one);
335 if (mantissa_long == one) {
336 mantissa_long >>= 1;
337 ++exponent;
338 }
339 mantissa_ = mantissa_long;
340 exponent_ = exponent;
341 }
342
343 // Multiply this value by a constant power of 2.
344 IntFloat operator*=(power_of_two x) {
345 exponent_ += x.value;
346 return *this;
347 }
348
349 T mantissa() const {
350 if (exponent_ < -log2_one) {
351 return 0;
352 } else {
353 return mantissa_;
354 }
355 }
356
357 T exponent() const {
358 if (exponent_ < -log2_one) {
359 return -log2_one;
360 } else if (exponent_ > log2_one) {
361 return log2_one;
362 } else {
363 return exponent_;
364 }
365 }
366};

368Interval get_quantized_min_max(ActivationFunction activation, int zero_point, double scale) {
369 int min = 0;
370 int max = 255;
371 if (activation == ActivationFunction::None) {
372 // nothing
373 } else if (activation == ActivationFunction::Relu) {
374 min = zero_point;
375 } else if (activation == ActivationFunction::Relu6) {
376 min = zero_point;
377 max = zero_point + (int)std::round(6.0 / scale);
378 } else if (activation == ActivationFunction::ReluN1To1) {
379 min = zero_point + (int)std::round(-1.0 / scale);
380 max = zero_point + (int)std::round(1.0 / scale);
381 } else {
382 HLOG(FATAL) << "Unsupported quantized activation function type.";
383 }
384 return {std::max(min, 0), std::min(max, 255)};
385}

387Interval get_output_range(ActivationFunction activation, const QuantizationInfo &quantization) {
388 const int output_zero = quantization.uniform_zero();
389 assert(output_zero >= 0 && output_zero <= 255);
390
391 const float output_scale = quantization.uniform_scale();
392
393 const auto output_range = get_quantized_min_max(activation, output_zero, output_scale);
394 assert(output_range.min >= 0 && output_range.min <= 255);
395 assert(output_range.max >= 0 && output_range.max <= 255);
396 assert(output_range.min <= output_range.max);
397
398 return output_range;
399}

422void add_uint8(const HalideBuffer<const void> &in1, const QuantizationInfo &in1q, int in1sign,
423 const HalideBuffer<const void> &in2, const QuantizationInfo &in2q, int in2sign,
424 const HalideBuffer<void> &out, const QuantizationInfo &outq,
425 ActivationFunction activation = ActivationFunction::None) {
426 const int in1_zero = in1q.uniform_zero();
427 const int in2_zero = in2q.uniform_zero();
428 const int out_zero = outq.uniform_zero();
429
430 const float in1_scale = in1q.uniform_scale() * (1 << add_output_shift);
431 const float in2_scale = in2q.uniform_scale() * (1 << add_output_shift);
432 const float out_scale = outq.uniform_scale() * (1 << add_input_shift);
433
434 const int in1_multiplier = std::lround(in1_scale / out_scale) * in1sign;
435 const int in2_multiplier = std::lround(in2_scale / out_scale) * in2sign;
436
437 const auto out_range = get_output_range(activation, outq);
438
439 auto add_rank2 = [&](halide_buffer_t *in1_buf, halide_buffer_t *in2_buf, halide_buffer_t *out_buf) {
440 add_uint8_uint8(in1_buf, in1_zero, in1_multiplier, in2_buf, in2_zero, in2_multiplier,
441 out_zero, out_range.min, out_range.max, out_buf);
442 };
443 elementwise_loop_nest<2>(add_rank2, in1, in2, out);
444}

446void mul_uint8(const HalideBuffer<const void> &in1, const QuantizationInfo &in1q,
447 const HalideBuffer<const void> &in2, const QuantizationInfo &in2q,
448 const HalideBuffer<void> &out, const QuantizationInfo &outq,
449 ActivationFunction activation = ActivationFunction::None) {
450 const int in1_zero = in1q.uniform_zero();
451 const int in2_zero = in2q.uniform_zero();
452 const int out_zero = outq.uniform_zero();
453
454 const float in1_scale = in1q.uniform_scale();
455 const float in2_scale = in2q.uniform_scale();
456 const float out_scale = outq.uniform_scale();
457
458 IntFloat<int32_t> multiplier(in1_scale * in2_scale / out_scale);
459 multiplier *= power_of_two(-2 * mul_input_shift);
460 assert(multiplier.exponent() <= 0);
461
462 const auto out_range = get_output_range(activation, outq);
463
464 auto mul_rank2 = [&](halide_buffer_t *in1_buf, halide_buffer_t *in2_buf, halide_buffer_t *out_buf) {
465 mul_uint8_uint8_uint8(in1_buf, in1_zero, in2_buf, in2_zero,
466 out_zero, multiplier.mantissa(), -multiplier.exponent(),
467 out_range.min, out_range.max, out_buf);
468 };
469 elementwise_loop_nest<2>(mul_rank2, in1, in2, out);
470}

472void requantize(const HalideBuffer<const void> &in, const QuantizationInfo &inq,
473 HalideBuffer<void> out, const QuantizationInfo &outq,
474 ActivationFunction activation = ActivationFunction::None) {
475 if (inq == outq) {
476 // Some of these are just copies, or no-ops.
477 if (is_alias(in.raw_buffer(), out.raw_buffer())) {
478 return;
479 } else {
480 out.copy_from(in);
481 }
482 } else if (in.type() == halide_type_of<uint8_t>() &&
483 out.type() == halide_type_of<uint8_t>()) {
484 // TODO: Maybe a dedicated pipeline for this would be better. It
485 // could be a little faster, and avoid some quantization error.
486 add_uint8(in, inq, 1, in, inq, 0, out, outq, activation);
487 } else {
488 HLOG(FATAL) << "Unable to requantize " << in.type() << " -> " << out.type() << "\n";
489 }
490}

545double dequantize_scalar(const Tensor *t) {
546 assert(t->rank() == 0);
547
548 const QuantizationInfo &q = t->quantization();
549 float scale = q.scale.empty() ? 1.0f : q.scale.front();
550 int zero = q.zero.empty() ? 0 : q.zero.front();
551
552 const auto &buf = t->buffer();
553 if (buf.type() == halide_type_of<uint8_t>()) {
554 return (as_scalar<uint8_t>(buf) - zero) * scale;
555 } else if (buf.type() == halide_type_of<int8_t>()) {
556 return (as_scalar<int8_t>(buf) - zero) * scale;
557 } else if (buf.type() == halide_type_of<uint16_t>()) {
558 return (as_scalar<uint16_t>(buf) - zero) * scale;
559 } else if (buf.type() == halide_type_of<int16_t>()) {
560 return (as_scalar<int16_t>(buf) - zero) * scale;
561 } else if (buf.type() == halide_type_of<uint32_t>()) {
562 return (as_scalar<uint32_t>(buf) - zero) * scale;
563 } else if (buf.type() == halide_type_of<int32_t>()) {
564 return (as_scalar<int32_t>(buf) - zero) * scale;
565 } else if (buf.type() == halide_type_of<float>()) {
566 return (as_scalar<float>(buf) - zero) * scale;
567 } else if (buf.type() == halide_type_of<double>()) {
568 return (as_scalar<double>(buf) - zero) * scale;
569 } else {
570 HLOG(FATAL) << "Unsupported type " << buf.type();
571 return std::numeric_limits<double>::quiet_NaN();
572 }
573}

600void BinaryOp::execute() {
601 const TensorPtr &in1 = input(0);
602 const TensorPtr &in2 = input(1);
603 const TensorPtr &out = output();
604
605 if (in1->type() == halide_type_of<uint8_t>() &&
606 in2->type() == halide_type_of<uint8_t>() &&
607 out->type() == halide_type_of<uint8_t>()) {
608 const auto &in1_buf = in1->buffer();
609 const auto &in2_buf = in2->buffer();
610 const auto &out_buf = out->buffer();
611
612 switch (op_) {
613 case Add:
614 case Sub:
615 add_uint8(in1_buf, in1->quantization(), 1, in2_buf, in2->quantization(), op_ == Add ? 1 : -1, out_buf, out->quantization(), activation_);
616 return;
617 case Mul:
618 mul_uint8(in1_buf, in1->quantization(), in2_buf, in2->quantization(), out_buf, out->quantization(), activation_);
619 return;
620 default:
621 break;
622 }
623 } else if (in1->type() == halide_type_of<int32_t>() &&
624 in2->type() == halide_type_of<int32_t>() &&
625 out->type() == halide_type_of<int32_t>()) {
626 const auto &in1_buf = in1->buffer<const int32_t>();
627 const auto &in2_buf = in2->buffer<const int32_t>();
628 const auto &out_buf = out->buffer<int32_t>();
629
630 // This is really slow, only intended to support scalar operations.
631 auto scalar_op = [&](int32_t a, int32_t b, int32_t &result) {
632 result = implement_binary<int32_t>(op_, a, b);
633 };
634 scalar_elementwise_loop_nest(scalar_op, in1_buf, in2_buf, out_buf);
635 return;
636 } else if (out->type() == halide_type_of<bool>() && out->rank() == 0) {
637 double in1_scalar = dequantize_scalar(in1.get());
638 double in2_scalar = dequantize_scalar(in2.get());
639 auto out_buf = out->buffer<bool>();
640
641 switch (op_) {
642 case Less:
643 out_buf() = in1_scalar < in2_scalar;
644 return;
645 case LessEqual:
646 out_buf() = in1_scalar <= in2_scalar;
647 return;
648 case Equal:
649 out_buf() = in1_scalar == in2_scalar;
650 return;
651 case NotEqual:
652 out_buf() = in1_scalar != in2_scalar;
653 return;
654 default:
655 break;
656 }
657 }
658 HLOG(FATAL)
659 << "Unsupported binary op " << to_string(op_)
660 << " for types " << in1->type() << ", " << in2->type() << ", " << out->type();
661}

919template<typename InputTypes, typename OutputTypes>
920bool can_use_elementwise_program(const Op *op) {
921 InputTypes input_types;
922 OutputTypes output_types;
923 if (op->input_count() > (int)input_types.size()) {
924 return false;
925 }
926 if (op->output_count() > (int)output_types.size()) {
927 return false;
928 }
929 for (int i = 0; i < op->input_count(); i++) {
930 if (op->input(i)->type() != input_types[i]) {
931 return false;
932 }
933 }
934 for (int i = 0; i < op->output_count(); i++) {
935 if (op->output(i)->type() != output_types[i]) {
936 return false;
937 }
938 }
939 return true;
940}

1104void PadOp::execute() {
1105 const TensorPtr &in = input(0);
1106 const TensorPtr &padding = input(1);
1107 const TensorPtr &out = output();
1108
1109 assert(padding->extent(0) == 2);
1110 assert(padding->extent(1) == in->rank());
1111
1112 const auto &padding_buf = padding->buffer<const int32_t>();
1113 if (out->is_dynamic()) {
1114 const int dims = in->rank();
1115 Box new_shape = in->bounds();
1116 for (int d = 0; d < dims; ++d) {
1117 const int idx = dims - d - 1;
1118 const int before_padding = padding_buf(0, idx);
1119 const int after_padding = padding_buf(1, idx);
1120 assert(before_padding >= 0 && after_padding >= 0);
1121 new_shape[d].max += before_padding + after_padding;
1122 }
1123 out->resize(new_shape);
1124 }
1125
1126 if (out->type().bytes() == 1) {
1127 auto input_buf = in->buffer();
1128 auto output_buf = out->buffer();
1129
1130 const int dims = input_buf.dimensions();
1131 for (int d = 0; d < input_buf.dimensions(); d++) {
1132 const int idx = dims - d - 1;
1133 input_buf.translate(d, padding_buf(0, idx));
1134 }
1135
1136 uint8_t pad_value = in->quantization().uniform_zero();
1137
1138 // TODO: should we pad_to_rank(4) the input and output bufs before the loop?
1139
1140 int fill_min_dim = 0;
1141 if (input_buf.dim(0).extent() == 3 && output_buf.dim(0).extent() == 4) {
1142 // copy can handle padding dimension 0, which is much faster than
1143 // filling the extra channel for interleaved 3/4 channel paddings.
1144 fill_min_dim = 1;
1145 }
1146 for (int d = output_buf.dimensions() - 1; d >= fill_min_dim; d--) {
1147 int input_min = input_buf.dim(d).min();
1148 int output_min = output_buf.dim(d).min();
1149 int input_max = input_buf.dim(d).max();
1150 int output_max = output_buf.dim(d).max();
1151 if (output_min < input_min) {
1152 auto before = output_buf.cropped(d, output_min, input_min - output_min);
1153 pad_to_rank(4, before);
1154 fill_uint8(pad_value, before);
1155 } else {
1156 input_min = output_min;
1157 }

1169 input_buf.dim(0).max() < output_buf.dim(0).max()) {
1170 pad_to_rank(4, input_buf);
1171 pad_to_rank(4, output_buf);

1316SmallVector<int, max_rank> ReshapeOp::calc_new_shape() const {
1317 const TensorPtr &in = input();
1318 const TensorPtr &shape = input(1);
1319
1320 SmallVector<int, max_rank> new_shape;
1321 if (shape) {
1322 const auto &shape_buf = shape->buffer<const int32_t>();
1323 assert(shape_buf.dimensions() == 1);
1324 new_shape.assign(shape_buf.begin(), shape_buf.end());
1325 }
1326 if (new_shape.size() == 1 && new_shape[0] == 0) {
1327 // Legacy tflite models use a shape parameter of [0] to indicate scalars,
1328 // so adjust accordingly.
1329 new_shape.clear();
1330 }
1331 std::reverse(new_shape.begin(), new_shape.end());
1332
1333 // One of the shape values can be -1, meaning "calculate it for me".
1334 int output_elements = 1;
1335 int stretch_dim = -1;
1336 for (size_t i = 0; i < new_shape.size(); ++i) {
1337 const int value = new_shape[i];
1338 if (value == -1) {
1339 assert(stretch_dim == -1);
1340 stretch_dim = i;
1341 } else {
1342 output_elements *= value;
1343 }
1344 }
1345 if (stretch_dim != -1) {
1346 new_shape[stretch_dim] = in->number_of_elements() / output_elements;
1347 output_elements *= new_shape[stretch_dim];
1348 assert(output_elements == (int)output()->number_of_elements());
1349 }
1350
1351 return new_shape;
1352}

1356 const TensorPtr &out = output();
1357
1358 SmallVector<int, max_rank> new_shape = calc_new_shape();

1627 const TensorPtr &in = input();
1628 const TensorPtr &out = output();
1629
1630 if (in->type() == halide_type_of<uint8_t>() && out->type() == halide_type_of<uint8_t>()) {
1631 auto in_buf = in->buffer();
1632 auto out_buf = out->buffer();
1633
1634 const int input_zero = in->quantization().uniform_zero();
1635 assert(input_zero >= 0 && input_zero <= 255);
1636 const float in_scale = in->quantization().uniform_scale();
1637
1638 const int left_shift = 6;
1639
1640 std::array<int16_t, 64> program_buffer;
1641 if (op_ == Logistic) {
1642 IntFloat<int16_t> in_multiplier(in_scale);
1643 in_multiplier *= power_of_two(-left_shift);
1644 assert(in_multiplier.exponent() <= 0);
1645
1646 assert(out->quantization().uniform_scale() == 1.0f / 256.0f);
1647 assert(out->quantization().uniform_zero() == 0);
1648
1649 // Build a program to implement the logistic op.
1650 ElementwiseAssembler p(program_buffer);
1651 auto input_zeroed = p.sub(p.input(0), input_zero);
1652 auto input_scaled = p.mul_shift(input_zeroed, in_multiplier.mantissa(), 15 - left_shift);
1653 auto result = p.logistic(8, input_scaled, -in_multiplier.exponent());
1654 auto program_buf = p.assemble({result});
1655
1656 auto logistic_rank2 = [&](halide_buffer_t *in_buf, halide_buffer_t *out_buf) {
1657 elementwise_5xuint8_1xuint8(in_buf, in_buf, in_buf, in_buf, in_buf, program_buf, out_buf);
1658 };
1659 elementwise_loop_nest<2>(logistic_rank2, in_buf, out_buf);
1660 return;
1661 } else if (op_ == Tanh) {
1662 IntFloat<int16_t> in_multiplier(in_scale);
1663 in_multiplier *= power_of_two(-left_shift);
1664 assert(in_multiplier.exponent() <= 0);
1665
1666 assert(out->quantization().uniform_scale() == 1.0f / 128.0f);
1667 assert(out->quantization().uniform_zero() == 128);
1668
1669 // Build a program to implement the tanh op.
1670 ElementwiseAssembler p(program_buffer);
1671 auto input_zeroed = p.sub(p.input(0), input_zero);
1672 auto input_scaled = p.mul_shift(input_zeroed, in_multiplier.mantissa(), 15 - left_shift);
1673 auto result = p.add(p.tanh(7, input_scaled, -in_multiplier.exponent()), 128);
1674 auto program_buf = p.assemble({result});
1675
1676 auto tanh_rank2 = [&](halide_buffer_t *in_buf, halide_buffer_t *out_buf) {
1677 elementwise_5xuint8_1xuint8(in_buf, in_buf, in_buf, in_buf, in_buf, program_buf, out_buf);
1678 };
1679 elementwise_loop_nest<2>(tanh_rank2, in_buf, out_buf);
1680 return;
1681 } else if (op_ == Negate) {
1682 add_uint8(in_buf, in->quantization(), -1, in_buf, in->quantization(), 0, out_buf, out->quantization());

1684 } else if (op_ == Square) {
