15void run_benchmark(const std::string &filename, const InterpreterOptions &options) {
16 if (!options.trace) {
17 // In trace mode, don't send *anything* to stdout
18 std::cout << filename;
19 }
20
21 std::vector<char> buffer = read_entire_file(filename);
22 std::unique_ptr<OpGroup> model = parse_tflite_model_from_buffer(buffer.data());
23
24 if (options.verbose) {
25 model->dump(std::cout);
26 }
27
28 Interpreter interpreter(std::move(model), options);
29
30 if (!options.trace) {
31 auto result = Halide::Tools::benchmark([&]() { interpreter.execute(); });
32 std::cout << ": " << result.wall_time * 1e6 << " us" << std::endl;
33
34 halide_profiler_report(nullptr);
35 halide_profiler_reset();
36 } else {
37 std::cout << std::endl;
38 interpreter.execute();
39 }
40}

44int main(int argc, char **argv) {
45 hannk::InterpreterOptions options;
46
47 for (int i = 1; i < argc; i++) {
48 if (!strcmp(argv[i], "--verbose")) {
49 options.verbose = true;
50 continue;
51 }
52 if (!strcmp(argv[i], "--trace")) {
53 options.trace = true;
54 continue;
55 }
56 if (argv[i][0] == '-') {
57 HLOG(ERROR) << "Unknown flag: " << argv[i] << ".\n";
58 exit(-1);
59 }
60 }
61
62 if (options.verbose && options.trace) {
63 HLOG(ERROR) << "You cannot specify --trace and --verbose at the same time.\n";
64 exit(-1);
65 }
66
67 for (int i = 1; i < argc; i++) {
68 if (!strncmp(argv[i], "--", 2)) {
69 continue;
70 }
71 hannk::run_benchmark(argv[i], options);
72 }
73
74 std::cout << "Done!\n";
75 return 0;
76}
