9class LensBlur : public Halide::Generator<LensBlur> {
10public:
11 Input<Buffer<uint8_t>> left_im{"left_im", 3};
12 Input<Buffer<uint8_t>> right_im{"right_im", 3};
13 // The number of displacements to consider
14 Input<int> slices{"slices", 32, 1, 64};
15 // The depth to focus on
16 Input<int> focus_depth{"focus_depth", 13, 1, 32};
17 // The increase in blur radius with misfocus depth
18 Input<float> blur_radius_scale{"blur_radius_scale", 0.5f, 0.0f, 1.0f};
19 // The number of samples of the aperture to use
20 Input<int> aperture_samples{"aperture_samples", 32, 1, 64};
21
22 Output<Buffer<float>> final{"final", 3};
23
24 void generate() {
25 /* THE ALGORITHM */
26 Expr maximum_blur_radius =
27 cast<int>(max(slices - focus_depth, focus_depth) * blur_radius_scale);
28
29 Func left = BoundaryConditions::repeat_edge(left_im);
30 Func right = BoundaryConditions::repeat_edge(right_im);
31
32 Func diff;
33 diff(x, y, z, c) = min(absd(left(x, y, c), right(x + 2 * z, y, c)),
34 absd(left(x, y, c), right(x + 2 * z + 1, y, c)));
35
36 Func cost;
37 cost(x, y, z) = (pow(cast<float>(diff(x, y, z, 0)), 2) +
38 pow(cast<float>(diff(x, y, z, 1)), 2) +
39 pow(cast<float>(diff(x, y, z, 2)), 2));
40
41 // Compute confidence of cost estimate at each pixel by taking the
42 // variance across the stack.
43 Func cost_confidence;
44 {
45 RDom r(0, slices);
46 Expr a = sum(pow(cost(x, y, r), 2)) / slices;
47 Expr b = pow(sum(cost(x, y, r) / slices), 2);
48 cost_confidence(x, y) = a - b;
49 }
50
51 // Do a push-pull thing to blur the cost volume with an
52 // exponential-decay type thing to inpaint over regions with low
53 // confidence.
54 Func cost_pyramid_push[8];
55 cost_pyramid_push[0](x, y, z, c) =
56 mux(c, {cost(x, y, z) * cost_confidence(x, y), cost_confidence(x, y)});
57
58 Expr w = left_im.dim(0).extent(), h = left_im.dim(1).extent();
59 for (int i = 1; i < 8; i++) {
60 cost_pyramid_push[i](x, y, z, c) = downsample(cost_pyramid_push[i - 1])(x, y, z, c);
61 w /= 2;
62 h /= 2;
63 cost_pyramid_push[i] = BoundaryConditions::repeat_edge(cost_pyramid_push[i], {{0, w}, {0, h}});
64 }
65
66 Func cost_pyramid_pull[8];
67 cost_pyramid_pull[7](x, y, z, c) = cost_pyramid_push[7](x, y, z, c);
68 for (int i = 6; i >= 0; i--) {
69 cost_pyramid_pull[i](x, y, z, c) = lerp(upsample(cost_pyramid_pull[i + 1])(x, y, z, c),
70 cost_pyramid_push[i](x, y, z, c),
71 0.5f);
72 }
73
74 Func filtered_cost;
75 filtered_cost(x, y, z) = (cost_pyramid_pull[0](x, y, z, 0) /
76 cost_pyramid_pull[0](x, y, z, 1));
77
78 // Assume the minimum cost slice is the correct depth.
79 Func depth;
80 {
81 RDom r(0, slices);
82 depth(x, y) = argmin(filtered_cost(x, y, r))[0];
83 }
84
85 Func bokeh_radius;
86 bokeh_radius(x, y) = abs(depth(x, y) - focus_depth) * blur_radius_scale;
87
88 Func bokeh_radius_squared;
89 bokeh_radius_squared(x, y) = pow(bokeh_radius(x, y), 2);
90
91 // Take a max filter of the bokeh radius to determine the
92 // worst-case bokeh radius to consider at each pixel. Makes the
93 // sampling more efficient below.
94 Func worst_case_bokeh_radius_y;
95 Func worst_case_bokeh_radius;
96 {
97 RDom r(-maximum_blur_radius, 2 * maximum_blur_radius + 1);
98 worst_case_bokeh_radius_y(x, y) = maximum(bokeh_radius(x, y + r));
99 worst_case_bokeh_radius(x, y) = maximum(worst_case_bokeh_radius_y(x + r, y));
100 }
101
102 Func input_with_alpha;
103 input_with_alpha(x, y, c) = mux(c, {cast<float>(left(x, y, 0)),
104 cast<float>(left(x, y, 1)),
105 cast<float>(left(x, y, 2)),
106 255.0f});
107
108 // Render a blurred image
109 Func output;
110 output(x, y, c) = input_with_alpha(x, y, c);
111
112 // The sample locations are a random function of x, y, and sample
113 // number (not c).
114 Expr worst_radius = worst_case_bokeh_radius(x, y);
115 Expr sample_u = (random_float() - 0.5f) * 2 * worst_radius;
116 Expr sample_v = (random_float() - 0.5f) * 2 * worst_radius;
117 sample_u = clamp(cast<int>(sample_u), -maximum_blur_radius, maximum_blur_radius);
118 sample_v = clamp(cast<int>(sample_v), -maximum_blur_radius, maximum_blur_radius);
119 Func sample_locations;
120 sample_locations(x, y, z) = {sample_u, sample_v};
121
122 RDom s(0, aperture_samples);
123 sample_u = sample_locations(x, y, z)[0];
124 sample_v = sample_locations(x, y, z)[1];
125 Expr sample_x = x + sample_u, sample_y = y + sample_v;
126 Expr r_squared = sample_u * sample_u + sample_v * sample_v;
127
128 // We use this sample if it's from a pixel whose bokeh influences
129 // this output pixel. Here's a crude approximation that ignores
130 // some subtleties of occlusion edges and inpaints behind objects.
131 Expr sample_is_within_bokeh_of_this_pixel =
132 r_squared < bokeh_radius_squared(x, y);
133
134 Expr this_pixel_is_within_bokeh_of_sample =
135 r_squared < bokeh_radius_squared(sample_x, sample_y);
136
137 Expr sample_is_in_front_of_this_pixel =
138 depth(sample_x, sample_y) < depth(x, y);
139
140 Func sample_weight;
141 sample_weight(x, y, z) =
142 select((sample_is_within_bokeh_of_this_pixel ||
143 sample_is_in_front_of_this_pixel) &&
144 this_pixel_is_within_bokeh_of_sample,
145 1.0f, 0.0f);
146
147 sample_x = x + sample_locations(x, y, s)[0];
148 sample_y = y + sample_locations(x, y, s)[1];
149 output(x, y, c) += sample_weight(x, y, s) * input_with_alpha(sample_x, sample_y, c);
150
151 // Normalize
152 final(x, y, c) = output(x, y, c) / output(x, y, 3);
153
154 /* ESTIMATES */
155 // (This can be useful in conjunction with RunGen and benchmarks as well
156 // as auto-schedule, so we do it in all cases.)
157 // Provide estimates on the input image
158 left_im.set_estimates({{0, 192}, {0, 320}, {0, 3}});
159 right_im.set_estimates({{0, 192}, {0, 320}, {0, 3}});
160 // Provide estimates on the parameters
161 slices.set_estimate(32);
162 focus_depth.set_estimate(13);
163 blur_radius_scale.set_estimate(0.5f);
164 aperture_samples.set_estimate(32);
165 // Provide estimates on the pipeline output
166 final.set_estimates({{0, 192}, {0, 320}, {0, 3}});
167
168 /* THE SCHEDULE */
169 if (auto_schedule) {
170 // nothing
171 } else if (get_target().has_gpu_feature()) {
172 // Manual GPU schedule
173 Var xi("xi"), yi("yi"), zi("zi");
174 cost_pyramid_push[0]
175 .compute_root()
176 .reorder(c, z, x, y)
177 .bound(c, 0, 2)
178 .unroll(c)
179 .gpu_tile(x, y, xi, yi, 16, 16);
180 cost.compute_at(cost_pyramid_push[0], xi);
181 cost_confidence.compute_at(cost_pyramid_push[0], xi);
182
183 for (int i = 1; i < 8; i++) {
184 cost_pyramid_push[i].compute_root().gpu_tile(x, y, z, xi, yi, zi, 8, 8, 8);
185 cost_pyramid_pull[i].compute_root().gpu_tile(x, y, z, xi, yi, zi, 8, 8, 8);
186 }
187
188 depth.compute_root()
189 .gpu_tile(x, y, xi, yi, 16, 16);
190 input_with_alpha.compute_root()
191 .reorder(c, x, y)
192 .unroll(c)
193 .gpu_tile(x, y, xi, yi, 16, 16);
194 worst_case_bokeh_radius_y
195 .compute_root()
196 .gpu_tile(x, y, xi, yi, 16, 16);
197 worst_case_bokeh_radius
198 .compute_root()
199 .gpu_tile(x, y, xi, yi, 16, 16);
200 final.compute_root()
201 .reorder(c, x, y)
202 .bound(c, 0, 3)
203 .unroll(c)
204 .gpu_tile(x, y, xi, yi, 16, 16);
205
206 output.compute_at(final, xi);
207 output.update().reorder(c, x, s).unroll(c);
208 sample_weight.compute_at(output, x);
209 sample_locations.compute_at(output, x);
210 } else {
211 // Manual CPU schedule
212 cost_pyramid_push[0]
213 .compute_root()
214 .reorder(c, z, x, y)
215 .bound(c, 0, 2)
216 .unroll(c)
217 .vectorize(x, 16)
218 .parallel(y, 4);
219 cost.compute_at(cost_pyramid_push[0], x)
220 .vectorize(x);
221 cost_confidence.compute_at(cost_pyramid_push[0], x)
222 .vectorize(x);
223
224 Var xi, yi, t;
225 for (int i = 1; i < 8; i++) {
226 cost_pyramid_push[i].compute_at(cost_pyramid_pull[1], t).vectorize(x, 8);
227 if (i > 1) {
228 cost_pyramid_pull[i]
229 .compute_at(cost_pyramid_pull[1], t)
230 .tile(x, y, xi, yi, 8, 2)
231 .vectorize(xi)
232 .unroll(yi);
233 }
234 }
235
236 cost_pyramid_pull[1]
237 .compute_root()
238 .fuse(z, c, t)
239 .parallel(t)
240 .tile(x, y, xi, yi, 8, 2)
241 .vectorize(xi)
242 .unroll(yi);
243 depth.compute_root()
244 .tile(x, y, xi, yi, 8, 2)
245 .vectorize(xi)
246 .unroll(yi)
247 .parallel(y, 8);
248 input_with_alpha.compute_root()
249 .reorder(c, x, y)
250 .unroll(c)
251 .vectorize(x, 8)
252 .parallel(y, 8);
253 worst_case_bokeh_radius_y
254 .compute_at(final, y)
255 .vectorize(x, 8);
256 final.compute_root()
257 .reorder(c, x, y)
258 .bound(c, 0, 3)
259 .unroll(c)
260 .vectorize(x, 8)
261 .parallel(y);
262 worst_case_bokeh_radius
263 .compute_at(final, y)
264 .vectorize(x, 8);
265 output.compute_at(final, x)
266 .vectorize(x);
267 output.update()
268 .reorder(c, x, s)
269 .vectorize(x)
270 .unroll(c);
271 sample_weight.compute_at(output, x).unroll(x);
272 sample_locations.compute_at(output, x).vectorize(x);
273 }
274 }

276private:
277 Var x, y, z, c;
278
279 // Downsample with a 1 3 3 1 filter
280 Func downsample(Func f) {
281 using Halide::_;
282 Func downx, downy;
283 downx(x, y, _) = (f(2 * x - 1, y, _) + 3.0f * (f(2 * x, y, _) + f(2 * x + 1, y, _)) + f(2 * x + 2, y, _)) / 8.0f;
284 downy(x, y, _) = (downx(x, 2 * y - 1, _) + 3.0f * (downx(x, 2 * y, _) + downx(x, 2 * y + 1, _)) + downx(x, 2 * y + 2, _)) / 8.0f;
285 return downy;
286 }
287
288 // Upsample using bilinear interpolation
289 Func upsample(Func f) {
290 using Halide::_;
291 Func upx, upy;
292 upx(x, y, _) = 0.25f * f((x / 2) - 1 + 2 * (x % 2), y, _) + 0.75f * f(x / 2, y, _);
293 upy(x, y, _) = 0.25f * upx(x, (y / 2) - 1 + 2 * (y % 2), _) + 0.75f * upx(x, y / 2, _);
294 return upy;
295 }
