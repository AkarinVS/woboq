38void show_usage_and_exit() {
39 fprintf(stderr,
40 "Usage:\n"
41 "\t./resample [-f scalefactor] "
42 "[-b benchmark_iterations] "
43 "[-i box|linear|cubic|lanczos] "
44 "[-t float32|uint8|uint16] in.png out.png\n");
45 exit(1);
46}

48void parse_commandline(int argc, char **argv) {
49 for (int i = 1; i < argc; i++) {
50 std::string arg = argv[i];
51 if (arg == "-f" && i + 1 < argc) {
52 scale_factor = atof(argv[++i]);
53 } else if (arg == "-i" && i + 1 < argc) {
54 interpolation_type = argv[++i];
55 } else if (arg == "-t" && i + 1 < argc) {
56 input_type = argv[++i];
57 } else if (arg == "-b" && i + 1 < argc) {
58 benchmark_iters = atoi(argv[++i]);
59 } else if (arg == "-p" && i + 1 < argc) {
60 packed = atoi(argv[++i]) != 0;
61 } else if (infile.empty()) {
62 infile = arg;
63 } else if (outfile.empty()) {
64 outfile = arg;
65 } else {
66 fprintf(stderr, "Unexpected command line option '%s'.\n", arg.c_str());
67 show_usage_and_exit();
68 }
69 }
70
71 if (infile.empty() || outfile.empty() || interpolation_type.empty()) {
72 show_usage_and_exit();
73 }
74}

76int main(int argc, char **argv) {
77 parse_commandline(argc, argv);
78
79 Halide::Runtime::Buffer<> in = Halide::Tools::load_image(infile);
80 int out_width = in.width() * scale_factor;
81 int out_height = in.height() * scale_factor;
82
83 decltype(&resize_box_float32_up) variants[3][2][4] =
84 {
85 {{&resize_box_float32_up,
86 &resize_cubic_float32_up,
87 &resize_linear_float32_up,
88 &resize_lanczos_float32_up},
89 {&resize_box_float32_down,
90 &resize_cubic_float32_down,
91 &resize_linear_float32_down,
92 &resize_lanczos_float32_down}},
93 {{&resize_box_uint8_up,
94 &resize_cubic_uint8_up,
95 &resize_linear_uint8_up,
96 &resize_lanczos_uint8_up},
97 {&resize_box_uint8_down,
98 &resize_cubic_uint8_down,
99 &resize_linear_uint8_down,
100 &resize_lanczos_uint8_down}},
101 {{&resize_box_uint16_up,
102 &resize_cubic_uint16_up,
103 &resize_linear_uint16_up,
104 &resize_lanczos_uint16_up},
105 {&resize_box_uint16_down,
106 &resize_cubic_uint16_down,
107 &resize_linear_uint16_down,
108 &resize_lanczos_uint16_down}}};
109
110 int interpolation_idx = 0;
111 if (interpolation_type == "box") {
112 interpolation_idx = 0;
113 } else if (interpolation_type == "cubic") {
114 interpolation_idx = 1;
115 } else if (interpolation_type == "linear") {
116 interpolation_idx = 2;
117 } else if (interpolation_type == "lanczos") {
118 interpolation_idx = 3;
119 } else {
120 fprintf(stderr, "Unknown interpolation type: %s\n", interpolation_type.c_str());
121 show_usage_and_exit();
122 }
123
124 int upsample_idx = scale_factor > 1.0f ? 0 : 1;
125
126 // Instead of just adapting to the actual type of the input, we'll
127 // convert it to the requested type to make it easier to benchmark
128 // lots of different types.
129 int type_idx = 0;
130 if (input_type == "float32") {
131 in = Halide::Tools::ImageTypeConversion::convert_image(in, halide_type_of<float>());
132 type_idx = 0;
133 } else if (input_type == "uint8") {
134 in = Halide::Tools::ImageTypeConversion::convert_image(in, halide_type_of<uint8_t>());
135 type_idx = 1;
136 } else if (input_type == "uint16") {
137 in = Halide::Tools::ImageTypeConversion::convert_image(in, halide_type_of<uint16_t>());
138 type_idx = 2;
139 } else {
140 fprintf(stderr, "Unhandled type: %s\n", input_type.c_str());
141 show_usage_and_exit();
142 }
143
144 Halide::Runtime::Buffer<> out(in.type(), out_width, out_height, 3);
145
146 auto resize_fn = variants[type_idx][upsample_idx][interpolation_idx];
147
148 double time = Halide::Tools::benchmark(benchmark_iters, benchmark_iters, [&]() { resize_fn(in, scale_factor, out); });
149 printf("planar %8s %8s %1.2f time: %f ms\n",
150 interpolation_type.c_str(), input_type.c_str(), scale_factor, time * 1000);
151
152 Halide::Tools::convert_and_save_image(out, outfile);
153
154 if (packed) {
155 // Also benchmark a packed memory layout. Don't bother to copy the
156 // actual data over, because we won't save the result. We just
157 // want to measure the runtime.
158 auto in_packed =
159 Halide::Runtime::Buffer<>::make_interleaved(in.type(), in.width(), in.height(), in.channels());
160 auto out_packed =
161 Halide::Runtime::Buffer<>::make_interleaved(out.type(), out.width(), out.height(), out.channels());
162 time = Halide::Tools::benchmark(benchmark_iters, benchmark_iters, [&]() { resize_fn(in_packed, scale_factor, out_packed); });
163 printf("packed %8s %8s %1.2f time: %f ms\n",
164 interpolation_type.c_str(), input_type.c_str(), scale_factor, time * 1000);
165 }
166
167 printf("Success!\n");
168 return 0;
169}
