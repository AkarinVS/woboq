1// This FFT is an implementation of the algorithm described in
2// http://research.microsoft.com/pubs/131400/fftgpusc08.pdf
3// This algorithm is more well suited to Halide than in-place
4// algorithms.

23template<typename T>
24Func make_real(const Buffer<T> &re) {
25 Func ret;
26 ret(x, y) = re(x, y);
27 return ret;
28}
29
30template<typename T>
31ComplexFunc make_complex(const Buffer<T> &re) {
32 ComplexFunc ret;
33 ret(x, y) = re(x, y);
34 return ret;
35}

37int main(int argc, char **argv) {
38 int W = 32;
39 int H = 32;
40 if (argc >= 3) {
41 W = atoi(argv[1]);
42 H = atoi(argv[2]);
43 }
44 std::string output_dir;
45 if (argc >= 4) {
46 output_dir = argv[3];
47 }
48
49 // Generate a random image to convolve with.
50 Buffer<float> in(W, H);
51 for (int y = 0; y < H; y++) {
52 for (int x = 0; x < W; x++) {
53 in(x, y) = (float)rand() / (float)RAND_MAX;
54 }
55 }
56
57 // Construct a box filter kernel centered on the origin.
58 const int box = 3;
59 Buffer<float> kernel(W, H);
60 for (int y = 0; y < H; y++) {
61 for (int x = 0; x < W; x++) {
62 int u = x < (W - x) ? x : (W - x);
63 int v = y < (H - y) ? y : (H - y);
64 kernel(x, y) = u <= box / 2 && v <= box / 2 ? 1.0f / (box * box) : 0.0f;
65 }
66 }
67
68 Target target = get_jit_target_from_environment();
69
70 Fft2dDesc fwd_desc;
71 Fft2dDesc inv_desc;
72 inv_desc.gain = 1.0f / (W * H);
73
74 Func filtered_c2c;
75 {
76 // Compute the DFT of the input and the kernel.
77 ComplexFunc dft_in = fft2d_c2c(make_complex(in), W, H, -1, target, fwd_desc);
78 ComplexFunc dft_kernel = fft2d_c2c(make_complex(kernel), W, H, -1, target, fwd_desc);
79 dft_in.compute_root();
80 dft_kernel.compute_root();
81
82 // Compute the convolution.
83 ComplexFunc dft_filtered("dft_filtered");
84 dft_filtered(x, y) = dft_in(x, y) * dft_kernel(x, y);
85
86 // Compute the inverse DFT to get the result.
87 ComplexFunc dft_out = fft2d_c2c(dft_filtered, W, H, 1, target, inv_desc);
88 dft_out.compute_root();
89
90 // Extract the real component and normalize.
91 filtered_c2c(x, y) = re(dft_out(x, y));
92 }
93
94 Func filtered_r2c;
95 {
96 // Compute the DFT of the input and the kernel.
97 ComplexFunc dft_in = fft2d_r2c(make_real(in), W, H, target, fwd_desc);
98 ComplexFunc dft_kernel = fft2d_r2c(make_real(kernel), W, H, target, fwd_desc);
99 dft_in.compute_root();
100 dft_kernel.compute_root();
101
102 // Compute the convolution.
103 ComplexFunc dft_filtered("dft_filtered");
104 dft_filtered(x, y) = dft_in(x, y) * dft_kernel(x, y);
105
106 // Compute the inverse DFT to get the result.
107 filtered_r2c = fft2d_c2r(dft_filtered, W, H, target, inv_desc);
108 }
109
110 Buffer<float> result_c2c = filtered_c2c.realize({W, H}, target);
111 Buffer<float> result_r2c = filtered_r2c.realize({W, H}, target);
112
113 for (int y = 0; y < H; y++) {
114 for (int x = 0; x < W; x++) {
115 float correct = 0;
116 for (int i = -box / 2; i <= box / 2; i++) {
117 for (int j = -box / 2; j <= box / 2; j++) {
118 correct += in((x + j + W) % W, (y + i + H) % H);
119 }
120 }
121 correct /= box * box;
122 if (fabs(result_c2c(x, y) - correct) > 1e-6f) {
123 printf("result_c2c(%d, %d) = %f instead of %f\n", x, y, result_c2c(x, y), correct);
124 return -1;
125 }
126 if (fabs(result_r2c(x, y) - correct) > 1e-6f) {
127 printf("result_r2c(%d, %d) = %f instead of %f\n", x, y, result_r2c(x, y), correct);
128 return -1;
129 }
130 }
131 }

133 // For a description of the methodology used here, see
134 // http://www.fftw.org/speed/method.html
135
136 // Take the minimum time over many of iterations to minimize
137 // noise.
138 const int samples = 100;
139 const int reps = 1000;
140
141 Var rep("rep");
142
143 Buffer<float> re_in = lambda(x, y, 0.0f).realize({W, H});
144 Buffer<float> im_in = lambda(x, y, 0.0f).realize({W, H});
145
146 printf("%12s %5s%11s%5s %5s%11s%5s\n", "", "", "Halide", "", "", "FFTW", "");
147 printf("%12s %10s %10s %10s %10s %10s\n", "DFT type", "Time (us)", "MFLOP/s", "Time (us)", "MFLOP/s", "Ratio");
148
149 ComplexFunc c2c_in;
150 // Read all reps from the same place in memory. This effectively
151 // benchmarks taking the FFT of cached inputs, which is a
152 // reasonable assumption for a well optimized program with good
153 // locality.
154 c2c_in(x, y, rep) = {re_in(x, y), im_in(x, y)};
155 Func bench_c2c = fft2d_c2c(c2c_in, W, H, -1, target, fwd_desc);
156 bench_c2c.compile_to_lowered_stmt(output_dir + "c2c.html", bench_c2c.infer_arguments(), HTML);
157 Realization R_c2c = bench_c2c.realize({W, H, reps}, target);
158 // Write all reps to the same place in memory. This means the
159 // output appears to be cached on all but the first
160 // iteration. This seems to match the behavior of FFTW's benchmark
161 // code, and like the input, it is a reasonable assumption for a well
162 // optimized real world system.
163 R_c2c[0].raw_buffer()->dim[2].stride = 0;
164 R_c2c[1].raw_buffer()->dim[2].stride = 0;
165
166 double halide_t = benchmark(samples, 1, [&]() { bench_c2c.realize(R_c2c); }) * 1e6 / reps;
167#ifdef WITH_FFTW
168 std::vector<std::pair<float, float>> fftw_c1(W * H);
169 std::vector<std::pair<float, float>> fftw_c2(W * H);
170 fftwf_plan c2c_plan = fftwf_plan_dft_2d(W, H, (fftwf_complex *)&fftw_c1[0], (fftwf_complex *)&fftw_c2[0], FFTW_FORWARD, FFTW_EXHAUSTIVE);
171 double fftw_t = benchmark(samples, reps, [&]() { fftwf_execute(c2c_plan); }) * 1e6;
172#else
173 double fftw_t = 0;
174#endif
175 printf("%12s %10.3f %10.2f %10.3f %10.2f %10.3g\n",
176 "c2c",
177 halide_t,
178 5 * W * H * (log2(W) + log2(H)) / halide_t,
179 fftw_t,
180 5 * W * H * (log2(W) + log2(H)) / fftw_t,
181 fftw_t / halide_t);
182
183 Func r2c_in;
184 // All reps read from the same input. See notes on c2c_in.
185 r2c_in(x, y, rep) = re_in(x, y);
186 Func bench_r2c = fft2d_r2c(r2c_in, W, H, target, fwd_desc);
187 bench_r2c.compile_to_lowered_stmt(output_dir + "r2c.html", bench_r2c.infer_arguments(), HTML);
188 Realization R_r2c = bench_r2c.realize({W, H / 2 + 1, reps}, target);
189 // Write all reps to the same place in memory. See notes on R_c2c.
190 R_r2c[0].raw_buffer()->dim[2].stride = 0;
191 R_r2c[1].raw_buffer()->dim[2].stride = 0;
192
193 halide_t = benchmark(samples, 1, [&]() { bench_r2c.realize(R_r2c); }) * 1e6 / reps;
194#ifdef WITH_FFTW
195 std::vector<float> fftw_r(W * H);
196 fftwf_plan r2c_plan = fftwf_plan_dft_r2c_2d(W, H, &fftw_r[0], (fftwf_complex *)&fftw_c1[0], FFTW_EXHAUSTIVE);
197 fftw_t = benchmark(samples, reps, [&]() { fftwf_execute(r2c_plan); }) * 1e6;
198#else
199 fftw_t = 0;
200#endif
201 printf("%12s %10.3f %10.2f %10.3f %10.2f %10.3g\n",
202 "r2c",
203 halide_t,
204 2.5 * W * H * (log2(W) + log2(H)) / halide_t,
205 fftw_t,
206 2.5 * W * H * (log2(W) + log2(H)) / fftw_t,
207 fftw_t / halide_t);
208
209 ComplexFunc c2r_in;
210 // All reps read from the same input. See notes on c2c_in.
211 c2r_in(x, y, rep) = {re_in(x, y), im_in(x, y)};
212 Func bench_c2r = fft2d_c2r(c2r_in, W, H, target, inv_desc);
213 bench_c2r.compile_to_lowered_stmt(output_dir + "c2r.html", bench_c2r.infer_arguments(), HTML);
214 Realization R_c2r = bench_c2r.realize({W, H, reps}, target);
215 // Write all reps to the same place in memory. See notes on R_c2c.
216 R_c2r[0].raw_buffer()->dim[2].stride = 0;
217
218 halide_t = benchmark(samples, 1, [&]() { bench_c2r.realize(R_c2r); }) * 1e6 / reps;
219#ifdef WITH_FFTW
220 fftwf_plan c2r_plan = fftwf_plan_dft_c2r_2d(W, H, (fftwf_complex *)&fftw_c1[0], &fftw_r[0], FFTW_EXHAUSTIVE);
221 fftw_t = benchmark(samples, reps, [&]() { fftwf_execute(c2r_plan); }) * 1e6;
222#else
223 fftw_t = 0;
224#endif
225 printf("%12s %10.3f %10.2f %10.3f %10.2f %10.3g\n",
226 "c2r",
227 halide_t,
228 2.5 * W * H * (log2(W) + log2(H)) / halide_t,
229 fftw_t,
230 2.5 * W * H * (log2(W) + log2(H)) / fftw_t,
231 fftw_t / halide_t);
232
233#ifdef WITH_FFTW
234 fftwf_destroy_plan(c2c_plan);
235 fftwf_destroy_plan(r2c_plan);
236 fftwf_destroy_plan(c2r_plan);
237#endif
238
239 return 0;
