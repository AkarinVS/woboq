1#include "Halide.h"
2
3namespace {
4
5using namespace Halide::ConciseCasts;
6
7class Hist : public Halide::Generator<Hist> {
8public:
9 Input<Buffer<uint8_t>> input{"input", 3};
10 Output<Buffer<uint8_t>> output{"output", 3};
11
12 void generate() {
13 Var x("x"), y("y"), c("c");
14
15 // Algorithm
16 Func Y("Y");
17 Y(x, y) = (0.299f * input(x, y, 0) +
18 0.587f * input(x, y, 1) +
19 0.114f * input(x, y, 2));
20
21 Func Cr("Cr");
22 Expr R = input(x, y, 0);
23 Cr(x, y) = (R - Y(x, y)) * 0.713f + 128;
24
25 Func Cb("Cb");
26 Expr B = input(x, y, 2);
27 Cb(x, y) = (B - Y(x, y)) * 0.564f + 128;
28
29 Func hist_rows("hist_rows");
30 hist_rows(x, y) = 0;
31 RDom rx(0, input.width());
32 Expr bin = cast<int>(clamp(Y(rx, y), 0, 255));
33 hist_rows(bin, y) += 1;
34
35 Func hist("hist");
36 hist(x) = 0;
37 RDom ry(0, input.height());
38 hist(x) += hist_rows(x, ry);
39
40 Func cdf("cdf");
41 cdf(x) = hist(0);
42 RDom b(1, 255);
43 cdf(b.x) = cdf(b.x - 1) + hist(b.x);
44
45 Func cdf_bin("cdf_bin");
46 cdf_bin(x, y) = u8(clamp(Y(x, y), 0, 255));
47
48 Func eq("equalize");
49 eq(x, y) = clamp(cdf(cdf_bin(x, y)) * (255.0f / (input.height() * input.width())), 0, 255);
50
51 Expr red = u8(clamp(eq(x, y) + (Cr(x, y) - 128) * 1.4f, 0, 255));
52 Expr green = u8(clamp(eq(x, y) - 0.343f * (Cb(x, y) - 128) - 0.711f * (Cr(x, y) - 128), 0, 255));
53 Expr blue = u8(clamp(eq(x, y) + 1.765f * (Cb(x, y) - 128), 0, 255));
54 output(x, y, c) = mux(c, {red, green, blue});
55
56 // Estimates (for autoscheduler; ignored otherwise)
57 {
58 input.dim(0).set_estimate(0, 1536);
59 input.dim(1).set_estimate(0, 2560);
60 input.dim(2).set_estimate(0, 3);
61 output.dim(0).set_estimate(0, 1536);
62 output.dim(1).set_estimate(0, 2560);
63 output.dim(2).set_estimate(0, 3);
64 }
65
66 // Schedule
67 if (!auto_schedule) {
68 cdf.bound(x, 0, 256);
69
70 Var xi("xi"), yi("yi");
71 if (get_target().has_gpu_feature()) {
72 // 0.197ms on a 2060 RTX
73 Var yii;
74 RVar rxo, rxi;
75
76 if (get_target().has_feature(Target::CUDA)) {
77 // Each thread below will use atomic integer adds
78 // to shared to compute the histogram of a single
79 // row.
80 hist_rows
81 .in()
82 .compute_root()
83 .split(x, x, xi, 64)
84 .vectorize(xi, 2)
85 .unroll(x)
86 .gpu_lanes(xi)
87 .gpu_blocks(y);
88
89 hist_rows
90 .store_in(MemoryType::GPUShared)
91 .compute_at(hist_rows.in(), y)
92 .split(x, x, xi, 64)
93 .vectorize(xi, 2)
94 .unroll(x)
95 .gpu_lanes(xi)
96 .update()
97 .split(rx, rxo, rxi, 32)
98 .reorder(rxi, rxo, y)
99 .atomic()
100 .gpu_lanes(rxi);
101
102 Y.clone_in(hist_rows)
103 .compute_at(hist_rows, rxo)
104 .store_in(MemoryType::Register)
105 .gpu_lanes(x);
106 } else {
107 hist_rows.compute_root()
108 .gpu_tile(x, y, xi, yi, 32, 8);
109
110 const int slice_width = 256;
111 // Get more parallelism by not just taking
112 // histograms of rows, but histograms of small
113 // pieces of each row.
114 hist_rows.update()
115 .split(rx, rxo, rxi, slice_width);
116 Var z, zi;
117 Func intm = hist_rows.update().rfactor(rxo, z);
118
119 intm.in()
120 .compute_root()
121 .gpu_tile(y, z, yi, zi, 16, 1);
122
123 intm.compute_at(intm.in(), y)
124 .split(x, x, xi, 16)
125 .gpu_threads(xi)
126 .update()
127 .gpu_threads(y);
128
129 // hist_rows now just sums up the mini-histograms
130 // along the z dimension.
131 hist_rows.update().gpu_tile(x, y, xi, yi, 32, 8);
132
133 if (!get_target().has_feature(Target::Metal) &&
134 !get_target().has_feature(Target::D3D12Compute)) {
135 // bound_extent doesn't currently work inside
136 // metal & d3d12compute kernels because we can't compile the
137 // assertion. For metal & d3d12compute we just inline the
138 // luma computation.
139 Y.clone_in(intm)
140 .compute_at(intm.in(), y)
141 .split(x, x, xi, 16)
142 .bound_extent(x, 16)
143 .gpu_threads(xi);
144 }
145 }
146 hist.compute_root()
147 .gpu_tile(x, xi, 16)
148 .update()
149 .gpu_tile(x, xi, 16);
150 cdf.compute_root()
151 .gpu_tile(x, xi, 16)
152 .update()
153 .gpu_single_thread();
154 output.compute_root()
155 .reorder(c, x, y)
156 .bound(c, 0, 3)
157 .unroll(c)
158 .gpu_tile(x, y, xi, yi, 128, 4)
159 .vectorize(xi, 4);
160 Cb.compute_at(output, xi).vectorize(x);
161 Cr.compute_at(output, xi).vectorize(x);
162 eq.compute_at(output, xi).vectorize(x);
163 // Stage the LUT into shared memory
164 cdf.in()
165 .compute_at(output, x)
166 .split(x, x, xi, 64)
167 .vectorize(xi, 2)
168 .gpu_threads(xi, x);
169 } else {
170 // Runtime is noisy. 0.8ms - 1.1ms on an Intel
171 // i9-9960X using 16 threads
172
173 const int vec = natural_vector_size<float>();
174 // Make separate copies of Y to use while
175 // histogramming and while computing the output. It's
176 // better to redundantly luminance than reload it, but
177 // you don't want to inline it into the histogram
178 // computation because then it doesn't vectorize.
179 RVar rxo, rxi;
180 Y.clone_in(hist_rows)
181 .compute_at(hist_rows.in(), y)
182 .vectorize(x, vec);
183
184 hist_rows.in()
185 .compute_root()
186 .vectorize(x, vec)
187 .parallel(y, 4);
188 hist_rows.compute_at(hist_rows.in(), y)
189 .vectorize(x, vec)
190 .update()
191 .reorder(y, rx)
192 .unroll(y);
193 hist.compute_root()
194 .vectorize(x, vec)
195 .update()
196 .reorder(x, ry)
197 .vectorize(x, vec)
198 .unroll(x, 4)
199 .parallel(x)
200 .reorder(ry, x);
201
202 cdf.compute_root();
203 output.reorder(c, x, y)
204 .bound(c, 0, 3)
205 .unroll(c)
206 .parallel(y, 8)
207 .vectorize(x, vec * 2);
208 }
209 }
210 }
211};
