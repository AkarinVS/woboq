6constexpr int maxJ = 20;
7
8class LocalLaplacian : public Halide::Generator<LocalLaplacian> {
9public:
10 GeneratorParam<int> pyramid_levels{"pyramid_levels", 8, 1, maxJ};
11
12 Input<Buffer<uint16_t>> input{"input", 3};
13 Input<int> levels{"levels"};
14 Input<float> alpha{"alpha"};
15 Input<float> beta{"beta"};
16
17 Output<Buffer<uint16_t>> output{"output", 3};
18
19 void generate() {
20 /* THE ALGORITHM */
21 const int J = pyramid_levels;
22
23 // Make the remapping function as a lookup table.
24 Func remap;
25 Expr fx = cast<float>(x) / 256.0f;
26 remap(x) = alpha * fx * exp(-fx * fx / 2.0f);
27
28 // Set a boundary condition
29 Func clamped = Halide::BoundaryConditions::repeat_edge(input);
30
31 // Convert to floating point
32 Func floating;
33 floating(x, y, c) = clamped(x, y, c) / 65535.0f;
34
35 // Get the luminance channel
36 Func gray;
37 gray(x, y) = 0.299f * floating(x, y, 0) + 0.587f * floating(x, y, 1) + 0.114f * floating(x, y, 2);
38
39 // Make the processed Gaussian pyramid.
40 Func gPyramid[maxJ];
41 // Do a lookup into a lut with 256 entires per intensity level
42 Expr level = k * (1.0f / (levels - 1));
43 Expr idx = gray(x, y) * cast<float>(levels - 1) * 256.0f;
44 idx = clamp(cast<int>(idx), 0, (levels - 1) * 256);
45 gPyramid[0](x, y, k) = beta * (gray(x, y) - level) + level + remap(idx - 256 * k);
46 for (int j = 1; j < J; j++) {
47 gPyramid[j](x, y, k) = downsample(gPyramid[j - 1])(x, y, k);
48 }
49
50 // Get its laplacian pyramid
51 Func lPyramid[maxJ];
52 lPyramid[J - 1](x, y, k) = gPyramid[J - 1](x, y, k);
53 for (int j = J - 2; j >= 0; j--) {
54 lPyramid[j](x, y, k) = gPyramid[j](x, y, k) - upsample(gPyramid[j + 1])(x, y, k);
55 }
56
57 // Make the Gaussian pyramid of the input
58 Func inGPyramid[maxJ];
59 inGPyramid[0](x, y) = gray(x, y);
60 for (int j = 1; j < J; j++) {
61 inGPyramid[j](x, y) = downsample(inGPyramid[j - 1])(x, y);
62 }
63
64 // Make the laplacian pyramid of the output
65 Func outLPyramid[maxJ];
66 for (int j = 0; j < J; j++) {
67 // Split input pyramid value into integer and floating parts
68 Expr level = inGPyramid[j](x, y) * cast<float>(levels - 1);
69 Expr li = clamp(cast<int>(level), 0, levels - 2);
70 Expr lf = level - cast<float>(li);
71 // Linearly interpolate between the nearest processed pyramid levels
72 outLPyramid[j](x, y) = (1.0f - lf) * lPyramid[j](x, y, li) + lf * lPyramid[j](x, y, li + 1);
73 }
74
75 // Make the Gaussian pyramid of the output
76 Func outGPyramid[maxJ];
77 outGPyramid[J - 1](x, y) = outLPyramid[J - 1](x, y);
78 for (int j = J - 2; j >= 0; j--) {
79 outGPyramid[j](x, y) = upsample(outGPyramid[j + 1])(x, y) + outLPyramid[j](x, y);
80 }
81
82 // Reintroduce color (Connelly: use eps to avoid scaling up noise w/ apollo3.png input)
83 Func color;
84 float eps = 0.01f;
85 color(x, y, c) = outGPyramid[0](x, y) * (floating(x, y, c) + eps) / (gray(x, y) + eps);
86
87 // Convert back to 16-bit
88 output(x, y, c) = cast<uint16_t>(clamp(color(x, y, c), 0.0f, 1.0f) * 65535.0f);
89
90 /* ESTIMATES */
91 // (This can be useful in conjunction with RunGen and benchmarks as well
92 // as auto-schedule, so we do it in all cases.)
93 input.set_estimates({{0, 1536}, {0, 2560}, {0, 3}});
94 // Provide estimates on the parameters
95 levels.set_estimate(8);
96 alpha.set_estimate(1);
97 beta.set_estimate(1);
98 // Provide estimates on the pipeline output
99 output.set_estimates({{0, 1536}, {0, 2560}, {0, 3}});
100
101 /* THE SCHEDULE */
102 if (auto_schedule) {
103 // Nothing.
104 } else if (get_target().has_gpu_feature()) {
105 // GPU schedule.
106 // 3.19ms on an RTX 2060.
107 remap.compute_root();
108 Var xi, yi;
109 output.compute_root().gpu_tile(x, y, xi, yi, 16, 8);
110 for (int j = 0; j < J; j++) {
111 int blockw = 16, blockh = 8;
112 if (j > 3) {
113 blockw = 2;
114 blockh = 2;
115 }
116 if (j > 0) {
117 inGPyramid[j].compute_root().gpu_tile(x, y, xi, yi, blockw, blockh);
118 gPyramid[j].compute_root().reorder(k, x, y).gpu_tile(x, y, xi, yi, blockw, blockh);
119 }
120 outGPyramid[j].compute_root().gpu_tile(x, y, xi, yi, blockw, blockh);
121 }
122 } else {
123 // CPU schedule.
124
125 // 21.4ms on an Intel i9-9960X using 32 threads at 3.7
126 // GHz, using the target x86-64-avx2.
127
128 // This app is dominated by data-dependent loads from
129 // memory, so we're better off leaving the AVX-512 units
130 // off in exchange for a higher clock, and we benefit from
131 // hyperthreading.
132
133 remap.compute_root();
134 Var yo;
135 output.reorder(c, x, y).split(y, yo, y, 64).parallel(yo).vectorize(x, 8);
136 gray.compute_root().parallel(y, 32).vectorize(x, 8);
137 for (int j = 1; j < 5; j++) {
138 inGPyramid[j]
139 .compute_root()
140 .parallel(y, 32)
141 .vectorize(x, 8);
142 gPyramid[j]
143 .compute_root()
144 .reorder_storage(x, k, y)
145 .reorder(k, y)
146 .parallel(y, 8)
147 .vectorize(x, 8);
148 outGPyramid[j]
149 .store_at(output, yo)
150 .compute_at(output, y)
151 .fold_storage(y, 4)
152 .vectorize(x, 8);
153 }
154 outGPyramid[0].compute_at(output, y).vectorize(x, 8);
155 for (int j = 5; j < J; j++) {
156 inGPyramid[j].compute_root();
157 gPyramid[j].compute_root().parallel(k);
158 outGPyramid[j].compute_root();
159 }
160 }
161
162 /* Optional tags to specify layout for HalideTraceViz */
163 {
164 Halide::Trace::FuncConfig cfg;
165 cfg.color_dim = 2;
166 cfg.max = 65535;
167 cfg.pos.x = 30;
168 cfg.pos.y = 100;
169 input.add_trace_tag(cfg.to_trace_tag());
170
171 cfg.pos.x = 1700;
172 output.add_trace_tag(cfg.to_trace_tag());
173 }
174
175 {
176 Halide::Trace::FuncConfig cfg;
177 cfg.store_cost = 5;
178 cfg.pos.x = 370;
179 cfg.pos.y = 100;
180 cfg.labels = {{"input pyramid", {-90, -68}}};
181 gray.add_trace_tag(cfg.to_trace_tag());
182 }
183
184 for (int i = 0; i < pyramid_levels; ++i) {
185 int y = 100;
186 for (int j = 0; j < i; ++j) {
187 y += 500 >> j;
188 }
189 {
190 int x = 370;
191 int store_cost = 1 << (i + 1);
192 Halide::Trace::FuncConfig cfg;
193 cfg.pos = {x, y};
194 cfg.store_cost = store_cost;
195 inGPyramid[i].add_trace_tag(cfg.to_trace_tag());
196 }
197 {
198 int x = 720;
199 int store_cost = 1 << i;
200 Halide::Trace::FuncConfig cfg;
201 cfg.strides = {{1, 0}, {0, 1}, {200, 0}};
202 cfg.pos = {x, y};
203 cfg.store_cost = store_cost;
204 if (i == 1) {
205 cfg.labels = {{"differently curved intermediate pyramids"}};
206 }
207 gPyramid[i].add_trace_tag(cfg.to_trace_tag());
208 }
209 {
210 int x = 1500;
211 int store_cost = (1 << i) * 10;
212 Halide::Trace::FuncConfig cfg;
213 cfg.pos = {x, y};
214 cfg.store_cost = store_cost;
215 if (i == 0) {
216 cfg.labels = {{"output pyramids"}};
217 cfg.pos = {x, 100};
218 }
219 outGPyramid[i].add_trace_tag(cfg.to_trace_tag());
220 }
221 }
222 }
223
224private:
225 Var x, y, c, k;
226
227 // Downsample with a 1 3 3 1 filter
228 Func downsample(Func f) {
229 using Halide::_;
230 Func downx, downy;
231 downx(x, y, _) = (f(2 * x - 1, y, _) + 3.0f * (f(2 * x, y, _) + f(2 * x + 1, y, _)) + f(2 * x + 2, y, _)) / 8.0f;
232 downy(x, y, _) = (downx(x, 2 * y - 1, _) + 3.0f * (downx(x, 2 * y, _) + downx(x, 2 * y + 1, _)) + downx(x, 2 * y + 2, _)) / 8.0f;
233 return downy;
234 }
235
236 // Upsample using bilinear interpolation
237 Func upsample(Func f) {
238 using Halide::_;
239 Func upx, upy;
240 upx(x, y, _) = lerp(f((x + 1) / 2, y, _), f((x - 1) / 2, y, _), ((x % 2) * 2 + 1) / 4.0f);
241 upy(x, y, _) = lerp(upx(x, (y + 1) / 2, _), upx(x, (y - 1) / 2, _), ((y % 2) * 2 + 1) / 4.0f);
242 return upy;
243 }
244};
