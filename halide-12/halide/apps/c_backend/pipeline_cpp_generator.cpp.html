51class PipelineCpp : public Halide::Generator<PipelineCpp> {
52public:
53 Input<Buffer<uint16_t>> input{"input", 2};
54 Output<Buffer<uint16_t>> output{"output", 2};
55
56 void generate() {
57 Var x, y;
58
59 assert(get_target().has_feature(Target::CPlusPlusMangling));
60
61 Expr add_all_the_things = cast<int>(0);
62 add_all_the_things += make_call_cpp_extern_toplevel(input(x, y), x + y);
63
64 add_all_the_things += make_call_cpp_extern_namespace(input(x, y), x + y);
65
66 add_all_the_things += make_call_cpp_extern_shared_namespace_1(input(x, y), x + y);
67 add_all_the_things += make_call_cpp_extern_shared_namespace_2(input(x, y), x + y);
68 add_all_the_things += make_call_cpp_extern_shared_namespace_3(input(x, y), x + y);
69
70 add_all_the_things += make_call_cpp_extern_nested_namespace_outer(input(x, y), x + y);
71 add_all_the_things += make_call_cpp_extern_nested_namespace_inner(input(x, y), x + y);
72
73 add_all_the_things += make_call_cpp_extern_shared_nested_1(input(x, y), x + y);
74 add_all_the_things += make_call_cpp_extern_shared_nested_2(input(x, y), x + y);
75 add_all_the_things += make_call_cpp_extern_shared_nested_3(input(x, y), x + y);
76 add_all_the_things += make_call_cpp_extern_shared_nested_4(input(x, y), x + y);
77
78 add_all_the_things += an_extern_c_func(cast<int32_t>(input(x, y)), cast<float>(x + y));
79
80 output(x, y) = cast<uint16_t>(add_all_the_things);
81 }
82};
