33/// A struct capturing PGO tunables.
34struct PGOOptions {
35 enum PGOAction { NoAction, IRInstr, IRUse, SampleUse };
36 enum CSPGOAction { NoCSAction, CSIRInstr, CSIRUse };
37 PGOOptions(std::string ProfileFile = "", std::string CSProfileGenFile = "",
38 std::string ProfileRemappingFile = "", PGOAction Action = NoAction,
39 CSPGOAction CSAction = NoCSAction,
40 bool DebugInfoForProfiling = false,
41 bool PseudoProbeForProfiling = false)
42 : ProfileFile(ProfileFile), CSProfileGenFile(CSProfileGenFile),
43 ProfileRemappingFile(ProfileRemappingFile), Action(Action),
44 CSAction(CSAction), DebugInfoForProfiling(DebugInfoForProfiling ||
45 (Action == SampleUse &&
46 !PseudoProbeForProfiling)),
47 PseudoProbeForProfiling(PseudoProbeForProfiling) {
48 // Note, we do allow ProfileFile.empty() for Action=IRUse LTO can
49 // callback with IRUse action without ProfileFile.
50
51 // If there is a CSAction, PGOAction cannot be IRInstr or SampleUse.
52 assert(this->CSAction == NoCSAction ||
53 (this->Action != IRInstr && this->Action != SampleUse));
54
55 // For CSIRInstr, CSProfileGenFile also needs to be nonempty.
56 assert(this->CSAction != CSIRInstr || !this->CSProfileGenFile.empty());
57
58 // If CSAction is CSIRUse, PGOAction needs to be IRUse as they share
59 // a profile.
60 assert(this->CSAction != CSIRUse || this->Action == IRUse);
61
62 // If neither Action nor CSAction, DebugInfoForProfiling or
63 // PseudoProbeForProfiling needs to be true.
64 assert(this->Action != NoAction || this->CSAction != NoCSAction ||
65 this->DebugInfoForProfiling || this->PseudoProbeForProfiling);
66
67 // Pseudo probe emission does not work with -fdebug-info-for-profiling since
68 // they both use the discriminator field of debug lines but for different
69 // purposes.
70 if (this->DebugInfoForProfiling && this->PseudoProbeForProfiling) {
71 report_fatal_error(
72 "Pseudo probes cannot be used with -debug-info-for-profiling", false);
73 }
74 }
75 std::string ProfileFile;
76 std::string CSProfileGenFile;
77 std::string ProfileRemappingFile;
78 PGOAction Action;
79 CSPGOAction CSAction;
80 bool DebugInfoForProfiling;
81 bool PseudoProbeForProfiling;
82};

149 /// A struct to capture parsed pass pipeline names.
150 ///
151 /// A pipeline is defined as a series of names, each of which may in itself
152 /// recursively contain a nested pipeline. A name is either the name of a pass
153 /// (e.g. "instcombine") or the name of a pipeline type (e.g. "cgscc"). If the
154 /// name is the name of a pass, the InnerPipeline is empty, since passes
155 /// cannot contain inner pipelines. See parsePassPipeline() for a more
156 /// detailed description of the textual pipeline format.
157 struct PipelineElement {
158 StringRef Name;
159 std::vector<PipelineElement> InnerPipeline;
160 };

162 /// LLVM-provided high-level optimization levels.
163 ///
164 /// This enumerates the LLVM-provided high-level optimization levels. Each
165 /// level has a specific goal and rationale.
166 class OptimizationLevel final {
167 unsigned SpeedLevel = 2;
168 unsigned SizeLevel = 0;
169 OptimizationLevel(unsigned SpeedLevel, unsigned SizeLevel)
170 : SpeedLevel(SpeedLevel), SizeLevel(SizeLevel) {
171 // Check that only valid combinations are passed.
172 assert(SpeedLevel <= 3 &&
173 "Optimization level for speed should be 0, 1, 2, or 3");
174 assert(SizeLevel <= 2 &&
175 "Optimization level for size should be 0, 1, or 2");
176 assert((SizeLevel == 0 || SpeedLevel == 2) &&
177 "Optimize for size should be encoded with speedup level == 2");
178 }
179
180 public:
181 OptimizationLevel() = default;
182 /// Disable as many optimizations as possible. This doesn't completely
183 /// disable the optimizer in all cases, for example always_inline functions
184 /// can be required to be inlined for correctness.
185 static const OptimizationLevel O0;
186
187 /// Optimize quickly without destroying debuggability.
188 ///
189 /// This level is tuned to produce a result from the optimizer as quickly
190 /// as possible and to avoid destroying debuggability. This tends to result
191 /// in a very good development mode where the compiled code will be
192 /// immediately executed as part of testing. As a consequence, where
193 /// possible, we would like to produce efficient-to-execute code, but not
194 /// if it significantly slows down compilation or would prevent even basic
195 /// debugging of the resulting binary.
196 ///
197 /// As an example, complex loop transformations such as versioning,
198 /// vectorization, or fusion don't make sense here due to the degree to
199 /// which the executed code differs from the source code, and the compile
200 /// time cost.
201 static const OptimizationLevel O1;
202 /// Optimize for fast execution as much as possible without triggering
203 /// significant incremental compile time or code size growth.
204 ///
205 /// The key idea is that optimizations at this level should "pay for
206 /// themselves". So if an optimization increases compile time by 5% or
207 /// increases code size by 5% for a particular benchmark, that benchmark
208 /// should also be one which sees a 5% runtime improvement. If the compile
209 /// time or code size penalties happen on average across a diverse range of
210 /// LLVM users' benchmarks, then the improvements should as well.
211 ///
212 /// And no matter what, the compile time needs to not grow superlinearly
213 /// with the size of input to LLVM so that users can control the runtime of
214 /// the optimizer in this mode.
215 ///
216 /// This is expected to be a good default optimization level for the vast
217 /// majority of users.
218 static const OptimizationLevel O2;
219 /// Optimize for fast execution as much as possible.
220 ///
221 /// This mode is significantly more aggressive in trading off compile time
222 /// and code size to get execution time improvements. The core idea is that
223 /// this mode should include any optimization that helps execution time on
224 /// balance across a diverse collection of benchmarks, even if it increases
225 /// code size or compile time for some benchmarks without corresponding
226 /// improvements to execution time.
227 ///
228 /// Despite being willing to trade more compile time off to get improved
229 /// execution time, this mode still tries to avoid superlinear growth in
230 /// order to make even significantly slower compile times at least scale
231 /// reasonably. This does not preclude very substantial constant factor
232 /// costs though.
233 static const OptimizationLevel O3;
234 /// Similar to \c O2 but tries to optimize for small code size instead of
235 /// fast execution without triggering significant incremental execution
236 /// time slowdowns.
237 ///
238 /// The logic here is exactly the same as \c O2, but with code size and
239 /// execution time metrics swapped.
240 ///
241 /// A consequence of the different core goal is that this should in general
242 /// produce substantially smaller executables that still run in
243 /// a reasonable amount of time.
244 static const OptimizationLevel Os;
245 /// A very specialized mode that will optimize for code size at any and all
246 /// costs.
247 ///
248 /// This is useful primarily when there are absolute size limitations and
249 /// any effort taken to reduce the size is worth it regardless of the
250 /// execution time impact. You should expect this level to produce rather
251 /// slow, but very small, code.
252 static const OptimizationLevel Oz;
253
254 bool isOptimizingForSpeed() const {
255 return SizeLevel == 0 && SpeedLevel > 0;
256 }
257
258 bool isOptimizingForSize() const { return SizeLevel > 0; }
259
260 bool operator==(const OptimizationLevel &Other) const {
261 return SizeLevel == Other.SizeLevel && SpeedLevel == Other.SpeedLevel;
262 }
263 bool operator!=(const OptimizationLevel &Other) const {
264 return SizeLevel != Other.SizeLevel || SpeedLevel != Other.SpeedLevel;
265 }
266
267 unsigned getSpeedupLevel() const { return SpeedLevel; }
268
269 unsigned getSizeLevel() const { return SizeLevel; }
270 };

317 /// Construct the core LLVM function canonicalization and simplification
318 /// pipeline.
319 ///
320 /// This is a long pipeline and uses most of the per-function optimization
321 /// passes in LLVM to canonicalize and simplify the IR. It is suitable to run
322 /// repeatedly over the IR and is not expected to destroy important
323 /// information about the semantics of the IR.
324 ///
325 /// Note that \p Level cannot be `O0` here. The pipelines produced are
326 /// only intended for use when attempting to optimize code. If frontends
327 /// require some transformations for semantic reasons, they should explicitly
328 /// build them.
329 ///
330 /// \p Phase indicates the current ThinLTO phase.
331 FunctionPassManager
332 buildFunctionSimplificationPipeline(OptimizationLevel Level,
333 ThinOrFullLTOPhase Phase);
334
335 /// Construct the core LLVM module canonicalization and simplification
336 /// pipeline.
337 ///
338 /// This pipeline focuses on canonicalizing and simplifying the entire module
339 /// of IR. Much like the function simplification pipeline above, it is
340 /// suitable to run repeatedly over the IR and is not expected to destroy
341 /// important information. It does, however, perform inlining and other
342 /// heuristic based simplifications that are not strictly reversible.
343 ///
344 /// Note that \p Level cannot be `O0` here. The pipelines produced are
345 /// only intended for use when attempting to optimize code. If frontends
346 /// require some transformations for semantic reasons, they should explicitly
347 /// build them.

349 /// \p Phase indicates the current ThinLTO phase.

458 /// Parse a textual pass pipeline description into a \c
459 /// ModulePassManager.
460 ///
461 /// The format of the textual pass pipeline description looks something like:
462 ///
463 /// module(function(instcombine,sroa),dce,cgscc(inliner,function(...)),...)
464 ///
465 /// Pass managers have ()s describing the nest structure of passes. All passes
466 /// are comma separated. As a special shortcut, if the very first pass is not
467 /// a module pass (as a module pass manager is), this will automatically form
468 /// the shortest stack of pass managers that allow inserting that first pass.
469 /// So, assuming function passes 'fpassN', CGSCC passes 'cgpassN', and loop
470 /// passes 'lpassN', all of these are valid:
471 ///
472 /// fpass1,fpass2,fpass3
473 /// cgpass1,cgpass2,cgpass3
474 /// lpass1,lpass2,lpass3
475 ///
476 /// And they are equivalent to the following (resp.):
477 ///
478 /// module(function(fpass1,fpass2,fpass3))
479 /// module(cgscc(cgpass1,cgpass2,cgpass3))
480 /// module(function(loop(lpass1,lpass2,lpass3)))
481 ///
482 /// This shortcut is especially useful for debugging and testing small pass
483 /// combinations.
484 ///
485 /// The sequence of passes aren't necessarily the exact same kind of pass.
486 /// You can mix different levels implicitly if adaptor passes are defined to
487 /// make them work. For example,
488 ///
489 /// mpass1,fpass1,fpass2,mpass2,lpass1
490 ///
491 /// This pipeline uses only one pass manager: the top-level module manager.
492 /// fpass1,fpass2 and lpass1 are added into the the top-level module manager
493 /// using only adaptor passes. No nested function/loop pass managers are
494 /// added. The purpose is to allow easy pass testing when the user
495 /// specifically want the pass to run under a adaptor directly. This is
496 /// preferred when a pipeline is largely of one type, but one or just a few
497 /// passes are of different types(See PassBuilder.cpp for examples).
498 Error parsePassPipeline(ModulePassManager &MPM, StringRef PipelineText);

507 Error parsePassPipeline(CGSCCPassManager &CGPM, StringRef PipelineText);

792/// This utility template takes care of adding require<> and invalidate<>
793/// passes for an analysis to a given \c PassManager. It is intended to be used
794/// during parsing of a pass pipeline when parsing a single PipelineName.
795/// When registering a new function analysis FancyAnalysis with the pass
796/// pipeline name "fancy-analysis", a matching ParsePipelineCallback could look
797/// like this:
798///
799/// static bool parseFunctionPipeline(StringRef Name, FunctionPassManager &FPM,
800/// ArrayRef<PipelineElement> P) {
801/// if (parseAnalysisUtilityPasses<FancyAnalysis>("fancy-analysis", Name,
802/// FPM))
803/// return true;
804/// return false;
805/// }
806template <typename AnalysisT, typename IRUnitT, typename AnalysisManagerT,
807 typename... ExtraArgTs>
808bool parseAnalysisUtilityPasses(
809 StringRef AnalysisName, StringRef PipelineName,
810 PassManager<IRUnitT, AnalysisManagerT, ExtraArgTs...> &PM) {
811 if (!PipelineName.endswith(">"))
812 return false;
813 // See if this is an invalidate<> pass name
814 if (PipelineName.startswith("invalidate<")) {
815 PipelineName = PipelineName.substr(11, PipelineName.size() - 12);
816 if (PipelineName != AnalysisName)
817 return false;
818 PM.addPass(InvalidateAnalysisPass<AnalysisT>());
819 return true;
820 }
821
822 // See if this is a require<> pass name
823 if (PipelineName.startswith("require<")) {
824 PipelineName = PipelineName.substr(8, PipelineName.size() - 9);
825 if (PipelineName != AnalysisName)
826 return false;
827 PM.addPass(RequireAnalysisPass<AnalysisT, IRUnitT, AnalysisManagerT,
828 ExtraArgTs...>());
829 return true;
830 }
831
832 return false;
833}
