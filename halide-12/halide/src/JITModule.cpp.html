60void load_opengl(bool needs_egl) {
61#if defined(__linux__)
62 if (have_symbol("glXGetCurrentContext") && have_symbol("glDeleteTextures")) {
63 debug(1) << "OpenGL support code already linked in...\n";
64 } else {
65 debug(1) << "Looking for OpenGL support code...\n";
66 string error;
67 if (needs_egl) {
68 // NVIDIA EGL prefers users to load libOpenGL.so instead of libGL.so
69 // The way we're using it, it seems like libGL.so.1 is a valid fallback.
70 // See here for more details: https://developer.nvidia.com/blog/linking-opengl-server-side-rendering
71 llvm::sys::DynamicLibrary::LoadLibraryPermanently("libOpenGL.so.0", &error);
72 if (!error.empty()) {
73 debug(1) << "Could not find libOpenGL.so.0 when EGL requested. Falling back to libGL.so.1\n";
74 llvm::sys::DynamicLibrary::LoadLibraryPermanently("libGL.so.1", &error);
75 }
76 user_assert(error.empty()) << "Could not find libOpenGL.so.0 or libGL.so.1\n";
77 llvm::sys::DynamicLibrary::LoadLibraryPermanently("libEGL.so.1", &error);
78 user_assert(error.empty()) << "Could not find libEGL.so.1\n";
79 } else {
80 llvm::sys::DynamicLibrary::LoadLibraryPermanently("libGL.so.1", &error);
81 user_assert(error.empty()) << "Could not find libGL.so\n";
82 llvm::sys::DynamicLibrary::LoadLibraryPermanently("libX11.so.6", &error);
83 user_assert(error.empty()) << "Could not find libX11.so.6\n";
84 }
85 }
86#elif defined(__APPLE__)
87 if (have_symbol("aglCreateContext") && have_symbol("glDeleteTextures")) {
88 debug(1) << "OpenGL support code already linked in...\n";
89 } else {
90 debug(1) << "Looking for OpenGL support code...\n";
91 string error;
92 llvm::sys::DynamicLibrary::LoadLibraryPermanently("/System/Library/Frameworks/AGL.framework/AGL", &error);
93 user_assert(error.empty()) << "Could not find AGL.framework\n";
94 llvm::sys::DynamicLibrary::LoadLibraryPermanently("/System/Library/Frameworks/OpenGL.framework/OpenGL", &error);
95 user_assert(error.empty()) << "Could not find OpenGL.framework\n";
96 }
97#else
98 internal_error << "JIT support for OpenGL on anything other than linux or OS X not yet implemented\n";
99#endif
100}

157// Retrieve a function pointer from an llvm module, possibly by compiling it.
158JITModule::Symbol compile_and_get_function(ExecutionEngine &ee, const string &name) {
159 debug(2) << "JIT Compiling " << name << "\n";
160 llvm::Function *fn = ee.FindFunctionNamed(name);
161 internal_assert(fn->getName() == name);
162 void *f = (void *)ee.getFunctionAddress(name);
163 if (!f) {
164 internal_error << "Compiling " << name << " returned nullptr\n";
165 }
166
167 JITModule::Symbol symbol(f);
168
169 debug(2) << "Function " << name << " is at " << f << "\n";
170
171 return symbol;
172}

174// Expand LLVM's search for symbols to include code contained in a set of JITModule.
175class HalideJITMemoryManager : public SectionMemoryManager {
176 std::vector<JITModule> modules;
177 std::vector<std::pair<uint8_t *, size_t>> code_pages;
178
179public:
180 HalideJITMemoryManager(const std::vector<JITModule> &modules)
181 : modules(modules) {
182 }
183
184 uint64_t getSymbolAddress(const std::string &name) override {
185 for (size_t i = 0; i < modules.size(); i++) {
186 const JITModule &m = modules[i];
187 std::map<std::string, JITModule::Symbol>::const_iterator iter = m.exports().find(name);
188 if (iter == m.exports().end() && starts_with(name, "_")) {
189 iter = m.exports().find(name.substr(1));
190 }
191 if (iter != m.exports().end()) {
192 return (uint64_t)iter->second.address;
193 }
194 }
195 uint64_t result = SectionMemoryManager::getSymbolAddress(name);
196#if defined(__GNUC__) && defined(__i386__)
197 // This is a workaround for an odd corner case (cross-compiling + testing
198 // Python bindings x86-32 on an x86-64 system): __udivdi3 is a helper function
199 // that GCC uses to do u64/u64 division on 32-bit systems; it's usually included
200 // by the linker on these systems as needed. When we JIT, LLVM will include references
201 // to this call; MCJIT fixes up these references by doing (roughly) dlopen(NULL)
202 // to look up the symbol. For normal JIT tests, this works fine, as dlopen(NULL)
203 // finds the test executable, which has the right lookups to locate it inside libHalide.so.
204 // If, however, we are running a JIT-via-Python test, dlopen(NULL) returns the
205 // CPython executable... which apparently *doesn't* include this as an exported
206 // function, so the lookup fails and crashiness ensues. So our workaround here is
207 // a bit icky, but expedient: check for this name if we can't find it elsewhere,
208 // and if so, return the one we know should be present. (Obviously, if other runtime
209 // helper functions of this sort crop up in the future, this should be expanded
210 // into a "builtins map".)
211 if (result == 0 && name == "__udivdi3") {
212 result = (uint64_t)&__udivdi3;
213 }
214#endif
215 internal_assert(result != 0)
216 << "HalideJITMemoryManager: unable to find address for " << name << "\n";
217 return result;
218 }
219
220 uint8_t *allocateCodeSection(uintptr_t size, unsigned alignment, unsigned section_id, StringRef section_name) override {
221 uint8_t *result = SectionMemoryManager::allocateCodeSection(size, alignment, section_id, section_name);
222 code_pages.emplace_back(result, size);
223 return result;
224 }
225};

245void JITModule::compile_module(std::unique_ptr<llvm::Module> m, const string &function_name, const Target &target,
246 const std::vector<JITModule> &dependencies,
247 const std::vector<std::string> &requested_exports) {
248
249 // Ensure that LLVM is initialized
250 CodeGen_LLVM::initialize_llvm();
251
252 // Make the execution engine
253 debug(2) << "Creating new execution engine\n";
254 debug(2) << "Target triple: " << m->getTargetTriple() << "\n";
255 string error_string;
256
257 string mcpu;
258 string mattrs;
259 llvm::TargetOptions options;
260 get_target_options(*m, options, mcpu, mattrs);
261
262 DataLayout initial_module_data_layout = m->getDataLayout();
263 string module_name = m->getModuleIdentifier();
264
265 llvm::EngineBuilder engine_builder((std::move(m)));
266 engine_builder.setTargetOptions(options);
267 engine_builder.setErrorStr(&error_string);
268 engine_builder.setEngineKind(llvm::EngineKind::JIT);
269 HalideJITMemoryManager *memory_manager = new HalideJITMemoryManager(dependencies);
270 engine_builder.setMCJITMemoryManager(std::unique_ptr<RTDyldMemoryManager>(memory_manager));
271
272 engine_builder.setOptLevel(CodeGenOpt::Aggressive);
273 if (!mcpu.empty()) {
274 engine_builder.setMCPU(mcpu);
275 }
276 std::vector<string> mattrs_array = {mattrs};
277 engine_builder.setMAttrs(mattrs_array);
278
279 TargetMachine *tm = engine_builder.selectTarget();
280 internal_assert(tm) << error_string << "\n";
281 DataLayout target_data_layout(tm->createDataLayout());
282 if (initial_module_data_layout != target_data_layout) {
283 internal_error << "Warning: data layout mismatch between module ("
284 << initial_module_data_layout.getStringRepresentation()
285 << ") and what the execution engine expects ("
286 << target_data_layout.getStringRepresentation() << ")\n";
287 }
288 ExecutionEngine *ee = engine_builder.create(tm);
289
290 if (!ee) {
291 std::cerr << error_string << "\n";
292 }
293 internal_assert(ee) << "Couldn't create execution engine\n";
294
295 // Do any target-specific initialization
296 std::vector<llvm::JITEventListener *> listeners;
297
298 if (target.arch == Target::X86) {
299 listeners.push_back(llvm::JITEventListener::createIntelJITEventListener());
300 }
301 // TODO: If this ever works in LLVM, this would allow profiling of JIT code with symbols with oprofile.
302 //listeners.push_back(llvm::createOProfileJITEventListener());
303
304 for (size_t i = 0; i < listeners.size(); i++) {
305 ee->RegisterJITEventListener(listeners[i]);
306 }
307
308 // Retrieve function pointers from the compiled module (which also
309 // triggers compilation)
310 debug(1) << "JIT compiling " << module_name
311 << " for " << target.to_string() << "\n";
312
313 std::map<std::string, Symbol> exports;
314
315 Symbol entrypoint;
316 Symbol argv_entrypoint;
317 if (!function_name.empty()) {
318 entrypoint = compile_and_get_function(*ee, function_name);
319 exports[function_name] = entrypoint;
320 argv_entrypoint = compile_and_get_function(*ee, function_name + "_argv");
321 exports[function_name + "_argv"] = argv_entrypoint;
322 }
323
324 for (size_t i = 0; i < requested_exports.size(); i++) {
325 exports[requested_exports[i]] = compile_and_get_function(*ee, requested_exports[i]);
326 }
327
328 debug(2) << "Finalizing object\n";
329 ee->finalizeObject();
330 // Do any target-specific post-compilation module meddling
331 for (size_t i = 0; i < listeners.size(); i++) {
332 ee->UnregisterJITEventListener(listeners[i]);
333 delete listeners[i];
334 }
335 listeners.clear();
336
337 // TODO: I don't think this is necessary, we shouldn't have any static constructors
338 ee->runStaticConstructorsDestructors(false);
339
340 // Stash the various objects that need to stay alive behind a reference-counted pointer.
341 jit_module->exports = exports;
342 jit_module->execution_engine = ee;
343 jit_module->dependencies = dependencies;
344 jit_module->entrypoint = entrypoint;
345 jit_module->argv_entrypoint = argv_entrypoint;
346 jit_module->name = function_name;
347}

349/*static*/
350JITModule JITModule::make_trampolines_module(const Target &target_arg,
351 const std::map<std::string, JITExtern> &externs,
352 const std::string &suffix,
353 const std::vector<JITModule> &deps) {
354 Target target = target_arg;
355 target.set_feature(Target::JIT);
356
357 JITModule result;
358 std::vector<std::pair<std::string, ExternSignature>> extern_signatures;
359 std::vector<std::string> requested_exports;
360 for (const std::pair<const std::string, JITExtern> &e : externs) {
361 const std::string &callee_name = e.first;
362 const std::string wrapper_name = callee_name + suffix;
363 const ExternCFunction &extern_c = e.second.extern_c_function();
364 result.add_extern_for_export(callee_name, extern_c);
365 requested_exports.push_back(wrapper_name);
366 extern_signatures.emplace_back(callee_name, extern_c.signature());
367 }
368
369 std::unique_ptr<llvm::Module> llvm_module = CodeGen_LLVM::compile_trampolines(
370 target, result.jit_module->context, suffix, extern_signatures);
371
372 result.compile_module(std::move(llvm_module), /*function_name*/ "", target, deps, requested_exports);
373
374 return result;
375}

381JITModule::Symbol JITModule::find_symbol_by_name(const std::string &name) const {
382 std::map<std::string, JITModule::Symbol>::iterator it = jit_module->exports.find(name);
383 if (it != jit_module->exports.end()) {
384 return it->second;
385 }
386 for (const JITModule &dep : jit_module->dependencies) {
387 JITModule::Symbol s = dep.find_symbol_by_name(name);
388 if (s.address) {
389 return s;
390 }
391 }
392 return JITModule::Symbol();

411static bool module_already_in_graph(const JITModuleContents *start, const JITModuleContents *target, std::set<const JITModuleContents *> &already_seen) {
412 if (start == target) {
413 return true;
414 }
415 if (already_seen.count(start) != 0) {
416 return false;
417 }
418 already_seen.insert(start);
419 for (const JITModule &dep_holder : start->dependencies) {
420 const JITModuleContents *dep = dep_holder.jit_module.get();
421 if (module_already_in_graph(dep, target, already_seen)) {
422 return true;
423 }
424 }
425 return false;
426}

428void JITModule::add_dependency(JITModule &dep) {
429 std::set<const JITModuleContents *> already_seen;
430 internal_assert(!module_already_in_graph(dep.jit_module.get(), jit_module.get(), already_seen)) << "JITModule::add_dependency: creating circular dependency graph.\n";
431 jit_module->dependencies.push_back(dep);
432}

612// The Halide runtime is broken up into pieces so that state can be
613// shared across JIT compilations that do not use the same target
614// options. At present, the split is into a MainShared module that
615// contains most of the runtime except for device API specific code
616// (GPU runtimes). There is one shared runtime per device API and a
617// the JITModule for a Func depends on all device API modules
618// specified in the target when it is JITted. (Instruction set variant
619// specific code, such as math routines, is inlined into the module
620// produced by compiling a Func so it can be specialized exactly for
621// each target.)
622enum RuntimeKind {
623 MainShared,
624 OpenCL,
625 Metal,
626 CUDA,
627 OpenGLCompute,
628 Hexagon,
629 D3D12Compute,
630 OpenCLDebug,
631 MetalDebug,
632 CUDADebug,
633 OpenGLComputeDebug,
634 HexagonDebug,
635 D3D12ComputeDebug,
636 MaxRuntimeKind
637};
638
639JITModule &shared_runtimes(RuntimeKind k) {
640 // We're already guarded by the shared_runtimes_mutex
641 static JITModule *m = nullptr;
642 if (!m) {
643 // Note that this is never freed. On windows this would invoke
644 // static destructors that use threading objects, and these
645 // don't work (crash or deadlock) after main exits.
646 m = new JITModule[MaxRuntimeKind];
647 }
648 return m[k];
649}

651JITModule &make_module(llvm::Module *for_module, Target target,
652 RuntimeKind runtime_kind, const std::vector<JITModule> &deps,
653 bool create) {
654
655 JITModule &runtime = shared_runtimes(runtime_kind);
656 if (!runtime.compiled() && create) {
657 // Ensure that JIT feature is set on target as it must be in
658 // order for the right runtime components to be added.
659 target.set_feature(Target::JIT);
660 // msan doesn't work for jit modules
661 target.set_feature(Target::MSAN, false);
662
663 Target one_gpu(target);
664 one_gpu.set_feature(Target::Debug, false);
665 one_gpu.set_feature(Target::OpenCL, false);
666 one_gpu.set_feature(Target::Metal, false);
667 one_gpu.set_feature(Target::CUDA, false);
668 one_gpu.set_feature(Target::HVX, false);
669 one_gpu.set_feature(Target::OpenGLCompute, false);
670 one_gpu.set_feature(Target::D3D12Compute, false);
671 string module_name;
672 switch (runtime_kind) {
673 case OpenCLDebug:
674 one_gpu.set_feature(Target::Debug);
675 one_gpu.set_feature(Target::OpenCL);
676 module_name = "debug_opencl";
677 break;
678 case OpenCL:
679 one_gpu.set_feature(Target::OpenCL);
680 module_name += "opencl";
681 break;
682 case MetalDebug:
683 one_gpu.set_feature(Target::Debug);
684 one_gpu.set_feature(Target::Metal);
685 load_metal();
686 module_name = "debug_metal";
687 break;
688 case Metal:
689 one_gpu.set_feature(Target::Metal);
690 module_name += "metal";
691 load_metal();
692 break;
693 case CUDADebug:
694 one_gpu.set_feature(Target::Debug);
695 one_gpu.set_feature(Target::CUDA);
696 module_name = "debug_cuda";
697 break;
698 case CUDA:
699 one_gpu.set_feature(Target::CUDA);
700 module_name += "cuda";
701 break;
702 case OpenGLComputeDebug:
703 one_gpu.set_feature(Target::Debug);
704 one_gpu.set_feature(Target::OpenGLCompute);
705 module_name = "debug_openglcompute";
706 load_opengl(one_gpu.has_feature(Target::EGL));
707 break;
708 case OpenGLCompute:
709 one_gpu.set_feature(Target::OpenGLCompute);
710 module_name += "openglcompute";
711 load_opengl(one_gpu.has_feature(Target::EGL));
712 break;
713 case HexagonDebug:
714 one_gpu.set_feature(Target::Debug);
715 one_gpu.set_feature(Target::HVX);
716 module_name = "debug_hexagon";
717 break;
718 case Hexagon:
719 one_gpu.set_feature(Target::HVX);
720 module_name += "hexagon";
721 break;
722 case D3D12ComputeDebug:
723 one_gpu.set_feature(Target::Debug);
724 one_gpu.set_feature(Target::D3D12Compute);
725 module_name = "debug_d3d12compute";
726 break;
727 case D3D12Compute:
728 one_gpu.set_feature(Target::D3D12Compute);
729 module_name += "d3d12compute";
730#if !defined(_WIN32)
731 internal_error << "JIT support for Direct3D 12 is only implemented on Windows 10 and above.\n";
732#endif
733 break;
734 default:
735 module_name = "shared runtime";
736 break;
737 }
738
739 // This function is protected by a mutex so this is thread safe.
740 auto module =
741 get_initial_module_for_target(one_gpu,
742 &runtime.jit_module->context,
743 true,
744 runtime_kind != MainShared);
745 if (for_module) {
746 clone_target_options(*for_module, *module);
747 }
748 module->setModuleIdentifier(module_name);
749
750 std::set<std::string> halide_exports_unique;
751
752 // Enumerate the functions.
753 for (auto &f : *module) {
754 // LLVM_Runtime_Linker has marked everything that should be exported as weak
755 if (f.hasWeakLinkage()) {
756 halide_exports_unique.insert(get_llvm_function_name(f));
757 }
758 }
759
760 std::vector<std::string> halide_exports(halide_exports_unique.begin(), halide_exports_unique.end());
761
762 runtime.compile_module(std::move(module), "", target, deps, halide_exports);
763
764 if (runtime_kind == MainShared) {
765 runtime_internal_handlers.custom_print =
766 hook_function(runtime.exports(), "halide_set_custom_print", print_handler);
767
768 runtime_internal_handlers.custom_malloc =
769 hook_function(runtime.exports(), "halide_set_custom_malloc", malloc_handler);
770
771 runtime_internal_handlers.custom_free =
772 hook_function(runtime.exports(), "halide_set_custom_free", free_handler);
773
774 runtime_internal_handlers.custom_do_task =
775 hook_function(runtime.exports(), "halide_set_custom_do_task", do_task_handler);
776
777 runtime_internal_handlers.custom_do_par_for =
778 hook_function(runtime.exports(), "halide_set_custom_do_par_for", do_par_for_handler);
779
780 runtime_internal_handlers.custom_error =
781 hook_function(runtime.exports(), "halide_set_error_handler", error_handler_handler);
782
783 runtime_internal_handlers.custom_trace =
784 hook_function(runtime.exports(), "halide_set_custom_trace", trace_handler);
785
786 runtime_internal_handlers.custom_get_symbol =
787 hook_function(shared_runtimes(MainShared).exports(), "halide_set_custom_get_symbol", get_symbol_handler);
788
789 runtime_internal_handlers.custom_load_library =
790 hook_function(shared_runtimes(MainShared).exports(), "halide_set_custom_load_library", load_library_handler);
791
792 runtime_internal_handlers.custom_get_library_symbol =
793 hook_function(shared_runtimes(MainShared).exports(), "halide_set_custom_get_library_symbol", get_library_symbol_handler);
794
795 active_handlers = runtime_internal_handlers;
796 merge_handlers(active_handlers, default_handlers);
797
798 if (default_cache_size != 0) {
799 runtime.memoization_cache_set_size(default_cache_size);
800 }
801
802 runtime.jit_module->name = "MainShared";
803 } else {
804 runtime.jit_module->name = "GPU";
805 }
806
807 uint64_t arg_addr =
808 runtime.jit_module->execution_engine->getGlobalValueAddress("halide_jit_module_argument");
809
810 internal_assert(arg_addr != 0);
811 *((void **)arg_addr) = runtime.jit_module.get();
812
813 uint64_t fun_addr = runtime.jit_module->execution_engine->getGlobalValueAddress("halide_jit_module_adjust_ref_count");
814 internal_assert(fun_addr != 0);
815 *(void (**)(void *arg, int32_t count))fun_addr = &adjust_module_ref_count;
816 }
817 return runtime;
818}

822/* Shared runtimes are stored as global state. The set needed is
823 * determined from the target and the retrieved. If one does not exist
824 * yet, it is made on the fly from the compiled in bitcode of the
825 * runtime modules. As with all JITModules, the shared runtime is ref
826 * counted, but a global keeps one ref alive until shutdown or when
827 * JITSharedRuntime::release_all is called. If
828 * JITSharedRuntime::release_all is called, the global state is reset
829 * and any newly compiled Funcs will get a new runtime. */
830std::vector<JITModule> JITSharedRuntime::get(llvm::Module *for_module, const Target &target, bool create) {
831 std::lock_guard<std::mutex> lock(shared_runtimes_mutex);
832
833 std::vector<JITModule> result;
834
835 JITModule m = make_module(for_module, target, MainShared, result, create);
836 if (m.compiled()) {
837 result.push_back(m);
838 }
839
840 // Add all requested GPU modules, each only depending on the main shared runtime.
841 std::vector<JITModule> gpu_modules;
842 if (target.has_feature(Target::OpenCL)) {
843 auto kind = target.has_feature(Target::Debug) ? OpenCLDebug : OpenCL;
844 JITModule m = make_module(for_module, target, kind, result, create);
845 if (m.compiled()) {
846 result.push_back(m);
847 }
848 }
849 if (target.has_feature(Target::Metal)) {
850 auto kind = target.has_feature(Target::Debug) ? MetalDebug : Metal;
851 JITModule m = make_module(for_module, target, kind, result, create);
852 if (m.compiled()) {
853 result.push_back(m);
854 }
855 }
856 if (target.has_feature(Target::CUDA)) {
857 auto kind = target.has_feature(Target::Debug) ? CUDADebug : CUDA;
858 JITModule m = make_module(for_module, target, kind, result, create);
859 if (m.compiled()) {
860 result.push_back(m);
861 }
862 }
863 if (target.has_feature(Target::OpenGLCompute)) {
864 auto kind = target.has_feature(Target::Debug) ? OpenGLComputeDebug : OpenGLCompute;
865 JITModule m = make_module(for_module, target, kind, result, create);
866 if (m.compiled()) {
867 result.push_back(m);
868 }
869 }
870 if (target.has_feature(Target::HVX)) {
871 auto kind = target.has_feature(Target::Debug) ? HexagonDebug : Hexagon;
872 JITModule m = make_module(for_module, target, kind, result, create);
873 if (m.compiled()) {
874 result.push_back(m);
875 }
876 }
877 if (target.has_feature(Target::D3D12Compute)) {
878 auto kind = target.has_feature(Target::Debug) ? D3D12ComputeDebug : D3D12Compute;
879 JITModule m = make_module(for_module, target, kind, result, create);
880 if (m.compiled()) {
881 result.push_back(m);
882 }
883 }
884
885 return result;
886}

900void JITSharedRuntime::release_all() {
901 std::lock_guard<std::mutex> lock(shared_runtimes_mutex);
902

904 shared_runtimes((RuntimeKind)(i - 1)) = JITModule();
