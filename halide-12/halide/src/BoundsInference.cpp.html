64/** Compute the bounds of the value of some variable defined by an
65 * inner let stmt or for loop. E.g. for the stmt:
66 *
67 *
68 * for x from 0 to 10:
69 * let y = x + 2;
70 *
71 * bounds_of_inner_var(y) would return 2 to 12, and
72 * bounds_of_inner_var(x) would return 0 to 10.
73 */
74class BoundsOfInnerVar : public IRVisitor {
75public:
76 Interval result;
77 BoundsOfInnerVar(const string &v)
78 : var(v) {
79 }
80
81private:
82 string var;
83 bool found = false;
84
85 using IRVisitor::visit;
86
87 void visit(const LetStmt *op) override {
88 if (op->name == var) {
89 result = Interval::single_point(op->value);
90 found = true;
91 } else if (!found) {
92 op->body.accept(this);
93 if (found) {
94 if (expr_uses_var(result.min, op->name)) {
95 result.min = Let::make(op->name, op->value, result.min);
96 }
97 if (expr_uses_var(result.max, op->name)) {
98 result.max = Let::make(op->name, op->value, result.max);
99 }
100 }
101 }
102 }
103
104 void visit(const Block *op) override {
105 // We're most likely to find our var at the end of a
106 // block. The start of the block could be unrelated producers.
107 op->rest.accept(this);
108 if (!found) {
109 op->first.accept(this);
110 }
111 }
112
113 void visit(const For *op) override {
114 // At this stage of lowering, loop_min and loop_max
115 // conveniently exist in scope.
116 Interval in(Variable::make(Int(32), op->name + ".loop_min"),
117 Variable::make(Int(32), op->name + ".loop_max"));
118
119 if (op->name == var) {
120 result = in;
121 found = true;
122 } else if (!found) {
123 op->body.accept(this);
124 if (found) {
125 Scope<Interval> scope;
126 scope.push(op->name, in);
127 if (expr_uses_var(result.min, op->name)) {
128 result.min = bounds_of_expr_in_scope(result.min, scope).min;
129 }
130 if (expr_uses_var(result.max, op->name)) {
131 result.max = bounds_of_expr_in_scope(result.max, scope).max;
132 }
133 }
134 }
135 }
136};

144size_t find_fused_group_index(const Function &producing_func,
145 const vector<vector<Function>> &fused_groups) {
146 const auto &iter = std::find_if(fused_groups.begin(), fused_groups.end(),
147 [&producing_func](const vector<Function> &group) {
148 return std::any_of(group.begin(), group.end(),
149 [&producing_func](const Function &f) {
150 return (f.name() == producing_func.name());
151 });
152 });
153 internal_assert(iter != fused_groups.end());
154 return iter - fused_groups.begin();
155}

157// Determine if the current producing stage is fused with other
158// stage (i.e. the consumer stage) at dimension 'var'.
159bool is_fused_with_others(const vector<vector<Function>> &fused_groups,
160 const vector<set<FusedPair>> &fused_pairs_in_groups,
161 const Function &producing_func, int producing_stage_index,
162 const string &consumer_name, int consumer_stage,
163 string var) {
164 if (producing_func.has_extern_definition()) {
165 return false;
166 }
167
168 // Find the fused group this producing stage belongs to.
169 size_t index = find_fused_group_index(producing_func, fused_groups);
170
171 const vector<Dim> &dims = (producing_stage_index == 0) ? producing_func.definition().schedule().dims() : producing_func.update(producing_stage_index - 1).schedule().dims();
172
173 size_t var_index;
174 {
175 const auto &iter = std::find_if(dims.begin(), dims.end(),
176 [&var](const Dim &d) { return var_name_match(d.var, var); });
177 if (iter == dims.end()) {
178 return false;
179 }
180 var_index = iter - dims.begin();
181 }
182
183 // Iterate over the fused pair list to check if the producer stage
184 // is fused with the consumer stage at 'var'
185 for (const auto &pair : fused_pairs_in_groups[index]) {
186 if (((pair.func_1 == consumer_name) && ((int)pair.stage_1 == consumer_stage)) ||
187 ((pair.func_2 == consumer_name) && ((int)pair.stage_2 == consumer_stage))) {
188 const auto &iter = std::find_if(dims.begin(), dims.end(),
189 [&pair](const Dim &d) { return var_name_match(d.var, pair.var_name); });
190 if (iter == dims.end()) {
191 continue;
192 }
193 size_t idx = iter - dims.begin();
194 if (var_index >= idx) {
195 return true;
196 }
197 }
198 }
199 return false;
200}

220 : cond(c), value(v) {

224 struct Stage {
225 Function func;
226 size_t stage; // 0 is the pure definition, 1 is the first update
227 string name;
228 vector<int> consumers;
229 map<pair<string, int>, Box> bounds;
230 vector<CondValue> exprs;
231 set<ReductionVariable, ReductionVariable::Compare> rvars;
232 string stage_prefix;
233 size_t fused_group_index;

235 // Computed expressions on the left and right-hand sides.
236 // Note that a function definition might have different LHS or reduction domain
237 // (if it's an update def) or RHS per specialization. All specializations
238 // of an init definition should have the same LHS.
239 // This also pushes all the reduction domains it encounters into the 'rvars'
240 // set for later use.
241 vector<vector<CondValue>> compute_exprs_helper(const Definition &def, bool is_update) {
242 vector<vector<CondValue>> result(2); // <args, values>
243
244 if (!def.defined()) {
245 return result;
246 }
247
248 // Default case (no specialization)
249 vector<Expr> predicates = def.split_predicate();
250 for (const ReductionVariable &rv : def.schedule().rvars()) {
251 rvars.insert(rv);
252 }
253
254 vector<vector<Expr>> vecs(2);
255 if (is_update) {
256 vecs[0] = def.args();
257 }
258 vecs[1] = def.values();
259
260 for (size_t i = 0; i < result.size(); ++i) {
261 for (const Expr &val : vecs[i]) {
262 if (!predicates.empty()) {
263 Expr cond_val = Call::make(val.type(),
264 Internal::Call::if_then_else,
265 {likely(predicates[0]), val, make_zero(val.type())},
266 Internal::Call::PureIntrinsic);
267 for (size_t i = 1; i < predicates.size(); ++i) {
268 cond_val = Call::make(cond_val.type(),
269 Internal::Call::if_then_else,
270 {likely(predicates[i]), cond_val, make_zero(cond_val.type())},
271 Internal::Call::PureIntrinsic);
272 }
273 result[i].push_back(CondValue(const_true(), cond_val));
274 } else {
275 result[i].push_back(CondValue(const_true(), val));
276 }
277 }
278 }
279
280 const vector<Specialization> &specializations = def.specializations();
281 for (size_t i = specializations.size(); i > 0; i--) {
282 Expr s_cond = specializations[i - 1].condition;
283 const Definition &s_def = specializations[i - 1].definition;
284
285 // Else case (i.e. specialization condition is false)
286 for (auto &vec : result) {
287 for (CondValue &cval : vec) {
288 cval.cond = simplify(!s_cond && cval.cond);
289 }
290 }
291
292 // Then case (i.e. specialization condition is true)
293 vector<vector<CondValue>> s_result = compute_exprs_helper(s_def, is_update);
294 for (auto &vec : s_result) {
295 for (CondValue &cval : vec) {
296 cval.cond = simplify(s_cond && cval.cond);
297 }
298 }
299 for (size_t i = 0; i < result.size(); i++) {
300 result[i].insert(result[i].end(), s_result[i].begin(), s_result[i].end());
301 }
302 }
303
304 // Optimization: If the args/values across specializations including
305 // the default case, are the same, we can combine those args/values
306 // into one arg/value with a const_true() condition for the purpose
307 // of bounds inference.
308 for (auto &vec : result) {
309 if (vec.size() > 1) {
310 bool all_equal = true;
311 Expr val = vec[0].value;
312 for (size_t i = 1; i < vec.size(); ++i) {
313 if (!equal(val, vec[i].value)) {
314 all_equal = false;
315 break;
316 }
317 }
318 if (all_equal) {
319 debug(4) << "compute_exprs: all values (size: " << vec.size() << ") "
320 << "(" << val << ") are equal, combine them together\n";
321 internal_assert(val.defined());
322 vec.clear();
323 vec.emplace_back(const_true(), val);
324 }
325 }
326 }
327 return result;
328 }

330 // Computed expressions on the left and right-hand sides. This also
331 // pushes all reduction domains it encounters into the 'rvars' set
332 // for later use.
333 void compute_exprs() {
334 // We need to clear 'exprs' and 'rvars' first, in case compute_exprs()
335 // is called multiple times.
336 exprs.clear();
337 rvars.clear();
338
339 bool is_update = (stage != 0);
340 vector<vector<CondValue>> result;
341 if (!is_update) {
342 result = compute_exprs_helper(func.definition(), is_update);
343 } else {
344 const Definition &def = func.update(stage - 1);
345 result = compute_exprs_helper(def, is_update);
346 }
347 internal_assert(result.size() == 2);
348 exprs = result[0];
349
350 if (func.extern_definition_proxy_expr().defined()) {
351 exprs.emplace_back(const_true(), func.extern_definition_proxy_expr());
352 }
353
354 exprs.insert(exprs.end(), result[1].begin(), result[1].end());
355
356 // For the purposes of computation bounds inference, we
357 // don't care what sites are loaded, just what sites need
358 // to have the correct value in them. So remap all selects
359 // to if_then_elses to get tighter bounds.
360 class SelectToIfThenElse : public IRMutator {
361 using IRMutator::visit;
362 Expr visit(const Select *op) override {
363 if (is_pure(op->condition)) {
364 return Call::make(op->type, Call::if_then_else,
365 {mutate(op->condition),
366 mutate(op->true_value),
367 mutate(op->false_value)},
368 Call::PureIntrinsic);
369 } else {
370 return IRMutator::visit(op);
371 }
372 }
373 } select_to_if_then_else;
374
375 for (auto &e : exprs) {
376 e.value = select_to_if_then_else.mutate(e.value);
377 }
378 }

380 // Check if the dimension at index 'dim_idx' is always pure (i.e. equal to 'dim')
381 // in the definition (including in its specializations)
382 bool is_dim_always_pure(const Definition &def, const string &dim, int dim_idx) {
383 const Variable *var = def.args()[dim_idx].as<Variable>();
384 if ((!var) || (var->name != dim)) {
385 return false;
386 }
387
388 for (const Specialization &s : def.specializations()) {
389 bool pure = is_dim_always_pure(s.definition, dim, dim_idx);
390 if (!pure) {
391 return false;
392 }
393 }
394 return true;
395 }

397 // Wrap a statement in let stmts defining the box
398 Stmt define_bounds(Stmt s,
399 const Function &producing_func,
400 const string &producing_stage_index,
401 int producing_stage_index_index,
402 const string &loop_level,
403 const vector<vector<Function>> &fused_groups,
404 const vector<set<FusedPair>> &fused_pairs_in_groups,
405 const set<string> &in_pipeline,
406 const set<string> &inner_productions,
407 const set<string> &has_extern_consumer,
408 const Target &target) {
409
410 // Merge all the relevant boxes.
411 Box b;
412
413 const vector<string> func_args = func.args();
414
415 size_t last_dot = loop_level.rfind('.');
416 string var = loop_level.substr(last_dot + 1);
417
418 for (const pair<const pair<string, int>, Box> &i : bounds) {
419 string func_name = i.first.first;
420 int func_stage_index = i.first.second;
421 string stage_name = func_name + ".s" + std::to_string(func_stage_index);
422 if (stage_name == producing_stage_index ||
423 inner_productions.count(func_name) ||
424 is_fused_with_others(fused_groups, fused_pairs_in_groups,
425 producing_func, producing_stage_index_index,
426 func_name, func_stage_index, var)) {
427 merge_boxes(b, i.second);
428 }
429 }
430
431 internal_assert(b.empty() || b.size() == func_args.size());
432
433 if (!b.empty()) {
434 // Optimization: If a dimension is pure in every update
435 // step of a func, then there exists a single bound for
436 // that dimension, instead of one bound per stage. Let's
437 // figure out what those dimensions are, and just have all
438 // stages but the last use the bounds for the last stage.
439 vector<bool> always_pure_dims(func_args.size(), true);
440 for (const Definition &def : func.updates()) {
441 for (size_t j = 0; j < always_pure_dims.size(); j++) {
442 bool pure = is_dim_always_pure(def, func_args[j], j);
443 if (!pure) {
444 always_pure_dims[j] = false;
445 }
446 }
447 }
448
449 if (stage < func.updates().size()) {
450 size_t stages = func.updates().size();
451 string last_stage = func.name() + ".s" + std::to_string(stages) + ".";
452 for (size_t i = 0; i < always_pure_dims.size(); i++) {
453 if (always_pure_dims[i]) {
454 const string &dim = func_args[i];
455 Expr min = Variable::make(Int(32), last_stage + dim + ".min");
456 Expr max = Variable::make(Int(32), last_stage + dim + ".max");
457 b[i] = Interval(min, max);
458 }
459 }
460 }
461 }
462
463 if (func.has_extern_definition() &&
464 !func.extern_definition_proxy_expr().defined()) {
465 // After we define our bounds required, we need to
466 // figure out what we're actually going to compute,
467 // and what inputs we need. To do this we:
468
469 // 1) Grab a handle on the bounds query results from one level up
470
471 // 2) Run the bounds query to let it round up the output size.
472
473 // 3) Shift the requested output box back inside of the
474 // bounds query result from one loop level up (in case
475 // it was rounded up)
476
477 // 4) then run the bounds query again to get the input
478 // sizes.
479
480 // Because we're wrapping a stmt, this happens in reverse order.
481
482 // 4)
483 s = do_bounds_query(s, in_pipeline, target);
484
485 if (!in_pipeline.empty()) {
486 // 3)
487 string outer_query_name = func.name() + ".outer_bounds_query";
488 Expr outer_query = Variable::make(type_of<struct halide_buffer_t *>(), outer_query_name);
489 string inner_query_name = func.name() + ".o0.bounds_query";
490 Expr inner_query = Variable::make(type_of<struct halide_buffer_t *>(), inner_query_name);
491 for (int i = 0; i < func.dimensions(); i++) {
492 Expr outer_min = Call::make(Int(32), Call::buffer_get_min,
493 {outer_query, i}, Call::Extern);
494 Expr outer_max = Call::make(Int(32), Call::buffer_get_max,
495 {outer_query, i}, Call::Extern);
496
497 Expr inner_min = Call::make(Int(32), Call::buffer_get_min,
498 {inner_query, i}, Call::Extern);
499 Expr inner_max = Call::make(Int(32), Call::buffer_get_max,
500 {inner_query, i}, Call::Extern);
501
502 // Push 'inner' inside of 'outer'
503 Expr shift = Min::make(0, outer_max - inner_max);
504 Expr new_min = inner_min + shift;
505 Expr new_max = inner_max + shift;
506
507 // Modify the region to be computed accordingly
508 s = LetStmt::make(func.name() + ".s0." + func_args[i] + ".max", new_max, s);
509 s = LetStmt::make(func.name() + ".s0." + func_args[i] + ".min", new_min, s);
510 }
511
512 // 2)
513 s = do_bounds_query(s, in_pipeline, target);
514
515 // 1)
516 s = LetStmt::make(func.name() + ".outer_bounds_query",
517 Variable::make(type_of<struct halide_buffer_t *>(), func.name() + ".o0.bounds_query"), s);
518 } else {
519 // If we're at the outermost loop, there is no
520 // bounds query result from one level up, but we
521 // still need to modify the region to be computed
522 // based on the bounds query result and then do
523 // another bounds query to ask for the required
524 // input size given that.
525
526 // 2)
527 string inner_query_name = func.name() + ".o0.bounds_query";
528 Expr inner_query = Variable::make(type_of<struct halide_buffer_t *>(), inner_query_name);
529 for (int i = 0; i < func.dimensions(); i++) {
530 Expr new_min = Call::make(Int(32), Call::buffer_get_min,
531 {inner_query, i}, Call::Extern);
532 Expr new_max = Call::make(Int(32), Call::buffer_get_max,
533 {inner_query, i}, Call::Extern);
534
535 s = LetStmt::make(func.name() + ".s0." + func_args[i] + ".max", new_max, s);
536 s = LetStmt::make(func.name() + ".s0." + func_args[i] + ".min", new_min, s);
537 }
538
539 s = do_bounds_query(s, in_pipeline, target);
540 }
541 }
542
543 if (in_pipeline.count(name) == 0) {
544 // Inject any explicit bounds
545 string prefix = name + ".s" + std::to_string(stage) + ".";
546
547 LoopLevel compute_at = func.schedule().compute_level();
548 LoopLevel store_at = func.schedule().store_level();
549
550 for (size_t i = 0; i < func.schedule().bounds().size(); i++) {
551 Bound bound = func.schedule().bounds()[i];
552 string min_var = prefix + bound.var + ".min";
553 string max_var = prefix + bound.var + ".max";
554 Expr min_required = Variable::make(Int(32), min_var);
555 Expr max_required = Variable::make(Int(32), max_var);
556
557 if (bound.extent.defined()) {
558 // If the Func is compute_at some inner loop, and
559 // only extent is bounded, then the min could
560 // actually move around, which makes the extent
561 // bound not actually useful for determining the
562 // max required from the point of view of
563 // producers.
564 if (bound.min.defined() ||
565 compute_at.is_root() ||
566 (compute_at.match(loop_level) &&
567 store_at.match(loop_level))) {
568 if (!bound.min.defined()) {
569 bound.min = min_required;
570 }
571 s = LetStmt::make(min_var, bound.min, s);
572 s = LetStmt::make(max_var, bound.min + bound.extent - 1, s);
573 }
574
575 // Save the unbounded values to use in bounds-checking assertions
576 s = LetStmt::make(min_var + "_unbounded", min_required, s);
577 s = LetStmt::make(max_var + "_unbounded", max_required, s);
578 }
579
580 if (bound.modulus.defined()) {
581 if (bound.remainder.defined()) {
582 min_required -= bound.remainder;
583 min_required = (min_required / bound.modulus) * bound.modulus;
584 min_required += bound.remainder;
585 Expr max_plus_one = max_required + 1;
586 max_plus_one -= bound.remainder;
587 max_plus_one = ((max_plus_one + bound.modulus - 1) / bound.modulus) * bound.modulus;
588 max_plus_one += bound.remainder;
589 max_required = max_plus_one - 1;
590 } else {
591 Expr extent = (max_required - min_required) + 1;
592 extent = simplify(((extent + bound.modulus - 1) / bound.modulus) * bound.modulus);
593 max_required = simplify(min_required + extent - 1);
594 }
595 s = LetStmt::make(min_var, min_required, s);
596 s = LetStmt::make(max_var, max_required, s);
597 }
598 }
599 }
600
601 for (size_t d = 0; d < b.size(); d++) {
602 string arg = name + ".s" + std::to_string(stage) + "." + func_args[d];
603
604 const bool clamp_to_outer_bounds =
605 !in_pipeline.empty() && has_extern_consumer.count(name);
606 if (clamp_to_outer_bounds) {
607 // Allocation bounds inference is going to have a
608 // bad time lifting the results of the bounds
609 // queries outwards. Help it out by insisting that
610 // the bounds are clamped to lie within the bounds
611 // one loop level up.
612 Expr outer_min = Variable::make(Int(32), arg + ".outer_min");
613 Expr outer_max = Variable::make(Int(32), arg + ".outer_max");
614 b[d].min = clamp(b[d].min, outer_min, outer_max);
615 b[d].max = clamp(b[d].max, outer_min, outer_max);
616 }
617
618 if (b[d].is_single_point()) {
619 s = LetStmt::make(arg + ".min", Variable::make(Int(32), arg + ".max"), s);
620 } else {
621 s = LetStmt::make(arg + ".min", b[d].min, s);
622 }
623 s = LetStmt::make(arg + ".max", b[d].max, s);
624
625 if (clamp_to_outer_bounds) {
626 s = LetStmt::make(arg + ".outer_min", Variable::make(Int(32), arg + ".min"), s);
627 s = LetStmt::make(arg + ".outer_max", Variable::make(Int(32), arg + ".max"), s);
628 }
629 }
630
631 if (stage > 0) {
632 for (const ReductionVariable &rvar : rvars) {
633 string arg = name + ".s" + std::to_string(stage) + "." + rvar.var;
634 s = LetStmt::make(arg + ".min", rvar.min, s);
635 s = LetStmt::make(arg + ".max", rvar.extent + rvar.min - 1, s);
636 }
637 }
638
639 return s;
640 }

642 Stmt do_bounds_query(Stmt s, const set<string> &in_pipeline, const Target &target) {
643
644 const string &extern_name = func.extern_function_name();
645 const vector<ExternFuncArgument> &args = func.extern_arguments();
646
647 vector<Expr> bounds_inference_args;
648
649 vector<pair<string, Expr>> lets;
650
651 // Iterate through all of the input args to the extern
652 // function building a suitable argument list for the
653 // extern function call. We need a query halide_buffer_t per
654 // producer and a query halide_buffer_t for the output
655
656 Expr null_handle = make_zero(Handle());
657
658 vector<pair<Expr, int>> buffers_to_annotate;
659 for (size_t j = 0; j < args.size(); j++) {
660 if (args[j].is_expr()) {
661 bounds_inference_args.push_back(args[j].expr);
662 } else if (args[j].is_func()) {
663 Function input(args[j].func);
664 for (int k = 0; k < input.outputs(); k++) {
665 string name = input.name() + ".o" + std::to_string(k) + ".bounds_query." + func.name();
666
667 BufferBuilder builder;
668 builder.type = input.output_types()[k];
669 builder.dimensions = input.dimensions();
670 Expr buf = builder.build();
671
672 lets.emplace_back(name, buf);
673 bounds_inference_args.push_back(Variable::make(type_of<struct halide_buffer_t *>(), name));
674 buffers_to_annotate.emplace_back(bounds_inference_args.back(), input.dimensions());
675 }
676 } else if (args[j].is_image_param() || args[j].is_buffer()) {
677 Parameter p = args[j].image_param;
678 Buffer<> b = args[j].buffer;
679 string name = args[j].is_image_param() ? p.name() : b.name();
680 int dims = args[j].is_image_param() ? p.dimensions() : b.dimensions();
681
682 Expr in_buf = Variable::make(type_of<struct halide_buffer_t *>(), name + ".buffer");
683
684 // Copy the input buffer into a query buffer to mutate.
685 string query_name = name + ".bounds_query." + func.name();
686
687 Expr alloca_size = Call::make(Int(32), Call::size_of_halide_buffer_t, {}, Call::Intrinsic);
688 Expr query_buf = Call::make(type_of<struct halide_buffer_t *>(), Call::alloca,
689 {alloca_size}, Call::Intrinsic);
690 Expr query_shape = Call::make(type_of<struct halide_dimension_t *>(), Call::alloca,
691 {(int)(sizeof(halide_dimension_t) * dims)}, Call::Intrinsic);
692 query_buf = Call::make(type_of<struct halide_buffer_t *>(), Call::buffer_init_from_buffer,
693 {query_buf, query_shape, in_buf}, Call::Extern);
694
695 lets.emplace_back(query_name, query_buf);
696 Expr buf = Variable::make(type_of<struct halide_buffer_t *>(), query_name, b, p, ReductionDomain());
697 bounds_inference_args.push_back(buf);
698 // Although we expect ImageParams to be properly initialized and sanitized by the caller,
699 // we create a copy with copy_memory (not msan-aware), so we need to annotate it as initialized.
700 buffers_to_annotate.emplace_back(bounds_inference_args.back(), dims);
701 } else {
702 internal_error << "Bad ExternFuncArgument type";
703 }
704 }
705
706 // Make the buffer_ts representing the output. They all
707 // use the same size, but have differing types.
708 for (int j = 0; j < func.outputs(); j++) {
709 BufferBuilder builder;
710 builder.type = func.output_types()[j];
711 builder.dimensions = func.dimensions();
712 for (const string &arg : func.args()) {
713 string prefix = func.name() + ".s" + std::to_string(stage) + "." + arg;
714 Expr min = Variable::make(Int(32), prefix + ".min");
715 Expr max = Variable::make(Int(32), prefix + ".max");
716 builder.mins.push_back(min);
717 builder.extents.push_back(max + 1 - min);
718 builder.strides.emplace_back(0);
719 }
720 Expr output_buffer_t = builder.build();
721
722 string buf_name = func.name() + ".o" + std::to_string(j) + ".bounds_query";
723 bounds_inference_args.push_back(Variable::make(type_of<struct halide_buffer_t *>(), buf_name));
724 // Since this is a temporary, internal-only buffer used for bounds inference,
725 // we need to mark it
726 buffers_to_annotate.emplace_back(bounds_inference_args.back(), func.dimensions());
727 lets.emplace_back(buf_name, output_buffer_t);
728 }
729
730 Stmt annotate;
731 if (target.has_feature(Target::MSAN)) {
732 // Mark the buffers as initialized before calling out.
733 for (const auto &p : buffers_to_annotate) {
734 Expr buffer = p.first;
735 int dimensions = p.second;
736 // Return type is really 'void', but no way to represent that in our IR.
737 // Precedent (from halide_print, etc) is to use Int(32) and ignore the result.
738 Expr sizeof_buffer_t = cast<uint64_t>(
739 Call::make(Int(32), Call::size_of_halide_buffer_t, {}, Call::Intrinsic));
740 Stmt mark_buffer =
741 Evaluate::make(Call::make(Int(32), "halide_msan_annotate_memory_is_initialized",
742 {buffer, sizeof_buffer_t}, Call::Extern));
743 Expr shape = Call::make(type_of<halide_dimension_t *>(), Call::buffer_get_shape, {buffer},
744 Call::Extern);
745 Expr shape_size = Expr((uint64_t)(sizeof(halide_dimension_t) * dimensions));
746 Stmt mark_shape =
747 Evaluate::make(Call::make(Int(32), "halide_msan_annotate_memory_is_initialized",
748 {shape, shape_size}, Call::Extern));
749
750 mark_buffer = Block::make(mark_buffer, mark_shape);
751 if (annotate.defined()) {
752 annotate = Block::make(annotate, mark_buffer);
753 } else {
754 annotate = mark_buffer;
755 }
756 }
757 }
758
759 // Make the extern call
760 Expr e = func.make_call_to_extern_definition(bounds_inference_args, target);
761
762 // Check if it succeeded
763 string result_name = unique_name('t');
764 Expr result = Variable::make(Int(32), result_name);
765 Expr error = Call::make(Int(32), "halide_error_bounds_inference_call_failed",
766 {extern_name, result}, Call::Extern);
767 Stmt check = AssertStmt::make(EQ::make(result, 0), error);
768
769 check = LetStmt::make(result_name, e, check);
770
771 if (annotate.defined()) {
772 check = Block::make(annotate, check);
773 }
774
775 // Now inner code is free to extract the fields from the halide_buffer_t
776 s = Block::make(check, s);
777
778 // Wrap in let stmts defining the args
779 for (size_t i = 0; i < lets.size(); i++) {
780 s = LetStmt::make(lets[i].first, lets[i].second, s);
781 }
782
783 return s;
784 }

811
812 BoundsInference(const vector<Function> &f,
813 const vector<vector<Function>> &fg,
814 const vector<set<FusedPair>> &fp,
815 const vector<Function> &outputs,
816 const FuncValueBounds &fb,
817 const Target &target)
818 : funcs(f), fused_groups(fg), fused_pairs_in_groups(fp), func_bounds(fb), target(target) {
819 internal_assert(!f.empty());
820
821 // Compute the intrinsic relationships between the stages of
822 // the functions.
823
824 // Figure out which functions will be inlined away
825 vector<bool> inlined(f.size());
826 for (size_t i = 0; i < inlined.size(); i++) {
827 if (i < f.size() - 1 &&
828 f[i].schedule().compute_level().is_inlined() &&
829 f[i].can_be_inlined()) {
830 inlined[i] = true;
831 } else {
832 inlined[i] = false;
833 }
834 }
835
836 // First lay out all the stages in their realization order.
837 // The functions are already in topologically sorted order, so
838 // this is straight-forward.
839 for (size_t i = 0; i < f.size(); i++) {
840
841 if (inlined[i]) {
842 continue;
843 }
844
845 Stage s;
846 s.func = f[i];
847 s.stage = 0;
848 s.name = s.func.name();
849 s.fused_group_index = find_fused_group_index(s.func, fused_groups);
850 s.compute_exprs();
851 s.stage_prefix = s.name + ".s0.";
852 stages.push_back(s);
853
854 for (size_t j = 0; j < f[i].updates().size(); j++) {
855 s.stage = (int)(j + 1);
856 s.stage_prefix = s.name + ".s" + std::to_string(s.stage) + ".";
857 s.compute_exprs();
858 stages.push_back(s);
859 }
860 }
861
862 // Do any pure inlining (TODO: This is currently slow)
863 for (size_t i = f.size(); i > 0; i--) {
864 Function func = f[i - 1];
865 if (inlined[i - 1]) {
866 for (size_t j = 0; j < stages.size(); j++) {
867 Stage &s = stages[j];
868 for (size_t k = 0; k < s.exprs.size(); k++) {
869 CondValue &cond_val = s.exprs[k];
870 internal_assert(cond_val.value.defined());
871 cond_val.value = inline_function(cond_val.value, func);
872 }
873 }
874 }
875 }
876
877 // Remove the inlined stages
878 vector<Stage> new_stages;
879 for (size_t i = 0; i < stages.size(); i++) {
880 if (!stages[i].func.schedule().compute_level().is_inlined() ||
881 !stages[i].func.can_be_inlined()) {
882 new_stages.push_back(stages[i]);
883 }
884 }
885 new_stages.swap(stages);
886
887 // Dump the stages post-inlining for debugging
888 /*
889 debug(0) << "Bounds inference stages after inlining: \n";
890 for (size_t i = 0; i < stages.size(); i++) {
891 debug(0) << " " << i << ") " << stages[i].name << "\n";
892 }
893 */
894
895 // Then compute relationships between them.
896 for (size_t i = 0; i < stages.size(); i++) {
897
898 Stage &consumer = stages[i];
899
900 // Set up symbols representing the bounds over which this
901 // stage will be computed.
902 Scope<Interval> scope;
903 consumer.populate_scope(scope);
904
905 // Compute all the boxes of the producers this consumer
906 // uses.
907 map<string, Box> boxes;
908 if (consumer.func.has_extern_definition() &&
909 !consumer.func.extern_definition_proxy_expr().defined()) {
910
911 const vector<ExternFuncArgument> &args = consumer.func.extern_arguments();
912 // Stage::define_bounds is going to compute a query
913 // halide_buffer_t per producer for bounds inference to
914 // use. We just need to extract those values.
915 for (size_t j = 0; j < args.size(); j++) {
916 if (args[j].is_func()) {
917 Function f(args[j].func);
918 has_extern_consumer.insert(f.name());
919 string stage_name = f.name() + ".s" + std::to_string(f.updates().size());
920 Box b(f.dimensions());
921 for (int d = 0; d < f.dimensions(); d++) {
922 string buf_name = f.name() + ".o0.bounds_query." + consumer.name;
923 Expr buf = Variable::make(type_of<struct halide_buffer_t *>(), buf_name);
924 Expr min = Call::make(Int(32), Call::buffer_get_min,
925 {buf, d}, Call::Extern);
926 Expr max = Call::make(Int(32), Call::buffer_get_max,
927 {buf, d}, Call::Extern);
928 b[d] = Interval(min, max);
929 }
930 merge_boxes(boxes[f.name()], b);
931 }
932 }
933 } else {
934 for (const auto &cval : consumer.exprs) {
935 map<string, Box> new_boxes;
936 new_boxes = boxes_required(cval.value, scope, func_bounds);
937 for (auto &i : new_boxes) {
938 // Add the condition on which this value is evaluated to the box before merging
939 Box &box = i.second;
940 box.used = cval.cond;
941 merge_boxes(boxes[i.first], box);
942 }
943 }
944 }
945
946 // Expand the bounds required of all the producers found
947 // (and we are checking until i, because stages are topologically sorted).
948 for (size_t j = 0; j < i; j++) {
949 Stage &producer = stages[j];
950 // A consumer depends on *all* stages of a producer, not just the last one.
951 const Box &b = boxes[producer.func.name()];
952
953 if (!b.empty()) {
954 // Check for unboundedness
955 for (size_t k = 0; k < b.size(); k++) {
956 if (!b[k].is_bounded()) {
957 std::ostringstream err;
958 if (consumer.stage == 0) {
959 err << "The pure definition ";
960 } else {
961 err << "Update definition number " << (consumer.stage - 1);
962 }
963 err << " of Function " << consumer.name
964 << " calls function " << producer.name
965 << " in an unbounded way in dimension " << k << "\n";
966 user_error << err.str();
967 }
968 }
969
970 // Dump out the region required of each stage for debugging.
971
972 /*
973 debug(0) << "Box required of " << producer.name
974 << " by " << consumer.name
975 << " stage " << consumer.stage << ":\n";
976 for (size_t k = 0; k < b.size(); k++) {
977 debug(0) << " " << b[k].min << " ... " << b[k].max << "\n";
978 }
979 debug(0) << "\n";
980 */
981
982 producer.bounds[{consumer.name, consumer.stage}] = b;
983 producer.consumers.push_back((int)i);
984 }
985 }
986 }
987
988 // The region required of the each output is expanded to include the size of the output buffer.
989 for (const Function &output : outputs) {
990 Box output_box;
991 string buffer_name = output.name();
992 if (output.outputs() > 1) {
993 // Use the output size of the first output buffer
994 buffer_name += ".0";
995 }
996 for (int d = 0; d < output.dimensions(); d++) {
997 Parameter buf = output.output_buffers()[0];
998 Expr min = Variable::make(Int(32), buffer_name + ".min." + std::to_string(d), buf);
999 Expr extent = Variable::make(Int(32), buffer_name + ".extent." + std::to_string(d), buf);
1000
1001 // Respect any output min and extent constraints
1002 Expr min_constraint = buf.min_constraint(d);
1003 Expr extent_constraint = buf.extent_constraint(d);
1004
1005 if (min_constraint.defined()) {
1006 min = min_constraint;
1007 }
1008 if (extent_constraint.defined()) {
1009 extent = extent_constraint;
1010 }
1011
1012 output_box.push_back(Interval(min, (min + extent) - 1));
1013 }
1014 for (size_t i = 0; i < stages.size(); i++) {
1015 Stage &s = stages[i];
1016 if (!s.func.same_as(output)) {
1017 continue;
1018 }
1019 s.bounds[{s.name, s.stage}] = output_box;
1020 }
1021 }
1022 }

1024 using IRMutator::visit;
1025
1026 Stmt visit(const For *op) override {
1027 // Don't recurse inside loops marked 'Extern', they will be
1028 // removed later.
1029 if (op->for_type == ForType::Extern) {
1030 return op;
1031 }
1032
1033 set<string> old_inner_productions;
1034 inner_productions.swap(old_inner_productions);
1035
1036 Stmt body = op->body;
1037
1038 // Walk inside of any let/if statements that don't depend on
1039 // bounds inference results so that we don't needlessly
1040 // complicate our bounds expressions.
1041 vector<pair<string, Expr>> wrappers;
1042 vector<ScopedBinding<>> bindings;
1043 while (true) {
1044 if (const LetStmt *let = body.as<LetStmt>()) {
1045 if (depends_on_bounds_inference(let->value)) {
1046 break;
1047 }
1048
1049 body = let->body;
1050 wrappers.emplace_back(let->name, let->value);
1051 bindings.emplace_back(let_vars_in_scope, let->name);
1052 } else if (const IfThenElse *if_then_else = body.as<IfThenElse>()) {
1053 if (depends_on_bounds_inference(if_then_else->condition) ||
1054 if_then_else->else_case.defined()) {
1055 break;
1056 }
1057
1058 body = if_then_else->then_case;
1059 wrappers.emplace_back(std::string(), if_then_else->condition);
1060 } else {
1061 break;
1062 }
1063 }
1064
1065 // If there are no pipelines at this loop level, we can skip
1066 // most of the work. Consider 'extern' for loops as pipelines
1067 // (we aren't recursing into these loops above).
1068 bool no_pipelines =
1069 body.as<For>() != nullptr &&
1070 body.as<For>()->for_type != ForType::Extern;
1071
1072 // Figure out which stage of which function we're producing
1073 int producing = -1;
1074 Function f;
1075 int stage_index = -1;
1076 string stage_name;
1077 for (size_t i = 0; i < stages.size(); i++) {
1078 if (starts_with(op->name, stages[i].stage_prefix)) {
1079 producing = i;
1080 f = stages[i].func;
1081 stage_index = (int)stages[i].stage;
1082 stage_name = stages[i].name + ".s" + std::to_string(stages[i].stage);
1083 break;
1084 }
1085 }
1086
1087 // Figure out how much of it we're producing
1088
1089 // Note: the case when functions are fused is a little bit tricky, so may need extra care:
1090 // when we're producing some of a Func A, at every loop belonging to A
1091 // you potentially need to define symbols for what box is being computed
1092 // of A (A.x.min, A.x.max ...), because that any other producer Func P nested
1093 // there is going to define its loop bounds in terms of these symbols, to ensure
1094 // it computes enough of itself to satisfy the consumer.
1095 // Now say we compute B with A, and say B consumes P, not A. Bounds inference
1096 // will see the shared loop, and think it belongs to A only. It will define A.x.min and
1097 // friends, but that's not very useful, because P's loops are in terms of B.x.min, B.x.max, etc.
1098 // So without a local definition of those symbols, P will use the one in the outer scope, and
1099 // compute way too much of itself. It'll still be correct, but it's massive over-compute.
1100 // The fix is to realize that in this loop belonging to A, we also potentially need to define
1101 // a box for B, because B belongs to the same fused group as A, so really this loop belongs to A and B.
1102 // We'll get the box using boxes_provided and only filtering for A and B after the fact
1103 // Note that even though the loops are fused, the boxes touched of A and B might be totally different,
1104 // because e.g. B could be double-resolution (as happens when fusing yuv computations), so this
1105 // is not just a matter of giving A's box B's name as an alias.
1106 map<string, Box> boxes_for_fused_group;
1107 map<string, Function> stage_name_to_func;
1108 if (!no_pipelines && producing >= 0 && !f.has_extern_definition()) {
1109 Scope<Interval> empty_scope;
1110 size_t last_dot = op->name.rfind('.');
1111 string var = op->name.substr(last_dot + 1);
1112
1113 set<pair<string, int>> fused_with_f;
1114 for (const auto &pair : fused_pairs_in_groups[stages[producing].fused_group_index]) {
1115 if (!((pair.func_1 == stages[producing].name) && ((int)pair.stage_1 == stage_index)) && is_fused_with_others(fused_groups, fused_pairs_in_groups,
1116 f, stage_index,
1117 pair.func_1, pair.stage_1, var)) {
1118 fused_with_f.insert(make_pair(pair.func_1, pair.stage_1));
1119 }
1120 if (!((pair.func_2 == stages[producing].name) && ((int)pair.stage_2 == stage_index)) && is_fused_with_others(fused_groups, fused_pairs_in_groups,
1121 f, stage_index,
1122 pair.func_2, pair.stage_2, var)) {
1123 fused_with_f.insert(make_pair(pair.func_2, pair.stage_2));
1124 }
1125 }
1126
1127 if (fused_with_f.empty()) {
1128 boxes_for_fused_group[stage_name] = box_provided(body, stages[producing].name, empty_scope, func_bounds);
1129 stage_name_to_func[stage_name] = f;
1130 internal_assert((int)boxes_for_fused_group[stage_name].size() == f.dimensions());
1131 } else {
1132 auto boxes = boxes_provided(body, empty_scope, func_bounds);
1133 boxes_for_fused_group[stage_name] = boxes[stages[producing].name];
1134 stage_name_to_func[stage_name] = f;
1135 internal_assert((int)boxes_for_fused_group[stage_name].size() == f.dimensions());
1136 for (const auto &fused : fused_with_f) {
1137 string fused_stage_name = fused.first + ".s" + std::to_string(fused.second);
1138 boxes_for_fused_group[fused_stage_name] = boxes[fused.first];
1139 for (const auto &fn : funcs) {
1140 if (fn.name() == fused.first) {
1141 stage_name_to_func[fused_stage_name] = fn;
1142 break;
1143 }
1144 }
1145 }
1146 }
1147 }
1148
1149 // Recurse.
1150 body = mutate(body);
1151
1152 if (!no_pipelines) {
1153 // We only care about the bounds of a func if:
1154 // A) We're not already in a pipeline over that func AND
1155 // B.1) There's a production of this func somewhere inside this loop OR
1156 // B.2) We're downstream (a consumer) of a func for which we care about the bounds.
1157 vector<bool> bounds_needed(stages.size(), false);
1158 for (size_t i = 0; i < stages.size(); i++) {
1159 if (inner_productions.count(stages[i].name)) {
1160 bounds_needed[i] = true;
1161 }
1162
1163 if (in_pipeline.count(stages[i].name)) {
1164 bounds_needed[i] = false;
1165 }
1166
1167 if (bounds_needed[i]) {
1168 for (size_t j = 0; j < stages[i].consumers.size(); j++) {
1169 bounds_needed[stages[i].consumers[j]] = true;
1170 }
1171 body = stages[i].define_bounds(
1172 body, f, stage_name, stage_index, op->name, fused_groups,
1173 fused_pairs_in_groups, in_pipeline, inner_productions,
1174 has_extern_consumer, target);
1175 }
1176 }
1177
1178 // Finally, define the production bounds for the thing
1179 // we're producing.
1180 if (producing >= 0 && !inner_productions.empty()) {
1181 for (const auto &b : boxes_for_fused_group) {
1182 const vector<string> &f_args = stage_name_to_func[b.first].args();
1183 const auto &box = b.second;
1184 internal_assert(f_args.size() == box.size());
1185 for (size_t i = 0; i < box.size(); i++) {
1186 internal_assert(box[i].is_bounded());
1187 string var = b.first + "." + f_args[i];
1188
1189 if (box[i].is_single_point()) {
1190 body = LetStmt::make(var + ".max", Variable::make(Int(32), var + ".min"), body);
1191 } else {
1192 body = LetStmt::make(var + ".max", box[i].max, body);
1193 }
1194
1195 body = LetStmt::make(var + ".min", box[i].min, body);
1196 }
1197 }
1198 }
1199
1200 // And the current bounds on its reduction variables, and
1201 // variables from extern for loops.
1202 if (producing >= 0) {
1203 const Stage &s = stages[producing];
1204 vector<string> vars;
1205 if (s.func.has_extern_definition()) {
1206 vars = s.func.args();
1207 }
1208 if (stages[producing].stage > 0) {
1209 for (const ReductionVariable &rv : s.rvars) {
1210 vars.push_back(rv.var);
1211 }
1212 }
1213 for (const string &i : vars) {
1214 string var = s.stage_prefix + i;
1215 Interval in = bounds_of_inner_var(var, body);
1216 if (in.is_bounded()) {
1217 // bounds_of_inner_var doesn't understand
1218 // GuardWithIf, but we know split rvars never
1219 // have inner bounds that exceed the outer
1220 // ones.
1221 if (!s.rvars.empty()) {
1222 in.min = max(in.min, Variable::make(Int(32), var + ".min"));
1223 in.max = min(in.max, Variable::make(Int(32), var + ".max"));
1224 }
1225
1226 body = LetStmt::make(var + ".min", in.min, body);
1227 body = LetStmt::make(var + ".max", in.max, body);
1228 } else {
1229 // If it's not found, we're already in the
1230 // scope of the injected let. The let was
1231 // probably lifted to an outer level.
1232 Expr val;
1233 if (let_vars_in_scope.contains(var + ".guarded")) {
1234 // Use a guarded version if it exists, for tighter bounds inference.
1235 val = Variable::make(Int(32), var + ".guarded");
1236 } else {
1237 val = Variable::make(Int(32), var);
1238 }
1239 body = LetStmt::make(var + ".min", val, body);
1240 body = LetStmt::make(var + ".max", val, body);
1241 }
1242 }
1243 }
1244 }
1245
1246 inner_productions.insert(old_inner_productions.begin(),
1247 old_inner_productions.end());
1248
1249 // Rewrap the let/if statements
1250 for (size_t i = wrappers.size(); i > 0; i--) {
1251 const auto &p = wrappers[i - 1];
1252 if (p.first.empty()) {
1253 body = IfThenElse::make(p.second, body);
1254 } else {
1255 body = LetStmt::make(p.first, p.second, body);
1256 }
1257 }
1258
1259 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, body);
1260 }
1261
1262 Scope<> let_vars_in_scope;
1263 Stmt visit(const LetStmt *op) override {
1264 ScopedBinding<> bind(let_vars_in_scope, op->name);
1265 return IRMutator::visit(op);
1266 }
1267
1268 Stmt visit(const ProducerConsumer *p) override {
1269 in_pipeline.insert(p->name);
1270 Stmt stmt = IRMutator::visit(p);
1271 in_pipeline.erase(p->name);
1272 inner_productions.insert(p->name);
1273 return stmt;
1274 }
1275};

1277} // namespace

1279Stmt bounds_inference(Stmt s,
1280 const vector<Function> &outputs,
1281 const vector<string> &order,
1282 const vector<vector<string>> &fused_groups,
1283 const map<string, Function> &env,
1284 const FuncValueBounds &func_bounds,
1285 const Target &target) {
1286
1287 vector<Function> funcs(order.size());
1288 for (size_t i = 0; i < order.size(); i++) {
1289 funcs[i] = env.find(order[i])->second;
1290 }
1291
1292 // Each element in 'fused_func_groups' indicates a group of functions
1293 // which loops should be fused together.
1294 vector<vector<Function>> fused_func_groups;
1295 for (const vector<string> &group : fused_groups) {
1296 vector<Function> fs;
1297 for (const string &fname : group) {
1298 fs.push_back(env.find(fname)->second);
1299 }
1300 fused_func_groups.push_back(fs);
1301 }
1302
1303 // For each fused group, collect the pairwise fused function stages.
1304 vector<set<FusedPair>> fused_pairs_in_groups;
1305 for (const vector<string> &group : fused_groups) {
1306 set<FusedPair> pairs;
1307 for (const string &fname : group) {
1308 Function f = env.find(fname)->second;
1309 if (!f.has_extern_definition()) {
1310 std::copy(f.definition().schedule().fused_pairs().begin(),
1311 f.definition().schedule().fused_pairs().end(),
1312 std::inserter(pairs, pairs.end()));
1313
1314 for (size_t i = 0; i < f.updates().size(); ++i) {
1315 std::copy(f.updates()[i].schedule().fused_pairs().begin(),
1316 f.updates()[i].schedule().fused_pairs().end(),
1317 std::inserter(pairs, pairs.end()));
1318 }
1319 }
1320 }
1321 fused_pairs_in_groups.push_back(pairs);
1322 }
1323
1324 // Add a note in the IR for where assertions on input images
1325 // should go. Those are handled by a later lowering pass.
1326 Expr marker = Call::make(Int(32), Call::add_image_checks_marker, {}, Call::Intrinsic);
1327 s = Block::make(Evaluate::make(marker), s);
1328
1329 // Add a synthetic outermost loop to act as 'root'.
1330 s = For::make("<outermost>", 0, 1, ForType::Serial, DeviceAPI::None, s);
1331
1332 s = BoundsInference(funcs, fused_func_groups, fused_pairs_in_groups,
1333 outputs, func_bounds, target)
1334 .mutate(s);
1335 return s.as<For>()->body;
1336}
