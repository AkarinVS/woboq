21// Some expressions are not worth lifting out into lets, even if they
22// occur redundantly many times. They may also be illegal to lift out
23// (e.g. calls with side-effects).
24// This list should at least avoid lifting the same cases as that of the
25// simplifier for lets, otherwise CSE and the simplifier will fight each
26// other pointlessly.
27bool should_extract(const Expr &e, bool lift_all) {
28 if (is_const(e)) {
29 return false;
30 }
31
32 if (e.as<Variable>()) {
33 return false;
34 }
35
36 if (lift_all) {
37 return true;
38 }
39
40 if (const Broadcast *a = e.as<Broadcast>()) {
41 return should_extract(a->value, false);
42 }
43
44 if (const Cast *a = e.as<Cast>()) {
45 return should_extract(a->value, false);
46 }
47
48 if (const Add *a = e.as<Add>()) {
49 return !(is_const(a->a) || is_const(a->b));
50 }
51
52 if (const Sub *a = e.as<Sub>()) {
53 return !(is_const(a->a) || is_const(a->b));
54 }
55
56 if (const Mul *a = e.as<Mul>()) {
57 return !(is_const(a->a) || is_const(a->b));
58 }
59
60 if (const Div *a = e.as<Div>()) {
61 return !(is_const(a->a) || is_const(a->b));
62 }
63
64 if (const Ramp *a = e.as<Ramp>()) {
65 return !is_const(a->stride);
66 }
67
68 return true;
69}

71// A global-value-numbering of expressions. Returns canonical form of
72// the Expr and writes out a global value numbering as a side-effect.
73class GVN : public IRMutator {
74public:
75 struct Entry {
76 Expr expr;
77 int use_count = 0;
78 // All consumer Exprs for which this is the last child Expr.
79 map<ExprWithCompareCache, int> uses;
80 Entry(const Expr &e)
81 : expr(e) {
82 }
83 };
84 vector<std::unique_ptr<Entry>> entries;
85
86 map<Expr, int, ExprCompare> shallow_numbering, output_numbering;
87 map<ExprWithCompareCache, int> leaves;
88
89 int number = -1;
90
91 IRCompareCache cache;
92
93 GVN()
94 : number(0), cache(8) {
95 }
96
97 Stmt mutate(const Stmt &s) override {
98 internal_error << "Can't call GVN on a Stmt: " << s << "\n";
99 return Stmt();
100 }
101
102 ExprWithCompareCache with_cache(const Expr &e) {
103 return ExprWithCompareCache(e, &cache);
104 }
105
106 Expr mutate(const Expr &e) override {
107 // Early out if we've already seen this exact Expr.
108 {
109 auto iter = shallow_numbering.find(e);
110 if (iter != shallow_numbering.end()) {
111 number = iter->second;
112 return entries[number]->expr;
113 }
114 }
115
116 // We haven't seen this exact Expr before. Rebuild it using
117 // things already in the numbering.
118 number = -1;
119 Expr new_e = IRMutator::mutate(e);
120
121 // 'number' is now set to the numbering for the last child of
122 // this Expr (or -1 if there are no children). Next we see if
123 // that child has an identical parent to this one.
124
125 auto &use_map = number == -1 ? leaves : entries[number]->uses;
126 auto p = use_map.emplace(with_cache(new_e), (int)entries.size());
127 auto iter = p.first;
128 bool novel = p.second;
129 if (novel) {
130 // This is a never-before-seen Expr
131 number = (int)entries.size();
132 iter->second = number;
133 entries.emplace_back(new Entry(new_e));
134 } else {
135 // This child already has a syntactically-equal parent
136 number = iter->second;
137 new_e = entries[number]->expr;
138 }
139
140 // Memorize this numbering for the old and new forms of this Expr
141 shallow_numbering[e] = number;
142 output_numbering[new_e] = number;
143 return new_e;
144 }
145};

147/** Fill in the use counts in a global value numbering. */
148class ComputeUseCounts : public IRGraphVisitor {
149 GVN &gvn;
150 bool lift_all;
151
152public:
153 ComputeUseCounts(GVN &g, bool l)
154 : gvn(g), lift_all(l) {
155 }
156
157 using IRGraphVisitor::include;
158 using IRGraphVisitor::visit;
159
160 void include(const Expr &e) override {
161 // If it's not the sort of thing we want to extract as a let,
162 // just use the generic visitor to increment use counts for
163 // the children.
164 debug(4) << "Include: " << e
165 << "; should extract: " << should_extract(e, lift_all) << "\n";
166 if (!should_extract(e, lift_all)) {
167 e.accept(this);
168 return;
169 }
170
171 // Find this thing's number.
172 auto iter = gvn.output_numbering.find(e);
173 if (iter != gvn.output_numbering.end()) {
174 gvn.entries[iter->second]->use_count++;
175 } else {
176 internal_error << "Expr not in shallow numbering: " << e << "\n";
177 }
178
179 // Visit the children if we haven't been here before.
180 IRGraphVisitor::include(e);
181 }
182};

198class RemoveLets : public IRGraphMutator {
199 using IRGraphMutator::visit;
200
201 Scope<Expr> scope;
202
203 Expr visit(const Variable *op) override {
204 if (scope.contains(op->name)) {
205 return scope.get(op->name);
206 } else {
207 return op;
208 }
209 }
210
211 Expr visit(const Let *op) override {
212 Expr new_value = mutate(op->value);
213 // When we enter a let, we invalidate all cached mutations
214 // with values that reference this var due to shadowing. When
215 // we leave a let, we similarly invalidate any cached
216 // mutations we learned on the inside that reference the var.
217
218 // A blunt way to handle this is to temporarily invalidate
219 // *all* mutations, so we never see the same Expr node
220 // on the inside and outside of a Let.
221 decltype(expr_replacements) tmp;
222 tmp.swap(expr_replacements);
223 ScopedBinding<Expr> bind(scope, op->name, new_value);
224 auto result = mutate(op->body);
225 tmp.swap(expr_replacements);
226 return result;
227 }
228};

230class CSEEveryExprInStmt : public IRMutator {
231 bool lift_all;
232 using IRMutator::visit;
233
234 Stmt visit(const Store *op) override {
235 // It's important to do CSE jointly on the index and value in
236 // a store to stop:
237 // f[x] = f[x] + y
238 // from turning into
239 // f[x] = f[z] + y
240 // due to the two equal x's indices being CSE'd differently due to the presence of y.
241 Expr dummy = Call::make(Int(32), Call::bundle, {op->value, op->index}, Call::PureIntrinsic);
242 dummy = common_subexpression_elimination(dummy, lift_all);
243 vector<pair<string, Expr>> lets;
244 while (const Let *let = dummy.as<Let>()) {
245 lets.emplace_back(let->name, let->value);
246 dummy = let->body;
247 }
248 const Call *bundle = Call::as_intrinsic(dummy, {Call::bundle});
249 internal_assert(bundle && bundle->args.size() == 2);
250 Stmt s = Store::make(op->name, bundle->args[0], bundle->args[1],
251 op->param, mutate(op->predicate), op->alignment);
252 for (auto it = lets.rbegin(); it != lets.rend(); it++) {
253 s = LetStmt::make(it->first, it->second, s);
254 }
255 return s;
256 }
257
258public:
259 using IRMutator::mutate;
260
261 Expr mutate(const Expr &e) override {
262 return common_subexpression_elimination(e, lift_all);
263 }
264
265 CSEEveryExprInStmt(bool l)
266 : lift_all(l) {
267 }
268};

272Expr common_subexpression_elimination(const Expr &e_in, bool lift_all) {
273 Expr e = e_in;
274
275 // Early-out for trivial cases.
276 if (is_const(e) || e.as<Variable>()) {
277 return e;
278 }
279
280 debug(4) << "\n\n\nInput to CSE " << e << "\n";
281
282 e = RemoveLets().mutate(e);
283
284 debug(4) << "After removing lets: " << e << "\n";
285
286 GVN gvn;
287 e = gvn.mutate(e);
288
289 ComputeUseCounts count_uses(gvn, lift_all);
290 count_uses.include(e);
291
292 debug(4) << "Canonical form without lets " << e << "\n";
293
294 // Figure out which ones we'll pull out as lets and variables.
295 vector<pair<string, Expr>> lets;
296 vector<Expr> new_version(gvn.entries.size());
297 map<Expr, Expr, ExprCompare> replacements;
298 for (size_t i = 0; i < gvn.entries.size(); i++) {
299 const auto &e = gvn.entries[i];
300 if (e->use_count > 1) {
301 string name = unique_name('t');
302 lets.emplace_back(name, e->expr);
303 // Point references to this expr to the variable instead.
304 replacements[e->expr] = Variable::make(e->expr.type(), name);
305 }
306 debug(4) << i << ": " << e->expr << ", " << e->use_count << "\n";
307 }
308
309 // Rebuild the expr to include references to the variables:
310 Replacer replacer(replacements);
311 e = replacer.mutate(e);
312
313 debug(4) << "With variables " << e << "\n";
314
315 // Wrap the final expr in the lets.
316 for (size_t i = lets.size(); i > 0; i--) {
317 Expr value = lets[i - 1].second;
318 // Drop this variable as an acceptable replacement for this expr.
319 replacer.erase(value);
320 // Use containing lets in the value.
321 value = replacer.mutate(lets[i - 1].second);
322 e = Let::make(lets[i - 1].first, value, e);
323 }
324
325 debug(4) << "With lets: " << e << "\n";
326
327 return e;
328}

330Stmt common_subexpression_elimination(const Stmt &s, bool lift_all) {
331 return CSEEveryExprInStmt(lift_all).mutate(s);
332}

338// Normalize all names in an expr so that expr compares can be done
339// without worrying about mere name differences.
340class NormalizeVarNames : public IRMutator {
341 int counter = 0;
342
343 map<string, string> new_names;
344
345 using IRMutator::visit;
346
347 Expr visit(const Variable *var) override {
348 map<string, string>::iterator iter = new_names.find(var->name);
349 if (iter == new_names.end()) {
350 return var;
351 } else {
352 return Variable::make(var->type, iter->second);
353 }
354 }
355
356 Expr visit(const Let *let) override {
357 string new_name = "t" + std::to_string(counter++);
358 new_names[let->name] = new_name;
359 Expr value = mutate(let->value);
360 Expr body = mutate(let->body);
361 return Let::make(new_name, value, body);
362 }
363
364public:
365 NormalizeVarNames() = default;
366};

368void check(const Expr &in, const Expr &correct) {
369 Expr result = common_subexpression_elimination(in);
370 NormalizeVarNames n;
371 result = n.mutate(result);
372 internal_assert(equal(result, correct))
373 << "Incorrect CSE:\n"
374 << in
375 << "\nbecame:\n"
376 << result
377 << "\ninstead of:\n"
378 << correct << "\n";
379}

381// Construct a nested block of lets. Variables of the form "tn" refer
382// to expr n in the vector.
383Expr ssa_block(vector<Expr> exprs) {
384 Expr e = exprs.back();
385 for (size_t i = exprs.size() - 1; i > 0; i--) {
386 string name = "t" + std::to_string(i - 1);
387 e = Let::make(name, exprs[i - 1], e);
388 }
389 return e;
390}

394void cse_test() {
395 Expr x = Variable::make(Int(32), "x");
396 Expr y = Variable::make(Int(32), "y");
397
398 Expr t[32], tf[32];
399 for (int i = 0; i < 32; i++) {
400 t[i] = Variable::make(Int(32), "t" + std::to_string(i));
401 tf[i] = Variable::make(Float(32), "t" + std::to_string(i));
402 }
403 Expr e, correct;
404
405 // This is fine as-is.
406 e = ssa_block({sin(x), tf[0] * tf[0]});
407 check(e, e);
408
409 // Test a simple case.
410 e = ((x * x + x) * (x * x + x)) + x * x;
411 e += e;
412 correct = ssa_block({x * x, // x*x
413 t[0] + x, // x*x + x
414 t[1] * t[1] + t[0], // (x*x + x)*(x*x + x) + x*x
415 t[2] + t[2]});
416 check(e, correct);
417
418 // Check for idempotence (also checks a case with lets)
419 check(correct, correct);
420
421 // Check a case with redundant lets
422 e = ssa_block({x * x,
423 x * x,
424 t[0] / t[1],
425 t[1] / t[1],
426 t[2] % t[3],
427 (t[4] + x * x) + x * x});
428 correct = ssa_block({x * x,
429 t[0] / t[0],
430 (t[1] % t[1] + t[0]) + t[0]});
431 check(e, correct);
432
433 // Check a case with nested lets with shared subexpressions
434 // between the lets, and repeated names.
435 Expr e1 = ssa_block({x * x, // a = x*x
436 t[0] + x, // b = a + x
437 t[1] * t[1] * t[0]}); // c = b * b * a
438 Expr e2 = ssa_block({x * x, // a again
439 t[0] - x, // d = a - x
440 t[1] * t[1] * t[0]}); // e = d * d * a
441 e = ssa_block({e1 + x * x, // f = c + a
442 e1 + e2, // g = c + e
443 t[0] + t[0] * t[1]}); // h = f + f * g
444
445 correct = ssa_block({x * x, // t0 = a = x*x
446 t[0] + x, // t1 = b = a + x = t0 + x
447 t[1] * t[1] * t[0], // t2 = c = b * b * a = t1 * t1 * t0
448 t[2] + t[0], // t3 = f = c + a = t2 + t0
449 t[0] - x, // t4 = d = a - x = t0 - x
450 t[3] + t[3] * (t[2] + t[4] * t[4] * t[0])}); // h (with g substituted in)
451 check(e, correct);
452
453 // Test it scales OK.
454 e = x;
455 for (int i = 0; i < 100; i++) {
456 e = e * e + e + i;
457 e = e * e - e * i;
458 }
459 Expr result = common_subexpression_elimination(e);
460
461 {
462 Expr pred = x * x + y * y > 0;
463 Expr index = select(x * x + y * y > 0, x * x + y * y + 2, x * x + y * y + 10);
464 Expr load = Load::make(Int(32), "buf", index, Buffer<>(), Parameter(), const_true(), ModulusRemainder());
465 Expr pred_load = Load::make(Int(32), "buf", index, Buffer<>(), Parameter(), pred, ModulusRemainder());
466 e = select(x * y > 10, x * y + 2, x * y + 3 + load) + pred_load;
467

470 Expr cse_pred_load = Load::make(Int(32), "buf", t[3], Buffer<>(), Parameter(), t2, ModulusRemainder());
471 correct = ssa_block({x * y,
472 x * x + y * y,
473 t[1] > 0,
474 select(t2, t[1] + 2, t[1] + 10),
475 select(t[0] > 10, t[0] + 2, t[0] + 3 + cse_load) + cse_pred_load});
