26// Loop partitioning only applies to things marked as 'likely'. Loads
27// through hand-written boundary conditions will produce clamped
28// ramps, which will turn into gathers. This pass injects likely
29// intrinsics so that these clamped ramps are picked up by loop
30// partitioning.
31class MarkClampedRampsAsLikely : public IRMutator {
32 using IRMutator::visit;
33 Expr visit(const Min *op) override {
34 if (in_index && op->a.as<Ramp>()) {
35 // No point recursing into the ramp - it can't contain
36 // another ramp.
37 return min(likely(op->a), mutate(op->b));
38 } else if (in_index && op->b.as<Ramp>()) {
39 return min(mutate(op->a), likely(op->b));
40 } else {
41 return IRMutator::visit(op);
42 }
43 }
44
45 Expr visit(const Max *op) override {
46 if (in_index && op->a.as<Ramp>()) {
47 return max(likely(op->a), mutate(op->b));
48 } else if (in_index && op->b.as<Ramp>()) {
49 return max(mutate(op->a), likely(op->b));
50 } else {
51 return IRMutator::visit(op);
52 }
53 }
54
55 Expr visit(const Load *op) override {
56 bool old_in_index = in_index;
57 in_index = true;
58 Expr expr = IRMutator::visit(op);
59 in_index = old_in_index;
60 return expr;
61 }
62
63 Stmt visit(const Store *op) override {
64 bool old_in_index = in_index;
65 in_index = true;
66 Expr index = mutate(op->index);
67 in_index = old_in_index;
68 Expr value = mutate(op->value);
69 Expr predicate = mutate(op->predicate);
70 if (predicate.same_as(op->predicate) && index.same_as(op->index) && value.same_as(op->value)) {
71 return op;
72 } else {
73 return Store::make(op->name, value, index, op->param, predicate, op->alignment);
74 }
75 }
76
77 bool in_index = false;
78};

108// The goal of loop partitioning is to split loops up into a prologue,
109// a clean steady state, and an epilogue. The next visitor
110// (FindSimplifications) finds a list of simplifications that can be
111// applied to produce that clean steady-state version of the loop
112// body. It tries to simplify selects, mins, and maxes to just their
113// likely branch. For example:
114//
115// select(a, likely(b), c) -> b
116// select(a, b, 5 + likely(c)) -> 5 + c
117// max(a, likely(b)) -> b
118//
119// These three simplifications are only valid if a is true, false, or
120// less than b, respectively. So we visit the loop body looking for
121// these sort of things, record the associated conditions, and try to
122// solve for a range of the loop variable for which all of our
123// conditions are true (by solving for each one and then taking the
124// intersection). That gives us the clean steady state.
125//
126// It may be that we can also make some simplifications to the
127// prologue or epilogue. For example, consider the case:
128//
129// select(x > 0, likely(expr_t), expr_f)
130//
131// It can simplify to expr_t when x > 0. However, if this is the sole
132// simplification which gives us a lower bound on x for the steady
133// state, we can also simplify this select in the prologue to just be
134// expr_f.
135//
136// Now consider this case:
137//
138// (select(x > a, likely(expr_t1), expr_f1) +
139// select(x > b, likely(expr_t2), expr_f2))
140//
141// The steady state starts at x == max(a, b), which we get from the
142// intersection of the intervals derived from each condition: x > a
143// and x > b. In the steady state, the expression simplifies to
144// expr_t1 + expr_t2. In the prologue we know that either x <= a or x
145// <= b, but we don't know which one might be true, so we can't make
146// any simplifications to the prologue.
147//
148// We may also encounter single conditions where we can simplify the
149// steady-state but not the prologue. Say we're splitting up a loop
150// over x and we encounter a condition that depends on a variable
151// introduced in some inner loop:
152//
153// for x:
154// for z from 0 to 10:
155// ... select(x > z, likely(expr_t), expr_f) ...
156//
157// This select definitely simplifies to expr_t when x > 9, because
158// that's the maximum value z could be, so we'll start the steady
159// state at x == 10. This means the prologue covers values like x ==
160// 5, where the select could be either true or false, so we can't make
161// any simplifications to the prologue.
162//
163// There are some simplifications that we won't be able to do. For
164// example, if we're partitioning the loop over x, and we encounter:
165//
166// for x:
167// for z from 0 to 10:
168// ... select(z < 5, likely(expr_t), expr_f)
169//
170// Restricting the range of x isn't going to simplify that expression
171// - it doesn't even depend on x. We just make all the simplifications
172// that we can, and take the intersection of the resulting regions. In
173// this case, we'll make that simplification later, when we do loop
174// partitioning over the loop in z. Some cases we'll never
175// handle. E.g. consider:
176//
177// for x:
178// ... select(a + x*(b + x*(c + x*(d + x*e))) > 0, likely(expr_t), expr_f)
179//
180// In order to simplify that we'd have to come up with a formula that
181// tells us an interval where a quintic is strictly positive. No such
182// general formula exists (because no formula exists for the roots),
183// so there's no programmatic way we can partition the loop over x to
184// make that condition simplify. Finally my Galois theory course pays
185// off. For failures like this, we just drop the likely tag. So loop
186// partitioning is best-effort, but it should always work for things
187// like x > a. A simpler case for which we bail is:
188//
189// for x:
190// ... select(x == 5, expr_t, likely(expr_f))
191//
192// This simplifies to the likely case in two disjoint ranges, but
193// we're only producing one steady state, and we have no reason to
194// believe one side is better than the other, so we just bail and drop
195// the likely tag.
196
197// First we define the struct that represents a single simplification
198// that can be applied to the steady state of the loop.
199struct Simplification {
200 // This condition is sufficient for the simplification to occur.
201 Expr condition;
202 // The expression we're simplifying
203 Expr old_expr;
204 // The replacement if the condition is true
205 Expr likely_value;
206 // The replacement if the condition is false. Not useful
207 // unless it's tight.
208 Expr unlikely_value;
209 // Is the condition necessary (as well as sufficient)?
210 bool tight;
211 // The interval over which this simplification applies. Comes from solving the condition.
212 Interval interval;
213};

243class FindSimplifications : public IRVisitor {
244 using IRVisitor::visit;
245
246 Scope<> depends_on_loop_var, depends_on_invalid_buffers;
247 Scope<> buffers;
248
249 void visit(const Allocate *op) override {
250 buffers.push(op->name);
251 IRVisitor::visit(op);
252 }
253
254 void new_simplification(Expr condition, Expr old, Expr likely_val, Expr unlikely_val) {
255 if (!expr_uses_vars(condition, depends_on_loop_var)) {
256 return;
257 }
258
259 if (expr_uses_vars(condition, depends_on_invalid_buffers) ||
260 expr_uses_invalid_buffers(condition, buffers)) {
261 // The condition refers to buffer allocated in the inner loop.
262 // We should throw away the condition
263 return;
264 }
265 condition = remove_likelies(condition);
266 Simplification s = {condition, std::move(old), std::move(likely_val), std::move(unlikely_val), true};
267 if (s.condition.type().is_vector()) {
268 s.condition = simplify(s.condition);
269 if (const Broadcast *b = s.condition.as<Broadcast>()) {
270 s.condition = b->value;
271 } else {
272 // Devectorize the condition
273 s.condition = and_condition_over_domain(s.condition, Scope<Interval>::empty_scope());
274 s.tight = false;
275 }
276 }
277 internal_assert(s.condition.type().is_scalar()) << s.condition << "\n";
278 simplifications.push_back(s);
279 }
280
281 void visit(const Min *op) override {
282 bool likely_a = has_uncaptured_likely_tag(op->a);
283 bool likely_b = has_uncaptured_likely_tag(op->b);
284
285 // If one side has an uncaptured likely, don't hunt for
286 // simplifications in the other side.
287 if (!likely_a) {
288 op->b.accept(this);
289 }
290 if (!likely_b) {
291 op->a.accept(this);
292 }
293
294 // Prefer the side that has an uncaptured top-level likely
295 // call. If neither does, prefer the side that contains any
296 // likely call at all.
297 if (!likely_a && !likely_b) {
298 likely_a = has_likely_tag(op->a);
299 likely_b = has_likely_tag(op->b);
300 }
301
302 if (likely_b && !likely_a) {
303 new_simplification(op->b <= op->a, op, op->b, op->a);
304 } else if (likely_a && !likely_b) {
305 new_simplification(op->a <= op->b, op, op->a, op->b);
306 }
307 }
308
309 void visit(const Max *op) override {
310 bool likely_a = has_uncaptured_likely_tag(op->a);
311 bool likely_b = has_uncaptured_likely_tag(op->b);
312
313 if (!likely_a) {
314 op->b.accept(this);
315 }
316 if (!likely_b) {
317 op->a.accept(this);
318 }
319
320 if (!likely_a && !likely_b) {
321 likely_a = has_likely_tag(op->a);
322 likely_b = has_likely_tag(op->b);
323 }
324
325 if (likely_b && !likely_a) {
326 new_simplification(op->b >= op->a, op, op->b, op->a);
327 } else if (likely_a && !likely_b) {
328 new_simplification(op->a >= op->b, op, op->a, op->b);
329 }
330 }
331
332 void visit_select(const Expr &condition, const Expr &old, const Expr &true_value, const Expr &false_value) {
333 condition.accept(this);
334
335 bool likely_t = has_uncaptured_likely_tag(true_value);
336 bool likely_f = has_uncaptured_likely_tag(false_value);
337
338 if (!likely_t && !likely_f) {
339 likely_t = has_likely_tag(true_value);
340 likely_f = has_likely_tag(false_value);
341 }
342
343 if (!likely_t) {
344 false_value.accept(this);
345 }
346 if (!likely_f) {
347 true_value.accept(this);
348 }
349
350 if (likely_t && !likely_f) {
351 new_simplification(condition, old, true_value, false_value);
352 } else if (likely_f && !likely_t) {
353 new_simplification(!condition, old, false_value, true_value);
354 }
355 }

357 void visit(const Select *op) override {
358 visit_select(op->condition, op, op->true_value, op->false_value);
359 }
360
361 void visit(const Call *op) override {
362 if (op->is_intrinsic(Call::if_then_else)) {
363 visit_select(op->args[0], op, op->args[1], op->args[2]);
364 } else {
365 IRVisitor::visit(op);
366 }

369 void visit(const IfThenElse *op) override {
370 // For select statements, mins, and maxes, you can mark the
371 // likely branch with likely. For if statements there's no way
372 // to mark the likely stmt. So if the condition of an if
373 // statement is marked as likely, treat it as likely true and
374 // partition accordingly.
375 IRVisitor::visit(op);
376 if (has_uncaptured_likely_tag(op->condition)) {
377 new_simplification(op->condition, op->condition, const_true(), const_false());
378 }
379 }
380
381 void visit(const For *op) override {
382 vector<Simplification> old;
383 old.swap(simplifications);
384 IRVisitor::visit(op);
385
386 // Relax all the new conditions using the loop bounds
387 for (Simplification &s : simplifications) {
388 if (expr_uses_var(s.condition, op->name)) {
389 Scope<Interval> varying;
390 varying.push(op->name, Interval(op->min, op->min + op->extent - 1));
391 Expr relaxed = and_condition_over_domain(s.condition, varying);
392 internal_assert(!expr_uses_var(relaxed, op->name))
393 << "Should not have had used the loop var (" << op->name
394 << ") any longer\n before: " << s.condition << "\n after: "
395 << relaxed << "\n";
396 if (!equal(relaxed, s.condition)) {
397 s.tight = false;
398 }
399 s.condition = relaxed;
400 }
401 }
402
403 simplifications.insert(simplifications.end(), old.begin(), old.end());
404 }

406 template<typename LetOrLetStmt>
407 void visit_let(const LetOrLetStmt *op) {
408 ScopedBinding<> bind_varying(expr_uses_vars(op->value, depends_on_loop_var),
409 depends_on_loop_var, op->name);
410 ScopedBinding<> bind_invalid(expr_uses_invalid_buffers(op->value, buffers) ||
411 expr_uses_vars(op->value, depends_on_invalid_buffers),
412 depends_on_invalid_buffers, op->name);
413 vector<Simplification> old;
414 old.swap(simplifications);
415 IRVisitor::visit(op);
416 for (Simplification &s : simplifications) {
417 if (expr_uses_var(s.condition, op->name)) {
418 s.condition = Let::make(op->name, op->value, s.condition);
419 }
420 }
421 simplifications.insert(simplifications.end(), old.begin(), old.end());
422 }
423
424 void visit(const LetStmt *op) override {
425 visit_let(op);
426 }
427
428 void visit(const Let *op) override {
429 visit_let(op);
430 }

494class PartitionLoops : public IRMutator {
495 using IRMutator::visit;
496
497 bool in_gpu_loop = false;
498
499 Stmt visit(const For *op) override {
500 Stmt body = op->body;
501
502 ScopedValue<bool> old_in_gpu_loop(in_gpu_loop, in_gpu_loop ||
503 CodeGen_GPU_Dev::is_gpu_var(op->name));
504
505 // If we're inside GPU kernel, and the body contains thread
506 // barriers or warp shuffles, it's not safe to partition loops.
507 if (in_gpu_loop && contains_warp_synchronous_logic(op)) {
508 return IRMutator::visit(op);
509 }
510
511 // Find simplifications in this loop body
512 FindSimplifications finder(op->name);
513 body.accept(&finder);
514
515 if (finder.simplifications.empty()) {
516 return IRMutator::visit(op);
517 }
518
519 debug(3) << "\n\n**** Partitioning loop over " << op->name << "\n";
520
521 vector<Expr> min_vals, max_vals;
522 vector<Simplification> middle_simps, prologue_simps, epilogue_simps;
523 bool lower_bound_is_tight = true, upper_bound_is_tight = true;
524 for (auto &s : finder.simplifications) {
525 // Solve for the interval over which this simplification is true.
526 s.interval = solve_for_inner_interval(s.condition, op->name);
527 if (s.tight) {
528 // Check if the solve is tight. I.e. the condition is
529 // definitely false outside of the interval.
530 Interval outer = solve_for_outer_interval(s.condition, op->name);
531 s.tight &= equal(outer.min, s.interval.min) && equal(outer.max, s.interval.max);
532 }
533
534 debug(3) << "\nSimplification: \n"
535 << " condition: " << s.condition << "\n"
536 << " old: " << s.old_expr << "\n"
537 << " new: " << s.likely_value << "\n"
538 << " min: " << s.interval.min << "\n"
539 << " max: " << s.interval.max << "\n"
540 << " tight: " << s.tight << "\n";
541
542 // Accept all non-empty intervals
543 if (!s.interval.is_empty()) {
544 if (s.interval.has_lower_bound()) {
545 Expr m = s.interval.min;
546 if (!s.tight) {
547 lower_bound_is_tight = false;
548 }
549 if (min_vals.empty()) {
550 min_vals.push_back(m);
551 } else if (equal(m, min_vals.back())) {
552 // We already have this min val
553 } else {
554 // This is a new distinct min val
555 min_vals.push_back(m);
556 lower_bound_is_tight = false;
557 }
558 }
559 if (s.interval.has_upper_bound()) {
560 Expr m = s.interval.max;
561 if (!s.tight) {
562 upper_bound_is_tight = false;
563 }
564 if (max_vals.empty()) {
565 max_vals.push_back(m);
566 } else if (equal(m, max_vals.back())) {
567 // We already have this max val
568 } else {
569 // This is a new distinct max val
570 max_vals.push_back(m);
571 upper_bound_is_tight = false;
572 }
573 }
574
575 // We'll apply this simplification to the
576 // steady-state.
577 middle_simps.push_back(s);
578 }
579 }
580
581 // In general we can't simplify the prologue - it may run up
582 // to after the epilogue starts for small images. However if
583 // we can prove the epilogue starts after the prologue ends,
584 // we're OK.
585 bool can_simplify_prologue = true;
586 for (const Expr &min_val : min_vals) {
587 for (const Expr &max_val : max_vals) {
588 Expr test = simplify(common_subexpression_elimination(min_val - 1 < max_val + 1));
589 if (!is_const_one(test)) {
590 can_simplify_prologue = false;
591 }
592 }
593 }
594
595 // Find simplifications we can apply to the prologue and epilogue.
596 for (const auto &s : middle_simps) {
597 // If it goes down to minus infinity, we can also
598 // apply it to the prologue
599 if (can_simplify_prologue &&
600 !s.interval.has_lower_bound()) {
601 prologue_simps.push_back(s);
602 }
603
604 // If it goes up to positive infinity, we can also
605 // apply it to the epilogue
606 if (!s.interval.has_upper_bound()) {
607 epilogue_simps.push_back(s);
608 }
609
610 // If our simplifications only contain one lower bound, and
611 // it's tight, then the reverse rule can be applied to the
612 // prologue.
613 if (can_simplify_prologue &&
614 s.interval.has_lower_bound() &&
615 lower_bound_is_tight) {
616 internal_assert(s.tight);
617 Simplification s2 = s;
618 // This condition is never used (we already solved
619 // for the interval), but it's nice for it to be
620 // correct.
621 s2.condition = !s2.condition;
622 std::swap(s2.likely_value, s2.unlikely_value);
623 prologue_simps.push_back(s2);
624 }
625 if (s.interval.has_upper_bound() &&
626 upper_bound_is_tight) {
627 internal_assert(s.tight);
628 Simplification s2 = s;
629 s2.condition = !s2.condition;
630 std::swap(s2.likely_value, s2.unlikely_value);
631 epilogue_simps.push_back(s2);
632 }
633 }
634
635 // Simplify each section of the loop.
636 Stmt simpler_body = MakeSimplifications(middle_simps).mutate(body);
637 Stmt prologue = MakeSimplifications(prologue_simps).mutate(body);
638 Stmt epilogue = MakeSimplifications(epilogue_simps).mutate(body);
639
640 bool make_prologue = !equal(prologue, simpler_body);
641 bool make_epilogue = !equal(epilogue, simpler_body);
642
643 // Recurse on the middle section.
644 simpler_body = mutate(simpler_body);
645
646 // Construct variables for the bounds of the simplified middle section
647 Expr min_steady = op->min, max_steady = op->extent + op->min;
648 Expr prologue_val, epilogue_val;
649 string prologue_name = unique_name(op->name + ".prologue");
650 string epilogue_name = unique_name(op->name + ".epilogue");
651
652 if (make_prologue) {
653 // They'll simplify better if you put them in
654 // lexicographic order. This puts things like (x+1) and
655 // (x+3) next to each other so that the simplifier sees
656 // them together and can drop one of them.
657 std::sort(min_vals.begin(), min_vals.end(), IRDeepCompare());
658 min_vals.push_back(op->min);
659 prologue_val = fold_left(min_vals, Max::make);
660 // Stop the prologue from running past the end of the loop
661 prologue_val = min(prologue_val, op->extent + op->min);
662 // prologue_val = print(prologue_val, prologue_name);
663 min_steady = Variable::make(Int(32), prologue_name);
664
665 internal_assert(!expr_uses_var(prologue_val, op->name));
666 }
667 if (make_epilogue) {
668 std::sort(max_vals.begin(), max_vals.end(), IRDeepCompare());
669 max_vals.push_back(op->min + op->extent - 1);
670 epilogue_val = fold_left(max_vals, Min::make) + 1;
671 // Stop the epilogue from running before the start of the loop/prologue
672 if (make_prologue) {
673 epilogue_val = max(epilogue_val, prologue_val);
674 } else {
675 epilogue_val = max(op->min, epilogue_val);
676 }
677 // epilogue_val = print(epilogue_val, epilogue_name);
678 max_steady = Variable::make(Int(32), epilogue_name);
679
680 internal_assert(!expr_uses_var(epilogue_val, op->name));
681 }
682
683 Stmt stmt;
684 // Bust simple serial for loops up into three.
685 if (op->for_type == ForType::Serial && !op->body.as<Acquire>()) {
686 stmt = For::make(op->name, min_steady, max_steady - min_steady,
687 op->for_type, op->device_api, simpler_body);
688
689 if (make_prologue) {
690 prologue = For::make(op->name, op->min, min_steady - op->min,
691 op->for_type, op->device_api, prologue);
692 stmt = Block::make(prologue, stmt);
693 }
694 if (make_epilogue) {
695 epilogue = For::make(op->name, max_steady, op->min + op->extent - max_steady,
696 op->for_type, op->device_api, epilogue);
697 stmt = Block::make(stmt, epilogue);
698 }
699 } else {
700 // For parallel for loops we could use a Fork node here,
701 // but that would introduce the more complicated parallel
702 // runtime into code that doesn't use async(), which may
703 // interfere with legacy overrides of
704 // halide_do_par_for. So for parallel for loops just put
705 // an if-then-else in the loop body. It should
706 // branch-predict to the steady state pretty well.
707 //
708 // Simple serial for loops that contain an Acquire node go
709 // into the task system as a single entity, but Block
710 // nodes do not, so we get a flatter task graph if we do
711 // the same trick.
712 Expr loop_var = Variable::make(Int(32), op->name);
713 stmt = simpler_body;
714 if (make_epilogue && make_prologue && equal(prologue, epilogue)) {
715 stmt = IfThenElse::make(min_steady <= loop_var && loop_var < max_steady, stmt, prologue);
716 } else {
717 if (make_epilogue) {
718 stmt = IfThenElse::make(loop_var < max_steady, stmt, epilogue);
719 }
720 if (make_prologue) {
721 stmt = IfThenElse::make(loop_var < min_steady, prologue, stmt);
722 }
723 }
724 stmt = For::make(op->name, op->min, op->extent, op->for_type, op->device_api, stmt);
725 }
726
727 if (make_epilogue) {
728 // Uncomment to include code that prints the epilogue value
729 //epilogue_val = print(epilogue_val, op->name, "epilogue");
730 stmt = LetStmt::make(epilogue_name, epilogue_val, stmt);
731 } else {
732 epilogue_val = op->min + op->extent;
733 }
734 if (make_prologue) {
735 // Uncomment to include code that prints the prologue value
736 //prologue_val = print(prologue_val, op->name, "prologue");
737 stmt = LetStmt::make(prologue_name, prologue_val, stmt);
738 } else {
739 prologue_val = op->min;
740 }
741
742 if (can_prove(epilogue_val <= prologue_val)) {
743 // The steady state is empty. I've made a huge
744 // mistake. Try to partition a loop further in.
745 return IRMutator::visit(op);
746 }
747
748 debug(3) << "Partition loop.\n"
749 << "Old: " << Stmt(op) << "\n"
750 << "New: " << stmt << "\n";
751
752 return stmt;
753 }

773// The loop partitioning logic can introduce if and let statements in
774// between GPU loop levels. This pass moves them inwards or outwards.
775class RenormalizeGPULoops : public IRMutator {
776 bool in_gpu_loop = false, in_thread_loop = false;
777
778 using IRMutator::visit;
779
780 // Track all vars that depend on GPU loop indices or loops inside GPU kernels.
781 Scope<> gpu_vars;
782
783 vector<pair<string, Expr>> lifted_lets;
784
785 Stmt visit(const For *op) override {
786 bool old_in_gpu_loop = in_gpu_loop;
787 Stmt stmt;
788
789 if (in_gpu_loop || CodeGen_GPU_Dev::is_gpu_var(op->name)) {
790 gpu_vars.push(op->name);
791 in_gpu_loop = true;
792 }
793
794 if (ends_with(op->name, "__thread_id_x")) {
795 internal_assert(!in_thread_loop);
796 in_thread_loop = true;
797 stmt = IRMutator::visit(op);
798 in_thread_loop = false;
799 } else {
800 stmt = IRMutator::visit(op);
801 }
802
803 if (in_gpu_loop && !old_in_gpu_loop) {
804 // This was the outermost GPU loop. Dump any lifted lets here.
805 while (!lifted_lets.empty()) {
806 stmt = LetStmt::make(lifted_lets.back().first,
807 lifted_lets.back().second,
808 stmt);
809 lifted_lets.pop_back();
810 }
811 }
812
813 in_gpu_loop = old_in_gpu_loop;
814 return stmt;
815 }
816
817 Stmt visit(const LetStmt *op) override {
818 if (!in_gpu_loop) {
819 return IRMutator::visit(op);
820 }
821
822 if (!expr_uses_vars(op->value, gpu_vars) && !expr_contains_load(op->value)) {
823 // This let value doesn't depend in the gpu vars. We
824 // should lift it outermost. Note that this might expand
825 // its scope to encompass other uses of the same name, so
826 // we'd better give it a new name.
827 string new_name = unique_name('t');
828 Expr new_var = Variable::make(op->value.type(), new_name);
829 lifted_lets.emplace_back(new_name, op->value);
830 return mutate(substitute(op->name, new_var, op->body));
831 }
832
833 gpu_vars.push(op->name);
834
835 if (in_thread_loop) {
836 return IRMutator::visit(op);
837 }
838
839 Stmt body = mutate(op->body);
840 const For *f = body.as<For>();
841 const Allocate *a = body.as<Allocate>();
842 // Move lets in-between gpu loop levels inwards.
843 if (f && in_gpu_loop && !in_thread_loop) {
844 internal_assert(!expr_uses_var(f->min, op->name) &&
845 !expr_uses_var(f->extent, op->name));
846 Stmt inner = LetStmt::make(op->name, op->value, f->body);
847 inner = For::make(f->name, f->min, f->extent, f->for_type, f->device_api, inner);
848 return mutate(inner);
849 } else if (a && in_gpu_loop && !in_thread_loop) {
850 internal_assert(a->extents.size() == 1);
851 if (expr_uses_var(a->extents[0], op->name)) {
852 // This var depends on the block index, and is used to
853 // define the size of shared memory. Can't move it
854 // inwards or outwards. Codegen will have to deal with
855 // it when it deduces how much shared or warp-level
856 // memory to allocate.
857 return IRMutator::visit(op);
858 } else {
859 Stmt inner = LetStmt::make(op->name, op->value, a->body);
860 inner = Allocate::make(a->name, a->type, a->memory_type, a->extents, a->condition, inner);
861 return mutate(inner);
862 }
863 } else {
864 return IRMutator::visit(op);
865 }
866 }
867
868 Stmt visit(const IfThenElse *op) override {
869 if (!in_gpu_loop || in_thread_loop) {
870 return IRMutator::visit(op);
871 }
872
873 internal_assert(op->else_case.defined())
874 << "PartitionLoops should only introduce if statements with an else branch\n";
875
876 Stmt then_case = mutate(op->then_case);
877 Stmt else_case = mutate(op->else_case);
878
879 if (equal(then_case, else_case)) {
880 // This can happen if the only difference between the
881 // cases was a let statement that we pulled out of the if.
882 return then_case;
883 }
884
885 const Allocate *allocate_a = then_case.as<Allocate>();
886 const Allocate *allocate_b = else_case.as<Allocate>();
887 const For *for_a = then_case.as<For>();
888 const For *for_b = else_case.as<For>();
889 const LetStmt *let_a = then_case.as<LetStmt>();
890 const LetStmt *let_b = else_case.as<LetStmt>();
891 if (allocate_a && allocate_b) {
892 Stmt inner = IfThenElse::make(op->condition, allocate_a->body, allocate_b->body);
893 inner = Allocate::make(allocate_a->name, allocate_a->type,
894 allocate_a->memory_type, allocate_a->extents,
895 allocate_a->condition, inner);
896 return mutate(inner);
897 } else if (let_a && let_b && let_a->name == let_b->name) {
898 string condition_name = unique_name('t');
899 Expr condition = Variable::make(op->condition.type(), condition_name);
900 Stmt inner = IfThenElse::make(condition, let_a->body, let_b->body);
901 inner = LetStmt::make(let_a->name, select(condition, let_a->value, let_b->value), inner);
902 inner = LetStmt::make(condition_name, op->condition, inner);
903 return mutate(inner);
904 } else if (let_a) {
905 string new_name = unique_name(let_a->name);
906 Stmt inner = let_a->body;
907 inner = substitute(let_a->name, Variable::make(let_a->value.type(), new_name), inner);
908 inner = IfThenElse::make(op->condition, inner, else_case);
909 inner = LetStmt::make(new_name, let_a->value, inner);
910 return mutate(inner);
911 } else if (let_b) {
912 string new_name = unique_name(let_b->name);
913 Stmt inner = let_b->body;
914 inner = substitute(let_b->name, Variable::make(let_b->value.type(), new_name), inner);
915 inner = IfThenElse::make(op->condition, then_case, inner);
916 inner = LetStmt::make(new_name, let_b->value, inner);
917 return mutate(inner);
918 } else if (for_a && for_b &&
919 for_a->name == for_b->name &&
920 for_a->min.same_as(for_b->min) &&
921 for_a->extent.same_as(for_b->extent)) {
922 Stmt inner = IfThenElse::make(op->condition, for_a->body, for_b->body);
923 inner = For::make(for_a->name, for_a->min, for_a->extent, for_a->for_type, for_a->device_api, inner);
924 return mutate(inner);
925 } else {
926 internal_error << "Unexpected construct inside if statement: " << Stmt(op) << "\n";
927 return Stmt();
928 }
929 }
930};

932// Expand selects of boolean conditions so that the partitioner can
933// consider them one-at-a-time.
934class ExpandSelects : public IRMutator {
935 using IRMutator::visit;
936
937 bool is_trivial(const Expr &e) {
938 return e.as<Variable>() || is_const(e);
939 }
940
941 Expr visit(const Select *op) override {
942 Expr condition = mutate(op->condition);
943 Expr true_value = mutate(op->true_value);
944 Expr false_value = mutate(op->false_value);
945 if (const Or *o = condition.as<Or>()) {
946 if (is_trivial(true_value)) {
947 return mutate(Select::make(o->a, true_value, Select::make(o->b, true_value, false_value)));
948 } else {
949 string var_name = unique_name('t');
950 Expr var = Variable::make(true_value.type(), var_name);
951 Expr expr = mutate(Select::make(o->a, var, Select::make(o->b, var, false_value)));
952 return Let::make(var_name, true_value, expr);
953 }
954 } else if (const And *a = condition.as<And>()) {
955 if (is_trivial(false_value)) {
956 return mutate(Select::make(a->a, Select::make(a->b, true_value, false_value), false_value));
957 } else {
958 string var_name = unique_name('t');
959 Expr var = Variable::make(false_value.type(), var_name);
960 Expr expr = mutate(Select::make(a->a, Select::make(a->b, true_value, var), var));
961 return Let::make(var_name, false_value, expr);
962 }
963 } else if (const Not *n = condition.as<Not>()) {
964 return mutate(Select::make(n->a, false_value, true_value));
965 } else if (condition.same_as(op->condition) &&
966 true_value.same_as(op->true_value) &&
967 false_value.same_as(op->false_value)) {
968 return op;
969 } else {
970 return Select::make(condition, true_value, false_value);
971 }
972 }
973};

975// Collapse selects back together
976class CollapseSelects : public IRMutator {
977 using IRMutator::visit;
978
979 Expr visit(const Select *op) override {
980 const Select *t = op->true_value.as<Select>();
981 const Select *f = op->false_value.as<Select>();
982
983 if (t && equal(t->false_value, op->false_value)) {
984 // select(a, select(b, t, f), f) -> select(a && b, t, f)
985 return mutate(select(op->condition && t->condition, t->true_value, op->false_value));
986 } else if (f && equal(op->true_value, f->true_value)) {
987 // select(a, t, select(b, t, f)) -> select(a || b, t, f)
988 return mutate(select(op->condition || f->condition, op->true_value, f->false_value));
989 } else {
990 return IRMutator::visit(op);
991 }
992 }
