64struct PipelineContents {
65 mutable RefCount ref_count;
66
67 // Cached lowered stmt
68 Module module;
69
70 // Name of the generated function
71 string name;
72
73 // Cached jit-compiled code
74 JITModule jit_module;
75 Target jit_target;
76
77 // Cached compiled JavaScript and/or wasm if defined */
78 WasmModule wasm_module;
79
80 /** Clear all cached state */
81 void invalidate_cache() {
82 module = Module("", Target());
83 jit_module = JITModule();
84 jit_target = Target();
85 inferred_args.clear();
86 wasm_module = WasmModule();
87 }
88
89 // The outputs
90 vector<Function> outputs;
91
92 // JIT custom overrides
93 JITHandlers jit_handlers;
94
95 /** The user context that's used when jitting. This is not
96 * settable by user code, but is reserved for internal use. Note
97 * that this is an Argument + Parameter (rather than a
98 * Param<void*>) so that we can exclude it from the
99 * ObjectInstanceRegistry. */
100 InferredArgument user_context_arg;
101
102 /** A set of custom passes to use when lowering this Func. */
103 vector<CustomLoweringPass> custom_lowering_passes;
104
105 /** The inferred arguments. Also the arguments to the main
106 * function in the jit_module above. The two must be updated
107 * together. */
108 vector<InferredArgument> inferred_args;
109
110 /** List of C funtions and Funcs to satisfy HalideExtern* and
111 * define_extern calls. */
112 std::map<std::string, JITExtern> jit_externs;
113
114 std::vector<Stmt> requirements;
115
116 bool trace_pipeline = false;
117
118 PipelineContents()
119 : module("", Target()) {
120 user_context_arg.arg = Argument("__user_context", Argument::InputScalar, type_of<const void *>(), 0, ArgumentEstimates{});
121 user_context_arg.param = Parameter(Handle(), false, 0, "__user_context");
122 }
123
124 ~PipelineContents() {
125 clear_custom_lowering_passes();
126 }
127
128 void clear_custom_lowering_passes() {
129 invalidate_cache();
130 for (size_t i = 0; i < custom_lowering_passes.size(); i++) {
131 if (custom_lowering_passes[i].deleter) {
132 custom_lowering_passes[i].deleter();
133 }
134 }
135 custom_lowering_passes.clear();
136 }

242Func Pipeline::get_func(size_t index) {
243 // Compute an environment
244 std::map<string, Function> env;
245 for (const Function &f : contents->outputs) {
246 std::map<string, Function> more_funcs = find_transitive_calls(f);
247 env.insert(more_funcs.begin(), more_funcs.end());
248 }
249 // Compute a topological order
250 vector<string> order = topological_order(contents->outputs, env);
251
252 user_assert(index < order.size())
253 << "Index value passed is " << index << "; however, there are only "
254 << order.size() << " functions in the pipeline.\n";
255 return Func(env.find(order[index])->second);
256}

369vector<Argument> Pipeline::infer_arguments(const Stmt &body) {
370 Stmt s = body;
371 if (!contents->requirements.empty()) {
372 s = Block::make(contents->requirements);
373 if (body.defined()) {
374 s = Block::make(s, body);
375 }
376 }
377 contents->inferred_args = ::infer_arguments(s, contents->outputs);
378
379 // Add the user context argument if it's not already there, or hook up our user context
380 // Parameter to any existing one.
381 bool has_user_context = false;
382 for (auto &arg : contents->inferred_args) {
383 if (arg.arg.name == contents->user_context_arg.arg.name) {
384 arg = contents->user_context_arg;
385 has_user_context = true;
386 }
387 }
388 if (!has_user_context) {
389 contents->inferred_args.push_back(contents->user_context_arg);
390 }
391
392 // Return the inferred argument types, minus any constant images
393 // (we'll embed those in the binary by default), and minus the user_context arg.
394 vector<Argument> result;
395 for (const InferredArgument &arg : contents->inferred_args) {
396 debug(2) << "Inferred argument: " << arg.arg.type << " " << arg.arg.name << "\n";
397 if (!arg.buffer.defined() &&
398 arg.arg.name != contents->user_context_arg.arg.name) {
399 result.push_back(arg.arg);
400 }
401 }
402
403 return result;
404}

406class FindExterns : public IRGraphVisitor {
407 using IRGraphVisitor::visit;
408
409 void visit(const Call *op) override {
410 IRGraphVisitor::visit(op);
411
412 if ((op->call_type == Call::Extern || op->call_type == Call::PureExtern) && externs.count(op->name) == 0) {
413 void *address = get_symbol_address(op->name.c_str());
414 if (address == nullptr && !starts_with(op->name, "_")) {
415 std::string underscored_name = "_" + op->name;
416 address = get_symbol_address(underscored_name.c_str());
417 }
418 if (address != nullptr) {
419 // TODO: here and below for arguments, we force types to scalar,
420 // which means this code cannot support functions which actually do
421 // take vectors. But generally the function is actually scalar and
422 // call sites which use vectors will have to be scalarized into a
423 // separate call per lane. Not sure there is anywhere to get
424 // information to make a distinction in the current design.
425 std::vector<Type> arg_types;
426 if (function_takes_user_context(op->name)) {
427 arg_types.push_back(type_of<void *>());
428 }
429 for (const Expr &e : op->args) {
430 arg_types.push_back(e.type().element_of());
431 }
432 bool is_void_return = op->type.bits() == 0 || op->name == "halide_print";
433 ExternSignature sig(is_void_return ? Type() : op->type.element_of(), is_void_return, arg_types);
434 ExternCFunction f(address, sig);
435 JITExtern jit_extern(f);
436 debug(2) << "FindExterns adds: " << op->name << "\n";
437 externs.emplace(op->name, jit_extern);
438 }
439 }
440 }
441
442public:
443 FindExterns(std::map<std::string, JITExtern> &externs)
444 : externs(externs) {
445 }
446
447 std::map<std::string, JITExtern> &externs;
448};

454Module Pipeline::compile_to_module(const vector<Argument> &args,
455 const string &fn_name,
456 const Target &target,
457 const LinkageType linkage_type) {
458 user_assert(defined()) << "Can't compile undefined Pipeline.\n";
459
460 for (const Function &f : contents->outputs) {
461 user_assert(f.has_pure_definition() || f.has_extern_definition())
462 << "Can't compile Pipeline with undefined output Func: " << f.name() << ".\n";
463 }
464
465 string new_fn_name(fn_name);
466 if (new_fn_name.empty()) {
467 new_fn_name = generate_function_name();
468 }
469 internal_assert(!new_fn_name.empty()) << "new_fn_name cannot be empty\n";
470 // TODO: Assert that the function name is legal
471
472 vector<Argument> lowering_args(args);
473
474 // If the target specifies user context but it's not in the args
475 // vector, add it at the start (the jit path puts it in there
476 // explicitly).
477 const bool requires_user_context = target.has_feature(Target::UserContext);
478 bool has_user_context = false;
479 for (const Argument &arg : lowering_args) {
480 if (arg.name == contents->user_context_arg.arg.name) {
481 has_user_context = true;
482 }
483 }
484 if (requires_user_context && !has_user_context) {
485 lowering_args.insert(lowering_args.begin(), contents->user_context_arg.arg);
486 }
487
488 const Module &old_module = contents->module;
489
490 bool same_compile = !old_module.functions().empty() && old_module.target() == target;
491 // Either generated name or one of the LoweredFuncs in the existing module has the same name.
492 same_compile = same_compile && fn_name.empty();
493 bool found_name = false;
494 for (const auto &lf : old_module.functions()) {
495 if (lf.name == fn_name) {
496 found_name = true;
497 break;
498 }
499 }
500 same_compile = same_compile && found_name;
501 // Number of args + number of outputs is the same as total args in existing LoweredFunc
502 same_compile = same_compile && (lowering_args.size() + outputs().size()) == old_module.functions().front().args.size();
503 // The initial args are the same.
504 same_compile = same_compile && std::equal(lowering_args.begin(), lowering_args.end(), old_module.functions().front().args.begin());
505 // Linkage is the same.
506 same_compile = same_compile && old_module.functions().front().linkage == linkage_type;
507 // The outputs of a Pipeline cannot change, so no need to test them.
508
509 if (same_compile) {
510 // We can avoid relowering and just reuse the existing module.
511 debug(2) << "Reusing old module\n";
512 } else {
513 vector<IRMutator *> custom_passes;
514 for (const CustomLoweringPass &p : contents->custom_lowering_passes) {
515 custom_passes.push_back(p.pass);
516 }
517
518 contents->module = lower(contents->outputs, new_fn_name, target, lowering_args,
519 linkage_type, contents->requirements, contents->trace_pipeline,
520 custom_passes);
521 }
522
523 return contents->module;
524}

526std::string Pipeline::generate_function_name() const {
527 user_assert(defined()) << "Pipeline is undefined\n";
528 // Come up with a name for a generated function

541Target Pipeline::get_compiled_jit_target() const {
542 const bool has_wasm = contents->wasm_module.contents.defined();
543 const bool has_native = contents->jit_module.compiled();
544 if (contents->jit_target.arch == Target::WebAssembly) {
545 internal_assert(has_wasm && !has_native);
546 } else if (!contents->jit_target.has_unknowns()) {

551 return contents->jit_target;

554void Pipeline::compile_jit(const Target &target_arg) {
555 user_assert(defined()) << "Pipeline is undefined\n";
556 user_assert(!target_arg.has_unknowns()) << "Cannot compile_jit() for target '" << target_arg << "'\n";
557
558 Target target(target_arg);
559 target.set_feature(Target::JIT);
560 target.set_feature(Target::UserContext);
561
562 // If we're re-jitting for the same target, we can just keep the old jit module.
563 if (get_compiled_jit_target() == target) {
564 debug(2) << "Reusing old jit module compiled for :\n"
565 << target << "\n";
566 return;
567 }
568
569 debug(2) << "jit-compiling for: " << target_arg << "\n";
570
571 // Clear all cached info in case there is an error.
572 contents->invalidate_cache();
573
574 contents->jit_target = target;
575
576 // Infer an arguments vector
577 infer_arguments();
578
579 // Don't actually use the return value - it embeds all constant
580 // images and we don't want to do that when jitting. Instead
581 // use the vector of parameters found to make a more complete
582 // arguments list.
583 vector<Argument> args;
584 for (const InferredArgument &arg : contents->inferred_args) {
585 args.push_back(arg.arg);
586 }
587
588 // Come up with a name for the generated function
589 string name = generate_function_name();
590
591 // Compile to a module and also compile any submodules.
592 Module module = compile_to_module(args, name, target).resolve_submodules();
593
594 std::map<std::string, JITExtern> lowered_externs = contents->jit_externs;
595
596 if (target.arch == Target::WebAssembly) {
597 FindExterns find_externs(lowered_externs);
598 for (const LoweredFunc &f : contents->module.functions()) {
599 f.body.accept(&find_externs);
600 }
601 if (debug::debug_level() >= 1) {
602 for (const auto &p : lowered_externs) {
603 debug(1) << "Found extern: " << p.first << "\n";
604 }
605 }
606
607 vector<Argument> args_and_outputs = args;
608 for (auto &out : contents->outputs) {
609 for (Type t : out.output_types()) {
610 args_and_outputs.emplace_back(out.name(), Argument::OutputBuffer, t, out.dimensions(), ArgumentEstimates{});
611 }
612 }
613
614 contents->wasm_module = WasmModule::compile(
615 module,
616 args_and_outputs,
617 contents->module.name(),
618 lowered_externs,
619 make_externs_jit_module(target, lowered_externs));
620 return;
621 }
622
623 auto f = module.get_function_by_name(name);
624
625 // Compile to jit module
626 JITModule jit_module(module, f, make_externs_jit_module(target_arg, lowered_externs));
627
628 // Dump bitcode to a file if the environment variable
629 // HL_GENBITCODE is defined to a nonzero value.
630 if (atoi(get_env_variable("HL_GENBITCODE").c_str())) {
631 string program_name = running_program_name();
632 if (program_name.empty()) {
633 program_name = "unknown" + unique_name('_').substr(1);
634 }
635 string file_name = program_name + "_" + name + "_" + unique_name('g').substr(1) + ".bc";
636 debug(4) << "Saving bitcode to: " << file_name << "\n";
637 module.compile({{Output::bitcode, file_name}});
638 }
639
640 contents->jit_module = jit_module;
641}

710Realization Pipeline::realize(vector<int32_t> sizes, const Target &target,
711 const ParamMap &param_map) {
712 user_assert(defined()) << "Pipeline is undefined\n";
713 vector<Buffer<>> bufs;
714 for (auto &out : contents->outputs) {
715 user_assert(out.has_pure_definition() || out.has_extern_definition()) << "Can't realize Pipeline with undefined output Func: " << out.name() << ".\n";
716 for (Type t : out.output_types()) {
717 bufs.emplace_back(t, nullptr, sizes);
718 }
719 }
720 Realization r(bufs);
721 // Do an output bounds query if we can. Otherwise just assume the
722 // output size is good.
723 if (!target.has_feature(Target::NoBoundsQuery)) {
724 realize(r, target, param_map);
725 }
726 for (size_t i = 0; i < r.size(); i++) {
727 r[i].allocate();
728 }
729 // Do the actual computation
730 realize(r, target, param_map);
731
732 // Crop back to the requested size if necessary
733 bool needs_crop = false;
734 vector<std::pair<int32_t, int32_t>> crop;
735 if (!target.has_feature(Target::NoBoundsQuery)) {
736 crop.resize(sizes.size());
737 for (size_t d = 0; d < sizes.size(); d++) {
738 needs_crop |= ((r[0].dim(d).extent() != sizes[d]) ||
739 (r[0].dim(d).min() != 0));
740 crop[d].first = 0;
741 crop[d].second = sizes[d];
742 }
743 }
744 for (size_t i = 0; i < r.size(); i++) {
745 if (needs_crop) {
746 r[i].crop(crop);
747 }
748 r[i].copy_to_host();
749 }
750 return r;
751}

808struct ErrorBuffer {
809 enum { MaxBufSize = 4096 };
810 char buf[MaxBufSize];
811 std::atomic<size_t> end{0};
812
813 void concat(const char *message) {
814 size_t len = strlen(message);
815
816 if (len && message[len - 1] != '\n') {
817 // Claim some extra space for a newline.
818 len++;
819 }
820
821 // Atomically claim some space in the buffer
822 size_t old_end = end.fetch_add(len);
823
824 if (old_end + len >= MaxBufSize - 1) {
825 // Out of space
826 return;
827 }
828
829 for (size_t i = 0; i < len - 1; i++) {
830 buf[old_end + i] = message[i];
831 }
832 if (buf[old_end + len - 2] != '\n') {
833 buf[old_end + len - 1] = '\n';
834 }
835 }
836
837 std::string str() const {
838 return std::string(buf, end);
839 }
840
841 static void handler(void *ctx, const char *message) {
842 if (ctx) {
843 JITUserContext *ctx1 = (JITUserContext *)ctx;
844 ErrorBuffer *buf = (ErrorBuffer *)ctx1->user_context;
845 buf->concat(message);
846 }
847 }
848};

854
855 JITFuncCallContext(const JITHandlers &handlers) {

868 << "custom_malloc: " << (void *)jit_context.handlers.custom_malloc << "\n"

928// Make a vector of void *'s to pass to the jit call using the
929// currently bound value for all of the params and image
930// params.
931void Pipeline::prepare_jit_call_arguments(RealizationArg &outputs, const Target &target,
932 const ParamMap &param_map, void *user_context,
933 bool is_bounds_inference, JITCallArgs &args_result) {
934 user_assert(defined()) << "Can't realize an undefined Pipeline\n";
935
936 size_t total_outputs = 0;
937 for (const Func &out : this->outputs()) {
938 total_outputs += out.outputs();
939 }
940 user_assert(outputs.size() == total_outputs)
941 << "Realization requires " << outputs.size() << " output(s) but pipeline produces "
942 << total_outputs << " result(s).\n";
943
944 JITModule &compiled_module = contents->jit_module;
945 internal_assert(compiled_module.argv_function() ||
946 contents->wasm_module.contents.defined());
947
948 const bool no_param_map = &param_map == &ParamMap::empty_map();
949
950 // Come up with the void * arguments to pass to the argv function
951 size_t arg_index = 0;
952 for (const InferredArgument &arg : contents->inferred_args) {
953 if (arg.param.defined()) {
954 if (arg.param.same_as(contents->user_context_arg.param)) {
955 args_result.store[arg_index++] = user_context;
956 } else {
957 Buffer<> *buf_out_param = nullptr;
958 const Parameter &p = no_param_map ? arg.param : param_map.map(arg.param, buf_out_param);
959 user_assert(is_bounds_inference || !buf_out_param)
960 << "Cannot pass Buffer<> pointers in parameters map to a compute call.\n";
961
962 if (p.is_buffer()) {
963 // ImageParam arg
964 Buffer<> buf = p.buffer();
965 if (buf.defined()) {
966 args_result.store[arg_index++] = p.raw_buffer();
967 } else {
968 // Unbound
969 args_result.store[arg_index++] = nullptr;
970 }
971 debug(2) << "JIT input ImageParam argument ";
972 } else {
973 args_result.store[arg_index++] = p.scalar_address();
974 debug(2) << "JIT input scalar argument ";
975 }
976 }
977 } else {
978 debug(2) << "JIT input Image argument ";
979 internal_assert(arg.buffer.defined());
980 args_result.store[arg_index++] = arg.buffer.raw_buffer();
981 }
982 const void *ptr = args_result.store[arg_index - 1];
983 debug(2) << arg.arg.name << " @ " << ptr << "\n";
984 }
985
986 // Then the outputs
987 if (outputs.r) {
988 for (size_t i = 0; i < outputs.r->size(); i++) {
989 const halide_buffer_t *buf = (*outputs.r)[i].raw_buffer();
990 args_result.store[arg_index++] = buf;
991 debug(2) << "JIT output buffer @ " << (const void *)buf << ", " << (const void *)buf->host << "\n";
992 }
993 } else if (outputs.buf) {
994 args_result.store[arg_index++] = outputs.buf;
995 debug(2) << "JIT output buffer @ " << (const void *)outputs.buf << ", " << (const void *)outputs.buf->host << "\n";
996 } else {
997 for (const Buffer<> &buffer : *outputs.buffer_list) {
998 const halide_buffer_t *buf = buffer.raw_buffer();
999 args_result.store[arg_index++] = buf;
1000 debug(2) << "JIT output buffer @ " << (const void *)buf << ", " << (const void *)buf->host << "\n";
1001 }
1002 }
1003}

1005std::vector<JITModule>
1006Pipeline::make_externs_jit_module(const Target &target,
1007 std::map<std::string, JITExtern> &externs_in_out) {
1008 std::vector<JITModule> result;
1009
1010 // Externs that are Funcs get their own JITModule. All standalone functions are
1011 // held in a single JITModule at the end of the list (if there are any).
1012 JITModule free_standing_jit_externs;
1013 for (std::map<std::string, JITExtern>::iterator iter = externs_in_out.begin();
1014 iter != externs_in_out.end();
1015 iter++) {
1016 Pipeline pipeline = iter->second.pipeline();
1017 if (pipeline.defined()) {
1018 PipelineContents &pipeline_contents(*pipeline.contents);
1019
1020 // Ensure that the pipeline is compiled.
1021 pipeline.compile_jit(target);
1022
1023 free_standing_jit_externs.add_dependency(pipeline_contents.jit_module);
1024 free_standing_jit_externs.add_symbol_for_export(iter->first, pipeline_contents.jit_module.entrypoint_symbol());
1025 void *address = pipeline_contents.jit_module.entrypoint_symbol().address;
1026 std::vector<Type> arg_types;
1027 // Add the arguments to the compiled pipeline
1028 for (const InferredArgument &arg : pipeline_contents.inferred_args) {
1029 // TODO: it's not clear whether arg.arg.type is correct for
1030 // the arg.is_buffer() case (AFAIK, is_buffer()==true isn't possible
1031 // in current trunk Halide, but may be in some side branches that
1032 // have not yet landed, e.g. JavaScript). Forcing it to be
1033 // the correct type here, just in case.
1034 arg_types.push_back(arg.arg.is_buffer() ? type_of<struct halide_buffer_t *>() : arg.arg.type);
1035 }
1036 // Add the outputs of the pipeline
1037 for (size_t i = 0; i < pipeline_contents.outputs.size(); i++) {
1038 arg_types.push_back(type_of<struct halide_buffer_t *>());
1039 }
1040 ExternSignature signature(Int(32), false, arg_types);
1041 iter->second = JITExtern(ExternCFunction(address, signature));
1042 } else {
1043 free_standing_jit_externs.add_extern_for_export(iter->first, iter->second.extern_c_function());
1044 }
1045 }
1046 if (free_standing_jit_externs.compiled() || !free_standing_jit_externs.exports().empty()) {
1047 result.push_back(free_standing_jit_externs);
1048 }
1049 return result;
1050}

1068void Pipeline::realize(RealizationArg outputs, const Target &t,
1069 const ParamMap &param_map) {
1070 Target target = t;
1071 user_assert(defined()) << "Can't realize an undefined Pipeline\n";
1072
1073 debug(2) << "Realizing Pipeline for " << target << "\n";
1074
1075 if (target.has_unknowns()) {
1076 // If we've already jit-compiled for a specific target, use that.
1077 target = get_compiled_jit_target();
1078 if (target.has_unknowns()) {
1079 // Otherwise get the target from the environment
1080 target = get_jit_target_from_environment();
1081 }
1082 }
1083
1084 // We need to make a context for calling the jitted function to
1085 // carry the the set of custom handlers. Here's how handlers get
1086 // called when running jitted code:
1087
1088 // There's a single shared module that includes runtime code like
1089 // posix_error_handler.cpp. This module is created the first time
1090 // you JIT something and is reused for all subsequent runs of
1091 // jitted code for any pipeline with the same target.
1092
1093 // To handle events like printing, tracing, or errors, the jitted
1094 // code calls things like halide_error or halide_print in the
1095 // shared runtime, which in turn call global function pointer
1096 // variables in the shared runtime (e.g. halide_error_handler,
1097 // halide_custom_print). When the shared module is created, we set
1098 // those variables to point to the global handlers in
1099 // JITModule.cpp (e.g. error_handler_handler, print_handler).
1100
1101 // Those global handlers use the user_context passed in to call
1102 // the right handler for this particular pipeline run. The
1103 // user_context is just a pointer to a JITUserContext, which is a
1104 // member of the JITFuncCallContext which we will declare now:
1105
1106 // Ensure the module is compiled.
1107 compile_jit(target);
1108
1109 // This has to happen after a runtime has been compiled in compile_jit.
1110 JITFuncCallContext jit_context(jit_handlers());
1111 void *user_context_storage = &jit_context.jit_context;
1112
1113 JITCallArgs args(contents->inferred_args.size() + outputs.size());
1114 prepare_jit_call_arguments(outputs, target, param_map,
1115 &user_context_storage, false, args);
1116
1117 // The handlers in the jit_context default to the default handlers
1118 // in the runtime of the shared module (e.g. halide_print_impl,
1119 // default_trace). As an example, here's what happens with a
1120 // halide_print call:
1121
1122 // 1) Before the pipeline runs, when the single shared runtime
1123 // module is created, halide_custom_print in posix_print.cpp is
1124 // set to print_handler in JITModule.cpp
1125
1126 // 2) When the jitted module is compiled, we tell llvm to resolve
1127 // calls to halide_print to the halide_print in the shared module
1128 // we made.
1129
1130 // 3) The user calls realize(), and the jitted code calls
1131 // halide_print in the shared runtime.
1132
1133 // 4) halide_print calls the function pointer halide_custom_print,
1134 // which is print_handler in JITModule.cpp
1135
1136 // 5) print_handler casts the user_context to a JITUserContext,
1137 // then calls the function pointer member handlers.custom_print,
1138 // which is either halide_print_impl in the runtime, or some other
1139 // function set by Pipeline::set_custom_print.
1140
1141 // Errors are slightly different, in that we always override the
1142 // default when jitting. We instead use ErrorBuffer::handler
1143 // above (this was set in jit_context's constructor). When
1144 // jit-compiled code encounters an error, it calls this handler,
1145 // which just records the fact there was an error and what the
1146 // message was, then returns back into jitted code. The jitted
1147 // code cleans up and returns early with an exit code. We record
1148 // this exit status below, then pass it to jit_context.finalize at
1149 // the end of this function. If it's non-zero,
1150 // jit_context.finalize passes the recorded error message to
1151 // halide_runtime_error, which either calls abort() or throws an
1152 // exception.
1153
1154 debug(2) << "Calling jitted function\n";
1155 int exit_status = call_jit_code(target, args);
1156 debug(2) << "Back from jitted function. Exit status was " << exit_status << "\n";
1157
1158 // If we're profiling, report runtimes and reset profiler stats.
1159 if (target.has_feature(Target::Profile)) {
1160 JITModule::Symbol report_sym =
1161 contents->jit_module.find_symbol_by_name("halide_profiler_report");
1162 JITModule::Symbol reset_sym =
1163 contents->jit_module.find_symbol_by_name("halide_profiler_reset");
1164 if (report_sym.address && reset_sym.address) {
1165 void *uc = &jit_context.jit_context;
1166 void (*report_fn_ptr)(void *) = (void (*)(void *))(report_sym.address);
1167 report_fn_ptr(uc);
1168
1169 void (*reset_fn_ptr)() = (void (*)())(reset_sym.address);
1170 reset_fn_ptr();
1171 }
1172 }
1173
1174 jit_context.finalize(exit_status);
1175}

1177void Pipeline::infer_input_bounds(RealizationArg outputs, const Target &target, const ParamMap &param_map) {
1178 user_assert(!target.has_feature(Target::NoBoundsQuery)) << "You may not call infer_input_bounds() with Target::NoBoundsQuery set.";
1179 compile_jit(target);
1180
1181 // This has to happen after a runtime has been compiled in compile_jit.
1182 JITFuncCallContext jit_context(jit_handlers());
1183 void *user_context_storage = &jit_context.jit_context;
1184
1185 size_t args_size = contents->inferred_args.size() + outputs.size();
1186 JITCallArgs args(args_size);
1187 prepare_jit_call_arguments(outputs, contents->jit_target, param_map,
1188 &user_context_storage, true, args);
1189
1190 struct TrackedBuffer {
1191 // The query buffer, and a backup to check for changes. We
1192 // want wrappers around actual halide_buffer_ts so that we can copy
1193 // the metadata, not shared pointers to a single buffer, so
1194 // it's simpler to use the runtime buffer class.
1195 Runtime::Buffer<> query, orig;
1196 };
1197 vector<TrackedBuffer> tracked_buffers(args_size);
1198
1199 vector<size_t> query_indices;
1200 for (size_t i = 0; i < contents->inferred_args.size(); i++) {
1201 if (args.store[i] == nullptr) {
1202 query_indices.push_back(i);
1203 InferredArgument ia = contents->inferred_args[i];
1204 internal_assert(ia.param.defined() && ia.param.is_buffer());
1205 // Make some empty Buffers of the right dimensionality
1206 vector<int> initial_shape(ia.param.dimensions(), 0);
1207 tracked_buffers[i].query = Runtime::Buffer<>(ia.param.type(), nullptr, initial_shape);
1208 tracked_buffers[i].orig = Runtime::Buffer<>(ia.param.type(), nullptr, initial_shape);
1209 args.store[i] = tracked_buffers[i].query.raw_buffer();
1210 }
1211 }
1212
1213 // No need to query if all the inputs are bound already.
1214 if (query_indices.empty()) {
1215 debug(2) << "All inputs are bound. No need for bounds inference\n";
1216 return;
1217 }
1218
1219 int iter = 0;
1220 const int max_iters = 16;
1221 for (iter = 0; iter < max_iters; iter++) {
1222 // Make a copy of the buffers that might be mutated
1223 for (TrackedBuffer &tb : tracked_buffers) {
1224 // Make a copy of the buffer sizes, etc.
1225 tb.orig = tb.query;
1226 }
1227
1228 Internal::debug(2) << "Calling jitted function\n";
1229 int exit_status = call_jit_code(contents->jit_target, args);
1230 jit_context.report_if_error(exit_status);
1231 Internal::debug(2) << "Back from jitted function\n";
1232 bool changed = false;
1233
1234 // Check if there were any changes
1235 for (TrackedBuffer &tb : tracked_buffers) {
1236 for (int i = 0; i < tb.query.dimensions(); i++) {
1237 if (tb.query.dim(i).min() != tb.orig.dim(i).min() ||
1238 tb.query.dim(i).extent() != tb.orig.dim(i).extent() ||
1239 tb.query.dim(i).stride() != tb.orig.dim(i).stride()) {
1240 changed = true;
1241 }
1242 }
1243 }
1244 if (!changed) {
1245 break;
1246 }
1247 }
1248
1249 jit_context.finalize(0);
1250
1251 user_assert(iter < max_iters)
1252 << "Inferring input bounds on Pipeline"
1253 << " didn't converge after " << max_iters
1254 << " iterations. There may be unsatisfiable constraints\n";
1255
1256 debug(2) << "Bounds inference converged after " << iter << " iterations\n";
1257
1258 // Now allocate the resulting buffers
1259 for (size_t i : query_indices) {
1260 InferredArgument ia = contents->inferred_args[i];
1261 Buffer<> *buf_out_param = nullptr;
1262 Parameter &p = param_map.map(ia.param, buf_out_param);
1263
1264 if (&p != &ia.param) {
1265 user_assert(buf_out_param != nullptr) << "Output Buffer<> arguments to infer_input_bounds in parameters map must be passed as pointers.\n";
1266 }
1267 internal_assert(!p.buffer().defined());
1268
1269 // Allocate enough memory with the right type and dimensionality.
1270 tracked_buffers[i].query.allocate();
1271
1272 if (buf_out_param != nullptr) {
1273 *buf_out_param = Buffer<>(*tracked_buffers[i].query.raw_buffer());
1274 } else {
1275 // Bind this parameter to this buffer, giving away the
1276 // buffer. The user retrieves it via ImageParam::get.
1277 p.set_buffer(Buffer<>(std::move(tracked_buffers[i].query)));
1278 }
1279 }
1280}

1285 user_assert(defined()) << "Can't infer input bounds on an undefined Pipeline.\n";

1288 bufs.emplace_back(t, sizes);
1289 }

1291 infer_input_bounds(r, target, param_map);
