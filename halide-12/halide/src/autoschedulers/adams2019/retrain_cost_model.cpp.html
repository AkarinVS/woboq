84 std::vector<float> parse_floats(const std::string &s) {
85 const char *c = s.c_str();
86 std::vector<float> v;
87 while (isspace(*c)) {
88 ++c;
89 }
90 while (*c) {
91 string f;
92 while (*c && !isspace(*c)) {
93 f += *c++;
94 }
95 v.push_back(std::atof(f.c_str()));
96 while (isspace(*c)) {
97 ++c;
98 }
99 }
100 return v;
101 }

124uint64_t hash_floats(uint64_t h, const float *begin, const float *end) {
125 while (begin != end) {
126 uint32_t bits = *((const uint32_t *)begin);
127 // From boost
128 h ^= (bits + 0x9e3779b9 + (h << 6) + (h >> 2));
129 begin++;
130 }
131 return h;
132}

134bool ends_with(const string &str, const string &suffix) {
135 if (str.size() < suffix.size()) {
136 return false;
137 }
138 size_t off = str.size() - suffix.size();
139 for (size_t i = 0; i < suffix.size(); i++) {
140 if (str[off + i] != suffix[i]) {
141 return false;
142 }
143 }
144 return true;
145}

147string leaf(const string &path) {
148 size_t slash_pos = path.rfind('/');
149#ifdef _WIN32
150 if (slash_pos == string::npos) {
151 // Windows is a thing
152 slash_pos = path.rfind('\\');
153 }
154#endif
155 if (slash_pos != string::npos) {
156 return path.substr(slash_pos + 1);
157 } else {
158 return path;
159 }

162// Load all the samples, reading filenames from stdin
163map<int, PipelineSample> load_samples(const Flags &flags) {
164 map<int, PipelineSample> result;
165 vector<float> scratch(10 * 1024 * 1024);
166
167 int best = -1;
168 float best_runtime = 1e20f;
169 string best_path;
170
171 size_t num_read = 0, num_unique = 0;
172 while (!std::cin.eof()) {
173 string s;
174 std::cin >> s;
175 if (s.empty()) {
176 continue;
177 }
178 if (!ends_with(s, ".sample")) {
179 std::cout << "Skipping file: " << s << "\n";
180 continue;
181 }
182 std::ifstream file(s);
183 file.read((char *)(scratch.data()), scratch.size() * sizeof(float));
184 const size_t floats_read = file.gcount() / sizeof(float);
185 const size_t num_features = floats_read - 3;
186 const size_t features_per_stage = head2_w + (head1_w + 1) * head1_h;
187 file.close();
188 // Note we do not check file.fail(). The various failure cases
189 // are handled below by checking the number of floats read. We
190 // expect truncated files if the benchmarking or
191 // autoscheduling procedure crashes and want to filter them
192 // out with a warning.
193
194 if (floats_read == scratch.size()) {
195 std::cout << "Too-large sample: " << s << " " << floats_read << "\n";
196 continue;
197 }
198 if (num_features % features_per_stage != 0) {
199 std::cout << "Truncated sample: " << s << " " << floats_read << "\n";
200 continue;
201 }
202 const size_t num_stages = num_features / features_per_stage;
203
204 const float runtime = scratch[num_features];
205 if (runtime > 100000) { // Don't try to predict runtime over 100s
206 std::cout << "Implausible runtime in ms: " << runtime << "\n";
207 continue;
208 }
209 // std::cout << "Runtime: " << runtime << "\n";
210
211 int pipeline_id = *((int32_t *)(&scratch[num_features + 1]));
212 const int schedule_id = *((int32_t *)(&scratch[num_features + 2]));
213
214 if (runtime < best_runtime) {
215 best_runtime = runtime;
216 best = schedule_id;
217 best_path = s;
218 }
219
220 PipelineSample &ps = result[pipeline_id];
221
222 if (ps.pipeline_features.data() == nullptr) {
223 ps.pipeline_id = pipeline_id;
224 ps.num_stages = (int)num_stages;
225 ps.pipeline_features = Buffer<float>(head1_w, head1_h, num_stages);
226 ps.fastest_runtime = 1e30f;
227 for (size_t i = 0; i < num_stages; i++) {
228 for (int x = 0; x < head1_w; x++) {
229 for (int y = 0; y < head1_h; y++) {
230 float f = scratch[i * features_per_stage + (x + 1) * 7 + y + head2_w];
231 if (f < 0 || std::isnan(f)) {
232 std::cout << "Negative or NaN pipeline feature: " << x << " " << y << " " << i << " " << f << "\n";
233 }
234 ps.pipeline_features(x, y, i) = f;
235 }
236 }
237 }
238
239 ps.pipeline_hash = hash_floats(0, ps.pipeline_features.begin(), ps.pipeline_features.end());
240 }
241
242 uint64_t schedule_hash = 0;
243 for (size_t i = 0; i < num_stages; i++) {
244 schedule_hash =
245 hash_floats(schedule_hash,
246 &scratch[i * features_per_stage],
247 &scratch[i * features_per_stage + head2_w]);
248 }
249
250 auto it = ps.schedules.find(schedule_hash);
251 if (it != ps.schedules.end()) {
252 // Keep the smallest runtime at the front
253 float best = it->second.runtimes[0];
254 if (runtime < best) {
255 it->second.runtimes.push_back(best);
256 it->second.runtimes[0] = runtime;
257 it->second.filename = s;
258 } else {
259 it->second.runtimes.push_back(runtime);
260 }
261 if (runtime < ps.fastest_runtime) {
262 ps.fastest_runtime = runtime;
263 ps.fastest_schedule_hash = schedule_hash;
264 }
265 } else {
266 Sample sample;
267 sample.filename = s;
268 sample.runtimes.push_back(runtime);
269 for (int i = 0; i < kModels; i++) {
270 sample.prediction[i] = 0.0;
271 }
272 sample.schedule_id = schedule_id;
273 sample.schedule_features = Buffer<float>(head2_w, num_stages);
274
275 bool ok = true;
276 for (size_t i = 0; i < num_stages; i++) {
277 for (int x = 0; x < head2_w; x++) {
278 float f = scratch[i * features_per_stage + x];
279 if (f < 0 || f > 1e14 || std::isnan(f)) {
280 std::cout << "Negative or implausibly large schedule feature: " << i << " " << x << " " << f << "\n";
281 // Something must have overflowed
282 ok = false;
283 }
284 sample.schedule_features(x, i) = f;
285 }
286 /*
287 if (sample.schedule_features(0, i) != sample.schedule_features(1, i)) {
288 std::cout << "Rejecting sliding window schedule for now\n";
289 ok = false;
290 }
291 */
292 }
293 if (ok) {
294 if (runtime < ps.fastest_runtime) {
295 ps.fastest_runtime = runtime;
296 ps.fastest_schedule_hash = schedule_hash;
297 }
298 ps.schedules.emplace(schedule_hash, std::move(sample));
299 num_unique++;
300 }
301 }
302 num_read++;
303
304 if (num_read % 10000 == 0) {
305 std::cout << "Samples loaded: " << num_read << " (" << num_unique << " unique)\n";
306 }
307 }
308
309 // Check the noise level
310 for (const auto &pipe : result) {
311 double variance_sum = 0;
312 size_t count = 0;
313 // Compute the weighted average of variances across all samples
314 for (const auto &p : pipe.second.schedules) {
315 if (p.second.runtimes.empty()) {
316 std::cerr << "Empty runtimes for schedule: " << p.first << "\n";
317 abort();
318 }
319 std::cout << "Unique sample: " << leaf(p.second.filename) << " : " << p.second.runtimes[0] << "\n";
320 if (p.second.runtimes.size() > 1) {
321 // Compute variance from samples
322 double mean = 0;
323 for (float f : p.second.runtimes) {
324 mean += f;
325 }
326 mean /= p.second.runtimes.size();
327 double variance = 0;
328 for (float f : p.second.runtimes) {
329 f -= mean;
330 variance += f * f;
331 }
332 variance_sum += variance;
333 count += p.second.runtimes.size() - 1;
334 }
335 }
336 if (count > 0) {
337 double stddev = std::sqrt(variance_sum / count);
338 std::cout << "Noise level: " << stddev << "\n";
339 }
340 }
341
342 std::cout << "Distinct pipelines: " << result.size() << "\n";
343
344 std::ostringstream o;
345 o << "Best runtime is " << best_runtime << " msec, from schedule id " << best << " in file " << best_path << "\n";
346 std::cout << o.str();
347 if (!flags.best_benchmark_path.empty()) {
348 std::ofstream f(flags.best_benchmark_path, std::ios_base::trunc);
349 f << o.str();
350 f.close();
351 assert(!f.fail());
352 }
353 if (!flags.best_schedule_path.empty()) {
354 // best_path points to a .sample file; look for a .schedule.h file in the same dir
355 size_t dot = best_path.rfind('.');
356 assert(dot != string::npos && best_path.substr(dot) == ".sample");
357 string schedule_file = best_path.substr(0, dot) + ".schedule.h";
358 std::ifstream src(schedule_file);
359 std::ofstream dst(flags.best_schedule_path);
360 dst << src.rdbuf();
361 assert(!src.fail());
362 assert(!dst.fail());
363 }
364
365 return result;
366}

370int main(int argc, char **argv) {
371 Flags flags(argc, argv);
372
373 auto samples = load_samples(flags);
374
375 // Iterate through the pipelines
376 vector<std::unique_ptr<DefaultCostModel>> tpp;
377 for (int i = 0; i < kModels; i++) {
378 tpp.emplace_back(make_default_cost_model(flags.initial_weights_path, flags.weights_out_path, flags.randomize_weights));
379 }
380
381 std::cout.setf(std::ios::fixed, std::ios::floatfield);
382 std::cout.precision(4);
383
384 auto seed = time(nullptr);
385 std::mt19937 rng((uint32_t)seed);
386
387 std::cout << "Iterating over " << samples.size() << " samples using seed = " << seed << "\n";
388 decltype(samples) validation_set;
389 uint64_t unique_schedules = 0;
390 if (samples.size() > 16) {
391 for (const auto &p : samples) {
392 unique_schedules += p.second.schedules.size();
393 // Whether or not a pipeline is part of the validation set
394 // can't be a call to rand. It must be a fixed property of a
395 // hash of some aspect of it. This way you don't accidentally
396 // do a training run where a validation set member was in the
397 // training set of a previous run. The id of the fastest
398 // schedule will do as a hash.
399 if ((p.second.pipeline_hash & 7) == 0) {
400 validation_set.insert(p);
401 }
402 }
403
404 for (const auto &p : validation_set) {
405 samples.erase(p.first);
406 }
407 }
408
409 std::cout << "Number of unique schedules: " << unique_schedules << "\n";
410
411 for (float learning_rate : flags.rates) {
412 float loss_sum[kModels] = {0}, loss_sum_counter[kModels] = {0};
413 float correct_ordering_rate_sum[kModels] = {0};
414 float correct_ordering_rate_count[kModels] = {0};
415 float v_correct_ordering_rate_sum[kModels] = {0};
416 float v_correct_ordering_rate_count[kModels] = {0};
417
418 for (int e = 0; e < flags.epochs; e++) {
419 int counter = 0;
420
421 float worst_miss = 0;
422 uint64_t worst_miss_pipeline_id = 0;
423 uint64_t worst_miss_schedule_id = 0;
424
425 struct Inversion {
426 int pipeline_id;
427 string f1, f2;
428 float p1, p2;
429 float r1, r2;
430 float badness = 0;
431 } worst_inversion;
432
433#if defined(_OPENMP)
434#pragma omp parallel for
435#endif
436 for (int model = 0; model < kModels; model++) {
437 for (int train = 0; train < 2; train++) {
438 auto &tp = tpp[model];
439
440 for (auto &p : train ? samples : validation_set) {
441 if (kModels > 1 && rng() & 1) {
442 continue; // If we are training multiple kModels, allow them to diverge.
443 }
444 if (p.second.schedules.size() < 8) {
445 continue;
446 }
447 tp->reset();
448 tp->set_pipeline_features(p.second.pipeline_features, flags.num_cores);
449
450 size_t batch_size = std::min((size_t)1024, p.second.schedules.size());
451
452 size_t fastest_idx = 0;
453 Halide::Runtime::Buffer<float> runtimes(batch_size);
454
455 size_t first = 0;
456 if (p.second.schedules.size() > 1024) {
457 first = rng() % (p.second.schedules.size() - 1024);
458 }
459
460 auto it = p.second.schedules.begin();
461 std::advance(it, first);
462 for (size_t j = 0; j < batch_size; j++) {
463 auto &sched = it->second;
464 Halide::Runtime::Buffer<float> buf;
465 tp->enqueue(p.second.num_stages, &buf, &sched.prediction[model]);
466 runtimes(j) = sched.runtimes[0];
467 if (runtimes(j) < runtimes(fastest_idx)) {
468 fastest_idx = j;
469 }
470 buf.copy_from(sched.schedule_features);
471 it++;
472 }
473
474 float loss = 0.0f;
475 if (train) {
476 loss = tp->backprop(runtimes, learning_rate);
477 assert(!std::isnan(loss));
478 loss_sum[model] += loss;
479 loss_sum_counter[model]++;
480
481 auto it = p.second.schedules.begin();
482 std::advance(it, first);
483 for (size_t j = 0; j < batch_size; j++) {
484 auto &sched = it->second;
485 float m = sched.runtimes[0] / (sched.prediction[model] + 1e-10f);
486 if (m > worst_miss) {
487 worst_miss = m;
488 worst_miss_pipeline_id = p.first;
489 worst_miss_schedule_id = it->first;
490 }
491 it++;
492 }
493 } else {
494 tp->evaluate_costs();
495 }
496
497 if (true) {
498 int good = 0, bad = 0;
499 for (auto &sched : p.second.schedules) {
500 auto &ref = p.second.schedules[p.second.fastest_schedule_hash];
501 if (sched.second.prediction[model] == 0) {
502 continue;
503 }
504 assert(sched.second.runtimes[0] >= ref.runtimes[0]);
505 float runtime_ratio = sched.second.runtimes[0] / ref.runtimes[0];
506 if (runtime_ratio <= 1.3f) {
507 continue; // Within 30% of the runtime of the best
508 }
509 if (sched.second.prediction[model] >= ref.prediction[model]) {
510 good++;
511 } else {
512 if (train) {
513 float badness = (sched.second.runtimes[0] - ref.runtimes[0]) * (ref.prediction[model] - sched.second.prediction[model]);
514 badness /= (ref.runtimes[0] * ref.runtimes[0]);
515 if (badness > worst_inversion.badness) {
516 worst_inversion.pipeline_id = p.first;
517 worst_inversion.badness = badness;
518 worst_inversion.r1 = ref.runtimes[0];
519 worst_inversion.r2 = sched.second.runtimes[0];
520 worst_inversion.p1 = ref.prediction[model];
521 worst_inversion.p2 = sched.second.prediction[model];
522 worst_inversion.f1 = ref.filename;
523 worst_inversion.f2 = sched.second.filename;
524 }
525 }
526 bad++;
527 }
528 }
529 if (train) {
530 correct_ordering_rate_sum[model] += good;
531 correct_ordering_rate_count[model] += good + bad;
532 } else {
533 v_correct_ordering_rate_sum[model] += good;
534 v_correct_ordering_rate_count[model] += good + bad;
535 }
536 }
537 }
538 }
539
540 counter++;
541 }
542
543 std::cout << "Loss: ";
544 for (int model = 0; model < kModels; model++) {
545 std::cout << loss_sum[model] / loss_sum_counter[model] << " ";
546 loss_sum[model] *= 0.9f;
547 loss_sum_counter[model] *= 0.9f;
548 }
549 if (kModels > 1) {
550 std::cout << "\n";
551 }
552 std::cout << " Rate: ";
553 int best_model = 0;
554 float best_rate = 0;
555 for (int model = 0; model < kModels; model++) {
556 float rate = correct_ordering_rate_sum[model] / correct_ordering_rate_count[model];
557 std::cout << rate << " ";
558 correct_ordering_rate_sum[model] *= 0.9f;
559 correct_ordering_rate_count[model] *= 0.9f;
560
561 rate = v_correct_ordering_rate_sum[model] / v_correct_ordering_rate_count[model];
562 if (rate < best_rate) {
563 best_model = model;
564 best_rate = rate;
565 }
566 std::cout << rate << " ";
567 v_correct_ordering_rate_sum[model] *= 0.9f;
568 v_correct_ordering_rate_count[model] *= 0.9f;
569 }
570
571 if (kModels > 1) {
572 std::cout << "\n";
573 }
574 if (samples.count(worst_miss_pipeline_id)) {
575 std::cout << " Worst: " << worst_miss << " " << leaf(samples[worst_miss_pipeline_id].schedules[worst_miss_schedule_id].filename) << "\n";
576 // samples[worst_miss_pipeline_id].schedules.erase(worst_miss_schedule_id);
577 } else {
578 std::cout << "\n";
579 }
580
581 if (worst_inversion.badness > 0) {
582 std::cout << "Worst inversion:\n"
583 << leaf(worst_inversion.f1) << " predicted: " << worst_inversion.p1 << " actual: " << worst_inversion.r1 << "\n"
584 << leaf(worst_inversion.f2) << " predicted: " << worst_inversion.p2 << " actual: " << worst_inversion.r2 << "\n";
585 if (samples.size() > 50000) {
586 // For robustness during training on large numbers
587 // of random pipelines, we discard poorly
588 // performing samples from the training set
589 // only. Some of them are weird degenerate
590 // pipelines.
591 samples.erase(worst_inversion.pipeline_id);
592 }
593 }
594
595 tpp[best_model]->save_weights();
596
597 if (loss_sum[best_model] < 1e-5f) {
598 std::cout << "Zero loss, returning early\n";
599 return 0;
600 }
601 }
602 }
603
604 // tpp.save_weights();
605
606 return 0;
607}
