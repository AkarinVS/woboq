17void Weights::randomize(uint32_t seed) {
18 std::mt19937 rng(seed);
19 // Fill the weights with random values
20 for_each_buffer([&rng](Buffer<float> &w) {
21 w.for_each_value([&rng](float &f) {
22 f = ((float)rng()) / ((float)std::mt19937::max()) - 0.5f;
23 });
24 });
25}

27/*
28 Structure of the .weights file format:
29
30 uint32 signature always 0x68776631 ('hwf1')
31 uint32 PipelineFeatures::version
32 uint32 ScheduleFeatures::version
33 uint32 buffer-count
34 uint32 dimension-count
35 uint32x(dimension-count) dimension-extent
36 float32x(element-count) data
37
38 (all values little-endian)
39*/

41bool Weights::load(std::istream &i) {
42 uint32_t signature;
43 i.read((char *)&signature, sizeof(signature));
44 if (i.fail() || signature != kSignature) {
45 return false;
46 }
47
48 i.read((char *)&pipeline_features_version, sizeof(pipeline_features_version));
49 if (i.fail()) {
50 return false;
51 }
52
53 i.read((char *)&schedule_features_version, sizeof(schedule_features_version));
54 if (i.fail()) {
55 return false;
56 }
57
58 uint32_t buffer_count;
59 i.read((char *)&buffer_count, sizeof(buffer_count));
60 if (i.fail() || buffer_count != 6) {
61 return false;
62 }
63
64 const auto load_one = [&i](Buffer<float> &buf) -> bool {
65 uint32_t dimension_count;
66 i.read((char *)&dimension_count, sizeof(dimension_count));
67 if (i.fail() || dimension_count != (uint32_t)buf.dimensions()) {
68 return false;
69 }
70 for (uint32_t d = 0; d < dimension_count; d++) {
71 uint32_t extent;
72 i.read((char *)&extent, sizeof(extent));
73 if (i.fail() || (int)extent != (int)buf.extent(d)) {
74 return false;
75 }
76 }
77 i.read((char *)(buf.data()), buf.size_in_bytes());
78 if (i.fail()) {
79 return false;
80 }
81 return true;
82 };
83
84 if (!load_one(head1_filter)) {
85 return false;
86 }
87 if (!load_one(head1_bias)) {
88 return false;
89 }
90 if (!load_one(head2_filter)) {
91 return false;
92 }
93 if (!load_one(head2_bias)) {
94 return false;
95 }
96 if (!load_one(conv1_filter)) {
97 return false;
98 }
99 if (!load_one(conv1_bias)) {
100 return false;
101 }
102
103 return true;
104}
