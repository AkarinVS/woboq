1/*
2 This file is the core of the autoscheduler. Most of the code here is
3 about navigating the search space and computing the
4 featurization. This also contains the top-level interface into the
5 autoscheduler.
6
7 The most interesting classes to look at are:
8
9 LoopNest Represents one node in our tree representation of loop nests. (Now in LoopNest.(h | cpp)).
10 State A state in the beam search. Holds a root loop nest. (Now in State.(h | cpp)).
11
12 Interesting functions below are:
13
14 generate_schedule The top-level entrypoint, which computes and applies a schedule to a Halide pipeline
15 optimal_schedule Runs the passes of the coarse-to-fine beam search
16 optimal_schedule_pass Runs a single pass of beam search
17 LoopNest::compute_features Recursively walks over a loop nest tree, computing our featurization using Halide's analysis tools.
18 LoopNest::apply Actually apply a computed schedule to a Halide pipeline
19 State::generate_children Generates successor states to a state in the beam search
20
21 Environment variables used (directly or indirectly):
22
23 HL_BEAM_SIZE
24 Beam size to use in the beam search. Defaults to 32. Use 1 to get a greedy search instead.
25
26 HL_CYOS
27 "Choose-your-own-schedule". If set to 1, lets you navigate the search tree by hand in the terminal. Whee! This is for debugging the autoscheduler.
28
29 HL_FEATURE_FILE -> output
30 *** DEPRECATED *** use the 'featurization' output from Generator instead
31 Write out a training featurization for the selected schedule into this file.
32 Needs to be converted to a sample file with the runtime using featurization_to_sample before it can be used to train.
33
34 HL_MACHINE_PARAMS
35 An architecture description string. Used by Halide master to configure the cost model. We only use the first term. Set it to the number of cores to target.
36
37 HL_PERMIT_FAILED_UNROLL
38 Set to 1 to tell Halide not to freak out if we try to unroll a loop that doesn't have a constant extent. Should generally not be necessary, but sometimes the autoscheduler's model for what will and will not turn into a constant during lowering is inaccurate, because Halide isn't perfect at constant-folding.
39
40 HL_SCHEDULE_FILE
41 *** DEPRECATED *** use the 'schedule' output from Generator instead
42 Write out a human-and-machine readable block of scheduling source code for the selected schedule into this file.
43
44 HL_RANDOM_DROPOUT
45 percent chance of accepting each state in the beam. Normalized by the number of decisions made, so 5 would be there's a 5 percent chance of never rejecting any states.
46
47 HL_SEED
48 Random seed used by the random dropout.
49
50 HL_WEIGHTS_DIR
51 When training or schedule, read weights from this directory or file
52 (if path ends in `.weights` it is written as a single file, otherwise a directory of files)
53
54 HL_NO_SUBTILING
55 If set to 1, limits the search space to that of Mullapudi et al.
56
57 HL_DEBUG_AUTOSCHEDULE
58 If set, is used for the debug log level for auto-schedule generation (overriding the
59 value of HL_DEBUG_CODEGEN, if any).
60
61 HL_AUTOSCHEDULE_MEMORY_LIMIT
62 If set, only consider schedules that allocate at most this much memory (measured in bytes).
63
64 HL_DISABLE_MEMOIZED_FEATURES
65 If set, features of possible schedules are always recalculated, and are not cached across passes.
66 (see Cache.h for more information)
67
68 HL_DISABLE_MEMOIZED_BLOCKS
69 If set, then tiling sizes are not cached across passes.
70 (see Cache.h for more information)
71
72 TODO: expose these settings by adding some means to pass args to
73 generator plugins instead of environment vars.
74*/

115struct ProgressBar {
116 void set(double progress) {
117 if (!draw_progress_bar) {
118 return;
119 }
120 counter++;
121 const int bits = 11;
122 if (counter & ((1 << bits) - 1)) {
123 return;
124 }
125 const int pos = (int)(progress * 78);
126 aslog(0) << "[";
127 for (int j = 0; j < 78; j++) {
128 if (j < pos) {
129 aslog(0) << ".";
130 } else if (j - 1 < pos) {
131 aslog(0) << "/-\\|"[(counter >> bits) % 4];
132 } else {
133 aslog(0) << " ";
134 }
135 }
136 aslog(0) << "]";
137 for (int j = 0; j < 80; j++) {
138 aslog(0) << "\b";
139 }
140 }
141
142 void clear() {
143 if (counter) {
144 for (int j = 0; j < 80; j++) {
145 aslog(0) << " ";
146 }
147 for (int j = 0; j < 80; j++) {
148 aslog(0) << "\b";
149 }
150 }
151 }
152
153private:
154 uint32_t counter = 0;
155 const bool draw_progress_bar = isatty(2);
156};

168// Decide whether or not to drop a beam search state. Used for
169// randomly exploring the search tree for autotuning and to generate
170// training data.
171bool random_dropout(std::mt19937 &rng, size_t num_decisions) {
172 static double random_dropout_threshold = get_dropout_threshold();
173 if (random_dropout_threshold >= 100) {
174 return false;
175 }
176
177 // The random dropout threshold is the chance that we operate
178 // entirely greedily and never discard anything.
179 double t = random_dropout_threshold;
180 t /= 100;
181 t = std::pow(t, 1.0f / num_decisions);
182 t *= 100;
183
184 uint32_t r = rng();
185 bool drop_it = (r % 100) >= t;
186 return drop_it;
187}

189// A priority queue of states, sorted according to increasing
190// cost. Never shrinks, to avoid reallocations.
191// Can't use std::priority_queue because it doesn't support unique_ptr.
192class StateQueue {
193private:
194 struct CompareStates {
195 bool operator()(const IntrusivePtr<State> &a, const IntrusivePtr<State> &b) const {
196 return a->cost > b->cost;
197 }
198 };
199
200 std::vector<IntrusivePtr<State>> storage;
201 size_t sz = 0;
202
203public:
204 void emplace(IntrusivePtr<State> &&s) {
205 if (sz >= storage.size()) {
206 storage.resize(std::max(sz * 2, (size_t)64));
207 }
208 internal_assert(sz < storage.size()) << sz << " " << storage.size() << "\n";
209 storage[sz] = std::move(s);
210 sz++;
211 std::push_heap(storage.begin(), storage.begin() + sz, CompareStates{});
212 }
213
214 IntrusivePtr<State> pop() {
215 internal_assert(sz <= storage.size()) << sz << " " << storage.size() << "\n";
216 std::pop_heap(storage.begin(), storage.begin() + sz, CompareStates{});
217 sz--;
218 return std::move(storage[sz]);
219 }
220
221 const IntrusivePtr<State> &top() {
222 return storage[0];
223 }
224
225 bool empty() const {
226 return sz == 0;
227 }
228
229 size_t size() const {
230 return sz;
231 }
232
233 void swap(StateQueue &other) {
234 storage.swap(other.storage);
235 std::swap(sz, other.sz);
236 }
237
238 IntrusivePtr<State> operator[](int idx) const {
239 return storage[idx];
240 }
241
242 void resort() {
243 std::make_heap(storage.begin(), storage.begin() + sz, CompareStates{});
244 }
245
246 void clear() {
247 for (size_t i = 0; i < sz; i++) {
248 storage[i] = IntrusivePtr<State>{};
249 }
250 sz = 0;
251 }
252};

262// A single pass of coarse-to-fine beam search.
263IntrusivePtr<State> optimal_schedule_pass(FunctionDAG &dag,
264 const vector<Function> &outputs,
265 const MachineParams &params,
266 CostModel *cost_model,
267 std::mt19937 &rng,
268 int beam_size,
269 int64_t memory_limit,
270 int pass_idx,
271 int num_passes,
272 ProgressBar &tick,
273 std::unordered_set<uint64_t> &permitted_hashes,
274 Cache *cache) {
275
276 if (cost_model) {
277 configure_pipeline_features(dag, params, cost_model);
278 }
279
280 StateQueue q, pending;
281
282 // The initial state, with no decisions made
283 {
284 IntrusivePtr<State> initial{new State};
285 initial->root = new LoopNest;
286 q.emplace(std::move(initial));
287 }
288
289 int expanded = 0;
290
291 std::function<void(IntrusivePtr<State> &&)> enqueue_new_children =
292 [&](IntrusivePtr<State> &&s) {
293 // aslog(0) << "\n** Generated child: ";
294 // s->dump();
295 // s->calculate_cost(dag, params, nullptr, true);
296
297 // Each child should have one more decision made than its parent state.
298 internal_assert(s->num_decisions_made == s->parent->num_decisions_made + 1);
299
300 int progress = s->num_decisions_made * beam_size + expanded;
301 size_t max_progress = dag.nodes.size() * beam_size * 2;
302
303 // Update the progress bar
304 tick.set(double(progress) / max_progress);
305 s->penalized = false;
306
307 // Add the state to the list of states to evaluate
308 q.emplace(std::move(s));
309 };
310
311 string cyos_str = get_env_variable("HL_CYOS");
312
313 // This loop is beam search over the sequence of decisions to make.
314 for (int i = 0;; i++) {
315 std::unordered_map<uint64_t, int> hashes;
316 q.swap(pending);
317
318 if (pending.empty()) {
319 if ((false) && beam_size < 1000) { // Intentional dead code. Extra parens to pacify clang-tidy.
320 // Total mortality. Double the beam size and
321 // restart. Disabled for now because total mortality
322 // may indicate a bug.
323 return optimal_schedule_pass(dag,
324 outputs,
325 params,
326 cost_model,
327 rng,
328 beam_size * 2,
329 memory_limit,
330 pass_idx,
331 num_passes,
332 tick,
333 permitted_hashes,
334 cache);
335 } else {
336 internal_error << "Ran out of legal states with beam size " << beam_size << "\n";
337 }
338 }
339
340 if ((int)pending.size() > beam_size * 10000) {
341 aslog(0) << "Warning: Huge number of states generated (" << pending.size() << ").\n";
342 }
343
344 expanded = 0;
345 while (expanded < beam_size && !pending.empty()) {
346
347 IntrusivePtr<State> state{pending.pop()};
348
349 if (beam_size > 1 && num_passes > 1) {
350 // We are doing coarse-to-fine beam search using the
351 // hashing strategy mentioned in the paper.
352 //
353 // We will lazily apply cost penalties to the queue
354 // according to structural uniqueness.
355 if (!state->penalized) {
356 uint64_t h1 = state->structural_hash(pass_idx + 1);
357 uint64_t h0 = state->structural_hash(pass_idx - 1);
358 // We penalize the cost of a state proportionately
359 // to how many states we've already seen with that
360 // hash.
361 int penalty = ++hashes[h1];
362 if (pass_idx > 0 && !permitted_hashes.count(h0)) {
363 // It's possible to get yourself into a state
364 // where the only things in the beam that match
365 // the hash were quick-rejected due to details not
366 // captured in the hash, so we apply a huge
367 // penalty, but leave the impermissible state in
368 // the beam.
369 penalty += 10;
370 }
371 if (penalty > 1) {
372 state->penalized = true;
373 state->cost *= penalty;
374 // After penalizing this state, if it's no
375 // longer the best, defer it. We set the
376 // 'penalized' flag so that we know not to
377 // penalize and defer it again.
378 if (!pending.empty() && state->cost > pending.top()->cost) {
379 pending.emplace(std::move(state));
380 continue;
381 }
382 }
383 }
384 }
385
386 // Random dropout
387 if (pending.size() > 1 && random_dropout(rng, dag.nodes.size() * 2)) {
388 continue;
389 }
390
391 if (state->num_decisions_made == 2 * (int)dag.nodes.size()) {
392 // We've reached the end of the pass. The first state
393 // must be the best, because we're pulling off a
394 // priority queue.
395 auto best = state;
396
397 // Bless the reasonable stuff in the beam as
398 // permissible states to visit again. We define
399 // reasonable as having a cost no more than 20% higher
400 // than the cost of the best thing. Only do this if
401 // there are more coarse-to-fine passes yet to come.
402 if (pass_idx + 1 < num_passes) {
403 int blessed = 0;
404 while (state->cost <= 1.2 * best->cost && blessed < beam_size) {
405 const State *s = state.get();
406 while (s) {
407 uint64_t h1 = s->structural_hash(pass_idx);
408 permitted_hashes.insert(h1);
409 s = s->parent.get();
410 }
411 if (pending.empty()) {
412 break;
413 }
414 state = pending.pop();
415 blessed++;
416 }
417 }
418
419 return best;
420 }
421
422 state->generate_children(dag, params, cost_model, memory_limit, enqueue_new_children, cache);
423 expanded++;
424 }
425
426 // Drop the other states unconsidered.
427 pending.clear();
428
429 if (cost_model) {
430 // Now evaluate all the costs and re-sort them in the priority queue
431 cost_model->evaluate_costs();
432 q.resort();
433 }
434
435 if (cyos_str == "1") {
436 // The user has set HL_CYOS, and wants to navigate the
437 // search space manually. Discard everything in the queue
438 // except for the user-chosen option.
439 aslog(0) << "\n--------------------\n";
440 aslog(0) << "Select a schedule:\n";
441 for (int choice_label = (int)q.size() - 1; choice_label >= 0; choice_label--) {
442 auto state = q[choice_label];
443 aslog(0) << "\n[" << choice_label << "]:\n";
444 state->dump();
445 state->calculate_cost(dag, params, cost_model, cache->options, memory_limit, true);
446 }
447 cost_model->evaluate_costs();
448
449 // Select next partial schedule to expand.
450 int selection = -1;
451 while (selection < 0 || selection >= (int)q.size()) {
452 aslog(0) << "\nEnter selection: ";
453 std::cin >> selection;
454 }
455
456 auto selected = q[selection];
457 selected->dump();
458 q.clear();
459 q.emplace(std::move(selected));
460 }
461 }

464// Performance coarse-to-fine beam search and return the best state found.
465IntrusivePtr<State> optimal_schedule(FunctionDAG &dag,
466 const vector<Function> &outputs,
467 const MachineParams &params,
468 CostModel *cost_model,
469 std::mt19937 &rng,
470 int beam_size,
471 int64_t memory_limit,
472 const CachingOptions &options) {
473
474 IntrusivePtr<State> best;
475
476 std::unordered_set<uint64_t> permitted_hashes;
477
478 // Set up cache with options and size.
479 Cache cache(options, dag.nodes.size());
480
481 // If the beam size is one, it's pointless doing multiple passes.
482 int num_passes = (beam_size == 1) ? 1 : 5;
483
484 string cyos_str = get_env_variable("HL_CYOS");
485 if (cyos_str == "1") {
486 // If the user is manually navigating the search space, don't
487 // ask them to do more than one pass.
488 num_passes = 1;
489 }
490
491 string num_passes_str = get_env_variable("HL_NUM_PASSES");
492 if (!num_passes_str.empty()) {
493 // The user has requested a non-standard number of passes.
494 num_passes = std::atoi(num_passes_str.c_str());
495 }
496
497 for (int i = 0; i < num_passes; i++) {
498 ProgressBar tick;
499
500 Timer timer;
501
502 auto pass = optimal_schedule_pass(dag, outputs, params, cost_model,
503 rng, beam_size, memory_limit,
504 i, num_passes, tick, permitted_hashes, &cache);
505
506 std::chrono::duration<double> total_time = timer.elapsed();
507 auto milli = std::chrono::duration_cast<std::chrono::milliseconds>(total_time).count();
508
509 tick.clear();
510
511 if (aslog::aslog_level() == 0) {
512 aslog(0) << "Pass " << i << " of " << num_passes << ", cost: " << pass->cost << ", time (ms): " << milli << "\n";
513 } else {
514 aslog(0) << "Pass " << i << " result: ";
515 pass->dump();
516 }
517
518 if (i == 0 || pass->cost < best->cost) {
519 // Track which pass produced the lowest-cost state. It's
520 // not necessarily the final one.
521 best = pass;
522 }
523 }
524
525 aslog(0) << "Best cost: " << best->cost << "\n";
526
527 if (options.cache_blocks) {
528 aslog(0) << "Cache (block) hits: " << cache.cache_hits << "\n";
529 aslog(0) << "Cache (block) misses: " << cache.cache_misses << "\n";
530 }
531
532 return best;
533}

538// The main entrypoint to generate a schedule for a pipeline.
539void generate_schedule(const std::vector<Function> &outputs,
540 const Target &target,
541 const MachineParams &params,
542 AutoSchedulerResults *auto_scheduler_results) {
543 aslog(0) << "generate_schedule for target=" << target.to_string() << "\n";
544
545 // Start a timer
546 HALIDE_TIC;
547
548 State::cost_calculations = 0;
549
550 // Get the seed for random dropout
551 string seed_str = get_env_variable("HL_SEED");
552 // Or use the time, if not set.
553 int seed = (int)time(nullptr);
554 if (!seed_str.empty()) {
555 seed = atoi(seed_str.c_str());
556 }
557 aslog(1) << "Dropout seed = " << seed << "\n";
558 std::mt19937 rng((uint32_t)seed);
559
560 // Get the beam size
561 string beam_size_str = get_env_variable("HL_BEAM_SIZE");
562 // Defaults to 32
563 size_t beam_size = 32;
564 if (!beam_size_str.empty()) {
565 beam_size = atoi(beam_size_str.c_str());
566 }
567
568 string weights_in_path = get_env_variable("HL_WEIGHTS_DIR");
569 string weights_out_path; // deliberately empty
570
571 string randomize_weights_str = get_env_variable("HL_RANDOMIZE_WEIGHTS");
572 bool randomize_weights = randomize_weights_str == "1";
573
574 string memory_limit_str = get_env_variable("HL_AUTOSCHEDULE_MEMORY_LIMIT");
575 int64_t memory_limit = memory_limit_str.empty() ? (uint64_t)(-1) : std::atoll(memory_limit_str.c_str());
576
577 // Analyse the Halide algorithm and construct our abstract representation of it
578 FunctionDAG dag(outputs, params, target);
579 if (aslog::aslog_level() > 0) {
580 dag.dump();
581 }
582
583 // Construct a cost model to use to evaluate states. Currently we
584 // just have the one, but it's an abstract interface, so others
585 // can be slotted in for experimentation.
586 std::unique_ptr<CostModel> cost_model = make_default_cost_model(weights_in_path, weights_out_path, randomize_weights);
587 internal_assert(cost_model != nullptr);
588
589 IntrusivePtr<State> optimal;
590
591 // Options generated from environment variables, decide whether or not to cache features and/or tilings.
592 CachingOptions cache_options = CachingOptions::MakeOptionsFromEnviron();
593
594 // Run beam search
595 optimal = optimal_schedule(dag, outputs, params, cost_model.get(), rng, beam_size, memory_limit, cache_options);
596
597 HALIDE_TOC;
598
599 aslog(1) << "Cost evaluated this many times: " << State::cost_calculations << "\n";
600
601 // Dump the schedule found
602 aslog(1) << "** Optimal schedule:\n";
603
604 // Just to get the debugging prints to fire
605 optimal->calculate_cost(dag, params, cost_model.get(), cache_options, memory_limit, aslog::aslog_level() > 0);
606
607 // Apply the schedules to the pipeline
608 optimal->apply_schedule(dag, params);
609
610 // Print out the schedule
611 if (aslog::aslog_level() > 0) {
612 optimal->dump();
613 }
614
615 string schedule_file = get_env_variable("HL_SCHEDULE_FILE");
616 if (!schedule_file.empty()) {
617 user_warning << "HL_SCHEDULE_FILE is deprecated; use the schedule output from Generator instead\n";
618 aslog(1) << "Writing schedule to " << schedule_file << "...\n";
619 std::ofstream f(schedule_file);
620 f << "// --- BEGIN machine-generated schedule\n"
621 << optimal->schedule_source
622 << "// --- END machine-generated schedule\n";
623 f.close();
624 internal_assert(!f.fail()) << "Failed to write " << schedule_file;
625 }
626
627 // Save the featurization, so that we can use this schedule as
628 // training data (once we've benchmarked it).
629 string feature_file = get_env_variable("HL_FEATURE_FILE");
630 if (!feature_file.empty()) {
631 user_warning << "HL_FEATURE_FILE is deprecated; use the featurization output from Generator instead\n";
632 std::ofstream binfile(feature_file, std::ios::binary | std::ios_base::trunc);
633 optimal->save_featurization(dag, params, cache_options, binfile);
634 binfile.close();
635 internal_assert(!binfile.fail()) << "Failed to write " << feature_file;
636 }
637
638 if (auto_scheduler_results) {
639 auto_scheduler_results->scheduler_name = "Adams2019";
640 auto_scheduler_results->schedule_source = optimal->schedule_source;
641 {
642 std::ostringstream out;
643 optimal->save_featurization(dag, params, cache_options, out);
644 auto_scheduler_results->featurization.resize(out.str().size());
645 memcpy(auto_scheduler_results->featurization.data(), out.str().data(), out.str().size());
646 }
647 }
648}

650struct Adams2019 {
651 void operator()(const Pipeline &p, const Target &target, const MachineParams &params, AutoSchedulerResults *results) {
652 std::vector<Function> outputs;
653 for (const Func &f : p.outputs()) {
654 outputs.push_back(f.function());
655 }
656 Autoscheduler::generate_schedule(outputs, target, params, results);
657 }

662// An alternative entrypoint for other uses
663void find_and_apply_schedule(FunctionDAG &dag,
664 const std::vector<Function> &outputs,
665 const MachineParams &params,
666 CostModel *cost_model,
667 int beam_size,
668 int64_t memory_limit,
669 StageMap<ScheduleFeatures> *schedule_features) {
670
671 std::mt19937 rng(12345);
672 CachingOptions cache_options = CachingOptions::MakeOptionsFromEnviron();
673 IntrusivePtr<State> optimal = optimal_schedule(dag, outputs, params, cost_model, rng, beam_size, memory_limit, cache_options);
674
675 // Apply the schedules
676 optimal->apply_schedule(dag, params);
677
678 if (schedule_features) {
679 optimal->compute_featurization(dag, params, schedule_features, cache_options);
680 }
681}
