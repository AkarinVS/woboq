22class StoreCollector : public IRMutator {
23public:
24 const std::string store_name;
25 const int store_stride, max_stores;
26 std::vector<Stmt> &let_stmts;
27 std::vector<Stmt> &stores;
28
29 StoreCollector(const std::string &name, int stride, int ms,
30 std::vector<Stmt> &lets, std::vector<Stmt> &ss)
31 : store_name(name), store_stride(stride), max_stores(ms),
32 let_stmts(lets), stores(ss), collecting(true) {
33 }
34
35private:
36 using IRMutator::visit;
37
38 // Don't enter any inner constructs for which it's not safe to pull out stores.
39 Stmt visit(const For *op) override {
40 collecting = false;
41 return op;
42 }
43 Stmt visit(const IfThenElse *op) override {
44 collecting = false;
45 return op;
46 }
47 Stmt visit(const ProducerConsumer *op) override {
48 collecting = false;
49 return op;
50 }
51 Stmt visit(const Allocate *op) override {
52 collecting = false;
53 return op;
54 }
55 Stmt visit(const Realize *op) override {
56 collecting = false;
57 return op;
58 }
59
60 bool collecting;
61 // These are lets that we've encountered since the last collected
62 // store. If we collect another store, these "potential" lets
63 // become lets used by the collected stores.
64 std::vector<Stmt> potential_lets;
65
66 Expr visit(const Load *op) override {
67 if (!collecting) {
68 return op;
69 }
70
71 // If we hit a load from the buffer we're trying to collect
72 // stores for, stop collecting to avoid reordering loads and
73 // stores from the same buffer.
74 if (op->name == store_name) {
75 collecting = false;
76 return op;
77 } else {
78 return IRMutator::visit(op);
79 }
80 }
81
82 Stmt visit(const Store *op) override {
83 if (!collecting) {
84 return op;
85 }
86
87 // By default, do nothing.
88 Stmt stmt = op;
89
90 if (stores.size() >= (size_t)max_stores) {
91 // Already have enough stores.
92 collecting = false;
93 return stmt;
94 }
95
96 // Make sure this Store doesn't do anything that causes us to
97 // stop collecting.
98 stmt = IRMutator::visit(op);
99 if (!collecting) {
100 return stmt;
101 }
102
103 if (op->name != store_name) {
104 // Not a store to the buffer we're looking for.
105 return stmt;
106 }
107
108 const Ramp *r = op->index.as<Ramp>();
109 if (!r || !is_const(r->stride, store_stride)) {
110 // Store doesn't store to the ramp we're looking
111 // for. Can't interleave it. Since we don't want to
112 // reorder stores, stop collecting.
113 collecting = false;
114 return stmt;
115 }
116
117 // This store is good, collect it and replace with a no-op.
118 stores.emplace_back(op);
119 stmt = Evaluate::make(0);
120
121 // Because we collected this store, we need to save the
122 // potential lets since the last collected store.
123 let_stmts.insert(let_stmts.end(), potential_lets.begin(), potential_lets.end());
124 potential_lets.clear();
125 return stmt;
126 }
127
128 Expr visit(const Call *op) override {
129 if (!op->is_pure()) {
130 // Avoid reordering calls to impure functions
131 collecting = false;
132 return op;
133 } else {
134 return IRMutator::visit(op);
135 }
136 }
137
138 Stmt visit(const LetStmt *op) override {
139 if (!collecting) {
140 return op;
141 }
142
143 // Walk inside the let chain
144 Stmt stmt = IRMutator::visit(op);
145
146 // If we're still collecting, we need to save the entire let chain as potential lets.
147 if (collecting) {
148 Stmt body;
149 do {
150 potential_lets.emplace_back(op);
151 body = op->body;
152 } while ((op = body.as<LetStmt>()));
153 }
154 return stmt;
155 }
156
157 Stmt visit(const Block *op) override {
158 if (!collecting) {
159 return op;
160 }
161
162 Stmt first = mutate(op->first);
163 Stmt rest = op->rest;
164 // We might have decided to stop collecting during mutation of first.
165 if (collecting) {
166 rest = mutate(rest);
167 }
168 return Block::make(first, rest);
169 }
170};

179class Deinterleaver : public IRGraphMutator {
180public:
181 Deinterleaver(int starting_lane, int lane_stride, int new_lanes, const Scope<> &lets)
182 : starting_lane(starting_lane),
183 lane_stride(lane_stride),
184 new_lanes(new_lanes),
185 external_lets(lets) {
186 }
187
188private:
189 int starting_lane;
190 int lane_stride;
191 int new_lanes;
192
193 // lets for which we have even and odd lane specializations
194 const Scope<> &external_lets;
195
196 using IRMutator::visit;
197
198 Expr visit(const VectorReduce *op) override {
199 std::vector<int> input_lanes;
200 int factor = op->value.type().lanes() / op->type.lanes();
201 for (int i = starting_lane; i < op->type.lanes(); i += lane_stride) {
202 for (int j = 0; j < factor; j++) {
203 input_lanes.push_back(i * factor + j);
204 }
205 }
206 Expr in = Shuffle::make({op->value}, input_lanes);
207 return VectorReduce::make(op->op, in, new_lanes);
208 }
209
210 Expr visit(const Broadcast *op) override {
211 if (new_lanes == 1) {
212 if (op->value.type().lanes() == 1) {
213 return op->value;
214 } else {
215 int old_starting_lane = starting_lane;
216 int old_lane_stride = lane_stride;
217 starting_lane = starting_lane % op->value.type().lanes();
218 lane_stride = op->value.type().lanes();
219 Expr e = mutate(op->value);
220 starting_lane = old_starting_lane;
221 lane_stride = old_lane_stride;
222 return e;
223 }
224 }
225 if (op->value.type().lanes() > 1) {
226 // There is probably a more efficient way to this.
227 return mutate(flatten_nested_ramps(op));
228 }
229
230 return Broadcast::make(op->value, new_lanes);
231 }
232
233 Expr visit(const Load *op) override {
234 if (op->type.is_scalar()) {
235 return op;
236 } else {
237 Type t = op->type.with_lanes(new_lanes);
238 ModulusRemainder align = op->alignment;
239 // TODO: Figure out the alignment of every nth lane
240 if (starting_lane != 0) {
241 align = ModulusRemainder();
242 }
243 return Load::make(t, op->name, mutate(op->index), op->image, op->param, mutate(op->predicate), align);
244 }
245 }
246
247 Expr visit(const Ramp *op) override {
248 int base_lanes = op->base.type().lanes();
249 if (base_lanes > 1) {
250 if (new_lanes == 1) {
251 int index = starting_lane / base_lanes;
252 Expr expr = op->base + cast(op->base.type(), index) * op->stride;
253 ScopedValue<int> old_starting_lane(starting_lane, starting_lane % base_lanes);
254 ScopedValue<int> old_lane_stride(lane_stride, base_lanes);
255 expr = mutate(expr);
256 return expr;
257 } else if (base_lanes == lane_stride &&
258 starting_lane < base_lanes) {
259 // Base class mutator actually works fine in this
260 // case, but we only want one lane from the base and
261 // one lane from the stride.
262 ScopedValue<int> old_new_lanes(new_lanes, 1);
263 return IRMutator::visit(op);
264 } else {
265 // There is probably a more efficient way to this by
266 // generalizing the two cases above.
267 return mutate(flatten_nested_ramps(op));
268 }
269 }
270 Expr expr = op->base + cast(op->base.type(), starting_lane) * op->stride;
271 internal_assert(expr.type() == op->base.type());
272 if (new_lanes > 1) {
273 expr = Ramp::make(expr, op->stride * lane_stride, new_lanes);
274 }
275 return expr;
276 }
277
278 Expr visit(const Variable *op) override {
279 if (op->type.is_scalar()) {
280 return op;
281 } else {
282
283 Type t = op->type.with_lanes(new_lanes);
284 if (external_lets.contains(op->name) &&
285 starting_lane == 0 &&
286 lane_stride == 2) {
287 return Variable::make(t, op->name + ".even_lanes", op->image, op->param, op->reduction_domain);
288 } else if (external_lets.contains(op->name) &&
289 starting_lane == 1 &&
290 lane_stride == 2) {
291 return Variable::make(t, op->name + ".odd_lanes", op->image, op->param, op->reduction_domain);
292 } else if (external_lets.contains(op->name) &&
293 starting_lane == 0 &&
294 lane_stride == 3) {
295 return Variable::make(t, op->name + ".lanes_0_of_3", op->image, op->param, op->reduction_domain);
296 } else if (external_lets.contains(op->name) &&
297 starting_lane == 1 &&
298 lane_stride == 3) {
299 return Variable::make(t, op->name + ".lanes_1_of_3", op->image, op->param, op->reduction_domain);
300 } else if (external_lets.contains(op->name) &&
301 starting_lane == 2 &&
302 lane_stride == 3) {
303 return Variable::make(t, op->name + ".lanes_2_of_3", op->image, op->param, op->reduction_domain);
304 } else {
305 // Uh-oh, we don't know how to deinterleave this vector expression
306 // Make llvm do it
307 std::vector<int> indices;
308 for (int i = 0; i < new_lanes; i++) {
309 indices.push_back(starting_lane + lane_stride * i);
310 }
311 return Shuffle::make({op}, indices);
312 }
313 }
314 }
315
316 Expr visit(const Cast *op) override {
317 if (op->type.is_scalar()) {
318 return op;
319 } else {
320 Type t = op->type.with_lanes(new_lanes);
321 return Cast::make(t, mutate(op->value));
322 }
323 }
324
325 Expr visit(const Call *op) override {
326 Type t = op->type.with_lanes(new_lanes);
327
328 // Don't mutate scalars
329 if (op->type.is_scalar()) {
330 return op;
331 } else {
332
333 // Vector calls are always parallel across the lanes, so we
334 // can just deinterleave the args.
335
336 // Beware of intrinsics for which this is not true!
337 std::vector<Expr> args(op->args.size());
338 for (size_t i = 0; i < args.size(); i++) {
339 args[i] = mutate(op->args[i]);
340 }
341
342 return Call::make(t, op->name, args, op->call_type,
343 op->func, op->value_index, op->image, op->param);
344 }
345 }
346
347 Expr visit(const Shuffle *op) override {
348 if (op->is_interleave()) {
349 // Special case where we can discard some of the vector arguments entirely.
350 internal_assert(starting_lane >= 0 && starting_lane < lane_stride);
351 if ((int)op->vectors.size() == lane_stride) {
352 return op->vectors[starting_lane];
353 } else if ((int)op->vectors.size() % lane_stride == 0) {
354 // Pick up every lane-stride vector.
355 std::vector<Expr> new_vectors(op->vectors.size() / lane_stride);
356 for (size_t i = 0; i < new_vectors.size(); i++) {
357 new_vectors[i] = op->vectors[i * lane_stride + starting_lane];
358 }
359 return Shuffle::make_interleave(new_vectors);
360 }
361 }
362
363 // Keep the same set of vectors and extract every nth numeric
364 // arg to the shuffle.
365 std::vector<int> indices;
366 for (int i = 0; i < new_lanes; i++) {
367 int idx = i * lane_stride + starting_lane;
368 indices.push_back(op->indices[idx]);
369 }
370 return Shuffle::make(op->vectors, indices);
371 }
372};

374Expr deinterleave(Expr e, int starting_lane, int lane_stride, int new_lanes, const Scope<> &lets) {
375 e = substitute_in_all_lets(e);
376 Deinterleaver d(starting_lane, lane_stride, new_lanes, lets);
377 e = d.mutate(e);
378 e = common_subexpression_elimination(e);
379 return simplify(e);
380}

417class Interleaver : public IRMutator {
418 Scope<> vector_lets;
419
420 using IRMutator::visit;
421
422 bool should_deinterleave = false;
423 int num_lanes;
424
425 Expr deinterleave_expr(const Expr &e) {
426 std::vector<Expr> exprs;
427 for (int i = 0; i < num_lanes; i++) {
428 Scope<> lets;
429 exprs.emplace_back(deinterleave(e, i, num_lanes, e.type().lanes() / num_lanes, lets));
430 }
431 return Shuffle::make_interleave(exprs);
432 }

434 template<typename T, typename Body>
435 Body visit_lets(const T *op) {
436 // Visit an entire chain of lets in a single method to conserve stack space.
437 struct Frame {
438 const T *op;
439 Expr new_value;
440 ScopedBinding<> binding;
441 Frame(const T *op, Expr v, Scope<void> &scope)
442 : op(op),
443 new_value(std::move(v)),
444 binding(new_value.type().is_vector(), scope, op->name) {
445 }
446 };
447 std::vector<Frame> frames;
448 Body result;
449
450 do {
451 result = op->body;
452 frames.emplace_back(op, mutate(op->value), vector_lets);
453 } while ((op = result.template as<T>()));
454
455 result = mutate(result);
456
457 for (auto it = frames.rbegin(); it != frames.rend(); it++) {
458 Expr value = std::move(it->new_value);
459
460 result = T::make(it->op->name, value, result);
461
462 // For vector lets, we may additionally need a let defining the even and odd lanes only
463 if (value.type().is_vector()) {
464 if (value.type().lanes() % 2 == 0) {
465 result = T::make(it->op->name + ".even_lanes", extract_even_lanes(value, vector_lets), result);
466 result = T::make(it->op->name + ".odd_lanes", extract_odd_lanes(value, vector_lets), result);
467 }
468 if (value.type().lanes() % 3 == 0) {
469 result = T::make(it->op->name + ".lanes_0_of_3", extract_mod3_lanes(value, 0, vector_lets), result);
470 result = T::make(it->op->name + ".lanes_1_of_3", extract_mod3_lanes(value, 1, vector_lets), result);
471 result = T::make(it->op->name + ".lanes_2_of_3", extract_mod3_lanes(value, 2, vector_lets), result);
472 }
473 }
474 }
475
476 return result;
477 }

487 Expr visit(const Ramp *op) override {
488 if (op->stride.type().is_vector() &&
489 is_const_one(op->stride) &&
490 !op->base.as<Ramp>() &&
491 !op->base.as<Broadcast>()) {
492 // We have a ramp with a computed vector base. If we
493 // deinterleave we'll get ramps of stride 1 with a
494 // computed scalar base.
495 should_deinterleave = true;
496 num_lanes = op->stride.type().lanes();
497 }
498 return IRMutator::visit(op);
499 }
500
501 Expr visit(const Mod *op) override {
502 const Ramp *r = op->a.as<Ramp>();
503 for (int i = 2; i <= 4; ++i) {
504 if (r &&
505 is_const(op->b, i) &&
506 (r->type.lanes() % i) == 0) {
507 should_deinterleave = true;
508 num_lanes = i;
509 break;
510 }
511 }
512 return IRMutator::visit(op);
513 }
514
515 Expr visit(const Div *op) override {
516 const Ramp *r = op->a.as<Ramp>();
517 for (int i = 2; i <= 4; ++i) {
518 if (r &&
519 is_const(op->b, i) &&
520 (r->type.lanes() % i) == 0 &&
521 r->type.lanes() > i) {
522 should_deinterleave = true;
523 num_lanes = i;
524 break;
525 }
526 }
527 return IRMutator::visit(op);
528 }

530 Expr visit(const Call *op) override {
531 if (!op->is_pure() &&
532 !op->is_intrinsic(Call::unsafe_promise_clamped) &&
533 !op->is_intrinsic(Call::promise_clamped)) {
534 // deinterleaving potentially changes the order of execution.
535 should_deinterleave = false;
536 }
537 return IRMutator::visit(op);
538 }

540 Expr visit(const Load *op) override {
541 bool old_should_deinterleave = should_deinterleave;
542 int old_num_lanes = num_lanes;
543
544 should_deinterleave = false;
545 Expr idx = mutate(op->index);
546 bool should_deinterleave_idx = should_deinterleave;
547
548 should_deinterleave = false;
549 Expr predicate = mutate(op->predicate);
550 bool should_deinterleave_predicate = should_deinterleave;
551
552 Expr expr;
553 if (should_deinterleave_idx && (should_deinterleave_predicate || is_const_one(predicate))) {
554 // If we want to deinterleave both the index and predicate
555 // (or the predicate is one), then deinterleave the
556 // resulting load.
557 expr = Load::make(op->type, op->name, idx, op->image, op->param, predicate, op->alignment);
558 expr = deinterleave_expr(expr);
559 } else if (should_deinterleave_idx) {
560 // If we only want to deinterleave the index and not the
561 // predicate, deinterleave the index prior to the load.
562 idx = deinterleave_expr(idx);
563 expr = Load::make(op->type, op->name, idx, op->image, op->param, predicate, op->alignment);
564 } else if (should_deinterleave_predicate) {
565 // Similarly, deinterleave the predicate prior to the load
566 // if we don't want to deinterleave the index.
567 predicate = deinterleave_expr(predicate);
568 expr = Load::make(op->type, op->name, idx, op->image, op->param, predicate, op->alignment);
569 } else if (!idx.same_as(op->index) || !predicate.same_as(op->index)) {
570 expr = Load::make(op->type, op->name, idx, op->image, op->param, predicate, op->alignment);
571 } else {
572 expr = op;
573 }
574
575 should_deinterleave = old_should_deinterleave;
576 num_lanes = old_num_lanes;
577 return expr;
578 }
579
580 Stmt visit(const Store *op) override {
581 bool old_should_deinterleave = should_deinterleave;
582 int old_num_lanes = num_lanes;
583
584 should_deinterleave = false;
585 Expr idx = mutate(op->index);
586 if (should_deinterleave) {
587 idx = deinterleave_expr(idx);
588 }
589
590 should_deinterleave = false;
591 Expr value = mutate(op->value);
592 if (should_deinterleave) {
593 value = deinterleave_expr(value);
594 }
595
596 should_deinterleave = false;
597 Expr predicate = mutate(op->predicate);
598 if (should_deinterleave) {
599 predicate = deinterleave_expr(predicate);
600 }
601
602 Stmt stmt = Store::make(op->name, value, idx, op->param, predicate, op->alignment);
603
604 should_deinterleave = old_should_deinterleave;
605 num_lanes = old_num_lanes;
606
607 return stmt;
608 }

610 HALIDE_NEVER_INLINE Stmt gather_stores(const Block *op) {
611 const LetStmt *let = op->first.as<LetStmt>();
612 const Store *store = op->first.as<Store>();
613
614 // Gather all the let stmts surrounding the first.
615 std::vector<Stmt> let_stmts;
616 while (let) {
617 let_stmts.emplace_back(let);
618 store = let->body.as<Store>();
619 let = let->body.as<LetStmt>();
620 }
621
622 // There was no inner store.
623 if (!store) {
624 return Stmt();
625 }
626
627 const Ramp *r0 = store->index.as<Ramp>();
628
629 // It's not a store of a ramp index.
630 if (!r0) {
631 return Stmt();
632 }
633
634 const int64_t *stride_ptr = as_const_int(r0->stride);
635
636 // The stride isn't a constant or is <= 1
637 if (!stride_ptr || *stride_ptr <= 1) {
638 return Stmt();
639 }
640
641 const int64_t stride = *stride_ptr;
642 const int lanes = r0->lanes;
643 const int64_t expected_stores = stride;
644
645 // Collect the rest of the stores.
646 std::vector<Stmt> stores;
647 stores.emplace_back(store);
648 Stmt rest = collect_strided_stores(op->rest, store->name,
649 stride, expected_stores,
650 let_stmts, stores);
651
652 // Check the store collector didn't collect too many
653 // stores (that would be a bug).
654 internal_assert(stores.size() <= (size_t)expected_stores);
655
656 // Not enough stores collected.
657 if (stores.size() != (size_t)expected_stores) {
658 return Stmt();
659 }
660
661 // Too many stores and lanes to represent in a single vector
662 // type.
663 int max_bits = sizeof(halide_type_t::lanes) * 8;
664 // mul_would_overflow is for signed types, but vector lanes
665 // are unsigned, so add a bit.
666 max_bits++;
667 if (mul_would_overflow(max_bits, stores.size(), lanes)) {
668 return Stmt();
669 }
670
671 Type t = store->value.type();
672 Expr base;
673 std::vector<Expr> args(stores.size());
674 std::vector<Expr> predicates(stores.size());
675
676 int min_offset = 0;
677 std::vector<int> offsets(stores.size());
678
679 std::string load_name;
680 Buffer<> load_image;
681 Parameter load_param;
682 for (size_t i = 0; i < stores.size(); ++i) {
683 const Ramp *ri = stores[i].as<Store>()->index.as<Ramp>();
684 internal_assert(ri);
685
686 // Mismatched store vector laness.
687 if (ri->lanes != lanes) {
688 return Stmt();
689 }
690
691 Expr diff = simplify(ri->base - r0->base);
692 const int64_t *offs = as_const_int(diff);
693
694 // Difference between bases is not constant.
695 if (!offs) {
696 return Stmt();
697 }
698
699 offsets[i] = *offs;
700 if (*offs < min_offset) {
701 min_offset = *offs;
702 }
703 }
704
705 // Gather the args for interleaving.
706 for (size_t i = 0; i < stores.size(); ++i) {
707 int j = offsets[i] - min_offset;
708 if (j == 0) {
709 base = stores[i].as<Store>()->index.as<Ramp>()->base;
710 }
711
712 // The offset is not between zero and the stride.
713 if (j < 0 || (size_t)j >= stores.size()) {
714 return Stmt();
715 }
716
717 // We already have a store for this offset.
718 if (args[j].defined()) {
719 return Stmt();
720 }
721
722 args[j] = stores[i].as<Store>()->value;
723 predicates[j] = stores[i].as<Store>()->predicate;
724 }
725
726 // One of the stores should have had the minimum offset.
727 internal_assert(base.defined());
728
729 // Generate a single interleaving store.
730 t = t.with_lanes(lanes * stores.size());
731 Expr index = Ramp::make(base, make_one(base.type()), t.lanes());
732 Expr value = Shuffle::make_interleave(args);
733 Expr predicate = Shuffle::make_interleave(predicates);
734 Stmt new_store = Store::make(store->name, value, index, store->param, predicate, ModulusRemainder());
735
736 // Continue recursively into the stuff that
737 // collect_strided_stores didn't collect.
738 Stmt stmt = Block::make(new_store, mutate(rest));
739
740 // Rewrap the let statements we pulled off.
741 while (!let_stmts.empty()) {
742 const LetStmt *let = let_stmts.back().as<LetStmt>();
743 stmt = LetStmt::make(let->name, let->value, stmt);
744 let_stmts.pop_back();
745 }
746
747 // Success!
748 return stmt;
749 }
750
751 Stmt visit(const Block *op) override {
752 Stmt s = gather_stores(op);
753 if (s.defined()) {
754 return s;
755 } else {
756 Stmt first = mutate(op->first);
757 Stmt rest = mutate(op->rest);
758 if (first.same_as(op->first) && rest.same_as(op->rest)) {
759 return op;
760 } else {
761 return Block::make(first, rest);
762 }
763 }
764 }
765
766public:
767 Interleaver() = default;
768};

777void check(Expr a, const Expr &even, const Expr &odd) {
778 a = simplify(a);
