34const int64_t *as_const_int_or_uint(const Expr &e) {

65ConstantInterval to_interval(Monotonic m) {
66 switch (m) {
67 case Monotonic::Constant:
68 return ConstantInterval::single_point(0);
69 case Monotonic::Increasing:
70 return ConstantInterval::bounded_below(0);
71 case Monotonic::Decreasing:
72 return ConstantInterval::bounded_above(0);
73 case Monotonic::Unknown:
74 return ConstantInterval::everything();
75 }
76 return ConstantInterval::everything();
77}

101// Helpers for doing arithmetic on ConstantIntervals that avoid generating
102// expressions of pos_inf/neg_inf.
103ConstantInterval add(const ConstantInterval &a, const ConstantInterval &b) {
104 ConstantInterval result;
105 result.min_defined = a.has_lower_bound() && b.has_lower_bound();
106 result.max_defined = a.has_upper_bound() && b.has_upper_bound();
107 if (result.has_lower_bound()) {
108 result.min = a.min + b.min;
109 }
110 if (result.has_upper_bound()) {
111 result.max = a.max + b.max;
112 }
113 return result;
114}

120ConstantInterval negate(const ConstantInterval &r) {
121 ConstantInterval result;
122 result.min_defined = r.has_upper_bound();
123 result.min = r.has_upper_bound() ? -r.max : 0;
124 result.max_defined = r.has_lower_bound();
125 result.max = r.has_lower_bound() ? -r.min : 0;
126 return result;
127}

137ConstantInterval multiply(const ConstantInterval &a, int64_t b) {
138 ConstantInterval result(a);
139 if (b < 0) {
140 result = negate(result);
141 b = -b;
142 }
143 if (result.has_lower_bound()) {
144 result.min *= b;
145 }
146 if (result.has_upper_bound()) {
147 result.max *= b;
148 }
149 return result;
150}

152ConstantInterval multiply(const ConstantInterval &a, const Expr &b) {
153 if (const int64_t *bi = as_const_int_or_uint(b)) {
154 return multiply(a, *bi);
155 }
156 return ConstantInterval::everything();

159ConstantInterval multiply(const ConstantInterval &a, const ConstantInterval &b) {
160 int64_t bounds[4];
161 int64_t *bounds_begin = &bounds[0];
162 int64_t *bounds_end = &bounds[0];
163 if (a.has_lower_bound() && b.has_lower_bound()) {
164 *bounds_end++ = a.min * b.min;
165 }
166 if (a.has_lower_bound() && b.has_upper_bound()) {
167 *bounds_end++ = a.min * b.max;
168 }
169 if (a.has_upper_bound() && b.has_lower_bound()) {
170 *bounds_end++ = a.max * b.min;
171 }
172 if (a.has_upper_bound() && b.has_upper_bound()) {
173 *bounds_end++ = a.max * b.max;
174 }
175 if (bounds_begin != bounds_end) {
176 ConstantInterval result = {
177 *std::min_element(bounds_begin, bounds_end),
178 *std::max_element(bounds_begin, bounds_end),
179 };
180 // There *must* be a better way than this... Even
181 // cutting half the cases with swapping isn't that much help.
182 if (!a.has_lower_bound()) {
183 if (may_be_negative(b)) result.max_defined = false; // NOLINT
184 if (may_be_positive(b)) result.min_defined = false; // NOLINT
185 }
186 if (!a.has_upper_bound()) {
187 if (may_be_negative(b)) result.min_defined = false; // NOLINT
188 if (may_be_positive(b)) result.max_defined = false; // NOLINT
189 }
190 if (!b.has_lower_bound()) {
191 if (may_be_negative(a)) result.max_defined = false; // NOLINT
192 if (may_be_positive(a)) result.min_defined = false; // NOLINT
193 }
194 if (!b.has_upper_bound()) {
195 if (may_be_negative(a)) result.min_defined = false; // NOLINT
196 if (may_be_positive(a)) result.max_defined = false; // NOLINT
197 }
198 return result;
199 } else {
200 return ConstantInterval::everything();
201 }
202}

204ConstantInterval divide(const ConstantInterval &a, int64_t b) {
205 ConstantInterval result(a);
206 if (b < 0) {
207 result = negate(result);
208 b = -b;
209 }
210 if (result.has_lower_bound()) {
211 result.min = div_imp(result.min, b);
212 }
213 if (result.has_upper_bound()) {
214 result.max = div_imp(result.max + b - 1, b);
215 }
216 return result;
217}

288 void visit(const Mul *op) override {
289 if (op->type.is_scalar()) {

345 a.accept(this);

367 void visit_lt(const Expr &a, const Expr &b) {
368 a.accept(this);
369 ConstantInterval ra = result;
370 b.accept(this);
371 ConstantInterval rb = result;
372 result = unify(negate(ra), rb);
373 // If the result is bounded, limit it to [-1, 1]. The largest
374 // difference possible is flipping from true to false or false
375 // to true.
376 if (result.has_lower_bound()) {
377 result.min = std::min<int64_t>(std::max<int64_t>(result.min, -1), 1);
378 }
379 if (result.has_upper_bound()) {
380 result.max = std::min<int64_t>(std::max<int64_t>(result.max, -1), 1);
381 }
382 }

421 void visit(const Select *op) override {
422 // The result is the unified bounds, added to the "bump" that happens when switching from true to false.
423 if (op->type.is_scalar()) {
424 op->condition.accept(this);
425 ConstantInterval rcond = result;
426
427 op->true_value.accept(this);
428 ConstantInterval ra = result;
429 op->false_value.accept(this);
430 ConstantInterval rb = result;
431 ConstantInterval unified = unify(ra, rb);
432
433 // TODO: How to handle unsigned values?
434 Expr delta = simplify(op->true_value - op->false_value);
435
436 Interval delta_bounds = find_constant_bounds(delta, bounds);
437 ConstantInterval adjusted_delta;
438 // TODO: Maybe we can do something with one-sided intervals?
439 if (delta_bounds.is_bounded()) {
440 ConstantInterval delta_low = multiply(rcond, delta_bounds.min);
441 ConstantInterval delta_high = multiply(rcond, delta_bounds.max);
442 adjusted_delta = ConstantInterval::make_union(delta_low, delta_high);
443 } else {
444 delta.accept(this);
445 ConstantInterval rdelta = result;
446 adjusted_delta = multiply(rcond, rdelta);
447 }
448
449 result = add(unified, adjusted_delta);
450 } else {
451 result = ConstantInterval::everything();
452 }
453 }

471 void visit(const Call *op) override {
472 // Some functions are known to be monotonic
473 if (Call::as_tag(op) ||
474 op->is_intrinsic(Call::return_second)) {
475 op->args.back().accept(this);
476 return;
477 }
478
479 if (op->is_intrinsic(Call::unsafe_promise_clamped) ||
480 op->is_intrinsic(Call::promise_clamped)) {
481 op->args[0].accept(this);
482 return;
483 }
484
485 if (op->is_intrinsic(Call::require)) {
486 // require() returns the value of the second arg in all non-failure cases
487 op->args[1].accept(this);
488 return;
489 }
490
491 if (!op->is_pure() || !is_constant(result)) {
492 // Even with constant args, the result could vary from one loop iteration to the next.
493 result = ConstantInterval::everything();
494 return;
495 }
496
497 for (size_t i = 0; i < op->args.size(); i++) {
498 op->args[i].accept(this);
499 if (!is_constant(result)) {
500 // One of the args is not constant.
501 result = ConstantInterval::everything();
502 return;
503 }
504 }
505 result = ConstantInterval::single_point(0);
506 }

508 void visit(const Let *op) override {
509 op->value.accept(this);
510
511 ScopedBinding<Interval> bounds_binding(bounds, op->name, find_constant_bounds(op->value, bounds));
512
513 if (is_constant(result)) {
514 // No point pushing it if it's constant w.r.t the var,
515 // because unknown variables are treated as constant.
516 op->body.accept(this);
517 } else {
518 ScopedBinding<ConstantInterval> scope_binding(scope, op->name, result);
519 op->body.accept(this);
520 }
521 }

534 void visit(const VectorReduce *op) override {
535 op->value.accept(this);
536 switch (op->op) {
537 case VectorReduce::Add:
538 case VectorReduce::SaturatingAdd:
539 result = multiply(result, op->value.type().lanes() / op->type.lanes());
540 break;
541 case VectorReduce::Min:
542 case VectorReduce::Max:
543 // These reductions are monotonic in the arg
544 break;
545 case VectorReduce::Mul:
546 case VectorReduce::And:
547 case VectorReduce::Or:
548 // These ones are not
549 if (!is_constant(result)) {
550 result = ConstantInterval::everything();
551 }
552 }

646Monotonic is_monotonic(const Expr &e, const std::string &var, const Scope<Monotonic> &scope) {
647 if (!e.defined()) {
648 return Monotonic::Unknown;
649 }
650 Scope<ConstantInterval> intervals_scope;
651 for (Scope<Monotonic>::const_iterator i = scope.cbegin(); i != scope.cend(); ++i) {
652 intervals_scope.push(i.name(), to_interval(i.value()));
653 }
654 return is_monotonic(e, var, intervals_scope);
655}

679void is_monotonic_test() {
680
681 Expr x = Variable::make(Int(32), "x");
682 Expr y = Variable::make(Int(32), "y");
683
684 check_increasing(x);
685 check_increasing(x + 4);
686 check_increasing(x + y);
687 check_increasing(x * 4);
688 check_increasing(x / 4);
689 check_increasing(min(x + 4, y + 4));
690 check_increasing(max(x + y, x - y));
691 check_increasing(x >= y);
692 check_increasing(x > y);
693
694 check_decreasing(-x);
695 check_decreasing(x * -4);
696 check_decreasing(x / -4);
697 check_decreasing(y - x);
698 check_decreasing(x < y);
699 check_decreasing(x <= y);
700
701 check_unknown(x == y);
702 check_unknown(x != y);
703 check_increasing(y <= x);
704 check_increasing(y < x);
705 check_decreasing(x <= y);
706 check_decreasing(x < y);
707 check_unknown(x * y);
708
709 // Not constant despite having constant args, because there's a side-effect.
710 check_unknown(Call::make(Int(32), "foo", {Expr(3)}, Call::Extern));
711
712 check_increasing(select(y == 2, x, x + 4));
713 check_decreasing(select(y == 2, -x, x * -4));
714
715 check_unknown(select(x > 2, x - 2, x));
716 check_unknown(select(x < 2, x, x - 2));
717 check_unknown(select(x > 2, -x + 2, -x));
718 check_unknown(select(x < 2, -x, -x + 2));
719 check_increasing(select(x > 2, x - 1, x));
720 check_increasing(select(x < 2, x, x - 1));
721 check_decreasing(select(x > 2, -x + 1, -x));
722 check_decreasing(select(x < 2, -x, -x + 1));
723
724 check_unknown(select(x < 2, x, x - 5));
725 check_unknown(select(x > 2, x - 5, x));
726
727 check_constant(y);
728
729 check_increasing(select(x < 17, y, y + 1));
730 check_increasing(select(x > 17, y, y - 1));
731 check_decreasing(select(x < 17, y, y - 1));
732 check_decreasing(select(x > 17, y, y + 1));
733
734 check_increasing(select(x % 2 == 0, x + 3, x + 3));
735
736 check_constant(select(y > 3, y + 23, y - 65));
737
738 check_decreasing(select(2 <= x, 0, 1));
739 check_increasing(select(2 <= x, 0, 1) + x);
740 check_decreasing(-min(x, 16));
741
742 check_unknown(select(0 < x, max(min(x, 4), 3), 4));
743
744 std::cout << "is_monotonic test passed" << std::endl;
745}
