21/* Find all the externally referenced buffers in a stmt */
22class FindBuffers : public IRGraphVisitor {
23public:
24 struct Result {
25 Buffer<> image;
26 Parameter param;
27 Type type;
28 int dimensions{0};
29 bool used_on_host{false};
30 };
31
32 map<string, Result> buffers;
33 bool in_device_loop = false;
34
35 using IRGraphVisitor::visit;
36
37 void visit(const For *op) override {
38 op->min.accept(this);
39 op->extent.accept(this);
40 bool old = in_device_loop;
41 if (op->device_api != DeviceAPI::None &&
42 op->device_api != DeviceAPI::Host) {
43 in_device_loop = true;
44 }
45 op->body.accept(this);
46 in_device_loop = old;
47 }
48
49 void visit(const Call *op) override {
50 IRGraphVisitor::visit(op);
51 if (op->image.defined()) {
52 Result &r = buffers[op->name];
53 r.image = op->image;
54 r.type = op->type.element_of();
55 r.dimensions = (int)op->args.size();
56 r.used_on_host = r.used_on_host || (!in_device_loop);
57 } else if (op->param.defined()) {
58 Result &r = buffers[op->name];
59 r.param = op->param;
60 r.type = op->type.element_of();
61 r.dimensions = (int)op->args.size();
62 r.used_on_host = r.used_on_host || (!in_device_loop);
63 }
64 }
65
66 void visit(const Provide *op) override {
67 IRGraphVisitor::visit(op);
68 if (op->values.size() == 1) {
69 auto it = buffers.find(op->name);
70 if (it != buffers.end() && !in_device_loop) {
71 it->second.used_on_host = true;
72 }
73 } else {
74 for (size_t i = 0; i < op->values.size(); i++) {
75 string name = op->name + "." + std::to_string(i);
76 auto it = buffers.find(name);
77 if (it != buffers.end() && !in_device_loop) {
78 it->second.used_on_host = true;
79 }
80 }
81 }
82 }
83
84 void visit(const Variable *op) override {
85 if (op->param.defined() &&
86 op->param.is_buffer() &&
87 buffers.find(op->param.name()) == buffers.end()) {
88 Result r;
89 r.param = op->param;
90 r.type = op->param.type();
91 r.dimensions = op->param.dimensions();
92 r.used_on_host = false;
93 buffers[op->param.name()] = r;
94 } else if (op->reduction_domain.defined()) {
95 // The bounds of reduction domains are not yet defined,
96 // and they may be the only reference to some parameters.
97 op->reduction_domain.accept(this);
98 }
99 }
100};

102class TrimStmtToPartsThatAccessBuffers : public IRMutator {
103 bool touches_buffer = false;
104 const map<string, FindBuffers::Result> &buffers;
105
106 using IRMutator::visit;
107
108 Expr visit(const Call *op) override {
109 touches_buffer |= (buffers.count(op->name) > 0);
110 return IRMutator::visit(op);
111 }
112 Stmt visit(const Provide *op) override {
113 touches_buffer |= (buffers.find(op->name) != buffers.end());
114 return IRMutator::visit(op);
115 }
116 Expr visit(const Variable *op) override {
117 if (op->type.is_handle() && op->param.defined() && op->param.is_buffer()) {
118 touches_buffer |= (buffers.find(op->param.name()) != buffers.end());
119 }
120 return IRMutator::visit(op);
121 }
122
123 Stmt visit(const Block *op) override {
124 bool old_touches_buffer = touches_buffer;
125 touches_buffer = false;
126 Stmt first = mutate(op->first);
127 old_touches_buffer |= touches_buffer;
128 if (!touches_buffer) {
129 first = Evaluate::make(0);
130 }
131 touches_buffer = false;
132 Stmt rest = mutate(op->rest);
133 old_touches_buffer |= touches_buffer;
134 if (!touches_buffer) {
135 rest = Evaluate::make(0);
136 }
137 touches_buffer = old_touches_buffer;
138 return Block::make(first, rest);
139 }
140
141public:
142 TrimStmtToPartsThatAccessBuffers(const map<string, FindBuffers::Result> &bufs)
143 : buffers(bufs) {
144 }
145};

147Stmt add_image_checks_inner(Stmt s,
148 const vector<Function> &outputs,
149 const Target &t,
150 const vector<string> &order,
151 const map<string, Function> &env,
152 const FuncValueBounds &fb,
153 bool will_inject_host_copies) {
154
155 bool no_asserts = t.has_feature(Target::NoAsserts);
156 bool no_bounds_query = t.has_feature(Target::NoBoundsQuery);
157
158 // First hunt for all the referenced buffers
159 FindBuffers finder;
160 map<string, FindBuffers::Result> &bufs = finder.buffers;
161
162 // Add the output buffer(s).
163 for (const Function &f : outputs) {
164 for (size_t i = 0; i < f.values().size(); i++) {
165 FindBuffers::Result output_buffer;
166 output_buffer.type = f.values()[i].type();
167 output_buffer.param = f.output_buffers()[i];
168 output_buffer.dimensions = f.dimensions();
169 if (f.values().size() > 1) {
170 bufs[f.name() + '.' + std::to_string(i)] = output_buffer;
171 } else {
172 bufs[f.name()] = output_buffer;
173 }
174 }
175 }
176
177 // Add the input buffer(s) and annotate which output buffers are
178 // used on host.
179 s.accept(&finder);
180
181 Scope<Interval> empty_scope;
182 Stmt sub_stmt = TrimStmtToPartsThatAccessBuffers(bufs).mutate(s);
183 map<string, Box> boxes = boxes_touched(sub_stmt, empty_scope, fb);
184
185 // Now iterate through all the buffers, creating a list of lets
186 // and a list of asserts.
187 vector<pair<string, Expr>> lets_overflow;
188 vector<pair<string, Expr>> lets_required;
189 vector<pair<string, Expr>> lets_constrained;
190 vector<pair<string, Expr>> lets_proposed;
191 vector<Stmt> dims_no_overflow_asserts;
192 vector<Stmt> asserts_required;
193 vector<Stmt> asserts_constrained;
194 vector<Stmt> asserts_proposed;
195 vector<Stmt> asserts_type_checks;
196 vector<Stmt> asserts_host_alignment;
197 vector<Stmt> asserts_host_non_null;
198 vector<Stmt> asserts_device_not_dirty;
199 vector<Stmt> buffer_rewrites;
200 vector<Stmt> msan_checks;
201
202 // Inject the code that conditionally returns if we're in inference mode
203 Expr maybe_return_condition = const_false();
204
205 // We're also going to apply the constraints to the required min
206 // and extent. To do this we have to substitute all references to
207 // the actual sizes of the input images in the constraints with
208 // references to the required sizes.
209 map<string, Expr> replace_with_required;
210
211 for (const pair<const string, FindBuffers::Result> &buf : bufs) {
212 const string &name = buf.first;
213
214 for (int i = 0; i < buf.second.dimensions; i++) {
215 string dim = std::to_string(i);
216
217 Expr min_required = Variable::make(Int(32), name + ".min." + dim + ".required");
218 replace_with_required[name + ".min." + dim] = min_required;
219
220 Expr extent_required = Variable::make(Int(32), name + ".extent." + dim + ".required");
221 replace_with_required[name + ".extent." + dim] = simplify(extent_required);
222
223 Expr stride_required = Variable::make(Int(32), name + ".stride." + dim + ".required");
224 replace_with_required[name + ".stride." + dim] = stride_required;
225 }
226 }
227
228 // We also want to build a map that lets us replace values passed
229 // in with the constrained version. This is applied to the rest of
230 // the lowered pipeline to take advantage of the constraints,
231 // e.g. for constant folding.
232 map<string, Expr> replace_with_constrained;
233
234 for (pair<const string, FindBuffers::Result> &buf : bufs) {
235 const string &name = buf.first;
236 Buffer<> &image = buf.second.image;
237 Parameter &param = buf.second.param;
238 Type type = buf.second.type;
239 int dimensions = buf.second.dimensions;
240 bool used_on_host = buf.second.used_on_host;
241
242 // Detect if this is one of the outputs of a multi-output pipeline.
243 bool is_output_buffer = false;
244 bool is_secondary_output_buffer = false;
245 string buffer_name = name;
246 for (const Function &f : outputs) {
247 for (size_t i = 0; i < f.output_buffers().size(); i++) {
248 if (param.defined() &&
249 param.same_as(f.output_buffers()[i])) {
250 is_output_buffer = true;
251 // If we're one of multiple output buffers, we should use the
252 // region inferred for the func in general.
253 buffer_name = f.name();
254 if (i > 0) {
255 is_secondary_output_buffer = true;
256 }
257 }
258 }
259 }
260
261 Box touched = boxes[buffer_name];
262 internal_assert(touched.empty() || (int)(touched.size()) == dimensions);
263
264 // The buffer may be used in one or more extern stage. If so we need to
265 // expand the box touched to include the results of the
266 // top-level bounds query calls to those extern stages.
267 if (param.defined()) {
268 // Find the extern users.
269 vector<string> extern_users;
270 for (size_t i = 0; i < order.size(); i++) {
271 Function f = env.find(order[i])->second;
272 if (f.has_extern_definition() &&
273 !f.extern_definition_proxy_expr().defined()) {
274 const vector<ExternFuncArgument> &args = f.extern_arguments();
275 for (size_t j = 0; j < args.size(); j++) {
276 if ((args[j].image_param.defined() &&
277 args[j].image_param.name() == param.name()) ||
278 (args[j].buffer.defined() &&
279 args[j].buffer.name() == param.name())) {
280 extern_users.push_back(order[i]);
281 }
282 }
283 }
284 }
285
286 // Expand the box by the result of the bounds query from each.
287 for (size_t i = 0; i < extern_users.size(); i++) {
288 const string &extern_user = extern_users[i];
289 Box query_box;
290 Expr query_buf = Variable::make(type_of<struct halide_buffer_t *>(),
291 param.name() + ".bounds_query." + extern_user);
292 for (int j = 0; j < dimensions; j++) {
293 Expr min = Call::make(Int(32), Call::buffer_get_min,
294 {query_buf, j}, Call::Extern);
295 Expr max = Call::make(Int(32), Call::buffer_get_max,
296 {query_buf, j}, Call::Extern);
297 query_box.push_back(Interval(min, max));
298 }
299 merge_boxes(touched, query_box);
300 }
301 }
302
303 ReductionDomain rdom;
304
305 // An expression returning whether or not we're in inference mode
306 string buf_name = name + ".buffer";
307 Expr handle = Variable::make(type_of<halide_buffer_t *>(), buf_name,
308 image, param, rdom);
309 Expr inference_mode = Call::make(Bool(), Call::buffer_is_bounds_query,
310 {handle}, Call::Extern);
311 maybe_return_condition = maybe_return_condition || inference_mode;
312
313 // Come up with a name to refer to this buffer in the error messages
314 string error_name = (is_output_buffer ? "Output" : "Input");
315 error_name += " buffer " + name;
316
317 if (!is_output_buffer && t.has_feature(Target::MSAN)) {
318 Expr buffer = Variable::make(type_of<struct halide_buffer_t *>(), buf_name);
319 Stmt check_contents = Evaluate::make(
320 Call::make(Int(32), "halide_msan_check_buffer_is_initialized", {buffer, Expr(buf_name)}, Call::Extern));
321 msan_checks.push_back(check_contents);
322 }
323
324 // Check the type matches the internally-understood type
325 {
326 string type_name = name + ".type";
327 Expr type_var = Variable::make(UInt(32), type_name, image, param, rdom);
328 uint32_t correct_type_bits = ((halide_type_t)type).as_u32();
329 Expr correct_type_expr = make_const(UInt(32), correct_type_bits);
330 Expr error = Call::make(Int(32), "halide_error_bad_type",
331 {error_name, type_var, correct_type_expr},
332 Call::Extern);
333 Stmt type_check = AssertStmt::make(type_var == correct_type_expr, error);
334 asserts_type_checks.push_back(type_check);
335 }
336
337 // Check the dimensions matches the internally-understood dimensions
338 {
339 string dimensions_name = name + ".dimensions";
340 Expr dimensions_given = Variable::make(Int(32), dimensions_name, image, param, rdom);
341 Expr error = Call::make(Int(32), "halide_error_bad_dimensions",
342 {error_name,
343 dimensions_given, make_const(Int(32), dimensions)},
344 Call::Extern);
345 asserts_type_checks.push_back(
346 AssertStmt::make(dimensions_given == dimensions, error));
347 }
348
349 if (touched.maybe_unused()) {
350 debug(3) << "Image " << name << " is only used when " << touched.used << "\n";
351 }
352
353 // Check that the region passed in (after applying constraints) is within the region used
354 debug(3) << "In image " << name << " region touched is:\n";
355
356 for (int j = 0; j < dimensions; j++) {
357 string dim = std::to_string(j);
358 string actual_min_name = name + ".min." + dim;
359 string actual_extent_name = name + ".extent." + dim;
360 string actual_stride_name = name + ".stride." + dim;
361 Expr actual_min = Variable::make(Int(32), actual_min_name, image, param, rdom);
362 Expr actual_extent = Variable::make(Int(32), actual_extent_name, image, param, rdom);
363 Expr actual_stride = Variable::make(Int(32), actual_stride_name, image, param, rdom);
364
365 if (!touched.empty() && !touched[j].is_bounded()) {
366 user_error << "Buffer " << name
367 << " may be accessed in an unbounded way in dimension "
368 << j << "\n";
369 }
370
371 Expr min_required = touched.empty() ? actual_min : touched[j].min;
372 Expr extent_required = touched.empty() ? actual_extent : (touched[j].max + 1 - touched[j].min);
373
374 if (touched.maybe_unused()) {
375 min_required = select(touched.used, min_required, actual_min);
376 extent_required = select(touched.used, extent_required, actual_extent);
377 }
378
379 string min_required_name = name + ".min." + dim + ".required";
380 string extent_required_name = name + ".extent." + dim + ".required";
381
382 Expr min_required_var = Variable::make(Int(32), min_required_name);
383 Expr extent_required_var = Variable::make(Int(32), extent_required_name);
384
385 lets_required.emplace_back(extent_required_name, extent_required);
386 lets_required.emplace_back(min_required_name, min_required);
387
388 Expr actual_max = actual_min + actual_extent - 1;
389 Expr max_required = min_required_var + extent_required_var - 1;
390
391 if (touched.maybe_unused()) {
392 max_required = select(touched.used, max_required, actual_max);
393 }
394
395 Expr oob_condition = actual_min <= min_required_var && actual_max >= max_required;
396
397 Expr oob_error = Call::make(Int(32), "halide_error_access_out_of_bounds",
398 {error_name, j, min_required_var, max_required, actual_min, actual_max},
399 Call::Extern);
400
401 asserts_required.push_back(AssertStmt::make(oob_condition, oob_error));
402
403 // Come up with a required stride to use in bounds
404 // inference mode. We don't assert it. It's just used to
405 // apply the constraints to to come up with a proposed
406 // stride. Strides actually passed in may not be in this
407 // order (e.g if storage is swizzled relative to dimension
408 // order).
409 Expr stride_required;
410 if (j == 0) {
411 stride_required = 1;
412 } else {
413 string last_dim = std::to_string(j - 1);
414 stride_required = (Variable::make(Int(32), name + ".stride." + last_dim + ".required") *
415 Variable::make(Int(32), name + ".extent." + last_dim + ".required"));
416 }
417 lets_required.emplace_back(name + ".stride." + dim + ".required", stride_required);
418
419 // On 32-bit systems, insert checks to make sure the total
420 // size of all input and output buffers is <= 2^31 - 1.
421 // And that no product of extents overflows 2^31 - 1. This
422 // second test is likely only needed if a fuse directive
423 // is used in the schedule to combine multiple extents,
424 // but it is here for extra safety. On 64-bit targets with the
425 // LargeBuffers feature, the maximum size is 2^63 - 1.
426 Expr max_size = make_const(UInt(64), t.maximum_buffer_size());
427 Expr max_extent = make_const(UInt(64), 0x7fffffff);
428 Expr actual_size = abs(cast<int64_t>(actual_extent) * actual_stride);
429 Expr allocation_size_error = Call::make(Int(32), "halide_error_buffer_allocation_too_large",
430 {name, actual_size, max_size}, Call::Extern);
431 Stmt check = AssertStmt::make(actual_size <= max_size, allocation_size_error);
432 dims_no_overflow_asserts.push_back(check);
433
434 // Don't repeat extents check for secondary buffers as extents must be the same as for the first one.
435 if (!is_secondary_output_buffer) {
436 if (j == 0) {
437 lets_overflow.emplace_back(name + ".total_extent." + dim, cast<int64_t>(actual_extent));
438 } else {
439 max_size = cast<int64_t>(max_size);
440 Expr last_dim = Variable::make(Int(64), name + ".total_extent." + std::to_string(j - 1));
441 Expr this_dim = actual_extent * last_dim;
442 Expr this_dim_var = Variable::make(Int(64), name + ".total_extent." + dim);
443 lets_overflow.emplace_back(name + ".total_extent." + dim, this_dim);
444 Expr error = Call::make(Int(32), "halide_error_buffer_extents_too_large",
445 {name, this_dim_var, max_size}, Call::Extern);
446 Stmt check = AssertStmt::make(this_dim_var <= max_size, error);
447 dims_no_overflow_asserts.push_back(check);
448 }
449
450 // It is never legal to have a negative buffer extent.
451 Expr negative_extent_condition = actual_extent >= 0;
452 Expr negative_extent_error = Call::make(Int(32), "halide_error_buffer_extents_negative",
453 {error_name, j, actual_extent}, Call::Extern);
454 asserts_required.push_back(AssertStmt::make(negative_extent_condition, negative_extent_error));
455 }
456 }
457
458 // Create code that mutates the input buffers if we're in bounds inference mode.
459 BufferBuilder builder;
460 builder.buffer_memory = Variable::make(type_of<struct halide_buffer_t *>(), buf_name);
461 builder.shape_memory = Call::make(type_of<struct halide_dimension_t *>(),
462 Call::buffer_get_shape, {builder.buffer_memory},
463 Call::Extern);
464 builder.type = type;
465 builder.dimensions = dimensions;
466 for (int i = 0; i < dimensions; i++) {
467 string dim = std::to_string(i);
468 builder.mins.push_back(Variable::make(Int(32), name + ".min." + dim + ".proposed"));
469 builder.extents.push_back(Variable::make(Int(32), name + ".extent." + dim + ".proposed"));
470 builder.strides.push_back(Variable::make(Int(32), name + ".stride." + dim + ".proposed"));
471 }
472 Stmt rewrite = Evaluate::make(builder.build());
473
474 rewrite = IfThenElse::make(inference_mode, rewrite);
475 buffer_rewrites.push_back(rewrite);
476
477 // Build the constraints tests and proposed sizes.
478 vector<pair<Expr, Expr>> constraints;
479 for (int i = 0; i < dimensions; i++) {
480 string dim = std::to_string(i);
481 string min_name = name + ".min." + dim;
482 string stride_name = name + ".stride." + dim;
483 string extent_name = name + ".extent." + dim;
484
485 Expr stride_constrained, extent_constrained, min_constrained;
486
487 Expr stride_orig = Variable::make(Int(32), stride_name, image, param, rdom);
488 Expr extent_orig = Variable::make(Int(32), extent_name, image, param, rdom);
489 Expr min_orig = Variable::make(Int(32), min_name, image, param, rdom);
490
491 Expr stride_required = Variable::make(Int(32), stride_name + ".required");
492 Expr extent_required = Variable::make(Int(32), extent_name + ".required");
493 Expr min_required = Variable::make(Int(32), min_name + ".required");
494
495 Expr stride_proposed = Variable::make(Int(32), stride_name + ".proposed");
496 Expr extent_proposed = Variable::make(Int(32), extent_name + ".proposed");
497 Expr min_proposed = Variable::make(Int(32), min_name + ".proposed");
498
499 debug(2) << "Injecting constraints for " << name << "." << i << "\n";
500 if (is_secondary_output_buffer) {
501 // For multi-output (Tuple) pipelines, output buffers
502 // beyond the first implicitly have their min and extent
503 // constrained to match the first output.
504
505 if (param.defined()) {
506 user_assert(!param.extent_constraint(i).defined() &&
507 !param.min_constraint(i).defined())
508 << "Can't constrain the min or extent of an output buffer beyond the "
509 << "first. They are implicitly constrained to have the same min and extent "
510 << "as the first output buffer.\n";
511
512 stride_constrained = param.stride_constraint(i);
513 } else if (image.defined() && (int)i < image.dimensions()) {
514 stride_constrained = image.dim(i).stride();
515 }
516
517 std::string min0_name = buffer_name + ".0.min." + dim;
518 if (replace_with_constrained.count(min0_name) > 0) {
519 min_constrained = replace_with_constrained[min0_name];
520 } else {
521 min_constrained = Variable::make(Int(32), min0_name);
522 }
523
524 std::string extent0_name = buffer_name + ".0.extent." + dim;
525 if (replace_with_constrained.count(extent0_name) > 0) {
526 extent_constrained = replace_with_constrained[extent0_name];
527 } else {
528 extent_constrained = Variable::make(Int(32), extent0_name);
529 }
530 } else if (image.defined() && (int)i < image.dimensions()) {
531 stride_constrained = image.dim(i).stride();
532 extent_constrained = image.dim(i).extent();
533 min_constrained = image.dim(i).min();
534 } else if (param.defined()) {
535 stride_constrained = param.stride_constraint(i);
536 extent_constrained = param.extent_constraint(i);
537 min_constrained = param.min_constraint(i);
538 }
539
540 if (stride_constrained.defined()) {
541 // Come up with a suggested stride by passing the
542 // required region through this constraint.
543 constraints.emplace_back(stride_orig, stride_constrained);
544 stride_constrained = substitute(replace_with_required, stride_constrained);
545 lets_proposed.emplace_back(stride_name + ".proposed", stride_constrained);
546 } else {
547 lets_proposed.emplace_back(stride_name + ".proposed", stride_required);
548 }
549
550 if (min_constrained.defined()) {
551 constraints.emplace_back(min_orig, min_constrained);
552 min_constrained = substitute(replace_with_required, min_constrained);
553 lets_proposed.emplace_back(min_name + ".proposed", min_constrained);
554 } else {
555 lets_proposed.emplace_back(min_name + ".proposed", min_required);
556 }
557
558 if (extent_constrained.defined()) {
559 constraints.emplace_back(extent_orig, extent_constrained);
560 extent_constrained = substitute(replace_with_required, extent_constrained);
561 lets_proposed.emplace_back(extent_name + ".proposed", extent_constrained);
562 } else {
563 lets_proposed.emplace_back(extent_name + ".proposed", extent_required);
564 }
565
566 // In bounds inference mode, make sure the proposed
567 // versions still satisfy the constraints.
568 Expr max_proposed = min_proposed + extent_proposed - 1;
569 Expr max_required = min_required + extent_required - 1;
570 Expr check = (min_proposed <= min_required) && (max_proposed >= max_required);
571 Expr error = Call::make(Int(32), "halide_error_constraints_make_required_region_smaller",
572 {error_name, i, min_proposed, max_proposed, min_required, max_required},
573 Call::Extern);
574 asserts_proposed.push_back(AssertStmt::make((!inference_mode) || check, error));
575
576 // stride_required is just a suggestion. It's ok if the
577 // constraints shuffle them around in ways that make it
578 // smaller.
579 /*
580 check = (stride_proposed >= stride_required);
581 error = "Applying the constraints to the required stride made it smaller";
582 asserts_proposed.push_back(AssertStmt::make((!inference_mode) || check, error, vector<Expr>()));
583 */
584 }
585
586 // Assert all the conditions, and set the new values
587 for (size_t i = 0; i < constraints.size(); i++) {
588 Expr var = constraints[i].first;
589 const string &name = var.as<Variable>()->name;
590 Expr constrained_var = Variable::make(Int(32), name + ".constrained");
591
592 std::ostringstream ss;
593 ss << constraints[i].second;
594 string constrained_var_str = ss.str();
595
596 lets_constrained.emplace_back(name + ".constrained", constraints[i].second);
597
598 // Substituting in complex expressions is not typically a good idea
599 if (constraints[i].second.as<Variable>() ||
600 is_const(constraints[i].second)) {
601 replace_with_constrained[name] = constrained_var;
602 }
603
604 Expr error = 0;
605 if (!no_asserts) {
606 error = Call::make(Int(32), "halide_error_constraint_violated",
607 {name, var, constrained_var_str, constrained_var},
608 Call::Extern);
609 }
610
611 // Check the var passed in equals the constrained version (when not in inference mode)
612 asserts_constrained.push_back(AssertStmt::make(var == constrained_var, error));
613 }
614
615 // For the buffers used on host, check the host field is non-null
616 Expr host_ptr = Variable::make(Handle(), name, image, param, ReductionDomain());
617 if (used_on_host) {
618 Expr error = Call::make(Int(32), "halide_error_host_is_null",
619 {error_name}, Call::Extern);
620 Expr check = (host_ptr != make_zero(host_ptr.type()));
621 if (touched.maybe_unused()) {
622 check = !touched.used || check;
623 }
624 asserts_host_non_null.push_back(AssertStmt::make(check, error));
625
626 if (!will_inject_host_copies) {
627 Expr device_dirty = Variable::make(Bool(), name + ".device_dirty",
628 image, param, ReductionDomain());
629
630 Expr error = Call::make(Int(32), "halide_error_device_dirty_with_no_device_support",
631 {error_name}, Call::Extern);
632
633 // If we have no device support, we can't handle
634 // device_dirty, so every buffer touched needs checking.
635 asserts_device_not_dirty.push_back(AssertStmt::make(!device_dirty, error));
636 }
637 }
638
639 // and check alignment of the host field
640 if (param.defined() && param.host_alignment() != param.type().bytes()) {
641 int alignment_required = param.host_alignment();
642 Expr u64t_host_ptr = reinterpret<uint64_t>(host_ptr);
643 Expr align_condition = (u64t_host_ptr % alignment_required) == 0;
644 Expr error = Call::make(Int(32), "halide_error_unaligned_host_ptr",
645 {name, alignment_required}, Call::Extern);
646 asserts_host_alignment.push_back(AssertStmt::make(align_condition, error));
647 }
648 }
649
650 auto prepend_stmts = [&](vector<Stmt> *stmts) {
651 while (!stmts->empty()) {
652 s = Block::make(std::move(stmts->back()), s);
653 stmts->pop_back();
654 }
655 };
656
657 auto prepend_lets = [&](vector<pair<string, Expr>> *lets) {
658 while (!lets->empty()) {
659 auto &p = lets->back();
660 s = LetStmt::make(p.first, std::move(p.second), s);
661 lets->pop_back();
662 }
663 };
664
665 if (!no_asserts) {
666 // Inject the code that checks the host pointers.
667 prepend_stmts(&asserts_host_non_null);
668 prepend_stmts(&asserts_host_alignment);
669 prepend_stmts(&asserts_device_not_dirty);
670 prepend_stmts(&dims_no_overflow_asserts);
671 prepend_lets(&lets_overflow);
672 }
673
674 // Replace uses of the var with the constrained versions in the
675 // rest of the program. We also need to respect the existence of
676 // constrained versions during storage flattening and bounds
677 // inference.
678 s = substitute(replace_with_constrained, s);
679
680 // Now we add a bunch of code to the top of the pipeline. This is
681 // all in reverse order compared to execution, as we incrementally
682 // prepending code.
683
684 // Inject the code that checks the constraints are correct. We
685 // need these regardless of how NoAsserts is set, because they are
686 // what gets Halide to actually exploit the constraint.
687 prepend_stmts(&asserts_constrained);
688
689 if (!no_asserts) {
690 prepend_stmts(&asserts_required);
691 prepend_stmts(&asserts_type_checks);
692 }
693
694 // Inject the code that returns early for inference mode.
695 if (!no_bounds_query) {
696 s = IfThenElse::make(!maybe_return_condition, s);
697 prepend_stmts(&buffer_rewrites);
698 }
699
700 if (!no_asserts) {
701 prepend_stmts(&asserts_proposed);
702 }
703
704 // Inject the code that defines the proposed sizes.
705 prepend_lets(&lets_proposed);
706
707 // Inject the code that defines the constrained sizes.
708 prepend_lets(&lets_constrained);
709
710 // Inject the code that defines the required sizes produced by bounds inference.
711 prepend_lets(&lets_required);
712
713 // Inject the code that checks that does msan checks. (Note that this ignores no_asserts.)
714 prepend_stmts(&msan_checks);
715
716 return s;
717}
