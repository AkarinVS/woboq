119void CodeGen_WebAssembly::init_module() {
120 CodeGen_Posix::init_module();
121
122 for (const WasmIntrinsic &i : intrinsic_defs) {
123 if (i.feature != Target::FeatureEnd && !target.has_feature(i.feature)) {
124 continue;
125 }
126
127 Type ret_type = i.ret_type;
128 vector<Type> arg_types;
129 arg_types.reserve(max_intrinsic_args);
130 for (halide_type_t i : i.arg_types) {
131 if (i.bits == 0) {
132 break;
133 }
134 arg_types.emplace_back(i);
135 }
136
137 auto *fn = declare_intrin_overload(i.name, ret_type, i.intrin_name, std::move(arg_types));
138 fn->addFnAttr(llvm::Attribute::ReadNone);
139 fn->addFnAttr(llvm::Attribute::NoUnwind);
140 }

143void CodeGen_WebAssembly::visit(const Cast *op) {
144#if LLVM_VERSION >= 130
145 struct Pattern {
146 std::string intrin; ///< Name of the intrinsic
147 Expr pattern; ///< The pattern to match against
148 Target::Feature required_feature;
149 };
150
151 // clang-format off
152 static const Pattern patterns[] = {
153 {"q15mulr_sat_s", i16_sat(rounding_shift_right(widening_mul(wild_i16x_, wild_i16x_), u16(15))), Target::WasmSimd128},
154 {"saturating_narrow", i8_sat(wild_i16x_), Target::WasmSimd128},
155 {"saturating_narrow", u8_sat(wild_i16x_), Target::WasmSimd128},
156 {"saturating_narrow", i16_sat(wild_i32x_), Target::WasmSimd128},
157 {"saturating_narrow", u16_sat(wild_i32x_), Target::WasmSimd128},
158 {"int_to_double", f64(wild_i32x_), Target::WasmSimd128},
159 {"int_to_double", f64(wild_u32x_), Target::WasmSimd128},
160 {"float_to_double", f64(wild_f32x_), Target::WasmSimd128},
161 };
162 // clang-format on
163
164 if (op->type.is_vector()) {
165 std::vector<Expr> matches;
166 for (const Pattern &p : patterns) {
167 if (!target.has_feature(p.required_feature)) {
168 continue;
169 }
170 if (expr_match(p.pattern, op, matches)) {
171 value = call_overloaded_intrin(op->type, p.intrin, matches);
172 if (value) {
173 return;
174 }
175 }
176 }
177 }
178#endif // LLVM_VERSION >= 130
179
180 CodeGen_Posix::visit(op);
181}

183void CodeGen_WebAssembly::codegen_vector_reduce(const VectorReduce *op, const Expr &init) {
184#if LLVM_VERSION >= 130
185 struct Pattern {
186 VectorReduce::Operator reduce_op;
187 int factor;
188 Expr pattern;
189 const char *intrin;
190 Target::Feature required_feature;
191 };
192 // clang-format off
193 static const Pattern patterns[] = {
194 {VectorReduce::Add, 2, i16(wild_i8x_), "pairwise_widening_add", Target::WasmSimd128},
195 {VectorReduce::Add, 2, u16(wild_u8x_), "pairwise_widening_add", Target::WasmSimd128},
196 {VectorReduce::Add, 2, i16(wild_u8x_), "pairwise_widening_add", Target::WasmSimd128},
197
198 {VectorReduce::Add, 2, i32(wild_i16x_), "pairwise_widening_add", Target::WasmSimd128},
199 {VectorReduce::Add, 2, u32(wild_u16x_), "pairwise_widening_add", Target::WasmSimd128},
200 {VectorReduce::Add, 2, i32(wild_u16x_), "pairwise_widening_add", Target::WasmSimd128},
201
202 {VectorReduce::Add, 2, i32(widening_mul(wild_i16x_, wild_i16x_)), "dot_product", Target::WasmSimd128},
203 };
204 // clang-format on
205
206 // Other values will be added soon, so this switch isn't actually pointless
207 using ValuePtr = llvm::Value *;
208 std::function<ValuePtr(ValuePtr, ValuePtr)> binop = nullptr;
209 switch (op->op) {
210 case VectorReduce::Add:
211 binop = [this](ValuePtr x, ValuePtr y) -> ValuePtr { return this->builder->CreateAdd(x, y); };
212 break;
213 default:
214 break;
215 }
216
217 const int factor = op->value.type().lanes() / op->type.lanes();
218 vector<Expr> matches;
219 for (const Pattern &p : patterns) {
220 if (op->op != p.reduce_op || (factor % p.factor) != 0) {
221 continue;
222 }
223 if (!target.has_feature(p.required_feature)) {
224 continue;
225 }
226 if (expr_match(p.pattern, op->value, matches)) {
227 if (factor != p.factor) {
228 Expr equiv = VectorReduce::make(op->op, op->value, op->value.type().lanes() / p.factor);
229 equiv = VectorReduce::make(op->op, equiv, op->type.lanes());
230 codegen_vector_reduce(equiv.as<VectorReduce>(), init);
231 return;
232 }
233
234 if (const Shuffle *s = matches[0].as<Shuffle>()) {
235 if (s->is_broadcast() && matches.size() == 2) {
236 // LLVM wants the broadcast as the second operand for the broadcasting
237 // variant of udot/sdot.
238 std::swap(matches[0], matches[1]);
239 }
240 }
241 value = call_overloaded_intrin(op->type, p.intrin, matches);
242 if (value) {
243 if (init.defined()) {
244 internal_assert(binop != nullptr) << "unsupported op";
245 ValuePtr x = value;
246 ValuePtr y = codegen(init);
247 value = binop(x, y);
248 }
249 return;
250 }
251 }
252 }
253#endif // LLVM_VERSION >= 130
254
255 CodeGen_Posix::codegen_vector_reduce(op, init);
256}

262string CodeGen_WebAssembly::mattrs() const {
263 std::ostringstream s;
264 string sep;
265
266 if (target.has_feature(Target::WasmSignExt)) {
267 s << sep << "+sign-ext";
268 sep = ",";
269 }
270
271 if (target.has_feature(Target::WasmSimd128)) {
272 s << sep << "+simd128";
273 sep = ",";
274 }
275
276 if (target.has_feature(Target::WasmSatFloatToInt)) {
277 s << sep << "+nontrapping-fptoint";
278 sep = ",";
279 }
280
281 if (target.has_feature(Target::WasmThreads)) {
282 // "WasmThreads" doesn't directly affect LLVM codegen,
283 // but it does end up requiring atomics, so be sure to enable them.
284 s << sep << ",+atomics";
285 sep = ",";
286 }
287
288 if (target.has_feature(Target::WasmBulkMemory)) {
289 s << sep << "+bulk-memory";
290 sep = ",";
291 }
292
293 user_assert(target.os == Target::WebAssemblyRuntime)
294 << "wasmrt is the only supported 'os' for WebAssembly at this time.";
295
296 return s.str();
297}
