6ParamMap::ParamMap(const std::initializer_list<ParamMapping> &init) {
7 for (const auto &pm : init) {
8 if (pm.parameter != nullptr) {
9 mapping[*pm.parameter] = ParamArg(pm);
10 } else if (pm.buf_out_param == nullptr) {
11 // TODO: there has to be a way to do this without the const_cast.
12 set(*pm.image_param, *const_cast<Buffer<> *>(&pm.buf), nullptr);
13 } else {
14 Buffer<> temp_undefined;
15 set(*pm.image_param, temp_undefined, pm.buf_out_param);
16 }
17 }
18}

20void ParamMap::set(const ImageParam &p, const Buffer<> &buf, Buffer<> *buf_out_param) {
21 Internal::Parameter v(p.type(), true, p.dimensions(), p.name());
22 v.set_buffer(buf);
23 ParamArg pa;
24 pa.mapped_param = v;
25 pa.buf_out_param = buf_out_param;
26 mapping[p.parameter()] = pa;
27};

29const Internal::Parameter &ParamMap::map(const Internal::Parameter &p, Buffer<> *&buf_out_param) const {
30 auto iter = mapping.find(p);
31 if (iter != mapping.end()) {
32 buf_out_param = iter->second.buf_out_param;
33 return iter->second.mapped_param;
34 } else {
35 buf_out_param = nullptr;
36 return p;
37 }
38}
