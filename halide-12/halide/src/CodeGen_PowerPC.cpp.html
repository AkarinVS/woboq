100void CodeGen_PowerPC::init_module() {
101 CodeGen_Posix::init_module();
102
103 for (const PowerPCIntrinsic &i : intrinsic_defs) {
104 if (i.feature != Target::FeatureEnd && !target.has_feature(i.feature)) {
105 continue;
106 }
107
108 Type ret_type = i.ret_type;
109 vector<Type> arg_types;
110 arg_types.reserve(max_intrinsic_args);
111 for (halide_type_t j : i.arg_types) {
112 if (j.bits == 0) {
113 break;
114 }
115 arg_types.emplace_back(j);
116 }
117
118 auto *fn = declare_intrin_overload(i.name, ret_type, i.intrin_name, std::move(arg_types));
119 fn->addFnAttr(llvm::Attribute::ReadNone);
120 fn->addFnAttr(llvm::Attribute::NoUnwind);
121 }
122}

124void CodeGen_PowerPC::visit(const Min *op) {
125 if (op->type.is_vector()) {
126 value = call_overloaded_intrin(op->type, "min", {op->a, op->b});
127 if (value) {
128 return;
129 }
130 }
131 return CodeGen_Posix::visit(op);
132}
133
134void CodeGen_PowerPC::visit(const Max *op) {
135 if (op->type.is_vector()) {
136 value = call_overloaded_intrin(op->type, "max", {op->a, op->b});
137 if (value) {
138 return;
139 }
140 }
141 return CodeGen_Posix::visit(op);
142}

144string CodeGen_PowerPC::mcpu() const {
145 if (target.bits == 32) {
146 return "ppc32";
147 } else {
148 if (target.has_feature(Target::POWER_ARCH_2_07)) {
149 return "pwr8";
150 } else if (target.has_feature(Target::VSX)) {
151 return "pwr7";
152 } else {
153 return "ppc64";
154 }
155 }
156}

158string CodeGen_PowerPC::mattrs() const {
159 string features;
160 string separator;
161 string enable;
162
163 features += "+altivec";
164 separator = ",";
165
166 enable = target.has_feature(Target::VSX) ? "+" : "-";
167 features += separator + enable + "vsx";
168 separator = ",";
169
170 enable = target.has_feature(Target::POWER_ARCH_2_07) ? "+" : "-";
171 features += separator + enable + "power8-altivec";
172 separator = ",";
173
174 // These move instructions are defined in POWER ISA 2.06 but we do
175 // not check for 2.06 currently. So disable this for anything
176 // lower than ISA 2.07
177 features += separator + enable + "direct-move";
178 separator = ",";
179
180 return features;
181}
