19// Permute a 32-bit unsigned integer using a fixed psuedorandom
20// permutation.
21Expr rng32(const Expr &x) {
22 internal_assert(x.type() == UInt(32));
23
24 // A polynomial P with coefficients C0 .. CN induces a permutation
25 // modulo 2^d iff:
26 // 1) P(0) != P(1) modulo 2
27 // 2) sum(i * Ci) is odd
28
29 // (See http://en.wikipedia.org/wiki/Permutation_polynomial#Rings_Z.2FpkZ)
30
31 // For a quadratic, this is only satisfied by:
32 // C0 anything
33 // C1 odd
34 // C2 even
35
36 // The coefficients defined above were chosen to satisfy this
37 // property.
38
39 // It's pretty random, but note that the quadratic term disappears
40 // if inputs are the multiples of 2^16, and so you get a linear
41 // sequence. However, *that* linear sequence probably varies in
42 // the low bits, so if you run in through the permutation again,
43 // you should break it up. All actual use of this runs it through
44 // multiple times in order to combine several inputs, so it should
45 // be ok. The other flaw is it's a permutation, so you get no
46 // collisions. Birthday paradox be damned.
47
48 // However, it's exceedingly cheap to compute, as it only uses
49 // vectorizable int32 muls and adds, and the resulting numbers:
50 // - Have the correct moments for a uniform distribution
51 // - Have no serial correlations in any of the bits
52 // - Have a completely flat power spectrum
53 // - Have no visible patterns
54
55 // So I declare this good enough for image processing.
56
57 // If it's just a const (which it often is), save the simplifier some work:
58 if (const uint64_t *i = as_const_uint(x)) {
59 return make_const(UInt(32), ((C2 * (*i)) + C1) * (*i) + C0);
60 }
61
62 return (((C2 * x) + C1) * x) + C0;
63}

66Expr random_int(const vector<Expr> &e) {
67 internal_assert(e.size());
68 internal_assert(e[0].type() == Int(32) || e[0].type() == UInt(32));
69 // Permute the first term
70 Expr result = rng32(cast(UInt(32), e[0]));
71 for (size_t i = 1; i < e.size(); i++) {
72 internal_assert(e[i].type() == Int(32) || e[i].type() == UInt(32));
73 // Add in the next term and permute again
74 string name = unique_name('R');
75 // If it's a const, save the simplifier some work
76 const uint64_t *ir = as_const_uint(result);
77 const uint64_t *ie = as_const_uint(e[i]);
78 if (ir && ie) {
79 result = rng32(make_const(UInt(32), (*ir) + (*ie)));
80 } else {
81 result = Let::make(name, result + cast<uint32_t>(e[i]),
82 rng32(Variable::make(UInt(32), name)));
83 }
84 }
85 return result;
86}

88Expr random_float(const vector<Expr> &e) {
89 Expr result = random_int(e);
90 // Set the exponent to one, and fill the mantissa with 23 random bits.
91 result = (127 << 23) | (cast<uint32_t>(result) >> 9);
92 // The clamp is purely for the benefit of bounds inference.
93 return clamp(reinterpret(Float(32), result) - 1.0f, 0.0f, 1.0f);
94}

98class LowerRandom : public IRMutator {
99 using IRMutator::visit;
100
101 Expr visit(const Call *op) override {
102 if (op->is_intrinsic(Call::random)) {
103 vector<Expr> args = op->args;
104 args.insert(args.end(), extra_args.begin(), extra_args.end());
105 if (op->type == Float(32)) {
106 return random_float(args);
107 } else if (op->type == Int(32)) {
108 return cast<int32_t>(random_int(args));
109 } else if (op->type == UInt(32)) {
110 return random_int(args);
111 } else {
112 internal_error << "The intrinsic random() returns an Int(32), UInt(32) or a Float(32).\n";
113 return Expr();
114 }
115 } else {
116 return IRMutator::visit(op);
117 }
118 }
119
120 vector<Expr> extra_args;
121
122public:
123 LowerRandom(const vector<VarOrRVar> &free_vars, int tag) {
124 extra_args.emplace_back(tag);
125 for (const VarOrRVar &v : free_vars) {
126 if (v.is_rvar) {
127 extra_args.push_back(v.rvar);
128 } else {
129 extra_args.push_back(v.var);
130 }
131 }
132 }
133};
134
135} // namespace

137Expr lower_random(const Expr &e, const vector<VarOrRVar> &free_vars, int tag) {
138 LowerRandom r(free_vars, tag);
139 return r.mutate(e);
140}
