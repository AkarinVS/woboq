6Expr Simplify::visit(const Mod *op, ExprInfo *bounds) {
7 ExprInfo a_bounds, b_bounds;
8 Expr a = mutate(op->a, &a_bounds);
9 Expr b = mutate(op->b, &b_bounds);
10
11 // We always combine bounds here, even if not requested, because
12 // we can use them to simplify down to a constant if the bounds
13 // are tight enough.
14 ExprInfo mod_bounds;
15
16 if (no_overflow_int(op->type)) {
17 // The result is at least zero.
18 mod_bounds.min_defined = true;
19 mod_bounds.min = 0;
20
21 // Mod by produces a result between 0
22 // and max(0, abs(modulus) - 1). However, if b is unbounded in
23 // either direction, abs(modulus) could be arbitrarily
24 // large.
25 if (b_bounds.max_defined && b_bounds.min_defined) {
26 mod_bounds.max_defined = true;
27 mod_bounds.max = 0; // When b == 0
28 mod_bounds.max = std::max(mod_bounds.max, b_bounds.max - 1); // When b > 0
29 mod_bounds.max = std::max(mod_bounds.max, -1 - b_bounds.min); // When b < 0
30 }
31
32 // If a is positive, mod can't make it larger
33 if (a_bounds.min_defined && a_bounds.min >= 0 && a_bounds.max_defined) {
34 if (mod_bounds.max_defined) {
35 mod_bounds.max = std::min(mod_bounds.max, a_bounds.max);
36 } else {
37 mod_bounds.max_defined = true;
38 mod_bounds.max = a_bounds.max;
39 }
40 }
41
42 mod_bounds.alignment = a_bounds.alignment % b_bounds.alignment;
43 mod_bounds.trim_bounds_using_alignment();
44 if (bounds) {
45 *bounds = mod_bounds;
46 }
47 }
48
49 if (may_simplify(op->type)) {
50 if (a_bounds.min_defined && a_bounds.min >= 0 &&
51 a_bounds.max_defined && b_bounds.min_defined && a_bounds.max < b_bounds.min) {
52 if (bounds) {
53 *bounds = a_bounds;
54 }
55 return a;
56 }
57
58 if (mod_bounds.min_defined && mod_bounds.max_defined && mod_bounds.min == mod_bounds.max) {
59 return make_const(op->type, mod_bounds.min);
60 }
61
62 int lanes = op->type.lanes();
63 auto rewrite = IRMatcher::rewriter(IRMatcher::mod(a, b), op->type);
64
65 if (rewrite(c0 % c1, fold(c0 % c1)) ||
66 rewrite(IRMatcher::Overflow() % x, a) ||
67 rewrite(x % IRMatcher::Overflow(), b) ||
68 rewrite(0 % x, 0) ||
69 rewrite(x % x, 0) ||
70 rewrite(x % 0, 0) ||
71 (!op->type.is_float() &&
72 rewrite(x % 1, 0))) {
73 return rewrite.result;
74 }
75
76 // clang-format off
77 if (EVAL_IN_LAMBDA
78 (rewrite(broadcast(x, c0) % broadcast(y, c0), broadcast(x % y, c0)) ||
79 (no_overflow_int(op->type) &&
80 (rewrite((x * c0) % c1, (x * fold(c0 % c1)) % c1, c1 > 0 && (c0 >= c1 || c0 < 0)) ||
81 rewrite((x + c0) % c1, (x + fold(c0 % c1)) % c1, c1 > 0 && (c0 >= c1 || c0 < 0)) ||
82 rewrite((x * c0) % c1, (x % fold(c1/c0)) * c0, c0 > 0 && c1 % c0 == 0) ||
83 rewrite((x * c0 + y) % c1, y % c1, c0 % c1 == 0) ||
84 rewrite((y + x * c0) % c1, y % c1, c0 % c1 == 0) ||
85 rewrite((x * c0 - y) % c1, (-y) % c1, c0 % c1 == 0) ||
86 rewrite((y - x * c0) % c1, y % c1, c0 % c1 == 0) ||
87 rewrite((x - y) % 2, (x + y) % 2) || // Addition and subtraction are the same modulo 2, because -1 == 1
88
89 rewrite(ramp(x, c0, c2) % broadcast(c1, c2), broadcast(x, c2) % broadcast(c1, c2), (c0 % c1 == 0)) ||
90 rewrite(ramp(x, c0, lanes) % broadcast(c1, lanes), ramp(x % c1, c0, lanes),
91 // First and last lanes are the same when...
92 can_prove((x % c1 + c0 * (lanes - 1)) / c1 == 0, this)) ||
93 rewrite(ramp(x * c0, c2, c3) % broadcast(c1, c3), ramp(x * fold(c0 % c1), fold(c2 % c1), c3) % c1, c1 > 0 && (c0 >= c1 || c0 < 0)) ||
94 rewrite(ramp(x + c0, c2, c3) % broadcast(c1, c3), ramp(x + fold(c0 % c1), fold(c2 % c1), c3) % c1, c1 > 0 && (c0 >= c1 || c0 < 0)) ||
95 rewrite(ramp(x * c0 + y, c2, c3) % broadcast(c1, c3), ramp(y, fold(c2 % c1), c3) % c1, c0 % c1 == 0) ||
96 rewrite(ramp(y + x * c0, c2, c3) % broadcast(c1, c3), ramp(y, fold(c2 % c1), c3) % c1, c0 % c1 == 0))))) {
97 return mutate(rewrite.result, bounds);
98 }
99 // clang-format on
100 }
101
102 if (a.same_as(op->a) && b.same_as(op->b)) {
103 return op;
104 } else {
105 return Mod::make(a, b);
106 }
107}
