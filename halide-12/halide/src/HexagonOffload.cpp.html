242void do_reloc(char *addr, uint32_t mask, uintptr_t val, bool is_signed, bool verify) {
243 uint32_t inst = *((uint32_t *)addr);
244 debug(4) << "Relocation in instruction: " << hex(inst) << "\n";
245 debug(4) << "val: " << hex(val) << "\n";
246 debug(4) << "mask: " << hex(mask) << "\n";
247
248 if (!mask) {
249 // The mask depends on the instruction. To implement
250 // relocations for new instructions see
251 // instruction_encodings.txt
252 // First print the bits so I can search for it in the
253 // instruction encodings.
254 debug(4) << "Instruction bits: ";
255 for (int i = 31; i >= 0; i--) {
256 debug(4) << (int)((inst >> i) & 1);
257 }
258 debug(4) << "\n";
259
260 if ((inst & (3 << 14)) == 0) {
261 // Some instructions are actually pairs of 16-bit
262 // subinstructions. See section 3.7 in the
263 // programmer's reference.
264 debug(4) << "Duplex!\n";
265
266 int iclass = ((inst >> 29) << 1) | ((inst >> 13) & 1);
267 debug(4) << "Class: " << hex(iclass) << "\n";
268 debug(4) << "Hi: ";
269 for (int i = 28; i >= 16; i--) {
270 debug(4) << (int)((inst >> i) & 1);
271 }
272 debug(4) << "\n";
273 debug(4) << "Lo: ";
274 for (int i = 12; i >= 0; i--) {
275 debug(4) << (int)((inst >> i) & 1);
276 }
277 debug(4) << "\n";
278
279 // We only know how to do the ones where the high
280 // subinstruction is an immediate assignment. (marked
281 // as A in table 9-4 in the programmer's reference
282 // manual).
283 internal_assert(iclass >= 3 && iclass <= 7);
284
285 // Pull out the subinstructions. They're the low 13
286 // bits of each half-word.
287 uint32_t hi = (inst >> 16) & ((1 << 13) - 1);
288 //uint32_t lo = inst & ((1 << 13) - 1);
289
290 // We only understand the ones where hi starts with 010
291 internal_assert((hi >> 10) == 2);
292
293 // Low 6 bits of val go in the following bits.
294 mask = 63 << 20;
295
296 } else if ((inst >> 24) == 72) {
297 // Example instruction encoding that has this high byte (ignoring bits 1 and 2):
298 // 0100 1ii0 000i iiii PPit tttt iiii iiii
299 debug(4) << "Instruction-specific case A\n";
300 mask = 0x061f20ff;
301 } else if ((inst >> 24) == 73) {
302 // 0100 1ii1 000i iiii PPii iiii iiid dddd
303 debug(4) << "Instruction-specific case B\n";
304 mask = 0x061f3fe0;
305 } else if ((inst >> 24) == 120) {
306 // 0111 1000 ii-i iiii PPii iiii iiid dddd
307 debug(4) << "Instruction-specific case C\n";
308 mask = 0x00df3fe0;
309 } else if ((inst >> 16) == 27209) {
310 // 0110 1010 0100 1001 PP-i iiii i--d dddd
311 mask = 0x00001f80;
312 } else if ((inst >> 25) == 72) {
313 // 1001 0ii0 101s ssss PPii iiii iiid dddd
314 // 1001 0ii1 000s ssss PPii iiii iiid dddd
315 mask = 0x06003fe0;
316 } else if ((inst >> 24) == 115 || (inst >> 24) == 124) {
317 // 0111 0011 -10sssss PP1iiiii iiiddddd
318 // 0111 0011 -11sssss PP1iiiii iiiddddd
319 // 0111 0011 0uusssss PP0iiiii iiiddddd
320 // 0111 0011 1uusssss PP0iiiii iiiddddd
321 // 0111 0011 -00sssss PP1iiiii iiiddddd
322 // 0111 0011 -01sssss PP1iiiii iiiddddd
323 // 0111 1100 0IIIIIII PPIiiiii iiiddddd
324 // 0111 0011 -11sssss PP1iiiii iiiddddd
325 mask = 0x00001fe0;
326
327 } else if ((inst >> 24) == 126) {
328 // 0111 1110 0uu0 iiii PP0i iiii iiid dddd
329 // 0111 1110 0uu0 iiii PP1i iiii iiid dddd
330 // 0111 1110 0uu1 iiii PP0i iiii iiid dddd
331 // 0111 1110 0uu1 iiii PP1i iiii iiid dddd
332 mask = 0x000f1fe0;
333 } else if ((inst >> 24) == 65 || (inst >> 24) == 77) {
334 // 0100 0001 000s ssss PP0t tiii iiid dddd
335 // 0100 0001 001s ssss PP0t tiii iiid dddd
336 // 0100 0001 010s ssss PP0t tiii iiid dddd
337 // 0100 0001 011s ssss PP0t tiii iiid dddd
338 // 0100 0001 100s ssss PP0t tiii iiid dddd
339 // 0100 0001 110s ssss PP0t tiii iiid dddd
340 // TODO: Add instructions to comment for mask 77.
341 mask = 0x000007e0;
342 } else if ((inst >> 21) == 540) {
343 // 0100 0011 100s ssss PP0t tiii iiid dddd
344 mask = 0x000007e0;
345 } else if ((inst >> 28) == 11) {
346 // 1011 iiii iiis ssss PPii iiii iiid dddd
347 mask = 0x0fe03fe0;
348 } else {
349 internal_error << "Unhandled instruction type! Instruction = " << inst << "\n";
350 }
351 }
352
353 uintptr_t old_val = val;
354 bool consumed_every_bit = false;
355 for (int i = 0; i < 32; i++) {
356 if (mask & (1 << i)) {
357 internal_assert((inst & (1 << i)) == 0);
358
359 // Consume a bit of val
360 int next_bit = val & 1;
361 if (is_signed) {
362 consumed_every_bit |= ((intptr_t)val) == -1;
363 val = ((intptr_t)val) >> 1;
364 } else {
365 val = ((uintptr_t)val) >> 1;
366 }
367 consumed_every_bit |= (val == 0);
368 inst |= (next_bit << i);
369 }
370 }
371
372 internal_assert(!verify || consumed_every_bit)
373 << "Relocation overflow inst=" << hex(inst)
374 << "mask=" << hex(mask) << " val=" << hex(old_val) << "\n";
375
376 debug(4) << "Relocated instruction: " << hex(inst) << "\n";
377
378 *((uint32_t *)addr) = inst;
379}

381void do_relocation(uint32_t fixup_offset, char *fixup_addr, uint32_t type,
382 const Symbol *sym, uint32_t sym_offset, int32_t addend,
383 Elf::Section &got) {
384 // Hexagon relocations are specified in section 11.5 in
385 // the Hexagon Application Binary Interface spec.
386
387 // Now we can define the variables from Table 11-5.
388 uint32_t S = sym_offset;
389 uint32_t P = fixup_offset;
390 intptr_t A = addend;
391 uint32_t GP = 0;
392
393 uint32_t G = got.contents_size();
394 for (const Relocation &r : got.relocations()) {
395 if (r.get_symbol() == sym) {
396 G = r.get_offset();
397 debug(2) << "Reusing G=" << G << " for symbol " << sym->get_name() << "\n";
398 break;
399 }
400 }
401
402 // Define some constants from table 11-3
403 const uint32_t Word32 = 0xffffffff;
404 const uint32_t Word16 = 0xffff;
405 const uint32_t Word8 = 0xff;
406 const uint32_t Word32_B22 = 0x01ff3ffe;
407 const uint32_t Word32_B15 = 0x00df20fe;
408 const uint32_t Word32_B13 = 0x00202ffe;
409 const uint32_t Word32_B9 = 0x003000fe;
410 const uint32_t Word32_B7 = 0x00001f18;
411 const uint32_t Word32_GP = 0; // The mask is instruction-specific
412 const uint32_t Word32_X26 = 0x0fff3fff;
413 const uint32_t Word32_U6 = 0; // The mask is instruction-specific
414 const uint32_t Word32_R6 = 0x000007e0;
415 const uint32_t Word32_LO = 0x00c03fff;
416 const bool truncate = false, verify = true;
417 const bool _unsigned = false, _signed = true;
418
419 bool needs_got_entry = false;
420
421 switch (type) {
422 case R_HEX_B22_PCREL:
423 do_reloc(fixup_addr, Word32_B22, intptr_t(S + A - P) >> 2, _signed, verify);
424 break;
425 case R_HEX_B15_PCREL:
426 // Untested
427 do_reloc(fixup_addr, Word32_B15, intptr_t(S + A - P) >> 2, _signed, verify);
428 break;
429 case R_HEX_B7_PCREL:
430 do_reloc(fixup_addr, Word32_B7, intptr_t(S + A - P) >> 2, _signed, verify);
431 break;
432 case R_HEX_LO16:
433 internal_error << "Not pic code " << type << "\n";
434 do_reloc(fixup_addr, Word32_LO, uintptr_t(S + A), _unsigned, truncate);
435 break;
436 case R_HEX_HI16:
437 internal_error << "Not pic code " << type << "\n";
438 do_reloc(fixup_addr, Word32_LO, uintptr_t(S + A) >> 16, _unsigned, truncate);
439 break;
440 case R_HEX_32:
441 internal_error << "Not pic code " << type << "\n";
442 do_reloc(fixup_addr, Word32, intptr_t(S + A), _unsigned, truncate);
443 break;
444 case R_HEX_16:
445 internal_error << "Not pic code " << type << "\n";
446 do_reloc(fixup_addr, Word16, uintptr_t(S + A), _unsigned, truncate);
447 break;
448 case R_HEX_8:
449 internal_error << "Not pic code " << type << "\n";
450 do_reloc(fixup_addr, Word8, uintptr_t(S + A), _unsigned, truncate);
451 break;
452 case R_HEX_GPREL16_0:
453 internal_error << "Not pic code " << type << "\n";
454 do_reloc(fixup_addr, Word32_GP, uintptr_t(S + A - GP), _unsigned, verify);
455 break;
456 case R_HEX_GPREL16_1:
457 internal_error << "Not pic code " << type << "\n";
458 do_reloc(fixup_addr, Word32_GP, uintptr_t(S + A - GP) >> 1, _unsigned, verify);
459 break;
460 case R_HEX_GPREL16_2:
461 internal_error << "Not pic code " << type << "\n";
462 do_reloc(fixup_addr, Word32_GP, uintptr_t(S + A - GP) >> 2, _unsigned, verify);
463 break;
464 case R_HEX_GPREL16_3:
465 internal_error << "Not pic code " << type << "\n";
466 do_reloc(fixup_addr, Word32_GP, uintptr_t(S + A - GP) >> 3, _unsigned, verify);
467 break;
468 case R_HEX_HL16:
469 internal_error << "Not pic code " << type << "\n";
470 do_reloc(fixup_addr, Word32_LO, uintptr_t(S + A) >> 16, _unsigned, truncate);
471 do_reloc(fixup_addr + 4, Word32_LO, uintptr_t(S + A), _unsigned, truncate);
472 break;
473 case R_HEX_B13_PCREL:
474 do_reloc(fixup_addr, Word32_B13, intptr_t(S + A - P) >> 2, _signed, verify);
475 break;
476 case R_HEX_B9_PCREL:
477 do_reloc(fixup_addr, Word32_B9, intptr_t(S + A - P) >> 2, _signed, verify);
478 break;
479 case R_HEX_B32_PCREL_X:
480 do_reloc(fixup_addr, Word32_X26, intptr_t(S + A - P) >> 6, _signed, truncate);
481 break;
482 case R_HEX_32_6_X:
483 internal_error << "Not pic code " << type << "\n";
484 do_reloc(fixup_addr, Word32_X26, uintptr_t(S + A) >> 6, _unsigned, verify);
485 break;
486 case R_HEX_B22_PCREL_X:
487 do_reloc(fixup_addr, Word32_B22, intptr_t(S + A - P) & 0x3f, _signed, verify);
488 break;
489 case R_HEX_B15_PCREL_X:
490 do_reloc(fixup_addr, Word32_B15, intptr_t(S + A - P) & 0x3f, _signed, verify);
491 break;
492 case R_HEX_B13_PCREL_X:
493 do_reloc(fixup_addr, Word32_B13, intptr_t(S + A - P) & 0x3f, _signed, verify);
494 break;
495 case R_HEX_B9_PCREL_X:
496 do_reloc(fixup_addr, Word32_B9, intptr_t(S + A - P) & 0x3f, _signed, verify);
497 break;
498 case R_HEX_B7_PCREL_X:
499 do_reloc(fixup_addr, Word32_B7, intptr_t(S + A - P) & 0x3f, _signed, verify);
500 break;
501 case R_HEX_16_X:
502 internal_error << "Not pic code " << type << "\n";
503 do_reloc(fixup_addr, Word32_U6, uintptr_t(S + A), _unsigned, truncate);
504 break;
505 case R_HEX_12_X:
506 internal_error << "Not pic code " << type << "\n";
507 do_reloc(fixup_addr, Word32_R6, uintptr_t(S + A), _unsigned, truncate);
508 break;
509 case R_HEX_11_X:
510 case R_HEX_10_X:
511 case R_HEX_9_X:
512 case R_HEX_8_X:
513 case R_HEX_7_X:
514 case R_HEX_6_X:
515 internal_error << "Not pic code " << type << "\n";
516 do_reloc(fixup_addr, Word32_U6, uintptr_t(S + A), _unsigned, truncate);
517 break;
518 case R_HEX_32_PCREL:
519 do_reloc(fixup_addr, Word32, intptr_t(S + A - P), _signed, verify);
520 break;
521 case R_HEX_6_PCREL_X:
522 do_reloc(fixup_addr, Word32_U6, uintptr_t(S + A - P), _unsigned, truncate);
523 break;
524 case R_HEX_GOT_32_6_X:
525 do_reloc(fixup_addr, Word32_X26, intptr_t(G) >> 6, _signed, truncate);
526 needs_got_entry = true;
527 break;
528 case R_HEX_GOT_16_X:
529 do_reloc(fixup_addr, Word32_U6, intptr_t(G), _signed, truncate);
530 needs_got_entry = true;
531 break;
532 case R_HEX_GOT_11_X:
533 do_reloc(fixup_addr, Word32_U6, uintptr_t(G), _unsigned, truncate);
534 needs_got_entry = true;
535 break;
536
537 default:
538 internal_error << "Unhandled relocation type " << type << "\n";
539 }
540
541 if (needs_got_entry && G == got.contents_size()) {
542 debug(2) << "Adding GOT entry " << G << " for symbol " << sym->get_name() << "\n";
543 got.append_contents((uint32_t)0);
544 got.add_relocation(Relocation(R_HEX_GLOB_DAT, G, 0, sym));
545 }
546}

548class HexagonLinker : public Linker {
549public:
550 uint32_t flags;
551
552 HexagonLinker(const Target &target) {
553 if (target.has_feature(Target::HVX_v66)) {
554 flags = Elf::EF_HEXAGON_MACH_V66;
555 } else if (target.has_feature(Target::HVX_v65)) {
556 flags = Elf::EF_HEXAGON_MACH_V65;
557 } else {
558 flags = Elf::EF_HEXAGON_MACH_V62;
559 }
560 }
561
562 uint16_t get_machine() override {
563 return EM_HEXAGON;
564 }
565 uint32_t get_flags() override {
566 return flags;
567 }
568 uint32_t get_version() override {
569 return EV_CURRENT;
570 }
571 void append_dynamic(Section &dynamic) override {
572 dynamic.append_contents((uint32_t)DT_HEXAGON_VER);
573 dynamic.append_contents((uint32_t)0x3);
574 }
575
576 uint64_t get_got_entry(Section &got, const Symbol &sym) override {
577 // Check if we already made a got entry for this symbol.
578 for (const Relocation &r : got.relocations()) {
579 if (r.get_symbol() == &sym && r.get_type() == R_HEX_GLOB_DAT) {
580 internal_assert(r.get_addend() == 0);
581 return r.get_offset();
582 }
583 }
584
585 uint64_t got_offset = got.contents_size();
586 got.append_contents((uint32_t)0);
587 got.add_relocation(Elf::Relocation(R_HEX_GLOB_DAT, got_offset, 0, &sym));
588 return got_offset;
589 }
590
591 bool needs_plt_entry(const Relocation &r) override {
592 return maybe_branch_inst(r.get_type());
593 }
594
595 Symbol add_plt_entry(const Symbol &sym, Section &plt, Section &got, const Symbol &got_sym) override {
596 if (got.contents_empty()) {
597 // The PLT hasn't been started, initialize it now.
598 plt.set_alignment(16);
599
600 std::vector<char> padding(64, (char)0);
601 // TODO: Make a .plt0 entry that supports lazy binding.
602 plt.set_contents(padding.begin(), padding.end());
603 }
604
605 static const uint8_t hexagon_plt1[] = {
606 0x00, 0x40, 0x00, 0x00, // { immext (#0) (Relocation:R_HEX_B32_PCREL_X)
607 0x0e, 0xc0, 0x49, 0x6a, // r14 = add (pc, ##GOTn@PCREL) } (Relocation:R_HEX_6_PCREL_X)
608 0x1c, 0xc0, 0x8e, 0x91, // r28 = memw (r14)
609 0x00, 0xc0, 0x9c, 0x52, // jumpr r28
610 };
611
612 debug(2) << "Adding PLT entry for symbol " << sym.get_name() << "\n";
613
614 // Add a GOT entry for this symbol.
615 uint64_t got_offset = got.contents_size();
616 got.append_contents((uint32_t)0);
617 got.add_relocation(Elf::Relocation(R_HEX_JMP_SLOT, got_offset, 0, &sym));
618
619 // Add the PLT code.
620 uint32_t plt_offset = plt.get_size();
621 plt.append_contents(hexagon_plt1, hexagon_plt1 + sizeof(hexagon_plt1));
622
623 plt.add_relocation(Relocation(R_HEX_B32_PCREL_X, plt_offset + 0, got_offset, &got_sym));
624 plt.add_relocation(Relocation(R_HEX_6_PCREL_X, plt_offset + 4, got_offset + 4, &got_sym));
625
626 // Make a symbol for the PLT entry.
627 Symbol plt_sym("plt_" + sym.get_name());
628 plt_sym
629 .set_type(Symbol::STT_FUNC)
630 .set_binding(Symbol::STB_LOCAL)
631 .define(&plt, plt_offset, sizeof(hexagon_plt1));
632
633 return plt_sym;
634 }
635
636 Relocation relocate(uint64_t fixup_offset, char *fixup_addr, uint64_t type,
637 const Elf::Symbol *sym, uint64_t sym_offset, int64_t addend,
638 Elf::Section &got) override {
639 if (type == R_HEX_32) {
640 // Don't do this relocation, generate a new R_HEX_RELATIVE relocation instead.
641 return Relocation(R_HEX_RELATIVE, fixup_offset, sym_offset + addend, nullptr);
642 }
643 do_relocation(fixup_offset, fixup_addr, type, sym, sym_offset, addend, got);
644 return Relocation();
645 }
646};

692class InjectHexagonRpc : public IRMutator {
693 std::map<std::string, Expr> state_bufs;
694
695 Module &device_code;
696
697 Expr state_var(const std::string &name, Type type) {
698 return Let::make(name, state_var_ptr(name, type),
699 Load::make(type_of<void *>(), name, 0,
700 Buffer<>(), Parameter(), const_true(), ModulusRemainder()));
701 }
702
703 Expr state_var_ptr(const std::string &name, Type type) {
704 Expr &buf = state_bufs[name];
705 if (!buf.defined()) {
706 auto storage = Buffer<void *>::make_scalar(name + "_buf");
707 storage() = nullptr;
708 buf = Variable::make(type_of<halide_buffer_t *>(), storage.name() + ".buffer", storage);
709 }
710 return Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
711 }
712
713 Expr module_state() {
714 return state_var("hexagon_module_state", type_of<void *>());
715 }
716
717 Expr module_state_ptr() {
718 return state_var_ptr("hexagon_module_state", type_of<void *>());
719 }
720
721 // Create a Buffer containing the given buffer/size, and return an
722 // expression for a pointer to the first element.
723 Expr buffer_ptr(const uint8_t *buffer, size_t size, const char *name) {
724 Buffer<uint8_t> code((int)size, name);
725 memcpy(code.data(), buffer, (int)size);
726 Expr buf = Variable::make(type_of<halide_buffer_t *>(), string(name) + ".buffer", code);
727 return Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
728 }
729
730 using IRMutator::visit;
731
732 Stmt visit(const For *loop) override {
733 if (loop->device_api != DeviceAPI::Hexagon) {
734 return IRMutator::visit(loop);
735 }
736
737 // Unrolling or loop partitioning might generate multiple
738 // loops with the same name, so we need to make them unique.
739 // There's a bit of a hack here: the offload_rpc. prefix is
740 // significant, it tells the Hexagon code generator to expect
741 // the arguments to be unpacked by the Hexagon remote-side RPC
742 // call, which doesn't work with standard buffers.
743 std::string hex_name = unique_name("offload_rpc." + loop->name);
744
745 // After moving this to Hexagon, it doesn't need to be marked
746 // Hexagon anymore.
747 Stmt body;
748 if (is_const_one(loop->extent)) {
749 body = LetStmt::make(loop->name, loop->min, loop->body);
750 } else {
751 body = For::make(loop->name, loop->min, loop->extent, loop->for_type,
752 DeviceAPI::None, loop->body);
753 }
754
755 // Build a closure for the device code.
756 // TODO: Should this move the body of the loop to Hexagon,
757 // or the loop itself? Currently, this moves the loop itself.
758 Closure c(body);
759
760 // A buffer parameter potentially generates 3 scalar parameters (min,
761 // extent, stride) per dimension. Pipelines with many buffers may
762 // generate extreme numbers of scalar parameters, which can cause
763 // problems for LLVM. This logic moves scalar parameters of the type
764 // matching the type of these scalars to a single buffer.
765 // TODO(dsharlet): Maybe this is Int(64) in some cases?
766 Type scalars_buffer_type = Int(32);
767 std::string scalars_buffer_name = "scalar_indices";
768 std::vector<Stmt> scalars_buffer_init;
769 for (auto i = c.vars.begin(); i != c.vars.end();) {
770 if (i->second == scalars_buffer_type) {
771 int index = scalars_buffer_init.size();
772 scalars_buffer_init.push_back(Store::make(scalars_buffer_name, Variable::make(scalars_buffer_type, i->first),
773 index, Parameter(), const_true(), ModulusRemainder()));
774 Expr replacement = Load::make(scalars_buffer_type, scalars_buffer_name, index, Buffer<>(),
775 Parameter(), const_true(), ModulusRemainder());
776 body = LetStmt::make(i->first, replacement, body);
777
778 i = c.vars.erase(i);
779 } else {
780 ++i;
781 }
782 }
783 if (!scalars_buffer_init.empty()) {
784 // If we put some scalars in the scalars buffer, add it to the closure.
785 Closure::Buffer scalars_buffer;
786 scalars_buffer.type = scalars_buffer_type;
787 scalars_buffer.dimensions = 1;
788 scalars_buffer.read = true;
789 scalars_buffer.write = false;
790 c.buffers[scalars_buffer_name] = scalars_buffer;
791 }
792 int scalars_buffer_extent = scalars_buffer_init.size();
793
794 // Make an argument list, and generate a function in the
795 // device_code module. The hexagon runtime code expects
796 // the arguments to appear in the order of (input buffers,
797 // output buffers, input scalars). Scalars must be last
798 // for the scalar arguments to shadow the symbols of the
799 // buffer that get generated by CodeGen_LLVM.
800 std::vector<LoweredArgument> input_buffers, output_buffers;
801 std::map<std::string, Parameter> replacement_params;
802 for (const auto &i : c.buffers) {
803 if (i.second.write) {
804 Argument::Kind kind = Argument::OutputBuffer;
805 output_buffers.emplace_back(i.first, kind, i.second.type, i.second.dimensions, ArgumentEstimates{});
806 } else {
807 Argument::Kind kind = Argument::InputBuffer;
808 input_buffers.emplace_back(i.first, kind, i.second.type, i.second.dimensions, ArgumentEstimates{});
809 }
810
811 // Build a parameter to replace.
812 Parameter p(i.second.type, true, i.second.dimensions);
813 // Assert that buffers are aligned to one HVX vector.
814 const int alignment = 128;
815 p.set_host_alignment(alignment);
816 // The other parameter constraints are already
817 // accounted for by the closure grabbing those
818 // arguments, so we only need to provide the host
819 // alignment.
820 replacement_params[i.first] = p;
821
822 // Add an assert to the body that validates the alignment of the
823 // buffer. These buffers are either allocated by FastRPC or
824 // halide_hexagon_device_interface buffers, either should be aligned
825 // to 128 bytes.
826 if (!device_code.target().has_feature(Target::NoAsserts)) {
827 Expr host_ptr = reinterpret<uint64_t>(Variable::make(Handle(), i.first));
828 Expr error = Call::make(Int(32), "halide_error_unaligned_host_ptr",
829 {i.first, alignment}, Call::Extern);
830 body = Block::make(AssertStmt::make(host_ptr % alignment == 0, error), body);
831 }
832
833 // Unpack buffer parameters into the scope. They come in as host/dev struct pairs.
834 Expr buf = Variable::make(Handle(), i.first + ".buffer");
835 Expr host_ptr = Call::make(Handle(), "_halide_hexagon_buffer_get_host", {buf}, Call::Extern);
836 Expr device_ptr = Call::make(Handle(), "_halide_hexagon_buffer_get_device", {buf}, Call::Extern);
837 body = LetStmt::make(i.first + ".device", device_ptr, body);
838 body = LetStmt::make(i.first, host_ptr, body);
839 }
840 body = replace_params(body, replacement_params);
841
842 std::vector<LoweredArgument> args;
843 args.insert(args.end(), input_buffers.begin(), input_buffers.end());
844 args.insert(args.end(), output_buffers.begin(), output_buffers.end());
845 for (const auto &i : c.vars) {
846 LoweredArgument arg(i.first, Argument::InputScalar, i.second, 0, ArgumentEstimates{});
847 args.push_back(arg);
848 }
849 device_code.append(LoweredFunc(hex_name, args, body, LinkageType::ExternalPlusMetadata));
850
851 // Generate a call to hexagon_device_run.
852 std::vector<Expr> arg_sizes;
853 std::vector<Expr> arg_ptrs;
854 std::vector<Expr> arg_flags;
855
856 for (const auto &i : c.buffers) {
857 // Buffers are passed to the hexagon host runtime as just device
858 // handles (uint64) and host (uint8*) fields. They correspond
859 // to the 'hexagon_device_pointer' struct declared elsewhere;
860 // we don't use that struct here because it's simple enough that
861 // just using `make_struct`() for it is simpler.
862 if (i.first != scalars_buffer_name) {
863 // If this isn't the scalars buffer, assume it has a '.buffer'
864 // description in the IR.
865 Expr buf = Variable::make(type_of<halide_buffer_t *>(), i.first + ".buffer");
866 Expr device = Call::make(UInt(64), Call::buffer_get_device, {buf}, Call::Extern);
867 Expr host = Call::make(Handle(), Call::buffer_get_host, {buf}, Call::Extern);
868 Expr pseudo_buffer = Call::make(Handle(), Call::make_struct, {device, host}, Call::Intrinsic);
869 arg_ptrs.push_back(pseudo_buffer);
870 arg_sizes.emplace_back((uint64_t)(pseudo_buffer.type().bytes()));
871 } else {
872 // If this is the scalars buffer, it doesn't have a .buffer
873 // field. Rather than make one, It's easier to just skip the
874 // buffer_get_host call and reference the allocation directly.
875 // TODO: This is a bit of an ugly hack, it would be nice to find
876 // a better way to identify buffers without a '.buffer' description.
877 Expr host = Variable::make(Handle(), i.first);
878 Expr pseudo_buffer = Call::make(Handle(), Call::make_struct, {make_zero(UInt(64)), host}, Call::Intrinsic);
879 arg_ptrs.push_back(pseudo_buffer);
880 arg_sizes.emplace_back((uint64_t)scalars_buffer_extent * scalars_buffer_type.bytes());
881 }
882
883 // In the flags parameter, bit 0 set indicates the
884 // buffer is read, bit 1 set indicates the buffer is
885 // written. If neither are set, the argument is a scalar.
886 int flags = 0;
887 if (i.second.read) {
888 flags |= 0x1;
889 }
890 if (i.second.write) {
891 flags |= 0x2;
892 }
893 arg_flags.emplace_back(flags);
894 }
895 for (const auto &i : c.vars) {
896 Expr arg = Variable::make(i.second, i.first);
897 Expr arg_ptr = Call::make(type_of<void *>(), Call::make_struct, {arg}, Call::Intrinsic);
898 arg_sizes.emplace_back((uint64_t)i.second.bytes());
899 arg_ptrs.push_back(arg_ptr);
900 arg_flags.emplace_back(0x0);
901 }
902
903 // The argument list is terminated with an argument of size 0.
904 arg_sizes.emplace_back((uint64_t)0);
905
906 std::string pipeline_name = hex_name + "_argv";
907 std::vector<Expr> params;
908 params.push_back(module_state());
909 params.emplace_back(pipeline_name);
910 params.push_back(state_var_ptr(hex_name, type_of<int>()));
911 params.push_back(Call::make(type_of<uint64_t *>(), Call::make_struct, arg_sizes, Call::Intrinsic));
912 params.push_back(Call::make(type_of<void **>(), Call::make_struct, arg_ptrs, Call::Intrinsic));
913 params.push_back(Call::make(type_of<int *>(), Call::make_struct, arg_flags, Call::Intrinsic));
914
915 Stmt offload_call = call_extern_and_assert("halide_hexagon_run", params);
916 if (!scalars_buffer_init.empty()) {
917 offload_call = Block::make(Block::make(scalars_buffer_init), offload_call);
918 }
919 offload_call = Allocate::make(scalars_buffer_name, scalars_buffer_type, MemoryType::Auto,
920 {Expr(scalars_buffer_extent)}, const_true(), offload_call);
921 return offload_call;
922 }
923
924public:
925 InjectHexagonRpc(Module &device_code)
926 : device_code(device_code) {
927 }
928
929 Stmt inject(Stmt s) {
930 s = mutate(s);
931
932 if (!device_code.functions().empty()) {
933 // Wrap the statement in calls to halide_initialize_kernels.
934 Expr runtime_buf_var = Variable::make(type_of<struct halide_buffer_t *>(), runtime_module_name + ".buffer");
935 Expr runtime_size = Call::make(Int(32), Call::buffer_get_extent, {runtime_buf_var, 0}, Call::Extern);
936 Expr runtime_ptr = Call::make(Handle(), Call::buffer_get_host, {runtime_buf_var}, Call::Extern);
937
938 Expr code_buf_var = Variable::make(type_of<struct halide_buffer_t *>(), pipeline_module_name + ".buffer");
939 Expr code_size = Call::make(Int(32), Call::buffer_get_extent, {code_buf_var, 0}, Call::Extern);
940 Expr code_ptr = Call::make(Handle(), Call::buffer_get_host, {code_buf_var}, Call::Extern);
941 Stmt init_kernels = call_extern_and_assert("halide_hexagon_initialize_kernels",
942 {module_state_ptr(), code_ptr, cast<uint64_t>(code_size), runtime_ptr, cast<uint64_t>(runtime_size)});
943 s = Block::make(init_kernels, s);
944 }
945
946 // TODO: This can probably go away due to general debug info at the submodule compile level.
947 debug(1) << "Hexagon device code module: " << device_code << "\n";
948
949 return s;
950 }
951};

955Stmt inject_hexagon_rpc(Stmt s, const Target &host_target,
956 Module &containing_module) {
957 // Make a new target for the device module.
958 Target target(Target::NoOS, Target::Hexagon, 32);
959 // There are two ways of offloading, on device and on host.
960 // In the former we have true QuRT available, while on the
961 // latter we simulate the Hexagon side code with a barebones
962 // Shim layer, ie. NO QURT!
963 if (host_target.arch == Target::ARM) {
964 target.os = Target::QuRT;
965 }
966
967 // These feature flags are propagated from the host target to the
968 // device module.
969 //
970 // TODO: We'd like Target::Debug to be in this list too, but trunk
971 // llvm currently disagrees with hexagon clang as to what
972 // constitutes valid debug info.
973 static const Target::Feature shared_features[] = {
974 Target::Profile,
975 Target::NoAsserts,
976 Target::HVX_128,
977 Target::HVX_v62,
978 Target::HVX_v65,
979 Target::HVX_v66,
980 Target::DisableLLVMLoopOpt,
981 };
982 for (Target::Feature i : shared_features) {
983 if (host_target.has_feature(i)) {
984 target = target.with_feature(i);
985 }
986 }
987
988 Module shared_runtime(runtime_module_name, target);
989 Module hexagon_module(pipeline_module_name, target.with_feature(Target::NoRuntime));
990 InjectHexagonRpc injector(hexagon_module);
991 s = injector.inject(s);
992
993 if (!hexagon_module.functions().empty()) {
994 containing_module.append(hexagon_module);
995 containing_module.append(shared_runtime);
996 }
997
998 return s;
999}

1001Buffer<uint8_t> compile_module_to_hexagon_shared_object(const Module &device_code) {
1002 llvm::LLVMContext context;
1003 std::unique_ptr<llvm::Module> llvm_module(compile_module_to_llvm_module(device_code, context));
1004
1005 // Write intermediate bitcode to disk if requested.
1006 // TODO: We really need something better than this. This won't
1007 // work in non-trivial JIT or AOT programs.
1008 std::string bitcode_dump_path = get_env_variable("HL_HEXAGON_DUMP_BITCODE");
1009 if (!bitcode_dump_path.empty()) {
1010 auto fd_ostream = make_raw_fd_ostream(bitcode_dump_path);
1011 compile_llvm_module_to_llvm_bitcode(*llvm_module, *fd_ostream);
1012 debug(0) << "Wrote Hexagon device bitcode to " << bitcode_dump_path;
1013 }
1014
1015 llvm::SmallVector<char, 4096> object;
1016 llvm::raw_svector_ostream object_stream(object);
1017 compile_llvm_module_to_object(*llvm_module, object_stream);
1018
1019 int min_debug_level = device_code.name() == runtime_module_name ? 3 : 2;
1020 if (debug::debug_level() >= min_debug_level) {
1021 debug(0) << "Hexagon device code assembly: "
1022 << "\n";
1023 llvm::SmallString<4096> assembly;
1024 llvm::raw_svector_ostream assembly_stream(assembly);
1025 compile_llvm_module_to_assembly(*llvm_module, assembly_stream);
1026 debug(0) << assembly.c_str() << "\n";
1027 }
1028
1029 auto obj = Elf::Object::parse_object(object.data(), object.size());
1030 internal_assert(obj);
1031
1032 // Generate just one .text section.
1033 obj->merge_text_sections();
1034
1035 // Make .bss a real section.
1036 auto bss = obj->find_section(".bss");
1037 if (bss != obj->sections_end()) {
1038 bss->set_alignment(128);
1039 // TODO: We should set the type to SHT_NOBITS
1040 // This will cause a difference in MemSize and FileSize like so:
1041 // FileSize = (MemSize - size_of_bss)
1042 // When the Hexagon loader is used on 8998 and later targets,
1043 // the difference is filled with zeroes thereby initializing the .bss
1044 // section.
1045 bss->set_type(Elf::Section::SHT_PROGBITS);
1046 std::fill(bss->contents_begin(), bss->contents_end(), 0);
1047 }
1048
1049 auto dtors = obj->find_section(".dtors");
1050 if (dtors != obj->sections_end()) {
1051 dtors->append_contents((uint32_t)0);
1052 }
1053
1054 // We call the constructors in ctors backwards starting from special
1055 // symbol __CTOR_END__ until we reach a 0 (NULL pointer value). So,
1056 // prepend the .ctors section with 0.
1057 auto ctors = obj->find_section(".ctors");
1058 if (ctors != obj->sections_end()) {
1059 ctors->prepend_contents((uint32_t)0);
1060 }
1061
1062 debug(2) << print_sections(*obj);
1063
1064 // Link into a shared object.
1065 std::string soname = "lib" + device_code.name() + ".so";
1066 Elf::HexagonLinker linker(device_code.target());
1067 std::vector<std::string> dependencies = {
1068 "libhalide_hexagon_remote_skel.so",
1069 };
1070 std::vector<char> shared_object = obj->write_shared_object(&linker, dependencies, soname);
1071
1072 std::string signer = get_env_variable("HL_HEXAGON_CODE_SIGNER");
1073 if (!signer.empty()) {
1074 // If signer is specified, shell out to a tool/script that will
1075 // sign the Hexagon code in a specific way. The tool is expected
1076 // to be of the form
1077 //
1078 // signer /path/to/unsigned.so /path/to/signed.so
1079 //
1080 // where unsigned and signed paths must not be the same file.
1081 // If the signed file already exists, it will be overwritten.
1082
1083 TemporaryFile input("hvx_unsigned", ".so");
1084 TemporaryFile output("hvx_signed", ".so");
1085
1086 debug(1) << "Signing Hexagon code: " << input.pathname() << " -> " << output.pathname() << "\n";
1087
1088 write_entire_file(input.pathname(), shared_object);
1089
1090 debug(1) << "Signing tool: (" << signer << ")\n";
1091 std::string cmd = signer + " " + input.pathname() + " " + output.pathname();
1092 int result = system(cmd.c_str());
1093 internal_assert(result == 0)
1094 << "HL_HEXAGON_CODE_SIGNER failed: result = " << result
1095 << " for cmd (" << cmd << ")";
1096
1097 shared_object = read_entire_file(output.pathname());
1098 }
1099
1100 Halide::Buffer<uint8_t> result_buf(shared_object.size(), device_code.name());
1101 memcpy(result_buf.data(), shared_object.data(), shared_object.size());
1102
1103 return result_buf;
1104}
