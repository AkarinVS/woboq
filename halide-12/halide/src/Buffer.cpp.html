20Expr buffer_accessor(const Buffer<> &buf, const std::vector<Expr> &args) {
21 std::vector<Expr> int_args;
22 for (const Expr &e : args) {
23 user_assert(Int(32).can_represent(e.type()))
24 << "Args to a call to an Image must be representable as 32-bit integers.\n";
25 if (equal(e, _)) {
26 // Expand the _ into the appropriate number of implicit vars.
27 int missing_dimensions = buf.dimensions() - (int)args.size() + 1;
28 for (int i = 0; i < missing_dimensions; i++) {
29 int_args.push_back(Var::implicit(i));
30 }
31 } else if (e.type() == Int(32)) {
32 int_args.push_back(e);
33 } else {
34 int_args.push_back(cast<int>(e));
35 }
36 }
37 Expr c = Call::make(buf, int_args);
38 user_assert(int_args.size() == (size_t)buf.dimensions())
39 << "Dimensionality mismatch accessing Buffer " << buf.name()
40 << ". There were " << int_args.size()
41 << " arguments, but the Buffer has " << buf.dimensions() << " dimensions:\n"
42 << " " << c << "\n";
43 return c;
44}
