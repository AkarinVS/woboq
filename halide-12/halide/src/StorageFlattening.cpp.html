27public:
28 FlattenDimensions(const map<string, pair<Function, int>> &e,
29 const vector<Function> &o,

31 : env(e), target(t) {
32 for (const auto &f : o) {

45 Expr make_shape_var(string name, const string &field, size_t dim,
46 const Buffer<> &buf, const Parameter &param) {
47 ReductionDomain rdom;
48 name = name + "." + field + "." + std::to_string(dim);
49 return Variable::make(Int(32), name, buf, param, rdom);
50 }

52 Expr flatten_args(const string &name, vector<Expr> args,
53 const Buffer<> &buf, const Parameter &param) {
54 bool internal = realizations.contains(name);
55 Expr idx = target.has_large_buffers() ? make_zero(Int(64)) : 0;
56 vector<Expr> mins(args.size()), strides(args.size());
57
58 for (size_t i = 0; i < args.size(); i++) {
59 strides[i] = make_shape_var(name, "stride", i, buf, param);
60 mins[i] = make_shape_var(name, "min", i, buf, param);
61 if (target.has_large_buffers()) {
62 strides[i] = cast<int64_t>(strides[i]);
63 }
64 }
65
66 Expr zero = target.has_large_buffers() ? make_zero(Int(64)) : 0;
67
68 // We peel off constant offsets so that multiple stencil
69 // taps can share the same base address.
70 Expr constant_term = zero;
71 for (size_t i = 0; i < args.size(); i++) {
72 const Add *add = args[i].as<Add>();
73 if (add && is_const(add->b)) {
74 constant_term += strides[i] * add->b;
75 args[i] = add->a;
76 }
77 }
78
79 if (internal) {
80 // f(x, y) -> f[(x-xmin)*xstride + (y-ymin)*ystride] This
81 // strategy makes sense when we expect x to cancel with
82 // something in xmin. We use this for internal allocations.
83 for (size_t i = 0; i < args.size(); i++) {
84 idx += (args[i] - mins[i]) * strides[i];
85 }
86 } else {
87 // f(x, y) -> f[x*stride + y*ystride - (xstride*xmin +
88 // ystride*ymin)]. The idea here is that the last term
89 // will be pulled outside the inner loop. We use this for
90 // external buffers, where the mins and strides are likely
91 // to be symbolic
92 Expr base = zero;
93 for (size_t i = 0; i < args.size(); i++) {
94 idx += args[i] * strides[i];
95 base += mins[i] * strides[i];
96 }
97 idx -= base;
98 }
99
100 if (!is_const_zero(constant_term)) {
101 idx += constant_term;
102 }
103
104 return idx;
105 }

109 Stmt visit(const Realize *op) override {
110 realizations.push(op->name);
111
112 if (op->memory_type == MemoryType::GPUTexture) {
113 textures.insert(op->name);
114 debug(2) << "found texture " << op->name << "\n";
115 }
116
117 Stmt body = mutate(op->body);
118
119 // Compute the size
120 vector<Expr> extents;
121 for (size_t i = 0; i < op->bounds.size(); i++) {
122 extents.push_back(op->bounds[i].extent);
123 extents[i] = mutate(extents[i]);
124 }
125 Expr condition = mutate(op->condition);
126
127 realizations.pop(op->name);
128
129 // The allocation extents of the function taken into account of
130 // the align_storage directives. It is only used to determine the
131 // host allocation size and the strides in halide_buffer_t objects (which
132 // also affects the device allocation in some backends).
133 vector<Expr> allocation_extents(extents.size());
134 vector<int> storage_permutation;
135 {
136 auto iter = env.find(op->name);
137 internal_assert(iter != env.end()) << "Realize node refers to function not in environment.\n";
138 Function f = iter->second.first;
139 const vector<StorageDim> &storage_dims = f.schedule().storage_dims();
140 const vector<string> &args = f.args();
141 for (size_t i = 0; i < storage_dims.size(); i++) {
142 for (size_t j = 0; j < args.size(); j++) {
143 if (args[j] == storage_dims[i].var) {
144 storage_permutation.push_back((int)j);
145 Expr alignment = storage_dims[i].alignment;
146 if (alignment.defined()) {
147 allocation_extents[j] = ((extents[j] + alignment - 1) / alignment) * alignment;
148 } else {
149 allocation_extents[j] = extents[j];
150 }
151 }
152 }
153 internal_assert(storage_permutation.size() == i + 1);
154 }
155 }
156
157 internal_assert(storage_permutation.size() == op->bounds.size());
158
159 Stmt stmt = body;
160 internal_assert(op->types.size() == 1);
161
162 // Make the names for the mins, extents, and strides
163 int dims = op->bounds.size();
164 vector<string> min_name(dims), extent_name(dims), stride_name(dims);
165 for (int i = 0; i < dims; i++) {
166 string d = std::to_string(i);
167 min_name[i] = op->name + ".min." + d;
168 stride_name[i] = op->name + ".stride." + d;
169 extent_name[i] = op->name + ".extent." + d;
170 }
171 vector<Expr> min_var(dims), extent_var(dims), stride_var(dims);
172 for (int i = 0; i < dims; i++) {
173 min_var[i] = Variable::make(Int(32), min_name[i]);
174 extent_var[i] = Variable::make(Int(32), extent_name[i]);
175 stride_var[i] = Variable::make(Int(32), stride_name[i]);
176 }
177
178 // Create a halide_buffer_t object for this allocation.
179 BufferBuilder builder;
180 builder.host = Variable::make(Handle(), op->name);
181 builder.type = op->types[0];
182 builder.dimensions = dims;
183 for (int i = 0; i < dims; i++) {
184 builder.mins.push_back(min_var[i]);
185 builder.extents.push_back(extent_var[i]);
186 builder.strides.push_back(stride_var[i]);
187 }
188 stmt = LetStmt::make(op->name + ".buffer", builder.build(), stmt);
189
190 // Make the allocation node
191 stmt = Allocate::make(op->name, op->types[0], op->memory_type, allocation_extents, condition, stmt);
192
193 // Compute the strides
194 for (int i = (int)op->bounds.size() - 1; i > 0; i--) {
195 int prev_j = storage_permutation[i - 1];
196 int j = storage_permutation[i];
197 Expr stride = stride_var[prev_j] * allocation_extents[prev_j];
198 stmt = LetStmt::make(stride_name[j], stride, stmt);
199 }
200
201 // Innermost stride is one
202 if (dims > 0) {
203 int innermost = storage_permutation.empty() ? 0 : storage_permutation[0];
204 stmt = LetStmt::make(stride_name[innermost], 1, stmt);
205 }
206
207 // Assign the mins and extents stored
208 for (size_t i = op->bounds.size(); i > 0; i--) {
209 stmt = LetStmt::make(min_name[i - 1], op->bounds[i - 1].min, stmt);
210 stmt = LetStmt::make(extent_name[i - 1], extents[i - 1], stmt);
211 }
212 return stmt;
213 }

215 Stmt visit(const Provide *op) override {
216 internal_assert(op->values.size() == 1);
217
218 Parameter output_buf;
219 auto it = env.find(op->name);
220 if (it != env.end()) {
221 const Function &f = it->second.first;
222 int idx = it->second.second;
223
224 // We only want to do this for actual pipeline outputs,
225 // even though every Function has an output buffer. Any
226 // constraints you set on the output buffer of a Func that
227 // isn't actually an output is ignored. This is a language
228 // wart.
229 if (outputs.count(f.name())) {
230 output_buf = f.output_buffers()[idx];
231 }
232 }
233
234 if (output_buf.defined()) {
235 if (output_buf.memory_type() == MemoryType::GPUTexture) {
236 textures.insert(op->name);
237 }
238 }
239
240 Expr value = mutate(op->values[0]);
241 if (in_gpu && textures.count(op->name)) {
242 Expr buffer_var =
243 Variable::make(type_of<halide_buffer_t *>(), op->name + ".buffer", output_buf);
244 vector<Expr> args(2);
245 args[0] = op->name;
246 args[1] = buffer_var;
247 for (size_t i = 0; i < op->args.size(); i++) {
248 Expr min = Variable::make(Int(32), op->name + ".min." + std::to_string(i));
249 args.push_back(op->args[i] - min);
250 }
251 args.push_back(value);
252 Expr store = Call::make(value.type(), Call::image_store,
253 args, Call::Intrinsic);
254 return Evaluate::make(store);
255 } else {
256 Expr idx = mutate(flatten_args(op->name, op->args, Buffer<>(), output_buf));
257 return Store::make(op->name, value, idx, output_buf, const_true(value.type().lanes()), ModulusRemainder());
258 }
259 }

261 Expr visit(const Call *op) override {
262 if (op->call_type == Call::Halide ||
263 op->call_type == Call::Image) {
264
265 debug(2) << " load call to " << op->name << " " << textures.count(op->name) << "\n";
266 if (op->param.defined()) {
267 debug(2) << " is param: "
268 << " " << op->param.name() << " " << op->param.memory_type()
269 << "\n";
270
271 if (op->param.memory_type() == MemoryType::GPUTexture) {
272 textures.insert(op->name);
273 }
274 }
275
276 internal_assert(op->value_index == 0);
277
278 if (in_gpu && textures.count(op->name)) {
279 ReductionDomain rdom;
280 Expr buffer_var =
281 Variable::make(type_of<halide_buffer_t *>(), op->name + ".buffer",
282 op->image, op->param, rdom);
283
284 // Create image_load("name", name.buffer, x - x_min, x_extent,
285 // y - y_min, y_extent, ...). Extents can be used by
286 // successive passes. OpenGL, for example, uses them
287 // for coordinate normalization.
288 vector<Expr> args(2);
289 args[0] = op->name;
290 args[1] = buffer_var;
291 for (size_t i = 0; i < op->args.size(); i++) {
292 Expr min = make_shape_var(op->name, "min", i, op->image, op->param);
293 Expr extent = make_shape_var(op->name, "extent", i, op->image, op->param);
294 args.push_back(mutate(op->args[i]) - min);
295 args.push_back(extent);
296 }
297
298 return Call::make(op->type,
299 Call::image_load,
300 args,
301 Call::PureIntrinsic,
302 FunctionPtr(),
303 0,
304 op->image,
305 op->param);
306 } else {
307 Expr idx = mutate(flatten_args(op->name, op->args, op->image, op->param));
308 return Load::make(op->type, op->name, idx, op->image, op->param,
309 const_true(op->type.lanes()), ModulusRemainder());
310 }
311
312 } else {
313 return IRMutator::visit(op);
314 }
315 }

317 Stmt visit(const Prefetch *op) override {
318 internal_assert(op->types.size() == 1)
319 << "Prefetch from multi-dimensional halide tuple should have been split\n";
320
321 Expr condition = mutate(op->condition);
322
323 vector<Expr> prefetch_min(op->bounds.size());
324 vector<Expr> prefetch_extent(op->bounds.size());
325 vector<Expr> prefetch_stride(op->bounds.size());
326 for (size_t i = 0; i < op->bounds.size(); i++) {
327 prefetch_min[i] = mutate(op->bounds[i].min);
328 prefetch_extent[i] = mutate(op->bounds[i].extent);
329 prefetch_stride[i] = Variable::make(Int(32), op->name + ".stride." + std::to_string(i), op->prefetch.param);
330 }
331
332 Expr base_offset = mutate(flatten_args(op->name, prefetch_min, Buffer<>(), op->prefetch.param));
333 Expr base_address = Variable::make(Handle(), op->name);
334 vector<Expr> args = {base_address, base_offset};
335
336 auto iter = env.find(op->name);
337 if (iter != env.end()) {
338 // Order the <min, extent> args based on the storage dims (i.e. innermost
339 // dimension should be first in args)
340 vector<int> storage_permutation;
341 {
342 Function f = iter->second.first;
343 const vector<StorageDim> &storage_dims = f.schedule().storage_dims();
344 const vector<string> &args = f.args();
345 for (size_t i = 0; i < storage_dims.size(); i++) {
346 for (size_t j = 0; j < args.size(); j++) {
347 if (args[j] == storage_dims[i].var) {
348 storage_permutation.push_back((int)j);
349 }
350 }
351 internal_assert(storage_permutation.size() == i + 1);
352 }
353 }
354 internal_assert(storage_permutation.size() == op->bounds.size());
355
356 for (size_t i = 0; i < op->bounds.size(); i++) {
357 internal_assert(storage_permutation[i] < (int)op->bounds.size());
358 args.push_back(prefetch_extent[storage_permutation[i]]);
359 args.push_back(prefetch_stride[storage_permutation[i]]);
360 }
361 } else {
362 for (size_t i = 0; i < op->bounds.size(); i++) {
363 args.push_back(prefetch_extent[i]);
364 args.push_back(prefetch_stride[i]);
365 }
366 }
367
368 // TODO: Consider generating a prefetch call for each tuple element.
369 Stmt prefetch_call = Evaluate::make(Call::make(op->types[0], Call::prefetch, args, Call::Intrinsic));
370 if (!is_const_one(condition)) {
371 prefetch_call = IfThenElse::make(condition, prefetch_call);
372 }
373 Stmt body = mutate(op->body);
374 return Block::make(prefetch_call, body);
375 }

377 Stmt visit(const For *op) override {
378 bool old_in_gpu = in_gpu;
379 if (op->for_type == ForType::GPUBlock ||
380 op->for_type == ForType::GPUThread) {
381 in_gpu = true;
382 }
383 Stmt stmt = IRMutator::visit(op);
384 in_gpu = old_in_gpu;
385 return stmt;
386 }

389// Realizations, stores, and loads must all be on types that are
390// multiples of 8-bits. This really only affects bools
391class PromoteToMemoryType : public IRMutator {
392 using IRMutator::visit;
393
394 Type upgrade(Type t) {
395 return t.with_bits(((t.bits() + 7) / 8) * 8);
396 }
397
398 Expr visit(const Load *op) override {
399 Type t = upgrade(op->type);
400 if (t != op->type) {
401 return Cast::make(op->type,
402 Load::make(t, op->name, mutate(op->index),
403 op->image, op->param, mutate(op->predicate), ModulusRemainder()));
404 } else {
405 return IRMutator::visit(op);
406 }
407 }
408
409 Stmt visit(const Store *op) override {
410 Type t = upgrade(op->value.type());
411 if (t != op->value.type()) {
412 return Store::make(op->name, Cast::make(t, mutate(op->value)), mutate(op->index),
413 op->param, mutate(op->predicate), ModulusRemainder());
414 } else {
415 return IRMutator::visit(op);
416 }
417 }
418
419 Stmt visit(const Allocate *op) override {
420 Type t = upgrade(op->type);
421 if (t != op->type) {
422 vector<Expr> extents;
423 for (const Expr &e : op->extents) {
424 extents.push_back(mutate(e));
425 }
426 return Allocate::make(op->name, t, op->memory_type, extents,
427 mutate(op->condition), mutate(op->body),
428 mutate(op->new_expr), op->free_function);
429 } else {
430 return IRMutator::visit(op);
431 }
432 }

437Stmt storage_flattening(Stmt s,
438 const vector<Function> &outputs,
439 const map<string, Function> &env,
440 const Target &target) {
441 // The OpenGL backend requires loop mins to be zero'd at this point.
442 s = zero_gpu_loop_mins(s);
443
444 // Make an environment that makes it easier to figure out which
445 // Function corresponds to a tuple component. foo.0, foo.1, foo.2,
446 // all point to the function foo.
447 map<string, pair<Function, int>> tuple_env;
448 for (const auto &p : env) {
449 if (p.second.outputs() > 1) {
450 for (int i = 0; i < p.second.outputs(); i++) {
451 tuple_env[p.first + "." + std::to_string(i)] = {p.second, i};
452 }
453 } else {
454 tuple_env[p.first] = {p.second, 0};
455 }
456 }
457
458 s = FlattenDimensions(tuple_env, outputs, target).mutate(s);
459 s = PromoteToMemoryType().mutate(s);
460 return s;
461}
