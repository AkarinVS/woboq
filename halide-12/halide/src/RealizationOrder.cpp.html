21void find_fused_groups_dfs(const string &current,
22 const map<string, set<string>> &fuse_adjacency_list,
23 set<string> &visited,
24 vector<string> &group) {
25 visited.insert(current);
26 group.push_back(current);
27
28 map<string, set<string>>::const_iterator iter = fuse_adjacency_list.find(current);
29 internal_assert(iter != fuse_adjacency_list.end());
30
31 for (const string &fn : iter->second) {
32 if (visited.find(fn) == visited.end()) {
33 find_fused_groups_dfs(fn, fuse_adjacency_list, visited, group);
34 }
35 }
36}

38pair<map<string, vector<string>>, map<string, string>>
39find_fused_groups(const map<string, Function> &env,
40 const map<string, set<string>> &fuse_adjacency_list) {
41 set<string> visited;
42 map<string, vector<string>> fused_groups;
43 map<string, string> group_name;
44
45 for (const auto &iter : env) {
46 const string &fn = iter.first;
47 if (visited.find(fn) == visited.end()) {
48 vector<string> group;
49 find_fused_groups_dfs(fn, fuse_adjacency_list, visited, group);
50
51 // Create a unique name for the fused group.
52 string rename = unique_name("_fg");
53 fused_groups.emplace(rename, group);
54 for (const auto &m : group) {
55 group_name.emplace(m, rename);
56 }
57 }
58 }
59 return {fused_groups, group_name};
60}

62void realization_order_dfs(const string &current,
63 const map<string, vector<string>> &graph,
64 set<string> &visited,
65 set<string> &result_set,
66 vector<string> &order) {
67 visited.insert(current);
68
69 const auto &iter = graph.find(current);
70 internal_assert(iter != graph.end());
71
72 for (const string &fn : iter->second) {
73 internal_assert(fn != current);
74 if (visited.find(fn) == visited.end()) {
75 realization_order_dfs(fn, graph, visited, result_set, order);
76 } else {
77 internal_assert(result_set.find(fn) != result_set.end())
78 << "Stuck in a loop computing a realization order. "
79 << "Perhaps this pipeline has a loop involving " << current << "?\n";
80 }
81 }
82
83 result_set.insert(current);
84 order.push_back(current);
85}

87// Check the validity of a pair of fused stages.
88void validate_fused_pair(const string &fn, size_t stage_index,
89 const map<string, Function> &env,
90 const map<string, map<string, Function>> &indirect_calls,
91 const FusedPair &p,
92 const vector<FusedPair> &func_fused_pairs) {
93 internal_assert((p.func_1 == fn) && (p.stage_1 == stage_index));
94
95 user_assert(env.count(p.func_2))
96 << "Illegal compute_with: \"" << p.func_2 << "\" is scheduled to be computed with \""
97 << p.func_1 << "\" but \"" << p.func_2 << "\" is not used anywhere.\n";
98
99 // Assert no compute_with of updates of the same Func and no duplicates
100 // (These technically should not have been possible from the front-end).
101 {
102 internal_assert(p.func_1 != p.func_2);
103 const auto &iter = std::find(func_fused_pairs.begin(), func_fused_pairs.end(), p);
104 internal_assert(iter == func_fused_pairs.end())
105 << "Found duplicates of fused pair (" << p.func_1 << ".s" << p.stage_1 << ", "
106 << p.func_2 << ".s" << p.stage_2 << ", " << p.var_name << ")\n";
107 }
108
109 // Assert no dependencies among the functions that are computed_with.
110 const auto &callees_1 = indirect_calls.find(p.func_1);
111 if (callees_1 != indirect_calls.end()) {
112 user_assert(callees_1->second.find(p.func_2) == callees_1->second.end())
113 << "Invalid compute_with: there is dependency between "
114 << p.func_1 << " and " << p.func_2 << "\n";
115 }
116 const auto &callees_2 = indirect_calls.find(p.func_2);
117 if (callees_2 != indirect_calls.end()) {
118 user_assert(callees_2->second.find(p.func_1) == callees_2->second.end())
119 << "Invalid compute_with: there is dependency between "
120 << p.func_1 << " and " << p.func_2 << "\n";
121 }
122}

124// Populate 'func_fused_pairs' and 'fuse_adjacency_list': a directed and
125// non-directed graph representing the compute_with dependencies between
126// functions.
127void collect_fused_pairs(const FusedPair &p,
128 vector<FusedPair> &func_fused_pairs,
129 map<string, vector<string>> &graph,
130 map<string, set<string>> &fuse_adjacency_list) {
131 fuse_adjacency_list[p.func_1].insert(p.func_2);
132 fuse_adjacency_list[p.func_2].insert(p.func_1);
133
134 func_fused_pairs.push_back(p);
135
136 // If there is a compute_with dependency between two functions, we need
137 // to update the pipeline DAG so that the computed realization order
138 // respects this dependency.
139 graph[p.func_1].push_back(p.func_2);
140}

142// Populate the 'fused_pairs' list in Schedule of each function stage.
143void populate_fused_pairs_list(const string &func, const Definition &def,
144 size_t stage_index, map<string, Function> &env) {
145 internal_assert(def.defined());
146 const LoopLevel &fuse_level = def.schedule().fuse_level().level;
147 if (fuse_level.is_inlined() || fuse_level.is_root()) {
148 // 'func' is not fused with anyone.
149 return;
150 }
151
152 auto iter = env.find(fuse_level.func());
153 user_assert(iter != env.end())
154 << "Illegal compute_with: \"" << func << "\" is scheduled to be computed with \""
155 << fuse_level.func() << "\" which is not used anywhere.\n";
156
157 Function &parent = iter->second;
158 user_assert(!parent.has_extern_definition())
159 << "Illegal compute_with: Func \"" << func << "\" is scheduled to be "
160 << "computed with extern Func \"" << parent.name() << "\"\n";
161
162 FusedPair pair(fuse_level.func(), fuse_level.stage_index(),
163 func, stage_index, fuse_level.var().name());
164 if (fuse_level.stage_index() == 0) {
165 parent.definition().schedule().fused_pairs().push_back(pair);
166 } else {
167 internal_assert(fuse_level.stage_index() > 0);
168 parent.update(fuse_level.stage_index() - 1).schedule().fused_pairs().push_back(pair);
169 }
170}

172// Make sure we don't have cyclic compute_with: if Func 'f' is computed after
173// Func 'g', Func 'g' should not be computed after Func 'f'.
174void check_no_cyclic_compute_with(const map<string, vector<FusedPair>> &fused_pairs_graph) {
175 for (const auto &iter : fused_pairs_graph) {
176 for (const auto &pair : iter.second) {
177 internal_assert(pair.func_1 != pair.func_2);
178 const auto &o_iter = fused_pairs_graph.find(pair.func_2);
179 if (o_iter == fused_pairs_graph.end()) {
180 continue;
181 }
182 const auto &it = std::find_if(o_iter->second.begin(), o_iter->second.end(),
183 [&pair](const FusedPair &other) {
184 return (pair.func_1 == other.func_2) && (pair.func_2 == other.func_1);
185 });
186 user_assert(it == o_iter->second.end())
187 << "Found cyclic dependencies between compute_with of "
188 << pair.func_1 << " and " << pair.func_2 << "\n";
189 }
190 }
191}

193// Check that stages are scheduled in the correct order with no compute_with
194// edge going back across other compute_with edge.
195// For example, some illegal cases include:
196// f.compute_with(g.update(0), var)
197// f.update(0).compute_with(g, var)
198// or
199// f.compute_with(g, var)
200// f.update(1).compute_with(g, var)
201// where f.update(0) will have to be computed after g, which means
202// that order of f will be f, f.update(1), f.update(0).
203void check_fused_stages_are_scheduled_in_order(const Function &f) {
204 map<string, pair<int, int>> max_stage_for_parent;
205 bool are_stages_consecutive = false;
206 for (size_t i = 0; i < f.updates().size() + 1; i++) {
207 const auto &def = (i == 0) ? f.definition() : f.update(i - 1);
208 const auto &fuse_level = def.schedule().fuse_level().level;
209 if (!fuse_level.is_inlined() && !fuse_level.is_root()) {
210 if (max_stage_for_parent.count(fuse_level.func()) == 0) {
211 max_stage_for_parent[fuse_level.func()] = {-1, -1};
212 }
213 const auto &max_stage = max_stage_for_parent[fuse_level.func()];
214 bool is_correct = (fuse_level.stage_index() > max_stage.second) ||
215 (fuse_level.stage_index() == max_stage.second && are_stages_consecutive);
216
217 user_assert(is_correct)
218 << "Invalid compute_with: impossible to establish correct stage order between "
219 << f.name() << ".s" << max_stage.first << " with "
220 << fuse_level.func() << ".s" << max_stage.second << " and "
221 << f.name() << ".s" << i << " with "
222 << fuse_level.func() << ".s" << fuse_level.stage_index() << "\n";
223
224 max_stage_for_parent[fuse_level.func()] = {i, fuse_level.stage_index()};
225 are_stages_consecutive = true;
226 } else {
227 are_stages_consecutive = false;
228 }
229 }

234pair<vector<string>, vector<vector<string>>> realization_order(
235 const vector<Function> &outputs, map<string, Function> &env) {
236
237 // Populate the fused_pairs list of each function definition (i.e. list of
238 // all function definitions that are to be computed with that function).
239 for (auto &iter : env) {
240 if (iter.second.has_extern_definition()) {
241 // Extern function should not be fused.
242 continue;
243 }
244
245 check_fused_stages_are_scheduled_in_order(iter.second);
246
247 populate_fused_pairs_list(iter.first, iter.second.definition(), 0, env);
248 for (size_t i = 0; i < iter.second.updates().size(); ++i) {
249 populate_fused_pairs_list(iter.first, iter.second.updates()[i], i + 1, env);
250 }
251 }
252
253 // Collect all indirect calls made by all the functions in "env".
254 map<string, map<string, Function>> indirect_calls;
255 for (const pair<const string, Function> &caller : env) {
256 map<string, Function> more_funcs = find_transitive_calls(caller.second);
257 indirect_calls.emplace(caller.first, more_funcs);
258 }
259
260 // 'graph' is a DAG representing the pipeline. Each function maps to the
261 // set describing its inputs.
262 map<string, vector<string>> graph;
263
264 // Make a directed and non-directed graph representing the compute_with
265 // dependencies between functions. Each function maps to the list of
266 // functions computed_with it.
267 map<string, vector<FusedPair>> fused_pairs_graph;
268 map<string, set<string>> fuse_adjacency_list;
269
270 for (const pair<const string, Function> &caller : env) {
271 // Find all compute_with (fused) pairs. We have to look at the update
272 // definitions as well since compute_with is defined per definition (stage).
273 vector<FusedPair> &func_fused_pairs = fused_pairs_graph[caller.first];
274 fuse_adjacency_list[caller.first]; // Make sure every Func in 'env' is allocated a slot
275 if (!caller.second.has_extern_definition()) {
276 for (const auto &p : caller.second.definition().schedule().fused_pairs()) {
277 validate_fused_pair(caller.first, 0, env, indirect_calls,
278 p, func_fused_pairs);
279 collect_fused_pairs(p, func_fused_pairs, graph, fuse_adjacency_list);
280 }
281 for (size_t i = 0; i < caller.second.updates().size(); ++i) {
282 for (const auto &p : caller.second.updates()[i].schedule().fused_pairs()) {
283 validate_fused_pair(caller.first, i + 1, env, indirect_calls,
284 p, func_fused_pairs);
285 collect_fused_pairs(p, func_fused_pairs, graph, fuse_adjacency_list);
286 }
287 }
288 }
289 }
290
291 check_no_cyclic_compute_with(fused_pairs_graph);
292
293 // Determine groups of functions which loops are to be fused together.
294 // 'fused_groups' maps a fused group to its members.
295 // 'group_name' maps a function to the name of the fused group it belongs to.
296 map<string, vector<string>> fused_groups;
297 map<string, string> group_name;
298 std::tie(fused_groups, group_name) = find_fused_groups(env, fuse_adjacency_list);
299
300 // Compute the DAG representing the pipeline
301 for (const pair<const string, Function> &caller : env) {
302 const string &caller_rename = group_name.at(caller.first);
303 // Create a dummy node representing the fused group and add input edge
304 // dependencies from the nodes representing member of the fused group
305 // to this dummy node.
306 graph[caller.first].push_back(caller_rename);
307 // Direct the calls to calls from the dummy node. This forces all the
308 // functions called by members of the fused group to be realized first.
309 vector<string> &s = graph[caller_rename];
310 for (const pair<const string, Function> &callee : find_direct_calls(caller.second)) {
311 if ((callee.first != caller.first) && // Skip calls to itself (i.e. update stages)
312 (std::find(s.begin(), s.end(), callee.first) == s.end())) {
313 s.push_back(callee.first);
314 }
315 }
316 }
317
318 // Compute the realization order of the fused groups (i.e. the dummy nodes)
319 // and also the realization order of the functions within a fused group.
320 vector<string> temp;
321 set<string> result_set;
322 set<string> visited;
323 for (const Function &f : outputs) {
324 if (visited.find(f.name()) == visited.end()) {
325 realization_order_dfs(f.name(), graph, visited, result_set, temp);
326 }
327 }
328
329 // Collect the realization order of the fused groups.
330 vector<vector<string>> group_order;
331 for (const auto &fn : temp) {
332 const auto &iter = fused_groups.find(fn);
333 if (iter != fused_groups.end()) {
334 group_order.push_back(iter->second);
335 }
336 }
337 // Sort the functions within a fused group based on the compute_with
338 // dependencies (i.e. parent of the fused loop should be realized after its
339 // children).
340 for (auto &group : group_order) {
341 std::sort(group.begin(), group.end(),
342 [&](const string &lhs, const string &rhs) {
343 const auto &iter_lhs = std::find(temp.begin(), temp.end(), lhs);
344 const auto &iter_rhs = std::find(temp.begin(), temp.end(), rhs);
345 return iter_lhs < iter_rhs;
346 });
347 }
348
349 // Collect the realization order of all functions within the pipeline.
350 vector<string> order;
351 for (const auto &group : group_order) {
352 for (const auto &f : group) {
353 order.push_back(f);
354 }
355 }
356
357 return {order, group_order};
358}

360vector<string> topological_order(const vector<Function> &outputs,
361 const map<string, Function> &env) {
362
363 // Make a DAG representing the pipeline. Each function maps to the
364 // set describing its inputs.
365 map<string, vector<string>> graph;
366
367 for (const pair<const string, Function> &caller : env) {
368 vector<string> s;
369 for (const pair<const string, Function> &callee : find_direct_calls(caller.second)) {
370 if ((callee.first != caller.first) && // Skip calls to itself (i.e. update stages)
371 (std::find(s.begin(), s.end(), callee.first) == s.end())) {
372 s.push_back(callee.first);
373 }
374 }
375 graph.emplace(caller.first, s);
376 }
377
378 vector<string> order;
379 set<string> result_set;
380 set<string> visited;
381 for (const Function &f : outputs) {
382 if (visited.find(f.name()) == visited.end()) {
383 realization_order_dfs(f.name(), graph, visited, result_set, order);
384 }
385 }
386
387 return order;
388}
