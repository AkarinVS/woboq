6Expr Simplify::visit(const EQ *op, ExprInfo *bounds) {
7 if (truths.count(op)) {
8 return const_true(op->type.lanes());
9 } else if (falsehoods.count(op)) {
10 return const_false(op->type.lanes());
11 }
12
13 if (!may_simplify(op->a.type())) {
14 Expr a = mutate(op->a, nullptr);
15 Expr b = mutate(op->b, nullptr);
16 if (a.same_as(op->a) && b.same_as(op->b)) {
17 return op;
18 } else {
19 return EQ::make(a, b);
20 }
21 }
22
23 if (op->a.type().is_bool()) {
24 Expr a = mutate(op->a, nullptr);
25 Expr b = mutate(op->b, nullptr);
26 if (should_commute(a, b)) {
27 std::swap(a, b);
28 }
29 const int lanes = op->type.lanes();
30 auto rewrite = IRMatcher::rewriter(IRMatcher::eq(a, b), op->type);
31 if (rewrite(x == 1, x)) {
32 return rewrite.result;
33 } else if (rewrite(x == 0, !x)) {
34 return mutate(rewrite.result, bounds);
35 } else if (rewrite(x == x, const_true(lanes))) {
36 return rewrite.result;
37 } else if (a.same_as(op->a) && b.same_as(op->b)) {
38 return op;
39 } else {
40 return EQ::make(a, b);
41 }
42 }
43
44 ExprInfo delta_bounds;
45 Expr delta = mutate(op->a - op->b, &delta_bounds);
46 const int lanes = op->type.lanes();
47
48 // If the delta is 0, then it's just x == x
49 if (is_const_zero(delta)) {
50 return const_true(lanes);
51 }
52
53 // Attempt to disprove using bounds analysis
54 if (delta_bounds.min_defined && delta_bounds.min > 0) {
55 return const_false(lanes);
56 }
57
58 if (delta_bounds.max_defined && delta_bounds.max < 0) {
59 return const_false(lanes);
60 }
61
62 // Attempt to disprove using modulus remainder analysis
63 if (delta_bounds.alignment.remainder != 0) {
64 return const_false(lanes);
65 }
66
67 auto rewrite = IRMatcher::rewriter(IRMatcher::eq(delta, 0), op->type, delta.type());
68
69 bool allowed_overflow = no_overflow(delta.type());
70
71 if (rewrite(broadcast(x, c0) == 0, broadcast(x == 0, c0)) ||
72 (allowed_overflow &&
73 (rewrite(x * y == 0, (x == 0) || (y == 0)) ||
74 rewrite(x * c0 + c1 == 0, x == fold((0 - c1) / c0), c1 % c0 == 0))) ||
75 rewrite(select(x, 0, y) == 0, x || (y == 0)) ||
76 rewrite(select(x, c0, y) == 0, !x && (y == 0), c0 != 0) ||
77 rewrite(select(x, y, 0) == 0, !x || (y == 0)) ||
78 rewrite(select(x, y, c0) == 0, x && (y == 0), c0 != 0) ||
79
80 rewrite(select(x, c0, y) + c1 == 0, x || (y == fold(-c1)), c0 + c1 == 0) ||
81 rewrite(select(x, y, c0) + c1 == 0, !x || (y == fold(-c1)), c0 + c1 == 0) ||
82 rewrite(select(x, c0, y) + c1 == 0, !x && (y == fold(-c1)), c0 + c1 != 0) ||
83 rewrite(select(x, y, c0) + c1 == 0, x && (y == fold(-c1)), c0 + c1 != 0) ||
84
85 rewrite(max(x, y) - y == 0, x <= y) ||
86 rewrite(min(x, y) - y == 0, y <= x) ||
87 rewrite(max(y, x) - y == 0, x <= y) ||
88 rewrite(min(y, x) - y == 0, y <= x) ||
89 rewrite(y - max(x, y) == 0, x <= y) ||
90 rewrite(y - min(x, y) == 0, y <= x) ||
91 rewrite(y - max(y, x) == 0, x <= y) ||
92 rewrite(y - min(y, x) == 0, y <= x) ||
93
94 // Guard against `c0 + c1` overflowing.
95 (allowed_overflow &&
96 (rewrite(max(x, c0) + c1 == 0, x == fold(-c1), c0 + c1 < 0) ||
97 rewrite(min(x, c0) + c1 == 0, x == fold(-c1), c0 + c1 > 0) ||
98 rewrite(max(x, c0) + c1 == 0, false, c0 + c1 > 0) ||
99 rewrite(min(x, c0) + c1 == 0, false, c0 + c1 < 0) ||
100 rewrite(max(x, c0) + c1 == 0, x <= c0, c0 + c1 == 0) ||
101 rewrite(min(x, c0) + c1 == 0, c0 <= x, c0 + c1 == 0))) ||
102
103 // Special case the above where c1 == 0
104 rewrite(max(x, c0) == 0, x == 0, c0 < 0) ||
105 rewrite(min(x, c0) == 0, x == 0, c0 > 0) ||
106 rewrite(max(x, c0) == 0, false, c0 > 0) ||
107 rewrite(min(x, c0) == 0, false, c0 < 0) ||
108 rewrite(max(x, 0) == 0, x <= 0) ||
109 rewrite(min(x, 0) == 0, 0 <= x) ||
110
111 false) {
112 return mutate(rewrite.result, bounds);
113 }
114
115 if (rewrite(c0 == 0, fold(c0 == 0)) ||
116 rewrite((x - y) + c0 == 0, x == y + fold(-c0)) ||
117 rewrite(x + c0 == 0, x == fold(-c0)) ||
118 rewrite(c0 - x == 0, x == c0) ||
119 rewrite(x - y == 0, x == y) ||
120 rewrite(x == 0, x == 0)) {
121 const EQ *eq = rewrite.result.as<EQ>();
122 if (eq &&
123 eq->a.same_as(op->a) &&
124 eq->b.same_as(op->b)) {
125 return op;
126 } else {
127 return rewrite.result;
128 }
129 }
130
131 // Unreachable. That last rewrite catches everything and
132 // constructs delta == 0 for us.
133 return Expr();
134}

136// ne redirects to not eq
137Expr Simplify::visit(const NE *op, ExprInfo *bounds) {
138 if (!may_simplify(op->a.type())) {
139 Expr a = mutate(op->a, nullptr);
140 Expr b = mutate(op->b, nullptr);
141 if (a.same_as(op->a) && b.same_as(op->b)) {
142 return op;
143 } else {
144 return NE::make(a, b);
145 }
146 }
147
148 Expr mutated = mutate(Not::make(EQ::make(op->a, op->b)), bounds);
149 if (const NE *ne = mutated.as<NE>()) {
150 if (ne->a.same_as(op->a) && ne->b.same_as(op->b)) {
151 return op;
152 }
153 }
154 return mutated;
155}
