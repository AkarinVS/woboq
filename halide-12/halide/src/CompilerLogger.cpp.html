22class ObfuscateNames : public IRMutator {
23 using IRMutator::visit;
24
25 std::map<std::string, std::string> remapping;
26
27 Expr visit(const Call *op) override {
28 std::vector<Expr> args;
29 for (const Expr &e : op->args) {
30 args.emplace_back(mutate(e));
31 }
32
33 std::string name = op->name;
34 if (op->call_type == Call::Extern ||
35 op->call_type == Call::ExternCPlusPlus ||
36 op->call_type == Call::Halide ||
37 op->call_type == Call::Image) {
38 name = remap(op->name);
39 }
40 return Call::make(op->type, name, args, op->call_type, op->func,
41 op->value_index, op->image, op->param);
42 }
43
44 Expr visit(const Let *op) override {
45 std::string name = remap(op->name);
46 Expr value = mutate(op->value);
47 Expr body = mutate(op->body);
48 return Let::make(name, std::move(value), std::move(body));
49 }
50
51 Expr visit(const Load *op) override {
52 std::string name = remap(op->name);
53 Expr index = mutate(op->index);
54 Expr predicate = mutate(op->predicate);
55 return Load::make(op->type, name, index, op->image, op->param,
56 predicate, op->alignment);
57 }
58
59 Expr visit(const Variable *op) override {
60 std::string name = remap(op->name);
61 return Variable::make(op->type, name, op->image, op->param, op->reduction_domain);
62 }
63
64public:
65 ObfuscateNames() = default;
66 ObfuscateNames(const std::initializer_list<std::pair<const std::string, std::string>> &values)
67 : remapping(values) {
68 }
69
70 std::string remap(const std::string &var_name) {
71 std::string anon_name = "anon" + std::to_string(remapping.size());
72 auto p = remapping.insert({var_name, std::move(anon_name)});
73 return p.first->second;
74 }
75};

123void JSONCompilerLogger::obfuscate() {
124 {
125 std::map<std::string, std::vector<Expr>> n;
126 for (const auto &it : matched_simplifier_rules) {
127 std::string rule = it.first;
128 for (const auto &e : it.second) {
129 ObfuscateNames obfuscater;
130 n[rule].emplace_back(obfuscater.mutate(e));
131 }
132 }
133 matched_simplifier_rules = n;
134 }
135 {
136 std::vector<std::pair<Expr, Expr>> n;
137 for (const auto &it : failed_to_prove_exprs) {
138 // Note that use a separate obfuscater for each pair, so each
139 // shares identifiers only with each other; this makes it simpler
140 // to post-process output from multiple unrelated Generators
141 // and combine Exprs with similar shapes.
142 ObfuscateNames obfuscater;
143 auto failed_to_prove = obfuscater.mutate(it.first);
144 auto original_expr = obfuscater.mutate(it.second);
145 n.emplace_back(std::move(failed_to_prove), std::move(original_expr));
146 }
147 failed_to_prove_exprs = n;
148 }
149}

207template<typename T>
208std::ostream &emit_pairs(std::ostream &o, int indent, const std::string &key, const T &pairs, bool comma = true) {
209 std::string spaces(indent, ' ');
210
211 emit_key(o, indent, key);
212 o << "{\n";
213 int commas_to_emit = (int)pairs.size() - 1;
214 for (const auto &it : pairs) {
215 const bool comma = (commas_to_emit > 0);
216 emit_key_value(o, indent + 1, it.first, it.second, comma);
217 commas_to_emit--;
218 }
219 o << spaces << "}";
220 emit_eol(o, comma);
221 return o;
222}
223
224template<typename T>
225std::ostream &emit_list(std::ostream &o, int indent, const T &list, bool comma = true) {
226 std::string spaces(indent, ' ');
227 std::string spaces_in(indent + 1, ' ');
228
229 o << spaces << "[\n";
230 int commas_to_emit = (int)list.size() - 1;
231 for (const auto &it : list) {
232 o << spaces_in;
233 emit_value(o, it);
234 emit_eol(o, commas_to_emit-- > 0);
235 }
236 o << spaces << "]";
237 emit_eol(o, comma);
238 return o;
239}

257std::ostream &JSONCompilerLogger::emit_to_stream(std::ostream &o) {
258 if (obfuscate_exprs) {
259 obfuscate();
260 }
261
262 // Output in JSON form
263
264 o << "{\n";
265
266 int indent = 1;
267 emit_optional_key_value(o, indent, "generator_name", generator_name);
268 emit_optional_key_value(o, indent, "function_name", function_name);
269 emit_optional_key_value(o, indent, "autoscheduler_name", autoscheduler_name);
270 emit_optional_key_value(o, indent, "target", target == Target() ? "" : target.to_string());
271 emit_optional_key_value(o, indent, "generator_args", generator_args);
272
273 if (object_code_size) {
274 emit_key_value(o, indent, "object_code_size", object_code_size);
275 }
276
277 // If these are present, emit them, even if value is zero
278 if (compilation_time.count(Phase::HalideLowering)) {
279 emit_key_value(o, indent, "compilation_time_halide_lowering", compilation_time[Phase::HalideLowering]);
280 }
281 if (compilation_time.count(Phase::LLVM)) {
282 emit_key_value(o, indent, "compilation_time_llvm", compilation_time[Phase::LLVM]);
283 }
284
285 if (!matched_simplifier_rules.empty()) {
286 emit_object_key_open(o, indent, "matched_simplifier_rules");
287
288 int commas_to_emit = (int)matched_simplifier_rules.size() - 1;
289 for (const auto &it : matched_simplifier_rules) {
290 const auto &loop_var = it.first;
291 emit_key(o, indent + 1, loop_var);
292 emit_eol(o, false);
293 emit_list(o, indent + 1, exprs_to_strings(it.second), (commas_to_emit-- > 0));
294 }
295
296 emit_object_key_close(o, indent);
297 }
298
299 if (!non_monotonic_loop_vars.empty()) {
300 emit_object_key_open(o, indent, "non_monotonic_loop_vars");
301
302 int commas_to_emit = (int)non_monotonic_loop_vars.size() - 1;
303 for (const auto &it : non_monotonic_loop_vars) {
304 const auto &loop_var = it.first;
305 emit_key(o, indent + 1, loop_var);
306 emit_eol(o, false);
307 emit_list(o, indent + 1, exprs_to_strings(it.second), (commas_to_emit-- > 0));
308 }
309
310 emit_object_key_close(o, indent);
311 }
312
313 if (!failed_to_prove_exprs.empty()) {
314 emit_object_key_open(o, indent, "failed_to_prove");
315
316 // We'll do deduplication here, during stringification.
317 std::map<std::string, std::set<std::string>> sorted;
318 for (const auto &it : failed_to_prove_exprs) {
319 const auto failed_to_prove_str = expr_to_string(it.first);
320 const auto original_expr_str = expr_to_string(it.second);
321 sorted[failed_to_prove_str].insert(original_expr_str);
322 }
323
324 int commas_to_emit = (int)sorted.size() - 1;
325 for (const auto &it : sorted) {
326 emit_key(o, indent + 1, it.first);
327 emit_eol(o, false);
328 emit_list(o, indent + 1, it.second, (commas_to_emit-- > 0));
329 }
330
331 emit_object_key_close(o, indent);
332 }
333
334 // Emit this last as a simple way to dodge the trailing-comma nonsense
335 o << " \"version\": \"HalideJSONCompilerLoggerV1\"\n";
336 o << "}\n";
337
338 return o;
339}
