122// This mutator rewrites calls and provides to a particular
123// func:
124// - Calls and Provides are shifted to be relative to the min.
125// - Provides additionally are rewritten to load values from the
126// previous iteration of the loop if they were computed in the
127// last iteration.
128class RollFunc : public IRMutator {
129 const Function &func;
130 int dim;
131 const string &loop_var;
132 const Interval &old_bounds;
133 const Interval &new_bounds;
134
135 Scope<Expr> scope;
136
137 // It helps simplify the shifted calls/provides to rebase the
138 // loops that are subtracted from to have a min of 0.
139 set<string> loops_to_rebase;
140 bool in_produce = false;
141
142 using IRMutator::visit;
143
144 Stmt visit(const ProducerConsumer *op) override {
145 bool produce_func = op->name == func.name() && op->is_producer;
146 ScopedValue<bool> old_in_produce(in_produce, in_produce || produce_func);
147 return IRMutator::visit(op);
148 }
149
150 Stmt visit(const Provide *op) override {
151 if (!(in_produce && op->name == func.name())) {
152 return IRMutator::visit(op);
153 }
154 vector<Expr> values = op->values;
155 for (Expr &i : values) {
156 i = mutate(i);
157 }
158 vector<Expr> args = op->args;
159 for (Expr &i : args) {
160 i = mutate(i);
161 }
162 bool sliding_up = old_bounds.max.same_as(new_bounds.max);
163 Expr is_new = sliding_up ? new_bounds.min <= args[dim] : args[dim] <= new_bounds.max;
164 args[dim] -= old_bounds.min;
165 vector<Expr> old_args = args;
166 Expr old_arg_dim = expand_expr(old_args[dim], scope);
167 old_args[dim] = substitute(loop_var, Variable::make(Int(32), loop_var) - 1, old_arg_dim);
168 for (int i = 0; i < (int)values.size(); i++) {
169 Type t = values[i].type();
170 Expr old_value =
171 Call::make(t, op->name, old_args, Call::Halide, func.get_contents(), i);
172 values[i] = Call::make(values[i].type(), Call::if_then_else, {is_new, values[i], likely(old_value)}, Call::PureIntrinsic);
173 }
174 if (const Variable *v = op->args[dim].as<Variable>()) {
175 // The subtractions above simplify more easily if the loop is rebased to 0.
176 loops_to_rebase.insert(v->name);
177 }
178 return Provide::make(func.name(), values, args);
179 }
180
181 Expr visit(const Call *op) override {
182 if (!(op->call_type == Call::Halide && op->name == func.name())) {
183 return IRMutator::visit(op);
184 }
185 vector<Expr> args = op->args;
186 for (Expr &i : args) {
187 i = mutate(i);
188 }
189 args[dim] -= old_bounds.min;
190 return Call::make(op->type, op->name, args, Call::Halide, op->func, op->value_index, op->image, op->param);
191 }
192
193 Stmt visit(const For *op) override {
194 Stmt result = IRMutator::visit(op);
195 op = result.as<For>();
196 internal_assert(op);
197 if (loops_to_rebase.count(op->name)) {
198 string new_name = op->name + ".rebased";
199 Stmt body = substitute(op->name, Variable::make(Int(32), new_name) + op->min, op->body);
200 result = For::make(new_name, 0, op->extent, op->for_type, op->device_api, body);
201 loops_to_rebase.erase(op->name);
202 }
203 return result;
204 }
205
206 Stmt visit(const LetStmt *op) override {
207 ScopedBinding<Expr> bind(scope, op->name, simplify(expand_expr(op->value, scope)));
208 return IRMutator::visit(op);
209 }
210
211public:
212 RollFunc(const Function &func, int dim, const string &loop_var,
213 const Interval &old_bounds, const Interval &new_bounds)
214 : func(func), dim(dim), loop_var(loop_var), old_bounds(old_bounds), new_bounds(new_bounds) {
215 }
216};

218// Perform sliding window optimization for a function over a
219// particular serial for loop
220class SlidingWindowOnFunctionAndLoop : public IRMutator {
221 Function func;
222 string loop_var;
223 Expr loop_min;
224 set<int> &slid_dimensions;
225 Scope<Expr> scope;
226
227 map<string, Expr> replacements;
228
229 using IRMutator::visit;
230
231 // Check if the dimension at index 'dim_idx' is always pure (i.e. equal to 'dim')
232 // in the definition (including in its specializations)
233 bool is_dim_always_pure(const Definition &def, const string &dim, int dim_idx) {
234 const Variable *var = def.args()[dim_idx].as<Variable>();
235 if ((!var) || (var->name != dim)) {
236 return false;
237 }
238
239 for (const auto &s : def.specializations()) {
240 bool pure = is_dim_always_pure(s.definition, dim, dim_idx);
241 if (!pure) {
242 return false;
243 }
244 }
245 return true;
246 }
247
248 Stmt visit(const ProducerConsumer *op) override {
249 if (op->is_producer) {
250 if (op->name != func.name()) {
251 return IRMutator::visit(op);
252 }
253
254 // We're interested in the case where exactly one of the
255 // dimensions of the buffer has a min/extent that depends
256 // on the loop_var.
257 string dim = "";
258 int dim_idx = 0;
259 Expr min_required, max_required;
260
261 debug(3) << "Considering sliding " << func.name()
262 << " along loop variable " << loop_var << "\n"
263 << "Region provided:\n";
264
265 string prefix = func.name() + ".s" + std::to_string(func.updates().size()) + ".";
266 const std::vector<string> func_args = func.args();
267 for (int i = 0; i < func.dimensions(); i++) {
268 if (slid_dimensions.count(i)) {
269 debug(3) << "Already slid over dimension " << i << ", so skipping it.\n";
270 continue;
271 }
272 // Look up the region required of this function's last stage
273 string var = prefix + func_args[i];
274 internal_assert(scope.contains(var + ".min") && scope.contains(var + ".max"));
275 Expr min_req = scope.get(var + ".min");
276 Expr max_req = scope.get(var + ".max");
277 min_req = expand_expr(min_req, scope);
278 max_req = expand_expr(max_req, scope);
279
280 debug(3) << func_args[i] << ":" << min_req << ", " << max_req << "\n";
281 if (expr_depends_on_var(min_req, loop_var) ||
282 expr_depends_on_var(max_req, loop_var)) {
283 if (!dim.empty()) {
284 dim = "";
285 min_required = Expr();
286 max_required = Expr();
287 break;
288 } else {
289 dim = func_args[i];
290 dim_idx = i;
291 min_required = min_req;
292 max_required = max_req;
293 }
294 } else if (!min_required.defined() &&
295 i == func.dimensions() - 1 &&
296 is_pure(min_req) &&
297 is_pure(max_req)) {
298 // The footprint doesn't depend on the loop var. Just compute everything on the first loop iteration.
299 dim = func_args[i];
300 dim_idx = i;
301 min_required = min_req;
302 max_required = max_req;
303 }
304 }
305
306 if (!min_required.defined()) {
307 debug(3) << "Could not perform sliding window optimization of "
308 << func.name() << " over " << loop_var << " because multiple "
309 << "dimensions of the function dependended on the loop var\n";
310 return op;
311 }
312
313 // If the function is not pure in the given dimension, give up. We also
314 // need to make sure that it is pure in all the specializations
315 bool pure = true;
316 for (const Definition &def : func.updates()) {
317 pure = is_dim_always_pure(def, dim, dim_idx);
318 if (!pure) {
319 break;
320 }
321 }
322 if (!pure) {
323 debug(3) << "Could not performance sliding window optimization of "
324 << func.name() << " over " << loop_var << " because the function "
325 << "scatters along the related axis.\n";
326 return op;
327 }
328
329 bool can_slide_up = false;
330 bool can_slide_down = false;
331
332 Monotonic monotonic_min = is_monotonic(min_required, loop_var);
333 Monotonic monotonic_max = is_monotonic(max_required, loop_var);
334
335 if (monotonic_min == Monotonic::Increasing ||
336 monotonic_min == Monotonic::Constant) {
337 can_slide_up = true;
338 } else if (monotonic_min == Monotonic::Unknown) {
339 if (get_compiler_logger()) {
340 get_compiler_logger()->record_non_monotonic_loop_var(loop_var, min_required);
341 }
342 }
343
344 if (monotonic_max == Monotonic::Decreasing ||
345 monotonic_max == Monotonic::Constant) {
346 can_slide_down = true;
347 } else if (monotonic_max == Monotonic::Unknown) {
348 if (get_compiler_logger()) {
349 get_compiler_logger()->record_non_monotonic_loop_var(loop_var, max_required);
350 }
351 }
352
353 if (!can_slide_up && !can_slide_down) {
354 debug(3) << "Not sliding " << func.name()
355 << " over dimension " << dim
356 << " along loop variable " << loop_var
357 << " because I couldn't prove it moved monotonically along that dimension\n"
358 << "Min is " << min_required << "\n"
359 << "Max is " << max_required << "\n";
360 return op;
361 }
362
363 // Ok, we've isolated a function, a dimension to slide
364 // along, and loop variable to slide over.
365 debug(3) << "Sliding " << func.name()
366 << " over dimension " << dim
367 << " along loop variable " << loop_var << "\n";
368
369 Expr loop_var_expr = Variable::make(Int(32), loop_var);
370
371 Expr prev_max_plus_one = substitute(loop_var, loop_var_expr - 1, max_required) + 1;
372 Expr prev_min_minus_one = substitute(loop_var, loop_var_expr - 1, min_required) - 1;
373
374 // If there's no overlap between adjacent iterations, we shouldn't slide.
375 if (can_prove(min_required >= prev_max_plus_one) ||
376 can_prove(max_required <= prev_min_minus_one)) {
377 debug(3) << "Not sliding " << func.name()
378 << " over dimension " << dim
379 << " along loop variable " << loop_var
380 << " there's no overlap in the region computed across iterations\n"
381 << "Min is " << min_required << "\n"
382 << "Max is " << max_required << "\n";
383 return op;
384 }
385
386 // Update the bounds of this producer assuming the previous iteration
387 // has run already.
388 Expr new_min, new_max;
389 if (can_slide_up) {
390 new_min = prev_max_plus_one;
391 new_max = max_required;
392 } else {
393 new_min = min_required;
394 new_max = prev_min_minus_one;
395 }
396
397 // See if we can find a new min for the loop that can warm up the
398 // sliding window. We're going to do this by building an equation
399 // that describes the constraints we have on our new loop min. The
400 // first constraint is that the new loop min is not after the
401 // loop min.
402 string new_loop_min_name = unique_name('x');
403 Expr new_loop_min_var = Variable::make(Int(32), new_loop_min_name);
404 Expr new_loop_min_eq = new_loop_min_var <= loop_min;
405 Expr new_min_at_new_loop_min = substitute(loop_var, new_loop_min_var, new_min);
406 Expr new_max_at_new_loop_min = substitute(loop_var, new_loop_min_var, new_max);
407 if (can_slide_up) {
408 // We need to find a new loop min that satisfies these constraints:
409 // - The new min at the new loop min needs to be before the min
410 // required at the original min.
411 // - The new max needs to be greater than the new min, both at the
412 // new loop min. This guarantees that the sliding window.
413 // Together, these conditions guarantee the sliding window is warmed
414 // up. The first condition checks that we reached the original loop
415 // min, and the second condition checks that the iterations before
416 // the original min weren't empty.
417 Expr min_required_at_loop_min = substitute(loop_var, loop_min, min_required);
418 new_loop_min_eq = new_loop_min_eq &&
419 new_min_at_new_loop_min <= min_required_at_loop_min &&
420 new_max_at_new_loop_min >= new_min_at_new_loop_min;
421 } else {
422 // When sliding down, the constraints are similar, just swapping
423 // the roles of the min and max.
424 Expr max_required_at_loop_min = substitute(loop_var, loop_min, max_required);
425 new_loop_min_eq = new_loop_min_eq &&
426 new_max_at_new_loop_min >= max_required_at_loop_min &&
427 new_min_at_new_loop_min <= new_max_at_new_loop_min;
428 }
429 // Try to solve the equation.
430 new_loop_min_eq = simplify(new_loop_min_eq);
431 Interval solve_result = solve_for_inner_interval(new_loop_min_eq, new_loop_min_name);
432 internal_assert(!new_loop_min.defined());
433 if (solve_result.has_upper_bound() && !equal(solve_result.max, loop_min)) {
434 new_loop_min = simplify(solve_result.max);
435
436 // We have a new loop min, so we an assume every iteration has
437 // a previous iteration. In order for this to be safe, we need
438 // the new min/max at the new loop min to be less than or equal to
439 // the min/max required at the original loop min.
440 Expr loop_var_expr = Variable::make(Int(32), loop_var);
441 Expr orig_loop_min_expr = Variable::make(Int(32), loop_var + ".loop_min.orig");
442 if (can_slide_up) {
443 Expr min_required_at_loop_min = substitute(loop_var, orig_loop_min_expr, min_required);
444 new_min = max(new_min, min_required_at_loop_min);
445 } else {
446 Expr max_required_at_loop_min = substitute(loop_var, orig_loop_min_expr, max_required);
447 new_max = min(new_max, max_required_at_loop_min);
448 }
449 } else {
450 // We couldn't find a suitable new loop min, we can't assume
451 // every iteration has a previous iteration. The first iteration
452 // will warm up the loop.
453 Expr need_explicit_warmup = loop_var_expr <= loop_min;
454 if (can_slide_up) {
455 new_min = select(need_explicit_warmup, min_required, likely_if_innermost(new_min));
456 } else {
457 new_max = select(need_explicit_warmup, max_required, likely_if_innermost(new_max));
458 }
459 }
460 new_min = simplify(new_min);
461 new_max = simplify(new_max);
462
463 debug(3) << "Sliding " << func.name() << ", " << dim << "\n"
464 << "Pushing min up from " << min_required << " to " << new_min << "\n"
465 << "Shrinking max from " << max_required << " to " << new_max << "\n"
466 << "Adjusting loop_min from " << loop_min << " to " << new_loop_min << "\n"
467 << "Equation is " << new_loop_min_eq << "\n";
468
469 slid_dimensions.insert(dim_idx);
470
471 // If we want to slide in registers, we're done here, we just need to
472 // save the updated bounds for later.
473 if (func.schedule().memory_type() == MemoryType::Register) {
474 this->dim_idx = dim_idx;
475 old_bounds = {min_required, max_required};
476 new_bounds = {new_min, new_max};
477 return op;
478 }
479
480 // If we aren't sliding in registers, we need to update the bounds of
481 // the producer to be only the bounds of the region newly computed.
482 internal_assert(replacements.empty());
483 if (can_slide_up) {
484 replacements[prefix + dim + ".min"] = new_min;
485 } else {
486 replacements[prefix + dim + ".max"] = new_max;
487 }
488
489 for (size_t i = 0; i < func.updates().size(); i++) {
490 string n = func.name() + ".s" + std::to_string(i) + "." + dim;
491 replacements[n + ".min"] = Variable::make(Int(32), prefix + dim + ".min");
492 replacements[n + ".max"] = Variable::make(Int(32), prefix + dim + ".max");
493 }
494
495 // Ok, we have a new min/max required and we're going to
496 // rewrite all the lets that define bounds required. Now
497 // we need to additionally expand the bounds required of
498 // the last stage to cover values produced by stages
499 // before the last one. Because, e.g., an intermediate
500 // stage may be unrolled, expanding its bounds provided.
501 Stmt result = op;
502 if (!func.updates().empty()) {
503 Box b = box_provided(op->body, func.name());
504 if (can_slide_up) {
505 string n = prefix + dim + ".min";
506 Expr var = Variable::make(Int(32), n);
507 result = LetStmt::make(n, min(var, b[dim_idx].min), result);
508 } else {
509 string n = prefix + dim + ".max";
510 Expr var = Variable::make(Int(32), n);
511 result = LetStmt::make(n, max(var, b[dim_idx].max), result);
512 }
513 }
514
515 return result;
516 } else if (!find_produce(op, func.name()) && new_loop_min.defined()) {
517 // The producer might have expanded the loop before the min to warm
518 // up the window. This consumer doesn't contain a producer that might
519 // be part of the warmup, so guard it with an if to only run it on
520 // the original loop bounds.
521 Expr loop_var_expr = Variable::make(Int(32), loop_var);
522 Expr orig_loop_min_expr = Variable::make(Int(32), loop_var + ".loop_min.orig");
523 Expr guard = likely_if_innermost(orig_loop_min_expr <= loop_var_expr);
524
525 // Put the if inside the consumer node, so semaphores end up outside the if.
526 // TODO: This is correct, but it produces slightly suboptimal code: if we
527 // didn't do this, the loop could likely be trimmed and the if simplified away.
528 Stmt body = mutate(op->body);
529 if (const IfThenElse *old_guard = body.as<IfThenElse>()) {
530 Expr x = Variable::make(Int(32), "*");
531 vector<Expr> matches;
532 if (expr_match(likely_if_innermost(x <= loop_var_expr), old_guard->condition, matches)) {
533 // There's already a condition on loop_var_expr here. Since we're
534 // adding a condition at the old loop min, this if must already be
535 // guarding more than we will.
536 guard = Expr();
537 }
538 }
539 if (guard.defined()) {
540 debug(3) << "Guarding body " << guard << "\n";
541 body = IfThenElse::make(guard, body);
542 }
543 if (body.same_as(op->body)) {
544 return op;
545 } else {
546 return ProducerConsumer::make_consume(op->name, body);
547 }
548 } else {
549 return IRMutator::visit(op);
550 }
551 }
552
553 Stmt visit(const For *op) override {
554 // It's not safe to enter an inner loop whose bounds depend on
555 // the var we're sliding over.
556 Expr min = expand_expr(op->min, scope);
557 Expr extent = expand_expr(op->extent, scope);
558 if (is_const_one(extent)) {
559 // Just treat it like a let
560 Stmt s = LetStmt::make(op->name, min, op->body);
561 s = mutate(s);
562 // Unpack it back into the for
563 const LetStmt *l = s.as<LetStmt>();
564 internal_assert(l);
565 return For::make(op->name, op->min, op->extent, op->for_type, op->device_api, l->body);
566 } else if (is_monotonic(min, loop_var) != Monotonic::Constant ||
567 is_monotonic(extent, loop_var) != Monotonic::Constant) {
568 debug(3) << "Not entering loop over " << op->name
569 << " because the bounds depend on the var we're sliding over: "
570 << min << ", " << extent << "\n";
571 return op;
572 } else {
573 return IRMutator::visit(op);
574 }
575 }
576
577 Stmt visit(const LetStmt *op) override {
578 ScopedBinding<Expr> bind(scope, op->name, simplify(expand_expr(op->value, scope)));
579 Stmt new_body = mutate(op->body);
580
581 Expr value = op->value;
582
583 map<string, Expr>::iterator iter = replacements.find(op->name);
584 if (iter != replacements.end()) {
585 value = iter->second;
586 replacements.erase(iter);
587 }
588
589 if (new_body.same_as(op->body) && value.same_as(op->value)) {
590 return op;
591 } else {
592 return LetStmt::make(op->name, value, new_body);
593 }
594 }

596public:
597 SlidingWindowOnFunctionAndLoop(Function f, string v, Expr v_min, set<int> &slid_dimensions)
598 : func(std::move(f)), loop_var(std::move(v)), loop_min(std::move(v_min)), slid_dimensions(slid_dimensions) {
599 }
600
601 Expr new_loop_min;
602 int dim_idx;
603 Interval old_bounds;
604 Interval new_bounds;
605
606 Stmt translate_loop(const Stmt &s) {
607 return RollFunc(func, dim_idx, loop_var, old_bounds, new_bounds).mutate(s);
608 }

611// In Stmt s, does the production of b depend on a?
612// We can't use produce/consume nodes to determine this, because they're "loose".
613// For example, we get this:
614//
615// produce a {
616// a(...) = ...
617// }
618// consume a {
619// produce b {
620// b(...) = ... // not depending on a
621// }
622// consume b {
623// c(...) = a(...) + b(...)
624// }
625// }
626//
627// When we'd rather see this:
628//
629// produce a {
630// a(...) = ...
631// }
632// produce b {
633// b(...) = ... // not depending on a
634// }
635// consume a {
636// consume b {
637// c(...) = a(...) + b(...)
638// }
639// }
640//
641// TODO: We might also need to figure out transitive dependencies...? If so, it
642// would be best to just fix the produce/consume relationships as above. We would
643// just be able to look for produce b inside produce a.
644class Dependencies : public IRVisitor {
645 using IRVisitor::visit;
646
647 const string &producer;
648 bool in_producer = false;
649
650 void visit(const ProducerConsumer *op) override {

672bool depends_on(const string &a, const string &b, const Stmt &s, map<string, bool> &cache) {
673 if (a == b) {
674 return true;
675 }
676 auto cached = cache.find(b);
677 if (cached != cache.end()) {
678 return cached->second;
679 }
680 Dependencies deps(b);
681 s.accept(&deps);
682 // Recursively search for dependencies.
683 for (const string &i : deps.dependencies) {
684 if (depends_on(a, i, s, cache)) {
685 cache[b] = true;
686 return true;
687 }
688 }
689 cache[b] = false;
690 return false;
691}

694 map<string, bool> cache;
695 return depends_on(a, b, s, cache);

724// Perform sliding window optimization for all functions
725class SlidingWindow : public IRMutator {
726 const map<string, Function> &env;
727
728 // A map of which dimensions we've already slid over, by Func name.
729 map<string, set<int>> slid_dimensions;
730
731 // Keep track of realizations we want to slide, from innermost to
732 // outermost.
733 list<Function> sliding;
734
735 using IRMutator::visit;
736
737 Stmt visit(const Realize *op) override {
738 // Find the args for this function
739 map<string, Function>::const_iterator iter = env.find(op->name);
740
741 // If it's not in the environment it's some anonymous
742 // realization that we should skip (e.g. an inlined reduction)
743 if (iter == env.end()) {
744 return IRMutator::visit(op);
745 }
746
747 // If the Function in question has the same compute_at level
748 // as its store_at level, skip it.
749 const FuncSchedule &sched = iter->second.schedule();
750 if (sched.compute_level() == sched.store_level()) {
751 return IRMutator::visit(op);
752 }
753
754 // We want to slide innermost first, so put it on the front of
755 // the list.
756 sliding.push_front(iter->second);
757 Stmt new_body = mutate(op->body);
758 sliding.pop_front();
759 // Remove tracking of slid dimensions when we're done realizing
760 // it in case a realization appears elsewhere.
761 auto slid_it = slid_dimensions.find(iter->second.name());
762 if (slid_it != slid_dimensions.end()) {
763 slid_dimensions.erase(slid_it);
764 }
765
766 if (new_body.same_as(op->body)) {
767 return op;
768 } else {
769 return Realize::make(op->name, op->types, op->memory_type,
770 op->bounds, op->condition, new_body);
771 }
772 }
773
774 Stmt visit(const For *op) override {
775 if (!(op->for_type == ForType::Serial || op->for_type == ForType::Unrolled)) {
776 return IRMutator::visit(op);
777 }
778 debug(3) << "Doing sliding window analysis on loop " << op->name << "\n";
779
780 string name = op->name;
781 Stmt body = op->body;
782 Expr loop_min = op->min;
783 Expr loop_extent = op->extent;
784 Expr loop_max = Variable::make(Int(32), op->name + ".loop_max");
785
786 list<pair<string, Expr>> prev_loop_mins;
787 list<pair<string, Expr>> new_lets;
788 for (const Function &func : sliding) {
789 debug(3) << "Doing sliding window analysis on function " << func.name() << "\n";
790
791 // Figure out where we should start sliding from. If no
792 // other func needs this func, we can just start at the
793 // original loop min.
794 Expr prev_loop_min = op->min;
795 // If a previously slid func needs this func to be warmed
796 // up, then we need to back up the loop to warm up this
797 // func before the already slid func starts warming up.
798 for (const auto &i : prev_loop_mins) {
799 if (depends_on(func.name(), i.first, body)) {
800 prev_loop_min = i.second;
801 break;
802 }
803 }
804
805 SlidingWindowOnFunctionAndLoop slider(func, name, prev_loop_min, slid_dimensions[func.name()]);
806 body = slider.mutate(body);
807
808 if (func.schedule().memory_type() == MemoryType::Register &&
809 slider.old_bounds.has_lower_bound()) {
810 body = slider.translate_loop(body);
811 }
812
813 if (slider.new_loop_min.defined()) {
814 Expr new_loop_min = slider.new_loop_min;
815 if (!prev_loop_min.same_as(loop_min)) {
816 // If we didn't start sliding from the previous
817 // loop min, we the old loop min might already
818 // be further back than this new one.
819 new_loop_min = min(new_loop_min, loop_min);
820 }
821
822 // Put this at the front of the list, so we find it first
823 // when checking subsequent funcs.
824 prev_loop_mins.emplace_front(func.name(), new_loop_min);
825
826 // Update the loop body to use the adjusted loop min.
827 string new_name = name + ".$n";
828 loop_min = Variable::make(Int(32), new_name + ".loop_min");
829 loop_extent = Variable::make(Int(32), new_name + ".loop_extent");
830 body = substitute({
831 {name, Variable::make(Int(32), new_name)},
832 {name + ".loop_min", loop_min},
833 {name + ".loop_extent", loop_extent},
834 },
835 body);
836 body = SubstitutePrefetchVar(name, new_name).mutate(body);
837
838 name = new_name;
839
840 // The new loop interval is the new loop min to the loop max.
841 new_lets.emplace_front(name + ".loop_min", new_loop_min);
842 new_lets.emplace_front(name + ".loop_min.orig", loop_min);
843 new_lets.emplace_front(name + ".loop_extent", (loop_max - loop_min) + 1);
844 }
845 }
846
847 body = mutate(body);
848
849 if (body.same_as(op->body) && loop_min.same_as(op->min) && loop_extent.same_as(op->extent) && name == op->name) {
850 return op;
851 } else {
852 Stmt result = For::make(name, loop_min, loop_extent, op->for_type, op->device_api, body);
853 if (!new_lets.empty()) {
854 result = LetStmt::make(name + ".loop_max", loop_max, result);
855 }
856 for (const auto &i : new_lets) {
857 result = LetStmt::make(i.first, i.second, result);
858 }
859 return result;
860 }
861 }
862
863 Stmt visit(const IfThenElse *op) override {
864 // Don't let specializations corrupt the tracking of which
865 // dimensions have been slid.
866 map<string, set<int>> old_slid_dimensions = slid_dimensions;
867 Stmt then_case = mutate(op->then_case);
868 slid_dimensions = old_slid_dimensions;
869 Stmt else_case = mutate(op->else_case);
870 slid_dimensions = old_slid_dimensions;
871 if (then_case.same_as(op->then_case) && else_case.same_as(op->else_case)) {
872 return op;
873 } else {
874 return IfThenElse::make(op->condition, then_case, else_case);
875 }
876 }
877
878public:
879 SlidingWindow(const map<string, Function> &e)
880 : env(e) {
881 }
882};

884// It is convenient to be able to assume that loops have a .loop_min.orig
885// let in addition to .loop_min. Most of these will get simplified away.
886class AddLoopMinOrig : public IRMutator {
887 using IRMutator::visit;
888
889 Stmt visit(const For *op) override {
890 Stmt body = mutate(op->body);
891 Expr min = mutate(op->min);
892 Expr extent = mutate(op->extent);
893
894 Stmt result;
895 if (body.same_as(op->body) && min.same_as(op->min) && extent.same_as(op->extent)) {
896 result = op;
897 } else {
898 result = For::make(op->name, min, extent, op->for_type, op->device_api, body);
899 }
900 return LetStmt::make(op->name + ".loop_min.orig", Variable::make(Int(32), op->name + ".loop_min"), result);
901 }
