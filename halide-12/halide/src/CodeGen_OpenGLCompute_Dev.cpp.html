120// Maps Halide types to appropriate GLSL types or emit error if no equivalent
121// type is available.
122Type CodeGen_OpenGLCompute_C::map_type(const Type &type) {
123 Type result = type;
124 if (type.is_scalar()) {
125 if (type.is_float()) {
126 user_assert(type.bits() <= 32)
127 << "GLSL: Can't represent a float with " << type.bits() << " bits.\n";
128 result = Float(32);
129 } else if (type.is_bool()) {
130 // unchanged
131 } else if (type.is_int() && type.bits() <= 32) {
132 result = Int(32);
133 } else if (type.is_uint() && type.bits() <= 32) {
134 result = UInt(32);
135 } else {
136 user_error << "GLSL: Can't represent type '" << type << "'.\n";
137 }
138 } else {
139 user_assert(type.lanes() <= 4)
140 << "GLSL: vector types wider than 4 aren't supported\n";
141 user_assert(type.is_bool() || type.is_int() || type.is_uint() || type.is_float())
142 << "GLSL: Can't represent vector type '" << type << "'.\n";
143 Type scalar_type = type.element_of();
144 result = map_type(scalar_type).with_lanes(type.lanes());
145 }
146 return result;
147}

153 return replace_all(mangled, "__", "XX");

156string CodeGen_OpenGLCompute_C::print_type(Type type, AppendSpaceIfNeeded space) {
157 ostringstream oss;
158 type = map_type(type);
159 if (type.is_scalar()) {
160 if (type.is_float()) {
161 oss << "float";
162 } else if (type.is_bool()) {
163 oss << "bool";
164 } else if (type.is_int()) {
165 oss << "int";
166 } else if (type.is_uint()) {
167 oss << "uint";
168 } else {
169 internal_error << "GLSL: invalid type '" << type << "' encountered.\n";
170 }
171 } else {
172 if (type.is_float()) {
173 // no prefix for float vectors
174 } else if (type.is_bool()) {
175 oss << "b";
176 } else if (type.is_int()) {
177 oss << "i";
178 } else if (type.is_uint()) {
179 oss << "u";
180 } else {
181 internal_error << "GLSL: invalid type '" << type << "' encountered.\n";
182 }
183 oss << "vec" << type.lanes();
184 }
185
186 if (space == AppendSpace) {
187 oss << " ";
188 }
189
190 return oss.str();
191}

193string simt_intrinsic(const string &name) {
194 if (ends_with(name, ".__thread_id_x")) {
195 return "gl_LocalInvocationID.x";
196 } else if (ends_with(name, ".__thread_id_y")) {
197 return "gl_LocalInvocationID.y";
198 } else if (ends_with(name, ".__thread_id_z")) {
199 return "gl_LocalInvocationID.z";
200 } else if (ends_with(name, ".__thread_id_w")) {
201 internal_error << "4-dimension loops with " << name << " are not supported\n";
202 } else if (ends_with(name, ".__block_id_x")) {
203 return "gl_WorkGroupID.x";
204 } else if (ends_with(name, ".__block_id_y")) {
205 return "gl_WorkGroupID.y";
206 } else if (ends_with(name, ".__block_id_z")) {
207 return "gl_WorkGroupID.z";
208 } else if (ends_with(name, ".__block_id_w")) {
209 internal_error << "4-dimension loops with " << name << " are not supported\n";
210 }
211 internal_error << "simt_intrinsic called on bad variable name: " << name << "\n";
212 return "";
213}

215int thread_loop_workgroup_index(const string &name) {
216 string ids[] = {".__thread_id_x",
217 ".__thread_id_y",
218 ".__thread_id_z",
219 ".__thread_id_w"};
220 for (size_t i = 0; i < sizeof(ids) / sizeof(string); i++) {
221 if (ends_with(name, ids[i])) {
222 return i;
223 }
224 }
225 return -1;

228void CodeGen_OpenGLCompute_C::visit(const FloatImm *op) {
229 ostringstream oss;
230 // Print integral numbers with trailing ".0". For fractional numbers use a
231 // precision of 9 digits, which should be enough to recover the binary
232 // float unambiguously from the decimal representation (if iostreams
233 // implements correct rounding).
234 const float truncated = (op->value < 0 ? std::ceil(op->value) : std::floor(op->value));
235 if (truncated == op->value) {
236 oss << std::fixed << std::setprecision(1) << op->value;
237 } else {
238 oss << std::setprecision(9) << op->value;
239 }
240 id = oss.str();
241}

263void CodeGen_OpenGLCompute_C::visit(const Mod *op) {
264 if (op->type.is_int() || op->type.is_uint()) {
265 // Just exploit the Euclidean identity
266 // FIXME: Why doesn't lower_euclidean_mod work for glsl?
267 // https://github.com/halide/Halide/issues/4979
268 Expr zero = make_zero(op->type);
269 Expr equiv = select(op->a == zero, zero,
270 op->a - (op->a / op->b) * op->b);
271 equiv = common_subexpression_elimination(equiv);
272 print_expr(equiv);
273 } else {
274 print_expr(Call::make(op->type, "mod", {op->a, op->b}, Call::Extern));
275 }

328void CodeGen_OpenGLCompute_C::visit(const Shuffle *op) {
329 // The halide Shuffle represents the llvm intrinisc
330 // shufflevector, however, for GLSL its use is limited to swizzling
331 // up to a four channel vec type.
332
333 internal_assert(op->vectors.size() == 1);
334
335 int shuffle_lanes = op->type.lanes();
336 internal_assert(shuffle_lanes <= 4);
337
338 string expr = print_expr(op->vectors[0]);
339
340 // Create a swizzle expression for the shuffle
341 string swizzle;
342 for (int i = 0; i != shuffle_lanes; ++i) {
343 int channel = op->indices[i];
344 internal_assert(channel < 4) << "Shuffle of invalid channel";
345 swizzle += get_lane_suffix(channel);
346 }
347
348 print_assignment(op->type, expr + "." + swizzle);
349}

351void CodeGen_OpenGLCompute_C::visit(const Call *op) {
352 if (op->is_intrinsic(Call::gpu_thread_barrier)) {
353 internal_assert(op->args.size() == 1) << "gpu_thread_barrier() intrinsic must specify memory fence type.\n";
354
355 const auto *fence_type_ptr = as_const_int(op->args[0]);
356 internal_assert(fence_type_ptr) << "gpu_thread_barrier() parameter is not a constant integer.\n";
357 auto fence_type = *fence_type_ptr;
358
359 stream << get_indent() << "barrier();\n";
360
361 // barrier() is an execution barrier; for memory behavior, we'll use the
362 // least-common-denominator groupMemoryBarrier(), because other fence types
363 // require extensions or GL 4.3 as a minumum.
364 if (fence_type & CodeGen_GPU_Dev::MemoryFenceType::Device ||
365 fence_type & CodeGen_GPU_Dev::MemoryFenceType::Shared) {
366 stream << "groupMemoryBarrier();\n";
367 }
368 print_assignment(op->type, "0");
369 } else if (op->is_intrinsic(Call::lerp)) {
370 // Implement lerp using GLSL's mix() function, which always uses
371 // floating point arithmetic.
372 Expr zero_val = op->args[0];
373 Expr one_val = op->args[1];
374 Expr weight = op->args[2];
375
376 internal_assert(weight.type().is_uint() || weight.type().is_float());
377 if (weight.type().is_uint()) {
378 // Normalize integer weights to [0.0f, 1.0f] range.
379 internal_assert(weight.type().bits() < 32);
380 weight = Div::make(Cast::make(Float(32), weight),
381 Cast::make(Float(32), weight.type().max()));
382 } else if (op->type.is_uint()) {
383 // Round float weights down to next multiple of (1/op->type.imax())
384 // to give same results as lerp based on integer arithmetic.
385 internal_assert(op->type.bits() < 32);
386 weight = floor(weight * op->type.max()) / op->type.max();
387 }
388
389 Type result_type = Float(32, op->type.lanes());
390 Expr e = Call::make(result_type, "mix", {zero_val, one_val, weight}, Call::Extern);
391
392 if (!op->type.is_float()) {
393 // Mirror rounding implementation of Halide's integer lerp.
394 e = Cast::make(op->type, floor(e + 0.5f));
395 }
396 print_expr(e);
397 return;
398 } else if (op->is_intrinsic(Call::abs)) {
399 internal_assert(op->args.size() == 1);
400 Expr a = op->args[0];
401 Type target_type = map_type(op->type);
402 if (op->type != Int(32)) {
403 print_assignment(target_type, print_type(target_type) + "(abs(" + print_expr(a) + "))");
404 } else {
405 print_assignment(target_type, "abs(" + print_expr(a) + ")");
406 }
407 return;
408 } else if (op->is_intrinsic(Call::absd)) {
409 internal_assert(op->args.size() == 2);
410 Expr a = op->args[0];
411 Expr b = op->args[1];
412 Expr e = cast(op->type, select(a < b, b - a, a - b));
413 print_expr(e);
414 return;
415 } else if (op->is_intrinsic(Call::return_second)) {
416 internal_assert(op->args.size() == 2);
417 // Simply discard the first argument, which is generally a call to
418 // 'halide_printf'.
419 print_assignment(op->type, print_expr(op->args[1]));
420 return;
421 } else if (op->name == "fast_inverse_f32") {
422 print_expr(make_one(op->type) / op->args[0]);
423 return;
424 } else if (op->name == "fast_inverse_sqrt_f32") {
425 print_expr(make_one(op->type) / sqrt(op->args[0]));
426 return;
427 } else if (op->name == "pow_f32") {
428 if (can_prove(op->args[0] > 0)) {
429 ostringstream rhs;
430 rhs << "pow(" << print_expr(op->args[0]) << ", " << print_expr(op->args[1]) << ")";
431 print_assignment(op->type, rhs.str());
432 return;
433 } else {
434 ostringstream base;
435 string a = print_expr(op->args[0]);
436 string b = print_expr(op->args[1]);
437 base << "pow(abs(" << a << "), " << b << ")";
438 string c = print_assignment(op->type, base.str());
439 Expr a_var = is_const(op->args[0]) ? op->args[0] : Variable::make(op->type, a);
440 Expr b_var = is_const(op->args[1]) ? op->args[1] : Variable::make(op->type, b);
441 Expr c_var = Variable::make(op->type, c);
442 // OpenGL isn't required to produce NaNs, so we return
443 // zero in the undefined case.
444 Expr equiv = select(a_var > 0 || b_var % 2 == 0, c_var,
445 b_var % 2 == 1, -c_var,
446 0.0f);
447 print_expr(simplify(equiv));
448 return;
449 }
450 } else if (op->is_intrinsic(Call::shift_right)) {
451 print_assignment(op->type, print_expr(op->args[0]) + " >> " + print_expr(op->args[1]));
452 } else if (op->is_intrinsic(Call::shift_left)) {
453 print_assignment(op->type, print_expr(op->args[0]) + " << " + print_expr(op->args[1]));
454 } else if (op->is_intrinsic(Call::bitwise_not)) {
455 print_assignment(op->type, "~" + print_expr(op->args[0]));
456 } else if (op->is_intrinsic(Call::bitwise_and)) {
457 print_assignment(op->type, print_expr(op->args[0]) + " & " + print_expr(op->args[1]));
458 } else if (op->is_intrinsic(Call::bitwise_or)) {
459 print_assignment(op->type, print_expr(op->args[0]) + " | " + print_expr(op->args[1]));
460 } else if (op->is_intrinsic(Call::bitwise_xor)) {
461 print_assignment(op->type, print_expr(op->args[0]) + " ^ " + print_expr(op->args[1]));
462 } else if (op->is_intrinsic(Call::div_round_to_zero)) {
463 print_assignment(op->type, print_expr(op->args[0]) + " / " + print_expr(op->args[1]));
464 } else if (op->is_intrinsic(Call::mod_round_to_zero)) {
465 print_assignment(op->type, print_expr(op->args[0]) + " % " + print_expr(op->args[1]));
466 } else {
467 auto it = builtin.find(op->name);
468 if (it == builtin.end()) {
469 user_error << "GLSL: unknown function '" << op->name << "' encountered.\n";
470 }
471
472 ostringstream rhs;
473 rhs << it->second << "(";
474 for (size_t i = 0; i < op->args.size(); i++) {
475 if (i > 0) {
476 rhs << ", ";
477 }
478 rhs << print_expr(op->args[i]);
479 }
480 rhs << ")";
481 print_assignment(op->type, rhs.str());
482 }

485void CodeGen_OpenGLCompute_C::visit(const Cast *op) {
486 Type value_type = op->value.type();
487 // If both types are represented by the same GLSL type, no explicit cast
488 // is necessary.
489 Type target_type = map_type(op->type);
490 if (target_type == map_type(value_type)) {
491 Expr value = op->value;
492 if (value_type.code() == Type::Float) {
493 // float->int conversions may need explicit truncation if an
494 // integer type is embedded into a float. (Note: overflows are
495 // considered undefined behavior, so we do nothing about values
496 // that are out of range of the target type.)
497 if (op->type.code() == Type::UInt) {
498 value = simplify(floor(value));
499 } else if (op->type.code() == Type::Int) {
500 value = simplify(trunc(value));
501 }
502 }
503 // FIXME: Overflow is not UB for most Halide types
504 // https://github.com/halide/Halide/issues/4975
505 value.accept(this);
506 } else {
507 print_assignment(target_type, print_type(target_type) + "(" + print_expr(op->value) + ")");
508 }
509}

511void CodeGen_OpenGLCompute_C::visit(const For *loop) {
512 user_assert(loop->for_type != ForType::GPULane)
513 << "The OpenGLCompute backend does not support the gpu_lanes() scheduling directive.";
514
515 if (CodeGen_GPU_Dev::is_gpu_var(loop->name)) {
516 internal_assert((loop->for_type == ForType::GPUBlock) ||
517 (loop->for_type == ForType::GPUThread))
518 << "kernel loop must be either gpu block or gpu thread\n";
519 internal_assert(is_const_zero(loop->min));
520
521 debug(4) << "loop extent is " << loop->extent << "\n";
522 //
523 // Need to extract workgroup size.
524 //
525 int index = thread_loop_workgroup_index(loop->name);
526 if (index >= 0) {
527 const IntImm *int_limit = loop->extent.as<IntImm>();
528 user_assert(int_limit != nullptr) << "For OpenGLCompute workgroup size must be a constant integer.\n";
529 int new_workgroup_size = int_limit->value;
530 user_assert(workgroup_size[index] == 0 ||
531 workgroup_size[index] == new_workgroup_size)
532 << "OpenGLCompute requires all gpu kernels have same workgroup size, "
533 << "but two different ones were encountered " << workgroup_size[index]
534 << " and " << new_workgroup_size
535 << " in dimension " << index << ".\n";
536 workgroup_size[index] = new_workgroup_size;
537 debug(4) << "Workgroup size for index " << index << " is " << workgroup_size[index] << "\n";
538 }
539
540 stream << get_indent() << print_type(Int(32)) << " " << print_name(loop->name)
541 << " = int(" << simt_intrinsic(loop->name) << ");\n";
542
543 loop->body.accept(this);
544
545 } else {
546 user_assert(loop->for_type != ForType::Parallel)
547 << "Cannot use parallel loops inside OpenGLCompute kernel\n";
548 CodeGen_C::visit(loop);
549 }
550}

552void CodeGen_OpenGLCompute_C::visit(const Ramp *op) {
553 ostringstream rhs;
554 rhs << print_type(op->type) << "(";
555
556 if (op->lanes > 4) {
557 internal_error << "GLSL: ramp lanes " << op->lanes << " is not supported\n";
558 }
559
560 rhs << print_expr(op->base);
561
562 for (int i = 1; i < op->lanes; ++i) {
563 rhs << ", " << print_expr(Add::make(op->base, Mul::make(i, op->stride)));
564 }
565
566 rhs << ")";
567 print_assignment(op->base.type(), rhs.str());
568}

593void CodeGen_OpenGLCompute_C::visit(const Store *op) {

601

609 // values from before the store.

688void CodeGen_OpenGLCompute_C::add_kernel(const Stmt &s,
689 const string &name,
690 const vector<DeviceArgument> &args) {
691
692 debug(2) << "Adding OpenGLCompute kernel " << name << "\n";
693 cache.clear();
694
695 if (target.os == Target::Android) {
696 stream << "#version 310 es\n"
697 << "#extension GL_ANDROID_extension_pack_es31a : require\n";
698 } else if (target.has_feature(Target::EGL)) {
699 stream << "#version 310 es\n";
700 } else {
701 stream << "#version 430\n";
702 }
703 stream << "float float_from_bits(int x) { return intBitsToFloat(int(x)); }\n";
704 stream << "#define halide_unused(x) (void)(x)\n";
705
706 for (size_t i = 0; i < args.size(); i++) {
707 if (args[i].is_buffer) {
708 //
709 // layout(binding = 10) buffer buffer10 {
710 // vec3 data[];
711 // } inBuffer;
712 //
713 stream << "layout(binding=" << i << ")"
714 << " buffer buffer" << i << " { "
715 << print_type(args[i].type) << " data[]; } "
716 << print_name(args[i].name) << ";\n";
717 } else {
718 stream << "layout(location = " << i << ") uniform " << print_type(args[i].type)
719 << " " << print_name(args[i].name) << ";\n";
720 }
721 }
722
723 // Find all the shared allocations and declare them at global scope.
724 FindSharedAllocations fsa;
725 s.accept(&fsa);
726 for (const Allocate *op : fsa.allocs) {
727 internal_assert(op->extents.size() == 1 && is_const(op->extents[0]));
728 stream << "shared "
729 << print_type(op->type) << " "
730 << print_name(op->name) << "["
731 << op->extents[0] << "];\n";
732 }
733
734 // We'll figure out the workgroup size while traversing the stmt
735 workgroup_size[0] = 0;
736 workgroup_size[1] = 0;
737 workgroup_size[2] = 0;
738
739 stream << "void main()\n{\n";
740 indent += 2;
741 print(s);
742 indent -= 2;
743 stream << "}\n";
744
745 // Declare the workgroup size.
746 indent += 2;
747 stream << "layout(local_size_x = " << workgroup_size[0];
748 if (workgroup_size[1] > 1) {
749 stream << ", local_size_y = " << workgroup_size[1];
750 }
751 if (workgroup_size[2] > 1) {
752 stream << ", local_size_z = " << workgroup_size[2];
753 }
754 stream << ") in;\n// end of kernel " << name << "\n";
755 indent -= 2;
756}

764void CodeGen_OpenGLCompute_C::visit(const Allocate *op) {
765 debug(2) << "OpenGLCompute: Allocate " << op->name << " of type " << op->type << " on device\n";
766
767 stream << get_indent();
768 Allocation alloc;
769 alloc.type = op->type;
770 allocations.push(op->name, alloc);
771
772 internal_assert(!op->extents.empty());
773 Expr extent = 1;
774 for (const Expr &e : op->extents) {
775 extent *= e;
776 }
777 extent = simplify(extent);
778 internal_assert(is_const(extent));
779
780 if (op->memory_type != MemoryType::GPUShared) {
781 stream << "{\n";
782 indent += 2;
783 stream << get_indent();
784 // Shared allocations were already declared at global scope.
785 stream << print_type(op->type) << " "
786 << print_name(op->name) << "["
787 << op->extents[0] << "];\n";
788 }
789 op->body.accept(this);
790
791 if (op->memory_type != MemoryType::GPUShared) {
792 indent -= 2;
793 stream << get_indent() << "}\n";
794 }
795}

800 allocations.pop(op->name);
