6Expr Simplify::visit(const Not *op, ExprInfo *bounds) {
7 Expr a = mutate(op->a, nullptr);
8
9 auto rewrite = IRMatcher::rewriter(IRMatcher::not_op(a), op->type);
10
11 if (rewrite(!c0, fold(!c0)) ||
12 rewrite(!(x < y), y <= x) ||
13 rewrite(!(x <= y), y < x) ||
14 rewrite(!(x == y), x != y) ||
15 rewrite(!(x != y), x == y) ||
16 rewrite(!!x, x)) {
17 return rewrite.result;
18 }
19
20 if (rewrite(!broadcast(x, c0), broadcast(!x, c0)) ||
21 rewrite(!intrin(Call::likely, x), intrin(Call::likely, !x)) ||
22 rewrite(!intrin(Call::likely_if_innermost, x), intrin(Call::likely_if_innermost, !x)) ||
23 rewrite(!(!x && y), x || !y) ||
24 rewrite(!(!x || y), x && !y) ||
25 rewrite(!(x && !y), !x || y) ||
26 rewrite(!(x || !y), !x && y) ||
27 false) {
28 return mutate(rewrite.result, bounds);
29 }
30
31 if (a.same_as(op->a)) {
32 return op;
33 } else {
34 return Not::make(a);
35 }
36}
