61ErrorReport::ErrorReport(const char *file, int line, const char *condition_string, int flags)
62 : flags(flags) {
63// Note that we deliberately try to put the entire message into a single line
64// (aside from newlines inserted by user code) to make it easy to filter
65// specific warnings or messages via (e.g.) grep.... unless we are likely to be
66// outputting to a proper terminal, in which case newlines are used to improve readability.
67#if defined(HALIDE_WITH_EXCEPTIONS)
68 const bool use_newlines = false;
69#else
70 const bool use_newlines = (custom_error_reporter == nullptr) && isatty(2);
71#endif
72 const char sep = use_newlines ? '\n' : ' ';
73
74 const std::string &source_loc = Introspection::get_source_location();
75 const char *what = (flags & Warning) ? "Warning" : "Error";
76 if (flags & User) {
77 // Only mention where inside of libHalide the error tripped if we have debug level > 0
78 debug(1) << "User error triggered at " << file << ":" << line << "\n";
79 if (condition_string) {
80 debug(1) << "Condition failed: " << condition_string << "\n";
81 }
82 msg << what << ":";
83 if (!source_loc.empty()) {
84 msg << " (at " << source_loc << ")";
85 }
86 msg << sep;
87 } else {
88 msg << "Internal " << what << " at " << file << ":" << line;
89 if (source_loc.empty()) {
90 msg << " triggered by user code at " << source_loc << ":";
91 }
92 msg << sep;
93 if (condition_string) {
94 msg << "Condition failed: " << condition_string << ":" << sep;
95 }
96 }
97}

99ErrorReport::~ErrorReport()
100#if __cplusplus >= 201100 || _MSC_VER >= 1900
101 noexcept(false)
102#endif
103{
104 if (!msg.str().empty() && msg.str().back() != '\n') {
105 msg << "\n";
106 }
107
108 if (custom_error_reporter != nullptr) {
109 if (flags & Warning) {
110 custom_error_reporter->warning(msg.str().c_str());
111 return;
112 } else {
113 custom_error_reporter->error(msg.str().c_str());
114 // error() should not have returned to us, but just in case
115 // it does, make sure we don't continue.
116 abort();
117 }
118 }
119
120 // TODO: Add an option to error out on warnings too
121 if (flags & Warning) {
122 std::cerr << msg.str();
123 return;
124 }
125
126#ifdef HALIDE_WITH_EXCEPTIONS
127 if (std::uncaught_exception()) {
128 // This should never happen - evaluating one of the arguments
129 // to the error message would have to throw an
130 // exception. Nonetheless, in case it does, preserve the
131 // exception already in flight and suppress this one.
132 return;
133 } else if (flags & Runtime) {
134 throw RuntimeError(msg.str());
135 } else if (flags & User) {
136 throw CompileError(msg.str());
137 } else {
138 throw InternalError(msg.str());
139 }
140#else
141 std::cerr << msg.str();
142 abort();
143#endif
144}
