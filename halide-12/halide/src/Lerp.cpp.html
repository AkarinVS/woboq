13Expr lower_lerp(Expr zero_val, Expr one_val, const Expr &weight) {
14
15 Expr result;
16
17 internal_assert(zero_val.type() == one_val.type());
18 internal_assert(weight.type().is_uint() || weight.type().is_float())
19 << "Bad weight type: " << weight.type() << "\n";
20
21 Type result_type = zero_val.type();
22
23 Expr bias_value = make_zero(result_type);
24 Type computation_type = result_type;
25
26 if (zero_val.type().is_int()) {
27 computation_type = UInt(zero_val.type().bits(), zero_val.type().lanes());
28 // We must take care to do the addition and subtraction of the
29 // bias while in the unsigned computation type, where overflow
30 // is well-defined.
31 bias_value = cast(computation_type, result_type.min());
32 }
33
34 // For signed integer types, just convert everything to unsigned
35 // and then back at the end to ensure proper rounding, etc.
36 // There is likely a better way to handle this.
37 if (result_type != computation_type) {
38 zero_val = Cast::make(computation_type, zero_val) - bias_value;
39 one_val = Cast::make(computation_type, one_val) - bias_value;
40 }
41
42 if (result_type.is_bool()) {
43 Expr half_weight;
44 if (weight.type().is_float()) {
45 half_weight = 0.5f;
46 } else {
47 half_weight = weight.type().max() / 2;
48 }
49
50 result = select(weight > half_weight, one_val, zero_val);
51 } else {
52 Expr typed_weight;
53 Expr inverse_typed_weight;
54
55 if (weight.type().is_float()) {
56 typed_weight = weight;
57 if (computation_type.is_uint()) {
58 // TODO: Verify this reduces to efficient code or
59 // figure out a better way to express a multiply
60 // of unsigned 2^32-1 by a double promoted weight
61 if (computation_type.bits() == 32) {
62 typed_weight =
63 Cast::make(computation_type,
64 cast<double>(Expr(65535.0f)) * cast<double>(Expr(65537.0f)) *
65 Cast::make(Float(64, typed_weight.type().lanes()), typed_weight));
66 } else {
67 typed_weight =
68 Cast::make(computation_type,
69 computation_type.max() * typed_weight);
70 }
71 inverse_typed_weight = computation_type.max() - typed_weight;
72 } else {
73 inverse_typed_weight = make_one(computation_type) - typed_weight;
74 }
75
76 } else {
77 if (computation_type.is_float()) {
78 int weight_bits = weight.type().bits();
79 Expr denom = make_const(computation_type, (ldexp(1.0, weight_bits) - 1));
80 typed_weight = Cast::make(computation_type, weight) / denom;
81 inverse_typed_weight = make_one(computation_type) - typed_weight;
82 } else {
83 // This code rescales integer weights to the right number of bits.
84 // It takes advantage of (2^n - 1) == (2^(n/2) - 1)(2^(n/2) + 1)
85 // e.g. 65535 = 255 * 257. (Ditto for the 32-bit equivalent.)
86 // To recale a weight of m bits to be n bits, we need to do:
87 // scaled_weight = (weight / (2^m - 1)) * (2^n - 1)
88 // which power of two values for m and n, results in a series like
89 // so:
90 // (2^(m/2) + 1) * (2^(m/4) + 1) ... (2^(n*2) + 1)
91 // The loop below computes a scaling constant and either multiples
92 // or divides by the constant and relies on lowering and llvm to
93 // generate efficient code for the operation.
94 int bit_size_difference = weight.type().bits() - computation_type.bits();
95 if (bit_size_difference == 0) {
96 typed_weight = weight;
97 } else {
98 typed_weight = Cast::make(computation_type, weight);
99
100 int bits_left = ::abs(bit_size_difference);
101 int shift_amount = std::min(computation_type.bits(), weight.type().bits());
102 uint64_t scaling_factor = 1;
103 while (bits_left != 0) {
104 internal_assert(bits_left > 0);
105 scaling_factor = scaling_factor + (scaling_factor << shift_amount);
106 bits_left -= shift_amount;
107 shift_amount *= 2;
108 }
109 if (bit_size_difference < 0) {
110 typed_weight =
111 Cast::make(computation_type, weight) *
112 cast(computation_type, (int32_t)scaling_factor);
113 } else {
114 typed_weight =
115 Cast::make(computation_type,
116 weight / cast(weight.type(), (int32_t)scaling_factor));
117 }
118 }
119 inverse_typed_weight =
120 Cast::make(computation_type,
121 computation_type.max() - typed_weight);
122 }
123 }
124
125 if (computation_type.is_float()) {
126 result = (zero_val * inverse_typed_weight +
127 one_val * typed_weight);
128 } else {
129 int32_t bits = computation_type.bits();
130 switch (bits) {
131 case 1:
132 result = select(typed_weight, one_val, zero_val);
133 break;
134 case 8:
135 case 16:
136 case 32: {
137 Expr zero_expand = Cast::make(UInt(2 * bits, computation_type.lanes()),
138 zero_val);
139 Expr one_expand = Cast::make(UInt(2 * bits, one_val.type().lanes()),
140 one_val);
141
142 Expr rounding = Cast::make(UInt(2 * bits), 1) << Cast::make(UInt(2 * bits), (bits - 1));
143 Expr divisor = Cast::make(UInt(2 * bits), 1) << Cast::make(UInt(2 * bits), bits);
144
145 Expr prod_sum = zero_expand * inverse_typed_weight +
146 one_expand * typed_weight + rounding;
147 Expr divided = ((prod_sum / divisor) + prod_sum) / divisor;
148
149 result = Cast::make(UInt(bits, computation_type.lanes()), divided);
150 break;
151 }
152 case 64:
153 // TODO: 64-bit lerp is not supported as current approach
154 // requires double-width multiply.
155 // There is an informative error message in IROperator.h.
156 internal_error << "Can't do a 64-bit lerp.\n";
157 break;
158 default:
159 break;
160 }
161 }
162
163 if (!is_const_zero(bias_value)) {
164 result = Cast::make(result_type, result + bias_value);
165 }
166 }
167
168 return simplify(common_subexpression_elimination(result));
169}
