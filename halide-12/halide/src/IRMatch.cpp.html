18void expr_match_test() {
19 vector<Expr> matches;
20 Expr w = Variable::make(Int(32), "*");
21 Expr fw = Variable::make(Float(32), "*");
22 Expr x = Variable::make(Int(32), "x");
23 Expr y = Variable::make(Int(32), "y");
24 Expr fx = Variable::make(Float(32), "fx");
25 Expr fy = Variable::make(Float(32), "fy");
26
27 Expr vec_wild = Variable::make(Int(32, 4), "*");
28
29 internal_assert(expr_match(w, 3, matches) &&
30 equal(matches[0], 3));
31
32 internal_assert(expr_match(w + 3, (y * 2) + 3, matches) &&
33 equal(matches[0], y * 2));
34
35 internal_assert(expr_match(fw * 17 + cast<float>(w + cast<int>(fw)),
36 (81.0f * fy) * 17 + cast<float>(x / 2 + cast<int>(x + 4.5f)), matches) &&
37 matches.size() == 3 &&
38 equal(matches[0], 81.0f * fy) &&
39 equal(matches[1], x / 2) &&
40 equal(matches[2], x + 4.5f));
41
42 internal_assert(!expr_match(fw + 17, fx + 18, matches) &&
43 matches.empty());
44 internal_assert(!expr_match((w * 2) + 17, fx + 17, matches) &&
45 matches.empty());
46 internal_assert(!expr_match(w * 3, 3 * x, matches) &&
47 matches.empty());
48
49 internal_assert(expr_match(vec_wild * 3, Ramp::make(x, y, 4) * 3, matches));
50
51 std::cout << "expr_match test passed" << std::endl;
52}

72 bool types_match(Type pattern_type, Type expr_type) {
73 bool bits_matches = (pattern_type.bits() == 0) || (pattern_type.bits() == expr_type.bits());
74 bool lanes_matches = (pattern_type.lanes() == 0) || (pattern_type.lanes() == expr_type.lanes());
75 bool code_matches = (pattern_type.code() == expr_type.code());
76 return bits_matches && lanes_matches && code_matches;
77 }

79 void visit(const IntImm *op) override {
80 const IntImm *e = expr.as<IntImm>();
81 if (!e ||
82 e->value != op->value ||
83 !types_match(op->type, e->type)) {
84 result = false;
85 }

97 void visit(const FloatImm *op) override {
98 const FloatImm *e = expr.as<FloatImm>();
99 // Note we use uint64_t equality instead of double equality to
100 // catch NaNs. We're checking for the same bits.
101 if (!e ||
102 reinterpret_bits<uint64_t>(e->value) !=
103 reinterpret_bits<uint64_t>(op->value) ||
104 !types_match(op->type, e->type)) {
105 result = false;
106 }
107 }

109 void visit(const Cast *op) override {
110 const Cast *e = expr.as<Cast>();
111 if (result && e && types_match(op->type, e->type)) {
112 expr = e->value;
113 op->value.accept(this);
114 } else {
115 result = false;
116 }
117 }
118
119 void visit(const Variable *op) override {
120 if (!result) {
121 return;
122 }
123
124 if (!types_match(op->type, expr.type())) {
125 result = false;
126 } else if (matches) {
127 if (op->name == "*") {
128 matches->push_back(expr);
129 } else {
130 const Variable *e = expr.as<Variable>();
131 result = e && (e->name == op->name);
132 }
133 } else if (var_matches) {
134 Expr &match = (*var_matches)[op->name];
135 if (match.defined()) {
136 result = equal(match, expr);
137 } else {
138 match = expr;
139 }
140 }
141 }

260 void visit(const Call *op) override {
261 const Call *e = expr.as<Call>();
262 if (result && e &&
263 types_match(op->type, e->type) &&
264 e->name == op->name &&
265 e->value_index == op->value_index &&
266 e->call_type == op->call_type &&
267 e->args.size() == op->args.size()) {
268 for (size_t i = 0; result && (i < e->args.size()); i++) {
269 expr = e->args[i];

302bool expr_match(const Expr &pattern, const Expr &expr, vector<Expr> &matches) {
303 matches.clear();
304 if (!pattern.defined() && !expr.defined()) {
305 return true;
306 }
307 if (!pattern.defined() || !expr.defined()) {
308 return false;
309 }
310
311 IRMatch eq(expr, matches);
312 pattern.accept(&eq);
313 if (eq.result) {
314 return true;
315 } else {
316 matches.clear();
317 return false;
318 }
319}

321bool expr_match(const Expr &pattern, const Expr &expr, map<string, Expr> &matches) {
322 // Explicitly don't clear matches. This allows usages to pre-match
323 // some variables.
324
325 if (!pattern.defined() && !expr.defined()) {
326 return true;
327 }
328 if (!pattern.defined() || !expr.defined()) {
329 return false;
330 }
331
332 IRMatch eq(expr, matches);
333 pattern.accept(&eq);
334 if (eq.result) {
335 return true;
336 } else {
337 matches.clear();
338 return false;
339 }

344class WithLanes : public IRMutator {
345 using IRMutator::visit;
346
347 int lanes;
348
349 Type with_lanes(Type t) const {
350 return t.with_lanes(lanes);
351 }
352
353 Expr visit(const Cast *op) override {
354 if (op->type.lanes() != lanes) {
355 return Cast::make(with_lanes(op->type), mutate(op->value));
356 } else {
357 return IRMutator::visit(op);
358 }
359 }
360
361 Expr visit(const Variable *op) override {
362 if (op->type.lanes() != lanes) {
363 return Variable::make(with_lanes(op->type), op->name);
364 } else {
365 return op;
366 }
367 }
368
369 Expr visit(const Broadcast *op) override {
370 if (op->type.lanes() != lanes) {
371 return Broadcast::make(op->value, lanes);
372 } else {
373 return IRMutator::visit(op);
374 }
375 }
376
377public:
378 WithLanes(int lanes)
379 : lanes(lanes) {
380 }
381};

392bool equal_helper(const Expr &a, const Expr &b) {
393 return equal(*a.get(), *b.get());

396template<typename Op>
397HALIDE_ALWAYS_INLINE bool equal_helper_binop(const BaseExprNode &a, const BaseExprNode &b) {
398 return (equal_helper(((const Op &)a).a, ((const Op &)b).a) &&
399 equal_helper(((const Op &)a).b, ((const Op &)b).b));
400}

402HALIDE_ALWAYS_INLINE
403bool equal_helper(int a, int b) {
404 return a == b;
405}
406
407template<typename T>
408HALIDE_ALWAYS_INLINE bool equal_helper(const std::vector<T> &a, const std::vector<T> &b) {
409 if (a.size() != b.size()) {
410 return false;
411 }
412 for (size_t i = 0; i < a.size(); i++) {
413 if (!equal_helper(a[i], b[i])) {
414 return false;
415 }
416 }
417 return true;
418}
419
420bool equal_helper(const BaseExprNode &a, const BaseExprNode &b) noexcept {
421 switch (a.node_type) {
422 case IRNodeType::IntImm:
423 return ((const IntImm &)a).value == ((const IntImm &)b).value;
424 case IRNodeType::UIntImm:
425 return ((const UIntImm &)a).value == ((const UIntImm &)b).value;
426 case IRNodeType::FloatImm:
427 return ((const FloatImm &)a).value == ((const FloatImm &)b).value;
428 case IRNodeType::StringImm:
429 return ((const StringImm &)a).value == ((const StringImm &)b).value;
430 case IRNodeType::Cast:
431 // While we know a and b have matching type, we don't know
432 // that the types of the values match, so use equal rather
433 // than equal_helper.
434 return equal(((const Cast &)a).value, ((const Cast &)b).value);
435 case IRNodeType::Variable:
436 return ((const Variable &)a).name == ((const Variable &)b).name;
437 case IRNodeType::Add:
438 return equal_helper_binop<Add>(a, b);
439 case IRNodeType::Sub:
440 return equal_helper_binop<Sub>(a, b);
441 case IRNodeType::Mul:
442 return equal_helper_binop<Mul>(a, b);
443 case IRNodeType::Div:
444 return equal_helper_binop<Div>(a, b);
445 case IRNodeType::Mod:
446 return equal_helper_binop<Mod>(a, b);
447 case IRNodeType::Min:
448 return equal_helper_binop<Min>(a, b);
449 case IRNodeType::Max:
450 return equal_helper_binop<Max>(a, b);
451 case IRNodeType::EQ:
452 return equal_helper_binop<EQ>(a, b);
453 case IRNodeType::NE:
454 return equal_helper_binop<NE>(a, b);
455 case IRNodeType::LT:
456 return equal_helper_binop<LT>(a, b);
457 case IRNodeType::LE:
458 return equal_helper_binop<LE>(a, b);
459 case IRNodeType::GT:
460 return equal_helper_binop<GT>(a, b);
461 case IRNodeType::GE:
462 return equal_helper_binop<GE>(a, b);
463 case IRNodeType::And:
464 return equal_helper_binop<And>(a, b);
465 case IRNodeType::Or:
466 return equal_helper_binop<Or>(a, b);
467 case IRNodeType::Not:
468 return equal_helper(((const Not &)a).a, ((const Not &)b).a);
469 case IRNodeType::Select:
470 return (equal_helper(((const Select &)a).condition, ((const Select &)b).condition) &&
471 equal_helper(((const Select &)a).true_value, ((const Select &)b).true_value) &&
472 equal_helper(((const Select &)a).false_value, ((const Select &)b).false_value));
473 case IRNodeType::Load:
474 return (((const Load &)a).name == ((const Load &)b).name &&
475 equal_helper(((const Load &)a).index, ((const Load &)b).index));
476 case IRNodeType::Ramp:
477 return (equal_helper(((const Ramp &)a).base, ((const Ramp &)b).base) &&
478 equal_helper(((const Ramp &)a).stride, ((const Ramp &)b).stride));
479 case IRNodeType::Broadcast:
480 return equal_helper(((const Broadcast &)a).value, ((const Broadcast &)b).value);
481 case IRNodeType::Call:
482 return (((const Call &)a).name == ((const Call &)b).name &&
483 ((const Call &)a).call_type == ((const Call &)b).call_type &&
484 ((const Call &)a).value_index == ((const Call &)b).value_index &&
485 equal_helper(((const Call &)a).args, ((const Call &)b).args));
486 case IRNodeType::Let:
487 return (((const Let &)a).name == ((const Let &)b).name &&
488 equal_helper(((const Let &)a).value, ((const Let &)b).value) &&
489 equal_helper(((const Let &)a).body, ((const Let &)b).body));
490 case IRNodeType::Shuffle:
491 return (equal_helper(((const Shuffle &)a).vectors, ((const Shuffle &)b).vectors) &&
492 equal_helper(((const Shuffle &)a).indices, ((const Shuffle &)b).indices));
493 case IRNodeType::VectorReduce:
494 // As with Cast above, we use equal instead of equal_helper
495 // here, because while we know a.type == b.type, we don't know
496 // if the types of the value fields also match. We could be
497 // comparing a reduction of an 8-vector down to a 4 vector to
498 // a reduction of a 16-vector down to a 4-vector.
499 return (((const VectorReduce &)a).op == ((const VectorReduce &)b).op &&
500 equal(((const VectorReduce &)a).value, ((const VectorReduce &)b).value));
501
502 // Explicitly list all the Stmts instead of using a default
503 // clause so that if new Exprs are added without being handled
504 // here we get a compile-time error.
505 case IRNodeType::LetStmt:
506 case IRNodeType::AssertStmt:
507 case IRNodeType::ProducerConsumer:
508 case IRNodeType::For:
509 case IRNodeType::Acquire:
510 case IRNodeType::Store:
511 case IRNodeType::Provide:
512 case IRNodeType::Allocate:
513 case IRNodeType::Free:
514 case IRNodeType::Realize:
515 case IRNodeType::Block:
516 case IRNodeType::Fork:
517 case IRNodeType::IfThenElse:
518 case IRNodeType::Evaluate:
519 case IRNodeType::Prefetch:
520 case IRNodeType::Atomic:
521 break;
522 }
523 return false;
524}
