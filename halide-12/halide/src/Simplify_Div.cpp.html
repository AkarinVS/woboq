6Expr Simplify::visit(const Div *op, ExprInfo *bounds) {
7 ExprInfo a_bounds, b_bounds;
8 Expr a = mutate(op->a, &a_bounds);
9 Expr b = mutate(op->b, &b_bounds);
10
11 if (bounds && no_overflow_int(op->type)) {
12 bounds->min = INT64_MAX;
13 bounds->max = INT64_MIN;
14
15 // Enumerate all possible values for the min and max and take the extreme values.
16 if (a_bounds.min_defined && b_bounds.min_defined && b_bounds.min != 0) {
17 int64_t v = div_imp(a_bounds.min, b_bounds.min);
18 bounds->min = std::min(bounds->min, v);
19 bounds->max = std::max(bounds->max, v);
20 }
21
22 if (a_bounds.min_defined && b_bounds.max_defined && b_bounds.max != 0) {
23 int64_t v = div_imp(a_bounds.min, b_bounds.max);
24 bounds->min = std::min(bounds->min, v);
25 bounds->max = std::max(bounds->max, v);
26 }
27
28 if (a_bounds.max_defined && b_bounds.max_defined && b_bounds.max != 0) {
29 int64_t v = div_imp(a_bounds.max, b_bounds.max);
30 bounds->min = std::min(bounds->min, v);
31 bounds->max = std::max(bounds->max, v);
32 }
33
34 if (a_bounds.max_defined && b_bounds.min_defined && b_bounds.min != 0) {
35 int64_t v = div_imp(a_bounds.max, b_bounds.min);
36 bounds->min = std::min(bounds->min, v);
37 bounds->max = std::max(bounds->max, v);
38 }
39
40 const bool b_positive = b_bounds.min_defined && b_bounds.min > 0;
41 const bool b_negative = b_bounds.max_defined && b_bounds.max < 0;
42
43 if ((b_positive && !b_bounds.max_defined) ||
44 (b_negative && !b_bounds.min_defined)) {
45 // Take limit as b -> +/- infinity
46 int64_t v = 0;
47 bounds->min = std::min(bounds->min, v);
48 bounds->max = std::max(bounds->max, v);
49 }
50
51 bounds->min_defined = ((a_bounds.min_defined && b_positive) ||
52 (a_bounds.max_defined && b_negative));
53 bounds->max_defined = ((a_bounds.max_defined && b_positive) ||
54 (a_bounds.min_defined && b_negative));
55
56 // That's as far as we can get knowing the sign of the
57 // denominator. For bounded numerators, we additionally know
58 // that div can't make anything larger in magnitude, so we can
59 // take the intersection with that.
60 if (a_bounds.max_defined && a_bounds.min_defined) {
61 int64_t v = std::max(a_bounds.max, -a_bounds.min);
62 if (bounds->min_defined) {
63 bounds->min = std::max(bounds->min, -v);
64 } else {
65 bounds->min = -v;
66 }
67 if (bounds->max_defined) {
68 bounds->max = std::min(bounds->max, v);
69 } else {
70 bounds->max = v;
71 }
72 bounds->min_defined = bounds->max_defined = true;
73 }
74
75 // Bounded numerator divided by constantish
76 // denominator can sometimes collapse things to a
77 // constant at this point
78 if (bounds->min_defined &&
79 bounds->max_defined &&
80 bounds->max == bounds->min) {
81 if (op->type.can_represent(bounds->min)) {
82 return make_const(op->type, bounds->min);
83 } else {
84 // Even though this is 'no-overflow-int', if the result
85 // we calculate can't fit into the destination type,
86 // we're better off returning an overflow condition than
87 // a known-wrong value. (Note that no_overflow_int() should
88 // only be true for signed integers.)
89 internal_assert(op->type.is_int());
90 return make_signed_integer_overflow(op->type);
91 }
92 }
93 // Code downstream can use min/max in calculated-but-unused arithmetic
94 // that can lead to UB (and thus, flaky failures under ASAN/UBSAN)
95 // if we leave them set to INT64_MAX/INT64_MIN; normalize to zero to avoid this.
96 if (!bounds->min_defined) {
97 bounds->min = 0;
98 }
99 if (!bounds->max_defined) {
100 bounds->max = 0;
101 }
102 bounds->alignment = a_bounds.alignment / b_bounds.alignment;
103 bounds->trim_bounds_using_alignment();
104 }
105
106 bool denominator_non_zero =
107 (no_overflow_int(op->type) &&
108 ((b_bounds.min_defined && b_bounds.min > 0) ||
109 (b_bounds.max_defined && b_bounds.max < 0) ||
110 (b_bounds.alignment.remainder != 0)));
111
112 if (may_simplify(op->type)) {
113
114 int lanes = op->type.lanes();
115
116 auto rewrite = IRMatcher::rewriter(IRMatcher::div(a, b), op->type);
117
118 if (rewrite(IRMatcher::Overflow() / x, a) ||
119 rewrite(x / IRMatcher::Overflow(), b) ||
120 rewrite(x / 1, x) ||
121 rewrite(c0 / c1, fold(c0 / c1)) ||
122 (!op->type.is_float() && rewrite(x / 0, 0)) ||
123 (!op->type.is_float() && denominator_non_zero && rewrite(x / x, 1)) ||
124 rewrite(0 / x, 0) ||
125 false) {
126 return rewrite.result;
127 }
128
129 int a_mod = a_bounds.alignment.modulus;
130 int a_rem = a_bounds.alignment.remainder;
131
132 // clang-format off
133 if (EVAL_IN_LAMBDA
134 (rewrite(broadcast(x, c0) / broadcast(y, c0), broadcast(x / y, c0)) ||
135 rewrite(select(x, c0, c1) / c2, select(x, fold(c0/c2), fold(c1/c2))) ||
136 (!op->type.is_float() &&
137 rewrite(x / x, select(x == 0, 0, 1))) ||
138 (no_overflow(op->type) &&
139 (// Fold repeated division
140 rewrite((x / c0) / c2, x / fold(c0 * c2), c0 > 0 && c2 > 0 && !overflows(c0 * c2)) ||
141 rewrite((x / c0 + c1) / c2, (x + fold(c1 * c0)) / fold(c0 * c2), c0 > 0 && c2 > 0 && !overflows(c0 * c2) && !overflows(c0 * c1)) ||
142 rewrite((x * c0) / c1, x / fold(c1 / c0), c1 % c0 == 0 && c0 > 0 && c1 / c0 != 0) ||
143 // Pull out terms that are a multiple of the denominator
144 rewrite((x * c0) / c1, x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
145
146 rewrite((x * c0 + y) / c1, y / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
147 rewrite((x * c0 - y) / c0, x + (0 - y) / c0) ||
148 rewrite((x * c1 - y) / c0, (0 - y) / c0 - x, c0 + c1 == 0) ||
149 rewrite((y + x * c0) / c1, y / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
150 rewrite((y - x * c0) / c1, y / c1 - x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
151
152 rewrite(((x * c0 + y) + z) / c1, (y + z) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
153 rewrite(((x * c0 - y) + z) / c1, (z - y) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
154 rewrite(((x * c0 + y) - z) / c1, (y - z) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
155 rewrite(((x * c0 - y) - z) / c0, x + (0 - y - z) / c0) ||
156 rewrite(((x * c1 - y) - z) / c0, (0 - y - z) / c0 - x, c0 + c1 == 0) ||
157
158 rewrite(((y + x * c0) + z) / c1, (y + z) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
159 rewrite(((y + x * c0) - z) / c1, (y - z) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
160 rewrite(((y - x * c0) - z) / c1, (y - z) / c1 - x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
161 rewrite(((y - x * c0) + z) / c1, (y + z) / c1 - x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
162
163 rewrite((z + (x * c0 + y)) / c1, (z + y) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
164 rewrite((z + (x * c0 - y)) / c1, (z - y) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
165 rewrite((z - (x * c0 - y)) / c1, (z + y) / c1 - x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
166 rewrite((z - (x * c0 + y)) / c1, (z - y) / c1 + x * fold(-c0 / c1), c0 % c1 == 0 && c1 > 0) ||
167
168 rewrite((z + (y + x * c0)) / c1, (z + y) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
169 rewrite((z - (y + x * c0)) / c1, (z - y) / c1 + x * fold(-c0 / c1), c0 % c1 == 0 && c1 > 0) ||
170 rewrite((z + (y - x * c0)) / c1, (z + y) / c1 - x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
171 rewrite((z - (y - x * c0)) / c1, (z - y) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
172
173 // For the next depth, stick to addition
174 rewrite((((x * c0 + y) + z) + w) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
175 rewrite((((y + x * c0) + z) + w) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
176 rewrite(((z + (x * c0 + y)) + w) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
177 rewrite(((z + (y + x * c0)) + w) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
178 rewrite((w + ((x * c0 + y) + z)) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
179 rewrite((w + ((y + x * c0) + z)) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
180 rewrite((w + (z + (x * c0 + y))) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
181 rewrite((w + (z + (y + x * c0))) / c1, (y + z + w) / c1 + x * fold(c0 / c1), c0 % c1 == 0 && c1 > 0) ||
182
183 /** In (x + c0) / c1, when can we pull the constant
184 addition out of the numerator? An obvious answer is
185 the constant is a multiple of the denominator, but
186 there are other cases too. The condition for the
187 rewrite to be correct is:
188
189 (x + c0) / c1 == x / c1 + c2
190
191 Say we know (x + c0) = a_mod * y + a_rem
192
193 (a_mod * y + a_rem) / c1 == (a_mod * y + a_rem - c0) / c1 + c2
194
195 If a_mod % c1 == 0, we can subtract the term in y
196 from both sides and get:
197
198 a_rem / c1 == (a_rem - c0) / c1 + c2
199
200 c2 == a_rem / c1 - (a_rem - c0) / c1
201
202 This is a sufficient and necessary condition for the case when x_mod % c1 == 0.
203 */
204 (no_overflow_int(op->type) &&
205 (rewrite((x + c0) / c1, x / c1 + fold(a_rem / c1 - (a_rem - c0) / c1), a_mod % c1 == 0) ||
206
207 /**
208 Now do the same thing for subtraction from a constant.
209
210 (c0 - x) / c1 == c2 - x / c1
211
212 where c0 - x == a_mod * y + a_rem
213
214 So x = c0 - a_mod * y - a_rem
215
216 (a_mod * y + a_rem) / c1 == c2 - (c0 - a_mod * y - a_rem) / c1
217
218 If a_mod % c1 == 0, we can pull that term out and cancel it:
219
220 a_rem / c1 == c2 - (c0 - a_rem) / c1
221
222 c2 == a_rem / c1 + (c0 - a_rem) / c1
223
224 */
225 rewrite((c0 - x)/c1, fold(a_rem / c1 + (c0 - a_rem) / c1) - x / c1, a_mod % c1 == 0) ||
226
227 // We can also pull it out when the constant is a
228 // multiple of the denominator.
229 rewrite((x + c0) / c1, x / c1 + fold(c0 / c1), c0 % c1 == 0) ||
230 rewrite((c0 - x) / c1, fold(c0 / c1) - x / c1, (c0 + 1) % c1 == 0))) ||
231
232 (denominator_non_zero &&
233 (rewrite((x + y)/x, y/x + 1) ||
234 rewrite((y + x)/x, y/x + 1) ||
235 rewrite((x - y)/x, (-y)/x + 1) ||
236 rewrite((y - x)/x, y/x - 1) ||
237 rewrite(((x + y) + z)/x, (y + z)/x + 1) ||
238 rewrite(((y + x) + z)/x, (y + z)/x + 1) ||
239 rewrite((z + (x + y))/x, (z + y)/x + 1) ||
240 rewrite((z + (y + x))/x, (z + y)/x + 1) ||
241 rewrite((x*y)/x, y) ||
242 rewrite((y*x)/x, y) ||
243 rewrite((x*y + z)/x, y + z/x) ||
244 rewrite((y*x + z)/x, y + z/x) ||
245 rewrite((z + x*y)/x, z/x + y) ||
246 rewrite((z + y*x)/x, z/x + y) ||
247 rewrite((x*y - z)/x, y + (-z)/x) ||
248 rewrite((y*x - z)/x, y + (-z)/x) ||
249 rewrite((z - x*y)/x, z/x - y) ||

251 false)) ||

253 (op->type.is_float() && rewrite(x/c0, x * fold(1/c0))))) ||
254 (no_overflow_int(op->type) &&
255 (
256 rewrite(ramp(x, c0, lanes) / broadcast(c1, lanes), ramp(x / c1, fold(c0 / c1), lanes), (c0 % c1 == 0)) ||
257 rewrite(ramp(x, c0, lanes) / broadcast(c1, lanes), broadcast(x / c1, lanes),
258 // First and last lanes are the same when...
259 can_prove((x % c1 + c0 * (lanes - 1)) / c1 == 0, this))
260 )) ||
261 (no_overflow_scalar_int(op->type) &&
262 (rewrite(x / -1, -x) ||
263 (denominator_non_zero && rewrite(c0 / y, select(y < 0, fold(-c0), c0), c0 == -1)) ||
264 rewrite((x * c0 + c1) / c2,
265 (x + fold(c1 / c0)) / fold(c2 / c0),
266 c2 > 0 && c0 > 0 && c2 % c0 == 0) ||
267 rewrite((x * c0 + c1) / c2,
268 x * fold(c0 / c2) + fold(c1 / c2),
269 c2 > 0 && c0 % c2 == 0) ||

271 rewrite((x % 2 + c0) / 2, x % 2 + fold(c0 / 2), c0 % 2 == 1))))) {
272 return mutate(rewrite.result, bounds);
273 }
