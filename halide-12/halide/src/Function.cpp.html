33// Weaken all the references to a particular Function to break
34// reference cycles. Also count the number of references found.
35class WeakenFunctionPtrs : public IRMutator {
36 using IRMutator::visit;
37
38 Expr visit(const Call *c) override {
39 Expr expr = IRMutator::visit(c);
40 c = expr.as<Call>();
41 internal_assert(c);
42 if (c->func.defined() &&
43 c->func.get() == func) {
44 FunctionPtr ptr = c->func;
45 ptr.weaken();
46 expr = Call::make(c->type, c->name, c->args, c->call_type,
47 ptr, c->value_index,
48 c->image, c->param);
49 count++;
50 }
51 return expr;
52 }
53 FunctionContents *func;
54
55public:
56 int count = 0;
57 WeakenFunctionPtrs(FunctionContents *f)
58 : func(f) {
59 }
60};

66 std::vector<Type> output_types;

68 // The names of the dimensions of the Function. Corresponds to the
69 // LHS of the pure definition if there is one. Is also the initial

75 // within the function.

80

82

84
85 std::vector<ExternFuncArgument> extern_arguments;
86 std::string extern_function_name;
87
88 NameMangling extern_mangling = NameMangling::Default;
89 DeviceAPI extern_function_device_api = DeviceAPI::Host;

94
95 bool frozen = false;

97 void accept(IRVisitor *visitor) const {
98 func_schedule.accept(visitor);
99
100 if (init_def.defined()) {
101 init_def.accept(visitor);
102 }
103 for (const Definition &def : updates) {
104 def.accept(visitor);
105 }
106
107 if (!extern_function_name.empty()) {
108 for (const ExternFuncArgument &i : extern_arguments) {
109 if (i.is_func()) {
110 user_assert(i.func.get() != this)
111 << "Extern Func has itself as an argument";
112 i.func->accept(visitor);
113 } else if (i.is_expr()) {
114 i.expr.accept(visitor);
115 }
116 }
117 if (extern_proxy_expr.defined()) {
118 extern_proxy_expr.accept(visitor);
119 }
120 }
121
122 for (const Parameter &i : output_buffers) {
123 for (size_t j = 0; j < args.size(); j++) {
124 if (i.min_constraint(j).defined()) {
125 i.min_constraint(j).accept(visitor);
126 }
127 if (i.stride_constraint(j).defined()) {
128 i.stride_constraint(j).accept(visitor);
129 }
130 if (i.extent_constraint(j).defined()) {
131 i.extent_constraint(j).accept(visitor);
132 }
133 }

180// All variables present in any part of a function definition must
181// either be pure args, elements of the reduction domain, parameters
182// (i.e. attached to some Parameter object), or part of a let node
183// internal to the expression
184struct CheckVars : public IRGraphVisitor {
185 vector<string> pure_args;
186 ReductionDomain reduction_domain;
187 Scope<> defined_internally;
188 const std::string name;
189 bool unbound_reduction_vars_ok = false;
190
191 CheckVars(const std::string &n)
192 : name(n) {
193 }
194
195 using IRVisitor::visit;
196
197 void visit(const Let *let) override {
198 let->value.accept(this);
199 ScopedBinding<> bind(defined_internally, let->name);
200 let->body.accept(this);
201 }
202
203 void visit(const Call *op) override {
204 IRGraphVisitor::visit(op);
205 if (op->name == name && op->call_type == Call::Halide) {
206 for (size_t i = 0; i < op->args.size(); i++) {
207 const Variable *var = op->args[i].as<Variable>();
208 if (!pure_args[i].empty()) {
209 user_assert(var && var->name == pure_args[i])
210 << "In definition of Func \"" << name << "\":\n"
211 << "All of a function's recursive references to itself"
212 << " must contain the same pure variables in the same"
213 << " places as on the left-hand-side.\n";
214 }
215 }
216 }
217 }
218
219 void visit(const Variable *var) override {
220 // Is it a parameter?
221 if (var->param.defined()) {
222 return;
223 }
224
225 // Was it defined internally by a let expression?
226 if (defined_internally.contains(var->name)) {
227 return;
228 }
229
230 // Is it a pure argument?
231 for (size_t i = 0; i < pure_args.size(); i++) {
232 if (var->name == pure_args[i]) {
233 return;
234 }
235 }
236
237 // Is it in a reduction domain?
238 if (var->reduction_domain.defined()) {
239 if (!reduction_domain.defined()) {
240 reduction_domain = var->reduction_domain;
241 return;
242 } else if (var->reduction_domain.same_as(reduction_domain)) {
243 // It's in a reduction domain we already know about
244 return;
245 } else {
246 user_error << "Multiple reduction domains found in definition of Func \"" << name << "\"\n";
247 }
248 } else if (reduction_domain.defined() && unbound_reduction_vars_ok) {
249 // Is it one of the RVars from the reduction domain we already
250 // know about (this can happen in the RDom predicate).
251 for (const ReductionVariable &rv : reduction_domain.domain()) {
252 if (rv.var == var->name) {
253 return;
254 }
255 }
256 }
257
258 user_error << "Undefined variable \"" << var->name << "\" in definition of Func \"" << name << "\"\n";
259 }
260};

333void Function::deep_copy(const FunctionPtr &copy, DeepCopyMap &copied_map) const {
334 internal_assert(copy.defined() && contents.defined())
335 << "Cannot deep-copy undefined Function\n";
336
337 // Add reference to this Function's deep-copy to the map in case of
338 // self-reference, e.g. self-reference in an Definition.
339 copied_map[contents] = copy;
340
341 debug(4) << "Deep-copy function contents: \"" << contents->name << "\"\n";
342
343 copy->name = contents->name;
344 copy->origin_name = contents->origin_name;
345 copy->args = contents->args;
346 copy->output_types = contents->output_types;
347 copy->debug_file = contents->debug_file;
348 copy->extern_function_name = contents->extern_function_name;
349 copy->extern_mangling = contents->extern_mangling;
350 copy->extern_function_device_api = contents->extern_function_device_api;
351 copy->extern_proxy_expr = contents->extern_proxy_expr;
352 copy->trace_loads = contents->trace_loads;
353 copy->trace_stores = contents->trace_stores;
354 copy->trace_realizations = contents->trace_realizations;
355 copy->trace_tags = contents->trace_tags;
356 copy->frozen = contents->frozen;
357 copy->output_buffers = contents->output_buffers;
358 copy->func_schedule = contents->func_schedule.deep_copy(copied_map);
359
360 // Copy the pure definition
361 if (contents->init_def.defined()) {
362 copy->init_def = contents->init_def.get_copy();
363 internal_assert(copy->init_def.is_init());
364 internal_assert(copy->init_def.schedule().rvars().empty())
365 << "Init definition shouldn't have reduction domain\n";
366 }
367
368 for (const Definition &def : contents->updates) {
369 internal_assert(!def.is_init());
370 Definition def_copy = def.get_copy();
371 internal_assert(!def_copy.is_init());
372 copy->updates.push_back(std::move(def_copy));
373 }
374
375 for (const ExternFuncArgument &e : contents->extern_arguments) {
376 ExternFuncArgument e_copy = deep_copy_extern_func_argument_helper(e, copied_map);
377 copy->extern_arguments.push_back(std::move(e_copy));
378 }
379}

386void Function::define(const vector<string> &args, vector<Expr> values) {
387 user_assert(!frozen())
388 << "Func " << name() << " cannot be given a new pure definition, "
389 << "because it has already been realized or used in the definition of another Func.\n";
390 user_assert(!has_extern_definition())
391 << "In pure definition of Func \"" << name() << "\":\n"
392 << "Func with extern definition cannot be given a pure definition.\n";
393 user_assert(!name().empty()) << "A Func may not have an empty name.\n";
394 for (size_t i = 0; i < values.size(); i++) {
395 user_assert(values[i].defined())
396 << "In pure definition of Func \"" << name() << "\":\n"
397 << "Undefined expression in right-hand-side of definition.\n";
398 }
399
400 // Make sure all the vars in the value are either args or are
401 // attached to some parameter
402 CheckVars check(name());
403 check.pure_args = args;
404 for (const auto &value : values) {
405 value.accept(&check);
406 }
407
408 // Freeze all called functions
409 FreezeFunctions freezer(name());
410 for (const auto &value : values) {
411 value.accept(&freezer);
412 }
413
414 // Make sure all the vars in the args have unique non-empty names
415 for (size_t i = 0; i < args.size(); i++) {
416 user_assert(!args[i].empty())
417 << "In pure definition of Func \"" << name() << "\":\n"
418 << "In left-hand-side of definition, argument "
419 << i << " has an empty name.\n";
420 for (size_t j = 0; j < i; j++) {
421 user_assert(args[i] != args[j])
422 << "In pure definition of Func \"" << name() << "\":\n"
423 << "In left-hand-side of definition, arguments "
424 << i << " and " << j
425 << " both have the name \"" + args[i] + "\"\n";
426 }
427 }
428
429 for (auto &value : values) {
430 value = common_subexpression_elimination(value);
431 }
432
433 // Tag calls to random() with the free vars
434 int tag = rand_counter++;
435 vector<VarOrRVar> free_vars;
436 free_vars.reserve(args.size());
437 for (const auto &arg : args) {
438 free_vars.emplace_back(Var(arg));
439 }
440 for (auto &value : values) {
441 value = lower_random(value, free_vars, tag);
442 }
443
444 user_assert(!check.reduction_domain.defined())
445 << "In pure definition of Func \"" << name() << "\":\n"
446 << "Reduction domain referenced in pure function definition.\n";
447
448 if (!contents.defined()) {
449 contents.strong = new FunctionGroup;
450 contents.strong->members.resize(1);
451 contents->name = unique_name('f');
452 contents->origin_name = contents->name;
453 }
454
455 user_assert(!contents->init_def.defined())
456 << "In pure definition of Func \"" << name() << "\":\n"
457 << "Func is already defined.\n";
458
459 contents->args = args;
460
461 std::vector<Expr> init_def_args;
462 init_def_args.resize(args.size());
463 for (size_t i = 0; i < args.size(); i++) {
464 init_def_args[i] = Var(args[i]);
465 }
466
467 ReductionDomain rdom;
468 contents->init_def = Definition(init_def_args, values, rdom, true);
469
470 for (size_t i = 0; i < args.size(); i++) {
471 Dim d = {args[i], ForType::Serial, DeviceAPI::None, DimType::PureVar};
472 contents->init_def.schedule().dims().push_back(d);
473 StorageDim sd = {args[i]};
474 contents->func_schedule.storage_dims().push_back(sd);
475 }
476
477 // Add the dummy outermost dim
478 {
479 Dim d = {Var::outermost().name(), ForType::Serial, DeviceAPI::None, DimType::PureVar};
480 contents->init_def.schedule().dims().push_back(d);
481 }
482
483 contents->output_types.resize(values.size());
484 for (size_t i = 0; i < contents->output_types.size(); i++) {
485 contents->output_types[i] = values[i].type();
486 }
487
488 for (size_t i = 0; i < values.size(); i++) {
489 string buffer_name = name();
490 if (values.size() > 1) {
491 buffer_name += '.' + std::to_string((int)i);
492 }
493 Parameter output(values[i].type(), true, args.size(), buffer_name);
494 contents->output_buffers.push_back(output);
495 }
496}

498void Function::define_update(const vector<Expr> &_args, vector<Expr> values) {
499 int update_idx = static_cast<int>(contents->updates.size());
500
501 user_assert(!name().empty())
502 << "Func has an empty name.\n";
503 user_assert(has_pure_definition())
504 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
505 << "Can't add an update definition without a pure definition first.\n";
506 user_assert(!frozen())
507 << "Func " << name() << " cannot be given a new update definition, "
508 << "because it has already been realized or used in the definition of another Func.\n";
509
510 for (size_t i = 0; i < values.size(); i++) {
511 user_assert(values[i].defined())
512 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
513 << "Undefined expression in right-hand-side of update.\n";
514 }
515
516 // Check the dimensionality matches
517 user_assert((int)_args.size() == dimensions())
518 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
519 << "Dimensionality of update definition must match dimensionality of pure definition.\n";
520
521 user_assert(values.size() == contents->init_def.values().size())
522 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
523 << "Number of tuple elements for update definition must "
524 << "match number of tuple elements for pure definition.\n";
525
526 const auto &pure_def_vals = contents->init_def.values();
527 for (size_t i = 0; i < values.size(); i++) {
528 // Check that pure value and the update value have the same
529 // type. Without this check, allocations may be the wrong size
530 // relative to what update code expects.
531 Type pure_type = pure_def_vals[i].type();
532 if (pure_type != values[i].type()) {
533 std::ostringstream err;
534 err << "In update definition " << update_idx << " of Func \"" << name() << "\":\n";
535 if (!values.empty()) {
536 err << "Tuple element " << i << " of update definition has type ";
537 } else {
538 err << "Update definition has type ";
539 }
540 err << values[i].type() << ", but pure definition has type " << pure_type;
541 user_error << err.str() << "\n";
542 }
543 values[i] = common_subexpression_elimination(values[i]);
544 }
545
546 vector<Expr> args(_args.size());
547 for (size_t i = 0; i < args.size(); i++) {
548 args[i] = common_subexpression_elimination(_args[i]);
549 }
550
551 // The pure args are those naked vars in the args that are not in
552 // a reduction domain and are not parameters and line up with the
553 // pure args in the pure definition.
554 bool pure = true;
555 vector<string> pure_args(args.size());
556 for (size_t i = 0; i < args.size(); i++) {
557 pure_args[i] = ""; // Will never match a var name
558 user_assert(args[i].defined())
559 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
560 << "Argument " << i
561 << " in left-hand-side of update definition is undefined.\n";
562 if (const Variable *var = args[i].as<Variable>()) {
563 if (!var->param.defined() &&
564 !var->reduction_domain.defined() &&
565 var->name == contents->args[i]) {
566 pure_args[i] = var->name;
567 } else {
568 pure = false;
569 }
570 } else {
571 pure = false;
572 }
573 }
574
575 // Make sure all the vars in the args and the value are either
576 // pure args, in the reduction domain, or a parameter. Also checks
577 // that recursive references to the function contain all the pure
578 // vars in the LHS in the correct places.
579 CheckVars check(name());
580 check.pure_args = pure_args;
581 for (const auto &arg : args) {
582 arg.accept(&check);
583 }
584 for (const auto &value : values) {
585 value.accept(&check);
586 }
587 if (check.reduction_domain.defined()) {
588 check.unbound_reduction_vars_ok = true;
589 check.reduction_domain.predicate().accept(&check);
590 }
591
592 // Freeze all called functions
593 FreezeFunctions freezer(name());
594 for (const auto &arg : args) {
595 arg.accept(&freezer);
596 }
597 for (const auto &value : values) {
598 value.accept(&freezer);
599 }
600
601 // Freeze the reduction domain if defined
602 if (check.reduction_domain.defined()) {
603 check.reduction_domain.predicate().accept(&freezer);
604 check.reduction_domain.freeze();
605 }
606
607 // Tag calls to random() with the free vars
608 vector<VarOrRVar> free_vars;
609 int num_free_vars = (int)pure_args.size();
610 if (check.reduction_domain.defined()) {
611 num_free_vars += (int)check.reduction_domain.domain().size();
612 }
613 free_vars.reserve(num_free_vars);
614 for (const auto &pure_arg : pure_args) {
615 if (!pure_arg.empty()) {
616 free_vars.emplace_back(Var(pure_arg));
617 }
618 }
619 if (check.reduction_domain.defined()) {
620 for (size_t i = 0; i < check.reduction_domain.domain().size(); i++) {
621 free_vars.emplace_back(RVar(check.reduction_domain, i));
622 }
623 }
624 int tag = rand_counter++;
625 for (auto &arg : args) {
626 arg = lower_random(arg, free_vars, tag);
627 }
628 for (auto &value : values) {
629 value = lower_random(value, free_vars, tag);
630 }
631 if (check.reduction_domain.defined()) {
632 check.reduction_domain.set_predicate(lower_random(check.reduction_domain.predicate(), free_vars, tag));
633 }
634
635 // The update value and args probably refer back to the
636 // function itself, introducing circular references and hence
637 // memory leaks. We need to break these cycles.
638 WeakenFunctionPtrs weakener(contents.get());
639 for (auto &arg : args) {
640 arg = weakener.mutate(arg);
641 }
642 for (auto &value : values) {
643 value = weakener.mutate(value);
644 }
645 if (check.reduction_domain.defined()) {
646 check.reduction_domain.set_predicate(
647 weakener.mutate(check.reduction_domain.predicate()));
648 }
649
650 Definition r(args, values, check.reduction_domain, false);
651 internal_assert(!r.is_init()) << "Should have been an update definition\n";
652
653 // First add any reduction domain
654 if (check.reduction_domain.defined()) {
655 for (size_t i = 0; i < check.reduction_domain.domain().size(); i++) {
656 // Is this RVar actually pure (safe to parallelize and
657 // reorder)? It's pure if one value of the RVar can never
658 // access from the same memory that another RVar is
659 // writing to.
660 const ReductionVariable &rvar = check.reduction_domain.domain()[i];
661 const string &v = rvar.var;
662
663 bool pure = can_parallelize_rvar(v, name(), r);
664 Dim d = {v, ForType::Serial, DeviceAPI::None,
665 pure ? DimType::PureRVar : DimType::ImpureRVar};
666 r.schedule().dims().push_back(d);
667 }
668 }
669
670 // Then add the pure args outside of that
671 for (const auto &pure_arg : pure_args) {
672 if (!pure_arg.empty()) {
673 Dim d = {pure_arg, ForType::Serial, DeviceAPI::None, DimType::PureVar};
674 r.schedule().dims().push_back(d);
675 }
676 }
677
678 // Then the dummy outermost dim
679 {
680 Dim d = {Var::outermost().name(), ForType::Serial, DeviceAPI::None, DimType::PureVar};
681 r.schedule().dims().push_back(d);
682 }
683
684 // If there's no recursive reference, no reduction domain, and all
685 // the args are pure, then this definition completely hides
686 // earlier ones!
687 if (!check.reduction_domain.defined() &&
688 weakener.count == 0 &&
689 pure) {
690 user_warning
691 << "In update definition " << update_idx << " of Func \"" << name() << "\":\n"
692 << "Update definition completely hides earlier definitions, "
693 << " because all the arguments are pure, it contains no self-references, "
694 << " and no reduction domain. This may be an accidental re-definition of "
695 << " an already-defined function.\n";
696 }
697
698 contents->updates.push_back(r);
699}

701void Function::define_extern(const std::string &function_name,
702 const std::vector<ExternFuncArgument> &extern_args,
703 const std::vector<Type> &types,
704 const std::vector<Var> &args,
705 NameMangling mangling,
706 DeviceAPI device_api) {
707
708 user_assert(!has_pure_definition() && !has_update_definition())
709 << "In extern definition for Func \"" << name() << "\":\n"
710 << "Func with a pure definition cannot have an extern definition.\n";
711
712 user_assert(!has_extern_definition())
713 << "In extern definition for Func \"" << name() << "\":\n"
714 << "Func already has an extern definition.\n";
715
716 std::vector<string> arg_names;
717 std::vector<Expr> arg_exprs;
718 for (size_t i = 0; i < args.size(); i++) {
719 arg_names.push_back(args[i].name());
720 arg_exprs.push_back(args[i]);
721 }
722 contents->args = arg_names;
723 contents->extern_function_name = function_name;
724 contents->extern_arguments = extern_args;
725 contents->output_types = types;
726 contents->extern_mangling = mangling;
727 contents->extern_function_device_api = device_api;
728
729 std::vector<Expr> values;
730 contents->output_buffers.clear();
731 for (size_t i = 0; i < types.size(); i++) {
732 string buffer_name = name();
733 if (types.size() > 1) {
734 buffer_name += '.' + std::to_string((int)i);
735 }
736 Parameter output(types[i], true, (int)args.size(), buffer_name);
737 contents->output_buffers.push_back(output);
738
739 values.push_back(undef(types[i]));
740 }
741
742 contents->init_def = Definition(arg_exprs, values, ReductionDomain(), true);
743
744 // Reset the storage dims to match the pure args
745 contents->func_schedule.storage_dims().clear();
746 contents->init_def.schedule().dims().clear();
747 for (size_t i = 0; i < args.size(); i++) {
748 contents->func_schedule.storage_dims().push_back(StorageDim{arg_names[i]});
749 contents->init_def.schedule().dims().push_back(
750 Dim{arg_names[i], ForType::Extern, DeviceAPI::None, DimType::PureVar});
751 }
752 // Add the dummy outermost dim
753 contents->init_def.schedule().dims().push_back(
754 Dim{Var::outermost().name(), ForType::Serial, DeviceAPI::None, DimType::PureVar});
755}

944void Function::lock_loop_levels() {
945 auto &schedule = contents->func_schedule;
946 schedule.compute_level().lock();
947 schedule.store_level().lock();
948 // If store_level is inlined, use the compute_level instead.
949 // (Note that we deliberately do *not* do the same if store_level
950 // is undefined.)
951 if (schedule.store_level().is_inlined()) {
952 schedule.store_level() = schedule.compute_level();
953 }
954 if (contents->init_def.defined()) {
955 contents->init_def.schedule().fuse_level().level.lock();
956 }
957 for (Definition &def : contents->updates) {
958 internal_assert(def.defined());
959 def.schedule().fuse_level().level.lock();
960 }

994const Call *Function::is_wrapper() const {
995 const vector<Expr> &rhs = values();
996 if (rhs.size() != 1) {
997 return nullptr;
998 }
999 const Call *call = rhs[0].as<Call>();
1000 if (!call) {
1001 return nullptr;
1002 }
1003 vector<Expr> expected_args;
1004 for (const string &v : args()) {
1005 expected_args.push_back(Variable::make(Int(32), v));
1006 }
1007 Expr expected_rhs =
1008 Call::make(call->type, call->name, expected_args, call->call_type,
1009 call->func, call->value_index, call->image, call->param);
1010 if (equal(rhs[0], expected_rhs)) {
1011 return call;
1012 } else {
1013 return nullptr;
1014 }

1019// Replace all calls to functions listed in 'substitutions' with their wrappers.
1020class SubstituteCalls : public IRMutator {
1021 using IRMutator::visit;
1022
1023 const map<FunctionPtr, FunctionPtr> &substitutions;
1024
1025 Expr visit(const Call *c) override {
1026 Expr expr = IRMutator::visit(c);
1027 c = expr.as<Call>();
1028 internal_assert(c);
1029
1030 if ((c->call_type == Call::Halide) &&
1031 c->func.defined() &&
1032 substitutions.count(c->func)) {
1033 auto it = substitutions.find(c->func);
1034 internal_assert(it != substitutions.end())
1035 << "Function not in environment: " << c->func->name << "\n";
1036 FunctionPtr subs = it->second;
1037 debug(4) << "...Replace call to Func \"" << c->name << "\" with "
1038 << "\"" << subs->name << "\"\n";
1039 expr = Call::make(c->type, subs->name, c->args, c->call_type,
1040 subs, c->value_index,
1041 c->image, c->param);
1042 }
1043 return expr;
1044 }
1045
1046public:
1047 SubstituteCalls(const map<FunctionPtr, FunctionPtr> &substitutions)
1048 : substitutions(substitutions) {
1049 }
1050};

1070// Deep copy an entire Function DAG.
1071pair<vector<Function>, map<string, Function>> deep_copy(
1072 const vector<Function> &outputs, const map<string, Function> &env) {
1073 vector<Function> copy_outputs;
1074 map<string, Function> copy_env;
1075
1076 // Create empty deep-copies of all Functions in 'env'
1077 DeepCopyMap copied_map; // Original Function -> Deep-copy
1078 IntrusivePtr<FunctionGroup> group(new FunctionGroup);
1079 group->members.resize(env.size());
1080 int i = 0;
1081 for (const auto &iter : env) {
1082 // Make a weak pointer to the function to use for within-group references.
1083 FunctionPtr ptr;
1084 ptr.weak = group.get();
1085 ptr.idx = i;
1086 ptr->name = iter.second.name();
1087 copied_map[iter.second.get_contents()] = ptr;
1088 i++;
1089 }
1090
1091 // Deep copy all Functions in 'env' into their corresponding empty copies
1092 for (const auto &iter : env) {
1093 iter.second.deep_copy(copied_map[iter.second.get_contents()], copied_map);
1094 }
1095
1096 // Need to substitute-in all old Function references in all Exprs referenced
1097 // within the Function with the deep-copy versions
1098 for (auto &iter : copied_map) {
1099 Function(iter.second).substitute_calls(copied_map);
1100 }
1101
1102 // Populate the env with the deep-copy version
1103 for (const auto &iter : copied_map) {
1104 FunctionPtr ptr = iter.second;
1105 copy_env.emplace(iter.first->name, Function(ptr));
1106 }
1107
1108 for (const auto &func : outputs) {
1109 const auto &iter = copied_map.find(func.get_contents());
1110 if (iter != copied_map.end()) {
1111 FunctionPtr ptr = iter->second;
1112 debug(4) << "Adding deep-copied version to outputs: " << func.name() << "\n";
1113 copy_outputs.emplace_back(ptr);
1114 } else {
1115 debug(4) << "Adding original version to outputs: " << func.name() << "\n";
1116 copy_outputs.push_back(func);
1117 }
1118 }
1119
1120 return {copy_outputs, copy_env};
1121}
