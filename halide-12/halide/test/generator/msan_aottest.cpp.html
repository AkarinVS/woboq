70extern "C" int msan_extern_stage(halide_buffer_t *in, halide_buffer_t *out) {
71 if (in->is_bounds_query() || out->is_bounds_query()) {
72 if (in->is_bounds_query()) {
73 assert(in->dimensions == 3);
74 in->dim[0].extent = 4;
75 in->dim[1].extent = 4;
76 in->dim[2].extent = 3;
77 in->dim[0].min = 0;
78 in->dim[1].min = 0;
79 in->dim[2].min = 0;
80 }
81 if (out->is_bounds_query()) {
82 assert(out->dimensions == 3);
83 out->dim[0].extent = 4;
84 out->dim[1].extent = 4;
85 out->dim[2].extent = 3;
86 out->dim[0].min = 0;
87 out->dim[1].min = 0;
88 out->dim[2].min = 0;
89 }
90 return 0;
91 }
92
93 if (in->type != out->type) {
94 fprintf(stderr, "type mismatch\n");
95 return -1;
96 }
97 if (skip_extern_copy) {
98 // Fill it with zero to mimic msan "poison".
99 Buffer<uint8_t>(*out).fill(0);
100 } else {
101 Buffer<uint8_t>(*out).copy_from(Buffer<uint8_t>(*in));
102 }
103 out->set_host_dirty();
104 return 0;
105}

132extern "C" int halide_msan_check_memory_is_initialized(void *user_context, const void *ptr, uint64_t len, const char *name) {
133 // printf("CHECK-MEM: %d:%p:%08x for buf %s\n", (int)check_stage, ptr, (unsigned int)len, name);
134 if (check_stage == CheckInputBuffer) {
135 if (len != sizeof(halide_buffer_t)) {
136 fprintf(stderr, "Failure: Expected sizeof(halide_buffer_t), saw %d\n", (unsigned int)len);
137 exit(-1);
138 }
139 check_stage = CheckInputShape;
140 } else if (check_stage == CheckInputShape) {
141 if (len != sizeof(halide_dimension_t) * 3) {
142 fprintf(stderr, "Failure: Expected sizeof(halide_dimension_t) * 3, saw %d\n", (unsigned int)len);
143 exit(-1);
144 }
145 check_stage = CheckInputContents;
146 } else if (check_stage == CheckInputContents) {
147 for (uint64_t i = 0; i < len; ++i) {
148 input_contents_uninitialized += (((const uint8_t *)ptr)[i] == 0);
149 }
150 input_contents_checked += len;
151 check_stage = CheckExternResultBuffer;
152 } else if (check_stage == CheckExternResultBuffer) {
153 if (len != sizeof(halide_buffer_t)) {
154 fprintf(stderr, "Failure: Expected sizeof(halide_buffer_t), saw %d\n", (unsigned int)len);
155 exit(-1);
156 }
157 check_stage = CheckExternResultShape;
158 } else if (check_stage == CheckExternResultShape) {
159 if (len != sizeof(halide_dimension_t) * 3) {
160 fprintf(stderr, "Failure: Expected sizeof(halide_dimension_t) * 3, saw %d\n", (unsigned int)len);
161 exit(-1);
162 }
163 check_stage = CheckExternResultContents;
164 } else if (check_stage == CheckExternResultContents) {
165 for (uint64_t i = 0; i < len; ++i) {
166 externresult_contents_uninitialized += (((const uint8_t *)ptr)[i] == 0);
167 }
168 externresult_contents_checked += len;
169 } else {
170 fprintf(stderr, "Failure: bad enum\n");
171 exit(-1);
172 }
173 return 0;
174}

176extern "C" int halide_msan_annotate_memory_is_initialized(void *user_context, const void *ptr, uint64_t len) {
177 // printf("ANNOTATE: %d:%p:%08x\n", (int)annotate_stage, ptr, (unsigned int)len);
178 if (annotate_stage == AnnotateBoundsInferenceBuffer) {
179 if (output_previous != nullptr || len != sizeof(halide_buffer_t)) {
180 fprintf(stderr, "Failure: Expected sizeof(halide_buffer_t), saw %d\n", (unsigned int)len);
181 exit(-1);
182 }
183 annotate_stage = AnnotateBoundsInferenceShape;
184 } else if (annotate_stage == AnnotateBoundsInferenceShape) {
185 if (output_previous != nullptr || len != sizeof(halide_dimension_t) * 3) {
186 fprintf(stderr, "Failure: Expected sizeof(halide_dimension_t) * 3, saw %d\n", (unsigned int)len);
187 exit(-1);
188 }
189 bounds_inference_count += 1;
190 if (bounds_inference_count == 4) {
191 annotate_stage = AnnotateIntermediateBuffer;
192 } else {
193 annotate_stage = AnnotateBoundsInferenceBuffer;
194 }
195 } else if (annotate_stage == AnnotateIntermediateBuffer) {
196 if (expect_intermediate_buffer_error) {
197 if (len != 80) {
198 fprintf(stderr, "Failure: Expected error message of len=80, saw %d bytes\n", (unsigned int)len);
199 exit(-1);
200 }
201 return 0; // stay in this state
202 }
203 if (output_previous != nullptr || len != sizeof(halide_buffer_t)) {
204 fprintf(stderr, "Failure: Expected sizeof(halide_buffer_t), saw %d\n", (unsigned int)len);
205 exit(-1);
206 }
207 annotate_stage = AnnotateIntermediateShape;
208 } else if (annotate_stage == AnnotateIntermediateShape) {
209 if (output_previous != nullptr || len != sizeof(halide_dimension_t) * 3) {
210 fprintf(stderr, "Failure: Expected sizeof(halide_dimension_t) * 3, saw %d\n", (unsigned int)len);
211 exit(-1);
212 }
213 annotate_stage = AnnotateOutputBuffer;
214 } else if (annotate_stage == AnnotateOutputBuffer) {
215 if (output_previous != nullptr || len != sizeof(halide_buffer_t)) {
216 fprintf(stderr, "Failure: Expected sizeof(halide_buffer_t), saw %d\n", (unsigned int)len);
217 exit(-1);
218 }
219 annotate_stage = AnnotateOutputShape;
220 } else if (annotate_stage == AnnotateOutputShape) {
221 if (output_previous != nullptr || len != sizeof(halide_dimension_t) * 3) {
222 fprintf(stderr, "Failure: Expected sizeof(halide_dimension_t) * 3, saw %d\n", (unsigned int)len);
223 exit(-1);
224 }
225 annotate_stage = AnnotateIntermediateContents;
226 } else if (annotate_stage == AnnotateIntermediateContents) {
227 if (output_previous != nullptr || len != 4 * 4 * 3) {
228 fprintf(stderr, "Failure: Expected %d, saw %d\n", 4 * 4 * 3, (unsigned int)len);
229 exit(-1);
230 }
231 annotate_stage = AnnotateOutputContents;
232 } else if (annotate_stage == AnnotateOutputContents) {
233 if (output_previous == nullptr) {
234 if (ptr != output_base) {
235 fprintf(stderr, "Failure: Expected base p %p but saw %p\n", output_base, ptr);
236 exit(-1);
237 }
238 if (ptr <= output_previous) {
239 fprintf(stderr, "Failure: Expected monotonic increase but saw %p -> %p\n", output_previous, ptr);
240 exit(-1);
241 }
242 output_previous = ptr;
243 }
244 output_contents_annotated += len;
245 } else {
246 fprintf(stderr, "Failure: bad enum\n");
247 exit(-1);
248 }
249 return 0;
250}

252template<typename T>
253void verify(const T &image) {
254 image.for_each_element([&](int x, int y, int c) {
255 int expected = 7;
256 for (int i = 0; i < 4; ++i) {
257 expected += (uint8_t)(i + y + c) | 0x01;
258 }
259 int actual = image(x, y, c);
260 if (actual != expected) {
261 fprintf(stderr, "Failure @ %d %d %d: expected %d, got %d\n", x, y, c, expected, actual);
262 exit(-1);
263 }
264 });
265}

267Buffer<uint8_t> make_input_for(const Buffer<uint8_t> &output) {
268 auto input = Buffer<uint8_t>::make_with_shape_of(output);
269 // Ensure that no 'valid' inputs are all-zero
270 input.for_each_element([&](int x, int y, int c) { input(x, y, c) = (uint8_t)(x + y + c) | 0x01; });
271 return input;
272}
