20 std::vector<Buffer<uint8_t>> extras;

27
28 Buffer<int16_t> typed_extra(kSize, kSize);
29 typed_extra.fill(4);
30 extra_value += 4;
31
32 // Funcs are aot-compiled as buffers.
33 Buffer<uint16_t> func_extra(kSize, kSize, 3);
34 func_extra.fill(5);
35 extra_value += 5;
36
37 const int extra_scalar = 7;
38 const int8_t extra_dynamic_scalar = 13;
39 extra_value += extra_scalar + extra_dynamic_scalar;
40
41 Buffer<int> output(kSize, kSize, 3);
42 Buffer<float> extra_buffer_output(kSize, kSize, 3);
43 Buffer<double> extra_func_output(kSize, kSize);
44
45 const int bias = 1;
46 int result = configure(input, bias,
47 // extra inputs are in the order they were added, after all predeclared inputs
48 extras[0], extras[1], extras[2], typed_extra, func_extra, extra_scalar, extra_dynamic_scalar,
49 output,
50 // extra outputs are in the order they were added, after all predeclared outputs
51 extra_buffer_output, extra_func_output);
52 if (result != 0) {
53 fprintf(stderr, "Result: %d\n", result);
54 exit(-1);
55 }
56
57 output.for_each_element([&](int x, int y, int c) {
58 assert(output(x, y, c) == input(x, y, c) + bias + extra_value);
59 });
60
61 extra_buffer_output.for_each_element([&](int x, int y, int c) {
62 assert(extra_buffer_output(x, y, c) == output(x, y, c));
63 });
64
65 extra_func_output.for_each_element([&](int x, int y) {
66 assert(extra_func_output(x, y) == output(x, y, 0));
67 });
68
69 printf("Success!\n");
70 return 0;
71}
