7template<typename T>
8Halide::Expr is_interleaved(const T &p, int channels = 3) {
9 return p.dim(0).stride() == channels && p.dim(2).stride() == 1 && p.dim(2).extent() == channels;
10}
11
12template<typename T>
13Halide::Expr is_planar(const T &p, int channels = 3) {
14 return p.dim(0).stride() == 1 && p.dim(2).extent() == channels;
15}
16
17class TiledBlur : public Halide::Generator<TiledBlur> {
18public:
19 Input<Buffer<uint8_t>> input{"input", 3};
20 Output<Buffer<uint8_t>> output{"output", 3};
21
22 void generate() {
23 Expr input_float = cast<float>(input(x, y, c)) / 255.f;
24
25 // This is the outermost pipeline, so input width and height
26 // are meaningful. If you want to be able to call this outer
27 // pipeline in a tiled fashion itself, then you should pass in
28 // width and height as params, as with the blur above.
29 brightened(x, y, c) = input_float * 1.2f;
30
31 tiled_blur.define_extern(
32 "blur2x2",
33 {brightened, input.dim(0).extent(), input.dim(1).extent()},
34 Float(32), 3);
35
36 Expr tiled_blur_brightened = tiled_blur(x, y, c) * 1.2f;
37
38 output(x, y, c) = saturating_cast<uint8_t>(tiled_blur_brightened * 255.f);
39 }
40
41 void schedule() {
42 Var xi, yi;
43 output.reorder(c, x, y).tile(x, y, xi, yi, 32, 32);
44 tiled_blur.compute_at(output, x);
45 brightened.compute_at(output, x);
46
47 // Let's see what tiled_blur decides that it needs from
48 // brightened. They should be 34x34 tiles, but clamped to fit
49 // within the input, so they'll often be 33x34, 34x33, or
50 // 33x33 near the boundaries
51 brightened.trace_realizations();
52
53 // Unset default constraints so that specialization works.
54 input.dim(0).set_stride(Expr());
55 output.dim(0).set_stride(Expr());
56
57 // Add specialization for input and output buffers that are both planar.
58 output.specialize(is_planar(input) && is_planar(output))
59 .vectorize(xi, natural_vector_size<float>());
60
61 // Add specialization for input and output buffers that are both interleaved.
62 output.specialize(is_interleaved(input) && is_interleaved(output));
63
64 // Note that other combinations (e.g. interleaved -> planar) will work
65 // but be relatively unoptimized.
66 }
67
68private:
69 Var x{"x"}, y{"y"}, c{"c"};
70 Func tiled_blur{"tiled_blur"};
71 Func brightened{"brightened"};
72};
73
74} // namespace
75
76HALIDE_REGISTER_GENERATOR(TiledBlur, tiled_blur)
