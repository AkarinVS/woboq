38int main(int argc, char **argv) {
39#if defined(TEST_OPENCL) || defined(TEST_CUDA)
40 const int W = 32, H = 32;
41 Buffer<int> input(W, H);
42 for (int y = 0; y < input.height(); y++) {
43 for (int x = 0; x < input.width(); x++) {
44 input(x, y) = x + y;
45 }
46 }
47
48 // Explicitly copy data to the GPU.
49 input.set_host_dirty();
50#if defined(TEST_OPENCL)
51 input.copy_to_device(halide_opencl_device_interface());
52#elif defined(TEST_CUDA)
53 input.copy_to_device(halide_cuda_device_interface());
54#endif
55
56 Buffer<int> output(W, H);
57
58 // Create halide_buffer_ts without host pointers.
59 halide_buffer_t input_no_host = *((halide_buffer_t *)input);
60 input_no_host.host = nullptr;
61
62 halide_buffer_t output_no_host = *((halide_buffer_t *)output);
63 // We need a fake pointer here to trick Halide into creating the
64 // device buffer (and not do bounds inference instead of running
65 // the pipeline). Halide will not dereference this pointer.
66 output_no_host.host = (uint8_t *)1;
67
68 gpu_only(&input_no_host, &output_no_host);
69
70 // Restore the host pointer and copy to host.
71 output_no_host.host = (uint8_t *)output.data();
72 halide_copy_to_host(nullptr, &output_no_host);
73
74 // Verify output.
75 for (int y = 0; y < H; y++) {
76 for (int x = 0; x < W; x++) {
77 if (input(x, y) * 2 != output(x, y)) {
78 printf("Error at %d, %d: %d != %d\n", x, y, input(x, y), output(x, y));
79 return -1;
80 }
81 }
82 }
83
84 printf("Success!\n");
85#else
86 printf("[SKIP] No GPU target enabled.\n");
87#endif
88 return 0;
89}
