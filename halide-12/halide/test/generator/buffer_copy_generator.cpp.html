3using namespace Halide;
4
5class BufferCopy : public Halide::Generator<BufferCopy> {
6public:
7 Input<Func> input{"input", Float(32), 2};
8
9 Output<Func> output{"output", Float(32), 2};
10
11 Func dev_1, host_1, dev_2;
12 Var x, y;
13
14 Expr check_eq(Expr a, Expr b, const char *name) {
15 return require(a == b, a, "!=", b, "@", name, "(", x, ",", y, ")");
16 }
17
18 void generate() {
19 // We're going to have a four stage pipeline where we do work
20 // on-device, on the host, on the device again, and finally on
21 // the host again. In between each stage we'll schedule an
22 // explicit copy in one direction or the other.
23
24 dev_1(x, y) = input(x, y) + 1;
25 host_1(x, y) = dev_1(x, y) + 1;
26 dev_2(x, y) = host_1(x, y) + 1;
27 output(x, y) = dev_2(x, y) + 1;
28 }
29
30 void schedule() {
31 if (get_target().has_gpu_feature()) {
32 Var tx, ty, xi, yi;
33
34 // Set up a complicated nested tiling so that the two of the
35 // buffer_copy stages are pulling a subset and two are pulling an entire buffer.
36 output.compute_root()
37 .tile(x, y, tx, ty, x, y, 64, 64);
38
39 // dev_1 does computed over 64x64 kernel launches
40 dev_1.compute_at(output, tx)
41 .gpu_tile(x, y, xi, yi, 8, 8);
42
43 // dev_2 does 32x32 kernel launches
44 dev_2.compute_at(output, tx)
45 .tile(x, y, tx, ty, x, y, 32, 32)
46 .gpu_tile(x, y, xi, yi, 8, 8);
47
48 // host_1 is computed per 32x32 tile of dev_2
49 host_1.compute_at(dev_2, tx);
50
51 // pulls a 64x64 subset of the input into a region of a GPU buffer to be consumed by dev_1
52 input.in(dev_1).copy_to_device().compute_at(output, tx).store_root();
53
54 // pulls a 32x32 subset from dev to host to be consumed by host_1
55 dev_1.in(host_1).copy_to_host().compute_at(dev_2, tx);
56
57 // pulls an entire 32x32 buffer back to the device to be consumed by dev_2
58 host_1.in(dev_2).copy_to_device().compute_at(dev_2, tx);
59
60 // pulls an entire 64x64 buffer back to the host to be consumed by the output
61 dev_2.in(output).copy_to_host().compute_at(output, tx);
62 }
63 }
64};
65
66HALIDE_REGISTER_GENERATOR(BufferCopy, buffer_copy);
