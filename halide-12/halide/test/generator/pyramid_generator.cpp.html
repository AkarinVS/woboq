1#include "Halide.h"

3namespace {
4
5class Pyramid : public Halide::Generator<Pyramid> {
6public:
7 GeneratorParam<int> levels{"levels", 1}; // deliberately wrong value, must be overridden to 10
8
9 Input<Func> input{"input", Float(32), 2};
10 Output<Func[]> pyramid{"pyramid", Float(32), 2};
11
12 void generate() {
13 Var x{"x"}, y{"y"};
14
15 pyramid.resize(levels);
16 pyramid[0](x, y) = input(x, y);
17 for (size_t i = 1; i < pyramid.size(); i++) {
18 Func p = pyramid[i - 1];
19 pyramid[i](x, y) = (p(2 * x, 2 * y) +
20 p(2 * x + 1, 2 * y) +
21 p(2 * x, 2 * y + 1) +
22 p(2 * x + 1, 2 * y + 1)) /
23 4;
24 }
25
26 // Be sure we set the 'schedule' member before we finish.
27 schedule = [=]() mutable {
28 for (Func p : pyramid) {
29 // No need to specify compute_root() for outputs
30 p.parallel(y);
31 // Vectorize if we're still wide enough at this level
32 const int v = natural_vector_size<float>();
33 p.specialize(p.output_buffer().width() >= v).vectorize(x, v);
34 }
35 };
36 }
37
38 // Note that you can define the schedule() method either as a conventional
39 // member method, *or*, a public std::function; for the latter approach,
40 // you must ensure the value is set by the generate() method.
41 // The main reason to do this is to capture the scheduling instructions
42 // via a lambda function, allowing you to keep intermediate Funcs and Vars
43 // as captured automatic variables in the generate() method, rather than
44 // member variables at class scope. There is no intrinsic reason to prefer
45 // either approach; it is purely a stylistic preference.
46 std::function<void()> schedule;
47};
48
49} // namespace
50
51HALIDE_REGISTER_GENERATOR(Pyramid, pyramid)
