3// By convention, Generators always go in a .cpp file, usually with no
4// corresponding .h file. They can be enclosed in any C++ namespaces
5// you like, but the anonymous namespace is often the best choice.
6//
7// It's normally considered Best Practice to have exactly one Generator
8// per .cpp file, and to have the .cpp file name match the generator name
9// with a "_generator" suffix (e.g., Generator with name "foo" should
10// live in "foo_generator.cpp"), as it tends to simplify build rules,
11// but neither of these are required.
12
13namespace {
14
15enum SomeEnum { Foo,
16 Bar };
17
18// Note the inheritance using the Curiously Recurring Template Pattern
19class Example : public Halide::Generator<Example> {
20public:
21 // GeneratorParamss, Inputs, and Outputs are (by convention)
22 // always public and always declared at the top of the Generator,
23 // in the order
24 // GeneratorParam(s)
25 // Input(s)
26 // Output(s)
27 //
28 // Note that the Inputs will appear in the C function
29 // call in the order they are declared. (GeneratorParams
30 // are always referenced by name, not position, so their order is irrelevant.)
31 //
32 // All Input variants declared as Generator members must have explicit
33 // names, and all such names must match the regex [A-Za-z_][A-Za-z_0-9]*
34 // (i.e., essentially a C/C++ variable name). By convention, the name should
35 // match the member-variable name.
36
37 // GeneratorParams can be float or ints: {default} or {default, min, max}
38 // (Note that if you want to specify min and max, you must specify both.)
39 GeneratorParam<float> compiletime_factor{"compiletime_factor", 1, 0, 100};
40 GeneratorParam<int> channels{"channels", 3};
41 // ...or enums: {default, name->value map}
42 GeneratorParam<SomeEnum> enummy{"enummy",
43 Foo,
44 {{"foo", Foo},
45 {"bar", Bar}}};
46 // ...or bools: {default}
47 GeneratorParam<bool> vectorize{"vectorize", true};
48 GeneratorParam<bool> parallelize{"parallelize", true};
49
50 // These are bad names that will produce errors at build time:
51 // GeneratorParam<bool> badname{ " flag", true };
52 // GeneratorParam<bool> badname{ "flag ", true };
53 // GeneratorParam<bool> badname{ "0flag ", true };
54 // GeneratorParam<bool> badname{ "", true };
55 // GeneratorParam<bool> badname{ "\001", true };
56 // GeneratorParam<bool> badname{ "a name? with! stuff*", true };
57
58 // Note that a leading underscore is legal-but-reserved in C,
59 // but it's outright forbidden here. (underscore after first char is ok.)
60 // GeneratorParam<bool> badname{ "_flag", true };
61
62 // We also forbid two underscores in a row.
63 // GeneratorParam<bool> badname{ "f__lag", true };
64
65 // Input<> are arguments passed to the filter when
66 // it is executed (as opposed to the Generator, during compilation).
67 // When jitting, there is effectively little difference between the
68 // two (at least for scalar values). Note that we set a default value of
69 // 1.0 so that invocations that don't set it explicitly use a predictable value.
70 Input<float> runtime_factor{"runtime_factor", 1.0};
71
72 Output<Func> output{"output", Int(32), 3};
73
74 void generate() {
75 Func f;
76 f(x, y) = max(x, y);
77 output(x, y, c) = cast(output.type(), f(x, y) * c * compiletime_factor * runtime_factor);
78 }
79
80 void schedule() {
81 runtime_factor.set_estimate(1);
82 output.set_estimates({{0, 32}, {0, 32}, {0, 3}});
83
84 if (!auto_schedule) {
85 output
86 .bound(c, 0, channels)
87 .reorder(c, x, y)
88 .unroll(c);
89 // Note that we can use the Generator method natural_vector_size()
90 // here; this produces the width of the SIMD vector being targeted
91 // divided by the width of the data type.
92 const int v = natural_vector_size(output.type());
93 if (parallelize && vectorize) {
94 output.parallel(y).vectorize(x, v);
95 } else if (parallelize) {
96 output.parallel(y);
97 } else if (vectorize) {
98 output.vectorize(x, v);
99 }
100 }
101 }
102
103private:
104 Var x{"x"}, y{"y"}, c{"c"};
105};
106
107} // namespace
108
109HALIDE_REGISTER_GENERATOR(Example, example)
