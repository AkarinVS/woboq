49// These functions replace the acquire/release implementation in src/runtime/opencl.cpp.
50// Since we don't parallelize access to the GPU in the schedule, we don't need synchronization
51// in our implementation of these functions.
52extern "C" int halide_acquire_cl_context(void *user_context, cl_context *ctx, cl_command_queue *q, bool create) {
53 if (user_context == nullptr) {
54 assert(!create);
55 *ctx = nullptr;
56 *q = nullptr;
57 } else {
58 const gpu_context *context = (const gpu_context *)user_context;
59 *ctx = context->cl_ctx;
60 *q = context->cl_q;
61 }
62 return 0;
63}

66 return 0;

154void run_kernels_on_thread(gpu_context context1, bool destroy_when_done) {
155 gpu_context context2;
156
157 Buffer<int32_t> buf1_in(W, H);
158 Buffer<int32_t> buf1_result(W, H);
159 buf1_in.fill(0);
160
161 const halide_device_interface_t *device_interface;
162
163 int val = 0;
164 for (int i = 0; i < 10; i++) {
165 init_context(context2);
166
167 Buffer<int32_t> buf2_in(W, H);
168 Buffer<int32_t> buf2_result(W, H);
169 buf2_in.fill(0);
170
171 gpu_multi_context_threaded_add(&context1, buf1_in, buf1_result);
172 gpu_multi_context_threaded_mul(&context1, buf1_result, buf1_in);
173 gpu_multi_context_threaded_add(&context1, buf1_in, buf1_result);
174
175 gpu_multi_context_threaded_add(&context2, buf2_in, buf2_result);
176 gpu_multi_context_threaded_mul(&context2, buf2_result, buf2_in);
177 gpu_multi_context_threaded_add(&context2, buf2_in, buf2_result);
178
179 buf1_result.copy_to_host(&context1);
180 buf2_result.copy_to_host(&context2);
181
182 val += 2;
183 val *= 2;
184 assert(buf1_result.all_equal(val + 2));
185 assert(buf2_result.all_equal(6));
186
187 device_interface = buf1_result.raw_buffer()->device_interface;
188
189 // About to destroy context, so ensure allocations are freed first.
190 buf2_in.device_free(&context2);
191 buf2_result.device_free(&context2);
192
193 if (device_interface != nullptr) {
194 halide_device_release(&context2, device_interface);
195 }
196 destroy_context(context2);
197 }
198
199 // About to destroy context, so ensure allocations are freed first.
200 buf1_in.device_free(&context1);
201 buf1_result.device_free(&context1);
202
203 if (destroy_when_done && device_interface != nullptr) {
204 halide_device_release(&context1, device_interface);
205 destroy_context(context1);
206 }
207}

209int main(int argc, char **argv) {
210 gpu_context contexta;
211 init_context(contexta);
212
213 gpu_context contextb;
214 init_context(contextb);
215
216 std::thread thread1(run_kernels_on_thread, contexta, false);
217 std::thread thread2(run_kernels_on_thread, contextb, false);
218
219 thread1.join();
220 thread2.join();
221
222 // Make sure using the same context on different threads works.
223 std::thread thread3(run_kernels_on_thread, contexta, HAS_MULTIPLE_CONTEXTS);
224 std::thread thread4(run_kernels_on_thread, contextb, HAS_MULTIPLE_CONTEXTS);
225
226 thread3.join();
227 thread4.join();
228
229 printf("Success!\n");
230 return 0;
231}
