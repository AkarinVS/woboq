5template<typename T>
6Halide::Expr is_interleaved(const T &p, int channels = 3) {
7 return p.dim(0).stride() == channels && p.dim(2).stride() == 1 && p.dim(2).extent() == channels;
8}
9
10template<typename T>
11Halide::Expr is_planar(const T &p, int channels = 3) {
12 return p.dim(0).stride() == 1 && p.dim(2).extent() == channels;
13}

15// A trivial 2x2 blur.
16class Blur2x2 : public Halide::Generator<Blur2x2> {
17public:
18 Input<Buffer<float>> input{"input", 3};
19 Input<int32_t> width{"width"};
20 Input<int32_t> height{"height"};
21
22 Output<Buffer<float>> blur{"blur", 3};
23
24 void generate() {
25 // We pass in parameters to tell us where the boundary
26 // condition kicks in; this allows us to decouple from the size of the
27 // input tile (if any).
28
29 // (In fact, if we are being used as an extern stage for tiled processing,
30 // clamping accesses to lie within the input tile using input.min() and
31 // input.extent() would tell the calling kernel we can cope with any size
32 // input, so it would always pass us 1x1 tiles.)
33
34 Func input_clamped =
35 Halide::BoundaryConditions::repeat_edge(input, {{0, width}, {0, height}});
36
37 blur(x, y, c) =
38 (input_clamped(x - 1, y, c) + input_clamped(x + 1, y, c) +
39 input_clamped(x, y - 1, c) + input_clamped(x, y + 1, c)) /
40 4.0f;
41 }
42
43 void schedule() {
44 // Unset default constraints so that specialization works.
45 input.dim(0).set_stride(Expr());
46 blur.dim(0).set_stride(Expr());
47
48 // Add specialization for input and output buffers that are both planar.
49 blur.specialize(is_planar(input) && is_planar(blur))
50 .vectorize(x, natural_vector_size<float>());
51
52 // Add specialization for input and output buffers that are both interleaved.
53 blur.specialize(is_interleaved(input) && is_interleaved(blur))
54 .reorder(c, x, y)
55 .vectorize(c);
56
57 // Note that other combinations (e.g. interleaved -> planar) will work
58 // but be relatively unoptimized.
59 }
60
61private:
62 Var x{"x"}, y{"y"}, c{"c"};
63};
64
65} // namespace
