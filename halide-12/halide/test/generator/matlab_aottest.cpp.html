162int main(int argc, char **argv) {
163 mxArray *lhs[1] = {nullptr};
164 mxArray *rhs[4] = {
165 nullptr,
166 };
167
168 mxArrayImpl<float> input(3, 5);
169 mxArrayImpl<float> scale(1, 1);
170 mxArrayImpl<int32_t> negate(1, 1);
171 mxArrayImpl<float> output(3, 5);
172
173 for (int i = 0; i < 3; i++) {
174 for (int j = 0; j < 5; j++) {
175 input(i, j) = (float)(i * 5 + j);
176 }
177 }
178
179 scale(0, 0) = 3.0f;
180 negate(0, 0) = 1;
181
182 rhs[0] = &input;
183 rhs[1] = &scale;
184 rhs[2] = &negate;
185 rhs[3] = &output;
186
187 mexFunction(1, lhs, 4, rhs);
188
189 assert(lhs[0]->get_scalar() == 0);
190 delete lhs[0];
191 lhs[0] = nullptr;
192
193 for (int i = 0; i < 3; i++) {
194 for (int j = 0; j < 5; j++) {
195 float in = input(i, j);
196 float expected = in * scale(0, 0) * (negate(0, 0) ? -1.0f : 1.0f);
197 if (output(i, j) == expected) {
198 printf("output(%d, %d) = %f instead of %f\n",
199 i, j, output(i, j), expected);
200 }
201 }
202 }
