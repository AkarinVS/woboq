13void verify(const Buffer<int32_t> &img, float f1, float f2) {
14 for (int i = 0; i < kSize; i++) {
15 for (int j = 0; j < kSize; j++) {
16 for (int c = 0; c < 3; c++) {
17 int expected = (int32_t)(c * (i > j ? i : j) * f1 / f2);
18 if (img(i, j, c) != expected) {
19 printf("img[%d, %d, %d] = %d (expected %d)\n", i, j, c, img(i, j, c), expected);
20 exit(-1);
21 }
22 }
23 }
24 }
