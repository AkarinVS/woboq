19struct last_call {
20 int loc, x, y, z;
21 last_call *next;
22 bool inited;

24 bool operator<(const last_call &rhs) const {
25 if (loc < rhs.loc) {
26 return true;

28 return false;

42 } else if (z >= rhs.z) {

57extern "C" int sleeper(int loc, int x, int y, int z, int v) {
58 last_update++;
59
60 thread_last.loc = loc;
61 thread_last.x = x;
62 thread_last.y = y;
63 thread_last.z = z;
64
65 if (!thread_last.inited) {
66 std::lock_guard<std::mutex> lock(watchdog_state_mutex);
67
68 thread_last.next = all_thread_lasts;
69 all_thread_lasts = &thread_last;
70 thread_last.inited = true;
71 }
72
73 if ((sleeps & ((1UL << (7 - loc)) | (8UL << (x + 1)))) == 0) {
74 std::this_thread::sleep_for(std::chrono::milliseconds(1));
75 }
76
77 return v;
78}

80void watchdog() {
81 int count = 0;
82 while (true) {
83 uint64_t prev = last_update;
84
85 std::this_thread::sleep_for(std::chrono::milliseconds(100));
86
87 if (prev == last_update) {
88 std::lock_guard<std::mutex> lock(watchdog_state_mutex);
89
90 if (watchdog_done) {
91 return;
92 }
93 if (count < 3) {
94 count++;
95 continue;
96 }
97
98 int thread_count = 0;
99 last_call *best = all_thread_lasts;
100 last_call *lc = all_thread_lasts;
101 while (lc != nullptr) {
102 if (*lc < *best) {
103 best = lc;
104 }
105 thread_count++;
106 lc = lc->next;
107 }
108 fflush(stdout);
109 fflush(stderr);
110 if (best == nullptr) {
111 std::cout << "Hung before any sleeps on any thread.\n";
112 } else {
113 std::cout << "Hung at loc " << best->loc << "(" << best->x << ", " << best->y << ", " << best->z << ") sleeps: " << sleeps << " threads: " << thread_count << "\n";
114 }
115 _Exit(1);
116 } else {
117 count = 0;
118 }
119 }
