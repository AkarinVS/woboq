15#ifndef ABSL_RANDOM_INTERNAL_RANDEN_TRAITS_H_
16#define ABSL_RANDOM_INTERNAL_RANDEN_TRAITS_H_
17
18// HERMETIC NOTE: The randen_hwaes target must not introduce duplicate
19// symbols from arbitrary system and other headers, since it may be built
20// with different flags from other targets, using different levels of
21// optimization, potentially introducing ODR violations.
22
23#include <cstddef>
24
25#include "absl/base/config.h"
26
27namespace absl {
28ABSL_NAMESPACE_BEGIN
29namespace random_internal {
30
31// RANDen = RANDom generator or beetroots in Swiss German.
32// 'Strong' (well-distributed, unpredictable, backtracking-resistant) random
33// generator, faster in some benchmarks than std::mt19937_64 and pcg64_c32.
34//
35// RandenTraits contains the basic algorithm traits, such as the size of the
36// state, seed, sponge, etc.
37struct RandenTraits {
38 // Size of the entire sponge / state for the randen PRNG.
39 static constexpr size_t kStateBytes = 256; // 2048-bit
40
41 // Size of the 'inner' (inaccessible) part of the sponge. Larger values would
42 // require more frequent calls to RandenGenerate.
43 static constexpr size_t kCapacityBytes = 16; // 128-bit
44
45 // Size of the default seed consumed by the sponge.
46 static constexpr size_t kSeedBytes = kStateBytes - kCapacityBytes;
47
48 // Largest size for which security proofs are known.
49 static constexpr size_t kFeistelBlocks = 16;
50
51 // Type-2 generalized Feistel => one round function for every two blocks.
52 static constexpr size_t kFeistelFunctions = kFeistelBlocks / 2; // = 8
53
54 // Ensures SPRP security and two full subblock diffusions.
55 // Must be > 4 * log2(kFeistelBlocks).
56 static constexpr size_t kFeistelRounds = 16 + 1;
57};
58
59} // namespace random_internal
60ABSL_NAMESPACE_END
61} // namespace absl
62
63#endif // ABSL_RANDOM_INTERNAL_RANDEN_TRAITS_H_
