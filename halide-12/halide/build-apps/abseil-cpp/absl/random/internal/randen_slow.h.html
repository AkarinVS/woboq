22namespace absl {
23ABSL_NAMESPACE_BEGIN
24namespace random_internal {
25
26// RANDen = RANDom generator or beetroots in Swiss German.
27// RandenSlow implements the basic state manipulation methods for
28// architectures lacking AES hardware acceleration intrinsics.
29class RandenSlow {
30 public:
31 // Size of the entire sponge / state for the randen PRNG.
32 static constexpr size_t kStateBytes = 256; // 2048-bit
33
34 // Size of the 'inner' (inaccessible) part of the sponge. Larger values would
35 // require more frequent calls to RandenGenerate.
36 static constexpr size_t kCapacityBytes = 16; // 128-bit
37
38 static void Generate(const void* keys, void* state_void);
39 static void Absorb(const void* seed_void, void* state_void);
40 static const void* GetKeys();
41};
42
43} // namespace random_internal
44ABSL_NAMESPACE_END
45} // namespace absl
