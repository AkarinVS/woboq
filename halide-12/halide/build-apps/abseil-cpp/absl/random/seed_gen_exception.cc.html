24static constexpr const char kExceptionMessage[] =
25 "Failed generating seed-material for URBG.";
26
27SeedGenException::~SeedGenException() = default;
28
29const char* SeedGenException::what() const noexcept {
30 return kExceptionMessage;
31}

33namespace random_internal {
34
35void ThrowSeedGenException() {
36#ifdef ABSL_HAVE_EXCEPTIONS
37 throw absl::SeedGenException();
38#else
39 std::cerr << kExceptionMessage << std::endl;
40 std::terminate();
41#endif
42}
