27// --------------------------------------------------------------------
28// Usage reporting interfaces
29
30namespace absl {
31ABSL_NAMESPACE_BEGIN
32namespace flags_internal {
33
34// The format to report the help messages in.
35enum class HelpFormat {
36 kHumanReadable,
37};
38
39// Outputs the help message describing specific flag.
40void FlagHelp(std::ostream& out, const flags_internal::CommandLineFlag& flag,
41 HelpFormat format = HelpFormat::kHumanReadable);
42
43// Produces the help messages for all flags matching the filter. A flag matches
44// the filter if it is defined in a file with a filename which includes
45// filter string as a substring. You can use '/' and '.' to restrict the
46// matching to a specific file names. For example:
47// FlagsHelp(out, "/path/to/file.");
48// restricts help to only flags which resides in files named like:
49// .../path/to/file.<ext>
50// for any extension 'ext'. If the filter is empty this function produces help
51// messages for all flags.
52void FlagsHelp(std::ostream& out, absl::string_view filter,
53 HelpFormat format, absl::string_view program_usage_message);
54
55// --------------------------------------------------------------------
56
57// If any of the 'usage' related command line flags (listed on the bottom of
58// this file) has been set this routine produces corresponding help message in
59// the specified output stream and returns:
60// 0 - if "version" or "only_check_flags" flags were set and handled.
61// 1 - if some other 'usage' related flag was set and handled.
62// -1 - if no usage flags were set on a commmand line.
63// Non negative return values are expected to be used as an exit code for a
64// binary.
65int HandleUsageFlags(std::ostream& out,
66 absl::string_view program_usage_message);
67
68} // namespace flags_internal
