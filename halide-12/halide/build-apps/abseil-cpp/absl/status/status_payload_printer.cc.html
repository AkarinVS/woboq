14#include "absl/status/status_payload_printer.h"

16#include <atomic>
17
18#include "absl/base/attributes.h"
19
20namespace absl {
21ABSL_NAMESPACE_BEGIN
22namespace status_internal {
23
24namespace {
25// Tried constant initialized global variable but it doesn't work with Lexan
26// (MSVC's `std::atomic` has trouble constant initializing).
27std::atomic<StatusPayloadPrinter>& GetStatusPayloadPrinterStorage() {
28 ABSL_CONST_INIT static std::atomic<StatusPayloadPrinter> instance{nullptr};
29 return instance;
30}
31} // namespace
32
33void SetStatusPayloadPrinter(StatusPayloadPrinter printer) {
34 GetStatusPayloadPrinterStorage().store(printer, std::memory_order_relaxed);
35}
36
37StatusPayloadPrinter GetStatusPayloadPrinter() {
38 return GetStatusPayloadPrinterStorage().load(std::memory_order_relaxed);
39}
40
41} // namespace status_internal
42ABSL_NAMESPACE_END
43} // namespace absl
