32std::string StatusCodeToString(StatusCode code) {
33 switch (code) {
34 case StatusCode::kOk:
35 return "OK";
36 case StatusCode::kCancelled:
37 return "CANCELLED";
38 case StatusCode::kUnknown:
39 return "UNKNOWN";
40 case StatusCode::kInvalidArgument:
41 return "INVALID_ARGUMENT";
42 case StatusCode::kDeadlineExceeded:
43 return "DEADLINE_EXCEEDED";
44 case StatusCode::kNotFound:
45 return "NOT_FOUND";
46 case StatusCode::kAlreadyExists:
47 return "ALREADY_EXISTS";
48 case StatusCode::kPermissionDenied:
49 return "PERMISSION_DENIED";
50 case StatusCode::kUnauthenticated:
51 return "UNAUTHENTICATED";
52 case StatusCode::kResourceExhausted:
53 return "RESOURCE_EXHAUSTED";
54 case StatusCode::kFailedPrecondition:
55 return "FAILED_PRECONDITION";
56 case StatusCode::kAborted:
57 return "ABORTED";
58 case StatusCode::kOutOfRange:
59 return "OUT_OF_RANGE";
60 case StatusCode::kUnimplemented:
61 return "UNIMPLEMENTED";
62 case StatusCode::kInternal:
63 return "INTERNAL";
64 case StatusCode::kUnavailable:
65 return "UNAVAILABLE";
66 case StatusCode::kDataLoss:
67 return "DATA_LOSS";
68 default:
69 return "";
70 }
71}

79static int FindPayloadIndexByUrl(const Payloads* payloads,
80 absl::string_view type_url) {
81 if (payloads == nullptr) return -1;
82
83 for (int i = 0; i < payloads->size(); ++i) {
84 if ((*payloads)[i].type_url == type_url) return i;
85 }
86
87 return -1;
88}

90// Convert canonical code to a value known to this binary.
91absl::StatusCode MapToLocalCode(int value) {
92 absl::StatusCode code = static_cast<absl::StatusCode>(value);
93 switch (code) {
94 case absl::StatusCode::kOk:
95 case absl::StatusCode::kCancelled:
96 case absl::StatusCode::kUnknown:
97 case absl::StatusCode::kInvalidArgument:
98 case absl::StatusCode::kDeadlineExceeded:
99 case absl::StatusCode::kNotFound:
100 case absl::StatusCode::kAlreadyExists:
101 case absl::StatusCode::kPermissionDenied:
102 case absl::StatusCode::kResourceExhausted:
103 case absl::StatusCode::kFailedPrecondition:
104 case absl::StatusCode::kAborted:
105 case absl::StatusCode::kOutOfRange:
106 case absl::StatusCode::kUnimplemented:
107 case absl::StatusCode::kInternal:
108 case absl::StatusCode::kUnavailable:
109 case absl::StatusCode::kDataLoss:
110 case absl::StatusCode::kUnauthenticated:
111 return code;
112 default:
113 return absl::StatusCode::kUnknown;
114 }
115}

118absl::optional<absl::Cord> Status::GetPayload(
119 absl::string_view type_url) const {
120 const auto* payloads = GetPayloads();
121 int index = status_internal::FindPayloadIndexByUrl(payloads, type_url);
122 if (index != -1) return (*payloads)[index].payload;
123
124 return absl::nullopt;
125}
126
127void Status::SetPayload(absl::string_view type_url, absl::Cord payload) {
128 if (ok()) return;
129
130 PrepareToModify();
131
132 status_internal::StatusRep* rep = RepToPointer(rep_);
133 if (!rep->payloads) {
134 rep->payloads = absl::make_unique<status_internal::Payloads>();
135 }
136
137 int index =
138 status_internal::FindPayloadIndexByUrl(rep->payloads.get(), type_url);
139 if (index != -1) {
140 (*rep->payloads)[index].payload = std::move(payload);
141 return;
142 }
143
144 rep->payloads->push_back({std::string(type_url), std::move(payload)});
145}
146
147bool Status::ErasePayload(absl::string_view type_url) {
148 int index = status_internal::FindPayloadIndexByUrl(GetPayloads(), type_url);
149 if (index != -1) {
150 PrepareToModify();
151 GetPayloads()->erase(GetPayloads()->begin() + index);
152 if (GetPayloads()->empty() && message().empty()) {
153 // Special case: If this can be represented inlined, it MUST be
154 // inlined (EqualsSlow depends on this behavior).
155 StatusCode c = static_cast<StatusCode>(raw_code());
156 Unref(rep_);
157 rep_ = CodeToInlinedRep(c);
158 }
159 return true;
160 }
161
162 return false;
163}
164
165void Status::ForEachPayload(
166 const std::function<void(absl::string_view, const absl::Cord&)>& visitor)
167 const {
168 if (auto* payloads = GetPayloads()) {
169 bool in_reverse =
170 payloads->size() > 1 && reinterpret_cast<uintptr_t>(payloads) % 13 > 6;
171
172 for (int index = 0; index < payloads->size(); ++index) {
173 const auto& elem =
174 (*payloads)[in_reverse ? payloads->size() - 1 - index : index];
175
176#ifdef NDEBUG
177 visitor(elem.type_url, elem.payload);
178#else
179 // In debug mode invaldiate the type url to prevent users from relying on
180 // this std::string lifetime.
181
182 // NOLINTNEXTLINE intentional extra conversion to force temporary.
183 visitor(std::string(elem.type_url), elem.payload);
184#endif // NDEBUG
185 }
186 }

201void Status::UnrefNonInlined(uintptr_t rep) {
202 status_internal::StatusRep* r = RepToPointer(rep);
203 // Fast path: if ref==1, there is no need for a RefCountDec (since
204 // this is the only reference and therefore no other thread is
205 // allowed to be mucking with r).
206 if (r->ref.load(std::memory_order_acquire) == 1 ||
207 r->ref.fetch_sub(1, std::memory_order_acq_rel) - 1 == 0) {
208 delete r;
209 }
210}

212uintptr_t Status::NewRep(absl::StatusCode code, absl::string_view msg,
213 std::unique_ptr<status_internal::Payloads> payloads) {
214 status_internal::StatusRep* rep = new status_internal::StatusRep;
215 rep->ref.store(1, std::memory_order_relaxed);
216 rep->code = code;
217 rep->message.assign(msg.data(), msg.size());
218 rep->payloads = std::move(payloads);
219 return PointerToRep(rep);
220}
221
222Status::Status(absl::StatusCode code, absl::string_view msg)
223 : rep_(CodeToInlinedRep(code)) {
224 if (code != absl::StatusCode::kOk && !msg.empty()) {
225 rep_ = NewRep(code, msg, nullptr);
226 }
227}

229int Status::raw_code() const {
230 if (IsInlined(rep_)) {
231 return static_cast<int>(InlinedRepToCode(rep_));
232 }
233 status_internal::StatusRep* rep = RepToPointer(rep_);
234 return static_cast<int>(rep->code);
235}
236
237absl::StatusCode Status::code() const {
238 return status_internal::MapToLocalCode(raw_code());
239}
240
241void Status::PrepareToModify() {
242 ABSL_RAW_CHECK(!ok(), "PrepareToModify shouldn't be called on OK status.");
243 if (IsInlined(rep_)) {
244 rep_ = NewRep(static_cast<absl::StatusCode>(raw_code()),
245 absl::string_view(), nullptr);
246 return;
247 }
248
249 uintptr_t rep_i = rep_;
250 status_internal::StatusRep* rep = RepToPointer(rep_);
251 if (rep->ref.load(std::memory_order_acquire) != 1) {
252 std::unique_ptr<status_internal::Payloads> payloads;
253 if (rep->payloads) {
254 payloads = absl::make_unique<status_internal::Payloads>(*rep->payloads);
255 }
256 rep_ = NewRep(rep->code, message(), std::move(payloads));
257 UnrefNonInlined(rep_i);
258 }
259}

261bool Status::EqualsSlow(const absl::Status& a, const absl::Status& b) {
262 if (IsInlined(a.rep_) != IsInlined(b.rep_)) return false;
263 if (a.message() != b.message()) return false;
264 if (a.raw_code() != b.raw_code()) return false;
265 if (a.GetPayloads() == b.GetPayloads()) return true;
266
267 const status_internal::Payloads no_payloads;
268 const status_internal::Payloads* larger_payloads =
269 a.GetPayloads() ? a.GetPayloads() : &no_payloads;
270 const status_internal::Payloads* smaller_payloads =
271 b.GetPayloads() ? b.GetPayloads() : &no_payloads;
272 if (larger_payloads->size() < smaller_payloads->size()) {
273 std::swap(larger_payloads, smaller_payloads);
274 }
275 if ((larger_payloads->size() - smaller_payloads->size()) > 1) return false;
276 // Payloads can be ordered differently, so we can't just compare payload
277 // vectors.
278 for (const auto& payload : *larger_payloads) {
279
280 bool found = false;
281 for (const auto& other_payload : *smaller_payloads) {
282 if (payload.type_url == other_payload.type_url) {
283 if (payload.payload != other_payload.payload) {
284 return false;
285 }
286 found = true;
287 break;
288 }
289 }
290 if (!found) return false;
291 }
292 return true;
293}

295std::string Status::ToStringSlow() const {
296 std::string text;
297 absl::StrAppend(&text, absl::StatusCodeToString(code()), ": ", message());
298 status_internal::StatusPayloadPrinter printer =
299 status_internal::GetStatusPayloadPrinter();
300 this->ForEachPayload([&](absl::string_view type_url,
301 const absl::Cord& payload) {
302 absl::optional<std::string> result;
303 if (printer) result = printer(type_url, payload);
304 absl::StrAppend(
305 &text, " [", type_url, "='",
306 result.has_value() ? *result : absl::CHexEscape(std::string(payload)),
307 "']");
308 });
309
310 return text;
311}
