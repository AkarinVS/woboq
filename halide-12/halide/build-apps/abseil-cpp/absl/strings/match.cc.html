19namespace absl {

22bool EqualsIgnoreCase(absl::string_view piece1, absl::string_view piece2) {
23 return (piece1.size() == piece2.size() &&
24 0 == absl::strings_internal::memcasecmp(piece1.data(), piece2.data(),
25 piece1.size()));
26 // memcasecmp uses absl::ascii_tolower().
27}
28
29bool StartsWithIgnoreCase(absl::string_view text, absl::string_view prefix) {
30 return (text.size() >= prefix.size()) &&
31 EqualsIgnoreCase(text.substr(0, prefix.size()), prefix);
32}
33
34bool EndsWithIgnoreCase(absl::string_view text, absl::string_view suffix) {
35 return (text.size() >= suffix.size()) &&
36 EqualsIgnoreCase(text.substr(text.size() - suffix.size()), suffix);
37}
