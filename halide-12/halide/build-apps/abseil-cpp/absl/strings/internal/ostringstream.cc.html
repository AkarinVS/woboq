21OStringStream::Buf::int_type OStringStream::overflow(int c) {
22 assert(s_);
23 if (!Buf::traits_type::eq_int_type(c, Buf::traits_type::eof()))
24 s_->push_back(static_cast<char>(c));
25 return 1;
26}
27
28std::streamsize OStringStream::xsputn(const char* s, std::streamsize n) {
29 assert(s_);
30 s_->append(s, n);
31 return n;
32}
