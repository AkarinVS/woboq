28namespace {
29template <typename T>
30[[noreturn]] void Throw(const T& error) {
31#ifdef ABSL_HAVE_EXCEPTIONS
32 throw error;
33#else
34 ABSL_RAW_LOG(FATAL, "%s", error.what());
35 std::abort();
36#endif
37}
38} // namespace
