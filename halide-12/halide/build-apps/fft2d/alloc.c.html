45int **alloc_2d_int(int n1, int n2)
46{
47 int **ii, *i;
48 int j;
49
50 ii = (int **) malloc(sizeof(int *) * n1);
51 alloc_error_check(ii);
52 i = (int *) malloc(sizeof(int) * n1 * n2);
53 alloc_error_check(i);
54 ii[0] = i;
55 for (j = 1; j < n1; j++) {
56 ii[j] = ii[j - 1] + n2;
57 }
58 return ii;
59}
