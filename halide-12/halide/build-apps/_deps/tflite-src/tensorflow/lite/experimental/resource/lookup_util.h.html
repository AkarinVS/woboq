25/// Helper class for accessing TFLite tensor data.
26template <typename T>
27class TensorReader {
28 public:
29 explicit TensorReader(const TfLiteTensor* input) {
30 input_data_ = GetTensorData<T>(input);
31 }
32
33 // Returns the corresponding scalar data at the given index position.
34 // In here, it does not check the validity of the index should be guaranteed
35 // in order not to harm the performance. Caller should take care of it.
36 T GetData(int index) { return input_data_[index]; }
37
38 private:
39 const T* input_data_;
40};
41
42/// Helper class for accessing TFLite tensor data. This specialized class is for
43/// std::string type.
44template <>
45class TensorReader<std::string> {
46 public:
47 explicit TensorReader(const TfLiteTensor* input) : input_(input) {}
48
49 // Returns the corresponding string data at the given index position.
50 // In here, it does not check the validity of the index should be guaranteed
51 // in order not to harm the performance. Caller should take care of it.
52 std::string GetData(int index) {
53 auto string_ref = GetString(input_, index);
54 return std::string(string_ref.str, string_ref.len);
55 }
56
57 private:
58 const TfLiteTensor* input_;
59};

61/// WARNING: Experimental interface, subject to change.
62/// Helper class for writing TFLite tensor data.
63template <typename ValueType>
64class TensorWriter {
65 public:
66 explicit TensorWriter(TfLiteTensor* values) {
67 output_data_ = GetTensorData<ValueType>(values);
68 }
69
70 // Sets the given value to the given index position of the tensor storage.
71 // In here, it does not check the validity of the index should be guaranteed
72 // in order not to harm the performance. Caller should take care of it.
73 void SetData(int index, ValueType& value) { output_data_[index] = value; }
74
75 // Commit updates. In this case, it does nothing since the SetData method
76 // writes data directly.
77 void Commit() {
78 // Noop.
79 }
80
81 private:
82 ValueType* output_data_;
83};
84
85/// WARNING: Experimental interface, subject to change.
86/// Helper class for writing TFLite tensor data. This specialized class is for
87/// std::string type.
88template <>
89class TensorWriter<std::string> {
90 public:
91 explicit TensorWriter(TfLiteTensor* values) : values_(values) {}
92
93 // Queues the given string value to the buffer regardless of the provided
94 // index.
95 // In here, it does not check the validity of the index should be guaranteed
96 // in order not to harm the performance. Caller should take care of it.
97 void SetData(int index, const std::string& value) {
98 buf_.AddString(value.data(), value.length());
99 }
100
101 // Commit updates. The stored data in DynamicBuffer will be written into the
102 // tensor storage.
103 void Commit() { buf_.WriteToTensor(values_, nullptr); }
104
105 private:
106 TfLiteTensor* values_;
107 DynamicBuffer buf_;
108};
