18#include "tensorflow/lite/c/common.h"
19#include "tensorflow/lite/experimental/resource/resource_base.h"

21namespace tflite {
22namespace resource {
23
24/// WARNING: Experimental interface, subject to change.
25// A resource variable class. It's similar to TensorFlow Resource
26// Variable, but it's identified with int32 ID in TFLite (instead of
27// using Resource handle like TensorFlow).
28//
29// TODO(b/137042749): TFLite converter cannot convert variables yet.
30// Variable functionalities are only tested with unit tests now.
31class ResourceVariable : public ResourceBase {
32 public:
33 ResourceVariable();
34 ResourceVariable(ResourceVariable&& other);
35
36 ResourceVariable(const ResourceVariable&) = delete;
37 ResourceVariable& operator=(const ResourceVariable&) = delete;
38
39 ~ResourceVariable() override;
40
41 // Assigns data from a tensor. Copies its type, shape and data over.
42 TfLiteStatus AssignFrom(const TfLiteTensor* tensor);
43
44 // Get the data tensor stored in the resource variable.
45 // Returns `nullptr` if the variable is never initialized by calling
46 // `AssignFrom`.
47 TfLiteTensor* GetTensor() { return is_initialized_ ? &tensor_ : nullptr; }
48
49 // Returns true if this resource variable is initialized.
50 bool IsInitialized() override { return is_initialized_; }
51
52 private:
53 // The tensor (and its buffer stored in `tensor_.data` is fully owned by
54 // the `ResourceVariable` object.
55 TfLiteTensor tensor_;
56 // True if `AssignFrom` function is every called.
57 // False if and only if `tensor_` is filled with zeros.
58 bool is_initialized_ = false;
59};
60
61// Creates a resource variable, shared among all the subgraphs with the given
62// resource id if there is an existing one.
63// WARNING: Experimental interface, subject to change.
64void CreateResourceVariableIfNotAvailable(ResourceMap* resources,
65 int resource_id);
66
67// Returns the corresponding resource variable, or nullptr if none.
68// WARNING: Experimental interface, subject to change.
69ResourceVariable* GetResourceVariable(ResourceMap* resources, int resource_id);
70
71} // namespace resource
72} // namespace tflite
