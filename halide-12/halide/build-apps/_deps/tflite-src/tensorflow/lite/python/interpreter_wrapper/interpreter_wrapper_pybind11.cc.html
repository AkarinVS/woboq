16#include "pybind11/functional.h"
17#include "pybind11/pybind11.h"
18#include "pybind11/pytypes.h"

23namespace py = pybind11;
24using tflite::interpreter_wrapper::InterpreterWrapper;
25
26PYBIND11_MODULE(_pywrap_tensorflow_interpreter_wrapper, m) {
27 m.doc() = R"pbdoc(
28 _pywrap_tensorflow_interpreter_wrapper
29 -----
30 )pbdoc";
31
32 // pybind11 suggests to convert factory functions into constructors, but
33 // when bytes are provided the wrapper will be confused which
34 // constructor to call.
35 m.def("CreateWrapperFromFile",
36 [](const std::string& model_path, int op_resolver_id,
37 const std::vector<std::string>& registerers,
38 bool preserve_all_tensors) {
39 std::string error;
40 auto* wrapper = ::InterpreterWrapper::CreateWrapperCPPFromFile(
41 model_path.c_str(), op_resolver_id, registerers, &error,
42 preserve_all_tensors);
43 if (!wrapper) {
44 throw std::invalid_argument(error);
45 }
46 return wrapper;
47 });
48 m.def(
49 "CreateWrapperFromFile",
50 [](const std::string& model_path, int op_resolver_id,
51 const std::vector<std::string>& registerers_by_name,
52 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
53 bool preserve_all_tensors) {
54 std::string error;
55 auto* wrapper = ::InterpreterWrapper::CreateWrapperCPPFromFile(
56 model_path.c_str(), op_resolver_id, registerers_by_name,
57 registerers_by_func, &error, preserve_all_tensors);
58 if (!wrapper) {
59 throw std::invalid_argument(error);
60 }
61 return wrapper;
62 });
63 m.def("CreateWrapperFromBuffer",
64 [](const py::bytes& data, int op_resolver_id,
65 const std::vector<std::string>& registerers,
66 bool preserve_all_tensors) {
67 std::string error;
68 auto* wrapper = ::InterpreterWrapper::CreateWrapperCPPFromBuffer(
69 data.ptr(), op_resolver_id, registerers, &error,
70 preserve_all_tensors);
71 if (!wrapper) {
72 throw std::invalid_argument(error);
73 }
74 return wrapper;
75 });
76 m.def(
77 "CreateWrapperFromBuffer",
78 [](const py::bytes& data, int op_resolver_id,
79 const std::vector<std::string>& registerers_by_name,
80 const std::vector<std::function<void(uintptr_t)>>& registerers_by_func,
81 bool preserve_all_tensors) {
82 std::string error;
83 auto* wrapper = ::InterpreterWrapper::CreateWrapperCPPFromBuffer(
84 data.ptr(), op_resolver_id, registerers_by_name,
85 registerers_by_func, &error, preserve_all_tensors);
86 if (!wrapper) {
87 throw std::invalid_argument(error);
88 }
89 return wrapper;
90 });
91 py::class_<InterpreterWrapper>(m, "InterpreterWrapper")
92 .def("AllocateTensors",
93 [](InterpreterWrapper& self) {
94 return tensorflow::PyoOrThrow(self.AllocateTensors());
95 })
96 .def("Invoke",
97 [](InterpreterWrapper& self) {

99 })
100 .def("InputIndices",
101 [](const InterpreterWrapper& self) {

103 })
104 .def("OutputIndices",
105 [](InterpreterWrapper& self) {
106 return tensorflow::PyoOrThrow(self.OutputIndices());
107 })
108 .def("ResizeInputTensor",
109 [](InterpreterWrapper& self, int i, py::handle& value, bool strict) {
110 return tensorflow::PyoOrThrow(
111 self.ResizeInputTensor(i, value.ptr(), strict));
112 })

114 .def("TensorName", &InterpreterWrapper::TensorName)

117 return tensorflow::PyoOrThrow(self.TensorType(i));

119 .def("TensorSize",
120 [](const InterpreterWrapper& self, int i) {
121 return tensorflow::PyoOrThrow(self.TensorSize(i));
122 })

124 [](const InterpreterWrapper& self, int i) {
125 return tensorflow::PyoOrThrow(self.TensorSizeSignature(i));

128 [](const InterpreterWrapper& self, int i) {
129 return tensorflow::PyoOrThrow(self.TensorSparsityParameters(i));

131 .def(

134 return tensorflow::PyoOrThrow(self.TensorQuantization(i));

136 R"pbdoc(
137 Deprecated in favor of TensorQuantizationParameters.
138 )pbdoc")
139 .def(
140 "TensorQuantizationParameters",
141 [](InterpreterWrapper& self, int i) {
142 return tensorflow::PyoOrThrow(self.TensorQuantizationParameters(i));
143 })
144 .def("SetTensor",
145 [](InterpreterWrapper& self, int i, py::handle& value) {

147 })

151 })

159 .def("GetOutputTensorFromSignatureDefName",
160 [](const InterpreterWrapper& self, const char* output_name,
161 const char* method_name) {
162 return tensorflow::PyoOrThrow(
163 self.GetOutputTensorFromSignatureDefName(output_name,
164 method_name));
165 })
166 .def("GetSignatureDefs",
167 [](InterpreterWrapper& self) {
168 return tensorflow::PyoOrThrow(self.GetSignatureDefs());
169 })
170 .def("ResetVariableTensors",
171 [](InterpreterWrapper& self) {
172 return tensorflow::PyoOrThrow(self.ResetVariableTensors());
173 })
174 .def("NumNodes", &InterpreterWrapper::NumNodes)
175 .def("NodeName", &InterpreterWrapper::NodeName)
176 .def("NodeInputs",
177 [](const InterpreterWrapper& self, int i) {
178 return tensorflow::PyoOrThrow(self.NodeInputs(i));
179 })
180 .def("NodeOutputs",
181 [](const InterpreterWrapper& self, int i) {
182 return tensorflow::PyoOrThrow(self.NodeOutputs(i));
183 })
184 .def(
185 "tensor",
186 [](InterpreterWrapper& self, py::handle& base_object, int i) {
187 return tensorflow::PyoOrThrow(self.tensor(base_object.ptr(), i));
188 },
189 R"pbdoc(
190 Returns a reference to tensor index i as a numpy array. The
191 base_object should be the interpreter object providing the memory.
192 )pbdoc")
193 .def(
194 "ModifyGraphWithDelegate",
195 // Address of the delegate is passed as an argument.
196 [](InterpreterWrapper& self, uintptr_t delegate_ptr) {
197 return tensorflow::PyoOrThrow(self.ModifyGraphWithDelegate(
198 reinterpret_cast<TfLiteDelegate*>(delegate_ptr)));
199 },
200 R"pbdoc(
201 Adds a delegate to the interpreter.
202 )pbdoc")
203 .def(
204 "SetNumThreads",
205 [](InterpreterWrapper& self, int num_threads) {
206 return tensorflow::PyoOrThrow(self.SetNumThreads(num_threads));
207 },
208 R"pbdoc(
209 ask the interpreter to set the number of threads to use.
210 )pbdoc")
211 .def("interpreter", [](InterpreterWrapper& self) {
212 return reinterpret_cast<intptr_t>(self.interpreter());
213 });
