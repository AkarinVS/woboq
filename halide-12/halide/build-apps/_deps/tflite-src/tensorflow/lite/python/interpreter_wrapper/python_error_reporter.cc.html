18namespace tflite {
19namespace interpreter_wrapper {
20
21// Report an error message
22int PythonErrorReporter::Report(const char* format, va_list args) {
23 char buf[1024];
24 int formatted = vsnprintf(buf, sizeof(buf), format, args);
25 buffer_ << buf;
26 return formatted;
27}
28
29// Set's a Python runtime exception with the last error.
30PyObject* PythonErrorReporter::exception() {
31 std::string last_message = message();
32 PyErr_SetString(PyExc_RuntimeError, last_message.c_str());
33 return nullptr;
34}
35
36// Gets the last error message and clears the buffer.
37std::string PythonErrorReporter::message() {
38 std::string value = buffer_.str();
39 buffer_.clear();
40 return value;
41}
