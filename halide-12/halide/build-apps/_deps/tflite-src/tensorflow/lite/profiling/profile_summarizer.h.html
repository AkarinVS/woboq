28namespace tflite {
29namespace profiling {
30
31// Creates a summary of operator invocations in the interpreter.
32class ProfileSummarizer {
33 public:
34 explicit ProfileSummarizer(
35 std::shared_ptr<ProfileSummaryFormatter> summary_formatter =
36 std::make_shared<ProfileSummaryDefaultFormatter>());
37 virtual ~ProfileSummarizer() {}
38
39 // Process profile events to update statistics for operator invocations.
40 void ProcessProfiles(const std::vector<const ProfileEvent*>& profile_stats,
41 const tflite::Interpreter& interpreter);
42
43 // Returns a string detailing the accumulated runtime stats in the format of
44 // summary_formatter_.
45 std::string GetOutputString() {
46 return summary_formatter_->GetOutputString(stats_calculator_map_,
47 *delegate_stats_calculator_);
48 }
49
50 std::string GetShortSummary() {
51 return summary_formatter_->GetShortSummary(stats_calculator_map_,
52 *delegate_stats_calculator_);
53 }
54
55 tensorflow::StatsCalculator* GetStatsCalculator(uint32_t subgraph_index);
56
57 bool HasProfiles() {
58 for (auto& stats_calc : stats_calculator_map_) {
59 auto subgraph_stats = stats_calc.second.get();
60 if (subgraph_stats->num_runs() >= 1) return true;
61 }
62 return false;
63 }
64
65 private:
66 // Map storing stats per subgraph.
67 std::map<uint32_t, std::unique_ptr<tensorflow::StatsCalculator>>
68 stats_calculator_map_;
69
70 std::unique_ptr<tensorflow::StatsCalculator> delegate_stats_calculator_;
71
72 // Summary formatter for customized output formats.
73 std::shared_ptr<ProfileSummaryFormatter> summary_formatter_;
74};
75
76} // namespace profiling
77} // namespace tflite
