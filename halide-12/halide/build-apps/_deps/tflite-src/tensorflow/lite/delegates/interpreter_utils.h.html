19#include "tensorflow/lite/interpreter.h"
20
21// Utility functions and classes for using delegates.
22
23namespace tflite {
24namespace delegates {
25class InterpreterUtils {
26 public:
27 /// Invokes an interpreter with automatic fallback from delegation to CPU.
28 ///
29 /// If using the delegate fails, the delegate is automatically undone and an
30 /// attempt made to return the interpreter to an invokable state.
31 ///
32 /// Allowing the fallback is suitable only if both of the following hold:
33 /// - The caller is known not to cache pointers to tensor data across Invoke()
34 /// calls.
35 /// - The model is not stateful (no variables, no LSTMs) or the state isn't
36 /// needed between batches.
37 ///
38 /// Returns one of the following three status codes:
39 /// 1. kTfLiteOk: Success. Output is valid.
40 /// 2. kTfLiteDelegateError: Delegate error but fallback succeeded. Output is
41 /// valid.
42 /// NOTE: This undoes all delegates previously applied to the Interpreter.
43 /// 3. kTfLiteError: Unexpected/runtime failure. Output is invalid.
44 /// WARNING: This is an experimental API and subject to change.
45 static TfLiteStatus InvokeWithCPUFallback(Interpreter* interpreter);
46};
47} // namespace delegates
48} // namespace tflite
49
50#endif // TENSORFLOW_LITE_DELEGATES_INTERPRETER_UTILS_H_
