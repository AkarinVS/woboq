88 void OnBenchmarkStart(const BenchmarkParams& params) override {
89 params_ = &params;
90 if (params_->Get<bool>("print_preinvoke_state")) {
91 TFLITE_LOG(INFO) << "\n====Printing out TfLite interpreter pre-invoke "
92 "state begins====";
93 tflite::PrintInterpreterState(interpreter_);
94 TFLITE_LOG(INFO) << "====Printing out TfLite interpreter pre-invoke "
95 "state ends====\n";
96 }
97 }
98
99 void OnBenchmarkEnd(const BenchmarkResults& results) override {
100 if (params_->Get<bool>("print_postinvoke_state")) {
101 TFLITE_LOG(INFO) << "\n====Printing out TfLite interpreter post-invoke "

103 tflite::PrintInterpreterState(interpreter_);
104 TFLITE_LOG(INFO) << "====Printing out TfLite interpreter post-invoke "
105 "state ends====\n";
106 }
107 }
108
109 private:
110 Interpreter* const interpreter_ = nullptr; // not own the memory.
111 const BenchmarkParams* params_ = nullptr; // not own the memory.
112};

122int GetNumElements(const TfLiteIntArray* dim_array) {
123 int num_elements = 1;
124 for (size_t i = 0; i < dim_array->size; i++) {
125 num_elements *= dim_array->data[i];
126 }
127 return num_elements;
128}

130void FillRandomString(tflite::DynamicBuffer* buffer,
131 const TfLiteIntArray* dim_array,
132 const std::function<std::string()>& random_func) {
133 int num_elements = GetNumElements(dim_array);
134 for (int i = 0; i < num_elements; ++i) {
135 auto str = random_func();
136 buffer->AddString(str.data(), str.length());
137 }
138}

140int FindLayerInfoIndex(std::vector<BenchmarkTfLiteModel::InputLayerInfo>* info,
141 const std::string& input_name,
142 const string& names_string) {
143 for (int i = 0; i < info->size(); ++i) {
144 if (info->at(i).name == input_name) {
145 return i;
146 }
147 }
148 TFLITE_LOG(FATAL) << "Cannot find the corresponding input_layer name("
149 << input_name << ") in --input_layer as " << names_string;
150 return -1;
151}

153TfLiteStatus PopulateInputValueRanges(
154 const std::string& names_string, const std::string& value_ranges_string,
155 std::vector<BenchmarkTfLiteModel::InputLayerInfo>* info) {
156 std::vector<std::string> value_ranges = Split(value_ranges_string, ':');
157 for (const auto& val : value_ranges) {
158 std::vector<std::string> name_range = Split(val, ',');
159 if (name_range.size() != 3) {
160 TFLITE_LOG(ERROR) << "Wrong input value range item specified: " << val;
161 return kTfLiteError;
162 }
163
164 // Ensure the specific input layer name exists.
165 int layer_info_idx = FindLayerInfoIndex(info, name_range[0], names_string);
166
167 // Parse the range value.
168 int low, high;
169 bool has_low = absl::SimpleAtoi(name_range[1], &low);
170 bool has_high = absl::SimpleAtoi(name_range[2], &high);
171 if (!has_low || !has_high || low > high) {
172 TFLITE_LOG(ERROR)
173 << "Wrong low and high value of the input value range specified: "
174 << val;
175 return kTfLiteError;
176 }
177 info->at(layer_info_idx).has_value_range = true;
178 info->at(layer_info_idx).low = low;
179 info->at(layer_info_idx).high = high;
180 }
181 return kTfLiteOk;
182}
183
184TfLiteStatus PopulateInputValueFiles(
185 const std::string& names_string, const std::string& value_files_string,
186 std::vector<BenchmarkTfLiteModel::InputLayerInfo>* info) {
187 std::vector<std::string> value_files = Split(value_files_string, ',');
188 for (const auto& val : value_files) {
189 std::vector<std::string> name_file = Split(val, ':');
190 if (name_file.size() != 2) {
191 TFLITE_LOG(ERROR) << "Wrong input value file item specified: " << val;
192 return kTfLiteError;
193 }
194
195 // Ensure the specific input layer name exists.
196 int layer_info_idx = FindLayerInfoIndex(info, name_file[0], names_string);
197 if (info->at(layer_info_idx).has_value_range) {
198 TFLITE_LOG(WARN)
199 << "The input_name:" << info->at(layer_info_idx).name
200 << " appears both in input_layer_value_files and "
201 "input_layer_value_range. The input_layer_value_range of the "
202 "input_name will be ignored.";
203 }
204 info->at(layer_info_idx).input_file_path = name_file[1];
205 }
206 return kTfLiteOk;
207}

209TfLiteStatus PopulateInputLayerInfo(
210 const std::string& names_string, const std::string& shapes_string,
211 const std::string& value_ranges_string,
212 const std::string& value_files_string,
213 std::vector<BenchmarkTfLiteModel::InputLayerInfo>* info) {
214 info->clear();
215 std::vector<std::string> names = Split(names_string, ',');
216 std::vector<std::string> shapes = Split(shapes_string, ':');
217
218 if (names.size() != shapes.size()) {
219 TFLITE_LOG(ERROR) << "The number of items in"
220 << " --input_layer_shape (" << shapes_string << ", with "
221 << shapes.size() << " items)"
222 << " must match the number of items in"
223 << " --input_layer (" << names_string << ", with "
224 << names.size() << " items)."
225 << " For example --input_layer=input1,input2"
226 << " --input_layer_shape=1,224,224,4:1,20";
227 return kTfLiteError;
228 }
229
230 for (int i = 0; i < names.size(); ++i) {
231 info->push_back(BenchmarkTfLiteModel::InputLayerInfo());
232 BenchmarkTfLiteModel::InputLayerInfo& input = info->back();
233
234 input.name = names[i];
235
236 TFLITE_TOOLS_CHECK(util::SplitAndParse(shapes[i], ',', &input.shape))
237 << "Incorrect size string specified: " << shapes[i];
238 for (int dim : input.shape) {
239 if (dim == -1) {
240 TFLITE_LOG(ERROR)
241 << "Any unknown sizes in the shapes (-1's) must be replaced"
242 << " with the size you want to benchmark with.";
243 return kTfLiteError;
244 }
245 }
246 }
247
248 // Populate input value range if it's specified.
249 TF_LITE_ENSURE_STATUS(
250 PopulateInputValueRanges(names_string, value_ranges_string, info));
251
252 // Populate input value files if it's specified.
253 TF_LITE_ENSURE_STATUS(
254 PopulateInputValueFiles(names_string, value_files_string, info));
255
256 return kTfLiteOk;
257}

439BenchmarkTfLiteModel::InputTensorData BenchmarkTfLiteModel::LoadInputTensorData(
440 const TfLiteTensor& t, const std::string& input_file_path) {
441 std::ifstream value_file(input_file_path, std::ios::binary);
442 if (!value_file.good()) {
443 TFLITE_LOG(FATAL) << "Failed to read the input_layer_value_file:"
444 << input_file_path;
445 }
446 InputTensorData t_data;
447 if (t.type == kTfLiteString) {
448 t_data.data = VoidUniquePtr(
449 static_cast<void*>(new tflite::DynamicBuffer()),
450 [](void* ptr) { delete static_cast<DynamicBuffer*>(ptr); });
451 std::string line;
452 size_t num_line = 0;
453 // Read the line with the delimiter '\0'.
454 while (std::getline(value_file, line, '\0')) {
455 num_line++;
456 static_cast<DynamicBuffer*>(t_data.data.get())
457 ->AddString(line.data(), line.length());
458 }
459 int num_elements = GetNumElements(t.dims);
460 if (num_line != num_elements) {
461 TFLITE_LOG(FATAL) << "The number of string in the input_layer_value_file("
462 << input_file_path << ") is " << num_line
463 << ". It should be " << num_elements << ".";
464 }
465 } else {
466 value_file.seekg(0, std::ios_base::end);
467 if (value_file.tellg() != t.bytes) {
468 TFLITE_LOG(FATAL) << "The size of " << input_file_path << " is "
469 << value_file.tellg() << " bytes. It should be "
470 << t.bytes << " bytes.";
471 }
472 t_data.bytes = t.bytes;
473 t_data.data =
474 VoidUniquePtr(static_cast<void*>(new char[t.bytes]),
475 [](void* ptr) { delete[] static_cast<char*>(ptr); });
476 value_file.clear();
477 value_file.seekg(0, std::ios_base::beg);
478 value_file.read(static_cast<char*>(t_data.data.get()), t.bytes);
479 }
480 return t_data;
481}

483BenchmarkTfLiteModel::InputTensorData
484BenchmarkTfLiteModel::CreateRandomTensorData(const TfLiteTensor& t,
485 const InputLayerInfo* layer_info) {
486 bool has_value_range = false;
487 int low_range = 0;
488 int high_range = 0;
489 if (layer_info) {
490 has_value_range = layer_info->has_value_range;
491 low_range = layer_info->low;
492 high_range = layer_info->high;
493 }
494 int num_elements = GetNumElements(t.dims);
495 switch (t.type) {
496 case kTfLiteFloat32: {
497 return CreateInputTensorData<float>(
498 num_elements, std::uniform_real_distribution<float>(-0.5f, 0.5f));
499 }
500 case kTfLiteFloat16: {
501 // TODO(b/138843274): Remove this preprocessor guard when bug is fixed.
502#if TFLITE_ENABLE_FP16_CPU_BENCHMARKS
503#if __GNUC__ && \
504 (__clang__ || __ARM_FP16_FORMAT_IEEE || __ARM_FP16_FORMAT_ALTERNATIVE)
505 // __fp16 is available on Clang or when __ARM_FP16_FORMAT_* is defined.
506 return CreateInputTensorData<__fp16>(
507 num_elements, std::uniform_real_distribution<float>(-0.5f, 0.5f));
508#else
509 TFLITE_LOG(FATAL) << "Don't know how to populate tensor " << t->name
510 << " of type FLOAT16 on this platform.";
511#endif
512#else
513 // You need to build with -DTFLITE_ENABLE_FP16_CPU_BENCHMARKS=1 using a
514 // compiler that supports __fp16 type. Note: when using Clang and *not*
515 // linking with compiler-rt, a definition of __gnu_h2f_ieee and
516 // __gnu_f2h_ieee must be supplied.
517 TFLITE_LOG(FATAL) << "Populating the tensor " << t.name
518 << " of type FLOAT16 is disabled.";
519#endif // TFLITE_ENABLE_FP16_CPU_BENCHMARKS
520 break;
521 }
522 case kTfLiteFloat64: {
523 return CreateInputTensorData<double>(
524 num_elements, std::uniform_real_distribution<double>(-0.5, 0.5));
525 }
526 case kTfLiteInt64: {
527 int low = has_value_range ? low_range : 0;
528 int high = has_value_range ? high_range : 99;
529 return CreateInputTensorData<int64_t>(
530 num_elements, std::uniform_int_distribution<int64_t>(low, high));
531 }
532 case kTfLiteInt32: {
533 int low = has_value_range ? low_range : 0;
534 int high = has_value_range ? high_range : 99;
535 return CreateInputTensorData<int32_t>(
536 num_elements, std::uniform_int_distribution<int32_t>(low, high));
537 }
538 case kTfLiteUInt32: {
539 int low = has_value_range ? low_range : 0;
540 int high = has_value_range ? high_range : 99;
541 return CreateInputTensorData<uint32_t>(
542 num_elements, std::uniform_int_distribution<uint32_t>(low, high));
543 }
544 case kTfLiteInt16: {
545 int low = has_value_range ? low_range : 0;
546 int high = has_value_range ? high_range : 99;
547 return CreateInputTensorData<int16_t>(
548 num_elements, std::uniform_int_distribution<int16_t>(low, high));
549 }
550 case kTfLiteUInt8: {
551 int low = has_value_range ? low_range : 0;
552 int high = has_value_range ? high_range : 254;
553 // std::uniform_int_distribution is specified not to support char types.
554 return CreateInputTensorData<uint8_t>(
555 num_elements, std::uniform_int_distribution<uint32_t>(low, high));
556 }
557 case kTfLiteInt8: {
558 int low = has_value_range ? low_range : -127;
559 int high = has_value_range ? high_range : 127;
560 // std::uniform_int_distribution is specified not to support char types.
561 return CreateInputTensorData<int8_t>(
562 num_elements, std::uniform_int_distribution<int32_t>(low, high));
563 }
564 case kTfLiteString: {
565 // TODO(haoliang): No need to cache string tensors right now.
566 break;
567 }
568 case kTfLiteBool: {
569 // According to std::uniform_int_distribution specification, non-int type
570 // is not supported.
571 return CreateInputTensorData<bool>(
572 num_elements, std::uniform_int_distribution<uint32_t>(0, 1));
573 }
574 default: {
575 TFLITE_LOG(FATAL) << "Don't know how to populate tensor " << t.name
576 << " of type " << t.type;
577 }
578 }
579 return InputTensorData();
580}

582TfLiteStatus BenchmarkTfLiteModel::PrepareInputData() {
583 CleanUp();
584
585 // Note the corresponding relation between 'interpreter_inputs' and 'inputs_'
586 // (i.e. the specified input layer info) has been checked in
587 // BenchmarkTfLiteModel::Init() before calling this function. So, we simply
588 // use the corresponding input layer info to initializethe input data value
589 // properly.
590 auto interpreter_inputs = interpreter_->inputs();
591 for (int i = 0; i < interpreter_inputs.size(); ++i) {
592 int tensor_index = interpreter_inputs[i];
593 const TfLiteTensor& t = *(interpreter_->tensor(tensor_index));
594 const InputLayerInfo* input_layer_info = nullptr;
595 // Note that when input layer parameters (i.e. --input_layer,
596 // --input_layer_shape) are not specified, inputs_ is empty.
597 if (!inputs_.empty()) input_layer_info = &inputs_[i];
598
599 InputTensorData t_data;
600 if (input_layer_info && !input_layer_info->input_file_path.empty()) {
601 t_data = LoadInputTensorData(t, input_layer_info->input_file_path);
602 } else {
603 t_data = CreateRandomTensorData(t, input_layer_info);
604 }
605 inputs_data_.push_back(std::move(t_data));
606 }
607 return kTfLiteOk;
608}

610TfLiteStatus BenchmarkTfLiteModel::ResetInputsAndOutputs() {
611 auto interpreter_inputs = interpreter_->inputs();
612 // Set the values of the input tensors from inputs_data_.
613 for (int j = 0; j < interpreter_inputs.size(); ++j) {
614 int i = interpreter_inputs[j];
615 TfLiteTensor* t = interpreter_->tensor(i);
616 if (t->type == kTfLiteString) {
617 if (inputs_data_[j].data) {
618 static_cast<DynamicBuffer*>(inputs_data_[j].data.get())
619 ->WriteToTensor(t, /*new_shape=*/nullptr);
620 } else {
621 tflite::DynamicBuffer buffer;
622 FillRandomString(&buffer, t->dims, []() {
623 return "we're have some friends over saturday to hang out in the "
624 "yard";
625 });
626 buffer.WriteToTensor(t, /*new_shape=*/nullptr);
627 }
628 } else {
629 std::memcpy(t->data.raw, inputs_data_[j].data.get(),
630 inputs_data_[j].bytes);
631 }
632 }
633
634 return kTfLiteOk;
635}

637TfLiteStatus BenchmarkTfLiteModel::InitInterpreter() {
638 auto resolver = GetOpResolver();
639 const int32_t num_threads = params_.Get<int32_t>("num_threads");
640 const bool use_caching = params_.Get<bool>("use_caching");
641 tflite::InterpreterBuilder(*model_, *resolver)(&interpreter_, num_threads);
642 if (!interpreter_) {
643 TFLITE_LOG(ERROR) << "Failed to initialize the interpreter";
644 return kTfLiteError;
645 }
646 // Manually enable caching behavior in TF Lite interpreter.
647 if (use_caching) {
648 external_context_.reset(new tflite::ExternalCpuBackendContext());
649 std::unique_ptr<tflite::CpuBackendContext> cpu_backend_context(
650 new tflite::CpuBackendContext());
651 cpu_backend_context->SetUseCaching(true);
652 cpu_backend_context->SetMaxNumThreads(num_threads);
653 external_context_->set_internal_backend_context(
654 std::move(cpu_backend_context));
655 interpreter_->SetExternalContext(kTfLiteCpuBackendContext,
656 external_context_.get());
657 }
658
659 return kTfLiteOk;
660}
661
662TfLiteStatus BenchmarkTfLiteModel::Init() {
663 TF_LITE_ENSURE_STATUS(LoadModel());
664 TF_LITE_ENSURE_STATUS(InitInterpreter());
665
666 // Install profilers if necessary right after interpreter is created so that
667 // any memory allocations inside the TFLite runtime could be recorded if the
668 // installed profiler profile memory usage information.
669 profiling_listener_ = MayCreateProfilingListener();
670 if (profiling_listener_) AddListener(profiling_listener_.get());
671
672 interpreter_state_printer_ = std::unique_ptr<BenchmarkListener>(
673 new InterpreterStatePrinter(interpreter_.get()));
674 AddListener(interpreter_state_printer_.get());
675
676 interpreter_->SetAllowFp16PrecisionForFp32(params_.Get<bool>("allow_fp16"));
677
678 owned_delegates_.clear();
679
680 // Contains all ids of TfLiteNodes that have been checked to see whether it's
681 // delegated or not.
682 std::unordered_set<int> checked_node_ids;
683 for (const auto& delegate_provider :
684 tools::GetRegisteredDelegateProviders()) {
685 auto delegate = delegate_provider->CreateTfLiteDelegate(params_);
686 // It's possible that a delegate of certain type won't be created as
687 // user-specified benchmark params tells not to.
688 if (delegate == nullptr) continue;
689 if (interpreter_->ModifyGraphWithDelegate(delegate.get()) != kTfLiteOk) {
690 TFLITE_LOG(ERROR) << "Failed to apply " << delegate_provider->GetName()
691 << " delegate.";
692 return kTfLiteError;
693 } else {
694 // Ideally, such delegate info should already be computed when the
695 // delegate is being applied to the model graph.
696 int num_delegated_kernels = 0;
697 for (int i = 0; i < interpreter_->execution_plan().size(); ++i) {
698 int node_id = interpreter_->execution_plan()[i];
699 if (checked_node_ids.find(node_id) != checked_node_ids.end()) {
700 continue;
701 }
702 const TfLiteNode& node =
703 interpreter_->node_and_registration(node_id)->first;
704
705 // Note that the 'delegate' here could be an ExternalDelegateWrapper
706 // object that wraps an actual external delegate, in which case,
707 // 'node.delegate' will be different from 'delegate' because
708 // 'node.delegate' refers to the actual external delegate.
709 if (node.delegate != nullptr) {
710 num_delegated_kernels++;
711 checked_node_ids.insert(node_id);
712 }
713 }
714 bool fully_delegated = (num_delegated_kernels == 1 &&
715 interpreter_->execution_plan().size() == 1);
716
717 if (params_.Get<bool>("require_full_delegation") && !fully_delegated) {
718 TFLITE_LOG(ERROR) << "Disallowed CPU fallback detected.";
719 return kTfLiteError;
720 }
721 if (fully_delegated) {
722 TFLITE_LOG(INFO) << "Explicitly applied "
723 << delegate_provider->GetName()
724 << " delegate, and the model graph will be completely"
725 << " executed by the delegate.";
726 } else if (num_delegated_kernels > 0) {
727 TFLITE_LOG(INFO) << "Explicitly applied "
728 << delegate_provider->GetName()
729 << " delegate, and the model graph will be partially"
730 << " executed by the delegate w/ "
731 << num_delegated_kernels << " delegate kernels.";
732 } else {
733 TFLITE_LOG(INFO)
734 << "Though " << delegate_provider->GetName()
735 << " delegate is explicitly applied, the model graph will not be"
736 << " executed by the delegate.";
737 }
738 }
739 owned_delegates_.emplace_back(std::move(delegate));
740 }
741
742 auto interpreter_inputs = interpreter_->inputs();
743
744 if (!inputs_.empty()) {
745 TFLITE_TOOLS_CHECK_EQ(inputs_.size(), interpreter_inputs.size())
746 << "Inputs mismatch: Model inputs #:" << inputs_.size()
747 << " expected: " << interpreter_inputs.size();
748 }
749
750 // Check if the tensor names match, and log a warning if it doesn't.
751 // TODO(ycling): Consider to make this an error again when the new converter
752 // create tensors with consistent naming.
753 for (int j = 0; j < inputs_.size(); ++j) {
754 const InputLayerInfo& input = inputs_[j];
755 int i = interpreter_inputs[j];
756 TfLiteTensor* t = interpreter_->tensor(i);
757 if (input.name != t->name) {
758 TFLITE_LOG(WARN) << "Tensor # " << i << " is named " << t->name
759 << " but flags call it " << input.name;
760 }
761
762 if (input.shape.size() != t->dims->size) {
763 TFLITE_LOG(ERROR) << "Input tensor #" << i << " should have "
764 << t->dims->size << " dimensions!";
765 return kTfLiteError;
766 }
767 }
768
769 // Resize all non-string tensors.
770 for (int j = 0; j < inputs_.size(); ++j) {
771 const InputLayerInfo& input = inputs_[j];
772 int i = interpreter_inputs[j];
773 TfLiteTensor* t = interpreter_->tensor(i);
774 if (t->type != kTfLiteString) {
775 interpreter_->ResizeInputTensor(i, input.shape);
776 }
777 }
778
779 if (interpreter_->AllocateTensors() != kTfLiteOk) {
780 TFLITE_LOG(ERROR) << "Failed to allocate tensors!";
781 return kTfLiteError;
782 }
783
784 ruy_profiling_listener_.reset(new RuyProfileListener());
785 AddListener(ruy_profiling_listener_.get());
786
787 return kTfLiteOk;
788}
