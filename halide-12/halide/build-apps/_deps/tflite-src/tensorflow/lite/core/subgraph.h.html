271 // Ensure the data in `tensor.data` is readable. In case delegate is used,
272 // it might require to copy the data from delegate buffer to raw memory.
273 // WARNING: This is an experimental API and subject to change.
274 // TODO(b/119495520): make this private when refactoring complete.
275 TfLiteStatus EnsureTensorDataIsReadable(int tensor_index) {
276 TfLiteTensor* t = &tensors_[tensor_index];
277 TF_LITE_ENSURE(&context_, t != nullptr);
278 if (t->data_is_stale) {
279 TF_LITE_ENSURE(&context_, t->delegate != nullptr);
280 TF_LITE_ENSURE(&context_, t->buffer_handle != kTfLiteNullBufferHandle);
281 TF_LITE_ENSURE(&context_, t->delegate->CopyFromBufferHandle != nullptr);
282 // TODO(b/120420546): we must add a test that exercise this code.
283 TF_LITE_ENSURE_STATUS(t->delegate->CopyFromBufferHandle(
284 &context_, t->delegate, t->buffer_handle, t));
285 t->data_is_stale = false;
286 }
287 return kTfLiteOk;
288 }

327 // Assigns (or reassigns) a custom memory allocation for the given tensor.
328 // `flags` is a bitmask, see TfLiteCustomAllocationFlags.
329 // The runtime does NOT take ownership of the underlying memory.
330 //
331 // NOTE: User needs to call AllocateTensors() after this. In case of input
332 // resizing, buffers will be checked for required data size during
333 // AllocateTensors().
334 //
335 // Parameters should satisfy the following conditions:
336 // 1. tensor->allocation_type == kTfLiteArenaRw or kTfLiteArenaRwPersistent
337 // In general, this is true for I/O tensors & variable tensors.
338 // 2. allocation->data has the appropriate permissions for runtime access
339 // (Read-only for inputs, Read-Write for others), and outlives Interpreter.
340 // 3. allocation->bytes >= tensor->bytes.
341 // This condition is checked again if any tensors are resized.
342 // 4. allocation->data should be aligned to kDefaultTensorAlignment
343 // defined in lite/util.h. (Currently 64 bytes)
344 // This check is skipped if kTfLiteCustomAllocationFlagsSkipAlignCheck is
345 // set through `flags`.
346 // TODO(b/182215910): Expand on this documentation in a g3doc.
347 //
348 // WARNING: This is an experimental interface that is subject to change.
349 TfLiteStatus SetCustomAllocationForTensor(
350 int tensor_index, const TfLiteCustomAllocation& allocation,
351 int64_t flags = kTfLiteCustomAllocationFlagsNone);

358 // SubgraphAwareProfiler wraps an actual TFLite profiler, such as a
359 // BufferedProfiler instance, and takes care of event profiling/tracing in a
360 // certain subgraph.
361 class SubgraphAwareProfiler : public Profiler {
362 public:
363 // Constructor should be called with the non-nullptr profiler argument.
364 SubgraphAwareProfiler(Profiler* profiler, int64_t subgraph_index)
365 : profiler_(profiler), subgraph_index_(subgraph_index) {}
366 ~SubgraphAwareProfiler() override {}
367
368 uint32_t BeginEvent(const char* tag, EventType event_type,
369 int64_t event_metadata1,
370 int64_t event_metadata2) override {
371 if (!profiler_) return 0;
372 return profiler_->BeginEvent(tag, event_type, event_metadata1,
373 subgraph_index_);
374 }
375
376 void EndEvent(uint32_t event_handle) override {
377 if (!profiler_) return;
378 profiler_->EndEvent(event_handle);
379 }
380
381 void EndEvent(uint32_t event_handle, int64_t event_metadata1,
382 int64_t event_metadata2) override {
383 if (!profiler_) return;
384 profiler_->EndEvent(event_handle, event_metadata1, event_metadata2);
385 }
386
387 void AddEvent(const char* tag, EventType event_type, uint64_t start,
388 uint64_t end, int64_t event_metadata1,
389 int64_t event_metadata2) override {
390 if (!profiler_) return;
391 profiler_->AddEvent(tag, event_type, start, end, event_metadata1,
392 subgraph_index_);
393 }
394
395 private:
396 // Not own the memory.
397 Profiler* const profiler_;
398 const int64_t subgraph_index_;

561 // WARNING: This is an experimental API and subject to change.
562 // Allow a delegate to look at the graph and modify the graph to handle
563 // parts of the graph themselves. After this is called, the graph may
564 // contain new nodes that replace 1 more nodes.
565 // NOTE: If tensors were allocated prior to delegate application, they will
566 // be reallocated if the graph was modified (i.e., the caller does *not* need
567 // to explicitly call |AllocateTensors()| again). If tensors were unallocated,
568 // they will remain unallocated after delegate application.
569 // Returns one of the following status codes:
570 // 1. kTfLiteOk: Delegation succeeded
571 // 2. kTfLiteDelegateError: Delegation failed due to an error *in the
572 // delegate*, or the delegate parameter was null. The Subgraph has been
573 // restored to its pre-delegation state.
574 // NOTE: This reverts all delegates previously applied to the Subgraph.
575 // 3. kTfLiteApplicationError : Delegation failed to be applied due to the
576 // incompatibility with the TfLite runtime, e.g., the model graph is already
577 // immutable when applying the delegate. However, the Subgraph is still in a
578 // invokable state.
579 // 4. kTfLiteError: Unexpected/runtime failure.
580 TfLiteStatus ModifyGraphWithDelegate(TfLiteDelegate* delegate);
