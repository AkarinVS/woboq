98// Returns true if at least one tensor in the given list is kTfLiteDynamic.
99template <typename TensorIntArray>
100bool HasDynamicTensorImpl(const TfLiteContext& context,
101 const TensorIntArray& int_array) {
102 for (int i : int_array) {
103 if (i == kTfLiteOptionalTensor) continue;
104 const TfLiteTensor& tensor = context.tensors[i];
105 if (tensor.allocation_type == kTfLiteDynamic) {
106 return true;
107 }
108 }
109 return false;
110}

117// Gets the legacy TfLiteQuantizationParams from the current TfLiteQuantization.
118TfLiteQuantizationParams GetLegacyQuantization(
119 const TfLiteQuantization& quantization) {
120 TfLiteQuantizationParams legacy_quantization;
121 legacy_quantization.scale = 0;
122 legacy_quantization.zero_point = 0;
123
124 // If the quantization type isn't affine, return the empty
125 // legacy_quantization.
126 if (quantization.type != kTfLiteAffineQuantization) {
127 return legacy_quantization;
128 }
129
130 auto* affine_quantization =
131 static_cast<TfLiteAffineQuantization*>(quantization.params);
132 if (!affine_quantization || !affine_quantization->scale ||
133 !affine_quantization->zero_point ||
134 affine_quantization->scale->size != 1 ||
135 affine_quantization->zero_point->size != 1) {
136 return legacy_quantization;
137 }
138
139 // We know its per-layer quantization now.
140 legacy_quantization.scale = affine_quantization->scale->data[0];
141 legacy_quantization.zero_point = affine_quantization->zero_point->data[0];
142 return legacy_quantization;
143}

158// An utility test to detect if the subgraph is abused:
159// 1. Detects if recursion exists in the graph (recursion is not currently
160// supported.
161// 2. Detects if the interpreter / subgraph is used in multiple subgraphs.
162// Note: It's clearly documented that the interpreter / subgraph are not
163// thread-safe. This serves as a check with possible false negatives
164// unless we switch to atomic boolean flags.
165class SubgraphGuard {
166 public:
167 SubgraphGuard(TfLiteContext* context, bool* is_subgraph_in_use)
168 : is_subgraph_in_use_(is_subgraph_in_use) {
169 if (*is_subgraph_in_use_) {
170 TF_LITE_KERNEL_LOG(
171 context,
172 "Subgraph is already in use. Using an interpreter or a subgraph in "
173 "multiple threads is not supported. Recursion in the graph is not "
174 "supported.");
175 status_ = kTfLiteError;
176 } else {
177 *is_subgraph_in_use_ = true;
178 }
179 }
180 ~SubgraphGuard() {
181 // If tht original status was OK, recover the boolean flag.
182 if (status_ == kTfLiteOk) {
183 *is_subgraph_in_use_ = false;
184 }
185 }
186
187 TfLiteStatus status() const { return status_; }
188
189 private:
190 TfLiteStatus status_ = kTfLiteOk;
191 bool* is_subgraph_in_use_;
192};

266Subgraph::~Subgraph() {
267 for (int node_index = 0; node_index < nodes_and_registration_.size();
268 ++node_index) {
269 CleanupNode(node_index);
270 }
271
272 for (size_t i = 0; i < context_.tensors_size; i++) {
273 TfLiteTensor* tensor = &context_.tensors[i];
274 if (tensor->buffer_handle != kTfLiteNullBufferHandle &&
275 tensor->delegate->FreeBufferHandle != nullptr) {
276 tensor->delegate->FreeBufferHandle(&context_, tensor->delegate,
277 &tensor->buffer_handle);
278 }
279 TfLiteTensorFree(tensor);
280 }

306// Copy a std::vector<int> to an existing TfLiteIntArray.
307// This is a low-level data manipulation function, and it's caller's
308// responsibility to ensure TfLiteIntArray has enough size.
309void CopyVectorToTfLiteIntArray(const std::vector<int>& vec,
310 TfLiteIntArray* arr) {
311 arr->size = vec.size();
312 memcpy(arr->data, vec.data(), sizeof(int) * arr->size);
313}

315// This function allocates a continuous memory space that contains a
316// TfLiteDelegateParams followed by a several TfLiteIntArray.
317// When calling `free` at TfLiteDelegateParams*, all the allocated space
318// will be freed together.
319//
320// +-----------------------------------+
321// | TfLiteDelegateParams |
322// | TfLiteDelegate* delegate; |
323// | TfLiteIntArray* nodes_to_replace; |--\
324// | TfLiteIntArray* input_tensors; |--+--\
325// | TfLiteIntArray* output_tensors; |--+--+--\
326// +-----------------------------------+ | | |
327// | TfLiteIntArray (variable size) |<-/ | |
328// +-----------------------------------+ | |
329// | TfLiteIntArray (variable size) |<----/ |
330// +-----------------------------------+ |
331// | TfLiteIntArray (variable size) |<-------/
332// +-----------------------------------+
333TfLiteDelegateParams* CreateDelegateParams(TfLiteDelegate* delegate,
334 const NodeSubset& node_subset) {
335 // Step 1: Calculate the allocation size.
336 int allocation_size = sizeof(TfLiteDelegateParams);
337
338 int nodes_to_replace_size =
339 TfLiteIntArrayGetSizeInBytes(node_subset.nodes.size());
340 allocation_size += nodes_to_replace_size;
341
342 int input_tensors_size =
343 TfLiteIntArrayGetSizeInBytes(node_subset.input_tensors.size());
344 allocation_size += input_tensors_size;
345
346 int output_tensors_size =
347 TfLiteIntArrayGetSizeInBytes(node_subset.output_tensors.size());
348 allocation_size += output_tensors_size;
349
350 // Step 2: Allocate the memory.
351 // Use `char*` for conveniently step through the allocated space by bytes.
352 char* allocation = static_cast<char*>(malloc(allocation_size));
353
354 // Step 3: Fill all data structures.
355 TfLiteDelegateParams* params =
356 reinterpret_cast<TfLiteDelegateParams*>(allocation);
357 params->delegate = delegate;
358 allocation += sizeof(TfLiteDelegateParams);
359
360 params->nodes_to_replace = reinterpret_cast<TfLiteIntArray*>(allocation);
361 CopyVectorToTfLiteIntArray(node_subset.nodes, params->nodes_to_replace);
362 allocation += nodes_to_replace_size;
363
364 params->input_tensors = reinterpret_cast<TfLiteIntArray*>(allocation);
365 CopyVectorToTfLiteIntArray(node_subset.input_tensors, params->input_tensors);
366 allocation += input_tensors_size;
367
368 params->output_tensors = reinterpret_cast<TfLiteIntArray*>(allocation);
369 CopyVectorToTfLiteIntArray(node_subset.output_tensors,
370 params->output_tensors);
371 allocation += output_tensors_size;
372
373 return params;
374}

376// Assumes that params is not nullptr.
377void PopulatePreviewDelegateParams(const NodeSubset& node_subset,
378 TfLiteDelegateParams* params) {
379 // Since these params are used for previewing partitioning, params->delegate
380 // is not required.
381 params->delegate = nullptr;
382
383 params->nodes_to_replace = TfLiteIntArrayCreate(node_subset.nodes.size());
384 CopyVectorToTfLiteIntArray(node_subset.nodes, params->nodes_to_replace);
385
386 params->input_tensors =
387 TfLiteIntArrayCreate(node_subset.input_tensors.size());
388 CopyVectorToTfLiteIntArray(node_subset.input_tensors, params->input_tensors);
389
390 params->output_tensors =
391 TfLiteIntArrayCreate(node_subset.output_tensors.size());
392 CopyVectorToTfLiteIntArray(node_subset.output_tensors,
393 params->output_tensors);
394}

398TfLiteStatus Subgraph::ReplaceNodeSubsetsWithDelegateKernels(
399 TfLiteRegistration registration, const TfLiteIntArray* nodes_to_replace,
400 TfLiteDelegate* delegate) {
401 // Ignore empty node replacement sets.
402 if (!nodes_to_replace->size) {
403 return kTfLiteOk;
404 }
405
406 // Annotate the registration as DELEGATE op.
407 registration.builtin_code = BuiltinOperator_DELEGATE;
408
409 // Analyze the graph to find all independent node_subsets that are either
410 // fully not-this-delegate or this-delegate computation.
411 InterpreterInfo info(this);
412 std::vector<NodeSubset> node_subsets;
413 PartitionGraphIntoIndependentNodeSubsets(&info, nodes_to_replace,
414 &node_subsets);
415
416 TFLITE_LOG(
417 tflite::TFLITE_LOG_INFO,
418 "Replacing %d node(s) with delegate (%s) node, yielding %zu partitions.",
419 nodes_to_replace->size,
420 registration.custom_name ? registration.custom_name : "unknown",
421 node_subsets.size());
422
423 execution_plan_.clear();
424
425 for (auto& node_subset : node_subsets) {
426 // Subsets claimed by the delegate should have a "macro" op created, the
427 // other node_subsets (kTfNonPartition) just have their nodes added back to
428 // the execution plan.
429 switch (node_subset.type) {
430 case NodeSubset::kTfNonPartition:
431 for (auto it = node_subset.nodes.begin(); it != node_subset.nodes.end();
432 ++it) {
433 execution_plan_.push_back(*it);
434 }
435 break;
436 case NodeSubset::kTfPartition: {
437 int node_index;
438
439 TfLiteDelegateParams* params =
440 CreateDelegateParams(delegate, node_subset);
441 TF_LITE_ENSURE_STATUS(AddNodeWithParameters(
442 node_subset.input_tensors, node_subset.output_tensors, {}, nullptr,
443 0, params, &registration, &node_index));
444
445 // Initialize the output tensors's delegate-related fields.
446 for (int tensor_index : node_subset.output_tensors) {
447 TfLiteTensor* tensor = &tensors_[tensor_index];
448 TF_LITE_ENSURE(&context_, tensor->delegate == nullptr ||
449 tensor->delegate == delegate);
450 tensor->delegate = delegate;
451 }
452
453 // Associate the node with the delegate.
454 TfLiteNode* node = &nodes_and_registration_[node_index].first;
455 node->delegate = delegate;
456 } break;
457 case NodeSubset::kTfUnexplored:
458 return kTfLiteError;
459 break;
460 }
461 }
462 return kTfLiteOk;
463}

522TfLiteStatus Subgraph::PreviewDelegatePartitioning(
523 const TfLiteIntArray* nodes_to_replace,
524 TfLiteDelegateParams** partition_params_array, int* num_partitions) {
525 // Ensure partitioning cache is empty.
526 FreeDelegatePartitioningData();
527 // Defaults.
528 if (!partition_params_array || !num_partitions) return kTfLiteError;
529 *partition_params_array = nullptr;
530 *num_partitions = 0;
531 if (!nodes_to_replace->size) {
532 return kTfLiteOk;
533 }
534
535 // Partition the execution plan into node subsets.
536 InterpreterInfo info(this);
537 std::vector<NodeSubset> node_subsets;
538 PartitionGraphIntoIndependentNodeSubsets(&info, nodes_to_replace,
539 &node_subsets);
540
541 // Create one TfLiteDelegateParams per node-subset which would be delegated.
542 for (auto& node_subset : node_subsets) {
543 if (node_subset.type != NodeSubset::kTfPartition) {
544 continue;
545 }
546 partitioning_preview_cache_.emplace_back();
547 PopulatePreviewDelegateParams(node_subset,
548 &partitioning_preview_cache_.back());
549 ++*num_partitions;
550 }
551
552 *partition_params_array = partitioning_preview_cache_.data();
553 return kTfLiteOk;
554}

600TfLiteStatus Subgraph::CheckTensorIndices(const char* label, const int* indices,
601 int length) {
602 // Making sure kTfLiteOptionalTensor is not re-defined to something other than
603 // -1.
604 static_assert(kTfLiteOptionalTensor == -1,
605 "kTfLiteOptionalTensor should be defined -1");
606
607 for (int i = 0; i < length; i++) {
608 int index = indices[i];
609 // Continue if index == kTfLiteOptionalTensor before additional comparisons
610 // below, size_t(-1) is always >= context_tensors_size.
611 if (index == kTfLiteOptionalTensor) {
612 continue;
613 }
614 if (index < 0 || static_cast<size_t>(index) >= context_.tensors_size) {
615 ReportError(
616 "Invalid tensor index %d in %s. The subgraph has %d tensors\n", index,
617 label, context_.tensors_size);
618 consistent_ = false;
619 return kTfLiteError;
620 }
621 }
622 return kTfLiteOk;
623}

625// We have two arrays and we need to check that elements from one array don't
626// show up in the other. We could sort both arrays and then iterate with two
627// pointers from start to finish always increasing the smaller one but since
628// these arrays are usually short (<25 elements for inputs, usually <3 for
629// outputs), this might be slower than the naive approach (if arrays have size n
630// and m, with n >> m ~ O(1), first approach is O(nlogn) whereas the other is
631// O(n)). Plus, sorting the input and output arrays might not be something we
632// want as it destroys ordering of elements.
633//
634// If it turns out that this is an issue, we can switch to the other algorithm.
635TfLiteStatus Subgraph::CheckInputAndOutputForOverlap(const int* input_indices,
636 int num_inputs,
637 const int* output_indices,
638 int num_outputs) {
639 for (int i = 0; i < num_inputs; i++) {
640 for (int j = 0; j < num_outputs; j++) {
641 if (input_indices[i] == output_indices[j]) {
642 ReportError("Tensor %d is both input %d and output %d\n",
643 input_indices[i], i, j);
644 consistent_ = false;
645 return kTfLiteError;
646 }
647 }

653// Multiply two sizes and return true if overflow occurred;
654// This is based off tensorflow/overflow.h but is simpler as we already
655// have unsigned numbers. It is also generalized to work where sizeof(size_t)
656// is not 8.
657TfLiteStatus MultiplyAndCheckOverflow(size_t a, size_t b, size_t* product) {
658 // Multiplying a * b where a and b are size_t cannot result in overflow in a
659 // size_t accumulator if both numbers have no non-zero bits in their upper
660 // half.
661 constexpr size_t size_t_bits = 8 * sizeof(size_t);
662 constexpr size_t overflow_upper_half_bit_position = size_t_bits / 2;
663 *product = a * b;
664 // If neither integers have non-zero bits past 32 bits can't overflow.
665 // Otherwise check using slow devision.
666 if (TFLITE_EXPECT_FALSE((a | b) >> overflow_upper_half_bit_position != 0)) {
667 if (a != 0 && *product / a != b) return kTfLiteError;
668 }
669 return kTfLiteOk;
670}

673TfLiteStatus Subgraph::BytesRequired(TfLiteType type, const int* dims,
674 size_t dims_size, size_t* bytes) {
675 TF_LITE_ENSURE(&context_, bytes != nullptr);
676 size_t count = 1;
677 for (int k = 0; k < dims_size; k++) {
678 size_t old_count = count;
679 TF_LITE_ENSURE_MSG(
680 &context_,
681 MultiplyAndCheckOverflow(old_count, dims[k], &count) == kTfLiteOk,
682 "BytesRequired number of elements overflowed.\n");
683 }
684 size_t type_size = 0;
685 TF_LITE_ENSURE_OK(&context_, GetSizeOfType(&context_, type, &type_size));
686 TF_LITE_ENSURE_MSG(
687 &context_, MultiplyAndCheckOverflow(type_size, count, bytes) == kTfLiteOk,
688 "BytesRequired number of bytes overflowed.\n");
689 return kTfLiteOk;

692TfLiteStatus Subgraph::AllocateTensors() {
693 TFLITE_SCOPED_TAGGED_DEFAULT_PROFILE(profiler_.get(), "AllocateTensors");
694
695 if (!consistent_) {
696 ReportError("AllocateTensors() called on inconsistent model.");
697 return kTfLiteError;
698 }
699
700 // Restore delegation state if applicable.
701 TF_LITE_ENSURE_STATUS(RedoAllDelegates());
702
703 // Explicit (re)allocation is necessary if nodes have been changed or tensors
704 // have been resized. For inputs marked as dynamic, we can't short-circuit the
705 // allocation as the client may have done the resize manually.
706 if (state_ != kStateUninvokable &&
707 !HasDynamicTensorImpl(context_, inputs())) {
708 if (memory_planner_ && !memory_planner_->HasNonPersistentMemory()) {
709 // If the only change was the release of non-persistent memory via
710 // ReleaseNonPersistentMemory(), just re-allocate it. For any other type
711 // of memory-planning change (for eg, ResizeInputTensor), the state would
712 // be kStateUninvokable.
713 memory_planner_->AcquireNonPersistentMemory();
714 }
715 return kTfLiteOk;
716 }
717
718 // Note `AllocateTensors` sometimes calls itself recursively above
719 // for delegates. Therefore only the logic below need to be guarded
720 // by `SubgraphGuard`.
721 SubgraphGuard guard(&context_, &is_subgraph_in_use_);
722 TF_LITE_ENSURE_OK(&context_, guard.status());
723
724 next_execution_plan_index_to_prepare_ = 0;
725 next_execution_plan_index_to_plan_allocation_ = 0;
726 next_original_execution_plan_index_to_prepare_ = 0;
727 if (memory_planner_) {
728 TF_LITE_ENSURE_STATUS(memory_planner_->ResetAllocations());
729 }
730
731 TF_LITE_ENSURE_STATUS(PrepareOpsAndTensors());
732
733 state_ = kStateInvokable;
734
735 // Reset the variable tensors to zero after (re)allocating the tensors.
736 // Developers shouldn't rely on the side effect of this function to reset
737 // variable tensors. They should call `ResetVariableTensors` directly
738 // instead.
739 ResetVariableTensors();
740
741 return kTfLiteOk;
742}

744// TODO(ycling): Support non-zero default values.
745TfLiteStatus Subgraph::ResetVariableTensors() {
746 for (auto& tensor : tensors_) {
747 if (!tensor.is_variable) {
748 continue;
749 }
750
751 if (tensor.allocation_type == kTfLiteArenaRwPersistent) {
752 // If variable tensors allocation type is `kTfLiteArenaRwPersistent`, then
753 // they must be allocated after the initial `PrepareOpsAndTensors()` is
754 // called.
755 TF_LITE_ENSURE(&context_, tensor.data.raw != nullptr);
756 tflite::ResetVariableTensor(&tensor);
757 } else {
758 // If variable tensors allocation type is not `kTfLiteArenaRwPersistent`,
759 // then it can only be `kTfLiteCustom` in which case, we do not reset it.
760 TF_LITE_ENSURE_EQ(&context_, tensor.allocation_type, kTfLiteCustom);
761 }
762 }
763 return kTfLiteOk;
764}

766TfLiteStatus Subgraph::AddNodeWithParameters(
767 const std::vector<int>& inputs, const std::vector<int>& outputs,
768 const std::vector<int>& intermediates, const char* init_data,
769 size_t init_data_size, void* builtin_data,
770 const TfLiteRegistration* registration, int* node_index) {
771 std::unique_ptr<void, decltype(free)*> builtin_data_deleter(builtin_data,
772 free);
773 if (state_ == kStateInvokableAndImmutable) {
774 ReportError("AddNodeWithParameters is disallowed when graph is immutable.");
775 return kTfLiteError;
776 }
777 state_ = kStateUninvokable;
778
779 TF_LITE_ENSURE_OK(&context_, CheckTensorIndices("node inputs", inputs.data(),
780 inputs.size()));
781 TF_LITE_ENSURE_OK(
782 &context_,
783 CheckTensorIndices("node outputs", outputs.data(), outputs.size()));
784
785 // For builtin ops, inputs and outputs must not overlap. Custom ops must do
786 // this check by themselves if they don't support overlapping tensors. This
787 // distinction is to allow custom ops to just forward a tensor, reusing it as
788 // both input and output.
789 if (builtin_data != nullptr) {
790 TF_LITE_ENSURE_OK(&context_, CheckInputAndOutputForOverlap(
791 inputs.data(), inputs.size(),
792 outputs.data(), outputs.size()));
793 }
794
795 int new_node_index = nodes_and_registration_.size();
796 if (node_index) *node_index = new_node_index;
797 nodes_and_registration_.resize(nodes_and_registration_.size() + 1);
798 auto& node_and_reg = nodes_and_registration_.back();
799 TfLiteNode& node = node_and_reg.first;
800 if (node.inputs) TfLiteIntArrayFree(node.inputs);
801 if (node.outputs) TfLiteIntArrayFree(node.outputs);
802 if (node.intermediates) TfLiteIntArrayFree(node.intermediates);
803 if (node.temporaries) TfLiteIntArrayFree(node.temporaries);
804
805 // NOTE, here we are not using move semantics yet, since our internal
806 // representation isn't std::vector, but in the future we would like to avoid
807 // copies, so we want the interface to take r-value references now.
808 node.inputs = ConvertVectorToTfLiteIntArray(inputs);
809 node.outputs = ConvertVectorToTfLiteIntArray(outputs);
810 node.intermediates = ConvertVectorToTfLiteIntArray(intermediates);
811 node.temporaries = TfLiteIntArrayCreate(0);
812 if (init_data) {
813 node.user_data = OpInit(*registration, init_data, init_data_size);
814 } else {
815 node.user_data = OpInit(
816 *registration, static_cast<const char*>(builtin_data_deleter.get()), 0);
817 }
818
819 node.builtin_data = builtin_data_deleter.release();
820 // TODO(ycling): Filling `custom_initial_data` and `custom_initial_data_size`
821 // properly for nodes generated by ReplaceNodeSubsetsWithDelegateKernels.
822
823 if (registration->builtin_code == BuiltinOperator_CUSTOM) {
824 // When it's a CUSTOM op, the `custom_options` field in the Flatbuffer
825 // `Operator` table is passed in.
826 node.custom_initial_data = init_data;
827 node.custom_initial_data_size = init_data_size;
828 } else {
829 node.custom_initial_data = nullptr;
830 node.custom_initial_data_size = 0;
831 }
832
833 node.delegate = nullptr;
834 // Copying of registration is required to support unresolved custom ops.
835 node_and_reg.second = *registration;
836 execution_plan_.push_back(new_node_index);
837 return kTfLiteOk;
838}

840TfLiteStatus Subgraph::ResizeInputTensor(int tensor_index,
841 const std::vector<int>& dims) {
842 const bool delegates_applied = !pre_delegation_execution_plan_.empty();
843 const bool graph_is_immutable = state_ == kStateInvokableAndImmutable;
844 if (graph_is_immutable && !delegates_applied) {
845 ReportError("ResizeInputTensor is disallowed when graph is immutable.");
846 return kTfLiteError;
847 }
848
849 // TODO(aselle): All bounds checks can be implemented as one-sided bounds
850 // checks by casting to unsigned for efficiency. Profile before doing this.
851 TF_LITE_ENSURE(&context_,
852 tensor_index < context_.tensors_size && tensor_index >= 0);
853 TfLiteTensor* tensor = &context_.tensors[tensor_index];
854
855 // Short-circuit the state change if the dimensions don't change, avoiding
856 // unnecessary (re)allocations.
857 //
858 // Note that it's required to check `tensor->data.raw != nullptr`. Otherwise
859 // the subgraph won't allocate memory for a dynamic tensor when its size
860 // is equal to the original tensor size.
861 if (tensor->data.raw != nullptr &&
862 EqualArrayAndTfLiteIntArray(tensor->dims, dims.size(), dims.data())) {
863 return kTfLiteOk;
864 }
865
866 if (graph_is_immutable) {
867 // Undo delegation if it resulted in the graph being immutable.
868 TF_LITE_ENSURE_STATUS(UndoAllDelegates());
869 }
870 state_ = kStateUninvokable;
871 return ResizeTensorImpl(tensor, ConvertVectorToTfLiteIntArray(dims));
872}

874TfLiteStatus Subgraph::ResizeInputTensorStrict(int tensor_index,
875 const std::vector<int>& dims) {
876 TF_LITE_ENSURE(&context_,
877 tensor_index < context_.tensors_size && tensor_index >= 0);
878 TfLiteTensor* tensor = &context_.tensors[tensor_index];
879
880 // Ensure that only unknown dimensions can be resized.
881 TF_LITE_ENSURE_EQ(&context_, tensor->dims->size, dims.size());
882 for (size_t idx = 0; idx < dims.size(); idx++) {
883 // `dims_signature` is not defined when no unknown dimensions are present.
884 int dim_signature;
885 if (tensor->dims_signature && tensor->dims_signature->size) {
886 dim_signature = tensor->dims_signature->data[idx];
887 } else {
888 dim_signature = tensor->dims->data[idx];
889 }
890
891 if (dim_signature != -1 && dim_signature != dims[idx]) {
892 ReportError(
893 "Attempting to resize dimension %d of tensor %d with value %d to %d. "
894 "ResizeInputTensorStrict only allows mutating unknown dimensions "
895 "identified by -1.",
896 idx, tensor_index, dim_signature, dims[idx]);
897 return kTfLiteError;
898 }
899 }
900
901 return ResizeInputTensor(tensor_index, dims);
902}

932TfLiteStatus Subgraph::PrepareOpsStartingAt(
933 int first_execution_plan_index, const std::vector<int>& execution_plan,
934 int* last_execution_plan_index_prepared) {
935 if (first_execution_plan_index == 0) {
936 // Forwarding inputs without modification won't be not evaluated in the
937 // operators. So, it needs to look up the subgraph's output tensors at the
938 // beginning.
939 has_dynamic_tensors_ = HasDynamicTensorImpl(context_, outputs());
940 }
941 for (int execution_plan_index = first_execution_plan_index;
942 execution_plan_index < execution_plan.size(); execution_plan_index++) {
943 int node_index = execution_plan[execution_plan_index];
944 TfLiteNode& node = nodes_and_registration_[node_index].first;
945 const TfLiteRegistration& registration =
946 nodes_and_registration_[node_index].second;
947 EnsureTensorsVectorCapacity();
948 if (OpPrepare(registration, &node) != kTfLiteOk) {
949 return ReportOpError(&context_, node, registration, node_index,
950 "failed to prepare");
951 }
952
953 *last_execution_plan_index_prepared = execution_plan_index;
954
955 // Discontinue if the node has dynamic outputs. Note that we don't
956 // stop for dynamic temporary tensors since they won't affect the
957 // sizes of other tensors in the graph.
958 if (HasDynamicTensor(context_, node.outputs)) {
959 has_dynamic_tensors_ = true;
960 return kTfLiteOk;
961 }
962 }
963 return kTfLiteOk;

966TfLiteStatus Subgraph::PrepareOpsAndTensors() {
967 if (!memory_planner_) {
968 memory_planner_.reset(new ArenaPlanner(
969 &context_, std::unique_ptr<GraphInfo>(new InterpreterInfo(this)),
970 /*preserve_inputs=*/true, preserve_all_tensors_,
971 kDefaultTensorAlignment));
972 memory_planner_->PlanAllocations();
973 }
974
975 // Prepare original execution plan if any applied delegate wants it.
976 // If any of the delegates is immutable, this won't be triggered
977 // post-delegation (since we undo/redo delegation). For all other cases, other
978 // delegates that do shape propagation themselves would still be able to.
979 bool prepare_original_plan = false;
980 if (!pre_delegation_execution_plan_.empty()) {
981 for (int i = 0; i < delegates_applied_.size(); ++i) {
982 if ((delegates_applied_[i]->flags &
983 kTfLiteDelegateFlagsRequirePropagatedShapes)) {
984 prepare_original_plan = true;
985 break;
986 }
987 }
988 }
989 if (prepare_original_plan) {
990 int last_original_exec_plan_index_prepared = 0;
991 TF_LITE_ENSURE_STATUS(PrepareOpsStartingAt(
992 next_execution_plan_index_to_prepare_, pre_delegation_execution_plan_,
993 &last_original_exec_plan_index_prepared));
994 next_original_execution_plan_index_to_prepare_ =
995 last_original_exec_plan_index_prepared + 1;
996 }
997
998 int last_exec_plan_index_prepared = 0;
999 TF_LITE_ENSURE_STATUS(
1000 PrepareOpsStartingAt(next_execution_plan_index_to_prepare_,
1001 execution_plan_, &last_exec_plan_index_prepared));
1002 next_execution_plan_index_to_prepare_ = last_exec_plan_index_prepared + 1;
1003
1004 // Execute arena allocations.
1005 TF_LITE_ENSURE_STATUS(memory_planner_->ExecuteAllocations(
1006 next_execution_plan_index_to_plan_allocation_,
1007 last_exec_plan_index_prepared));
1008
1009 // Ensure custom allocations are large enough for applicable tensors.
1010 // This causes some extra validations for cases with dynamic tensors, but the
1011 // overhead should be minimal since the number of custom-allocated tensors
1012 // will typically be low.
1013 for (int i = 0; i < custom_allocations_.size(); ++i) {
1014 auto index_and_alloc = custom_allocations_[i];
1015 TfLiteTensor* tensor_at_index = tensor(index_and_alloc.first);
1016 const auto& alloc = index_and_alloc.second;
1017 TF_LITE_ENSURE_EQ(context(), tensor_at_index->allocation_type,
1018 kTfLiteCustom);
1019 if (alloc.bytes < tensor_at_index->bytes) {
1020 ReportError("Custom allocation is too small for tensor idx: %d",
1021 index_and_alloc.first);
1022 return kTfLiteError;
1023 }
1024 }
1025
1026 next_execution_plan_index_to_plan_allocation_ =
1027 last_exec_plan_index_prepared + 1;
1028
1029 return kTfLiteOk;
1030}

1032TfLiteStatus Subgraph::Invoke() {
1033 SubgraphGuard guard(&context_, &is_subgraph_in_use_);
1034 TF_LITE_ENSURE_OK(&context_, guard.status());
1035
1036 if (!consistent_) {
1037 ReportError("Invoke called on model that is not consistent.");
1038 return kTfLiteError;
1039 }
1040
1041 TfLiteStatus status = kTfLiteOk;
1042 if (state_ == kStateUninvokable) {
1043 ReportError("Invoke called on model that is not ready.");
1044 return kTfLiteError;
1045 } else if (memory_planner_ && !memory_planner_->HasNonPersistentMemory()) {
1046 ReportError("Non-persistent memory is not available.");
1047 return kTfLiteError;
1048 }
1049 TFLITE_SCOPED_TAGGED_DEFAULT_PROFILE(profiler_.get(), "Invoke");
1050
1051 // Invocations are always done in node order.
1052 // Note that calling Invoke repeatedly will cause the original memory plan to
1053 // be reused, unless either ResizeInputTensor() or AllocateTensors() has been
1054 // called.
1055 for (int execution_plan_index = 0;
1056 execution_plan_index < execution_plan_.size(); execution_plan_index++) {
1057 if (execution_plan_index == next_execution_plan_index_to_prepare_) {
1058 TF_LITE_ENSURE_STATUS(PrepareOpsAndTensors());
1059 TF_LITE_ENSURE(&context_, next_execution_plan_index_to_prepare_ >=
1060 execution_plan_index);
1061 }
1062 int node_index = execution_plan_[execution_plan_index];
1063 TfLiteNode& node = nodes_and_registration_[node_index].first;
1064 const TfLiteRegistration& registration =
1065 nodes_and_registration_[node_index].second;
1066
1067 const char* op_name = nullptr;
1068 if (profiler_) op_name = GetTFLiteOpName(registration);
1069 TFLITE_SCOPED_TAGGED_OPERATOR_PROFILE(profiler_.get(), op_name, node_index);
1070
1071 // TODO(ycling): This is an extra loop through inputs to check if the data
1072 // need to be copied from Delegate buffer to raw memory, which is often not
1073 // needed. We may want to cache this in prepare to know if this needs to be
1074 // done for a node or not.
1075 for (int i = 0; i < node.inputs->size; ++i) {
1076 int tensor_index = node.inputs->data[i];
1077 if (tensor_index == kTfLiteOptionalTensor) {
1078 continue;
1079 }
1080 TfLiteTensor* tensor = &tensors_[tensor_index];
1081 if (tensor->delegate && tensor->delegate != node.delegate &&
1082 tensor->data_is_stale) {
1083 TF_LITE_ENSURE_STATUS(EnsureTensorDataIsReadable(tensor_index));
1084 }
1085 if (tensor->data.raw == nullptr && tensor->bytes > 0) {
1086 if (registration.builtin_code == kTfLiteBuiltinReshape && i == 1 &&
1087 tensor->dims->size != 1) {
1088 // In general, having a tensor here with no buffer will be an error.
1089 // However, for the reshape operator, the second input tensor is
1090 // sometimes only used for the shape, not for the data. Thus, null
1091 // buffer is ok in this situation.
1092 // The situation where null buffer is not ok for reshape operator is
1093 // only when there are 2 inputs given to the node and the one
1094 // corresponding to the shape (i == 1) is a vector that contains all
1095 // dimensions. See `GetOutputShape()` function in
1096 // `tensorflow/lite/kernels/reshape.cc`
1097 continue;
1098 } else {
1099 // In all other cases, we need to return an error as otherwise we will
1100 // trigger a null pointer dereference (likely).
1101 ReportError("Input tensor %d lacks data", tensor_index);
1102 return kTfLiteError;
1103 }
1104 }
1105 }
1106
1107 if (check_cancelled_func_ != nullptr &&
1108 check_cancelled_func_(cancellation_data_)) {
1109 ReportError("Client requested cancel during Invoke()");
1110 return kTfLiteError;
1111 }
1112
1113 EnsureTensorsVectorCapacity();
1114 tensor_resized_since_op_invoke_ = false;
1115 if (OpInvoke(registration, &node) != kTfLiteOk) {
1116 return ReportOpError(&context_, node, registration, node_index,
1117 "failed to invoke");
1118 }
1119
1120 // Force execution prep for downstream ops if the latest op triggered the
1121 // resize of a dynamic tensor.
1122 if (tensor_resized_since_op_invoke_ &&
1123 HasDynamicTensor(context_, node.outputs)) {
1124 next_execution_plan_index_to_prepare_ = execution_plan_index + 1;
1125
1126 // This happens when an intermediate dynamic tensor is resized.
1127 // We don't have to prepare all the ops, but we need to recompute
1128 // the allocation plan.
1129 if (next_execution_plan_index_to_plan_allocation_ >
1130 next_execution_plan_index_to_prepare_) {
1131 next_execution_plan_index_to_plan_allocation_ =
1132 next_execution_plan_index_to_prepare_;
1133 if (memory_planner_) {
1134 TF_LITE_ENSURE_STATUS(memory_planner_->ResetAllocationsAfter(
1135 next_execution_plan_index_to_plan_allocation_ - 1));
1136 }
1137 }
1138 }
1139 }
1140
1141 return status;
1142}

1144TfLiteStatus Subgraph::ResizeTensor(TfLiteContext* context,
1145 TfLiteTensor* tensor,
1146 TfLiteIntArray* new_size) {
1147 // If the dimensions don't change, avoiding
1148 // unnecessary (re)allocations.
1149 //
1150 // Note that it's required to check `tensor->data.raw != nullptr`. Otherwise
1151 // the subgraph won't allocate memory for a dynamic tensor when its size
1152 // is equal to the original tensor size.
1153 if (tensor->data.raw != nullptr &&
1154 EqualArrayAndTfLiteIntArray(tensor->dims, new_size->size,
1155 new_size->data)) {
1156 // A number of clients assume |new_size| remains valid upon success, so
1157 // swap it in as the new (but logically identical) tensor dims.
1158 TfLiteIntArrayFree(tensor->dims);
1159 tensor->dims = new_size;
1160 return kTfLiteOk;
1161 }

1163 // Note here that context->impl_ is recovering the this pointer for an
1164 // instance of Interpreter to call into the member function ResizeTensorImpl
1165 // (this function is static).
1166 return static_cast<Subgraph*>(context->impl_)
1167 ->ResizeTensorImpl(tensor, new_size);
1168}

1240 int tensor_index, TfLiteType type, const char* name, const size_t rank,
1241 const int* dims, TfLiteQuantization quantization, const char* buffer,
1242 size_t bytes, const Allocation* allocation, TfLiteSparsity* sparsity) {
1243 // Ensure quantization cleanup on failure.
1244 ScopedTfLiteQuantization scoped_quantization(&quantization);
1245 ScopedTfLiteSparsity scoped_sparsity(sparsity);
1246 if (state_ == kStateInvokableAndImmutable) {
1247 ReportError(
1248 "SetTensorParametersReadOnly is disallowed when graph is immutable.");
1249 return kTfLiteError;
1250 }
1251
1252 TF_LITE_ENSURE(&context_,
1253 tensor_index < context_.tensors_size && tensor_index >= 0);
1254
1255 // For most tensors we know exactly how much memory is necessary so we can
1256 // ensure the buffer is large enough. However, we need to skip string tensors
1257 // and sparse tensors because their sizes change with the contents.
1258 // TODO(b/145615516): Extend BytesRequired to check sparse tensors.
1259 if (type != kTfLiteString && type != kTfLiteResource &&
1260 type != kTfLiteVariant && sparsity == nullptr) {
1261 size_t required_bytes;
1262 TF_LITE_ENSURE_OK(&context_,
1263 BytesRequired(type, dims, rank, &required_bytes));
1264 TF_LITE_ENSURE_EQ(&context_, required_bytes, bytes);
1265 }
1266
1267 TfLiteTensor& tensor = context_.tensors[tensor_index];
1268 if (type == tensor.type &&
1269 EqualArrayAndTfLiteIntArray(tensor.dims, rank, dims)) {
1270 // Fast path which does not invalidate the invokable property.
1271 TfLiteTensorDataFree(&tensor);
1272 TfLiteQuantizationFree(&tensor.quantization);
1273 tensor.data.raw = const_cast<char*>(buffer);
1274 if (!tensor.dims) tensor.dims = ConvertArrayToTfLiteIntArray(rank, dims);
1275 tensor.params = GetLegacyQuantization(quantization);
1276 tensor.quantization = *scoped_quantization.release();
1277 tensor.sparsity = scoped_sparsity.release();
1278 tensor.allocation_type = kTfLiteMmapRo;
1279 tensor.allocation = allocation;
1280 } else {
1281 state_ = kStateUninvokable;
1282 TfLiteTensorReset(type, name, ConvertArrayToTfLiteIntArray(rank, dims),
1283 GetLegacyQuantization(quantization),
1284 const_cast<char*>(buffer), bytes, kTfLiteMmapRo,
1285 allocation, false, &tensor);
1286 // TODO(suharshs): Update TfLiteTensorReset to include the new quantization
1287 // if there are other required callers.
1288 tensor.quantization = *scoped_quantization.release();
1289 tensor.sparsity = scoped_sparsity.release();
1290 }
1291 return kTfLiteOk;
1292}

1296// bytes. The lifetime of buffer must be ensured to be greater or equal
1297// to Interpreter.
1298TfLiteStatus Subgraph::SetTensorParametersReadWrite(
1299 int tensor_index, TfLiteType type, const char* name, const size_t rank,
1300 const int* dims, TfLiteQuantization quantization, bool is_variable,
1301 const size_t rank_dims_signature, const int* dims_signature) {
1302 // Ensure quantization cleanup on failure.
1303 ScopedTfLiteQuantization scoped_quantization(&quantization);
1304 if (state_ == kStateInvokableAndImmutable) {
1305 ReportError(
1306 "SetTensorParametersReadWrite is disallowed when graph is immutable.");
1307 return kTfLiteError;
1308 }
1309 TF_LITE_ENSURE(&context_,
1310 tensor_index < context_.tensors_size && tensor_index >= 0);
1311 size_t required_bytes = 0;
1312 if (type != kTfLiteString && type != kTfLiteResource &&
1313 type != kTfLiteVariant) {
1314 // These types will be allocated in our arena so we need to record how
1315 // many bytes we will need based on the dimensions. String tensors are
1316 // allocated dynamically and we can't know ahead of time how much space
1317 // they will require.
1318 TF_LITE_ENSURE_OK(&context_,
1319 BytesRequired(type, dims, rank, &required_bytes));
1320 }
1321
1322 TfLiteAllocationType allocation_type = kTfLiteArenaRw;
1323 if (type == kTfLiteString || type == kTfLiteResource ||
1324 type == kTfLiteVariant) {
1325 if (is_variable) {
1326 // We don't have a real use case for string variable tensor.
1327 ReportError("String variable tensor isn't supported.");
1328 return kTfLiteError;
1329 }
1330 allocation_type = kTfLiteDynamic;
1331 } else if (is_variable) {
1332 allocation_type = kTfLiteArenaRwPersistent;
1333 }
1334
1335 TfLiteTensor& tensor = context_.tensors[tensor_index];
1336 TfLiteTensorReset(type, name, ConvertArrayToTfLiteIntArray(rank, dims),
1337 GetLegacyQuantization(quantization),
1338 /*buffer=*/nullptr, required_bytes, allocation_type,
1339 nullptr, is_variable, &tensor);
1340 // TODO(suharshs): Update TfLiteTensorReset to include the new quantization
1341 // if there are other required callers.
1342 tensor.quantization = *scoped_quantization.release();
1343 tensor.dims_signature =
1344 ConvertArrayToTfLiteIntArray(rank_dims_signature, dims_signature);
1345 return kTfLiteOk;
1346}

1357TfLiteStatus Subgraph::ResizeTensorImpl(TfLiteTensor* tensor,
1358 TfLiteIntArray* new_size) {
1359 // Note that in theory we could resize kTfLiteArenaRwPersistent tensors too.
1360 if (tensor->allocation_type == kTfLiteArenaRw ||
1361 tensor->allocation_type == kTfLiteDynamic ||
1362 tensor->allocation_type == kTfLiteArenaRwPersistent ||
1363 tensor->allocation_type == kTfLitePersistentRo ||
1364 tensor->allocation_type == kTfLiteCustom) {
1365 tensor_resized_since_op_invoke_ |=
1366 TfLiteIntArrayEqual(tensor->dims, new_size) == 0;
1367 if (tensor->type != kTfLiteString && tensor->type != kTfLiteResource &&
1368 tensor->type != kTfLiteVariant) {
1369 size_t bytesRequired;
1370 TfLiteStatus status = BytesRequired(tensor->type, new_size->data,
1371 new_size->size, &bytesRequired);
1372 if (status != kTfLiteOk) {
1373 TfLiteIntArrayFree(new_size);
1374 return kTfLiteError;
1375 }
1376
1377 // Realloc space for heap-allocated tensors.
1378 TfLiteTensorRealloc(bytesRequired, tensor);
1379 tensor->bytes = bytesRequired;
1380 }
1381 if (tensor->dims) TfLiteIntArrayFree(tensor->dims);
1382 tensor->dims = new_size;
1383
1384 // Reset arena-allocated tensors; they will be allocated later.
1385 if (tensor->allocation_type == kTfLiteArenaRw ||
1386 tensor->allocation_type == kTfLiteArenaRwPersistent) {
1387 tensor->data.raw = nullptr;
1388 }
1389 } else {
1390 // kTfLiteMmapRo tensors are stored in the flatbuffer and are therefore
1391 // of fixed size.
1392 TfLiteIntArrayFree(new_size);
1393 ReportError("Attempting to resize a fixed-size tensor.");
1394 return kTfLiteError;
1395 }
1396 return kTfLiteOk;

1431TfLiteStatus Subgraph::UndoAllDelegates() {
1432 // Return early if there is nothing to reset to.
1433 if (pre_delegation_execution_plan_.empty()) return kTfLiteOk;
1434
1435 // First free all delegate nodes.
1436 for (int execution_plan_index = 0;
1437 execution_plan_index < execution_plan_.size(); ++execution_plan_index) {
1438 int node_index = execution_plan_[execution_plan_index];
1439 TfLiteNode& node = nodes_and_registration_[node_index].first;
1440 if (node.delegate == nullptr) {
1441 continue;
1442 }
1443 CleanupNode(node_index);
1444 }
1445
1446 // Reset execution plan.
1447 execution_plan_ = pre_delegation_execution_plan_;
1448 pre_delegation_execution_plan_.clear();
1449
1450 // Handling FP16 delegation (if applies).
1451 //
1452 // First pass through execution plan to remember mapping of FP16
1453 // dequantizations in the graph.
1454 // This is required because delegates that support FP16 could remap supported
1455 // nodes' inputs to point to their fp16 versions (if delegate supports fp16
1456 // acceleration). This remapping is performed in FP16GraphPartitionHelper in
1457 // delegates/utils. We need to undo this remapping to ensure CPU kernels work.
1458 std::vector<int> fp16_to_fp32(tensors_size(), -1);
1459 for (int execution_plan_index = 0;
1460 execution_plan_index < execution_plan_.size(); ++execution_plan_index) {
1461 int node_index = execution_plan_[execution_plan_index];
1462 auto& node_and_reg = nodes_and_registration_[node_index];
1463 const TfLiteNode& node = node_and_reg.first;
1464 const TfLiteRegistration& reg = node_and_reg.second;
1465 if (reg.builtin_code == kTfLiteBuiltinDequantize &&
1466 node.inputs->size == 1 && node.outputs->size == 1) {
1467 const int input_idx = node.inputs->data[0];
1468 if (tensors_[input_idx].type == kTfLiteFloat16) {
1469 fp16_to_fp32[input_idx] = node.outputs->data[0];
1470 }
1471 }
1472 }
1473 // Second pass through the execution plan to remap applicable nodes' fp16
1474 // inputs to their original fp32 versions. Note that if a CPU kernel does
1475 // support fp16, the model will not contain a DEQUANTIZE for its constant
1476 // input.
1477 for (int execution_plan_index = 0;
1478 execution_plan_index < execution_plan_.size(); ++execution_plan_index) {
1479 int node_index = execution_plan_[execution_plan_index];
1480 auto& node_and_reg = nodes_and_registration_[node_index];
1481 const TfLiteNode& node = node_and_reg.first;
1482 const TfLiteRegistration& reg = node_and_reg.second;
1483 if (reg.builtin_code == kTfLiteBuiltinDequantize) continue;
1484 for (int i = 0; i < node.inputs->size; ++i) {
1485 const int original_input_idx = node.inputs->data[i];
1486 if (tensors_[original_input_idx].type == kTfLiteFloat16) {
1487 node.inputs->data[i] = fp16_to_fp32[original_input_idx];
1488 }
1489 }
1490 }
1491
1492 // Delegate nodes are appended to nodes_and_registration_. Therefore,
1493 // cleanup nodes_and_registration_ to only contain nodes from
1494 // pre_delegation_execution_plan_.
1495 int max_retained_node_index = 0;
1496 for (int execution_plan_index = 0;
1497 execution_plan_index < execution_plan_.size(); ++execution_plan_index) {
1498 max_retained_node_index = std::max(max_retained_node_index,
1499 execution_plan_[execution_plan_index]);
1500 }
1501 nodes_and_registration_.resize(max_retained_node_index + 1);
1502 // After undoing delegates, the graph is uninvokable, but mutable.
1503 state_ = kStateUninvokable;
1504
1505 delegates_undone_ = true;
1506 return kTfLiteOk;
1507}

1509TfLiteStatus Subgraph::RedoAllDelegates() {
1510 if (!delegates_undone_) return kTfLiteOk;
1511
1512 delegates_undone_ = false;
1513 std::vector<TfLiteDelegate*> delegates_to_apply;
1514 delegates_applied_.swap(delegates_to_apply);
1515 for (auto* delegate : delegates_to_apply) {
1516 TF_LITE_ENSURE_STATUS(ModifyGraphWithDelegate(delegate));
1517 }
1518 return kTfLiteOk;
1519}

1531void Subgraph::EnsureTensorsVectorCapacity() {
1532 const size_t required_capacity = tensors_.size() + kTensorsCapacityHeadroom;
1533 if (required_capacity > tensors_.capacity()) {
1534 // Whenever it's required to increase the vector capacity, make it at
1535 // least twice bigger. The behavior is consistent with the default
1536 // behavior of GCC STL's `std::vector::resize()`. This avoids frequently
1537 // allocating and copying the underlying buffer.
1538 size_t reserved_capacity =
1539 std::max(required_capacity, tensors_.capacity() * 2);
1540 tensors_.reserve(reserved_capacity);
1541 context_.tensors = tensors_.data();
1542 }

1555TfLiteStatus Subgraph::ModifyGraphWithDelegate(TfLiteDelegate* delegate) {
1556 TFLITE_SCOPED_TAGGED_DEFAULT_PROFILE(profiler_.get(),
1557 "ModifyGraphWithDelegate");
1558
1559 if (delegate == nullptr) {
1560 ReportError("Null delegate.");
1561 return kTfLiteDelegateError;
1562 }
1563
1564 // Restore delegation state if applicable.
1565 TF_LITE_ENSURE_STATUS(RedoAllDelegates());
1566
1567 if (state_ == kStateInvokableAndImmutable) {
1568 ReportError(
1569 "ModifyGraphWithDelegate is disallowed when graph is immutable.");
1570 return kTfLiteApplicationError;
1571 }
1572
1573 if (!(delegate->flags & kTfLiteDelegateFlagsAllowDynamicTensors)) {
1574 int last_execution_plan_index_prepared;
1575 TF_LITE_ENSURE_OK(
1576 &context_, PrepareOpsStartingAt(0, execution_plan_,
1577 &last_execution_plan_index_prepared));
1578 if (has_dynamic_tensors_) {
1579 // Make sure that we are in a defined ready state before returning.
1580 // Plan and allocate tensors before returning.
1581 TF_LITE_ENSURE_OK(&context_, EnsureMemoryAllocations());
1582 ReportError(
1583 "Attempting to use a delegate that only supports static-sized "
1584 "tensors with a graph that has dynamic-sized tensors.");
1585 return kTfLiteApplicationError;
1586 }
1587 }
1588
1589 const bool was_invokable_before_delegate = state_ == kStateInvokable;
1590 if (delegates_applied_.empty()) {
1591 // This is the first delegate being applied, so remember original execution
1592 // plan.
1593 // TODO(b/119623453): Restore execution plan to this state if delegate
1594 // application fails.
1595 pre_delegation_execution_plan_ = execution_plan_;
1596 }
1597
1598 // TODO(aselle): Consider if it is worth storing pointers to delegates.
1599 // Setup additional context interface.
1600 SwitchToDelegateContext();
1601
1602 auto reset_delegation_if_not_ok = [this](TfLiteStatus status) {
1603 if (status != kTfLiteOk) {
1604 TF_LITE_ENSURE_STATUS(RemoveAllDelegates());
1605 ReportError(
1606 "Restored original execution plan after delegate application "
1607 "failure.");
1608 return kTfLiteDelegateError;
1609 }
1610 return kTfLiteOk;
1611 };
1612
1613 TfLiteStatus status = delegate->Prepare(&context_, delegate);
1614
1615 // Remove additional context info.
1616 SwitchToKernelContext();
1617
1618 TF_LITE_ENSURE_STATUS(reset_delegation_if_not_ok(status));
1619
1620 if (!(delegate->flags & kTfLiteDelegateFlagsAllowDynamicTensors)) {
1621 // Reset the state to force tensor/op reallocation.
1622 state_ = kStateUninvokable;
1623 TF_LITE_ENSURE_STATUS(
1624 reset_delegation_if_not_ok(EnsureMemoryAllocations()));
1625 // After using a delegate which doesn't support dynamic tensors, make the
1626 // entire graph immutable.
1627 state_ = kStateInvokableAndImmutable;
1628 } else if (was_invokable_before_delegate) {
1629 // If the graph was invokable prior to delegate application, flush
1630 // allocation now to leave it in a consistent state.
1631 TF_LITE_ENSURE_STATUS(
1632 reset_delegation_if_not_ok(EnsureMemoryAllocations()));
1633 }
1634 delegates_applied_.push_back(delegate);
1635
1636 return status;
1637}

1639TfLiteStatus Subgraph::SetCustomAllocationForTensor(
1640 int tensor_index, const TfLiteCustomAllocation& allocation, int64_t flags) {
1641 TfLiteTensor* tensor = &context_.tensors[tensor_index];
1642 TF_LITE_ENSURE(context(),
1643 (tensor->allocation_type == kTfLiteArenaRw ||
1644 tensor->allocation_type == kTfLiteArenaRwPersistent ||
1645 tensor->allocation_type == kTfLiteCustom));
1646 // Don't check allocation.bytes here, we do that after all ops are prepared
1647 // to allow tensor shape propagation.
1648 TF_LITE_ENSURE(context(), allocation.data != nullptr);
1649 if (!(flags & kTfLiteCustomAllocationFlagsSkipAlignCheck)) {
1650 const intptr_t data_ptr_value = reinterpret_cast<intptr_t>(allocation.data);
1651 TF_LITE_ENSURE(context(), data_ptr_value % kDefaultTensorAlignment == 0);
1652 }
1653
1654 // If tensor already has a custom alloc, just reassign.
1655 const auto alloc_it = std::find_if(
1656 custom_allocations_.begin(), custom_allocations_.end(),
1657 [tensor_index](
1658 const std::pair<int, TfLiteCustomAllocation>& existing_alloc) {
1659 return existing_alloc.first == tensor_index;
1660 });
1661 if (alloc_it == custom_allocations_.end()) {
1662 custom_allocations_.emplace_back(tensor_index, allocation);
1663 } else {
1664 // If tensor already has a custom alloc, just reassign.
1665 alloc_it->second = allocation;
1666 }
1667
1668 tensor->allocation_type = kTfLiteCustom;
1669 tensor->data.data = allocation.data;
1670
1671 return kTfLiteOk;
1672}
