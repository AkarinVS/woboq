27const TfLiteRegistration* MutableOpResolver::FindOp(tflite::BuiltinOperator op,
28 int version) const {
29 auto it = builtins_.find(std::make_pair(op, version));
30 return it != builtins_.end() ? &it->second : nullptr;
31}
32
33const TfLiteRegistration* MutableOpResolver::FindOp(const char* op,
34 int version) const {
35 auto it = custom_ops_.find(std::make_pair(op, version));
36 return it != custom_ops_.end() ? &it->second : nullptr;
37}
38
39void MutableOpResolver::AddBuiltin(tflite::BuiltinOperator op,
40 const TfLiteRegistration* registration,
41 int version) {
42 TfLiteRegistration new_registration = *registration;
43 new_registration.custom_name = nullptr;
44 new_registration.builtin_code = op;
45 new_registration.version = version;
46 auto op_key = std::make_pair(op, version);
47 builtins_[op_key] = new_registration;
48}
49
50void MutableOpResolver::AddBuiltin(tflite::BuiltinOperator op,
51 const TfLiteRegistration* registration,
52 int min_version, int max_version) {
53 for (int version = min_version; version <= max_version; ++version) {
54 AddBuiltin(op, registration, version);
55 }
56}
57
58void MutableOpResolver::AddCustom(const char* name,
59 const TfLiteRegistration* registration,
60 int version) {
61 TfLiteRegistration new_registration = *registration;
62 new_registration.builtin_code = BuiltinOperator_CUSTOM;
63 new_registration.custom_name = name;
64 new_registration.version = version;
65 auto op_key = std::make_pair(name, version);
66 custom_ops_[op_key] = new_registration;
67}
68
69void MutableOpResolver::AddCustom(const char* name,
70 const TfLiteRegistration* registration,
71 int min_version, int max_version) {
72 for (int version = min_version; version <= max_version; ++version) {
73 AddCustom(name, registration, version);
74 }
75}
76
77void MutableOpResolver::AddAll(const MutableOpResolver& other) {
78 // map::insert does not replace existing elements, and map::insert_or_assign
79 // wasn't added until C++17.
80 for (const auto& other_builtin : other.builtins_) {
81 builtins_[other_builtin.first] = other_builtin.second;
82 }
83 for (const auto& other_custom_op : other.custom_ops_) {
84 custom_ops_[other_custom_op.first] = other_custom_op.second;
85 }
86}
