35// A class to represent the information of a memory arena that's used in TfLite
36// runtime for holding allocated memory of tensors. The information includes
37// the following:
38// 1. The memory allocation type.
39// 2. The tensor id of the tensor that has the most amount of memory allocated,
40// and the memory size.
41// 3. The estimated memory boundary and size of the arena.
42class MemoryArenaInfo {
43 public:
44 explicit MemoryArenaInfo(TfLiteAllocationType type)
45 : allocation_type_(type) {}
46
47 void Update(size_t tensor_index, const TfLiteTensor& tensor) {
48 if (tensor.allocation_type != allocation_type_) return;
49 if (tensor.data.data == nullptr) return;
50 if (tensor.bytes > max_tensor_mem_bytes_) {
51 max_tensor_mem_bytes_ = tensor.bytes;
52 max_tensor_id_ = tensor_index;
53 }
54
55 size_t current_start_addr = reinterpret_cast<size_t>(tensor.data.data);
56
57 size_t current_end_addr = current_start_addr + tensor.bytes;
58 if (current_start_addr < min_tensor_start_addr_) {
59 min_tensor_start_addr_ = current_start_addr;
60 }
61 if (current_end_addr > max_tensor_end_addr_) {
62 max_tensor_end_addr_ = current_end_addr;
63 }
64 }
65
66 void Print() const {
67 printf("%s Info: ", AllocTypeName(allocation_type_));
68 if (max_tensor_end_addr_ == 0) {
69 printf("not holding any allocation.\n");
70 return;
71 }
72 printf("\nTensor %zu has the max size %zu bytes (%.1f MB).\n",
73 max_tensor_id_, max_tensor_mem_bytes_,
74 static_cast<float>(max_tensor_mem_bytes_) / (1 << 20));
75 printf("This memory arena is estimated as[0x%zx, 0x%zx), taking %.1f MB.\n",
76 max_tensor_end_addr_, min_tensor_start_addr_,
77 static_cast<float>(max_tensor_end_addr_ - min_tensor_start_addr_) /
78 (1 << 20));
79 }
80
81 private:
82 TfLiteAllocationType allocation_type_;
83 size_t max_tensor_mem_bytes_ = 0;
84 // the index of the tensor that has the max memory size.
85 size_t max_tensor_id_ = -1;
86 size_t min_tensor_start_addr_ = std::numeric_limits<size_t>::max();
87 size_t max_tensor_end_addr_ = 0;
88};

90void PrintIntVector(const std::vector<int>& v) {
91 if (v.empty()) {
92 printf("(null)\n");
93 return;
94 }
95
96 printf("[");
97 for (int i = 0; i < v.size() - 1; ++i) {
98 printf("%d,", v[i]);
99 }
100 printf("%d]\n", v.back());
101}
102
103void PrintTfLiteIntVector(const TfLiteIntArray* v) {
104 if (!v || v->size <= 0) {
105 printf("(null)\n");
106 return;
107 }
108 printf("[");
109 for (int k = 0; k < v->size - 1; k++) {
110 printf("%d,", v->data[k]);
111 }
112 printf("%d]\n", v->data[v->size - 1]);
113}

115const char* TensorTypeName(TfLiteType type) {
116 switch (type) {
117 case kTfLiteNoType:
118 return "kTfLiteNoType";
119 case kTfLiteFloat32:
120 return "kTfLiteFloat32";
121 case kTfLiteInt32:
122 return "kTfLiteInt32";
123 case kTfLiteUInt32:
124 return "kTfLiteUInt32";
125 case kTfLiteUInt8:
126 return "kTfLiteUInt8";
127 case kTfLiteInt8:
128 return "kTfLiteInt8";
129 case kTfLiteInt64:
130 return "kTfLiteInt64";
131 case kTfLiteUInt64:
132 return "kTfLiteUInt64";
133 case kTfLiteString:
134 return "kTfLiteString";
135 case kTfLiteBool:
136 return "kTfLiteBool";
137 case kTfLiteInt16:
138 return "kTfLiteInt16";
139 case kTfLiteComplex64:
140 return "kTfLiteComplex64";
141 case kTfLiteComplex128:
142 return "kTfLiteComplex128";
143 case kTfLiteFloat16:
144 return "kTfLiteFloat16";
145 case kTfLiteFloat64:
146 return "kTfLiteFloat64";
147 case kTfLiteResource:
148 return "kTfLiteResource";

175std::string TruncateString(const char* str, int size_limit,
176 bool truncate_at_end = false) {
177 if (str == nullptr) return "(nil)";
178
179 std::string truncated(str);
180 const size_t length = truncated.size();
181 if (length <= size_limit) return truncated;
182
183 if (size_limit <= 3) return std::string(size_limit, '.');
184
185 if (truncate_at_end) {
186 truncated.resize(size_limit);
187 // Change the the last 3 chars to "..." to imply truncation.
188 truncated.replace(size_limit - 3, 3, "...");
189 } else {
190 truncated.erase(0, length - size_limit);
191 // Change the the first 3 chars to "..." to imply truncation.
192 truncated.replace(0, 3, "...");
193 }
194 return truncated;
195}

199// Prints a dump of what tensors and what nodes are in the interpreter.
200void PrintInterpreterState(Interpreter* interpreter) {
201 const size_t num_subgraphs = interpreter->subgraphs_size();
202 printf("Interpreter has %zu subgraphs.\n\n", num_subgraphs);
203
204 for (int i = 0; i < num_subgraphs; ++i) {
205 const Subgraph& subgraph = *(interpreter->subgraph(i));
206 printf("-----------Subgraph-%d has %zu tensors and %zu nodes------------\n",
207 i, subgraph.tensors_size(), subgraph.nodes_size());
208 printf("Inputs: ");
209 PrintIntVector(subgraph.inputs());
210 printf("Outputs: ");
211 PrintIntVector(subgraph.outputs());
212 printf("\n");
213
214 printf("Tensor %3s %-25s %-15s %-18s %10s\n", "ID", "Name", "Type",
215 "AllocType", "Size");
216 MemoryArenaInfo rw_info(kTfLiteArenaRw);
217 MemoryArenaInfo rw_persistent_info(kTfLiteArenaRwPersistent);
218 for (size_t tensor_index = 0; tensor_index < subgraph.tensors_size();
219 tensor_index++) {
220 const TfLiteTensor* tensor =
221 subgraph.tensor(static_cast<int>(tensor_index));
222 printf("Tensor %3zu %-25s %-15s %-18s %10zuB (%4.1f MB) ", tensor_index,
223 TruncateString(tensor->name, 25, /*truncate_at_end*/ true).c_str(),
224 TruncateString(TensorTypeName(tensor->type), 15).c_str(),
225 TruncateString(AllocTypeName(tensor->allocation_type), 18).c_str(),
226 tensor->bytes, (static_cast<float>(tensor->bytes) / (1 << 20)));
227 PrintTfLiteIntVector(tensor->dims);
228 rw_info.Update(tensor_index, *tensor);
229 rw_persistent_info.Update(tensor_index, *tensor);
230 }
231 printf("\n");
232 rw_info.Print();
233 printf("\n");
234 rw_persistent_info.Print();
235 printf("\n");
236 for (size_t node_index = 0; node_index < subgraph.nodes_size();
237 node_index++) {
238 const std::pair<TfLiteNode, TfLiteRegistration>* node_and_reg =
239 subgraph.node_and_registration(static_cast<int>(node_index));
240 const TfLiteNode& node = node_and_reg->first;
241 const TfLiteRegistration& reg = node_and_reg->second;
242 if (reg.custom_name != nullptr) {
243 printf("Node %3zu Operator Custom Name %s\n", node_index,
244 reg.custom_name);
245 } else {
246 printf("Node %3zu Operator Builtin Code %3d %s\n", node_index,
247 reg.builtin_code, EnumNamesBuiltinOperator()[reg.builtin_code]);
248 }
249 printf(" Input Tensors:");
250 PrintTfLiteIntVector(node.inputs);
251 printf(" Output Tensors:");
252 PrintTfLiteIntVector(node.outputs);
253 if (node.intermediates && node.intermediates->size) {
254 printf(" Intermediate Tensors:");
255 PrintTfLiteIntVector(node.intermediates);
256 }
257 if (node.temporaries && node.temporaries->size) {
258 printf(" Temporary Tensors:");
259 PrintTfLiteIntVector(node.temporaries);
260 }
261 }
262 printf("--------------Subgraph-%d dump has completed--------------\n\n", i);
263 }
264}
