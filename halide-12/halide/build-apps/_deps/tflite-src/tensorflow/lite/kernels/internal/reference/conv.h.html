25inline void Conv(const ConvParams& params, const RuntimeShape& input_shape,
26 const float* input_data, const RuntimeShape& filter_shape,
27 const float* filter_data, const RuntimeShape& bias_shape,
28 const float* bias_data, const RuntimeShape& output_shape,
29 float* output_data, const RuntimeShape& im2col_shape,
30 float* im2col_data) {
31 const int stride_width = params.stride_width;
32 const int stride_height = params.stride_height;
33 const int dilation_width_factor = params.dilation_width_factor;
34 const int dilation_height_factor = params.dilation_height_factor;
35 const int pad_width = params.padding_values.width;
36 const int pad_height = params.padding_values.height;
37 const float output_activation_min = params.float_activation_min;
38 const float output_activation_max = params.float_activation_max;
39 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
40 TFLITE_DCHECK_EQ(filter_shape.DimensionsCount(), 4);
41 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
42
43 (void)im2col_data; // only used in optimized code.
44 (void)im2col_shape; // only used in optimized code.
45 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
46 const int input_depth = MatchingDim(input_shape, 3, filter_shape, 3);
47 const int output_depth = MatchingDim(filter_shape, 0, output_shape, 3);
48 if (bias_data) {
49 TFLITE_DCHECK_EQ(bias_shape.FlatSize(), output_depth);
50 }
51 const int input_height = input_shape.Dims(1);
52 const int input_width = input_shape.Dims(2);
53 const int filter_height = filter_shape.Dims(1);
54 const int filter_width = filter_shape.Dims(2);
55 const int output_height = output_shape.Dims(1);
56 const int output_width = output_shape.Dims(2);
57 for (int batch = 0; batch < batches; ++batch) {
58 for (int out_y = 0; out_y < output_height; ++out_y) {
59 const int in_y_origin = (out_y * stride_height) - pad_height;
60 for (int out_x = 0; out_x < output_width; ++out_x) {
61 const int in_x_origin = (out_x * stride_width) - pad_width;
62 for (int out_channel = 0; out_channel < output_depth; ++out_channel) {
63 float total = 0.f;
64 for (int filter_y = 0; filter_y < filter_height; ++filter_y) {
65 const int in_y = in_y_origin + dilation_height_factor * filter_y;
66 for (int filter_x = 0; filter_x < filter_width; ++filter_x) {
67 const int in_x = in_x_origin + dilation_width_factor * filter_x;
68
69 // Zero padding by omitting the areas outside the image.
70 const bool is_point_inside_image =
71 (in_x >= 0) && (in_x < input_width) && (in_y >= 0) &&
72 (in_y < input_height);
73
74 if (!is_point_inside_image) {
75 continue;
76 }
77
78 for (int in_channel = 0; in_channel < input_depth; ++in_channel) {
79 float input_value = input_data[Offset(input_shape, batch, in_y,
80 in_x, in_channel)];
81 float filter_value = filter_data[Offset(
82 filter_shape, out_channel, filter_y, filter_x, in_channel)];
83 total += (input_value * filter_value);
84 }
85 }
86 }
87 float bias_value = 0.0f;
88 if (bias_data) {
89 bias_value = bias_data[out_channel];
90 }
91 output_data[Offset(output_shape, batch, out_y, out_x, out_channel)] =
92 ActivationFunctionWithMinMax(total + bias_value,
93 output_activation_min,
94 output_activation_max);
95 }
96 }
97 }
98 }
99}

101inline void Conv(const ConvParams& params, const RuntimeShape& input_shape,
102 const uint8_t* input_data, const RuntimeShape& filter_shape,
103 const uint8_t* filter_data, const RuntimeShape& bias_shape,
104 const int32_t* bias_data, const RuntimeShape& output_shape,
105 uint8_t* output_data, const RuntimeShape& im2col_shape,
106 uint8_t* im2col_data, void* cpu_backend_context) {
107 (void)cpu_backend_context; // only used in optimized code.
108 (void)im2col_data; // only used in optimized code.
109 (void)im2col_shape; // only used in optimized code.
110 const int stride_width = params.stride_width;
111 const int stride_height = params.stride_height;
112 const int dilation_width_factor = params.dilation_width_factor;
113 const int dilation_height_factor = params.dilation_height_factor;
114 const int pad_width = params.padding_values.width;
115 const int pad_height = params.padding_values.height;
116 const int32_t input_offset = params.input_offset;
117 const int32_t filter_offset = params.weights_offset;
118 const int32_t output_offset = params.output_offset;
119 const int32_t output_multiplier = params.output_multiplier;
120 const int output_shift = params.output_shift;
121 const int32_t output_activation_min = params.quantized_activation_min;
122 const int32_t output_activation_max = params.quantized_activation_max;
123 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
124
125 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
126 TFLITE_DCHECK_EQ(filter_shape.DimensionsCount(), 4);
127 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
128 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
129 const int input_depth = MatchingDim(input_shape, 3, filter_shape, 3);
130 const int output_depth = MatchingDim(filter_shape, 0, output_shape, 3);
131 if (bias_data) {
132 TFLITE_DCHECK_EQ(bias_shape.FlatSize(), output_depth);
133 }
134 const int input_height = input_shape.Dims(1);
135 const int input_width = input_shape.Dims(2);
136 const int filter_height = filter_shape.Dims(1);
137 const int filter_width = filter_shape.Dims(2);
138 const int output_height = output_shape.Dims(1);
139 const int output_width = output_shape.Dims(2);
140 for (int batch = 0; batch < batches; ++batch) {
141 for (int out_y = 0; out_y < output_height; ++out_y) {
142 const int in_y_origin = (out_y * stride_height) - pad_height;
143 for (int out_x = 0; out_x < output_width; ++out_x) {
144 const int in_x_origin = (out_x * stride_width) - pad_width;
145 for (int out_channel = 0; out_channel < output_depth; ++out_channel) {
146 int32_t acc = 0;
147 for (int filter_y = 0; filter_y < filter_height; ++filter_y) {
148 const int in_y = in_y_origin + dilation_height_factor * filter_y;
149 for (int filter_x = 0; filter_x < filter_width; ++filter_x) {
150 const int in_x = in_x_origin + dilation_width_factor * filter_x;
151
152 // Zero padding by omitting the areas outside the image.
153 const bool is_point_inside_image =
154 (in_x >= 0) && (in_x < input_width) && (in_y >= 0) &&
155 (in_y < input_height);
156
157 if (!is_point_inside_image) {
158 continue;
159 }
160
161 for (int in_channel = 0; in_channel < input_depth; ++in_channel) {
162 int32_t input_val = input_data[Offset(input_shape, batch, in_y,
163 in_x, in_channel)];
164 int32_t filter_val = filter_data[Offset(
165 filter_shape, out_channel, filter_y, filter_x, in_channel)];
166 acc +=
167 (filter_val + filter_offset) * (input_val + input_offset);
168 }
169 }
170 }
171 if (bias_data) {
172 acc += bias_data[out_channel];
173 }
174 acc = MultiplyByQuantizedMultiplier(acc, output_multiplier,
175 output_shift);
176 acc += output_offset;
177 acc = std::max(acc, output_activation_min);
178 acc = std::min(acc, output_activation_max);
179 output_data[Offset(output_shape, batch, out_y, out_x, out_channel)] =
180 static_cast<uint8_t>(acc);
181 }
182 }
183 }
184 }

187inline void HybridConvPerChannel(
188 const ConvParams& params, float* scaling_factors_ptr,
189 const RuntimeShape& input_shape, const int8_t* input_data,
190 const RuntimeShape& filter_shape, const int8_t* filter_data,
191 const RuntimeShape& bias_shape, const float* bias_data,
192 const RuntimeShape& output_shape, float* output_data,
193 const RuntimeShape& im2col_shape, int8_t* im2col_data,
194 const float* per_channel_scale, int32_t* input_offset) {
195 (void)im2col_data; // only used in optimized code.
196 (void)im2col_shape; // only used in optimized code.
197 const int stride_width = params.stride_width;
198 const int stride_height = params.stride_height;
199 const int dilation_width_factor = params.dilation_width_factor;
200 const int dilation_height_factor = params.dilation_height_factor;
201 const int pad_width = params.padding_values.width;
202 const int pad_height = params.padding_values.height;
203 const float output_activation_min = params.float_activation_min;
204 const float output_activation_max = params.float_activation_max;
205 TFLITE_DCHECK_EQ(input_shape.DimensionsCount(), 4);
206 TFLITE_DCHECK_EQ(filter_shape.DimensionsCount(), 4);
207 TFLITE_DCHECK_EQ(output_shape.DimensionsCount(), 4);
208 const int batches = MatchingDim(input_shape, 0, output_shape, 0);
209 const int input_depth = MatchingDim(input_shape, 3, filter_shape, 3);
210 const int output_depth = MatchingDim(filter_shape, 0, output_shape, 3);
211 if (bias_data) {
212 TFLITE_DCHECK_EQ(bias_shape.FlatSize(), output_depth);
213 }
214 const int input_height = input_shape.Dims(1);
215 const int input_width = input_shape.Dims(2);
216 const int filter_height = filter_shape.Dims(1);
217 const int filter_width = filter_shape.Dims(2);
218 const int output_height = output_shape.Dims(1);
219 const int output_width = output_shape.Dims(2);
220 for (int batch = 0; batch < batches; ++batch) {
221 for (int out_y = 0; out_y < output_height; ++out_y) {
222 for (int out_x = 0; out_x < output_width; ++out_x) {
223 for (int out_channel = 0; out_channel < output_depth; ++out_channel) {
224 const int in_x_origin = (out_x * stride_width) - pad_width;
225 const int in_y_origin = (out_y * stride_height) - pad_height;
226 int32_t acc = 0;
227 for (int filter_y = 0; filter_y < filter_height; ++filter_y) {
228 for (int filter_x = 0; filter_x < filter_width; ++filter_x) {
229 for (int in_channel = 0; in_channel < input_depth; ++in_channel) {
230 const int in_x = in_x_origin + dilation_width_factor * filter_x;
231 const int in_y =
232 in_y_origin + dilation_height_factor * filter_y;
233 // If the location is outside the bounds of the input image,
234 // use zero as a default value.
235 if ((in_x >= 0) && (in_x < input_width) && (in_y >= 0) &&
236 (in_y < input_height)) {
237 int32_t input_val = input_data[Offset(
238 input_shape, batch, in_y, in_x, in_channel)];
239 int32_t filter_val =
240 filter_data[Offset(filter_shape, out_channel, filter_y,
241 filter_x, in_channel)];
242 acc += filter_val * (input_val - input_offset[batch]);
243 }
244 }
245 }
246 }
247 float acc_float =
248 acc * per_channel_scale[out_channel] * scaling_factors_ptr[batch];
249 if (bias_data) {
250 acc_float += bias_data[out_channel];
251 }
252 output_data[Offset(output_shape, batch, out_y, out_x, out_channel)] =
253 ActivationFunctionWithMinMax(acc_float, output_activation_min,
254 output_activation_max);
255 }
256 }
257 }
