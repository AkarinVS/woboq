30struct OpData {
31 // This implementation uses a random generator from the standard C++ library
32 // on the platform where TFLite is build. This is different from the TF
33 // version of the kernel that uses custom implementations of random
34 // generator, different for different hardware.
35 std::default_random_engine rng;
36};
37
38namespace {
39
40template <typename T, typename dist_type>
41void RandomUniformSample(std::default_random_engine& rng, T* buffer,
42 size_t buffer_size, T min_value, T max_value) {
43 dist_type dist(min_value, max_value);
44 std::generate(buffer, buffer + buffer_size, [&]() { return dist(rng); });
45}

47TfLiteIntArray* CreateDimensionsFromTensor(const TfLiteTensor* tensor) {
48 const int output_dims = tflite::SizeOfDimension(tensor, 0);
49 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(output_dims);
50 for (int i = 0; i < output_dims; i++) {
51 output_shape->data[i] = tensor->data.i32[i];
52 }
53 return output_shape;
54}
55} // namespace
56void* Init(TfLiteContext* context, const char* buffer, size_t length) {
57 return new OpData();
58}
59
60void Free(TfLiteContext* context, void* buffer) {
61 delete reinterpret_cast<OpData*>(buffer);
62}

64TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
65 // TODO(b/169611265): Handle optional seed input.
66 TF_LITE_ENSURE(context, tflite::NumInputs(node) >= 1);
67 TF_LITE_ENSURE_EQ(context, tflite::NumOutputs(node), 1);
68
69 // Input is a shape tensor.
70 const TfLiteTensor* input = tflite::GetInput(context, node, 0);
71 TF_LITE_ENSURE_EQ(context, tflite::NumDimensions(input), 1);
72 TfLiteTensor* output = tflite::GetOutput(context, node, 0);
73 if (!IsConstantTensor(input)) {
74 SetTensorToDynamic(output);
75 return kTfLiteOk;
76 }
77 return context->ResizeTensor(context, output,
78 CreateDimensionsFromTensor(input));
79}

81TfLiteStatus EvalFloat(TfLiteContext* context, TfLiteNode* node) {
82 OpData* params = reinterpret_cast<OpData*>(node->user_data);
83 TF_LITE_ENSURE(context, params != nullptr);
84
85 TfLiteTensor* output = tflite::GetOutput(context, node, 0);
86 if (IsDynamicTensor(output)) {
87 const TfLiteTensor* input = tflite::GetInput(context, node, 0);
88 TF_LITE_ENSURE_OK(context,
89 context->ResizeTensor(context, output,
90 CreateDimensionsFromTensor(input)));
91 }
92 const size_t output_size = tflite::NumElements(output);
93 switch (output->type) {
94 case kTfLiteFloat32:
95 RandomUniformSample<float, std::uniform_real_distribution<float>>(
96 params->rng, GetTensorData<float>(output), output_size, 0.f, 1.f);
97 break;
98 case kTfLiteFloat64:
99 RandomUniformSample<double, std::uniform_real_distribution<double>>(
100 params->rng, GetTensorData<double>(output), output_size, 0.f, 1.f);
101 break;
102 default:
103 TF_LITE_KERNEL_LOG(context,
104 "Unsupported output datatype for RandomUniform: %s",
105 TfLiteTypeGetName(output->type));
106 return kTfLiteError;
107 }
108
109 return kTfLiteOk;
110}

112int64_t IntValueFromTensor(const TfLiteTensor* tensor) {
113 switch (tensor->type) {
114 case kTfLiteInt8:
115 return *GetTensorData<int8_t>(tensor);
116 case kTfLiteInt32:
117 return *GetTensorData<int32_t>(tensor);
118 case kTfLiteInt64:
119 return *GetTensorData<int64_t>(tensor);
120 default:
121 return -1;
122 }
123}
124
125TfLiteStatus EvalInt(TfLiteContext* context, TfLiteNode* node) {
126 OpData* params = reinterpret_cast<OpData*>(node->user_data);
127 TF_LITE_ENSURE(context, params != nullptr);
128
129 TF_LITE_ENSURE(context, tflite::NumInputs(node) >= 3);
130 TfLiteTensor* output = tflite::GetOutput(context, node, 0);
131 if (IsDynamicTensor(output)) {
132 const TfLiteTensor* input = tflite::GetInput(context, node, 0);
133 TF_LITE_ENSURE_OK(context,
134 context->ResizeTensor(context, output,
135 CreateDimensionsFromTensor(input)));
136 }
137 int64_t min_value = IntValueFromTensor(tflite::GetInput(context, node, 1));
138 int64_t max_value = IntValueFromTensor(tflite::GetInput(context, node, 2));
139 TF_LITE_ENSURE(context, min_value < max_value);
140 size_t output_size = tflite::NumElements(output);
141 switch (output->type) {
142 case kTfLiteInt8:
143 RandomUniformSample<int8_t, std::uniform_int_distribution<int32_t>>(
144 params->rng, GetTensorData<int8_t>(output), output_size, min_value,
145 max_value);
146 break;
147 case kTfLiteInt32:
148 RandomUniformSample<int32_t, std::uniform_int_distribution<int32_t>>(
149 params->rng, GetTensorData<int32_t>(output), output_size, min_value,
150 max_value);
151 break;
152 case kTfLiteInt64:
153 RandomUniformSample<int64_t, std::uniform_int_distribution<int64_t>>(
154 params->rng, GetTensorData<int64_t>(output), output_size, min_value,
155 max_value);
156 break;
157 default:
158 TF_LITE_KERNEL_LOG(context,
159 "Unsupported output datatype for RandomUniformInt: %s",
160 TfLiteTypeGetName(output->type));
161 return kTfLiteError;
162 }
163
164 return kTfLiteOk;
165}
