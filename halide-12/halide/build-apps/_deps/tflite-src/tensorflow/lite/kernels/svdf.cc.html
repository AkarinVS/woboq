40struct OpData {
41 int scratch_tensor_index;
42 bool float_weights_time_initialized;
43 int32 effective_scale_1_a;
44 int effective_scale_1_b;
45 int32 effective_scale_2_a;
46 int effective_scale_2_b;
47 bool compute_row_sums = false;
48};
49
50} // namespace
51
52// Input tensors.
53constexpr int kInputTensor = 0;
54constexpr int kWeightsFeatureTensor = 1;
55constexpr int kWeightsTimeTensor = 2;
56constexpr int kBiasTensor = 3;
57// This is a variable tensor, and will be modified by this op.
58constexpr int kStateTensor = 4;
59
60// Output tensor.
61constexpr int kOutputTensor = 0;
62
63void* Init(TfLiteContext* context, const char* buffer, size_t length) {
64 auto* op_data = new OpData();
65 op_data->float_weights_time_initialized = false;
66 // Note: only needs 6 scratch tensors when is_hybrid_op, only 1 otherwise.
67 context->AddTensors(context, /*tensors_to_add=*/6,
68 &op_data->scratch_tensor_index);
69 return op_data;
70}
71
72void Free(TfLiteContext* context, void* buffer) {
73 delete reinterpret_cast<OpData*>(buffer);
74}

76TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
77 const auto* params = reinterpret_cast<TfLiteSVDFParams*>(node->builtin_data);
78 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
79 int scratch_tensor_index = op_data->scratch_tensor_index;
80
81 // Check we have all the inputs and outputs we need.
82 TF_LITE_ENSURE_EQ(context, node->outputs->size, 1);
83 TF_LITE_ENSURE_EQ(context, node->inputs->size, 5);
84
85 const TfLiteTensor* input;
86 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
87 const TfLiteTensor* weights_feature;
88 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kWeightsFeatureTensor,
89 &weights_feature));
90 const TfLiteTensor* weights_time;
91 TF_LITE_ENSURE_OK(
92 context, GetInputSafe(context, node, kWeightsTimeTensor, &weights_time));
93
94 TF_LITE_ENSURE(context,
95 input->type == kTfLiteFloat32 || input->type == kTfLiteInt8);
96
97 // Check all the parameters of tensor match within themselves and match the
98 // input configuration.
99 const int rank = params->rank;
100 const int batch_size = input->dims->data[0];
101 const int num_filters = weights_feature->dims->data[0];
102 TF_LITE_ENSURE(context, rank != 0);
103 TF_LITE_ENSURE_EQ(context, num_filters % rank, 0);
104 const int num_units = num_filters / rank;
105 const int memory_size = weights_time->dims->data[1];
106 TF_LITE_ENSURE_EQ(context, input->dims->data[1],
107 weights_feature->dims->data[1]);
108 TF_LITE_ENSURE_EQ(context, weights_time->dims->data[0], num_filters);
109
110 const TfLiteTensor* bias = GetOptionalInputTensor(context, node, kBiasTensor);
111 if (bias) {
112 TF_LITE_ENSURE_EQ(context, bias->dims->data[0], num_units);
113 }
114
115 const TfLiteTensor* state;
116 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kStateTensor, &state));
117 TfLiteTensor* output;
118 TF_LITE_ENSURE_OK(context,
119 GetOutputSafe(context, node, kOutputTensor, &output));
120
121 // Check the shape of input state tensors.
122 TF_LITE_ENSURE_EQ(context, NumDimensions(state), 2);
123 TF_LITE_ENSURE_EQ(context, SizeOfDimension(state, 0), batch_size);
124 TF_LITE_ENSURE_EQ(context, SizeOfDimension(state, 1),
125 memory_size * num_filters);
126
127 // Resize output.
128 TfLiteIntArray* output_size_array = TfLiteIntArrayCreate(2);
129 output_size_array->data[0] = batch_size;
130 output_size_array->data[1] = num_units;
131 TF_LITE_ENSURE_OK(context,
132 context->ResizeTensor(context, output, output_size_array));
133
134 // The weights are of consistent type, so it suffices to check one.
135 const bool is_hybrid_op = IsHybridOp(input, weights_feature);
136 const bool is_full_integer = input->type == kTfLiteInt8;
137
138 // Resize scratch.
139 TfLiteIntArrayFree(node->temporaries);
140 if (is_hybrid_op) {
141 node->temporaries = TfLiteIntArrayCreate(6);
142 } else if (is_full_integer) {
143 node->temporaries = TfLiteIntArrayCreate(2);
144 } else {
145 node->temporaries = TfLiteIntArrayCreate(1);
146 }
147 node->temporaries->data[0] = scratch_tensor_index;
148
149 TfLiteIntArray* scratch_size_array = TfLiteIntArrayCreate(2);
150 scratch_size_array->data[0] = batch_size;
151 scratch_size_array->data[1] = num_filters;
152
153 TfLiteTensor* scratch_tensor;
154 TF_LITE_ENSURE_OK(
155 context, GetTemporarySafe(context, node, /*index=*/0, &scratch_tensor));
156
157 // The scratch buffer is of type int32 for full integer svdf and it's of type
158 // float32 for hybrid and float case.
159 if (is_full_integer) {
160 scratch_tensor->type = kTfLiteInt32;
161 } else {
162 scratch_tensor->type = kTfLiteFloat32;
163 }
164 scratch_tensor->allocation_type = kTfLiteArenaRw;
165 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scratch_tensor,
166 scratch_size_array));
167
168 if (is_hybrid_op) {
169 op_data->compute_row_sums = true;
170 // Tell interpreter to allocate temporary tensors to store quantized values
171 // of input tensors.
172 node->temporaries->data[1] = scratch_tensor_index + 1;
173 TfLiteTensor* input_quantized;
174 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
175 &input_quantized));
176 input_quantized->type = weights_feature->type;
177 input_quantized->allocation_type = kTfLiteArenaRw;
178 if (!TfLiteIntArrayEqual(input_quantized->dims, input->dims)) {
179 TfLiteIntArray* input_quantized_size = TfLiteIntArrayCopy(input->dims);
180 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
181 input_quantized_size));
182 }
183
184 // Tell interpreter to allocate temporary tensors to store scaling factors.
185 node->temporaries->data[2] = scratch_tensor_index + 2;
186 TfLiteTensor* scaling_factors;
187 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/2,
188 &scaling_factors));
189 scaling_factors->type = kTfLiteFloat32;
190 scaling_factors->allocation_type = kTfLiteArenaRw;
191 int scaling_dims[1] = {batch_size};
192 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
193 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
194 scaling_factors_size->data[0] = batch_size;
195 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
196 scaling_factors_size));
197 }
198
199 // Used to store dequantized weights_time matrix for hybrid computation of
200 // matmul(state, weights_time), which occurs in floating point.
201 node->temporaries->data[3] = scratch_tensor_index + 3;
202 TfLiteTensor* float_weights_time;
203 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/3,
204 &float_weights_time));
205 float_weights_time->type = kTfLiteFloat32;
206 // Persistent so that we can compute the dequantized weights only once.
207 float_weights_time->allocation_type = kTfLiteArenaRwPersistent;
208 if (!TfLiteIntArrayEqual(float_weights_time->dims, weights_time->dims)) {
209 TfLiteIntArray* float_weights_time_size =
210 TfLiteIntArrayCopy(weights_time->dims);
211 TF_LITE_ENSURE_OK(context,
212 context->ResizeTensor(context, float_weights_time,
213 float_weights_time_size));
214 }
215
216 node->temporaries->data[4] = scratch_tensor_index + 4;
217 TfLiteTensor* zero_points;
218 TF_LITE_ENSURE_OK(
219 context, GetTemporarySafe(context, node, /*index=*/4, &zero_points));
220 zero_points->type = kTfLiteFloat32;
221 zero_points->allocation_type = kTfLiteArenaRw;
222 int zero_points_dims[1] = {batch_size};
223 if (!TfLiteIntArrayEqualsArray(zero_points->dims, 1, zero_points_dims)) {
224 TfLiteIntArray* zero_points_size = TfLiteIntArrayCreate(1);
225 zero_points_size->data[0] = zero_points_dims[0];
226 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, zero_points,
227 zero_points_size));
228 }
229
230 node->temporaries->data[5] = scratch_tensor_index + 5;
231 TfLiteTensor* row_sums;
232 TF_LITE_ENSURE_OK(context,
233 GetTemporarySafe(context, node, /*index=*/5, &row_sums));
234 row_sums->type = kTfLiteFloat32;
235 row_sums->allocation_type = kTfLiteArenaRwPersistent;
236 int row_sums_dims[1] = {num_filters};
237 if (!TfLiteIntArrayEqualsArray(row_sums->dims, 1, row_sums_dims)) {
238 TfLiteIntArray* row_sums_size = TfLiteIntArrayCreate(1);
239 row_sums_size->data[0] = row_sums_dims[0];
240 TF_LITE_ENSURE_OK(
241 context, context->ResizeTensor(context, row_sums, row_sums_size));
242 }
243 }
244 if (is_full_integer) {
245 // Allocated one extra tensor.
246 TfLiteIntArray* output_temp_size_array = TfLiteIntArrayCreate(2);
247 output_temp_size_array->data[0] = num_units;
248 output_temp_size_array->data[1] = batch_size;
249 node->temporaries->data[1] = scratch_tensor_index + 1;
250 TfLiteTensor* output_temp;
251 TF_LITE_ENSURE_OK(
252 context, GetTemporarySafe(context, node, /*index=*/1, &output_temp));
253 output_temp->type = kTfLiteInt32;
254 output_temp->allocation_type = kTfLiteArenaRw;
255 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, output_temp,
256 output_temp_size_array));
257
258 // Calculate effective scales.
259 auto* input_params =
260 reinterpret_cast<TfLiteAffineQuantization*>(input->quantization.params);
261 auto* weights_feature_params = reinterpret_cast<TfLiteAffineQuantization*>(
262 weights_feature->quantization.params);
263 auto* state_params =
264 reinterpret_cast<TfLiteAffineQuantization*>(state->quantization.params);
265 auto* weight_time_params = reinterpret_cast<TfLiteAffineQuantization*>(
266 weights_time->quantization.params);
267 auto* output_params = reinterpret_cast<TfLiteAffineQuantization*>(
268 output->quantization.params);
269 const double effective_scale_1 = input_params->scale->data[0] *
270 weights_feature_params->scale->data[0] /
271 state_params->scale->data[0];
272 const double effective_scale_2 = state_params->scale->data[0] *
273 weight_time_params->scale->data[0] /
274 output_params->scale->data[0];
275 QuantizeMultiplier(effective_scale_1, &op_data->effective_scale_1_a,
276 &op_data->effective_scale_1_b);
277 QuantizeMultiplier(effective_scale_2, &op_data->effective_scale_2_a,
278 &op_data->effective_scale_2_b);
279 }
280 return kTfLiteOk;
281}
282
283TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
284 auto* params = reinterpret_cast<TfLiteSVDFParams*>(node->builtin_data);
285 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
286
287 const TfLiteTensor* input;
288 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
289 const TfLiteTensor* weights_feature;
290 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kWeightsFeatureTensor,
291 &weights_feature));
292 const TfLiteTensor* weights_time;
293 TF_LITE_ENSURE_OK(
294 context, GetInputSafe(context, node, kWeightsTimeTensor, &weights_time));
295 const TfLiteTensor* bias = GetOptionalInputTensor(context, node, kBiasTensor);
296
297 TfLiteTensor* scratch;
298 TF_LITE_ENSURE_OK(context,
299 GetTemporarySafe(context, node, /*index=*/0, &scratch));
300
301 TfLiteTensor* state = GetVariableInput(context, node, kStateTensor);
302 TfLiteTensor* output;
303 TF_LITE_ENSURE_OK(context,
304 GetOutputSafe(context, node, kOutputTensor, &output));
305
306 switch (weights_feature->type) {
307 case kTfLiteFloat32: {
308 reference_ops::EvalFloatSVDF(
309 params, GetTensorShape(input), GetTensorData<float>(input),
310 GetTensorShape(weights_feature),
311 GetTensorData<float>(weights_feature), GetTensorShape(weights_time),
312 GetTensorData<float>(weights_time), GetTensorShape(bias),
313 GetTensorData<float>(bias), GetTensorData<float>(scratch),
314 GetTensorData<float>(state), GetTensorShape(output),
315 GetTensorData<float>(output));
316 return kTfLiteOk;
317 break;
318 }
319 case kTfLiteUInt8:
320 case kTfLiteInt8: {
321 if (input->type == kTfLiteFloat32) {
322 TfLiteTensor* input_quantized;
323 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
324 &input_quantized));
325 TfLiteTensor* scaling_factors;
326 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/2,
327 &scaling_factors));
328 TfLiteTensor* float_weights_time;
329 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/3,
330 &float_weights_time));
331 TfLiteTensor* zero_points;
332 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/4,
333 &zero_points));
334 TfLiteTensor* row_sums;
335 TF_LITE_ENSURE_OK(
336 context, GetTemporarySafe(context, node, /*index=*/5, &row_sums));
337 // Dequantize weights time.
338 // TODO(alanchiao): this dequantization initialization only needs to
339 // happen once per model and should theoretically be placed in either
340 // Init or Prepare. However, TFLite doesn't allocate float_weights_time
341 // until the Eval function.
342 // TODO(alanchiao): refactor logic out into dequantize function.
343 if (!op_data->float_weights_time_initialized) {
344 const float dequantization_scale = weights_time->params.scale;
345 const int8_t* weights_time_ptr = GetTensorData<int8_t>(weights_time);
346 float* float_weights_time_ptr =
347 GetTensorData<float>(float_weights_time);
348 for (int i = 0; i < NumElements(float_weights_time); ++i) {
349 float_weights_time_ptr[i] =
350 weights_time_ptr[i] * dequantization_scale;
351 }
352 op_data->float_weights_time_initialized = true;
353 }
354
355 int32_t* zero_points_ptr = nullptr;
356 int32_t* row_sums_ptr = nullptr;
357 if (params->asymmetric_quantize_inputs && row_sums != nullptr) {
358 zero_points_ptr = GetTensorData<int32_t>(zero_points);
359 row_sums_ptr = GetTensorData<int32_t>(row_sums);
360 }
361
362 reference_ops::EvalHybridSVDF(
363 params, GetTensorShape(input), GetTensorData<float>(input),
364 GetTensorShape(weights_feature),
365 GetTensorData<int8_t>(weights_feature),
366 weights_feature->params.scale, GetTensorShape(float_weights_time),
367 GetTensorData<float>(float_weights_time), GetTensorShape(bias),
368 GetTensorData<float>(bias), GetTensorData<float>(scratch),
369 GetTensorData<float>(scaling_factors),
370 GetTensorData<int8_t>(input_quantized), GetTensorData<float>(state),
371 GetTensorShape(output), GetTensorData<float>(output),
372 zero_points_ptr, row_sums_ptr, &op_data->compute_row_sums);
373 return kTfLiteOk;
374 } else {
375 auto* input_params = reinterpret_cast<TfLiteAffineQuantization*>(
376 input->quantization.params);
377 auto* output_params = reinterpret_cast<TfLiteAffineQuantization*>(
378 output->quantization.params);
379 TfLiteTensor* output_temp;
380 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
381 &output_temp));
382
383 // Currently supports only ReLU.
384 // TODO(jianlijianli): support other activations.
385 TF_LITE_ENSURE_EQ(context, params->activation, kTfLiteActRelu);
386
387 reference_ops::EvalIntegerSVDF(
388 params, GetTensorShape(input), GetTensorData<int8_t>(input),
389 GetTensorShape(weights_feature),
390 GetTensorData<int8_t>(weights_feature),
391 GetTensorShape(weights_time), GetTensorData<int16_t>(weights_time),
392 GetTensorShape(bias), GetTensorData<int32_t>(bias),
393 GetTensorData<int16_t>(state), GetTensorShape(output),
394 GetTensorData<int8_t>(output), GetTensorData<int32_t>(scratch),
395 GetTensorData<int32_t>(output_temp), op_data->effective_scale_1_a,
396 op_data->effective_scale_1_b, op_data->effective_scale_2_a,
397 op_data->effective_scale_2_b, input_params->zero_point->data[0],
398 output_params->zero_point->data[0]);
399 return kTfLiteOk;
400 }
401 break;
402 }
403 default:
404 context->ReportError(context, "Type %s not currently supported.",
405 TfLiteTypeGetName(weights_feature->type));
406 return kTfLiteError;
407 }
