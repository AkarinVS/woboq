34constexpr int kInputMultipliers = 1;
35constexpr int kOutputTensor = 0;

38template <typename T>
39TfLiteIntArray* MultiplyShapeDims(const TfLiteIntArray& shape,
40 const TfLiteTensor* multipliers,
41 int num_dimensions) {
42 const T* multipliers_v = GetTensorData<T>(multipliers);
43
44 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(num_dimensions);
45 for (int i = 0; i < num_dimensions; ++i) {
46 output_shape->data[i] = shape.data[i] * multipliers_v[i];
47 }
48 return output_shape;
49}

51TfLiteStatus ResizeOutput(TfLiteContext* context, TfLiteNode* node) {
52 const TfLiteTensor* input;
53 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
54 TfLiteTensor* output;
55 TF_LITE_ENSURE_OK(context,
56 GetOutputSafe(context, node, kOutputTensor, &output));
57 const TfLiteTensor* multipliers;
58 TF_LITE_ENSURE_OK(
59 context, GetInputSafe(context, node, kInputMultipliers, &multipliers));
60
61 const int num_dimensions = NumDimensions(input);
62 const int num_multipliers = NumElements(multipliers);
63 TF_LITE_ENSURE_EQ(context, num_dimensions, num_multipliers);
64 switch (multipliers->type) {
65 case kTfLiteInt32:
66 return context->ResizeTensor(
67 context, output,
68 MultiplyShapeDims<int32_t>(*input->dims, multipliers,
69 num_dimensions));
70 case kTfLiteInt64:
71 return context->ResizeTensor(
72 context, output,
73 MultiplyShapeDims<int64_t>(*input->dims, multipliers,
74 num_dimensions));
75 default:
76 context->ReportError(
77 context, "Multipliers of type '%s' are not supported by tile.",
78 TfLiteTypeGetName(multipliers->type));
79 return kTfLiteError;
80 }
81}

83template <typename T, typename M>
84void CopyMultipleTimes(const T* in_data, int32_t in_size, M multiplier,
85 T* out_data) {
86 for (M i = 0; i < multiplier; ++i) {
87 const T* in_end = in_data + in_size;
88 T* new_out_data = std::copy(in_data, in_end, out_data);
89 in_data = out_data;
90 out_data = new_out_data;
91 }
92}

94template <typename M>
95void CopyStringMultipleTimes(const TfLiteTensor* in_data, int in_data_index,
96 const int dimension_size, M multiplier,
97 DynamicBuffer* buffer) {
98 for (M i = 0; i < multiplier; ++i) {
99 for (int j = 0; j < dimension_size; ++j) {
100 const auto string_ref = GetString(in_data, in_data_index + j);
101 buffer->AddString(string_ref.str, string_ref.len);
102 }
103 }
104}

106template <typename T, typename M>
107std::pair<int, int> TileOneDimension(const TfLiteIntArray& in_dimensions,
108 const T* in_data, const M* multipliers,
109 T* out_data, int dimension) {
110 const int dimension_size = in_dimensions.data[dimension];
111 if (dimension == in_dimensions.size - 1) {
112 CopyMultipleTimes(in_data, dimension_size, multipliers[dimension],
113 out_data);
114 return std::make_pair(
115 dimension_size,
116 dimension_size * static_cast<int>(multipliers[dimension]));
117 }
118 int total_stride_size = 0, total_tiled_stride_size = 0;
119 const T* copy_from_data = in_data;
120 T* copy_to_data = out_data;
121 for (int i = 0; i < dimension_size; ++i) {
122 int stride_size = 0, tiled_stride_size = 0;
123 std::tie(stride_size, tiled_stride_size) =
124 TileOneDimension(in_dimensions, copy_from_data, multipliers,
125 copy_to_data, dimension + 1);
126 copy_from_data += stride_size;
127 copy_to_data += tiled_stride_size;
128 total_stride_size += stride_size;
129 total_tiled_stride_size += tiled_stride_size;
130 }
131 CopyMultipleTimes(out_data, total_tiled_stride_size,
132 multipliers[dimension] - 1,
133 out_data + total_tiled_stride_size);
134 return std::make_pair(
135 total_stride_size,
136 static_cast<int>(total_tiled_stride_size * multipliers[dimension]));
137}

139template <typename M>
140std::pair<int, int> TileStringOneDimension(
141 const TfLiteIntArray& in_dimensions, const TfLiteTensor* in_data,
142 int in_data_index, const M* multipliers, DynamicBuffer* buffer,
143 int buffer_index, int dimension, TfLiteTensor* out_data) {
144 const int dimension_size = in_dimensions.data[dimension];
145 if (dimension == in_dimensions.size - 1) {
146 CopyStringMultipleTimes(in_data, in_data_index, dimension_size,
147 multipliers[dimension], buffer);
148 return {dimension_size,
149 dimension_size * static_cast<int>(multipliers[dimension])};
150 }
151
152 int total_stride_size = 0, total_tiled_stride_size = 0;
153 for (int i = 0; i < dimension_size; ++i) {
154 int stride_size, tiled_stride_size;
155 std::tie(stride_size, tiled_stride_size) = TileStringOneDimension(
156 in_dimensions, in_data, in_data_index + total_stride_size, multipliers,
157 buffer, buffer_index + total_tiled_stride_size, dimension + 1,
158 out_data);
159 total_stride_size += stride_size;
160 total_tiled_stride_size += tiled_stride_size;
161 }
162
163 buffer->WriteToTensor(out_data, /*new_shape=*/nullptr);
164 CopyStringMultipleTimes(out_data, buffer_index, total_tiled_stride_size,
165 multipliers[dimension] - 1, buffer);
166
167 return {total_stride_size,
168 total_tiled_stride_size * static_cast<int>(multipliers[dimension])};
169}

171template <typename T>
172void Tile(const TfLiteIntArray& in_dimensions, const TfLiteTensor* in_data,
173 const TfLiteTensor* multipliers, TfLiteTensor* out_data) {
174 // Doing recursively tiling from top to down dimension.
175 switch (multipliers->type) {
176 case kTfLiteInt32:
177 TileOneDimension(in_dimensions, GetTensorData<T>(in_data),
178 GetTensorData<int32_t>(multipliers),
179 GetTensorData<T>(out_data), 0);
180 break;
181 case kTfLiteInt64:
182 TileOneDimension(in_dimensions, GetTensorData<T>(in_data),
183 GetTensorData<int64_t>(multipliers),
184 GetTensorData<T>(out_data), 0);
185 break;
186 default:
187 break;
188 }
189}
190
191void TileString(const TfLiteIntArray& in_dimensions,
192 const TfLiteTensor* in_data, const TfLiteTensor* multipliers,
193 DynamicBuffer* buffer, TfLiteTensor* out_data) {
194 // Doing recursively tiling from top to down dimension.
195 switch (multipliers->type) {
196 case kTfLiteInt32:
197 TileStringOneDimension(in_dimensions, in_data, 0,
198 GetTensorData<int32_t>(multipliers), buffer, 0, 0,
199 out_data);
200 break;
201 case kTfLiteInt64:
202 TileStringOneDimension(in_dimensions, in_data, 0,
203 GetTensorData<int64_t>(multipliers), buffer, 0, 0,
204 out_data);
205 break;
206 default:
207 break;
208 }

212TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
213 TF_LITE_ENSURE_EQ(context, NumInputs(node), 2);
214 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
215
216 const TfLiteTensor* input;
217 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
218
219 TfLiteTensor* output;
220 TF_LITE_ENSURE_OK(context,
221 GetOutputSafe(context, node, kOutputTensor, &output));
222 TF_LITE_ENSURE_TYPES_EQ(context, input->type, output->type);
223
224 const TfLiteTensor* multipliers;
225 TF_LITE_ENSURE_OK(
226 context, GetInputSafe(context, node, kInputMultipliers, &multipliers));
227 // Only int32 and int64 multipliers type is supported.
228 if (multipliers->type != kTfLiteInt32 && multipliers->type != kTfLiteInt64) {
229 context->ReportError(context,
230 "Multipliers of type '%s' are not supported by tile.",
231 TfLiteTypeGetName(multipliers->type));
232 return kTfLiteError;
233 }
234
235 if (IsConstantTensor(multipliers)) {
236 TF_LITE_ENSURE_OK(context, ResizeOutput(context, node));
237 } else {
238 SetTensorToDynamic(output);
239 }
240 return kTfLiteOk;
241}
242
243TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
244 const TfLiteTensor* input;
245 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
246 TfLiteTensor* output;
247 TF_LITE_ENSURE_OK(context,
248 GetOutputSafe(context, node, kOutputTensor, &output));
249 const TfLiteTensor* multipliers;
250 TF_LITE_ENSURE_OK(
251 context, GetInputSafe(context, node, kInputMultipliers, &multipliers));
252
253 if (IsDynamicTensor(output)) {
254 TF_LITE_ENSURE_OK(context, ResizeOutput(context, node));
255 }
256
257 switch (output->type) {
258 case kTfLiteFloat32:
259 Tile<float>(*(input->dims), input, multipliers, output);
260 break;
261 case kTfLiteUInt8:
262 Tile<uint8_t>(*(input->dims), input, multipliers, output);
263 break;
264 case kTfLiteInt32:
265 Tile<int32_t>(*(input->dims), input, multipliers, output);
266 break;
267 case kTfLiteInt64:
268 Tile<int64_t>(*(input->dims), input, multipliers, output);
269 break;
270 case kTfLiteString: {
271 DynamicBuffer buffer;
272 TileString(*(input->dims), input, multipliers, &buffer, output);
273 buffer.WriteToTensor(output, /*new_shape=*/nullptr);
274 break;
275 }
276 case kTfLiteBool:
277 Tile<bool>(*(input->dims), input, multipliers, output);
278 break;
279 default:
280 context->ReportError(context, "Type '%s' is not supported by tile.",
281 TfLiteTypeGetName(output->type));
282 return kTfLiteError;
283 }
284 return kTfLiteOk;
285}
