31struct OpData {
32 int scratch_tensor_index;
33 bool compute_row_sums = false;
34};
35
36} // namespace

38constexpr int kInputTensor = 0;
39constexpr int kWeightsTensor = 1;
40constexpr int kRecurrentWeightsTensor = 2;
41constexpr int kBiasTensor = 3;
42constexpr int kHiddenStateTensor = 4;
43
44// Output tensor.
45constexpr int kOutputTensor = 0;
46
47void* Init(TfLiteContext* context, const char* buffer, size_t length) {
48 auto* op_data = new OpData();
49 context->AddTensors(context, /*tensors_to_add=*/6,
50 &op_data->scratch_tensor_index);
51 return op_data;
52}
53
54void Free(TfLiteContext* context, void* buffer) {
55 delete reinterpret_cast<OpData*>(buffer);
56}

58TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
59 // Check we have all the inputs and outputs we need.
60 TF_LITE_ENSURE_EQ(context, node->inputs->size, 5);
61 TF_LITE_ENSURE_EQ(context, node->outputs->size, 1);
62
63 const TfLiteTensor* input;
64 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
65 const TfLiteTensor* input_weights;
66 TF_LITE_ENSURE_OK(
67 context, GetInputSafe(context, node, kWeightsTensor, &input_weights));
68 const TfLiteTensor* recurrent_weights;
69 TF_LITE_ENSURE_OK(
70 context,
71 GetInputSafe(context, node, kRecurrentWeightsTensor, &recurrent_weights));
72 const TfLiteTensor* bias;
73 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kBiasTensor, &bias));
74 const TfLiteTensor* hidden_state;
75 TF_LITE_ENSURE_OK(
76 context, GetInputSafe(context, node, kHiddenStateTensor, &hidden_state));
77
78 // Check all the parameters of tensor match within themselves and match the
79 // input configuration.
80 const int batch_size = input->dims->data[0];
81 const int num_units = input_weights->dims->data[0];
82 TF_LITE_ENSURE_EQ(context, input->dims->data[1],
83 input_weights->dims->data[1]);
84 TF_LITE_ENSURE_EQ(context, input_weights->dims->data[0], bias->dims->data[0]);
85 TF_LITE_ENSURE_EQ(context, recurrent_weights->dims->data[0],
86 bias->dims->data[0]);
87 TF_LITE_ENSURE_EQ(context, recurrent_weights->dims->data[1],
88 bias->dims->data[0]);
89 TF_LITE_ENSURE_TYPES_EQ(context, input->type, kTfLiteFloat32);
90 TF_LITE_ENSURE_TYPES_EQ(context, input_weights->type,
91 recurrent_weights->type);
92 TF_LITE_ENSURE_EQ(context, NumDimensions(hidden_state), 2);
93 TF_LITE_ENSURE_EQ(context, hidden_state->dims->data[0], batch_size);
94 TF_LITE_ENSURE_EQ(context, hidden_state->dims->data[1], num_units);
95
96 TfLiteTensor* output;
97 TF_LITE_ENSURE_OK(context,
98 GetOutputSafe(context, node, kOutputTensor, &output));
99
100 // Resize output.
101 TfLiteIntArray* output_size_array = TfLiteIntArrayCreate(2);
102 output_size_array->data[0] = batch_size;
103 output_size_array->data[1] = num_units;
104 TF_LITE_ENSURE_OK(context,
105 context->ResizeTensor(context, output, output_size_array));
106
107 const bool is_hybrid = IsHybridOp(input, input_weights);
108
109 // Allocate temporary tensors to store quantized values of input and
110 // hidden_state tensors.
111 if (is_hybrid) {
112 auto* op_data = reinterpret_cast<OpData*>(node->user_data);
113 op_data->compute_row_sums = true;
114 TfLiteIntArrayFree(node->temporaries);
115 node->temporaries = TfLiteIntArrayCreate(6);
116 node->temporaries->data[0] = op_data->scratch_tensor_index;
117 TfLiteTensor* input_quantized;
118 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/0,
119 &input_quantized));
120 input_quantized->type = input_weights->type;
121 input_quantized->allocation_type = kTfLiteArenaRw;
122 if (!TfLiteIntArrayEqual(input_quantized->dims, input->dims)) {
123 TfLiteIntArray* input_quantized_size = TfLiteIntArrayCopy(input->dims);
124 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
125 input_quantized_size));
126 }
127 node->temporaries->data[1] = op_data->scratch_tensor_index + 1;
128 TfLiteTensor* hidden_state_quantized;
129 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/1,
130 &hidden_state_quantized));
131 hidden_state_quantized->type = input_weights->type;
132 hidden_state_quantized->allocation_type = kTfLiteArenaRw;
133 if (!TfLiteIntArrayEqual(hidden_state_quantized->dims,
134 hidden_state->dims)) {
135 TfLiteIntArray* hidden_state_quantized_size =
136 TfLiteIntArrayCopy(hidden_state->dims);
137 TF_LITE_ENSURE_OK(context,
138 context->ResizeTensor(context, hidden_state_quantized,
139 hidden_state_quantized_size));
140 }
141 node->temporaries->data[2] = op_data->scratch_tensor_index + 2;
142 TfLiteTensor* scaling_factors;
143 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/2,
144 &scaling_factors));
145 scaling_factors->type = kTfLiteFloat32;
146 scaling_factors->allocation_type = kTfLiteArenaRw;
147 int scaling_dims[1] = {batch_size};
148 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
149 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
150 scaling_factors_size->data[0] = batch_size;
151 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
152 scaling_factors_size));
153 }
154 node->temporaries->data[3] = op_data->scratch_tensor_index + 3;
155 TfLiteTensor* accum_scratch;
156 TF_LITE_ENSURE_OK(
157 context, GetTemporarySafe(context, node, /*index=*/3, &accum_scratch));
158 accum_scratch->type = kTfLiteInt32;
159 accum_scratch->allocation_type = kTfLiteArenaRw;
160 int accum_scratch_dims[2] = {num_units, batch_size};
161 if (!TfLiteIntArrayEqualsArray(accum_scratch->dims, 2,
162 accum_scratch_dims)) {
163 TfLiteIntArray* accum_scratch_size = TfLiteIntArrayCreate(2);
164 accum_scratch_size->data[0] = accum_scratch_dims[0];
165 accum_scratch_size->data[1] = accum_scratch_dims[1];
166 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, accum_scratch,
167 accum_scratch_size));
168 }
169 node->temporaries->data[4] = op_data->scratch_tensor_index + 4;
170 TfLiteTensor* zero_points;
171 TF_LITE_ENSURE_OK(
172 context, GetTemporarySafe(context, node, /*index=*/4, &zero_points));
173 zero_points->type = kTfLiteInt32;
174 zero_points->allocation_type = kTfLiteArenaRw;
175 int zero_points_dims[1] = {batch_size};
176 if (!TfLiteIntArrayEqualsArray(zero_points->dims, 1, zero_points_dims)) {
177 TfLiteIntArray* zero_points_size = TfLiteIntArrayCreate(1);
178 zero_points_size->data[0] = batch_size;
179 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, zero_points,
180 zero_points_size));
181 }
182 node->temporaries->data[5] = op_data->scratch_tensor_index + 5;
183 TfLiteTensor* row_sums;
184 TF_LITE_ENSURE_OK(context,
185 GetTemporarySafe(context, node, /*index=*/5, &row_sums));
186 row_sums->type = kTfLiteInt32;
187 row_sums->allocation_type = kTfLiteArenaRwPersistent;
188 int row_sums_dims[2] = {2, num_units};
189 if (!TfLiteIntArrayEqualsArray(row_sums->dims, 2, row_sums_dims)) {
190 TfLiteIntArray* row_sums_size = TfLiteIntArrayCreate(2);
191 row_sums_size->data[0] = row_sums_dims[0];
192 row_sums_size->data[1] = row_sums_dims[1];
193 TF_LITE_ENSURE_OK(
194 context, context->ResizeTensor(context, row_sums, row_sums_size));
195 }
196 }
197 return kTfLiteOk;
198}

200TfLiteStatus EvalFloat(const TfLiteTensor* input,
201 const TfLiteTensor* input_weights,
202 const TfLiteTensor* recurrent_weights,
203 const TfLiteTensor* bias, const TfLiteRNNParams* params,
204 TfLiteTensor* hidden_state, TfLiteTensor* output) {
205 const int batch_size = input->dims->data[0];
206 const int num_units = input_weights->dims->data[0];
207 const int input_size = input->dims->data[1];
208 const int output_batch_leading_dim =
209 output->dims->data[output->dims->size - 1];
210
211 // Initialize the pointer to hidden state.
212 float* hidden_state_ptr_batch = GetTensorData<float>(hidden_state);
213 // Initialize the pointer to input and output.
214 const float* input_ptr_batch = GetTensorData<float>(input);
215 float* output_ptr_batch = GetTensorData<float>(output);
216 // Initialize input_weights, recurrent_weights and bias.
217 const float* input_weights_ptr = GetTensorData<float>(input_weights);
218 const float* recurrent_weights_ptr = GetTensorData<float>(recurrent_weights);
219 const float* bias_ptr = GetTensorData<float>(bias);
220
221 kernel_utils::RnnBatchStep(
222 input_ptr_batch, input_weights_ptr, recurrent_weights_ptr, bias_ptr,
223 input_size, num_units, batch_size, output_batch_leading_dim,
224 params->activation, hidden_state_ptr_batch, output_ptr_batch);
225 return kTfLiteOk;
226}
227
228TfLiteStatus EvalHybrid(const TfLiteTensor* input,
229 const TfLiteTensor* input_weights,
230 const TfLiteTensor* recurrent_weights,
231 const TfLiteTensor* bias, const TfLiteRNNParams* params,
232 TfLiteTensor* input_scratch,
233 TfLiteTensor* hidden_state_scratch,
234 TfLiteTensor* scaling_factors,
235 TfLiteTensor* hidden_state, TfLiteTensor* output,
236 TfLiteTensor* zero_points, TfLiteTensor* accum_scratch,
237 TfLiteTensor* row_sums, bool* compute_row_sums) {
238 const int batch_size = input->dims->data[0];
239 const int num_units = input_weights->dims->data[0];
240 const int input_size = input->dims->data[1];
241 const int output_batch_leading_dim =
242 output->dims->data[output->dims->size - 1];
243
244 // Initialize the pointer to hidden state.
245 float* hidden_state_ptr_batch = GetTensorData<float>(hidden_state);
246 // Initialize the pointer to input and output.
247 const float* input_ptr_batch = GetTensorData<float>(input);
248 float* output_ptr_batch = GetTensorData<float>(output);
249 // Initialize input_weights, recurrent_weights and bias.
250 const int8_t* input_weights_ptr = GetTensorData<int8_t>(input_weights);
251 const int8_t* recurrent_weights_ptr =
252 GetTensorData<int8_t>(recurrent_weights);
253 const float* bias_ptr = GetTensorData<float>(bias);
254 // Get the scale of the quantized weights.
255 float input_weights_scale = input_weights->params.scale;
256 float recurrent_weights_scale = recurrent_weights->params.scale;
257 // Initialize temporary storage for quantized values.
258 int8_t* quantized_input_ptr = GetTensorData<int8_t>(input_scratch);
259 int8_t* quantized_hidden_state_ptr =
260 GetTensorData<int8_t>(hidden_state_scratch);
261 float* scaling_factors_ptr = GetTensorData<float>(scaling_factors);
262 int32_t* accum_scratch_ptr = GetTensorData<int32_t>(accum_scratch);
263 int32_t* zero_points_ptr = nullptr;
264 int32_t* row_sums_ptr = nullptr;
265 if (params->asymmetric_quantize_inputs) {
266 zero_points_ptr = GetTensorData<int32_t>(zero_points);
267 row_sums_ptr = GetTensorData<int32_t>(row_sums);
268 }
269 kernel_utils::RnnBatchStep(
270 input_ptr_batch, input_weights_ptr, input_weights_scale,
271 recurrent_weights_ptr, recurrent_weights_scale, bias_ptr, input_size,
272 num_units, batch_size, output_batch_leading_dim, params->activation,
273 quantized_input_ptr, quantized_hidden_state_ptr, scaling_factors_ptr,
274 hidden_state_ptr_batch, output_ptr_batch,
275 params->asymmetric_quantize_inputs, zero_points_ptr, accum_scratch_ptr,
276 row_sums_ptr, compute_row_sums);
277 return kTfLiteOk;
278}

280TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
281 auto* params = reinterpret_cast<TfLiteRNNParams*>(node->builtin_data);
282 auto* op_data = reinterpret_cast<OpData*>(node->user_data);
283 const TfLiteTensor* input;
284 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
285 const TfLiteTensor* input_weights;
286 TF_LITE_ENSURE_OK(
287 context, GetInputSafe(context, node, kWeightsTensor, &input_weights));
288 const TfLiteTensor* recurrent_weights;
289 TF_LITE_ENSURE_OK(
290 context,
291 GetInputSafe(context, node, kRecurrentWeightsTensor, &recurrent_weights));
292 const TfLiteTensor* bias;
293 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kBiasTensor, &bias));
294 TfLiteTensor* hidden_state =
295 GetVariableInput(context, node, kHiddenStateTensor);
296 TF_LITE_ENSURE(context, hidden_state != nullptr);
297 TfLiteTensor* output;
298 TF_LITE_ENSURE_OK(context,
299 GetOutputSafe(context, node, kOutputTensor, &output));
300
301 // We already checked that weight types are consistent, so branch on one.
302 switch (input_weights->type) {
303 case kTfLiteFloat32:
304 return EvalFloat(input, input_weights, recurrent_weights, bias, params,
305 hidden_state, output);
306 case kTfLiteUInt8:
307 case kTfLiteInt8: {
308 // TODO(mirkov): implement eval with quantized inputs as well.
309 TfLiteTensor* input_quantized;
310 TF_LITE_ENSURE_OK(context,
311 GetTemporarySafe(context, node, 0, &input_quantized));
312 TfLiteTensor* hidden_state_quantized;
313 TF_LITE_ENSURE_OK(
314 context, GetTemporarySafe(context, node, 1, &hidden_state_quantized));
315 TfLiteTensor* scaling_factors;
316 TF_LITE_ENSURE_OK(context,
317 GetTemporarySafe(context, node, 2, &scaling_factors));
318 TfLiteTensor* accum_scratch;
319 TF_LITE_ENSURE_OK(context,
320 GetTemporarySafe(context, node, 3, &accum_scratch));
321 TfLiteTensor* zero_points;
322 TF_LITE_ENSURE_OK(context,
323 GetTemporarySafe(context, node, 4, &zero_points));
324 TfLiteTensor* row_sums;
325 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, 5, &row_sums));
326 return EvalHybrid(input, input_weights, recurrent_weights, bias, params,
327 input_quantized, hidden_state_quantized,
328 scaling_factors, hidden_state, output, zero_points,
329 accum_scratch, row_sums, &op_data->compute_row_sums);
330 }
331 default:
332 TF_LITE_KERNEL_LOG(context, "Type %s not currently supported.",
333 TfLiteTypeGetName(input_weights->type));
334 return kTfLiteError;
335 }
336 return kTfLiteOk;
337}
