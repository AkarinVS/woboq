34// Propagate tensor shapes and types from `src_tensor_indices` in `src_subgraph`
35// to `dst_tensor_indices` in `dst_subgraph`.
36//
37// When `resize_subgraph_inputs` is true, the function calls subgraphs's
38// `ResizeInputTensor` function, and it may trigger the memory planner to
39// reallocate memory.
40// When `resize_subgraph_inputs` is false, it implies `context` belongs to
41// `dst_subgraph`. The function calls `context->ResizeTensor`. This happens
42// when resizing `While` op's outputs.
43template <typename SrcVector, typename DstVector>
44TfLiteStatus CopyTensorsShapeAndType(TfLiteContext* context,
45 Subgraph* src_subgraph,
46 const SrcVector& src_tensor_indices,
47 Subgraph* dst_subgraph,
48 const DstVector& dst_tensor_indices,
49 bool resize_subgraph_inputs) {
50 TF_LITE_ENSURE_EQ(context, src_tensor_indices.size(),
51 dst_tensor_indices.size());
52 for (int i = 0; i < src_tensor_indices.size(); ++i) {
53 const TfLiteTensor* src_tensor =
54 src_subgraph->tensor(src_tensor_indices[i]);
55
56 TfLiteTensor* dst_tensor = dst_subgraph->tensor(dst_tensor_indices[i]);
57 if (resize_subgraph_inputs) {
58 std::vector<int> dims(src_tensor->dims->data,
59 src_tensor->dims->data + src_tensor->dims->size);
60 dst_subgraph->ResizeInputTensor(dst_tensor_indices[i], dims);
61 } else {
62 TF_LITE_ENSURE_OK(
63 context, context->ResizeTensor(context, dst_tensor,
64 TfLiteIntArrayCopy(src_tensor->dims)));
65 }
66 dst_tensor->type = src_tensor->type;
67 }
68 return kTfLiteOk;
69}

71// Copy the tensors data from tensors `src_tensor_indices` in `src_subgraph`
72// to `dst_tensor_indices` in `dst_subgraph`.
73template <typename SrcVector, typename DstVector>
74TfLiteStatus CopyTensorsData(TfLiteContext* context, Subgraph* src_subgraph,
75 const SrcVector& src_tensor_indices,
76 Subgraph* dst_subgraph,
77 const DstVector& dst_tensor_indices) {
78 TF_LITE_ENSURE_EQ(context, src_tensor_indices.size(),
79 dst_tensor_indices.size());
80 for (int i = 0; i < src_tensor_indices.size(); ++i) {
81 const TfLiteTensor* src_tensor =
82 src_subgraph->tensor(src_tensor_indices[i]);
83 TfLiteTensor* dst_tensor = dst_subgraph->tensor(dst_tensor_indices[i]);
84 if (IsDynamicTensor(dst_tensor)) {
85 TfLiteTensorRealloc(src_tensor->bytes, dst_tensor);
86 }
87 TF_LITE_ENSURE_EQ(context, src_tensor->bytes, dst_tensor->bytes);
88 memcpy(dst_tensor->data.raw, src_tensor->data.raw, src_tensor->bytes);
89 }
90 return kTfLiteOk;
91}

93TfLiteStatus CheckCondOutput(TfLiteContext* context,
94 const TfLiteTensor* cond_output) {
95 // The condition output must be a single boolean value.
96 TF_LITE_ENSURE_TYPES_EQ(context, cond_output->type, kTfLiteBool);
97 if (cond_output->dims->size == 0) {
98 // It's okay if it's a 0D scalar.
99 return kTfLiteOk;
100 }
101 // Otherwise it must be 1D with shape [1].
102 TF_LITE_ENSURE_EQ(context, cond_output->dims->size, 1);
103 TF_LITE_ENSURE_EQ(context, cond_output->dims->data[0], 1);
104 return kTfLiteOk;
105}

109struct OpData {
110 int cond_subgraph_index;
111 int body_subgraph_index;
112 bool cond_has_dynamic_output_tensors;
113 bool body_has_dynamic_output_tensors;
114};
115
116void* Init(TfLiteContext* context, const char* buffer, size_t length) {
117 auto* op_data = new OpData;
118 const auto* params = reinterpret_cast<const TfLiteWhileParams*>(buffer);
119 op_data->cond_subgraph_index = params->cond_subgraph_index;
120 op_data->body_subgraph_index = params->body_subgraph_index;
121 op_data->cond_has_dynamic_output_tensors = false;
122 op_data->body_has_dynamic_output_tensors = false;
123 return op_data;
124}
125
126void Free(TfLiteContext* context, void* buffer) {
127 delete reinterpret_cast<OpData*>(buffer);
128}

130TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
131 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
132 int num_inputs = node->inputs->size;
133 // The number of outputs should be the same as number of inputs.
134 TF_LITE_ENSURE_EQ(context, node->outputs->size, num_inputs);
135
136 // Check subgraph indices and get subgraphs.
137 Subgraph* this_subgraph = reinterpret_cast<Subgraph*>(context->impl_);
138 auto* subgraphs = this_subgraph->GetSubgraphs();
139 TF_LITE_ENSURE(context, op_data->cond_subgraph_index < subgraphs->size());
140 TF_LITE_ENSURE(context, op_data->body_subgraph_index < subgraphs->size());
141
142 Subgraph* cond_subgraph = (*subgraphs)[op_data->cond_subgraph_index].get();
143 Subgraph* body_subgraph = (*subgraphs)[op_data->body_subgraph_index].get();
144
145 // Check input & output count of the condition subgraph.
146 TF_LITE_ENSURE_EQ(context, cond_subgraph->inputs().size(), num_inputs);
147 TF_LITE_ENSURE_EQ(context, cond_subgraph->outputs().size(), 1);
148
149 // Check input & output count of the body subgraph.
150 TF_LITE_ENSURE_EQ(context, body_subgraph->inputs().size(), num_inputs);
151 TF_LITE_ENSURE_EQ(context, body_subgraph->outputs().size(), num_inputs);
152
153 // Prepare and check the condition subgraph.
154 TF_LITE_ENSURE_OK(
155 context, CopyTensorsShapeAndType(
156 context, this_subgraph, TfLiteIntArrayView(node->inputs),
157 cond_subgraph, cond_subgraph->inputs(), true));
158 TF_LITE_ENSURE_OK(context, cond_subgraph->AllocateTensors());
159 TfLiteTensor* cond_output =
160 cond_subgraph->tensor(cond_subgraph->outputs()[0]);
161 // This should rarely happens. In most cases the output is static with shape
162 // [1]. However theoretically intermediate tensors in the cond subgraph
163 // can be dynamic.
164 if (IsDynamicTensor(cond_output)) {
165 op_data->cond_has_dynamic_output_tensors = true;
166 } else {
167 TF_LITE_ENSURE_STATUS(CheckCondOutput(context, cond_output));
168 }
169
170 // Prepare and check the body subgraph.
171 TF_LITE_ENSURE_OK(
172 context, CopyTensorsShapeAndType(
173 context, this_subgraph, TfLiteIntArrayView(node->inputs),
174 body_subgraph, body_subgraph->inputs(), true));
175 TF_LITE_ENSURE_OK(context, body_subgraph->AllocateTensors());
176 if (body_subgraph->HasDynamicTensors()) {
177 op_data->body_has_dynamic_output_tensors = true;
178 } else {
179 for (int i = 0; i < num_inputs; ++i) {
180 TfLiteTensor* body_input =
181 body_subgraph->tensor(body_subgraph->inputs()[i]);
182 TfLiteTensor* body_output =
183 body_subgraph->tensor(body_subgraph->outputs()[i]);
184 TF_LITE_ENSURE_TYPES_EQ(context, body_input->type, body_output->type);
185
186 TF_LITE_ENSURE(context, !IsDynamicTensor(body_output));
187 if (!TfLiteIntArrayEqual(body_input->dims, body_output->dims)) {
188 // If the output shape of the body subgraph is static w.r.t. a fixed
189 // input size, but it's different from input size, it's still considered
190 // dynamic. For example: If a subgraph keeps padding its input with a
191 // fixed padding, the output shape is static w.r.t the input shape and
192 // padding, but running it in a loop will keep bloating the tensor.
193 op_data->body_has_dynamic_output_tensors = true;
194 break;
195 }
196 }
197 }
198 for (int i = 0; i < num_inputs; ++i) {
199 TfLiteTensor* output;
200 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, i, &output));
201 if (op_data->body_has_dynamic_output_tensors) {
202 SetTensorToDynamic(output);
203 } else {
204 TfLiteTensor* body_output =
205 body_subgraph->tensor(body_subgraph->outputs()[i]);
206 TfLiteIntArray* output_size = TfLiteIntArrayCopy(body_output->dims);
207 TF_LITE_ENSURE_OK(context,
208 context->ResizeTensor(context, output, output_size));
209 }
210 }
211 return kTfLiteOk;
212}

214TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
215 const OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
216 Subgraph* this_subgraph = reinterpret_cast<Subgraph*>(context->impl_);
217 auto* subgraphs = this_subgraph->GetSubgraphs();
218 Subgraph* cond_subgraph = (*subgraphs)[op_data->cond_subgraph_index].get();
219 Subgraph* body_subgraph = (*subgraphs)[op_data->body_subgraph_index].get();
220
221 // The follow graph illustrates the current implementation.
222 //
223 // This Subgraph Cond Subgraph Body Subgraph
224 // +-----------+ (1) +------------+ (3) +------------+
225 // | WHILE |-------->| SUBGRAPH |-------->| SUBGRAPH |
226 // | INPUT | /| INPUT |<----- | INPUT |
227 // +-----------+ / +------------+ \ +------------+
228 // / | \ |
229 // (6) / | (2) (5) \ | (4)
230 // / v \ v
231 // +-----------+ / +------------+ +------------+
232 // | WHILE |<-- | SUBGRAPH | | SUBGRAPH |
233 // | OUTPUT | | OUTPUT | | OUTPUT |
234 // +-----------+ +------------+ +------------+
235 //
236 // (1) Copy the inputs of WHILE op to the inputs of condition subgraph.
237 // (2) Invoke condition subgraph.
238 // Jump to step 5 if result is false.
239 // (3) Copy the inputs of condition subgraph to the inputs of body subgraph.
240 // (4) Invoke body subgraph.
241 // (5) Copy the outputs of body subgraph to the inputs condition subgraph.
242 // Jump back to step 2!
243 // (6) Copy the inputs of condition subgraph to the outputs of WHILE op.
244 //
245 // If the body subgraph has dynamic sized outputs, it's required to resize the
246 // tensor before copying in step 1, 3, 4 and 6.
247 //
248 // Note the flow is carefully designed to handle the dynamic sized output
249 // case. The loop invariant is: The newest value is in the inputs of condition
250 // subgraph. This is always true before step 2.
251 //
252 // This is the best we can do without sharing tensor buffer across subgraph
253 // boundary. Currently we copy the input / output between the subgraphs. This
254 // isn't optimized yet and a lot of redundant copies are made.
255 // TODO(b/120234921): Optimize and avoid copying tensors between subgraphs.
256
257 if (op_data->body_has_dynamic_output_tensors) {
258 // If body subgraph has dynamic outputs, the input of condition subgraph may
259 // be changed in the last invocation and may need resizing.
260 TF_LITE_ENSURE_OK(
261 context, CopyTensorsShapeAndType(
262 context, this_subgraph, TfLiteIntArrayView(node->inputs),
263 cond_subgraph, cond_subgraph->inputs(), true));
264 TF_LITE_ENSURE_OK(context, cond_subgraph->AllocateTensors());
265 }
266 TF_LITE_ENSURE_OK(
267 context,
268 CopyTensorsData(context, this_subgraph, TfLiteIntArrayView(node->inputs),
269 cond_subgraph, cond_subgraph->inputs()));
270
271 while (true) {
272 TF_LITE_ENSURE_OK(context, cond_subgraph->Invoke());
273 int cond_subgraph_output_index = cond_subgraph->outputs()[0];
274 cond_subgraph->EnsureTensorDataIsReadable(cond_subgraph_output_index);
275 TfLiteTensor* cond_output =
276 cond_subgraph->tensor(cond_subgraph_output_index);
277 if (op_data->cond_has_dynamic_output_tensors) {
278 TF_LITE_ENSURE_STATUS(CheckCondOutput(context, cond_output));
279 }
280
281 if (!cond_output->data.b[0]) {
282 break;
283 }
284 if (op_data->body_has_dynamic_output_tensors) {
285 TF_LITE_ENSURE_OK(context,
286 CopyTensorsShapeAndType(
287 context, cond_subgraph, cond_subgraph->inputs(),
288 body_subgraph, body_subgraph->inputs(), true));
289 TF_LITE_ENSURE_OK(context, body_subgraph->AllocateTensors());
290 }
291
292 TF_LITE_ENSURE_OK(
293 context,
294 CopyTensorsData(context, cond_subgraph, cond_subgraph->inputs(),
295 body_subgraph, body_subgraph->inputs()));
296
297 TF_LITE_ENSURE_OK(context, body_subgraph->Invoke());
298
299 for (int tensor_index : body_subgraph->outputs()) {
300 body_subgraph->EnsureTensorDataIsReadable(tensor_index);
301 }
302
303 if (op_data->body_has_dynamic_output_tensors) {
304 TF_LITE_ENSURE_OK(context,
305 CopyTensorsShapeAndType(
306 context, body_subgraph, body_subgraph->outputs(),
307 cond_subgraph, cond_subgraph->inputs(), true));
308 TF_LITE_ENSURE_OK(context, cond_subgraph->AllocateTensors());
309 }
310
311 TF_LITE_ENSURE_OK(
312 context,
313 CopyTensorsData(context, body_subgraph, body_subgraph->outputs(),
314 cond_subgraph, cond_subgraph->inputs()));
315 }
316
317 // Note that copying from body's output will fail if body is never invoked.
318 // TODO(b/120234921): Optimize and avoid copying tensors between subgraphs.
319 if (op_data->body_has_dynamic_output_tensors) {
320 TF_LITE_ENSURE_OK(
321 context, CopyTensorsShapeAndType(
322 context, cond_subgraph, cond_subgraph->inputs(),
323 this_subgraph, TfLiteIntArrayView(node->outputs), false));
324 }
325
326 TF_LITE_ENSURE_OK(
327 context,
328 CopyTensorsData(context, cond_subgraph, cond_subgraph->inputs(),
329 this_subgraph, TfLiteIntArrayView(node->outputs)));
330 return kTfLiteOk;
331}
