89template <typename T>
90inline void Relu(const RuntimeShape& input_shape, const T* input_data,
91 const RuntimeShape& output_shape, T* output_data) {
92 const int flat_size = MatchingFlatSize(input_shape, output_shape);
93 for (int i = 0; i < flat_size; ++i) {
94 const T val = input_data[i];
95 const T lower = 0;
96 const T clamped = val < lower ? lower : val;
97 output_data[i] = clamped;
98 }
99}

128template <typename T>
129inline void ReluX(const tflite::ReluParams& params,
130 const RuntimeShape& input_shape, const T* input_data,
131 const RuntimeShape& output_shape, T* output_data) {
132 ruy::profiler::ScopeLabel label("Quantized ReluX (not fused)");
133 const int flat_size = MatchingFlatSize(input_shape, output_shape);
134 for (int i = 0; i < flat_size; ++i) {
135 const int32 val = static_cast<int32_t>(input_data[i]);
136 int32 clamped = params.output_offset +
137 MultiplyByQuantizedMultiplier(val - params.input_offset,
138 params.output_multiplier,
139 params.output_shift);
140 clamped = std::max(params.quantized_activation_min, clamped);
141 clamped = std::min(params.quantized_activation_max, clamped);
142 output_data[i] = static_cast<T>(clamped);
143 }
144}

163// TODO(jiawen): We can implement BroadcastMul on buffers of arbitrary
164// dimensionality if the runtime code does a single loop over one dimension
165// that handles broadcasting as the base case. The code generator would then
166// generate max(D1, D2) nested for loops.
167inline void BroadcastMulFivefold(const ArithmeticParams& unswitched_params,
168 const RuntimeShape& unswitched_input1_shape,
169 const uint8* unswitched_input1_data,
170 const RuntimeShape& unswitched_input2_shape,
171 const uint8* unswitched_input2_data,
172 const RuntimeShape& output_shape,
173 uint8* output_data) {
174 ArithmeticParams switched_params = unswitched_params;
175 switched_params.input1_offset = unswitched_params.input2_offset;
176 switched_params.input2_offset = unswitched_params.input1_offset;
177
178 const bool use_unswitched =
179 unswitched_params.broadcast_category ==
180 tflite::BroadcastableOpCategory::kFirstInputBroadcastsFast;
181
182 const ArithmeticParams& params =
183 use_unswitched ? unswitched_params : switched_params;
184 const uint8* input1_data =
185 use_unswitched ? unswitched_input1_data : unswitched_input2_data;
186 const uint8* input2_data =
187 use_unswitched ? unswitched_input2_data : unswitched_input1_data;
188
189 // Fivefold nested loops. The second input resets its position for each
190 // iteration of the second loop. The first input resets its position at the
191 // beginning of the fourth loop. The innermost loop is an elementwise Mul of
192 // sections of the arrays.
193 uint8* output_data_ptr = output_data;
194 const uint8* input1_data_ptr = input1_data;
195 const uint8* input2_data_reset = input2_data;
196 int y0 = params.broadcast_shape[0];
197 int y1 = params.broadcast_shape[1];
198 int y2 = params.broadcast_shape[2];
199 int y3 = params.broadcast_shape[3];
200 int y4 = params.broadcast_shape[4];
201 for (int i0 = 0; i0 < y0; ++i0) {
202 const uint8* input2_data_ptr;
203 for (int i1 = 0; i1 < y1; ++i1) {
204 input2_data_ptr = input2_data_reset;
205 for (int i2 = 0; i2 < y2; ++i2) {
206 for (int i3 = 0; i3 < y3; ++i3) {
207 MulElementwise(y4, params, input1_data_ptr, input2_data_ptr,
208 output_data_ptr);
209 input2_data_ptr += y4;
210 output_data_ptr += y4;
211 }
212 input1_data_ptr += y4;
213 }
214 }
215 input2_data_reset = input2_data_ptr;
216 }

219inline void Mul(const ArithmeticParams& params,
220 const RuntimeShape& input1_shape, const int16* input1_data,
221 const RuntimeShape& input2_shape, const int16* input2_data,
222 const RuntimeShape& output_shape, int16* output_data) {
223 ruy::profiler::ScopeLabel label("Mul/Int16");
224
225 const int flat_size =
226 MatchingElementsSize(input1_shape, input2_shape, output_shape);
227
228 for (int i = 0; i < flat_size; i++) {
229 // F0 uses 0 integer bits, range [-1, 1].
230 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
231
232 F0 unclamped_result =
233 F0::FromRaw(input1_data[i]) * F0::FromRaw(input2_data[i]);
234 output_data[i] = unclamped_result.raw();
235 }
236}

238inline void Mul(const ArithmeticParams& params,
239 const RuntimeShape& input1_shape, const int16* input1_data,
240 const RuntimeShape& input2_shape, const int16* input2_data,
241 const RuntimeShape& output_shape, uint8* output_data) {
242 ruy::profiler::ScopeLabel label("Mul/Int16Uint8");
243 int32 output_offset = params.output_offset;
244 int32 output_activation_min = params.quantized_activation_min;
245 int32 output_activation_max = params.quantized_activation_max;
246 TFLITE_DCHECK_LE(output_activation_min, output_activation_max);
247
248 const int flat_size =
249 MatchingElementsSize(input1_shape, input2_shape, output_shape);
250
251 for (int i = 0; i < flat_size; i++) {
252 // F0 uses 0 integer bits, range [-1, 1].
253 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
254
255 F0 unclamped_result =
256 F0::FromRaw(input1_data[i]) * F0::FromRaw(input2_data[i]);
257 int16 rescaled_result =
258 gemmlowp::RoundingDivideByPOT(unclamped_result.raw(), 8);
259 int16 clamped_result =
260 std::min<int16>(output_activation_max - output_offset, rescaled_result);
261 clamped_result =
262 std::max<int16>(output_activation_min - output_offset, clamped_result);
263 output_data[i] = output_offset + clamped_result;
264 }
265}

267inline void Sub16(const ArithmeticParams& params,
268 const RuntimeShape& input1_shape, const int16_t* input1_data,
269 const RuntimeShape& input2_shape, const int16_t* input2_data,
270 const RuntimeShape& output_shape, int16_t* output_data) {
271 ruy::profiler::ScopeLabel label("Sub/Int16");
272 const int input1_shift = params.input1_shift;
273 const int flat_size =
274 MatchingElementsSize(input1_shape, input2_shape, output_shape);
275 const int16 output_activation_min = params.quantized_activation_min;
276 const int16 output_activation_max = params.quantized_activation_max;
277
278 TFLITE_DCHECK(input1_shift == 0 || params.input2_shift == 0);
279 TFLITE_DCHECK_LE(input1_shift, 0);
280 TFLITE_DCHECK_LE(params.input2_shift, 0);
281 const int16* not_shift_input = input1_shift == 0 ? input1_data : input2_data;
282 const int16* shift_input = input1_shift == 0 ? input2_data : input1_data;
283 const int input_right_shift =
284 input1_shift == 0 ? -params.input2_shift : -input1_shift;
285
286 if (input1_shift == 0) {
287 // F0 uses 0 integer bits, range [-1, 1].
288 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
289 for (int i = 0; i < flat_size; ++i) {
290 F0 input_ready_scaled = F0::FromRaw(not_shift_input[i]);
291 F0 scaled_input = F0::FromRaw(
292 gemmlowp::RoundingDivideByPOT(shift_input[i], input_right_shift));
293 F0 result = SaturatingSub(input_ready_scaled, scaled_input);
294 const int16 raw_output = result.raw();
295 const int16 clamped_output = std::min(
296 output_activation_max, std::max(output_activation_min, raw_output));
297 output_data[i] = clamped_output;
298 }
299 } else {
300 // F0 uses 0 integer bits, range [-1, 1].
301 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
302 for (int i = 0; i < flat_size; ++i) {
303 F0 input_ready_scaled = F0::FromRaw(not_shift_input[i]);
304 F0 scaled_input = F0::FromRaw(
305 gemmlowp::RoundingDivideByPOT(shift_input[i], input_right_shift));
306 F0 result = SaturatingSub(scaled_input, input_ready_scaled);
307 const int16 raw_output = result.raw();
308 const int16 clamped_output = std::min(
309 output_activation_max, std::max(output_activation_min, raw_output));
310 output_data[i] = clamped_output;
311 }
312 }

315template <typename Scalar>
316void Pack(const PackParams& params, const RuntimeShape* const* input_shapes,
317 const Scalar* const* input_data, const RuntimeShape& output_shape,
318 Scalar* output_data) {
319 ruy::profiler::ScopeLabel label("Pack");
320 const int dimensions = output_shape.DimensionsCount();
321 int axis = params.axis;
322 int inputs_count = params.inputs_count;
323
324 int outer_size = 1;
325 for (int i = 0; i < axis; i++) {
326 outer_size *= output_shape.Dims(i);
327 }
328 int copy_size = 1;
329 for (int i = params.axis + 1; i < dimensions; i++) {
330 copy_size *= output_shape.Dims(i);
331 }
332 TFLITE_DCHECK_EQ((**input_shapes).FlatSize(), copy_size * outer_size);
333
334 for (int i = 0; i < inputs_count; ++i) {
335 for (int k = 0; k < outer_size; k++) {
336 const Scalar* input_ptr = input_data[i] + copy_size * k;
337 int loc = k * inputs_count * copy_size + i * copy_size;
338 memcpy(output_data + loc, input_ptr, copy_size * sizeof(Scalar));
339 }
340 }
341}

343template <typename Scalar>
344void Unpack(const UnpackParams& params, const RuntimeShape& input_shape,
345 const Scalar* input_data, const RuntimeShape& output_shape,
346 Scalar* const* output_datas) {
347 ruy::profiler::ScopeLabel label("Unpack");
348 const int dimensions = input_shape.DimensionsCount();
349 const int outputs_count = params.num_split;
350
351 int outer_size = 1;
352 int axis = params.axis;
353 if (axis < 0) {
354 axis += dimensions;
355 }
356 TFLITE_DCHECK_GE(axis, 0);
357 TFLITE_DCHECK_LT(axis, dimensions);
358 for (int i = 0; i < axis; ++i) {
359 outer_size *= input_shape.Dims(i);
360 }
361 int copy_size = 1;
362 for (int i = axis + 1; i < dimensions; ++i) {
363 copy_size *= input_shape.Dims(i);
364 }
365 TFLITE_DCHECK_EQ(output_shape.FlatSize(), copy_size * outer_size);
366
367 for (int i = 0; i < outputs_count; ++i) {
368 for (int k = 0; k < outer_size; k++) {
369 Scalar* output_ptr = output_datas[i] + copy_size * k;
370 int loc = k * outputs_count * copy_size + i * copy_size;
371 memcpy(output_ptr, input_data + loc, copy_size * sizeof(Scalar));
372 }
373 }
374}

376template <typename Scalar>
377void PackWithScaling(const PackParams& params,
378 const RuntimeShape* const* input_shapes,
379 const uint8* const* input_data,
380 const RuntimeShape& output_shape, uint8* output_data) {
381 ruy::profiler::ScopeLabel label("PackWithScaling");
382 const int dimensions = output_shape.DimensionsCount();
383 int axis = params.axis;
384 const int32* input_zeropoint = params.input_zeropoint;
385 const float* input_scale = params.input_scale;
386 int inputs_count = params.inputs_count;
387 const int32 output_zeropoint = params.output_zeropoint;
388 const float output_scale = params.output_scale;
389
390 int outer_size = 1;
391 for (int i = 0; i < axis; i++) {
392 outer_size *= output_shape.Dims(i);
393 }
394 int copy_size = 1;
395 for (int i = axis + 1; i < dimensions; i++) {
396 copy_size *= output_shape.Dims(i);
397 }
398 TFLITE_DCHECK_EQ((**input_shapes).FlatSize(), copy_size * outer_size);
399
400 Scalar* output_ptr = output_data;
401 const float inverse_output_scale = 1.f / output_scale;
402 for (int k = 0; k < outer_size; k++) {
403 for (int i = 0; i < inputs_count; ++i) {
404 if (input_zeropoint[i] == output_zeropoint &&
405 input_scale[i] == output_scale) {
406 memcpy(output_ptr, input_data[i] + k * copy_size,
407 copy_size * sizeof(Scalar));
408 } else {
409 assert(false);
410 const float scale = input_scale[i] * inverse_output_scale;
411 const float bias = -input_zeropoint[i] * scale;
412 auto input_ptr = input_data[i];
413 for (int j = 0; j < copy_size; ++j) {
414 const int32_t value =
415 static_cast<int32_t>(std::round(input_ptr[j] * scale + bias)) +
416 output_zeropoint;
417 output_ptr[j] =
418 static_cast<uint8_t>(std::max(std::min(255, value), 0));
419 }
420 }
421 output_ptr += copy_size;
422 }
423 }

438inline void LstmCell(
439 const LstmCellParams& params, const RuntimeShape& unextended_input_shape,
440 const float* input_data, const RuntimeShape& unextended_prev_activ_shape,
441 const float* prev_activ_data, const RuntimeShape& weights_shape,
442 const float* weights_data, const RuntimeShape& unextended_bias_shape,
443 const float* bias_data, const RuntimeShape& unextended_prev_state_shape,
444 const float* prev_state_data,
445 const RuntimeShape& unextended_output_state_shape, float* output_state_data,
446 const RuntimeShape& unextended_output_activ_shape, float* output_activ_data,
447 const RuntimeShape& unextended_concat_temp_shape, float* concat_temp_data,
448 const RuntimeShape& unextended_activ_temp_shape, float* activ_temp_data) {
449 TFLITE_DCHECK_LE(unextended_input_shape.DimensionsCount(), 4);
450 TFLITE_DCHECK_LE(unextended_prev_activ_shape.DimensionsCount(), 4);
451 TFLITE_DCHECK_LE(unextended_bias_shape.DimensionsCount(), 4);
452 TFLITE_DCHECK_LE(unextended_prev_state_shape.DimensionsCount(), 4);
453 TFLITE_DCHECK_LE(unextended_output_state_shape.DimensionsCount(), 4);
454 TFLITE_DCHECK_LE(unextended_output_activ_shape.DimensionsCount(), 4);
455 TFLITE_DCHECK_LE(unextended_concat_temp_shape.DimensionsCount(), 4);
456 TFLITE_DCHECK_LE(unextended_activ_temp_shape.DimensionsCount(), 4);
457 const RuntimeShape input_shape =
458 RuntimeShape::ExtendedShape(4, unextended_input_shape);
459 const RuntimeShape prev_activ_shape =
460 RuntimeShape::ExtendedShape(4, unextended_prev_activ_shape);
461 const RuntimeShape bias_shape =
462 RuntimeShape::ExtendedShape(4, unextended_bias_shape);
463 const RuntimeShape prev_state_shape =
464 RuntimeShape::ExtendedShape(4, unextended_prev_state_shape);
465 const RuntimeShape output_state_shape =
466 RuntimeShape::ExtendedShape(4, unextended_output_state_shape);
467 const RuntimeShape output_activ_shape =
468 RuntimeShape::ExtendedShape(4, unextended_output_activ_shape);
469 const RuntimeShape concat_temp_shape =
470 RuntimeShape::ExtendedShape(4, unextended_concat_temp_shape);
471 const RuntimeShape activ_temp_shape =
472 RuntimeShape::ExtendedShape(4, unextended_activ_temp_shape);
473 TFLITE_DCHECK_GE(weights_shape.DimensionsCount(), 2);
474
475 const int weights_dim_count = weights_shape.DimensionsCount();
476 const int batches =
477 MatchingDim(input_shape, 0, prev_activ_shape, 0, prev_state_shape, 0,
478 output_state_shape, 0, output_activ_shape, 0);
479 const int height =
480 MatchingDim(input_shape, 1, prev_activ_shape, 1, prev_state_shape, 1,
481 output_state_shape, 1, output_activ_shape, 1);
482 const int width =
483 MatchingDim(input_shape, 2, prev_activ_shape, 2, prev_state_shape, 2,
484 output_state_shape, 2, output_activ_shape, 2);
485 const int input_depth = input_shape.Dims(3);
486 const int prev_activ_depth = prev_activ_shape.Dims(3);
487 const int total_input_depth = prev_activ_depth + input_depth;
488 TFLITE_DCHECK_EQ(weights_shape.Dims(weights_dim_count - 1),
489 total_input_depth);
490 TFLITE_DCHECK_EQ(FlatSizeSkipDim(bias_shape, 3), 1);
491 const int intern_activ_depth =
492 MatchingDim(weights_shape, weights_dim_count - 2, bias_shape, 3);
493 TFLITE_DCHECK_EQ(weights_shape.FlatSize(),
494 intern_activ_depth * total_input_depth);
495 TFLITE_DCHECK_EQ(intern_activ_depth % 4, 0);
496 const int output_depth =
497 MatchingDim(prev_state_shape, 3, prev_activ_shape, 3, output_state_shape,
498 3, output_activ_shape, 3);
499 TFLITE_DCHECK_EQ(output_depth, intern_activ_depth / 4);
500
501 // Concatenate prev_activ and input data together
502 std::vector<float const*> concat_input_arrays_data;
503 std::vector<RuntimeShape const*> concat_input_arrays_shapes;
504 concat_input_arrays_data.push_back(input_data);
505 concat_input_arrays_data.push_back(prev_activ_data);
506 concat_input_arrays_shapes.push_back(&input_shape);
507 concat_input_arrays_shapes.push_back(&prev_activ_shape);
508 tflite::ConcatenationParams concat_params;
509 concat_params.axis = 3;
510 concat_params.inputs_count = concat_input_arrays_data.size();
511 Concatenation(concat_params, &(concat_input_arrays_shapes[0]),
512 &(concat_input_arrays_data[0]), concat_temp_shape,
513 concat_temp_data);
514
515 // Fully connected
516 tflite::FullyConnectedParams fc_params;
517 fc_params.float_activation_min = std::numeric_limits<float>::lowest();
518 fc_params.float_activation_max = std::numeric_limits<float>::max();
519 FullyConnected(fc_params, concat_temp_shape, concat_temp_data, weights_shape,
520 weights_data, bias_shape, bias_data, activ_temp_shape,
521 activ_temp_data);
522
523 // Memory state update (the LSTM "guts")
524 for (int b = 0; b < batches; ++b) {
525 for (int w = 0; w < width; ++w) {
526 for (int h = 0; h < height; ++h) {
527 for (int c = 0; c < output_depth; ++c) {
528 const float input_gate =
529 1.f /
530 (1.f + std::exp(-activ_temp_data[Offset(activ_temp_shape, b, h, w,
531 0 * output_depth + c)]));
532 const float new_input = std::tanh(activ_temp_data[Offset(
533 activ_temp_shape, b, h, w, 1 * output_depth + c)]);
534 const float forget_gate =
535 1.f /
536 (1.f + std::exp(-activ_temp_data[Offset(activ_temp_shape, b, h, w,
537 2 * output_depth + c)]));
538 const float output_gate =
539 1.f /
540 (1.f + std::exp(-activ_temp_data[Offset(activ_temp_shape, b, h, w,
541 3 * output_depth + c)]));
542 const float new_state =
543 input_gate * new_input +
544 forget_gate *
545 prev_state_data[Offset(prev_state_shape, b, h, w, c)];
546 output_state_data[Offset(output_state_shape, b, h, w, c)] = new_state;
547 output_activ_data[Offset(output_activ_shape, b, h, w, c)] =
548 output_gate * std::tanh(new_state);
549 }
550 }
551 }
552 }
553}

555// Quantized LSTM cell implementation.
556// The quantization of the input, output arrays is as follows:
557// - The input activations are quantized as uint8 on the interval
558// [-1, 127/128].
559// The rationale for that is that is the natural interval for output
560// activations (see next point) and these need to be concatenated together.
561// We could accommodate different ranges by re-scaling, but we empirically
562// found that setting the input activations range to be [-1, 127/128] in the
563// first place, removing the need for re-scaling, greatly improves accuracy.
564// - The output activations are quantized as uint8 on the interval
565// [-1, 127/128].
566// The rationale for that is that the definition of a LSTM cell makes them
567// intrinsically constrained in [-1, 1]; tweaking that to [-1, 127/128]
568// makes for simpler, more accurate fixed-point arithmetic.
569// - The output-at-previous-timestep state array is obviously quantized as
570// the output activations.
571// - The internal LSTM memory (not the output-at-previous-timestep, the other
572// internal state array) is int16-quantized and may use any power-of-two,
573// symmetric range i.e. [-2^N, 2^N * 32767/32768] for any N, which we call
574// StateIntegerBits below, see the below discussion of that template
575// parameter ("The StateIntegerBits template parameter").
576// - The output of the internal fully-connected node is int16-quantized
577// on the interval [-8, 8 * 32767/32768], the rationale for which is
578// explained just below ("Why [-8, 8] for fully-connected output?").
579//
580//
581// === The StateIntegerBits template parameter ===
582//
583// The StateIntegerBits template parameter controls the fixed-point format used
584// to represent the internal memory of the LSTM cell (not the
585// output-at-previous-timestep, the other internal state array). It's currently
586// a template parameter so that the model can control that. The most typical
587// value for StateIntegerBits is 4. Other plausible values are anywhere between
588// 3 and 5. We might eventually standardize on a single supported value, e.g. 4,
589// and drop that template parameter. The reason why it can't be a runtime
590// parameter is that this controls the fixed-point format used, i.e. we need to
591// generate actually different code based on it. In particular, we generate code
592// for a fixed-point tanh() implementation for that format, which internally
593// uses a fixed-point exp() implementation, which internally uses a
594// barrel-shifter with a number of steps that depends on StateIntegerBits.
595// Another consequence of that is that a higher value of StateIntegerBits
596// results in a more expensive implementation (more barrel shifter steps
597// needed).
598//
599//
600// === Why [-8, 8] for fully-connected output? ===
601//
602// This array is only fed to Logistic and Tanh functions, for which
603// the quantized implementation will want to use fixed-point arithmetic,
604// requiring a power-of-two representation interval. Thus, we should right
605// away quantize this array to a power-of-two interval; otherwise,
606// implementation will need to rescale that, losing any benefit that a tighter
607// representation interval might otherwise yield, while introducing some
608// numerical error and computational overhead.
609//
610// Now, Logistic and Tanh
611// are nearly constant (nearly equal to their horizontal asymptotes)
612// outside of a small bounded interval around 0:
613//
614// Logistic(4) = 1 - 1.8e-2 Tanh(4) = 1 - 6.7e-4
615// Logistic(8) = 1 - 3.4e-4 Tanh(8) = 1 - 2.3e-7
616// Logistic(16) = 1 - 1.1e-7 Tanh(16) = 1 - 2.5e-14
617//
618// From this, we see that clamping to [-4, 4] would be too inaccurate
619// (the error of 1.8e-2 on Logistic would be felt even in 8bit precision)
620// while clamping to [-16, 16] would make no difference even in float32.
621// However, for a fixed-point implementation in 16-bit integers, using 5
622// integer bits to represent the [-16, 16] range would leave only 11
623// fractional bits, giving an increment of 2^-11 = 4.9e-4 between consecutive
624// representable values. Notice that is higher than the
625// worst-case clamping error with clamping to [-8, 8]: 3.4e-4 for Logistic.
626// Using [-8, 8] thus seems like the better compromise overall, enjoying
627// an increment of 2.4e-4 between representable values and a worst-case
628// clamping error of 3.4e-4, both better than the increment of 4.9e-4 with
629// [-16, 16].
630//
631// Moreover, all other things being equal, it is nice to choose the narrower
632// representation range, as that makes the implementation of fixed-point
633// math functions a little cheaper (each integer bit requires an additional
634// barrel-shifter atep in the implementation of exp(-x)). That is further
635// reason to prefer [-8, 8] over [-16, 16]. The choice of [-16, 16] would make
636// sense for 32-bit float or 32-bit fixed-point quantization, but we are
637// aiming for 16-bit fixed-point quantization of these internal nodes here.
638//
639template <int StateIntegerBits>
640inline void LstmCell(const LstmCellParams& params,
641 const RuntimeShape& unextended_input_shape,
642 const uint8* input_data_uint8,
643 const RuntimeShape& unextended_prev_activ_shape,
644 const uint8* prev_activ_data_uint8,
645 const RuntimeShape& weights_shape,
646 const uint8* weights_data_uint8,
647 const RuntimeShape& unextended_bias_shape,
648 const int32* bias_data_int32,
649 const RuntimeShape& unextended_prev_state_shape,
650 const int16* prev_state_data_int16,
651 const RuntimeShape& unextended_output_state_shape,
652 int16* output_state_data_int16,
653 const RuntimeShape& unextended_output_activ_shape,
654 uint8* output_activ_data_uint8,
655 const RuntimeShape& unextended_concat_temp_shape,
656 uint8* concat_temp_data_uint8,
657 const RuntimeShape& unextended_activ_temp_shape,
658 int16* activ_temp_data_int16, void* gemmlowp_context) {
659 (void)gemmlowp_context; // only used in optimized code.
660 int32 weights_zero_point = params.weights_zero_point;
661 int32 accum_multiplier = params.accum_multiplier;
662 int accum_shift = params.accum_shift;
663 TFLITE_DCHECK_LE(unextended_input_shape.DimensionsCount(), 4);
664 TFLITE_DCHECK_LE(unextended_prev_activ_shape.DimensionsCount(), 4);
665 TFLITE_DCHECK_LE(unextended_bias_shape.DimensionsCount(), 4);
666 TFLITE_DCHECK_LE(unextended_prev_state_shape.DimensionsCount(), 4);
667 TFLITE_DCHECK_LE(unextended_output_state_shape.DimensionsCount(), 4);
668 TFLITE_DCHECK_LE(unextended_output_activ_shape.DimensionsCount(), 4);
669 TFLITE_DCHECK_LE(unextended_concat_temp_shape.DimensionsCount(), 4);
670 TFLITE_DCHECK_LE(unextended_activ_temp_shape.DimensionsCount(), 4);
671 const RuntimeShape input_shape =
672 RuntimeShape::ExtendedShape(4, unextended_input_shape);
673 const RuntimeShape prev_activ_shape =
674 RuntimeShape::ExtendedShape(4, unextended_prev_activ_shape);
675 const RuntimeShape bias_shape =
676 RuntimeShape::ExtendedShape(4, unextended_bias_shape);
677 const RuntimeShape prev_state_shape =
678 RuntimeShape::ExtendedShape(4, unextended_prev_state_shape);
679 const RuntimeShape output_state_shape =
680 RuntimeShape::ExtendedShape(4, unextended_output_state_shape);
681 const RuntimeShape output_activ_shape =
682 RuntimeShape::ExtendedShape(4, unextended_output_activ_shape);
683 const RuntimeShape concat_temp_shape =
684 RuntimeShape::ExtendedShape(4, unextended_concat_temp_shape);
685 const RuntimeShape activ_temp_shape =
686 RuntimeShape::ExtendedShape(4, unextended_activ_temp_shape);
687 TFLITE_DCHECK_GE(weights_shape.DimensionsCount(), 2);
688
689 // Gather dimensions information, and perform consistency checks.
690 const int weights_dim_count = weights_shape.DimensionsCount();
691 const int outer_size = MatchingFlatSizeSkipDim(
692 input_shape, 3, prev_activ_shape, prev_state_shape, output_state_shape,
693 output_activ_shape);
694 const int input_depth = input_shape.Dims(3);
695 const int prev_activ_depth = prev_activ_shape.Dims(3);
696 const int total_input_depth = prev_activ_depth + input_depth;
697 TFLITE_DCHECK_EQ(weights_shape.Dims(weights_dim_count - 1),
698 total_input_depth);
699 const int intern_activ_depth =
700 MatchingDim(weights_shape, weights_dim_count - 2, bias_shape, 3);
701 TFLITE_DCHECK_EQ(weights_shape.FlatSize(),
702 intern_activ_depth * total_input_depth);
703 TFLITE_DCHECK_EQ(FlatSizeSkipDim(bias_shape, 3), 1);
704 TFLITE_DCHECK_EQ(intern_activ_depth % 4, 0);
705 const int output_depth =
706 MatchingDim(prev_state_shape, 3, prev_activ_shape, 3, output_state_shape,
707 3, output_activ_shape, 3);
708 TFLITE_DCHECK_EQ(output_depth, intern_activ_depth / 4);
709 const int fc_batches = FlatSizeSkipDim(activ_temp_shape, 3);
710 const int fc_output_depth =
711 MatchingDim(weights_shape, weights_dim_count - 2, activ_temp_shape, 3);
712 const int fc_accum_depth = total_input_depth;
713 TFLITE_DCHECK_EQ(fc_output_depth, 4 * output_depth);
714
715 // Depth-concatenate prev_activ and input data together.
716 uint8 const* concat_input_arrays_data[2] = {input_data_uint8,
717 prev_activ_data_uint8};
718 const RuntimeShape* concat_input_arrays_shapes[2] = {&input_shape,
719 &prev_activ_shape};
720 tflite::ConcatenationParams concat_params;
721 concat_params.axis = 3;
722 concat_params.inputs_count = 2;
723 Concatenation(concat_params, concat_input_arrays_shapes,
724 concat_input_arrays_data, concat_temp_shape,
725 concat_temp_data_uint8);
726
727 // Implementation of the fully connected node inside the LSTM cell.
728 // The operands are 8-bit integers, the accumulators are internally 32bit
729 // integers, and the output is 16-bit fixed-point with 3 integer bits so
730 // the output range is [-2^3, 2^3] == [-8, 8]. The rationale for that
731 // is explained in the function comment above.
732 for (int b = 0; b < fc_batches; ++b) {
733 for (int out_c = 0; out_c < fc_output_depth; ++out_c) {
734 // Internal accumulation.
735 // Initialize accumulator with the bias-value.
736 int32 accum = bias_data_int32[out_c];
737 // Accumulation loop.
738 for (int d = 0; d < fc_accum_depth; ++d) {
739 int16 input_val = concat_temp_data_uint8[b * fc_accum_depth + d] - 128;
740 int16 weights_val =
741 weights_data_uint8[out_c * fc_accum_depth + d] - weights_zero_point;
742 accum += input_val * weights_val;
743 }
744 // Down-scale the final int32 accumulator to the scale used by our
745 // (16-bit, using 3 integer bits) fixed-point format. The quantized
746 // multiplier and shift here have been pre-computed offline
747 // (e.g. by toco).
748 accum =
749 MultiplyByQuantizedMultiplier(accum, accum_multiplier, accum_shift);
750 // Saturate, cast to int16, and store to the temporary activations array.
751 accum = std::max(-32768, std::min(32767, accum));
752 activ_temp_data_int16[out_c + fc_output_depth * b] = accum;
753 }
754 }
755
756 // Rest of the LSTM cell: tanh and logistic math functions, and some adds
757 // and muls, all done in 16-bit fixed-point.
758 for (int b = 0; b < outer_size; ++b) {
759 for (int c = 0; c < output_depth; ++c) {
760 // Define the fixed-point data types that we will use here. All use
761 // int16 as the underlying integer type i.e. all are 16-bit fixed-point.
762 // They only differ by the number of integral vs. fractional bits,
763 // determining the range of values that they can represent.
764 //
765 // F0 uses 0 integer bits, range [-1, 1].
766 // This is the return type of math functions such as tanh, logistic,
767 // whose range is in [-1, 1].
768 using F0 = gemmlowp::FixedPoint<std::int16_t, 0>;
769 // F3 uses 3 integer bits, range [-8, 8].
770 // This is the range of the previous fully-connected node's output,
771 // which is our input here.
772 using F3 = gemmlowp::FixedPoint<std::int16_t, 3>;
773 // FS uses StateIntegerBits integer bits, range [-2^StateIntegerBits,
774 // 2^StateIntegerBits]. It's used to represent the internal state, whose
775 // number of integer bits is currently dictated by the model. See comment
776 // on the StateIntegerBits template parameter above.
777 using FS = gemmlowp::FixedPoint<std::int16_t, StateIntegerBits>;
778 // Implementation of input gate, using fixed-point logistic function.
779 F3 input_gate_input = F3::FromRaw(
780 activ_temp_data_int16[b * fc_output_depth + 0 * output_depth + c]);
781 F0 input_gate_output = gemmlowp::logistic(input_gate_input);
782 // Implementation of input modulation gate, using fixed-point tanh
783 // function.
784 F3 input_modulation_gate_input = F3::FromRaw(
785 activ_temp_data_int16[b * fc_output_depth + 1 * output_depth + c]);
786 F0 input_modulation_gate_output =
787 gemmlowp::tanh(input_modulation_gate_input);
788 // Implementation of forget gate, using fixed-point logistic function.
789 F3 forget_gate_input = F3::FromRaw(
790 activ_temp_data_int16[b * fc_output_depth + 2 * output_depth + c]);
791 F0 forget_gate_output = gemmlowp::logistic(forget_gate_input);
792 // Implementation of output gate, using fixed-point logistic function.
793 F3 output_gate_input = F3::FromRaw(
794 activ_temp_data_int16[b * fc_output_depth + 3 * output_depth + c]);
795 F0 output_gate_output = gemmlowp::logistic(output_gate_input);
796 // Implementation of internal multiplication nodes, still in fixed-point.
797 F0 input_times_input_modulation =
798 input_gate_output * input_modulation_gate_output;
799 FS prev_state = FS::FromRaw(prev_state_data_int16[b * output_depth + c]);
800 FS prev_state_times_forget_state = forget_gate_output * prev_state;
801 // Implementation of internal addition node, saturating.
802 FS new_state = gemmlowp::SaturatingAdd(
803 gemmlowp::Rescale<StateIntegerBits>(input_times_input_modulation),
804 prev_state_times_forget_state);
805 // Implementation of last internal Tanh node, still in fixed-point.
806 // Since a Tanh fixed-point implementation is specialized for a given
807 // number or integer bits, and each specialization can have a substantial
808 // code size, and we already used above a Tanh on an input with 3 integer
809 // bits, and per the table in the above function comment there is no
810 // significant accuracy to be lost by clamping to [-8, +8] for a
811 // 3-integer-bits representation, let us just do that. This helps people
812 // porting this to targets where code footprint must be minimized.
813 F3 new_state_f3 = gemmlowp::Rescale<3>(new_state);
814 F0 output_activ_int16 = output_gate_output * gemmlowp::tanh(new_state_f3);
815 // Store the new internal state back to memory, as 16-bit integers.
816 // Note: here we store the original value with StateIntegerBits, not
817 // the rescaled 3-integer-bits value fed to tanh.
818 output_state_data_int16[b * output_depth + c] = new_state.raw();
819 // Down-scale the output activations to 8-bit integers, saturating,
820 // and store back to memory.
821 int16 rescaled_output_activ =
822 gemmlowp::RoundingDivideByPOT(output_activ_int16.raw(), 8);
823 int16 clamped_output_activ =
824 std::max<int16>(-128, std::min<int16>(127, rescaled_output_activ));
825 output_activ_data_uint8[b * output_depth + c] =
826 128 + clamped_output_activ;
827 }
828 }
829}

832void Split(const SplitParams& params, const RuntimeShape& input_shape,
833 const Scalar* input_data, const RuntimeShape* const* output_shapes,
834 Scalar* const* output_data) {
835 ruy::profiler::ScopeLabel label("Split");
836 const int split_dimensions = input_shape.DimensionsCount();
837 int axis = params.axis < 0 ? params.axis + split_dimensions : params.axis;
838 int outputs_count = params.num_split;
839 TFLITE_DCHECK_LT(axis, split_dimensions);
840
841 int64_t split_size = 0;
842 for (int i = 0; i < outputs_count; i++) {
843 TFLITE_DCHECK_EQ(output_shapes[i]->DimensionsCount(), split_dimensions);
844 for (int j = 0; j < split_dimensions; j++) {
845 if (j != axis) {
846 MatchingDim(*output_shapes[i], j, input_shape, j);
847 }
848 }
849 split_size += output_shapes[i]->Dims(axis);
850 }
851 TFLITE_DCHECK_EQ(split_size, input_shape.Dims(axis));
852 int64_t outer_size = 1;
853 for (int i = 0; i < axis; ++i) {
854 outer_size *= input_shape.Dims(i);
855 }
856 // For all output arrays,
857 // FlatSize() = outer_size * Dims(axis) * base_inner_size;
858 int64_t base_inner_size = 1;
859 for (int i = axis + 1; i < split_dimensions; ++i) {
860 base_inner_size *= input_shape.Dims(i);
861 }
862
863 const Scalar* input_ptr = input_data;
864 for (int k = 0; k < outer_size; k++) {
865 for (int i = 0; i < outputs_count; ++i) {
866 const int copy_size = output_shapes[i]->Dims(axis) * base_inner_size;
867 memcpy(output_data[i] + k * copy_size, input_ptr,
868 copy_size * sizeof(Scalar));
869 input_ptr += copy_size;
870 }
871 }

879 const tflite::LocalResponseNormalizationParams& op_params,
880 const RuntimeShape& input_shape, const float* input_data,
881 const RuntimeShape& output_shape, float* output_data) {
882 const int trailing_dim = input_shape.DimensionsCount() - 1;

884 MatchingFlatSizeSkipDim(input_shape, trailing_dim, output_shape);
885 const int depth =
886 MatchingDim(input_shape, trailing_dim, output_shape, trailing_dim);
887
888 for (int i = 0; i < outer_size; ++i) {
889 for (int c = 0; c < depth; ++c) {

894 const float input_val = input_data[i * depth + input_c];
895 accum += input_val * input_val;
896 }
897 const float multiplier =
898 std::pow(op_params.bias + op_params.alpha * accum, -op_params.beta);
899 output_data[i * depth + c] = input_data[i * depth + c] * multiplier;
900 }

904inline void LogSoftmax(const SoftmaxParams& params,
905 const RuntimeShape& input_shape, const float* input_data,
906 const RuntimeShape& output_shape, float* output_data) {
907 const int trailing_dim = input_shape.DimensionsCount() - 1;
908 const int outer_size =
909 MatchingFlatSizeSkipDim(input_shape, trailing_dim, output_shape);
910 const int depth =
911 MatchingDim(input_shape, trailing_dim, output_shape, trailing_dim);
912
913 for (int i = 0; i < outer_size; ++i) {
914 // Find max element value which we'll use to ensure numerical stability
915 // taking advantage of the following equality:
916 // log(exp(x[i])/sum(exp(x[i]))) == log(exp(x[i]+C)/sum(exp(x[i]+C)))
917 float max = std::numeric_limits<float>::lowest();
918 for (int c = 0; c < depth; ++c) {
919 max = std::max(max, input_data[i * depth + c]);
920 }
921
922 // Compute sum.
923 float sum = 0.f;
924 for (int c = 0; c < depth; ++c) {
925 sum += std::exp(input_data[i * depth + c] - max);
926 }
927
928 // Compute result.
929 const float log_sum = std::log(sum);
930 for (int c = 0; c < depth; ++c) {
931 output_data[i * depth + c] = input_data[i * depth + c] - max - log_sum;
932 }
933 }
934}

936inline void LogSoftmax(const SoftmaxParams& params,
937 const RuntimeShape& input_shape, const uint8* input_data,
938 const RuntimeShape& output_shape, uint8* output_data) {
939 ruy::profiler::ScopeLabel label("LogSoftmax/8bit");
940 const int32 input_multiplier = params.input_multiplier;
941 const int32 input_left_shift = params.input_left_shift;
942 const int32 reverse_scaling_divisor = params.reverse_scaling_divisor;
943 const int32 reverse_scaling_right_shift = params.reverse_scaling_right_shift;
944 const int diff_min = params.diff_min;
945 // The representation chosen for the input to the exp() function is Q5.26.
946 // We need to leave extra space since values that we skip might be as large
947 // as -32 before multiplying by input_beta_multiplier, and therefore as
948 // large as -16 afterwards. Note that exp(-8) is definitely not
949 // insignificant to accumulation, but exp(-16) definitely is.
950 static constexpr int kScaledDiffIntegerBits = 5;
951 static constexpr int kAccumulationIntegerBits = 12;
952 static constexpr int kOutputIntegerBits = 4;
953 using FixedPointScaledDiff =
954 gemmlowp::FixedPoint<int32, kScaledDiffIntegerBits>;
955 using FixedPointAccum = gemmlowp::FixedPoint<int32, kAccumulationIntegerBits>;
956
957 const int trailing_dim = input_shape.DimensionsCount() - 1;
958 const int outer_size =
959 MatchingFlatSizeSkipDim(input_shape, trailing_dim, output_shape);
960 const int depth =
961 MatchingDim(input_shape, trailing_dim, output_shape, trailing_dim);
962
963 for (int i = 0; i < outer_size; ++i) {
964 uint8 max_in_row = 0;
965 for (int c = 0; c < depth; ++c) {
966 max_in_row = std::max(max_in_row, input_data[i * depth + c]);
967 }
968
969 FixedPointAccum sum_of_exps = FixedPointAccum::Zero();
970 for (int c = 0; c < depth; ++c) {
971 int32 input_diff =
972 static_cast<int32>(input_data[i * depth + c]) - max_in_row;
973 if (input_diff >= diff_min) {
974 const int32 input_diff_rescaled =
975 MultiplyByQuantizedMultiplierGreaterThanOne(
976 input_diff, input_multiplier, input_left_shift);
977 const FixedPointScaledDiff scaled_diff_f8 =
978 FixedPointScaledDiff::FromRaw(input_diff_rescaled);
979 sum_of_exps = sum_of_exps + gemmlowp::Rescale<kAccumulationIntegerBits>(
980 exp_on_negative_values(scaled_diff_f8));
981 }
982 }
983
984 const int32 fixed_log_sum_of_exps =
985 log_x_for_x_greater_than_or_equal_to_1<kScaledDiffIntegerBits>(
986 sum_of_exps)
987 .raw();
988
989 // rescaled_diff_min is smallest representable in
990 // Q(kScaledDiffIntegerBits).(31-kScaledDiffIntegerBits) plus the
991 // log-sub-exps that will be subtracted in the loop.
992 //
993 // The thresholds diff_min, etc are negative.
994 const int rescaled_diff_min =
995 fixed_log_sum_of_exps + std::numeric_limits<int32>::lowest();
996 const int adjusted_diff_min =
997 std::max(diff_min - 1, // Note use of > below instead of >= above.
998 MultiplyByQuantizedMultiplierSmallerThanOneExp(
999 rescaled_diff_min, reverse_scaling_divisor,
1000 -reverse_scaling_right_shift));
1001
1002 for (int c = 0; c < depth; ++c) {
1003 int32 input_diff =
1004 static_cast<int32>(input_data[i * depth + c]) - max_in_row;
1005 if (input_diff > adjusted_diff_min) {
1006 const int32 input_diff_rescaled =
1007 MultiplyByQuantizedMultiplierGreaterThanOne(
1008 input_diff, input_multiplier, input_left_shift);
1009 int32 unsat_output =
1010 gemmlowp::RoundingDivideByPOT(
1011 (input_diff_rescaled - fixed_log_sum_of_exps),
1012 31 - kScaledDiffIntegerBits - kOutputIntegerBits) +
1013 255;
1014
1015 output_data[i * depth + c] = static_cast<uint8>(
1016 std::max(std::min(unsat_output, static_cast<int32>(255)), 0));
1017 } else {
1018 // Set output to smallest value.
1019 output_data[i * depth + c] = 0;
1020 }
1021 }
1022 }

1058// Common subroutine for both `GatherNd` and `GatherNdString`.
1059struct GatherNdHelperResult {
1060 int n_slices;
1061 int slice_size;
1062 int indices_nd;
1063 std::vector<int> dims_to_count;
1064};

1066// Returns common values being used on both `GatherNd` and `GatherNdString`.
1067inline GatherNdHelperResult GatherNdHelper(const RuntimeShape& params_shape,
1068 const RuntimeShape& indices_shape) {
1069 GatherNdHelperResult ret;
1070 ret.n_slices = 1;
1071 ret.slice_size = 1;
1072 const int indices_dims = indices_shape.DimensionsCount();
1073 ret.indices_nd = indices_shape.Dims(indices_dims - 1);
1074 const int params_dims = params_shape.DimensionsCount();
1075 for (int i = 0; i < indices_dims - 1; ++i) {
1076 ret.n_slices *= indices_shape.Dims(i);
1077 }
1078 for (int i = ret.indices_nd; i < params_dims; ++i) {
1079 ret.slice_size *= params_shape.Dims(i);
1080 }
1081
1082 int remain_flat_size = params_shape.FlatSize();
1083 ret.dims_to_count = std::vector<int>(ret.indices_nd, 0);
1084 for (int i = 0; i < ret.indices_nd; ++i) {
1085 ret.dims_to_count[i] = remain_flat_size / params_shape.Dims(i);
1086 remain_flat_size = ret.dims_to_count[i];
1087 }
1088
1089 return ret;
1090}

1194template <typename T>
1195inline void ResizeBilinear(const tflite::ResizeBilinearParams& op_params,
1196 const RuntimeShape& unextended_input_shape,
1197 const T* input_data,
1198 const RuntimeShape& unextended_output_size_shape,
1199 const int32* output_size_data,
1200 const RuntimeShape& unextended_output_shape,
1201 T* output_data) {
1202 // If half_pixel_centers is True, align_corners must be False.
1203 TFLITE_DCHECK(!op_params.half_pixel_centers || !op_params.align_corners);
1204 TFLITE_DCHECK_LE(unextended_input_shape.DimensionsCount(), 4);
1205 TFLITE_DCHECK_LE(unextended_output_size_shape.DimensionsCount(), 4);
1206 TFLITE_DCHECK_LE(unextended_output_shape.DimensionsCount(), 4);
1207 const RuntimeShape input_shape =
1208 RuntimeShape::ExtendedShape(4, unextended_input_shape);
1209 const RuntimeShape output_size_shape =
1210 RuntimeShape::ExtendedShape(4, unextended_output_size_shape);
1211 const RuntimeShape output_shape =
1212 RuntimeShape::ExtendedShape(4, unextended_output_shape);
1213
1214 int32 batches = MatchingDim(input_shape, 0, output_shape, 0);
1215 int32 input_height = input_shape.Dims(1);
1216 int32 input_width = input_shape.Dims(2);
1217 int32 depth = MatchingDim(input_shape, 3, output_shape, 3);
1218
1219 TFLITE_DCHECK_EQ(output_size_shape.Dims(0), 1);
1220 TFLITE_DCHECK_EQ(output_size_shape.Dims(1), 1);
1221 TFLITE_DCHECK_EQ(output_size_shape.Dims(2), 1);
1222 TFLITE_DCHECK_EQ(output_size_shape.Dims(3), 2);
1223 int32 output_height = output_size_data[Offset(output_size_shape, 0, 0, 0, 0)];
1224 int32 output_width = output_size_data[Offset(output_size_shape, 0, 0, 0, 1)];
1225
1226 float height_scale = static_cast<float>(input_height) / output_height;
1227 float width_scale = static_cast<float>(input_width) / output_width;
1228 if (op_params.align_corners && output_height > 1) {
1229 height_scale = static_cast<float>(input_height - 1) / (output_height - 1);
1230 }
1231 if (op_params.align_corners && output_width > 1) {
1232 width_scale = static_cast<float>(input_width - 1) / (output_width - 1);
1233 }
1234 const float rounding_offset = std::numeric_limits<T>::is_integer ? .5f : .0f;
1235
1236 for (int b = 0; b < batches; ++b) {
1237 for (int y = 0; y < output_height; ++y) {
1238 float input_y;
1239 int32 y0, y1;
1240 ComputeInterpolationValues(y, height_scale, op_params.half_pixel_centers,
1241 input_height, &input_y, &y0, &y1);
1242 for (int x = 0; x < output_width; ++x) {
1243 float input_x;
1244 int32 x0, x1;
1245 ComputeInterpolationValues(x, width_scale, op_params.half_pixel_centers,
1246 input_width, &input_x, &x0, &x1);
1247 for (int c = 0; c < depth; ++c) {
1248 T interpolation =
1249 static_cast<T>(input_data[Offset(input_shape, b, y0, x0, c)] *
1250 (1 - (input_y - y0)) * (1 - (input_x - x0)) +
1251 input_data[Offset(input_shape, b, y1, x0, c)] *
1252 (input_y - y0) * (1 - (input_x - x0)) +
1253 input_data[Offset(input_shape, b, y0, x1, c)] *
1254 (1 - (input_y - y0)) * (input_x - x0) +
1255 input_data[Offset(input_shape, b, y1, x1, c)] *
1256 (input_y - y0) * (input_x - x0) +
1257 rounding_offset);
1258 output_data[Offset(output_shape, b, y, x, c)] = interpolation;
1259 }
1260 }
1261 }
1262 }
1263}

1290 TFLITE_DCHECK_LE(unextended_output_size_shape.DimensionsCount(), 4);
1291 TFLITE_DCHECK_LE(unextended_output_shape.DimensionsCount(), 4);

1295 RuntimeShape::ExtendedShape(4, unextended_output_size_shape);

1297 RuntimeShape::ExtendedShape(4, unextended_output_shape);
1298
1299 const int32 batches = MatchingDim(input_shape, 0, output_shape, 0);
1300 const int32 input_height = input_shape.Dims(1);
1301 const int32 input_width = input_shape.Dims(2);
1302 const int32 depth = MatchingDim(input_shape, 3, output_shape, 3);
1303
1304 TFLITE_DCHECK_EQ(output_size_shape.Dims(0), 1);
1305 TFLITE_DCHECK_EQ(output_size_shape.Dims(1), 1);
1306 TFLITE_DCHECK_EQ(output_size_shape.Dims(2), 1);
1307 TFLITE_DCHECK_EQ(output_size_shape.Dims(3), 2);
1308 const int32 output_height =
1309 output_size_data[Offset(output_size_shape, 0, 0, 0, 0)];

1311 output_size_data[Offset(output_size_shape, 0, 0, 0, 1)];

1314 ((1 << 10) * input_height + output_height / 2) / output_height;

1317 if (op_params.align_corners && output_height > 1) {
1318 height_scale_10 =
1319 ((1 << 10) * (input_height - 1) + (output_height - 1) / 2) /
1320 (output_height - 1);
1321 }
1322 if (op_params.align_corners && output_width > 1) {
1323 width_scale_10 = ((1 << 10) * (input_width - 1) + (output_width - 1) / 2) /
1324 (output_width - 1);
1325 }

1328 for (int y = 0; y < output_height; ++y) {

1331 op_params.half_pixel_centers,
1332 input_height, &input_y, &y0, &y1);
1333 for (int x = 0; x < output_width; ++x) {

1335 ComputeInterpolationValuesInteger(x, width_scale_10,
1336 op_params.half_pixel_centers,
1337 input_width, &input_x, &x0, &x1);
1338 for (int c = 0; c < depth; ++c) {
1339 const int64_t output_20_ll =
1340 static_cast<int64_t>(
1341 input_data[Offset(input_shape, b, y0, x0, c)]) *
1342 ((1 << 10) - (input_y - (1 << 10) * y0)) *
1343 ((1 << 10) - (input_x - (1 << 10) * x0));
1344 const int64_t output_20_lu =
1345 static_cast<int64_t>(
1346 input_data[Offset(input_shape, b, y1, x0, c)]) *
1347 (input_y - (1 << 10) * y0) *
1348 ((1 << 10) - (input_x - (1 << 10) * x0));
1349 const int64_t output_20_rl =
1350 static_cast<int64_t>(
1351 input_data[Offset(input_shape, b, y0, x1, c)]) *
1352 ((1 << 10) - (input_y - (1 << 10) * y0)) *

1354 const int64_t output_20_ru =
1355 static_cast<int64_t>(
1356 input_data[Offset(input_shape, b, y1, x1, c)]) *
1357 (input_y - (1 << 10) * y0) * (input_x - (1 << 10) * x0);

1359 output_20_ll + output_20_lu + output_20_rl + output_20_ru;

1362 static_cast<T>((output_20 + round) / (1 << 20));

1370template <typename T>
1371inline void Slice(const tflite::SliceParams& op_params,
1372 const RuntimeShape& input_shape,
1373 const RuntimeShape& output_shape,
1374 SequentialTensorWriter<T>* writer) {
1375 const RuntimeShape ext_shape = RuntimeShape::ExtendedShape(5, input_shape);
1376 TFLITE_DCHECK_LE(op_params.begin_count, 5);
1377 TFLITE_DCHECK_LE(op_params.size_count, 5);
1378 const int begin_count = op_params.begin_count;
1379 const int size_count = op_params.size_count;
1380 // We front-pad the begin and size vectors.
1381 std::array<int, 5> start;
1382 std::array<int, 5> stop;
1383 for (int i = 0; i < 5; ++i) {
1384 int padded_i = 5 - i;
1385 start[i] =
1386 begin_count < padded_i ? 0 : op_params.begin[begin_count - padded_i];
1387 stop[i] =
1388 (size_count < padded_i || op_params.size[size_count - padded_i] == -1)
1389 ? ext_shape.Dims(i)
1390 : start[i] + op_params.size[size_count - padded_i];
1391 }
1392
1393 for (int i0 = start[0]; i0 < stop[0]; ++i0) {
1394 for (int i1 = start[1]; i1 < stop[1]; ++i1) {
1395 for (int i2 = start[2]; i2 < stop[2]; ++i2) {
1396 for (int i3 = start[3]; i3 < stop[3]; ++i3) {
1397 for (int i4 = start[4]; i4 < stop[4]; ++i4) {
1398 writer->Write(Offset(ext_shape, i0, i1, i2, i3, i4));
1399 }
1400 }
1401 }
1402 }
1403 }

1524template <typename D, typename T>
1525void BroadcastSelect4DSlow(const RuntimeShape& input_condition_shape,
1526 const D* input_condition_data,
1527 const RuntimeShape& input_x_shape,
1528 const T* input_x_data,
1529 const RuntimeShape& input_y_shape,
1530 const T* input_y_data,
1531 const RuntimeShape& output_shape, T* output_data) {
1532 TFLITE_DCHECK_LE(input_condition_shape.DimensionsCount(), 4);
1533 TFLITE_DCHECK_LE(input_x_shape.DimensionsCount(), 4);
1534 TFLITE_DCHECK_LE(input_y_shape.DimensionsCount(), 4);
1535 TFLITE_DCHECK_LE(output_shape.DimensionsCount(), 4);
1536
1537 const RuntimeShape extended_output_shape =
1538 RuntimeShape::ExtendedShape(4, output_shape);
1539
1540 NdArrayDesc<4> desc_condition;
1541 NdArrayDesc<4> desc_x;
1542 NdArrayDesc<4> desc_y;
1543 NdArrayDescsForElementwiseBroadcast(input_condition_shape, input_x_shape,
1544 input_y_shape, &desc_condition, &desc_x,
1545 &desc_y);
1546
1547 // In Tensorflow, the dimensions are canonically named (batch_number, row,
1548 // col, channel), with extents (batches, height, width, depth), with the
1549 // trailing dimension changing most rapidly (channels has the smallest
1550 // stride, typically 1 element).
1551 //
1552 // In generated C code, we store arrays with the dimensions reversed. The
1553 // first dimension has smallest stride.
1554 //
1555 // We name our variables by their Tensorflow convention, but generate C code
1556 // nesting loops such that the innermost loop has the smallest stride for
1557 // the best cache behavior.
1558 for (int b = 0; b < extended_output_shape.Dims(0); ++b) {
1559 for (int y = 0; y < extended_output_shape.Dims(1); ++y) {
1560 for (int x = 0; x < extended_output_shape.Dims(2); ++x) {
1561 for (int c = 0; c < extended_output_shape.Dims(3); ++c) {
1562 const int condition_index =
1563 SubscriptToIndex(desc_condition, b, y, x, c);
1564 const int x_index = SubscriptToIndex(desc_x, b, y, x, c);
1565 const int y_index = SubscriptToIndex(desc_y, b, y, x, c);
1566 output_data[Offset(extended_output_shape, b, y, x, c)] =
1567 input_condition_data[condition_index] ? input_x_data[x_index]
1568 : input_y_data[y_index];
1569 }
1570 }
1571 }
1572 }

1575template <typename D, typename T>
1576void SelectTrueCoords(const RuntimeShape& input_condition_shape,
1577 const D* input_condition_data, T* output_data) {
1578 const size_t size = input_condition_shape.FlatSize();
1579 if (size == 0) {
1580 // Dimension is zero, in which case we don't need to output.
1581 return;
1582 }
1583 const size_t cond_rank = input_condition_shape.DimensionsCount();
1584
1585 std::vector<int> dims_to_count(cond_rank, 0);
1586 int cur_flat_size = size;
1587 for (int i = 0; i < cond_rank; ++i) {
1588 dims_to_count[i] = cur_flat_size / input_condition_shape.Dims(i);
1589 cur_flat_size = dims_to_count[i];
1590 }
1591
1592 int output_index = 0;
1593 for (int i = 0; i < size; ++i) {
1594 if (input_condition_data[i]) {
1595 // Insert the coordinate of the current item (row major) into output.
1596 int flat_index = i;
1597 for (int j = 0; j < cond_rank; ++j) {
1598 int coord_j = flat_index / dims_to_count[j];
1599 output_data[output_index * cond_rank + j] = coord_j;
1600 flat_index %= dims_to_count[j];
1601 }
1602 output_index++;
1603 }
1604 }
1605}

1701 outer_size *= input_shape.Dims(i);
1702 }

1705 for (int i = axis + 1; i < input_shape.DimensionsCount(); ++i) {

1710 for (int i = 0; i < outer_size; ++i) {
1711 for (int j = 0; j < dims_at_axis; ++j) {
1712 const int start_pos = (i * dims_at_axis + j) * copy_size;
1713 Scalar* output_ptr = output_data + start_pos;
1714 int loc = (i * dims_at_axis + dims_at_axis - j - 1) * copy_size;
1715 memcpy(output_ptr, input_data + loc, copy_size * sizeof(Scalar));

1720template <typename Scalar, typename TS>
1721void ReverseSequence(const TS* seq_lengths, const int seq_dim,
1722 const int batch_dim, const RuntimeShape& input_shape,
1723 const Scalar* input_data, const RuntimeShape& output_shape,
1724 Scalar* output_data) {
1725 ruy::profiler::ScopeLabel label("ReverseSequence");
1726
1727 int outer_size = 1;
1728 int outer_dim = std::min(batch_dim, seq_dim);
1729 int medium_dim = std::max(batch_dim, seq_dim);
1730 for (int i = 0; i < outer_dim; ++i) {
1731 outer_size *= input_shape.Dims(i);
1732 }
1733
1734 int medium_size = 1;
1735 for (int i = outer_dim + 1; i < medium_dim; ++i) {
1736 medium_size *= input_shape.Dims(i);
1737 }
1738
1739 int copy_size = 1;
1740 for (int i = medium_dim + 1; i < input_shape.DimensionsCount(); ++i) {
1741 copy_size *= input_shape.Dims(i);
1742 }
1743
1744 const int dims_at_outer_dim = input_shape.Dims(outer_dim);
1745 const int dims_at_medium_dim = input_shape.Dims(medium_dim);
1746
1747 Scalar* output_ptr;
1748 if (batch_dim > seq_dim) {
1749 for (int i = 0; i < outer_size; ++i) {
1750 for (int j = 0; j < dims_at_outer_dim; ++j) {
1751 const int in_pos_base = (i * dims_at_outer_dim + j) * medium_size;
1752 for (int p = 0; p < medium_size; ++p) {
1753 for (int q = 0; q < dims_at_medium_dim; ++q) {
1754 const int in_pos =
1755 ((in_pos_base + p) * dims_at_medium_dim + q) * copy_size;
1756 const Scalar* in_ptr = input_data + in_pos;
1757 int sl = seq_lengths[q] - 1;
1758 if (j > sl) {
1759 output_ptr = output_data + in_pos;
1760 } else {
1761 const int out_pos_base =
1762 (i * dims_at_outer_dim + sl - j) * medium_size;
1763 const int out_pos =
1764 ((out_pos_base + p) * dims_at_medium_dim + q) * copy_size;
1765 output_ptr = output_data + out_pos;
1766 }
1767 memcpy(output_ptr, in_ptr, copy_size * sizeof(Scalar));
1768 }
1769 }
1770 }
1771 }
1772 } else if (batch_dim < seq_dim) {
1773 for (int i = 0; i < outer_size; ++i) {
1774 for (int j = 0; j < dims_at_outer_dim; ++j) {
1775 const int in_pos_base = (i * dims_at_outer_dim + j) * medium_size;
1776 int sl = seq_lengths[j] - 1;
1777 const int out_pos_base = (i * dims_at_outer_dim + j) * medium_size;
1778 for (int p = 0; p < medium_size; ++p) {
1779 for (int q = 0; q < dims_at_medium_dim; ++q) {
1780 const int in_pos =
1781 ((in_pos_base + p) * dims_at_medium_dim + q) * copy_size;
1782 const Scalar* in_ptr = input_data + in_pos;
1783 if (q > sl) {
1784 output_ptr = output_data + in_pos;
1785 } else {
1786 const int out_pos =
1787 ((out_pos_base + p) * dims_at_medium_dim + sl - q) *
1788 copy_size;
1789 output_ptr = output_data + out_pos;
1790 }
1791 memcpy(output_ptr, in_ptr, copy_size * sizeof(Scalar));
1792 }
1793 }
1794 }
1795 }
