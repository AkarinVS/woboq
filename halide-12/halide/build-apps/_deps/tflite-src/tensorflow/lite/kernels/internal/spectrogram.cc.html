29// Returns the default Hann window function for the spectrogram.
30void GetPeriodicHann(int window_length, std::vector<double>* window) {
31 // Some platforms don't have M_PI, so define a local constant here.
32 const double pi = std::atan(1.0) * 4.0;
33 window->resize(window_length);
34 for (int i = 0; i < window_length; ++i) {
35 (*window)[i] = 0.5 - 0.5 * cos((2.0 * pi * i) / window_length);
36 }
37}
38} // namespace

40bool Spectrogram::Initialize(int window_length, int step_length) {
41 std::vector<double> window;
42 GetPeriodicHann(window_length, &window);
43 return Initialize(window, step_length);
44}

46inline int Log2Floor(uint32_t n) {
47 if (n == 0) return -1;
48 int log = 0;
49 uint32_t value = n;
50 for (int i = 4; i >= 0; --i) {
51 int shift = (1 << i);
52 uint32_t x = value >> shift;
53 if (x != 0) {
54 value = x;
55 log += shift;
56 }
57 }
58 return log;
59}
60
61inline int Log2Ceiling(uint32_t n) {
62 int floor = Log2Floor(n);
63 if (n == (n & ~(n - 1))) // zero or a power of two
64 return floor;
65 else
66 return floor + 1;
67}
68
69inline uint32_t NextPowerOfTwo(uint32_t value) {
70 int exponent = Log2Ceiling(value);
71 // DCHECK_LT(exponent, std::numeric_limits<uint32>::digits);
72 return 1 << exponent;
73}

75bool Spectrogram::Initialize(const std::vector<double>& window,
76 int step_length) {
77 window_length_ = window.size();
78 window_ = window; // Copy window.
79 if (window_length_ < 2) {
80 // LOG(ERROR) << "Window length too short.";
81 initialized_ = false;
82 return false;
83 }
84
85 step_length_ = step_length;
86 if (step_length_ < 1) {
87 // LOG(ERROR) << "Step length must be positive.";
88 initialized_ = false;
89 return false;
90 }
91
92 fft_length_ = NextPowerOfTwo(window_length_);
93 // CHECK(fft_length_ >= window_length_);
94 output_frequency_channels_ = 1 + fft_length_ / 2;
95
96 // Allocate 2 more than what rdft needs, so we can rationalize the layout.
97 fft_input_output_.assign(fft_length_ + 2, 0.0);
98
99 int half_fft_length = fft_length_ / 2;
100 fft_double_working_area_.assign(half_fft_length, 0.0);
101 fft_integer_working_area_.assign(2 + static_cast<int>(sqrt(half_fft_length)),
102 0);
103 // Set flag element to ensure that the working areas are initialized
104 // on the first call to cdft. It's redundant given the assign above,
105 // but keep it as a reminder.
106 fft_integer_working_area_[0] = 0;
107 input_queue_.clear();
108 samples_to_next_step_ = window_length_;
109 initialized_ = true;
110 return true;
111}

113template <class InputSample, class OutputSample>
114bool Spectrogram::ComputeComplexSpectrogram(
115 const std::vector<InputSample>& input,
116 std::vector<std::vector<complex<OutputSample>>>* output) {
117 if (!initialized_) {
118 // LOG(ERROR) << "ComputeComplexSpectrogram() called before successful call
119 // "
120 // << "to Initialize().";
121 return false;
122 }
123 // CHECK(output);
124 output->clear();
125 int input_start = 0;
126 while (GetNextWindowOfSamples(input, &input_start)) {
127 // DCHECK_EQ(input_queue_.size(), window_length_);
128 ProcessCoreFFT(); // Processes input_queue_ to fft_input_output_.
129 // Add a new slice vector onto the output, to save new result to.
130 output->resize(output->size() + 1);
131 // Get a reference to the newly added slice to fill in.
132 auto& spectrogram_slice = output->back();
133 spectrogram_slice.resize(output_frequency_channels_);
134 for (int i = 0; i < output_frequency_channels_; ++i) {
135 // This will convert double to float if it needs to.
136 spectrogram_slice[i] = complex<OutputSample>(
137 fft_input_output_[2 * i], fft_input_output_[2 * i + 1]);
138 }
139 }
140 return true;
141}

155template <class InputSample, class OutputSample>
156bool Spectrogram::ComputeSquaredMagnitudeSpectrogram(
157 const std::vector<InputSample>& input,
158 std::vector<std::vector<OutputSample>>* output) {
159 if (!initialized_) {
160 // LOG(ERROR) << "ComputeSquaredMagnitudeSpectrogram() called before "
161 // << "successful call to Initialize().";
162 return false;
163 }
164 // CHECK(output);
165 output->clear();
166 int input_start = 0;
167 while (GetNextWindowOfSamples(input, &input_start)) {
168 // DCHECK_EQ(input_queue_.size(), window_length_);
169 ProcessCoreFFT(); // Processes input_queue_ to fft_input_output_.
170 // Add a new slice vector onto the output, to save new result to.
171 output->resize(output->size() + 1);
172 // Get a reference to the newly added slice to fill in.
173 auto& spectrogram_slice = output->back();
174 spectrogram_slice.resize(output_frequency_channels_);
175 for (int i = 0; i < output_frequency_channels_; ++i) {
176 // Similar to the Complex case, except storing the norm.
177 // But the norm function is known to be a performance killer,
178 // so do it this way with explicit real and imaginary temps.
179 const double re = fft_input_output_[2 * i];
180 const double im = fft_input_output_[2 * i + 1];
181 // Which finally converts double to float if it needs to.
182 spectrogram_slice[i] = re * re + im * im;
183 }
184 }
185 return true;
186}

197// Return true if a full window of samples is prepared; manage the queue.
198template <class InputSample>
199bool Spectrogram::GetNextWindowOfSamples(const std::vector<InputSample>& input,
200 int* input_start) {
201 auto input_it = input.begin() + *input_start;
202 int input_remaining = input.end() - input_it;
203 if (samples_to_next_step_ > input_remaining) {
204 // Copy in as many samples are left and return false, no full window.
205 input_queue_.insert(input_queue_.end(), input_it, input.end());
206 *input_start += input_remaining; // Increases it to input.size().
207 samples_to_next_step_ -= input_remaining;
208 return false; // Not enough for a full window.
209 } else {
210 // Copy just enough into queue to make a new window, then trim the
211 // front off the queue to make it window-sized.
212 input_queue_.insert(input_queue_.end(), input_it,
213 input_it + samples_to_next_step_);
214 *input_start += samples_to_next_step_;
215 input_queue_.erase(
216 input_queue_.begin(),
217 input_queue_.begin() + input_queue_.size() - window_length_);
218 // DCHECK_EQ(window_length_, input_queue_.size());
219 samples_to_next_step_ = step_length_; // Be ready for next time.
220 return true; // Yes, input_queue_ now contains exactly a window-full.
221 }
222}

224void Spectrogram::ProcessCoreFFT() {
225 for (int j = 0; j < window_length_; ++j) {
226 fft_input_output_[j] = input_queue_[j] * window_[j];
227 }
228 // Zero-pad the rest of the input buffer.
229 for (int j = window_length_; j < fft_length_; ++j) {
230 fft_input_output_[j] = 0.0;
231 }
232 const int kForwardFFT = 1; // 1 means forward; -1 reverse.
233 // This real FFT is a fair amount faster than using cdft here.
234 rdft(fft_length_, kForwardFFT, &fft_input_output_[0],
235 &fft_integer_working_area_[0], &fft_double_working_area_[0]);
236 // Make rdft result look like cdft result;
237 // unpack the last real value from the first position's imag slot.
238 fft_input_output_[fft_length_] = fft_input_output_[1];
239 fft_input_output_[fft_length_ + 1] = 0;
240 fft_input_output_[1] = 0;
241}
