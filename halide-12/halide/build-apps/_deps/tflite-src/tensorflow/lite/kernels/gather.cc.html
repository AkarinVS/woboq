31constexpr int kInputTensor = 0;
32constexpr int kInputPositions = 1;
33constexpr int kOutputTensor = 0;
34
35TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
36 TF_LITE_ENSURE_EQ(context, NumInputs(node), 2);
37 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
38
39 const auto* params =
40 reinterpret_cast<const TfLiteGatherParams*>(node->builtin_data);
41 const TfLiteTensor* input;
42 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
43 const TfLiteTensor* positions;
44 TF_LITE_ENSURE_OK(context,
45 GetInputSafe(context, node, kInputPositions, &positions));
46 TfLiteTensor* output;
47 TF_LITE_ENSURE_OK(context,
48 GetOutputSafe(context, node, kOutputTensor, &output));
49
50 switch (positions->type) {
51 case kTfLiteInt64:
52 case kTfLiteInt32:
53 break;
54 default:
55 context->ReportError(
56 context, "Positions of type '%s' are not supported by gather.",
57 TfLiteTypeGetName(positions->type));
58 return kTfLiteError;
59 }
60
61 // Assign to output the input type.
62 output->type = input->type;
63
64 // Check conditions for different types.
65 switch (input->type) {
66 case kTfLiteFloat32:
67 case kTfLiteUInt8:
68 case kTfLiteInt8:
69 case kTfLiteInt16:
70 case kTfLiteInt64:
71 case kTfLiteInt32:
72 case kTfLiteBool:
73 break;
74 case kTfLiteString: {
75 // Only 1D input is supported.
76 TF_LITE_ENSURE_EQ(context, NumDimensions(input), 1);
77 } break;
78 default:
79 context->ReportError(context, "Type '%s' is not supported by gather.",
80 TfLiteTypeGetName(input->type));
81 return kTfLiteError;
82 }
83
84 int axis = params->axis;
85 if (axis < 0) {
86 axis += NumDimensions(input);
87 }
88 TF_LITE_ENSURE(context, 0 <= axis && axis < NumDimensions(input));
89
90 int batch_dims = params->batch_dims;
91 // batch_dims should be in range: [-rank(positions), rank(positions)].
92 // Negative batch_dims is added with rank of positions.
93 if (batch_dims < 0) {
94 batch_dims += NumDimensions(positions);
95 }
96 TF_LITE_ENSURE(context, batch_dims <= axis);
97 TF_LITE_ENSURE(context, 0 <= batch_dims && batch_dims < NumDimensions(input));
98 TF_LITE_ENSURE(context, batch_dims <= NumDimensions(positions));
99 for (int i = 0; i < batch_dims; ++i) {
100 TF_LITE_ENSURE_EQ(context, input->dims->data[i], positions->dims->data[i]);
101 }
102
103 const int num_dimensions =
104 NumDimensions(input) + NumDimensions(positions) - 1 - batch_dims;
105 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(num_dimensions);
106 int output_index = 0;
107 for (int i = 0; i < axis; ++i) {
108 output_shape->data[output_index++] = input->dims->data[i];
109 }
110 for (int i = batch_dims; i < positions->dims->size; ++i) {
111 output_shape->data[output_index++] = positions->dims->data[i];
112 }
113 for (int i = axis + 1; i < input->dims->size; ++i) {
114 output_shape->data[output_index++] = input->dims->data[i];
115 }
116 return context->ResizeTensor(context, output, output_shape);
117}

119template <typename InputT, typename PositionsT>
120TfLiteStatus Gather(const TfLiteGatherParams& params, const TfLiteTensor* input,
121 const TfLiteTensor* positions, TfLiteTensor* output) {
122 tflite::GatherParams op_params;
123 op_params.axis = params.axis;
124 op_params.batch_dims = params.batch_dims;
125 optimized_ops::Gather(op_params, GetTensorShape(input),
126 GetTensorData<InputT>(input), GetTensorShape(positions),
127 GetTensorData<PositionsT>(positions),
128 GetTensorShape(output), GetTensorData<InputT>(output));
129 return kTfLiteOk;
130}

132template <typename PositionT>
133TfLiteStatus GatherStrings(TfLiteContext* context, const TfLiteTensor* input,
134 const TfLiteTensor* positions,
135 TfLiteTensor* output) {
136 DynamicBuffer buffer;
137 const PositionT* indexes = GetTensorData<PositionT>(positions);
138 const PositionT num_strings = GetStringCount(input);
139 const int num_indexes = NumElements(positions);
140
141 for (int i = 0; i < num_indexes; ++i) {
142 const PositionT pos = indexes[i];
143 TF_LITE_ENSURE(context, pos < num_strings);
144 const auto string_ref = GetString(input, pos);
145 buffer.AddString(string_ref.str, string_ref.len);
146 }
147 buffer.WriteToTensor(output, /*new_shape=*/nullptr);
148 return kTfLiteOk;
149}
150
151TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
152 const auto* params =
153 reinterpret_cast<const TfLiteGatherParams*>(node->builtin_data);
154 const TfLiteTensor* input;
155 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputTensor, &input));
156 const TfLiteTensor* positions;
157 TF_LITE_ENSURE_OK(context,
158 GetInputSafe(context, node, kInputPositions, &positions));
159 TfLiteTensor* output;
160 TF_LITE_ENSURE_OK(context,
161 GetOutputSafe(context, node, kOutputTensor, &output));
162
163 if (positions->type == kTfLiteInt32) {
164 switch (input->type) {
165 case kTfLiteFloat32:
166 return Gather<float, int32_t>(*params, input, positions, output);
167 case kTfLiteUInt8:
168 return Gather<uint8_t, int32_t>(*params, input, positions, output);
169 case kTfLiteInt8:
170 return Gather<int8_t, int32_t>(*params, input, positions, output);
171 case kTfLiteInt16:
172 return Gather<int16_t, int32_t>(*params, input, positions, output);
173 case kTfLiteInt32:
174 return Gather<int32_t, int32_t>(*params, input, positions, output);
175 case kTfLiteInt64:
176 return Gather<int64_t, int32_t>(*params, input, positions, output);
177 case kTfLiteBool:
178 return Gather<bool, int32_t>(*params, input, positions, output);
179 case kTfLiteString:
180 return GatherStrings<int32_t>(context, input, positions, output);
181 default:
182 context->ReportError(context, "Type '%s' is not supported by gather.",
183 TfLiteTypeGetName(input->type));
184 return kTfLiteError;
185 }
186 }
187 if (positions->type == kTfLiteInt64) {
188 switch (input->type) {
189 case kTfLiteFloat32:
190 return Gather<float, int64_t>(*params, input, positions, output);
191 case kTfLiteUInt8:
192 return Gather<uint8_t, int64_t>(*params, input, positions, output);
193 case kTfLiteInt8:
194 return Gather<int8_t, int64_t>(*params, input, positions, output);
195 case kTfLiteInt16:
196 return Gather<int16_t, int64_t>(*params, input, positions, output);
197 case kTfLiteInt32:
198 return Gather<int32_t, int64_t>(*params, input, positions, output);
199 case kTfLiteInt64:
200 return Gather<int64_t, int64_t>(*params, input, positions, output);
201 case kTfLiteBool:
202 return Gather<bool, int64_t>(*params, input, positions, output);
203 case kTfLiteString:
204 return GatherStrings<int64_t>(context, input, positions, output);
205 default:
206 context->ReportError(context, "Type '%s' is not supported by gather.",
207 TfLiteTypeGetName(input->type));
208 return kTfLiteError;
209 }
210 }
211 context->ReportError(context,
212 "Positions of type '%s' are not supported by gather.",
213 TfLiteTypeGetName(positions->type));
214 return kTfLiteError;
215}

218TfLiteRegistration* Register_GATHER() {
219 static TfLiteRegistration r = {nullptr, nullptr, gather::Prepare,
220 gather::Eval};
