45struct OpData {
46 int32_t multiplier;
47 int32_t shift;
48 int input_offset;
49 int output_offset;
50 bool needs_rescale;
51};

53bool IsNumericSupportedType(const TfLiteType type) {
54 return type == kTfLiteFloat32;
55}
56
57bool IsLogicalSupportedType(const TfLiteType type) {
58 return type == kTfLiteBool;
59}
60
61bool IsAbsSupportedType(const TfLiteType type) {
62 return type == kTfLiteFloat32 || type == kTfLiteInt8 || type == kTfLiteInt16;
63}
64
65bool IsRsqrtSupportedType(const TfLiteType type) {
66 return type == kTfLiteFloat32 || type == kTfLiteInt8;
67}

69inline void SetAbsOutputMultiplier(const float input_scale,
70 const float output_scale,
71 int32_t* multiplier, int32_t* shift) {
72 QuantizeMultiplier(input_scale / output_scale, multiplier, shift);
73}
74
75inline void SetRsqrtOutputMultiplier(const float input_scale,
76 const float output_scale,
77 int32_t* multiplier, int32_t* shift) {
78 const double scale = 1. / (std::sqrt(input_scale) * output_scale);
79 QuantizeMultiplier(scale, multiplier, shift);
80}

82typedef bool (*IsSupportedType)(TfLiteType);
83template <IsSupportedType is_supported_type, const char* op_name>
84TfLiteStatus GenericPrepare(TfLiteContext* context, TfLiteNode* node) {
85 TF_LITE_ENSURE_EQ(context, NumInputs(node), 1);
86 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
87 const TfLiteTensor* input;
88 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
89 TfLiteTensor* output;
90 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
91 TF_LITE_ENSURE_TYPES_EQ(context, input->type, output->type);
92 if (!is_supported_type(input->type)) {
93 TF_LITE_UNSUPPORTED_TYPE(context, input->type, op_name);
94 }
95 if (input->type == kTfLiteInt8 || input->type == kTfLiteInt16) {
96 TfLiteTensor* output = GetOutput(context, node, 0);
97 auto* op_data = static_cast<OpData*>(node->user_data);
98 TF_LITE_ENSURE_EQ(context, input->quantization.type,
99 kTfLiteAffineQuantization);
100 TF_LITE_ENSURE_EQ(context, output->quantization.type,
101 kTfLiteAffineQuantization);
102 const auto* input_params =
103 reinterpret_cast<TfLiteAffineQuantization*>(input->quantization.params);
104 const auto* output_params = reinterpret_cast<TfLiteAffineQuantization*>(
105 output->quantization.params);
106 TF_LITE_ENSURE(context, input_params != nullptr);
107 TF_LITE_ENSURE(context, input_params->scale != nullptr);
108 TF_LITE_ENSURE(context, input_params->scale->size > 0);
109 TF_LITE_ENSURE(context, input_params->zero_point->size > 0);
110 TF_LITE_ENSURE(context, output_params != nullptr);
111 TF_LITE_ENSURE(context, output_params->scale != nullptr);
112 TF_LITE_ENSURE(context, output_params->scale->size > 0);
113 TF_LITE_ENSURE(context, output_params->zero_point->size > 0);
114 op_data->input_offset = input_params->zero_point->data[0];
115 op_data->output_offset = output_params->zero_point->data[0];
116 if (input->type == kTfLiteInt16) {
117 TF_LITE_ENSURE_EQ(context, op_data->input_offset, 0);
118 TF_LITE_ENSURE_EQ(context, op_data->output_offset, 0);
119 }
120 const float input_scale = input_params->scale->data[0];
121 const float output_scale = output_params->scale->data[0];
122 op_data->needs_rescale = input_scale != output_scale;
123 if (op_name == kAbsName && op_data->needs_rescale) {
124 SetAbsOutputMultiplier(input_scale, output_scale, &op_data->multiplier,
125 &op_data->shift);
126 } else if (op_name == kRsqrtName) {
127 SetRsqrtOutputMultiplier(input_scale, output_scale, &op_data->multiplier,
128 &op_data->shift);
129 }
130 }
131 return context->ResizeTensor(context, output,
132 TfLiteIntArrayCopy(input->dims));
133}

135template <typename T>
136inline TfLiteStatus EvalImpl(TfLiteContext* context, TfLiteNode* node,
137 std::function<T(T)> func,
138 std::function<TfLiteStatus(T)> validate_input_func,
139 TfLiteType expected_type) {
140 const TfLiteTensor* input;
141 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
142 TfLiteTensor* output;
143 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
144 TF_LITE_ENSURE_TYPES_EQ(context, input->type, expected_type);
145 const int64_t num_elements = NumElements(input);
146 const T* in_data = GetTensorData<T>(input);
147 T* out_data = GetTensorData<T>(output);
148 for (int64_t i = 0; i < num_elements; ++i) {
149 if (validate_input_func) {
150 TF_LITE_ENSURE_OK(context, validate_input_func(in_data[i]));
151 }
152 out_data[i] = func(in_data[i]);
153 }
154 return kTfLiteOk;
155}

157template <typename T>
158inline TfLiteStatus EvalImpl(TfLiteContext* context, TfLiteNode* node,
159 std::function<T(T)> func,
160 TfLiteType expected_type) {
161 return EvalImpl<T>(context, node, func, /*validate_input_func=*/nullptr,
162 expected_type);
163}
164
165inline TfLiteStatus EvalNumeric(TfLiteContext* context, TfLiteNode* node,
166 float float_func(float)) {
167 return EvalImpl<float>(context, node, float_func, kTfLiteFloat32);
168}

170inline TfLiteStatus EvalLogical(TfLiteContext* context, TfLiteNode* node,
171 bool bool_func(bool)) {
172 return EvalImpl<bool>(context, node, bool_func, kTfLiteBool);
173}

184template <typename T>
185TfLiteStatus AbsEvalQuantized(TfLiteContext* context, TfLiteNode* node,
186 TfLiteType type) {
187 const auto* op_data = static_cast<const OpData*>(node->user_data);
188 const int kMin = std::numeric_limits<T>::min();
189 const int kMax = std::numeric_limits<T>::max();
190
191 std::function<T(T)> func = [&](T i) {
192 const int32_t value = std::abs(i - op_data->input_offset);
193 if (!op_data->needs_rescale) {
194 return static_cast<T>(
195 std::min(std::max(value + op_data->output_offset, kMin), kMax));
196 }
197 const int32_t output = MultiplyByQuantizedMultiplier(
198 value, op_data->multiplier, op_data->shift) +
199 op_data->output_offset;
200 return static_cast<T>(std::min(std::max(output, kMin), kMax));
201 };
202
203 return EvalImpl<T>(context, node, func, type);
204}

206TfLiteStatus AbsEval(TfLiteContext* context, TfLiteNode* node) {
207 const TfLiteType type = GetInput(context, node, 0)->type;
208 switch (type) {
209 case kTfLiteFloat32:
210 return EvalImpl<float>(context, node, std::abs<float>, type);
211 case kTfLiteInt8:
212 return AbsEvalQuantized<int8_t>(context, node, type);
213 case kTfLiteInt16:
214 return AbsEvalQuantized<int16_t>(context, node, type);
215 default:
216 TF_LITE_KERNEL_LOG(context, "Current data type %s is not supported.",
217 TfLiteTypeGetName(type));
218 return kTfLiteError;
219 }
220}

238TfLiteStatus RsqrtEvalQuantized(TfLiteContext* context, TfLiteNode* node,
239 TfLiteType type) {
240 const auto* op_data = static_cast<const OpData*>(node->user_data);
241 const int kMin = std::numeric_limits<int8_t>::min();
242 const int kMax = std::numeric_limits<int8_t>::max();
243 std::function<TfLiteStatus(int8_t)> validate_input_func = [&](int8_t i) {
244 TF_LITE_ENSURE_MSG(context, i >= op_data->input_offset,
245 "Rsqrt is only defined for positive values");
246 return kTfLiteOk;
247 };
248
249 std::function<int8_t(int8_t)> func = [&](int8_t i) {
250 const int32_t value = (i - op_data->input_offset);
251 const int32_t kShift = 20; // Shift to keep value integer.
252 if (value == 0) {
253 // Assume that any value close to 0 represents the max output value.
254 return static_cast<int8_t>(kMax);
255 }
256 int32_t inv_sqrt_multiplier;
257 int inv_sqrt_shift;
258 GetInvSqrtQuantizedMultiplierExp(value, kReverseShift, &inv_sqrt_multiplier,
259 &inv_sqrt_shift);
260 const int32_t data = MultiplyByQuantizedMultiplier(1, inv_sqrt_multiplier,
261 inv_sqrt_shift + kShift);
262 const int32_t output =
263 MultiplyByQuantizedMultiplier(data, op_data->multiplier,
264 op_data->shift - kShift) +
265 op_data->output_offset;
266 return static_cast<int8_t>(std::min(std::max(output, kMin), kMax));
267 };
268
269 return EvalImpl<int8_t>(context, node, func, validate_input_func, type);
270}

272TfLiteStatus RsqrtEval(TfLiteContext* context, TfLiteNode* node) {
273 const TfLiteType type = GetInput(context, node, 0)->type;
274 switch (type) {
275 case kTfLiteFloat32:
276 return EvalImpl<float>(
277 context, node, [](float f) { return 1.f / std::sqrt(f); }, type);
278 case kTfLiteInt8:
279 return RsqrtEvalQuantized(context, node, type);
280 default:
281 TF_LITE_KERNEL_LOG(context, "Current data type %s is not supported.",
282 TfLiteTypeGetName(type));
283 return kTfLiteError;
284 }
285}
