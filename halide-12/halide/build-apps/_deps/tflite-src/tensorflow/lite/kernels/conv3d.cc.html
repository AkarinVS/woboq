37enum KernelType {
38 kReference,
39 kGenericOptimized,
40};

42// Struct to carry data from Prepare to Eval.
43const int kTensorNotAllocated = -1;
44static constexpr size_t kMaxIm2colBufferSizeMobile = 1024 * 1024 * 1024; // 1GB
45
46struct OpData {
47 Padding3DValues padding;
48 int im2col_tensor_id = kTensorNotAllocated;
49 int transposed_filter_tensor_id = kTensorNotAllocated;
50
51 bool need_im2col = false;
52 bool need_transposed_filter = false;
53
54 // Disable im2col if the temporary im2col tensor requires too much memory
55 // (i.e. >= kMaxIm2colBufferSizeMobile).
56 bool im2col_oversized = false;
57
58 int32_t im2col_index;
59 int32_t transposed_filter_index;
60};

62void* Init(TfLiteContext* context, const char* buffer, size_t length) {
63 auto* opdata = new OpData;
64 return opdata;
65}

68 delete static_cast<OpData*>(buffer);

71TfLiteStatus AllocateTemporaryTensorsIfRequired(
72 KernelType kernel_type, TfLiteContext* context, TfLiteNode* node,
73 OpData* opdata, TfLiteConv3DParams* params, const TfLiteTensor* filter,
74 size_t im2col_bytes) {
75 int temporaries_count = 0;
76 const bool need_dilated_im2col = params->dilation_width_factor != 1 ||
77 params->dilation_height_factor != 1 ||
78 params->dilation_depth_factor != 1;
79 const bool need_non_dilated_im2col =
80 params->stride_depth != 1 || params->stride_width != 1 ||
81 params->stride_height != 1 || filter->dims->data[2] != 1 ||
82 filter->dims->data[1] != 1 || filter->dims->data[0] != 1;
83
84 opdata->need_im2col = (kernel_type == kGenericOptimized) &&
85 (need_dilated_im2col || need_non_dilated_im2col);
86 // TODO(b/183455632): Add transposing logic in converter so constant folding
87 // might work on constant filter tensor.
88 opdata->need_transposed_filter = (kernel_type == kGenericOptimized);
89
90 // On mobile platforms, the generic optimized kernel will not be used if the
91 // temporary im2col tensor requires too much memory.
92 if (IsMobilePlatform() && opdata->need_im2col &&
93 im2col_bytes >= kMaxIm2colBufferSizeMobile) {
94 opdata->need_im2col = false;
95 opdata->need_transposed_filter = false;
96 opdata->im2col_oversized = true;
97 }
98
99 if (opdata->need_im2col && opdata->im2col_tensor_id == kTensorNotAllocated) {
100 TF_LITE_ENSURE_OK(
101 context, context->AddTensors(context, 1, &opdata->im2col_tensor_id));
102 opdata->im2col_index = temporaries_count++;
103 }
104
105 if (opdata->need_transposed_filter &&
106 opdata->transposed_filter_tensor_id == kTensorNotAllocated) {
107 TF_LITE_ENSURE_OK(
108 context,
109 context->AddTensors(context, 1, &opdata->transposed_filter_tensor_id));
110 opdata->transposed_filter_index = temporaries_count++;
111 }
112
113 TfLiteIntArrayFree(node->temporaries);
114 node->temporaries = TfLiteIntArrayCreate(temporaries_count);
115 return kTfLiteOk;
116}

118TfLiteStatus Prepare(KernelType kernel_type, TfLiteContext* context,
119 TfLiteNode* node) {
120 auto* params = static_cast<TfLiteConv3DParams*>(node->builtin_data);
121 OpData* opdata = reinterpret_cast<OpData*>(node->user_data);
122
123 // Check number of inputs/outputs.
124 TF_LITE_ENSURE(context, node->inputs->size == 2 || node->inputs->size == 3);
125 TF_LITE_ENSURE_EQ(context, node->outputs->size, 1);
126 TfLiteTensor* output;
127 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
128 const TfLiteTensor* input;
129 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
130 const TfLiteTensor* filter;
131 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 1, &filter));
132
133 // Check dimensionality of input, filter.
134 TF_LITE_ENSURE_EQ(context, input->dims->size, 5);
135 TF_LITE_ENSURE_EQ(context, filter->dims->size, 5);
136
137 // Check input channels matching filter.
138 TF_LITE_ENSURE_EQ(context, input->dims->data[4], filter->dims->data[3]);
139
140 // Check types.
141 TfLiteType input_type = input->type;
142 TF_LITE_ENSURE_TYPES_EQ(context, input_type, kTfLiteFloat32);
143 TF_LITE_ENSURE_TYPES_EQ(context, filter->type, kTfLiteFloat32);
144 TF_LITE_ENSURE_TYPES_EQ(context, output->type, input_type);
145
146 // Check bias.
147 const TfLiteTensor* bias = GetInput(context, node, 2);
148 if (bias) {
149 TF_LITE_ENSURE_TYPES_EQ(context, bias->type, input_type);
150 TF_LITE_ENSURE_EQ(context, NumElements(bias), SizeOfDimension(filter, 4));
151 }
152
153 // Filter has shape of [filter_depth, filter_height, filter_width,
154 // in_channels, out_channels].
155 int batches = input->dims->data[0];
156 int channels_out = filter->dims->data[4];
157 int depth = input->dims->data[1];
158 int height = input->dims->data[2];
159 int width = input->dims->data[3];
160 int filter_depth = filter->dims->data[0];
161 int filter_height = filter->dims->data[1];
162 int filter_width = filter->dims->data[2];
163 int input_channel = filter->dims->data[3];
164
165 // Matching GetWindowedOutputSize in TensorFlow.
166 int out_width, out_height, out_depth;
167 opdata->padding = ComputePadding3DValues(
168 params->stride_height, params->stride_width, params->stride_depth,
169 params->dilation_height_factor, params->dilation_width_factor,
170 params->dilation_depth_factor, height, width, depth, filter_height,
171 filter_width, filter_depth, params->padding, &out_height, &out_width,
172 &out_depth);
173
174 TfLiteIntArray* output_size = TfLiteIntArrayCreate(5);
175 output_size->data[0] = batches;
176 output_size->data[1] = out_depth;
177 output_size->data[2] = out_height;
178 output_size->data[3] = out_width;
179 output_size->data[4] = channels_out;
180 TF_LITE_ENSURE_OK(context,
181 context->ResizeTensor(context, output, output_size));
182
183 // Allocate temporary tensors.
184 size_t input_type_size;
185 TF_LITE_ENSURE_STATUS(GetSizeOfType(context, input->type, &input_type_size));
186 const size_t im2col_bytes = batches * out_depth * out_height * out_width *
187 input_channel * filter_depth * filter_height *
188 filter_width * input_type_size;
189 TF_LITE_ENSURE_OK(context, AllocateTemporaryTensorsIfRequired(
190 kernel_type, context, node, opdata, params,
191 filter, im2col_bytes));
192
193 if (opdata->need_im2col) {
194 TfLiteIntArray* im2col_size = TfLiteIntArrayCreate(5);
195 im2col_size->data[0] = output_size->data[0];
196 im2col_size->data[1] = output_size->data[1];
197 im2col_size->data[2] = output_size->data[2];
198 im2col_size->data[3] = output_size->data[3];
199 im2col_size->data[4] =
200 input_channel * filter_depth * filter_height * filter_width;
201
202 TfLiteTensor* im2col;
203 node->temporaries->data[opdata->im2col_index] = opdata->im2col_tensor_id;
204 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node,
205 opdata->im2col_index, &im2col));
206 im2col->type = input->type;
207 im2col->allocation_type = kTfLiteArenaRw;
208 TF_LITE_ENSURE_OK(context,
209 context->ResizeTensor(context, im2col, im2col_size));
210 }
211
212 if (opdata->need_transposed_filter) {
213 TfLiteIntArray* transposed_filter_size = TfLiteIntArrayCreate(5);
214 transposed_filter_size->data[0] = filter->dims->data[4];
215 transposed_filter_size->data[1] = filter->dims->data[0];
216 transposed_filter_size->data[2] = filter->dims->data[1];
217 transposed_filter_size->data[3] = filter->dims->data[2];
218 transposed_filter_size->data[4] = filter->dims->data[3];
219
220 TfLiteTensor* transposed_filter;
221 node->temporaries->data[opdata->transposed_filter_index] =
222 opdata->transposed_filter_tensor_id;
223 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node,
224 opdata->transposed_filter_index,
225 &transposed_filter));
226 transposed_filter->type = filter->type;
227 transposed_filter->allocation_type = kTfLiteArenaRw;
228 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, transposed_filter,
229 transposed_filter_size));
230 }
231 return kTfLiteOk;
232}

234template <KernelType kernel_type>
235TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
236 return Prepare(kernel_type, context, node);
237}

239void EvalFloat(KernelType kernel_type, TfLiteContext* context, TfLiteNode* node,
240 TfLiteConv3DParams* params, OpData* opdata,
241 const TfLiteTensor* input, const TfLiteTensor* filter,
242 const TfLiteTensor* bias, TfLiteTensor* im2col,
243 TfLiteTensor* tranposed_filter, TfLiteTensor* output) {
244 float output_activation_min, output_activation_max;
245 CalculateActivationRange(params->activation, &output_activation_min,
246 &output_activation_max);
247
248 Conv3DParams runtime_params;
249 runtime_params.padding_values = opdata->padding;
250 runtime_params.stride_depth = params->stride_depth;
251 runtime_params.stride_height = params->stride_height;
252 runtime_params.stride_width = params->stride_width;
253 runtime_params.dilation_depth = params->dilation_depth_factor;
254 runtime_params.dilation_height = params->dilation_height_factor;
255 runtime_params.dilation_width = params->dilation_width_factor;
256 runtime_params.float_activation_min = output_activation_min;
257 runtime_params.float_activation_max = output_activation_max;
258 switch (kernel_type) {
259 case kReference: {
260 reference_ops::Conv3D(runtime_params, GetTensorShape(input),
261 GetTensorData<float>(input), GetTensorShape(filter),
262 GetTensorData<float>(filter), GetTensorShape(bias),
263 GetTensorData<float>(bias), GetTensorShape(output),
264 GetTensorData<float>(output));
265 break;
266 }
267 case kGenericOptimized: {
268 optimized_ops::Conv3D(
269 runtime_params, GetTensorShape(input), GetTensorData<float>(input),
270 GetTensorShape(filter), GetTensorData<float>(filter),
271 GetTensorShape(bias), GetTensorData<float>(bias),
272 GetTensorShape(output), GetTensorData<float>(output),
273 GetTensorShape(im2col), GetTensorData<float>(im2col),
274 GetTensorShape(tranposed_filter),
275 GetTensorData<float>(tranposed_filter),
276 CpuBackendContext::GetFromContext(context));
277 } break;
278 }
279}

281TfLiteStatus Eval(KernelType kernel_type, TfLiteContext* context,
282 TfLiteNode* node) {
283 auto* params = reinterpret_cast<TfLiteConv3DParams*>(node->builtin_data);
284 OpData* opdata = reinterpret_cast<OpData*>(node->user_data);
285
286 TfLiteTensor* output;
287 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, 0, &output));
288 const TfLiteTensor* input;
289 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 0, &input));
290 const TfLiteTensor* filter;
291 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, 1, &filter));
292 const TfLiteTensor* bias = GetInput(context, node, 2);
293
294 TfLiteTensor* im2col = opdata->need_im2col
295 ? &context->tensors[opdata->im2col_tensor_id]
296 : nullptr;
297 TfLiteTensor* transposed_filter =
298 opdata->need_transposed_filter
299 ? &context->tensors[opdata->transposed_filter_tensor_id]
300 : nullptr;
301
302 // Fallback to reference execution path when im2col is needed but disabled.
303 if (opdata->im2col_oversized) {
304 kernel_type = kReference;
305 }
306
307 switch (input->type) {
308 case kTfLiteFloat32:
309 EvalFloat(kernel_type, context, node, params, opdata, input, filter, bias,
310 im2col, transposed_filter, output);
311 break;
312 default:
313 TF_LITE_KERNEL_LOG(context, "Type %s currently not supported.",
314 TfLiteTypeGetName(input->type));
315 return kTfLiteError;
316 }
317 return kTfLiteOk;
318}
319
320template <KernelType kernel_type>
321TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
322 return Eval(kernel_type, context, node);
323}
