15#ifndef TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_DENSIFY_H_
16#define TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_DENSIFY_H_
17
18#include <vector>
19
20#include "tensorflow/lite/c/common.h"
21#include "tensorflow/lite/kernels/internal/common.h"
22#include "tensorflow/lite/kernels/internal/types.h"
23#include "tensorflow/lite/tools/optimize/sparsity/format_converter.h"
24
25namespace tflite {
26namespace reference_ops {
27
28template <typename T>
29inline void Densify(const TfLiteSparsity* sparsity,
30 const RuntimeShape& input_shape, const T* input_data,
31 const RuntimeShape& output_shape, T* output_data,
32 TfLiteContext* context) {
33 const int dims_count = output_shape.DimensionsCount();
34 std::vector<int> vector_shape(dims_count);
35 for (int i = 0; i < dims_count; i++) {
36 vector_shape[i] = output_shape.Dims(i);
37 }
38
39 tflite::optimize::sparsity::FormatConverter<T> converter(vector_shape,
40 *sparsity);
41 converter.SparseToDense(input_data, output_shape.FlatSize(), output_data,
42 context);
43}
44
45} // namespace reference_ops
46} // namespace tflite
47#endif // TENSORFLOW_LITE_KERNELS_INTERNAL_REFERENCE_DENSIFY_H_
