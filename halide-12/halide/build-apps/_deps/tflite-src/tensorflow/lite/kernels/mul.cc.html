41// This file has three implementation of Mul.
42enum KernelType {
43 kReference,
44 kGenericOptimized, // Neon-free
45 kNeonOptimized,
46};
47
48constexpr int kInputTensor1 = 0;
49constexpr int kInputTensor2 = 1;
50constexpr int kOutputTensor = 0;
51
52struct OpData {
53 // Parameters used in the quantized paths where the output is 8bit
54 int32 output_activation_min;
55 int32 output_activation_max;
56
57 // Parameters used in all quantized paths
58 int32_t output_multiplier;
59 int output_shift;
60};
61
62void* Init(TfLiteContext* context, const char* buffer, size_t length) {
63 auto* data = new OpData;
64 return data;
65}
66
67void Free(TfLiteContext* context, void* buffer) {
68 delete reinterpret_cast<OpData*>(buffer);
69}

71TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
72 auto* params = reinterpret_cast<TfLiteMulParams*>(node->builtin_data);
73 OpData* data = reinterpret_cast<OpData*>(node->user_data);
74
75 TF_LITE_ENSURE_EQ(context, NumInputs(node), 2);
76 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
77
78 const TfLiteTensor* input1;
79 TF_LITE_ENSURE_OK(context,
80 GetInputSafe(context, node, kInputTensor1, &input1));
81 const TfLiteTensor* input2;
82 TF_LITE_ENSURE_OK(context,
83 GetInputSafe(context, node, kInputTensor2, &input2));
84 TfLiteTensor* output;
85 TF_LITE_ENSURE_OK(context,
86 GetOutputSafe(context, node, kOutputTensor, &output));
87
88 TF_LITE_ENSURE_TYPES_EQ(context, input1->type, input2->type);
89
90 const bool requires_broadcast = !HaveSameShapes(input1, input2);
91
92 TfLiteIntArray* output_size = nullptr;
93 if (requires_broadcast) {
94 TF_LITE_ENSURE_OK(context, CalculateShapeForBroadcast(
95 context, input1, input2, &output_size));
96 } else {
97 output_size = TfLiteIntArrayCopy(input1->dims);
98 }
99
100 if (output->type == kTfLiteUInt8 || output->type == kTfLiteInt8 ||
101 output->type == kTfLiteInt16) {
102 TF_LITE_ENSURE_STATUS(CalculateActivationRangeQuantized(
103 context, params->activation, output, &data->output_activation_min,
104 &data->output_activation_max));
105 double real_multiplier =
106 input1->params.scale * input2->params.scale / output->params.scale;
107 QuantizeMultiplier(real_multiplier, &data->output_multiplier,
108 &data->output_shift);
109 }
110
111 return context->ResizeTensor(context, output, output_size);
112}

114template <KernelType kernel_type>
115void EvalMul(TfLiteContext* context, TfLiteNode* node, TfLiteMulParams* params,
116 const OpData* data, const TfLiteTensor* input1,
117 const TfLiteTensor* input2, TfLiteTensor* output) {
118 tflite::ArithmeticParams op_params;
119 const bool need_broadcast = optimized_ops::ProcessBroadcastShapes(
120 GetTensorShape(input1), GetTensorShape(input2), &op_params);
121#define TF_LITE_MUL(type, opname, data_type) \
122 data_type output_activation_min, output_activation_max; \
123 CalculateActivationRange(params->activation, &output_activation_min, \
124 &output_activation_max); \
125 SetActivationParams(output_activation_min, output_activation_max, \
126 &op_params); \
127 type::opname(op_params, GetTensorShape(input1), \
128 GetTensorData<data_type>(input1), GetTensorShape(input2), \
129 GetTensorData<data_type>(input2), GetTensorShape(output), \
130 GetTensorData<data_type>(output))
131
132 if (output->type == kTfLiteInt32) {
133 if (kernel_type == kReference) {
134 if (need_broadcast) {
135 TF_LITE_MUL(reference_ops, BroadcastMul4DSlow, int32_t);
136 } else {
137 TF_LITE_MUL(reference_ops, Mul, int32_t);
138 }
139 } else {
140 if (need_broadcast) {
141 TF_LITE_MUL(optimized_ops, BroadcastMul4DSlow, int32_t);
142 } else {
143 TF_LITE_MUL(optimized_ops, Mul, int32_t);
144 }
145 }
146 } else if (output->type == kTfLiteFloat32) {
147 if (kernel_type == kReference) {
148 if (need_broadcast) {
149 TF_LITE_MUL(reference_ops, BroadcastMul4DSlow, float);
150 } else {
151 TF_LITE_MUL(reference_ops, Mul, float);
152 }
153 } else {
154 if (need_broadcast) {
155 TF_LITE_MUL(optimized_ops, BroadcastMulDispatch, float);
156 } else {
157 TF_LITE_MUL(optimized_ops, Mul, float);
158 }
159 }
160 }
161#undef TF_LITE_MUL
162}

164template <KernelType kernel_type>
165TfLiteStatus EvalQuantized(TfLiteContext* context, TfLiteNode* node,
166 TfLiteMulParams* params, const OpData* data,
167 const TfLiteTensor* input1,
168 const TfLiteTensor* input2, TfLiteTensor* output) {
169 if (input1->type == input2->type && input1->type == output->type &&
170 (input1->type == kTfLiteUInt8 || input1->type == kTfLiteInt8 ||
171 input1->type == kTfLiteInt16)) {
172 tflite::ArithmeticParams op_params;
173 SetActivationParams(data->output_activation_min,
174 data->output_activation_max, &op_params);
175 op_params.input1_offset = -input1->params.zero_point;
176 op_params.input2_offset = -input2->params.zero_point;
177 op_params.output_offset = output->params.zero_point;
178 op_params.output_multiplier = data->output_multiplier;
179 op_params.output_shift = data->output_shift;
180 bool need_broadcast = optimized_ops::ProcessBroadcastShapes(
181 GetTensorShape(input1), GetTensorShape(input2), &op_params);
182#define TF_LITE_MUL(type, opname, dtype) \
183 type::opname(op_params, GetTensorShape(input1), \
184 GetTensorData<dtype>(input1), GetTensorShape(input2), \
185 GetTensorData<dtype>(input2), GetTensorShape(output), \
186 GetTensorData<dtype>(output))
187 if (input1->type == kTfLiteInt8) {
188 if (kernel_type == kReference) {
189 if (need_broadcast) {
190 TF_LITE_MUL(reference_integer_ops, BroadcastMul4DSlow, int8_t);
191 } else {
192 TF_LITE_MUL(reference_integer_ops, Mul, int8_t);
193 }
194 } else {
195 if (need_broadcast) {
196 TF_LITE_MUL(optimized_integer_ops, BroadcastMulDispatch, int8_t);
197 } else {
198 TF_LITE_MUL(optimized_integer_ops, Mul, int8_t);
199 }
200 }
201 } else if (input1->type == kTfLiteInt16) {
202 // We have this check, because in case of int16
203 // input1_val*input2_val can overflow int32:
204 // see MulElementwise -
205 // tensorflow/lite/kernels/internal/reference/integer_ops/mul.h in case of
206 // 16-bit this function is used in symmetric quantization, so offset
207 // should be zero.
208 TF_LITE_ENSURE_EQ(context, op_params.input1_offset, 0.0);
209 TF_LITE_ENSURE_EQ(context, op_params.input2_offset, 0.0);
210 TF_LITE_ENSURE_EQ(context, op_params.output_offset, 0.0);
211
212 if (need_broadcast) {
213 TF_LITE_MUL(reference_integer_ops, BroadcastMul4DSlow, int16_t);
214 } else {
215 TF_LITE_MUL(reference_integer_ops, Mul, int16_t);
216 }
217 } else {
218 // type == kTfLiteUInt8
219 if (kernel_type == kReference) {
220 if (need_broadcast) {
221 TF_LITE_MUL(reference_ops, BroadcastMul4DSlow, uint8_t);
222 } else {
223 TF_LITE_MUL(reference_ops, Mul, uint8_t);
224 }
225 } else {
226 if (need_broadcast) {
227 TF_LITE_MUL(optimized_ops, BroadcastMulDispatch, uint8_t);
228 } else {
229 TF_LITE_MUL(optimized_ops, Mul, uint8_t);
230 }
231 }
232 }
233#undef TF_LITE_MUL
234 } else if (input1->type == kTfLiteInt16 && input2->type == kTfLiteInt16 &&
235 (output->type == kTfLiteUInt8 || output->type == kTfLiteInt8)) {
236#define TF_LITE_MUL(type, opname, output_dtype) \
237 tflite::ArithmeticParams op_params; \
238 SetActivationParams(data->output_activation_min, \
239 data->output_activation_max, &op_params); \
240 op_params.output_offset = output->params.zero_point; \
241 type::opname(op_params, GetTensorShape(input1), \
242 GetTensorData<int16_t>(input1), GetTensorShape(input2), \
243 GetTensorData<int16_t>(input2), GetTensorShape(output), \
244 GetTensorData<output_dtype>(output))
245 if (output->type == kTfLiteInt8) {
246 TF_LITE_MUL(reference_integer_ops, Mul, int8_t);
247 } else {
248 if (kernel_type == kReference) {
249 TF_LITE_MUL(reference_ops, Mul, uint8_t);
250 } else {
251 TF_LITE_MUL(optimized_ops, Mul, uint8_t);
252 }
253 }
254#undef TF_LITE_MUL
255 } else {
256 context->ReportError(
257 context, "Unsupported combination of input and output types in Mul.");
258 return kTfLiteError;
259 }
260 return kTfLiteOk;
261}
262
263template <KernelType kernel_type>
264TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
265 auto* params = reinterpret_cast<TfLiteMulParams*>(node->builtin_data);
266 OpData* data = reinterpret_cast<OpData*>(node->user_data);
267
268 const TfLiteTensor* input1;
269 TF_LITE_ENSURE_OK(context,
270 GetInputSafe(context, node, kInputTensor1, &input1));
271 const TfLiteTensor* input2;
272 TF_LITE_ENSURE_OK(context,
273 GetInputSafe(context, node, kInputTensor2, &input2));
274 TfLiteTensor* output;
275 TF_LITE_ENSURE_OK(context,
276 GetOutputSafe(context, node, kOutputTensor, &output));
277
278 if (output->type == kTfLiteFloat32 || output->type == kTfLiteInt32) {
279 EvalMul<kernel_type>(context, node, params, data, input1, input2, output);
280 } else if (output->type == kTfLiteUInt8 || output->type == kTfLiteInt8 ||
281 output->type == kTfLiteInt16) {
282 TF_LITE_ENSURE_OK(
283 context, EvalQuantized<kernel_type>(context, node, params, data, input1,
284 input2, output));
285 } else {
286 context->ReportError(context,
287 "Mul only supports FLOAT32, INT32 and quantized UINT8,"
288 " INT8 and INT16 now, got %d.",
289 output->type);
290 return kTfLiteError;
291 }
292
293 return kTfLiteOk;
294}

299 static TfLiteRegistration r = {mul::Init, mul::Free, mul::Prepare,

301 return &r;
302}

305 static TfLiteRegistration r = {mul::Init, mul::Free, mul::Prepare,
306 mul::Eval<mul::kGenericOptimized>};
307 return &r;
308}

310TfLiteRegistration* Register_MUL_NEON_OPT() {

314}
