33struct OpContext {
34 OpContext(TfLiteContext* context, TfLiteNode* node) {
35 params = reinterpret_cast<TfLiteSplitVParams*>(node->builtin_data);
36 input = GetInput(context, node, 0);
37 size_splits = GetInput(context, node, 1);
38 axis = GetInput(context, node, 2);
39 }
40 TfLiteSplitVParams* params;
41 const TfLiteTensor* input;
42 const TfLiteTensor* size_splits;
43 const TfLiteTensor* axis;
44};

46TfLiteStatus UseDynamicOutputTensors(TfLiteContext* context, TfLiteNode* node) {
47 for (int i = 0; i < NumOutputs(node); ++i) {
48 TfLiteTensor* tensor;
49 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, i, &tensor));
50 SetTensorToDynamic(tensor);
51 }
52 return kTfLiteOk;
53}

55template <typename T>
56void GetSizeSplitsVector(const TfLiteTensor* size_splits,
57 std::vector<int64_t>* size_splits_vector) {
58 const auto num_elements = NumElements(size_splits);
59 for (int i = 0; i < num_elements; ++i) {
60 size_splits_vector->push_back(GetTensorData<T>(size_splits)[i]);
61 }
62}
63
64TfLiteStatus ResizeOutputTensors(TfLiteContext* context, TfLiteNode* node,
65 const TfLiteTensor* input,
66 const TfLiteTensor* size_splits,
67 const TfLiteTensor* axis) {
68 int axis_value = GetTensorData<int>(axis)[0];
69 if (axis_value < 0) {
70 axis_value += NumDimensions(input);
71 }
72
73 std::vector<int64_t> size_splits_vector;
74 if (size_splits->type == kTfLiteInt32) {
75 GetSizeSplitsVector<int32_t>(size_splits, &size_splits_vector);
76 } else if (size_splits->type == kTfLiteInt64) {
77 GetSizeSplitsVector<int64_t>(size_splits, &size_splits_vector);
78 } else {
79 context->ReportError(context, "size_splits only support type int32|int64.");
80 return kTfLiteError;
81 }
82
83 int minus_one_index = -1;
84 int64_t size_splits_sum = 0;
85
86 for (int i = 0; i < size_splits_vector.size(); ++i) {
87 if (size_splits_vector.at(i) == -1) {
88 if (minus_one_index == -1) {
89 minus_one_index = i;
90 } else {
91 context->ReportError(context,
92 "The size_splits contains more than one -1.");
93 }
94 } else {
95 size_splits_sum += size_splits_vector.at(i);
96 }
97 }
98
99 TF_LITE_ENSURE(context, axis_value >= 0);
100 TF_LITE_ENSURE(context, axis_value < NumDimensions(input));
101 const int input_size = SizeOfDimension(input, axis_value);
102
103 if (minus_one_index != -1) {
104 if (size_splits_sum > input_size) {
105 context->ReportError(
106 context,
107 "The sum of size_splits must be less than the dimension of value.");
108 } else {
109 size_splits_vector[minus_one_index] = input_size - size_splits_sum;
110 }
111 } else if (size_splits_sum != input_size) {
112 context->ReportError(
113 context,
114 "The size_splits must sum to the dimension of value along axis.");
115 }
116
117 for (int i = 0; i < NumOutputs(node); ++i) {
118 TfLiteIntArray* output_dims = TfLiteIntArrayCopy(input->dims);
119 output_dims->data[axis_value] = size_splits_vector.at(i);
120 TfLiteTensor* output;
121 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, i, &output));
122 TF_LITE_ENSURE_STATUS(context->ResizeTensor(context, output, output_dims));
123 }
124
125 return kTfLiteOk;
126}

128TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
129 TF_LITE_ENSURE_EQ(context, NumInputs(node), 3);
130
131 OpContext op_context(context, node);
132
133 TF_LITE_ENSURE_EQ(context, NumOutputs(node), op_context.params->num_splits);
134
135 auto input_type = op_context.input->type;
136 TF_LITE_ENSURE(context,
137 input_type == kTfLiteFloat32 || input_type == kTfLiteUInt8 ||
138 input_type == kTfLiteInt16 || input_type == kTfLiteInt32 ||
139 input_type == kTfLiteInt64 || input_type == kTfLiteInt8);
140 for (int i = 0; i < NumOutputs(node); ++i) {
141 TfLiteTensor* tensor;
142 TF_LITE_ENSURE_OK(context, GetOutputSafe(context, node, i, &tensor));
143 tensor->type = input_type;
144 }
145
146 auto size_splits = op_context.size_splits;
147 TF_LITE_ENSURE_EQ(context, NumDimensions(size_splits), 1);
148 TF_LITE_ENSURE_EQ(context, NumOutputs(node), NumElements(size_splits));
149
150 // If we know the contents of the 'size_splits' tensor and the 'axis' tensor,
151 // resize all outputs. Otherwise, wait until Eval().
152 if (IsConstantTensor(op_context.size_splits) &&
153 IsConstantTensor(op_context.axis)) {
154 return ResizeOutputTensors(context, node, op_context.input,
155 op_context.size_splits, op_context.axis);
156 } else {
157 return UseDynamicOutputTensors(context, node);
158 }
159}
160
161TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
162 OpContext op_context(context, node);
163
164 // When the 'size_splits' and the 'axis' tensor is non-const we can't resize
165 // output tensors in Prepare(), and we have to do it now.
166 if (!IsConstantTensor(op_context.axis) ||
167 !IsConstantTensor(op_context.size_splits)) {
168 TF_LITE_ENSURE_OK(
169 context, ResizeOutputTensors(context, node, op_context.input,
170 op_context.size_splits, op_context.axis));
171 }
172
173 int axis_value = GetTensorData<int>(op_context.axis)[0];
174
175 // Use split function to build the outputs since they share the same logic.
176#define TF_LITE_SPLIT_V(scalar) \
177 VectorOfTensors<scalar> all_outputs(*context, *node->outputs); \
178 tflite::SplitParams op_params; \
179 op_params.num_split = NumOutputs(node); \
180 op_params.axis = axis_value; \
181 reference_ops::Split(op_params, GetTensorShape(op_context.input), \
182 GetTensorData<scalar>(op_context.input), \
183 all_outputs.shapes(), all_outputs.data());
184 switch (op_context.input->type) {
185 case kTfLiteFloat32: {
186 TF_LITE_SPLIT_V(float);
187 break;
188 }
189 case kTfLiteUInt8: {
190 TF_LITE_SPLIT_V(uint8_t);
191 break;
192 }
193 case kTfLiteInt16: {
194 TF_LITE_SPLIT_V(int16_t);
195 break;
196 }
197 case kTfLiteInt32: {
198 TF_LITE_SPLIT_V(int32_t);
199 break;
200 }
201 case kTfLiteInt64: {
202 TF_LITE_SPLIT_V(int64_t);
203 break;
204 }
205 case kTfLiteInt8: {
206 TF_LITE_SPLIT_V(int8_t);
207 break;
208 }
209 default:
210 context->ReportError(context, "Type %s currently not supported.",
211 TfLiteTypeGetName(op_context.input->type));
212 return kTfLiteError;
213 }
214#undef TF_LITE_SPLIT_V
215
216 return kTfLiteOk;
217}
