333OpResolver::TfLiteDelegatePtrVector BuiltinOpResolver::GetDelegates(
334 int num_threads) const {
335 OpResolver::TfLiteDelegatePtrVector delegates;
336 auto xnnpack_delegate = tflite::MaybeCreateXNNPACKDelegate(num_threads);
337 if (xnnpack_delegate != nullptr) {
338 delegates.push_back(std::move(xnnpack_delegate));
339 }
340 return delegates;
341}
