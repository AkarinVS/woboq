23// TODO(renjieliu): Migrate others to use ComputePaddingWithLeftover.
24inline int ComputePadding(int stride, int dilation_rate, int in_size,
25 int filter_size, int out_size) {
26 int effective_filter_size = (filter_size - 1) * dilation_rate + 1;
27 int padding = ((out_size - 1) * stride + effective_filter_size - in_size) / 2;
28 return padding > 0 ? padding : 0;
29}
30
31// It's not guaranteed that padding is symmetric. It's important to keep
32// offset for algorithms need all paddings.
33inline int ComputePaddingWithOffset(int stride, int dilation_rate, int in_size,
34 int filter_size, int out_size,
35 int* offset) {
36 int effective_filter_size = (filter_size - 1) * dilation_rate + 1;
37 int total_padding =
38 ((out_size - 1) * stride + effective_filter_size - in_size);
39 total_padding = total_padding > 0 ? total_padding : 0;
40 *offset = total_padding % 2;
41 return total_padding / 2;
42}
43
44// Matching GetWindowedOutputSize in TensorFlow.
45inline int ComputeOutSize(TfLitePadding padding, int image_size,
46 int filter_size, int stride, int dilation_rate = 1) {
47 int effective_filter_size = (filter_size - 1) * dilation_rate + 1;
48
49 // TODO(b/186448822): This uses 0 since the function has no other way to
50 // report error case
51 if (stride == 0) return 0;
52
53 switch (padding) {
54 case kTfLitePaddingSame:
55 return (image_size + stride - 1) / stride;
56 case kTfLitePaddingValid:
57 return (image_size + stride - effective_filter_size) / stride;
58 default:
59 return 0;
60 }
61}
62
63inline TfLitePaddingValues ComputePaddingHeightWidth(
64 int stride_height, int stride_width, int dilation_rate_height,
65 int dilation_rate_width, int in_height, int in_width, int filter_height,
66 int filter_width, TfLitePadding padding, int* out_height, int* out_width) {
67 *out_width = ComputeOutSize(padding, in_width, filter_width, stride_width,
68 dilation_rate_width);
69 *out_height = ComputeOutSize(padding, in_height, filter_height, stride_height,
70 dilation_rate_height);
71
72 TfLitePaddingValues padding_values;
73 int offset = 0;
74 padding_values.height =
75 ComputePaddingWithOffset(stride_height, dilation_rate_height, in_height,
76 filter_height, *out_height, &offset);
77 padding_values.height_offset = offset;
78 padding_values.width =
79 ComputePaddingWithOffset(stride_width, dilation_rate_width, in_width,
80 filter_width, *out_width, &offset);
81 padding_values.width_offset = offset;
82 return padding_values;
83}
84
85inline Padding3DValues ComputePadding3DValues(
86 int stride_height, int stride_width, int stride_depth,
87 int dilation_rate_height, int dilation_rate_width, int dilation_rate_depth,
88 int in_height, int in_width, int in_depth, int filter_height,
89 int filter_width, int filter_depth, TfLitePadding padding, int* out_height,
90 int* out_width, int* out_depth) {
91 *out_width = ComputeOutSize(padding, in_width, filter_width, stride_width,
92 dilation_rate_width);
93 *out_height = ComputeOutSize(padding, in_height, filter_height, stride_height,
94 dilation_rate_height);
95 *out_depth = ComputeOutSize(padding, in_depth, filter_depth, stride_depth,
96 dilation_rate_depth);
97
98 Padding3DValues padding_values;
99 int offset = 0;
100 padding_values.depth =
101 ComputePaddingWithOffset(stride_depth, dilation_rate_depth, in_depth,
102 filter_depth, *out_depth, &offset);
103 padding_values.depth_offset = offset;
104 padding_values.height =
105 ComputePaddingWithOffset(stride_height, dilation_rate_height, in_height,
106 filter_height, *out_height, &offset);
107 padding_values.height_offset = offset;
108 padding_values.width =
109 ComputePaddingWithOffset(stride_width, dilation_rate_width, in_width,
110 filter_width, *out_width, &offset);
111 padding_values.width_offset = offset;
112 return padding_values;
113}
