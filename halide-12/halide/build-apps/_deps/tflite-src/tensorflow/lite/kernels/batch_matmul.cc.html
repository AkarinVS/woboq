42static const int kInputLHSTensor = 0;
43static const int kInputRHSTensor = 1;
44static const int kOutputTensor = 0;

46static const int kNumTempTensorsForAdjoints = 2;
47static const int kNumTempTensorsForHybrid = 5;

49// This file has two implementations of Transpose.
50enum KernelType {
51 kReference,
52 kGenericOptimized,
53};

55struct OpData {
56 // The scaling factor from input to output (aka the 'real multiplier') can
57 // be represented as a fixed point multiplier plus a left shift.
58 int32_t output_multiplier;
59 int output_shift;
60 // The range of the fused activation layer. For example for kNone and
61 // uint8_t these would be 0 and 255.
62 int32_t output_activation_min;
63 int32_t output_activation_max;
64 // The index of the temporary tensors where we store transposed LHS/RHS.
65 int scratch_tensor_index;
66 bool rhs_transposed;
67 bool compute_row_sums = false;
68};
69
70struct OpContext {
71 OpContext(TfLiteContext* context, TfLiteNode* node) {
72 params = reinterpret_cast<TfLiteBatchMatMulParams*>(node->builtin_data);
73 lhs = GetInput(context, node, kInputLHSTensor);
74 rhs = GetInput(context, node, kInputRHSTensor);
75 output = GetOutput(context, node, 0);
76 }
77 TfLiteBatchMatMulParams* params;
78 const TfLiteTensor* lhs;
79 const TfLiteTensor* rhs;
80 TfLiteTensor* output;
81};
82
83void* Init(TfLiteContext* context, const char* buffer, size_t length) {
84 auto* op_data = new OpData();
85 // If the RHS is constant, we only transpose once.
86 op_data->rhs_transposed = false;
87 // Creates the temp tensors to store the transposed LHS and/or RHS, and
88 // extra buffers for the quantized case.
89 context->AddTensors(context,
90 kNumTempTensorsForAdjoints + kNumTempTensorsForHybrid,
91 &op_data->scratch_tensor_index);
92 return op_data;
93}
94
95void Free(TfLiteContext* context, void* buffer) {
96 delete static_cast<OpData*>(buffer);
97}

99TfLiteStatus ResizeOutputTensor(TfLiteContext* context,
100 const RuntimeShape& extended_lhs_shape,
101 const RuntimeShape& extended_rhs_shape,
102 bool adj_x, bool adj_y, int output_rank,
103 TfLiteTensor* output) {
104 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(output_rank);
105 // Fill in any broadcast dimensions.
106 for (int i = 0; i < output_rank - 2; ++i) {
107 const int lhs_dim = extended_lhs_shape.Dims(i);
108 const int rhs_dim = extended_rhs_shape.Dims(i);
109 int broadcast_dim = lhs_dim;
110 if ((lhs_dim != rhs_dim) && (lhs_dim == 1)) {
111 broadcast_dim = rhs_dim;
112 }
113 output_shape->data[i] = broadcast_dim;
114 }
115 // Fill in the matmul dimensions.
116 int lhs_rows_index = adj_x ? output_rank - 1 : output_rank - 2;
117 int rhs_cols_index = adj_y ? output_rank - 2 : output_rank - 1;
118
119 output_shape->data[output_rank - 2] = extended_lhs_shape.Dims(lhs_rows_index);
120 output_shape->data[output_rank - 1] = extended_rhs_shape.Dims(rhs_cols_index);
121 TfLiteStatus stat = context->ResizeTensor(context, output, output_shape);
122 return stat;
123}

125// Initializes temp tensors to store transposed operands.
126TfLiteStatus InitializeTemporaries(TfLiteContext* context, TfLiteNode* node,
127 OpContext* op_context) {
128 // Create temporary tensors to hold transposed LHS/RHS.
129 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
130 const TfLiteTensor* lhs = op_context->lhs;
131 const TfLiteTensor* rhs = op_context->rhs;
132 TfLiteIntArrayFree(node->temporaries);
133 // For "hybrid" quantization, we impose the constraint that the LHS
134 // is float (typically an activation from a prior layer) and the RHS
135 // is quantized int8.
136 bool is_hybrid =
137 (op_context->lhs->type == kTfLiteFloat32 && rhs->type == kTfLiteInt8);
138 if (is_hybrid) {
139 node->temporaries = TfLiteIntArrayCreate(kNumTempTensorsForAdjoints +
140 kNumTempTensorsForHybrid);
141 } else {
142 node->temporaries = TfLiteIntArrayCreate(kNumTempTensorsForAdjoints);
143 }
144
145 const int lhs_rank = NumDimensions(lhs);
146 const int rhs_rank = NumDimensions(rhs);
147 const int batch_size = op_context->params->adj_x
148 ? lhs->dims->data[lhs_rank - 2]
149 : lhs->dims->data[lhs_rank - 1];
150 const int num_units = rhs->dims->data[rhs_rank - 1];
151
152 // Temp tensor for Transposed LHS;
153 {
154 node->temporaries->data[0] = op_data->scratch_tensor_index;
155 TfLiteTensor* scratch_buffer;
156 TF_LITE_ENSURE_OK(
157 context, GetTemporarySafe(context, node, /*index=*/0, &scratch_buffer));
158 TfLiteIntArray* scratch_buffer_size = TfLiteIntArrayCreate(lhs_rank);
159 for (int i = 0; i < lhs_rank - 2; ++i) {
160 scratch_buffer_size->data[i] = lhs->dims->data[i];
161 }
162 // Swap last two dimensions.
163 scratch_buffer_size->data[lhs_rank - 2] = lhs->dims->data[lhs_rank - 1];
164 scratch_buffer_size->data[lhs_rank - 1] = lhs->dims->data[lhs_rank - 2];
165
166 scratch_buffer->type = op_context->lhs->type;
167 scratch_buffer->allocation_type = kTfLiteArenaRw;
168 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scratch_buffer,
169 scratch_buffer_size));
170 }
171
172 // We need a temp buffer for the RHS if we need to transpose the RHS. We
173 // transpose by default, so that the two inputs (LHS and RHS) are in a proper
174 // layout for our fast matrix multiplication routines. If the transpose flag
175 // is set by the caller, the data is already in the desired layout.
176 {
177 node->temporaries->data[1] = op_data->scratch_tensor_index + 1;
178 TfLiteTensor* scratch_buffer;
179 TF_LITE_ENSURE_OK(
180 context, GetTemporarySafe(context, node, /*index=*/1, &scratch_buffer));
181 const TfLiteTensor* rhs = op_context->rhs;
182 int rhs_rank = NumDimensions(rhs);
183 TfLiteIntArray* scratch_buffer_size = TfLiteIntArrayCreate(rhs_rank);
184 for (int i = 0; i < rhs_rank - 2; ++i) {
185 scratch_buffer_size->data[i] = rhs->dims->data[i];
186 }
187 // Swap last two dimensions.
188 scratch_buffer_size->data[rhs_rank - 2] = rhs->dims->data[rhs_rank - 1];
189 scratch_buffer_size->data[rhs_rank - 1] = rhs->dims->data[rhs_rank - 2];
190
191 if (IsConstantTensor(op_context->rhs)) {
192 scratch_buffer->allocation_type = kTfLiteArenaRwPersistent;
193 } else {
194 scratch_buffer->allocation_type = kTfLiteArenaRw;
195 }
196 scratch_buffer->type = op_context->rhs->type;
197 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scratch_buffer,
198 scratch_buffer_size));
199 }
200
201 // If we have to perform on-the-fly quantization (with quantized weights and
202 // float inputs) first we need to quantize the inputs. Allocate temporary
203 // buffer to store the intermediate quantized values, the batch scaling
204 // factors, the accumulator buffer (optimized version), the input offsets,
205 // and the sums of the rows for each weights matrix.
206 // RHS = weights, LHS = inputs
207 if (is_hybrid) {
208 // Calculate the total number of LHS batches.
209 int num_batches = 1;
210 for (int i = 0; i < lhs_rank - 2; ++i) {
211 num_batches *= lhs->dims->data[i];
212 }
213 int num_weights_matrices = 1;
214 for (int i = 0; i < rhs_rank - 2; ++i) {
215 num_weights_matrices *= rhs->dims->data[i];
216 }
217 op_data->compute_row_sums = true;
218 node->temporaries->data[2] = op_data->scratch_tensor_index + 2;
219 TfLiteTensor* input_quantized;
220 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/2,
221 &input_quantized));
222 input_quantized->type = op_context->rhs->type;
223 input_quantized->allocation_type = kTfLiteArenaRw;
224
225 TfLiteIntArray* input_quantized_size =
226 TfLiteIntArrayCopy(op_context->lhs->dims);
227 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_quantized,
228 input_quantized_size));
229
230 node->temporaries->data[3] = op_data->scratch_tensor_index + 3;
231 TfLiteTensor* scaling_factors;
232 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/3,
233 &scaling_factors));
234 scaling_factors->type = kTfLiteFloat32;
235 scaling_factors->allocation_type = kTfLiteArenaRw;
236 // Total size of scaling factors is batch size * number of total batches
237 int scaling_dims[1] = {num_batches * batch_size};
238 if (!TfLiteIntArrayEqualsArray(scaling_factors->dims, 1, scaling_dims)) {
239 TfLiteIntArray* scaling_factors_size = TfLiteIntArrayCreate(1);
240 scaling_factors_size->data[0] = batch_size;
241 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, scaling_factors,
242 scaling_factors_size));
243 }
244
245 node->temporaries->data[4] = op_data->scratch_tensor_index + 4;
246 TfLiteTensor* accum_scratch;
247 TF_LITE_ENSURE_OK(
248 context, GetTemporarySafe(context, node, /*index=*/4, &accum_scratch));
249 accum_scratch->type = kTfLiteInt32;
250 accum_scratch->allocation_type = kTfLiteArenaRw;
251 int accum_scratch_dims[2] = {num_units, batch_size};
252 if (!TfLiteIntArrayEqualsArray(accum_scratch->dims, 2,
253 accum_scratch_dims)) {
254 TfLiteIntArray* accum_size = TfLiteIntArrayCreate(2);
255 accum_size->data[0] = num_units;
256 accum_size->data[1] = batch_size;
257 TF_LITE_ENSURE_OK(
258 context, context->ResizeTensor(context, accum_scratch, accum_size));
259 }
260
261 node->temporaries->data[5] = op_data->scratch_tensor_index + 5;
262 TfLiteTensor* input_offsets;
263 TF_LITE_ENSURE_OK(
264 context, GetTemporarySafe(context, node, /*index=*/5, &input_offsets));
265 input_offsets->type = kTfLiteInt32;
266 input_offsets->allocation_type = kTfLiteArenaRw;
267 if (!TfLiteIntArrayEqualsArray(input_offsets->dims, 1, scaling_dims)) {
268 TfLiteIntArray* input_offsets_size = TfLiteIntArrayCreate(1);
269 input_offsets_size->data[0] = num_batches * batch_size;
270 TF_LITE_ENSURE_OK(context, context->ResizeTensor(context, input_offsets,
271 input_offsets_size));
272 }
273 node->temporaries->data[6] = op_data->scratch_tensor_index + 6;
274 TfLiteTensor* row_sums;
275 TF_LITE_ENSURE_OK(context,
276 GetTemporarySafe(context, node, /*index=*/6, &row_sums));
277 row_sums->type = kTfLiteInt32;
278 row_sums->allocation_type = kTfLiteArenaRwPersistent;
279 int row_sums_dims[1] = {num_weights_matrices * num_units};
280 if (!TfLiteIntArrayEqualsArray(row_sums->dims, 1, row_sums_dims)) {
281 TfLiteIntArray* row_sums_size = TfLiteIntArrayCreate(1);
282 row_sums_size->data[0] = row_sums_dims[0];
283 TF_LITE_ENSURE_OK(
284 context, context->ResizeTensor(context, row_sums, row_sums_size));
285 }
286 }
287
288 return kTfLiteOk;
289}

291TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
292 TF_LITE_ENSURE_EQ(context, NumInputs(node), 2);
293 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 1);
294
295 OpContext op_context(context, node);
296 TF_LITE_ENSURE_OK(context, InitializeTemporaries(context, node, &op_context));
297 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
298
299 bool adj_x = op_context.params->adj_x;
300 bool adj_y = op_context.params->adj_y;
301
302 const TfLiteTensor* lhs_data;
303 TF_LITE_ENSURE_OK(context,
304 GetInputSafe(context, node, kInputLHSTensor, &lhs_data));
305 const TfLiteTensor* rhs_data;
306 TF_LITE_ENSURE_OK(context,
307 GetInputSafe(context, node, kInputRHSTensor, &rhs_data));
308 TfLiteTensor* output;
309 TF_LITE_ENSURE_OK(context,
310 GetOutputSafe(context, node, kOutputTensor, &output));
311
312 // Note that quantized inference requires that all tensors have their
313 // parameters set. This is usually done during quantized training.
314 if (lhs_data->type == kTfLiteInt8 || lhs_data->type == kTfLiteInt16) {
315 double real_multiplier = 0.0;
316 TF_LITE_ENSURE_STATUS(GetQuantizedConvolutionMultipler(
317 context, lhs_data, rhs_data, output, &real_multiplier));
318 int exponent;
319 QuantizeMultiplier(real_multiplier, &op_data->output_multiplier, &exponent);
320 op_data->output_shift = exponent;
321 // BatchMatMul has no fused activation functions. Therefore, set
322 // output activation min and max to min and max of int8_t or int16_t
323 // type.
324 if (lhs_data->type == kTfLiteInt8) {
325 op_data->output_activation_min = std::numeric_limits<int8_t>::min();
326 op_data->output_activation_max = std::numeric_limits<int8_t>::max();
327 } else {
328 op_data->output_activation_min = std::numeric_limits<int16_t>::min();
329 op_data->output_activation_max = std::numeric_limits<int16_t>::max();
330 }
331 }
332
333 if (lhs_data->type == kTfLiteInt16) {
334 TF_LITE_ENSURE_EQ(context, lhs_data->params.zero_point, 0);
335 TF_LITE_ENSURE_EQ(context, rhs_data->params.zero_point, 0);
336 TF_LITE_ENSURE_EQ(context, output->params.zero_point, 0);
337 }
338
339 TF_LITE_ENSURE(context, lhs_data->type == kTfLiteFloat32 ||
340 lhs_data->type == kTfLiteInt8 ||
341 lhs_data->type == kTfLiteInt16);
342 TF_LITE_ENSURE(context, rhs_data->type == kTfLiteFloat32 ||
343 rhs_data->type == kTfLiteInt8 ||
344 rhs_data->type == kTfLiteInt16);
345 // Either we have a hybrid quantization with a float32 and an int8 input,
346 // otherwise both inputs should be of the same type.
347 TF_LITE_ENSURE(context, (lhs_data->type == kTfLiteFloat32 &&
348 rhs_data->type == kTfLiteInt8) ||
349 lhs_data->type == rhs_data->type);
350 // Support dimensions between 2 and 4, inclusive.
351 TF_LITE_ENSURE(context, NumDimensions(lhs_data) >= 2);
352 TF_LITE_ENSURE(context, NumDimensions(lhs_data) <= 4);
353 TF_LITE_ENSURE(context, NumDimensions(rhs_data) >= 2);
354 TF_LITE_ENSURE(context, NumDimensions(rhs_data) <= 4);
355
356 const int lhs_rank = NumDimensions(lhs_data);
357 const int rhs_rank = NumDimensions(rhs_data);
358 const int output_rank = std::max(lhs_rank, rhs_rank);
359 const RuntimeShape extended_lhs_shape =
360 RuntimeShape::ExtendedShape(output_rank, GetTensorShape(lhs_data));
361 const RuntimeShape extended_rhs_shape =
362 RuntimeShape::ExtendedShape(output_rank, GetTensorShape(rhs_data));
363
364 // Ensure any batch dimensions obey broacasting rules.
365 for (int i = 0; i < output_rank - 2; ++i) {
366 const int lhs_dim = extended_lhs_shape.Dims(i);
367 const int rhs_dim = extended_rhs_shape.Dims(i);
368 if (lhs_dim != rhs_dim) {
369 if (lhs_dim != 1) {
370 TF_LITE_ENSURE_EQ(context, rhs_dim, 1);
371 }
372 }
373 }
374 // Ensure other dimensions work for matrix multiplication.
375 int accum_dim_lhs = adj_x ? extended_lhs_shape.Dims(output_rank - 2)
376 : extended_lhs_shape.Dims(output_rank - 1);
377 int accum_dim_rhs = adj_y ? extended_rhs_shape.Dims(output_rank - 1)
378 : extended_rhs_shape.Dims(output_rank - 2);
379
380 TF_LITE_ENSURE_EQ(context, accum_dim_lhs, accum_dim_rhs);
381 TfLiteStatus status =
382 ResizeOutputTensor(context, extended_lhs_shape, extended_rhs_shape, adj_x,
383 adj_y, output_rank, output);
384 return status;
385}

387template <typename scalar>
388void TransposeRowsColumnsImpl(const TfLiteTensor* tensor_in,
389 const scalar* input, TfLiteTensor* tensor_out,
390 scalar* output) {
391 RuntimeShape transposed_shape(GetTensorShape(tensor_in));
392 RuntimeShape shape(GetTensorShape(tensor_in));
393 TransposeParams params;
394 int rank = NumDimensions(tensor_in);
395 params.perm_count = rank;
396 for (int i = 0; i < rank - 2; ++i) {
397 params.perm[i] = i;
398 }
399 // Transpose the last two dimensions.
400 params.perm[rank - 2] = rank - 1;
401 params.perm[rank - 1] = rank - 2;
402 transposed_shape.SetDim(rank - 1, shape.Dims(rank - 2));
403 transposed_shape.SetDim(rank - 2, shape.Dims(rank - 1));
404 optimized_ops::Transpose(params, shape, input, transposed_shape, output);
405}
406
407TfLiteStatus TransposeRowsColumns(TfLiteContext* context,
408 const TfLiteTensor* tensor_in,
409 TfLiteTensor* tensor_out) {
410 if (tensor_in->type == kTfLiteFloat32) {
411 TransposeRowsColumnsImpl<float>(tensor_in, GetTensorData<float>(tensor_in),
412 tensor_out,
413 GetTensorData<float>(tensor_out));
414 return kTfLiteOk;
415 } else if (tensor_in->type == kTfLiteInt8) {
416 TransposeRowsColumnsImpl<int8_t>(
417 tensor_in, GetTensorData<int8_t>(tensor_in), tensor_out,
418 GetTensorData<int8_t>(tensor_out));
419 return kTfLiteOk;
420 } else if (tensor_in->type == kTfLiteInt16) {
421 TransposeRowsColumnsImpl<int16_t>(
422 tensor_in, GetTensorData<int16_t>(tensor_in), tensor_out,
423 GetTensorData<int16_t>(tensor_out));
424 return kTfLiteOk;
425 } else {
426 TF_LITE_KERNEL_LOG(
427 context, "Can only transpose tensors with float, int8 or int16 type.");
428 return kTfLiteError;
429 }
430}

432RuntimeShape SwapRowColumnDims(const RuntimeShape& shape) {
433 RuntimeShape swapped_shape(shape);
434 const int32_t dims = shape.DimensionsCount();
435 swapped_shape.SetDim(dims - 2, shape.Dims(dims - 1));
436 swapped_shape.SetDim(dims - 1, shape.Dims(dims - 2));
437 return swapped_shape;
438}

440template <KernelType kernel_type>
441TfLiteStatus EvalHybrid(TfLiteContext* context, TfLiteNode* node, OpData* data,
442 const RuntimeShape& input_shape,
443 const TfLiteTensor* input,
444 const RuntimeShape& filter_shape,
445 const TfLiteTensor* filter,
446 TfLiteTensor* input_quantized,
447 TfLiteTensor* scaling_factors,
448 TfLiteTensor* accum_scratch, TfLiteTensor* row_sums,
449 TfLiteTensor* input_offsets, TfLiteTensor* output) {
450 const auto* params =
451 reinterpret_cast<TfLiteBatchMatMulParams*>(node->builtin_data);
452 const int32_t num_input_dims = input_shape.DimensionsCount();
453
454 // Input row/cols have been swapped at this point, so dims are
455 // {input_size, num_batches}
456 const int input_size = input_shape.Dims(num_input_dims - 2);
457 const int batch_size = input_shape.Dims(num_input_dims - 1);
458
459 int num_batches_to_quantize = batch_size;
460 for (int i = 0; i < input_shape.DimensionsCount() - 2; ++i) {
461 num_batches_to_quantize *= input_shape.Dims(i);
462 }
463 // Quantize input from float to uint8 + quantization params (scaling factor).
464 float* scaling_factors_ptr = GetTensorData<float>(scaling_factors);
465 int32_t* input_offset_ptr = nullptr;
466 int32_t* row_sums_ptr = nullptr;
467 input_offset_ptr = GetTensorData<int32_t>(input_offsets);
468 row_sums_ptr = GetTensorData<int32_t>(row_sums);
469 if (!params->asymmetric_quantize_inputs) {
470 memset(input_offset_ptr, 0, input_offsets->bytes);
471 }
472 int8_t* quant_data = GetTensorData<int8_t>(input_quantized);
473 const int8_t* filter_data = GetTensorData<int8_t>(filter);
474 const float* input_ptr = GetTensorData<float>(input);
475 // Quantize each batch independently.
476 tensor_utils::BatchQuantizeFloats(input_ptr, num_batches_to_quantize,
477 input_size, quant_data, scaling_factors_ptr,
478 input_offset_ptr,
479 params->asymmetric_quantize_inputs);
480 for (int b = 0; b < num_batches_to_quantize; ++b) {
481 // Incorporate scaling of the filter.
482 scaling_factors_ptr[b] *= filter->params.scale;
483 }
484
485 RuntimeShape output_shape = GetTensorShape(output);
486 int output_size = 1;
487 for (int i = 0; i < output_shape.DimensionsCount(); ++i) {
488 output_size *= output_shape.Dims(i);
489 }
490 std::fill_n(GetTensorData<float>(output), output_size, 0.0f);
491 if (kernel_type == kGenericOptimized) {
492 optimized_ops::BatchMatMul(
493 filter_shape, filter_data, input_shape, quant_data, scaling_factors_ptr,
494 input_offset_ptr, row_sums_ptr, GetTensorShape(output),
495 GetTensorData<int32_t>(accum_scratch), GetTensorData<float>(output),
496 &(data->compute_row_sums), CpuBackendContext::GetFromContext(context));
497 } else {
498 reference_ops::BatchMatMul(
499 filter_shape, filter_data, input_shape, quant_data, scaling_factors_ptr,
500 input_offset_ptr, row_sums_ptr, GetTensorShape(output),
501 GetTensorData<float>(output), &(data->compute_row_sums));
502 }
503
504 return kTfLiteOk;
505}

507template <KernelType kernel_type>
508TfLiteStatus EvalInt8(TfLiteContext* context, const OpData* data,
509 const RuntimeShape& lhs_shape, const TfLiteTensor* lhs,
510 const RuntimeShape& rhs_shape, const TfLiteTensor* rhs,
511 const RuntimeShape& output_shape, TfLiteTensor* output) {
512 // Reuse params struct from FullyConnected Op.
513 FullyConnectedParams op_params;
514 int32_t input_offset = -lhs->params.zero_point;
515 int32_t filter_offset = -rhs->params.zero_point;
516 int32_t output_offset = output->params.zero_point;
517 op_params.input_offset = input_offset;
518 op_params.weights_offset = filter_offset;
519 op_params.output_offset = output_offset;
520 op_params.output_multiplier = data->output_multiplier;
521 op_params.output_shift = data->output_shift;
522 op_params.quantized_activation_min = data->output_activation_min;
523 op_params.quantized_activation_max = data->output_activation_max;
524 op_params.lhs_cacheable = IsConstantTensor(lhs);
525 op_params.rhs_cacheable = IsConstantTensor(rhs);
526
527 if (kernel_type == kReference) {
528 reference_ops::BatchMatMul<int8_t, int32_t>(
529 op_params, rhs_shape, GetTensorData<int8_t>(rhs), lhs_shape,
530 GetTensorData<int8_t>(lhs), GetTensorShape(output),
531 GetTensorData<int8_t>(output));
532 } else {
533 optimized_ops::BatchMatMul(op_params, rhs_shape, GetTensorData<int8_t>(rhs),
534 lhs_shape, GetTensorData<int8_t>(lhs),
535 GetTensorShape(output),
536 GetTensorData<int8_t>(output),
537 CpuBackendContext::GetFromContext(context));
538 }
539 return kTfLiteOk;
540}

569template <KernelType kernel_type>
570TfLiteStatus EvalQuantized(TfLiteContext* context, TfLiteNode* node,
571 OpData* data, const RuntimeShape& lhs_shape,
572 const TfLiteTensor* lhs,
573 const RuntimeShape& rhs_shape,
574 const TfLiteTensor* rhs, TfLiteTensor* output) {
575 if (lhs->type == kTfLiteFloat32 && rhs->type == kTfLiteInt8) {
576 TfLiteTensor* input_quantized;
577 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/2,
578 &input_quantized));
579 TfLiteTensor* scaling_factors;
580 TF_LITE_ENSURE_OK(context, GetTemporarySafe(context, node, /*index=*/3,
581 &scaling_factors));
582 TfLiteTensor* accum_scratch;
583 TF_LITE_ENSURE_OK(
584 context, GetTemporarySafe(context, node, /*index=*/4, &accum_scratch));
585 TfLiteTensor* input_offsets;
586 TF_LITE_ENSURE_OK(
587 context, GetTemporarySafe(context, node, /*index=*/5, &input_offsets));
588 TfLiteTensor* row_sums;
589 TF_LITE_ENSURE_OK(context,
590 GetTemporarySafe(context, node, /*index=*/6, &row_sums));
591 return EvalHybrid<kernel_type>(
592 context, node, data, lhs_shape, lhs, rhs_shape, rhs, input_quantized,
593 scaling_factors, accum_scratch, row_sums, input_offsets, output);
594 } else if (lhs->type == kTfLiteInt8 && rhs->type == kTfLiteInt8) {
595 return EvalInt8<kernel_type>(context, data, lhs_shape, lhs, rhs_shape, rhs,
596 GetTensorShape(output), output);
597 } else if (lhs->type == kTfLiteInt16 && rhs->type == kTfLiteInt16) {
598 return EvalInt16<kernel_type>(context, data, lhs_shape, lhs, rhs_shape, rhs,
599 GetTensorShape(output), output);
600 } else {
601 TF_LITE_KERNEL_LOG(
602 context,
603 "Currently only hybrid, int8 and int16 quantization are supported.\n");
604 return kTfLiteError;
605 }
606 return kTfLiteOk;
607}

609TfLiteTensor* GetTempRhs(TfLiteContext* context, TfLiteNode* node,
610 const TfLiteTensor* rhs) {
611 TfLiteTensor* transposed_rhs = GetTemporary(context, node, 1);
612 if (transposed_rhs == nullptr) {
613 return nullptr;
614 }
615
616 if (rhs->type == kTfLiteInt8 || rhs->type == kTfLiteInt16) {
617 // Get the quantization params from the RHS tensor.
618 transposed_rhs->params.scale = rhs->params.scale;
619 transposed_rhs->params.zero_point = rhs->params.zero_point;
620 }
621 return transposed_rhs;
622}

624TfLiteTensor* GetTempLhs(TfLiteContext* context, TfLiteNode* node,
625 const TfLiteTensor* lhs) {
626 TfLiteTensor* transposed_lhs = GetTemporary(context, node, 0);
627 if (transposed_lhs == nullptr) {
628 return nullptr;
629 }
630
631 if (lhs->type == kTfLiteInt8 || lhs->type == kTfLiteInt16) {
632 // Get the quantization params from the LHS tensor.
633 transposed_lhs->params.scale = lhs->params.scale;
634 transposed_lhs->params.zero_point = lhs->params.zero_point;
635 }
636 return transposed_lhs;

639// Perform a batch matrix multiply on
640// LHS <..., A, B> X RHS<..., B, C>
641// where the leading dimensions of LHS and RHS obey broadcasting rules
642// (this Op will apply broadcasting rules).
643// We assume that LHS and RHS are both row oriented (adjacent values in memory
644// are in the same row) and will output in the same memory layout. However,
645// our fast GEMM libraries assume RCC layout (LHS row oriented,
646// RHS column oriented, output column oriented). Therefore, we perform
647// RHS <..., C, B> X LHS <..., B, A>
648// where output is a C X A column-oriented, which is equivalent to
649// A X C row-oriented.
650template <KernelType kernel_type>
651TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
652 OpContext op_context(context, node);
653 OpData* op_data = reinterpret_cast<OpData*>(node->user_data);
654 const TfLiteTensor* lhs;
655 TF_LITE_ENSURE_OK(context,
656 GetInputSafe(context, node, kInputLHSTensor, &lhs));
657 const TfLiteTensor* rhs;
658 TF_LITE_ENSURE_OK(context,
659 GetInputSafe(context, node, kInputRHSTensor, &rhs));
660 TfLiteTensor* output;
661 TF_LITE_ENSURE_OK(context,
662 GetOutputSafe(context, node, kOutputTensor, &output));
663 RuntimeShape orig_lhs_shape = GetTensorShape(lhs);
664 RuntimeShape orig_rhs_shape = GetTensorShape(rhs);
665
666 bool adj_y = op_context.params->adj_y;
667 bool adj_x = op_context.params->adj_x;
668
669 const TfLiteTensor* rhs_tensor = adj_y ? rhs : GetTempRhs(context, node, rhs);
670 const TfLiteTensor* lhs_tensor = adj_x ? GetTempLhs(context, node, lhs) : lhs;
671 if (!adj_y) {
672 // TODO(b/154760341) Constant tensors should already be transposed, but
673 // we transpose once if necessary for now.
674 if (!(IsConstantTensor(rhs) && op_data->rhs_transposed)) {
675 TransposeRowsColumns(context, rhs, GetTemporary(context, node, 1));
676 op_data->rhs_transposed = true;
677 }
678 }
679 if (adj_x) {
680 TransposeRowsColumns(context, lhs, GetTemporary(context, node, 0));
681 }
682 RuntimeShape rhs_shape =
683 adj_y ? orig_rhs_shape : SwapRowColumnDims(orig_rhs_shape);
684 RuntimeShape lhs_shape =
685 adj_x ? orig_lhs_shape : SwapRowColumnDims(orig_lhs_shape);
686
687 switch (rhs->type) {
688 case kTfLiteFloat32:
689 // Note we pass RHS args first, LHS args second. See note above.
690 if (kernel_type == kGenericOptimized) {
691 optimized_ops::BatchMatMul(rhs_shape, GetTensorData<float>(rhs_tensor),
692 lhs_shape, GetTensorData<float>(lhs_tensor),
693 GetTensorShape(output),
694 GetTensorData<float>(output),
695 CpuBackendContext::GetFromContext(context));
696 } else {
697 reference_ops::BatchMatMul(rhs_shape, GetTensorData<float>(rhs_tensor),
698 lhs_shape, GetTensorData<float>(lhs_tensor),
699 GetTensorShape(output),
700 GetTensorData<float>(output));
701 }
702 break;
703 case kTfLiteInt8:
704 case kTfLiteInt16:
705 EvalQuantized<kernel_type>(context, node, op_data, lhs_shape, lhs_tensor,
706 rhs_shape, rhs_tensor, output);
707 break;
708 default:
709 TF_LITE_KERNEL_LOG(context,
710 "Currently BatchMatMul doesn't support type: %s",
711 TfLiteTypeGetName(lhs->type));
712 return kTfLiteError;
713 }
714 return kTfLiteOk;
715}
