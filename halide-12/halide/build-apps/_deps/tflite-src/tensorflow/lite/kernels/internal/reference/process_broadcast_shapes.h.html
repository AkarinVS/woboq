24// Consolidates dimensions in broadcast inputs, checks for five-fold pattern.
25//
26// For example, if sequence of dimensions of one input is
27// ..., 1, 3, 1, 7, 9, 5,... and the other is ..., 2, 3, 1, 7, 1, 1, ...
28// we can consolidate these as
29// ..., 1, 3*7, 9*5, ... and 2, 3*7, 1.
30//
31// The category is updated in the less-frequent case of shapes that are
32// not suited to a fivefold-loop broadcast.
33//
34// Falls back to generic pattern when it does not know how to process properly.
35//
36// Returns true iff there is some sort of broadcast, which includes five-fold
37// patterns and falling back to generic broadcast.
38inline bool ProcessBroadcastShapes(const RuntimeShape& shape0,
39 const RuntimeShape& shape1,
40 tflite::ArithmeticParams* params) {
41 const int dims_count =
42 std::max(shape0.DimensionsCount(), shape1.DimensionsCount());
43
44 params->broadcast_category = BroadcastableOpCategory::kGenericBroadcast;
45 RuntimeShape scalar_shape(dims_count, 1);
46
47 auto extended_shape0 = RuntimeShape::ExtendedShape(dims_count, shape0);
48 auto extended_shape1 = RuntimeShape::ExtendedShape(dims_count, shape1);
49
50 // Check for "exact" match, implicitly accepting any scalar shapes.
51 if (extended_shape0 == extended_shape1) {
52 params->broadcast_category = BroadcastableOpCategory::kNonBroadcast;
53 return false;
54 }
55
56 for (int i = dims_count - 1; i >= 0; --i) {
57 if (extended_shape0.Dims(i) == extended_shape1.Dims(i)) {
58 continue;
59 } else if (extended_shape0.Dims(i) == 1) {
60 params->broadcast_category =
61 BroadcastableOpCategory::kFirstInputBroadcastsFast;
62 break;
63 } else if (extended_shape1.Dims(i) == 1) {
64 params->broadcast_category =
65 BroadcastableOpCategory::kSecondInputBroadcastsFast;
66 break;
67 } else {
68 // This case is erroneous: there is a dimension that does not match and
69 // is not a broadcast from one shape to the other.
70 params->broadcast_category = BroadcastableOpCategory::kGenericBroadcast;
71 return true;
72 }
73 }
74
75 if (params->broadcast_category !=
76 BroadcastableOpCategory::kFirstInputBroadcastsFast &&
77 params->broadcast_category !=
78 BroadcastableOpCategory::kSecondInputBroadcastsFast) {
79 // This is unreachable because at least one else clause in the above loop
80 // must be reached.
81 TFLITE_DCHECK(false);
82 params->broadcast_category = BroadcastableOpCategory::kNonBroadcast;
83 return false;
84 }
85
86 // From this point it is assumed contractually that corresponding dimensions
87 // in shape0 and shape1 are either (a) equal or (b) one or other equals 1.
88 const bool swap_inputs = params->broadcast_category ==
89 BroadcastableOpCategory::kSecondInputBroadcastsFast;
90 const RuntimeShape* shape_a =
91 swap_inputs ? &extended_shape1 : &extended_shape0;
92 const RuntimeShape* shape_b =
93 swap_inputs ? &extended_shape0 : &extended_shape1;
94
95 int i = dims_count - 1;
96 params->broadcast_shape[0] = 1;
97 params->broadcast_shape[1] = 1;
98 params->broadcast_shape[2] = 1;
99 params->broadcast_shape[3] = 1;
100 params->broadcast_shape[4] = 1;
101 // y_0 is greedy: include dims if both or neither equal 1: in other words,
102 // test for equality rather than (shape_a->Dims(i) != 1).
103 while (i >= 0 && shape_a->Dims(i) == shape_b->Dims(i)) {
104 params->broadcast_shape[4] *= shape_b->Dims(i);
105 --i;
106 }
107 // Here either input_a or input_b has dim of 1 (if i >= 0). If it is input_b
108 // that has the unit dimension, the next two loops are not entered.
109 while (i >= 0 && shape_a->Dims(i) == 1) {
110 params->broadcast_shape[3] *= shape_b->Dims(i);
111 --i;
112 }
113 while (i >= 0 && shape_a->Dims(i) == shape_b->Dims(i)) {
114 params->broadcast_shape[2] *= shape_a->Dims(i);
115 --i;
116 }
117 // Here either input_a or input_b has dim of 1 (if i >= 0).
118 while (i >= 0 && shape_b->Dims(i) == 1) {
119 params->broadcast_shape[1] *= shape_a->Dims(i);
120 --i;
121 }
122 while (i >= 0 && shape_a->Dims(i) == shape_b->Dims(i)) {
123 params->broadcast_shape[0] *= shape_b->Dims(i);
124 --i;
125 }
126
127 // Rarer case is when the broadcast dimensions cannot be handled by a fivefold
128 // loop.
129 if (i >= 0) {
130 params->broadcast_category = BroadcastableOpCategory::kGenericBroadcast;
131 }
132 return true;
133}
