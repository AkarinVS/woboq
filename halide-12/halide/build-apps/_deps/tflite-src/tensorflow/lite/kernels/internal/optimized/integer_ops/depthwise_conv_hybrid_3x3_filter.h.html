128 public:
129 static inline void Run(const float* input_scale,
130 const int8* input_ptr,
131 const int8* filter_ptr, const float* bias_ptr,
132 float* output_ptr, int64_t input_depth,
133 int64_t input_row_size, int32 output_window_height,
134 int32 output_window_width,
135 const float* per_channel_scales,
136 const DepthwiseConvParams* params_ptr) {
137 const int64_t input_width_increment = 2 * input_depth;
138 const int64_t input_height_increment = 2 * input_row_size;
139 const int64_t output_height_increment = 2 * 4 * params_ptr->output_row_size;
140 TFLITE_DCHECK_EQ(params_ptr->filter_offset, 0);

142#define DEPTHWISECONV_LABEL_HEIGHT_2_LOOP "1"
143#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP "2"
144#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_1_LEFTOVER "3"
145#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LEFTOVER "4"
146#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_AFTER_LOOP "5"
147#define DEPTHWISECONV_LABEL_HEIGHT_2_AFTER_LOOP "6"
148#define DEPTHWISECONV_LABEL_HEIGHT_1 "7"
149#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_2_LOOP "8"
150#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_1_LEFTOVER "9"
151#define DEPTHWISECONV_LABEL_HEIGHT_1_WIDTH_2_LEFTOVER "10"
152#define DEPTHWISECONV_LABEL_HEIGHT_1_END "11"
153
154 asm volatile(
155 // Performs depthwise convolutions for a window specified by
156 // |output_window_height| and |output_window_width|. The inner-most loop
157 // processes 2x2 outputs, and any leftovers at the end.
158 //
159 // Algorithm works as follows:
160 //
161 // 1. Load filters of 8 depth (8x3x3). Registers v0--v8 hold filter
162 // values.
163 // 2. For 2 output heights at a time:
164 // i. For 2 output widths at a time, load inputs for a 2x1 (2
165 // height, 1 width) output window (4x3 input window).
166 // Registers v9--v20 hold input values. Mul-add with
167 // accumulators v21--v24. Then run activation, downquantize
168 // and store. Repeat for the next 2x1 output window,
169 // leveraging overlapping inputs.
170 // ii. Handle single leftover width if exists.
171 // 3. Handle single leftover height if exists.
172 // i. For 2 output widths at a time, load inputs for a 1x2 (1
173 // height, 2 width) output window (3x4 input window).
174 // Registers v9--v20 hold input values. Mul-add with
175 // accumulators v21--v24. Then run activation, downquantize
176 // and store. Repeat for the next 1x2 output window,
177 // leveraging overlapping inputs.
178 // ii. Handle single leftover width if exists.
179 //
180 // Loads are placed as soon as the register is no longer needed and
181 // interleaved with arithmetic operations to take advantage of
182 // dual-issue pipelines. We also add input offsets as far from the loads
183 // as possible to give loads enough cycles to fetch data from memory.
184 //
185 // This logic is copied and modified from the non-per-channel quantized
186 // part.
187 // However, the challenges are how to plan the registers allocation
188 // wisely: 25 NEON registers are already reserved for inputs, filters,
189 // and outputs; also, 2 registers (v30, v31) are used for output
190 // min/max, while another 2 registers (v26, v29) are used for input
191 // offset & output offset, so that's total 25 + 2 + 2 = 29 already.
192 // But we need 4 more registers to hold the output multiplier & output
193 // right shift (we only have 3).
194 //
195 // So here's the plan:
196 // v27 (which held duplicated output multiplier previously) will hold
197 // the first 4 values of the output_multiplier_ptr (we have 8 in total);
198 // v30 (which held duplicated output right shift previously) will hold
199 // the first 4 values of the output_shift_ptr (we have 8 in total);
200 // lastly, v28 will hold the last 4 values of output_mulitplier and v31
201 // (previously occupied by activations) will hold the last 4 values of
202 // output_shift. Then v25 will be used for output activation min while
203 // output activation max will just reuse oother registers, like v24.
204 //
205 // Set "constant" registers. These registers may be replaced with temp
206 // values from time to time when there are not enough NEON registers.
207 // We use x9--x15 general purpose registers as they are caller-saved
208 // temporary registers (see
209 // http://infocenter.arm.com/help/topic/com.arm.doc.ihi0055b/IHI0055B_aapcs64.pdf). // NOLINT
210 "ldr w9, [%[params_ptr], #" STR(OFFSET_INPUT_OFFSET) "]\n"
211 "ldr x3, [%[params_ptr], #" STR(OFFSET_OUTPUT_DEPTH) "]\n"
212 "cmp %w[output_window_height], #2\n"
213 "ldr w4, [%[params_ptr], #" STR(OFFSET_FLOAT_OUTPUT_ACTIVATION_MIN) "]\n"
214 "ldr w0, [%[params_ptr], #" STR(OFFSET_FLOAT_OUTPUT_ACTIVATION_MAX) "]\n"
215 "dup v25.4s, w4\n"
216 "dup v29.4s, w0\n"
217 "ldr x1, [%[params_ptr], #" STR(OFFSET_OUTPUT_ROW_SIZE) "]\n"
218 "mov x4, #4\n"
219 "mul x1, x1, x4\n"
220 "mul x4, x4, x3\n"
221
222 // Load per_channel scales and bias (float).
223 "ldr w2, [%[input_scale]]\n"
224 "ld1 {v27.4s, v28.4s}, [%[per_channel_scales]]\n"
225 "ld1 {v30.4s, v31.4s}, [%[bias_ptr]]\n"
226 "dup v26.4s, w2\n"
227 "fmul v27.4s, v27.4s, v26.4s\n"
228 "fmul v28.4s, v28.4s, v26.4s\n"
229 "dup v26.8h, w9\n"
230
231 // Load filters and add offsets.
232 "ld1 {v0.8b}, [%[filter_ptr]], x3\n"
233 "ld1 {v1.8b}, [%[filter_ptr]], x3\n"
234 "sshll v0.8h, v0.8b, #0\n"
235 "ld1 {v2.8b}, [%[filter_ptr]], x3\n"
236 "sshll v1.8h, v1.8b, #0\n"
237 "ld1 {v3.8b}, [%[filter_ptr]], x3\n"
238 "sshll v2.8h, v2.8b, #0\n"
239 "ld1 {v4.8b}, [%[filter_ptr]], x3\n"
240 "sshll v3.8h, v3.8b, #0\n"
241 "ld1 {v5.8b}, [%[filter_ptr]], x3\n"
242 "sshll v4.8h, v4.8b, #0\n"
243 "ld1 {v6.8b}, [%[filter_ptr]], x3\n"
244 "sshll v5.8h, v5.8b, #0\n"
245 "ld1 {v7.8b}, [%[filter_ptr]], x3\n"
246 "sshll v6.8h, v6.8b, #0\n"
247 "ld1 {v8.8b}, [%[filter_ptr]], x3\n"
248 "sshll v7.8h, v7.8b, #0\n"
249 "sshll v8.8h, v8.8b, #0\n"
250
251 "blt " DEPTHWISECONV_LABEL_HEIGHT_2_AFTER_LOOP "f\n"
252
253 //"loop_%=:\n"
254 DEPTHWISECONV_LABEL_HEIGHT_2_LOOP ":\n"
255 // This loop processes 2x2 outputs. To avoid register exhaustion,
256 // inputs for the left 2 outputs are loaded first, then the right
257 // two outputs.
258 "mov x11, %[input_ptr]\n"
259 "mov x12, x11\n"
260 "ld1 {v9.8b}, [x12], %[input_depth]\n"
261 "add x13, x11, %[input_row_size]\n"
262 "ld1 {v10.8b}, [x12], %[input_depth]\n"
263 "add x14, x13, %[input_row_size]\n"
264 "ld1 {v11.8b}, [x12], %[input_depth]\n"
265 "add x15, x14, %[input_row_size]\n"
266 "ld1 {v12.8b}, [x13], %[input_depth]\n"
267 "mov w5, %w[output_window_width]\n"
268 "ld1 {v13.8b}, [x13], %[input_depth]\n"
269 "mov x6, %[output_ptr]\n"
270 "ld1 {v14.8b}, [x13], %[input_depth]\n"
271 "add x7, %[output_ptr], x1\n"
272 "ld1 {v15.8b}, [x14], %[input_depth]\n"
273 // The height 2 / width 2 loop loads an extra 2x1 outputs (2 height,
274 // 1 width) in anticipation for the next iteration. Make sure
275 // |output_window_width| is large enough to handle the additional
276 // loads, otherwise jump to specific the appropriate label to handle
277 // smaller widths.
278 "cmp w5, #2\n"
279 "saddw v9.8h, v26.8h, v9.8b\n"
280 "ld1 {v16.8b}, [x14], %[input_depth]\n"
281 "saddw v10.8h, v26.8h, v10.8b\n"
282 "ld1 {v17.8b}, [x14], %[input_depth]\n"
283 "saddw v11.8h, v26.8h, v11.8b\n"
284 "ld1 {v18.8b}, [x15], %[input_depth]\n"
285 "saddw v12.8h, v26.8h, v12.8b\n"
286 "ld1 {v19.8b}, [x15], %[input_depth]\n"
287 "saddw v13.8h, v26.8h, v13.8b\n"
288 "ld1 {v20.8b}, [x15], %[input_depth]\n"
289 "saddw v14.8h, v26.8h, v14.8b\n"
290
291 "movi v21.4s, #0\n"
292 "saddw v15.8h, v26.8h, v15.8b\n"
293 "movi v22.4s, #0\n"
294 "saddw v16.8h, v26.8h, v16.8b\n"
295 "movi v23.4s, #0\n"
296 "saddw v17.8h, v26.8h, v17.8b\n"
297 "movi v24.4s, #0\n"
298
299 "saddw v18.8h, v26.8h, v18.8b\n"
300 "saddw v19.8h, v26.8h, v19.8b\n"
301 "saddw v20.8h, v26.8h, v20.8b\n"
302
303 "beq " DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LEFTOVER "f\n"
304 "cmp w5, #1\n"
305 "beq " DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_1_LEFTOVER "f\n"
306
307 //"loop_%=:\n"
308 DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP ":\n"
309 // Mul-add left outputs.
310 "smlal v21.4s, v0.4h, v9.4h\n"
311 "subs w5, w5, #2\n"
312 "smlal2 v22.4s, v0.8h, v9.8h\n"
313 "cmp w5, #3\n"
314 "smlal v23.4s, v0.4h, v12.4h\n"
315 "ld1 {v9.8b}, [x12]\n"
316 "smlal2 v24.4s, v0.8h, v12.8h\n"
317 "smlal v21.4s, v1.4h, v10.4h\n"
318 "smlal2 v22.4s, v1.8h, v10.8h\n"
319 "smlal v23.4s, v1.4h, v13.4h\n"
320 "smlal2 v24.4s, v1.8h, v13.8h\n"
321 "smlal v21.4s, v2.4h, v11.4h\n"
322 "smlal2 v22.4s, v2.8h, v11.8h\n"
323 "smlal v23.4s, v2.4h, v14.4h\n"
324 "smlal2 v24.4s, v2.8h, v14.8h\n"
325 "smlal v21.4s, v3.4h, v12.4h\n"
326 "smlal2 v22.4s, v3.8h, v12.8h\n"
327 "ld1 {v12.8b}, [x13]\n"
328 "smlal v23.4s, v3.4h, v15.4h\n"
329 "smlal2 v24.4s, v3.8h, v15.8h\n"
330 "smlal v21.4s, v4.4h, v13.4h\n"
331 "smlal2 v22.4s, v4.8h, v13.8h\n"
332 "smlal v23.4s, v4.4h, v16.4h\n"
333 "smlal2 v24.4s, v4.8h, v16.8h\n"
334 "smlal v21.4s, v5.4h, v14.4h\n"
335 "smlal2 v22.4s, v5.8h, v14.8h\n"
336 "smlal v23.4s, v5.4h, v17.4h\n"
337 "smlal2 v24.4s, v5.8h, v17.8h\n"
338 "smlal v21.4s, v6.4h, v15.4h\n"
339 "smlal2 v22.4s, v6.8h, v15.8h\n"
340 "ld1 {v15.8b}, [x14]\n"
341 "smlal v23.4s, v6.4h, v18.4h\n"
342 "smlal2 v24.4s, v6.8h, v18.8h\n"
343 "ld1 {v18.8b}, [x15]\n"
344 "smlal v21.4s, v7.4h, v16.4h\n"
345 "smlal2 v22.4s, v7.8h, v16.8h\n"
346 "smlal v23.4s, v7.4h, v19.4h\n"
347 "smlal2 v24.4s, v7.8h, v19.8h\n"
348 "smlal v21.4s, v8.4h, v17.4h\n"
349 "smlal2 v22.4s, v8.8h, v17.8h\n"
350 "smlal v23.4s, v8.4h, v20.4h\n"
351 "smlal2 v24.4s, v8.8h, v20.8h\n"
352
353 // Cast to float.
354 "scvtf v21.4s, v21.4s\n"
355 "scvtf v22.4s, v22.4s\n"
356 "scvtf v23.4s, v23.4s\n"
357 "scvtf v24.4s, v24.4s\n"
358 // Multiply by per channel scale.
359 "fmul v21.4s, v21.4s, v27.4s\n"
360 "fmul v22.4s, v22.4s, v28.4s\n"
361 "fmul v23.4s, v23.4s, v27.4s\n"
362 "fmul v24.4s, v24.4s, v28.4s\n"
363 // Add bias.
364 "fadd v21.4s, v21.4s, v30.4s\n"
365 "fadd v22.4s, v22.4s, v31.4s\n"
366 "fadd v23.4s, v23.4s, v30.4s\n"
367 "fadd v24.4s, v24.4s, v31.4s\n"
368 // Clamp range.
369 "fmax v21.4s, v21.4s, v25.4s\n"
370 "fmin v21.4s, v21.4s, v29.4s\n"
371 "fmax v22.4s, v22.4s, v25.4s\n"
372 "fmin v22.4s, v22.4s, v29.4s\n"
373 "fmax v23.4s, v23.4s, v25.4s\n"
374 "fmin v23.4s, v23.4s, v29.4s\n"
375 "fmax v24.4s, v24.4s, v25.4s\n"
376 "fmin v24.4s, v24.4s, v29.4s\n"
377 // Store to float.
378 "st1 {v21.4s, v22.4s}, [x6], x4\n"
379 "st1 {v23.4s, v24.4s}, [x7], x4\n"
380 // Reset to int
381 "fcvtms v21.4s, v21.4s\n"
382 "fcvtms v22.4s, v22.4s\n"
383 "fcvtms v23.4s, v23.4s\n"
384 "fcvtms v24.4s, v24.4s\n"
385
386 "movi v22.4s, #0\n"
387 "movi v24.4s, #0\n"
388 "saddw v9.8h, v26.8h, v9.8b\n"
389 "saddw v12.8h, v26.8h, v12.8b\n"
390 "saddw v15.8h, v26.8h, v15.8b\n"
391 "movi v21.4s, #0\n"
392 "saddw v18.8h, v26.8h, v18.8b\n"
393 "movi v23.4s, #0\n"
394
395 // Mul-add right outputs.
396 "smlal v21.4s, v0.4h, v10.4h\n"
397 "add x11, x11, %[input_width_increment]\n"
398 "smlal2 v22.4s, v0.8h, v10.8h\n"
399 "mov x12, x11\n"
400 "smlal v23.4s, v0.4h, v13.4h\n"
401 "add x13, x11, %[input_row_size]\n"
402 "smlal2 v24.4s, v0.8h, v13.8h\n"
403 "add x14, x13, %[input_row_size]\n"
404 "smlal v21.4s, v1.4h, v11.4h\n"
405 "add x15, x14, %[input_row_size]\n"
406 "smlal2 v22.4s, v1.8h, v11.8h\n"
407 "smlal v23.4s, v1.4h, v14.4h\n"
408 "smlal2 v24.4s, v1.8h, v14.8h\n"
409 "smlal v21.4s, v2.4h, v9.4h\n"
410 "smlal2 v22.4s, v2.8h, v9.8h\n"
411 "ld1 {v9.8b}, [x12], %[input_depth]\n"
412 "smlal v23.4s, v2.4h, v12.4h\n"
413 "ld1 {v10.8b}, [x12], %[input_depth]\n"
414 "smlal2 v24.4s, v2.8h, v12.8h\n"
415 "ld1 {v11.8b}, [x12], %[input_depth]\n"
416 "smlal v21.4s, v3.4h, v13.4h\n"
417 "smlal2 v22.4s, v3.8h, v13.8h\n"
418 "smlal v23.4s, v3.4h, v16.4h\n"
419 "smlal2 v24.4s, v3.8h, v16.8h\n"
420 "smlal v21.4s, v4.4h, v14.4h\n"
421 "smlal2 v22.4s, v4.8h, v14.8h\n"
422 "smlal v23.4s, v4.4h, v17.4h\n"
423 "smlal2 v24.4s, v4.8h, v17.8h\n"
424 "smlal v21.4s, v5.4h, v12.4h\n"
425 "smlal2 v22.4s, v5.8h, v12.8h\n"
426 "ld1 {v12.8b}, [x13], %[input_depth]\n"
427 "smlal v23.4s, v5.4h, v15.4h\n"

1072 const int64_t input_height_increment = 4 * input_row_size;
1073 const int64_t output_height_increment = 2 * 4 * params_ptr->output_row_size;
1074 TFLITE_DCHECK_EQ(params_ptr->filter_offset, 0);

1077#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_2_LOOP "2"
1078#define DEPTHWISECONV_LABEL_HEIGHT_2_WIDTH_1_LEFTOVER "3"

1087
1088 asm volatile(
1089 // Performs depthwise convolutions for a window specified by
1090 // |output_window_height| and |output_window_width|. The inner-most loop
1091 // processes 2x2 outputs, and any leftovers at the end.
1092 //
1093 // Algorithm works as follows:
1094 //
1095 // 1. Load filters of 8 depth (8x3x3). Registers v0--v8 hold filter
1096 // values.
1097 // 2. For 2 output heights at a time:
1098 // i. For 2 output widths at a time at stride 2, a 5x5 input
1099 // window is required. To avoid register exhaustion, we load
1100 // the first 2 rows of the 5x5 input window into registers
1101 // v9--v18, and use the same registers to load the next 2
1102 // rows, and finally v9--v13 to load the last row.
1103 // Accumulators for all 2x2 outputs are reserved by registers
1104 // v21-v22 (top left output), v23-v24 (top right output),
1105 // v19-v20 (bottom left output), v25-v26 (bottom right
1106 // output).
1107 // ii. Handle single leftover width if exists.
1108 // 3. Handle single leftover height if exists.
1109 // i. For 2 output widths at a time at stride 2, load inputs for
1110 // a 1x2 (1 height, 2 width) output window (3x5 input
1111 // window). Registers v9--v24 hold input values. Mul-add with
1112 // accumulators v24--v27.
1113 // ii. Handle single leftover width if exists.
1114 //
1115 // Loads are placed as soon as the register is no longer needed and
1116 // interleaved with arithmetic operations to take advantage of
1117 // dual-issue pipelines. We also add input offsets as far from the loads
1118 // as possible to give loads enough cycles to fetch data from memory.
1119 //
1120 // This logic is copied and modified from the non-per-channel quantized
1121 // part.
1122 // The register planning here is really tricky:
1123 // v0-v29 are all used at least once for either filter/input/output,
1124 // some of them are used for output shift and output mulitplier, or
1125 // input/output offset.
1126 // Only v30 & v31 are only used for output activation min/max.
1127 // For per-channel case, we need 4 registers to hold output shift &
1128 // output multiplier. However, given the reality, we simply cannot do
1129 // that without reloading.
1130 //
1131 // So here's the plan:
1132 // We hold output_multiplier in v30 & v31, and we will load output_shift
1133 // into two consecutive registers each time before use.
1134 // We will duplicate output min & max before needed.
1135 // Sometimes we may borrow registers from input offset or bias, we will
1136 // dup them back after use.
1137 //
1138
1139 // Set "constant" registers. These registers may be replaced with temp

3049 const RuntimeShape& bias_shape, const float* bias_data,

3053 DepthwiseConvParams params;
3054 const int32 stride_width = rt_params.stride_width;
3055 const int32 stride_height = rt_params.stride_height;
3056 const int32 pad_width = rt_params.padding_values.width;
3057 const int32 pad_height = rt_params.padding_values.height;
3058 const int32 depth_multiplier = rt_params.depth_multiplier;
3059 const float output_activation_min = rt_params.float_activation_min;
3060 const float output_activation_max = rt_params.float_activation_max;

3062
3063 params.input_depth = input_shape.Dims(3);
3064 params.input_width = input_shape.Dims(2);
3065 params.input_height = input_shape.Dims(1);
3066 params.input_row_size = params.input_depth * params.input_width;
3067 params.stride_width = stride_width; params.stride_height = stride_height;
3068 params.output_depth = MatchingDim(filter_shape, 3, output_shape, 3);
3069 params.output_width = output_shape.Dims(2);
3070 params.output_height = output_shape.Dims(1);
3071 params.output_row_size = params.output_depth * params.output_width;
3072 params.filter_offset = filter_offset;
3073 params.float_output_activation_min = output_activation_min;
3074 params.float_output_activation_max = output_activation_max;
3075
3076 const int32 filter_height = filter_shape.Dims(1);
3077 const int32 filter_width = filter_shape.Dims(2);
3078 params.filter_row_size = params.output_depth * filter_width;
3079
3080 // Algorithm assumes below constraints. It is optimized for depth
3081 // multiplier of 1, 3x3 filter, no padding and strides 1 and 2.
3082 TFLITE_DCHECK(params.output_depth == params.input_depth * depth_multiplier);
3083 TFLITE_DCHECK(depth_multiplier == 1);
3084 TFLITE_DCHECK(filter_height == 3);
3085 TFLITE_DCHECK(filter_width == 3);
3086 TFLITE_DCHECK(stride_height == 1 || stride_height == 2);
3087 TFLITE_DCHECK(stride_width == 1 || stride_width == 2);
3088 TFLITE_DCHECK(stride_width == stride_height);
3089 TFLITE_DCHECK(pad_height == 0 || pad_height == 1);
3090 TFLITE_DCHECK(pad_width == 0 || pad_width == 1);
3091 TFLITE_DCHECK(pad_width == pad_height);
3092 TFLITE_DCHECK(thread_dim == 0 || thread_dim == 1);

3094 const int32 batches = MatchingDim(input_shape, 0, output_shape, 0);
3095 const int64_t input_batch_size = params.input_row_size * params.input_height;
3096 const int64_t output_batch_size =
3097 params.output_row_size * params.output_height;
3098
3099 ShuffleParams one_row_shuffle_params, two_row_shuffle_params,
3100 four_row_shuffle_params, eight_row_shuffle_params;
3101 if (stride_width == 1) {
3102 one_row_shuffle_params = ShuffleParams(30, 1, 1, 1);
3103 two_row_shuffle_params = ShuffleParams(22, 2, 1, 1);
3104 four_row_shuffle_params = ShuffleParams(14, 4, 1, 1);
3105 eight_row_shuffle_params = ShuffleParams(8, 8, 1, 1);
3106 } else {
3107 one_row_shuffle_params = ShuffleParams(14, 1, 2, 2);
3108 two_row_shuffle_params = ShuffleParams(8, 2, 2, 2);
3109 four_row_shuffle_params = ShuffleParams(4, 4, 2, 2);
3110 eight_row_shuffle_params = ShuffleParams(2, 8, 2, 2);
3111 }
3112
3113 using conv_multirow_func_t =
3114 decltype(
3115 &DepthwiseConvHybridMultiRowPerChannel<output_rounding, 1, 1>::Run);
3116 conv_multirow_func_t conv_multirow_func =
3117 DepthwiseConvHybridMultiRowPerChannel<output_rounding, 1, 1>::Run;
3118 if (stride_width == 2) {
3119 conv_multirow_func =
3120 DepthwiseConvHybridMultiRowPerChannel<output_rounding, 2, 2>::Run;

3124 int8 shuffle_workspace[kDepthwiseConvScratchWorkspaceSize];

3127 int batch_end = batches;
3128 int row_start = 0;
3129 int row_end = params.output_height;

3134 TFLITE_DCHECK_LE(thread_end, batches);
3135 batch_start = thread_start;
3136 batch_end = thread_end;
3137 break;

3139 TFLITE_DCHECK_GE(thread_start, 0);
3140 TFLITE_DCHECK_LE(thread_end, params.output_height);
3141 row_start = thread_start;
3142 row_end = thread_end;

3148 const int8* input_ptr = input_data + b * input_batch_size;
3149 float* output_ptr = output_data + b * output_batch_size;
3150 params.input_offset = -input_offsets[b];

3152 int32 out_y = row_start;
3153 int32 end_x = params.output_width;
3154 int32 end_y = row_end;
3155 if (pad_width == 1 && pad_height == 1) {
3156 DepthwiseConvHybridHandlePaddingPerChannel<output_rounding>(
