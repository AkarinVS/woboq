34// Draws a sample from a categorical distribution.
35template <typename FloatType, typename IntegralType>
36TfLiteStatus MultinomialSample(std::default_random_engine& rng,
37 const FloatType* logits, int logits_size,
38 IntegralType* outputs, int output_size) {
39 // Computes arg_max(cumsum(exp(logits)) > rand()).
40 // TODO(b/169166131): Remove hard-coded double for constrained use-cases.
41 std::vector<double> cumulative_odds;
42 cumulative_odds.reserve(logits_size);
43 double last_odds = 0.0;
44
45 // Compute max logit for numerical stability.
46 FloatType max_logit = std::numeric_limits<FloatType>::lowest();
47 for (int i = 0; i < logits_size; i++) {
48 max_logit = std::max(max_logit, logits[i]);
49 }
50
51 for (int i = 0; i < logits_size; i++) {
52 FloatType odds = std::exp(logits[i] - max_logit) + last_odds;
53 cumulative_odds.push_back(odds);
54 last_odds = odds;
55 }
56
57 std::uniform_real_distribution<double> distribution{0.0,
58 cumulative_odds.back()};
59
60 for (int i = 0; i < output_size; i++) {
61 double sample = distribution(rng);
62 auto it = std::lower_bound(cumulative_odds.begin(), cumulative_odds.end(),
63 sample);
64 if (it == cumulative_odds.end()) {
65 // This should be impossible by construction.
66 return kTfLiteError;
67 }
68 *outputs++ = static_cast<IntegralType>(it - cumulative_odds.begin());
69 }
70 return kTfLiteOk;
71}
72
73template <typename FloatType>
74TfLiteStatus MultinomialSample(TfLiteContext* context,
75 std::default_random_engine& rng,
76 const FloatType* logits, int logits_size,
77 TfLiteTensor* output, int outputs_offset,
78 int output_size) {
79 switch (output->type) {
80 case kTfLiteInt32:
81 return MultinomialSample<FloatType, int32_t>(
82 rng, logits, logits_size,
83 GetTensorData<int32_t>(output) + outputs_offset, output_size);
84 break;
85 case kTfLiteInt64:
86 return MultinomialSample<FloatType, int64_t>(
87 rng, logits, logits_size,
88 GetTensorData<int64_t>(output) + outputs_offset, output_size);
89 break;
90 default:
91 TF_LITE_KERNEL_LOG(context,
92 "Unsupported datatype for multinomial output: %s",
93 TfLiteTypeGetName(output->type));
94 return kTfLiteError;
95 }
96}
97
98TfLiteStatus MultinomialSample(TfLiteContext* context,
99 std::default_random_engine& rng,
100 const TfLiteTensor* logits, int logits_offset,
101 int logits_size, TfLiteTensor* output,
102 int outputs_offset, int output_size) {
103 switch (logits->type) {
104 case kTfLiteFloat16:
105 TF_LITE_KERNEL_LOG(context, "TfLiteFloat16 is currently not supported.");
106 return kTfLiteError;
107 break;
108 case kTfLiteFloat32:
109 TF_LITE_ENSURE_OK(
110 context,
111 MultinomialSample<float>(
112 context, rng, GetTensorData<float>(logits) + logits_offset,
113 logits_size, output, outputs_offset, output_size));
114 break;
115 case kTfLiteFloat64:
116 TF_LITE_ENSURE_OK(
117 context,
118 MultinomialSample<double>(
119 context, rng, GetTensorData<double>(logits) + logits_offset,
120 logits_size, output, outputs_offset, output_size));
121 break;
122 default:
123 TF_LITE_KERNEL_LOG(context,
124 "Unsupported datatype for multinomial logit input: %s",
125 TfLiteTypeGetName(output->type));
126 return kTfLiteError;
127 }
128 return kTfLiteOk;
129}

139TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
140 // TODO(b/169166131): Handle optional seed input.
141 TF_LITE_ENSURE_EQ(context, tflite::NumInputs(node), 2);
142 TF_LITE_ENSURE_EQ(context, tflite::NumOutputs(node), 1);
143
144 // 'logits' is a float matrix [batch_size, num_categories]
145 const TfLiteTensor* logits_input = tflite::GetInput(context, node, 0);
146 TF_LITE_ENSURE_EQ(context, tflite::NumDimensions(logits_input), 2);
147 int batch_size = tflite::SizeOfDimension(logits_input, 0);
148
149 // 'num_samples' is an int scalar.
150 const TfLiteTensor* num_samples_input = tflite::GetInput(context, node, 1);
151 TF_LITE_ENSURE_EQ(context, tflite::NumDimensions(num_samples_input), 0);
152 // TODO(b/169166131): Allow different integer input types.
153 TF_LITE_ENSURE_EQ(context, num_samples_input->type, kTfLiteInt32);
154 // TODO(b/169166131): Support dynamic output tensors.
155 TF_LITE_ENSURE(context, IsConstantTensor(num_samples_input));
156
157 int num_samples = *num_samples_input->data.i32;
158
159 TfLiteIntArray* output_shape = TfLiteIntArrayCreate(2);
160 output_shape->data[0] = batch_size;
161 output_shape->data[1] = num_samples;
162
163 TfLiteTensor* output = tflite::GetOutput(context, node, 0);
164 // ResizeTensor takes ownership of output_shape.
165 return context->ResizeTensor(context, output, output_shape);
166}

168TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
169 // TODO(b/169166131): Handle optional seed input.
170 MultinomialParams* params =
171 reinterpret_cast<MultinomialParams*>(node->user_data);
172 TF_LITE_ENSURE(context, params != nullptr);
173
174 const TfLiteTensor* logits = tflite::GetInput(context, node, 0);
175 int batch_size = tflite::SizeOfDimension(logits, 0);
176 int logits_size = tflite::SizeOfDimension(logits, 1);
177
178 const TfLiteTensor* num_samples_input = tflite::GetInput(context, node, 1);
179 int output_size = *num_samples_input->data.i32;
180
181 TfLiteTensor* output = tflite::GetOutput(context, node, 0);
182
183 for (int batch = 0; batch < batch_size; ++batch) {
184 int logits_offset = logits_size * batch;
185 int output_offset = output_size * batch;
186
187 TF_LITE_ENSURE_OK(
188 context,
189 MultinomialSample(context, params->rng, logits, logits_offset,
190 logits_size, output, output_offset, output_size));
191 }
192
193 return kTfLiteOk;
194}
