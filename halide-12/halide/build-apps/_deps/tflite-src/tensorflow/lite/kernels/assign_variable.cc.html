29constexpr int kInputVariableId = 0;
30constexpr int kInputValue = 1;
31
32TfLiteStatus Prepare(TfLiteContext* context, TfLiteNode* node) {
33 TF_LITE_ENSURE_EQ(context, NumInputs(node), 2);
34 // TODO(b/137042749): TFLite infrastructure (converter, delegate) doesn't
35 // fully support 0-output ops yet. Currently it works if we manually crfat
36 // a TFLite graph that contains variable ops. Note:
37 // * The TFLite Converter need to be changed to be able to produce an op
38 // with 0 output.
39 // * The delegation code need to be changed to handle 0 output ops. However
40 // everything still works fine when variable ops aren't used.
41 TF_LITE_ENSURE_EQ(context, NumOutputs(node), 0);
42
43 const TfLiteTensor* input_resource_id_tensor;
44 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputVariableId,
45 &input_resource_id_tensor));
46 TF_LITE_ENSURE_EQ(context, input_resource_id_tensor->type, kTfLiteInt32);
47 TF_LITE_ENSURE_EQ(context, NumElements(input_resource_id_tensor), 1);
48
49 return kTfLiteOk;
50}
51
52TfLiteStatus Eval(TfLiteContext* context, TfLiteNode* node) {
53 Subgraph* subgraph = reinterpret_cast<Subgraph*>(context->impl_);
54
55 const TfLiteTensor* input_resource_id_tensor;
56 TF_LITE_ENSURE_OK(context, GetInputSafe(context, node, kInputVariableId,
57 &input_resource_id_tensor));
58 const TfLiteTensor* input_value_tensor;
59 TF_LITE_ENSURE_OK(
60 context, GetInputSafe(context, node, kInputValue, &input_value_tensor));
61
62 int resource_id = input_resource_id_tensor->data.i32[0];
63 auto& resources = subgraph->resources();
64 resource::CreateResourceVariableIfNotAvailable(&resources, resource_id);
65 auto* variable = resource::GetResourceVariable(&resources, resource_id);
66 TF_LITE_ENSURE(context, variable != nullptr);
67 variable->AssignFrom(input_value_tensor);
68
69 return kTfLiteOk;
70}
71
72} // namespace assign_variable
73
74TfLiteRegistration* Register_ASSIGN_VARIABLE() {
75 static TfLiteRegistration r = {nullptr, nullptr, assign_variable::Prepare,
76 assign_variable::Eval};
77 return &r;
78}
