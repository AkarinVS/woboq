15/** \ingroup SparseCore_Module
16 *
17 * \class SparseMatrix
18 *
19 * \brief A versatible sparse matrix representation
20 *
21 * This class implements a more versatile variants of the common \em compressed row/column storage format.
22 * Each colmun's (resp. row) non zeros are stored as a pair of value with associated row (resp. colmiun) index.
23 * All the non zeros are stored in a single large buffer. Unlike the \em compressed format, there might be extra
24 * space in between the nonzeros of two successive colmuns (resp. rows) such that insertion of new non-zero
25 * can be done with limited memory reallocation and copies.
26 *
27 * A call to the function makeCompressed() turns the matrix into the standard \em compressed format
28 * compatible with many library.
29 *
30 * More details on this storage sceheme are given in the \ref TutorialSparse "manual pages".
31 *
32 * \tparam _Scalar the scalar type, i.e. the type of the coefficients
33 * \tparam _Options Union of bit flags controlling the storage scheme. Currently the only possibility
34 * is ColMajor or RowMajor. The default is 0 which means column-major.
35 * \tparam _StorageIndex the type of the indices. It has to be a \b signed type (e.g., short, int, std::ptrdiff_t). Default is \c int.
36 *
37 * \warning In %Eigen 3.2, the undocumented type \c SparseMatrix::Index was improperly defined as the storage index type (e.g., int),
38 * whereas it is now (starting from %Eigen 3.3) deprecated and always defined as Eigen::Index.
39 * Codes making use of \c SparseMatrix::Index, might thus likely have to be changed to use \c SparseMatrix::StorageIndex instead.
40 *
41 * This class can be extended with the help of the plugin mechanism described on the page
42 * \ref TopicCustomizing_Plugins by defining the preprocessor symbol \c EIGEN_SPARSEMATRIX_PLUGIN.
43 */
44

188 /** \returns the value of the matrix at position \a i, \a j
189 * This function returns Scalar(0) if the element is an explicit \em zero */
190 inline Scalar coeff(Index row, Index col) const
191 {
192 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
193
194 const Index outer = IsRowMajor ? row : col;
195 const Index inner = IsRowMajor ? col : row;
196 Index end = m_innerNonZeros ? m_outerIndex[outer] + m_innerNonZeros[outer] : m_outerIndex[outer+1];
197 return m_data.atInRange(m_outerIndex[outer], end, StorageIndex(inner));
198 }

200 /** \returns a non-const reference to the value of the matrix at position \a i, \a j
201 *
202 * If the element does not exist then it is inserted via the insert(Index,Index) function
203 * which itself turns the matrix into a non compressed form if that was not the case.
204 *
205 * This is a O(log(nnz_j)) operation (binary search) plus the cost of insert(Index,Index)
206 * function if the element does not already exist.
207 */
208 inline Scalar& coeffRef(Index row, Index col)
209 {
210 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
211
212 const Index outer = IsRowMajor ? row : col;
213 const Index inner = IsRowMajor ? col : row;
214
215 Index start = m_outerIndex[outer];
216 Index end = m_innerNonZeros ? m_outerIndex[outer] + m_innerNonZeros[outer] : m_outerIndex[outer+1];
217 eigen_assert(end>=start && "you probably called coeffRef on a non finalized matrix");
218 if(end<=start)
219 return insert(row,col);
220 const Index p = m_data.searchLowerIndex(start,end-1,StorageIndex(inner));
221 if((p<end) && (m_data.index(p)==inner))
222 return m_data.value(p);
223 else
224 return insert(row,col);
225 }

227 /** \returns a reference to a novel non zero coefficient with coordinates \a row x \a col.
228 * The non zero coefficient must \b not already exist.
229 *
230 * If the matrix \c *this is in compressed mode, then \c *this is turned into uncompressed
231 * mode while reserving room for 2 x this->innerSize() non zeros if reserve(Index) has not been called earlier.
232 * In this case, the insertion procedure is optimized for a \e sequential insertion mode where elements are assumed to be
233 * inserted by increasing outer-indices.
234 *
235 * If that's not the case, then it is strongly recommended to either use a triplet-list to assemble the matrix, or to first
236 * call reserve(const SizesType &) to reserve the appropriate number of non-zero elements per inner vector.
237 *
238 * Assuming memory has been appropriately reserved, this function performs a sorted insertion in O(1)
239 * if the elements of each inner vector are inserted in increasing inner index order, and in O(nnz_j) for a random insertion.
240 *
241 */
242 Scalar& insert(Index row, Index col);

298 template<class SizesType>
299 inline void reserveInnerVectors(const SizesType& reserveSizes)
300 {
301 if(isCompressed())
302 {
303 Index totalReserveSize = 0;
304 // turn the matrix into non-compressed mode
305 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
306 if (!m_innerNonZeros) internal::throw_std_bad_alloc();
307
308 // temporarily use m_innerSizes to hold the new starting points.
309 StorageIndex* newOuterIndex = m_innerNonZeros;
310
311 StorageIndex count = 0;
312 for(Index j=0; j<m_outerSize; ++j)
313 {
314 newOuterIndex[j] = count;
315 count += reserveSizes[j] + (m_outerIndex[j+1]-m_outerIndex[j]);
316 totalReserveSize += reserveSizes[j];
317 }
318 m_data.reserve(totalReserveSize);
319 StorageIndex previousOuterIndex = m_outerIndex[m_outerSize];
320 for(Index j=m_outerSize-1; j>=0; --j)
321 {
322 StorageIndex innerNNZ = previousOuterIndex - m_outerIndex[j];
323 for(Index i=innerNNZ-1; i>=0; --i)
324 {
325 m_data.index(newOuterIndex[j]+i) = m_data.index(m_outerIndex[j]+i);
326 m_data.value(newOuterIndex[j]+i) = m_data.value(m_outerIndex[j]+i);
327 }
328 previousOuterIndex = m_outerIndex[j];
329 m_outerIndex[j] = newOuterIndex[j];
330 m_innerNonZeros[j] = innerNNZ;
331 }
332 m_outerIndex[m_outerSize] = m_outerIndex[m_outerSize-1] + m_innerNonZeros[m_outerSize-1] + reserveSizes[m_outerSize-1];
333
334 m_data.resize(m_outerIndex[m_outerSize]);
335 }
336 else
337 {
338 StorageIndex* newOuterIndex = static_cast<StorageIndex*>(std::malloc((m_outerSize+1)*sizeof(StorageIndex)));
339 if (!newOuterIndex) internal::throw_std_bad_alloc();
340
341 StorageIndex count = 0;
342 for(Index j=0; j<m_outerSize; ++j)
343 {
344 newOuterIndex[j] = count;
345 StorageIndex alreadyReserved = (m_outerIndex[j+1]-m_outerIndex[j]) - m_innerNonZeros[j];
346 StorageIndex toReserve = std::max<StorageIndex>(reserveSizes[j], alreadyReserved);
347 count += toReserve + m_innerNonZeros[j];
348 }
349 newOuterIndex[m_outerSize] = count;
350
351 m_data.resize(count);
352 for(Index j=m_outerSize-1; j>=0; --j)
353 {
354 Index offset = newOuterIndex[j] - m_outerIndex[j];
355 if(offset>0)
356 {
357 StorageIndex innerNNZ = m_innerNonZeros[j];
358 for(Index i=innerNNZ-1; i>=0; --i)
359 {
360 m_data.index(newOuterIndex[j]+i) = m_data.index(m_outerIndex[j]+i);
361 m_data.value(newOuterIndex[j]+i) = m_data.value(m_outerIndex[j]+i);
362 }
363 }
364 }
365
366 std::swap(m_outerIndex, newOuterIndex);
367 std::free(newOuterIndex);
368 }

421 /** \internal
422 * Must be called after inserting a set of non zero entries using the low level compressed API.
423 */
424 inline void finalize()
425 {
426 if(isCompressed())
427 {
428 StorageIndex size = internal::convert_index<StorageIndex>(m_data.size());
429 Index i = m_outerSize;
430 // find the last filled column
431 while (i>=0 && m_outerIndex[i]==0)
432 --i;
433 ++i;
434 while (i<=m_outerSize)
435 {
436 m_outerIndex[i] = size;
437 ++i;
438 }
439 }

464 /** Turns the matrix into the \em compressed format.
465 */
466 void makeCompressed()
467 {
468 if(isCompressed())
469 return;
470
471 eigen_internal_assert(m_outerIndex!=0 && m_outerSize>0);
472
473 Index oldStart = m_outerIndex[1];
474 m_outerIndex[1] = m_innerNonZeros[0];
475 for(Index j=1; j<m_outerSize; ++j)
476 {
477 Index nextOldStart = m_outerIndex[j+1];
478 Index offset = oldStart - m_outerIndex[j];
479 if(offset>0)
480 {
481 for(Index k=0; k<m_innerNonZeros[j]; ++k)
482 {
483 m_data.index(m_outerIndex[j]+k) = m_data.index(oldStart+k);
484 m_data.value(m_outerIndex[j]+k) = m_data.value(oldStart+k);
485 }
486 }
487 m_outerIndex[j+1] = m_outerIndex[j] + m_innerNonZeros[j];
488 oldStart = nextOldStart;
489 }
490 std::free(m_innerNonZeros);
491 m_innerNonZeros = 0;
492 m_data.resize(m_outerIndex[m_outerSize]);
493 m_data.squeeze();
494 }

496 /** Turns the matrix into the uncompressed mode */
497 void uncompress()
498 {
499 if(m_innerNonZeros != 0)
500 return;
501 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
502 for (Index i = 0; i < m_outerSize; i++)
503 {
504 m_innerNonZeros[i] = m_outerIndex[i+1] - m_outerIndex[i];
505 }
506 }

514 /** Turns the matrix into compressed format, and suppresses all nonzeros which do not satisfy the predicate \a keep.
515 * The functor type \a KeepFunc must implement the following function:
516 * \code
517 * bool operator() (const Index& row, const Index& col, const Scalar& value) const;
518 * \endcode
519 * \sa prune(Scalar,RealScalar)
520 */
521 template<typename KeepFunc>
522 void prune(const KeepFunc& keep = KeepFunc())
523 {
524 // TODO optimize the uncompressed mode to avoid moving and allocating the data twice
525 makeCompressed();
526
527 StorageIndex k = 0;
528 for(Index j=0; j<m_outerSize; ++j)
529 {
530 Index previousStart = m_outerIndex[j];
531 m_outerIndex[j] = k;
532 Index end = m_outerIndex[j+1];
533 for(Index i=previousStart; i<end; ++i)
534 {
535 if(keep(IsRowMajor?j:m_data.index(i), IsRowMajor?m_data.index(i):j, m_data.value(i)))
536 {
537 m_data.value(k) = m_data.value(i);
538 m_data.index(k) = m_data.index(i);
539 ++k;
540 }
541 }
542 }
543 m_outerIndex[m_outerSize] = k;
544 m_data.resize(k,0);
545 }

547 /** Resizes the matrix to a \a rows x \a cols matrix leaving old values untouched.
548 *
549 * If the sizes of the matrix are decreased, then the matrix is turned to \b uncompressed-mode
550 * and the storage of the out of bounds coefficients is kept and reserved.
551 * Call makeCompressed() to pack the entries and squeeze extra memory.
552 *
553 * \sa reserve(), setZero(), makeCompressed()
554 */
555 void conservativeResize(Index rows, Index cols)
556 {
557 // No change
558 if (this->rows() == rows && this->cols() == cols) return;
559
560 // If one dimension is null, then there is nothing to be preserved
561 if(rows==0 || cols==0) return resize(rows,cols);
562
563 Index innerChange = IsRowMajor ? cols - this->cols() : rows - this->rows();
564 Index outerChange = IsRowMajor ? rows - this->rows() : cols - this->cols();
565 StorageIndex newInnerSize = convert_index(IsRowMajor ? cols : rows);
566
567 // Deals with inner non zeros
568 if (m_innerNonZeros)
569 {
570 // Resize m_innerNonZeros
571 StorageIndex *newInnerNonZeros = static_cast<StorageIndex*>(std::realloc(m_innerNonZeros, (m_outerSize + outerChange) * sizeof(StorageIndex)));
572 if (!newInnerNonZeros) internal::throw_std_bad_alloc();
573 m_innerNonZeros = newInnerNonZeros;
574
575 for(Index i=m_outerSize; i<m_outerSize+outerChange; i++)
576 m_innerNonZeros[i] = 0;
577 }
578 else if (innerChange < 0)
579 {
580 // Inner size decreased: allocate a new m_innerNonZeros
581 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc((m_outerSize+outerChange+1) * sizeof(StorageIndex)));
582 if (!m_innerNonZeros) internal::throw_std_bad_alloc();
583 for(Index i = 0; i < m_outerSize; i++)
584 m_innerNonZeros[i] = m_outerIndex[i+1] - m_outerIndex[i];
585 }
586
587 // Change the m_innerNonZeros in case of a decrease of inner size
588 if (m_innerNonZeros && innerChange < 0)
589 {
590 for(Index i = 0; i < m_outerSize + (std::min)(outerChange, Index(0)); i++)
591 {
592 StorageIndex &n = m_innerNonZeros[i];
593 StorageIndex start = m_outerIndex[i];
594 while (n > 0 && m_data.index(start+n-1) >= newInnerSize) --n;
595 }
596 }
597
598 m_innerSize = newInnerSize;
599
600 // Re-allocate outer index structure if necessary
601 if (outerChange == 0)
602 return;
603
604 StorageIndex *newOuterIndex = static_cast<StorageIndex*>(std::realloc(m_outerIndex, (m_outerSize + outerChange + 1) * sizeof(StorageIndex)));
605 if (!newOuterIndex) internal::throw_std_bad_alloc();
606 m_outerIndex = newOuterIndex;
607 if (outerChange > 0)
608 {
609 StorageIndex lastIdx = m_outerSize == 0 ? 0 : m_outerIndex[m_outerSize];
610 for(Index i=m_outerSize; i<m_outerSize+outerChange+1; i++)
611 m_outerIndex[i] = lastIdx;
612 }
613 m_outerSize += outerChange;
614 }

617 *
618 * This function does not free the currently allocated memory. To release as much as memory as possible,

910 /** \internal assign \a diagXpr to the diagonal of \c *this
911 * There are different strategies:
912 * 1 - if *this is overwritten (Func==assign_op) or *this is empty, then we can work treat *this as a dense vector expression.
913 * 2 - otherwise, for each diagonal coeff,
914 * 2.a - if it already exists, then we update it,
915 * 2.b - otherwise, if *this is uncompressed and that the current inner-vector has empty room for at least 1 element, then we perform an in-place insertion.
916 * 2.c - otherwise, we'll have to reallocate and copy everything, so instead of doing so for each new element, it is recorded in a std::vector.
917 * 3 - at the end, if some entries failed to be inserted in-place, then we alloc a new buffer, copy each chunk at the right position, and insert the new elements.
918 *
919 * TODO: some piece of code could be isolated and reused for a general in-place update strategy.
920 * TODO: if we start to defer the insertion of some elements (i.e., case 2.c executed once),
921 * then it *might* be better to disable case 2.b since they will have to be copied anyway.
922 */
923 template<typename DiagXpr, typename Func>
924 void assignDiagonal(const DiagXpr diagXpr, const Func& assignFunc)
925 {
926 Index n = diagXpr.size();
927
928 const bool overwrite = internal::is_same<Func, internal::assign_op<Scalar,Scalar> >::value;
929 if(overwrite)
930 {
931 if((this->rows()!=n) || (this->cols()!=n))
932 this->resize(n, n);
933 }
934
935 if(m_data.size()==0 || overwrite)
936 {
937 typedef Array<StorageIndex,Dynamic,1> ArrayXI;
938 this->makeCompressed();
939 this->resizeNonZeros(n);
940 Eigen::Map<ArrayXI>(this->innerIndexPtr(), n).setLinSpaced(0,StorageIndex(n)-1);
941 Eigen::Map<ArrayXI>(this->outerIndexPtr(), n+1).setLinSpaced(0,StorageIndex(n));
942 Eigen::Map<Array<Scalar,Dynamic,1> > values = this->coeffs();
943 values.setZero();
944 internal::call_assignment_no_alias(values, diagXpr, assignFunc);
945 }
946 else
947 {
948 bool isComp = isCompressed();
949 internal::evaluator<DiagXpr> diaEval(diagXpr);
950 std::vector<IndexPosPair> newEntries;
951
952 // 1 - try in-place update and record insertion failures
953 for(Index i = 0; i<n; ++i)
954 {
955 internal::LowerBoundIndex lb = this->lower_bound(i,i);
956 Index p = lb.value;
957 if(lb.found)
958 {
959 // the coeff already exists
960 assignFunc.assignCoeff(m_data.value(p), diaEval.coeff(i));
961 }
962 else if((!isComp) && m_innerNonZeros[i] < (m_outerIndex[i+1]-m_outerIndex[i]))
963 {
964 // non compressed mode with local room for inserting one element
965 m_data.moveChunk(p, p+1, m_outerIndex[i]+m_innerNonZeros[i]-p);
966 m_innerNonZeros[i]++;
967 m_data.value(p) = Scalar(0);
968 m_data.index(p) = StorageIndex(i);
969 assignFunc.assignCoeff(m_data.value(p), diaEval.coeff(i));
970 }
971 else
972 {
973 // defer insertion
974 newEntries.push_back(IndexPosPair(i,p));
975 }
976 }
977 // 2 - insert deferred entries
978 Index n_entries = Index(newEntries.size());
979 if(n_entries>0)
980 {
981 Storage newData(m_data.size()+n_entries);
982 Index prev_p = 0;
983 Index prev_i = 0;
984 for(Index k=0; k<n_entries;++k)
985 {
986 Index i = newEntries[k].i;
987 Index p = newEntries[k].p;
988 internal::smart_copy(m_data.valuePtr()+prev_p, m_data.valuePtr()+p, newData.valuePtr()+prev_p+k);
989 internal::smart_copy(m_data.indexPtr()+prev_p, m_data.indexPtr()+p, newData.indexPtr()+prev_p+k);
990 for(Index j=prev_i;j<i;++j)
991 m_outerIndex[j+1] += k;
992 if(!isComp)
993 m_innerNonZeros[i]++;
994 prev_p = p;
995 prev_i = i;
996 newData.value(p+k) = Scalar(0);
997 newData.index(p+k) = StorageIndex(i);
998 assignFunc.assignCoeff(newData.value(p+k), diaEval.coeff(i));
999 }
1000 {
1001 internal::smart_copy(m_data.valuePtr()+prev_p, m_data.valuePtr()+m_data.size(), newData.valuePtr()+prev_p+n_entries);
1002 internal::smart_copy(m_data.indexPtr()+prev_p, m_data.indexPtr()+m_data.size(), newData.indexPtr()+prev_p+n_entries);
1003 for(Index j=prev_i+1;j<=m_outerSize;++j)
1004 m_outerIndex[j] += n_entries;
1005 }
1006 m_data.swap(newData);
1007 }
1008 }
1009 }

1031template<typename InputIterator, typename SparseMatrixType, typename DupFunctor>
1032void set_from_triplets(const InputIterator& begin, const InputIterator& end, SparseMatrixType& mat, DupFunctor dup_func)
1033{
1034 enum { IsRowMajor = SparseMatrixType::IsRowMajor };
1035 typedef typename SparseMatrixType::Scalar Scalar;
1036 typedef typename SparseMatrixType::StorageIndex StorageIndex;
1037 SparseMatrix<Scalar,IsRowMajor?ColMajor:RowMajor,StorageIndex> trMat(mat.rows(),mat.cols());
1038
1039 if(begin!=end)
1040 {
1041 // pass 1: count the nnz per inner-vector
1042 typename SparseMatrixType::IndexVector wi(trMat.outerSize());
1043 wi.setZero();
1044 for(InputIterator it(begin); it!=end; ++it)
1045 {
1046 eigen_assert(it->row()>=0 && it->row()<mat.rows() && it->col()>=0 && it->col()<mat.cols());
1047 wi(IsRowMajor ? it->col() : it->row())++;
1048 }
1049
1050 // pass 2: insert all the elements into trMat
1051 trMat.reserve(wi);
1052 for(InputIterator it(begin); it!=end; ++it)
1053 trMat.insertBackUncompressed(it->row(),it->col()) = it->value();
1054
1055 // pass 3:
1056 trMat.collapseDuplicates(dup_func);
1057 }
1058
1059 // pass 4: transposed copy -> implicit sorting
1060 mat = trMat;
1061}

1066/** Fill the matrix \c *this with the list of \em triplets defined by the iterator range \a begin - \a end.
1067 *
1068 * A \em triplet is a tuple (i,j,value) defining a non-zero element.
1069 * The input list of triplets does not have to be sorted, and can contains duplicated elements.
1070 * In any case, the result is a \b sorted and \b compressed sparse matrix where the duplicates have been summed up.
1071 * This is a \em O(n) operation, with \em n the number of triplet elements.
1072 * The initial contents of \c *this is destroyed.
1073 * The matrix \c *this must be properly resized beforehand using the SparseMatrix(Index,Index) constructor,
1074 * or the resize(Index,Index) method. The sizes are not extracted from the triplet list.
1075 *
1076 * The \a InputIterators value_type must provide the following interface:
1077 * \code
1078 * Scalar value() const; // the value
1079 * Scalar row() const; // the row index i
1080 * Scalar col() const; // the column index j
1081 * \endcode
1082 * See for instance the Eigen::Triplet template class.
1083 *
1084 * Here is a typical usage example:
1085 * \code
1086 typedef Triplet<double> T;
1087 std::vector<T> tripletList;
1088 triplets.reserve(estimation_of_entries);
1089 for(...)
1090 {
1091 // ...
1092 tripletList.push_back(T(i,j,v_ij));
1093 }
1094 SparseMatrixType m(rows,cols);
1095 m.setFromTriplets(tripletList.begin(), tripletList.end());
1096 // m is ready to go!
1097 * \endcode
1098 *
1099 * \warning The list of triplets is read multiple times (at least twice). Therefore, it is not recommended to define
1100 * an abstract iterator over a complex data-structure that would be expensive to evaluate. The triplets should rather
1101 * be explicitly stored into a std::vector for instance.
1102 */
1103template<typename Scalar, int _Options, typename _StorageIndex>
1104template<typename InputIterators>
1105void SparseMatrix<Scalar,_Options,_StorageIndex>::setFromTriplets(const InputIterators& begin, const InputIterators& end)
1106{
1107 internal::set_from_triplets<InputIterators, SparseMatrix<Scalar,_Options,_StorageIndex> >(begin, end, *this, internal::scalar_sum_op<Scalar,Scalar>());
1108}

1115 * \code
1116 * mat.setFromTriplets(triplets.begin(), triplets.end(), [] (const Scalar&,const Scalar &b) { return b; });
1117 * \endcode

1127template<typename Scalar, int _Options, typename _StorageIndex>
1128template<typename DupFunctor>
1129void SparseMatrix<Scalar,_Options,_StorageIndex>::collapseDuplicates(DupFunctor dup_func)
1130{
1131 eigen_assert(!isCompressed());
1132 // TODO, in practice we should be able to use m_innerNonZeros for that task
1133 IndexVector wi(innerSize());
1134 wi.fill(-1);
1135 StorageIndex count = 0;
1136 // for each inner-vector, wi[inner_index] will hold the position of first element into the index/value buffers
1137 for(Index j=0; j<outerSize(); ++j)
1138 {
1139 StorageIndex start = count;
1140 Index oldEnd = m_outerIndex[j]+m_innerNonZeros[j];
1141 for(Index k=m_outerIndex[j]; k<oldEnd; ++k)
1142 {
1143 Index i = m_data.index(k);
1144 if(wi(i)>=start)
1145 {
1146 // we already meet this entry => accumulate it
1147 m_data.value(wi(i)) = dup_func(m_data.value(wi(i)), m_data.value(k));
1148 }
1149 else
1150 {
1151 m_data.value(count) = m_data.value(k);
1152 m_data.index(count) = m_data.index(k);
1153 wi(i) = count;
1154 ++count;
1155 }
1156 }
1157 m_outerIndex[j] = start;
1158 }
1159 m_outerIndex[m_outerSize] = count;
1160
1161 // turn the matrix into compressed form
1162 std::free(m_innerNonZeros);
1163 m_innerNonZeros = 0;
1164 m_data.resize(m_outerIndex[m_outerSize]);
1165}

1228 }

1230 {
1231 if(other.isRValue())
1232 {
1233 initAssignment(other.derived());
1234 }
1235 // there is no special optimization
1236 return Base::operator=(other.derived());

1240template<typename _Scalar, int _Options, typename _StorageIndex>
1241typename SparseMatrix<_Scalar,_Options,_StorageIndex>::Scalar& SparseMatrix<_Scalar,_Options,_StorageIndex>::insert(Index row, Index col)
1242{
1243 eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
1244
1245 const Index outer = IsRowMajor ? row : col;
1246 const Index inner = IsRowMajor ? col : row;
1247
1248 if(isCompressed())
1249 {
1250 if(nonZeros()==0)
1251 {
1252 // reserve space if not already done
1253 if(m_data.allocatedSize()==0)
1254 m_data.reserve(2*m_innerSize);
1255
1256 // turn the matrix into non-compressed mode
1257 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
1258 if(!m_innerNonZeros) internal::throw_std_bad_alloc();
1259
1260 memset(m_innerNonZeros, 0, (m_outerSize)*sizeof(StorageIndex));
1261
1262 // pack all inner-vectors to the end of the pre-allocated space
1263 // and allocate the entire free-space to the first inner-vector
1264 StorageIndex end = convert_index(m_data.allocatedSize());
1265 for(Index j=1; j<=m_outerSize; ++j)
1266 m_outerIndex[j] = end;
1267 }
1268 else
1269 {
1270 // turn the matrix into non-compressed mode
1271 m_innerNonZeros = static_cast<StorageIndex*>(std::malloc(m_outerSize * sizeof(StorageIndex)));
1272 if(!m_innerNonZeros) internal::throw_std_bad_alloc();
1273 for(Index j=0; j<m_outerSize; ++j)
1274 m_innerNonZeros[j] = m_outerIndex[j+1]-m_outerIndex[j];
1275 }
1276 }
1277
1278 // check whether we can do a fast "push back" insertion
1279 Index data_end = m_data.allocatedSize();
1280
1281 // First case: we are filling a new inner vector which is packed at the end.
1282 // We assume that all remaining inner-vectors are also empty and packed to the end.
1283 if(m_outerIndex[outer]==data_end)
1284 {
1285 eigen_internal_assert(m_innerNonZeros[outer]==0);
1286
1287 // pack previous empty inner-vectors to end of the used-space
1288 // and allocate the entire free-space to the current inner-vector.
1289 StorageIndex p = convert_index(m_data.size());
1290 Index j = outer;
1291 while(j>=0 && m_innerNonZeros[j]==0)
1292 m_outerIndex[j--] = p;
1293
1294 // push back the new element
1295 ++m_innerNonZeros[outer];
1296 m_data.append(Scalar(0), inner);
1297
1298 // check for reallocation
1299 if(data_end != m_data.allocatedSize())
1300 {
1301 // m_data has been reallocated
1302 // -> move remaining inner-vectors back to the end of the free-space
1303 // so that the entire free-space is allocated to the current inner-vector.
1304 eigen_internal_assert(data_end < m_data.allocatedSize());
1305 StorageIndex new_end = convert_index(m_data.allocatedSize());
1306 for(Index k=outer+1; k<=m_outerSize; ++k)
1307 if(m_outerIndex[k]==data_end)
1308 m_outerIndex[k] = new_end;
1309 }
1310 return m_data.value(p);
1311 }
1312
1313 // Second case: the next inner-vector is packed to the end
1314 // and the current inner-vector end match the used-space.
1315 if(m_outerIndex[outer+1]==data_end && m_outerIndex[outer]+m_innerNonZeros[outer]==m_data.size())
1316 {
1317 eigen_internal_assert(outer+1==m_outerSize || m_innerNonZeros[outer+1]==0);
1318
1319 // add space for the new element
1320 ++m_innerNonZeros[outer];
1321 m_data.resize(m_data.size()+1);
1322
1323 // check for reallocation
1324 if(data_end != m_data.allocatedSize())
1325 {
1326 // m_data has been reallocated
1327 // -> move remaining inner-vectors back to the end of the free-space
1328 // so that the entire free-space is allocated to the current inner-vector.
1329 eigen_internal_assert(data_end < m_data.allocatedSize());
1330 StorageIndex new_end = convert_index(m_data.allocatedSize());
1331 for(Index k=outer+1; k<=m_outerSize; ++k)
1332 if(m_outerIndex[k]==data_end)
1333 m_outerIndex[k] = new_end;
1334 }
1335
1336 // and insert it at the right position (sorted insertion)
1337 Index startId = m_outerIndex[outer];
1338 Index p = m_outerIndex[outer]+m_innerNonZeros[outer]-1;
1339 while ( (p > startId) && (m_data.index(p-1) > inner) )
1340 {
1341 m_data.index(p) = m_data.index(p-1);
1342 m_data.value(p) = m_data.value(p-1);
1343 --p;
1344 }
1345
1346 m_data.index(p) = convert_index(inner);
1347 return (m_data.value(p) = Scalar(0));
1348 }
1349
1350 if(m_data.size() != m_data.allocatedSize())
1351 {
1352 // make sure the matrix is compatible to random un-compressed insertion:
1353 m_data.resize(m_data.allocatedSize());
1354 this->reserveInnerVectors(Array<StorageIndex,Dynamic,1>::Constant(m_outerSize, 2));
1355 }
1356
1357 return insertUncompressed(row,col);
1358}

1361EIGEN_DONT_INLINE typename SparseMatrix<_Scalar,_Options,_StorageIndex>::Scalar& SparseMatrix<_Scalar,_Options,_StorageIndex>::insertUncompressed(Index row, Index col)
1362{
1363 eigen_assert(!isCompressed());
1364

1366 const StorageIndex inner = convert_index(IsRowMajor ? col : row);
1367

1381 m_data.value(p) = m_data.value(p-1);
1382 --p;
1383 }

1392template<typename _Scalar, int _Options, typename _StorageIndex>
1393EIGEN_DONT_INLINE typename SparseMatrix<_Scalar,_Options,_StorageIndex>::Scalar& SparseMatrix<_Scalar,_Options,_StorageIndex>::insertCompressed(Index row, Index col)
1394{
1395 eigen_assert(isCompressed());
1396
1397 const Index outer = IsRowMajor ? row : col;
1398 const Index inner = IsRowMajor ? col : row;
1399
1400 Index previousOuter = outer;
1401 if (m_outerIndex[outer+1]==0)
1402 {
1403 // we start a new inner vector
1404 while (previousOuter>=0 && m_outerIndex[previousOuter]==0)
1405 {
1406 m_outerIndex[previousOuter] = convert_index(m_data.size());
1407 --previousOuter;
1408 }
1409 m_outerIndex[outer+1] = m_outerIndex[outer];
1410 }
1411
1412 // here we have to handle the tricky case where the outerIndex array
1413 // starts with: [ 0 0 0 0 0 1 ...] and we are inserted in, e.g.,
1414 // the 2nd inner vector...
1415 bool isLastVec = (!(previousOuter==-1 && m_data.size()!=0))
1416 && (std::size_t(m_outerIndex[outer+1]) == m_data.size());
1417
1418 std::size_t startId = m_outerIndex[outer];
1419 // FIXME let's make sure sizeof(long int) == sizeof(std::size_t)
1420 std::size_t p = m_outerIndex[outer+1];
1421 ++m_outerIndex[outer+1];
1422
1423 double reallocRatio = 1;
1424 if (m_data.allocatedSize()<=m_data.size())
1425 {
1426 // if there is no preallocated memory, let's reserve a minimum of 32 elements
1427 if (m_data.size()==0)
1428 {
1429 m_data.reserve(32);
1430 }
1431 else
1432 {
1433 // we need to reallocate the data, to reduce multiple reallocations
1434 // we use a smart resize algorithm based on the current filling ratio
1435 // in addition, we use double to avoid integers overflows
1436 double nnzEstimate = double(m_outerIndex[outer])*double(m_outerSize)/double(outer+1);
1437 reallocRatio = (nnzEstimate-double(m_data.size()))/double(m_data.size());
1438 // furthermore we bound the realloc ratio to:
1439 // 1) reduce multiple minor realloc when the matrix is almost filled
1440 // 2) avoid to allocate too much memory when the matrix is almost empty
1441 reallocRatio = (std::min)((std::max)(reallocRatio,1.5),8.);
1442 }
1443 }
1444 m_data.resize(m_data.size()+1,reallocRatio);
1445
1446 if (!isLastVec)
1447 {
1448 if (previousOuter==-1)
1449 {
1450 // oops wrong guess.
1451 // let's correct the outer offsets
1452 for (Index k=0; k<=(outer+1); ++k)
1453 m_outerIndex[k] = 0;
1454 Index k=outer+1;
1455 while(m_outerIndex[k]==0)
1456 m_outerIndex[k++] = 1;
1457 while (k<=m_outerSize && m_outerIndex[k]!=0)
1458 m_outerIndex[k++]++;
1459 p = 0;
1460 --k;
1461 k = m_outerIndex[k]-1;
1462 while (k>0)
1463 {
1464 m_data.index(k) = m_data.index(k-1);
1465 m_data.value(k) = m_data.value(k-1);
1466 k--;
1467 }
1468 }
1469 else
1470 {
1471 // we are not inserting into the last inner vec
1472 // update outer indices:
1473 Index j = outer+2;
1474 while (j<=m_outerSize && m_outerIndex[j]!=0)
1475 m_outerIndex[j++]++;
1476 --j;
1477 // shift data of last vecs:
1478 Index k = m_outerIndex[j]-1;
1479 while (k>=Index(p))
1480 {
1481 m_data.index(k) = m_data.index(k-1);
1482 m_data.value(k) = m_data.value(k-1);
1483 k--;
1484 }
1485 }
1486 }
1487
1488 while ( (p > startId) && (m_data.index(p-1) > inner) )
1489 {
1490 m_data.index(p) = m_data.index(p-1);
1491 m_data.value(p) = m_data.value(p-1);
1492 --p;
1493 }
1494
1495 m_data.index(p) = inner;
1496 return (m_data.value(p) = Scalar(0));
1497}
