15/** \internal the return type of conjugate() */
16typedef typename internal::conditional<NumTraits<Scalar>::IsComplex,
17 const CwiseUnaryOp<internal::scalar_conjugate_op<Scalar>, const Derived>,
18 const Derived&
19 >::type ConjugateReturnType;

21typedef typename internal::conditional<NumTraits<Scalar>::IsComplex,

23 const Derived&
24 >::type RealReturnType;

154/// \brief Apply a unary operator coefficient-wise
155/// \param[in] func Functor implementing the unary operator
156/// \tparam CustomUnaryOp Type of \a func
157/// \returns An expression of a custom coefficient-wise unary operator \a func of *this
158///
159/// The function \c ptr_fun() from the C++ standard library can be used to make functors out of normal functions.
160///
161/// Example:
162/// \include class_CwiseUnaryOp_ptrfun.cpp
163/// Output: \verbinclude class_CwiseUnaryOp_ptrfun.out
164///
165/// Genuine functors allow for more possibilities, for instance it may contain a state.
166///
167/// Example:
168/// \include class_CwiseUnaryOp.cpp
169/// Output: \verbinclude class_CwiseUnaryOp.out
170///
171EIGEN_DOC_UNARY_ADDONS(unaryExpr,unary function)
172///
173/// \sa unaryViewExpr, binaryExpr, class CwiseUnaryOp
174///
175template<typename CustomUnaryOp>
176EIGEN_DEVICE_FUNC
177inline const CwiseUnaryOp<CustomUnaryOp, const Derived>
178unaryExpr(const CustomUnaryOp& func = CustomUnaryOp()) const
179{
180 return CwiseUnaryOp<CustomUnaryOp, const Derived>(derived(), func);
181}

183/// \returns an expression of a custom coefficient-wise unary operator \a func of *this

186/// of the custom unary operator.

189/// \include class_CwiseUnaryOp.cpp
190/// Output: \verbinclude class_CwiseUnaryOp.out
191///
192EIGEN_DOC_UNARY_ADDONS(unaryViewExpr,unary function)
193///
194/// \sa unaryExpr, binaryExpr class CwiseUnaryOp
195///

197EIGEN_DEVICE_FUNC
198inline const CwiseUnaryView<CustomViewOp, const Derived>
199unaryViewExpr(const CustomViewOp& func = CustomViewOp()) const

202}
