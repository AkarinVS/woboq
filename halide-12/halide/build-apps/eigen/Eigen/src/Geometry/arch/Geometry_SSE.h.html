18template<class Derived, class OtherDerived>
19struct quat_product<Architecture::SSE, Derived, OtherDerived, float>
20{
21 enum {
22 AAlignment = traits<Derived>::Alignment,
23 BAlignment = traits<OtherDerived>::Alignment,
24 ResAlignment = traits<Quaternion<float> >::Alignment
25 };
26 static inline Quaternion<float> run(const QuaternionBase<Derived>& _a, const QuaternionBase<OtherDerived>& _b)
27 {
28 evaluator<typename Derived::Coefficients> ae(_a.coeffs());
29 evaluator<typename OtherDerived::Coefficients> be(_b.coeffs());
30 Quaternion<float> res;
31 const Packet4f mask = _mm_setr_ps(0.f,0.f,0.f,-0.f);
32 Packet4f a = ae.template packet<AAlignment,Packet4f>(0);
33 Packet4f b = be.template packet<BAlignment,Packet4f>(0);
34 Packet4f s1 = pmul(vec4f_swizzle1(a,1,2,0,2),vec4f_swizzle1(b,2,0,1,2));
35 Packet4f s2 = pmul(vec4f_swizzle1(a,3,3,3,1),vec4f_swizzle1(b,0,1,2,1));
36 pstoret<float,Packet4f,ResAlignment>(
37 &res.x(),
38 padd(psub(pmul(a,vec4f_swizzle1(b,3,3,3,3)),
39 pmul(vec4f_swizzle1(a,2,0,1,0),
40 vec4f_swizzle1(b,1,2,0,0))),
41 pxor(mask,padd(s1,s2))));
42
43 return res;
44 }
45};

47template<class Derived>
48struct quat_conj<Architecture::SSE, Derived, float>
49{
50 enum {
51 ResAlignment = traits<Quaternion<float> >::Alignment
52 };
53 static inline Quaternion<float> run(const QuaternionBase<Derived>& q)
54 {
55 evaluator<typename Derived::Coefficients> qe(q.coeffs());
56 Quaternion<float> res;
57 const Packet4f mask = _mm_setr_ps(-0.f,-0.f,-0.f,0.f);
58 pstoret<float,Packet4f,ResAlignment>(&res.x(), pxor(mask, qe.template packet<traits<Derived>::Alignment,Packet4f>(0)));
59 return res;
60 }
61};

64template<typename VectorLhs,typename VectorRhs>
65struct cross3_impl<Architecture::SSE,VectorLhs,VectorRhs,float,true>
66{
67 enum {
68 ResAlignment = traits<typename plain_matrix_type<VectorLhs>::type>::Alignment
69 };
70 static inline typename plain_matrix_type<VectorLhs>::type
71 run(const VectorLhs& lhs, const VectorRhs& rhs)
72 {
73 evaluator<VectorLhs> lhs_eval(lhs);
74 evaluator<VectorRhs> rhs_eval(rhs);
75 Packet4f a = lhs_eval.template packet<traits<VectorLhs>::Alignment,Packet4f>(0);
76 Packet4f b = rhs_eval.template packet<traits<VectorRhs>::Alignment,Packet4f>(0);
77 Packet4f mul1 = pmul(vec4f_swizzle1(a,1,2,0,3),vec4f_swizzle1(b,2,0,1,3));
78 Packet4f mul2 = pmul(vec4f_swizzle1(a,2,0,1,3),vec4f_swizzle1(b,1,2,0,3));
79 typename plain_matrix_type<VectorLhs>::type res;
80 pstoret<float,Packet4f,ResAlignment>(&res.x(),psub(mul1,mul2));
81 return res;
82 }
83};

88template<class Derived, class OtherDerived>
89struct quat_product<Architecture::SSE, Derived, OtherDerived, double>
90{
91 enum {
92 BAlignment = traits<OtherDerived>::Alignment,
93 ResAlignment = traits<Quaternion<double> >::Alignment
94 };
95
96 static inline Quaternion<double> run(const QuaternionBase<Derived>& _a, const QuaternionBase<OtherDerived>& _b)
97 {
98 const Packet2d mask = _mm_castsi128_pd(_mm_set_epi32(0x0,0x0,0x80000000,0x0));
99
100 Quaternion<double> res;
101
102 evaluator<typename Derived::Coefficients> ae(_a.coeffs());
103 evaluator<typename OtherDerived::Coefficients> be(_b.coeffs());
104
105 const double* a = _a.coeffs().data();
106 Packet2d b_xy = be.template packet<BAlignment,Packet2d>(0);
107 Packet2d b_zw = be.template packet<BAlignment,Packet2d>(2);
108 Packet2d a_xx = pset1<Packet2d>(a[0]);
109 Packet2d a_yy = pset1<Packet2d>(a[1]);
110 Packet2d a_zz = pset1<Packet2d>(a[2]);
111 Packet2d a_ww = pset1<Packet2d>(a[3]);
112
113 // two temporaries:
114 Packet2d t1, t2;
115
116 /*
117 * t1 = ww*xy + yy*zw
118 * t2 = zz*xy - xx*zw
119 * res.xy = t1 +/- swap(t2)
120 */
121 t1 = padd(pmul(a_ww, b_xy), pmul(a_yy, b_zw));
122 t2 = psub(pmul(a_zz, b_xy), pmul(a_xx, b_zw));
123#ifdef EIGEN_VECTORIZE_SSE3
124 EIGEN_UNUSED_VARIABLE(mask)
125 pstoret<double,Packet2d,ResAlignment>(&res.x(), _mm_addsub_pd(t1, preverse(t2)));
126#else
127 pstoret<double,Packet2d,ResAlignment>(&res.x(), padd(t1, pxor(mask,preverse(t2))));
128#endif
129
130 /*
131 * t1 = ww*zw - yy*xy
132 * t2 = zz*zw + xx*xy
133 * res.zw = t1 -/+ swap(t2) = swap( swap(t1) +/- t2)
134 */
135 t1 = psub(pmul(a_ww, b_zw), pmul(a_yy, b_xy));
136 t2 = padd(pmul(a_zz, b_zw), pmul(a_xx, b_xy));
137#ifdef EIGEN_VECTORIZE_SSE3
138 EIGEN_UNUSED_VARIABLE(mask)
139 pstoret<double,Packet2d,ResAlignment>(&res.z(), preverse(_mm_addsub_pd(preverse(t1), t2)));
140#else
141 pstoret<double,Packet2d,ResAlignment>(&res.z(), psub(t1, pxor(mask,preverse(t2))));
142#endif
143
144 return res;
145}
146};

148template<class Derived>
149struct quat_conj<Architecture::SSE, Derived, double>
150{
151 enum {
152 ResAlignment = traits<Quaternion<double> >::Alignment
153 };
154 static inline Quaternion<double> run(const QuaternionBase<Derived>& q)
155 {
156 evaluator<typename Derived::Coefficients> qe(q.coeffs());
157 Quaternion<double> res;
158 const Packet2d mask0 = _mm_setr_pd(-0.,-0.);
159 const Packet2d mask2 = _mm_setr_pd(-0.,0.);
160 pstoret<double,Packet2d,ResAlignment>(&res.x(), pxor(mask0, qe.template packet<traits<Derived>::Alignment,Packet2d>(0)));
161 pstoret<double,Packet2d,ResAlignment>(&res.z(), pxor(mask2, qe.template packet<traits<Derived>::Alignment,Packet2d>(2)));
162 return res;
163 }
164};
