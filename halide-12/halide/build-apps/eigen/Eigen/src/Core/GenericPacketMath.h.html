140/** \internal Wrapper to ensure that multiple packet types can map to the same
141 same underlying vector type. */
142template<typename T, int unique_id = 0>
143struct eigen_packet_wrapper
144{
145 EIGEN_ALWAYS_INLINE operator T&() { return m_val; }
146 EIGEN_ALWAYS_INLINE operator const T&() const { return m_val; }
147 EIGEN_ALWAYS_INLINE eigen_packet_wrapper() {}
148 EIGEN_ALWAYS_INLINE eigen_packet_wrapper(const T &v) : m_val(v) {}
149 EIGEN_ALWAYS_INLINE eigen_packet_wrapper& operator=(const T &v) {
150 m_val = v;
151 return *this;
152 }
153
154 T m_val;
155};

408/** \internal equivalent to
409 * \code
410 * a0 = pload1(a+0);
411 * a1 = pload1(a+1);
412 * a2 = pload1(a+2);
413 * a3 = pload1(a+3);
414 * \endcode
415 * \sa pset1, pload1, ploaddup, pbroadcast2

418inline void pbroadcast4(const typename unpacket_traits<Packet>::type *a,
419 Packet& a0, Packet& a1, Packet& a2, Packet& a3)

469/** \internal tries to do cache prefetching of \a addr */
470template<typename Scalar> EIGEN_DEVICE_FUNC inline void prefetch(const Scalar* addr)
471{
472#if defined(EIGEN_HIP_DEVICE_COMPILE)
473 // do nothing
474#elif defined(EIGEN_CUDA_ARCH)
475#if defined(__LP64__)
476 // 64-bit pointer operand constraint for inlined asm
477 asm(" prefetch.L1 [ %1 ];" : "=l"(addr) : "l"(addr));
478#else
479 // 32-bit pointer operand constraint for inlined asm
480 asm(" prefetch.L1 [ %1 ];" : "=r"(addr) : "r"(addr));
481#endif
482#elif (!EIGEN_COMP_MSVC) && (EIGEN_COMP_GNUC || EIGEN_COMP_CLANG || EIGEN_COMP_ICC)
483 __builtin_prefetch(addr);
484#endif
485}

523/** \internal \returns true if any coeffs of \a a means "true"
524 * It is supposed to be called on values returned by pcmp_*.
525 */
526template<typename Packet> EIGEN_DEVICE_FUNC inline bool predux_any(const Packet& a)
527{
528 // Dirty but generic implementation where "true" is assumed to be non 0 and all the sames.
529 // It is expected that "true" is either:
530 // - Scalar(1)
531 // - bits full of ones (NaN for floats),
532 // - or first bit equals to 1 (1 for ints, smallest denormal for floats).
533 // For all these cases, taking the sum is just fine, and this boils down to a no-op for scalars.
534 typedef typename unpacket_traits<Packet>::type Scalar;
535 return numext::not_equal_strict(predux(a), Scalar(0));
536}

655 * The pointer \a from must be aligned on a \a Alignment bytes boundary. */
