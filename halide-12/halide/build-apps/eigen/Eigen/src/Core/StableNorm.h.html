17template<typename ExpressionType, typename Scalar>
18inline void stable_norm_kernel(const ExpressionType& bl, Scalar& ssq, Scalar& scale, Scalar& invScale)
19{
20 Scalar maxCoeff = bl.cwiseAbs().maxCoeff();
21
22 if(maxCoeff>scale)
23 {
24 ssq = ssq * numext::abs2(scale/maxCoeff);
25 Scalar tmp = Scalar(1)/maxCoeff;
26 if(tmp > NumTraits<Scalar>::highest())
27 {
28 invScale = NumTraits<Scalar>::highest();
29 scale = Scalar(1)/invScale;
30 }
31 else if(maxCoeff>NumTraits<Scalar>::highest()) // we got a INF
32 {
33 invScale = Scalar(1);
34 scale = maxCoeff;
35 }
36 else
37 {
38 scale = maxCoeff;
39 invScale = tmp;
40 }
41 }
42 else if(maxCoeff!=maxCoeff) // we got a NaN
43 {
44 scale = maxCoeff;
45 }
46
47 // TODO if the maxCoeff is much much smaller than the current scale,
48 // then we can neglect this sub vector
49 if(scale>Scalar(0)) // if scale==0, then bl is 0
50 ssq += (bl*invScale).squaredNorm();
51}

53template<typename VectorType, typename RealScalar>
54void stable_norm_impl_inner_step(const VectorType &vec, RealScalar& ssq, RealScalar& scale, RealScalar& invScale)
55{
56 typedef typename VectorType::Scalar Scalar;
57 const Index blockSize = 4096;
58
59 typedef typename internal::nested_eval<VectorType,2>::type VectorTypeCopy;
60 typedef typename internal::remove_all<VectorTypeCopy>::type VectorTypeCopyClean;
61 const VectorTypeCopy copy(vec);
62
63 enum {
64 CanAlign = ( (int(VectorTypeCopyClean::Flags)&DirectAccessBit)
65 || (int(internal::evaluator<VectorTypeCopyClean>::Alignment)>0) // FIXME Alignment)>0 might not be enough
66 ) && (blockSize*sizeof(Scalar)*2<EIGEN_STACK_ALLOCATION_LIMIT)
67 && (EIGEN_MAX_STATIC_ALIGN_BYTES>0) // if we cannot allocate on the stack, then let's not bother about this optimization
68 };
69 typedef typename internal::conditional<CanAlign, Ref<const Matrix<Scalar,Dynamic,1,0,blockSize,1>, internal::evaluator<VectorTypeCopyClean>::Alignment>,
70 typename VectorTypeCopyClean::ConstSegmentReturnType>::type SegmentWrapper;
71 Index n = vec.size();
72
73 Index bi = internal::first_default_aligned(copy);
74 if (bi>0)
75 internal::stable_norm_kernel(copy.head(bi), ssq, scale, invScale);
76 for (; bi<n; bi+=blockSize)
77 internal::stable_norm_kernel(SegmentWrapper(copy.segment(bi,numext::mini(blockSize, n - bi))), ssq, scale, invScale);
78}

80template<typename VectorType>
81typename VectorType::RealScalar
82stable_norm_impl(const VectorType &vec, typename enable_if<VectorType::IsVectorAtCompileTime>::type* = 0 )
83{
84 using std::sqrt;
85 using std::abs;
86
87 Index n = vec.size();
88
89 if(n==1)
90 return abs(vec.coeff(0));
91
92 typedef typename VectorType::RealScalar RealScalar;
93 RealScalar scale(0);
94 RealScalar invScale(1);
95 RealScalar ssq(0); // sum of squares
96
97 stable_norm_impl_inner_step(vec, ssq, scale, invScale);
98
99 return scale * sqrt(ssq);
100}
101
102template<typename MatrixType>
103typename MatrixType::RealScalar
104stable_norm_impl(const MatrixType &mat, typename enable_if<!MatrixType::IsVectorAtCompileTime>::type* = 0 )
105{
106 using std::sqrt;
107
108 typedef typename MatrixType::RealScalar RealScalar;
109 RealScalar scale(0);
110 RealScalar invScale(1);
111 RealScalar ssq(0); // sum of squares
112
113 for(Index j=0; j<mat.outerSize(); ++j)
114 stable_norm_impl_inner_step(mat.innerVector(j), ssq, scale, invScale);
115 return scale * sqrt(ssq);
116}

118template<typename Derived>
119inline typename NumTraits<typename traits<Derived>::Scalar>::Real
120blueNorm_impl(const EigenBase<Derived>& _vec)
121{
122 typedef typename Derived::RealScalar RealScalar;
123 using std::pow;
124 using std::sqrt;
125 using std::abs;
126 const Derived& vec(_vec.derived());
127 static bool initialized = false;
128 static RealScalar b1, b2, s1m, s2m, rbig, relerr;
129 if(!initialized)
130 {
131 int ibeta, it, iemin, iemax, iexp;
132 RealScalar eps;
133 // This program calculates the machine-dependent constants
134 // bl, b2, slm, s2m, relerr overfl
135 // from the "basic" machine-dependent numbers
136 // nbig, ibeta, it, iemin, iemax, rbig.
137 // The following define the basic machine-dependent constants.
138 // For portability, the PORT subprograms "ilmaeh" and "rlmach"
139 // are used. For any specific computer, each of the assignment
140 // statements can be replaced
141 ibeta = std::numeric_limits<RealScalar>::radix; // base for floating-point numbers
142 it = NumTraits<RealScalar>::digits(); // number of base-beta digits in mantissa
143 iemin = std::numeric_limits<RealScalar>::min_exponent; // minimum exponent
144 iemax = std::numeric_limits<RealScalar>::max_exponent; // maximum exponent
145 rbig = (std::numeric_limits<RealScalar>::max)(); // largest floating-point number
146
147 iexp = -((1-iemin)/2);
148 b1 = RealScalar(pow(RealScalar(ibeta),RealScalar(iexp))); // lower boundary of midrange
149 iexp = (iemax + 1 - it)/2;
150 b2 = RealScalar(pow(RealScalar(ibeta),RealScalar(iexp))); // upper boundary of midrange
151
152 iexp = (2-iemin)/2;
153 s1m = RealScalar(pow(RealScalar(ibeta),RealScalar(iexp))); // scaling factor for lower range
154 iexp = - ((iemax+it)/2);
155 s2m = RealScalar(pow(RealScalar(ibeta),RealScalar(iexp))); // scaling factor for upper range
156
157 eps = RealScalar(pow(double(ibeta), 1-it));
158 relerr = sqrt(eps); // tolerance for neglecting asml
159 initialized = true;
160 }
161 Index n = vec.size();
162 RealScalar ab2 = b2 / RealScalar(n);
163 RealScalar asml = RealScalar(0);
164 RealScalar amed = RealScalar(0);
165 RealScalar abig = RealScalar(0);
166
167 for(Index j=0; j<vec.outerSize(); ++j)
168 {
169 for(typename Derived::InnerIterator it(vec, j); it; ++it)
170 {
171 RealScalar ax = abs(it.value());
172 if(ax > ab2) abig += numext::abs2(ax*s2m);
173 else if(ax < b1) asml += numext::abs2(ax*s1m);
174 else amed += numext::abs2(ax);
175 }
176 }
177 if(amed!=amed)
178 return amed; // we got a NaN
179 if(abig > RealScalar(0))
180 {
181 abig = sqrt(abig);
182 if(abig > rbig) // overflow, or *this contains INF values
183 return abig; // return INF
184 if(amed > RealScalar(0))
185 {
186 abig = abig/s2m;
187 amed = sqrt(amed);
188 }
189 else
190 return abig/s2m;
191 }
192 else if(asml > RealScalar(0))
193 {
194 if (amed > RealScalar(0))
195 {
196 abig = sqrt(amed);
197 amed = sqrt(asml) / s1m;
198 }
199 else
200 return sqrt(asml)/s1m;
201 }
202 else
203 return sqrt(amed);
204 asml = numext::mini(abig, amed);
205 abig = numext::maxi(abig, amed);
206 if(asml <= abig*relerr)
207 return abig;
208 else
209 return abig * sqrt(RealScalar(1) + numext::abs2(asml/abig));
210}

214/** \returns the \em l2 norm of \c *this avoiding underflow and overflow.
215 * This version use a blockwise two passes algorithm:
216 * 1 - find the absolute largest coefficient \c s
217 * 2 - compute \f$ s \Vert \frac{*this}{s} \Vert \f$ in a standard way
218 *
219 * For architecture/scalar types supporting vectorization, this version
220 * is faster than blueNorm(). Otherwise the blueNorm() is much faster.
221 *
222 * \sa norm(), blueNorm(), hypotNorm()
223 */
224template<typename Derived>
225inline typename NumTraits<typename internal::traits<Derived>::Scalar>::Real
226MatrixBase<Derived>::stableNorm() const
227{
228 return internal::stable_norm_impl(derived());
229}
230
231/** \returns the \em l2 norm of \c *this using the Blue's algorithm.
232 * A Portable Fortran Program to Find the Euclidean Norm of a Vector,
233 * ACM TOMS, Vol 4, Issue 1, 1978.
234 *
235 * For architecture/scalar types without vectorization, this version
236 * is much faster than stableNorm(). Otherwise the stableNorm() is faster.
237 *
238 * \sa norm(), stableNorm(), hypotNorm()
239 */
240template<typename Derived>
241inline typename NumTraits<typename internal::traits<Derived>::Scalar>::Real
242MatrixBase<Derived>::blueNorm() const
243{
244 return internal::blueNorm_impl(*this);
245}
246
247/** \returns the \em l2 norm of \c *this avoiding undeflow and overflow.
248 * This version use a concatenation of hypot() calls, and it is very slow.
249 *
250 * \sa norm(), stableNorm()
251 */
252template<typename Derived>
253inline typename NumTraits<typename internal::traits<Derived>::Scalar>::Real
254MatrixBase<Derived>::hypotNorm() const
255{
256 if(size()==1)
257 return numext::abs(coeff(0,0));
258 else
259 return this->cwiseAbs().redux(internal::scalar_hypot_op<RealScalar>());
260}
