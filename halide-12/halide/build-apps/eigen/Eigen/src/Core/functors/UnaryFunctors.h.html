17/** \internal

20 * \sa class CwiseUnaryOp, MatrixBase::operator-

23 EIGEN_EMPTY_STRUCT_CTOR(scalar_opposite_op)

31{ enum {
32 Cost = NumTraits<Scalar>::AddCost,
33 PacketAccess = packet_traits<Scalar>::HasNegate };

70/* Avoid recomputing abs when we know the score and they are the same. Not a true Eigen functor. */
71template<typename Scalar, typename=void> struct abs_knowing_score
72{
73 EIGEN_EMPTY_STRUCT_CTOR(abs_knowing_score)
74 typedef typename NumTraits<Scalar>::Real result_type;
75 template<typename Score>
76 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const result_type operator() (const Scalar& a, const Score&) const { return numext::abs(a); }
77};
78template<typename Scalar> struct abs_knowing_score<Scalar, typename scalar_score_coeff_op<Scalar>::Score_is_abs>
79{
80 EIGEN_EMPTY_STRUCT_CTOR(abs_knowing_score)
81 typedef typename NumTraits<Scalar>::Real result_type;
82 template<typename Scal>
83 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const result_type operator() (const Scal&, const result_type& a) const { return a; }

104/** \internal
105 * \brief Template functor to compute the conjugate of a complex value
106 *
107 * \sa class CwiseUnaryOp, MatrixBase::conjugate()
108 */
109template<typename Scalar> struct scalar_conjugate_op {
110 EIGEN_EMPTY_STRUCT_CTOR(scalar_conjugate_op)
111 EIGEN_DEVICE_FUNC
112 EIGEN_STRONG_INLINE const Scalar operator() (const Scalar& a) const { using numext::conj; return conj(a); }
113 template<typename Packet>
114 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Packet packetOp(const Packet& a) const { return internal::pconj(a); }
115};
116template<typename Scalar>
117struct functor_traits<scalar_conjugate_op<Scalar> >
118{
119 enum {
120 Cost = 0,
121 // Yes the cost is zero even for complexes because in most cases for which
122 // the cost is used, conjugation turns to be a no-op. Some examples:
123 // cost(a*conj(b)) == cost(a*b)
124 // cost(a+conj(b)) == cost(a+b)
125 // <etc.
126 // If we don't set it to zero, then:
127 // A.conjugate().lazyProduct(B.conjugate())
128 // will bake its operands. We definitely don't want that!
129 PacketAccess = packet_traits<Scalar>::HasConj
130 };
131};

271/** \internal
272 *
273 * \brief Template functor to compute the exponential of a scalar
274 *
275 * \sa class CwiseUnaryOp, Cwise::exp()
276 */
277template<typename Scalar> struct scalar_exp_op {
278 EIGEN_EMPTY_STRUCT_CTOR(scalar_exp_op)
279 EIGEN_DEVICE_FUNC inline const Scalar operator() (const Scalar& a) const { return numext::exp(a); }
280 template <typename Packet>
281 EIGEN_DEVICE_FUNC inline Packet packetOp(const Packet& a) const { return internal::pexp(a); }
282};
283template <typename Scalar>
284struct functor_traits<scalar_exp_op<Scalar> > {
285 enum {
286 PacketAccess = packet_traits<Scalar>::HasExp,
287 // The following numbers are based on the AVX implementation.
288#ifdef EIGEN_VECTORIZE_FMA
289 // Haswell can issue 2 add/mul/madd per cycle.
290 Cost =
291 (sizeof(Scalar) == 4
292 // float: 8 pmadd, 4 pmul, 2 padd/psub, 6 other
293 ? (8 * NumTraits<Scalar>::AddCost + 6 * NumTraits<Scalar>::MulCost)
294 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
295 : (14 * NumTraits<Scalar>::AddCost +
296 6 * NumTraits<Scalar>::MulCost +
297 scalar_div_cost<Scalar,packet_traits<Scalar>::HasDiv>::value))
298#else
299 Cost =
300 (sizeof(Scalar) == 4
301 // float: 7 pmadd, 6 pmul, 4 padd/psub, 10 other
302 ? (21 * NumTraits<Scalar>::AddCost + 13 * NumTraits<Scalar>::MulCost)
303 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
304 : (23 * NumTraits<Scalar>::AddCost +
305 12 * NumTraits<Scalar>::MulCost +
306 scalar_div_cost<Scalar,packet_traits<Scalar>::HasDiv>::value))
307#endif
308 };
309};

977#ifndef EIGEN_GPU_COMPILE_PHASE
978/** \internal
979 * \brief Template specialization of the logistic function for float.
980 *
981 * Uses just a 9/10-degree rational interpolant which
982 * interpolates 1/(1+exp(-x)) - 0.5 up to a couple of ulps in the range
983 * [-9, 18]. Below -9 we use the more accurate approximation
984 * 1/(1+exp(-x)) ~= exp(x), and above 18 the logistic function is 1 withing
985 * one ulp. The shifted logistic is interpolated because it was easier to
986 * make the fit converge.
987 *
988 */
989template <>
990struct scalar_logistic_op<float> {
991 EIGEN_EMPTY_STRUCT_CTOR(scalar_logistic_op)
992 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE float operator()(const float& x) const {
993 return packetOp(x);
994 }
995
996 template <typename Packet> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
997 Packet packetOp(const Packet& _x) const {
998 const Packet cutoff_lower = pset1<Packet>(-9.f);
999 const Packet lt_mask = pcmp_lt<Packet>(_x, cutoff_lower);
1000 const bool any_small = predux_any(lt_mask);
1001
1002 // The upper cut-off is the smallest x for which the rational approximation evaluates to 1.
1003 // Choosing this value saves us a few instructions clamping the results at the end.
1004#ifdef EIGEN_VECTORIZE_FMA
1005 const Packet cutoff_upper = pset1<Packet>(15.7243833541870117f);
1006#else
1007 const Packet cutoff_upper = pset1<Packet>(15.6437711715698242f);
1008#endif
1009 const Packet x = pmin(_x, cutoff_upper);
1010
1011 // The monomial coefficients of the numerator polynomial (odd).
1012 const Packet alpha_1 = pset1<Packet>(2.48287947061529e-01f);
1013 const Packet alpha_3 = pset1<Packet>(8.51377133304701e-03f);
1014 const Packet alpha_5 = pset1<Packet>(6.08574864600143e-05f);
1015 const Packet alpha_7 = pset1<Packet>(1.15627324459942e-07f);
1016 const Packet alpha_9 = pset1<Packet>(4.37031012579801e-11f);
1017
1018 // The monomial coefficients of the denominator polynomial (even).
1019 const Packet beta_0 = pset1<Packet>(9.93151921023180e-01f);
1020 const Packet beta_2 = pset1<Packet>(1.16817656904453e-01f);
1021 const Packet beta_4 = pset1<Packet>(1.70198817374094e-03f);
1022 const Packet beta_6 = pset1<Packet>(6.29106785017040e-06f);
1023 const Packet beta_8 = pset1<Packet>(5.76102136993427e-09f);
1024 const Packet beta_10 = pset1<Packet>(6.10247389755681e-13f);
1025
1026 // Since the polynomials are odd/even, we need x^2.
1027 const Packet x2 = pmul(x, x);
1028
1029 // Evaluate the numerator polynomial p.
1030 Packet p = pmadd(x2, alpha_9, alpha_7);
1031 p = pmadd(x2, p, alpha_5);
1032 p = pmadd(x2, p, alpha_3);
1033 p = pmadd(x2, p, alpha_1);
1034 p = pmul(x, p);
1035
1036 // Evaluate the denominator polynomial q.
1037 Packet q = pmadd(x2, beta_10, beta_8);
1038 q = pmadd(x2, q, beta_6);
1039 q = pmadd(x2, q, beta_4);
1040 q = pmadd(x2, q, beta_2);
1041 q = pmadd(x2, q, beta_0);
1042 // Divide the numerator by the denominator and shift it up.
1043 const Packet logistic = padd(pdiv(p, q), pset1<Packet>(0.5f));
1044 if (EIGEN_PREDICT_FALSE(any_small)) {
1045 const Packet exponential = pexp(_x);
1046 return pselect(lt_mask, exponential, logistic);
1047 } else {
1048 return logistic;
1049 }
1050 }
1051};
1052#endif // #ifndef EIGEN_GPU_COMPILE_PHASE

1054template <typename T>
1055struct functor_traits<scalar_logistic_op<T> > {
1056 enum {
1057 // The cost estimate for float here here is for the common(?) case where
1058 // all arguments are greater than -9.
1059 Cost = scalar_div_cost<T, packet_traits<T>::HasDiv>::value +
1060 (internal::is_same<T, float>::value
1061 ? NumTraits<T>::AddCost * 15 + NumTraits<T>::MulCost * 11
1062 : NumTraits<T>::AddCost * 2 +
1063 functor_traits<scalar_exp_op<T> >::Cost),
1064 PacketAccess =
1065 packet_traits<T>::HasAdd && packet_traits<T>::HasDiv &&
1066 (internal::is_same<T, float>::value
1067 ? packet_traits<T>::HasMul && packet_traits<T>::HasMax &&
1068 packet_traits<T>::HasMin
1069 : packet_traits<T>::HasNegate && packet_traits<T>::HasExp)
1070 };
1071};
