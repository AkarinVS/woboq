45/** \defgroup flags Flags
46 * \ingroup Core_Module
47 *
48 * These are the possible bits which can be OR'ed to constitute the flags of a matrix or
49 * expression.
50 *
51 * It is important to note that these flags are a purely compile-time notion. They are a compile-time property of
52 * an expression type, implemented as enum's. They are not stored in memory at runtime, and they do not incur any
53 * runtime overhead.
54 *
55 * \sa MatrixBase::Flags
56 */
57
58/** \ingroup flags
59 *
60 * for a matrix, this means that the storage order is row-major.
61 * If this bit is not set, the storage order is column-major.
62 * For an expression, this determines the storage order of
63 * the matrix created by evaluation of that expression.
64 * \sa \blank \ref TopicStorageOrders */
65const unsigned int RowMajorBit = 0x1;
66
67/** \ingroup flags
68 * means the expression should be evaluated by the calling expression */
69const unsigned int EvalBeforeNestingBit = 0x2;

71/** \ingroup flags
72 * \deprecated

77/** \ingroup flags
78 *
79 * Short version: means the expression might be vectorized
80 *
81 * Long version: means that the coefficients can be handled by packets
82 * and start at a memory location whose alignment meets the requirements
83 * of the present CPU architecture for optimized packet access. In the fixed-size
84 * case, there is the additional condition that it be possible to access all the
85 * coefficients by packets (this implies the requirement that the size be a multiple of 16 bytes,
86 * and that any nontrivial strides don't break the alignment). In the dynamic-size case,
87 * there is no such condition on the total size and strides, so it might not be possible to access
88 * all coeffs by packets.
89 *
90 * \note This bit can be set regardless of whether vectorization is actually enabled.
91 * To check for actual vectorizability, see \a ActualPacketAccessBit.
92 */
93const unsigned int PacketAccessBit = 0x8;
94
95#ifdef EIGEN_VECTORIZE
96/** \ingroup flags
97 *
98 * If vectorization is enabled (EIGEN_VECTORIZE is defined) this constant
99 * is set to the value \a PacketAccessBit.
100 *
101 * If vectorization is not enabled (EIGEN_VECTORIZE is not defined) this constant
102 * is set to the value 0.
103 */
104const unsigned int ActualPacketAccessBit = PacketAccessBit;
105#else
106const unsigned int ActualPacketAccessBit = 0x0;
107#endif
108
109/** \ingroup flags
110 *
111 * Short version: means the expression can be seen as 1D vector.
112 *
113 * Long version: means that one can access the coefficients
114 * of this expression by coeff(int), and coeffRef(int) in the case of a lvalue expression. These
115 * index-based access methods are guaranteed
116 * to not have to do any runtime computation of a (row, col)-pair from the index, so that it
117 * is guaranteed that whenever it is available, index-based access is at least as fast as
118 * (row,col)-based access. Expressions for which that isn't possible don't have the LinearAccessBit.
119 *
120 * If both PacketAccessBit and LinearAccessBit are set, then the
121 * packets of this expression can be accessed by packet(int), and writePacket(int) in the case of a
122 * lvalue expression.
123 *
124 * Typically, all vector expressions have the LinearAccessBit, but there is one exception:
125 * Product expressions don't have it, because it would be troublesome for vectorization, even when the
126 * Product is a vector expression. Thus, vector Product expressions allow index-based coefficient access but
127 * not index-based packet access, so they don't have the LinearAccessBit.
128 */
129const unsigned int LinearAccessBit = 0x10;
130
131/** \ingroup flags
132 *
133 * Means the expression has a coeffRef() method, i.e. is writable as its individual coefficients are directly addressable.
134 * This rules out read-only expressions.
135 *
136 * Note that DirectAccessBit and LvalueBit are mutually orthogonal, as there are examples of expression having one but note
137 * the other:
138 * \li writable expressions that don't have a very simple memory layout as a strided array, have LvalueBit but not DirectAccessBit
139 * \li Map-to-const expressions, for example Map<const Matrix>, have DirectAccessBit but not LvalueBit
140 *
141 * Expressions having LvalueBit also have their coeff() method returning a const reference instead of returning a new value.
142 */
143const unsigned int LvalueBit = 0x20;
144
145/** \ingroup flags
146 *
147 * Means that the underlying array of coefficients can be directly accessed as a plain strided array. The memory layout
148 * of the array of coefficients must be exactly the natural one suggested by rows(), cols(),
149 * outerStride(), innerStride(), and the RowMajorBit. This rules out expressions such as Diagonal, whose coefficients,
150 * though referencable, do not have such a regular memory layout.
151 *
152 * See the comment on LvalueBit for an explanation of how LvalueBit and DirectAccessBit are mutually orthogonal.
153 */
154const unsigned int DirectAccessBit = 0x40;
155
156/** \deprecated \ingroup flags
157 *
158 * means the first coefficient packet is guaranteed to be aligned.
159 * An expression cannot has the AlignedBit without the PacketAccessBit flag.
160 * In other words, this means we are allow to perform an aligned packet access to the first element regardless
161 * of the expression kind:
162 * \code
163 * expression.packet<Aligned>(0);
164 * \endcode
165 */
166EIGEN_DEPRECATED const unsigned int AlignedBit = 0x80;
167
168const unsigned int NestByRefBit = 0x100;

170/** \ingroup flags
171 *
172 * for an expression, this means that the storage order
173 * can be either row-major or column-major.
174 * The precise choice will be decided at evaluation time or when
175 * combined with other expressions.
176 * \sa \blank \ref RowMajorBit, \ref TopicStorageOrders */
177const unsigned int NoPreferredStorageOrderBit = 0x200;
178
179/** \ingroup flags
180 *
181 * Means that the underlying coefficients can be accessed through pointers to the sparse (un)compressed storage format,
182 * that is, the expression provides:
183 * \code
184 inline const Scalar* valuePtr() const;
185 inline const Index* innerIndexPtr() const;
186 inline const Index* outerIndexPtr() const;
187 inline const Index* innerNonZeroPtr() const;
188 \endcode
189 */
190const unsigned int CompressedAccessBit = 0x400;

272/** \internal \ingroup enums
273 * Enum to specify how to traverse the entries of a matrix. */
274enum TraversalType {
275 /** \internal Default traversal, no vectorization, no index-based access */
276 DefaultTraversal,
277 /** \internal No vectorization, use index-based access to have only one for loop instead of 2 nested loops */
278 LinearTraversal,
279 /** \internal Equivalent to a slice vectorization for fixed-size matrices having good alignment
280 * and good size */
281 InnerVectorizedTraversal,
282 /** \internal Vectorization path using a single loop plus scalar loops for the
283 * unaligned boundaries */
284 LinearVectorizedTraversal,
285 /** \internal Generic vectorization path using one vectorized loop per row/column with some
286 * scalar loops to handle the unaligned boundaries */
287 SliceVectorizedTraversal,
288 /** \internal Special case to properly handle incompatible scalar types or other defecting cases*/
289 InvalidTraversal,
290 /** \internal Evaluate all entries at once */
291 AllAtOnceTraversal
292};
