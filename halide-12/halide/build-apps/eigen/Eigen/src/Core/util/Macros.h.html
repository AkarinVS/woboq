445// Unify CUDA/HIPCC
446
447#if defined(EIGEN_CUDACC) || defined(EIGEN_HIPCC)
448//
449// If either EIGEN_CUDACC or EIGEN_HIPCC is defined, then define EIGEN_GPUCC
450//
451#define EIGEN_GPUCC
452//
453// EIGEN_HIPCC implies the HIP compiler and is used to tweak Eigen code for use in HIP kernels
454// EIGEN_CUDACC implies the CUDA compiler and is used to tweak Eigen code for use in CUDA kernels
455//
456// In most cases the same tweaks are required to the Eigen code to enable in both the HIP and CUDA kernels.
457// For those cases, the corresponding code should be guarded with
458// #if defined(EIGEN_GPUCC)
459// instead of
460// #if defined(EIGEN_CUDACC) || defined(EIGEN_HIPCC)
461//
462// For cases where the tweak is specific to HIP, the code should be guarded with
463// #if defined(EIGEN_HIPCC)
464//
465// For cases where the tweak is specific to CUDA, the code should be guarded with
466// #if defined(EIGEN_CUDACC)
467//
468#endif
469
470#if defined(EIGEN_CUDA_ARCH) || defined(EIGEN_HIP_DEVICE_COMPILE)
471//
472// If either EIGEN_CUDA_ARCH or EIGEN_HIP_DEVICE_COMPILE is defined, then define EIGEN_GPU_COMPILE_PHASE
473//
474#define EIGEN_GPU_COMPILE_PHASE
475//
476// GPU compilers (HIPCC, NVCC) typically do two passes over the source code,
477// + one to compile the source for the "host" (ie CPU)
478// + another to compile the source for the "device" (ie. GPU)
479//
480// Code that needs to enabled only during the either the "host" or "device" compilation phase
481// needs to be guarded with a macro that indicates the current compilation phase
482//
483// EIGEN_HIP_DEVICE_COMPILE implies the device compilation phase in HIP
484// EIGEN_CUDA_ARCH implies the device compilation phase in CUDA
485//
486// In most cases, the "host" / "device" specific code is the same for both HIP and CUDA
487// For those cases, the code should be guarded with
488// #if defined(EIGEN_GPU_COMPILE_PHASE)
489// instead of
490// #if defined(EIGEN_CUDA_ARCH) || defined(EIGEN_HIP_DEVICE_COMPILE)
491//
492// For cases where the tweak is specific to HIP, the code should be guarded with
493// #if defined(EIGEN_HIP_DEVICE_COMPILE)
494//
495// For cases where the tweak is specific to CUDA, the code should be guarded with
496// #if defined(EIGEN_CUDA_ARCH)
497//
498#endif

503#define SYCL_DEVICE_ONLY

1126// EIGEN_SIZE_MIN_PREFER_DYNAMIC gives the min between compile-time sizes. 0 has absolute priority, followed by 1,
1127// followed by Dynamic, followed by other finite values. The reason for giving Dynamic the priority over
1128// finite values is that min(3, Dynamic) should be Dynamic, since that could be anything between 0 and 3.
1129#define EIGEN_SIZE_MIN_PREFER_DYNAMIC(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
1130 : ((int)a == 1 || (int)b == 1) ? 1 \
1131 : ((int)a == Dynamic || (int)b == Dynamic) ? Dynamic \
1132 : ((int)a <= (int)b) ? (int)a : (int)b)
1133
1134// EIGEN_SIZE_MIN_PREFER_FIXED is a variant of EIGEN_SIZE_MIN_PREFER_DYNAMIC comparing MaxSizes. The difference is that finite values
1135// now have priority over Dynamic, so that min(3, Dynamic) gives 3. Indeed, whatever the actual value is
1136// (between 0 and 3), it is not more than 3.
1137#define EIGEN_SIZE_MIN_PREFER_FIXED(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
1138 : ((int)a == 1 || (int)b == 1) ? 1 \
1139 : ((int)a == Dynamic && (int)b == Dynamic) ? Dynamic \
1140 : ((int)a == Dynamic) ? (int)b \
1141 : ((int)b == Dynamic) ? (int)a \
1142 : ((int)a <= (int)b) ? (int)a : (int)b)
1143
1144// see EIGEN_SIZE_MIN_PREFER_DYNAMIC. No need for a separate variant for MaxSizes here.
1145#define EIGEN_SIZE_MAX(a,b) (((int)a == Dynamic || (int)b == Dynamic) ? Dynamic \
1146 : ((int)a >= (int)b) ? (int)a : (int)b)
