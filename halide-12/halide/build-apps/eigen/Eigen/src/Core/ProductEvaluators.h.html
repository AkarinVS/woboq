20/** \internal
21 * Evaluator of a product expression.
22 * Since products require special treatments to handle all possible cases,
23 * we simply defer the evaluation logic to a product_evaluator class
24 * which offers more partial specialization possibilities.
25 *
26 * \sa class product_evaluator
27 */
28template<typename Lhs, typename Rhs, int Options>
29struct evaluator<Product<Lhs, Rhs, Options> >
30 : public product_evaluator<Product<Lhs, Rhs, Options> >
31{
32 typedef Product<Lhs, Rhs, Options> XprType;
33 typedef product_evaluator<XprType> Base;
34
35 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE explicit evaluator(const XprType& xpr) : Base(xpr) {}
36};

93// This is the default evaluator implementation for products:
94// It creates a temporary and call generic_product_impl
95template<typename Lhs, typename Rhs, int Options, int ProductTag, typename LhsShape, typename RhsShape>
96struct product_evaluator<Product<Lhs, Rhs, Options>, ProductTag, LhsShape, RhsShape>
97 : public evaluator<typename Product<Lhs, Rhs, Options>::PlainObject>
98{
99 typedef Product<Lhs, Rhs, Options> XprType;
100 typedef typename XprType::PlainObject PlainObject;
101 typedef evaluator<PlainObject> Base;
102 enum {
103 Flags = Base::Flags | EvalBeforeNestingBit
104 };
105
106 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
107 explicit product_evaluator(const XprType& xpr)
108 : m_result(xpr.rows(), xpr.cols())
109 {
110 ::new (static_cast<Base*>(this)) Base(m_result);
111
112// FIXME shall we handle nested_eval here?,
113// if so, then we must take care at removing the call to nested_eval in the specializations (e.g., in permutation_matrix_product, transposition_matrix_product, etc.)
114// typedef typename internal::nested_eval<Lhs,Rhs::ColsAtCompileTime>::type LhsNested;
115// typedef typename internal::nested_eval<Rhs,Lhs::RowsAtCompileTime>::type RhsNested;
116// typedef typename internal::remove_all<LhsNested>::type LhsNestedCleaned;
117// typedef typename internal::remove_all<RhsNested>::type RhsNestedCleaned;
118//
119// const LhsNested lhs(xpr.lhs());
120// const RhsNested rhs(xpr.rhs());
121//
122// generic_product_impl<LhsNestedCleaned, RhsNestedCleaned>::evalTo(m_result, lhs, rhs);
123
124 generic_product_impl<Lhs, Rhs, LhsShape, RhsShape, ProductTag>::evalTo(m_result, xpr.lhs(), xpr.rhs());
125 }
126
127protected:
128 PlainObject m_result;
129};

135template< typename DstXprType, typename Lhs, typename Rhs, int Options, typename Scalar>
136struct Assignment<DstXprType, Product<Lhs,Rhs,Options>, internal::assign_op<Scalar,Scalar>, Dense2Dense,
137 typename enable_if<(Options==DefaultProduct || Options==AliasFreeProduct)>::type>
138{
139 typedef Product<Lhs,Rhs,Options> SrcXprType;
140 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
141 void run(DstXprType &dst, const SrcXprType &src, const internal::assign_op<Scalar,Scalar> &)
142 {
143 Index dstRows = src.rows();
144 Index dstCols = src.cols();
145 if((dst.rows()!=dstRows) || (dst.cols()!=dstCols))
146 dst.resize(dstRows, dstCols);
147 // FIXME shall we handle nested_eval here?
148 generic_product_impl<Lhs, Rhs>::evalTo(dst, src.lhs(), src.rhs());
149 }
150};

245template<typename Lhs, typename Rhs>
246struct generic_product_impl<Lhs,Rhs,DenseShape,DenseShape,InnerProduct>
247{
248 template<typename Dst>
249 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void evalTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
250 {
251 dst.coeffRef(0,0) = (lhs.transpose().cwiseProduct(rhs)).sum();
252 }
253
254 template<typename Dst>
255 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void addTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
256 {
257 dst.coeffRef(0,0) += (lhs.transpose().cwiseProduct(rhs)).sum();
258 }
259
260 template<typename Dst>
261 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void subTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
262 { dst.coeffRef(0,0) -= (lhs.transpose().cwiseProduct(rhs)).sum(); }
263};

270// Column major result
271template<typename Dst, typename Lhs, typename Rhs, typename Func>
272void EIGEN_DEVICE_FUNC outer_product_selector_run(Dst& dst, const Lhs &lhs, const Rhs &rhs, const Func& func, const false_type&)
273{
274 evaluator<Rhs> rhsEval(rhs);
275 ei_declare_local_nested_eval(Lhs,lhs,Rhs::SizeAtCompileTime,actual_lhs);
276 // FIXME if cols is large enough, then it might be useful to make sure that lhs is sequentially stored
277 // FIXME not very good if rhs is real and lhs complex while alpha is real too
278 const Index cols = dst.cols();
279 for (Index j=0; j<cols; ++j)
280 func(dst.col(j), rhsEval.coeff(Index(0),j) * actual_lhs);
281}
282
283// Row major result
284template<typename Dst, typename Lhs, typename Rhs, typename Func>
285void EIGEN_DEVICE_FUNC outer_product_selector_run(Dst& dst, const Lhs &lhs, const Rhs &rhs, const Func& func, const true_type&)
286{
287 evaluator<Lhs> lhsEval(lhs);
288 ei_declare_local_nested_eval(Rhs,rhs,Lhs::SizeAtCompileTime,actual_rhs);
289 // FIXME if rows is large enough, then it might be useful to make sure that rhs is sequentially stored
290 // FIXME not very good if lhs is real and rhs complex while alpha is real too
291 const Index rows = dst.rows();
292 for (Index i=0; i<rows; ++i)
293 func(dst.row(i), lhsEval.coeff(i,Index(0)) * actual_rhs);
294}

296template<typename Lhs, typename Rhs>
297struct generic_product_impl<Lhs,Rhs,DenseShape,DenseShape,OuterProduct>
298{
299 template<typename T> struct is_row_major : internal::conditional<(int(T::Flags)&RowMajorBit), internal::true_type, internal::false_type>::type {};
300 typedef typename Product<Lhs,Rhs>::Scalar Scalar;
301
302 // TODO it would be nice to be able to exploit our *_assign_op functors for that purpose
303 struct set { template<typename Dst, typename Src> EIGEN_DEVICE_FUNC void operator()(const Dst& dst, const Src& src) const { dst.const_cast_derived() = src; } };
304 struct add { template<typename Dst, typename Src> EIGEN_DEVICE_FUNC void operator()(const Dst& dst, const Src& src) const { dst.const_cast_derived() += src; } };
305 struct sub { template<typename Dst, typename Src> EIGEN_DEVICE_FUNC void operator()(const Dst& dst, const Src& src) const { dst.const_cast_derived() -= src; } };
306 struct adds {
307 Scalar m_scale;
308 explicit adds(const Scalar& s) : m_scale(s) {}
309 template<typename Dst, typename Src> void EIGEN_DEVICE_FUNC operator()(const Dst& dst, const Src& src) const {
310 dst.const_cast_derived() += m_scale * src;
311 }
312 };
313
314 template<typename Dst>
315 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void evalTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
316 {
317 internal::outer_product_selector_run(dst, lhs, rhs, set(), is_row_major<Dst>());
318 }
319
320 template<typename Dst>
321 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void addTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
322 {
323 internal::outer_product_selector_run(dst, lhs, rhs, add(), is_row_major<Dst>());
324 }
325
326 template<typename Dst>
327 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void subTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)
328 {
329 internal::outer_product_selector_run(dst, lhs, rhs, sub(), is_row_major<Dst>());
330 }
331
332 template<typename Dst>
333 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void scaleAndAddTo(Dst& dst, const Lhs& lhs, const Rhs& rhs, const Scalar& alpha)
334 {
335 internal::outer_product_selector_run(dst, lhs, rhs, adds(alpha), is_row_major<Dst>());
336 }
337
338};

354
355 template<typename Dst>
356 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void subTo(Dst& dst, const Lhs& lhs, const Rhs& rhs)

363};

414 // This is a special evaluation path called from generic_product_impl<...,GemmProduct> in file GeneralMatrixMatrix.h
415 // This variant tries to extract scalar multiples from both the LHS and RHS and factor them out. For instance:
416 // dst {,+,-}= (s1*A)*(B*s2)
417 // will be rewritten as:
418 // dst {,+,-}= (s1*s2) * (A.lazyProduct(B))
419 // There are at least four benefits of doing so:
420 // 1 - huge performance gain for heap-allocated matrix types as it save costly allocations.
421 // 2 - it is faster than simply by-passing the heap allocation through stack allocation.
422 // 3 - it makes this fallback consistent with the heavy GEMM routine.
423 // 4 - it fully by-passes huge stack allocation attempts when multiplying huge fixed-size matrices.
424 // (see https://stackoverflow.com/questions/54738495)
425 // For small fixed sizes matrices, howver, the gains are less obvious, it is sometimes x2 faster, but sometimes x3 slower,
426 // and the behavior depends also a lot on the compiler... This is why this re-writting strategy is currently
427 // enabled only when falling back from the main GEMM.
428 template<typename Dst, typename Func>
429 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
430 void eval_dynamic(Dst& dst, const Lhs& lhs, const Rhs& rhs, const Func &func)
431 {
432 enum {
433 HasScalarFactor = blas_traits<Lhs>::HasScalarFactor || blas_traits<Rhs>::HasScalarFactor,
434 ConjLhs = blas_traits<Lhs>::NeedToConjugate,
435 ConjRhs = blas_traits<Rhs>::NeedToConjugate
436 };
437 // FIXME: in c++11 this should be auto, and extractScalarFactor should also return auto
438 // this is important for real*complex_mat
439 Scalar actualAlpha = blas_traits<Lhs>::extractScalarFactor(lhs)
440 * blas_traits<Rhs>::extractScalarFactor(rhs);
441 eval_dynamic_impl(dst,
442 blas_traits<Lhs>::extract(lhs).template conjugateIf<ConjLhs>(),
443 blas_traits<Rhs>::extract(rhs).template conjugateIf<ConjRhs>(),
444 func,
445 actualAlpha,
446 typename conditional<HasScalarFactor,true_type,false_type>::type());
447 }
448
449protected:
450
451 template<typename Dst, typename LhsT, typename RhsT, typename Func, typename Scalar>
452 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
453 void eval_dynamic_impl(Dst& dst, const LhsT& lhs, const RhsT& rhs, const Func &func, const Scalar& s /* == 1 */, false_type)
454 {
455 EIGEN_UNUSED_VARIABLE(s);
456 eigen_internal_assert(s==Scalar(1));
457 call_restricted_packet_assignment_no_alias(dst, lhs.lazyProduct(rhs), func);
458 }
459
460 template<typename Dst, typename LhsT, typename RhsT, typename Func, typename Scalar>
461 static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
462 void eval_dynamic_impl(Dst& dst, const LhsT& lhs, const RhsT& rhs, const Func &func, const Scalar& s, true_type)
463 {
464 call_restricted_packet_assignment_no_alias(dst, s * lhs.lazyProduct(rhs), func);
465 }
466};

532 ColsAtCompileTime = RhsNestedCleaned::ColsAtCompileTime,
533 InnerSize = EIGEN_SIZE_MIN_PREFER_FIXED(LhsNestedCleaned::ColsAtCompileTime, RhsNestedCleaned::RowsAtCompileTime),
534 MaxRowsAtCompileTime = LhsNestedCleaned::MaxRowsAtCompileTime,

541 enum {
542
543 LhsCoeffReadCost = LhsEtorType::CoeffReadCost,
544 RhsCoeffReadCost = RhsEtorType::CoeffReadCost,
545 CoeffReadCost = InnerSize==0 ? NumTraits<Scalar>::ReadCost
546 : InnerSize == Dynamic ? HugeCost
547 : InnerSize * (NumTraits<Scalar>::MulCost + LhsCoeffReadCost + RhsCoeffReadCost)
548 + (InnerSize - 1) * NumTraits<Scalar>::AddCost,
549
550 Unroll = CoeffReadCost <= EIGEN_UNROLLING_LIMIT,
551
552 LhsFlags = LhsEtorType::Flags,
553 RhsFlags = RhsEtorType::Flags,
554
555 LhsRowMajor = LhsFlags & RowMajorBit,
556 RhsRowMajor = RhsFlags & RowMajorBit,
557
558 LhsVecPacketSize = unpacket_traits<LhsVecPacketType>::size,
559 RhsVecPacketSize = unpacket_traits<RhsVecPacketType>::size,
560
561 // Here, we don't care about alignment larger than the usable packet size.
562 LhsAlignment = EIGEN_PLAIN_ENUM_MIN(LhsEtorType::Alignment,LhsVecPacketSize*int(sizeof(typename LhsNestedCleaned::Scalar))),
563 RhsAlignment = EIGEN_PLAIN_ENUM_MIN(RhsEtorType::Alignment,RhsVecPacketSize*int(sizeof(typename RhsNestedCleaned::Scalar))),
564
565 SameType = is_same<typename LhsNestedCleaned::Scalar,typename RhsNestedCleaned::Scalar>::value,
566
567 CanVectorizeRhs = bool(RhsRowMajor) && (RhsFlags & PacketAccessBit) && (ColsAtCompileTime!=1),
568 CanVectorizeLhs = (!LhsRowMajor) && (LhsFlags & PacketAccessBit) && (RowsAtCompileTime!=1),
569
570 EvalToRowMajor = (MaxRowsAtCompileTime==1&&MaxColsAtCompileTime!=1) ? 1
571 : (MaxColsAtCompileTime==1&&MaxRowsAtCompileTime!=1) ? 0
572 : (bool(RhsRowMajor) && !CanVectorizeLhs),
573
574 Flags = ((unsigned int)(LhsFlags | RhsFlags) & HereditaryBits & ~RowMajorBit)
575 | (EvalToRowMajor ? RowMajorBit : 0)
576 // TODO enable vectorization for mixed types
577 | (SameType && (CanVectorizeLhs || CanVectorizeRhs) ? PacketAccessBit : 0)
578 | (XprType::IsVectorAtCompileTime ? LinearAccessBit : 0),
579
580 LhsOuterStrideBytes = int(LhsNestedCleaned::OuterStrideAtCompileTime) * int(sizeof(typename LhsNestedCleaned::Scalar)),
581 RhsOuterStrideBytes = int(RhsNestedCleaned::OuterStrideAtCompileTime) * int(sizeof(typename RhsNestedCleaned::Scalar)),
582
583 Alignment = bool(CanVectorizeLhs) ? (LhsOuterStrideBytes<=0 || (int(LhsOuterStrideBytes) % EIGEN_PLAIN_ENUM_MAX(1,LhsAlignment))!=0 ? 0 : LhsAlignment)
584 : bool(CanVectorizeRhs) ? (RhsOuterStrideBytes<=0 || (int(RhsOuterStrideBytes) % EIGEN_PLAIN_ENUM_MAX(1,RhsAlignment))!=0 ? 0 : RhsAlignment)
585 : 0,
586
587 /* CanVectorizeInner deserves special explanation. It does not affect the product flags. It is not used outside
588 * of Product. If the Product itself is not a packet-access expression, there is still a chance that the inner
589 * loop of the product might be vectorized. This is the meaning of CanVectorizeInner. Since it doesn't affect
590 * the Flags, it is safe to make this value depend on ActualPacketAccessBit, that doesn't affect the ABI.
591 */
592 CanVectorizeInner = SameType
593 && LhsRowMajor
594 && (!RhsRowMajor)
595 && (LhsFlags & RhsFlags & ActualPacketAccessBit)
596 && (InnerSize % packet_traits<Scalar>::size == 0)
597 };

599 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const CoeffReturnType coeff(Index row, Index col) const
600 {
601 return (m_lhs.row(row).transpose().cwiseProduct( m_rhs.col(col) )).sum();
602 }
603
604 /* Allow index-based non-packet access. It is impossible though to allow index-based packed access,
605 * which is why we don't set the LinearAccessBit.
606 * TODO: this seems possible when the result is a vector
607 */
608 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
609 const CoeffReturnType coeff(Index index) const
610 {
611 const Index row = (RowsAtCompileTime == 1 || MaxRowsAtCompileTime==1) ? 0 : index;
612 const Index col = (RowsAtCompileTime == 1 || MaxRowsAtCompileTime==1) ? index : 0;
613 return (m_lhs.row(row).transpose().cwiseProduct( m_rhs.col(col) )).sum();
614 }

626 }

634 return packet<LoadMode,PacketType>(row,col);

642 RhsEtorType m_rhsImpl;
643
644 // TODO: Get rid of m_innerDim if known at compile time
645 Index m_innerDim;

827public:
828 enum {
829 CoeffReadCost = NumTraits<Scalar>::MulCost + evaluator<MatrixType>::CoeffReadCost + evaluator<DiagonalType>::CoeffReadCost,
830

832 DiagFlags = evaluator<DiagonalType>::Flags,
833
834 _StorageOrder = (Derived::MaxRowsAtCompileTime==1 && Derived::MaxColsAtCompileTime!=1) ? RowMajor
835 : (Derived::MaxColsAtCompileTime==1 && Derived::MaxRowsAtCompileTime!=1) ? ColMajor
836 : MatrixFlags & RowMajorBit ? RowMajor : ColMajor,
837 _SameStorageOrder = _StorageOrder == (MatrixFlags & RowMajorBit ? RowMajor : ColMajor),

840 ||(int(_StorageOrder) == RowMajor && int(ProductOrder) == OnTheRight)),
841 _SameTypes = is_same<typename MatrixType::Scalar, typename DiagonalType::Scalar>::value,

886 };

888 m_diagImpl.template packet<DiagonalPacketLoadMode,PacketType>(id));
889 }

892 evaluator<MatrixType> m_matImpl;
893};

995template<typename ExpressionType, int Side, bool Transposed>
996struct permutation_matrix_product<ExpressionType, Side, Transposed, DenseShape>
997{
998 typedef typename nested_eval<ExpressionType, 1>::type MatrixType;
999 typedef typename remove_all<MatrixType>::type MatrixTypeCleaned;
1000
1001 template<typename Dest, typename PermutationType>
1002 static inline void run(Dest& dst, const PermutationType& perm, const ExpressionType& xpr)
1003 {
1004 MatrixType mat(xpr);
1005 const Index n = Side==OnTheLeft ? mat.rows() : mat.cols();
1006 // FIXME we need an is_same for expression that is not sensitive to constness. For instance
1007 // is_same_xpr<Block<const Matrix>, Block<Matrix> >::value should be true.
1008 //if(is_same<MatrixTypeCleaned,Dest>::value && extract_data(dst) == extract_data(mat))
1009 if(is_same_dense(dst, mat))
1010 {
1011 // apply the permutation inplace
1012 Matrix<bool,PermutationType::RowsAtCompileTime,1,0,PermutationType::MaxRowsAtCompileTime> mask(perm.size());
1013 mask.fill(false);
1014 Index r = 0;
1015 while(r < perm.size())
1016 {
1017 // search for the next seed
1018 while(r<perm.size() && mask[r]) r++;
1019 if(r>=perm.size())
1020 break;
1021 // we got one, let's follow it until we are back to the seed
1022 Index k0 = r++;
1023 Index kPrev = k0;
1024 mask.coeffRef(k0) = true;
1025 for(Index k=perm.indices().coeff(k0); k!=k0; k=perm.indices().coeff(k))
1026 {
1027 Block<Dest, Side==OnTheLeft ? 1 : Dest::RowsAtCompileTime, Side==OnTheRight ? 1 : Dest::ColsAtCompileTime>(dst, k)
1028 .swap(Block<Dest, Side==OnTheLeft ? 1 : Dest::RowsAtCompileTime, Side==OnTheRight ? 1 : Dest::ColsAtCompileTime>
1029 (dst,((Side==OnTheLeft) ^ Transposed) ? k0 : kPrev));
1030
1031 mask.coeffRef(k) = true;
1032 kPrev = k;
1033 }
1034 }
1035 }
1036 else
1037 {
1038 for(Index i = 0; i < n; ++i)
1039 {
1040 Block<Dest, Side==OnTheLeft ? 1 : Dest::RowsAtCompileTime, Side==OnTheRight ? 1 : Dest::ColsAtCompileTime>
1041 (dst, ((Side==OnTheLeft) ^ Transposed) ? perm.indices().coeff(i) : i)
1042
1043 =
1044
1045 Block<const MatrixTypeCleaned,Side==OnTheLeft ? 1 : MatrixTypeCleaned::RowsAtCompileTime,Side==OnTheRight ? 1 : MatrixTypeCleaned::ColsAtCompileTime>
1046 (mat, ((Side==OnTheRight) ^ Transposed) ? perm.indices().coeff(i) : i);
1047 }
1048 }
1049 }
1050};

1103template<typename ExpressionType, int Side, bool Transposed, typename ExpressionShape>
1104struct transposition_matrix_product
1105{
1106 typedef typename nested_eval<ExpressionType, 1>::type MatrixType;
1107 typedef typename remove_all<MatrixType>::type MatrixTypeCleaned;
1108
1109 template<typename Dest, typename TranspositionType>
1110 static inline void run(Dest& dst, const TranspositionType& tr, const ExpressionType& xpr)
1111 {
1112 MatrixType mat(xpr);
1113 typedef typename TranspositionType::StorageIndex StorageIndex;
1114 const Index size = tr.size();
1115 StorageIndex j = 0;
1116
1117 if(!is_same_dense(dst,mat))
1118 dst = mat;
1119
1120 for(Index k=(Transposed?size-1:0) ; Transposed?k>=0:k<size ; Transposed?--k:++k)
1121 if(Index(j=tr.coeff(k))!=k)
1122 {
1123 if(Side==OnTheLeft) dst.row(k).swap(dst.row(j));
1124 else if(Side==OnTheRight) dst.col(k).swap(dst.col(j));
1125 }
1126 }
1127};
