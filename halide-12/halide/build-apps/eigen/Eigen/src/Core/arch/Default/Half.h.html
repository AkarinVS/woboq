28// Standard 16-bit float type, mostly useful for GPUs. Defines a new
29// type Eigen::half (inheriting either from CUDA's or HIP's __half struct) with
30// operator overloads such that it behaves basically as an arithmetic
31// type. It will be quite slow on CPUs (so it is recommended to stay
32// in fp32 for CPUs, except for simple parameter conversions, I/O
33// to disk and the likes), but fast on GPUs.

426union float32_bits {
427 unsigned int u;
428 float f;
429};

431EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC __half_raw float_to_half_rtne(float ff) {
432#if (defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300) || \
433 (defined(EIGEN_HAS_HIP_FP16) && defined(EIGEN_HIP_DEVICE_COMPILE))
434 __half tmp_ff = __float2half(ff);
435 return *(__half_raw*)&tmp_ff;
436
437#elif defined(EIGEN_HAS_FP16_C)
438 __half_raw h;
439 h.x = _cvtss_sh(ff, 0);
440 return h;
441
442#else
443 float32_bits f; f.f = ff;
444
445 const float32_bits f32infty = { 255 << 23 };
446 const float32_bits f16max = { (127 + 16) << 23 };
447 const float32_bits denorm_magic = { ((127 - 15) + (23 - 10) + 1) << 23 };
448 unsigned int sign_mask = 0x80000000u;
449 __half_raw o;
450 o.x = static_cast<unsigned short>(0x0u);
451
452 unsigned int sign = f.u & sign_mask;
453 f.u ^= sign;
454
455 // NOTE all the integer compares in this function can be safely
456 // compiled into signed compares since all operands are below
457 // 0x80000000. Important if you want fast straight SSE2 code
458 // (since there's no unsigned PCMPGTD).
459
460 if (f.u >= f16max.u) { // result is Inf or NaN (all exponent bits set)
461 o.x = (f.u > f32infty.u) ? 0x7e00 : 0x7c00; // NaN->qNaN and Inf->Inf
462 } else { // (De)normalized number or zero
463 if (f.u < (113 << 23)) { // resulting FP16 is subnormal or zero
464 // use a magic value to align our 10 mantissa bits at the bottom of
465 // the float. as long as FP addition is round-to-nearest-even this
466 // just works.
467 f.f += denorm_magic.f;
468
469 // and one integer subtract of the bias later, we have our final float!
470 o.x = static_cast<unsigned short>(f.u - denorm_magic.u);
471 } else {
472 unsigned int mant_odd = (f.u >> 13) & 1; // resulting mantissa is odd
473
474 // update exponent, rounding bias part 1
475 f.u += ((unsigned int)(15 - 127) << 23) + 0xfff;
476 // rounding bias part 2
477 f.u += mant_odd;
478 // take the bits!
479 o.x = static_cast<unsigned short>(f.u >> 13);
480 }
481 }
482
483 o.x |= static_cast<unsigned short>(sign >> 16);
484 return o;
485#endif
486}

488EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC float half_to_float(__half_raw h) {
489#if (defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300) || \
490 (defined(EIGEN_HAS_HIP_FP16) && defined(EIGEN_HIP_DEVICE_COMPILE))
491 return __half2float(h);
492
493#elif defined(EIGEN_HAS_FP16_C)
494 return _cvtsh_ss(h.x);
495
496#else
497 const float32_bits magic = { 113 << 23 };
498 const unsigned int shifted_exp = 0x7c00 << 13; // exponent mask after shift
499 float32_bits o;
500
501 o.u = (h.x & 0x7fff) << 13; // exponent/mantissa bits
502 unsigned int exp = shifted_exp & o.u; // just the exponent
503 o.u += (127 - 15) << 23; // exponent adjust
504
505 // handle exponent special cases
506 if (exp == shifted_exp) { // Inf/NaN?
507 o.u += (128 - 16) << 23; // extra exp adjust
508 } else if (exp == 0) { // Zero/Denormal?
509 o.u += 1 << 23; // extra exp adjust
510 o.f -= magic.f; // renormalize
511 }
512
513 o.u |= (h.x & 0x8000) << 16; // sign bit
514 return o.f;
515#endif
516}
