101 /** \returns an expression of the coefficient-wise power of \a x to the given constant \a exponent.
102 *
103 * \tparam ScalarExponent is the scalar type of \a exponent. It must be compatible with the scalar type of the given expression (\c Derived::Scalar).
104 *
105 * \sa ArrayBase::pow()
106 *
107 * \relates ArrayBase
108 */
109#ifdef EIGEN_PARSED_BY_DOXYGEN
110 template<typename Derived,typename ScalarExponent>
111 inline const CwiseBinaryOp<internal::scalar_pow_op<Derived::Scalar,ScalarExponent>,Derived,Constant<ScalarExponent> >
112 pow(const Eigen::ArrayBase<Derived>& x, const ScalarExponent& exponent);
113#else
114 template <typename Derived,typename ScalarExponent>
115 EIGEN_DEVICE_FUNC inline
116 EIGEN_MSVC10_WORKAROUND_BINARYOP_RETURN_TYPE(
117 const EIGEN_EXPR_BINARYOP_SCALAR_RETURN_TYPE(Derived,typename internal::promote_scalar_arg<typename Derived::Scalar
118 EIGEN_COMMA ScalarExponent EIGEN_COMMA
119 EIGEN_SCALAR_BINARY_SUPPORTED(pow,typename Derived::Scalar,ScalarExponent)>::type,pow))
120 pow(const Eigen::ArrayBase<Derived>& x, const ScalarExponent& exponent)
121 {
122 typedef typename internal::promote_scalar_arg<typename Derived::Scalar,ScalarExponent,
123 EIGEN_SCALAR_BINARY_SUPPORTED(pow,typename Derived::Scalar,ScalarExponent)>::type PromotedExponent;
124 return EIGEN_EXPR_BINARYOP_SCALAR_RETURN_TYPE(Derived,PromotedExponent,pow)(x.derived(),
125 typename internal::plain_constant_type<Derived,PromotedExponent>::type(x.derived().rows(), x.derived().cols(), internal::scalar_constant_op<PromotedExponent>(exponent)));
126 }
127#endif

129 /** \returns an expression of the coefficient-wise power of \a x to the given array of \a exponents.
130 *
131 * This function computes the coefficient-wise power.
132 *
133 * Example: \include Cwise_array_power_array.cpp
134 * Output: \verbinclude Cwise_array_power_array.out
135 *
136 * \sa ArrayBase::pow()
137 *
138 * \relates ArrayBase
139 */
140 template<typename Derived,typename ExponentDerived>
141 inline const Eigen::CwiseBinaryOp<Eigen::internal::scalar_pow_op<typename Derived::Scalar, typename ExponentDerived::Scalar>, const Derived, const ExponentDerived>
142 pow(const Eigen::ArrayBase<Derived>& x, const Eigen::ArrayBase<ExponentDerived>& exponents)
143 {
144 return Eigen::CwiseBinaryOp<Eigen::internal::scalar_pow_op<typename Derived::Scalar, typename ExponentDerived::Scalar>, const Derived, const ExponentDerived>(
145 x.derived(),
146 exponents.derived()
147 );
148 }

150 /** \returns an expression of the coefficient-wise power of the scalar \a x to the given array of \a exponents.
151 *
152 * This function computes the coefficient-wise power between a scalar and an array of exponents.
153 *
154 * \tparam Scalar is the scalar type of \a x. It must be compatible with the scalar type of the given array expression (\c Derived::Scalar).
155 *
156 * Example: \include Cwise_scalar_power_array.cpp
157 * Output: \verbinclude Cwise_scalar_power_array.out
158 *
159 * \sa ArrayBase::pow()
160 *
161 * \relates ArrayBase
162 */
163#ifdef EIGEN_PARSED_BY_DOXYGEN
164 template<typename Scalar,typename Derived>
165 inline const CwiseBinaryOp<internal::scalar_pow_op<Scalar,Derived::Scalar>,Constant<Scalar>,Derived>
166 pow(const Scalar& x,const Eigen::ArrayBase<Derived>& x);
167#else
168 template <typename Scalar, typename Derived>
169 EIGEN_DEVICE_FUNC inline
170 EIGEN_MSVC10_WORKAROUND_BINARYOP_RETURN_TYPE(
171 const EIGEN_SCALAR_BINARYOP_EXPR_RETURN_TYPE(typename internal::promote_scalar_arg<typename Derived::Scalar
172 EIGEN_COMMA Scalar EIGEN_COMMA
173 EIGEN_SCALAR_BINARY_SUPPORTED(pow,Scalar,typename Derived::Scalar)>::type,Derived,pow))
174 pow(const Scalar& x, const Eigen::ArrayBase<Derived>& exponents) {
175 typedef typename internal::promote_scalar_arg<typename Derived::Scalar,Scalar,
176 EIGEN_SCALAR_BINARY_SUPPORTED(pow,Scalar,typename Derived::Scalar)>::type PromotedScalar;
177 return EIGEN_SCALAR_BINARYOP_EXPR_RETURN_TYPE(PromotedScalar,Derived,pow)(
178 typename internal::plain_constant_type<Derived,PromotedScalar>::type(exponents.derived().rows(), exponents.derived().cols(), internal::scalar_constant_op<PromotedScalar>(x)), exponents.derived());
179 }
180#endif
