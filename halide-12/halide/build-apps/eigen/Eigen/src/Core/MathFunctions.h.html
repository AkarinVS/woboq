30/** \internal \class global_math_functions_filtering_base
31 *
32 * What it does:
33 * Defines a typedef 'type' as follows:
34 * - if type T has a member typedef Eigen_BaseClassForSpecializationOfGlobalMathFuncImpl, then
35 * global_math_functions_filtering_base<T>::type is a typedef for it.
36 * - otherwise, global_math_functions_filtering_base<T>::type is a typedef for T.
37 *
38 * How it's used:
39 * To allow to defined the global math functions (like sin...) in certain cases, like the Array expressions.
40 * When you do sin(array1+array2), the object array1+array2 has a complicated expression type, all what you want to know
41 * is that it inherits ArrayBase. So we implement a partial specialization of sin_impl for ArrayBase<Derived>.
42 * So we must make sure to use sin_impl<ArrayBase<Derived> > and not sin_impl<Derived>, otherwise our partial specialization
43 * won't be used. How does sin know that? That's exactly what global_math_functions_filtering_base tells it.
44 *
45 * How it's implemented:
46 * SFINAE in the style of enable_if. Highly susceptible of breaking compilers. With GCC, it sure does work, but if you replace
47 * the typename dummy by an integer template parameter, it doesn't work anymore!
48 */
49
50template<typename T, typename dummy = void>
51struct global_math_functions_filtering_base
52{
53 typedef T type;
54};
55
56template<typename T> struct always_void { typedef void type; };
57
58template<typename T>
59struct global_math_functions_filtering_base
60 <T,
61 typename always_void<typename T::Eigen_BaseClassForSpecializationOfGlobalMathFuncImpl>::type
62 >
63{
64 typedef typename T::Eigen_BaseClassForSpecializationOfGlobalMathFuncImpl type;
65};

67#define EIGEN_MATHFUNC_IMPL(func, scalar) Eigen::internal::func##_impl<typename Eigen::internal::global_math_functions_filtering_base<scalar>::type>
68#define EIGEN_MATHFUNC_RETVAL(func, scalar) typename Eigen::internal::func##_retval<typename Eigen::internal::global_math_functions_filtering_base<scalar>::type>::type

528* Implementation of expm1 *

531// This implementation is based on GSL Math's expm1.
532namespace std_fallback {
533 // fallback expm1 implementation in case there is no expm1(Scalar) function in namespace of Scalar,
534 // or that there is no suitable std::expm1 function available. Implementation
535 // attributed to Kahan. See: http://www.plunk.org/~hatch/rightway.php.
536 template<typename Scalar>
537 EIGEN_DEVICE_FUNC inline Scalar expm1(const Scalar& x) {
538 EIGEN_STATIC_ASSERT_NON_INTEGER(Scalar)
539 typedef typename NumTraits<Scalar>::Real RealScalar;
540
541 EIGEN_USING_STD_MATH(exp);
542 Scalar u = exp(x);
543 if (numext::equal_strict(u, Scalar(1))) {
544 return x;
545 }
546 Scalar um1 = u - RealScalar(1);
547 if (numext::equal_strict(um1, Scalar(-1))) {
548 return RealScalar(-1);
549 }
550
551 EIGEN_USING_STD_MATH(log);
552 Scalar logu = log(u);
553 return numext::equal_strict(u, logu) ? u : (u - RealScalar(1)) * x / logu;
554 }
555}
556
557template<typename Scalar>
558struct expm1_impl {
559 EIGEN_DEVICE_FUNC static inline Scalar run(const Scalar& x)
560 {
561 EIGEN_STATIC_ASSERT_NON_INTEGER(Scalar)
562 #if EIGEN_HAS_CXX11_MATH
563 using std::expm1;
564 #else
565 using std_fallback::expm1;
566 #endif
567 return expm1(x);
568 }
569};
570
571// Specialization for complex types that are not supported by std::expm1.
572template <typename RealScalar>
573struct expm1_impl<std::complex<RealScalar> > {
574 EIGEN_DEVICE_FUNC static inline std::complex<RealScalar> run(
575 const std::complex<RealScalar>& x) {
576 EIGEN_STATIC_ASSERT_NON_INTEGER(RealScalar)
577 RealScalar xr = x.real();
578 RealScalar xi = x.imag();
579 // expm1(z) = exp(z) - 1
580 // = exp(x + i * y) - 1
581 // = exp(x) * (cos(y) + i * sin(y)) - 1
582 // = exp(x) * cos(y) - 1 + i * exp(x) * sin(y)
583 // Imag(expm1(z)) = exp(x) * sin(y)
584 // Real(expm1(z)) = exp(x) * cos(y) - 1
585 // = exp(x) * cos(y) - 1.
586 // = expm1(x) + exp(x) * (cos(y) - 1)
587 // = expm1(x) + exp(x) * (2 * sin(y / 2) ** 2)
588
589 // TODO better use numext::expm1 and numext::sin (but that would require forward declarations or moving this specialization down).
590 RealScalar erm1 = expm1_impl<RealScalar>::run(xr);
591 RealScalar er = erm1 + RealScalar(1.);
592 EIGEN_USING_STD_MATH(sin);
593 RealScalar sin2 = sin(xi / RealScalar(2.));
594 sin2 = sin2 * sin2;
595 RealScalar s = sin(xi);
596 RealScalar real_part = erm1 - RealScalar(2.) * er * sin2;
597 return std::complex<RealScalar>(real_part, er * s);
598 }
599};

612 // fallback log1p implementation in case there is no log1p(Scalar) function in namespace of Scalar,
613 // or that there is no suitable std::log1p function available
614 template<typename Scalar>
615 EIGEN_DEVICE_FUNC inline Scalar log1p(const Scalar& x) {
616 EIGEN_STATIC_ASSERT_NON_INTEGER(Scalar)
617 typedef typename NumTraits<Scalar>::Real RealScalar;
618 EIGEN_USING_STD_MATH(log);
619 Scalar x1p = RealScalar(1) + x;
620 Scalar log_1p = log(x1p);
621 const bool is_small = numext::equal_strict(x1p, Scalar(1));
622 const bool is_inf = numext::equal_strict(x1p, log_1p);
623 return (is_small || is_inf) ? x : x * (log_1p / (x1p - RealScalar(1)));
624 }
625}

774template<typename Scalar>
775struct random_default_impl<Scalar, false, true>
776{
777 static inline Scalar run(const Scalar& x, const Scalar& y)
778 {
779 if (y <= x)
780 return x;
781 // ScalarU is the unsigned counterpart of Scalar, possibly Scalar itself.
782 typedef typename make_unsigned<Scalar>::type ScalarU;
783 // ScalarX is the widest of ScalarU and unsigned int.
784 // We'll deal only with ScalarX and unsigned int below thus avoiding signed
785 // types and arithmetic and signed overflows (which are undefined behavior).
786 typedef typename conditional<(ScalarU(-1) > unsigned(-1)), ScalarU, unsigned>::type ScalarX;
787 // The following difference doesn't overflow, provided our integer types are two's
788 // complement and have the same number of padding bits in signed and unsigned variants.
789 // This is the case in most modern implementations of C++.
790 ScalarX range = ScalarX(y) - ScalarX(x);
791 ScalarX offset = 0;
792 ScalarX divisor = 1;
793 ScalarX multiplier = 1;
794 const unsigned rand_max = RAND_MAX;
795 if (range <= rand_max) divisor = (rand_max + 1) / (range + 1);
796 else multiplier = 1 + range / (rand_max + 1);
797 // Rejection sampling.
798 do {
799 offset = (unsigned(std::rand()) * multiplier) / divisor;
800 } while (offset > range);
801 return Scalar(ScalarX(x) + offset);
802 }
803
804 static inline Scalar run()
805 {
806#ifdef EIGEN_MAKING_DOCS
807 return run(Scalar(NumTraits<Scalar>::IsSigned ? -10 : 0), Scalar(10));
808#else
809 enum { rand_bits = meta_floor_log2<(unsigned int)(RAND_MAX)+1>::value,
810 scalar_bits = sizeof(Scalar) * CHAR_BIT,
811 shift = EIGEN_PLAIN_ENUM_MAX(0, int(rand_bits) - int(scalar_bits)),
812 offset = NumTraits<Scalar>::IsSigned ? (1 << (EIGEN_PLAIN_ENUM_MIN(rand_bits,scalar_bits)-1)) : 0
813 };
814 return Scalar((std::rand() >> shift) - offset);
815#endif
816 }
817};

1338 v |= v >> 1;
1339 v |= v >> 2;
1340 v |= v >> 4;
1341 v |= v >> 8;
1342 v |= v >> 16;
1343 return table[(v * 0x07C4ACDDU) >> 27];
