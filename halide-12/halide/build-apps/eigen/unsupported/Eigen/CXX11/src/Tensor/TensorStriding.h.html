99// Eval as rvalue
100template<typename Strides, typename ArgType, typename Device>
101struct TensorEvaluator<const TensorStridingOp<Strides, ArgType>, Device>
102{
103 typedef TensorStridingOp<Strides, ArgType> XprType;
104 typedef typename XprType::Index Index;
105 static const int NumDims = internal::array_size<typename TensorEvaluator<ArgType, Device>::Dimensions>::value;
106 typedef DSizes<Index, NumDims> Dimensions;
107 typedef typename XprType::Scalar Scalar;
108 typedef typename XprType::CoeffReturnType CoeffReturnType;
109 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
110 static const int PacketSize = PacketType<CoeffReturnType, Device>::size;
111 typedef StorageMemory<CoeffReturnType, Device> Storage;
112 typedef typename Storage::Type EvaluatorPointerType;
113
114 enum {
115 IsAligned = /*TensorEvaluator<ArgType, Device>::IsAligned*/false,
116 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
117 BlockAccess = false,
118 PreferBlockAccess = TensorEvaluator<ArgType, Device>::PreferBlockAccess,
119 Layout = TensorEvaluator<ArgType, Device>::Layout,
120 CoordAccess = false, // to be implemented
121 RawAccess = false
122 };
123
124 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
125 typedef internal::TensorBlockNotImplemented TensorBlock;
126 //===--------------------------------------------------------------------===//
127
128 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
129 : m_impl(op.expression(), device)
130 {
131 m_dimensions = m_impl.dimensions();
132 for (int i = 0; i < NumDims; ++i) {
133 m_dimensions[i] =Eigen::numext::ceil(static_cast<float>(m_dimensions[i]) / op.strides()[i]);
134 }
135
136 const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
137 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
138 m_outputStrides[0] = 1;
139 m_inputStrides[0] = 1;
140 for (int i = 1; i < NumDims; ++i) {
141 m_outputStrides[i] = m_outputStrides[i-1] * m_dimensions[i-1];
142 m_inputStrides[i] = m_inputStrides[i-1] * input_dims[i-1];
143 m_inputStrides[i-1] *= op.strides()[i-1];
144 }
145 m_inputStrides[NumDims-1] *= op.strides()[NumDims-1];
146 } else { // RowMajor
147 m_outputStrides[NumDims-1] = 1;
148 m_inputStrides[NumDims-1] = 1;
149 for (int i = NumDims - 2; i >= 0; --i) {
150 m_outputStrides[i] = m_outputStrides[i+1] * m_dimensions[i+1];
151 m_inputStrides[i] = m_inputStrides[i+1] * input_dims[i+1];
152 m_inputStrides[i+1] *= op.strides()[i+1];
153 }
154 m_inputStrides[0] *= op.strides()[0];
155 }
156 }
157
158
159 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Dimensions& dimensions() const { return m_dimensions; }
160
161 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE bool evalSubExprsIfNeeded(EvaluatorPointerType/*data*/) {
162 m_impl.evalSubExprsIfNeeded(NULL);
163 return true;
164 }
165 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void cleanup() {
166 m_impl.cleanup();
167 }

169 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE CoeffReturnType coeff(Index index) const
170 {
171 return m_impl.coeff(srcCoeff(index));
172 }

174 template<int LoadMode>
175 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
176 {
177 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
178 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
179
180 Index inputIndices[] = {0, 0};
181 Index indices[] = {index, index + PacketSize - 1};
182 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
183 EIGEN_UNROLL_LOOP
184 for (int i = NumDims - 1; i > 0; --i) {
185 const Index idx0 = indices[0] / m_outputStrides[i];
186 const Index idx1 = indices[1] / m_outputStrides[i];
187 inputIndices[0] += idx0 * m_inputStrides[i];
188 inputIndices[1] += idx1 * m_inputStrides[i];
189 indices[0] -= idx0 * m_outputStrides[i];
190 indices[1] -= idx1 * m_outputStrides[i];
191 }
192 inputIndices[0] += indices[0] * m_inputStrides[0];
193 inputIndices[1] += indices[1] * m_inputStrides[0];
194 } else { // RowMajor
195 EIGEN_UNROLL_LOOP
196 for (int i = 0; i < NumDims - 1; ++i) {
197 const Index idx0 = indices[0] / m_outputStrides[i];
198 const Index idx1 = indices[1] / m_outputStrides[i];
199 inputIndices[0] += idx0 * m_inputStrides[i];
200 inputIndices[1] += idx1 * m_inputStrides[i];
201 indices[0] -= idx0 * m_outputStrides[i];
202 indices[1] -= idx1 * m_outputStrides[i];
203 }
204 inputIndices[0] += indices[0] * m_inputStrides[NumDims-1];
205 inputIndices[1] += indices[1] * m_inputStrides[NumDims-1];
206 }
207 if (inputIndices[1] - inputIndices[0] == PacketSize - 1) {
208 PacketReturnType rslt = m_impl.template packet<Unaligned>(inputIndices[0]);
209 return rslt;
210 }
211 else {
212 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
213 values[0] = m_impl.coeff(inputIndices[0]);
214 values[PacketSize-1] = m_impl.coeff(inputIndices[1]);
215 EIGEN_UNROLL_LOOP
216 for (int i = 1; i < PacketSize-1; ++i) {
217 values[i] = coeff(index+i);
218 }
219 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
220 return rslt;
221 }
222 }

224 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost costPerCoeff(bool vectorized) const {
225 double compute_cost = (NumDims - 1) * (TensorOpCost::AddCost<Index>() +
226 TensorOpCost::MulCost<Index>() +
227 TensorOpCost::DivCost<Index>()) +
228 TensorOpCost::MulCost<Index>();
229 if (vectorized) {
230 compute_cost *= 2; // packet() computes two indices
231 }
232 const int innerDim = (static_cast<int>(Layout) == static_cast<int>(ColMajor)) ? 0 : (NumDims - 1);
233 return m_impl.costPerCoeff(vectorized && m_inputStrides[innerDim] == 1) +
234 // Computation is not vectorized per se, but it is done once per packet.
235 TensorOpCost(0, 0, compute_cost, vectorized, PacketSize);
236 }
237
238 EIGEN_DEVICE_FUNC typename Storage::Type data() const { return NULL; }
239
240#ifdef EIGEN_USE_SYCL
241 // binding placeholder accessors to a command group handler for SYCL
242 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void bind(cl::sycl::handler &cgh) const {
243 m_impl.bind(cgh);
244 }
245#endif
246 protected:
247 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index srcCoeff(Index index) const
248 {
249 Index inputIndex = 0;
250 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
251 EIGEN_UNROLL_LOOP
252 for (int i = NumDims - 1; i > 0; --i) {
253 const Index idx = index / m_outputStrides[i];
254 inputIndex += idx * m_inputStrides[i];
255 index -= idx * m_outputStrides[i];
256 }
257 inputIndex += index * m_inputStrides[0];
258 } else { // RowMajor
259 EIGEN_UNROLL_LOOP
260 for (int i = 0; i < NumDims - 1; ++i) {
261 const Index idx = index / m_outputStrides[i];
262 inputIndex += idx * m_inputStrides[i];
263 index -= idx * m_outputStrides[i];
264 }
265 inputIndex += index * m_inputStrides[NumDims-1];
266 }
267 return inputIndex;
268 }
269
270 Dimensions m_dimensions;
271 array<Index, NumDims> m_outputStrides;
272 array<Index, NumDims> m_inputStrides;
273 TensorEvaluator<ArgType, Device> m_impl;
274};

310 template <int StoreMode> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
311 void writePacket(Index index, const PacketReturnType& x)
312 {
313 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
314 eigen_assert(index+PacketSize-1 < this->dimensions().TotalSize());
315
316 Index inputIndices[] = {0, 0};
317 Index indices[] = {index, index + PacketSize - 1};
318 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
319 EIGEN_UNROLL_LOOP
320 for (int i = NumDims - 1; i > 0; --i) {
321 const Index idx0 = indices[0] / this->m_outputStrides[i];
322 const Index idx1 = indices[1] / this->m_outputStrides[i];
323 inputIndices[0] += idx0 * this->m_inputStrides[i];
324 inputIndices[1] += idx1 * this->m_inputStrides[i];
325 indices[0] -= idx0 * this->m_outputStrides[i];
326 indices[1] -= idx1 * this->m_outputStrides[i];
327 }
328 inputIndices[0] += indices[0] * this->m_inputStrides[0];
329 inputIndices[1] += indices[1] * this->m_inputStrides[0];
330 } else { // RowMajor
331 EIGEN_UNROLL_LOOP
332 for (int i = 0; i < NumDims - 1; ++i) {
333 const Index idx0 = indices[0] / this->m_outputStrides[i];
334 const Index idx1 = indices[1] / this->m_outputStrides[i];
335 inputIndices[0] += idx0 * this->m_inputStrides[i];
336 inputIndices[1] += idx1 * this->m_inputStrides[i];
337 indices[0] -= idx0 * this->m_outputStrides[i];
338 indices[1] -= idx1 * this->m_outputStrides[i];
339 }
340 inputIndices[0] += indices[0] * this->m_inputStrides[NumDims-1];
341 inputIndices[1] += indices[1] * this->m_inputStrides[NumDims-1];
342 }
343 if (inputIndices[1] - inputIndices[0] == PacketSize - 1) {
344 this->m_impl.template writePacket<Unaligned>(inputIndices[0], x);
345 }
346 else {
347 EIGEN_ALIGN_MAX Scalar values[PacketSize];
348 internal::pstore<Scalar, PacketReturnType>(values, x);
349 this->m_impl.coeffRef(inputIndices[0]) = values[0];
350 this->m_impl.coeffRef(inputIndices[1]) = values[PacketSize-1];
351 EIGEN_UNROLL_LOOP
352 for (int i = 1; i < PacketSize-1; ++i) {
353 this->coeffRef(index+i) = values[i];
354 }
355 }
356 }
357};
