96template<typename Broadcast, typename ArgType, typename Device>
97struct TensorEvaluator<const TensorBroadcastingOp<Broadcast, ArgType>, Device>
98{
99 typedef TensorBroadcastingOp<Broadcast, ArgType> XprType;
100 typedef typename XprType::Index Index;
101 static const int NumDims = internal::array_size<typename TensorEvaluator<ArgType, Device>::Dimensions>::value;
102 typedef DSizes<Index, NumDims> Dimensions;
103 typedef typename XprType::Scalar Scalar;
104 typedef typename TensorEvaluator<ArgType, Device>::Dimensions InputDimensions;
105 typedef typename XprType::CoeffReturnType CoeffReturnType;
106 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
107 static const int PacketSize = PacketType<CoeffReturnType, Device>::size;
108 protected: // all the non-static fields must have the same access control, otherwise the TensorEvaluator wont be standard layout;
109 bool isCopy, nByOne, oneByN;
110 public:
111 typedef StorageMemory<CoeffReturnType, Device> Storage;
112 typedef typename Storage::Type EvaluatorPointerType;
113
114 enum {
115 IsAligned = TensorEvaluator<ArgType, Device>::IsAligned,
116 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
117 BlockAccess = TensorEvaluator<ArgType, Device>::BlockAccess,
118 PreferBlockAccess = true,
119 Layout = TensorEvaluator<ArgType, Device>::Layout,
120 RawAccess = false
121 };
122
123 typedef typename internal::remove_const<Scalar>::type ScalarNoConst;
124
125 // We do block based broadcasting using a trick with 2x tensor rank and 0
126 // strides. See block method implementation for details.
127 typedef DSizes<Index, 2 * NumDims> BroadcastDimensions;
128
129 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
130 typedef internal::TensorBlockDescriptor<NumDims, Index> TensorBlockDesc;
131 typedef internal::TensorBlockScratchAllocator<Device> TensorBlockScratch;
132
133 typedef typename TensorEvaluator<const ArgType, Device>::TensorBlock
134 ArgTensorBlock;
135
136 typedef typename internal::TensorMaterializedBlock<ScalarNoConst, NumDims,
137 Layout, Index>

141 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op,
142 const Device& device)
143 : isCopy(false), nByOne(false), oneByN(false),
144 m_device(device), m_broadcast(op.broadcast()), m_impl(op.expression(), device)
145 {
146
147 // The broadcasting op doesn't change the rank of the tensor. One can't broadcast a scalar
148 // and store the result in a scalar. Instead one should reshape the scalar into a a N-D
149 // tensor with N >= 1 of 1 element first and then broadcast.
150 EIGEN_STATIC_ASSERT((NumDims > 0), YOU_MADE_A_PROGRAMMING_MISTAKE);
151 const InputDimensions& input_dims = m_impl.dimensions();
152 isCopy = true;
153 for (int i = 0; i < NumDims; ++i) {
154 eigen_assert(input_dims[i] > 0);
155 m_dimensions[i] = input_dims[i] * m_broadcast[i];
156 if (m_broadcast[i] != 1) {
157 isCopy = false;
158 }
159 }
160
161 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
162 m_inputStrides[0] = 1;
163 m_outputStrides[0] = 1;
164 for (int i = 1; i < NumDims; ++i) {
165 m_inputStrides[i] = m_inputStrides[i-1] * input_dims[i-1];
166 m_outputStrides[i] = m_outputStrides[i-1] * m_dimensions[i-1];
167 }
168 } else {
169 m_inputStrides[NumDims-1] = 1;
170 m_outputStrides[NumDims-1] = 1;
171 for (int i = NumDims-2; i >= 0; --i) {
172 m_inputStrides[i] = m_inputStrides[i+1] * input_dims[i+1];
173 m_outputStrides[i] = m_outputStrides[i+1] * m_dimensions[i+1];
174 }
175 }
176
177 if (input_dims[0] == 1) {
178 oneByN = true;
179 for (int i = 1; i < NumDims; ++i) {
180 if (m_broadcast[i] != 1) {
181 oneByN = false;
182 break;
183 }
184 }
185 } else if (input_dims[NumDims-1] == 1) {
186 nByOne = true;
187 for (int i = 0; i < NumDims-1; ++i) {
188 if (m_broadcast[i] != 1) {
189 nByOne = false;
190 break;
191 }
192 }
193 }
194
195 // Handle special format like NCHW, its input shape is '[1, N..., 1]' and
196 // broadcast shape is '[N, 1..., N]'
197 if (!oneByN && !nByOne) {
198 if (input_dims[0] == 1 && input_dims[NumDims-1] == 1 && NumDims > 2) {
199 nByOne = true;
200 oneByN = true;
201 for (int i = 1; i < NumDims-1; ++i) {
202 if (m_broadcast[i] != 1) {
203 nByOne = false;
204 oneByN = false;
205 break;
206 }
207 }
208 }
209 }
210 }

231 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE CoeffReturnType coeff(Index index) const
232 {
233 if (internal::is_input_scalar<typename internal::remove_all<InputDimensions>::type>::value) {
234 return m_impl.coeff(0);
235 }
236
237 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
238 if (isCopy) {
239 return m_impl.coeff(index);
240 } else {
241 return coeffColMajor(index);
242 }
243 } else {
244 if (isCopy) {
245 return m_impl.coeff(index);
246 } else {
247 return coeffRowMajor(index);
248 }
249 }
250 }

252 // TODO: attempt to speed this up. The integer divisions and modulo are slow
253 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index indexColMajor(Index index) const {
254 Index inputIndex = 0;
255 EIGEN_UNROLL_LOOP
256 for (int i = NumDims - 1; i > 0; --i) {
257 const Index idx = index / m_outputStrides[i];
258 if (internal::index_statically_eq<Broadcast>(i, 1)) {
259 eigen_assert(idx < m_impl.dimensions()[i]);
260 inputIndex += idx * m_inputStrides[i];
261 } else {
262 if (internal::index_statically_eq<InputDimensions>(i, 1)) {
263 eigen_assert(idx % m_impl.dimensions()[i] == 0);
264 } else {
265 inputIndex += (idx % m_impl.dimensions()[i]) * m_inputStrides[i];
266 }
267 }
268 index -= idx * m_outputStrides[i];
269 }
270 if (internal::index_statically_eq<Broadcast>(0, 1)) {
271 eigen_assert(index < m_impl.dimensions()[0]);
272 inputIndex += index;
273 } else {
274 if (internal::index_statically_eq<InputDimensions>(0, 1)) {
275 eigen_assert(index % m_impl.dimensions()[0] == 0);
276 } else {
277 inputIndex += (index % m_impl.dimensions()[0]);
278 }
279 }
280 return inputIndex;
281 }

288 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index indexRowMajor(Index index) const {
289 Index inputIndex = 0;
290 EIGEN_UNROLL_LOOP
291 for (int i = 0; i < NumDims - 1; ++i) {
292 const Index idx = index / m_outputStrides[i];
293 if (internal::index_statically_eq<Broadcast>(i, 1)) {
294 eigen_assert(idx < m_impl.dimensions()[i]);
295 inputIndex += idx * m_inputStrides[i];
296 } else {
297 if (internal::index_statically_eq<InputDimensions>(i, 1)) {
298 eigen_assert(idx % m_impl.dimensions()[i] == 0);
299 } else {
300 inputIndex += (idx % m_impl.dimensions()[i]) * m_inputStrides[i];
301 }
302 }
303 index -= idx * m_outputStrides[i];
304 }
305 if (internal::index_statically_eq<Broadcast>(NumDims - 1, 1)) {
306 eigen_assert(index < m_impl.dimensions()[NumDims - 1]);
307 inputIndex += index;
308 } else {
309 if (internal::index_statically_eq<InputDimensions>(NumDims - 1, 1)) {
310 eigen_assert(index % m_impl.dimensions()[NumDims - 1] == 0);
311 } else {
312 inputIndex += (index % m_impl.dimensions()[NumDims - 1]);
313 }
314 }
315 return inputIndex;
316 }

323 template<int LoadMode>
324 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE PacketReturnType packet(Index index) const
325 {
326 if (internal::is_input_scalar<typename internal::remove_all<InputDimensions>::type>::value) {
327 return internal::pset1<PacketReturnType>(m_impl.coeff(0));
328 }
329
330 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
331 if (isCopy) {
332 #ifdef EIGEN_GPU_COMPILE_PHASE
333 // See PR 437: on NVIDIA P100 and K20m we observed a x3-4 speed up by enforcing
334 // unaligned loads here. The reason is unclear though.
335 return m_impl.template packet<Unaligned>(index);
336 #else
337 return m_impl.template packet<LoadMode>(index);
338 #endif
339 } else if (oneByN && !nByOne) {
340 return packetNByOne<LoadMode>(index);
341 } else if (!oneByN && nByOne) {
342 return packetOneByN<LoadMode>(index);
343 } else if (oneByN && nByOne) {
344 return packetOneByNByOne<LoadMode>(index);
345 } else {
346 return packetColMajor<LoadMode>(index);
347 }
348 } else {
349 if (isCopy) {
350 #ifdef EIGEN_GPU_COMPILE_PHASE
351 // See above.
352 return m_impl.template packet<Unaligned>(index);
353 #else
354 return m_impl.template packet<LoadMode>(index);
355 #endif
356 } else if (oneByN && !nByOne) {
357 return packetOneByN<LoadMode>(index);
358 } else if (!oneByN && nByOne) {
359 return packetNByOne<LoadMode>(index);
360 } else if (oneByN && nByOne) {
361 return packetOneByNByOne<LoadMode>(index);
362 } else {
363 return packetRowMajor<LoadMode>(index);
364 }
365 }
366 }

368 template<int LoadMode>
369 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packetOneByNByOne
370 (Index index) const
371 {
372 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
373 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
374
375 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
376 Index startDim, endDim;
377 Index inputIndex, outputOffset, batchedIndex;
378
379 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
380 startDim = NumDims - 1;
381 endDim = 1;
382 } else {
383 startDim = 0;
384 endDim = NumDims - 2;
385 }
386
387 batchedIndex = index % m_outputStrides[startDim];
388 inputIndex = batchedIndex / m_outputStrides[endDim];
389 outputOffset = batchedIndex % m_outputStrides[endDim];
390
391 if (outputOffset + PacketSize <= m_outputStrides[endDim]) {
392 values[0] = m_impl.coeff(inputIndex);
393 return internal::pload1<PacketReturnType>(values);
394 } else {
395 EIGEN_UNROLL_LOOP
396 for (int i = 0, cur = 0; i < PacketSize; ++i, ++cur) {
397 if (outputOffset + cur < m_outputStrides[endDim]) {
398 values[i] = m_impl.coeff(inputIndex);
399 } else {
400 ++inputIndex;
401 inputIndex = (inputIndex == m_inputStrides[startDim] ? 0 : inputIndex);
402 values[i] = m_impl.coeff(inputIndex);
403 outputOffset = 0;
404 cur = 0;
405 }
406 }
407 return internal::pload<PacketReturnType>(values);
408 }
409 }

411 template<int LoadMode>
412 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packetOneByN(Index index) const
413 {
414 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
415 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
416
417 Index dim, inputIndex;
418
419 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
420 dim = NumDims - 1;
421 } else {
422 dim = 0;
423 }
424
425 inputIndex = index % m_inputStrides[dim];
426 if (inputIndex + PacketSize <= m_inputStrides[dim]) {
427 return m_impl.template packet<Unaligned>(inputIndex);
428 } else {
429 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
430 EIGEN_UNROLL_LOOP
431 for (int i = 0; i < PacketSize; ++i) {
432 if (inputIndex > m_inputStrides[dim]-1) {
433 inputIndex = 0;
434 }
435 values[i] = m_impl.coeff(inputIndex++);
436 }
437 return internal::pload<PacketReturnType>(values);
438 }
439 }

441 template<int LoadMode>
442 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packetNByOne(Index index) const
443 {
444 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
445 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
446
447 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
448 Index dim, inputIndex, outputOffset;
449
450 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
451 dim = 1;
452 } else {
453 dim = NumDims - 2;
454 }
455
456 inputIndex = index / m_outputStrides[dim];
457 outputOffset = index % m_outputStrides[dim];
458 if (outputOffset + PacketSize <= m_outputStrides[dim]) {
459 values[0] = m_impl.coeff(inputIndex);
460 return internal::pload1<PacketReturnType>(values);
461 } else {
462 EIGEN_UNROLL_LOOP
463 for (int i = 0, cur = 0; i < PacketSize; ++i, ++cur) {
464 if (outputOffset + cur < m_outputStrides[dim]) {
465 values[i] = m_impl.coeff(inputIndex);
466 } else {
467 values[i] = m_impl.coeff(++inputIndex);
468 outputOffset = 0;
469 cur = 0;
470 }
471 }
472 return internal::pload<PacketReturnType>(values);
473 }

476 // Ignore the LoadMode and always use unaligned loads since we can't guarantee
477 // the alignment at compile time.
478 template<int LoadMode>
479 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packetColMajor(Index index) const
480 {
481 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
482 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
483
484 const Index originalIndex = index;
485
486 Index inputIndex = 0;
487 EIGEN_UNROLL_LOOP
488 for (int i = NumDims - 1; i > 0; --i) {
489 const Index idx = index / m_outputStrides[i];
490 if (internal::index_statically_eq<Broadcast>(i, 1)) {
491 eigen_assert(idx < m_impl.dimensions()[i]);
492 inputIndex += idx * m_inputStrides[i];
493 } else {
494 if (internal::index_statically_eq<InputDimensions>(i, 1)) {
495 eigen_assert(idx % m_impl.dimensions()[i] == 0);
496 } else {
497 inputIndex += (idx % m_impl.dimensions()[i]) * m_inputStrides[i];
498 }
499 }
500 index -= idx * m_outputStrides[i];
501 }
502 Index innermostLoc;
503 if (internal::index_statically_eq<Broadcast>(0, 1)) {
504 eigen_assert(index < m_impl.dimensions()[0]);
505 innermostLoc = index;
506 } else {
507 if (internal::index_statically_eq<InputDimensions>(0, 1)) {
508 eigen_assert(index % m_impl.dimensions()[0] == 0);
509 innermostLoc = 0;
510 } else {
511 innermostLoc = index % m_impl.dimensions()[0];
512 }
513 }
514 inputIndex += innermostLoc;
515
516 // Todo: this could be extended to the second dimension if we're not
517 // broadcasting alongside the first dimension, and so on.
518 if (innermostLoc + PacketSize <= m_impl.dimensions()[0]) {
519 return m_impl.template packet<Unaligned>(inputIndex);
520 } else {
521 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
522 values[0] = m_impl.coeff(inputIndex);
523 EIGEN_UNROLL_LOOP
524 for (int i = 1; i < PacketSize; ++i) {
525 if (innermostLoc + i < m_impl.dimensions()[0]) {
526 values[i] = m_impl.coeff(inputIndex+i);
527 } else {
528 values[i] = coeffColMajor(originalIndex+i);
529 }
530 }
531 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
532 return rslt;
533 }
534 }

536 template<int LoadMode>
537 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packetRowMajor(Index index) const
538 {
539 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
540 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
541
542 const Index originalIndex = index;
543
544 Index inputIndex = 0;
545 EIGEN_UNROLL_LOOP
546 for (int i = 0; i < NumDims - 1; ++i) {
547 const Index idx = index / m_outputStrides[i];
548 if (internal::index_statically_eq<Broadcast>(i, 1)) {
549 eigen_assert(idx < m_impl.dimensions()[i]);
550 inputIndex += idx * m_inputStrides[i];
551 } else {
552 if (internal::index_statically_eq<InputDimensions>(i, 1)) {
553 eigen_assert(idx % m_impl.dimensions()[i] == 0);
554 } else {
555 inputIndex += (idx % m_impl.dimensions()[i]) * m_inputStrides[i];
556 }
557 }
558 index -= idx * m_outputStrides[i];
559 }
560 Index innermostLoc;
561 if (internal::index_statically_eq<Broadcast>(NumDims-1, 1)) {
562 eigen_assert(index < m_impl.dimensions()[NumDims-1]);
563 innermostLoc = index;
564 } else {
565 if (internal::index_statically_eq<InputDimensions>(NumDims-1, 1)) {
566 eigen_assert(index % m_impl.dimensions()[NumDims-1] == 0);
567 innermostLoc = 0;
568 } else {
569 innermostLoc = index % m_impl.dimensions()[NumDims-1];
570 }
571 }
572 inputIndex += innermostLoc;
573
574 // Todo: this could be extended to the second dimension if we're not
575 // broadcasting alongside the first dimension, and so on.
576 if (innermostLoc + PacketSize <= m_impl.dimensions()[NumDims-1]) {
577 return m_impl.template packet<Unaligned>(inputIndex);
578 } else {
579 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
580 values[0] = m_impl.coeff(inputIndex);
581 EIGEN_UNROLL_LOOP
582 for (int i = 1; i < PacketSize; ++i) {
583 if (innermostLoc + i < m_impl.dimensions()[NumDims-1]) {
584 values[i] = m_impl.coeff(inputIndex+i);
585 } else {
586 values[i] = coeffRowMajor(originalIndex+i);
587 }
588 }
589 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
590 return rslt;
591 }
592 }

594 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost
595 costPerCoeff(bool vectorized) const {
596 double compute_cost = TensorOpCost::AddCost<Index>();
597 if (!isCopy && NumDims > 0) {
598 EIGEN_UNROLL_LOOP
599 for (int i = NumDims - 1; i > 0; --i) {
600 compute_cost += TensorOpCost::DivCost<Index>();
601 if (internal::index_statically_eq<Broadcast>(i, 1)) {
602 compute_cost +=
603 TensorOpCost::MulCost<Index>() + TensorOpCost::AddCost<Index>();
604 } else {
605 if (!internal::index_statically_eq<InputDimensions>(i, 1)) {
606 compute_cost += TensorOpCost::MulCost<Index>() +
607 TensorOpCost::ModCost<Index>() +
608 TensorOpCost::AddCost<Index>();
609 }
610 }
611 compute_cost +=
612 TensorOpCost::MulCost<Index>() + TensorOpCost::AddCost<Index>();
613 }
614 }
615 return m_impl.costPerCoeff(vectorized) +
616 TensorOpCost(0, 0, compute_cost, vectorized, PacketSize);
617 }

619 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
620 internal::TensorBlockResourceRequirements getResourceRequirements() const {
621 // TODO(wuke): Targeting L1 size is 30% faster than targeting L{-1} on large
622 // tensors. But this might need further tuning.
623 const size_t target_size = m_device.firstLevelCacheSize();
624 return internal::TensorBlockResourceRequirements::merge(
625 m_impl.getResourceRequirements(),
626 internal::TensorBlockResourceRequirements::skewed<Scalar>(target_size));
627 }

629 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorBlock
630 block(TensorBlockDesc& desc, TensorBlockScratch& scratch,
631 bool /*root_of_expr_ast*/ = false) const {
632 BlockBroadcastingParams params = blockBroadcastingParams(desc);
633
634 if (params.inner_dim_size == 0 || params.bcast_dim_size == 0) {
635 return emptyBlock();
636 }
637
638 // Prepare storage for the materialized broadcasting result.
639 const typename TensorBlock::Storage block_storage =
640 TensorBlock::prepareStorage(desc, scratch);
641 ScalarNoConst* materialized_output = block_storage.data();
642
643 // We potentially will need to materialize input blocks.
644 size_t materialized_input_size = 0;
645 ScalarNoConst* materialized_input = NULL;
646
647 // Initialize block broadcating iterator state for outer dimensions (outer
648 // with regard to bcast dimension). Dimension in this array are always in
649 // inner_most -> outer_most order (col major layout).
650 array<BlockBroadcastingIteratorState, NumDims> it;
651 int idx = 0;
652
653 for (int i = params.inner_dim_count + 1; i < NumDims; ++i) {
654 const Index dim = IsColMajor ? i : NumDims - 1 - i;
655 it[idx].size = params.output_dims[dim];
656 it[idx].count = 0;
657 it[idx].output_stride = m_outputStrides[dim];
658 it[idx].output_span = it[idx].output_stride * (it[idx].size - 1);
659 idx++;
660 }
661
662 // Write output into the beginning of `materialized_output`.
663 Index output_offset = 0;
664
665 // We will fill output block by broadcasting along the bcast dim, and
666 // iterating over outer dimension.
667 const Index output_size = NumDims == 0 ? 1 : params.output_dims.TotalSize();
668
669 for (Index num_output_coeffs = 0; num_output_coeffs < output_size;) {
670 ScalarNoConst* bcast_output = materialized_output + num_output_coeffs;
671 Index bcast_offset = desc.offset() + output_offset;
672
673 // Broadcast along the bcast dimension.
674 num_output_coeffs += BroadcastBlockAlongBcastDim(
675 params, bcast_offset, scratch, bcast_output, &materialized_input,
676 &materialized_input_size);
677
678 // Switch to the next outer dimension.
679 for (int j = 0; j < idx; ++j) {
680 if (++it[j].count < it[j].size) {
681 output_offset += it[j].output_stride;
682 break;
683 }
684 it[j].count = 0;
685 output_offset -= it[j].output_span;
686 }
687 }
688
689 return block_storage.AsTensorMaterializedBlock();
690 }

704 private:
705 static const bool IsColMajor =
706 static_cast<int>(Layout) == static_cast<int>(ColMajor);
707
708 // We will build a general case block broadcasting on top of broadcasting
709 // primitive that will do broadcasting only for the inner dimension(s) along
710 // the first dimension smaller than the input size (it's called `bcast_dim`).
711 //
712 // Example:
713 // dim: 0 1 2 (ColMajor)
714 // input size: [9, 3, 6]
715 // block size: [9, 2, 6]
716 //
717 // We will compute broadcasted block by iterating over the outer dimensions
718 // before `bcast_dim` (only dimension `2` in this example) and computing
719 // broadcasts along the `bcast_dim` (dimension `1` in this example).
720
721 // BlockBroadcastingParams holds precomputed parameters for broadcasting a
722 // single block along the broadcasting dimension. Sizes and strides along the
723 // `bcast_dim` might be invalid, they will be adjusted later in
724 // `BroadcastBlockAlongBcastDim`.
725 struct BlockBroadcastingParams {
726 Dimensions input_dims; // input expression dimensions
727 Dimensions output_dims; // output block sizes
728 Dimensions output_strides; // output block strides
729
730 int inner_dim_count; // count inner dimensions matching in size
731 int bcast_dim; // broadcasting dimension index
732 Index bcast_dim_size; // broadcasting dimension size
733 Index inner_dim_size; // inner dimensions size
734
735 // Block sizes and strides for the input block where all dimensions before
736 // `bcast_dim` are equal to `1`.
737 Dimensions input_block_sizes;
738 Dimensions input_block_strides;
739
740 // Block sizes and strides for blocks with extra dimensions and strides `0`.
741 BroadcastDimensions bcast_block_sizes;
742 BroadcastDimensions bcast_block_strides;
743 BroadcastDimensions bcast_input_strides;
744 };
745
746 struct BlockBroadcastingIteratorState {
747 Index size;
748 Index count;
749 Index output_stride;
750 Index output_span;
751 };

753 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE BlockBroadcastingParams
754 blockBroadcastingParams(TensorBlockDesc& desc) const {
755 BlockBroadcastingParams params;
756
757 params.input_dims = Dimensions(m_impl.dimensions());
758
759 // Output block sizes and strides.
760 params.output_dims = desc.dimensions();
761 params.output_strides = internal::strides<Layout>(params.output_dims);
762
763 // Find the broadcasting dimension (first dimension with output size smaller
764 // that the input size).
765 params.bcast_dim = 0;
766 params.bcast_dim_size = 1;
767 params.inner_dim_size = 1;
768
769 // Count the number of inner dimensions that have the same size in the block
770 // and in the broadcast expression.
771 params.inner_dim_count = 0;
772
773 for (int i = 0; i < NumDims; ++i) {
774 const int dim = IsColMajor ? i : NumDims - i - 1;
775
776 if (params.output_dims[dim] == m_dimensions[dim]) {
777 params.inner_dim_size *= params.output_dims[dim];
778 ++params.inner_dim_count;
779 continue;
780 }
781
782 // First non-matching dimension is the broadcasting dimension.
783 eigen_assert(params.output_dims[dim] < m_dimensions[dim]);
784 params.bcast_dim = dim;
785 params.bcast_dim_size = params.output_dims[dim];
786 break;
787 }
788
789 // Calculate the input block size for looking into the input.
790 for (int i = 0; i < params.inner_dim_count; ++i) {
791 const int dim = IsColMajor ? i : NumDims - i - 1;
792 params.input_block_sizes[dim] = params.input_dims[dim];
793 }
794 for (int i = params.inner_dim_count; i < NumDims; ++i) {
795 const int dim = IsColMajor ? i : NumDims - i - 1;
796 params.input_block_sizes[dim] = 1;
797 }
798 params.input_block_strides =
799 internal::strides<Layout>(params.input_block_sizes);
800
801 // Broadcast with the 0-stride trick: Create 1 extra dim for each
802 // broadcast, set the input stride to 0.
803 //
804 // When ColMajor:
805 //
806 // - bcast_block_sizes:
807 // [d_0, b_0, d_1, b_1, ...]
808 //
809 // - bcast_block_strides:
810 // [output_block_strides[0], output_block_strides[0] * d_0,
811 // output_block_strides[1], output_block_strides[1] * d_1,
812 // ...]
813 //
814 // - bcast_input_strides:
815 // [input_block_strides[0], 0,
816 // input_block_strides[1], 0,
817 // ...].
818 //
819 for (int i = 0; i < params.inner_dim_count; ++i) {
820 const int dim = IsColMajor ? i : NumDims - i - 1;
821
822 const int copy_dim = IsColMajor ? 2 * i : 2 * NumDims - 2 * i - 1;
823 const int broadcast_dim = IsColMajor ? copy_dim + 1 : copy_dim - 1;
824
825 params.bcast_block_sizes[copy_dim] = params.input_dims[dim];
826 params.bcast_block_sizes[broadcast_dim] = m_broadcast[dim];
827 params.bcast_block_strides[copy_dim] = params.output_strides[dim];
828 params.bcast_block_strides[broadcast_dim] =
829 params.output_strides[dim] * params.input_dims[dim];
830 params.bcast_input_strides[copy_dim] = params.input_block_strides[dim];
831 params.bcast_input_strides[broadcast_dim] = 0;
832 }
833
834 for (int i = 2 * params.inner_dim_count; i < 2 * NumDims; ++i) {
835 const int dim = IsColMajor ? i : 2 * NumDims - i - 1;
836 params.bcast_block_sizes[dim] = 1;
837 params.bcast_block_strides[dim] = 0;
838 params.bcast_input_strides[dim] = 0;
839 }
840
841 return params;
842 }

850 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index BroadcastBlockAlongBcastDim(
851 BlockBroadcastingParams params, Index bcast_offset,
852 TensorBlockScratch& scratch, ScalarNoConst* materialized_output,
853 ScalarNoConst** materialized_input,
854 size_t* materialized_input_size) const {
855 if (params.bcast_dim_size == 1) {
856 // We just need one block read using the ready-set values above.
857 return BroadcastBlock(
858 params.input_block_sizes, params.input_block_strides,
859 params.bcast_block_sizes, params.bcast_block_strides,
860 params.bcast_input_strides, bcast_offset, 0, scratch,
861 materialized_output, materialized_input, materialized_input_size);
862
863 } else if (params.input_dims[params.bcast_dim] == 1) {
864 // Broadcast bcast dimension (< NumDims) by bcast_dim_size.
865 const int broadcast_bcast_dim =
866 IsColMajor ? 2 * params.inner_dim_count + 1
867 : 2 * NumDims - 2 * params.inner_dim_count - 2;
868
869 params.bcast_block_sizes[broadcast_bcast_dim] = params.bcast_dim_size;
870 params.bcast_input_strides[broadcast_bcast_dim] = 0;
871 params.bcast_block_strides[broadcast_bcast_dim] =
872 params.output_strides[params.bcast_dim];
873
874 return BroadcastBlock(
875 params.input_block_sizes, params.input_block_strides,
876 params.bcast_block_sizes, params.bcast_block_strides,
877 params.bcast_input_strides, bcast_offset, 0, scratch,
878 materialized_output, materialized_input, materialized_input_size);
879
880 } else {
881 // Keep track of the total number of the coefficients written to the
882 // output block.
883 Index num_output_coeffs = 0;
884
885 // The general case. Let's denote the output block as
886 //
887 // x[..., a:a+bcast_dim_size, :, ..., :]
888 //
889 // where a:a+bcast_dim_size is a slice on the bcast_dim dimension
890 // (< NumDims). We need to split the a:a+bcast_dim_size into possibly 3
891 // sub-blocks:
892 //
893 // (1) a:b, where b is the smallest multiple of
894 // input_dims[bcast_dim_start] in [a, a+bcast_dim_size].
895 //
896 // (2) b:c, where c is the largest multiple of input_dims[bcast_dim_start]
897 // in [a, a+bcast_dim_size].
898 //
899 // (3) c:a+bcast_dim_size .
900 //
901 // Or, when b and c do not exist, we just need to process the whole block
902 // together.
903
904 // Find a.
905 const Index bcast_dim_left_index =
906 bcast_offset / m_outputStrides[params.bcast_dim];
907
908 // Find b and c.
909 const Index input_bcast_dim_size = params.input_dims[params.bcast_dim];
910
911 // First multiple after a. This is b when <= bcast_dim_left_index +
912 // bcast_dim_size.
913 const Index first_multiple =
914 divup<Index>(bcast_dim_left_index, input_bcast_dim_size) *
915 input_bcast_dim_size;
916
917 if (first_multiple <= bcast_dim_left_index + params.bcast_dim_size) {
918 // b exists, so does c. Find it.
919 const Index last_multiple =
920 (bcast_dim_left_index + params.bcast_dim_size) /
921 input_bcast_dim_size * input_bcast_dim_size;
922 const int copy_bcast_dim =
923 IsColMajor ? 2 * params.inner_dim_count
924 : 2 * NumDims - 2 * params.inner_dim_count - 1;
925 const int broadcast_bcast_dim =
926 IsColMajor ? 2 * params.inner_dim_count + 1
927 : 2 * NumDims - 2 * params.inner_dim_count - 2;
928
929 if (first_multiple > bcast_dim_left_index) {
930 const Index head_size = first_multiple - bcast_dim_left_index;
931 params.input_block_sizes[params.bcast_dim] = head_size;
932 params.bcast_block_sizes[copy_bcast_dim] = head_size;
933 params.bcast_input_strides[copy_bcast_dim] =
934 params.input_block_strides[params.bcast_dim];
935 params.bcast_block_strides[copy_bcast_dim] =
936 params.output_strides[params.bcast_dim];
937 params.bcast_block_sizes[broadcast_bcast_dim] = 1;
938 params.bcast_input_strides[broadcast_bcast_dim] = 0;
939 params.bcast_block_strides[broadcast_bcast_dim] =
940 params.output_strides[params.bcast_dim] *
941 params.input_dims[params.bcast_dim];
942
943 num_output_coeffs += BroadcastBlock(
944 params.input_block_sizes, params.input_block_strides,
945 params.bcast_block_sizes, params.bcast_block_strides,
946 params.bcast_input_strides, bcast_offset, 0, scratch,
947 materialized_output, materialized_input, materialized_input_size);
948 }
949 if (first_multiple < last_multiple) {
950 params.input_block_sizes[params.bcast_dim] = input_bcast_dim_size;
951 params.bcast_block_sizes[copy_bcast_dim] = input_bcast_dim_size;
952 params.bcast_input_strides[copy_bcast_dim] =
953 params.input_block_strides[params.bcast_dim];
954 params.bcast_block_strides[copy_bcast_dim] =
955 params.output_strides[params.bcast_dim];
956 params.bcast_block_sizes[broadcast_bcast_dim] =
957 (last_multiple - first_multiple) / input_bcast_dim_size;
958 params.bcast_input_strides[broadcast_bcast_dim] = 0;
959 params.bcast_block_strides[broadcast_bcast_dim] =
960 params.output_strides[params.bcast_dim] *
961 params.input_dims[params.bcast_dim];
962 const Index offset = (first_multiple - bcast_dim_left_index) *
963 m_outputStrides[params.bcast_dim];
964
965 num_output_coeffs += BroadcastBlock(
966 params.input_block_sizes, params.input_block_strides,
967 params.bcast_block_sizes, params.bcast_block_strides,
968 params.bcast_input_strides, bcast_offset, offset, scratch,
969 materialized_output, materialized_input, materialized_input_size);
970 }
971 if (last_multiple < bcast_dim_left_index + params.bcast_dim_size) {
972 const Index tail_size =
973 bcast_dim_left_index + params.bcast_dim_size - last_multiple;
974 params.input_block_sizes[params.bcast_dim] = tail_size;
975 params.bcast_block_sizes[copy_bcast_dim] = tail_size;
976 params.bcast_input_strides[copy_bcast_dim] =
977 params.input_block_strides[params.bcast_dim];
978 params.bcast_block_strides[copy_bcast_dim] =
979 params.output_strides[params.bcast_dim];
980 params.bcast_block_sizes[broadcast_bcast_dim] = 1;
981 params.bcast_input_strides[broadcast_bcast_dim] = 0;
982 params.bcast_block_strides[broadcast_bcast_dim] =
983 params.output_strides[params.bcast_dim] *
984 params.input_dims[params.bcast_dim];
985 const Index offset = (last_multiple - bcast_dim_left_index) *
986 m_outputStrides[params.bcast_dim];
987
988 num_output_coeffs += BroadcastBlock(
989 params.input_block_sizes, params.input_block_strides,
990 params.bcast_block_sizes, params.bcast_block_strides,
991 params.bcast_input_strides, bcast_offset, offset, scratch,
992 materialized_output, materialized_input, materialized_input_size);
993 }
994 } else {
995 // b and c do not exist.
996 const int copy_bcast_dim =
997 IsColMajor ? 2 * params.inner_dim_count
998 : 2 * NumDims - 2 * params.inner_dim_count - 1;
999 params.input_block_sizes[params.bcast_dim] = params.bcast_dim_size;
1000 params.bcast_block_sizes[copy_bcast_dim] = params.bcast_dim_size;
1001 params.bcast_input_strides[copy_bcast_dim] =
1002 params.input_block_strides[params.bcast_dim];
1003 params.bcast_block_strides[copy_bcast_dim] =
1004 params.output_strides[params.bcast_dim];
1005
1006 num_output_coeffs += BroadcastBlock(
1007 params.input_block_sizes, params.input_block_strides,
1008 params.bcast_block_sizes, params.bcast_block_strides,
1009 params.bcast_input_strides, bcast_offset, 0, scratch,
1010 materialized_output, materialized_input, materialized_input_size);
1011 }
1012
1013 return num_output_coeffs;
1014 }
1015 }

1017 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index BroadcastBlock(
1018 const Dimensions& input_block_sizes,
1019 const Dimensions& input_block_strides,
1020 const BroadcastDimensions& bcast_block_sizes,
1021 const BroadcastDimensions& bcast_block_strides,
1022 const BroadcastDimensions& bcast_input_strides, Index bcast_offset,
1023 Index offset, TensorBlockScratch& scratch,
1024 ScalarNoConst* materialized_output, ScalarNoConst** materialized_input,
1025 size_t* materialized_input_size) const {
1026 // ---------------------------------------------------------------------- //
1027 // Tensor block descriptor for reading block from the input.
1028 const Index input_offset = bcast_offset + offset;
1029 TensorBlockDesc input_desc(
1030 IsColMajor ? indexColMajor(input_offset) : indexRowMajor(input_offset),
1031 input_block_sizes);
1032
1033 ArgTensorBlock input_block = m_impl.block(input_desc, scratch);
1034
1035 // ---------------------------------------------------------------------- //
1036 // Materialize input block into a temporary memory buffer only if it's not
1037 // already available in the arg block.
1038 const ScalarNoConst* input_buffer = NULL;
1039
1040 if (input_block.data() != NULL) {
1041 // Input block already has raw data, there is no need to materialize it.
1042 input_buffer = input_block.data();
1043
1044 } else {
1045 // Otherwise we have to do block assignment into a temporary buffer.
1046
1047 // Maybe reuse previously allocated buffer, or allocate a new one with a
1048 // scratch allocator.
1049 const size_t input_total_size = input_block_sizes.TotalSize();
1050 if (*materialized_input == NULL ||
1051 *materialized_input_size < input_total_size) {
1052 *materialized_input_size = input_total_size;
1053 void* mem = scratch.allocate(*materialized_input_size * sizeof(Scalar));
1054 *materialized_input = static_cast<ScalarNoConst*>(mem);
1055 }

1057 typedef internal::TensorBlockAssignment<
1058 ScalarNoConst, NumDims, typename ArgTensorBlock::XprType, Index>
1059 TensorBlockAssignment;
1060
1061 TensorBlockAssignment::Run(
1062 TensorBlockAssignment::target(input_block_sizes, input_block_strides,
1063 *materialized_input),
1064 input_block.expr());
1065
1066 input_buffer = *materialized_input;
1067 }
1068
1069 // ---------------------------------------------------------------------- //
1070 // Copy data from materialized input block to the materialized output, using
1071 // given broadcast strides (strides with zeroes).
1072 typedef internal::TensorBlockIO<ScalarNoConst, Index, 2 * NumDims, Layout>
1073 TensorBlockIO;
1074
1075 typename TensorBlockIO::Src src(bcast_input_strides, input_buffer);
1076 typename TensorBlockIO::Dst dst(bcast_block_sizes, bcast_block_strides,
1077 materialized_output + offset);
1078
1079 return TensorBlockIO::Copy(dst, src);
