99template<typename NewDimensions, typename ArgType, typename Device>
100struct TensorEvaluator<const TensorReshapingOp<NewDimensions, ArgType>, Device>
101{
102 typedef TensorReshapingOp<NewDimensions, ArgType> XprType;
103 typedef NewDimensions Dimensions;
104
105 typedef typename XprType::Index Index;
106 typedef typename XprType::Scalar Scalar;
107 typedef typename XprType::CoeffReturnType CoeffReturnType;
108 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
109 typedef StorageMemory<CoeffReturnType, Device> Storage;
110 typedef typename Storage::Type EvaluatorPointerType;
111 typedef StorageMemory<typename internal::remove_const<CoeffReturnType>::type, Device> ConstCastStorage;
112
113 static const int NumOutputDims = internal::array_size<Dimensions>::value;
114 static const int NumInputDims = internal::array_size<typename TensorEvaluator<ArgType, Device>::Dimensions>::value;
115
116 enum ReshapingKind {
117 // We do not use layout information to determine reshaping kind.
118 // Depending on the layout `N` can be inner or outer dimension.
119 OneByN = 0, // expr.reshape(1, N)
120 NByOne = 1, // expr.reshape(N, 1)
121 Runtime = 2 // Reshape dimensions are dynamic (specified at runtime).
122 };
123
124 // clang-format off
125 static const ReshapingKind kind =
126#if defined(EIGEN_HAS_INDEX_LIST)
127 (NumOutputDims == 2 && internal::index_statically_eq<NewDimensions>(/*index=*/0, /*value=*/1)) ? OneByN
128 : (NumOutputDims == 2 && internal::index_statically_eq<NewDimensions>(/*index=*/1, /*value=*/1)) ? NByOne
129 : Runtime;
130#else
131 Runtime;
132#endif
133 // clang-format on
134
135 enum {
136 IsAligned = TensorEvaluator<ArgType, Device>::IsAligned,
137 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
138 // For trivial reshapes with raw access to underlying data we will provide
139 // zero overhead block access.
140 // TODO(ezhulenev): Consider adding block access without raw access?
141 BlockAccess = TensorEvaluator<ArgType, Device>::RawAccess &&
142 NumInputDims > 0 && NumOutputDims > 0,
143 PreferBlockAccess = false,
144 Layout = TensorEvaluator<ArgType, Device>::Layout,
145 CoordAccess = false, // to be implemented
146 RawAccess = TensorEvaluator<ArgType, Device>::RawAccess
147 };
148
149 typedef typename internal::remove_const<Scalar>::type ScalarNoConst;
150
151 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
152 typedef internal::TensorBlockDescriptor<NumOutputDims, Index> TensorBlockDesc;
153 typedef internal::TensorBlockScratchAllocator<Device> TensorBlockScratch;
154
155 typedef
156 typename internal::TensorMaterializedBlock<ScalarNoConst, NumOutputDims,
157 Layout, Index>
158 TensorBlock;
159 //===--------------------------------------------------------------------===//
160
161 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
162 : m_impl(op.expression(), device), m_dimensions(op.dimensions())
163 {
164 // The total size of the reshaped tensor must be equal to the total size
165 // of the input tensor.
166 eigen_assert(internal::array_prod(m_impl.dimensions()) == internal::array_prod(op.dimensions()));
167 }

183 m_impl.cleanup();

188 return m_impl.coeff(index);

206 // required in block(OutputTensorBlock* output_block) const
207 // For C++03 compatibility this must be defined outside the method
208 struct BlockIteratorState {
209 Index stride;
210 Index span;
211 Index size;
212 Index count;
213 };
214
215 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorBlock
216 block(TensorBlockDesc& desc, TensorBlockScratch& scratch,
217 bool /*root_of_expr_ast*/ = false) const {
218 eigen_assert(m_impl.data() != NULL);
219 eigen_assert((kind == Runtime) ||
220 (kind == OneByN && desc.dimensions()[0] == 1) ||
221 (kind == NByOne && desc.dimensions()[1] == 1));
222
223 if (kind == OneByN || kind == NByOne) {
224 // We can guarantee at compile time that block is just a contiguous slice
225 // of the underlying expression memory buffer.
226 return TensorBlock(internal::TensorBlockKind::kView,
227 m_impl.data() + desc.offset(), desc.dimensions());
228 } else {
229 // This will do additional runtime checks, and in the end it might be also
230 // a view, or it might be a block materialized in the temporary buffer.
231 return TensorBlock::materialize(m_impl.data(), m_dimensions, desc,
232 scratch);
233 }
234 }
235
236 EIGEN_DEVICE_FUNC typename Storage::Type data() const {
237 return constCast(m_impl.data());
238 }
239
240 EIGEN_DEVICE_FUNC const TensorEvaluator<ArgType, Device>& impl() const { return m_impl; }
241
242 #ifdef EIGEN_USE_SYCL
243 // binding placeholder accessors to a command group handler for SYCL
244 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void bind(cl::sycl::handler &cgh) const {
245 m_impl.bind(cgh);
246 }
247 #endif
248 protected:
249 TensorEvaluator<ArgType, Device> m_impl;
250 NewDimensions m_dimensions;
251};

439template<typename StartIndices, typename Sizes, typename ArgType, typename Device>
440struct TensorEvaluator<const TensorSlicingOp<StartIndices, Sizes, ArgType>, Device>
441{
442 typedef TensorSlicingOp<StartIndices, Sizes, ArgType> XprType;
443 static const int NumDims = internal::array_size<Sizes>::value;
444
445 typedef typename XprType::Index Index;
446 typedef typename XprType::Scalar Scalar;
447 typedef typename XprType::CoeffReturnType CoeffReturnType;
448 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
449 typedef Sizes Dimensions;
450 typedef StorageMemory<CoeffReturnType, Device> Storage;
451 typedef StorageMemory<typename internal::remove_const<CoeffReturnType>::type, Device> ConstCastStorage;
452 typedef typename Storage::Type EvaluatorPointerType;
453
454 enum {
455 // Alignment can't be guaranteed at compile time since it depends on the
456 // slice offsets and sizes.
457 IsAligned = false,
458 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
459 BlockAccess = TensorEvaluator<ArgType, Device>::BlockAccess &&
460 // FIXME: Temporary workaround for bug in slicing of bool tensors.
461 !internal::is_same<typename internal::remove_const<Scalar>::type, bool>::value,
462 PreferBlockAccess = true,
463 Layout = TensorEvaluator<ArgType, Device>::Layout,
464 CoordAccess = false,
465 RawAccess = false
466 };
467
468 typedef typename internal::remove_const<Scalar>::type ScalarNoConst;
469
470 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
471 typedef internal::TensorBlockDescriptor<NumDims, Index> TensorBlockDesc;
472 typedef internal::TensorBlockScratchAllocator<Device> TensorBlockScratch;
473
474 // Tensor slicing does not change the block type.
475 typedef typename TensorEvaluator<const ArgType, Device>::TensorBlock
476 TensorBlock;
477 //===--------------------------------------------------------------------===//
478
479 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
480 : m_impl(op.expression(), device), m_device(device), m_dimensions(op.sizes()), m_offsets(op.startIndices())
481 {
482 for (Index i = 0; i < internal::array_size<Dimensions>::value; ++i) {
483 eigen_assert(m_impl.dimensions()[i] >= op.sizes()[i] + op.startIndices()[i]);
484 }
485
486 m_is_identity = true;
487 for (int i = 0; i < internal::array_size<Dimensions>::value; ++i) {
488 eigen_assert(m_impl.dimensions()[i] >=
489 op.sizes()[i] + op.startIndices()[i]);
490 if (m_impl.dimensions()[i] != op.sizes()[i] ||
491 op.startIndices()[i] != 0) {
492 m_is_identity = false;
493 }
494 }
495
496 // No strides for scalars.
497 if (NumDims == 0) return;
498
499 const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
500 const Sizes& output_dims = op.sizes();
501 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
502 m_inputStrides[0] = 1;
503 for (int i = 1; i < NumDims; ++i) {
504 m_inputStrides[i] = m_inputStrides[i-1] * input_dims[i-1];
505 }
506
507 // Don't initialize m_fastOutputStrides[0] since it won't ever be accessed.
508 m_outputStrides[0] = 1;
509 for (int i = 1; i < NumDims; ++i) {
510 m_outputStrides[i] = m_outputStrides[i-1] * output_dims[i-1];
511 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
512 }
513 } else {
514 m_inputStrides[NumDims-1] = 1;
515 for (int i = NumDims - 2; i >= 0; --i) {
516 m_inputStrides[i] = m_inputStrides[i+1] * input_dims[i+1];
517 }
518
519 // Don't initialize m_fastOutputStrides[NumDims-1] since it won't ever be accessed.
520 m_outputStrides[NumDims-1] = 1;
521 for (int i = NumDims - 2; i >= 0; --i) {
522 m_outputStrides[i] = m_outputStrides[i+1] * output_dims[i+1];
523 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
524 }
525 }
526 }

528 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Dimensions& dimensions() const { return m_dimensions; }
529
530 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE bool evalSubExprsIfNeeded(EvaluatorPointerType data) {
531 m_impl.evalSubExprsIfNeeded(NULL);
532 if (!NumTraits<typename internal::remove_const<Scalar>::type>::RequireInitialization
533 && data && m_impl.data()) {
534 Index contiguous_values = 1;
535 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
536 for (int i = 0; i < NumDims; ++i) {
537 contiguous_values *= dimensions()[i];
538 if (dimensions()[i] != m_impl.dimensions()[i]) {
539 break;
540 }
541 }
542 } else {
543 for (int i = NumDims-1; i >= 0; --i) {
544 contiguous_values *= dimensions()[i];
545 if (dimensions()[i] != m_impl.dimensions()[i]) {
546 break;
547 }
548 }
549 }
550 // Use memcpy if it's going to be faster than using the regular evaluation.
551 const MemcpyTriggerForSlicing<Index, Device, BlockAccess> trigger(m_device);
552 if (trigger(internal::array_prod(dimensions()), contiguous_values)) {
553 EvaluatorPointerType src = (EvaluatorPointerType)m_impl.data();
554 for (Index i = 0; i < internal::array_prod(dimensions()); i += contiguous_values) {
555 Index offset = srcCoeff(i);
556 m_device.memcpy((void*)(m_device.get(data + i)), m_device.get(src+offset), contiguous_values * sizeof(Scalar));
557 }
558 return false;
559 }
560 }
561 return true;
562 }

585 template<int LoadMode>
586 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
587 {
588 const int packetSize = PacketType<CoeffReturnType, Device>::size;
589 EIGEN_STATIC_ASSERT((packetSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
590 eigen_assert(index+packetSize-1 < internal::array_prod(dimensions()));
591
592 if (m_is_identity) {
593 return m_impl.template packet<LoadMode>(index);
594 }
595
596 Index inputIndices[] = {0, 0};
597 Index indices[] = {index, index + packetSize - 1};
598 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
599 EIGEN_UNROLL_LOOP
600 for (int i = NumDims - 1; i > 0; --i) {
601 const Index idx0 = indices[0] / m_fastOutputStrides[i];
602 const Index idx1 = indices[1] / m_fastOutputStrides[i];
603 inputIndices[0] += (idx0 + m_offsets[i]) * m_inputStrides[i];
604 inputIndices[1] += (idx1 + m_offsets[i]) * m_inputStrides[i];
605 indices[0] -= idx0 * m_outputStrides[i];
606 indices[1] -= idx1 * m_outputStrides[i];
607 }
608 inputIndices[0] += (indices[0] + m_offsets[0]);
609 inputIndices[1] += (indices[1] + m_offsets[0]);
610 } else {
611 EIGEN_UNROLL_LOOP
612 for (int i = 0; i < NumDims - 1; ++i) {
613 const Index idx0 = indices[0] / m_fastOutputStrides[i];
614 const Index idx1 = indices[1] / m_fastOutputStrides[i];
615 inputIndices[0] += (idx0 + m_offsets[i]) * m_inputStrides[i];
616 inputIndices[1] += (idx1 + m_offsets[i]) * m_inputStrides[i];
617 indices[0] -= idx0 * m_outputStrides[i];
618 indices[1] -= idx1 * m_outputStrides[i];
619 }

621 inputIndices[1] += (indices[1] + m_offsets[NumDims-1]);
622 }
623 if (inputIndices[1] - inputIndices[0] == packetSize - 1) {
624 PacketReturnType rslt = m_impl.template packet<Unaligned>(inputIndices[0]);
625 return rslt;
626 }
627 else {
628 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[packetSize];
629 values[0] = m_impl.coeff(inputIndices[0]);
630 values[packetSize-1] = m_impl.coeff(inputIndices[1]);
631 EIGEN_UNROLL_LOOP
632 for (int i = 1; i < packetSize-1; ++i) {
633 values[i] = coeff(index+i);
634 }
635 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
636 return rslt;
637 }

661 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE typename Storage::Type data() const {
662 typename Storage::Type result = constCast(m_impl.data());
663 if (result) {
664 Index offset = 0;
665 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
666 for (int i = 0; i < NumDims; ++i) {
667 if (m_dimensions[i] != m_impl.dimensions()[i]) {
668 offset += m_offsets[i] * m_inputStrides[i];
669 for (int j = i+1; j < NumDims; ++j) {
670 if (m_dimensions[j] > 1) {

673 offset += m_offsets[j] * m_inputStrides[j];
674 }
675 break;
676 }

680 if (m_dimensions[i] != m_impl.dimensions()[i]) {

686 offset += m_offsets[j] * m_inputStrides[j];

708 EIGEN_UNROLL_LOOP
709 for (int i = NumDims - 1; i > 0; --i) {
710 const Index idx = index / m_fastOutputStrides[i];
711 inputIndex += (idx + m_offsets[i]) * m_inputStrides[i];
712 index -= idx * m_outputStrides[i];
713 }
714 inputIndex += (index + m_offsets[0]);
715 } else {
716 EIGEN_UNROLL_LOOP
717 for (int i = 0; i < NumDims - 1; ++i) {
718 const Index idx = index / m_fastOutputStrides[i];
719 inputIndex += (idx + m_offsets[i]) * m_inputStrides[i];
720 index -= idx * m_outputStrides[i];
721 }
722 inputIndex += (index + m_offsets[NumDims-1]);
723 }

783 template <int StoreMode> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
784 void writePacket(Index index, const PacketReturnType& x)
785 {
786 if (this->m_is_identity) {
787 this->m_impl.template writePacket<StoreMode>(index, x);
788 return;
789 }
790
791 const int packetSize = PacketType<CoeffReturnType, Device>::size;
792 Index inputIndices[] = {0, 0};
793 Index indices[] = {index, index + packetSize - 1};
794 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
795 EIGEN_UNROLL_LOOP
796 for (int i = NumDims - 1; i > 0; --i) {
797 const Index idx0 = indices[0] / this->m_fastOutputStrides[i];
798 const Index idx1 = indices[1] / this->m_fastOutputStrides[i];
799 inputIndices[0] += (idx0 + this->m_offsets[i]) * this->m_inputStrides[i];
800 inputIndices[1] += (idx1 + this->m_offsets[i]) * this->m_inputStrides[i];
801 indices[0] -= idx0 * this->m_outputStrides[i];
802 indices[1] -= idx1 * this->m_outputStrides[i];
803 }
804 inputIndices[0] += (indices[0] + this->m_offsets[0]);
805 inputIndices[1] += (indices[1] + this->m_offsets[0]);
806 } else {
807 EIGEN_UNROLL_LOOP
808 for (int i = 0; i < NumDims - 1; ++i) {
809 const Index idx0 = indices[0] / this->m_fastOutputStrides[i];
810 const Index idx1 = indices[1] / this->m_fastOutputStrides[i];
811 inputIndices[0] += (idx0 + this->m_offsets[i]) * this->m_inputStrides[i];
812 inputIndices[1] += (idx1 + this->m_offsets[i]) * this->m_inputStrides[i];
813 indices[0] -= idx0 * this->m_outputStrides[i];
814 indices[1] -= idx1 * this->m_outputStrides[i];
815 }
816 inputIndices[0] += (indices[0] + this->m_offsets[NumDims-1]);
817 inputIndices[1] += (indices[1] + this->m_offsets[NumDims-1]);
818 }
819 if (inputIndices[1] - inputIndices[0] == packetSize - 1) {
820 this->m_impl.template writePacket<StoreMode>(inputIndices[0], x);
821 }
822 else {
823 EIGEN_ALIGN_MAX CoeffReturnType values[packetSize];
824 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
825 this->m_impl.coeffRef(inputIndices[0]) = values[0];
826 this->m_impl.coeffRef(inputIndices[1]) = values[packetSize-1];
827 EIGEN_UNROLL_LOOP
828 for (int i = 1; i < packetSize-1; ++i) {
829 this->coeffRef(index+i) = values[i];
830 }
831 }
832 }

941 enum {
942 // Alignment can't be guaranteed at compile time since it depends on the
943 // slice offsets and sizes.
944 IsAligned = false,
945 PacketAccess = false,
946 BlockAccess = false,

956 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
957 : m_impl(op.expression(), device),
958 m_device(device),
959 m_strides(op.strides())
960 {
961 // Handle degenerate intervals by gracefully clamping and allowing m_dimensions to be zero
962 DSizes<Index, NumDims> startIndicesClamped, stopIndicesClamped;
963 for (ptrdiff_t i = 0; i < internal::array_size<Dimensions>::value; ++i) {
964 eigen_assert(m_strides[i] != 0 && "0 stride is invalid");
965 if (m_strides[i] > 0) {
966 startIndicesClamped[i] =
967 clamp(op.startIndices()[i], 0, m_impl.dimensions()[i]);
968 stopIndicesClamped[i] =
969 clamp(op.stopIndices()[i], 0, m_impl.dimensions()[i]);
970 } else {
971 /* implies m_strides[i] < 0 by assert */
972 startIndicesClamped[i] =
973 clamp(op.startIndices()[i], -1, m_impl.dimensions()[i] - 1);
974 stopIndicesClamped[i] =
975 clamp(op.stopIndices()[i], -1, m_impl.dimensions()[i] - 1);
976 }
977 m_startIndices[i] = startIndicesClamped[i];
978 }
979
980 typedef typename TensorEvaluator<ArgType, Device>::Dimensions InputDimensions;
981 const InputDimensions& input_dims = m_impl.dimensions();
982
983 // check for degenerate intervals and compute output tensor shape
984 bool degenerate = false;
985 m_is_identity = true;
986 for (int i = 0; i < NumDims; i++) {
987 Index interval = stopIndicesClamped[i] - startIndicesClamped[i];
988 if (interval == 0 || ((interval < 0) != (m_strides[i] < 0))) {
989 m_dimensions[i] = 0;
990 degenerate = true;
991 } else {
992 m_dimensions[i] =
993 (interval / m_strides[i]) + (interval % m_strides[i] != 0 ? 1 : 0);
994 eigen_assert(m_dimensions[i] >= 0);
995 }
996 if (m_strides[i] != 1 || interval != m_impl.dimensions()[i]) {
997 m_is_identity = false;
998 }
999 }
1000
1001 Strides output_dims = m_dimensions;
1002
1003 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
1004 m_inputStrides[0] = m_strides[0];
1005 m_offsets[0] = startIndicesClamped[0];
1006 Index previousDimProduct = 1;
1007 for (int i = 1; i < NumDims; ++i) {
1008 previousDimProduct *= input_dims[i-1];
1009 m_inputStrides[i] = previousDimProduct * m_strides[i];
1010 m_offsets[i] = startIndicesClamped[i] * previousDimProduct;
1011 }
1012
1013 // Don't initialize m_fastOutputStrides[0] since it won't ever be accessed.
1014 m_outputStrides[0] = 1;
1015 for (int i = 1; i < NumDims; ++i) {
1016 m_outputStrides[i] = m_outputStrides[i-1] * output_dims[i-1];
1017 // NOTE: if tensor is degenerate, we send 1 to prevent TensorIntDivisor constructor crash
1018 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(degenerate ? 1 : m_outputStrides[i]);
1019 }
1020 } else {
1021 m_inputStrides[NumDims-1] = m_strides[NumDims-1];
1022 m_offsets[NumDims-1] = startIndicesClamped[NumDims-1];
1023 Index previousDimProduct = 1;
1024 for (int i = NumDims - 2; i >= 0; --i) {
1025 previousDimProduct *= input_dims[i+1];
1026 m_inputStrides[i] = previousDimProduct * m_strides[i];
1027 m_offsets[i] = startIndicesClamped[i] * previousDimProduct;
1028 }
1029
1030 m_outputStrides[NumDims-1] = 1;
1031 for (int i = NumDims - 2; i >= 0; --i) {
1032 m_outputStrides[i] = m_outputStrides[i+1] * output_dims[i+1];
1033 // NOTE: if tensor is degenerate, we send 1 to prevent TensorIntDivisor constructor crash
1034 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(degenerate ? 1 : m_outputStrides[i]);
1035 }
1036 }
1037 }
