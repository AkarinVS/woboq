15/** \class TensorLayoutSwap
16 * \ingroup CXX11_Tensor_Module
17 *
18 * \brief Swap the layout from col-major to row-major, or row-major
19 * to col-major, and invert the order of the dimensions.
20 *
21 * Beware: the dimensions are reversed by this operation. If you want to
22 * preserve the ordering of the dimensions, you need to combine this
23 * operation with a shuffle.
24 *
25 * \example:
26 * Tensor<float, 2, ColMajor> input(2, 4);
27 * Tensor<float, 2, RowMajor> output = input.swap_layout();
28 * eigen_assert(output.dimension(0) == 4);
29 * eigen_assert(output.dimension(1) == 2);
30 *
31 * array<int, 2> shuffle(1, 0);
32 * output = input.swap_layout().shuffle(shuffle);
33 * eigen_assert(output.dimension(0) == 2);
34 * eigen_assert(output.dimension(1) == 4);
35 *
36 */
37namespace internal {
38template<typename XprType>
39struct traits<TensorLayoutSwapOp<XprType> > : public traits<XprType>
40{
41 typedef typename XprType::Scalar Scalar;
42 typedef traits<XprType> XprTraits;
43 typedef typename XprTraits::StorageKind StorageKind;
44 typedef typename XprTraits::Index Index;
45 typedef typename XprType::Nested Nested;
46 typedef typename remove_reference<Nested>::type _Nested;
47 static const int NumDimensions = traits<XprType>::NumDimensions;
48 static const int Layout = (traits<XprType>::Layout == ColMajor) ? RowMajor : ColMajor;
49 typedef typename XprTraits::PointerType PointerType;
50};
51
52template<typename XprType>
53struct eval<TensorLayoutSwapOp<XprType>, Eigen::Dense>
54{
55 typedef const TensorLayoutSwapOp<XprType>& type;
56};

68template<typename XprType>
69class TensorLayoutSwapOp : public TensorBase<TensorLayoutSwapOp<XprType>, WriteAccessors>
70{
71 public:
72 typedef typename Eigen::internal::traits<TensorLayoutSwapOp>::Scalar Scalar;
73 typedef typename Eigen::NumTraits<Scalar>::Real RealScalar;
74 typedef typename internal::remove_const<typename XprType::CoeffReturnType>::type CoeffReturnType;
75 typedef typename Eigen::internal::nested<TensorLayoutSwapOp>::type Nested;
76 typedef typename Eigen::internal::traits<TensorLayoutSwapOp>::StorageKind StorageKind;
77 typedef typename Eigen::internal::traits<TensorLayoutSwapOp>::Index Index;
78
79 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorLayoutSwapOp(const XprType& expr)
80 : m_xpr(expr) {}
81
82 EIGEN_DEVICE_FUNC
83 const typename internal::remove_all<typename XprType::Nested>::type&
84 expression() const { return m_xpr; }
85
86 EIGEN_DEVICE_FUNC
87 EIGEN_STRONG_INLINE TensorLayoutSwapOp& operator = (const TensorLayoutSwapOp& other)
88 {
89 typedef TensorAssignOp<TensorLayoutSwapOp, const TensorLayoutSwapOp> Assign;
90 Assign assign(*this, other);
91 internal::TensorExecutor<const Assign, DefaultDevice>::run(assign, DefaultDevice());
92 return *this;
93 }
94
95 template<typename OtherDerived>
96 EIGEN_DEVICE_FUNC
97 EIGEN_STRONG_INLINE TensorLayoutSwapOp& operator = (const OtherDerived& other)
98 {
99 typedef TensorAssignOp<TensorLayoutSwapOp, const OtherDerived> Assign;
100 Assign assign(*this, other);
101 internal::TensorExecutor<const Assign, DefaultDevice>::run(assign, DefaultDevice());
102 return *this;
103 }
104
105 protected:
106 typename XprType::Nested m_xpr;
107};

110// Eval as rvalue
111template<typename ArgType, typename Device>
112struct TensorEvaluator<const TensorLayoutSwapOp<ArgType>, Device>
113{
114 typedef TensorLayoutSwapOp<ArgType> XprType;
115 typedef typename XprType::Index Index;
116 static const int NumDims = internal::array_size<typename TensorEvaluator<ArgType, Device>::Dimensions>::value;
117 typedef DSizes<Index, NumDims> Dimensions;
118
119 enum {
120 IsAligned = TensorEvaluator<ArgType, Device>::IsAligned,
121 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
122 BlockAccess = false,
123 PreferBlockAccess = TensorEvaluator<ArgType, Device>::PreferBlockAccess,
124 Layout = (static_cast<int>(TensorEvaluator<ArgType, Device>::Layout) == static_cast<int>(ColMajor)) ? RowMajor : ColMajor,
125 CoordAccess = false, // to be implemented
126 RawAccess = TensorEvaluator<ArgType, Device>::RawAccess
127 };
128
129 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
130 typedef internal::TensorBlockNotImplemented TensorBlock;
131 //===--------------------------------------------------------------------===//
132
133 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
134 : m_impl(op.expression(), device)
135 {
136 for(int i = 0; i < NumDims; ++i) {
137 m_dimensions[i] = m_impl.dimensions()[NumDims-1-i];
138 }
139 }
140
141#ifdef EIGEN_USE_SYCL
142 // binding placeholder accessors to a command group handler for SYCL
143 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void bind(cl::sycl::handler &cgh) const {
144 m_impl.bind(cgh);
145 }
146#endif
147
148 typedef typename XprType::Scalar Scalar;
149 typedef typename XprType::CoeffReturnType CoeffReturnType;
150 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
151 typedef StorageMemory<CoeffReturnType, Device> Storage;
152 typedef typename Storage::Type EvaluatorPointerType;
153
154 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Dimensions& dimensions() const { return m_dimensions; }
155
156 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE bool evalSubExprsIfNeeded(EvaluatorPointerType data) {
157 return m_impl.evalSubExprsIfNeeded(data);
158 }
159 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void cleanup() {
160 m_impl.cleanup();
161 }
162
163 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE CoeffReturnType coeff(Index index) const
164 {
165 return m_impl.coeff(index);
166 }
167
168 template<int LoadMode>
169 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
170 {
171 return m_impl.template packet<LoadMode>(index);
172 }
173
174 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost costPerCoeff(bool vectorized) const {
175 return m_impl.costPerCoeff(vectorized);
176 }
177
178 EIGEN_DEVICE_FUNC typename Storage::Type data() const {
179 return constCast(m_impl.data());
180 }
181
182 const TensorEvaluator<ArgType, Device>& impl() const { return m_impl; }
183
184 protected:
185 TensorEvaluator<ArgType, Device> m_impl;
186 Dimensions m_dimensions;
187};

203 Layout = (static_cast<int>(TensorEvaluator<ArgType, Device>::Layout) == static_cast<int>(ColMajor)) ? RowMajor : ColMajor,
204 CoordAccess = false // to be implemented
205 };
206
207 //===- Tensor block evaluation strategy (see TensorBlock.h) -------------===//
208 typedef internal::TensorBlockNotImplemented TensorBlock;
209 //===--------------------------------------------------------------------===//

211 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)

218 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;

228 }
