19EIGEN_DEVICE_FUNC uint64_t get_random_seed() {
20#if defined(EIGEN_GPU_COMPILE_PHASE)
21 // We don't support 3d kernels since we currently only use 1 and
22 // 2d kernels.
23 gpu_assert(threadIdx.z == 0);
24 return clock64() +
25 blockIdx.x * blockDim.x + threadIdx.x +
26 gridDim.x * blockDim.x * (blockIdx.y * blockDim.y + threadIdx.y);
27
28#elif defined _WIN32
29 // Use the current time as a baseline.
30 SYSTEMTIME st;
31 GetSystemTime(&st);
32 int time = st.wSecond + 1000 * st.wMilliseconds;
33 // Mix in a random number to make sure that we get different seeds if
34 // we try to generate seeds faster than the clock resolution.
35 // We need 2 random values since the generator only generate 16 bits at
36 // a time (https://msdn.microsoft.com/en-us/library/398ax69y.aspx)
37 int rnd1 = ::rand();
38 int rnd2 = ::rand();
39 uint64_t rnd = (rnd1 | rnd2 << 16) ^ time;
40 return rnd;
41
42#elif defined __APPLE__
43 // Same approach as for win32, except that the random number generator
44 // is better (// https://developer.apple.com/legacy/library/documentation/Darwin/Reference/ManPages/man3/random.3.html#//apple_ref/doc/man/3/random).
45 uint64_t rnd = ::random() ^ mach_absolute_time();
46 return rnd;
47
48#elif defined __native_client__
49 // Same approach as for win32, except using clock_gettime
50 timespec ts;
51 clock_gettime(CLOCK_REALTIME, &ts);
52 int rnd1 = ::rand();
53 int rnd2 = ::rand();
54 uint64_t rnd = (rnd1 | rnd2 << 16) ^ ts.tv_nsec;
55 return rnd;
56
57#else
58 // Augment the current time with pseudo random number generation
59 // to ensure that we get different seeds if we try to generate seeds
60 // faster than the clock resolution.
61 timespec ts;
62 clock_gettime(CLOCK_REALTIME, &ts);
63 uint64_t rnd = ::random() ^ ts.tv_nsec;
64 return rnd;
65#endif
66}

68static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE unsigned PCG_XSH_RS_generator(uint64_t* state, uint64_t stream) {
69 // TODO: Unify with the implementation in the non blocking thread pool.
70 uint64_t current = *state;
71 // Update the internal state
72 *state = current * 6364136223846793005ULL + (stream << 1 | 1);
73 // Generate the random output (using the PCG-XSH-RS scheme)
74 return static_cast<unsigned>((current ^ (current >> 22)) >> (22 + (current >> 61)));
75}
76
77static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE uint64_t PCG_XSH_RS_state(uint64_t seed) {
78 seed = seed ? seed : get_random_seed();
79 return seed * 6364136223846793005ULL + 0xda3e39cb94b95bdbULL;
80}

85template <typename T> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
86T RandomToTypeUniform(uint64_t* state, uint64_t stream) {
87 unsigned rnd = PCG_XSH_RS_generator(state, stream);
88 return static_cast<T>(rnd);
89}

92template <> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
93Eigen::half RandomToTypeUniform<Eigen::half>(uint64_t* state, uint64_t stream) {
94 Eigen::half result;
95 // Generate 10 random bits for the mantissa
96 unsigned rnd = PCG_XSH_RS_generator(state, stream);
97 result.x = static_cast<uint16_t>(rnd & 0x3ffu);
98 // Set the exponent
99 result.x |= (static_cast<uint16_t>(15) << 10);
100 // Return the final result
101 return result - Eigen::half(1.0f);
102}

104template <> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
105Eigen::bfloat16 RandomToTypeUniform<Eigen::bfloat16>(uint64_t* state, uint64_t stream) {
106 Eigen::bfloat16 result;
107 // Generate 7 random bits for the mantissa
108 unsigned rnd = PCG_XSH_RS_generator(state, stream);
109 result.value = static_cast<uint16_t>(rnd & 0x7fu);
110 // Set the exponent
111 result.value |= (static_cast<uint16_t>(127) << 7);
112 // Return the final result
113 return result - Eigen::bfloat16(1.0f);
114}

116template <> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
117float RandomToTypeUniform<float>(uint64_t* state, uint64_t stream) {
118 typedef union {
119 uint32_t raw;
120 float fp;
121 } internal;
122 internal result;
123 // Generate 23 random bits for the mantissa mantissa
124 const unsigned rnd = PCG_XSH_RS_generator(state, stream);
125 result.raw = rnd & 0x7fffffu;
126 // Set the exponent
127 result.raw |= (static_cast<uint32_t>(127) << 23);
128 // Return the final result
129 return result.fp - 1.0f;
130}

132template <> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
133double RandomToTypeUniform<double>(uint64_t* state, uint64_t stream) {
134 typedef union {
135 uint64_t raw;
136 double dp;
137 } internal;
138 internal result;
139 result.raw = 0;
140 // Generate 52 random bits for the mantissa
141 // First generate the upper 20 bits
142 unsigned rnd1 = PCG_XSH_RS_generator(state, stream) & 0xfffffu;
143 // The generate the lower 32 bits
144 unsigned rnd2 = PCG_XSH_RS_generator(state, stream);
145 result.raw = (static_cast<uint64_t>(rnd1) << 32) | rnd2;
146 // Set the exponent
147 result.raw |= (static_cast<uint64_t>(1023) << 52);
148 // Return the final result
149 return result.dp - 1.0;
150}

163template <typename T> class UniformRandomGenerator {
164 public:
165 static const bool PacketAccess = true;
166
167 // Uses the given "seed" if non-zero, otherwise uses a random seed.
168 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE UniformRandomGenerator(
169 uint64_t seed = 0) {
170 m_state = PCG_XSH_RS_state(seed);
171 #ifdef EIGEN_USE_SYCL
172 // In SYCL it is not possible to build PCG_XSH_RS_state in one step.
173 // Therefor, we need two step to initializate the m_state.
174 // IN SYCL, the constructor of the functor is s called on the CPU
175 // and we get the clock seed here from the CPU. However, This seed is
176 //the same for all the thread. As unlike CUDA, the thread.ID, BlockID, etc is not a global function.
177 // and only available on the Operator() function (which is called on the GPU).
178 // Thus for CUDA (((CLOCK + global_thread_id)* 6364136223846793005ULL) + 0xda3e39cb94b95bdbULL) is passed to each thread
179 // but for SYCL ((CLOCK * 6364136223846793005ULL) + 0xda3e39cb94b95bdbULL) is passed to each thread and each thread adds
180 // the (global_thread_id* 6364136223846793005ULL) for itself only once, in order to complete the construction
181 // similar to CUDA Therefore, the thread Id injection is not available at this stage.
182 //However when the operator() is called the thread ID will be avilable. So inside the opeator,
183 // we add the thrreadID, BlockId,... (which is equivalent of i)
184 //to the seed and construct the unique m_state per thead similar to cuda.
185 m_exec_once =false;
186 #endif
187 }
188 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE UniformRandomGenerator(
189 const UniformRandomGenerator& other) {
190 m_state = other.m_state;
191 #ifdef EIGEN_USE_SYCL
192 m_exec_once =other.m_exec_once;
193 #endif
194 }
195
196 template<typename Index> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
197 T operator()(Index i) const {
198 #ifdef EIGEN_USE_SYCL
199 if(!m_exec_once) {
200 // This is the second stage of adding thread Id to the CPU clock seed and build unique seed per thread
201 // The (i * 6364136223846793005ULL) is the remaining part of the PCG_XSH_RS_state on the GPU side
202 m_state += (i * 6364136223846793005ULL);
203 m_exec_once =true;
204 }
205 #endif
206 T result = RandomToTypeUniform<T>(&m_state, i);
207 return result;
208 }
209
210 template<typename Packet, typename Index> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
211 Packet packetOp(Index i) const {
212 const int packetSize = internal::unpacket_traits<Packet>::size;
213 EIGEN_ALIGN_MAX T values[packetSize];
214 #ifdef EIGEN_USE_SYCL
215 if(!m_exec_once) {
216 // This is the second stage of adding thread Id to the CPU clock seed and build unique seed per thread
217 m_state += (i * 6364136223846793005ULL);
218 m_exec_once =true;
219 }
220 #endif
221 EIGEN_UNROLL_LOOP
222 for (int j = 0; j < packetSize; ++j) {
223 values[j] = RandomToTypeUniform<T>(&m_state, i);
224 }
225 return internal::pload<Packet>(values);
226 }
227
228 private:
229 mutable uint64_t m_state;
230 #ifdef EIGEN_USE_SYCL
231 mutable bool m_exec_once;
232 #endif
233};

235template <typename Scalar>
236struct functor_traits<UniformRandomGenerator<Scalar> > {
237 enum {
238 // Rough estimate for floating point, multiplied by ceil(sizeof(T) / sizeof(float)).
239 Cost = 12 * NumTraits<Scalar>::AddCost *
240 ((sizeof(Scalar) + sizeof(float) - 1) / sizeof(float)),
241 PacketAccess = UniformRandomGenerator<Scalar>::PacketAccess
242 };
243};

247template <typename T> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
248T RandomToTypeNormal(uint64_t* state, uint64_t stream) {
249 // Use the ratio of uniform method to generate numbers following a normal
250 // distribution. See for example Numerical Recipes chapter 7.3.9 for the
251 // details.
252 T u, v, q;
253 do {
254 u = RandomToTypeUniform<T>(state, stream);
255 v = T(1.7156) * (RandomToTypeUniform<T>(state, stream) - T(0.5));
256 const T x = u - T(0.449871);
257 const T y = numext::abs(v) + T(0.386595);
258 q = x*x + y * (T(0.196)*y - T(0.25472)*x);
259 } while (q > T(0.27597) &&
260 (q > T(0.27846) || v*v > T(-4) * numext::log(u) * u*u));
261
262 return v/u;
263}

345template <typename Scalar>
346struct functor_traits<NormalRandomGenerator<Scalar> > {
347 enum {
348 // On average, we need to generate about 3 random numbers
349 // 15 mul, 8 add, 1.5 logs
350 Cost = 3 * functor_traits<UniformRandomGenerator<Scalar> >::Cost +
351 15 * NumTraits<Scalar>::AddCost + 8 * NumTraits<Scalar>::AddCost +
352 3 * functor_traits<scalar_log_op<Scalar> >::Cost / 2,
353 PacketAccess = NormalRandomGenerator<Scalar>::PacketAccess
354 };
355};
