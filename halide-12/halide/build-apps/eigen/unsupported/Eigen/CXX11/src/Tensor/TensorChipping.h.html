130template<DenseIndex DimId, typename ArgType, typename Device>
131struct TensorEvaluator<const TensorChippingOp<DimId, ArgType>, Device>
132{
133 typedef TensorChippingOp<DimId, ArgType> XprType;
134 static const int NumInputDims = internal::array_size<typename TensorEvaluator<ArgType, Device>::Dimensions>::value;
135 static const int NumDims = NumInputDims-1;
136 typedef typename XprType::Index Index;
137 typedef DSizes<Index, NumDims> Dimensions;
138 typedef typename XprType::Scalar Scalar;
139 typedef typename XprType::CoeffReturnType CoeffReturnType;
140 typedef typename PacketType<CoeffReturnType, Device>::type PacketReturnType;
141 static const int PacketSize = PacketType<CoeffReturnType, Device>::size;
142 typedef StorageMemory<CoeffReturnType, Device> Storage;
143 typedef typename Storage::Type EvaluatorPointerType;
144
145 enum {
146 // Alignment can't be guaranteed at compile time since it depends on the
147 // slice offsets.
148 IsAligned = false,
149 Layout = TensorEvaluator<ArgType, Device>::Layout,
150 PacketAccess = TensorEvaluator<ArgType, Device>::PacketAccess,
151 BlockAccess = TensorEvaluator<ArgType, Device>::BlockAccess,
152 // Chipping of outer-most dimension is a trivial operation, because we can
153 // read and write directly from the underlying tensor using single offset.
154 IsOuterChipping = (static_cast<int>(Layout) == ColMajor && DimId == NumInputDims - 1) ||
155 (static_cast<int>(Layout) == RowMajor && DimId == 0),
156 // Chipping inner-most dimension.
157 IsInnerChipping = (static_cast<int>(Layout) == ColMajor && DimId == 0) ||
158 (static_cast<int>(Layout) == RowMajor && DimId == NumInputDims - 1),
159 // Prefer block access if the underlying expression prefers it, otherwise
160 // only if chipping is not trivial.
161 PreferBlockAccess = TensorEvaluator<ArgType, Device>::PreferBlockAccess ||
162 !IsOuterChipping,
163 CoordAccess = false, // to be implemented
164 RawAccess = false
165 };

167 typedef typename internal::remove_const<Scalar>::type ScalarNoConst;

177

183 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op, const Device& device)
184 : m_impl(op.expression(), device), m_dim(op.dim()), m_device(device)
185 {
186 EIGEN_STATIC_ASSERT((NumInputDims >= 1), YOU_MADE_A_PROGRAMMING_MISTAKE);
187 eigen_assert(NumInputDims > m_dim.actualDim());
188
189 const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
190 eigen_assert(op.offset() < input_dims[m_dim.actualDim()]);
191
192 int j = 0;
193 for (int i = 0; i < NumInputDims; ++i) {
194 if (i != m_dim.actualDim()) {
195 m_dimensions[j] = input_dims[i];
196 ++j;
197 }
198 }
199
200 m_stride = 1;
201 m_inputStride = 1;
202 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
203 for (int i = 0; i < m_dim.actualDim(); ++i) {
204 m_stride *= input_dims[i];
205 m_inputStride *= input_dims[i];
206 }
207 } else {
208 for (int i = NumInputDims-1; i > m_dim.actualDim(); --i) {
209 m_stride *= input_dims[i];
210 m_inputStride *= input_dims[i];
211 }
212 }
213 m_inputStride *= input_dims[m_dim.actualDim()];
214 m_inputOffset = m_stride * op.offset();
215 }

233 template<int LoadMode>
234 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
235 {
236 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
237 eigen_assert(index+PacketSize-1 < dimensions().TotalSize());
238
239 if (isInnerChipping()) {
240 // m_stride is equal to 1, so let's avoid the integer division.
241 eigen_assert(m_stride == 1);
242 Index inputIndex = index * m_inputStride + m_inputOffset;
243 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
244 EIGEN_UNROLL_LOOP
245 for (int i = 0; i < PacketSize; ++i) {
246 values[i] = m_impl.coeff(inputIndex);
247 inputIndex += m_inputStride;
248 }
249 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
250 return rslt;
251 } else if (isOuterChipping()) {
252 // m_stride is always greater than index, so let's avoid the integer division.
253 eigen_assert(m_stride > index);
254 return m_impl.template packet<LoadMode>(index + m_inputOffset);
255 } else {
256 const Index idx = index / m_stride;
257 const Index rem = index - idx * m_stride;
258 if (rem + PacketSize <= m_stride) {
259 Index inputIndex = idx * m_inputStride + m_inputOffset + rem;
260 return m_impl.template packet<LoadMode>(inputIndex);
261 } else {
262 // Cross the stride boundary. Fallback to slow path.
263 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
264 EIGEN_UNROLL_LOOP
265 for (int i = 0; i < PacketSize; ++i) {
266 values[i] = coeff(index);
267 ++index;
268 }
269 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
270 return rslt;
271 }
272 }
273 }

275 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost
276 costPerCoeff(bool vectorized) const {
277 double cost = 0;
278 if ((static_cast<int>(Layout) == static_cast<int>(ColMajor) &&
279 m_dim.actualDim() == 0) ||
280 (static_cast<int>(Layout) == static_cast<int>(RowMajor) &&
281 m_dim.actualDim() == NumInputDims - 1)) {
282 cost += TensorOpCost::MulCost<Index>() + TensorOpCost::AddCost<Index>();
283 } else if ((static_cast<int>(Layout) == static_cast<int>(ColMajor) &&
284 m_dim.actualDim() == NumInputDims - 1) ||
285 (static_cast<int>(Layout) == static_cast<int>(RowMajor) &&
286 m_dim.actualDim() == 0)) {
287 cost += TensorOpCost::AddCost<Index>();
288 } else {
289 cost += 3 * TensorOpCost::MulCost<Index>() + TensorOpCost::DivCost<Index>() +
290 3 * TensorOpCost::AddCost<Index>();
291 }
292
293 return m_impl.costPerCoeff(vectorized) +
294 TensorOpCost(0, 0, cost, vectorized, PacketSize);
295 }

305 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorBlock
306 block(TensorBlockDesc& desc, TensorBlockScratch& scratch,
307 bool root_of_expr_ast = false) const {
308 const Index chip_dim = m_dim.actualDim();
309
310 DSizes<Index, NumInputDims> input_block_dims;
311 for (int i = 0; i < NumInputDims; ++i) {
312 input_block_dims[i]
313 = i < chip_dim ? desc.dimension(i)
314 : i > chip_dim ? desc.dimension(i - 1)
315 : 1;
316 }
317
318 ArgTensorBlockDesc arg_desc(srcCoeff(desc.offset()), input_block_dims);
319
320 // Try to reuse destination buffer for materializing argument block.
321 if (desc.HasDestinationBuffer()) {
322 DSizes<Index, NumInputDims> arg_destination_strides;
323 for (int i = 0; i < NumInputDims; ++i) {
324 arg_destination_strides[i]
325 = i < chip_dim ? desc.destination().strides()[i]
326 : i > chip_dim ? desc.destination().strides()[i - 1]
327 : 0; // for dimensions of size `1` stride should never be used.
328 }
329
330 arg_desc.template AddDestinationBuffer<Layout>(
331 desc.destination().template data<ScalarNoConst>(),
332 arg_destination_strides);
333 }
334
335 ArgTensorBlock arg_block = m_impl.block(arg_desc, scratch, root_of_expr_ast);
336 if (!arg_desc.HasDestinationBuffer()) desc.DropDestinationBuffer();
337
338 if (arg_block.data() != NULL) {
339 // Forward argument block buffer if possible.
340 return TensorBlock(arg_block.kind(), arg_block.data(),
341 desc.dimensions());
342
343 } else {
344 // Assign argument block expression to a buffer.
345
346 // Prepare storage for the materialized chipping result.
347 const typename TensorBlock::Storage block_storage =
348 TensorBlock::prepareStorage(desc, scratch);
349
350 typedef internal::TensorBlockAssignment<
351 ScalarNoConst, NumInputDims, typename ArgTensorBlock::XprType, Index>
352 TensorBlockAssignment;
353
354 TensorBlockAssignment::Run(
355 TensorBlockAssignment::target(
356 arg_desc.dimensions(),
357 internal::strides<Layout>(arg_desc.dimensions()),
358 block_storage.data()),
359 arg_block.expr());
360
361 return block_storage.AsTensorMaterializedBlock();
362 }
363 }

365 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE typename Storage::Type data() const {
366 typename Storage::Type result = constCast(m_impl.data());
367 if (isOuterChipping() && result) {
368 return result + m_inputOffset;
369 } else {
370 return NULL;
371 }
372 }

374 // binding placeholder accessors to a command group handler for SYCL
375 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void bind(cl::sycl::handler &cgh) const {
376 m_impl.bind(cgh);
377 }
378#endif

380 protected:
381 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index srcCoeff(Index index) const
382 {
383 Index inputIndex;
384 if (isInnerChipping()) {
385 // m_stride is equal to 1, so let's avoid the integer division.
386 eigen_assert(m_stride == 1);
387 inputIndex = index * m_inputStride + m_inputOffset;
388 } else if (isOuterChipping()) {
389 // m_stride is always greater than index, so let's avoid the integer
390 // division.
391 eigen_assert(m_stride > index);
392 inputIndex = index + m_inputOffset;
393 } else {
394 const Index idx = index / m_stride;
395 inputIndex = idx * m_inputStride + m_inputOffset;
396 index -= idx * m_stride;
397 inputIndex += index;
398 }
399 return inputIndex;
400 }
401
402 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE bool isInnerChipping() const {
403 return IsInnerChipping ||
404 (static_cast<int>(Layout) == ColMajor && m_dim.actualDim() == 0) ||
405 (static_cast<int>(Layout) == RowMajor && m_dim.actualDim() == NumInputDims - 1);
406 }
407
408 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE bool isOuterChipping() const {
409 return IsOuterChipping ||
410 (static_cast<int>(Layout) == ColMajor && m_dim.actualDim() == NumInputDims-1) ||
411 (static_cast<int>(Layout) == RowMajor && m_dim.actualDim() == 0);
412 }

414 Dimensions m_dimensions;
415 Index m_stride;
416 Index m_inputOffset;
417 Index m_inputStride;
418 TensorEvaluator<ArgType, Device> m_impl;
419 const internal::DimensionId<DimId> m_dim;
420 const Device EIGEN_DEVICE_REF m_device;

461 template <int StoreMode> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
462 void writePacket(Index index, const PacketReturnType& x)
463 {
464 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
465
466 if (this->isInnerChipping()) {
467 // m_stride is equal to 1, so let's avoid the integer division.
468 eigen_assert(this->m_stride == 1);
469 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
470 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
471 Index inputIndex = index * this->m_inputStride + this->m_inputOffset;
472 EIGEN_UNROLL_LOOP
473 for (int i = 0; i < PacketSize; ++i) {
474 this->m_impl.coeffRef(inputIndex) = values[i];
475 inputIndex += this->m_inputStride;
476 }
477 } else if (this->isOuterChipping()) {
478 // m_stride is always greater than index, so let's avoid the integer division.
479 eigen_assert(this->m_stride > index);
480 this->m_impl.template writePacket<StoreMode>(index + this->m_inputOffset, x);
481 } else {
482 const Index idx = index / this->m_stride;
483 const Index rem = index - idx * this->m_stride;
484 if (rem + PacketSize <= this->m_stride) {
485 const Index inputIndex = idx * this->m_inputStride + this->m_inputOffset + rem;
486 this->m_impl.template writePacket<StoreMode>(inputIndex, x);
487 } else {
488 // Cross stride boundary. Fallback to slow path.
489 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
490 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
491 EIGEN_UNROLL_LOOP
492 for (int i = 0; i < PacketSize; ++i) {
493 this->coeffRef(index) = values[i];
494 ++index;
495 }
496 }
497 }
498 }

500 template <typename TensorBlock>
501 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void writeBlock(
502 const TensorBlockDesc& desc, const TensorBlock& block) {
503 assert(this->m_impl.data() != NULL);
504
505 const Index chip_dim = this->m_dim.actualDim();
506
507 DSizes<Index, NumInputDims> input_block_dims;
508 for (int i = 0; i < NumInputDims; ++i) {
509 input_block_dims[i] = i < chip_dim ? desc.dimension(i)
510 : i > chip_dim ? desc.dimension(i - 1)
511 : 1;
512 }
513
514 typedef TensorReshapingOp<const DSizes<Index, NumInputDims>,
515 const typename TensorBlock::XprType>
516 TensorBlockExpr;
517
518 typedef internal::TensorBlockAssignment<Scalar, NumInputDims,
519 TensorBlockExpr, Index>
520 TensorBlockAssign;
521
522 TensorBlockAssign::Run(
523 TensorBlockAssign::target(
524 input_block_dims,
525 internal::strides<Layout>(this->m_impl.dimensions()),
526 this->m_impl.data(), this->srcCoeff(desc.offset())),
527 block.expr().reshape(input_block_dims));
528 }
