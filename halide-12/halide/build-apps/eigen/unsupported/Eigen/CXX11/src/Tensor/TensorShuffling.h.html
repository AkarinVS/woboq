136 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorEvaluator(const XprType& op,
137 const Device& device)
138 : m_device(device),
139 m_impl(op.expression(), device)
140 {
141 const typename TensorEvaluator<ArgType, Device>::Dimensions& input_dims = m_impl.dimensions();
142 const Shuffle& shuffle = op.shufflePermutation();
143 m_is_identity = true;
144 for (int i = 0; i < NumDims; ++i) {
145 m_shuffle[i] = static_cast<int>(shuffle[i]);
146 m_dimensions[i] = input_dims[shuffle[i]];
147 m_inverseShuffle[shuffle[i]] = i;
148 if (m_is_identity && shuffle[i] != i) {
149 m_is_identity = false;
150 }
151 }
152
153 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
154 m_unshuffledInputStrides[0] = 1;
155 m_outputStrides[0] = 1;
156
157 for (int i = 1; i < NumDims; ++i) {
158 m_unshuffledInputStrides[i] =
159 m_unshuffledInputStrides[i - 1] * input_dims[i - 1];
160 m_outputStrides[i] = m_outputStrides[i - 1] * m_dimensions[i - 1];
161 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
162 }
163 } else {
164 m_unshuffledInputStrides[NumDims - 1] = 1;
165 m_outputStrides[NumDims - 1] = 1;
166 for (int i = NumDims - 2; i >= 0; --i) {
167 m_unshuffledInputStrides[i] =
168 m_unshuffledInputStrides[i + 1] * input_dims[i + 1];
169 m_outputStrides[i] = m_outputStrides[i + 1] * m_dimensions[i + 1];
170 m_fastOutputStrides[i] = internal::TensorIntDivisor<Index>(m_outputStrides[i]);
171 }
172 }
173
174 for (int i = 0; i < NumDims; ++i) {
175 m_inputStrides[i] = m_unshuffledInputStrides[shuffle[i]];
176 }
177 }

207 template <int LoadMode, typename Self, bool ImplPacketAccess>
208 struct PacketLoader {
209 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
210 static PacketReturnType Run(const Self& self, Index index) {
211 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
212 EIGEN_UNROLL_LOOP
213 for (int i = 0; i < PacketSize; ++i) {
214 values[i] = self.coeff(index + i);
215 }
216 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
217 return rslt;
218 }
219 };
220
221 template<int LoadMode, typename Self>
222 struct PacketLoader<LoadMode, Self, true> {
223 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
224 static PacketReturnType Run(const Self& self, Index index) {
225 if (self.m_is_identity) {
226 return self.m_impl.template packet<LoadMode>(index);
227 } else {
228 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
229 EIGEN_UNROLL_LOOP
230 for (int i = 0; i < PacketSize; ++i) {
231 values[i] = self.coeff(index + i);
232 }
233 PacketReturnType rslt = internal::pload<PacketReturnType>(values);
234 return rslt;
235 }
236 }
237 };

239 template<int LoadMode>
240 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE PacketReturnType packet(Index index) const
241 {
242 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
243 eigen_assert(index + PacketSize - 1 < dimensions().TotalSize());
244 return PacketLoader<LoadMode, Self, TensorEvaluator<ArgType, Device>::PacketAccess>::Run(*this, index);
245 }

247 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
248 internal::TensorBlockResourceRequirements getResourceRequirements() const {
249 static const int inner_dim =
250 Layout == static_cast<int>(ColMajor) ? 0 : NumDims - 1;
251
252 const size_t target_size = m_device.firstLevelCacheSize();
253 const bool inner_dim_shuffled = m_shuffle[inner_dim] != inner_dim;
254
255 // Shuffled inner dimensions leads to a random memory access, which is not
256 // captured by default cost model bytes loaded/stored. We add this cost
257 // explicitly. The number of cycles picked based on the benchmarks.
258 // TODO(ezhulenev): This number was picked based on a very questionable
259 // benchmarks, add benchmarks that are representative of real workloads.
260 using BlockRequirements = internal::TensorBlockResourceRequirements;
261 if (inner_dim_shuffled) {
262 return BlockRequirements::uniform<Scalar>(target_size)
263 .addCostPerCoeff({0, 0, NumDims * 28});
264 } else {
265 return BlockRequirements::skewed<Scalar>(target_size);
266 }
267 }

269 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorBlock
270 block(TensorBlockDesc& desc, TensorBlockScratch& scratch,
271 bool root_of_expr_ast = false) const {
272 assert(m_impl.data() != NULL);
273
274 typedef internal::TensorBlockIO<ScalarNoConst, Index, NumDims, Layout>
275 TensorBlockIO;
276 typedef typename TensorBlockIO::Dst TensorBlockIODst;
277 typedef typename TensorBlockIO::Src TensorBlockIOSrc;
278
279 const typename TensorBlock::Storage block_storage =
280 TensorBlock::prepareStorage(
281 desc, scratch, /*allow_strided_storage=*/root_of_expr_ast);
282
283 typename TensorBlockIO::Dimensions input_strides(m_unshuffledInputStrides);
284 TensorBlockIOSrc src(input_strides, m_impl.data(), srcCoeff(desc.offset()));
285
286 TensorBlockIODst dst(block_storage.dimensions(), block_storage.strides(),
287 block_storage.data());
288
289 typename TensorBlockIO::DimensionsMap dst_to_src_dim_map(m_shuffle);
290 TensorBlockIO::Copy(dst, src, dst_to_src_dim_map);
291
292 return block_storage.AsTensorMaterializedBlock();
293 }

295 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE TensorOpCost costPerCoeff(bool vectorized) const {
296 const double compute_cost = m_is_identity ? TensorOpCost::AddCost<Index>() :
297 NumDims * (2 * TensorOpCost::AddCost<Index>() +
298 2 * TensorOpCost::MulCost<Index>() +
299 TensorOpCost::DivCost<Index>());
300 return m_impl.costPerCoeff(vectorized) +
301 TensorOpCost(0, 0, compute_cost, m_is_identity /* vectorized */, PacketSize);
302 }
303
304 EIGEN_DEVICE_FUNC typename Storage::Type data() const { return NULL; }
305
306#ifdef EIGEN_USE_SYCL
307 // binding placeholder accessors to a command group handler for SYCL
308 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void bind(cl::sycl::handler &cgh) const {
309 m_impl.bind(cgh);
310 }
311#endif
312 protected:
313 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index GetBlockOutputIndex(
314 Index input_index,
315 const DSizes<Index, NumDims>& input_block_strides,
316 const DSizes<Index, NumDims>& output_block_strides,
317 const DSizes<internal::TensorIntDivisor<Index>, NumDims>& fast_input_block_strides) const {
318 Index output_index = 0;
319 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
320 for (int i = NumDims - 1; i > 0; --i) {
321 const Index idx = input_index / fast_input_block_strides[i];
322 output_index += idx * output_block_strides[m_inverseShuffle[i]];
323 input_index -= idx * input_block_strides[i];
324 }
325 return output_index + input_index *
326 output_block_strides[m_inverseShuffle[0]];
327 } else {
328 for (int i = 0; i < NumDims - 1; ++i) {
329 const Index idx = input_index / fast_input_block_strides[i];
330 output_index += idx * output_block_strides[m_inverseShuffle[i]];
331 input_index -= idx * input_block_strides[i];
332 }
333 return output_index + input_index *
334 output_block_strides[m_inverseShuffle[NumDims - 1]];
335 }
336 }
337
338 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Index srcCoeff(Index index) const {
339 Index inputIndex = 0;
340 if (static_cast<int>(Layout) == static_cast<int>(ColMajor)) {
341 for (int i = NumDims - 1; i > 0; --i) {
342 const Index idx = index / m_fastOutputStrides[i];
343 inputIndex += idx * m_inputStrides[i];
344 index -= idx * m_outputStrides[i];
345 }
346 return inputIndex + index * m_inputStrides[0];
347 } else {
348 for (int i = 0; i < NumDims - 1; ++i) {
349 const Index idx = index / m_fastOutputStrides[i];
350 inputIndex += idx * m_inputStrides[i];
351 index -= idx * m_outputStrides[i];
352 }
353 return inputIndex + index * m_inputStrides[NumDims - 1];
354 }
355 }
356
357 Dimensions m_dimensions;
358 bool m_is_identity;
359 array<int, NumDims> m_shuffle;
360 array<Index, NumDims> m_inverseShuffle; // TODO(ezhulenev): Make it int type.
361 array<Index, NumDims> m_outputStrides;
362 array<internal::TensorIntDivisor<Index>, NumDims> m_fastOutputStrides;
363 array<Index, NumDims> m_inputStrides;
364 array<Index, NumDims> m_unshuffledInputStrides;
365
366 const Device EIGEN_DEVICE_REF m_device;
367 TensorEvaluator<ArgType, Device> m_impl;
368};

411 template <int StoreMode> EIGEN_STRONG_INLINE
412 void writePacket(Index index, const PacketReturnType& x)
413 {
414 EIGEN_STATIC_ASSERT((PacketSize > 1), YOU_MADE_A_PROGRAMMING_MISTAKE)
415
416 EIGEN_ALIGN_MAX typename internal::remove_const<CoeffReturnType>::type values[PacketSize];
417 internal::pstore<CoeffReturnType, PacketReturnType>(values, x);
418 EIGEN_UNROLL_LOOP
419 for (int i = 0; i < PacketSize; ++i) {
420 this->coeffRef(index+i) = values[i];
421 }
422 }

424 template <typename TensorBlock>
425 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void writeBlock(
426 const TensorBlockDesc& desc, const TensorBlock& block) {
427 eigen_assert(this->m_impl.data() != NULL);
428
429 typedef internal::TensorBlockIO<ScalarNoConst, Index, NumDims, Layout>
430 TensorBlockIO;
431 typedef typename TensorBlockIO::Dst TensorBlockIODst;
432 typedef typename TensorBlockIO::Src TensorBlockIOSrc;
433
434 const Scalar* block_buffer = block.data();
435
436 // TODO(ezhulenev): TensorBlockIO should be able to read from any Eigen
437 // expression with coefficient and packet access as `src`.
438 void* mem = NULL;
439 if (block_buffer == NULL) {
440 mem = this->m_device.allocate(desc.size() * sizeof(Scalar));
441 ScalarNoConst* buf = static_cast<ScalarNoConst*>(mem);
442
443 typedef internal::TensorBlockAssignment<
444 ScalarNoConst, NumDims, typename TensorBlock::XprType, Index>
445 TensorBlockAssignment;
446
447 TensorBlockAssignment::Run(
448 TensorBlockAssignment::target(
449 desc.dimensions(), internal::strides<Layout>(desc.dimensions()),
450 buf),
451 block.expr());
452
453 block_buffer = buf;
454 }
455
456 // Read from block.
457 TensorBlockIOSrc src(internal::strides<Layout>(desc.dimensions()),
458 block_buffer);
459
460 // Write to the output buffer.
461 typename TensorBlockIO::Dimensions output_strides(
462 this->m_unshuffledInputStrides);
463 typename TensorBlockIO::Dimensions output_dimensions;
464 for (int i = 0; i < NumDims; ++i) {
465 output_dimensions[this->m_shuffle[i]] = desc.dimension(i);
466 }
467 TensorBlockIODst dst(output_dimensions, output_strides, this->m_impl.data(),
468 this->srcCoeff(desc.offset()));
469
470 // Reorder dimensions according to the shuffle.
471 typename TensorBlockIO::DimensionsMap dst_to_src_dim_map;
472 for (int i = 0; i < NumDims; ++i) {
473 dst_to_src_dim_map[i] = static_cast<int>(this->m_inverseShuffle[i]);
474 }
475 TensorBlockIO::Copy(dst, src, dst_to_src_dim_map);
476
477 // Deallocate temporary buffer used for the block materialization.
478 if (mem != NULL) this->m_device.deallocate(mem);
479 }
