42struct FuncInfo {
43 int min_coords[16];
44 int max_coords[16];
45 int dimensions;
46 halide_type_t type;
47 Buffer<> values;
48
49 FuncInfo() = default;
50 FuncInfo(Packet *p) {
51 int real_dims = p->dimensions / p->type.lanes;
52 if (real_dims > 16) {
53 fprintf(stderr, "Error: found trace packet with dimensionality > 16. Aborting.\n");
54 exit(-1);
55 }
56 for (int i = 0; i < real_dims; i++) {
57 min_coords[i] = INT32_MAX;
58 max_coords[i] = INT32_MIN;
59 }
60 dimensions = real_dims;
61 type = p->type;
62 type.lanes = 1;
63 }
64
65 void add_preprocess(Packet *p) {
66 int real_dims = p->dimensions / p->type.lanes;
67 int lanes = p->type.lanes;
68
69 halide_type_t scalar_type = p->type;
70 scalar_type.lanes = 1;
71
72 if (scalar_type != type) {
73 fprintf(stderr, "Error: packet type doesn't match previous packets of same Func. Aborting.\n");
74 exit(-1);
75 }
76 if (real_dims != dimensions) {
77 fprintf(stderr, "Error: packet dimensionality doesn't match previous packets of same Func. Aborting.\n");
78 exit(-1);
79 }
80
81 for (int lane = 0; lane < lanes; lane++) {
82 for (int i = 0; i < real_dims; i++) {
83 if (p->coordinates()[lanes * i + lane] < min_coords[i]) {
84 min_coords[i] = p->coordinates()[lanes * i + lane];
85 }
86 if (p->coordinates()[lanes * i + lane] > max_coords[i]) {
87 max_coords[i] = p->coordinates()[lanes * i + lane];
88 }
89 }
90 }
91 }
92
93 void allocate() {
94 std::vector<int> extents;
95 for (int i = 0; i < dimensions; i++) {
96 extents.push_back(max_coords[i] - min_coords[i] + 1);
97 }
98 values = Buffer<>(type, extents);
99 if (values.data() == nullptr) {
100 fprintf(stderr, "Memory allocation failure. Aborting.\n");
101 exit(-1);
102 }
103 }
104
105 void add(Packet *p) {
106 halide_type_t scalar_type = p->type;
107 scalar_type.lanes = 1;
108 if (scalar_type == halide_type_of<float>()) {
109 add_typed<float>(p);
110 } else if (scalar_type == halide_type_of<double>()) {
111 add_typed<double>(p);
112 } else if (scalar_type == halide_type_of<uint8_t>()) {
113 add_typed<uint8_t>(p);
114 } else if (scalar_type == halide_type_of<uint16_t>()) {
115 add_typed<uint16_t>(p);
116 } else if (scalar_type == halide_type_of<uint32_t>()) {
117 add_typed<uint32_t>(p);
118 } else if (scalar_type == halide_type_of<uint64_t>()) {
119 add_typed<uint64_t>(p);
120 } else if (scalar_type == halide_type_of<int8_t>()) {
121 add_typed<int8_t>(p);
122 } else if (scalar_type == halide_type_of<int16_t>()) {
123 add_typed<int16_t>(p);
124 } else if (scalar_type == halide_type_of<int32_t>()) {
125 add_typed<int32_t>(p);
126 } else if (scalar_type == halide_type_of<int64_t>()) {
127 add_typed<int64_t>(p);
128 } else if (scalar_type == halide_type_of<bool>()) {
129 add_typed<bool>(p);
130 } else {
131 printf("Packet with unknown type\n");
132 exit(-1);
133 }
134 }
135
136 template<typename T>
137 void add_typed(Packet *p) {
138 Buffer<T> &buf = values.as<T>();
139 int lanes = p->type.lanes;
140
141 if (!allocated()) {
142 fprintf(stderr, "Packet storage not allocated. Aborting.\n");
143 exit(-1);
144 }
145
146 if (p->dimensions < 0) {
147 fprintf(stderr, "Negative dimensionality found. Aborting.\n");
148 exit(-1);
149 }
150
151 for (int lane = 0; lane < lanes; lane++) {
152 int coord[16];
153 for (int i = 0; i < dimensions; i++) {
154 coord[i] = p->coordinates()[lanes * i + lane] - min_coords[i];
155 }
156 buf(coord) = p->get_value_as<T>(lane);
157 }
158 }
159
160 bool allocated() const {
161 return (values.data() != nullptr);
162 }
163};

165bool check_and_continue(bool condition, const char *msg) {
166 if (!condition) {
167 fprintf(stderr, "Failed to dump func: %s\n", msg);
168 }
169 return condition;
170}

172void dump_func(string name, FuncInfo &func, BufferOutputOpts output_opts) {
173 // Remove special characters
174 for (size_t i = 0; i < name.size(); i++) {
175 if (!std::isalnum(name[i])) {
176 name[i] = '_';
177 }
178 }
179
180 string filename;
181 switch (output_opts.type) {
182 case BufferOutputOpts::PNG:
183 filename = name + ".png";
184 break;
185 case BufferOutputOpts::JPG:
186 filename = name + ".jpg";
187 break;
188 case BufferOutputOpts::PGM:
189 filename = name + ".pgm";
190 break;
191 case BufferOutputOpts::TMP:
192 filename = name + ".tmp";
193 break;
194 case BufferOutputOpts::MAT:
195 filename = name + ".mat";
196 break;
197 default:
198 exit(1);
199 }
200
201 printf("[INFO] Dumping func '%s' to file: %s\n", name.c_str(), filename.c_str());
202
203 // Rely on save_image to do type-checking
204 Halide::Tools::convert_and_save_image<Buffer<>, check_and_continue>(func.values, filename);
205}

207void finish_dump(map<string, FuncInfo> &func_info, BufferOutputOpts output_opts) {
208 printf("\nTrace stats:\n");
209 printf(" Funcs:\n");
210 for (auto &pair : func_info) {
211 const string &name = pair.first;
212 FuncInfo &info = pair.second;
213 printf(" %s:\n", name.c_str());
214
215 // Type
216 printf(" Type: ");
217 if (info.type.code == halide_type_code_t::halide_type_int) {
218 printf("int%d\n", info.type.bits);
219 } else if (info.type.code == halide_type_code_t::halide_type_uint) {
220 printf("uint%d\n", info.type.bits);
221 } else if (info.type.code == halide_type_code_t::halide_type_float) {
222 printf("float%d\n", info.type.bits);
223 } else {
224 fprintf(stderr, "Unsupported Func type. Aborting.\n");
225 exit(-1);
226 }
227
228 // Dimensions
229 printf(" Dimensions: %d\n", info.dimensions);
230
231 // Size of the func
232 printf(" Size: ");
233 for (int idx = 0; idx < info.dimensions; idx++) {
234 if (idx > 0) {
235 printf("x");
236 }
237 printf("%d", (info.max_coords[idx] - info.min_coords[idx]) + 1);
238 }
239 printf("\n");
240
241 // Minima
242 printf(" Minimum stored to in each dim: {");
243 for (int idx = 0; idx < info.dimensions; idx++) {
244 if (idx > 0) {
245 printf(", ");
246 }
247 printf("%d", info.min_coords[idx]);
248 }
249 printf("}\n");
250
251 // Maxima
252 printf(" Maximum stored to in each dim: {");
253 for (int idx = 0; idx < info.dimensions; idx++) {
254 if (idx > 0) {
255 printf(", ");
256 }
257 printf("%d", info.max_coords[idx]);
258 }
259 printf("}\n");
260 }
261
262 for (auto &pair : func_info) {
263 string name = pair.first;
264 FuncInfo &info = pair.second;
265 dump_func(name, info, output_opts);
266 }
267
268 printf("Done.\n");
269}

271void usage(char *const *argv) {
272 const string usage =
273 "Usage: " + string(argv[0]) +
274 " -i trace_file -t {png,jpg,pgm,tmp,mat}\n"
275 "\n"
276 "This tool reads a binary trace produced by Halide, and dumps all\n"
277 "Funcs into individual image files in the current directory.\n"
278 "To generate a suitable binary trace, use Func::trace_stores(), or the\n"
279 "target features trace_stores and trace_realizations, and run with\n"
280 "HL_TRACE_FILE=<filename>.\n";
281 fprintf(stderr, "%s\n", usage.c_str());
282 exit(1);
283}

285int main(int argc, char *const *argv) {
286 char *buf_filename = nullptr;
287 char *buf_imagetype = nullptr;
288 BufferOutputOpts outputopts;
289 for (int i = 1; i < argc - 1; i++) {
290 string arg = argv[i];
291 if (arg == "-t") {
292 i++;
293 buf_imagetype = argv[i];
294 } else if (arg == "-i") {
295 i++;
296 buf_filename = argv[i];
297 }
298 }
299
300 if (buf_filename == nullptr) {
301 usage(argv);
302 }
303 if (buf_imagetype == nullptr) {
304 usage(argv);
305 }
306
307 string imagetype(buf_imagetype);
308 if (imagetype == "jpg") {
309 outputopts.type = BufferOutputOpts::JPG;
310 } else if (imagetype == "png") {
311 outputopts.type = BufferOutputOpts::PNG;
312 } else if (imagetype == "pgm") {
313 outputopts.type = BufferOutputOpts::PGM;
314 } else if (imagetype == "tmp") {
315 outputopts.type = BufferOutputOpts::TMP;
316 } else if (imagetype == "mat") {
317 outputopts.type = BufferOutputOpts::MAT;
318 } else {
319 usage(argv);
320 }
321
322 FILE *file_desc = fopen(buf_filename, "r");
323 if (file_desc == nullptr) {
324 fprintf(stderr, "[Error opening file: %s. Exiting.\n", argv[1]);
325 exit(1);
326 }
327
328 printf("[INFO] Starting parse of binary trace...\n");
329 int packet_count = 0;
330
331 map<string, FuncInfo> func_info;
332
333 printf("[INFO] First pass...\n");
334
335 for (;;) {
336 Packet p;
337 if (!p.read_from_filedesc(file_desc)) {
338 printf("[INFO] Finished pass 1 after %d packets.\n", packet_count);
339 break;
340 }
341
342 // Packet read was successful.
343 packet_count++;
344 if ((packet_count % 100000) == 0) {
345 printf("[INFO] Pass 1: Read %d packets so far.\n", packet_count);
346 }
347
348 // Check if this was a store packet.
349 if ((p.event == halide_trace_store) || (p.event == halide_trace_load)) {
350 if (func_info.find(string(p.func())) == func_info.end()) {
351 printf("[INFO] Found Func with tracked accesses: %s\n", p.func());
352 func_info[string(p.func())] = FuncInfo(&p);
353 }
354 func_info[string(p.func())].add_preprocess(&p);
355 }
356 }
357
358 packet_count = 0;
359 fseek(file_desc, 0, SEEK_SET);
360 if (ferror(file_desc)) {
361 fprintf(stderr, "Error: couldn't seek back to beginning of trace file. Aborting.\n");
362 exit(-1);
363 }
364
365 for (auto &pair : func_info) {
366 pair.second.allocate();
367 }
368
369 for (;;) {
370 Packet p;
371 if (!p.read_from_filedesc(file_desc)) {
372 printf("[INFO] Finished pass 2 after %d packets.\n", packet_count);
373 if (file_desc != nullptr) {
374 fclose(file_desc);
375 }
376 finish_dump(func_info, outputopts);
377 exit(0);
378 }
379
380 // Packet read was successful.
381 packet_count++;
382 if ((packet_count % 100000) == 0) {
383 printf("[INFO] Pass 2: Read %d packets so far.\n", packet_count);
384 }
385
386 // Check if this was a store packet.
387 if ((p.event == halide_trace_store) || (p.event == halide_trace_load)) {
388 if (func_info.find(string(p.func())) == func_info.end()) {
389 fprintf(stderr, "Unable to find Func on 2nd pass. Aborting.\n");
390 exit(-1);
391 }
392 func_info[string(p.func())].add(&p);
393 }
394 }
