87/* Retrieve command line arguments as UTF-8. */
88static int get_argv_utf8( int *argc_ptr, char ***argv_ptr )
89{
90 int ret = 0;
91 wchar_t **argv_utf16 = CommandLineToArgvW( GetCommandLineW(), argc_ptr );
92 if( argv_utf16 )
93 {
94 int argc = *argc_ptr;
95 int offset = (argc+1) * sizeof(char*);
96 int size = offset;
97
98 for( int i = 0; i < argc; i++ )
99 size += WideCharToMultiByte( CP_UTF8, 0, argv_utf16[i], -1, NULL, 0, NULL, NULL );
100
101 char **argv = *argv_ptr = malloc( size );
102 if( argv )
103 {
104 for( int i = 0; i < argc; i++ )
105 {
106 argv[i] = (char*)argv + offset;
107 offset += WideCharToMultiByte( CP_UTF8, 0, argv_utf16[i], -1, argv[i], size-offset, NULL, NULL );
108 }
109 argv[argc] = NULL;
110 ret = 1;
111 }
112 LocalFree( argv_utf16 );
113 }
114 return ret;
115}

264/* logging and printing for within the cli system */
265static int cli_log_level = X264_LOG_INFO;
266void x264_cli_log( const char *name, int i_level, const char *fmt, ... )
267{
268 if( i_level > cli_log_level )
269 return;
270 char *s_level;
271 switch( i_level )
272 {
273 case X264_LOG_ERROR:
274 s_level = "error";
275 break;
276 case X264_LOG_WARNING:
277 s_level = "warning";
278 break;
279 case X264_LOG_INFO:
280 s_level = "info";
281 break;
282 case X264_LOG_DEBUG:
283 s_level = "debug";
284 break;
285 default:
286 s_level = "unknown";
287 break;
288 }
289 fprintf( stderr, "%s [%s]: ", name, s_level );
290 va_list arg;
291 va_start( arg, fmt );
292 x264_vfprintf( stderr, fmt, arg );
293 va_end( arg );
294}

358REALIGN_STACK int main( int argc, char **argv )
359{
360 if( argc == 4 && !strcmp( argv[1], "--autocomplete" ) )
361 return x264_cli_autocomplete( argv[2], argv[3] );
362
363 x264_param_t param;
364 cli_opt_t opt = {0};
365 int ret = 0;
366
367 FAIL_IF_ERROR( x264_threading_init(), "unable to initialize threading\n" );
368
369#ifdef _WIN32
370 FAIL_IF_ERROR( !get_argv_utf8( &argc, &argv ), "unable to convert command line to UTF-8\n" );
371
372 GetConsoleTitleW( org_console_title, CONSOLE_TITLE_SIZE );
373 _setmode( _fileno( stdin ), _O_BINARY );
374 _setmode( _fileno( stdout ), _O_BINARY );
375 _setmode( _fileno( stderr ), _O_BINARY );
376#endif
377
378 x264_param_default( &param );
379 /* Parse command line */
380 if( parse( argc, argv, &param, &opt ) < 0 )
381 ret = -1;
382
383#ifdef _WIN32
384 /* Restore title; it can be changed by input modules */
385 SetConsoleTitleW( org_console_title );
386#endif
387
388 /* Control-C handler */
389 signal( SIGINT, sigint_handler );
390
391 if( !ret )
392 ret = encode( &param, &opt );
393
394 /* clean up handles */
395 if( filter.free )
396 filter.free( opt.hin );
397 else if( opt.hin )
398 cli_input.close_file( opt.hin );
399 if( opt.hout )
400 cli_output.close_file( opt.hout, 0, 0 );
401 if( opt.tcfile_out )
402 fclose( opt.tcfile_out );
403 if( opt.qpfile )
404 fclose( opt.qpfile );
405 x264_param_cleanup( &param );
406
407#ifdef _WIN32
408 SetConsoleTitleW( org_console_title );
409 free( argv );
410#endif
411
412 return ret;
413}

421static char *stringify_names( char *buf, const char * const names[] )
422{
423 int i = 0;
424 char *p = buf;
425 for( p[0] = 0; names[i]; i++ )
426 if( *names[i] )
427 {
428 if( p != buf )
429 p += sprintf( p, ", " );
430 p += sprintf( p, "%s", names[i] );
431 }
432 return buf;
433}

440static void print_csp_name_internal( const char *name, size_t *line_len, int last )
441{
442 if( name )
443 {
444 size_t name_len = strlen( name );
445 if( *line_len + name_len > (80 - SEPARATOR_LEN) )
446 {
447 printf( "\n" INDENT );
448 *line_len = INDENT_LEN;
449 }
450 printf( "%s", name );
451 *line_len += name_len;
452 if( !last )
453 {
454 printf( SEPARATOR );
455 *line_len += SEPARATOR_LEN;
456 }
457 }
458}

1185static int select_output( const char *muxer, char *filename, x264_param_t *param )
1186{
1187 const char *ext = get_filename_extension( filename );
1188 if( !strcmp( filename, "-" ) || strcasecmp( muxer, "auto" ) )
1189 ext = muxer;
1190
1191 if( !strcasecmp( ext, "mp4" ) )
1192 {
1193#if HAVE_GPAC || HAVE_LSMASH
1194 cli_output = mp4_output;

1201 }
1202#else
1203 x264_cli_log( "x264", X264_LOG_ERROR, "not compiled with MP4 output support\n" );

1205#endif
1206 }
1207 else if( !strcasecmp( ext, "mkv" ) )
1208 {
1209 cli_output = mkv_output;
1210 param->b_annexb = 0;
1211 param->b_repeat_headers = 0;
1212 }

1214 {
1215 cli_output = flv_output;
1216 param->b_annexb = 0;
1217 param->b_repeat_headers = 0;
1218 }

1224static int select_input( const char *demuxer, char *used_demuxer, char *filename,
1225 hnd_t *p_handle, video_info_t *info, cli_input_opt_t *opt )
1226{
1227 int b_auto = !strcasecmp( demuxer, "auto" );
1228 const char *ext = b_auto ? get_filename_extension( filename ) : "";
1229 int b_regular = strcmp( filename, "-" );
1230 if( !b_regular && b_auto )
1231 ext = "raw";
1232 b_regular = b_regular && x264_is_regular_file_path( filename );
1233 if( b_regular )
1234 {
1235 FILE *f = x264_fopen( filename, "r" );
1236 if( f )
1237 {
1238 b_regular = x264_is_regular_file( f );
1239 fclose( f );
1240 }
1241 }
1242 const char *module = b_auto ? ext : demuxer;
1243
1244 if( !strcasecmp( module, "avs" ) || !strcasecmp( ext, "d2v" ) || !strcasecmp( ext, "dga" ) )
1245 {
1246#if HAVE_AVS
1247 cli_input = avs_input;
1248 module = "avs";
1249#else
1250 x264_cli_log( "x264", X264_LOG_ERROR, "not compiled with AVS input support\n" );
1251 return -1;
1252#endif
1253 }
1254 else if( !strcasecmp( module, "y4m" ) )
1255 cli_input = y4m_input;
1256 else if( !strcasecmp( module, "raw" ) || !strcasecmp( ext, "yuv" ) )
1257 cli_input = raw_input;
1258 else
1259 {
1260#if HAVE_FFMS
1261 if( b_regular && (b_auto || !strcasecmp( demuxer, "ffms" )) &&
1262 !ffms_input.open_file( filename, p_handle, info, opt ) )
1263 {
1264 module = "ffms";
1265 b_auto = 0;
1266 cli_input = ffms_input;
1267 }
1268#endif
1269#if HAVE_LAVF
1270 if( (b_auto || !strcasecmp( demuxer, "lavf" )) &&
1271 !lavf_input.open_file( filename, p_handle, info, opt ) )
1272 {
1273 module = "lavf";
1274 b_auto = 0;
1275 cli_input = lavf_input;
1276 }
1277#endif
1278#if HAVE_AVS
1279 if( b_regular && (b_auto || !strcasecmp( demuxer, "avs" )) &&
1280 !avs_input.open_file( filename, p_handle, info, opt ) )
1281 {
1282 module = "avs";
1283 b_auto = 0;
1284 cli_input = avs_input;
1285 }
1286#endif
1287 if( b_auto && !raw_input.open_file( filename, p_handle, info, opt ) )
1288 {
1289 module = "raw";
1290 b_auto = 0;
1291 cli_input = raw_input;
1292 }
1293

1298 return 0;
1299}

1301static int init_vid_filters( char *sequence, hnd_t *handle, video_info_t *info, x264_param_t *param, int output_csp )
1302{
1303 x264_register_vid_filters();
1304
1305 /* intialize baseline filters */
1306 if( x264_init_vid_filter( "source", handle, &filter, info, param, NULL ) ) /* wrap demuxer into a filter */
1307 return -1;
1308 if( x264_init_vid_filter( "resize", handle, &filter, info, param, "normcsp" ) ) /* normalize csps to be of a known/supported format */
1309 return -1;
1310 if( x264_init_vid_filter( "fix_vfr_pts", handle, &filter, info, param, NULL ) ) /* fix vfr pts */
1311 return -1;
1312
1313 /* parse filter chain */
1314 for( char *p = sequence; p && *p; )
1315 {
1316 int tok_len = strcspn( p, "/" );
1317 int p_len = strlen( p );
1318 p[tok_len] = 0;
1319 int name_len = strcspn( p, ":" );
1320 p[name_len] = 0;
1321 name_len += name_len != tok_len;
1322 if( x264_init_vid_filter( p, handle, &filter, info, param, p + name_len ) )
1323 return -1;
1324 p += X264_MIN( tok_len+1, p_len );
1325 }
1326
1327 /* force end result resolution */
1328 if( !param->i_width && !param->i_height )
1329 {
1330 param->i_height = info->height;
1331 param->i_width = info->width;
1332 }
1333 /* force the output csp to what the user specified (or the default) */
1334 param->i_csp = info->csp;
1335 int csp = info->csp & X264_CSP_MASK;
1336 if( output_csp == X264_CSP_I400 && csp != X264_CSP_I400 )
1337 param->i_csp = X264_CSP_I400;
1338 else if( output_csp == X264_CSP_I420 && (csp < X264_CSP_I420 || csp >= X264_CSP_I422) )
1339 param->i_csp = X264_CSP_I420;
1340 else if( output_csp == X264_CSP_I422 && (csp < X264_CSP_I422 || csp >= X264_CSP_I444) )
1341 param->i_csp = X264_CSP_I422;
1342 else if( output_csp == X264_CSP_I444 && (csp < X264_CSP_I444 || csp >= X264_CSP_BGR) )
1343 param->i_csp = X264_CSP_I444;
1344 else if( output_csp == X264_CSP_RGB && (csp < X264_CSP_BGR || csp > X264_CSP_RGB) )
1345 param->i_csp = X264_CSP_RGB;
1346 param->i_csp |= info->csp & X264_CSP_HIGH_DEPTH;
1347 /* if the output range is not forced, assign it to the input one now */
1348 if( param->vui.b_fullrange == RANGE_AUTO )
1349 param->vui.b_fullrange = info->fullrange;
1350
1351 if( x264_init_vid_filter( "resize", handle, &filter, info, param, NULL ) )
1352 return -1;
1353
1354 char args[20], name[20];
1355 sprintf( args, "bit_depth=%d", param->i_bitdepth );
1356 sprintf( name, "depth_%d", param->i_bitdepth );
1357
1358 if( x264_init_vid_filter( name, handle, &filter, info, param, args ) )
1359 return -1;
1360
1361 return 0;
1362}

1386static int parse( int argc, char **argv, x264_param_t *param, cli_opt_t *opt )
1387{
1388 char *input_filename = NULL;
1389 const char *demuxer = x264_demuxer_names[0];
1390 char *output_filename = NULL;
1391 const char *muxer = x264_muxer_names[0];
1392 char *tcfile_name = NULL;
1393 x264_param_t defaults;
1394 char *profile = NULL;
1395 char *vid_filters = NULL;
1396 int b_thread_input = 0;
1397 int b_turbo = 1;
1398 int b_user_ref = 0;
1399 int b_user_fps = 0;
1400 int b_user_interlaced = 0;
1401 cli_input_opt_t input_opt;
1402 cli_output_opt_t output_opt;
1403 char *preset = NULL;
1404 char *tune = NULL;
1405
1406 /* Presets are applied before all other options. */
1407 for( optind = 0;; )
1408 {
1409 int c = getopt_long( argc, argv, short_options, long_options, NULL );
1410 if( c == -1 )
1411 break;
1412 if( c == OPT_PRESET )
1413 preset = optarg;
1414 if( c == OPT_TUNE )
1415 tune = optarg;
1416 else if( c == '?' )
1417 return -1;
1418 }
1419
1420 if( preset && !strcasecmp( preset, "placebo" ) )
1421 b_turbo = 0;
1422
1423 if( (preset || tune) && x264_param_default_preset( param, preset, tune ) < 0 )
1424 return -1;
1425
1426 x264_param_default( &defaults );
1427 cli_log_level = defaults.i_log_level;
1428
1429 memset( &input_opt, 0, sizeof(cli_input_opt_t) );
1430 memset( &output_opt, 0, sizeof(cli_output_opt_t) );
1431 input_opt.bit_depth = 8;
1432 input_opt.input_range = input_opt.output_range = param->vui.b_fullrange = RANGE_AUTO;
1433 int output_csp = defaults.i_csp;
1434 opt->b_progress = 1;
1435
1436 /* Parse command line options */
1437 for( optind = 0;; )
1438 {
1439 int b_error = 0;
1440 int long_options_index = -1;
1441
1442 int c = getopt_long( argc, argv, short_options, long_options, &long_options_index );
1443
1444 if( c == -1 )
1445 {
1446 break;
1447 }
1448
1449 switch( c )
1450 {
1451 case 'h':
1452 help( &defaults, 0 );
1453 exit(0);
1454 case OPT_LONGHELP:
1455 help( &defaults, 1 );
1456 exit(0);
1457 case OPT_FULLHELP:
1458 help( &defaults, 2 );
1459 exit(0);
1460 case 'V':
1461 print_version_info();
1462 exit(0);
1463 case OPT_FRAMES:
1464 param->i_frame_total = X264_MAX( atoi( optarg ), 0 );
1465 break;
1466 case OPT_SEEK:
1467 opt->i_seek = X264_MAX( atoi( optarg ), 0 );
1468 break;
1469 case 'o':
1470 output_filename = optarg;
1471 break;
1472 case OPT_MUXER:
1473 FAIL_IF_ERROR( parse_enum_name( optarg, x264_muxer_names, &muxer ), "Unknown muxer `%s'\n", optarg );
1474 break;
1475 case OPT_DEMUXER:
1476 FAIL_IF_ERROR( parse_enum_name( optarg, x264_demuxer_names, &demuxer ), "Unknown demuxer `%s'\n", optarg );
1477 break;
1478 case OPT_INDEX:
1479 input_opt.index_file = optarg;
1480 break;
1481 case OPT_QPFILE:
1482 opt->qpfile = x264_fopen( optarg, "rb" );
1483 FAIL_IF_ERROR( !opt->qpfile, "can't open qpfile `%s'\n", optarg );
1484 if( !x264_is_regular_file( opt->qpfile ) )
1485 {
1486 x264_cli_log( "x264", X264_LOG_ERROR, "qpfile incompatible with non-regular file `%s'\n", optarg );
1487 fclose( opt->qpfile );
1488 return -1;
1489 }
1490 break;
1491 case OPT_THREAD_INPUT:
1492 b_thread_input = 1;
1493 break;
1494 case OPT_QUIET:
1495 cli_log_level = param->i_log_level = X264_LOG_NONE;
1496 break;
1497 case 'v':
1498 cli_log_level = param->i_log_level = X264_LOG_DEBUG;
1499 break;
1500 case OPT_LOG_LEVEL:
1501 if( !parse_enum_value( optarg, x264_log_level_names, &cli_log_level ) )
1502 cli_log_level += X264_LOG_NONE;
1503 else
1504 cli_log_level = atoi( optarg );
1505 param->i_log_level = cli_log_level;
1506 break;
1507 case OPT_NOPROGRESS:
1508 opt->b_progress = 0;
1509 break;
1510 case OPT_TUNE:
1511 case OPT_PRESET:
1512 break;
1513 case OPT_PROFILE:
1514 profile = optarg;
1515 break;
1516 case OPT_SLOWFIRSTPASS:
1517 b_turbo = 0;
1518 break;
1519 case 'r':
1520 b_user_ref = 1;
1521 goto generic_option;
1522 case OPT_FPS:
1523 b_user_fps = 1;
1524 param->b_vfr_input = 0;
1525 goto generic_option;
1526 case OPT_INTERLACED:
1527 b_user_interlaced = 1;
1528 goto generic_option;
1529 case OPT_TCFILE_IN:
1530 tcfile_name = optarg;
1531 break;
1532 case OPT_TCFILE_OUT:
1533 opt->tcfile_out = x264_fopen( optarg, "wb" );
1534 FAIL_IF_ERROR( !opt->tcfile_out, "can't open `%s'\n", optarg );
1535 break;
1536 case OPT_TIMEBASE:
1537 input_opt.timebase = optarg;
1538 break;
1539 case OPT_PULLDOWN:
1540 FAIL_IF_ERROR( parse_enum_value( optarg, x264_pulldown_names, &opt->i_pulldown ), "Unknown pulldown `%s'\n", optarg );
1541 break;
1542 case OPT_VIDEO_FILTER:
1543 vid_filters = optarg;
1544 break;

1546 input_opt.format = optarg;

1549 input_opt.resolution = optarg;

1552 input_opt.colorspace = optarg;

1555 input_opt.bit_depth = atoi( optarg );

1558 param->i_bitdepth = atoi( optarg );

1561 output_opt.use_dts_compress = 1;

1566#if X264_CHROMA_FORMAT

1568#else
1569 static const uint8_t output_csp_fix[] = { X264_CSP_I400, X264_CSP_I420, X264_CSP_I422, X264_CSP_I444, X264_CSP_RGB };

1597 }
1598
1599 b_error |= x264_param_parse( param, long_options[long_options_index].name, optarg );
1600 }
1601 }

1603 if( b_error )
1604 {
1605 const char *name = long_options_index > 0 ? long_options[long_options_index].name : argv[optind-2];
1606 x264_cli_log( "x264", X264_LOG_ERROR, "invalid argument: %s = %s\n", name, optarg );
1607 return -1;
1608 }

1617 return -1;

1620 FAIL_IF_ERROR( optind > argc - 1 || !output_filename, "No %s file. Run x264 --help for a list of options.\n",
1621 optind > argc - 1 ? "input" : "output" );
1622
1623 if( select_output( muxer, output_filename, param ) )
1624 return -1;
1625 FAIL_IF_ERROR( cli_output.open_file( output_filename, &opt->hout, &output_opt ), "could not open output file `%s'\n", output_filename );
1626
1627 input_filename = argv[optind++];
1628 video_info_t info = {0};
1629 char demuxername[5];
1630

1632 info.csp = param->i_csp;
1633 info.fps_num = param->i_fps_num;
1634 info.fps_den = param->i_fps_den;
1635 info.fullrange = input_opt.input_range == RANGE_PC;
1636 info.interlaced = param->b_interlaced;
1637 if( param->vui.i_sar_width > 0 && param->vui.i_sar_height > 0 )
1638 {
1639 info.sar_width = param->vui.i_sar_width;
1640 info.sar_height = param->vui.i_sar_height;
1641 }
1642 info.tff = param->b_tff;
1643 info.vfr = param->b_vfr_input;
1644
1645 input_opt.seek = opt->i_seek;
1646 input_opt.progress = opt->b_progress;
1647 input_opt.output_csp = output_csp;
1648
1649 if( select_input( demuxer, demuxername, input_filename, &opt->hin, &info, &input_opt ) )
1650 return -1;
1651

1656 x264_reduce_fraction( &info.fps_num, &info.fps_den );

1658 info.height, info.interlaced ? 'i' : 'p', info.sar_width, info.sar_height,
1659 info.fps_num, info.fps_den, info.vfr ? 'v' : 'c' );

1662 info.width > MAX_RESOLUTION || info.height > MAX_RESOLUTION,

1666 {

1670 }

1699 info.sar_height = param->vui.i_sar_height;

1701 if( b_user_fps )
1702 {

1718 {
1719 i_user_timebase_num = info.timebase_num;

1729 }

1733 info.tff = param->b_tff;
1734 }
1735 if( input_opt.input_range != RANGE_AUTO )
1736 info.fullrange = input_opt.input_range;
1737
1738 if( init_vid_filters( vid_filters, &opt->hin, &info, param, output_csp ) )
1739 return -1;
1740
1741 /* set param flags from the post-filtered video */
1742 param->b_vfr_input = info.vfr;
1743 param->i_fps_num = info.fps_num;
1744 param->i_fps_den = info.fps_den;
1745 param->i_timebase_num = info.timebase_num;
1746 param->i_timebase_den = info.timebase_den;
1747 param->vui.i_sar_width = info.sar_width;
1748 param->vui.i_sar_height = info.sar_height;
1749
1750 info.num_frames = X264_MAX( info.num_frames - opt->i_seek, 0 );
1751 if( (!info.num_frames || param->i_frame_total < info.num_frames)
1752 && param->i_frame_total > 0 )
1753 info.num_frames = param->i_frame_total;
1754 param->i_frame_total = info.num_frames;
1755
1756 if( !b_user_interlaced && info.interlaced )
1757 {
1758#if HAVE_INTERLACED
1759 x264_cli_log( "x264", X264_LOG_WARNING, "input appears to be interlaced, enabling %cff interlaced mode.\n"
1760 " If you want otherwise, use --no-interlaced or --%cff\n",
1761 info.tff ? 't' : 'b', info.tff ? 'b' : 't' );
1762 param->b_interlaced = 1;
1763 param->b_tff = !!info.tff;
1764#else
1765 x264_cli_log( "x264", X264_LOG_WARNING, "input appears to be interlaced, but not compiled with interlaced support\n" );
1766#endif
1767 }
1768 /* if the user never specified the output range and the input is now rgb, default it to pc */
1769 int csp = param->i_csp & X264_CSP_MASK;
1770 if( csp >= X264_CSP_BGR && csp <= X264_CSP_RGB )
1771 {
1772 if( input_opt.output_range == RANGE_AUTO )
1773 param->vui.b_fullrange = RANGE_PC;
1774 /* otherwise fail if they specified tv */
1775 FAIL_IF_ERROR( !param->vui.b_fullrange, "RGB must be PC range" );
1776 }
1777
1778 /* Automatically reduce reference frame count to match the user's target level
1779 * if the user didn't explicitly set a reference frame count. */
1780 if( !b_user_ref )
1781 {
1782 int mbs = (((param->i_width)+15)>>4) * (((param->i_height)+15)>>4);
1783 for( int i = 0; x264_levels[i].level_idc != 0; i++ )
1784 if( param->i_level_idc == x264_levels[i].level_idc )
1785 {
1786 while( mbs * param->i_frame_reference > x264_levels[i].dpb && param->i_frame_reference > 1 )
1787 param->i_frame_reference--;
1788 break;
1789 }
1790 }
1791
1792
1793 return 0;
1794}

1796static void parse_qpfile( cli_opt_t *opt, x264_picture_t *pic, int i_frame )
1797{
1798 int num = -1;
1799 char type;
1800 while( num < i_frame )
1801 {
1802 int64_t file_pos = ftell( opt->qpfile );
1803 int qp = -1;
1804 int ret = fscanf( opt->qpfile, "%d %c%*[ \t]%d\n", &num, &type, &qp );
1805 pic->i_type = X264_TYPE_AUTO;
1806 pic->i_qpplus1 = X264_QP_AUTO;
1807 if( num > i_frame || ret == EOF )
1808 {
1809 if( file_pos < 0 || fseek( opt->qpfile, file_pos, SEEK_SET ) )
1810 {
1811 x264_cli_log( "x264", X264_LOG_ERROR, "qpfile seeking failed\n" );
1812 fclose( opt->qpfile );
1813 opt->qpfile = NULL;
1814 }
1815 break;
1816 }
1817 if( num < i_frame && ret >= 2 )
1818 continue;
1819 if( ret == 3 && qp >= 0 )
1820 pic->i_qpplus1 = qp+1;
1821 if ( type == 'I' ) pic->i_type = X264_TYPE_IDR;
1822 else if( type == 'i' ) pic->i_type = X264_TYPE_I;
1823 else if( type == 'K' ) pic->i_type = X264_TYPE_KEYFRAME;
1824 else if( type == 'P' ) pic->i_type = X264_TYPE_P;
1825 else if( type == 'B' ) pic->i_type = X264_TYPE_BREF;
1826 else if( type == 'b' ) pic->i_type = X264_TYPE_B;
1827 else ret = 0;
1828 if( ret < 2 || qp < -1 || qp > QP_MAX )
1829 {
1830 x264_cli_log( "x264", X264_LOG_ERROR, "can't parse qpfile for frame %d\n", i_frame );
1831 fclose( opt->qpfile );
1832 opt->qpfile = NULL;
1833 break;
1834 }
1835 }

1858static int64_t print_status( int64_t i_start, int64_t i_previous, int i_frame, int i_frame_total, int64_t i_file, x264_param_t *param, int64_t last_ts )

1906static int encode( x264_param_t *param, cli_opt_t *opt )
1907{
1908 x264_t *h = NULL;
1909 x264_picture_t pic;
1910 cli_pic_t cli_pic;
1911 const cli_pulldown_t *pulldown = NULL; // shut up gcc
1912
1913 int i_frame = 0;
1914 int i_frame_output = 0;
1915 int64_t i_end, i_previous = 0, i_start = 0;
1916 int64_t i_file = 0;
1917 int i_frame_size;
1918 int64_t last_dts = 0;
1919 int64_t prev_dts = 0;
1920 int64_t first_dts = 0;
1921# define MAX_PTS_WARNING 3 /* arbitrary */
1922 int pts_warning_cnt = 0;
1923 int64_t largest_pts = -1;
1924 int64_t second_largest_pts = -1;
1925 int64_t ticks_per_frame;
1926 double duration;
1927 double pulldown_pts = 0;
1928 int retval = 0;
1929
1930 opt->b_progress &= param->i_log_level < X264_LOG_DEBUG;
1931
1932 /* set up pulldown */
1933 if( opt->i_pulldown && !param->b_vfr_input )
1934 {
1935 param->b_pulldown = 1;
1936 param->b_pic_struct = 1;
1937 pulldown = &pulldown_values[opt->i_pulldown];
1938 param->i_timebase_num = param->i_fps_den;
1939 FAIL_IF_ERROR2( fmod( param->i_fps_num * pulldown->fps_factor, 1 ),
1940 "unsupported framerate for chosen pulldown\n" );
1941 param->i_timebase_den = param->i_fps_num * pulldown->fps_factor;
1942 }
1943
1944 h = x264_encoder_open( param );
1945 FAIL_IF_ERROR2( !h, "x264_encoder_open failed\n" );
1946
1947 x264_encoder_parameters( h, param );
1948
1949 FAIL_IF_ERROR2( cli_output.set_param( opt->hout, param ), "can't set outfile param\n" );
1950
1951 i_start = x264_mdate();
1952
1953 /* ticks/frame = ticks/second / frames/second */
1954 ticks_per_frame = (int64_t)param->i_timebase_den * param->i_fps_den / param->i_timebase_num / param->i_fps_num;
1955 FAIL_IF_ERROR2( ticks_per_frame < 1 && !param->b_vfr_input, "ticks_per_frame invalid: %"PRId64"\n", ticks_per_frame );
1956 ticks_per_frame = X264_MAX( ticks_per_frame, 1 );
1957
1958 if( !param->b_repeat_headers )
1959 {
1960 // Write SPS/PPS/SEI
1961 x264_nal_t *headers;
1962 int i_nal;
1963
1964 FAIL_IF_ERROR2( x264_encoder_headers( h, &headers, &i_nal ) < 0, "x264_encoder_headers failed\n" );
1965 FAIL_IF_ERROR2( (i_file = cli_output.write_headers( opt->hout, headers )) < 0, "error writing headers to output file\n" );
1966 }
1967
1968 if( opt->tcfile_out )
1969 fprintf( opt->tcfile_out, "# timecode format v2\n" );
1970
1971 /* Encode frames */
1972 for( ; !b_ctrl_c && (i_frame < param->i_frame_total || !param->i_frame_total); i_frame++ )
1973 {
1974 if( filter.get_frame( opt->hin, &cli_pic, i_frame + opt->i_seek ) )
1975 break;
1976 x264_picture_init( &pic );
1977 convert_cli_to_lib_pic( &pic, &cli_pic );
1978
1979 if( !param->b_vfr_input )
1980 pic.i_pts = i_frame;
1981
1982 if( opt->i_pulldown && !param->b_vfr_input )
1983 {
1984 pic.i_pic_struct = pulldown->pattern[ i_frame % pulldown->mod ];
1985 pic.i_pts = (int64_t)( pulldown_pts + 0.5 );
1986 pulldown_pts += pulldown_frame_duration[pic.i_pic_struct];
1987 }
1988 else if( opt->timebase_convert_multiplier )
1989 pic.i_pts = (int64_t)( pic.i_pts * opt->timebase_convert_multiplier + 0.5 );
1990
1991 if( pic.i_pts <= largest_pts )
1992 {
1993 if( cli_log_level >= X264_LOG_DEBUG || pts_warning_cnt < MAX_PTS_WARNING )
1994 x264_cli_log( "x264", X264_LOG_WARNING, "non-strictly-monotonic pts at frame %d (%"PRId64" <= %"PRId64")\n",
1995 i_frame, pic.i_pts, largest_pts );
1996 else if( pts_warning_cnt == MAX_PTS_WARNING )
1997 x264_cli_log( "x264", X264_LOG_WARNING, "too many nonmonotonic pts warnings, suppressing further ones\n" );
1998 pts_warning_cnt++;
1999 pic.i_pts = largest_pts + ticks_per_frame;
2000 }
2001
2002 second_largest_pts = largest_pts;
2003 largest_pts = pic.i_pts;
2004 if( opt->tcfile_out )
2005 fprintf( opt->tcfile_out, "%.6f\n", pic.i_pts * ((double)param->i_timebase_num / param->i_timebase_den) * 1e3 );
2006
2007 if( opt->qpfile )
2008 parse_qpfile( opt, &pic, i_frame + opt->i_seek );
2009
2010 prev_dts = last_dts;
2011 i_frame_size = encode_frame( h, opt->hout, &pic, &last_dts );
2012 if( i_frame_size < 0 )
2013 {
2014 b_ctrl_c = 1; /* lie to exit the loop */
2015 retval = -1;
2016 }
2017 else if( i_frame_size )
2018 {
2019 i_file += i_frame_size;
2020 i_frame_output++;
2021 if( i_frame_output == 1 )
2022 first_dts = prev_dts = last_dts;
2023 }
2024
2025 if( filter.release_frame( opt->hin, &cli_pic, i_frame + opt->i_seek ) )
2026 break;
2027
2028 /* update status line (up to 1000 times per input file) */
2029 if( opt->b_progress && i_frame_output )
2030 i_previous = print_status( i_start, i_previous, i_frame_output, param->i_frame_total, i_file, param, 2 * last_dts - prev_dts - first_dts );
2031 }
2032 /* Flush delayed frames */
2033 while( !b_ctrl_c && x264_encoder_delayed_frames( h ) )
2034 {
2035 prev_dts = last_dts;
2036 i_frame_size = encode_frame( h, opt->hout, NULL, &last_dts );
2037 if( i_frame_size < 0 )
2038 {
2039 b_ctrl_c = 1; /* lie to exit the loop */
2040 retval = -1;
2041 }
2042 else if( i_frame_size )
2043 {
2044 i_file += i_frame_size;
2045 i_frame_output++;
2046 if( i_frame_output == 1 )
2047 first_dts = prev_dts = last_dts;
2048 }
2049 if( opt->b_progress && i_frame_output )
2050 i_previous = print_status( i_start, i_previous, i_frame_output, param->i_frame_total, i_file, param, 2 * last_dts - prev_dts - first_dts );
2051 }
2052fail:
2053 if( pts_warning_cnt >= MAX_PTS_WARNING && cli_log_level < X264_LOG_DEBUG )
2054 x264_cli_log( "x264", X264_LOG_WARNING, "%d suppressed nonmonotonic pts warnings\n", pts_warning_cnt-MAX_PTS_WARNING );
2055
2056 /* duration algorithm fails when only 1 frame is output */
2057 if( i_frame_output == 1 )
