29static int open_file( char *psz_filename, hnd_t *p_handle, cli_output_opt_t *opt )
30{
31 if( !strcmp( psz_filename, "-" ) )
32 *p_handle = stdout;
33 else if( !(*p_handle = x264_fopen( psz_filename, "w+b" )) )
34 return -1;
35
36 return 0;
37}
38
39static int set_param( hnd_t handle, x264_param_t *p_param )
40{
41 return 0;
42}
43
44static int write_headers( hnd_t handle, x264_nal_t *p_nal )
45{
46 int size = p_nal[0].i_payload + p_nal[1].i_payload + p_nal[2].i_payload;
47
48 if( fwrite( p_nal[0].p_payload, size, 1, (FILE*)handle ) )
49 return size;
50 return -1;
51}
52
53static int write_frame( hnd_t handle, uint8_t *p_nalu, int i_size, x264_picture_t *p_picture )
54{
55 if( fwrite( p_nalu, i_size, 1, (FILE*)handle ) )
56 return i_size;
57 return -1;
58}
59
60static int close_file( hnd_t handle, int64_t largest_pts, int64_t second_largest_pts )
61{
62 if( !handle || handle == stdout )
63 return 0;
64
65 return fclose( (FILE*)handle );
66}
67
68const cli_output_t raw_output = { open_file, set_param, write_headers, write_frame, close_file };
