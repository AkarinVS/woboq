26#include "video.h"
27
28#define NAME "select_every"
29#define FAIL_IF_ERROR( cond, ... ) FAIL_IF_ERR( cond, NAME, __VA_ARGS__ )
30
31#define MAX_PATTERN_SIZE 100 /* arbitrary */
32
33typedef struct
34{
35 hnd_t prev_hnd;
36 cli_vid_filter_t prev_filter;
37
38 int *pattern;
39 int pattern_len;
40 int step_size;
41 int vfr;
42 int64_t pts;
43} selvry_hnd_t;
44
45cli_vid_filter_t select_every_filter;
46
47static void help( int longhelp )
48{
49 printf( " "NAME":step,offset1[,...]\n" );
50 if( !longhelp )
51 return;
52 printf( " apply a selection pattern to input frames\n"
53 " step: the number of frames in the pattern\n"
54 " offsets: the offset into the step to select a frame\n"
55 " see: http://avisynth.nl/index.php/Select#SelectEvery\n" );
56}
57
58static int init( hnd_t *handle, cli_vid_filter_t *filter, video_info_t *info, x264_param_t *param, char *opt_string )
59{
60 selvry_hnd_t *h = malloc( sizeof(selvry_hnd_t) );
61 if( !h )
62 return -1;
63 h->pattern_len = 0;
64 h->step_size = 0;
65 int offsets[MAX_PATTERN_SIZE];
66 for( char *tok, *p = opt_string, UNUSED *saveptr = NULL; (tok = strtok_r( p, ",", &saveptr )); p = NULL )
67 {
68 int val = x264_otoi( tok, -1 );
69 if( p )
70 {
71 FAIL_IF_ERROR( val <= 0, "invalid step `%s'\n", tok );
72 h->step_size = val;
73 continue;
74 }
75 FAIL_IF_ERROR( val < 0 || val >= h->step_size, "invalid offset `%s'\n", tok );
76 FAIL_IF_ERROR( h->pattern_len >= MAX_PATTERN_SIZE, "max pattern size %d reached\n", MAX_PATTERN_SIZE );
77 offsets[h->pattern_len++] = val;
78 }
79 FAIL_IF_ERROR( !h->step_size, "no step size provided\n" );
80 FAIL_IF_ERROR( !h->pattern_len, "no offsets supplied\n" );
81
82 h->pattern = malloc( h->pattern_len * sizeof(int) );
83 if( !h->pattern )
84 return -1;
85 memcpy( h->pattern, offsets, h->pattern_len * sizeof(int) );
86
87 /* determine required cache size to maintain pattern. */
88 intptr_t max_rewind = 0;
89 int min = h->step_size;
90 for( int i = h->pattern_len-1; i >= 0; i-- )
91 {
92 min = X264_MIN( min, offsets[i] );
93 if( i )
94 max_rewind = X264_MAX( max_rewind, offsets[i-1] - min + 1 );
95 /* reached maximum rewind size */
96 if( max_rewind == h->step_size )
97 break;
98 }
99 char name[20];
100 sprintf( name, "cache_%d", param->i_bitdepth );
101 if( x264_init_vid_filter( name, handle, filter, info, param, (void*)max_rewind ) )
102 return -1;
103
104 /* done initing, overwrite properties */
105 if( h->step_size != h->pattern_len )
106 {
107 info->num_frames = (uint64_t)info->num_frames * h->pattern_len / h->step_size;
108 info->fps_den *= h->step_size;
109 info->fps_num *= h->pattern_len;
110 x264_reduce_fraction( &info->fps_num, &info->fps_den );
111 if( info->vfr )
112 {
113 info->timebase_den *= h->pattern_len;
114 info->timebase_num *= h->step_size;
115 x264_reduce_fraction( &info->timebase_num, &info->timebase_den );
116 }
117 }
118
119 h->pts = 0;
120 h->vfr = info->vfr;
121 h->prev_filter = *filter;
122 h->prev_hnd = *handle;
123 *filter = select_every_filter;
124 *handle = h;
125
126 return 0;
127}

129static int get_frame( hnd_t handle, cli_pic_t *output, int frame )
130{
131 selvry_hnd_t *h = handle;
132 int pat_frame = h->pattern[frame % h->pattern_len] + frame / h->pattern_len * h->step_size;
133 if( h->prev_filter.get_frame( h->prev_hnd, output, pat_frame ) )
134 return -1;
135 if( h->vfr )
136 {
137 output->pts = h->pts;
138 h->pts += output->duration;
139 }
140 return 0;
141}
142
143static int release_frame( hnd_t handle, cli_pic_t *pic, int frame )
144{
145 selvry_hnd_t *h = handle;
146 int pat_frame = h->pattern[frame % h->pattern_len] + frame / h->pattern_len * h->step_size;
147 return h->prev_filter.release_frame( h->prev_hnd, pic, pat_frame );
148}
149
150static void free_filter( hnd_t handle )
151{
152 selvry_hnd_t *h = handle;
153 h->prev_filter.free( h->prev_hnd );
154 free( h->pattern );
155 free( h );
156}
157
158cli_vid_filter_t select_every_filter = { NAME, help, init, get_frame, release_frame, free_filter, NULL };
