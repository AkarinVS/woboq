28#define FAIL_IF_ERROR( cond, ... ) FAIL_IF_ERR( cond, "x264", __VA_ARGS__ )

30void x264_cli_plane_copy( uint8_t *dst, int i_dst, uint8_t *src, int i_src, int w, int h )
31{
32 while( h-- )
33 {
34 memcpy( dst, src, w );
35 dst += i_dst;
36 src += i_src;
37 }
38}

40int x264_cli_pic_copy( cli_pic_t *out, cli_pic_t *in )
41{
42 int csp = in->img.csp & X264_CSP_MASK;
43 FAIL_IF_ERROR( x264_cli_csp_is_invalid( in->img.csp ), "invalid colorspace arg %d\n", in->img.csp );
44 FAIL_IF_ERROR( in->img.csp != out->img.csp || in->img.height != out->img.height
45 || in->img.width != out->img.width, "incompatible frame properties\n" );
46 /* copy data */
47 out->duration = in->duration;
48 out->pts = in->pts;
49 out->opaque = in->opaque;
50
51 for( int i = 0; i < out->img.planes; i++ )
52 {
53 int height = in->img.height * x264_cli_csps[csp].height[i];
54 int width = in->img.width * x264_cli_csps[csp].width[i];
55 width *= x264_cli_csp_depth_factor( in->img.csp );
56 x264_cli_plane_copy( out->img.plane[i], out->img.stride[i], in->img.plane[i],
57 in->img.stride[i], width, height );
58 }
59 return 0;
60}
