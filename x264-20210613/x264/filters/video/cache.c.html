30#define cache_filter x264_glue3(cache, BIT_DEPTH, filter)
31#if BIT_DEPTH == 8
32#define NAME "cache_8"
33#else
34#define NAME "cache_10"
35#endif
36
37#define LAST_FRAME (h->first_frame + h->cur_size - 1)
38
39typedef struct
40{
41 hnd_t prev_hnd;
42 cli_vid_filter_t prev_filter;
43
44 int max_size;
45 int first_frame; /* first cached frame */
46 cli_pic_t **cache;
47 int cur_size;
48 int eof; /* frame beyond end of the file */
49} cache_hnd_t;
50
51cli_vid_filter_t cache_filter;
52
53static int init( hnd_t *handle, cli_vid_filter_t *filter, video_info_t *info, x264_param_t *param, char *opt_string )
54{
55 intptr_t size = (intptr_t)opt_string;
56 /* upon a <= 0 cache request, do nothing */
57 if( size <= 0 )
58 return 0;
59 cache_hnd_t *h = calloc( 1, sizeof(cache_hnd_t) );
60 if( !h )
61 return -1;
62
63 h->max_size = size;
64 h->cache = malloc( (h->max_size+1) * sizeof(cli_pic_t*) );
65 if( !h->cache )
66 return -1;
67
68 for( int i = 0; i < h->max_size; i++ )
69 {
70 h->cache[i] = malloc( sizeof(cli_pic_t) );
71 if( !h->cache[i] || x264_cli_pic_alloc( h->cache[i], info->csp, info->width, info->height ) )
72 return -1;
73 }
74 h->cache[h->max_size] = NULL; /* require null terminator for list methods */
75
76 h->prev_filter = *filter;
77 h->prev_hnd = *handle;
78 *handle = h;
79 *filter = cache_filter;
80
81 return 0;
82}
83
84static void fill_cache( cache_hnd_t *h, int frame )
85{
86 /* shift frames out of the cache as the frame request is beyond the filled cache */
87 int shift = frame - LAST_FRAME;
88 /* no frames to shift or no frames left to read */
89 if( shift <= 0 || h->eof )
90 return;
91 /* the next frames to read are either
92 * A) starting at the end of the current cache, or
93 * B) starting at a new frame that has the end of the cache at the desired frame
94 * and proceeding to fill the entire cache */
95 int cur_frame = X264_MAX( h->first_frame + h->cur_size, frame - h->max_size + 1 );
96 /* the new starting point is either
97 * A) the current one shifted the number of frames entering/leaving the cache, or
98 * B) at a new frame that has the end of the cache at the desired frame. */
99 h->first_frame = X264_MIN( h->first_frame + shift, cur_frame );
100 h->cur_size = X264_MAX( h->cur_size - shift, 0 );
101 while( h->cur_size < h->max_size )
102 {
103 cli_pic_t temp;
104 /* the old front frame is going to shift off, overwrite it with the new frame */
105 cli_pic_t *cache = h->cache[0];
106 if( h->prev_filter.get_frame( h->prev_hnd, &temp, cur_frame ) ||
107 x264_cli_pic_copy( cache, &temp ) ||
108 h->prev_filter.release_frame( h->prev_hnd, &temp, cur_frame ) )
109 {
110 h->eof = cur_frame;
111 return;
112 }
113 /* the read was successful, shift the frame off the front to the end */
114 x264_frame_push( (void*)h->cache, x264_frame_shift( (void*)h->cache ) );
115 cur_frame++;
116 h->cur_size++;
117 }
118}
119
120static int get_frame( hnd_t handle, cli_pic_t *output, int frame )
121{
122 cache_hnd_t *h = handle;
123 FAIL_IF_ERR( frame < h->first_frame, NAME, "frame %d is before first cached frame %d \n", frame, h->first_frame );
124 fill_cache( h, frame );
125 if( frame > LAST_FRAME ) /* eof */
126 return -1;
127 int idx = frame - (h->eof ? h->eof - h->max_size : h->first_frame);
128 *output = *h->cache[idx];
129 return 0;
130}
131
132static int release_frame( hnd_t handle, cli_pic_t *pic, int frame )
133{
134 /* the parent filter's frame has already been released so do nothing here */
135 return 0;
136}
137
138static void free_filter( hnd_t handle )
139{
140 cache_hnd_t *h = handle;
141 h->prev_filter.free( h->prev_hnd );
142 for( int i = 0; i < h->max_size; i++ )
143 {
144 x264_cli_pic_clean( h->cache[i] );
145 free( h->cache[i] );
146 }
147 free( h->cache );
148 free( h );
149}
150
151cli_vid_filter_t cache_filter = { NAME, NULL, init, get_frame, release_frame, free_filter, NULL };
