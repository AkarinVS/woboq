51static int depth_filter_csp_is_supported( int csp )
52{
53 int csp_mask = csp & X264_CSP_MASK;
54 return csp_mask == X264_CSP_I400 ||
55 csp_mask == X264_CSP_I420 ||
56 csp_mask == X264_CSP_I422 ||
57 csp_mask == X264_CSP_I444 ||
58 csp_mask == X264_CSP_YV12 ||
59 csp_mask == X264_CSP_YV16 ||
60 csp_mask == X264_CSP_YV24 ||
61 csp_mask == X264_CSP_NV12 ||
62 csp_mask == X264_CSP_NV21 ||
63 csp_mask == X264_CSP_NV16 ||
64 csp_mask == X264_CSP_BGR ||
65 csp_mask == X264_CSP_RGB ||
66 csp_mask == X264_CSP_BGRA;
67}

69static int csp_num_interleaved( int csp, int plane )
70{
71 int csp_mask = csp & X264_CSP_MASK;
72 return (csp_mask == X264_CSP_NV12 || csp_mask == X264_CSP_NV21 || csp_mask == X264_CSP_NV16) && plane == 1 ? 2 :
73 csp_mask == X264_CSP_BGR || csp_mask == X264_CSP_RGB ? 3 :
74 csp_mask == X264_CSP_BGRA ? 4 :
75 1;
76}

78/* The dithering algorithm is based on Sierra-2-4A error diffusion. It has been
79 * written in such a way so that if the source has been upconverted using the
80 * same algorithm as used in scale_image, dithering down to the source bit
81 * depth again is lossless. */
82#define DITHER_PLANE( pitch ) \
83static void dither_plane_##pitch( pixel *dst, int dst_stride, uint16_t *src, int src_stride, \
84 int width, int height, int16_t *errors ) \
85{ \
86 const int lshift = 16-BIT_DEPTH; \
87 const int rshift = 16-BIT_DEPTH+2; \
88 const int half = 1 << (16-BIT_DEPTH+1); \
89 const int pixel_max = (1 << BIT_DEPTH)-1; \
90 memset( errors, 0, (width+1) * sizeof(int16_t) ); \
91 for( int y = 0; y < height; y++, src += src_stride, dst += dst_stride ) \
92 { \
93 int err = 0; \
94 for( int x = 0; x < width; x++ ) \
95 { \
96 err = err*2 + errors[x] + errors[x+1]; \
97 dst[x*pitch] = x264_clip3( ((src[x*pitch]<<2)+err+half) >> rshift, 0, pixel_max ); \
98 errors[x] = err = src[x*pitch] - (dst[x*pitch] << lshift); \
99 } \
100 } \
101}
102
103DITHER_PLANE( 1 )
104DITHER_PLANE( 2 )
105DITHER_PLANE( 3 )
106DITHER_PLANE( 4 )

108static void dither_image( cli_image_t *out, cli_image_t *img, int16_t *error_buf )
109{
110 int csp_mask = img->csp & X264_CSP_MASK;
111 for( int i = 0; i < img->planes; i++ )
112 {
113 int num_interleaved = csp_num_interleaved( img->csp, i );
114 int height = x264_cli_csps[csp_mask].height[i] * img->height;
115 int width = x264_cli_csps[csp_mask].width[i] * img->width / num_interleaved;
116
117#define CALL_DITHER_PLANE( pitch, off ) \
118 dither_plane_##pitch( ((pixel*)out->plane[i])+off, out->stride[i]/SIZEOF_PIXEL, \
119 ((uint16_t*)img->plane[i])+off, img->stride[i]/2, width, height, error_buf )
120
121 if( num_interleaved == 4 )
122 {
123 CALL_DITHER_PLANE( 4, 0 );
124 CALL_DITHER_PLANE( 4, 1 );
125 CALL_DITHER_PLANE( 4, 2 );
126 CALL_DITHER_PLANE( 4, 3 ); //we probably can skip this one
127 }
128 else if( num_interleaved == 3 )
129 {
130 CALL_DITHER_PLANE( 3, 0 );
131 CALL_DITHER_PLANE( 3, 1 );
132 CALL_DITHER_PLANE( 3, 2 );
133 }
134 else if( num_interleaved == 2 )
135 {
136 CALL_DITHER_PLANE( 2, 0 );
137 CALL_DITHER_PLANE( 2, 1 );
138 }
139 else //if( num_interleaved == 1 )
140 {
141 CALL_DITHER_PLANE( 1, 0 );
142 }
143 }
144}

146static void scale_image( cli_image_t *output, cli_image_t *img )
147{
148 int csp_mask = img->csp & X264_CSP_MASK;
149 const int shift = BIT_DEPTH - 8;
150 for( int i = 0; i < img->planes; i++ )
151 {
152 uint8_t *src = img->plane[i];
153 uint16_t *dst = (uint16_t*)output->plane[i];
154 int height = x264_cli_csps[csp_mask].height[i] * img->height;
155 int width = x264_cli_csps[csp_mask].width[i] * img->width;
156
157 for( int j = 0; j < height; j++ )
158 {
159 for( int k = 0; k < width; k++ )
160 dst[k] = src[k] << shift;
161
162 src += img->stride[i];
163 dst += output->stride[i]/2;
164 }
165 }
166}

168static int get_frame( hnd_t handle, cli_pic_t *output, int frame )
169{
170 depth_hnd_t *h = handle;
171
172 if( h->prev_filter.get_frame( h->prev_hnd, output, frame ) )
173 return -1;
174
175 if( h->bit_depth < 16 && output->img.csp & X264_CSP_HIGH_DEPTH )
176 {
177 dither_image( &h->buffer.img, &output->img, h->error_buf );
178 output->img = h->buffer.img;
179 }
180 else if( h->bit_depth > 8 && !(output->img.csp & X264_CSP_HIGH_DEPTH) )
181 {
182 scale_image( &h->buffer.img, &output->img );
183 output->img = h->buffer.img;
184 }
185 return 0;
186}

188static int release_frame( hnd_t handle, cli_pic_t *pic, int frame )
189{
190 depth_hnd_t *h = handle;
191 return h->prev_filter.release_frame( h->prev_hnd, pic, frame );
192}

194static void free_filter( hnd_t handle )
195{
196 depth_hnd_t *h = handle;
197 h->prev_filter.free( h->prev_hnd );
198 x264_cli_pic_clean( &h->buffer );
199 x264_free( h );
200}

202static int init( hnd_t *handle, cli_vid_filter_t *filter, video_info_t *info,
203 x264_param_t *param, char *opt_string )
204{
205 int ret = 0;
206 int change_fmt = (info->csp ^ param->i_csp) & X264_CSP_HIGH_DEPTH;
207 int csp = ~(~info->csp ^ change_fmt);
208 int bit_depth = 8*x264_cli_csp_depth_factor( csp );
209
210 if( opt_string )
211 {
212 static const char * const optlist[] = { "bit_depth", NULL };
213 char **opts = x264_split_options( opt_string, optlist );
214
215 if( opts )
216 {
217 char *str_bit_depth = x264_get_option( "bit_depth", opts );
218 bit_depth = x264_otoi( str_bit_depth, -1 );
219
220 ret = bit_depth < 8 || bit_depth > 16;
221 csp = bit_depth > 8 ? csp | X264_CSP_HIGH_DEPTH : csp & ~X264_CSP_HIGH_DEPTH;
222 change_fmt = (info->csp ^ csp) & X264_CSP_HIGH_DEPTH;
223 free( opts );
224 }
225 else
226 ret = 1;
227 }
228
229 FAIL_IF_ERROR( bit_depth != BIT_DEPTH, "this filter supports only bit depth %d\n", BIT_DEPTH );
230 FAIL_IF_ERROR( ret, "unsupported bit depth conversion.\n" );
231
232 /* only add the filter to the chain if it's needed */
233 if( change_fmt || bit_depth != 8 * x264_cli_csp_depth_factor( csp ) )
234 {
235 FAIL_IF_ERROR( !depth_filter_csp_is_supported(csp), "unsupported colorspace.\n" );
236 depth_hnd_t *h = x264_malloc( sizeof(depth_hnd_t) + (info->width+1)*sizeof(int16_t) );
237
238 if( !h )
239 return -1;
240
241 h->error_buf = (int16_t*)(h + 1);
242 h->dst_csp = csp;
243 h->bit_depth = bit_depth;
244 h->prev_hnd = *handle;
245 h->prev_filter = *filter;
246
247 if( x264_cli_pic_alloc( &h->buffer, h->dst_csp, info->width, info->height ) )
248 {
249 x264_free( h );
250 return -1;
251 }
252
253 *handle = h;
254 *filter = depth_filter;
255 info->csp = h->dst_csp;
256 }
257
258 return 0;
259}
