32char **x264_split_options( const char *opt_str, const char * const *options )
33{
34 int opt_count = 0, options_count = 0, found_named = 0;
35 size_t size = 0;
36 const char *opt = opt_str;
37
38 if( !opt_str )
39 return NULL;
40
41 while( options[options_count] )
42 options_count++;
43
44 do
45 {
46 size_t length = strcspn( opt, "=," );
47 if( opt[length] == '=' )
48 {
49 const char * const *option = options;
50 while( *option && (strlen( *option ) != length || strncmp( opt, *option, length )) )
51 option++;
52
53 RETURN_IF_ERROR( !*option, "Invalid option '%.*s'\n", length, opt );
54 found_named = 1;
55 length += strcspn( opt + length, "," );
56 }
57 else
58 {
59 RETURN_IF_ERROR( opt_count >= options_count, "Too many options given\n" );
60 RETURN_IF_ERROR( found_named, "Ordered option given after named\n" );
61 size += strlen( options[opt_count] ) + 1;
62 }
63 opt_count++;
64 opt += length;
65 } while( *opt++ );
66
67 size_t offset = 2 * (opt_count+1) * sizeof(char*);
68 size += offset + (opt - opt_str);
69 char **opts = calloc( 1, size );
70 RETURN_IF_ERROR( !opts, "malloc failed\n" );
71
72#define insert_opt( src, length )\
73do {\
74 opts[i++] = memcpy( (char*)opts + offset, src, length );\
75 offset += length + 1;\
76 src += length + 1;\
77} while( 0 )
78
79 for( int i = 0; i < 2*opt_count; )
80 {
81 size_t length = strcspn( opt_str, "=," );
82 if( opt_str[length] == '=' )
83 {
84 insert_opt( opt_str, length );
85 length = strcspn( opt_str, "," );
86 }
87 else
88 {
89 const char *option = options[i/2];
90 size_t option_length = strlen( option );
91 insert_opt( option, option_length );
92 }
93 insert_opt( opt_str, length );
94 }
95
96 assert( offset == size );
97 return opts;
98}

100char *x264_get_option( const char *name, char **split_options )
101{
102 if( split_options )
103 {
104 int last_i = -1;
105 for( int i = 0; split_options[i]; i += 2 )
106 if( !strcmp( split_options[i], name ) )
107 last_i = i;
108 if( last_i >= 0 && split_options[last_i+1][0] )
109 return split_options[last_i+1];
110 }
111 return NULL;
112}

114int x264_otob( const char *str, int def )
115{
116 if( str )
117 return !strcasecmp( str, "true" ) || !strcmp( str, "1" ) || !strcasecmp( str, "yes" );
118 return def;
119}

121double x264_otof( const char *str, double def )
122{
123 double ret = def;
124 if( str )
125 {
126 char *end;
127 ret = strtod( str, &end );
128 if( end == str || *end != '\0' )
129 ret = def;
130 }
131 return ret;
132}
133
134int x264_otoi( const char *str, int def )
135{
136 int ret = def;
137 if( str )
138 {
139 char *end;
140 ret = strtol( str, &end, 0 );
141 if( end == str || *end != '\0' )
142 ret = def;
143 }
144 return ret;
145}
146
147char *x264_otos( char *str, char *def )
148{
149 return str ? str : def;
150}
