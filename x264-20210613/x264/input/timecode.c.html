28#define FAIL_IF_ERROR( cond, ... ) FAIL_IF_ERR( cond, "timecode", __VA_ARGS__ )

30typedef struct
31{
32 cli_input_t input;
33 hnd_t p_handle;
34 int auto_timebase_num;
35 int auto_timebase_den;
36 uint64_t timebase_num;
37 uint64_t timebase_den;
38 int stored_pts_num;
39 int64_t *pts;
40 double assume_fps;
41 double last_timecode;
42} timecode_hnd_t;

44static inline double sigexp10( double value, double *exponent )
45{
46 /* This function separates significand and exp10 from double floating point. */
47 *exponent = pow( 10, floor( log10( value ) ) );
48 return value / *exponent;
49}
50
51#define DOUBLE_EPSILON 5e-6
52#define MKV_TIMEBASE_DEN 1000000000
53
54static double correct_fps( double fps, timecode_hnd_t *h )
55{
56 int i = 1;
57 uint64_t fps_num, fps_den;
58 double exponent;
59 double fps_sig = sigexp10( fps, &exponent );
60 while( 1 )
61 {
62 fps_den = i * h->timebase_num;
63 fps_num = round( fps_den * fps_sig ) * exponent;
64 FAIL_IF_ERROR( fps_num > UINT32_MAX, "tcfile fps correction failed.\n"
65 " Specify an appropriate timebase manually or remake tcfile.\n" );
66 if( fabs( ((double)fps_num / fps_den) / exponent - fps_sig ) < DOUBLE_EPSILON )
67 break;
68 ++i;
69 }
70 if( h->auto_timebase_den )
71 {
72 h->timebase_den = h->timebase_den ? lcm( h->timebase_den, fps_num ) : fps_num;
73 if( h->timebase_den > UINT32_MAX )
74 h->auto_timebase_den = 0;
75 }
76 return (double)fps_num / fps_den;
77}

79static int try_mkv_timebase_den( double *fpss, timecode_hnd_t *h, int loop_num )
80{
81 h->timebase_num = 0;
82 h->timebase_den = MKV_TIMEBASE_DEN;
83 for( int num = 0; num < loop_num; num++ )
84 {
85 uint64_t fps_den;
86 double exponent;
87 double fps_sig = sigexp10( fpss[num], &exponent );
88 fps_den = round( MKV_TIMEBASE_DEN / fps_sig ) / exponent;
89 h->timebase_num = fps_den && h->timebase_num ? gcd( h->timebase_num, fps_den ) : fps_den;
90 FAIL_IF_ERROR( h->timebase_num > UINT32_MAX || !h->timebase_num, "automatic timebase generation failed.\n"
91 " Specify timebase manually.\n" );
92 }
93 return 0;
94}

96static int parse_tcfile( FILE *tcfile_in, timecode_hnd_t *h, video_info_t *info )
97{
98 char buff[256];
99 int ret, tcfv, num, seq_num, timecodes_num;
100 double *timecodes = NULL;
101 double *fpss = NULL;
102
103 ret = fgets( buff, sizeof(buff), tcfile_in ) != NULL &&
104 (sscanf( buff, "# timecode format v%d", &tcfv ) == 1 || sscanf( buff, "# timestamp format v%d", &tcfv ) == 1);
105 FAIL_IF_ERROR( !ret || (tcfv != 1 && tcfv != 2), "unsupported timecode format\n" );
106#define NO_TIMECODE_LINE (buff[0] == '#' || buff[0] == '\n' || buff[0] == '\r')
107 if( tcfv == 1 )
108 {
109 int64_t file_pos;
110 double assume_fps, seq_fps;
111 int start, end = -1;
112 int prev_start = -1, prev_end = -1;
113
114 h->assume_fps = 0;
115 for( num = 2; fgets( buff, sizeof(buff), tcfile_in ) != NULL; num++ )
116 {
117 if( NO_TIMECODE_LINE )
118 continue;
119 FAIL_IF_ERROR( sscanf( buff, "assume %lf", &h->assume_fps ) != 1 && sscanf( buff, "Assume %lf", &h->assume_fps ) != 1,
120 "tcfile parsing error: assumed fps not found\n" );
121 break;
122 }
123 FAIL_IF_ERROR( h->assume_fps <= 0, "invalid assumed fps %.6f\n", h->assume_fps );
124
125 file_pos = ftell( tcfile_in );
126 h->stored_pts_num = 0;
127 for( seq_num = 0; fgets( buff, sizeof(buff), tcfile_in ) != NULL; num++ )
128 {
129 if( NO_TIMECODE_LINE )
130 {
131 if( sscanf( buff, "# TDecimate Mode 3: Last Frame = %d", &end ) == 1 )
132 h->stored_pts_num = end + 1;
133 continue;
134 }
135 ret = sscanf( buff, "%d,%d,%lf", &start, &end, &seq_fps );
136 FAIL_IF_ERROR( ret != 3 && ret != EOF, "invalid input tcfile\n" );
137 FAIL_IF_ERROR( start > end || start <= prev_start || end <= prev_end || seq_fps <= 0,
138 "invalid input tcfile at line %d: %s\n", num, buff );
139 prev_start = start;
140 prev_end = end;
141 if( h->auto_timebase_den || h->auto_timebase_num )
142 ++seq_num;
143 }
144 if( !h->stored_pts_num )
145 h->stored_pts_num = end + 2;
146 timecodes_num = h->stored_pts_num;
147 fseek( tcfile_in, file_pos, SEEK_SET );
148
149 timecodes = malloc( timecodes_num * sizeof(double) );
150 if( !timecodes )
151 return -1;
152 if( h->auto_timebase_den || h->auto_timebase_num )
153 {
154 fpss = malloc( (seq_num + 1) * sizeof(double) );
155 if( !fpss )
156 goto fail;
157 }
158
159 assume_fps = correct_fps( h->assume_fps, h );
160 if( assume_fps < 0 )
161 goto fail;
162 timecodes[0] = 0;
163 for( num = seq_num = 0; num < timecodes_num - 1 && fgets( buff, sizeof(buff), tcfile_in ) != NULL; )
164 {
165 if( NO_TIMECODE_LINE )
166 continue;
167 ret = sscanf( buff, "%d,%d,%lf", &start, &end, &seq_fps );
168 if( ret != 3 )
169 start = end = timecodes_num - 1;
170 for( ; num < start && num < timecodes_num - 1; num++ )
171 timecodes[num + 1] = timecodes[num] + 1 / assume_fps;
172 if( num < timecodes_num - 1 )
173 {
174 if( h->auto_timebase_den || h->auto_timebase_num )
175 fpss[seq_num++] = seq_fps;
176 seq_fps = correct_fps( seq_fps, h );
177 if( seq_fps < 0 )
178 goto fail;
179 for( num = start; num <= end && num < timecodes_num - 1; num++ )
180 timecodes[num + 1] = timecodes[num] + 1 / seq_fps;
181 }
182 }
183 for( ; num < timecodes_num - 1; num++ )
184 timecodes[num + 1] = timecodes[num] + 1 / assume_fps;
185 if( h->auto_timebase_den || h->auto_timebase_num )
186 fpss[seq_num] = h->assume_fps;
187
188 if( h->auto_timebase_num && !h->auto_timebase_den )
189 {
190 double exponent;
191 double assume_fps_sig, seq_fps_sig;
192 if( try_mkv_timebase_den( fpss, h, seq_num + 1 ) < 0 )
193 goto fail;
194 fseek( tcfile_in, file_pos, SEEK_SET );
195 assume_fps_sig = sigexp10( h->assume_fps, &exponent );
196 assume_fps = MKV_TIMEBASE_DEN / ( round( MKV_TIMEBASE_DEN / assume_fps_sig ) / exponent );
197 for( num = 0; num < timecodes_num - 1 && fgets( buff, sizeof(buff), tcfile_in ) != NULL; )
198 {
199 if( NO_TIMECODE_LINE )
200 continue;
201 ret = sscanf( buff, "%d,%d,%lf", &start, &end, &seq_fps );
202 if( ret != 3 )
203 start = end = timecodes_num - 1;
204 seq_fps_sig = sigexp10( seq_fps, &exponent );
205 seq_fps = MKV_TIMEBASE_DEN / ( round( MKV_TIMEBASE_DEN / seq_fps_sig ) / exponent );
206 for( ; num < start && num < timecodes_num - 1; num++ )
207 timecodes[num + 1] = timecodes[num] + 1 / assume_fps;
208 for( num = start; num <= end && num < timecodes_num - 1; num++ )
209 timecodes[num + 1] = timecodes[num] + 1 / seq_fps;
210 }
211 for( ; num < timecodes_num - 1; num++ )
212 timecodes[num + 1] = timecodes[num] + 1 / assume_fps;
213 }
214 if( fpss )
215 {
216 free( fpss );
217 fpss = NULL;
218 }
219
220 h->assume_fps = assume_fps;
221 h->last_timecode = timecodes[timecodes_num - 1];
222 }
223 else /* tcfv == 2 */
224 {
225 int64_t file_pos = ftell( tcfile_in );
226
227 h->stored_pts_num = 0;
228 while( fgets( buff, sizeof(buff), tcfile_in ) != NULL )
229 {
230 if( NO_TIMECODE_LINE )
231 {
232 if( !h->stored_pts_num )
233 file_pos = ftell( tcfile_in );
234 continue;
235 }
236 h->stored_pts_num++;
237 }
238 timecodes_num = h->stored_pts_num;
239 FAIL_IF_ERROR( !timecodes_num, "input tcfile doesn't have any timecodes!\n" );
240 fseek( tcfile_in, file_pos, SEEK_SET );
241
242 timecodes = malloc( timecodes_num * sizeof(double) );
243 if( !timecodes )
244 return -1;
245
246 num = 0;
247 if( fgets( buff, sizeof(buff), tcfile_in ) != NULL )
248 {
249 ret = sscanf( buff, "%lf", &timecodes[0] );
250 timecodes[0] *= 1e-3; /* Timecode format v2 is expressed in milliseconds. */
251 FAIL_IF_ERROR( ret != 1, "invalid input tcfile for frame 0\n" );
252 for( num = 1; num < timecodes_num && fgets( buff, sizeof(buff), tcfile_in ) != NULL; )
253 {
254 if( NO_TIMECODE_LINE )
255 continue;
256 ret = sscanf( buff, "%lf", &timecodes[num] );
257 timecodes[num] *= 1e-3; /* Timecode format v2 is expressed in milliseconds. */
258 FAIL_IF_ERROR( ret != 1 || timecodes[num] <= timecodes[num - 1],
259 "invalid input tcfile for frame %d\n", num );
260 ++num;
261 }
262 }
263 FAIL_IF_ERROR( num < timecodes_num, "failed to read input tcfile for frame %d", num );
264
265 if( timecodes_num == 1 )
266 h->timebase_den = info->fps_num;
267 else if( h->auto_timebase_den )
268 {
269 fpss = malloc( (timecodes_num - 1) * sizeof(double) );
270 if( !fpss )
271 goto fail;
272 for( num = 0; num < timecodes_num - 1; num++ )
273 {
274 fpss[num] = 1 / (timecodes[num + 1] - timecodes[num]);
275 if( h->auto_timebase_den )
276 {
277 int i = 1;
278 uint64_t fps_num, fps_den;
279 double exponent;
280 double fps_sig = sigexp10( fpss[num], &exponent );
281 while( 1 )
282 {
283 fps_den = i * h->timebase_num;
284 fps_num = round( fps_den * fps_sig ) * exponent;
285 if( fps_num > UINT32_MAX || fabs( ((double)fps_num / fps_den) / exponent - fps_sig ) < DOUBLE_EPSILON )
286 break;
287 ++i;
288 }
289 h->timebase_den = fps_num && h->timebase_den ? lcm( h->timebase_den, fps_num ) : fps_num;
290 if( h->timebase_den > UINT32_MAX )
291 {
292 h->auto_timebase_den = 0;
293 continue;
294 }
295 }
296 }
297 if( h->auto_timebase_num && !h->auto_timebase_den )
298 if( try_mkv_timebase_den( fpss, h, timecodes_num - 1 ) < 0 )
299 goto fail;
300 free( fpss );
301 fpss = NULL;
302 }
303
304 if( timecodes_num > 1 )
305 h->assume_fps = 1 / (timecodes[timecodes_num - 1] - timecodes[timecodes_num - 2]);
306 else
307 h->assume_fps = (double)info->fps_num / info->fps_den;
308 h->last_timecode = timecodes[timecodes_num - 1];
309 }
310#undef NO_TIMECODE_LINE
311 if( h->auto_timebase_den || h->auto_timebase_num )
312 {
313 uint64_t i = gcd( h->timebase_num, h->timebase_den );
314 h->timebase_num /= i;
315 h->timebase_den /= i;
316 x264_cli_log( "timecode", X264_LOG_INFO, "automatic timebase generation %"PRIu64"/%"PRIu64"\n", h->timebase_num, h->timebase_den );
317 }
318 else FAIL_IF_ERROR( h->timebase_den > UINT32_MAX || !h->timebase_den, "automatic timebase generation failed.\n"
319 " Specify an appropriate timebase manually.\n" );
320
321 h->pts = malloc( h->stored_pts_num * sizeof(int64_t) );
322 if( !h->pts )
323 goto fail;
324 for( num = 0; num < h->stored_pts_num; num++ )
325 {
326 h->pts[num] = timecodes[num] * ((double)h->timebase_den / h->timebase_num) + 0.5;
327 FAIL_IF_ERROR( num > 0 && h->pts[num] <= h->pts[num - 1], "invalid timebase or timecode for frame %d\n", num );
328 }
329
330 free( timecodes );
331 return 0;
332
333fail:
334 if( timecodes )
335 free( timecodes );
336 if( fpss )
337 free( fpss );
338 return -1;
339}

344static int open_file( char *psz_filename, hnd_t *p_handle, video_info_t *info, cli_input_opt_t *opt )
345{
346 int ret = 0;
347 FILE *tcfile_in;
348 timecode_hnd_t *h = malloc( sizeof(timecode_hnd_t) );
349 FAIL_IF_ERROR( !h, "malloc failed\n" );
350 h->input = cli_input;
351 h->p_handle = *p_handle;
352 h->pts = NULL;
353 if( opt->timebase )
354 {
355 ret = sscanf( opt->timebase, "%"SCNu64"/%"SCNu64, &h->timebase_num, &h->timebase_den );
356 if( ret == 1 )
357 {
358 h->timebase_num = strtoul( opt->timebase, NULL, 10 );
359 h->timebase_den = 0; /* set later by auto timebase generation */
360 }
361 FAIL_IF_ERROR( h->timebase_num > UINT32_MAX || h->timebase_den > UINT32_MAX,
362 "timebase you specified exceeds H.264 maximum\n" );
363 }
364 h->auto_timebase_num = !ret;
365 h->auto_timebase_den = ret < 2;
366 if( h->auto_timebase_num )
367 h->timebase_num = info->fps_den; /* can be changed later by auto timebase generation */
368 if( h->auto_timebase_den )
369 h->timebase_den = 0; /* set later by auto timebase generation */
370
371 tcfile_in = x264_fopen( psz_filename, "rb" );
372 FAIL_IF_ERROR( !tcfile_in, "can't open `%s'\n", psz_filename );
373 if( !x264_is_regular_file( tcfile_in ) )
374 {
375 x264_cli_log( "timecode", X264_LOG_ERROR, "tcfile input incompatible with non-regular file `%s'\n", psz_filename );
376 fclose( tcfile_in );
377 return -1;
378 }
379
380 if( parse_tcfile( tcfile_in, h, info ) < 0 )
381 {
382 if( h->pts )
383 free( h->pts );
384 fclose( tcfile_in );
385 return -1;
386 }
387 fclose( tcfile_in );
388
389 info->timebase_num = h->timebase_num;
390 info->timebase_den = h->timebase_den;
391 info->vfr = 1;
392
393 *p_handle = h;
394 return 0;
395}

397static int64_t get_frame_pts( timecode_hnd_t *h, int frame, int real_frame )
398{
399 if( frame < h->stored_pts_num )
400 return h->pts[frame];
401 else
402 {
403 if( h->pts && real_frame )
404 {
405 x264_cli_log( "timecode", X264_LOG_INFO, "input timecode file missing data for frame %d and later\n"
406 " assuming constant fps %.6f\n", frame, h->assume_fps );
407 free( h->pts );
408 h->pts = NULL;
409 }
410 double timecode = h->last_timecode + 1 / h->assume_fps;
411 if( real_frame )
412 h->last_timecode = timecode;
413 return timecode * ((double)h->timebase_den / h->timebase_num) + 0.5;
414 }
415}

417static int read_frame( cli_pic_t *pic, hnd_t handle, int frame )
418{
419 timecode_hnd_t *h = handle;
420 if( h->input.read_frame( pic, h->p_handle, frame ) )
421 return -1;
422
423 pic->pts = get_frame_pts( h, frame, 1 );
424 pic->duration = get_frame_pts( h, frame + 1, 0 ) - pic->pts;
425
426 return 0;
427}
