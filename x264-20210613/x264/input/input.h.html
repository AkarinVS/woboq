37/* options that are used by only some demuxers */
38typedef struct
39{
40 char *index_file;
41 char *format;
42 char *resolution;
43 char *colorspace;
44 int bit_depth;
45 char *timebase;
46 int seek;
47 int progress;
48 int output_csp; /* convert to this csp, if applicable */
49 int output_range; /* user desired output range */
50 int input_range; /* user override input range */
51} cli_input_opt_t;
52
53/* properties of the source given by the demuxer */
54typedef struct
55{
56 int csp; /* colorspace of the input */
57 uint32_t fps_num;
58 uint32_t fps_den;
59 int fullrange; /* has 2^bit_depth-1 instead of 219*2^(bit_depth-8) ranges (YUV only) */
60 int width;
61 int height;
62 int interlaced;
63 int num_frames;
64 uint32_t sar_width;
65 uint32_t sar_height;
66 int tff;
67 int thread_safe; /* demuxer is thread_input safe */
68 uint32_t timebase_num;
69 uint32_t timebase_den;
70 int vfr;
71} video_info_t;
72
73/* image data type used by x264cli */
74typedef struct
75{
76 int csp; /* colorspace */
77 int width; /* width of the picture */
78 int height; /* height of the picture */
79 int planes; /* number of planes */
80 uint8_t *plane[4]; /* pointers for each plane */
81 int stride[4]; /* strides for each plane */
82} cli_image_t;
83
84typedef struct
85{
86 cli_image_t img;
87 int64_t pts; /* input pts */
88 int64_t duration; /* frame duration - used for vfr */
89 void *opaque; /* opaque handle */
90} cli_pic_t;
91
92typedef struct
93{
94 int (*open_file)( char *psz_filename, hnd_t *p_handle, video_info_t *info, cli_input_opt_t *opt );
95 int (*picture_alloc)( cli_pic_t *pic, hnd_t handle, int csp, int width, int height );
96 int (*read_frame)( cli_pic_t *pic, hnd_t handle, int i_frame );
97 int (*release_frame)( cli_pic_t *pic, hnd_t handle );
98 void (*picture_clean)( cli_pic_t *pic, hnd_t handle );
99 int (*close_file)( hnd_t handle );
100} cli_input_t;
