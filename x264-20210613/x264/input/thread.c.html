27#include "input.h"
28#include "common/common.h"
29
30#define thread_input x264_glue3(thread, BIT_DEPTH, input)
31
32typedef struct
33{
34 cli_input_t input;
35 hnd_t p_handle;
36 cli_pic_t pic;
37 x264_threadpool_t *pool;
38 int next_frame;
39 int frame_total;
40 struct thread_input_arg_t *next_args;
41} thread_hnd_t;
42
43typedef struct thread_input_arg_t
44{
45 thread_hnd_t *h;
46 cli_pic_t *pic;
47 int i_frame;
48 int status;
49} thread_input_arg_t;
50
51static int open_file( char *psz_filename, hnd_t *p_handle, video_info_t *info, cli_input_opt_t *opt )
52{
53 thread_hnd_t *h = malloc( sizeof(thread_hnd_t) );
54 FAIL_IF_ERR( !h || cli_input.picture_alloc( &h->pic, *p_handle, info->csp, info->width, info->height ),
55 "x264", "malloc failed\n" );
56 h->input = cli_input;
57 h->p_handle = *p_handle;
58 h->next_frame = -1;
59 h->next_args = malloc( sizeof(thread_input_arg_t) );
60 if( !h->next_args )
61 return -1;
62 h->next_args->h = h;
63 h->next_args->status = 0;
64 h->frame_total = info->num_frames;
65
66 if( x264_threadpool_init( &h->pool, 1, NULL, NULL ) )
67 return -1;
68
69 *p_handle = h;
70 return 0;
71}
72
73static void read_frame_thread_int( thread_input_arg_t *i )
74{
75 i->status = i->h->input.read_frame( i->pic, i->h->p_handle, i->i_frame );
76}
77
78static int read_frame( cli_pic_t *p_pic, hnd_t handle, int i_frame )
79{
80 thread_hnd_t *h = handle;
81 int ret = 0;
82
83 if( h->next_frame >= 0 )
84 {
85 x264_threadpool_wait( h->pool, h->next_args );
86 ret |= h->next_args->status;
87 }
88
89 if( h->next_frame == i_frame )
90 XCHG( cli_pic_t, *p_pic, h->pic );
91 else
92 {
93 if( h->next_frame >= 0 )
94 thread_input.release_frame( &h->pic, handle );
95 ret |= h->input.read_frame( p_pic, h->p_handle, i_frame );
96 }
97
98 if( !h->frame_total || i_frame+1 < h->frame_total )
99 {
100 h->next_frame =
101 h->next_args->i_frame = i_frame+1;
102 h->next_args->pic = &h->pic;
103 x264_threadpool_run( h->pool, (void*)read_frame_thread_int, h->next_args );
104 }
105 else
106 h->next_frame = -1;
107
108 return ret;
109}
110
111static int release_frame( cli_pic_t *pic, hnd_t handle )
112{
113 thread_hnd_t *h = handle;
114 if( h->input.release_frame )
115 return h->input.release_frame( pic, h->p_handle );
116 return 0;
117}
118
119static int picture_alloc( cli_pic_t *pic, hnd_t handle, int csp, int width, int height )
120{
121 thread_hnd_t *h = handle;
122 return h->input.picture_alloc( pic, h->p_handle, csp, width, height );
123}
124
125static void picture_clean( cli_pic_t *pic, hnd_t handle )
126{
127 thread_hnd_t *h = handle;
128 h->input.picture_clean( pic, h->p_handle );
129}
130
131static int close_file( hnd_t handle )
132{
133 thread_hnd_t *h = handle;
134 x264_threadpool_delete( h->pool );
135 h->input.picture_clean( &h->pic, h->p_handle );
136 h->input.close_file( h->p_handle );
137 free( h->next_args );
138 free( h );
139 return 0;
140}
141
142const cli_input_t thread_input = { open_file, picture_alloc, read_frame, release_frame, picture_clean, close_file };
