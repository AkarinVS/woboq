66#define LOAD_AVS_FUNC(name, continue_on_fail)\
67{\
68 h->func.name = (void*)avs_address( h->library, #name );\
69 if( !continue_on_fail && !h->func.name )\
70 goto fail;\
71}
72
73#define LOAD_AVS_FUNC_ALIAS(name, alias, continue_on_fail)\
74{\
75 if( !h->func.name )\
76 h->func.name = (void*)avs_address( h->library, alias );\
77 if( !continue_on_fail && !h->func.name )\
78 goto fail;\
79}

127/* load the library and functions we require from it */
128static int custom_avs_load_library( avs_hnd_t *h )
129{
130 h->library = avs_open();
131 if( !h->library )
132 return -1;
133 LOAD_AVS_FUNC( avs_clip_get_error, 0 );
134 LOAD_AVS_FUNC( avs_create_script_environment, 0 );
135 LOAD_AVS_FUNC( avs_delete_script_environment, 1 );
136 LOAD_AVS_FUNC( avs_get_error, 1 );
137 LOAD_AVS_FUNC( avs_get_frame, 0 );
138 LOAD_AVS_FUNC( avs_get_video_info, 0 );
139 LOAD_AVS_FUNC( avs_function_exists, 0 );
140 LOAD_AVS_FUNC( avs_invoke, 0 );
141 LOAD_AVS_FUNC( avs_release_clip, 0 );
142 LOAD_AVS_FUNC( avs_release_value, 0 );
143 LOAD_AVS_FUNC( avs_release_video_frame, 0 );
144 LOAD_AVS_FUNC( avs_take_clip, 0 );
145#if !USE_AVXSYNTH
146 // AviSynth+ extension
147 LOAD_AVS_FUNC( avs_is_rgb48, 1 );
148 LOAD_AVS_FUNC_ALIAS( avs_is_rgb48, "_avs_is_rgb48@4", 1 );
149 LOAD_AVS_FUNC( avs_is_rgb64, 1 );
150 LOAD_AVS_FUNC_ALIAS( avs_is_rgb64, "_avs_is_rgb64@4", 1 );
151 LOAD_AVS_FUNC( avs_is_yuv444p16, 1 );
152 LOAD_AVS_FUNC( avs_is_yuv422p16, 1 );
153 LOAD_AVS_FUNC( avs_is_yuv420p16, 1 );
154 LOAD_AVS_FUNC( avs_is_y16, 1 );
155 LOAD_AVS_FUNC( avs_is_yuv444ps, 1 );
156 LOAD_AVS_FUNC( avs_is_yuv422ps, 1 );
157 LOAD_AVS_FUNC( avs_is_yuv420ps, 1 );
158 LOAD_AVS_FUNC( avs_is_y32, 1 );
159 LOAD_AVS_FUNC( avs_is_444, 1 );
160 LOAD_AVS_FUNC( avs_is_422, 1 );
161 LOAD_AVS_FUNC( avs_is_420, 1 );
162 LOAD_AVS_FUNC( avs_is_y, 1 );
163 LOAD_AVS_FUNC( avs_is_yuva, 1 );
164 LOAD_AVS_FUNC( avs_is_planar_rgb, 1 );
165 LOAD_AVS_FUNC( avs_is_planar_rgba, 1 );
166 LOAD_AVS_FUNC( avs_num_components, 1 );
167 LOAD_AVS_FUNC( avs_component_size, 1 );
168 LOAD_AVS_FUNC( avs_bits_per_component, 1 );
169#endif
170 return 0;
171fail:
172 avs_close( h->library );
173 h->library = NULL;
174 return -1;
175}

209/* generate a filter sequence to try based on the filename extension */
210static void avs_build_filter_sequence( char *filename_ext, const char *filter[AVS_MAX_SEQUENCE+1] )
211{
212 int i = 0;
213#if USE_AVXSYNTH
214 const char *all_purpose[] = { "FFVideoSource", 0 };
215#else
216 const char *all_purpose[] = { "FFmpegSource2", "DSS2", "DirectShowSource", 0 };
217 if( !strcasecmp( filename_ext, "avi" ) )
218 filter[i++] = "AVISource";
219 if( !strcasecmp( filename_ext, "d2v" ) )
220 filter[i++] = "MPEG2Source";
221 if( !strcasecmp( filename_ext, "dga" ) )
222 filter[i++] = "AVCSource";
223#endif
224 for( int j = 0; all_purpose[j] && i < AVS_MAX_SEQUENCE; j++ )
225 filter[i++] = all_purpose[j];
226}

228static AVS_Value update_clip( avs_hnd_t *h, const AVS_VideoInfo **vi, AVS_Value res, AVS_Value release )
229{
230 h->func.avs_release_clip( h->clip );
231 h->clip = h->func.avs_take_clip( res, h->env );
232 h->func.avs_release_value( release );
233 *vi = h->func.avs_get_video_info( h->clip );
234 return res;
235}

237static float get_avs_version( avs_hnd_t *h )
238{
239/* AvxSynth has its version defined starting at 4.0, even though it's based on
240 AviSynth 2.5.8. This is troublesome for get_avs_version and working around
241 the new colorspaces in 2.6. So if AvxSynth is detected, explicitly define
242 the version as 2.58. */
243#if USE_AVXSYNTH
244 return 2.58f;
245#else
246 FAIL_IF_ERROR( !h->func.avs_function_exists( h->env, "VersionNumber" ), "VersionNumber does not exist\n" );
247 AVS_Value ver = h->func.avs_invoke( h->env, "VersionNumber", avs_new_value_array( NULL, 0 ), NULL );
248 FAIL_IF_ERROR( avs_is_error( ver ), "unable to determine avisynth version: %s\n", avs_as_error( ver ) );
249 FAIL_IF_ERROR( !avs_is_float( ver ), "VersionNumber did not return a float value\n" );
250 float ret = avs_as_float( ver );
251 h->func.avs_release_value( ver );
252 return ret;
253#endif
254}

256#ifdef _WIN32
257static char *utf16_to_ansi( const wchar_t *utf16 )
258{
259 BOOL invalid;
260 int len = WideCharToMultiByte( CP_ACP, WC_NO_BEST_FIT_CHARS, utf16, -1, NULL, 0, NULL, &invalid );
261 if( len && !invalid )
262 {
263 char *ansi = malloc( len * sizeof( char ) );
264 if( ansi )
265 {
266 if( WideCharToMultiByte( CP_ACP, WC_NO_BEST_FIT_CHARS, utf16, -1, ansi, len, NULL, &invalid ) && !invalid )
267 return ansi;
268 free( ansi );
269 }
270 }
271 return NULL;
272}

274static char *utf8_to_ansi( const char *filename )
275{
276 char *ansi = NULL;
277 wchar_t *filename_utf16 = x264_utf8_to_utf16( filename );
278 if( filename_utf16 )
279 {
280 /* Check if the filename already is valid ANSI. */
281 if( !(ansi = utf16_to_ansi( filename_utf16 )) )
282 {
283 /* Check for a legacy 8.3 short filename. */
284 int len = GetShortPathNameW( filename_utf16, NULL, 0 );
285 if( len )
286 {
287 wchar_t *short_utf16 = malloc( len * sizeof( wchar_t ) );
288 if( short_utf16 )
289 {
290 if( GetShortPathNameW( filename_utf16, short_utf16, len ) )
291 ansi = utf16_to_ansi( short_utf16 );
292 free( short_utf16 );
293 }
294 }
295 }
296 free( filename_utf16 );
297 }
298 return ansi;
299}

302static int open_file( char *psz_filename, hnd_t *p_handle, video_info_t *info, cli_input_opt_t *opt )
303{
304 FILE *fh = x264_fopen( psz_filename, "r" );
305 if( !fh )
306 return -1;
307 int b_regular = x264_is_regular_file( fh );
308 fclose( fh );
309 FAIL_IF_ERROR( !b_regular, "AVS input is incompatible with non-regular file `%s'\n", psz_filename );
310
311 avs_hnd_t *h = calloc( 1, sizeof(avs_hnd_t) );
312 if( !h )
313 return -1;
314 FAIL_IF_ERROR( custom_avs_load_library( h ), "failed to load avisynth\n" );
315 h->env = h->func.avs_create_script_environment( AVS_INTERFACE_25 );
316 if( h->func.avs_get_error )
317 {
318 const char *error = h->func.avs_get_error( h->env );
319 FAIL_IF_ERROR( error, "%s\n", error );
320 }
321 float avs_version = get_avs_version( h );
322 if( avs_version <= 0 )
323 return -1;
324 x264_cli_log( "avs", X264_LOG_DEBUG, "using avisynth version %.2f\n", avs_version );
325
326#ifdef _WIN32
327 /* Avisynth doesn't support Unicode filenames. */
328 char *ansi_filename = utf8_to_ansi( psz_filename );
329 FAIL_IF_ERROR( !ansi_filename, "invalid ansi filename\n" );
330 AVS_Value arg = avs_new_value_string( ansi_filename );
331#else
332 AVS_Value arg = avs_new_value_string( psz_filename );
333#endif
334
335 AVS_Value res;
336 char *filename_ext = get_filename_extension( psz_filename );
337
338 if( !strcasecmp( filename_ext, "avs" ) )
339 {
340 res = h->func.avs_invoke( h->env, "Import", arg, NULL );
341#ifdef _WIN32
342 free( ansi_filename );
343#endif
344 FAIL_IF_ERROR( avs_is_error( res ), "%s\n", avs_as_error( res ) );
345 /* check if the user is using a multi-threaded script and apply distributor if necessary.
346 adapted from avisynth's vfw interface */
347 AVS_Value mt_test = h->func.avs_invoke( h->env, "GetMTMode", avs_new_value_bool( 0 ), NULL );
348 int mt_mode = avs_is_int( mt_test ) ? avs_as_int( mt_test ) : 0;
349 h->func.avs_release_value( mt_test );
350 if( mt_mode > 0 && mt_mode < 5 )
351 {
352 AVS_Value temp = h->func.avs_invoke( h->env, "Distributor", res, NULL );
353 h->func.avs_release_value( res );
354 res = temp;
355 }
356 }
357 else /* non script file */
358 {
359 /* cycle through known source filters to find one that works */
360 const char *filter[AVS_MAX_SEQUENCE+1] = { 0 };
361 avs_build_filter_sequence( filename_ext, filter );
362 int i;
363 for( i = 0; filter[i]; i++ )
364 {
365 x264_cli_log( "avs", X264_LOG_INFO, "trying %s... ", filter[i] );
366 if( !h->func.avs_function_exists( h->env, filter[i] ) )
367 {
368 x264_cli_printf( X264_LOG_INFO, "not found\n" );
369 continue;
370 }
371 if( !strncasecmp( filter[i], "FFmpegSource", 12 ) )
372 {
373 x264_cli_printf( X264_LOG_INFO, "indexing... " );
374 fflush( stderr );
375 }
376 res = h->func.avs_invoke( h->env, filter[i], arg, NULL );
377 if( !avs_is_error( res ) )
378 {
379 x264_cli_printf( X264_LOG_INFO, "succeeded\n" );
380 break;
381 }
382 x264_cli_printf( X264_LOG_INFO, "failed\n" );
383 }
384#ifdef _WIN32
385 free( ansi_filename );
386#endif
387 FAIL_IF_ERROR( !filter[i], "unable to find source filter to open `%s'\n", psz_filename );
388 }
389 FAIL_IF_ERROR( !avs_is_clip( res ), "`%s' didn't return a video clip\n", psz_filename );
390 h->clip = h->func.avs_take_clip( res, h->env );
391 const AVS_VideoInfo *vi = h->func.avs_get_video_info( h->clip );
392 FAIL_IF_ERROR( !avs_has_video( vi ), "`%s' has no video data\n", psz_filename );
393 /* if the clip is made of fields instead of frames, call weave to make them frames */
394 if( avs_is_field_based( vi ) )
395 {
396 x264_cli_log( "avs", X264_LOG_WARNING, "detected fieldbased (separated) input, weaving to frames\n" );
397 AVS_Value tmp = h->func.avs_invoke( h->env, "Weave", res, NULL );
398 FAIL_IF_ERROR( avs_is_error( tmp ), "couldn't weave fields into frames: %s\n", avs_as_error( tmp ) );
399 res = update_clip( h, &vi, tmp, res );
400 info->interlaced = 1;
401 info->tff = avs_is_tff( vi );
402 }
403#if !HAVE_SWSCALE
404 /* if swscale is not available, convert the CSP if necessary */
405 FAIL_IF_ERROR( avs_version < 2.6f && (opt->output_csp == X264_CSP_I400 || opt->output_csp == X264_CSP_I422 || opt->output_csp == X264_CSP_I444),
406 "avisynth >= 2.6 is required for i400/i422/i444 output\n" );
407 if( (opt->output_csp == X264_CSP_I400 && !AVS_IS_Y( vi )) ||
408 (opt->output_csp == X264_CSP_I420 && !AVS_IS_420( vi )) ||
409 (opt->output_csp == X264_CSP_I422 && !AVS_IS_422( vi )) ||
410 (opt->output_csp == X264_CSP_I444 && !AVS_IS_444( vi )) ||
411 (opt->output_csp == X264_CSP_RGB && !avs_is_rgb( vi )) )
412 {
413 const char *csp;
414 if( AVS_IS_AVISYNTHPLUS )
415 {
416 csp = opt->output_csp == X264_CSP_I400 ? "Y" :
417 opt->output_csp == X264_CSP_I420 ? "YUV420" :
418 opt->output_csp == X264_CSP_I422 ? "YUV422" :
419 opt->output_csp == X264_CSP_I444 ? "YUV444" :
420 "RGB";
421 }
422 else
423 {
424 csp = opt->output_csp == X264_CSP_I400 ? "Y8" :
425 opt->output_csp == X264_CSP_I420 ? "YV12" :
426 opt->output_csp == X264_CSP_I422 ? "YV16" :
427 opt->output_csp == X264_CSP_I444 ? "YV24" :
428 "RGB";
429 }
430 x264_cli_log( "avs", X264_LOG_WARNING, "converting input clip to %s\n", csp );
431 if( opt->output_csp != X264_CSP_I400 )
432 {
433 FAIL_IF_ERROR( opt->output_csp < X264_CSP_I444 && (vi->width&1),
434 "input clip width not divisible by 2 (%dx%d)\n", vi->width, vi->height );
435 FAIL_IF_ERROR( opt->output_csp == X264_CSP_I420 && info->interlaced && (vi->height&3),
436 "input clip height not divisible by 4 (%dx%d)\n", vi->width, vi->height );
437 FAIL_IF_ERROR( (opt->output_csp == X264_CSP_I420 || info->interlaced) && (vi->height&1),
438 "input clip height not divisible by 2 (%dx%d)\n", vi->width, vi->height );
439 }
440 char conv_func[16];
441 snprintf( conv_func, sizeof(conv_func), "ConvertTo%s", csp );
442 AVS_Value arg_arr[3];
443 const char *arg_name[3];
444 int arg_count = 1;
445 arg_arr[0] = res;
446 arg_name[0] = NULL;
447 if( opt->output_csp != X264_CSP_I400 )
448 {
449 arg_arr[arg_count] = avs_new_value_bool( info->interlaced );
450 arg_name[arg_count] = "interlaced";
451 arg_count++;
452 }
453 /* if doing a rgb <-> yuv conversion then range is handled via 'matrix'. though it's only supported in 2.56+ */
454 char matrix[7];
455 if( avs_version >= 2.56f && ((opt->output_csp == X264_CSP_RGB && avs_is_yuv( vi )) || (opt->output_csp != X264_CSP_RGB && avs_is_rgb( vi ))) )
456 {
457 // if converting from yuv, then we specify the matrix for the input, otherwise use the output's.
458 int use_pc_matrix = avs_is_yuv( vi ) ? opt->input_range == RANGE_PC : opt->output_range == RANGE_PC;
459 snprintf( matrix, sizeof(matrix), "%s601", use_pc_matrix ? "PC." : "Rec" ); /* FIXME: use correct coefficients */
460 arg_arr[arg_count] = avs_new_value_string( matrix );
461 arg_name[arg_count] = "matrix";
462 arg_count++;
463 // notification that the input range has changed to the desired one
464 opt->input_range = opt->output_range;
465 }
466 AVS_Value res2 = h->func.avs_invoke( h->env, conv_func, avs_new_value_array( arg_arr, arg_count ), arg_name );
467 FAIL_IF_ERROR( avs_is_error( res2 ), "couldn't convert input clip to %s: %s\n", csp, avs_as_error( res2 ) );
468 res = update_clip( h, &vi, res2, res );
469 }
470 /* if swscale is not available, change the range if necessary. This only applies to YUV-based CSPs however */
471 if( avs_is_yuv( vi ) && opt->output_range != RANGE_AUTO && ((opt->input_range == RANGE_PC) != opt->output_range) )
472 {
473 const char *levels = opt->output_range ? "TV->PC" : "PC->TV";
474 x264_cli_log( "avs", X264_LOG_WARNING, "performing %s conversion\n", levels );
475 AVS_Value arg_arr[2];
476 arg_arr[0] = res;
477 arg_arr[1] = avs_new_value_string( levels );
478 const char *arg_name[] = { NULL, "levels" };
479 AVS_Value res2 = h->func.avs_invoke( h->env, "ColorYUV", avs_new_value_array( arg_arr, 2 ), arg_name );
480 FAIL_IF_ERROR( avs_is_error( res2 ), "couldn't convert range: %s\n", avs_as_error( res2 ) );
481 res = update_clip( h, &vi, res2, res );
482 // notification that the input range has changed to the desired one
483 opt->input_range = opt->output_range;
484 }
485#endif
486
487 h->func.avs_release_value( res );
488
489 info->width = vi->width;
490 info->height = vi->height;
491 info->fps_num = vi->fps_numerator;
492 info->fps_den = vi->fps_denominator;

537static int picture_alloc( cli_pic_t *pic, hnd_t handle, int csp, int width, int height )
538{

554static int read_frame( cli_pic_t *pic, hnd_t handle, int i_frame )
555{
556 static const int plane[3] = { AVS_PLANAR_Y, AVS_PLANAR_U, AVS_PLANAR_V };
557 avs_hnd_t *h = handle;
558 if( i_frame >= h->num_frames )
559 return -1;
560 AVS_VideoFrame *frm = pic->opaque = h->func.avs_get_frame( h->clip, i_frame );
561 const char *err = h->func.avs_clip_get_error( h->clip );
562 FAIL_IF_ERROR( err, "%s occurred while reading frame %d\n", err, i_frame );
563 for( int i = 0; i < pic->img.planes; i++ )
564 {
565 /* explicitly cast away the const attribute to avoid a warning */
566 pic->img.plane[i] = (uint8_t*)avs_get_read_ptr_p( frm, plane[i] );
567 pic->img.stride[i] = avs_get_pitch_p( frm, plane[i] );
568 }
569 return 0;
570}
