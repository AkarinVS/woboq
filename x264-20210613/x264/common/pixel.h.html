31// SSD assumes all args aligned
32// other cmp functions assume first arg aligned
33typedef int (*x264_pixel_cmp_t)( pixel *, intptr_t, pixel *, intptr_t );
34typedef void (*x264_pixel_cmp_x3_t)( pixel *, pixel *, pixel *, pixel *, intptr_t, int[3] );
35typedef void (*x264_pixel_cmp_x4_t)( pixel *, pixel *, pixel *, pixel *, pixel *, intptr_t, int[4] );
36
37enum
38{
39 PIXEL_16x16 = 0,
40 PIXEL_16x8 = 1,
41 PIXEL_8x16 = 2,
42 PIXEL_8x8 = 3,
43 PIXEL_8x4 = 4,
44 PIXEL_4x8 = 5,
45 PIXEL_4x4 = 6,
46
47 /* Subsampled chroma only */
48 PIXEL_4x16 = 7, /* 4:2:2 */
49 PIXEL_4x2 = 8,
50 PIXEL_2x8 = 9, /* 4:2:2 */
51 PIXEL_2x4 = 10,
52 PIXEL_2x2 = 11,
53};

70static const uint8_t x264_luma2chroma_pixel[4][7] =
71{
72 { 0 },
73 { PIXEL_8x8, PIXEL_8x4, PIXEL_4x8, PIXEL_4x4, PIXEL_4x2, PIXEL_2x4, PIXEL_2x2 }, /* 4:2:0 */
74 { PIXEL_8x16, PIXEL_8x8, PIXEL_4x16, PIXEL_4x8, PIXEL_4x4, PIXEL_2x8, PIXEL_2x4 }, /* 4:2:2 */
75 { PIXEL_16x16, PIXEL_16x8, PIXEL_8x16, PIXEL_8x8, PIXEL_8x4, PIXEL_4x8, PIXEL_4x4 }, /* 4:4:4 */
76};

78typedef struct
79{
80 x264_pixel_cmp_t sad[8];
81 x264_pixel_cmp_t ssd[8];
82 x264_pixel_cmp_t satd[8];
83 x264_pixel_cmp_t ssim[7];
84 x264_pixel_cmp_t sa8d[4];
85 x264_pixel_cmp_t mbcmp[8]; /* either satd or sad for subpel refine and mode decision */
86 x264_pixel_cmp_t mbcmp_unaligned[8]; /* unaligned mbcmp for subpel */
87 x264_pixel_cmp_t fpelcmp[8]; /* either satd or sad for fullpel motion search */
88 x264_pixel_cmp_x3_t fpelcmp_x3[7];
89 x264_pixel_cmp_x4_t fpelcmp_x4[7];
90 x264_pixel_cmp_t sad_aligned[8]; /* Aligned SAD for mbcmp */
91 int (*vsad)( pixel *, intptr_t, int );
92 int (*asd8)( pixel *pix1, intptr_t stride1, pixel *pix2, intptr_t stride2, int height );
93 uint64_t (*sa8d_satd[1])( pixel *pix1, intptr_t stride1, pixel *pix2, intptr_t stride2 );
94
95 uint64_t (*var[4])( pixel *pix, intptr_t stride );
96 int (*var2[4])( pixel *fenc, pixel *fdec, int ssd[2] );
97 uint64_t (*hadamard_ac[4])( pixel *pix, intptr_t stride );
98
99 void (*ssd_nv12_core)( pixel *pixuv1, intptr_t stride1,
100 pixel *pixuv2, intptr_t stride2, int width, int height,
101 uint64_t *ssd_u, uint64_t *ssd_v );
102 void (*ssim_4x4x2_core)( const pixel *pix1, intptr_t stride1,
103 const pixel *pix2, intptr_t stride2, int sums[2][4] );
104 float (*ssim_end4)( int sum0[5][4], int sum1[5][4], int width );
105
106 /* multiple parallel calls to cmp. */
107 x264_pixel_cmp_x3_t sad_x3[7];
108 x264_pixel_cmp_x4_t sad_x4[7];
109 x264_pixel_cmp_x3_t satd_x3[7];
110 x264_pixel_cmp_x4_t satd_x4[7];
111
112 /* abs-diff-sum for successive elimination.
113 * may round width up to a multiple of 16. */
114 int (*ads[7])( int enc_dc[4], uint16_t *sums, int delta,
115 uint16_t *cost_mvx, int16_t *mvs, int width, int thresh );
116
117 /* calculate satd or sad of V, H, and DC modes. */
118 void (*intra_mbcmp_x3_16x16)( pixel *fenc, pixel *fdec, int res[3] );
119 void (*intra_satd_x3_16x16) ( pixel *fenc, pixel *fdec, int res[3] );
120 void (*intra_sad_x3_16x16) ( pixel *fenc, pixel *fdec, int res[3] );
121 void (*intra_mbcmp_x3_4x4) ( pixel *fenc, pixel *fdec, int res[3] );
122 void (*intra_satd_x3_4x4) ( pixel *fenc, pixel *fdec, int res[3] );
123 void (*intra_sad_x3_4x4) ( pixel *fenc, pixel *fdec, int res[3] );
124 void (*intra_mbcmp_x3_chroma)( pixel *fenc, pixel *fdec, int res[3] );
125 void (*intra_satd_x3_chroma) ( pixel *fenc, pixel *fdec, int res[3] );
126 void (*intra_sad_x3_chroma) ( pixel *fenc, pixel *fdec, int res[3] );
127 void (*intra_mbcmp_x3_8x16c) ( pixel *fenc, pixel *fdec, int res[3] );
128 void (*intra_satd_x3_8x16c) ( pixel *fenc, pixel *fdec, int res[3] );
129 void (*intra_sad_x3_8x16c) ( pixel *fenc, pixel *fdec, int res[3] );
130 void (*intra_mbcmp_x3_8x8c) ( pixel *fenc, pixel *fdec, int res[3] );
131 void (*intra_satd_x3_8x8c) ( pixel *fenc, pixel *fdec, int res[3] );
132 void (*intra_sad_x3_8x8c) ( pixel *fenc, pixel *fdec, int res[3] );
133 void (*intra_mbcmp_x3_8x8) ( pixel *fenc, pixel edge[36], int res[3] );
134 void (*intra_sa8d_x3_8x8) ( pixel *fenc, pixel edge[36], int res[3] );
135 void (*intra_sad_x3_8x8) ( pixel *fenc, pixel edge[36], int res[3] );
136 /* find minimum satd or sad of all modes, and set fdec.
137 * may be NULL, in which case just use pred+satd instead. */
138 int (*intra_mbcmp_x9_4x4)( pixel *fenc, pixel *fdec, uint16_t *bitcosts );
139 int (*intra_satd_x9_4x4) ( pixel *fenc, pixel *fdec, uint16_t *bitcosts );
140 int (*intra_sad_x9_4x4) ( pixel *fenc, pixel *fdec, uint16_t *bitcosts );
141 int (*intra_mbcmp_x9_8x8)( pixel *fenc, pixel *fdec, pixel edge[36], uint16_t *bitcosts, uint16_t *satds );
142 int (*intra_sa8d_x9_8x8) ( pixel *fenc, pixel *fdec, pixel edge[36], uint16_t *bitcosts, uint16_t *satds );
143 int (*intra_sad_x9_8x8) ( pixel *fenc, pixel *fdec, pixel edge[36], uint16_t *bitcosts, uint16_t *satds );
144} x264_pixel_function_t;
