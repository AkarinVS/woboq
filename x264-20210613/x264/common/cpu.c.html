103static sigjmp_buf jmpbuf;
104static volatile sig_atomic_t canjump = 0;

106static void sigill_handler( int sig )
107{
108 if( !canjump )
109 {
110 signal( sig, SIG_DFL );
111 raise( sig );
112 }
113
114 canjump = 0;
115 siglongjmp( jmpbuf, 1 );
116}

119#if HAVE_MMX
120int x264_cpu_cpuid_test( void );
121void x264_cpu_cpuid( uint32_t op, uint32_t *eax, uint32_t *ebx, uint32_t *ecx, uint32_t *edx );
122uint64_t x264_cpu_xgetbv( int xcr );
123
124uint32_t x264_cpu_detect( void )
125{
126 uint32_t cpu = 0;
127 uint32_t eax, ebx, ecx, edx;
128 uint32_t vendor[4] = {0};
129 uint32_t max_extended_cap, max_basic_cap;
130
131#if !ARCH_X86_64
132 if( !x264_cpu_cpuid_test() )
133 return 0;
134#endif
135
136 x264_cpu_cpuid( 0, &max_basic_cap, vendor+0, vendor+2, vendor+1 );
137 if( max_basic_cap == 0 )
138 return 0;
139
140 x264_cpu_cpuid( 1, &eax, &ebx, &ecx, &edx );
141 if( edx&0x00800000 )
142 cpu |= X264_CPU_MMX;
143 else
144 return cpu;
145 if( edx&0x02000000 )
146 cpu |= X264_CPU_MMX2|X264_CPU_SSE;
147 if( edx&0x04000000 )
148 cpu |= X264_CPU_SSE2;
149 if( ecx&0x00000001 )
150 cpu |= X264_CPU_SSE3;
151 if( ecx&0x00000200 )
152 cpu |= X264_CPU_SSSE3|X264_CPU_SSE2_IS_FAST;
153 if( ecx&0x00080000 )
154 cpu |= X264_CPU_SSE4;
155 if( ecx&0x00100000 )
156 cpu |= X264_CPU_SSE42;
157
158 if( ecx&0x08000000 ) /* XGETBV supported and XSAVE enabled by OS */
159 {
160 uint64_t xcr0 = x264_cpu_xgetbv( 0 );
161 if( (xcr0&0x6) == 0x6 ) /* XMM/YMM state */
162 {
163 if( ecx&0x10000000 )
164 cpu |= X264_CPU_AVX;
165 if( ecx&0x00001000 )
166 cpu |= X264_CPU_FMA3;
167
168 if( max_basic_cap >= 7 )
169 {
170 x264_cpu_cpuid( 7, &eax, &ebx, &ecx, &edx );
171 if( ebx&0x00000008 )
172 cpu |= X264_CPU_BMI1;
173 if( ebx&0x00000100 )
174 cpu |= X264_CPU_BMI2;
175 if( ebx&0x00000020 )
176 cpu |= X264_CPU_AVX2;
177
178 if( (xcr0&0xE0) == 0xE0 ) /* OPMASK/ZMM state */
179 {
180 if( (ebx&0xD0030000) == 0xD0030000 )
181 cpu |= X264_CPU_AVX512;
182 }
183 }
184 }
185 }
186
187 x264_cpu_cpuid( 0x80000000, &eax, &ebx, &ecx, &edx );
188 max_extended_cap = eax;
189
190 if( max_extended_cap >= 0x80000001 )
191 {
192 x264_cpu_cpuid( 0x80000001, &eax, &ebx, &ecx, &edx );
193
194 if( ecx&0x00000020 )
195 cpu |= X264_CPU_LZCNT; /* Supported by Intel chips starting with Haswell */
196 if( ecx&0x00000040 ) /* SSE4a, AMD only */
197 {
198 int family = ((eax>>8)&0xf) + ((eax>>20)&0xff);
199 cpu |= X264_CPU_SSE2_IS_FAST; /* Phenom and later CPUs have fast SSE units */
200 if( family == 0x14 )
201 {
202 cpu &= ~X264_CPU_SSE2_IS_FAST; /* SSSE3 doesn't imply fast SSE anymore... */
203 cpu |= X264_CPU_SSE2_IS_SLOW; /* Bobcat has 64-bit SIMD units */
204 cpu |= X264_CPU_SLOW_PALIGNR; /* palignr is insanely slow on Bobcat */
205 }
206 if( family == 0x16 )
207 {
208 cpu |= X264_CPU_SLOW_PSHUFB; /* Jaguar's pshufb isn't that slow, but it's slow enough
209 * compared to alternate instruction sequences that this
210 * is equal or faster on almost all such functions. */
211 }
212 }
213
214 if( cpu & X264_CPU_AVX )
215 {
216 if( ecx&0x00000800 ) /* XOP */
217 cpu |= X264_CPU_XOP;
218 if( ecx&0x00010000 ) /* FMA4 */
219 cpu |= X264_CPU_FMA4;
220 }
221
222 if( !strcmp((char*)vendor, "AuthenticAMD") )
223 {
224 if( edx&0x00400000 )
225 cpu |= X264_CPU_MMX2;
226 if( (cpu&X264_CPU_SSE2) && !(cpu&X264_CPU_SSE2_IS_FAST) )
227 cpu |= X264_CPU_SSE2_IS_SLOW; /* AMD CPUs come in two types: terrible at SSE and great at it */
228 }
229 }
230
231 if( !strcmp((char*)vendor, "GenuineIntel") )
232 {
233 x264_cpu_cpuid( 1, &eax, &ebx, &ecx, &edx );
234 int family = ((eax>>8)&0xf) + ((eax>>20)&0xff);
235 int model = ((eax>>4)&0xf) + ((eax>>12)&0xf0);
236 if( family == 6 )
237 {
238 /* Detect Atom CPU */
239 if( model == 28 )
240 {
241 cpu |= X264_CPU_SLOW_ATOM;
242 cpu |= X264_CPU_SLOW_PSHUFB;
243 }
244 /* Conroe has a slow shuffle unit. Check the model number to make sure not
245 * to include crippled low-end Penryns and Nehalems that don't have SSE4. */
246 else if( (cpu&X264_CPU_SSSE3) && !(cpu&X264_CPU_SSE4) && model < 23 )
247 cpu |= X264_CPU_SLOW_SHUFFLE;
248 }
249 }
250
251 if( (!strcmp((char*)vendor, "GenuineIntel") || !strcmp((char*)vendor, "CyrixInstead")) && !(cpu&X264_CPU_SSE42))
252 {
253 /* cacheline size is specified in 3 places, any of which may be missing */
254 x264_cpu_cpuid( 1, &eax, &ebx, &ecx, &edx );
255 int cache = (ebx&0xff00)>>5; // cflush size
256 if( !cache && max_extended_cap >= 0x80000006 )
257 {
258 x264_cpu_cpuid( 0x80000006, &eax, &ebx, &ecx, &edx );
259 cache = ecx&0xff; // cacheline size
260 }
261 if( !cache && max_basic_cap >= 2 )
262 {
263 // Cache and TLB Information
264 static const char cache32_ids[] = { 0x0a, 0x0c, 0x41, 0x42, 0x43, 0x44, 0x45, 0x82, 0x83, 0x84, 0x85, 0 };
265 static const char cache64_ids[] = { 0x22, 0x23, 0x25, 0x29, 0x2c, 0x46, 0x47, 0x49, 0x60, 0x66, 0x67,
266 0x68, 0x78, 0x79, 0x7a, 0x7b, 0x7c, 0x7c, 0x7f, 0x86, 0x87, 0 };
267 uint32_t buf[4];
268 int max, i = 0;
269 do {
270 x264_cpu_cpuid( 2, buf+0, buf+1, buf+2, buf+3 );
271 max = buf[0]&0xff;
272 buf[0] &= ~0xff;
273 for( int j = 0; j < 4; j++ )
274 if( !(buf[j]>>31) )
275 while( buf[j] )
276 {
277 if( strchr( cache32_ids, buf[j]&0xff ) )
278 cache = 32;
279 if( strchr( cache64_ids, buf[j]&0xff ) )
280 cache = 64;
281 buf[j] >>= 8;
282 }
283 } while( ++i < max );
284 }
285
286 if( cache == 32 )
287 cpu |= X264_CPU_CACHELINE_32;
288 else if( cache == 64 )
289 cpu |= X264_CPU_CACHELINE_64;
290 else
291 x264_log_internal( X264_LOG_WARNING, "unable to determine cacheline size\n" );
292 }
293
294#if STACK_ALIGNMENT < 16
295 cpu |= X264_CPU_STACK_MOD4;
296#endif
297
298 return cpu;
299}

303#if SYS_MACOSX || SYS_OPENBSD || SYS_FREEBSD
304#include <sys/sysctl.h>
305uint32_t x264_cpu_detect( void )
306{
307 /* Thank you VLC */
308 uint32_t cpu = 0;
309#if SYS_OPENBSD
310 int selectors[2] = { CTL_MACHDEP, CPU_ALTIVEC };
311#elif SYS_MACOSX
312 int selectors[2] = { CTL_HW, HW_VECTORUNIT };
313#endif
314 int has_altivec = 0;
315 size_t length = sizeof( has_altivec );
316#if SYS_MACOSX || SYS_OPENBSD
317 int error = sysctl( selectors, 2, &has_altivec, &length, NULL, 0 );
318#else
319 int error = sysctlbyname( "hw.altivec", &has_altivec, &length, NULL, 0 );
320#endif
321
322 if( error == 0 && has_altivec != 0 )
323 cpu |= X264_CPU_ALTIVEC;
324
325 return cpu;
326}

330uint32_t x264_cpu_detect( void )
331{
332#ifdef __NO_FPRS__
333 return 0;
334#else
335 static void (*oldsig)( int );
336
337 oldsig = signal( SIGILL, sigill_handler );
338 if( sigsetjmp( jmpbuf, 1 ) )
339 {
340 signal( SIGILL, oldsig );
341 return 0;
342 }
343
344 canjump = 1;
345 asm volatile( "mtspr 256, %0\n\t"
346 "vand 0, 0, 0\n\t"
347 :
348 : "r"(-1) );
349 canjump = 0;
350
351 signal( SIGILL, oldsig );
352
353 return X264_CPU_ALTIVEC;
354#endif
355}

358#elif HAVE_ARMV6
359
360void x264_cpu_neon_test( void );
361int x264_cpu_fast_neon_mrc_test( void );
362
363uint32_t x264_cpu_detect( void )
364{
365 uint32_t flags = 0;
366 flags |= X264_CPU_ARMV6;
367
368 // don't do this hack if compiled with -mfpu=neon
369#if !HAVE_NEON
370 static void (* oldsig)( int );
371 oldsig = signal( SIGILL, sigill_handler );
372 if( sigsetjmp( jmpbuf, 1 ) )
373 {
374 signal( SIGILL, oldsig );
375 return flags;
376 }
377
378 canjump = 1;
379 x264_cpu_neon_test();
380 canjump = 0;
381 signal( SIGILL, oldsig );
382#endif
383
384 flags |= X264_CPU_NEON;
385
386 // fast neon -> arm (Cortex-A9) detection relies on user access to the
387 // cycle counter; this assumes ARMv7 performance counters.
388 // NEON requires at least ARMv7, ARMv8 may require changes here, but
389 // hopefully this hacky detection method will have been replaced by then.
390 // Note that there is potential for a race condition if another program or
391 // x264 instance disables or reinits the counters while x264 is using them,
392 // which may result in incorrect detection and the counters stuck enabled.
393 // right now Apple does not seem to support performance counters for this test
394#ifndef __MACH__
395 flags |= x264_cpu_fast_neon_mrc_test() ? X264_CPU_FAST_NEON_MRC : 0;
396#endif
397 // TODO: write dual issue test? currently it's A8 (dual issue) vs. A9 (fast mrc)
398 return flags;
399}

428int x264_cpu_num_processors( void )
429{
430#if !HAVE_THREAD
431 return 1;
432
433#elif SYS_WINDOWS
434 return x264_pthread_num_processors_np();
435
436#elif SYS_CYGWIN || SYS_SunOS || SYS_OPENBSD
437 return sysconf( _SC_NPROCESSORS_ONLN );
438
439#elif SYS_LINUX
440#ifdef __ANDROID__
441 // Android NDK does not expose sched_getaffinity
442 return sysconf( _SC_NPROCESSORS_CONF );
443#else
444 cpu_set_t p_aff;
445 memset( &p_aff, 0, sizeof(p_aff) );
446 if( sched_getaffinity( 0, sizeof(p_aff), &p_aff ) )
447 return 1;
448#if HAVE_CPU_COUNT
449 return CPU_COUNT(&p_aff);
450#else
451 int np = 0;
452 for( size_t bit = 0; bit < 8 * sizeof(p_aff); bit++ )
453 np += (((uint8_t *)&p_aff)[bit / 8] >> (bit % 8)) & 1;
454 return np;
455#endif
456#endif
457
458#elif SYS_BEOS
459 system_info info;
460 get_system_info( &info );
461 return info.cpu_count;
462
463#elif SYS_MACOSX || SYS_FREEBSD
464 int ncpu;
465 size_t length = sizeof( ncpu );
466 if( sysctlbyname("hw.ncpu", &ncpu, &length, NULL, 0) )
467 {
468 ncpu = 1;
469 }
470 return ncpu;
471
472#else
473 return 1;
474#endif
475}
