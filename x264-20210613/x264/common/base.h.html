60/* Unions for type-punning.
61 * Mn: load or store n bits, aligned, native-endian
62 * CPn: copy n bits, aligned, native-endian
63 * we don't use memcpy for CPn because memcpy's args aren't assumed to be aligned */
64typedef union { uint16_t i; uint8_t b[2]; } MAY_ALIAS x264_union16_t;
65typedef union { uint32_t i; uint16_t w[2]; uint8_t b[4]; } MAY_ALIAS x264_union32_t;
66typedef union { uint64_t i; uint32_t d[2]; uint16_t w[4]; uint8_t b[8]; } MAY_ALIAS x264_union64_t;
67typedef struct { uint64_t i[2]; } x264_uint128_t;
68typedef union { x264_uint128_t i; uint64_t q[2]; uint32_t d[4]; uint16_t w[8]; uint8_t b[16]; } MAY_ALIAS x264_union128_t;
69#define M16(src) (((x264_union16_t*)(src))->i)
70#define M32(src) (((x264_union32_t*)(src))->i)
71#define M64(src) (((x264_union64_t*)(src))->i)
72#define M128(src) (((x264_union128_t*)(src))->i)
73#define M128_ZERO ((x264_uint128_t){{0,0}})
74#define CP16(dst,src) M16(dst) = M16(src)
75#define CP32(dst,src) M32(dst) = M32(src)
76#define CP64(dst,src) M64(dst) = M64(src)
77#define CP128(dst,src) M128(dst) = M128(src)

216/* Not a general-purpose function; multiplies input by -1/6 to convert
217 * qp to qscale. */
218static ALWAYS_INLINE int x264_exp2fix8( float x )
219{
220 int i = x*(-64.f/6.f) + 512.5f;
221 if( i < 0 ) return 0;
222 if( i > 1023 ) return 0xffff;
223 return (x264_exp2_lut[i&63]+256) << (i>>6) >> 8;
224}
225
226static ALWAYS_INLINE float x264_log2( uint32_t x )
227{
228 int lz = x264_clz( x );
229 return x264_log2_lut[(x<<lz>>24)&0x7f] + x264_log2_lz_lut[lz];
230}

232static ALWAYS_INLINE int x264_median( int a, int b, int c )
233{
234 int t = (a-b)&((a-b)>>31);
235 a -= t;
236 b += t;
237 b -= (b-c)&((b-c)>>31);
238 b += (a-b)&((a-b)>>31);
239 return b;
240}
241
242static ALWAYS_INLINE void x264_median_mv( int16_t *dst, int16_t *a, int16_t *b, int16_t *c )
243{
244 dst[0] = x264_median( a[0], b[0], c[0] );
245 dst[1] = x264_median( a[1], b[1], c[1] );
246}

248static ALWAYS_INLINE int x264_predictor_difference( int16_t (*mvc)[2], intptr_t i_mvc )
249{
250 int sum = 0;
251 for( int i = 0; i < i_mvc-1; i++ )
252 {
253 sum += abs( mvc[i][0] - mvc[i+1][0] )
254 + abs( mvc[i][1] - mvc[i+1][1] );
255 }
256 return sum;
257}
258
259static ALWAYS_INLINE uint16_t x264_cabac_mvd_sum( uint8_t *mvdleft, uint8_t *mvdtop )
260{
261 int amvd0 = mvdleft[0] + mvdtop[0];
262 int amvd1 = mvdleft[1] + mvdtop[1];
263 amvd0 = (amvd0 > 2) + (amvd0 > 32);
264 amvd1 = (amvd1 > 2) + (amvd1 > 32);
265 return amvd0 + (amvd1<<8);
266}

314/* Macros for merging multiple allocations into a single large malloc, for improved
315 * use with huge pages. */
316
317/* Needs to be enough to contain any set of buffers that use combined allocations */
318#define PREALLOC_BUF_SIZE 1024
319
320#define PREALLOC_INIT\
321 int prealloc_idx = 0;\
322 int64_t prealloc_size = 0;\
323 uint8_t **preallocs[PREALLOC_BUF_SIZE];
324
325#define PREALLOC( var, size )\
326do {\
327 var = (void*)(intptr_t)prealloc_size;\
328 preallocs[prealloc_idx++] = (uint8_t**)&var;\
329 prealloc_size += ALIGN((int64_t)(size), NATIVE_ALIGN);\
330} while( 0 )
331
332#define PREALLOC_END( ptr )\
333do {\
334 CHECKED_MALLOC( ptr, prealloc_size );\
335 while( prealloc_idx-- )\
336 *preallocs[prealloc_idx] = (uint8_t*)((intptr_t)(*preallocs[prealloc_idx]) + (intptr_t)ptr);\
337} while( 0 )
