54static int name( pixel *pix1, intptr_t i_stride_pix1, \

67 return i_sum; \

110uint64_t x264_pixel_ssd_wxh( x264_pixel_function_t *pf, pixel *pix1, intptr_t i_pix1,
111 pixel *pix2, intptr_t i_pix2, int i_width, int i_height )
112{
113 uint64_t i_ssd = 0;
114 int y;
115 int align = !(((intptr_t)pix1 | (intptr_t)pix2 | i_pix1 | i_pix2) & 15);
116
117#define SSD(size) i_ssd += pf->ssd[size]( pix1 + y*i_pix1 + x, i_pix1, \
118 pix2 + y*i_pix2 + x, i_pix2 );
119 for( y = 0; y < i_height-15; y += 16 )
120 {
121 int x = 0;
122 if( align )
123 for( ; x < i_width-15; x += 16 )
124 SSD(PIXEL_16x16);
125 for( ; x < i_width-7; x += 8 )
126 SSD(PIXEL_8x16);
127 }
128 if( y < i_height-7 )
129 for( int x = 0; x < i_width-7; x += 8 )
130 SSD(PIXEL_8x8);
131#undef SSD
132
133#define SSD1 { int d = pix1[y*i_pix1+x] - pix2[y*i_pix2+x]; i_ssd += d*d; }
134 if( i_width & 7 )
135 {
136 for( y = 0; y < (i_height & ~7); y++ )
137 for( int x = i_width & ~7; x < i_width; x++ )
138 SSD1;
139 }
140 if( i_height & 7 )
141 {
142 for( y = i_height & ~7; y < i_height; y++ )
143 for( int x = 0; x < i_width; x++ )
144 SSD1;
145 }
146#undef SSD1
147
148 return i_ssd;
149}

151static void pixel_ssd_nv12_core( pixel *pixuv1, intptr_t stride1, pixel *pixuv2, intptr_t stride2,
152 int width, int height, uint64_t *ssd_u, uint64_t *ssd_v )
153{
154 *ssd_u = 0, *ssd_v = 0;
155 for( int y = 0; y < height; y++, pixuv1+=stride1, pixuv2+=stride2 )
156 for( int x = 0; x < width; x++ )
157 {
158 int du = pixuv1[2*x] - pixuv2[2*x];
159 int dv = pixuv1[2*x+1] - pixuv2[2*x+1];
160 *ssd_u += du*du;
161 *ssd_v += dv*dv;
162 }
163}

165void x264_pixel_ssd_nv12( x264_pixel_function_t *pf, pixel *pix1, intptr_t i_pix1, pixel *pix2, intptr_t i_pix2,
166 int i_width, int i_height, uint64_t *ssd_u, uint64_t *ssd_v )
167{
168 pf->ssd_nv12_core( pix1, i_pix1, pix2, i_pix2, i_width&~7, i_height, ssd_u, ssd_v );
169 if( i_width&7 )
170 {
171 uint64_t tmp[2];
172 pixel_ssd_nv12_core( pix1+(i_width&~7), i_pix1, pix2+(i_width&~7), i_pix2, i_width&7, i_height, &tmp[0], &tmp[1] );
173 *ssd_u += tmp[0];
174 *ssd_v += tmp[1];
175 }

182static uint64_t name( pixel *pix, intptr_t i_stride ) \
183{ \

201/****************************************************************************
202 * pixel_var2_wxh
203 ****************************************************************************/
204#define PIXEL_VAR2_C( name, h, shift ) \
205static int name( pixel *fenc, pixel *fdec, int ssd[2] ) \
206{ \
207 int sum_u = 0, sum_v = 0, sqr_u = 0, sqr_v = 0; \
208 for( int y = 0; y < h; y++ ) \
209 { \
210 for( int x = 0; x < 8; x++ ) \
211 { \
212 int diff_u = fenc[x] - fdec[x]; \
213 int diff_v = fenc[x+FENC_STRIDE/2] - fdec[x+FDEC_STRIDE/2]; \
214 sum_u += diff_u; \
215 sum_v += diff_v; \
216 sqr_u += diff_u * diff_u; \
217 sqr_v += diff_v * diff_v; \
218 } \
219 fenc += FENC_STRIDE; \
220 fdec += FDEC_STRIDE; \
221 } \
222 ssd[0] = sqr_u; \
223 ssd[1] = sqr_v; \
224 return sqr_u - ((int64_t)sum_u * sum_u >> shift) + \
225 sqr_v - ((int64_t)sum_v * sum_v >> shift); \
226}

231#if BIT_DEPTH > 8
232 typedef uint32_t sum_t;
233 typedef uint64_t sum2_t;
234#else
235 typedef uint16_t sum_t;
236 typedef uint32_t sum2_t;
237#endif
238#define BITS_PER_SUM (8 * sizeof(sum_t))
239
240#define HADAMARD4(d0, d1, d2, d3, s0, s1, s2, s3) {\
241 sum2_t t0 = s0 + s1;\
242 sum2_t t1 = s0 - s1;\
243 sum2_t t2 = s2 + s3;\
244 sum2_t t3 = s2 - s3;\
245 d0 = t0 + t2;\
246 d2 = t0 - t2;\
247 d1 = t1 + t3;\
248 d3 = t1 - t3;\
249}
250
251// in: a pseudo-simd number of the form x+(y<<16)
252// return: abs(x)+(abs(y)<<16)
253static ALWAYS_INLINE sum2_t abs2( sum2_t a )
254{
255 sum2_t s = ((a>>(BITS_PER_SUM-1))&(((sum2_t)1<<BITS_PER_SUM)+1))*((sum_t)-1);
256 return (a+s)^s;
257}

263static NOINLINE int x264_pixel_satd_4x4( pixel *pix1, intptr_t i_pix1, pixel *pix2, intptr_t i_pix2 )
264{
265 sum2_t tmp[4][2];
266 sum2_t a0, a1, a2, a3, b0, b1;
267 sum2_t sum = 0;
268 for( int i = 0; i < 4; i++, pix1 += i_pix1, pix2 += i_pix2 )
269 {
270 a0 = pix1[0] - pix2[0];
271 a1 = pix1[1] - pix2[1];
272 b0 = (a0+a1) + ((a0-a1)<<BITS_PER_SUM);
273 a2 = pix1[2] - pix2[2];
274 a3 = pix1[3] - pix2[3];
275 b1 = (a2+a3) + ((a2-a3)<<BITS_PER_SUM);
276 tmp[i][0] = b0 + b1;
277 tmp[i][1] = b0 - b1;
278 }
279 for( int i = 0; i < 2; i++ )
280 {
281 HADAMARD4( a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i] );
282 a0 = abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
283 sum += ((sum_t)a0) + (a0>>BITS_PER_SUM);
284 }
285 return sum >> 1;
286}

288static NOINLINE int x264_pixel_satd_8x4( pixel *pix1, intptr_t i_pix1, pixel *pix2, intptr_t i_pix2 )
289{
290 sum2_t tmp[4][4];
291 sum2_t a0, a1, a2, a3;
292 sum2_t sum = 0;
293 for( int i = 0; i < 4; i++, pix1 += i_pix1, pix2 += i_pix2 )
294 {
295 a0 = (pix1[0] - pix2[0]) + ((sum2_t)(pix1[4] - pix2[4]) << BITS_PER_SUM);
296 a1 = (pix1[1] - pix2[1]) + ((sum2_t)(pix1[5] - pix2[5]) << BITS_PER_SUM);
297 a2 = (pix1[2] - pix2[2]) + ((sum2_t)(pix1[6] - pix2[6]) << BITS_PER_SUM);
298 a3 = (pix1[3] - pix2[3]) + ((sum2_t)(pix1[7] - pix2[7]) << BITS_PER_SUM);
299 HADAMARD4( tmp[i][0], tmp[i][1], tmp[i][2], tmp[i][3], a0,a1,a2,a3 );
300 }
301 for( int i = 0; i < 4; i++ )
302 {
303 HADAMARD4( a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i] );
304 sum += abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
305 }
306 return (((sum_t)sum) + (sum>>BITS_PER_SUM)) >> 1;
307}

332static NOINLINE int sa8d_8x8( pixel *pix1, intptr_t i_pix1, pixel *pix2, intptr_t i_pix2 )
333{
334 sum2_t tmp[8][4];
335 sum2_t a0, a1, a2, a3, a4, a5, a6, a7, b0, b1, b2, b3;
336 sum2_t sum = 0;
337 for( int i = 0; i < 8; i++, pix1 += i_pix1, pix2 += i_pix2 )
338 {
339 a0 = pix1[0] - pix2[0];
340 a1 = pix1[1] - pix2[1];
341 b0 = (a0+a1) + ((a0-a1)<<BITS_PER_SUM);
342 a2 = pix1[2] - pix2[2];
343 a3 = pix1[3] - pix2[3];
344 b1 = (a2+a3) + ((a2-a3)<<BITS_PER_SUM);
345 a4 = pix1[4] - pix2[4];
346 a5 = pix1[5] - pix2[5];
347 b2 = (a4+a5) + ((a4-a5)<<BITS_PER_SUM);
348 a6 = pix1[6] - pix2[6];
349 a7 = pix1[7] - pix2[7];
350 b3 = (a6+a7) + ((a6-a7)<<BITS_PER_SUM);
351 HADAMARD4( tmp[i][0], tmp[i][1], tmp[i][2], tmp[i][3], b0,b1,b2,b3 );
352 }
353 for( int i = 0; i < 4; i++ )
354 {
355 HADAMARD4( a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i] );
356 HADAMARD4( a4, a5, a6, a7, tmp[4][i], tmp[5][i], tmp[6][i], tmp[7][i] );
357 b0 = abs2(a0+a4) + abs2(a0-a4);
358 b0 += abs2(a1+a5) + abs2(a1-a5);
359 b0 += abs2(a2+a6) + abs2(a2-a6);
360 b0 += abs2(a3+a7) + abs2(a3-a7);
361 sum += (sum_t)b0 + (b0>>BITS_PER_SUM);
362 }
363 return sum;
364}

381static NOINLINE uint64_t pixel_hadamard_ac( pixel *pix, intptr_t stride )
382{
383 sum2_t tmp[32];
384 sum2_t a0, a1, a2, a3, dc;
385 sum2_t sum4 = 0, sum8 = 0;
386 for( int i = 0; i < 8; i++, pix+=stride )
387 {
388 sum2_t *t = tmp + (i&3) + (i&4)*4;
389 a0 = (pix[0]+pix[1]) + ((sum2_t)(pix[0]-pix[1])<<BITS_PER_SUM);
390 a1 = (pix[2]+pix[3]) + ((sum2_t)(pix[2]-pix[3])<<BITS_PER_SUM);
391 t[0] = a0 + a1;
392 t[4] = a0 - a1;
393 a2 = (pix[4]+pix[5]) + ((sum2_t)(pix[4]-pix[5])<<BITS_PER_SUM);
394 a3 = (pix[6]+pix[7]) + ((sum2_t)(pix[6]-pix[7])<<BITS_PER_SUM);
395 t[8] = a2 + a3;
396 t[12] = a2 - a3;
397 }
398 for( int i = 0; i < 8; i++ )
399 {
400 HADAMARD4( a0, a1, a2, a3, tmp[i*4+0], tmp[i*4+1], tmp[i*4+2], tmp[i*4+3] );
401 tmp[i*4+0] = a0;
402 tmp[i*4+1] = a1;
403 tmp[i*4+2] = a2;
404 tmp[i*4+3] = a3;
405 sum4 += abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
406 }
407 for( int i = 0; i < 8; i++ )
408 {
409 HADAMARD4( a0,a1,a2,a3, tmp[i], tmp[8+i], tmp[16+i], tmp[24+i] );
410 sum8 += abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
411 }
412 dc = (sum_t)(tmp[0] + tmp[8] + tmp[16] + tmp[24]);
413 sum4 = (sum_t)sum4 + (sum4>>BITS_PER_SUM) - dc;
414 sum8 = (sum_t)sum8 + (sum8>>BITS_PER_SUM) - dc;
415 return ((uint64_t)sum8<<32) + sum4;
416}

418#define HADAMARD_AC(w,h) \
419static uint64_t x264_pixel_hadamard_ac_##w##x##h( pixel *pix, intptr_t stride )\
420{\
421 uint64_t sum = pixel_hadamard_ac( pix, stride );\
422 if( w==16 )\
423 sum += pixel_hadamard_ac( pix+8, stride );\

426 if( w==16 && h==16 )\
427 sum += pixel_hadamard_ac( pix+8*stride+8, stride );\
428 return ((sum>>34)<<32) + ((uint32_t)sum>>1);\
429}

622/****************************************************************************
623 * structural similarity metric
624 ****************************************************************************/
625static void ssim_4x4x2_core( const pixel *pix1, intptr_t stride1,
626 const pixel *pix2, intptr_t stride2,
627 int sums[2][4] )
628{
629 for( int z = 0; z < 2; z++ )
630 {
631 uint32_t s1 = 0, s2 = 0, ss = 0, s12 = 0;
632 for( int y = 0; y < 4; y++ )
633 for( int x = 0; x < 4; x++ )
634 {
635 int a = pix1[x+y*stride1];
636 int b = pix2[x+y*stride2];
637 s1 += a;
638 s2 += b;
639 ss += a*a;
640 ss += b*b;
641 s12 += a*b;
642 }
643 sums[z][0] = s1;
644 sums[z][1] = s2;
645 sums[z][2] = ss;
646 sums[z][3] = s12;
647 pix1 += 4;
648 pix2 += 4;
649 }
650}
651
652static float ssim_end1( int s1, int s2, int ss, int s12 )
653{
654/* Maximum value for 10-bit is: ss*64 = (2^10-1)^2*16*4*64 = 4286582784, which will overflow in some cases.
655 * s1*s1, s2*s2, and s1*s2 also obtain this value for edge cases: ((2^10-1)*16*4)^2 = 4286582784.
656 * Maximum value for 9-bit is: ss*64 = (2^9-1)^2*16*4*64 = 1069551616, which will not overflow. */
657#if BIT_DEPTH > 9
658#define type float
659 static const float ssim_c1 = .01*.01*PIXEL_MAX*PIXEL_MAX*64;
660 static const float ssim_c2 = .03*.03*PIXEL_MAX*PIXEL_MAX*64*63;
661#else
662#define type int
663 static const int ssim_c1 = (int)(.01*.01*PIXEL_MAX*PIXEL_MAX*64 + .5);
664 static const int ssim_c2 = (int)(.03*.03*PIXEL_MAX*PIXEL_MAX*64*63 + .5);
665#endif
666 type fs1 = s1;
667 type fs2 = s2;
668 type fss = ss;
669 type fs12 = s12;
670 type vars = fss*64 - fs1*fs1 - fs2*fs2;
671 type covar = fs12*64 - fs1*fs2;
672 return (float)(2*fs1*fs2 + ssim_c1) * (float)(2*covar + ssim_c2)
673 / ((float)(fs1*fs1 + fs2*fs2 + ssim_c1) * (float)(vars + ssim_c2));
674#undef type
675}
676
677static float ssim_end4( int sum0[5][4], int sum1[5][4], int width )
678{
679 float ssim = 0.0;
680 for( int i = 0; i < width; i++ )
681 ssim += ssim_end1( sum0[i][0] + sum0[i+1][0] + sum1[i][0] + sum1[i+1][0],
682 sum0[i][1] + sum0[i+1][1] + sum1[i][1] + sum1[i+1][1],
683 sum0[i][2] + sum0[i+1][2] + sum1[i][2] + sum1[i+1][2],
684 sum0[i][3] + sum0[i+1][3] + sum1[i][3] + sum1[i+1][3] );
685 return ssim;
686}

688float x264_pixel_ssim_wxh( x264_pixel_function_t *pf,
689 pixel *pix1, intptr_t stride1,
690 pixel *pix2, intptr_t stride2,
691 int width, int height, void *buf, int *cnt )
692{
693 int z = 0;
694 float ssim = 0.0;
695 int (*sum0)[4] = buf;
696 int (*sum1)[4] = sum0 + (width >> 2) + 3;
697 width >>= 2;
698 height >>= 2;
699 for( int y = 1; y < height; y++ )
700 {
701 for( ; z <= y; z++ )
702 {
703 XCHG( void*, sum0, sum1 );
704 for( int x = 0; x < width; x+=2 )
705 pf->ssim_4x4x2_core( &pix1[4*(x+z*stride1)], stride1, &pix2[4*(x+z*stride2)], stride2, &sum0[x] );
706 }
707 for( int x = 0; x < width-1; x += 4 )
708 ssim += pf->ssim_end4( sum0+x, sum1+x, X264_MIN(4,width-x-1) );
709 }
710 *cnt = (height-1) * (width-1);
711 return ssim;
712}

714static int pixel_vsad( pixel *src, intptr_t stride, int height )
715{
716 int score = 0;
717 for( int i = 1; i < height; i++, src += stride )
718 for( int j = 0; j < 16; j++ )
719 score += abs(src[j] - src[j+stride]);
720 return score;
721}
722
723int x264_field_vsad( x264_t *h, int mb_x, int mb_y )
724{
725 int score_field, score_frame;
726 int stride = h->fenc->i_stride[0];
727 int mb_stride = h->mb.i_mb_stride;
728 pixel *fenc = h->fenc->plane[0] + 16 * (mb_x + mb_y * stride);
729 int mb_xy = mb_x + mb_y*mb_stride;
730
731 /* We don't want to analyze pixels outside the frame, as it gives inaccurate results. */
732 int mbpair_height = X264_MIN( h->param.i_height - mb_y * 16, 32 );
733 score_frame = h->pixf.vsad( fenc, stride, mbpair_height );
734 score_field = h->pixf.vsad( fenc, stride*2, mbpair_height >> 1 );
735 score_field += h->pixf.vsad( fenc+stride, stride*2, mbpair_height >> 1 );
736
737 if( mb_x > 0 )
738 score_field += 512 - h->mb.field[mb_xy -1]*1024;
739 if( mb_y > 0 )
740 score_field += 512 - h->mb.field[mb_xy-mb_stride]*1024;
741
742 return (score_field < score_frame);
743}

745static int pixel_asd8( pixel *pix1, intptr_t stride1, pixel *pix2, intptr_t stride2, int height )
746{
747 int sum = 0;
748 for( int y = 0; y < height; y++, pix1 += stride1, pix2 += stride2 )
749 for( int x = 0; x < 8; x++ )
750 sum += pix1[x] - pix2[x];
751 return abs( sum );
752}

754/****************************************************************************
755 * successive elimination
756 ****************************************************************************/
757static int x264_pixel_ads4( int enc_dc[4], uint16_t *sums, int delta,
758 uint16_t *cost_mvx, int16_t *mvs, int width, int thresh )
759{
760 int nmv = 0;
761 for( int i = 0; i < width; i++, sums++ )
762 {
763 int ads = abs( enc_dc[0] - sums[0] )
764 + abs( enc_dc[1] - sums[8] )
765 + abs( enc_dc[2] - sums[delta] )
766 + abs( enc_dc[3] - sums[delta+8] )
767 + cost_mvx[i];
768 if( ads < thresh )
769 mvs[nmv++] = i;
770 }
771 return nmv;
772}

774static int x264_pixel_ads2( int enc_dc[2], uint16_t *sums, int delta,
775 uint16_t *cost_mvx, int16_t *mvs, int width, int thresh )
776{
777 int nmv = 0;
778 for( int i = 0; i < width; i++, sums++ )
779 {
780 int ads = abs( enc_dc[0] - sums[0] )
781 + abs( enc_dc[1] - sums[delta] )
782 + cost_mvx[i];
783 if( ads < thresh )
784 mvs[nmv++] = i;
785 }
786 return nmv;
787}

789static int x264_pixel_ads1( int enc_dc[1], uint16_t *sums, int delta,
790 uint16_t *cost_mvx, int16_t *mvs, int width, int thresh )
791{
792 int nmv = 0;
793 for( int i = 0; i<width; i++, sums++ )
794 {
795 int ads = abs( enc_dc[0] - sums[0] )
796 + cost_mvx[i];
797 if( ads < thresh )
798 mvs[nmv++] = i;
799 }
800 return nmv;
801}
