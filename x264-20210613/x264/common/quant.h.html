42 void (*idct_dequant_2x4_dc)( dctcoef dct[8], dctcoef dct4x4[8][16], int dequant_mf[6][16], int i_qp );

57 int (*coeff_level_run4)( dctcoef *dct, x264_run_level_t *runlevel );

68 int (*trellis_cabac_dc)( TRELLIS_PARAMS, int num_coefs );
69 int (*trellis_cabac_chroma_422_dc)( TRELLIS_PARAMS );
