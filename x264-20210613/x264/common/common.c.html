29/****************************************************************************
30 * x264_log:
31 ****************************************************************************/
32void x264_log( x264_t *h, int i_level, const char *psz_fmt, ... )
33{
34 if( !h || i_level <= h->param.i_log_level )
35 {
36 va_list arg;
37 va_start( arg, psz_fmt );
38 if( !h )
39 x264_log_default( NULL, i_level, psz_fmt, arg );
40 else
41 h->param.pf_log( h->param.p_log_private, i_level, psz_fmt, arg );
42 va_end( arg );
43 }
44}
