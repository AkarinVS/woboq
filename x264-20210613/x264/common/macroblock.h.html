371static ALWAYS_INLINE uint32_t pack16to32( uint32_t a, uint32_t b )
372{
373#if WORDS_BIGENDIAN
374 return b + (a<<16);
375#else
376 return a + (b<<16);
377#endif
378}
379static ALWAYS_INLINE uint32_t pack8to16( uint32_t a, uint32_t b )
380{
381#if WORDS_BIGENDIAN
382 return b + (a<<8);
383#else
384 return a + (b<<8);
385#endif
386}
387static ALWAYS_INLINE uint32_t pack8to32( uint32_t a, uint32_t b, uint32_t c, uint32_t d )
388{
389#if WORDS_BIGENDIAN
390 return d + (c<<8) + (b<<16) + (a<<24);
391#else
392 return a + (b<<8) + (c<<16) + (d<<24);
393#endif
394}
395static ALWAYS_INLINE uint32_t pack16to32_mask( int a, int b )
396{
397#if WORDS_BIGENDIAN
398 return (b&0xFFFF) + ((uint32_t)a<<16);
399#else
400 return (a&0xFFFF) + ((uint32_t)b<<16);
401#endif
402}
403static ALWAYS_INLINE uint64_t pack32to64( uint32_t a, uint32_t b )
404{
405#if WORDS_BIGENDIAN
406 return b + ((uint64_t)a<<32);
407#else
408 return a + ((uint64_t)b<<32);
409#endif
410}

420static ALWAYS_INLINE int x264_mb_predict_intra4x4_mode( x264_t *h, int idx )
421{
422 const int ma = h->mb.cache.intra4x4_pred_mode[x264_scan8[idx] - 1];
423 const int mb = h->mb.cache.intra4x4_pred_mode[x264_scan8[idx] - 8];
424 const int m = X264_MIN( x264_mb_pred_mode4x4_fix(ma),
425 x264_mb_pred_mode4x4_fix(mb) );
426
427 if( m < 0 )
428 return I_PRED_4x4_DC;
429
430 return m;
431}
432static ALWAYS_INLINE int x264_mb_predict_non_zero_code( x264_t *h, int idx )
433{
434 const int za = h->mb.cache.non_zero_count[x264_scan8[idx] - 1];
435 const int zb = h->mb.cache.non_zero_count[x264_scan8[idx] - 8];
436
437 int i_ret = za + zb;
438
439 if( i_ret < 0x80 )
440 i_ret = ( i_ret + 1 ) >> 1;
441 return i_ret & 0x7f;
442}

450/* x264_mb_transform_8x8_allowed:
451 * check whether any partition is smaller than 8x8 (or at least
452 * might be, according to just partition type.)
453 * doesn't check for cbp */
454static ALWAYS_INLINE int x264_mb_transform_8x8_allowed( x264_t *h )
455{
456 if( !h->pps->b_transform_8x8_mode )
457 return 0;
458 if( h->mb.i_type != P_8x8 )
459 return x264_transform_allowed[h->mb.i_type];
460 return M32( h->mb.i_sub_partition ) == D_L0_8x8*0x01010101;
461}
