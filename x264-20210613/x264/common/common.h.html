774static ALWAYS_INLINE int x264_predictor_roundclip( int16_t (*dst)[2], int16_t (*mvc)[2], int i_mvc, int16_t mv_limit[2][2], uint32_t pmv )
775{
776 int cnt = 0;
777 for( int i = 0; i < i_mvc; i++ )
778 {
779 int mx = (mvc[i][0] + 2) >> 2;
780 int my = (mvc[i][1] + 2) >> 2;
781 uint32_t mv = pack16to32_mask(mx, my);
782 if( !mv || mv == pmv ) continue;
783 dst[cnt][0] = x264_clip3( mx, mv_limit[0][0], mv_limit[1][0] );
784 dst[cnt][1] = x264_clip3( my, mv_limit[0][1], mv_limit[1][1] );
785 cnt++;
786 }
787 return cnt;
788}

790static ALWAYS_INLINE int x264_predictor_clip( int16_t (*dst)[2], int16_t (*mvc)[2], int i_mvc, int16_t mv_limit[2][2], uint32_t pmv )
791{
792 int cnt = 0;
793 int qpel_limit[4] = {mv_limit[0][0] << 2, mv_limit[0][1] << 2, mv_limit[1][0] << 2, mv_limit[1][1] << 2};
794 for( int i = 0; i < i_mvc; i++ )
795 {
796 uint32_t mv = M32( mvc[i] );
797 int mx = mvc[i][0];
798 int my = mvc[i][1];
799 if( !mv || mv == pmv ) continue;
800 dst[cnt][0] = x264_clip3( mx, qpel_limit[0], qpel_limit[2] );
801 dst[cnt][1] = x264_clip3( my, qpel_limit[1], qpel_limit[3] );
802 cnt++;
803 }
804 return cnt;
805}
