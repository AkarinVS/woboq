45static void dct4x4dc( dctcoef d[16] )
46{
47 dctcoef tmp[16];
48
49 for( int i = 0; i < 4; i++ )
50 {
51 int s01 = d[i*4+0] + d[i*4+1];
52 int d01 = d[i*4+0] - d[i*4+1];
53 int s23 = d[i*4+2] + d[i*4+3];
54 int d23 = d[i*4+2] - d[i*4+3];
55
56 tmp[0*4+i] = s01 + s23;
57 tmp[1*4+i] = s01 - s23;
58 tmp[2*4+i] = d01 - d23;
59 tmp[3*4+i] = d01 + d23;
60 }
61
62 for( int i = 0; i < 4; i++ )
63 {
64 int s01 = tmp[i*4+0] + tmp[i*4+1];
65 int d01 = tmp[i*4+0] - tmp[i*4+1];
66 int s23 = tmp[i*4+2] + tmp[i*4+3];
67 int d23 = tmp[i*4+2] - tmp[i*4+3];
68
69 d[i*4+0] = ( s01 + s23 + 1 ) >> 1;
70 d[i*4+1] = ( s01 - s23 + 1 ) >> 1;
71 d[i*4+2] = ( d01 - d23 + 1 ) >> 1;
72 d[i*4+3] = ( d01 + d23 + 1 ) >> 1;
73 }
74}

76static void idct4x4dc( dctcoef d[16] )
77{
78 dctcoef tmp[16];
79
80 for( int i = 0; i < 4; i++ )
81 {
82 int s01 = d[i*4+0] + d[i*4+1];
83 int d01 = d[i*4+0] - d[i*4+1];
84 int s23 = d[i*4+2] + d[i*4+3];
85 int d23 = d[i*4+2] - d[i*4+3];
86
87 tmp[0*4+i] = s01 + s23;
88 tmp[1*4+i] = s01 - s23;
89 tmp[2*4+i] = d01 - d23;
90 tmp[3*4+i] = d01 + d23;
91 }
92
93 for( int i = 0; i < 4; i++ )
94 {
95 int s01 = tmp[i*4+0] + tmp[i*4+1];
96 int d01 = tmp[i*4+0] - tmp[i*4+1];
97 int s23 = tmp[i*4+2] + tmp[i*4+3];
98 int d23 = tmp[i*4+2] - tmp[i*4+3];
99
100 d[i*4+0] = s01 + s23;
101 d[i*4+1] = s01 - s23;
102 d[i*4+2] = d01 - d23;
103 d[i*4+3] = d01 + d23;
104 }

107static void dct2x4dc( dctcoef dct[8], dctcoef dct4x4[8][16] )
108{
109 int a0 = dct4x4[0][0] + dct4x4[1][0];
110 int a1 = dct4x4[2][0] + dct4x4[3][0];
111 int a2 = dct4x4[4][0] + dct4x4[5][0];
112 int a3 = dct4x4[6][0] + dct4x4[7][0];
113 int a4 = dct4x4[0][0] - dct4x4[1][0];
114 int a5 = dct4x4[2][0] - dct4x4[3][0];
115 int a6 = dct4x4[4][0] - dct4x4[5][0];
116 int a7 = dct4x4[6][0] - dct4x4[7][0];
117 int b0 = a0 + a1;
118 int b1 = a2 + a3;
119 int b2 = a4 + a5;
120 int b3 = a6 + a7;
121 int b4 = a0 - a1;
122 int b5 = a2 - a3;
123 int b6 = a4 - a5;
124 int b7 = a6 - a7;
125 dct[0] = b0 + b1;
126 dct[1] = b2 + b3;
127 dct[2] = b0 - b1;
128 dct[3] = b2 - b3;
129 dct[4] = b4 - b5;
130 dct[5] = b6 - b7;
131 dct[6] = b4 + b5;
132 dct[7] = b6 + b7;
133 dct4x4[0][0] = 0;
134 dct4x4[1][0] = 0;
135 dct4x4[2][0] = 0;
136 dct4x4[3][0] = 0;
137 dct4x4[4][0] = 0;
138 dct4x4[5][0] = 0;
139 dct4x4[6][0] = 0;
140 dct4x4[7][0] = 0;
141}

143static inline void pixel_sub_wxh( dctcoef *diff, int i_size,
144 pixel *pix1, int i_pix1, pixel *pix2, int i_pix2 )
145{
146 for( int y = 0; y < i_size; y++ )
147 {
148 for( int x = 0; x < i_size; x++ )
149 diff[x + y*i_size] = pix1[x] - pix2[x];
150 pix1 += i_pix1;
151 pix2 += i_pix2;
152 }
153}

155static void sub4x4_dct( dctcoef dct[16], pixel *pix1, pixel *pix2 )
156{
157 dctcoef d[16];
158 dctcoef tmp[16];
159
160 pixel_sub_wxh( d, 4, pix1, FENC_STRIDE, pix2, FDEC_STRIDE );
161
162 for( int i = 0; i < 4; i++ )
163 {
164 int s03 = d[i*4+0] + d[i*4+3];
165 int s12 = d[i*4+1] + d[i*4+2];
166 int d03 = d[i*4+0] - d[i*4+3];
167 int d12 = d[i*4+1] - d[i*4+2];
168
169 tmp[0*4+i] = s03 + s12;
170 tmp[1*4+i] = 2*d03 + d12;
171 tmp[2*4+i] = s03 - s12;
172 tmp[3*4+i] = d03 - 2*d12;
173 }
174
175 for( int i = 0; i < 4; i++ )
176 {
177 int s03 = tmp[i*4+0] + tmp[i*4+3];
178 int s12 = tmp[i*4+1] + tmp[i*4+2];
179 int d03 = tmp[i*4+0] - tmp[i*4+3];
180 int d12 = tmp[i*4+1] - tmp[i*4+2];
181
182 dct[i*4+0] = s03 + s12;
183 dct[i*4+1] = 2*d03 + d12;
184 dct[i*4+2] = s03 - s12;
185 dct[i*4+3] = d03 - 2*d12;
186 }
187}

205static int sub4x4_dct_dc( pixel *pix1, pixel *pix2 )
206{
207 int sum = 0;
208 for( int i=0; i<4; i++, pix1 += FENC_STRIDE, pix2 += FDEC_STRIDE )
209 sum += pix1[0] + pix1[1] + pix1[2] + pix1[3]
210 - pix2[0] - pix2[1] - pix2[2] - pix2[3];
211 return sum;
212}

214static void sub8x8_dct_dc( dctcoef dct[4], pixel *pix1, pixel *pix2 )
215{
216 dct[0] = sub4x4_dct_dc( &pix1[0], &pix2[0] );
217 dct[1] = sub4x4_dct_dc( &pix1[4], &pix2[4] );
218 dct[2] = sub4x4_dct_dc( &pix1[4*FENC_STRIDE+0], &pix2[4*FDEC_STRIDE+0] );
219 dct[3] = sub4x4_dct_dc( &pix1[4*FENC_STRIDE+4], &pix2[4*FDEC_STRIDE+4] );
220
221 /* 2x2 DC transform */
222 int d0 = dct[0] + dct[1];
223 int d1 = dct[2] + dct[3];
224 int d2 = dct[0] - dct[1];
225 int d3 = dct[2] - dct[3];
226 dct[0] = d0 + d1;
227 dct[1] = d0 - d1;
228 dct[2] = d2 + d3;
229 dct[3] = d2 - d3;
230}

232static void sub8x16_dct_dc( dctcoef dct[8], pixel *pix1, pixel *pix2 )
233{
234 int a0 = sub4x4_dct_dc( &pix1[ 0*FENC_STRIDE+0], &pix2[ 0*FDEC_STRIDE+0] );
235 int a1 = sub4x4_dct_dc( &pix1[ 0*FENC_STRIDE+4], &pix2[ 0*FDEC_STRIDE+4] );
236 int a2 = sub4x4_dct_dc( &pix1[ 4*FENC_STRIDE+0], &pix2[ 4*FDEC_STRIDE+0] );
237 int a3 = sub4x4_dct_dc( &pix1[ 4*FENC_STRIDE+4], &pix2[ 4*FDEC_STRIDE+4] );
238 int a4 = sub4x4_dct_dc( &pix1[ 8*FENC_STRIDE+0], &pix2[ 8*FDEC_STRIDE+0] );
239 int a5 = sub4x4_dct_dc( &pix1[ 8*FENC_STRIDE+4], &pix2[ 8*FDEC_STRIDE+4] );
240 int a6 = sub4x4_dct_dc( &pix1[12*FENC_STRIDE+0], &pix2[12*FDEC_STRIDE+0] );
241 int a7 = sub4x4_dct_dc( &pix1[12*FENC_STRIDE+4], &pix2[12*FDEC_STRIDE+4] );
242
243 /* 2x4 DC transform */
244 int b0 = a0 + a1;
245 int b1 = a2 + a3;
246 int b2 = a4 + a5;
247 int b3 = a6 + a7;
248 int b4 = a0 - a1;
249 int b5 = a2 - a3;
250 int b6 = a4 - a5;
251 int b7 = a6 - a7;
252 a0 = b0 + b1;
253 a1 = b2 + b3;
254 a2 = b4 + b5;
255 a3 = b6 + b7;
256 a4 = b0 - b1;
257 a5 = b2 - b3;
258 a6 = b4 - b5;
259 a7 = b6 - b7;
260 dct[0] = a0 + a1;
261 dct[1] = a2 + a3;
262 dct[2] = a0 - a1;
263 dct[3] = a2 - a3;
264 dct[4] = a4 - a5;
265 dct[5] = a6 - a7;
266 dct[6] = a4 + a5;
267 dct[7] = a6 + a7;
268}

270static void add4x4_idct( pixel *p_dst, dctcoef dct[16] )
271{
272 dctcoef d[16];
273 dctcoef tmp[16];
274
275 for( int i = 0; i < 4; i++ )
276 {
277 int s02 = dct[0*4+i] + dct[2*4+i];
278 int d02 = dct[0*4+i] - dct[2*4+i];
279 int s13 = dct[1*4+i] + (dct[3*4+i]>>1);
280 int d13 = (dct[1*4+i]>>1) - dct[3*4+i];
281
282 tmp[i*4+0] = s02 + s13;
283 tmp[i*4+1] = d02 + d13;
284 tmp[i*4+2] = d02 - d13;
285 tmp[i*4+3] = s02 - s13;
286 }
287
288 for( int i = 0; i < 4; i++ )
289 {
290 int s02 = tmp[0*4+i] + tmp[2*4+i];
291 int d02 = tmp[0*4+i] - tmp[2*4+i];
292 int s13 = tmp[1*4+i] + (tmp[3*4+i]>>1);
293 int d13 = (tmp[1*4+i]>>1) - tmp[3*4+i];
294
295 d[0*4+i] = ( s02 + s13 + 32 ) >> 6;
296 d[1*4+i] = ( d02 + d13 + 32 ) >> 6;
297 d[2*4+i] = ( d02 - d13 + 32 ) >> 6;
298 d[3*4+i] = ( s02 - s13 + 32 ) >> 6;
299 }
300
301
302 for( int y = 0; y < 4; y++ )
303 {
304 for( int x = 0; x < 4; x++ )
305 p_dst[x] = x264_clip_pixel( p_dst[x] + d[y*4+x] );
306 p_dst += FDEC_STRIDE;
307 }
308}

327 * 8x8 transform:

330#define DCT8_1D {\
331 int s07 = SRC(0) + SRC(7);\
332 int s16 = SRC(1) + SRC(6);\
333 int s25 = SRC(2) + SRC(5);\
334 int s34 = SRC(3) + SRC(4);\
335 int a0 = s07 + s34;\
336 int a1 = s16 + s25;\
337 int a2 = s07 - s34;\
338 int a3 = s16 - s25;\
339 int d07 = SRC(0) - SRC(7);\
340 int d16 = SRC(1) - SRC(6);\
341 int d25 = SRC(2) - SRC(5);\
342 int d34 = SRC(3) - SRC(4);\
343 int a4 = d16 + d25 + (d07 + (d07>>1));\
344 int a5 = d07 - d34 - (d25 + (d25>>1));\
345 int a6 = d07 + d34 - (d16 + (d16>>1));\
346 int a7 = d16 - d25 + (d34 + (d34>>1));\
347 DST(0) = a0 + a1 ;\
348 DST(1) = a4 + (a7>>2);\
349 DST(2) = a2 + (a3>>1);\
350 DST(3) = a5 + (a6>>2);\
351 DST(4) = a0 - a1 ;\
352 DST(5) = a6 - (a5>>2);\
353 DST(6) = (a2>>1) - a3 ;\
354 DST(7) = (a4>>2) - a7 ;\
355}

362
363#define SRC(x) tmp[x*8+i]
364#define DST(x) tmp[x*8+i]
365 for( int i = 0; i < 8; i++ )
366 DCT8_1D
367#undef SRC
368#undef DST
369
370#define SRC(x) tmp[i*8+x]
371#define DST(x) dct[x*8+i]
372 for( int i = 0; i < 8; i++ )
373 DCT8_1D
374#undef SRC
375#undef DST
376}

386#define IDCT8_1D {\
387 int a0 = SRC(0) + SRC(4);\
388 int a2 = SRC(0) - SRC(4);\
389 int a4 = (SRC(2)>>1) - SRC(6);\
390 int a6 = (SRC(6)>>1) + SRC(2);\
391 int b0 = a0 + a6;\
392 int b2 = a2 + a4;\
393 int b4 = a2 - a4;\
394 int b6 = a0 - a6;\
395 int a1 = -SRC(3) + SRC(5) - SRC(7) - (SRC(7)>>1);\
396 int a3 = SRC(1) + SRC(7) - SRC(3) - (SRC(3)>>1);\
397 int a5 = -SRC(1) + SRC(7) + SRC(5) + (SRC(5)>>1);\
398 int a7 = SRC(3) + SRC(5) + SRC(1) + (SRC(1)>>1);\
399 int b1 = (a7>>2) + a1;\
400 int b3 = a3 + (a5>>2);\
401 int b5 = (a3>>2) - a5;\
402 int b7 = a7 - (a1>>2);\
403 DST(0, b0 + b7);\
404 DST(1, b2 + b5);\
405 DST(2, b4 + b3);\
406 DST(3, b6 + b1);\
407 DST(4, b6 - b1);\
408 DST(5, b4 - b3);\
409 DST(6, b2 - b5);\
410 DST(7, b0 - b7);\
411}

426 for( int i = 0; i < 8; i++ )
427 IDCT8_1D
428#undef SRC

430}
