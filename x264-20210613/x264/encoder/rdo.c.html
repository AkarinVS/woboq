75static ALWAYS_INLINE uint64_t cached_hadamard( x264_t *h, int size, int x, int y )
76{
77 static const uint8_t hadamard_shift_x[4] = {4, 4, 3, 3};
78 static const uint8_t hadamard_shift_y[4] = {4-0, 3-0, 4-1, 3-1};
79 static const uint8_t hadamard_offset[4] = {0, 1, 3, 5};
80 int cache_index = (x >> hadamard_shift_x[size]) + (y >> hadamard_shift_y[size])
81 + hadamard_offset[size];
82 uint64_t res = h->mb.pic.fenc_hadamard_cache[cache_index];
83 if( res )
84 return res - 1;
85 else
86 {
87 pixel *fenc = h->mb.pic.p_fenc[0] + x + y*FENC_STRIDE;
88 res = h->pixf.hadamard_ac[size]( fenc, FENC_STRIDE );
89 h->mb.pic.fenc_hadamard_cache[cache_index] = res + 1;
90 return res;
91 }
92}

94static ALWAYS_INLINE int cached_satd( x264_t *h, int size, int x, int y )
95{
96 static const uint8_t satd_shift_x[3] = {3, 2, 2};
97 static const uint8_t satd_shift_y[3] = {2-1, 3-2, 2-2};
98 static const uint8_t satd_offset[3] = {0, 8, 16};
99 int cache_index = (x >> satd_shift_x[size - PIXEL_8x4]) + (y >> satd_shift_y[size - PIXEL_8x4])
100 + satd_offset[size - PIXEL_8x4];
101 int res = h->mb.pic.fenc_satd_cache[cache_index];
102 if( res )
103 return res - 1;
104 else
105 {
106 pixel *fenc = h->mb.pic.p_fenc[0] + x + y*FENC_STRIDE;
107 int dc = h->pixf.sad[size]( fenc, FENC_STRIDE, (pixel*)x264_zero, 0 ) >> 1;
108 res = h->pixf.satd[size]( fenc, FENC_STRIDE, (pixel*)x264_zero, 0 ) - dc;
109 h->mb.pic.fenc_satd_cache[cache_index] = res + 1;
110 return res;
111 }
112}

114/* Psy RD distortion metric: SSD plus "Absolute Difference of Complexities" */
115/* SATD and SA8D are used to measure block complexity. */
116/* The difference between SATD and SA8D scores are both used to avoid bias from the DCT size. Using SATD */
117/* only, for example, results in overusage of 8x8dct, while the opposite occurs when using SA8D. */
118
119/* FIXME: Is there a better metric than averaged SATD/SA8D difference for complexity difference? */
120/* Hadamard transform is recursive, so a SATD+SA8D can be done faster by taking advantage of this fact. */
121/* This optimization can also be used in non-RD transform decision. */
122
123static inline int ssd_plane( x264_t *h, int size, int p, int x, int y )
124{
125 int satd = 0;
126 pixel *fdec = h->mb.pic.p_fdec[p] + x + y*FDEC_STRIDE;
127 pixel *fenc = h->mb.pic.p_fenc[p] + x + y*FENC_STRIDE;
128 if( p == 0 && h->mb.i_psy_rd )
129 {
130 /* If the plane is smaller than 8x8, we can't do an SA8D; this probably isn't a big problem. */
131 if( size <= PIXEL_8x8 )
132 {
133 uint64_t fdec_acs = h->pixf.hadamard_ac[size]( fdec, FDEC_STRIDE );
134 uint64_t fenc_acs = cached_hadamard( h, size, x, y );
135 satd = abs((int32_t)fdec_acs - (int32_t)fenc_acs)
136 + abs((int32_t)(fdec_acs>>32) - (int32_t)(fenc_acs>>32));
137 satd >>= 1;
138 }
139 else
140 {
141 int dc = h->pixf.sad[size]( fdec, FDEC_STRIDE, (pixel*)x264_zero, 0 ) >> 1;
142 satd = abs(h->pixf.satd[size]( fdec, FDEC_STRIDE, (pixel*)x264_zero, 0 ) - dc - cached_satd( h, size, x, y ));
143 }
144 satd = (satd * h->mb.i_psy_rd * h->mb.i_psy_rd_lambda + 128) >> 8;
145 }
146 return h->pixf.ssd[size](fenc, FENC_STRIDE, fdec, FDEC_STRIDE) + satd;
147}

149static inline int ssd_mb( x264_t *h )
150{
151 int i_ssd = ssd_plane( h, PIXEL_16x16, 0, 0, 0 );
152 if( CHROMA_FORMAT )
153 {
154 int chroma_size = h->luma2chroma_pixel[PIXEL_16x16];
155 int chroma_ssd = ssd_plane( h, chroma_size, 1, 0, 0 ) + ssd_plane( h, chroma_size, 2, 0, 0 );
156 i_ssd += ((uint64_t)chroma_ssd * h->mb.i_chroma_lambda2_offset + 128) >> 8;
157 }
158 return i_ssd;
159}

161static int rd_cost_mb( x264_t *h, int i_lambda2 )
162{
163 int b_transform_bak = h->mb.b_transform_8x8;
164 int i_ssd;
165 int i_bits;
166 int type_bak = h->mb.i_type;
167
168 x264_macroblock_encode( h );
169
170 if( h->mb.b_deblock_rdo )
171 x264_macroblock_deblock( h );
172
173 i_ssd = ssd_mb( h );
174
175 if( IS_SKIP( h->mb.i_type ) )
176 {
177 i_bits = (1 * i_lambda2 + 128) >> 8;
178 }
179 else if( h->param.b_cabac )
180 {
181 x264_cabac_t cabac_tmp;
182 COPY_CABAC;
183 macroblock_size_cabac( h, &cabac_tmp );
184 i_bits = ( (uint64_t)cabac_tmp.f8_bits_encoded * i_lambda2 + 32768 ) >> 16;
185 }
186 else
187 {
188 macroblock_size_cavlc( h );
189 i_bits = ( (uint64_t)h->out.bs.i_bits_encoded * i_lambda2 + 128 ) >> 8;
190 }
191
192 h->mb.b_transform_8x8 = b_transform_bak;
193 h->mb.i_type = type_bak;
194
195 return X264_MIN( i_ssd + i_bits, COST_MAX );
196}

200static uint64_t rd_cost_subpart( x264_t *h, int i_lambda2, int i4, int i_pixel )
201{
202 uint64_t i_ssd, i_bits;
203
204 x264_macroblock_encode_p4x4( h, i4 );
205 if( i_pixel == PIXEL_8x4 )
206 x264_macroblock_encode_p4x4( h, i4+1 );
207 if( i_pixel == PIXEL_4x8 )
208 x264_macroblock_encode_p4x4( h, i4+2 );
209
210 i_ssd = ssd_plane( h, i_pixel, 0, block_idx_x[i4]*4, block_idx_y[i4]*4 );
211 if( CHROMA444 )
212 {
213 int chromassd = ssd_plane( h, i_pixel, 1, block_idx_x[i4]*4, block_idx_y[i4]*4 )
214 + ssd_plane( h, i_pixel, 2, block_idx_x[i4]*4, block_idx_y[i4]*4 );
215 chromassd = ((uint64_t)chromassd * h->mb.i_chroma_lambda2_offset + 128) >> 8;
216 i_ssd += chromassd;
217 }
218
219 if( h->param.b_cabac )
220 {
221 x264_cabac_t cabac_tmp;
222 COPY_CABAC;
223 subpartition_size_cabac( h, &cabac_tmp, i4, i_pixel );
224 i_bits = ( (uint64_t)cabac_tmp.f8_bits_encoded * i_lambda2 + 128 ) >> 8;
225 }
226 else
227 i_bits = subpartition_size_cavlc( h, i4, i_pixel );
228
229 return (i_ssd<<8) + i_bits;
230}

232uint64_t x264_rd_cost_part( x264_t *h, int i_lambda2, int i4, int i_pixel )
233{
234 uint64_t i_ssd, i_bits;
235 int i8 = i4 >> 2;
236
237 if( i_pixel == PIXEL_16x16 )
238 {
239 int i_cost = rd_cost_mb( h, i_lambda2 );
240 return i_cost;
241 }
242
243 if( i_pixel > PIXEL_8x8 )
244 return rd_cost_subpart( h, i_lambda2, i4, i_pixel );
245
246 h->mb.i_cbp_luma = 0;
247
248 x264_macroblock_encode_p8x8( h, i8 );
249 if( i_pixel == PIXEL_16x8 )
250 x264_macroblock_encode_p8x8( h, i8+1 );
251 if( i_pixel == PIXEL_8x16 )
252 x264_macroblock_encode_p8x8( h, i8+2 );
253
254 int ssd_x = 8*(i8&1);
255 int ssd_y = 8*(i8>>1);
256 i_ssd = ssd_plane( h, i_pixel, 0, ssd_x, ssd_y );
257 if( CHROMA_FORMAT )
258 {
259 int chroma_size = h->luma2chroma_pixel[i_pixel];
260 int chroma_ssd = ssd_plane( h, chroma_size, 1, ssd_x>>CHROMA_H_SHIFT, ssd_y>>CHROMA_V_SHIFT )
261 + ssd_plane( h, chroma_size, 2, ssd_x>>CHROMA_H_SHIFT, ssd_y>>CHROMA_V_SHIFT );
262 i_ssd += ((uint64_t)chroma_ssd * h->mb.i_chroma_lambda2_offset + 128) >> 8;
263 }
264
265 if( h->param.b_cabac )
266 {
267 x264_cabac_t cabac_tmp;
268 COPY_CABAC;
269 partition_size_cabac( h, &cabac_tmp, i8, i_pixel );
270 i_bits = ( (uint64_t)cabac_tmp.f8_bits_encoded * i_lambda2 + 128 ) >> 8;
271 }
272 else
273 i_bits = (uint64_t)partition_size_cavlc( h, i8, i_pixel ) * i_lambda2;
274
275 return (i_ssd<<8) + i_bits;
276}

377#define TRELLIS_SCORE_MAX (~0ULL) // marks the node as invalid
378#define TRELLIS_SCORE_BIAS (1ULL<<60) // bias so that all valid scores are positive, even after negative contributions from psy
379#define CABAC_SIZE_BITS 8
380#define LAMBDA_BITS 4
381
382/* precalculate the cost of coding various combinations of bits in a single context */
383void x264_rdo_init( void )
384{
385 for( int i_prefix = 0; i_prefix < 15; i_prefix++ )
386 {
387 for( int i_ctx = 0; i_ctx < 128; i_ctx++ )
388 {
389 int f8_bits = 0;
390 uint8_t ctx = i_ctx;
391
392 for( int i = 1; i < i_prefix; i++ )
393 f8_bits += x264_cabac_size_decision2( &ctx, 1 );
394 if( i_prefix > 0 && i_prefix < 14 )
395 f8_bits += x264_cabac_size_decision2( &ctx, 0 );
396 f8_bits += 1 << CABAC_SIZE_BITS; //sign
397
398 x264_cabac_size_unary[i_prefix][i_ctx] = f8_bits;
399 x264_cabac_transition_unary[i_prefix][i_ctx] = ctx;
400 }
401 }
402 for( int i_ctx = 0; i_ctx < 128; i_ctx++ )
403 {
404 int f8_bits = 0;
405 uint8_t ctx = i_ctx;
406
407 for( int i = 0; i < 5; i++ )
408 f8_bits += x264_cabac_size_decision2( &ctx, 1 );
409 f8_bits += 1 << CABAC_SIZE_BITS; //sign
410
411 cabac_size_5ones[i_ctx] = f8_bits;
412 cabac_transition_5ones[i_ctx] = ctx;
413 }
414}

460// encode all values of the dc coef in a block which is known to have no ac
461static NOINLINE
462int trellis_dc_shortcut( int sign_coef, int quant_coef, int unquant_mf, int coef_weight, int lambda2, uint8_t *cabac_state, int cost_sig )
463{
464 uint64_t bscore = TRELLIS_SCORE_MAX;
465 int ret = 0;
466 int q = abs( quant_coef );
467 for( int abs_level = q-1; abs_level <= q; abs_level++ )
468 {
469 int unquant_abs_level = (unquant_mf * abs_level + 128) >> 8;
470
471 /* Optimize rounding for DC coefficients in DC-only luma 4x4/8x8 blocks. */
472 int d = sign_coef - ((SIGN(unquant_abs_level, sign_coef) + 8)&~15);
473 uint64_t score = (uint64_t)d*d * coef_weight;
474
475 /* code the proposed level, and count how much entropy it would take */
476 if( abs_level )
477 {
478 unsigned f8_bits = cost_sig;
479 int prefix = X264_MIN( abs_level - 1, 14 );
480 f8_bits += x264_cabac_size_decision_noup2( cabac_state+1, prefix > 0 );
481 f8_bits += x264_cabac_size_unary[prefix][cabac_state[5]];
482 if( abs_level >= 15 )
483 f8_bits += bs_size_ue_big( abs_level - 15 ) << CABAC_SIZE_BITS;
484 score += (uint64_t)f8_bits * lambda2 >> ( CABAC_SIZE_BITS - LAMBDA_BITS );
485 }
486
487 COPY2_IF_LT( bscore, score, ret, abs_level );
488 }
489 return SIGN(ret, sign_coef);
490}

492// encode one value of one coef in one context
493static ALWAYS_INLINE
494int trellis_coef( int j, int const_level, int abs_level, int prefix, int suffix_cost,
495 int node_ctx, int level1_ctx, int levelgt1_ctx, uint64_t ssd, int cost_siglast[3],
496 trellis_node_t *nodes_cur, trellis_node_t *nodes_prev,
497 trellis_level_t *level_tree, int levels_used, int lambda2, uint8_t *level_state )
498{
499 uint64_t score = nodes_prev[j].score + ssd;
500 /* code the proposed level, and count how much entropy it would take */
501 unsigned f8_bits = cost_siglast[ j ? 1 : 2 ];
502 uint8_t level1_state = (j >= 3) ? nodes_prev[j].cabac_state[level1_ctx>>2] : level_state[level1_ctx];
503 f8_bits += x264_cabac_entropy[level1_state ^ (const_level > 1)];
504 uint8_t levelgt1_state;
505 if( const_level > 1 )
506 {
507 levelgt1_state = j >= 6 ? nodes_prev[j].cabac_state[levelgt1_ctx-6] : level_state[levelgt1_ctx];
508 f8_bits += x264_cabac_size_unary[prefix][levelgt1_state] + suffix_cost;
509 }
510 else
511 f8_bits += 1 << CABAC_SIZE_BITS;
512 score += (uint64_t)f8_bits * lambda2 >> ( CABAC_SIZE_BITS - LAMBDA_BITS );
513
514 /* save the node if it's better than any existing node with the same cabac ctx */
515 if( score < nodes_cur[node_ctx].score )
516 {
517 nodes_cur[node_ctx].score = score;
518 if( j == 2 || (j <= 3 && node_ctx == 4) ) // init from input state
519 M32(nodes_cur[node_ctx].cabac_state) = M32(level_state+12);
520 else if( j >= 3 )
521 M32(nodes_cur[node_ctx].cabac_state) = M32(nodes_prev[j].cabac_state);
522 if( j >= 3 ) // skip the transition if we're not going to reuse the context
523 nodes_cur[node_ctx].cabac_state[level1_ctx>>2] = x264_cabac_transition[level1_state][const_level > 1];
524 if( const_level > 1 && node_ctx == 7 )
525 nodes_cur[node_ctx].cabac_state[levelgt1_ctx-6] = x264_cabac_transition_unary[prefix][levelgt1_state];
526 nodes_cur[node_ctx].level_idx = nodes_prev[j].level_idx;
527 SET_LEVEL( nodes_cur[node_ctx], nodes_prev[j], abs_level );
528 }
529 return levels_used;
530}

640static ALWAYS_INLINE
641int quant_trellis_cabac( x264_t *h, dctcoef *dct,
642 udctcoef *quant_mf, udctcoef *quant_bias, const int *unquant_mf,
643 const uint8_t *zigzag, int ctx_block_cat, int lambda2, int b_ac,
644 int b_chroma, int dc, int num_coefs, int idx )
645{
646 ALIGNED_ARRAY_64( dctcoef, orig_coefs, [64] );
647 ALIGNED_ARRAY_64( dctcoef, quant_coefs, [64] );
648 const uint32_t *coef_weight1 = num_coefs == 64 ? x264_dct8_weight_tab : x264_dct4_weight_tab;
649 const uint32_t *coef_weight2 = num_coefs == 64 ? x264_dct8_weight2_tab : x264_dct4_weight2_tab;
650 const int b_interlaced = MB_INTERLACED;
651 uint8_t *cabac_state_sig = &h->cabac.state[ x264_significant_coeff_flag_offset[b_interlaced][ctx_block_cat] ];
652 uint8_t *cabac_state_last = &h->cabac.state[ x264_last_coeff_flag_offset[b_interlaced][ctx_block_cat] ];
653 int levelgt1_ctx = b_chroma && dc ? 8 : 9;
654
655 if( dc )
656 {
657 if( num_coefs == 16 )
658 {
659 memcpy( orig_coefs, dct, sizeof(dctcoef)*16 );
660 if( !h->quantf.quant_4x4_dc( dct, quant_mf[0] >> 1, quant_bias[0] << 1 ) )
661 return 0;
662 h->zigzagf.scan_4x4( quant_coefs, dct );
663 }
664 else
665 {
666 memcpy( orig_coefs, dct, sizeof(dctcoef)*num_coefs );
667 int nz = h->quantf.quant_2x2_dc( &dct[0], quant_mf[0] >> 1, quant_bias[0] << 1 );
668 if( num_coefs == 8 )
669 nz |= h->quantf.quant_2x2_dc( &dct[4], quant_mf[0] >> 1, quant_bias[0] << 1 );
670 if( !nz )
671 return 0;
672 for( int i = 0; i < num_coefs; i++ )
673 quant_coefs[i] = dct[zigzag[i]];
674 }
675 }
676 else
677 {
678 if( num_coefs == 64 )
679 {
680 h->mc.memcpy_aligned( orig_coefs, dct, sizeof(dctcoef)*64 );
681 if( !h->quantf.quant_8x8( dct, quant_mf, quant_bias ) )
682 return 0;
683 h->zigzagf.scan_8x8( quant_coefs, dct );
684 }
685 else //if( num_coefs == 16 )
686 {
687 memcpy( orig_coefs, dct, sizeof(dctcoef)*16 );
688 if( !h->quantf.quant_4x4( dct, quant_mf, quant_bias ) )
689 return 0;
690 h->zigzagf.scan_4x4( quant_coefs, dct );
691 }
692 }
693
694 int last_nnz = h->quantf.coeff_last[ctx_block_cat]( quant_coefs+b_ac )+b_ac;
695 uint8_t *cabac_state = &h->cabac.state[ x264_coeff_abs_level_m1_offset[ctx_block_cat] ];
696
697 /* shortcut for dc-only blocks.
698 * this doesn't affect the output, but saves some unnecessary computation. */
699 if( last_nnz == 0 && !dc )
700 {
701 int cost_sig = x264_cabac_size_decision_noup2( &cabac_state_sig[0], 1 )
702 + x264_cabac_size_decision_noup2( &cabac_state_last[0], 1 );
703 dct[0] = trellis_dc_shortcut( orig_coefs[0], quant_coefs[0], unquant_mf[0], coef_weight2[0], lambda2, cabac_state, cost_sig );
704 return !!dct[0];
705 }
706
707#if HAVE_MMX && ARCH_X86_64
708 uint64_t level_state0;
709 memcpy( &level_state0, cabac_state, sizeof(uint64_t) );
710 uint16_t level_state1;
711 memcpy( &level_state1, cabac_state+8, sizeof(uint16_t) );
712#define TRELLIS_ARGS unquant_mf, zigzag, lambda2, last_nnz, orig_coefs, quant_coefs, dct,\
713 cabac_state_sig, cabac_state_last, level_state0, level_state1
714 if( num_coefs == 16 && !dc )
715 if( b_chroma || !h->mb.i_psy_trellis )
716 return h->quantf.trellis_cabac_4x4( TRELLIS_ARGS, b_ac );
717 else
718 return h->quantf.trellis_cabac_4x4_psy( TRELLIS_ARGS, b_ac, h->mb.pic.fenc_dct4[idx&15], h->mb.i_psy_trellis );
719 else if( num_coefs == 64 && !dc )
720 if( b_chroma || !h->mb.i_psy_trellis )
721 return h->quantf.trellis_cabac_8x8( TRELLIS_ARGS, b_interlaced );
722 else
723 return h->quantf.trellis_cabac_8x8_psy( TRELLIS_ARGS, b_interlaced, h->mb.pic.fenc_dct8[idx&3], h->mb.i_psy_trellis);
724 else if( num_coefs == 8 && dc )
725 return h->quantf.trellis_cabac_chroma_422_dc( TRELLIS_ARGS );
726 else if( dc )
727 return h->quantf.trellis_cabac_dc( TRELLIS_ARGS, num_coefs-1 );
728#endif
729
730 // (# of coefs) * (# of ctx) * (# of levels tried) = 1024
731 // we don't need to keep all of those: (# of coefs) * (# of ctx) would be enough,
732 // but it takes more time to remove dead states than you gain in reduced memory.
733 trellis_level_t level_tree[64*8*2];
734 int levels_used = 1;
735 /* init trellis */
736 trellis_node_t nodes[2][8];
737 trellis_node_t *nodes_cur = nodes[0];
738 trellis_node_t *nodes_prev = nodes[1];
739 trellis_node_t *bnode;
740 for( int j = 1; j < 4; j++ )
741 nodes_cur[j].score = TRELLIS_SCORE_MAX;
742 nodes_cur[0].score = TRELLIS_SCORE_BIAS;
743 nodes_cur[0].level_idx = 0;
744 level_tree[0].abs_level = 0;
745 level_tree[0].next = 0;
746 ALIGNED_4( uint8_t level_state[16] );
747 memcpy( level_state, cabac_state, 10 );
748 level_state[12] = cabac_state[0]; // packed subset for copying into trellis_node_t
749 level_state[13] = cabac_state[4];
750 level_state[14] = cabac_state[8];
751 level_state[15] = cabac_state[9];
752
753 idx &= num_coefs == 64 ? 3 : 15;
754
755 // coefs are processed in reverse order, because that's how the abs value is coded.
756 // last_coef and significant_coef flags are normally coded in forward order, but
757 // we have to reverse them to match the levels.
758 // in 4x4 blocks, last_coef and significant_coef use a separate context for each
759 // position, so the order doesn't matter, and we don't even have to update their contexts.
760 // in 8x8 blocks, some positions share contexts, so we'll just have to hope that
761 // cabac isn't too sensitive.
762 int i = last_nnz;
763#define TRELLIS_LOOP(ctx_hi)\
764 for( ; i >= b_ac; i-- )\
765 {\
766 /* skip 0s: this doesn't affect the output, but saves some unnecessary computation. */\
767 if( !quant_coefs[i] )\
768 {\
769 /* no need to calculate ssd of 0s: it's the same in all nodes.\
770 * no need to modify level_tree for ctx=0: it starts with an infinite loop of 0s.\
771 * subtracting from one score is equivalent to adding to the rest. */\
772 if( !ctx_hi )\
773 {\
774 int sigindex = !dc && num_coefs == 64 ? x264_significant_coeff_flag_offset_8x8[b_interlaced][i] :\
775 b_chroma && dc && num_coefs == 8 ? x264_coeff_flag_offset_chroma_422_dc[i] : i;\
776 uint64_t cost_sig0 = x264_cabac_size_decision_noup2( &cabac_state_sig[sigindex], 0 )\
777 * (uint64_t)lambda2 >> ( CABAC_SIZE_BITS - LAMBDA_BITS );\
778 nodes_cur[0].score -= cost_sig0;\
779 }\
780 for( int j = 1; j < (ctx_hi?8:4); j++ )\
781 SET_LEVEL( nodes_cur[j], nodes_cur[j], 0 );\
782 continue;\
783 }\
784\
785 int sign_coef = orig_coefs[zigzag[i]];\
786 int abs_coef = abs( sign_coef );\
787 int q = abs( quant_coefs[i] );\
788 int cost_siglast[3]; /* { zero, nonzero, nonzero-and-last } */\
789 XCHG( trellis_node_t*, nodes_cur, nodes_prev );\
790 for( int j = ctx_hi; j < 8; j++ )\
791 nodes_cur[j].score = TRELLIS_SCORE_MAX;\
792\
793 if( i < num_coefs-1 || ctx_hi )\
794 {\
795 int sigindex = !dc && num_coefs == 64 ? x264_significant_coeff_flag_offset_8x8[b_interlaced][i] :\
796 b_chroma && dc && num_coefs == 8 ? x264_coeff_flag_offset_chroma_422_dc[i] : i;\
797 int lastindex = !dc && num_coefs == 64 ? x264_last_coeff_flag_offset_8x8[i] :\
798 b_chroma && dc && num_coefs == 8 ? x264_coeff_flag_offset_chroma_422_dc[i] : i;\
799 cost_siglast[0] = x264_cabac_size_decision_noup2( &cabac_state_sig[sigindex], 0 );\
800 int cost_sig1 = x264_cabac_size_decision_noup2( &cabac_state_sig[sigindex], 1 );\
801 cost_siglast[1] = x264_cabac_size_decision_noup2( &cabac_state_last[lastindex], 0 ) + cost_sig1;\
802 if( !ctx_hi )\
803 cost_siglast[2] = x264_cabac_size_decision_noup2( &cabac_state_last[lastindex], 1 ) + cost_sig1;\
804 }\
805 else\
806 {\
807 cost_siglast[0] = cost_siglast[1] = cost_siglast[2] = 0;\
808 }\
809\
810 /* there are a few cases where increasing the coeff magnitude helps,\
811 * but it's only around .003 dB, and skipping them ~doubles the speed of trellis.\
812 * could also try q-2: that sometimes helps, but also sometimes decimates blocks\
813 * that are better left coded, especially at QP > 40. */\
814 uint64_t ssd0[2], ssd1[2];\
815 for( int k = 0; k < 2; k++ )\
816 {\
817 int abs_level = q-1+k;\
818 int unquant_abs_level = (((dc?unquant_mf[0]<<1:unquant_mf[zigzag[i]]) * abs_level + 128) >> 8);\
819 int d = abs_coef - unquant_abs_level;\
820 /* Psy trellis: bias in favor of higher AC coefficients in the reconstructed frame. */\
821 if( h->mb.i_psy_trellis && i && !dc && !b_chroma )\
822 {\
823 int orig_coef = (num_coefs == 64) ? h->mb.pic.fenc_dct8[idx][zigzag[i]] : h->mb.pic.fenc_dct4[idx][zigzag[i]];\
824 int predicted_coef = orig_coef - sign_coef;\
825 int psy_value = abs(unquant_abs_level + SIGN(predicted_coef, sign_coef));\
826 int psy_weight = coef_weight1[zigzag[i]] * h->mb.i_psy_trellis;\
827 ssd1[k] = (uint64_t)d*d * coef_weight2[zigzag[i]] - psy_weight * psy_value;\
828 }\
829 else\
830 /* FIXME: for i16x16 dc is this weight optimal? */\
831 ssd1[k] = (uint64_t)d*d * (dc?256:coef_weight2[zigzag[i]]);\
832 ssd0[k] = ssd1[k];\
833 if( !i && !dc && !ctx_hi )\
834 {\
835 /* Optimize rounding for DC coefficients in DC-only luma 4x4/8x8 blocks. */\
836 d = sign_coef - ((SIGN(unquant_abs_level, sign_coef) + 8)&~15);\
837 ssd0[k] = (uint64_t)d*d * coef_weight2[zigzag[i]];\
838 }\
839 }\
840\
841 /* argument passing imposes some significant overhead here. gcc's interprocedural register allocation isn't up to it. */\
842 switch( q )\
843 {\
844 case 1:\
845 ssd1[0] += (uint64_t)cost_siglast[0] * lambda2 >> ( CABAC_SIZE_BITS - LAMBDA_BITS );\
846 levels_used = trellis_coef0_##ctx_hi( ssd0[0]-ssd1[0], nodes_cur, nodes_prev, level_tree, levels_used );\
847 levels_used = trellis_coef1_##ctx_hi( ssd0[1]-ssd1[0], ssd1[1]-ssd1[0], cost_siglast, nodes_cur, nodes_prev, level_tree, levels_used, lambda2, level_state );\
848 goto next##ctx_hi;\
849 case 2:\
850 levels_used = trellis_coef1_##ctx_hi( ssd0[0], ssd1[0], cost_siglast, nodes_cur, nodes_prev, level_tree, levels_used, lambda2, level_state );\
851 levels_used = trellis_coefn_##ctx_hi( q, ssd0[1], ssd1[1], cost_siglast, nodes_cur, nodes_prev, level_tree, levels_used, lambda2, level_state, levelgt1_ctx );\
852 goto next1;\
853 default:\
854 levels_used = trellis_coefn_##ctx_hi( q-1, ssd0[0], ssd1[0], cost_siglast, nodes_cur, nodes_prev, level_tree, levels_used, lambda2, level_state, levelgt1_ctx );\
855 levels_used = trellis_coefn_##ctx_hi( q, ssd0[1], ssd1[1], cost_siglast, nodes_cur, nodes_prev, level_tree, levels_used, lambda2, level_state, levelgt1_ctx );\
856 goto next1;\
857 }\
858 next##ctx_hi:;\
859 }\
860 /* output levels from the best path through the trellis */\
861 bnode = &nodes_cur[ctx_hi];\
862 for( int j = ctx_hi+1; j < (ctx_hi?8:4); j++ )\
863 if( nodes_cur[j].score < bnode->score )\
864 bnode = &nodes_cur[j];
865
866 // keep 2 versions of the main quantization loop, depending on which subsets of the node_ctxs are live
867 // node_ctx 0..3, i.e. having not yet encountered any coefs that might be quantized to >1
868 TRELLIS_LOOP(0);
869
870 if( bnode == &nodes_cur[0] )
871 {
872 /* We only need to zero an empty 4x4 block. 8x8 can be
873 implicitly emptied via zero nnz, as can dc. */
874 if( num_coefs == 16 && !dc )
875 memset( dct, 0, 16 * sizeof(dctcoef) );
876 return 0;
877 }
878
879 if( 0 ) // accessible only by goto, not fallthrough
880 {
881 // node_ctx 1..7 (ctx0 ruled out because we never try both level0 and level2+ on the same coef)
882 TRELLIS_LOOP(1);
883 }
884
885 int level = bnode->level_idx;
886 for( i = b_ac; i <= last_nnz; i++ )
887 {
888 dct[zigzag[i]] = SIGN(level_tree[level].abs_level, dct[zigzag[i]]);
889 level = level_tree[level].next;
890 }
891
892 return 1;
893}

895/* FIXME: This is a gigantic hack. See below.
896 *
897 * CAVLC is much more difficult to trellis than CABAC.
898 *
899 * CABAC has only three states to track: significance map, last, and the
900 * level state machine.
901 * CAVLC, by comparison, has five: coeff_token (trailing + total),
902 * total_zeroes, zero_run, and the level state machine.
903 *
904 * I know of no paper that has managed to design a close-to-optimal trellis
905 * that covers all five of these and isn't exponential-time. As a result, this
906 * "trellis" isn't: it's just a QNS search. Patches welcome for something better.
907 * It's actually surprisingly fast, albeit not quite optimal. It's pretty close
908 * though; since CAVLC only has 2^16 possible rounding modes (assuming only two
909 * roundings as options), a bruteforce search is feasible. Testing shows
910 * that this QNS is reasonably close to optimal in terms of compression.
911 *
912 * TODO:
913 * Don't bother changing large coefficients when it wouldn't affect bit cost
914 * (e.g. only affecting bypassed suffix bits).
915 * Don't re-run all parts of CAVLC bit cost calculation when not necessary.
916 * e.g. when changing a coefficient from one non-zero value to another in
917 * such a way that trailing ones and suffix length isn't affected. */
918static ALWAYS_INLINE
919int quant_trellis_cavlc( x264_t *h, dctcoef *dct,
920 const udctcoef *quant_mf, const int *unquant_mf,
921 const uint8_t *zigzag, int ctx_block_cat, int lambda2, int b_ac,
922 int b_chroma, int dc, int num_coefs, int idx, int b_8x8 )
923{
924 ALIGNED_ARRAY_16( dctcoef, quant_coefs,[2],[16] );
925 ALIGNED_ARRAY_16( dctcoef, coefs,[16] );
926 const uint32_t *coef_weight1 = b_8x8 ? x264_dct8_weight_tab : x264_dct4_weight_tab;
927 const uint32_t *coef_weight2 = b_8x8 ? x264_dct8_weight2_tab : x264_dct4_weight2_tab;
928 int delta_distortion[16];
929 int64_t score = 1ULL<<62;
930 int i, j;
931 const int f = 1<<15;
932 int nC = b_chroma && dc ? 3 + (num_coefs>>2)
933 : ct_index[x264_mb_predict_non_zero_code( h, !b_chroma && dc ? (idx - LUMA_DC)*16 : idx )];
934
935 for( i = 0; i < 16; i += 16/sizeof(*coefs) )
936 M128( &coefs[i] ) = M128_ZERO;
937
938 /* Code for handling 8x8dct -> 4x4dct CAVLC munging. Input/output use a different
939 * step/start/end than internal processing. */
940 int step = 1;
941 int start = b_ac;
942 int end = num_coefs - 1;
943 if( b_8x8 )
944 {
945 start = idx&3;
946 end = 60 + start;
947 step = 4;
948 }
949 idx &= 15;
950
951 lambda2 <<= LAMBDA_BITS;
952
953 /* Find last non-zero coefficient. */
954 for( i = end; i >= start; i -= step )
955 if( (unsigned)(dct[zigzag[i]] * (dc?quant_mf[0]>>1:quant_mf[zigzag[i]]) + f-1) >= 2*f )
956 break;
957
958 if( i < start )
959 goto zeroblock;
960
961 /* Prepare for QNS search: calculate distortion caused by each DCT coefficient
962 * rounding to be searched.
963 *
964 * We only search two roundings (nearest and nearest-1) like in CABAC trellis,
965 * so we just store the difference in distortion between them. */
966 int last_nnz = b_8x8 ? i >> 2 : i;
967 int coef_mask = 0;
968 int round_mask = 0;
969 for( i = b_ac, j = start; i <= last_nnz; i++, j += step )
970 {
971 int coef = dct[zigzag[j]];
972 int abs_coef = abs(coef);
973 int sign = coef < 0 ? -1 : 1;
974 int nearest_quant = ( f + abs_coef * (dc?quant_mf[0]>>1:quant_mf[zigzag[j]]) ) >> 16;
975 quant_coefs[1][i] = quant_coefs[0][i] = sign * nearest_quant;
976 coefs[i] = quant_coefs[1][i];
977 if( nearest_quant )
978 {
979 /* We initialize the trellis with a deadzone halfway between nearest rounding
980 * and always-round-down. This gives much better results than initializing to either
981 * extreme.
982 * FIXME: should we initialize to the deadzones used by deadzone quant? */
983 int deadzone_quant = ( f/2 + abs_coef * (dc?quant_mf[0]>>1:quant_mf[zigzag[j]]) ) >> 16;
984 int unquant1 = (((dc?unquant_mf[0]<<1:unquant_mf[zigzag[j]]) * (nearest_quant-0) + 128) >> 8);
985 int unquant0 = (((dc?unquant_mf[0]<<1:unquant_mf[zigzag[j]]) * (nearest_quant-1) + 128) >> 8);
986 int d1 = abs_coef - unquant1;
987 int d0 = abs_coef - unquant0;
988 delta_distortion[i] = (d0*d0 - d1*d1) * (dc?256:coef_weight2[zigzag[j]]);
989
990 /* Psy trellis: bias in favor of higher AC coefficients in the reconstructed frame. */
991 if( h->mb.i_psy_trellis && j && !dc && !b_chroma )
992 {
993 int orig_coef = b_8x8 ? h->mb.pic.fenc_dct8[idx>>2][zigzag[j]] : h->mb.pic.fenc_dct4[idx][zigzag[j]];
994 int predicted_coef = orig_coef - coef;
995 int psy_weight = coef_weight1[zigzag[j]];
996 int psy_value0 = h->mb.i_psy_trellis * abs(predicted_coef + unquant0 * sign);
997 int psy_value1 = h->mb.i_psy_trellis * abs(predicted_coef + unquant1 * sign);
998 delta_distortion[i] += (psy_value0 - psy_value1) * psy_weight;
999 }
1000
1001 quant_coefs[0][i] = sign * (nearest_quant-1);
1002 if( deadzone_quant != nearest_quant )
1003 coefs[i] = quant_coefs[0][i];
1004 else
1005 round_mask |= 1 << i;
1006 }
1007 else
1008 delta_distortion[i] = 0;
1009 coef_mask |= (!!coefs[i]) << i;
1010 }
1011
1012 /* Calculate the cost of the starting state. */
1013 h->out.bs.i_bits_encoded = 0;
1014 if( !coef_mask )
1015 bs_write_vlc( &h->out.bs, x264_coeff0_token[nC] );
1016 else
1017 cavlc_block_residual_internal( h, ctx_block_cat, coefs + b_ac, nC );
1018 score = (int64_t)h->out.bs.i_bits_encoded * lambda2;
1019
1020 /* QNS loop: pick the change that improves RD the most, apply it, repeat.
1021 * coef_mask and round_mask are used to simplify tracking of nonzeroness
1022 * and rounding modes chosen. */
1023 while( 1 )
1024 {
1025 int64_t iter_score = score;
1026 int iter_distortion_delta = 0;
1027 int iter_coef = -1;
1028 int iter_mask = coef_mask;
1029 int iter_round = round_mask;
1030 for( i = b_ac; i <= last_nnz; i++ )
1031 {
1032 if( !delta_distortion[i] )
1033 continue;
1034
1035 /* Set up all the variables for this iteration. */
1036 int cur_round = round_mask ^ (1 << i);
1037 int round_change = (cur_round >> i)&1;
1038 int old_coef = coefs[i];
1039 int new_coef = quant_coefs[round_change][i];
1040 int cur_mask = (coef_mask&~(1 << i))|(!!new_coef << i);
1041 int cur_distortion_delta = delta_distortion[i] * (round_change ? -1 : 1);
1042 int64_t cur_score = cur_distortion_delta;
1043 coefs[i] = new_coef;
1044
1045 /* Count up bits. */
1046 h->out.bs.i_bits_encoded = 0;
1047 if( !cur_mask )
1048 bs_write_vlc( &h->out.bs, x264_coeff0_token[nC] );
1049 else
1050 cavlc_block_residual_internal( h, ctx_block_cat, coefs + b_ac, nC );
1051 cur_score += (int64_t)h->out.bs.i_bits_encoded * lambda2;
1052
1053 coefs[i] = old_coef;
1054 if( cur_score < iter_score )
1055 {
1056 iter_score = cur_score;
1057 iter_coef = i;
1058 iter_mask = cur_mask;
1059 iter_round = cur_round;
1060 iter_distortion_delta = cur_distortion_delta;
1061 }
1062 }
1063 if( iter_coef >= 0 )
1064 {
1065 score = iter_score - iter_distortion_delta;
1066 coef_mask = iter_mask;
1067 round_mask = iter_round;
1068 coefs[iter_coef] = quant_coefs[((round_mask >> iter_coef)&1)][iter_coef];
1069 /* Don't try adjusting coefficients we've already adjusted.
1070 * Testing suggests this doesn't hurt results -- and sometimes actually helps. */
1071 delta_distortion[iter_coef] = 0;
1072 }
1073 else
1074 break;
1075 }
1076
1077 if( coef_mask )
1078 {
1079 for( i = b_ac, j = start; i < num_coefs; i++, j += step )
1080 dct[zigzag[j]] = coefs[i];
1081 return 1;
1082 }
1083
1084zeroblock:
1085 if( !dc )
1086 {
1087 if( b_8x8 )
1088 for( i = start; i <= end; i+=step )
1089 dct[zigzag[i]] = 0;
1090 else
1091 memset( dct, 0, 16*sizeof(dctcoef) );
1092 }
1093 return 0;
1094}
