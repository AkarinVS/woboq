34#define ZIG(i,y,x) level[i] = dct[x*2+y];
35static inline void zigzag_scan_2x2_dc( dctcoef level[4], dctcoef dct[4] )
36{
37 ZIG(0,0,0)
38 ZIG(1,0,1)
39 ZIG(2,1,0)
40 ZIG(3,1,1)
41}
42#undef ZIG

44static inline void zigzag_scan_2x4_dc( dctcoef level[8], dctcoef dct[8] )
45{
46 level[0] = dct[0];
47 level[1] = dct[2];
48 level[2] = dct[1];
49 level[3] = dct[4];
50 level[4] = dct[6];
51 level[5] = dct[3];
52 level[6] = dct[5];
53 level[7] = dct[7];
54}

56#define IDCT_DEQUANT_2X2_START \
57 int d0 = dct[0] + dct[1]; \
58 int d1 = dct[2] + dct[3]; \
59 int d2 = dct[0] - dct[1]; \
60 int d3 = dct[2] - dct[3]; \
61 int dmf = dequant_mf[i_qp%6][0] << i_qp/6;
62
63static inline void idct_dequant_2x2_dc( dctcoef dct[4], dctcoef dct4x4[4][16], int dequant_mf[6][16], int i_qp )
64{
65 IDCT_DEQUANT_2X2_START
66 dct4x4[0][0] = (d0 + d1) * dmf >> 5;
67 dct4x4[1][0] = (d0 - d1) * dmf >> 5;
68 dct4x4[2][0] = (d2 + d3) * dmf >> 5;
69 dct4x4[3][0] = (d2 - d3) * dmf >> 5;
70}
71
72static inline void idct_dequant_2x2_dconly( dctcoef dct[4], int dequant_mf[6][16], int i_qp )
73{
74 IDCT_DEQUANT_2X2_START
75 dct[0] = (d0 + d1) * dmf >> 5;
76 dct[1] = (d0 - d1) * dmf >> 5;
77 dct[2] = (d2 + d3) * dmf >> 5;
78 dct[3] = (d2 - d3) * dmf >> 5;
79}
80#undef IDCT_2X2_DEQUANT_START

82static inline void dct2x2dc( dctcoef d[4], dctcoef dct4x4[4][16] )
83{
84 int d0 = dct4x4[0][0] + dct4x4[1][0];
85 int d1 = dct4x4[2][0] + dct4x4[3][0];
86 int d2 = dct4x4[0][0] - dct4x4[1][0];
87 int d3 = dct4x4[2][0] - dct4x4[3][0];
88 d[0] = d0 + d1;
89 d[2] = d2 + d3;
90 d[1] = d0 - d1;
91 d[3] = d2 - d3;
92 dct4x4[0][0] = 0;
93 dct4x4[1][0] = 0;
94 dct4x4[2][0] = 0;
95 dct4x4[3][0] = 0;
96}

98static ALWAYS_INLINE int array_non_zero( dctcoef *v, int i_count )
99{
100 if( WORD_SIZE == 8 )
101 {
102 for( int i = 0; i < i_count; i += 8/sizeof(dctcoef) )
103 if( M64( &v[i] ) )
104 return 1;
105 }
106 else
107 {
108 for( int i = 0; i < i_count; i += 4/sizeof(dctcoef) )
109 if( M32( &v[i] ) )
110 return 1;
111 }
112 return 0;
113}

115/* All encoding functions must output the correct CBP and NNZ values.
116 * The entropy coding functions will check CBP first, then NNZ, before
117 * actually reading the DCT coefficients. NNZ still must be correct even
118 * if CBP is zero because of the use of NNZ values for context selection.
119 * "NNZ" need only be 0 or 1 rather than the exact coefficient count because
120 * that is only needed in CAVLC, and will be calculated by CAVLC's residual
121 * coding and stored as necessary. */
122
123/* This means that decimation can be done merely by adjusting the CBP and NNZ
124 * rather than memsetting the coefficients. */
125
126static void mb_encode_i16x16( x264_t *h, int p, int i_qp )
127{
128 pixel *p_src = h->mb.pic.p_fenc[p];
129 pixel *p_dst = h->mb.pic.p_fdec[p];
130
131 ALIGNED_ARRAY_64( dctcoef, dct4x4,[16],[16] );
132 ALIGNED_ARRAY_64( dctcoef, dct_dc4x4,[16] );
133
134 int nz, block_cbp = 0;
135 int decimate_score = h->mb.b_dct_decimate ? 0 : 9;
136 int i_quant_cat = p ? CQM_4IC : CQM_4IY;
137 int i_mode = h->mb.i_intra16x16_pred_mode;
138
139 if( h->mb.b_lossless )
140 x264_predict_lossless_16x16( h, p, i_mode );
141 else
142 h->predict_16x16[i_mode]( h->mb.pic.p_fdec[p] );
143
144 if( h->mb.b_lossless )
145 {
146 for( int i = 0; i < 16; i++ )
147 {
148 int oe = block_idx_xy_fenc[i];
149 int od = block_idx_xy_fdec[i];
150 nz = h->zigzagf.sub_4x4ac( h->dct.luma4x4[16*p+i], p_src+oe, p_dst+od, &dct_dc4x4[block_idx_yx_1d[i]] );
151 h->mb.cache.non_zero_count[x264_scan8[16*p+i]] = nz;
152 block_cbp |= nz;
153 }
154 h->mb.i_cbp_luma |= block_cbp * 0xf;
155 h->mb.cache.non_zero_count[x264_scan8[LUMA_DC+p]] = array_non_zero( dct_dc4x4, 16 );
156 h->zigzagf.scan_4x4( h->dct.luma16x16_dc[p], dct_dc4x4 );
157 return;
158 }
159
160 CLEAR_16x16_NNZ( p );
161
162 h->dctf.sub16x16_dct( dct4x4, p_src, p_dst );
163
164 if( h->mb.b_noise_reduction )
165 for( int idx = 0; idx < 16; idx++ )
166 h->quantf.denoise_dct( dct4x4[idx], h->nr_residual_sum[0], h->nr_offset[0], 16 );
167
168 for( int idx = 0; idx < 16; idx++ )
169 {
170 dct_dc4x4[block_idx_xy_1d[idx]] = dct4x4[idx][0];
171 dct4x4[idx][0] = 0;
172 }
173
174 if( h->mb.b_trellis )
175 {
176 for( int idx = 0; idx < 16; idx++ )
177 if( x264_quant_4x4_trellis( h, dct4x4[idx], i_quant_cat, i_qp, ctx_cat_plane[DCT_LUMA_AC][p], 1, !!p, idx ) )
178 {
179 block_cbp = 0xf;
180 h->zigzagf.scan_4x4( h->dct.luma4x4[16*p+idx], dct4x4[idx] );
181 h->quantf.dequant_4x4( dct4x4[idx], h->dequant4_mf[i_quant_cat], i_qp );
182 if( decimate_score < 6 ) decimate_score += h->quantf.decimate_score15( h->dct.luma4x4[16*p+idx] );
183 h->mb.cache.non_zero_count[x264_scan8[16*p+idx]] = 1;
184 }
185 }
186 else
187 {
188 for( int i8x8 = 0; i8x8 < 4; i8x8++ )
189 {
190 nz = h->quantf.quant_4x4x4( &dct4x4[i8x8*4], h->quant4_mf[i_quant_cat][i_qp], h->quant4_bias[i_quant_cat][i_qp] );
191 if( nz )
192 {
193 block_cbp = 0xf;
194 FOREACH_BIT( idx, i8x8*4, nz )
195 {
196 h->zigzagf.scan_4x4( h->dct.luma4x4[16*p+idx], dct4x4[idx] );
197 h->quantf.dequant_4x4( dct4x4[idx], h->dequant4_mf[i_quant_cat], i_qp );
198 if( decimate_score < 6 ) decimate_score += h->quantf.decimate_score15( h->dct.luma4x4[16*p+idx] );
199 h->mb.cache.non_zero_count[x264_scan8[16*p+idx]] = 1;
200 }
201 }
202 }
203 }
204
205 /* Writing the 16 CBFs in an i16x16 block is quite costly, so decimation can save many bits. */
206 /* More useful with CAVLC, but still useful with CABAC. */
207 if( decimate_score < 6 )
208 {
209 CLEAR_16x16_NNZ( p );
210 block_cbp = 0;
211 }
212 else
213 h->mb.i_cbp_luma |= block_cbp;
214
215 h->dctf.dct4x4dc( dct_dc4x4 );
216 if( h->mb.b_trellis )
217 nz = x264_quant_luma_dc_trellis( h, dct_dc4x4, i_quant_cat, i_qp, ctx_cat_plane[DCT_LUMA_DC][p], 1, LUMA_DC+p );
218 else
219 nz = h->quantf.quant_4x4_dc( dct_dc4x4, h->quant4_mf[i_quant_cat][i_qp][0]>>1, h->quant4_bias[i_quant_cat][i_qp][0]<<1 );
220
221 h->mb.cache.non_zero_count[x264_scan8[LUMA_DC+p]] = nz;
222 if( nz )
223 {
224 h->zigzagf.scan_4x4( h->dct.luma16x16_dc[p], dct_dc4x4 );
225
226 /* output samples to fdec */
227 h->dctf.idct4x4dc( dct_dc4x4 );
228 h->quantf.dequant_4x4_dc( dct_dc4x4, h->dequant4_mf[i_quant_cat], i_qp ); /* XXX not inversed */
229 if( block_cbp )
230 for( int i = 0; i < 16; i++ )
231 dct4x4[i][0] = dct_dc4x4[block_idx_xy_1d[i]];
232 }
233
234 /* put pixels to fdec */
235 if( block_cbp )
236 h->dctf.add16x16_idct( p_dst, dct4x4 );
237 else if( nz )
238 h->dctf.add16x16_idct_dc( p_dst, dct_dc4x4 );
239}

241/* Round down coefficients losslessly in DC-only chroma blocks.
242 * Unlike luma blocks, this can't be done with a lookup table or
243 * other shortcut technique because of the interdependencies
244 * between the coefficients due to the chroma DC transform. */
245static ALWAYS_INLINE int mb_optimize_chroma_dc( x264_t *h, dctcoef *dct_dc, int dequant_mf[6][16], int i_qp, int chroma422 )
246{
247 int dmf = dequant_mf[i_qp%6][0] << i_qp/6;
248
249 /* If the QP is too high, there's no benefit to rounding optimization. */
250 if( dmf > 32*64 )
251 return 1;
252
253 if( chroma422 )
254 return h->quantf.optimize_chroma_2x4_dc( dct_dc, dmf );
255 else
256 return h->quantf.optimize_chroma_2x2_dc( dct_dc, dmf );
257}

259static ALWAYS_INLINE void mb_encode_chroma_internal( x264_t *h, int b_inter, int i_qp, int chroma422 )
260{
261 int nz, nz_dc;
262 int b_decimate = b_inter && h->mb.b_dct_decimate;
263 int (*dequant_mf)[16] = h->dequant4_mf[CQM_4IC + b_inter];
264 ALIGNED_ARRAY_16( dctcoef, dct_dc,[8] );
265 h->mb.i_cbp_chroma = 0;
266 h->nr_count[2] += h->mb.b_noise_reduction * 4;
267
268 M16( &h->mb.cache.non_zero_count[x264_scan8[16]] ) = 0;
269 M16( &h->mb.cache.non_zero_count[x264_scan8[18]] ) = 0;
270 M16( &h->mb.cache.non_zero_count[x264_scan8[32]] ) = 0;
271 M16( &h->mb.cache.non_zero_count[x264_scan8[34]] ) = 0;
272 if( chroma422 )
273 {
274 M16( &h->mb.cache.non_zero_count[x264_scan8[24]] ) = 0;
275 M16( &h->mb.cache.non_zero_count[x264_scan8[26]] ) = 0;
276 M16( &h->mb.cache.non_zero_count[x264_scan8[40]] ) = 0;
277 M16( &h->mb.cache.non_zero_count[x264_scan8[42]] ) = 0;
278 }
279
280 /* Early termination: check variance of chroma residual before encoding.
281 * Don't bother trying early termination at low QPs.
282 * Values are experimentally derived. */
283 if( b_decimate && i_qp >= (h->mb.b_trellis ? 12 : 18) && !h->mb.b_noise_reduction )
284 {
285 int thresh = chroma422 ? (x264_lambda2_tab[i_qp] + 16) >> 5 : (x264_lambda2_tab[i_qp] + 32) >> 6;
286 ALIGNED_ARRAY_8( int, ssd,[2] );
287 int chromapix = chroma422 ? PIXEL_8x16 : PIXEL_8x8;
288
289 if( h->pixf.var2[chromapix]( h->mb.pic.p_fenc[1], h->mb.pic.p_fdec[1], ssd ) < thresh*4 )
290 {
291 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+0]] = 0;
292 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+1]] = 0;
293
294 for( int ch = 0; ch < 2; ch++ )
295 {
296 if( ssd[ch] > thresh )
297 {
298 pixel *p_src = h->mb.pic.p_fenc[1+ch];
299 pixel *p_dst = h->mb.pic.p_fdec[1+ch];
300
301 if( chroma422 )
302 /* Cannot be replaced by two calls to sub8x8_dct_dc since the hadamard transform is different */
303 h->dctf.sub8x16_dct_dc( dct_dc, p_src, p_dst );
304 else
305 h->dctf.sub8x8_dct_dc( dct_dc, p_src, p_dst );
306
307 if( h->mb.b_trellis )
308 nz_dc = x264_quant_chroma_dc_trellis( h, dct_dc, i_qp+3*chroma422, !b_inter, CHROMA_DC+ch );
309 else
310 {
311 nz_dc = 0;
312 for( int i = 0; i <= chroma422; i++ )
313 nz_dc |= h->quantf.quant_2x2_dc( &dct_dc[4*i], h->quant4_mf[CQM_4IC+b_inter][i_qp+3*chroma422][0] >> 1,
314 h->quant4_bias[CQM_4IC+b_inter][i_qp+3*chroma422][0] << 1 );
315 }
316
317 if( nz_dc )
318 {
319 if( !mb_optimize_chroma_dc( h, dct_dc, dequant_mf, i_qp+3*chroma422, chroma422 ) )
320 continue;
321 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+ch]] = 1;
322 if( chroma422 )
323 {
324 zigzag_scan_2x4_dc( h->dct.chroma_dc[ch], dct_dc );
325 h->quantf.idct_dequant_2x4_dconly( dct_dc, dequant_mf, i_qp+3 );
326 }
327 else
328 {
329 zigzag_scan_2x2_dc( h->dct.chroma_dc[ch], dct_dc );
330 idct_dequant_2x2_dconly( dct_dc, dequant_mf, i_qp );
331 }
332
333 for( int i = 0; i <= chroma422; i++ )
334 h->dctf.add8x8_idct_dc( p_dst + 8*i*FDEC_STRIDE, &dct_dc[4*i] );
335 h->mb.i_cbp_chroma = 1;
336 }
337 }
338 }
339 return;
340 }
341 }
342
343 for( int ch = 0; ch < 2; ch++ )
344 {
345 pixel *p_src = h->mb.pic.p_fenc[1+ch];
346 pixel *p_dst = h->mb.pic.p_fdec[1+ch];
347 int i_decimate_score = b_decimate ? 0 : 7;
348 int nz_ac = 0;
349
350 ALIGNED_ARRAY_64( dctcoef, dct4x4,[8],[16] );
351
352 if( h->mb.b_lossless )
353 {
354 static const uint8_t chroma422_scan[8] = { 0, 2, 1, 5, 3, 6, 4, 7 };
355
356 for( int i = 0; i < (chroma422?8:4); i++ )
357 {
358 int oe = 4*(i&1) + 4*(i>>1)*FENC_STRIDE;
359 int od = 4*(i&1) + 4*(i>>1)*FDEC_STRIDE;
360 nz = h->zigzagf.sub_4x4ac( h->dct.luma4x4[16+i+(chroma422?i&4:0)+ch*16], p_src+oe, p_dst+od,
361 &h->dct.chroma_dc[ch][chroma422?chroma422_scan[i]:i] );
362 h->mb.cache.non_zero_count[x264_scan8[16+i+(chroma422?i&4:0)+ch*16]] = nz;
363 h->mb.i_cbp_chroma |= nz;
364 }
365 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+ch]] = array_non_zero( h->dct.chroma_dc[ch], chroma422?8:4 );
366 continue;
367 }
368
369 for( int i = 0; i <= chroma422; i++ )
370 h->dctf.sub8x8_dct( &dct4x4[4*i], p_src + 8*i*FENC_STRIDE, p_dst + 8*i*FDEC_STRIDE );
371
372 if( h->mb.b_noise_reduction )
373 for( int i = 0; i < (chroma422?8:4); i++ )
374 h->quantf.denoise_dct( dct4x4[i], h->nr_residual_sum[2], h->nr_offset[2], 16 );
375
376 if( chroma422 )
377 h->dctf.dct2x4dc( dct_dc, dct4x4 );
378 else
379 dct2x2dc( dct_dc, dct4x4 );
380
381 /* calculate dct coeffs */
382 for( int i8x8 = 0; i8x8 < (chroma422?2:1); i8x8++ )
383 {
384 if( h->mb.b_trellis )
385 {
386 for( int i4x4 = 0; i4x4 < 4; i4x4++ )
387 {
388 if( x264_quant_4x4_trellis( h, dct4x4[i8x8*4+i4x4], CQM_4IC+b_inter, i_qp, DCT_CHROMA_AC, !b_inter, 1, 0 ) )
389 {
390 int idx = 16+ch*16+i8x8*8+i4x4;
391 h->zigzagf.scan_4x4( h->dct.luma4x4[idx], dct4x4[i8x8*4+i4x4] );
392 h->quantf.dequant_4x4( dct4x4[i8x8*4+i4x4], dequant_mf, i_qp );
393 if( i_decimate_score < 7 )
394 i_decimate_score += h->quantf.decimate_score15( h->dct.luma4x4[idx] );
395 h->mb.cache.non_zero_count[x264_scan8[idx]] = 1;
396 nz_ac = 1;
397 }
398 }
399 }
400 else
401 {
402 nz = h->quantf.quant_4x4x4( &dct4x4[i8x8*4], h->quant4_mf[CQM_4IC+b_inter][i_qp],
403 h->quant4_bias[CQM_4IC+b_inter][i_qp] );
404 nz_ac |= nz;
405
406 FOREACH_BIT( i4x4, 0, nz )
407 {
408 int idx = 16+ch*16+i8x8*8+i4x4;
409
410 h->zigzagf.scan_4x4( h->dct.luma4x4[idx], dct4x4[i8x8*4+i4x4] );
411 h->quantf.dequant_4x4( dct4x4[i8x8*4+i4x4], dequant_mf, i_qp );
412 if( i_decimate_score < 7 )
413 i_decimate_score += h->quantf.decimate_score15( h->dct.luma4x4[idx] );
414 h->mb.cache.non_zero_count[x264_scan8[idx]] = 1;
415 }
416 }
417 }
418
419 if( h->mb.b_trellis )
420 nz_dc = x264_quant_chroma_dc_trellis( h, dct_dc, i_qp+3*chroma422, !b_inter, CHROMA_DC+ch );
421 else
422 {
423 nz_dc = 0;
424 for( int i = 0; i <= chroma422; i++ )
425 nz_dc |= h->quantf.quant_2x2_dc( &dct_dc[4*i], h->quant4_mf[CQM_4IC+b_inter][i_qp+3*chroma422][0] >> 1,
426 h->quant4_bias[CQM_4IC+b_inter][i_qp+3*chroma422][0] << 1 );
427 }
428
429 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+ch]] = nz_dc;
430
431 if( i_decimate_score < 7 || !nz_ac )
432 {
433 /* Decimate the block */
434 M16( &h->mb.cache.non_zero_count[x264_scan8[16+16*ch]] ) = 0;
435 M16( &h->mb.cache.non_zero_count[x264_scan8[18+16*ch]] ) = 0;
436 if( chroma422 )
437 {
438 M16( &h->mb.cache.non_zero_count[x264_scan8[24+16*ch]] ) = 0;
439 M16( &h->mb.cache.non_zero_count[x264_scan8[26+16*ch]] ) = 0;
440 }
441
442 if( !nz_dc ) /* Whole block is empty */
443 continue;
444 if( !mb_optimize_chroma_dc( h, dct_dc, dequant_mf, i_qp+3*chroma422, chroma422 ) )
445 {
446 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+ch]] = 0;
447 continue;
448 }
449 /* DC-only */
450 if( chroma422 )
451 {
452 zigzag_scan_2x4_dc( h->dct.chroma_dc[ch], dct_dc );
453 h->quantf.idct_dequant_2x4_dconly( dct_dc, dequant_mf, i_qp+3 );
454 }
455 else
456 {
457 zigzag_scan_2x2_dc( h->dct.chroma_dc[ch], dct_dc );
458 idct_dequant_2x2_dconly( dct_dc, dequant_mf, i_qp );
459 }
460
461 for( int i = 0; i <= chroma422; i++ )
462 h->dctf.add8x8_idct_dc( p_dst + 8*i*FDEC_STRIDE, &dct_dc[4*i] );
463 }
464 else
465 {
466 h->mb.i_cbp_chroma = 1;
467
468 if( nz_dc )
469 {
470 if( chroma422 )
471 {
472 zigzag_scan_2x4_dc( h->dct.chroma_dc[ch], dct_dc );
473 h->quantf.idct_dequant_2x4_dc( dct_dc, dct4x4, dequant_mf, i_qp+3 );
474 }
475 else
476 {
477 zigzag_scan_2x2_dc( h->dct.chroma_dc[ch], dct_dc );
478 idct_dequant_2x2_dc( dct_dc, dct4x4, dequant_mf, i_qp );
479 }
480 }
481
482 for( int i = 0; i <= chroma422; i++ )
483 h->dctf.add8x8_idct( p_dst + 8*i*FDEC_STRIDE, &dct4x4[4*i] );
484 }
485 }

487 /* 0 = none, 1 = DC only, 2 = DC+AC */
488 h->mb.i_cbp_chroma += (h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+0]] |
489 h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+1]] | h->mb.i_cbp_chroma);
490}

493{
494 if( CHROMA_FORMAT == CHROMA_420 )
495 mb_encode_chroma_internal( h, b_inter, i_qp, 0 );
496 else
497 mb_encode_chroma_internal( h, b_inter, i_qp, 1 );
498}

526void x264_predict_lossless_chroma( x264_t *h, int i_mode )
527{
528 int height = 16 >> CHROMA_V_SHIFT;
529 if( i_mode == I_PRED_CHROMA_V )
530 {
531 h->mc.copy[PIXEL_8x8]( h->mb.pic.p_fdec[1], FDEC_STRIDE, h->mb.pic.p_fenc[1]-FENC_STRIDE, FENC_STRIDE, height );
532 h->mc.copy[PIXEL_8x8]( h->mb.pic.p_fdec[2], FDEC_STRIDE, h->mb.pic.p_fenc[2]-FENC_STRIDE, FENC_STRIDE, height );
533 memcpy( h->mb.pic.p_fdec[1], h->mb.pic.p_fdec[1]-FDEC_STRIDE, 8*SIZEOF_PIXEL );
534 memcpy( h->mb.pic.p_fdec[2], h->mb.pic.p_fdec[2]-FDEC_STRIDE, 8*SIZEOF_PIXEL );
535 }
536 else if( i_mode == I_PRED_CHROMA_H )
537 {
538 h->mc.copy[PIXEL_8x8]( h->mb.pic.p_fdec[1], FDEC_STRIDE, h->mb.pic.p_fenc[1]-1, FENC_STRIDE, height );
539 h->mc.copy[PIXEL_8x8]( h->mb.pic.p_fdec[2], FDEC_STRIDE, h->mb.pic.p_fenc[2]-1, FENC_STRIDE, height );
540 x264_copy_column8( h->mb.pic.p_fdec[1]+4*FDEC_STRIDE, h->mb.pic.p_fdec[1]+4*FDEC_STRIDE-1 );
541 x264_copy_column8( h->mb.pic.p_fdec[2]+4*FDEC_STRIDE, h->mb.pic.p_fdec[2]+4*FDEC_STRIDE-1 );
542 if( CHROMA_FORMAT == CHROMA_422 )
543 {
544 x264_copy_column8( h->mb.pic.p_fdec[1]+12*FDEC_STRIDE, h->mb.pic.p_fdec[1]+12*FDEC_STRIDE-1 );
545 x264_copy_column8( h->mb.pic.p_fdec[2]+12*FDEC_STRIDE, h->mb.pic.p_fdec[2]+12*FDEC_STRIDE-1 );
546 }
547 }
548 else
549 {
550 h->predict_chroma[i_mode]( h->mb.pic.p_fdec[1] );
551 h->predict_chroma[i_mode]( h->mb.pic.p_fdec[2] );
552 }
553}

555void x264_predict_lossless_4x4( x264_t *h, pixel *p_dst, int p, int idx, int i_mode )
556{
557 int stride = h->fenc->i_stride[p] << MB_INTERLACED;
558 pixel *p_src = h->mb.pic.p_fenc_plane[p] + block_idx_x[idx]*4 + block_idx_y[idx]*4 * stride;
559
560 if( i_mode == I_PRED_4x4_V )
561 {
562 h->mc.copy[PIXEL_4x4]( p_dst, FDEC_STRIDE, p_src-stride, stride, 4 );
563 memcpy( p_dst, p_dst-FDEC_STRIDE, 4*SIZEOF_PIXEL );
564 }
565 else if( i_mode == I_PRED_4x4_H )
566 {
567 h->mc.copy[PIXEL_4x4]( p_dst, FDEC_STRIDE, p_src-1, stride, 4 );
568 for( int i = 0; i < 4; i++ )
569 p_dst[i*FDEC_STRIDE] = p_dst[i*FDEC_STRIDE-1];
570 }
571 else
572 h->predict_4x4[i_mode]( p_dst );
573}

575void x264_predict_lossless_8x8( x264_t *h, pixel *p_dst, int p, int idx, int i_mode, pixel edge[36] )
576{
577 int stride = h->fenc->i_stride[p] << MB_INTERLACED;
578 pixel *p_src = h->mb.pic.p_fenc_plane[p] + (idx&1)*8 + (idx>>1)*8*stride;
579
580 if( i_mode == I_PRED_8x8_V )
581 {
582 h->mc.copy[PIXEL_8x8]( p_dst, FDEC_STRIDE, p_src-stride, stride, 8 );
583 memcpy( p_dst, &edge[16], 8*SIZEOF_PIXEL );
584 }
585 else if( i_mode == I_PRED_8x8_H )
586 {
587 h->mc.copy[PIXEL_8x8]( p_dst, FDEC_STRIDE, p_src-1, stride, 8 );
588 for( int i = 0; i < 8; i++ )
589 p_dst[i*FDEC_STRIDE] = edge[14-i];
590 }
591 else
592 h->predict_8x8[i_mode]( p_dst, edge );
593}

595void x264_predict_lossless_16x16( x264_t *h, int p, int i_mode )

599
600 if( i_mode == I_PRED_16x16_V )
601 {
602 h->mc.copy[PIXEL_16x16]( p_dst, FDEC_STRIDE, h->mb.pic.p_fenc_plane[p]-stride, stride, 16 );
603 memcpy( p_dst, p_dst-FDEC_STRIDE, 16*SIZEOF_PIXEL );
604 }
605 else if( i_mode == I_PRED_16x16_H )
606 {
607 h->mc.copy_16x16_unaligned( p_dst, FDEC_STRIDE, h->mb.pic.p_fenc_plane[p]-1, stride, 16 );
608 for( int i = 0; i < 16; i++ )
609 p_dst[i*FDEC_STRIDE] = p_dst[i*FDEC_STRIDE-1];
610 }
611 else
612 h->predict_16x16[i_mode]( p_dst );
613}

615/*****************************************************************************
616 * x264_macroblock_encode:
617 *****************************************************************************/
618static ALWAYS_INLINE void macroblock_encode_internal( x264_t *h, int plane_count, int chroma )
619{
620 int i_qp = h->mb.i_qp;
621 int b_decimate = h->mb.b_dct_decimate;
622 int b_force_no_skip = 0;
623 int nz;
624 h->mb.i_cbp_luma = 0;
625 for( int p = 0; p < plane_count; p++ )
626 h->mb.cache.non_zero_count[x264_scan8[LUMA_DC+p]] = 0;
627
628 if( h->mb.i_type == I_PCM )
629 {
630 /* if PCM is chosen, we need to store reconstructed frame data */
631 for( int p = 0; p < plane_count; p++ )
632 h->mc.copy[PIXEL_16x16]( h->mb.pic.p_fdec[p], FDEC_STRIDE, h->mb.pic.p_fenc[p], FENC_STRIDE, 16 );
633 if( chroma )
634 {
635 int height = 16 >> CHROMA_V_SHIFT;
636 h->mc.copy[PIXEL_8x8] ( h->mb.pic.p_fdec[1], FDEC_STRIDE, h->mb.pic.p_fenc[1], FENC_STRIDE, height );
637 h->mc.copy[PIXEL_8x8] ( h->mb.pic.p_fdec[2], FDEC_STRIDE, h->mb.pic.p_fenc[2], FENC_STRIDE, height );
638 }
639 return;
640 }
641
642 if( !h->mb.b_allow_skip )
643 {
644 b_force_no_skip = 1;
645 if( IS_SKIP(h->mb.i_type) )
646 {
647 if( h->mb.i_type == P_SKIP )
648 h->mb.i_type = P_L0;
649 else if( h->mb.i_type == B_SKIP )
650 h->mb.i_type = B_DIRECT;
651 }
652 }
653
654 if( h->mb.i_type == P_SKIP )
655 {
656 /* don't do pskip motion compensation if it was already done in macroblock_analyse */
657 if( !h->mb.b_skip_mc )
658 {
659 int mvx = x264_clip3( h->mb.cache.mv[0][x264_scan8[0]][0],
660 h->mb.mv_min[0], h->mb.mv_max[0] );
661 int mvy = x264_clip3( h->mb.cache.mv[0][x264_scan8[0]][1],
662 h->mb.mv_min[1], h->mb.mv_max[1] );
663
664 for( int p = 0; p < plane_count; p++ )
665 h->mc.mc_luma( h->mb.pic.p_fdec[p], FDEC_STRIDE,
666 &h->mb.pic.p_fref[0][0][p*4], h->mb.pic.i_stride[p],
667 mvx, mvy, 16, 16, &h->sh.weight[0][p] );
668
669 if( chroma )
670 {
671 int v_shift = CHROMA_V_SHIFT;
672 int height = 16 >> v_shift;
673
674 /* Special case for mv0, which is (of course) very common in P-skip mode. */
675 if( mvx | mvy )
676 h->mc.mc_chroma( h->mb.pic.p_fdec[1], h->mb.pic.p_fdec[2], FDEC_STRIDE,
677 h->mb.pic.p_fref[0][0][4], h->mb.pic.i_stride[1],
678 mvx, 2*mvy>>v_shift, 8, height );
679 else
680 h->mc.load_deinterleave_chroma_fdec( h->mb.pic.p_fdec[1], h->mb.pic.p_fref[0][0][4],
681 h->mb.pic.i_stride[1], height );
682
683 if( h->sh.weight[0][1].weightfn )
684 h->sh.weight[0][1].weightfn[8>>2]( h->mb.pic.p_fdec[1], FDEC_STRIDE,
685 h->mb.pic.p_fdec[1], FDEC_STRIDE,
686 &h->sh.weight[0][1], height );
687 if( h->sh.weight[0][2].weightfn )
688 h->sh.weight[0][2].weightfn[8>>2]( h->mb.pic.p_fdec[2], FDEC_STRIDE,
689 h->mb.pic.p_fdec[2], FDEC_STRIDE,
690 &h->sh.weight[0][2], height );
691 }
692 }
693
694 macroblock_encode_skip( h );
695 return;
696 }
697 if( h->mb.i_type == B_SKIP )
698 {
699 /* don't do bskip motion compensation if it was already done in macroblock_analyse */
700 if( !h->mb.b_skip_mc )
701 x264_mb_mc( h );
702 macroblock_encode_skip( h );
703 return;
704 }
705
706 if( h->mb.i_type == I_16x16 )
707 {
708 h->mb.b_transform_8x8 = 0;
709
710 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
711 mb_encode_i16x16( h, p, i_qp );
712 }
713 else if( h->mb.i_type == I_8x8 )
714 {
715 h->mb.b_transform_8x8 = 1;
716 /* If we already encoded 3 of the 4 i8x8 blocks, we don't have to do them again. */
717 if( h->mb.i_skip_intra )
718 {
719 h->mc.copy[PIXEL_16x16]( h->mb.pic.p_fdec[0], FDEC_STRIDE, h->mb.pic.i8x8_fdec_buf, 16, 16 );
720 M32( &h->mb.cache.non_zero_count[x264_scan8[ 0]] ) = h->mb.pic.i8x8_nnz_buf[0];
721 M32( &h->mb.cache.non_zero_count[x264_scan8[ 2]] ) = h->mb.pic.i8x8_nnz_buf[1];
722 M32( &h->mb.cache.non_zero_count[x264_scan8[ 8]] ) = h->mb.pic.i8x8_nnz_buf[2];
723 M32( &h->mb.cache.non_zero_count[x264_scan8[10]] ) = h->mb.pic.i8x8_nnz_buf[3];
724 h->mb.i_cbp_luma = h->mb.pic.i8x8_cbp;
725 /* In RD mode, restore the now-overwritten DCT data. */
726 if( h->mb.i_skip_intra == 2 )
727 h->mc.memcpy_aligned( h->dct.luma8x8, h->mb.pic.i8x8_dct_buf, sizeof(h->mb.pic.i8x8_dct_buf) );
728 }
729 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
730 {
731 for( int i = (p == 0 && h->mb.i_skip_intra) ? 3 : 0; i < 4; i++ )
732 {
733 int i_mode = h->mb.cache.intra4x4_pred_mode[x264_scan8[4*i]];
734 x264_mb_encode_i8x8( h, p, i, i_qp, i_mode, NULL, 1 );
735 }
736 }
737 }
738 else if( h->mb.i_type == I_4x4 )
739 {
740 h->mb.b_transform_8x8 = 0;
741 /* If we already encoded 15 of the 16 i4x4 blocks, we don't have to do them again. */
742 if( h->mb.i_skip_intra )
743 {
744 h->mc.copy[PIXEL_16x16]( h->mb.pic.p_fdec[0], FDEC_STRIDE, h->mb.pic.i4x4_fdec_buf, 16, 16 );
745 M32( &h->mb.cache.non_zero_count[x264_scan8[ 0]] ) = h->mb.pic.i4x4_nnz_buf[0];
746 M32( &h->mb.cache.non_zero_count[x264_scan8[ 2]] ) = h->mb.pic.i4x4_nnz_buf[1];
747 M32( &h->mb.cache.non_zero_count[x264_scan8[ 8]] ) = h->mb.pic.i4x4_nnz_buf[2];
748 M32( &h->mb.cache.non_zero_count[x264_scan8[10]] ) = h->mb.pic.i4x4_nnz_buf[3];
749 h->mb.i_cbp_luma = h->mb.pic.i4x4_cbp;
750 /* In RD mode, restore the now-overwritten DCT data. */
751 if( h->mb.i_skip_intra == 2 )
752 h->mc.memcpy_aligned( h->dct.luma4x4, h->mb.pic.i4x4_dct_buf, sizeof(h->mb.pic.i4x4_dct_buf) );
753 }
754 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
755 {
756 for( int i = (p == 0 && h->mb.i_skip_intra) ? 15 : 0; i < 16; i++ )
757 {
758 pixel *p_dst = &h->mb.pic.p_fdec[p][block_idx_xy_fdec[i]];
759 int i_mode = h->mb.cache.intra4x4_pred_mode[x264_scan8[i]];
760
761 if( (h->mb.i_neighbour4[i] & (MB_TOPRIGHT|MB_TOP)) == MB_TOP )
762 /* emulate missing topright samples */
763 MPIXEL_X4( &p_dst[4-FDEC_STRIDE] ) = PIXEL_SPLAT_X4( p_dst[3-FDEC_STRIDE] );
764
765 x264_mb_encode_i4x4( h, p, i, i_qp, i_mode, 1 );
766 }
767 }
768 }
769 else /* Inter MB */
770 {
771 int i_decimate_mb = 0;
772
773 /* Don't repeat motion compensation if it was already done in non-RD transform analysis */
774 if( !h->mb.b_skip_mc )
775 x264_mb_mc( h );
776
777 if( h->mb.b_lossless )
778 {
779 if( h->mb.b_transform_8x8 )
780 for( int p = 0; p < plane_count; p++ )
781 for( int i8x8 = 0; i8x8 < 4; i8x8++ )
782 {
783 int x = i8x8&1;
784 int y = i8x8>>1;
785 nz = h->zigzagf.sub_8x8( h->dct.luma8x8[p*4+i8x8], h->mb.pic.p_fenc[p] + 8*x + 8*y*FENC_STRIDE,
786 h->mb.pic.p_fdec[p] + 8*x + 8*y*FDEC_STRIDE );
787 STORE_8x8_NNZ( p, i8x8, nz );
788 h->mb.i_cbp_luma |= nz << i8x8;
789 }
790 else
791 for( int p = 0; p < plane_count; p++ )
792 for( int i4x4 = 0; i4x4 < 16; i4x4++ )
793 {
794 nz = h->zigzagf.sub_4x4( h->dct.luma4x4[p*16+i4x4],
795 h->mb.pic.p_fenc[p]+block_idx_xy_fenc[i4x4],
796 h->mb.pic.p_fdec[p]+block_idx_xy_fdec[i4x4] );
797 h->mb.cache.non_zero_count[x264_scan8[p*16+i4x4]] = nz;
798 h->mb.i_cbp_luma |= nz << (i4x4>>2);
799 }
800 }
801 else if( h->mb.b_transform_8x8 )
802 {
803 ALIGNED_ARRAY_64( dctcoef, dct8x8,[4],[64] );
804 b_decimate &= !h->mb.b_trellis || !h->param.b_cabac; // 8x8 trellis is inherently optimal decimation for CABAC
805
806 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
807 {
808 int quant_cat = p ? CQM_8PC : CQM_8PY;
809 CLEAR_16x16_NNZ( p );
810 h->dctf.sub16x16_dct8( dct8x8, h->mb.pic.p_fenc[p], h->mb.pic.p_fdec[p] );
811 h->nr_count[1+!!p*2] += h->mb.b_noise_reduction * 4;
812
813 int plane_cbp = 0;
814 for( int idx = 0; idx < 4; idx++ )
815 {
816 nz = x264_quant_8x8( h, dct8x8[idx], i_qp, ctx_cat_plane[DCT_LUMA_8x8][p], 0, p, idx );
817
818 if( nz )
819 {
820 h->zigzagf.scan_8x8( h->dct.luma8x8[p*4+idx], dct8x8[idx] );
821 if( b_decimate )
822 {
823 int i_decimate_8x8 = h->quantf.decimate_score64( h->dct.luma8x8[p*4+idx] );
824 i_decimate_mb += i_decimate_8x8;
825 if( i_decimate_8x8 >= 4 )
826 plane_cbp |= 1<<idx;
827 }
828 else
829 plane_cbp |= 1<<idx;
830 }
831 }
832
833 if( i_decimate_mb >= 6 || !b_decimate )
834 {
835 h->mb.i_cbp_luma |= plane_cbp;
836 FOREACH_BIT( idx, 0, plane_cbp )
837 {
838 h->quantf.dequant_8x8( dct8x8[idx], h->dequant8_mf[quant_cat], i_qp );
839 h->dctf.add8x8_idct8( &h->mb.pic.p_fdec[p][8*(idx&1) + 8*(idx>>1)*FDEC_STRIDE], dct8x8[idx] );
840 STORE_8x8_NNZ( p, idx, 1 );
841 }
842 }
843 }
844 }
845 else
846 {
847 ALIGNED_ARRAY_64( dctcoef, dct4x4,[16],[16] );
848 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
849 {
850 int quant_cat = p ? CQM_4PC : CQM_4PY;
851 CLEAR_16x16_NNZ( p );
852 h->dctf.sub16x16_dct( dct4x4, h->mb.pic.p_fenc[p], h->mb.pic.p_fdec[p] );
853
854 if( h->mb.b_noise_reduction )
855 {
856 h->nr_count[0+!!p*2] += 16;
857 for( int idx = 0; idx < 16; idx++ )
858 h->quantf.denoise_dct( dct4x4[idx], h->nr_residual_sum[0+!!p*2], h->nr_offset[0+!!p*2], 16 );
859 }
860
861 int plane_cbp = 0;
862 for( int i8x8 = 0; i8x8 < 4; i8x8++ )
863 {
864 int i_decimate_8x8 = b_decimate ? 0 : 6;
865 int nnz8x8 = 0;
866 if( h->mb.b_trellis )
867 {
868 for( int i4x4 = 0; i4x4 < 4; i4x4++ )
869 {
870 int idx = i8x8*4+i4x4;
871 if( x264_quant_4x4_trellis( h, dct4x4[idx], quant_cat, i_qp, ctx_cat_plane[DCT_LUMA_4x4][p], 0, !!p, p*16+idx ) )
872 {
873 h->zigzagf.scan_4x4( h->dct.luma4x4[p*16+idx], dct4x4[idx] );
874 h->quantf.dequant_4x4( dct4x4[idx], h->dequant4_mf[quant_cat], i_qp );
875 if( i_decimate_8x8 < 6 )
876 i_decimate_8x8 += h->quantf.decimate_score16( h->dct.luma4x4[p*16+idx] );
877 h->mb.cache.non_zero_count[x264_scan8[p*16+idx]] = 1;
878 nnz8x8 = 1;
879 }
880 }
881 }
882 else
883 {
884 nnz8x8 = nz = h->quantf.quant_4x4x4( &dct4x4[i8x8*4], h->quant4_mf[quant_cat][i_qp], h->quant4_bias[quant_cat][i_qp] );
885 if( nz )
886 {
887 FOREACH_BIT( idx, i8x8*4, nz )
888 {
889 h->zigzagf.scan_4x4( h->dct.luma4x4[p*16+idx], dct4x4[idx] );
890 h->quantf.dequant_4x4( dct4x4[idx], h->dequant4_mf[quant_cat], i_qp );
891 if( i_decimate_8x8 < 6 )
892 i_decimate_8x8 += h->quantf.decimate_score16( h->dct.luma4x4[p*16+idx] );
893 h->mb.cache.non_zero_count[x264_scan8[p*16+idx]] = 1;
894 }
895 }
896 }
897 if( nnz8x8 )
898 {
899 i_decimate_mb += i_decimate_8x8;
900 if( i_decimate_8x8 < 4 )
901 STORE_8x8_NNZ( p, i8x8, 0 );
902 else
903 plane_cbp |= 1<<i8x8;
904 }
905 }
906
907 if( i_decimate_mb < 6 )
908 {
909 plane_cbp = 0;
910 CLEAR_16x16_NNZ( p );
911 }
912 else
913 {
914 h->mb.i_cbp_luma |= plane_cbp;
915 FOREACH_BIT( i8x8, 0, plane_cbp )
916 {
917 h->dctf.add8x8_idct( &h->mb.pic.p_fdec[p][(i8x8&1)*8 + (i8x8>>1)*8*FDEC_STRIDE], &dct4x4[i8x8*4] );
918 }
919 }
920 }
921 }
922 }
923
924 /* encode chroma */
925 if( chroma )
926 {
927 if( IS_INTRA( h->mb.i_type ) )
928 {
929 int i_mode = h->mb.i_chroma_pred_mode;
930 if( h->mb.b_lossless )
931 x264_predict_lossless_chroma( h, i_mode );
932 else
933 {
934 h->predict_chroma[i_mode]( h->mb.pic.p_fdec[1] );
935 h->predict_chroma[i_mode]( h->mb.pic.p_fdec[2] );
936 }
937 }
938
939 /* encode the 8x8 blocks */
940 x264_mb_encode_chroma( h, !IS_INTRA( h->mb.i_type ), h->mb.i_chroma_qp );
941 }
942 else
943 h->mb.i_cbp_chroma = 0;
944
945 /* store cbp */
946 int cbp = h->mb.i_cbp_chroma << 4 | h->mb.i_cbp_luma;
947 if( h->param.b_cabac )
948 cbp |= h->mb.cache.non_zero_count[x264_scan8[LUMA_DC ]] << 8
949 | h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+0]] << 9
950 | h->mb.cache.non_zero_count[x264_scan8[CHROMA_DC+1]] << 10;
951 h->mb.cbp[h->mb.i_mb_xy] = cbp;
952
953 /* Check for P_SKIP
954 * XXX: in the me perhaps we should take x264_mb_predict_mv_pskip into account
955 * (if multiple mv give same result)*/
956 if( !b_force_no_skip )
957 {
958 if( h->mb.i_type == P_L0 && h->mb.i_partition == D_16x16 &&
959 !(h->mb.i_cbp_luma | h->mb.i_cbp_chroma) &&
960 M32( h->mb.cache.mv[0][x264_scan8[0]] ) == M32( h->mb.cache.pskip_mv )
961 && h->mb.cache.ref[0][x264_scan8[0]] == 0 )
962 {
963 h->mb.i_type = P_SKIP;
964 }
965
966 /* Check for B_SKIP */
967 if( h->mb.i_type == B_DIRECT && !(h->mb.i_cbp_luma | h->mb.i_cbp_chroma) )
968 {
969 h->mb.i_type = B_SKIP;
970 }
971 }
972}

984/*****************************************************************************
985 * x264_macroblock_probe_skip:
986 * Check if the current MB could be encoded as a [PB]_SKIP
987 *****************************************************************************/
988static ALWAYS_INLINE int macroblock_probe_skip_internal( x264_t *h, int b_bidir, int plane_count, int chroma )
989{
990 ALIGNED_ARRAY_64( dctcoef, dct4x4,[8],[16] );
991 ALIGNED_ARRAY_64( dctcoef, dctscan,[16] );
992 ALIGNED_4( int16_t mvp[2] );
993 int i_qp = h->mb.i_qp;
994
995 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
996 {
997 int quant_cat = p ? CQM_4PC : CQM_4PY;
998 if( !b_bidir )
999 {
1000 /* Get the MV */
1001 mvp[0] = x264_clip3( h->mb.cache.pskip_mv[0], h->mb.mv_min[0], h->mb.mv_max[0] );
1002 mvp[1] = x264_clip3( h->mb.cache.pskip_mv[1], h->mb.mv_min[1], h->mb.mv_max[1] );
1003
1004 /* Motion compensation */
1005 h->mc.mc_luma( h->mb.pic.p_fdec[p], FDEC_STRIDE,
1006 &h->mb.pic.p_fref[0][0][p*4], h->mb.pic.i_stride[p],
1007 mvp[0], mvp[1], 16, 16, &h->sh.weight[0][p] );
1008 }
1009
1010 for( int i8x8 = 0, i_decimate_mb = 0; i8x8 < 4; i8x8++ )
1011 {
1012 int fenc_offset = (i8x8&1) * 8 + (i8x8>>1) * FENC_STRIDE * 8;
1013 int fdec_offset = (i8x8&1) * 8 + (i8x8>>1) * FDEC_STRIDE * 8;
1014
1015 h->dctf.sub8x8_dct( dct4x4, h->mb.pic.p_fenc[p] + fenc_offset,
1016 h->mb.pic.p_fdec[p] + fdec_offset );
1017
1018 if( h->mb.b_noise_reduction )
1019 for( int i4x4 = 0; i4x4 < 4; i4x4++ )
1020 h->quantf.denoise_dct( dct4x4[i4x4], h->nr_residual_sum[0+!!p*2], h->nr_offset[0+!!p*2], 16 );
1021
1022 int nz = h->quantf.quant_4x4x4( dct4x4, h->quant4_mf[quant_cat][i_qp], h->quant4_bias[quant_cat][i_qp] );
1023 FOREACH_BIT( idx, 0, nz )
1024 {
1025 h->zigzagf.scan_4x4( dctscan, dct4x4[idx] );
1026 i_decimate_mb += h->quantf.decimate_score16( dctscan );
1027 if( i_decimate_mb >= 6 )
1028 return 0;
1029 }
1030 }
1031 }
1032
1033 if( chroma == CHROMA_420 || chroma == CHROMA_422 )
1034 {
1035 i_qp = h->mb.i_chroma_qp;
1036 int chroma422 = chroma == CHROMA_422;
1037 int thresh = chroma422 ? (x264_lambda2_tab[i_qp] + 16) >> 5 : (x264_lambda2_tab[i_qp] + 32) >> 6;
1038 int ssd;
1039 ALIGNED_ARRAY_16( dctcoef, dct_dc,[8] );
1040
1041 if( !b_bidir )
1042 {
1043 /* Special case for mv0, which is (of course) very common in P-skip mode. */
1044 if( M32( mvp ) )
1045 h->mc.mc_chroma( h->mb.pic.p_fdec[1], h->mb.pic.p_fdec[2], FDEC_STRIDE,
1046 h->mb.pic.p_fref[0][0][4], h->mb.pic.i_stride[1],
1047 mvp[0], mvp[1] * (1<<chroma422), 8, chroma422?16:8 );
1048 else
1049 h->mc.load_deinterleave_chroma_fdec( h->mb.pic.p_fdec[1], h->mb.pic.p_fref[0][0][4],
1050 h->mb.pic.i_stride[1], chroma422?16:8 );
1051 }
1052
1053 for( int ch = 0; ch < 2; ch++ )
1054 {
1055 pixel *p_src = h->mb.pic.p_fenc[1+ch];
1056 pixel *p_dst = h->mb.pic.p_fdec[1+ch];
1057
1058 if( !b_bidir && h->sh.weight[0][1+ch].weightfn )
1059 h->sh.weight[0][1+ch].weightfn[8>>2]( h->mb.pic.p_fdec[1+ch], FDEC_STRIDE,
1060 h->mb.pic.p_fdec[1+ch], FDEC_STRIDE,
1061 &h->sh.weight[0][1+ch], chroma422?16:8 );
1062
1063 /* there is almost never a termination during chroma, but we can't avoid the check entirely */
1064 /* so instead we check SSD and skip the actual check if the score is low enough. */
1065 ssd = h->pixf.ssd[chroma422?PIXEL_8x16:PIXEL_8x8]( p_dst, FDEC_STRIDE, p_src, FENC_STRIDE );
1066 if( ssd < thresh )
1067 continue;
1068
1069 /* The vast majority of chroma checks will terminate during the DC check or the higher
1070 * threshold check, so we can save time by doing a DC-only DCT. */
1071 if( h->mb.b_noise_reduction )
1072 {
1073 for( int i = 0; i <= chroma422; i++ )
1074 h->dctf.sub8x8_dct( &dct4x4[4*i], p_src + 8*i*FENC_STRIDE, p_dst + 8*i*FDEC_STRIDE );
1075
1076 for( int i4x4 = 0; i4x4 < (chroma422?8:4); i4x4++ )
1077 {
1078 h->quantf.denoise_dct( dct4x4[i4x4], h->nr_residual_sum[2], h->nr_offset[2], 16 );
1079 dct_dc[i4x4] = dct4x4[i4x4][0];
1080 dct4x4[i4x4][0] = 0;
1081 }
1082 }
1083 else
1084 {
1085 if( chroma422 )
1086 h->dctf.sub8x16_dct_dc( dct_dc, p_src, p_dst );
1087 else
1088 h->dctf.sub8x8_dct_dc( dct_dc, p_src, p_dst );
1089 }
1090
1091 for( int i = 0; i <= chroma422; i++ )
1092 if( h->quantf.quant_2x2_dc( &dct_dc[4*i], h->quant4_mf[CQM_4PC][i_qp+3*chroma422][0] >> 1,
1093 h->quant4_bias[CQM_4PC][i_qp+3*chroma422][0] << 1 ) )
1094 return 0;
1095
1096 /* If there wasn't a termination in DC, we can check against a much higher threshold. */
1097 if( ssd < thresh*4 )
1098 continue;
1099
1100 if( !h->mb.b_noise_reduction )
1101 for( int i = 0; i <= chroma422; i++ )
1102 {
1103 h->dctf.sub8x8_dct( &dct4x4[4*i], p_src + 8*i*FENC_STRIDE, p_dst + 8*i*FDEC_STRIDE );
1104 dct4x4[i*4+0][0] = 0;
1105 dct4x4[i*4+1][0] = 0;
1106 dct4x4[i*4+2][0] = 0;
1107 dct4x4[i*4+3][0] = 0;
1108 }
1109
1110 /* calculate dct coeffs */
1111 for( int i8x8 = 0, i_decimate_mb = 0; i8x8 < (chroma422?2:1); i8x8++ )
1112 {
1113 int nz = h->quantf.quant_4x4x4( &dct4x4[i8x8*4], h->quant4_mf[CQM_4PC][i_qp], h->quant4_bias[CQM_4PC][i_qp] );
1114 FOREACH_BIT( idx, i8x8*4, nz )
1115 {
1116 h->zigzagf.scan_4x4( dctscan, dct4x4[idx] );
1117 i_decimate_mb += h->quantf.decimate_score15( dctscan );
1118 if( i_decimate_mb >= 7 )
1119 return 0;
1120 }
1121 }
1122 }
1123 }
1124
1125 h->mb.b_skip_mc = 1;
1126 return 1;
1127}

1141/****************************************************************************
1142 * DCT-domain noise reduction / adaptive deadzone
1143 * from libavcodec
1144 ****************************************************************************/
1145
1146void x264_noise_reduction_update( x264_t *h )
1147{
1148 h->nr_offset = h->nr_offset_denoise;
1149 h->nr_residual_sum = h->nr_residual_sum_buf[0];
1150 h->nr_count = h->nr_count_buf[0];
1151 for( int cat = 0; cat < 3 + CHROMA444; cat++ )
1152 {
1153 int dct8x8 = cat&1;
1154 int size = dct8x8 ? 64 : 16;
1155 const uint32_t *weight = dct8x8 ? x264_dct8_weight2_tab : x264_dct4_weight2_tab;
1156
1157 if( h->nr_count[cat] > (dct8x8 ? (1<<16) : (1<<18)) )
1158 {
1159 for( int i = 0; i < size; i++ )
1160 h->nr_residual_sum[cat][i] >>= 1;
1161 h->nr_count[cat] >>= 1;
1162 }
1163
1164 for( int i = 0; i < size; i++ )
1165 h->nr_offset[cat][i] =
1166 ((uint64_t)h->param.analyse.i_noise_reduction * h->nr_count[cat]
1167 + h->nr_residual_sum[cat][i]/2)
1168 / ((uint64_t)h->nr_residual_sum[cat][i] * weight[i]/256 + 1);
1169
1170 /* Don't denoise DC coefficients */
1171 h->nr_offset[cat][0] = 0;
1172 }
1173}

1175/*****************************************************************************
1176 * RD only; 4 calls to this do not make up for one macroblock_encode.
1177 * doesn't transform chroma dc.
1178 *****************************************************************************/
1179static ALWAYS_INLINE void macroblock_encode_p8x8_internal( x264_t *h, int i8, int plane_count, int chroma )
1180{
1181 int b_decimate = h->mb.b_dct_decimate;
1182 int i_qp = h->mb.i_qp;
1183 int x = i8&1;
1184 int y = i8>>1;
1185 int nz;
1186 int chroma422 = chroma == CHROMA_422;
1187
1188 h->mb.i_cbp_chroma = 0;
1189 h->mb.i_cbp_luma &= ~(1 << i8);
1190
1191 if( !h->mb.b_skip_mc )
1192 x264_mb_mc_8x8( h, i8 );
1193
1194 if( h->mb.b_lossless )
1195 {
1196 for( int p = 0; p < plane_count; p++ )
1197 {
1198 pixel *p_fenc = h->mb.pic.p_fenc[p] + 8*x + 8*y*FENC_STRIDE;
1199 pixel *p_fdec = h->mb.pic.p_fdec[p] + 8*x + 8*y*FDEC_STRIDE;
1200 int nnz8x8 = 0;
1201 if( h->mb.b_transform_8x8 )
1202 {
1203 nnz8x8 = h->zigzagf.sub_8x8( h->dct.luma8x8[4*p+i8], p_fenc, p_fdec );
1204 STORE_8x8_NNZ( p, i8, nnz8x8 );
1205 }
1206 else
1207 {
1208 for( int i4 = i8*4; i4 < i8*4+4; i4++ )
1209 {
1210 nz = h->zigzagf.sub_4x4( h->dct.luma4x4[16*p+i4],
1211 h->mb.pic.p_fenc[p]+block_idx_xy_fenc[i4],
1212 h->mb.pic.p_fdec[p]+block_idx_xy_fdec[i4] );
1213 h->mb.cache.non_zero_count[x264_scan8[16*p+i4]] = nz;
1214 nnz8x8 |= nz;
1215 }
1216 }
1217 h->mb.i_cbp_luma |= nnz8x8 << i8;
1218 }
1219 if( chroma == CHROMA_420 || chroma == CHROMA_422 )
1220 {
1221 for( int ch = 0; ch < 2; ch++ )
1222 {
1223 dctcoef dc;
1224 pixel *p_fenc = h->mb.pic.p_fenc[1+ch] + 4*x + (chroma422?8:4)*y*FENC_STRIDE;
1225 pixel *p_fdec = h->mb.pic.p_fdec[1+ch] + 4*x + (chroma422?8:4)*y*FDEC_STRIDE;
1226
1227 for( int i4x4 = 0; i4x4 <= chroma422; i4x4++ )
1228 {
1229 int offset = chroma422 ? 8*y + 2*i4x4 + x : i8;
1230 nz = h->zigzagf.sub_4x4ac( h->dct.luma4x4[16+offset+ch*16], p_fenc+4*i4x4*FENC_STRIDE, p_fdec+4*i4x4*FDEC_STRIDE, &dc );
1231 h->mb.cache.non_zero_count[x264_scan8[16+offset+ch*16]] = nz;
1232 }
1233 }
1234 h->mb.i_cbp_chroma = 0x02;
1235 }
1236 }
1237 else
1238 {
1239 if( h->mb.b_transform_8x8 )
1240 {
1241 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
1242 {
1243 int quant_cat = p ? CQM_8PC : CQM_8PY;
1244 pixel *p_fenc = h->mb.pic.p_fenc[p] + 8*x + 8*y*FENC_STRIDE;
1245 pixel *p_fdec = h->mb.pic.p_fdec[p] + 8*x + 8*y*FDEC_STRIDE;
1246 ALIGNED_ARRAY_64( dctcoef, dct8x8,[64] );
1247
1248 h->dctf.sub8x8_dct8( dct8x8, p_fenc, p_fdec );
1249 int nnz8x8 = x264_quant_8x8( h, dct8x8, i_qp, ctx_cat_plane[DCT_LUMA_8x8][p], 0, p, i8 );
1250 if( nnz8x8 )
1251 {
1252 h->zigzagf.scan_8x8( h->dct.luma8x8[4*p+i8], dct8x8 );
1253
1254 if( b_decimate && !h->mb.b_trellis )
1255 nnz8x8 = 4 <= h->quantf.decimate_score64( h->dct.luma8x8[4*p+i8] );
1256
1257 if( nnz8x8 )
1258 {
1259 h->quantf.dequant_8x8( dct8x8, h->dequant8_mf[quant_cat], i_qp );
1260 h->dctf.add8x8_idct8( p_fdec, dct8x8 );
1261 STORE_8x8_NNZ( p, i8, 1 );
1262 h->mb.i_cbp_luma |= 1 << i8;
1263 }
1264 else
1265 STORE_8x8_NNZ( p, i8, 0 );
1266 }
1267 else
1268 STORE_8x8_NNZ( p, i8, 0 );
1269 }
1270 }
1271 else
1272 {
1273 for( int p = 0; p < plane_count; p++, i_qp = h->mb.i_chroma_qp )
1274 {
1275 int quant_cat = p ? CQM_4PC : CQM_4PY;
1276 pixel *p_fenc = h->mb.pic.p_fenc[p] + 8*x + 8*y*FENC_STRIDE;
1277 pixel *p_fdec = h->mb.pic.p_fdec[p] + 8*x + 8*y*FDEC_STRIDE;
1278 int i_decimate_8x8 = b_decimate ? 0 : 4;
1279 ALIGNED_ARRAY_64( dctcoef, dct4x4,[4],[16] );
1280 int nnz8x8 = 0;
1281
1282 h->dctf.sub8x8_dct( dct4x4, p_fenc, p_fdec );
1283 STORE_8x8_NNZ( p, i8, 0 );
1284
1285 if( h->mb.b_noise_reduction )
1286 for( int idx = 0; idx < 4; idx++ )
1287 h->quantf.denoise_dct( dct4x4[idx], h->nr_residual_sum[0+!!p*2], h->nr_offset[0+!!p*2], 16 );
1288
1289 if( h->mb.b_trellis )
1290 {
1291 for( int i4x4 = 0; i4x4 < 4; i4x4++ )
1292 {
1293 if( x264_quant_4x4_trellis( h, dct4x4[i4x4], quant_cat, i_qp, ctx_cat_plane[DCT_LUMA_4x4][p], 0, !!p, i8*4+i4x4+p*16 ) )
1294 {
1295 h->zigzagf.scan_4x4( h->dct.luma4x4[p*16+i8*4+i4x4], dct4x4[i4x4] );
1296 h->quantf.dequant_4x4( dct4x4[i4x4], h->dequant4_mf[quant_cat], i_qp );
1297 if( i_decimate_8x8 < 4 )
1298 i_decimate_8x8 += h->quantf.decimate_score16( h->dct.luma4x4[p*16+i8*4+i4x4] );
1299 h->mb.cache.non_zero_count[x264_scan8[p*16+i8*4+i4x4]] = 1;
1300 nnz8x8 = 1;
1301 }
1302 }
1303 }
1304 else
1305 {
1306 nnz8x8 = nz = h->quantf.quant_4x4x4( dct4x4, h->quant4_mf[quant_cat][i_qp], h->quant4_bias[quant_cat][i_qp] );
1307 if( nz )
1308 {
1309 FOREACH_BIT( i4x4, 0, nz )
1310 {
1311 h->zigzagf.scan_4x4( h->dct.luma4x4[p*16+i8*4+i4x4], dct4x4[i4x4] );
1312 h->quantf.dequant_4x4( dct4x4[i4x4], h->dequant4_mf[quant_cat], i_qp );
1313 if( i_decimate_8x8 < 4 )
1314 i_decimate_8x8 += h->quantf.decimate_score16( h->dct.luma4x4[p*16+i8*4+i4x4] );
1315 h->mb.cache.non_zero_count[x264_scan8[p*16+i8*4+i4x4]] = 1;
1316 }
1317 }
1318 }
1319 if( nnz8x8 )
1320 {
1321 /* decimate this 8x8 block */
1322 if( i_decimate_8x8 < 4 )
1323 STORE_8x8_NNZ( p, i8, 0 );
1324 else
1325 {
1326 h->dctf.add8x8_idct( p_fdec, dct4x4 );
1327 h->mb.i_cbp_luma |= 1 << i8;
1328 }
1329 }
1330 }
1331 }
1332
1333 if( chroma == CHROMA_420 || chroma == CHROMA_422 )
1334 {
1335 i_qp = h->mb.i_chroma_qp;
1336 for( int ch = 0; ch < 2; ch++ )
1337 {
1338 ALIGNED_ARRAY_64( dctcoef, dct4x4,[2],[16] );
1339 pixel *p_fenc = h->mb.pic.p_fenc[1+ch] + 4*x + (chroma422?8:4)*y*FENC_STRIDE;
1340 pixel *p_fdec = h->mb.pic.p_fdec[1+ch] + 4*x + (chroma422?8:4)*y*FDEC_STRIDE;
1341
1342 for( int i4x4 = 0; i4x4 <= chroma422; i4x4++ )
1343 {
1344 h->dctf.sub4x4_dct( dct4x4[i4x4], p_fenc + 4*i4x4*FENC_STRIDE, p_fdec + 4*i4x4*FDEC_STRIDE );
1345
1346 if( h->mb.b_noise_reduction )
1347 h->quantf.denoise_dct( dct4x4[i4x4], h->nr_residual_sum[2], h->nr_offset[2], 16 );
1348 dct4x4[i4x4][0] = 0;
1349
1350 if( h->mb.b_trellis )
1351 nz = x264_quant_4x4_trellis( h, dct4x4[i4x4], CQM_4PC, i_qp, DCT_CHROMA_AC, 0, 1, 0 );
1352 else
1353 nz = h->quantf.quant_4x4( dct4x4[i4x4], h->quant4_mf[CQM_4PC][i_qp], h->quant4_bias[CQM_4PC][i_qp] );
1354
1355 int offset = chroma422 ? ((5*i8) & 0x09) + 2*i4x4 : i8;
1356 h->mb.cache.non_zero_count[x264_scan8[16+offset+ch*16]] = nz;
1357 if( nz )
1358 {
1359 h->zigzagf.scan_4x4( h->dct.luma4x4[16+offset+ch*16], dct4x4[i4x4] );
1360 h->quantf.dequant_4x4( dct4x4[i4x4], h->dequant4_mf[CQM_4PC], i_qp );
1361 h->dctf.add4x4_idct( p_fdec + 4*i4x4*FDEC_STRIDE, dct4x4[i4x4] );
1362 }
1363 }
1364 }
1365 h->mb.i_cbp_chroma = 0x02;
1366 }
1367 }
1368}

1370void x264_macroblock_encode_p8x8( x264_t *h, int i8 )
1371{
1372 if( CHROMA_FORMAT == CHROMA_420 )
