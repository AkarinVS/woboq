231static int list_contains( const char * const *list, const char *s )
232{
233 if( *s )
234 for( ; *list; list++ )
235 if( !strcmp( *list, s ) )
236 return 1;
237 return 0;
238}
239
240static void suggest( const char *s, const char *cur, int cur_len )
241{
242 if( s && *s && !strncmp( s, cur, cur_len ) )
243 printf( "%s\n", s );
244}
245
246static void suggest_lower( const char *s, const char *cur, int cur_len )
247{
248 if( s && *s && !strncasecmp( s, cur, cur_len ) )
249 {
250 for( ; *s; s++ )
251 putchar( *s < 'A' || *s > 'Z' ? *s : *s | 0x20 );
252 putchar( '\n' );
253 }
254}
255
256static void suggest_num_range( int start, int end, const char *cur, int cur_len )
257{
258 char buf[16];
259 for( int i = start; i <= end; i++ )
260 {
261 snprintf( buf, sizeof( buf ), "%d", i );
262 suggest( buf, cur, cur_len );
263 }
264}

266#if HAVE_LAVF
267/* Suggest each token in a string separated by delimiters. */
268static void suggest_token( const char *s, int delim, const char *cur, int cur_len )
269{
270 if( s && *s )
271 {
272 for( const char *tok_end; (tok_end = strchr( s, delim )); s = tok_end + 1 )
273 {
274 int tok_len = tok_end - s;
275 if( tok_len && tok_len >= cur_len && !strncmp( s, cur, cur_len ) )
276 printf( "%.*s\n", tok_len, s );
277 }
278 suggest( s, cur, cur_len );
279 }
280}
281#endif

283#define OPT( opt ) else if( !strcmp( prev, opt ) )
284#define OPT2( opt1, opt2 ) else if( !strcmp( prev, opt1 ) || !strcmp( prev, opt2 ) )
285#define OPT_TYPE( type ) list_contains( opts_##type, prev )
286
287#define suggest( s ) suggest( s, cur, cur_len )
288#define suggest_lower( s ) suggest_lower( s, cur, cur_len )
289#define suggest_list( list ) for( const char * const *s = list; *s; s++ ) suggest( *s )
290#define suggest_num_range( start, end ) suggest_num_range( start, end, cur, cur_len )
291#define suggest_token( s, delim ) suggest_token( s, delim, cur, cur_len )
292
293int x264_cli_autocomplete( const char *prev, const char *cur )
294{
295 int cur_len = strlen( cur );
296 if( 0 );
297 OPT( "--alternative-transfer" )
298 suggest_list( x264_transfer_names );
299 OPT( "--aq-mode" )
300 suggest_num_range( 0, 3 );
301 OPT( "--asm" )
302 for( const x264_cpu_name_t *cpu = x264_cpu_names; cpu->flags; cpu++ )
303 suggest_lower( cpu->name );
304 OPT( "--avcintra-class" )
305 suggest_list( x264_avcintra_class_names );
306 OPT( "--avcintra-flavor" )
307 suggest_list( x264_avcintra_flavor_names );
308 OPT( "--b-adapt" )
309 suggest_num_range( 0, 2 );
310 OPT( "--b-pyramid" )
311 suggest_list( x264_b_pyramid_names );
312 OPT( "--colormatrix" )
313 suggest_list( x264_colmatrix_names );
314 OPT( "--colorprim" )
315 suggest_list( x264_colorprim_names );
316 OPT( "--cqm" )
317 suggest_list( x264_cqm_names );
318 OPT( "--demuxer" )
319 suggest_list( x264_demuxer_names );
320 OPT( "--direct" )
321 suggest_list( x264_direct_pred_names );
322 OPT( "--frame-packing" )
323 suggest_num_range( 0, 7 );
324 OPT( "--input-csp" )
325 {
326 for( int i = X264_CSP_NONE+1; i < X264_CSP_CLI_MAX; i++ )
327 suggest( x264_cli_csps[i].name );
328#if HAVE_LAVF
329 for( const AVPixFmtDescriptor *d = NULL; (d = av_pix_fmt_desc_next( d )); )
330 suggest( d->name );
331#endif
332 }
333 OPT( "--input-fmt" )
334 {
335#if HAVE_LAVF
336 void *i = NULL;
337 for( const AVInputFormat *f; (f = av_demuxer_iterate( &i )); )
338 suggest_token( f->name, ',' );
339#endif
340 }
341 OPT( "--input-range" )
342 suggest_list( x264_range_names );
343 OPT( "--level" )
344 suggest_list( level_names );
345 OPT( "--log-level" )
346 suggest_list( x264_log_level_names );
347 OPT( "--me" )
348 suggest_list( x264_motion_est_names );
349 OPT( "--muxer" )
350 suggest_list( x264_muxer_names );
351 OPT( "--nal-hrd" )
352 suggest_list( x264_nal_hrd_names );
353 OPT( "--output-csp" )
354 suggest_list( x264_output_csp_names );
355 OPT( "--output-depth" )
356 {
357#if HAVE_BITDEPTH8
358 suggest( "8" );
359#endif
360#if HAVE_BITDEPTH10
361 suggest( "10" );
362#endif
363 }
364 OPT( "--overscan" )
365 suggest_list( x264_overscan_names );
366 OPT2( "--partitions", "-A" )
367 suggest_list( x264_partition_names );
368 OPT2( "--pass", "-p" )
369 suggest_num_range( 1, 3 );
370 OPT( "--preset" )
371 suggest_list( x264_preset_names );
372 OPT( "--profile" )
373 suggest_list( x264_valid_profile_names );
374 OPT( "--pulldown" )
375 suggest_list( x264_pulldown_names );
376 OPT( "--range" )
377 suggest_list( x264_range_names );
378 OPT2( "--subme", "-m" )
379 suggest_num_range( 0, 11 );
380 OPT( "--transfer" )
381 suggest_list( x264_transfer_names );
382 OPT2( "--trellis", "-t" )
383 suggest_num_range( 0, 2 );
384 OPT( "--tune" )
385 suggest_list( x264_tune_names );
386 OPT( "--videoformat" )
387 suggest_list( x264_vidformat_names );
388 OPT( "--weightp" )
389 suggest_num_range( 0, 2 );
390 else if( !OPT_TYPE( nosuggest ) && !OPT_TYPE( special ) )
391 {
392 if( OPT_TYPE( filename ) || strncmp( cur, "--", 2 ) )
393 return 1; /* Fall back to default shell filename autocomplete. */
394
395 /* Suggest options. */
396 suggest_list( opts_suggest );
397 suggest_list( opts_nosuggest );
398 suggest_list( opts_filename );
399 suggest_list( opts_standalone );
400
401 /* Only suggest special options if no other options have been specified. */
402 if( !*prev )
403 suggest_list( opts_special );
404 }
405
406 return 0;
407}
