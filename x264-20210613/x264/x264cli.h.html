50static inline uint64_t gcd( uint64_t a, uint64_t b )
51{
52 while( 1 )
53 {
54 int64_t c = a % b;
55 if( !c )
56 return b;
57 a = b;
58 b = c;
59 }
60}
61
62static inline uint64_t lcm( uint64_t a, uint64_t b )
63{
64 return ( a / gcd( a, b ) ) * b;
65}

67static inline char *get_filename_extension( char *filename )
68{
69 char *ext = filename + strlen( filename );
70 while( *ext != '.' && ext > filename )
71 ext--;
72 ext += *ext == '.';
73 return ext;
74}
