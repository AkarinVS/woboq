316struct perf_branch_entry {
317 uint64_t from;
318 uint64_t to;
319 uint64_t mispred:1, /* target mispredicted */
320 predicted:1,/* target predicted */
321 in_tx:1, /* in transaction */
322 abort:1, /* transaction abort */
323 cycles:16, /* cycle count to last branch */
324 type:4, /* branch type */
325 reserved:40;
326};
327
328/*
329 * branch stack layout:
330 * nr: number of taken branches stored in entries[]
331 *
332 * Note that nr can vary from sample to sample
333 * branches (to, from) are stored from most recent
334 * to least recent, i.e., entries[0] contains the most
335 * recent branch.
336 */
337struct perf_branch_stack {
338 uint64_t nr;
339 struct perf_branch_entry entries[0];
340};

516/*
517 * perf_event_open() syscall stub
518 */
519static inline int
520perf_event_open(
521 struct perf_event_attr *hw_event_uptr,
522 pid_t pid,
523 int cpu,
524 int group_fd,
525 unsigned long flags)
526{
527 return syscall(
528 __NR_perf_event_open, hw_event_uptr, pid, cpu, group_fd, flags);
529}

555/* type of opcode (load/store/prefetch,code) */
556#define PERF_MEM_OP_NA 0x01 /* not available */
557#define PERF_MEM_OP_LOAD 0x02 /* load instruction */
558#define PERF_MEM_OP_STORE 0x04 /* store instruction */
559#define PERF_MEM_OP_PFETCH 0x08 /* prefetch */
560#define PERF_MEM_OP_EXEC 0x10 /* code (execution) */
561#define PERF_MEM_OP_SHIFT 0
562
563/* memory hierarchy (memory level, hit or miss) */
564#define PERF_MEM_LVL_NA 0x01 /* not available */
565#define PERF_MEM_LVL_HIT 0x02 /* hit level */
566#define PERF_MEM_LVL_MISS 0x04 /* miss level */
567#define PERF_MEM_LVL_L1 0x08 /* L1 */
568#define PERF_MEM_LVL_LFB 0x10 /* Line Fill Buffer */
569#define PERF_MEM_LVL_L2 0x20 /* L2 */
570#define PERF_MEM_LVL_L3 0x40 /* L3 */
571#define PERF_MEM_LVL_LOC_RAM 0x80 /* Local DRAM */
572#define PERF_MEM_LVL_REM_RAM1 0x100 /* Remote DRAM (1 hop) */
573#define PERF_MEM_LVL_REM_RAM2 0x200 /* Remote DRAM (2 hops) */
574#define PERF_MEM_LVL_REM_CCE1 0x400 /* Remote Cache (1 hop) */
575#define PERF_MEM_LVL_REM_CCE2 0x800 /* Remote Cache (2 hops) */
576#define PERF_MEM_LVL_IO 0x1000 /* I/O memory */
577#define PERF_MEM_LVL_UNC 0x2000 /* Uncached memory */
578#define PERF_MEM_LVL_SHIFT 5
579
580#define PERF_MEM_REMOTE_REMOTE 0x01 /* Remote */
581#define PERF_MEM_REMOTE_SHIFT 37
582
583/* snoop mode */
584#define PERF_MEM_SNOOP_NA 0x01 /* not available */
585#define PERF_MEM_SNOOP_NONE 0x02 /* no snoop */
586#define PERF_MEM_SNOOP_HIT 0x04 /* snoop hit */
587#define PERF_MEM_SNOOP_MISS 0x08 /* snoop miss */
588#define PERF_MEM_SNOOP_HITM 0x10 /* snoop hit modified */
589#define PERF_MEM_SNOOP_SHIFT 19
590
591#define PERF_MEM_SNOOPX_FWD 0x01 /* forward */
592#define PERF_MEM_SNOOPX_SHIFT 37
593
594/* locked instruction */
595#define PERF_MEM_LOCK_NA 0x01 /* not available */
596#define PERF_MEM_LOCK_LOCKED 0x02 /* locked transaction */
597#define PERF_MEM_LOCK_SHIFT 24
598
599/* TLB access */
600#define PERF_MEM_TLB_NA 0x01 /* not available */
601#define PERF_MEM_TLB_HIT 0x02 /* hit level */
602#define PERF_MEM_TLB_MISS 0x04 /* miss level */
603#define PERF_MEM_TLB_L1 0x08 /* L1 */
604#define PERF_MEM_TLB_L2 0x10 /* L2 */
605#define PERF_MEM_TLB_WK 0x20 /* Hardware Walker*/
606#define PERF_MEM_TLB_OS 0x40 /* OS fault handler */
607#define PERF_MEM_TLB_SHIFT 26
