28#pragma GCC visibility push(default)
29
30#ifdef __cplusplus
31extern "C" {
32#endif
33
34/*
35 * use with PFM_OS_PERF, PFM_OS_PERF_EXT for pfm_get_os_event_encoding()
36 */
37typedef struct {
38 struct perf_event_attr *attr; /* in/out: perf_event struct pointer */
39 char **fstr; /* out/in: fully qualified event string */
40 size_t size; /* sizeof struct */
41 int idx; /* out: opaque event identifier */
42 int cpu; /* out: cpu to program, -1 = not set */
43 int flags; /* out: perf_event_open() flags */
44 int pad0; /* explicit 64-bit mode padding */
45} pfm_perf_encode_arg_t;
46
47#if __WORDSIZE == 64
48#define PFM_PERF_ENCODE_ABI0 40 /* includes 4-byte padding */
49#else
50#define PFM_PERF_ENCODE_ABI0 28
51#endif
52/*
53 * old interface, maintained for backward compatibility with older versions o
54 * the library. Should use pfm_get_os_event_encoding() now
55 */
56extern pfm_err_t pfm_get_perf_event_encoding(const char *str,
57 int dfl_plm,
58 struct perf_event_attr *output,
59 char **fstr,
60 int *idx);
61
62#ifdef __cplusplus /* extern C */
63}
64#endif
65
66#pragma GCC visibility pop
67
