221 auto &ctx = R.getASTContext();
222 BuiltinType::Kind kind = R.find("kind").readBuiltinTypeKind();
223
224 switch (kind) {
225#define IMAGE_TYPE(IMGTYPE, ID, SINGLETON_ID, ACCESS, SUFFIX) \
226 case BuiltinType::ID: return ctx.SINGLETON_ID;
227#include "clang/Basic/OpenCLImageTypes.def"
228
229#define EXT_OPAQUE_TYPE(EXTTYPE, ID, EXT) \
230 case BuiltinType::ID: return ctx.ID##Ty;
231#include "clang/Basic/OpenCLExtensionTypes.def"

233#define SVE_TYPE(NAME, ID, SINGLETON_ID) \
234 case BuiltinType::ID: return ctx.SINGLETON_ID;
235#include "clang/Basic/AArch64SVEACLETypes.def"

238 case BuiltinType::ID: return ctx.ID##Ty;

241#define BUILTIN_TYPE(ID, SINGLETON_ID) \
242 case BuiltinType::ID: return ctx.SINGLETON_ID;
243#include "clang/AST/BuiltinTypes.def"

442 QualType readInjectedClassNameType() {
443 auto &ctx = R.getASTContext();
444 Decl* declaration = R.find("declaration").readDeclRef();
445 QualType injectedSpecializationType = R.find("injectedSpecializationType").readQualType();
446
447 // FIXME: ASTContext::getInjectedClassNameType is not currently suitable
448 // for AST reading, too much interdependencies.
449 const Type *T = nullptr;
450 auto typeDecl = cast<CXXRecordDecl>(declaration);
451 for (auto *DI = typeDecl; DI; DI = DI->getPreviousDecl()) {
452 if (const Type *existing = DI->getTypeForDecl()) {
453 T = existing;
454 break;
455 }
456 }
457 if (!T) {
458 T = new (ctx, TypeAlignment)
459 InjectedClassNameType(typeDecl, injectedSpecializationType);
460 for (auto *DI = typeDecl; DI; DI = DI->getPreviousDecl())
461 DI->setTypeForDecl(T);
462 }
463 return QualType(T, 0);
464

648 QualType readTemplateSpecializationType() {
649 auto &ctx = R.getASTContext();
650 bool dependent = R.find("dependent").readBool();
651 TemplateName templateName = R.find("templateName").readTemplateName();
652 llvm::SmallVector<TemplateArgument, 8> templateArguments_buffer_0;
653 llvm::ArrayRef<TemplateArgument> templateArguments = R.find("templateArguments").template readArray<TemplateArgument>(templateArguments_buffer_0);
654 llvm::Optional<QualType> underlyingType = R.find("underlyingType").template readOptional<QualType>();
655
656 QualType result;
657 if (!underlyingType.hasValue()) {
658 result = ctx.getCanonicalTemplateSpecializationType(templateName,
659 templateArguments);
660 } else {
661 result = ctx.getTemplateSpecializationType(templateName,
662 templateArguments,
663 *underlyingType);
664 }
665 if (dependent)
