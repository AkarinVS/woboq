512template <class Impl>
513class BasicReaderBase {
514 ASTContext &C;
515protected:
516 BasicReaderBase(ASTContext &ctx) : C(ctx) {}
517public:
518 ASTContext &getASTContext() { return C; }
519 Impl &asImpl() { return static_cast<Impl&>(*this); }
520 APValue readAPValue() {
521 auto &ctx = asImpl().getASTContext();
522 auto &&subR = asImpl().readObject();
523 APValue::ValueKind kind = subR.find("kind").readAPValueKind();
524 switch (kind) {
525 case APValue::None: {
526 return APValue();
527 }
528
529 case APValue::FixedPoint: {
530 llvm::FixedPointSemantics semantics = subR.find("semantics").readFixedPointSemantics();
531 llvm::APSInt value = subR.find("value").readAPSInt();
532
533 return APValue(llvm::APFixedPoint(std::move(value), semantics));
534
535 }
536
537 case APValue::ComplexInt: {
538 llvm::APSInt real = subR.find("real").readAPSInt();
539 llvm::APSInt imag = subR.find("imag").readAPSInt();
540 return APValue(real, imag);
541 }
542
543 case APValue::ComplexFloat: {
544 uint32_t semantics = subR.find("semantics").readUInt32();
545 llvm::APInt real = subR.find("real").readAPInt();
546 llvm::APInt imag = subR.find("imag").readAPInt();
547
548 const llvm::fltSemantics &sema = llvm::APFloatBase::EnumToSemantics(
549 static_cast<llvm::APFloatBase::Semantics>(semantics));
550 return APValue(llvm::APFloat(sema, real),
551 llvm::APFloat(sema, imag));
552
553 }
554
555 case APValue::Vector: {
556 llvm::SmallVector<APValue, 8> elements_buffer_0;
557 llvm::ArrayRef<APValue> elements = subR.find("elements").template readArray<APValue>(elements_buffer_0);
558
559 APValue result;
560 result.MakeVector();
561 unsigned length = elements.size();
562 (void)result.setVectorUninit(length);
563 for (unsigned i = 0; i < length; i++)
564 result.getVectorElt(i) = elements[i];
565 return result;
566
567 }
568
569 case APValue::Indeterminate: {
570 return APValue::IndeterminateValue();
571 }
572
573 case APValue::Array: {
574 uint32_t totalLength = subR.find("totalLength").readUInt32();
575 bool hasFiller = subR.find("hasFiller").readBool();
576 llvm::SmallVector<APValue, 8> elements_buffer_0;
577 llvm::ArrayRef<APValue> elements = subR.find("elements").template readArray<APValue>(elements_buffer_0);
578
579 APValue result;
580 unsigned initLength = elements.size() - (hasFiller ? 1 : 0);
581 result.MakeArray(initLength, totalLength);
582 for (unsigned i = 0; i < initLength; ++i)
583 result.getArrayInitializedElt(i) = elements[i];
584 if (hasFiller)
585 result.getArrayFiller() = elements.back();
586 return result;
587
588 }
589
590 case APValue::Struct: {
591 llvm::SmallVector<APValue, 8> bases_buffer_0;
592 llvm::ArrayRef<APValue> bases = subR.find("bases").template readArray<APValue>(bases_buffer_0);
593 llvm::SmallVector<APValue, 8> fields_buffer_0;
594 llvm::ArrayRef<APValue> fields = subR.find("fields").template readArray<APValue>(fields_buffer_0);
595
596 APValue result;
597 result.MakeStruct(bases.size(), fields.size());
598 for (unsigned i = 0; i < bases.size(); ++i)
599 result.getStructBase(i) = bases[i];
600 for (unsigned i = 0; i < fields.size(); ++i)
601 result.getStructField(i) = fields[i];
602 return result;
603
604 }
605
606 case APValue::Union: {
607 Decl* fieldDecl = subR.find("fieldDecl").readDeclRef();
608 APValue value = subR.find("value").readAPValue();
609
610 return APValue(cast<clang::FieldDecl>(fieldDecl), std::move(value));
611
612 }
613
614 case APValue::AddrLabelDiff: {
615 Stmt* lhs = subR.find("lhs").readStmtRef();
616 Stmt* rhs = subR.find("rhs").readStmtRef();
617
618 return APValue(cast<AddrLabelExpr>(lhs), cast<AddrLabelExpr>(rhs));
619
620 }
621
622 case APValue::Int: {
623 llvm::APSInt value = subR.find("value").readAPSInt();
624 return APValue(value);
625 }
626
627 case APValue::MemberPointer: {
628 bool isDerived = subR.find("isDerived").readBool();
629 ValueDecl* member = subR.find("member").readValueDeclRef();
630 llvm::SmallVector<CXXRecordDecl*, 8> memberPath_buffer_0;
631 llvm::ArrayRef<CXXRecordDecl*> memberPath = subR.find("memberPath").template readArray<CXXRecordDecl*>(memberPath_buffer_0);
632
633 APValue result;
634 unsigned pathSize = memberPath.size();
635 const CXXRecordDecl **pathArray =
636 result.setMemberPointerUninit(member, isDerived, pathSize).data();
637 for (unsigned i = 0; i < pathSize; ++i)
638 pathArray[i] = memberPath[i]->getCanonicalDecl();
639 return result;
640
641 }
642
643 case APValue::LValue: {
644 bool hasLValuePath = subR.find("hasLValuePath").readBool();
645 bool isLValueOnePastTheEnd = subR.find("isLValueOnePastTheEnd").readBool();
646 bool isExpr = subR.find("isExpr").readBool();
647 bool isTypeInfo = subR.find("isTypeInfo").readBool();
648 bool hasBase = subR.find("hasBase").readBool();
649 bool isNullPtr = subR.find("isNullPtr").readBool();
650 llvm::Optional<QualType> typeInfo;
651 if ( hasBase && isTypeInfo ) {
652 typeInfo.emplace(subR.find("typeInfo").readQualType());
653 }
654 llvm::Optional<QualType> type;
655 if ( hasBase && isTypeInfo ) {
656 type.emplace(subR.find("type").readQualType());
657 }
658 llvm::Optional<uint32_t> callIndex;
659 if ( hasBase && !isTypeInfo ) {
660 callIndex.emplace(subR.find("callIndex").readUInt32());
661 }
662 llvm::Optional<uint32_t> version;
663 if ( hasBase && !isTypeInfo ) {
664 version.emplace(subR.find("version").readUInt32());
665 }
666 llvm::Optional<Stmt*> stmt;
667 if ( hasBase && !isTypeInfo && isExpr ) {
668 stmt.emplace(subR.find("stmt").readStmtRef());
669 }
670 llvm::Optional<Decl*> decl;
671 if ( hasBase && !isTypeInfo && !isExpr ) {
672 decl.emplace(subR.find("decl").readDeclRef());
673 }
674 uint32_t offsetQuantity = subR.find("offsetQuantity").readUInt32();
675 llvm::SmallVector<APValue::LValuePathEntry, 8> lvaluePath_buffer_0;
676 llvm::Optional<APValue::LValuePathSerializationHelper> lvaluePath;
677 if ( hasLValuePath ) {
678 lvaluePath.emplace(subR.find("lvaluePath").readLValuePathSerializationHelper(lvaluePath_buffer_0));
679 }
680
681 (void)ctx;
682 APValue::LValueBase base;
683 QualType elemTy;
684 if (hasBase) {
685 if (isTypeInfo) {
686 base = APValue::LValueBase::getTypeInfo(
687 TypeInfoLValue(typeInfo.getValue().getTypePtr()), type.getValue());
688 elemTy = base.getTypeInfoType();
689 } else if (isExpr) {
690 base = APValue::LValueBase(cast<Expr>(stmt.getValue()),
691 callIndex.getValue(), version.getValue());
692 elemTy = base.get<const Expr *>()->getType();
693 } else {
694 base = APValue::LValueBase(cast<ValueDecl>(decl.getValue()),
695 callIndex.getValue(), version.getValue());
696 elemTy = base.get<const ValueDecl *>()->getType();
697 }
698 }
699 CharUnits offset = CharUnits::fromQuantity(offsetQuantity);
700 APValue result;
701 result.MakeLValue();
702 if (!hasLValuePath) {
703 result.setLValue(base, offset, APValue::NoLValuePath{}, isNullPtr);
704 return result;
705 }
706 auto pathLength = lvaluePath->Path.size();
707 APValue::LValuePathEntry *path = result.setLValueUninit(
708 base, offset, pathLength, isLValueOnePastTheEnd, isNullPtr).data();
709 assert(lvaluePath->getType() == elemTy && "Unexpected type reference!");
710 llvm::copy(lvaluePath->Path, path);
711 return result;
712
713 }
714
715 case APValue::Float: {
716 llvm::APInt value = subR.find("value").readAPInt();
717 uint32_t semantics = subR.find("semantics").readUInt32();
718
719 const llvm::fltSemantics &floatSema = llvm::APFloatBase::EnumToSemantics(
720 static_cast<llvm::APFloatBase::Semantics>(semantics));
721 return APValue(llvm::APFloat(floatSema, value));
722
723 }
724
725 }
726 llvm_unreachable("bad APValue::ValueKind");
727 }

878 TemplateArgument readTemplateArgument() {

928 llvm::Optional<uint32_t> numExpansions = subR.find("numExpansions").template readOptional<uint32_t>();
929
930 auto numExpansionsUnsigned =

946

949 for (size_t i = 0, e = elements.size(); i != e; ++i)

964 TemplateName readTemplateName() {
965 auto &ctx = asImpl().getASTContext();
966 auto &&subR = asImpl().readObject();
967 TemplateName::NameKind kind = subR.find("kind").readTemplateNameKind();
968 switch (kind) {
969 case TemplateName::Template: {

971

973
974 }
975
976 case TemplateName::OverloadedTemplate: {
977 llvm::SmallVector<NamedDecl*, 8> overloads_buffer_0;
978 llvm::ArrayRef<NamedDecl*> overloads = subR.find("overloads").template readArray<NamedDecl*>(overloads_buffer_0);
979
980 // Copy into an UnresolvedSet to satisfy the interface.
981 UnresolvedSet<8> overloadSet;
982 for (auto overload : overloads) {
983 overloadSet.addDecl(overload);
984 }
985
986 return ctx.getOverloadedTemplateName(overloadSet.begin(),
987 overloadSet.end());
988

993

999 NestedNameSpecifier * qualifier = subR.find("qualifier").readNestedNameSpecifier();
1000 bool hasTemplateKeyword = subR.find("hasTemplateKeyword").readBool();
1001 TemplateDecl* declaration = subR.find("declaration").readTemplateDeclRef();
1002
1003 return ctx.getQualifiedTemplateName(qualifier, hasTemplateKeyword,

1011 llvm::Optional<OverloadedOperatorKind> operatorKind;
1012 if ( !identifier ) {
1013 operatorKind.emplace(subR.find("operatorKind").readOverloadedOperatorKind());
1014 }
1015
1016 if (identifier) {
1017 return ctx.getDependentTemplateName(qualifier, *identifier);
1018 } else {
1019 return ctx.getDependentTemplateName(qualifier, *operatorKind);
1020 }
1021
1022 }
