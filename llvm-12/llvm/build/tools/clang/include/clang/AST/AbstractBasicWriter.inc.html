733template <class Impl>
734class BasicWriterBase {
735 ASTContext &C;
736protected:
737 BasicWriterBase(ASTContext &ctx) : C(ctx) {}
738public:
739 ASTContext &getASTContext() { return C; }
740 Impl &asImpl() { return static_cast<Impl&>(*this); }
741 void writeAPValue(const APValue & node) {
742 auto &&subW = asImpl().writeObject();
743 APValue::ValueKind kind = (node.getKind());
744 subW.find("kind").writeAPValueKind(kind);
745 switch (kind) {
746 case APValue::None: {
747 return;
748 }
749
750 case APValue::FixedPoint: {
751 llvm::FixedPointSemantics semantics = ( node.getFixedPoint().getSemantics() );
752 subW.find("semantics").writeFixedPointSemantics(semantics);
753 llvm::APSInt value = ( node.getFixedPoint().getValue() );
754 subW.find("value").writeAPSInt(value);
755 return;
756 }
757
758 case APValue::ComplexInt: {
759 llvm::APSInt real = ( node.getComplexIntReal() );
760 subW.find("real").writeAPSInt(real);
761 llvm::APSInt imag = ( node.getComplexIntImag() );
762 subW.find("imag").writeAPSInt(imag);
763 return;
764 }
765
766 case APValue::ComplexFloat: {
767
768 auto sema = llvm::APFloatBase::SemanticsToEnum(
769 node.getComplexFloatReal().getSemantics());
770 assert(sema == llvm::APFloatBase::SemanticsToEnum(
771 node.getComplexFloatImag().getSemantics()));
772
773 uint32_t semantics = ( static_cast<uint32_t>(sema) );
774 subW.find("semantics").writeUInt32(semantics);
775 llvm::APInt real = ( node.getComplexFloatReal().bitcastToAPInt() );
776 subW.find("real").writeAPInt(real);
777 llvm::APInt imag = ( node.getComplexFloatImag().bitcastToAPInt() );
778 subW.find("imag").writeAPInt(imag);
779 return;
780 }

782 case APValue::Vector: {
783
784 SmallVector<APValue, 4> buffer;
785 unsigned len = node.getVectorLength();
786 for (unsigned i = 0; i < len; ++i)
787 buffer.push_back(node.getVectorElt(i));
788
789 llvm::ArrayRef<APValue> elements = ( buffer );
790 subW.find("elements").writeArray(elements);
791 return;
792 }
793
794 case APValue::Indeterminate: {
795 return;
796 }
797
798 case APValue::Array: {
799
800 SmallVector<APValue, 4> buffer{};
801 unsigned initLength = node.getArrayInitializedElts();
802 for (unsigned i = 0; i < initLength; ++i)
803 buffer.push_back(node.getArrayInitializedElt(i));
804 if (node.hasArrayFiller())
805 buffer.push_back(node.getArrayFiller());
806
807 uint32_t totalLength = ( node.getArraySize() );
808 subW.find("totalLength").writeUInt32(totalLength);
809 bool hasFiller = ( node.hasArrayFiller() );
810 subW.find("hasFiller").writeBool(hasFiller);
811 llvm::ArrayRef<APValue> elements = ( buffer );
812 subW.find("elements").writeArray(elements);
813 return;
814 }

816 case APValue::Struct: {
817
818 SmallVector<APValue, 4> structBases;
819 unsigned numBases = node.getStructNumBases();
820 for (unsigned i = 0; i < numBases; ++i)
821 structBases.push_back(node.getStructBase(i));
822 SmallVector<APValue, 4> structFields;
823 unsigned numFields = node.getStructNumFields();
824 for (unsigned i = 0; i < numFields; ++i)
825 structFields.push_back(node.getStructField(i));
826
827 llvm::ArrayRef<APValue> bases = ( structBases );
828 subW.find("bases").writeArray(bases);
829 llvm::ArrayRef<APValue> fields = ( structFields );
830 subW.find("fields").writeArray(fields);
831 return;
832 }
833
834 case APValue::Union: {
835 const Decl* fieldDecl = ( node.getUnionField() );
836 subW.find("fieldDecl").writeDeclRef(fieldDecl);
837 APValue value = ( node.getUnionValue() );
838 subW.find("value").writeAPValue(value);
839 return;
840 }
841
842 case APValue::AddrLabelDiff: {
843 const Stmt* lhs = ( const_cast<AddrLabelExpr *>(node.getAddrLabelDiffLHS()) );
844 subW.find("lhs").writeStmtRef(lhs);
845 const Stmt* rhs = ( const_cast<AddrLabelExpr *>(node.getAddrLabelDiffRHS()) );
846 subW.find("rhs").writeStmtRef(rhs);
847 return;
848 }
849
850 case APValue::Int: {
851 llvm::APSInt value = ( node.getInt() );
852 subW.find("value").writeAPSInt(value);
853 return;
854 }
855
856 case APValue::MemberPointer: {
857 bool isDerived = ( node.isMemberPointerToDerivedMember() );
858 subW.find("isDerived").writeBool(isDerived);
859 const ValueDecl* member = ( node.getMemberPointerDecl() );
860 subW.find("member").writeValueDeclRef(member);
861 llvm::ArrayRef<const CXXRecordDecl*> memberPath = ( node.getMemberPointerPath() );
862 subW.find("memberPath").writeArray(memberPath);
863 return;
864 }
865
866 case APValue::LValue: {
867
868 auto lvalueBase = node.getLValueBase();
869 const Expr *expr =
870 lvalueBase ? lvalueBase.dyn_cast<const Expr *>() : nullptr;
871 bool lvalueBaseIsExpr = (bool) expr;
872 bool lvalueBaseIsTypeInfo = lvalueBase.is<TypeInfoLValue>();
873 QualType elemTy;
874 if (lvalueBase) {
875 if (lvalueBaseIsTypeInfo) {
876 elemTy = lvalueBase.getTypeInfoType();
877 } else if (lvalueBaseIsExpr) {
878 elemTy = expr->getType();
879 } else {
880 elemTy = lvalueBase.get<const ValueDecl *>()->getType();
881 }
882 }
883
884 bool hasLValuePath = ( node.hasLValuePath() );
885 subW.find("hasLValuePath").writeBool(hasLValuePath);
886 bool isLValueOnePastTheEnd = ( node.isLValueOnePastTheEnd() );
887 subW.find("isLValueOnePastTheEnd").writeBool(isLValueOnePastTheEnd);
888 bool isExpr = ( lvalueBaseIsExpr );
889 subW.find("isExpr").writeBool(isExpr);
890 bool isTypeInfo = ( lvalueBaseIsTypeInfo );
891 subW.find("isTypeInfo").writeBool(isTypeInfo);
892 bool hasBase = ( static_cast<bool>(lvalueBase) );
893 subW.find("hasBase").writeBool(hasBase);
894 bool isNullPtr = ( node.isNullPointer() );
895 subW.find("isNullPtr").writeBool(isNullPtr);
896 if ( hasBase && isTypeInfo ) {
897 QualType typeInfo = (
898 QualType(node.getLValueBase().get<TypeInfoLValue>().getType(), 0)
899 );
900 subW.find("typeInfo").writeQualType(typeInfo);
901 }
902 if ( hasBase && isTypeInfo ) {
903 QualType type = ( node.getLValueBase().getTypeInfoType() );
904 subW.find("type").writeQualType(type);
905 }
906 if ( hasBase && !isTypeInfo ) {
907 uint32_t callIndex = ( node.getLValueBase().getCallIndex() );
908 subW.find("callIndex").writeUInt32(callIndex);
909 }
910 if ( hasBase && !isTypeInfo ) {
911 uint32_t version = ( node.getLValueBase().getVersion() );
912 subW.find("version").writeUInt32(version);
913 }
914 if ( hasBase && !isTypeInfo && isExpr ) {
915 const Stmt* stmt = ( const_cast<Expr *>(expr) );
916 subW.find("stmt").writeStmtRef(stmt);
917 }
918 if ( hasBase && !isTypeInfo && !isExpr ) {
919 const Decl* decl = ( lvalueBase.get<const ValueDecl *>() );
920 subW.find("decl").writeDeclRef(decl);
921 }
922 uint32_t offsetQuantity = ( node.getLValueOffset().getQuantity() );
923 subW.find("offsetQuantity").writeUInt32(offsetQuantity);
924 if ( hasLValuePath ) {
925 APValue::LValuePathSerializationHelper lvaluePath = (
926 APValue::LValuePathSerializationHelper(node.getLValuePath(), elemTy)
927 );
928 subW.find("lvaluePath").writeLValuePathSerializationHelper(lvaluePath);
929 }
930 return;
931 }
932
933 case APValue::Float: {
934 llvm::APInt value = ( node.getFloat().bitcastToAPInt() );
935 subW.find("value").writeAPInt(value);
936 uint32_t semantics = (
937 static_cast<uint32_t>(
938 llvm::APFloatBase::SemanticsToEnum(node.getFloat().getSemantics()))
939 );
940 subW.find("semantics").writeUInt32(semantics);
941 return;
942 }
943
944 }
945 llvm_unreachable("bad APValue::ValueKind");
