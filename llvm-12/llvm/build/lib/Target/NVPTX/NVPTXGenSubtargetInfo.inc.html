196// ParseSubtargetFeatures - Parses features string setting specified
197// subtarget options.
198void llvm::NVPTXSubtarget::ParseSubtargetFeatures(StringRef CPU, StringRef TuneCPU, StringRef FS) {
199 LLVM_DEBUG(dbgs() << "\nFeatures:" << FS);
200 LLVM_DEBUG(dbgs() << "\nCPU:" << CPU);
201 LLVM_DEBUG(dbgs() << "\nTuneCPU:" << TuneCPU << "\n\n");
202 InitMCProcessorInfo(CPU, TuneCPU, FS);
203 const FeatureBitset &Bits = getFeatureBits();
204 if (Bits[NVPTX::PTX32] && PTXVersion < 32) PTXVersion = 32;
205 if (Bits[NVPTX::PTX40] && PTXVersion < 40) PTXVersion = 40;
206 if (Bits[NVPTX::PTX41] && PTXVersion < 41) PTXVersion = 41;
207 if (Bits[NVPTX::PTX42] && PTXVersion < 42) PTXVersion = 42;
208 if (Bits[NVPTX::PTX43] && PTXVersion < 43) PTXVersion = 43;
209 if (Bits[NVPTX::PTX50] && PTXVersion < 50) PTXVersion = 50;
210 if (Bits[NVPTX::PTX60] && PTXVersion < 60) PTXVersion = 60;
211 if (Bits[NVPTX::PTX61] && PTXVersion < 61) PTXVersion = 61;
212 if (Bits[NVPTX::PTX63] && PTXVersion < 63) PTXVersion = 63;
213 if (Bits[NVPTX::PTX64] && PTXVersion < 64) PTXVersion = 64;
214 if (Bits[NVPTX::PTX65] && PTXVersion < 65) PTXVersion = 65;
215 if (Bits[NVPTX::PTX70] && PTXVersion < 70) PTXVersion = 70;
216 if (Bits[NVPTX::SM20] && SmVersion < 20) SmVersion = 20;
217 if (Bits[NVPTX::SM21] && SmVersion < 21) SmVersion = 21;
218 if (Bits[NVPTX::SM30] && SmVersion < 30) SmVersion = 30;
219 if (Bits[NVPTX::SM32] && SmVersion < 32) SmVersion = 32;
220 if (Bits[NVPTX::SM35] && SmVersion < 35) SmVersion = 35;
221 if (Bits[NVPTX::SM37] && SmVersion < 37) SmVersion = 37;
222 if (Bits[NVPTX::SM50] && SmVersion < 50) SmVersion = 50;
223 if (Bits[NVPTX::SM52] && SmVersion < 52) SmVersion = 52;
224 if (Bits[NVPTX::SM53] && SmVersion < 53) SmVersion = 53;
225 if (Bits[NVPTX::SM60] && SmVersion < 60) SmVersion = 60;
226 if (Bits[NVPTX::SM61] && SmVersion < 61) SmVersion = 61;
227 if (Bits[NVPTX::SM62] && SmVersion < 62) SmVersion = 62;
228 if (Bits[NVPTX::SM70] && SmVersion < 70) SmVersion = 70;
229 if (Bits[NVPTX::SM72] && SmVersion < 72) SmVersion = 72;
230 if (Bits[NVPTX::SM75] && SmVersion < 75) SmVersion = 75;
231 if (Bits[NVPTX::SM80] && SmVersion < 80) SmVersion = 80;
