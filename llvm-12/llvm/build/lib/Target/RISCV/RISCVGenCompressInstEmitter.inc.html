15 unsigned PredicateIndex) {
16 switch (PredicateIndex) {
17 default:

20 case 1: {
21 // uimm10_lsb00nonzero

25 return false;

33 if (MCOp.evaluateAsConstantImm(Imm))
34 return (Imm != 0) && isInt<6>(Imm);
35 return MCOp.isBareSymbolRef();
36
37 }
38 case 3: {
39 // simm6
40
41 int64_t Imm;
42 if (MCOp.evaluateAsConstantImm(Imm))
43 return isInt<6>(Imm);
44 return MCOp.isBareSymbolRef();
45
46 }
47 case 4: {
48 // simm10_lsb0000nonzero
49
50 int64_t Imm;
51 if (!MCOp.evaluateAsConstantImm(Imm))
52 return false;
53 return isShiftedInt<6, 4>(Imm) && (Imm != 0);
54
55 }
56 case 5: {
57 // simm9_lsb0
58
59 int64_t Imm;
60 if (MCOp.evaluateAsConstantImm(Imm))
61 return isShiftedInt<8, 1>(Imm);
62 return MCOp.isBareSymbolRef();
63
64
65 }
66 case 6: {
67 // uimm8_lsb000
68
69 int64_t Imm;
70 if (!MCOp.evaluateAsConstantImm(Imm))
71 return false;
72 return isShiftedUInt<5, 3>(Imm);
73
74 }
75 case 7: {
76 // uimm9_lsb000
77
78 int64_t Imm;
79 if (!MCOp.evaluateAsConstantImm(Imm))
80 return false;
81 return isShiftedUInt<6, 3>(Imm);
82
83 }
84 case 8: {
85 // uimm7_lsb00
86
87 int64_t Imm;
88 if (!MCOp.evaluateAsConstantImm(Imm))
89 return false;
90 return isShiftedUInt<5, 2>(Imm);
91
92 }
93 case 9: {
94 // uimm8_lsb00
95
96 int64_t Imm;
97 if (!MCOp.evaluateAsConstantImm(Imm))
98 return false;
99 return isShiftedUInt<6, 2>(Imm);
100
101 }
102 case 10: {
103 // simm12_lsb0
104
105 int64_t Imm;
106 if (MCOp.evaluateAsConstantImm(Imm))
107 return isShiftedInt<11, 1>(Imm);
108 return MCOp.isBareSymbolRef();
109
110 }
111 case 11: {
112 // c_lui_imm
113
114 int64_t Imm;
115 if (MCOp.evaluateAsConstantImm(Imm))
116 return (Imm != 0) && (isUInt<5>(Imm) ||
117 (Imm >= 0xfffe0 && Imm <= 0xfffff));
118 return MCOp.isBareSymbolRef();
119
120 }
121 case 12: {
122 // uimmlog2xlennonzero
123
124 int64_t Imm;
125 if (!MCOp.evaluateAsConstantImm(Imm))
126 return false;
127 if (STI.getTargetTriple().isArch64Bit())
128 return isUInt<6>(Imm) && (Imm != 0);
129 return isUInt<5>(Imm) && (Imm != 0);
130
131 }
132 }

1018static bool RISCVValidateMCOperand(const MCOperand &MCOp,
1019 const MCSubtargetInfo &STI,
1020 unsigned PredicateIndex) {
1021 switch (PredicateIndex) {
1022 default:
1023 llvm_unreachable("Unknown MCOperandPredicate kind");
1024 break;
1025 case 1: {
1026 // simm12
1027
1028 int64_t Imm;
1029 if (MCOp.evaluateAsConstantImm(Imm))
1030 return isInt<12>(Imm);
1031 return MCOp.isBareSymbolRef();
1032
1033 }
1034 case 2: {
1035 // simm13_lsb0
1036
1037 int64_t Imm;
1038 if (MCOp.evaluateAsConstantImm(Imm))
1039 return isShiftedInt<12, 1>(Imm);
1040 return MCOp.isBareSymbolRef();
1041
1042 }
1043 case 3: {
1044 // simm21_lsb0_jal
1045
1046 int64_t Imm;
1047 if (MCOp.evaluateAsConstantImm(Imm))
1048 return isShiftedInt<20, 1>(Imm);
1049 return MCOp.isBareSymbolRef();
1050
1051 }
1052 case 4: {
1053 // uimm20_lui
1054
1055 int64_t Imm;
1056 if (MCOp.evaluateAsConstantImm(Imm))
1057 return isUInt<20>(Imm);
1058 return MCOp.isBareSymbolRef();
1059
1060 }
1061 case 5: {
1062 // uimmlog2xlen
1063
1064 int64_t Imm;
1065 if (!MCOp.evaluateAsConstantImm(Imm))
1066 return false;
1067 if (STI.getTargetTriple().isArch64Bit())
1068 return isUInt<6>(Imm);
1069 return isUInt<5>(Imm);
1070
1071 }
1072 }

1906static bool RISCVValidateMachineOperand(const MachineOperand &MO,
1907 const RISCVSubtarget *Subtarget,
1908 unsigned PredicateIndex) {
1909 int64_t Imm = MO.getImm();
1910 switch (PredicateIndex) {
1911 default:
1912 llvm_unreachable("Unknown ImmLeaf Predicate kind");
1913 break;
1914 case 1: {
1915 // simm6nonzero
1916 return (Imm != 0) && isInt<6>(Imm);
1917 }
1918 case 2: {
1919 // simm10_lsb0000nonzero
1920 return (Imm != 0) && isShiftedInt<6, 4>(Imm);
1921 }
1922 case 3: {
1923 // uimm10_lsb00nonzero
1924 return isShiftedUInt<8, 2>(Imm) && (Imm != 0);
1925 }
1926 case 4: {
1927 // simm6
1928 return isInt<6>(Imm);
1929 }
1930 case 5: {
1931 // simm9_lsb0
1932 return isShiftedInt<8, 1>(Imm);
1933 }
1934 case 6: {
1935 // uimm8_lsb000
1936 return isShiftedUInt<5, 3>(Imm);
1937 }
1938 case 7: {
1939 // uimm9_lsb000
1940 return isShiftedUInt<6, 3>(Imm);
1941 }
1942 case 8: {
1943 // uimm7_lsb00
1944 return isShiftedUInt<5, 2>(Imm);
1945 }
1946 case 9: {
1947 // uimm8_lsb00
1948 return isShiftedUInt<6, 2>(Imm);
1949 }
1950 case 10: {
1951 // simm12_lsb0
1952 return isShiftedInt<11, 1>(Imm);
1953 }
1954 case 11: {
1955 // c_lui_imm
1956 return (Imm != 0) &&
1957 (isUInt<5>(Imm) ||
1958 (Imm >= 0xfffe0 && Imm <= 0xfffff));
1959 }
1960 case 12: {
1961 // uimmlog2xlennonzero
1962
1963 if (Subtarget->is64Bit())
1964 return isUInt<6>(Imm) && (Imm != 0);
1965 return isUInt<5>(Imm) && (Imm != 0);
1966
1967 }
1968 }
