18// Helper functions for extracting fields from encoded instructions.
19// InsnType must either be integral or an APInt-like object that must:
20// * Have a static const max_size_in_bits equal to the number of bits in the
21// encoding.
22// * be default-constructible and copy-constructible
23// * be constructible from a uint64_t
24// * be constructible from an APInt (this can be private)
25// * Support getBitsSet(loBit, hiBit)
26// * be convertible to uint64_t
27// * Support the ~, &, ==, !=, and |= operators with other objects of the same type
28// * Support shift (<<, >>) with signed and unsigned integers on the RHS
29// * Support put (<<) to raw_ostream&
30template <typename InsnType>
31#if defined(_MSC_VER) && !defined(__clang__)
32__declspec(noinline)
33#endif
34static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
35 unsigned numBits, std::true_type) {
36 assert(startBit + numBits <= 64 && "Cannot support >64-bit extractions!");
37 assert(startBit + numBits <= (sizeof(InsnType) * 8) &&
38 "Instruction field out of bounds!");
39 InsnType fieldMask;
40 if (numBits == sizeof(InsnType) * 8)
41 fieldMask = (InsnType)(-1LL);
42 else
43 fieldMask = (((InsnType)1 << numBits) - 1) << startBit;
44 return (insn & fieldMask) >> startBit;
45}

47template <typename InsnType>
48static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
49 unsigned numBits, std::false_type) {
50 assert(startBit + numBits <= InsnType::max_size_in_bits && "Instruction field out of bounds!");
51 InsnType fieldMask = InsnType::getBitsSet(0, numBits);
52 return (insn >> startBit) & fieldMask;
53}

801template <typename InsnType>
802static DecodeStatus decodeInstruction(const uint8_t DecodeTable[], MCInst &MI,
803 InsnType insn, uint64_t Address,
804 const void *DisAsm,
805 const MCSubtargetInfo &STI) {
806 const FeatureBitset &Bits = STI.getFeatureBits();
807
808 const uint8_t *Ptr = DecodeTable;
809 InsnType CurFieldValue = 0;
810 DecodeStatus S = MCDisassembler::Success;
811 while (true) {
812 ptrdiff_t Loc = Ptr - DecodeTable;
813 switch (*Ptr) {
814 default:
815 errs() << Loc << ": Unexpected decode table opcode!\n";
816 return MCDisassembler::Fail;
817 case MCD::OPC_ExtractField: {
818 unsigned Start = *++Ptr;
819 unsigned Len = *++Ptr;
820 ++Ptr;
821 CurFieldValue = fieldFromInstruction(insn, Start, Len);
822 LLVM_DEBUG(dbgs() << Loc << ": OPC_ExtractField(" << Start << ", "
823 << Len << "): " << CurFieldValue << "\n");
824 break;
825 }
826 case MCD::OPC_FilterValue: {
827 // Decode the field value.
828 unsigned Len;
829 InsnType Val = decodeULEB128(++Ptr, &Len);
830 Ptr += Len;
831 // NumToSkip is a plain 24-bit integer.
832 unsigned NumToSkip = *Ptr++;
833 NumToSkip |= (*Ptr++) << 8;
834 NumToSkip |= (*Ptr++) << 16;
835
836 // Perform the filter operation.
837 if (Val != CurFieldValue)
838 Ptr += NumToSkip;
839 LLVM_DEBUG(dbgs() << Loc << ": OPC_FilterValue(" << Val << ", " << NumToSkip
840 << "): " << ((Val != CurFieldValue) ? "FAIL:" : "PASS:")
841 << " continuing at " << (Ptr - DecodeTable) << "\n");
842
843 break;
844 }
845 case MCD::OPC_CheckField: {
846 unsigned Start = *++Ptr;
847 unsigned Len = *++Ptr;
848 InsnType FieldValue = fieldFromInstruction(insn, Start, Len);
849 // Decode the field value.
850 InsnType ExpectedValue = decodeULEB128(++Ptr, &Len);
851 Ptr += Len;
852 // NumToSkip is a plain 24-bit integer.
853 unsigned NumToSkip = *Ptr++;
854 NumToSkip |= (*Ptr++) << 8;
855 NumToSkip |= (*Ptr++) << 16;
856
857 // If the actual and expected values don't match, skip.
858 if (ExpectedValue != FieldValue)
859 Ptr += NumToSkip;
860 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckField(" << Start << ", "
861 << Len << ", " << ExpectedValue << ", " << NumToSkip
862 << "): FieldValue = " << FieldValue << ", ExpectedValue = "
863 << ExpectedValue << ": "
864 << ((ExpectedValue == FieldValue) ? "PASS\n" : "FAIL\n"));
865 break;
866 }
867 case MCD::OPC_CheckPredicate: {
868 unsigned Len;
869 // Decode the Predicate Index value.
870 unsigned PIdx = decodeULEB128(++Ptr, &Len);
871 Ptr += Len;
872 // NumToSkip is a plain 24-bit integer.
873 unsigned NumToSkip = *Ptr++;
874 NumToSkip |= (*Ptr++) << 8;
875 NumToSkip |= (*Ptr++) << 16;
876 // Check the predicate.
877 bool Pred;
878 if (!(Pred = checkDecoderPredicate(PIdx, Bits)))
879 Ptr += NumToSkip;
880 (void)Pred;
881 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckPredicate(" << PIdx << "): "
882 << (Pred ? "PASS\n" : "FAIL\n"));
883
884 break;
885 }
886 case MCD::OPC_Decode: {
887 unsigned Len;
888 // Decode the Opcode value.
889 unsigned Opc = decodeULEB128(++Ptr, &Len);
890 Ptr += Len;
891 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
892 Ptr += Len;
893
894 MI.clear();
895 MI.setOpcode(Opc);
896 bool DecodeComplete;
897 S = decodeToMCInst(S, DecodeIdx, insn, MI, Address, DisAsm, DecodeComplete);
898 assert(DecodeComplete);
899
900 LLVM_DEBUG(dbgs() << Loc << ": OPC_Decode: opcode " << Opc
901 << ", using decoder " << DecodeIdx << ": "
902 << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
903 return S;
904 }
905 case MCD::OPC_TryDecode: {
906 unsigned Len;
907 // Decode the Opcode value.
908 unsigned Opc = decodeULEB128(++Ptr, &Len);
909 Ptr += Len;
910 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
911 Ptr += Len;
912 // NumToSkip is a plain 24-bit integer.
913 unsigned NumToSkip = *Ptr++;
914 NumToSkip |= (*Ptr++) << 8;
915 NumToSkip |= (*Ptr++) << 16;
916
917 // Perform the decode operation.
918 MCInst TmpMI;
919 TmpMI.setOpcode(Opc);
920 bool DecodeComplete;
921 S = decodeToMCInst(S, DecodeIdx, insn, TmpMI, Address, DisAsm, DecodeComplete);
922 LLVM_DEBUG(dbgs() << Loc << ": OPC_TryDecode: opcode " << Opc
923 << ", using decoder " << DecodeIdx << ": ");
924
925 if (DecodeComplete) {
926 // Decoding complete.
927 LLVM_DEBUG(dbgs() << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
928 MI = TmpMI;
929 return S;
930 } else {
931 assert(S == MCDisassembler::Fail);
932 // If the decoding was incomplete, skip.
933 Ptr += NumToSkip;
934 LLVM_DEBUG(dbgs() << "FAIL: continuing at " << (Ptr - DecodeTable) << "\n");
935 // Reset decode status. This also drops a SoftFail status that could be
936 // set before the decode attempt.
937 S = MCDisassembler::Success;
938 }
939 break;
940 }
941 case MCD::OPC_SoftFail: {
942 // Decode the mask values.
943 unsigned Len;
944 InsnType PositiveMask = decodeULEB128(++Ptr, &Len);
945 Ptr += Len;
946 InsnType NegativeMask = decodeULEB128(Ptr, &Len);
947 Ptr += Len;
948 bool Fail = (insn & PositiveMask) || (~insn & NegativeMask);
949 if (Fail)
950 S = MCDisassembler::SoftFail;
951 LLVM_DEBUG(dbgs() << Loc << ": OPC_SoftFail: " << (Fail ? "FAIL\n" : "PASS\n"));
952 break;
953 }
954 case MCD::OPC_Fail: {
955 LLVM_DEBUG(dbgs() << Loc << ": OPC_Fail\n");
956 return MCDisassembler::Fail;
957 }
958 }
959 }
960 llvm_unreachable("bogosity detected in disassembler state machine!");
961}
