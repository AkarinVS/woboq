161void WebAssemblyAsmParser::
162convertToMCInst(unsigned Kind, MCInst &Inst, unsigned Opcode,
163 const OperandVector &Operands) {
164 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
165 const uint8_t *Converter = ConversionTable[Kind];
166 unsigned OpIdx;
167 Inst.setOpcode(Opcode);
168 for (const uint8_t *p = Converter; *p; p += 2) {
169 OpIdx = *(p + 1);
170 switch (*p) {
171 default: llvm_unreachable("invalid conversion entry!");
172 case CVT_Reg:
173 static_cast<WebAssemblyOperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
174 break;
175 case CVT_Tied: {
176 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
177 std::begin(TiedAsmOperandTable)) &&
178 "Tied operand not found");
179 unsigned TiedResOpnd = TiedAsmOperandTable[OpIdx][0];
180 if (TiedResOpnd != (uint8_t)-1)
181 Inst.addOperand(Inst.getOperand(TiedResOpnd));
182 break;
183 }
184 case CVT_imm_95_0:
185 Inst.addOperand(MCOperand::createImm(0));
186 break;
187 case CVT_95_addImmOperands:
188 static_cast<WebAssemblyOperand &>(*Operands[OpIdx]).addImmOperands(Inst, 1);
189 break;
190 case CVT_95_addBrListOperands:
191 static_cast<WebAssemblyOperand &>(*Operands[OpIdx]).addBrListOperands(Inst, 1);
192 break;
193 case CVT_95_addFPImmOperands:
194 static_cast<WebAssemblyOperand &>(*Operands[OpIdx]).addFPImmOperands(Inst, 1);
195 break;
196 }
197 }

281static unsigned validateOperandClass(MCParsedAsmOperand &GOp, MatchClassKind Kind) {
282 WebAssemblyOperand &Operand = (WebAssemblyOperand &)GOp;
283 if (Kind == InvalidMatchClass)
284 return MCTargetAsmParser::Match_InvalidOperand;
285
286 if (Operand.isToken() && Kind <= MCK_LAST_TOKEN)
287 return isSubclass(matchTokenString(Operand.getToken()), Kind) ?
288 MCTargetAsmParser::Match_Success :
289 MCTargetAsmParser::Match_InvalidOperand;
290
291 switch (Kind) {
292 default: break;
293 // 'BrList' class
294 case MCK_BrList: {
295 DiagnosticPredicate DP(Operand.isBrList());
296 if (DP.isMatch())
297 return MCTargetAsmParser::Match_Success;
298 break;
299 }
300 // 'FPImm' class
301 case MCK_FPImm: {
302 DiagnosticPredicate DP(Operand.isFPImm());
303 if (DP.isMatch())
304 return MCTargetAsmParser::Match_Success;
305 break;
306 }
307 // 'Imm' class
308 case MCK_Imm: {
309 DiagnosticPredicate DP(Operand.isImm());
310 if (DP.isMatch())
311 return MCTargetAsmParser::Match_Success;
312 break;
313 }
314 } // end switch (Kind)
315
316 if (Operand.isReg()) {
317 MatchClassKind OpKind;
318 switch (Operand.getReg()) {
319 default: OpKind = InvalidMatchClass; break;
320 case WebAssembly::FP32: OpKind = MCK_I32; break;
321 case WebAssembly::FP64: OpKind = MCK_I64; break;
322 case WebAssembly::SP32: OpKind = MCK_I32; break;
323 case WebAssembly::SP64: OpKind = MCK_I64; break;
324 case WebAssembly::I32_0: OpKind = MCK_I32; break;
325 case WebAssembly::I64_0: OpKind = MCK_I64; break;
326 case WebAssembly::F32_0: OpKind = MCK_F32; break;
327 case WebAssembly::F64_0: OpKind = MCK_F64; break;
328 case WebAssembly::V128_0: OpKind = MCK_V128; break;
329 case WebAssembly::FUNCREF_0: OpKind = MCK_FUNCREF; break;
330 case WebAssembly::EXTERNREF_0: OpKind = MCK_EXTERNREF; break;
331 }
332 return isSubclass(OpKind, Kind) ? (unsigned)MCTargetAsmParser::Match_Success :
333 getDiagKindFromRegisterClass(Kind);
334 }
335
336 if (Kind > MCK_LAST_TOKEN && Kind <= MCK_LAST_REGISTER)
337 return getDiagKindFromRegisterClass(Kind);
338
339 return MCTargetAsmParser::Match_InvalidOperand;
340}

391static bool checkAsmTiedOperandConstraints(const WebAssemblyAsmParser&AsmParser,
392 unsigned Kind,
393 const OperandVector &Operands,
394 uint64_t &ErrorInfo) {
395 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
396 const uint8_t *Converter = ConversionTable[Kind];
397 for (const uint8_t *p = Converter; *p; p += 2) {
398 switch (*p) {
399 case CVT_Tied: {
400 unsigned OpIdx = *(p + 1);
401 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
402 std::begin(TiedAsmOperandTable)) &&
403 "Tied operand not found");
404 unsigned OpndNum1 = TiedAsmOperandTable[OpIdx][1];
405 unsigned OpndNum2 = TiedAsmOperandTable[OpIdx][2];
406 if (OpndNum1 != OpndNum2) {
407 auto &SrcOp1 = Operands[OpndNum1];
408 auto &SrcOp2 = Operands[OpndNum2];
409 if (SrcOp1->isReg() && SrcOp2->isReg()) {
410 if (!AsmParser.regsEqual(*SrcOp1, *SrcOp2)) {
411 ErrorInfo = OpndNum2;
412 return false;
413 }
414 }
415 }
416 break;
417 }
418 default:
419 break;
420 }
421 }
422 return true;
423}

1309unsigned WebAssemblyAsmParser::
1310MatchInstructionImpl(const OperandVector &Operands,
1311 MCInst &Inst,
1312 uint64_t &ErrorInfo,
1313 FeatureBitset &MissingFeatures,
1314 bool matchingInlineAsm, unsigned VariantID) {
1315 // Eliminate obvious mismatches.
1316 if (Operands.size() > 17) {
1317 ErrorInfo = 17;
1318 return Match_InvalidOperand;
1319 }
1320
1321 // Get the current feature set.
1322 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1323
1324 // Get the instruction mnemonic, which is the first token.
1325 StringRef Mnemonic = ((WebAssemblyOperand &)*Operands[0]).getToken();
1326
1327 // Some state to try to produce better error messages.
1328 bool HadMatchOtherThanFeatures = false;
1329 bool HadMatchOtherThanPredicate = false;
1330 unsigned RetCode = Match_InvalidOperand;
1331 MissingFeatures.set();
1332 // Set ErrorInfo to the operand that mismatches if it is
1333 // wrong for all instances of the instruction.
1334 ErrorInfo = ~0ULL;
1335 // Find the appropriate table for this asm variant.
1336 const MatchEntry *Start, *End;
1337 switch (VariantID) {
1338 default: llvm_unreachable("invalid variant!");
1339 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1340 }
1341 // Search the table.
1342 auto MnemonicRange = std::equal_range(Start, End, Mnemonic, LessOpcode());
1343
1344 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "AsmMatcher: found " <<
1345 std::distance(MnemonicRange.first, MnemonicRange.second) <<
1346 " encodings with mnemonic '" << Mnemonic << "'\n");
1347
1348 // Return a more specific error code if no mnemonics match.
1349 if (MnemonicRange.first == MnemonicRange.second)
1350 return Match_MnemonicFail;
1351
1352 for (const MatchEntry *it = MnemonicRange.first, *ie = MnemonicRange.second;
1353 it != ie; ++it) {
1354 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1355 bool HasRequiredFeatures =
1356 (AvailableFeatures & RequiredFeatures) == RequiredFeatures;
1357 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Trying to match opcode "
1358 << MII.getName(it->Opcode) << "\n");
1359 // equal_range guarantees that instruction mnemonic matches.
1360 assert(Mnemonic == it->getMnemonic());
1361 bool OperandsValid = true;
1362 for (unsigned FormalIdx = 0, ActualIdx = 1; FormalIdx != 16; ++FormalIdx) {
1363 auto Formal = static_cast<MatchClassKind>(it->Classes[FormalIdx]);
1364 DEBUG_WITH_TYPE("asm-matcher",
1365 dbgs() << " Matching formal operand class " << getMatchClassName(Formal)
1366 << " against actual operand at index " << ActualIdx);
1367 if (ActualIdx < Operands.size())
1368 DEBUG_WITH_TYPE("asm-matcher", dbgs() << " (";
1369 Operands[ActualIdx]->print(dbgs()); dbgs() << "): ");
1370 else
1371 DEBUG_WITH_TYPE("asm-matcher", dbgs() << ": ");
1372 if (ActualIdx >= Operands.size()) {
1373 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "actual operand index out of range ");
1374 OperandsValid = (Formal == InvalidMatchClass) || isSubclass(Formal, OptionalMatchClass);
1375 if (!OperandsValid) ErrorInfo = ActualIdx;
1376 break;
1377 }
1378 MCParsedAsmOperand &Actual = *Operands[ActualIdx];
1379 unsigned Diag = validateOperandClass(Actual, Formal);
1380 if (Diag == Match_Success) {
1381 DEBUG_WITH_TYPE("asm-matcher",
1382 dbgs() << "match success using generic matcher\n");
1383 ++ActualIdx;
1384 continue;
1385 }
1386 // If the generic handler indicates an invalid operand
1387 // failure, check for a special case.
1388 if (Diag != Match_Success) {
1389 unsigned TargetDiag = validateTargetOperandClass(Actual, Formal);
1390 if (TargetDiag == Match_Success) {
1391 DEBUG_WITH_TYPE("asm-matcher",
1392 dbgs() << "match success using target matcher\n");
1393 ++ActualIdx;
1394 continue;
1395 }
1396 // If the target matcher returned a specific error code use
1397 // that, else use the one from the generic matcher.
1398 if (TargetDiag != Match_InvalidOperand && HasRequiredFeatures)
1399 Diag = TargetDiag;
1400 }
1401 // If current formal operand wasn't matched and it is optional
1402 // then try to match next formal operand
1403 if (Diag == Match_InvalidOperand && isSubclass(Formal, OptionalMatchClass)) {
1404 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "ignoring optional operand\n");
1405 continue;
1406 }
1407 // If this operand is broken for all of the instances of this
1408 // mnemonic, keep track of it so we can report loc info.
1409 // If we already had a match that only failed due to a
1410 // target predicate, that diagnostic is preferred.
1411 if (!HadMatchOtherThanPredicate &&
1412 (it == MnemonicRange.first || ErrorInfo <= ActualIdx)) {
1413 if (HasRequiredFeatures && (ErrorInfo != ActualIdx || Diag != Match_InvalidOperand))
1414 RetCode = Diag;
1415 ErrorInfo = ActualIdx;
1416 }
1417 // Otherwise, just reject this instance of the mnemonic.
1418 OperandsValid = false;
1419 break;
1420 }
1421
1422 if (!OperandsValid) {
1423 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Opcode result: multiple "
1424 "operand mismatches, ignoring "
1425 "this opcode\n");
1426 continue;
1427 }
1428 if (!HasRequiredFeatures) {
1429 HadMatchOtherThanFeatures = true;
1430 FeatureBitset NewMissingFeatures = RequiredFeatures & ~AvailableFeatures;
1431 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Missing target features:";
1432 for (unsigned I = 0, E = NewMissingFeatures.size(); I != E; ++I)
1433 if (NewMissingFeatures[I])
1434 dbgs() << ' ' << I;
1435 dbgs() << "\n");
1436 if (NewMissingFeatures.count() <=
1437 MissingFeatures.count())
1438 MissingFeatures = NewMissingFeatures;
1439 continue;
1440 }
1441
1442 Inst.clear();
1443
1444 Inst.setOpcode(it->Opcode);
1445 // We have a potential match but have not rendered the operands.
1446 // Check the target predicate to handle any context sensitive
1447 // constraints.
1448 // For example, Ties that are referenced multiple times must be
1449 // checked here to ensure the input is the same for each match
1450 // constraints. If we leave it any later the ties will have been
1451 // canonicalized
1452 unsigned MatchResult;
1453 if ((MatchResult = checkEarlyTargetMatchPredicate(Inst, Operands)) != Match_Success) {
1454 Inst.clear();
1455 DEBUG_WITH_TYPE(
1456 "asm-matcher",
1457 dbgs() << "Early target match predicate failed with diag code "
1458 << MatchResult << "\n");
1459 RetCode = MatchResult;
1460 HadMatchOtherThanPredicate = true;
1461 continue;
1462 }
1463
1464 if (matchingInlineAsm) {
1465 convertToMapAndConstraints(it->ConvertFn, Operands);
1466 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1467 return Match_InvalidTiedOperand;
1468
1469 return Match_Success;
1470 }
1471
1472 // We have selected a definite instruction, convert the parsed
1473 // operands into the appropriate MCInst.
1474 convertToMCInst(it->ConvertFn, Inst, it->Opcode, Operands);
1475
1476 // We have a potential match. Check the target predicate to
1477 // handle any context sensitive constraints.
1478 if ((MatchResult = checkTargetMatchPredicate(Inst)) != Match_Success) {
1479 DEBUG_WITH_TYPE("asm-matcher",
1480 dbgs() << "Target match predicate failed with diag code "
1481 << MatchResult << "\n");
1482 Inst.clear();
1483 RetCode = MatchResult;
1484 HadMatchOtherThanPredicate = true;
1485 continue;
1486 }
1487
1488 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1489 return Match_InvalidTiedOperand;
1490
1491 DEBUG_WITH_TYPE(
1492 "asm-matcher",
1493 dbgs() << "Opcode result: complete match, selecting this opcode\n");
1494 return Match_Success;
1495 }
1496
1497 // Okay, we had no match. Try to return a useful error code.
1498 if (HadMatchOtherThanPredicate || !HadMatchOtherThanFeatures)
1499 return RetCode;
1500
1501 ErrorInfo = 0;
1502 return Match_MissingFeature;
1503}

1511static std::string WebAssemblyMnemonicSpellCheck(StringRef S, const FeatureBitset &FBS, unsigned VariantID) {
1512 const unsigned MaxEditDist = 2;
1513 std::vector<StringRef> Candidates;
1514 StringRef Prev = "";
1515
1516 // Find the appropriate table for this asm variant.
1517 const MatchEntry *Start, *End;
1518 switch (VariantID) {
1519 default: llvm_unreachable("invalid variant!");
1520 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1521 }
1522
1523 for (auto I = Start; I < End; I++) {
1524 // Ignore unsupported instructions.
1525 const FeatureBitset &RequiredFeatures = FeatureBitsets[I->RequiredFeaturesIdx];
1526 if ((FBS & RequiredFeatures) != RequiredFeatures)
1527 continue;
1528
1529 StringRef T = I->getMnemonic();
1530 // Avoid recomputing the edit distance for the same string.
1531 if (T.equals(Prev))
1532 continue;
1533
1534 Prev = T;
1535 unsigned Dist = S.edit_distance(T, false, MaxEditDist);
1536 if (Dist <= MaxEditDist)
1537 Candidates.push_back(T);
1538 }
1539
1540 if (Candidates.empty())
1541 return "";
1542
1543 std::string Res = ", did you mean: ";
1544 unsigned i = 0;
1545 for (; i < Candidates.size() - 1; i++)
1546 Res += Candidates[i].str() + ", ";
1547 return Res + Candidates[i].str() + "?";
1548}
