149// ParseSubtargetFeatures - Parses features string setting specified
150// subtarget options.
151void llvm::WebAssemblySubtarget::ParseSubtargetFeatures(StringRef CPU, StringRef TuneCPU, StringRef FS) {
152 LLVM_DEBUG(dbgs() << "\nFeatures:" << FS);
153 LLVM_DEBUG(dbgs() << "\nCPU:" << CPU);
154 LLVM_DEBUG(dbgs() << "\nTuneCPU:" << TuneCPU << "\n\n");
155 InitMCProcessorInfo(CPU, TuneCPU, FS);
156 const FeatureBitset &Bits = getFeatureBits();
157 if (Bits[WebAssembly::FeatureAtomics]) HasAtomics = true;
158 if (Bits[WebAssembly::FeatureBulkMemory]) HasBulkMemory = true;
159 if (Bits[WebAssembly::FeatureExceptionHandling]) HasExceptionHandling = true;
160 if (Bits[WebAssembly::FeatureMultivalue]) HasMultivalue = true;
161 if (Bits[WebAssembly::FeatureMutableGlobals]) HasMutableGlobals = true;
162 if (Bits[WebAssembly::FeatureNontrappingFPToInt]) HasNontrappingFPToInt = true;
163 if (Bits[WebAssembly::FeatureReferenceTypes]) HasReferenceTypes = true;
164 if (Bits[WebAssembly::FeatureSIMD128] && SIMDLevel < SIMD128) SIMDLevel = SIMD128;
165 if (Bits[WebAssembly::FeatureSignExt]) HasSignExt = true;
166 if (Bits[WebAssembly::FeatureTailCall]) HasTailCall = true;
167 if (Bits[WebAssembly::FeatureUnimplementedSIMD128] && SIMDLevel < UnimplementedSIMD128) SIMDLevel = UnimplementedSIMD128;
168}
