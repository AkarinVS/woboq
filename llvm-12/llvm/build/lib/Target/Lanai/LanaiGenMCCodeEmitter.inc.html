661 op &= UINT64_C(31);
662 op <<= 23;
663 Value |= op;
664 // op: src
665 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
666 Value |= (op & UINT64_C(2031616)) << 2;
667 Value |= (op & UINT64_C(65535));
668 break;
669 }
670 case Lanai::BRIND_CC: {
671 // op: Rs1
672 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
673 op &= UINT64_C(31);
674 op <<= 18;
675 Value |= op;
676 // op: DDDI
677 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
678 Value |= (op & UINT64_C(1)) << 16;
679 Value |= (op & UINT64_C(14)) >> 1;
680 break;
681 }
682 case Lanai::SCC: {
683 // op: Rs1
684 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
685 op &= UINT64_C(31);
686 op <<= 18;
687 Value |= op;
688 // op: DDDI
689 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
690 Value |= (op & UINT64_C(14)) << 24;
691 Value |= (op & UINT64_C(1));
692 break;
693 }
694 case Lanai::SFSUB_F_RR: {
695 // op: Rs1
696 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
697 op &= UINT64_C(31);
698 op <<= 18;
699 Value |= op;
700 // op: Rs2
701 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
702 op &= UINT64_C(31);
703 op <<= 11;
704 Value |= op;
705 break;
706 }
707 case Lanai::BRIND_CCA: {
708 // op: Rs1
709 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
710 op &= UINT64_C(31);
711 op <<= 18;
712 Value |= op;
713 // op: Rs2
714 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
715 op &= UINT64_C(31);
716 op <<= 11;
717 Value |= op;
718 // op: DDDI
719 op = getMachineOpValue(MI, MI.getOperand(2), Fixups, STI);
720 Value |= (op & UINT64_C(1)) << 16;
721 Value |= (op & UINT64_C(14)) >> 1;
722 break;
723 }
724 case Lanai::SFSUB_F_RI_HI:
725 case Lanai::SFSUB_F_RI_LO: {
726 // op: Rs1
727 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
728 op &= UINT64_C(31);
729 op <<= 18;
730 Value |= op;
731 // op: imm16
732 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
733 op &= UINT64_C(65535);
734 Value |= op;
735 break;
736 }
737 case Lanai::JR: {
738 // op: Rs2
739 op = getMachineOpValue(MI, MI.getOperand(0), Fixups, STI);
740 op &= UINT64_C(31);
741 op <<= 11;
742 Value |= op;
743 break;
744 }
745 case Lanai::BT: {
746 // op: addr
747 op = getBranchTargetOpValue(MI, 0, Fixups, STI);
748 op &= UINT64_C(33554428);
749 Value |= op;
750 break;
751 }
752 case Lanai::BRCC: {
753 // op: addr
754 op = getBranchTargetOpValue(MI, 0, Fixups, STI);
755 op &= UINT64_C(33554428);
756 Value |= op;
757 // op: DDDI
758 op = getMachineOpValue(MI, MI.getOperand(1), Fixups, STI);
759 Value |= (op & UINT64_C(14)) << 24;
760 Value |= (op & UINT64_C(1));

762 }

765 raw_string_ostream Msg(msg);
766 Msg << "Not supported instr: " << MI;
767 report_fatal_error(Msg.str());
768 }
