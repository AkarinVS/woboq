757/// printInstruction - This method is automatically generated by tablegen
758/// from the instruction set description.
759void LanaiInstPrinter::printInstruction(const MCInst *MI, uint64_t Address, raw_ostream &O) {
760 O << "\t";
761
762 auto MnemonicInfo = getMnemonic(MI);
763
764 O << MnemonicInfo.first;
765
766 uint32_t Bits = MnemonicInfo.second;
767 assert(Bits != 0 && "Cannot print this instruction.");
768
769 // Fragment 0 encoded into 4 bits for 12 unique commands.
770 switch ((Bits >> 10) & 15) {
771 default: llvm_unreachable("Invalid command number.");
772 case 0:
773 // DBG_VALUE, DBG_INSTR_REF, DBG_LABEL, BUNDLE, LIFETIME_START, LIFETIME_...
774 return;
775 break;
776 case 1:
777 // ADJCALLSTACKDOWN, ADJCALLSTACKUP, ADJDYNALLOC, BT, JR, SFSUB_F_RI_HI, ...
778 printOperand(MI, 0, O);
779 break;
780 case 2:
781 // ADDC_F_I_HI, ADDC_F_I_LO, ADDC_I_HI, ADDC_I_LO, ADD_F_I_HI, ADD_F_I_LO...
782 printOperand(MI, 1, O);
783 O << ", ";
784 break;
785 case 3:
786 // ADDC_F_R, ADDC_R, ADD_F_R, ADD_R, AND_F_R, AND_R, OR_F_R, OR_R, SHL_F_...
787 printPredicateOperand(MI, 3, O);
788 O << "\t";
789 printOperand(MI, 1, O);
790 O << ", ";
791 printOperand(MI, 2, O);
792 O << ", ";
793 printOperand(MI, 0, O);
794 return;
795 break;
796 case 4:
797 // BRCC, BRIND_CC, BRR, SCC
798 printCCOperand(MI, 1, O);
799 break;
800 case 5:
801 // BRIND_CCA
802 printCCOperand(MI, 2, O);
803 O << "\t";
804 printOperand(MI, 0, O);
805 O << " add ";
806 printOperand(MI, 1, O);
807 return;
808 break;
809 case 6:
810 // LDADDR
811 printMemImmOperand(MI, 1, O);
812 O << ", ";
813 printOperand(MI, 0, O);
814 return;
815 break;
816 case 7:
817 // LDBs_RI, LDBz_RI, LDHs_RI, LDHz_RI
818 printMemSplsOperand(MI, 1, O);
819 O << ", ";
820 printOperand(MI, 0, O);
821 return;
822 break;
823 case 8:
824 // LDBs_RR, LDBz_RR, LDHs_RR, LDHz_RR, LDW_RR, LDWz_RR
825 printMemRrOperand(MI, 1, O);
826 O << ", ";
827 printOperand(MI, 0, O);
828 return;
829 break;
830 case 9:
831 // LDW_RI
832 printMemRiOperand(MI, 1, O);
833 O << ", ";
834 printOperand(MI, 0, O);
835 return;
836 break;
837 case 10:
838 // MOVHI
839 printHi16ImmOperand(MI, 1, O);
840 O << ", ";
841 printOperand(MI, 0, O);
842 return;
843 break;
844 case 11:
845 // SELECT
846 printCCOperand(MI, 3, O);
847 O << ' ';
848 printOperand(MI, 1, O);
849 O << ", ";
850 printOperand(MI, 2, O);
851 O << ", ";
852 printOperand(MI, 0, O);
853 return;
854 break;
855 }
856
857
858 // Fragment 1 encoded into 4 bits for 10 unique commands.
859 switch ((Bits >> 14) & 15) {
860 default: llvm_unreachable("Invalid command number.");
861 case 0:
862 // ADJCALLSTACKDOWN, ADJCALLSTACKUP, ADJDYNALLOC
863 O << ' ';
864 printOperand(MI, 1, O);
865 return;
866 break;
867 case 1:
868 // ADDC_F_I_HI, ADDC_I_HI, ADD_F_I_HI, ADD_I_HI, OR_F_I_HI, OR_I_HI, SUBB...
869 printHi16ImmOperand(MI, 2, O);
870 O << ", ";
871 printOperand(MI, 0, O);
872 return;
873 break;
874 case 2:
875 // ADDC_F_I_LO, ADDC_I_LO, ADD_F_I_LO, ADD_I_LO, OR_F_I_LO, OR_I_LO, SA_F...
876 printOperand(MI, 2, O);
877 O << ", ";
878 printOperand(MI, 0, O);
879 return;
880 break;
881 case 3:
882 // AND_F_I_HI, AND_I_HI
883 printHi16AndImmOperand(MI, 2, O);
884 O << ", ";
885 printOperand(MI, 0, O);
886 return;
887 break;
888 case 4:
889 // AND_F_I_LO, AND_I_LO
890 printLo16AndImmOperand(MI, 2, O);
891 O << ", ";
892 printOperand(MI, 0, O);
893 return;
894 break;
895 case 5:
896 // BRCC, BRIND_CC, SCC
897 O << "\t";
898 printOperand(MI, 0, O);
899 return;
900 break;
901 case 6:
902 // BRR
903 O << ".r\t";
904 printOperand(MI, 0, O);
905 return;
906 break;
907 case 7:
908 // BT, JR
909 return;
910 break;
911 case 8:
912 // LEADZ, POPC, SLI, TRAILZ
913 printOperand(MI, 0, O);
914 return;
915 break;
916 case 9:
917 // SFSUB_F_RI_HI, SFSUB_F_RI_LO, SFSUB_F_RR, STADDR, STB_RI, STB_RR, STH_...
918 O << ", ";
919 break;
920 }
921
922
923 // Fragment 2 encoded into 3 bits for 6 unique commands.
924 switch ((Bits >> 18) & 7) {
925 default: llvm_unreachable("Invalid command number.");
926 case 0:
927 // SFSUB_F_RI_HI
928 printHi16ImmOperand(MI, 1, O);
929 O << ", %r0";
930 return;
931 break;
932 case 1:
933 // SFSUB_F_RI_LO, SFSUB_F_RR
934 printOperand(MI, 1, O);
935 O << ", %r0";
936 return;
937 break;
938 case 2:
939 // STADDR
940 printMemImmOperand(MI, 1, O);
941 return;
942 break;
943 case 3:
944 // STB_RI, STH_RI
945 printMemSplsOperand(MI, 1, O);
946 return;
947 break;
948 case 4:
949 // STB_RR, STH_RR, SW_RR
950 printMemRrOperand(MI, 1, O);
951 return;
952 break;
953 case 5:
954 // SW_RI
955 printMemRiOperand(MI, 1, O);
956 return;
957 break;
958 }

960}

1080 /* 0 */ "mov $\xFF\x03\x01, $\x01\0"
1081 /* 13 */ "mov $\x03, $\x01\0"
