341void LanaiAsmParser::
342convertToMCInst(unsigned Kind, MCInst &Inst, unsigned Opcode,
343 const OperandVector &Operands) {
344 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
345 const uint8_t *Converter = ConversionTable[Kind];
346 unsigned OpIdx;
347 Inst.setOpcode(Opcode);
348 for (const uint8_t *p = Converter; *p; p += 2) {
349 OpIdx = *(p + 1);
350 switch (*p) {
351 default: llvm_unreachable("invalid conversion entry!");
352 case CVT_Reg:
353 static_cast<LanaiOperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
354 break;
355 case CVT_Tied: {
356 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
357 std::begin(TiedAsmOperandTable)) &&
358 "Tied operand not found");
359 unsigned TiedResOpnd = TiedAsmOperandTable[OpIdx][0];
360 if (TiedResOpnd != (uint8_t)-1)
361 Inst.addOperand(Inst.getOperand(TiedResOpnd));
362 break;
363 }
364 case CVT_95_addImmOperands:
365 static_cast<LanaiOperand &>(*Operands[OpIdx]).addImmOperands(Inst, 1);
366 break;
367 case CVT_95_Reg:
368 static_cast<LanaiOperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
369 break;
370 case CVT_95_addHiImm16Operands:
371 static_cast<LanaiOperand &>(*Operands[OpIdx]).addHiImm16Operands(Inst, 1);
372 break;
373 case CVT_95_addLoImm16Operands:
374 static_cast<LanaiOperand &>(*Operands[OpIdx]).addLoImm16Operands(Inst, 1);
375 break;
376 case CVT_95_addCondCodeOperands:
377 static_cast<LanaiOperand &>(*Operands[OpIdx]).addCondCodeOperands(Inst, 1);
378 break;
379 case CVT_95_addHiImm16AndOperands:
380 static_cast<LanaiOperand &>(*Operands[OpIdx]).addHiImm16AndOperands(Inst, 1);
381 break;
382 case CVT_95_addLoImm16AndOperands:
383 static_cast<LanaiOperand &>(*Operands[OpIdx]).addLoImm16AndOperands(Inst, 1);
384 break;
385 case CVT_95_addBrTargetOperands:
386 static_cast<LanaiOperand &>(*Operands[OpIdx]).addBrTargetOperands(Inst, 1);
387 break;
388 case CVT_95_addMemImmOperands:
389 static_cast<LanaiOperand &>(*Operands[OpIdx]).addMemImmOperands(Inst, 1);
390 break;
391 case CVT_95_addMemRegImmOperands:
392 static_cast<LanaiOperand &>(*Operands[OpIdx]).addMemRegImmOperands(Inst, 3);
393 break;
394 case CVT_95_addMemRegRegOperands:

859static bool checkAsmTiedOperandConstraints(const LanaiAsmParser&AsmParser,
860 unsigned Kind,
861 const OperandVector &Operands,
862 uint64_t &ErrorInfo) {
863 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
864 const uint8_t *Converter = ConversionTable[Kind];
865 for (const uint8_t *p = Converter; *p; p += 2) {
866 switch (*p) {
867 case CVT_Tied: {
868 unsigned OpIdx = *(p + 1);
869 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
870 std::begin(TiedAsmOperandTable)) &&
871 "Tied operand not found");
872 unsigned OpndNum1 = TiedAsmOperandTable[OpIdx][1];
873 unsigned OpndNum2 = TiedAsmOperandTable[OpIdx][2];
874 if (OpndNum1 != OpndNum2) {
875 auto &SrcOp1 = Operands[OpndNum1];
876 auto &SrcOp2 = Operands[OpndNum2];
877 if (SrcOp1->isReg() && SrcOp2->isReg()) {
878 if (!AsmParser.regsEqual(*SrcOp1, *SrcOp2)) {
879 ErrorInfo = OpndNum2;
880 return false;
881 }
882 }
883 }
884 break;
885 }
886 default:
887 break;
888 }
889 }
890 return true;
891}

1040unsigned LanaiAsmParser::
1041MatchInstructionImpl(const OperandVector &Operands,
1042 MCInst &Inst,
1043 uint64_t &ErrorInfo,
1044 FeatureBitset &MissingFeatures,
1045 bool matchingInlineAsm, unsigned VariantID) {
1046 // Eliminate obvious mismatches.
1047 if (Operands.size() > 8) {
1048 ErrorInfo = 8;
1049 return Match_InvalidOperand;
1050 }
1051
1052 // Get the current feature set.
1053 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1054
1055 // Get the instruction mnemonic, which is the first token.
1056 StringRef Mnemonic = ((LanaiOperand &)*Operands[0]).getToken();
1057
1058 // Some state to try to produce better error messages.
1059 bool HadMatchOtherThanFeatures = false;
1060 bool HadMatchOtherThanPredicate = false;
1061 unsigned RetCode = Match_InvalidOperand;
1062 MissingFeatures.set();
1063 // Set ErrorInfo to the operand that mismatches if it is
1064 // wrong for all instances of the instruction.
1065 ErrorInfo = ~0ULL;
1066 // Find the appropriate table for this asm variant.
1067 const MatchEntry *Start, *End;
1068 switch (VariantID) {
1069 default: llvm_unreachable("invalid variant!");
1070 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1071 }
1072 // Search the table.
1073 auto MnemonicRange = std::equal_range(Start, End, Mnemonic, LessOpcode());
1074
1075 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "AsmMatcher: found " <<
1076 std::distance(MnemonicRange.first, MnemonicRange.second) <<
1077 " encodings with mnemonic '" << Mnemonic << "'\n");
1078
1079 // Return a more specific error code if no mnemonics match.
1080 if (MnemonicRange.first == MnemonicRange.second)
1081 return Match_MnemonicFail;
1082
1083 for (const MatchEntry *it = MnemonicRange.first, *ie = MnemonicRange.second;
1084 it != ie; ++it) {
1085 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1086 bool HasRequiredFeatures =
1087 (AvailableFeatures & RequiredFeatures) == RequiredFeatures;
1088 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Trying to match opcode "
1089 << MII.getName(it->Opcode) << "\n");
1090 // equal_range guarantees that instruction mnemonic matches.
1091 assert(Mnemonic == it->getMnemonic());
1092 bool OperandsValid = true;
1093 for (unsigned FormalIdx = 0, ActualIdx = 1; FormalIdx != 7; ++FormalIdx) {
1094 auto Formal = static_cast<MatchClassKind>(it->Classes[FormalIdx]);
1095 DEBUG_WITH_TYPE("asm-matcher",
1096 dbgs() << " Matching formal operand class " << getMatchClassName(Formal)
1097 << " against actual operand at index " << ActualIdx);
1098 if (ActualIdx < Operands.size())
1099 DEBUG_WITH_TYPE("asm-matcher", dbgs() << " (";
1100 Operands[ActualIdx]->print(dbgs()); dbgs() << "): ");
1101 else
1102 DEBUG_WITH_TYPE("asm-matcher", dbgs() << ": ");
1103 if (ActualIdx >= Operands.size()) {
1104 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "actual operand index out of range ");
1105 OperandsValid = (Formal == InvalidMatchClass) || isSubclass(Formal, OptionalMatchClass);
1106 if (!OperandsValid) ErrorInfo = ActualIdx;
1107 break;
1108 }
1109 MCParsedAsmOperand &Actual = *Operands[ActualIdx];
1110 unsigned Diag = validateOperandClass(Actual, Formal);
1111 if (Diag == Match_Success) {
1112 DEBUG_WITH_TYPE("asm-matcher",
1113 dbgs() << "match success using generic matcher\n");
1114 ++ActualIdx;
1115 continue;
1116 }
1117 // If the generic handler indicates an invalid operand
1118 // failure, check for a special case.
1119 if (Diag != Match_Success) {
1120 unsigned TargetDiag = validateTargetOperandClass(Actual, Formal);
1121 if (TargetDiag == Match_Success) {
1122 DEBUG_WITH_TYPE("asm-matcher",
1123 dbgs() << "match success using target matcher\n");
1124 ++ActualIdx;
1125 continue;
1126 }
1127 // If the target matcher returned a specific error code use
1128 // that, else use the one from the generic matcher.
1129 if (TargetDiag != Match_InvalidOperand && HasRequiredFeatures)
1130 Diag = TargetDiag;
1131 }
1132 // If current formal operand wasn't matched and it is optional
1133 // then try to match next formal operand
1134 if (Diag == Match_InvalidOperand && isSubclass(Formal, OptionalMatchClass)) {
1135 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "ignoring optional operand\n");
1136 continue;
1137 }
1138 // If this operand is broken for all of the instances of this
1139 // mnemonic, keep track of it so we can report loc info.
1140 // If we already had a match that only failed due to a
1141 // target predicate, that diagnostic is preferred.
1142 if (!HadMatchOtherThanPredicate &&
1143 (it == MnemonicRange.first || ErrorInfo <= ActualIdx)) {
1144 if (HasRequiredFeatures && (ErrorInfo != ActualIdx || Diag != Match_InvalidOperand))
1145 RetCode = Diag;
1146 ErrorInfo = ActualIdx;
1147 }
1148 // Otherwise, just reject this instance of the mnemonic.
1149 OperandsValid = false;
1150 break;
1151 }
1152
1153 if (!OperandsValid) {
1154 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Opcode result: multiple "
1155 "operand mismatches, ignoring "
1156 "this opcode\n");
1157 continue;
1158 }
1159 if (!HasRequiredFeatures) {
1160 HadMatchOtherThanFeatures = true;
1161 FeatureBitset NewMissingFeatures = RequiredFeatures & ~AvailableFeatures;
1162 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Missing target features:";
1163 for (unsigned I = 0, E = NewMissingFeatures.size(); I != E; ++I)
1164 if (NewMissingFeatures[I])
1165 dbgs() << ' ' << I;
1166 dbgs() << "\n");
1167 if (NewMissingFeatures.count() <=
1168 MissingFeatures.count())
1169 MissingFeatures = NewMissingFeatures;
1170 continue;
1171 }
1172
1173 Inst.clear();
1174
1175 Inst.setOpcode(it->Opcode);
1176 // We have a potential match but have not rendered the operands.
1177 // Check the target predicate to handle any context sensitive
1178 // constraints.
1179 // For example, Ties that are referenced multiple times must be
1180 // checked here to ensure the input is the same for each match
1181 // constraints. If we leave it any later the ties will have been
1182 // canonicalized
1183 unsigned MatchResult;
1184 if ((MatchResult = checkEarlyTargetMatchPredicate(Inst, Operands)) != Match_Success) {
1185 Inst.clear();
1186 DEBUG_WITH_TYPE(
1187 "asm-matcher",
1188 dbgs() << "Early target match predicate failed with diag code "
1189 << MatchResult << "\n");
1190 RetCode = MatchResult;
1191 HadMatchOtherThanPredicate = true;
1192 continue;
1193 }
1194
1195 if (matchingInlineAsm) {
1196 convertToMapAndConstraints(it->ConvertFn, Operands);
1197 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1198 return Match_InvalidTiedOperand;
1199
1200 return Match_Success;
1201 }
1202
1203 // We have selected a definite instruction, convert the parsed
1204 // operands into the appropriate MCInst.
1205 convertToMCInst(it->ConvertFn, Inst, it->Opcode, Operands);
1206
1207 // We have a potential match. Check the target predicate to
1208 // handle any context sensitive constraints.
1209 if ((MatchResult = checkTargetMatchPredicate(Inst)) != Match_Success) {
1210 DEBUG_WITH_TYPE("asm-matcher",
1211 dbgs() << "Target match predicate failed with diag code "
1212 << MatchResult << "\n");
1213 Inst.clear();
1214 RetCode = MatchResult;
1215 HadMatchOtherThanPredicate = true;
1216 continue;
1217 }
1218
1219 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1220 return Match_InvalidTiedOperand;
1221
1222 DEBUG_WITH_TYPE(
1223 "asm-matcher",
1224 dbgs() << "Opcode result: complete match, selecting this opcode\n");
1225 return Match_Success;
1226 }
1227
1228 // Okay, we had no match. Try to return a useful error code.
1229 if (HadMatchOtherThanPredicate || !HadMatchOtherThanFeatures)
1230 return RetCode;
1231
1232 ErrorInfo = 0;
1233 return Match_MissingFeature;
1234}

1306OperandMatchResultTy LanaiAsmParser::
1307MatchOperandParserImpl(OperandVector &Operands,
1308 StringRef Mnemonic,
1309 bool ParseForAllFeatures) {
1310 // Get the current feature set.
1311 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1312
1313 // Get the next operand index.
1314 unsigned NextOpNum = Operands.size() - 1;
1315 // Search the table.
1316 auto MnemonicRange =
1317 std::equal_range(std::begin(OperandMatchTable), std::end(OperandMatchTable),
1318 Mnemonic, LessOpcodeOperand());
1319
1320 if (MnemonicRange.first == MnemonicRange.second)
1321 return MatchOperand_NoMatch;
1322
1323 for (const OperandMatchEntry *it = MnemonicRange.first,
1324 *ie = MnemonicRange.second; it != ie; ++it) {
1325 // equal_range guarantees that instruction mnemonic matches.
1326 assert(Mnemonic == it->getMnemonic());
1327
1328 // check if the available features match
1329 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1330 if (!ParseForAllFeatures && (AvailableFeatures & RequiredFeatures) != RequiredFeatures)
1331 continue;
1332
1333 // check if the operand in question has a custom parser.
1334 if (!(it->OperandMask & (1 << NextOpNum)))
1335 continue;
1336
1337 // call custom parse method to handle the operand
1338 OperandMatchResultTy Result = tryCustomParseOperand(Operands, it->Class);
1339 if (Result != MatchOperand_NoMatch)
1340 return Result;
1341 }
1342
1343 // Okay, we had no match.
1344 return MatchOperand_NoMatch;
1345}

1354 const unsigned MaxEditDist = 2;
1355 std::vector<StringRef> Candidates;
1356 StringRef Prev = "";
1357
1358 // Find the appropriate table for this asm variant.

1365 for (auto I = Start; I < End; I++) {
1366 // Ignore unsupported instructions.

1368 if ((FBS & RequiredFeatures) != RequiredFeatures)

1371 StringRef T = I->getMnemonic();
1372 // Avoid recomputing the edit distance for the same string.
1373 if (T.equals(Prev))
1374 continue;

1376 Prev = T;
1377 unsigned Dist = S.edit_distance(T, false, MaxEditDist);
1378 if (Dist <= MaxEditDist)
1379 Candidates.push_back(T);

1382 if (Candidates.empty())
1383 return "";
1384
1385 std::string Res = ", did you mean: ";
1386 unsigned i = 0;
1387 for (; i < Candidates.size() - 1; i++)
1388 Res += Candidates[i].str() + ", ";
1389 return Res + Candidates[i].str() + "?";
1390}
