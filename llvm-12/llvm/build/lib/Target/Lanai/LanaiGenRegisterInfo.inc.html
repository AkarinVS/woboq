782 struct MaskRolOp {
783 LaneBitmask Mask;
784 uint8_t RotateLeft;
785 };
786 static const MaskRolOp LaneMaskComposeSequences[] = {
787 { LaneBitmask(0xFFFFFFFFFFFFFFFF), 0 }, { LaneBitmask::getNone(), 0 } // Sequence 0
788 };
789 static const MaskRolOp *const CompositeSequences[] = {
790 &LaneMaskComposeSequences[0] // to sub_32
791 };
792
793LaneBitmask LanaiGenRegisterInfo::composeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
794 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
795 LaneBitmask Result;
796 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
797 LaneBitmask::Type M = LaneMask.getAsInteger() & Ops->Mask.getAsInteger();
798 if (unsigned S = Ops->RotateLeft)
799 Result |= LaneBitmask((M << S) | (M >> (LaneBitmask::BitWidth - S)));
800 else
801 Result |= LaneBitmask(M);
802 }
803 return Result;
804}
805
806LaneBitmask LanaiGenRegisterInfo::reverseComposeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
807 LaneMask &= getSubRegIndexLaneMask(IdxA);
808 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
809 LaneBitmask Result;
810 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
811 LaneBitmask::Type M = LaneMask.getAsInteger();
812 if (unsigned S = Ops->RotateLeft)
813 Result |= LaneBitmask((M >> S) | (M << (LaneBitmask::BitWidth - S)));
814 else
815 Result |= LaneBitmask(M);
816 }
817 return Result;
818}
