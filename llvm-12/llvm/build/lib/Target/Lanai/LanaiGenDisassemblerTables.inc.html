18// Helper functions for extracting fields from encoded instructions.
19// InsnType must either be integral or an APInt-like object that must:
20// * Have a static const max_size_in_bits equal to the number of bits in the
21// encoding.
22// * be default-constructible and copy-constructible
23// * be constructible from a uint64_t
24// * be constructible from an APInt (this can be private)
25// * Support getBitsSet(loBit, hiBit)
26// * be convertible to uint64_t
27// * Support the ~, &, ==, !=, and |= operators with other objects of the same type
28// * Support shift (<<, >>) with signed and unsigned integers on the RHS
29// * Support put (<<) to raw_ostream&
30template <typename InsnType>
31#if defined(_MSC_VER) && !defined(__clang__)
32__declspec(noinline)
33#endif
34static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
35 unsigned numBits, std::true_type) {
36 assert(startBit + numBits <= 64 && "Cannot support >64-bit extractions!");
37 assert(startBit + numBits <= (sizeof(InsnType) * 8) &&
38 "Instruction field out of bounds!");
39 InsnType fieldMask;
40 if (numBits == sizeof(InsnType) * 8)
41 fieldMask = (InsnType)(-1LL);
42 else
43 fieldMask = (((InsnType)1 << numBits) - 1) << startBit;
44 return (insn & fieldMask) >> startBit;
45}

47template <typename InsnType>
48static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
49 unsigned numBits, std::false_type) {
50 assert(startBit + numBits <= InsnType::max_size_in_bits && "Instruction field out of bounds!");
51 InsnType fieldMask = InsnType::getBitsSet(0, numBits);
52 return (insn >> startBit) & fieldMask;
53}

55template <typename InsnType>
56static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
57 unsigned numBits) {
58 return fieldFromInstruction(insn, startBit, numBits, std::is_integral<InsnType>());
59}

452template <typename InsnType>
453static DecodeStatus decodeInstruction(const uint8_t DecodeTable[], MCInst &MI,
454 InsnType insn, uint64_t Address,
455 const void *DisAsm,
456 const MCSubtargetInfo &STI) {
457 const FeatureBitset &Bits = STI.getFeatureBits();
458
459 const uint8_t *Ptr = DecodeTable;
460 InsnType CurFieldValue = 0;
461 DecodeStatus S = MCDisassembler::Success;
462 while (true) {
463 ptrdiff_t Loc = Ptr - DecodeTable;
464 switch (*Ptr) {
465 default:
466 errs() << Loc << ": Unexpected decode table opcode!\n";
467 return MCDisassembler::Fail;
468 case MCD::OPC_ExtractField: {
469 unsigned Start = *++Ptr;
470 unsigned Len = *++Ptr;
471 ++Ptr;
472 CurFieldValue = fieldFromInstruction(insn, Start, Len);
473 LLVM_DEBUG(dbgs() << Loc << ": OPC_ExtractField(" << Start << ", "
474 << Len << "): " << CurFieldValue << "\n");
475 break;
476 }
477 case MCD::OPC_FilterValue: {
478 // Decode the field value.
479 unsigned Len;
480 InsnType Val = decodeULEB128(++Ptr, &Len);
481 Ptr += Len;
482 // NumToSkip is a plain 24-bit integer.
483 unsigned NumToSkip = *Ptr++;
484 NumToSkip |= (*Ptr++) << 8;
485 NumToSkip |= (*Ptr++) << 16;
486
487 // Perform the filter operation.
488 if (Val != CurFieldValue)
489 Ptr += NumToSkip;
490 LLVM_DEBUG(dbgs() << Loc << ": OPC_FilterValue(" << Val << ", " << NumToSkip
491 << "): " << ((Val != CurFieldValue) ? "FAIL:" : "PASS:")
492 << " continuing at " << (Ptr - DecodeTable) << "\n");
493
494 break;
495 }
496 case MCD::OPC_CheckField: {
497 unsigned Start = *++Ptr;
498 unsigned Len = *++Ptr;
499 InsnType FieldValue = fieldFromInstruction(insn, Start, Len);
500 // Decode the field value.
501 InsnType ExpectedValue = decodeULEB128(++Ptr, &Len);
502 Ptr += Len;
503 // NumToSkip is a plain 24-bit integer.
504 unsigned NumToSkip = *Ptr++;
505 NumToSkip |= (*Ptr++) << 8;
506 NumToSkip |= (*Ptr++) << 16;
507
508 // If the actual and expected values don't match, skip.
509 if (ExpectedValue != FieldValue)
510 Ptr += NumToSkip;
511 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckField(" << Start << ", "
512 << Len << ", " << ExpectedValue << ", " << NumToSkip
513 << "): FieldValue = " << FieldValue << ", ExpectedValue = "
514 << ExpectedValue << ": "
515 << ((ExpectedValue == FieldValue) ? "PASS\n" : "FAIL\n"));
516 break;
517 }
518 case MCD::OPC_CheckPredicate: {
519 unsigned Len;
520 // Decode the Predicate Index value.
521 unsigned PIdx = decodeULEB128(++Ptr, &Len);
522 Ptr += Len;
523 // NumToSkip is a plain 24-bit integer.
524 unsigned NumToSkip = *Ptr++;
525 NumToSkip |= (*Ptr++) << 8;
526 NumToSkip |= (*Ptr++) << 16;
527 // Check the predicate.
528 bool Pred;
529 if (!(Pred = checkDecoderPredicate(PIdx, Bits)))
530 Ptr += NumToSkip;
531 (void)Pred;
532 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckPredicate(" << PIdx << "): "
533 << (Pred ? "PASS\n" : "FAIL\n"));
534
535 break;
536 }
537 case MCD::OPC_Decode: {
538 unsigned Len;
539 // Decode the Opcode value.
540 unsigned Opc = decodeULEB128(++Ptr, &Len);
541 Ptr += Len;
542 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
543 Ptr += Len;
544
545 MI.clear();
546 MI.setOpcode(Opc);
547 bool DecodeComplete;
548 S = decodeToMCInst(S, DecodeIdx, insn, MI, Address, DisAsm, DecodeComplete);
549 assert(DecodeComplete);
550
551 LLVM_DEBUG(dbgs() << Loc << ": OPC_Decode: opcode " << Opc
552 << ", using decoder " << DecodeIdx << ": "
553 << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
554 return S;
555 }
556 case MCD::OPC_TryDecode: {
557 unsigned Len;
558 // Decode the Opcode value.
559 unsigned Opc = decodeULEB128(++Ptr, &Len);
560 Ptr += Len;
561 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
562 Ptr += Len;
563 // NumToSkip is a plain 24-bit integer.
564 unsigned NumToSkip = *Ptr++;
565 NumToSkip |= (*Ptr++) << 8;
566 NumToSkip |= (*Ptr++) << 16;
567
568 // Perform the decode operation.
569 MCInst TmpMI;
570 TmpMI.setOpcode(Opc);
571 bool DecodeComplete;
572 S = decodeToMCInst(S, DecodeIdx, insn, TmpMI, Address, DisAsm, DecodeComplete);
573 LLVM_DEBUG(dbgs() << Loc << ": OPC_TryDecode: opcode " << Opc
574 << ", using decoder " << DecodeIdx << ": ");
575
576 if (DecodeComplete) {
577 // Decoding complete.
578 LLVM_DEBUG(dbgs() << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
579 MI = TmpMI;
580 return S;
581 } else {
582 assert(S == MCDisassembler::Fail);
583 // If the decoding was incomplete, skip.
584 Ptr += NumToSkip;
585 LLVM_DEBUG(dbgs() << "FAIL: continuing at " << (Ptr - DecodeTable) << "\n");
586 // Reset decode status. This also drops a SoftFail status that could be
587 // set before the decode attempt.
588 S = MCDisassembler::Success;
589 }
590 break;
591 }
592 case MCD::OPC_SoftFail: {
593 // Decode the mask values.
594 unsigned Len;
595 InsnType PositiveMask = decodeULEB128(++Ptr, &Len);
596 Ptr += Len;
597 InsnType NegativeMask = decodeULEB128(Ptr, &Len);
598 Ptr += Len;
599 bool Fail = (insn & PositiveMask) || (~insn & NegativeMask);
600 if (Fail)
601 S = MCDisassembler::SoftFail;
602 LLVM_DEBUG(dbgs() << Loc << ": OPC_SoftFail: " << (Fail ? "FAIL\n" : "PASS\n"));
603 break;
604 }
605 case MCD::OPC_Fail: {
606 LLVM_DEBUG(dbgs() << Loc << ": OPC_Fail\n");
607 return MCDisassembler::Fail;
608 }
609 }
610 }
611 llvm_unreachable("bogosity detected in disassembler state machine!");
612}
