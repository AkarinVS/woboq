863::resolveSchedClass(unsigned SchedClass, const MachineInstr *MI, const TargetSchedModel *SchedModel) const {
864
865 const SIInstrInfo *TII =
866 static_cast<const SIInstrInfo*>(SchedModel->getInstrInfo());
867 (void)TII;
868
869 switch (SchedClass) {
870 case 24: // V_ACCVGPR_WRITE_B32_e64
871 if (SchedModel->getProcessorID() == 1) { // SIQuarterSpeedModel
872 if (TII->hasVGPRUses(*MI))
873 return 29; // Write64Bit_MIVGPRRead
874 return 30; // Write64Bit_ReadDefault
875 }
876 if (SchedModel->getProcessorID() == 3) { // SIFullSpeedModel
877 if (TII->hasVGPRUses(*MI))
878 return 29; // Write64Bit_MIVGPRRead
879 return 30; // Write64Bit_ReadDefault
880 }
881 break;
882 case 28: // COPY
883 if (SchedModel->getProcessorID() == 1) { // SIQuarterSpeedModel
884 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) <= 32)
885 return 31; // Write32Bit
886 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) > 32)
887 return 32; // Write64Bit
888 return 33; // WriteSALU
889 }
890 if (SchedModel->getProcessorID() == 2) { // GFX10SpeedModel
891 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) <= 32)
892 return 31; // Write32Bit
893 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) > 32)
894 return 32; // Write64Bit
895 return 33; // WriteSALU
896 }
897 if (SchedModel->getProcessorID() == 3) { // SIFullSpeedModel
898 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) <= 32)
899 return 31; // Write32Bit
900 if (TII->isVGPRCopy(*MI) && TII->getOpSize(*MI, 0) > 32)
901 return 32; // Write64Bit
902 return 33; // WriteSALU
903 }
904 break;
905 };
906 report_fatal_error("Expected a variant SchedClass");
