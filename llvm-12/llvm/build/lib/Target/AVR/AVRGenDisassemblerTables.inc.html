18// Helper functions for extracting fields from encoded instructions.
19// InsnType must either be integral or an APInt-like object that must:
20// * Have a static const max_size_in_bits equal to the number of bits in the
21// encoding.
22// * be default-constructible and copy-constructible
23// * be constructible from a uint64_t
24// * be constructible from an APInt (this can be private)
25// * Support getBitsSet(loBit, hiBit)
26// * be convertible to uint64_t
27// * Support the ~, &, ==, !=, and |= operators with other objects of the same type
28// * Support shift (<<, >>) with signed and unsigned integers on the RHS
29// * Support put (<<) to raw_ostream&
30template <typename InsnType>
31#if defined(_MSC_VER) && !defined(__clang__)
32__declspec(noinline)
33#endif
34static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
35 unsigned numBits, std::true_type) {
36 assert(startBit + numBits <= 64 && "Cannot support >64-bit extractions!");
37 assert(startBit + numBits <= (sizeof(InsnType) * 8) &&
38 "Instruction field out of bounds!");
39 InsnType fieldMask;
40 if (numBits == sizeof(InsnType) * 8)
41 fieldMask = (InsnType)(-1LL);
42 else
43 fieldMask = (((InsnType)1 << numBits) - 1) << startBit;
44 return (insn & fieldMask) >> startBit;
45}
46
47template <typename InsnType>
48static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
49 unsigned numBits, std::false_type) {
50 assert(startBit + numBits <= InsnType::max_size_in_bits && "Instruction field out of bounds!");
51 InsnType fieldMask = InsnType::getBitsSet(0, numBits);
52 return (insn >> startBit) & fieldMask;
53}

55template <typename InsnType>
56static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
57 unsigned numBits) {
58 return fieldFromInstruction(insn, startBit, numBits, std::is_integral<InsnType>());
59}

595template <typename InsnType>
596static DecodeStatus decodeInstruction(const uint8_t DecodeTable[], MCInst &MI,
597 InsnType insn, uint64_t Address,
598 const void *DisAsm,
599 const MCSubtargetInfo &STI) {
600 const FeatureBitset &Bits = STI.getFeatureBits();
601
602 const uint8_t *Ptr = DecodeTable;
603 InsnType CurFieldValue = 0;
604 DecodeStatus S = MCDisassembler::Success;
605 while (true) {
606 ptrdiff_t Loc = Ptr - DecodeTable;
607 switch (*Ptr) {
608 default:
609 errs() << Loc << ": Unexpected decode table opcode!\n";
610 return MCDisassembler::Fail;
611 case MCD::OPC_ExtractField: {
612 unsigned Start = *++Ptr;
613 unsigned Len = *++Ptr;
614 ++Ptr;
615 CurFieldValue = fieldFromInstruction(insn, Start, Len);
616 LLVM_DEBUG(dbgs() << Loc << ": OPC_ExtractField(" << Start << ", "
617 << Len << "): " << CurFieldValue << "\n");
618 break;
619 }
620 case MCD::OPC_FilterValue: {
621 // Decode the field value.
622 unsigned Len;
623 InsnType Val = decodeULEB128(++Ptr, &Len);
624 Ptr += Len;
625 // NumToSkip is a plain 24-bit integer.
626 unsigned NumToSkip = *Ptr++;
627 NumToSkip |= (*Ptr++) << 8;
628 NumToSkip |= (*Ptr++) << 16;
629
630 // Perform the filter operation.
631 if (Val != CurFieldValue)
632 Ptr += NumToSkip;
633 LLVM_DEBUG(dbgs() << Loc << ": OPC_FilterValue(" << Val << ", " << NumToSkip
634 << "): " << ((Val != CurFieldValue) ? "FAIL:" : "PASS:")
635 << " continuing at " << (Ptr - DecodeTable) << "\n");
636
637 break;
638 }
639 case MCD::OPC_CheckField: {
640 unsigned Start = *++Ptr;
641 unsigned Len = *++Ptr;
642 InsnType FieldValue = fieldFromInstruction(insn, Start, Len);
643 // Decode the field value.
644 InsnType ExpectedValue = decodeULEB128(++Ptr, &Len);
645 Ptr += Len;
646 // NumToSkip is a plain 24-bit integer.
647 unsigned NumToSkip = *Ptr++;
648 NumToSkip |= (*Ptr++) << 8;
649 NumToSkip |= (*Ptr++) << 16;
650
651 // If the actual and expected values don't match, skip.
652 if (ExpectedValue != FieldValue)
653 Ptr += NumToSkip;
654 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckField(" << Start << ", "
655 << Len << ", " << ExpectedValue << ", " << NumToSkip
656 << "): FieldValue = " << FieldValue << ", ExpectedValue = "
657 << ExpectedValue << ": "
658 << ((ExpectedValue == FieldValue) ? "PASS\n" : "FAIL\n"));
659 break;
660 }
661 case MCD::OPC_CheckPredicate: {
662 unsigned Len;
663 // Decode the Predicate Index value.
664 unsigned PIdx = decodeULEB128(++Ptr, &Len);
665 Ptr += Len;
666 // NumToSkip is a plain 24-bit integer.
667 unsigned NumToSkip = *Ptr++;
668 NumToSkip |= (*Ptr++) << 8;
669 NumToSkip |= (*Ptr++) << 16;
670 // Check the predicate.
671 bool Pred;
672 if (!(Pred = checkDecoderPredicate(PIdx, Bits)))
673 Ptr += NumToSkip;
674 (void)Pred;
675 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckPredicate(" << PIdx << "): "
676 << (Pred ? "PASS\n" : "FAIL\n"));
677
678 break;
679 }
680 case MCD::OPC_Decode: {
681 unsigned Len;
682 // Decode the Opcode value.
683 unsigned Opc = decodeULEB128(++Ptr, &Len);
684 Ptr += Len;
685 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
686 Ptr += Len;
687
688 MI.clear();
689 MI.setOpcode(Opc);
690 bool DecodeComplete;
691 S = decodeToMCInst(S, DecodeIdx, insn, MI, Address, DisAsm, DecodeComplete);
692 assert(DecodeComplete);
693
694 LLVM_DEBUG(dbgs() << Loc << ": OPC_Decode: opcode " << Opc
695 << ", using decoder " << DecodeIdx << ": "
696 << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
697 return S;
698 }
699 case MCD::OPC_TryDecode: {
700 unsigned Len;
701 // Decode the Opcode value.
702 unsigned Opc = decodeULEB128(++Ptr, &Len);
703 Ptr += Len;
704 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
705 Ptr += Len;
706 // NumToSkip is a plain 24-bit integer.
707 unsigned NumToSkip = *Ptr++;
708 NumToSkip |= (*Ptr++) << 8;
709 NumToSkip |= (*Ptr++) << 16;
710
711 // Perform the decode operation.
712 MCInst TmpMI;
713 TmpMI.setOpcode(Opc);
714 bool DecodeComplete;
715 S = decodeToMCInst(S, DecodeIdx, insn, TmpMI, Address, DisAsm, DecodeComplete);
716 LLVM_DEBUG(dbgs() << Loc << ": OPC_TryDecode: opcode " << Opc
717 << ", using decoder " << DecodeIdx << ": ");
718
719 if (DecodeComplete) {
720 // Decoding complete.
721 LLVM_DEBUG(dbgs() << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
722 MI = TmpMI;
723 return S;
724 } else {
725 assert(S == MCDisassembler::Fail);
726 // If the decoding was incomplete, skip.
727 Ptr += NumToSkip;
728 LLVM_DEBUG(dbgs() << "FAIL: continuing at " << (Ptr - DecodeTable) << "\n");
729 // Reset decode status. This also drops a SoftFail status that could be
730 // set before the decode attempt.
731 S = MCDisassembler::Success;
732 }
733 break;
734 }
735 case MCD::OPC_SoftFail: {
736 // Decode the mask values.
737 unsigned Len;
738 InsnType PositiveMask = decodeULEB128(++Ptr, &Len);
739 Ptr += Len;
740 InsnType NegativeMask = decodeULEB128(Ptr, &Len);
741 Ptr += Len;
742 bool Fail = (insn & PositiveMask) || (~insn & NegativeMask);
743 if (Fail)
744 S = MCDisassembler::SoftFail;
745 LLVM_DEBUG(dbgs() << Loc << ": OPC_SoftFail: " << (Fail ? "FAIL\n" : "PASS\n"));
746 break;
747 }
748 case MCD::OPC_Fail: {
749 LLVM_DEBUG(dbgs() << Loc << ": OPC_Fail\n");
750 return MCDisassembler::Fail;
751 }
752 }
753 }
754 llvm_unreachable("bogosity detected in disassembler state machine!");
755}
