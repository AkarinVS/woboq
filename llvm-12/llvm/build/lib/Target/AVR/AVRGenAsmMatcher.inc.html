529void AVRAsmParser::
530convertToMCInst(unsigned Kind, MCInst &Inst, unsigned Opcode,
531 const OperandVector &Operands) {
532 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
533 const uint8_t *Converter = ConversionTable[Kind];
534 unsigned OpIdx;
535 Inst.setOpcode(Opcode);
536 for (const uint8_t *p = Converter; *p; p += 2) {
537 OpIdx = *(p + 1);
538 switch (*p) {
539 default: llvm_unreachable("invalid conversion entry!");
540 case CVT_Reg:
541 static_cast<AVROperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
542 break;
543 case CVT_Tied: {
544 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
545 std::begin(TiedAsmOperandTable)) &&
546 "Tied operand not found");
547 unsigned TiedResOpnd = TiedAsmOperandTable[OpIdx][0];
548 if (TiedResOpnd != (uint8_t)-1)
549 Inst.addOperand(Inst.getOperand(TiedResOpnd));
550 break;
551 }
552 case CVT_95_Reg:
553 static_cast<AVROperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
554 break;
555 case CVT_95_addImmOperands:
556 static_cast<AVROperand &>(*Operands[OpIdx]).addImmOperands(Inst, 1);
557 break;
558 case CVT_imm_95_0:
559 Inst.addOperand(MCOperand::createImm(0));
560 break;
561 case CVT_imm_95_5:
562 Inst.addOperand(MCOperand::createImm(5));
563 break;
564 case CVT_imm_95_7:
565 Inst.addOperand(MCOperand::createImm(7));
566 break;
567 case CVT_imm_95_6:
568 Inst.addOperand(MCOperand::createImm(6));
569 break;
570 case CVT_imm_95_3:
571 Inst.addOperand(MCOperand::createImm(3));
572 break;
573 case CVT_95_addImmCom8Operands:
574 static_cast<AVROperand &>(*Operands[OpIdx]).addImmCom8Operands(Inst, 1);
575 break;
576 case CVT_imm_95_2:
577 Inst.addOperand(MCOperand::createImm(2));
578 break;
579 case CVT_imm_95_4:

598void AVRAsmParser::
599convertToMapAndConstraints(unsigned Kind,
600 const OperandVector &Operands) {
601 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
602 unsigned NumMCOperands = 0;
603 const uint8_t *Converter = ConversionTable[Kind];
604 for (const uint8_t *p = Converter; *p; p += 2) {
605 switch (*p) {
606 default: llvm_unreachable("invalid conversion entry!");
607 case CVT_Reg:
608 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
609 Operands[*(p + 1)]->setConstraint("r");
610 ++NumMCOperands;
611 break;
612 case CVT_Tied:
613 ++NumMCOperands;
614 break;
615 case CVT_95_Reg:
616 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
617 Operands[*(p + 1)]->setConstraint("r");
618 NumMCOperands += 1;
619 break;
620 case CVT_95_addImmOperands:

1165static bool checkAsmTiedOperandConstraints(const AVRAsmParser&AsmParser,
1166 unsigned Kind,
1167 const OperandVector &Operands,
1168 uint64_t &ErrorInfo) {
1169 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
1170 const uint8_t *Converter = ConversionTable[Kind];
1171 for (const uint8_t *p = Converter; *p; p += 2) {
1172 switch (*p) {
1173 case CVT_Tied: {
1174 unsigned OpIdx = *(p + 1);
1175 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
1176 std::begin(TiedAsmOperandTable)) &&
1177 "Tied operand not found");
1178 unsigned OpndNum1 = TiedAsmOperandTable[OpIdx][1];
1179 unsigned OpndNum2 = TiedAsmOperandTable[OpIdx][2];
1180 if (OpndNum1 != OpndNum2) {
1181 auto &SrcOp1 = Operands[OpndNum1];
1182 auto &SrcOp2 = Operands[OpndNum2];
1183 if (SrcOp1->isReg() && SrcOp2->isReg()) {
1184 if (!AsmParser.regsEqual(*SrcOp1, *SrcOp2)) {
1185 ErrorInfo = OpndNum2;
1186 return false;
1187 }
1188 }
1189 }
1190 break;
1191 }
1192 default:
1193 break;
1194 }
1195 }
1196 return true;

1416unsigned AVRAsmParser::
1417MatchInstructionImpl(const OperandVector &Operands,
1418 MCInst &Inst,
1419 uint64_t &ErrorInfo,
1420 FeatureBitset &MissingFeatures,
1421 bool matchingInlineAsm, unsigned VariantID) {
1422 // Eliminate obvious mismatches.
1423 if (Operands.size() > 4) {
1424 ErrorInfo = 4;
1425 return Match_InvalidOperand;
1426 }
1427
1428 // Get the current feature set.
1429 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1430
1431 // Get the instruction mnemonic, which is the first token.
1432 StringRef Mnemonic = ((AVROperand &)*Operands[0]).getToken();
1433
1434 // Some state to try to produce better error messages.
1435 bool HadMatchOtherThanFeatures = false;
1436 bool HadMatchOtherThanPredicate = false;
1437 unsigned RetCode = Match_InvalidOperand;
1438 MissingFeatures.set();
1439 // Set ErrorInfo to the operand that mismatches if it is
1440 // wrong for all instances of the instruction.
1441 ErrorInfo = ~0ULL;
1442 // Find the appropriate table for this asm variant.
1443 const MatchEntry *Start, *End;
1444 switch (VariantID) {
1445 default: llvm_unreachable("invalid variant!");
1446 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1447 }
1448 // Search the table.
1449 auto MnemonicRange = std::equal_range(Start, End, Mnemonic, LessOpcode());
1450
1451 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "AsmMatcher: found " <<
1452 std::distance(MnemonicRange.first, MnemonicRange.second) <<
1453 " encodings with mnemonic '" << Mnemonic << "'\n");
1454
1455 // Return a more specific error code if no mnemonics match.
1456 if (MnemonicRange.first == MnemonicRange.second)
1457 return Match_MnemonicFail;
1458
1459 for (const MatchEntry *it = MnemonicRange.first, *ie = MnemonicRange.second;
1460 it != ie; ++it) {
1461 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1462 bool HasRequiredFeatures =
1463 (AvailableFeatures & RequiredFeatures) == RequiredFeatures;
1464 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Trying to match opcode "
1465 << MII.getName(it->Opcode) << "\n");
1466 // equal_range guarantees that instruction mnemonic matches.
1467 assert(Mnemonic == it->getMnemonic());
1468 bool OperandsValid = true;
1469 for (unsigned FormalIdx = 0, ActualIdx = 1; FormalIdx != 3; ++FormalIdx) {
1470 auto Formal = static_cast<MatchClassKind>(it->Classes[FormalIdx]);
1471 DEBUG_WITH_TYPE("asm-matcher",
1472 dbgs() << " Matching formal operand class " << getMatchClassName(Formal)
1473 << " against actual operand at index " << ActualIdx);
1474 if (ActualIdx < Operands.size())
1475 DEBUG_WITH_TYPE("asm-matcher", dbgs() << " (";
1476 Operands[ActualIdx]->print(dbgs()); dbgs() << "): ");
1477 else
1478 DEBUG_WITH_TYPE("asm-matcher", dbgs() << ": ");
1479 if (ActualIdx >= Operands.size()) {
1480 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "actual operand index out of range ");
1481 OperandsValid = (Formal == InvalidMatchClass) || isSubclass(Formal, OptionalMatchClass);
1482 if (!OperandsValid) ErrorInfo = ActualIdx;
1483 break;
1484 }
1485 MCParsedAsmOperand &Actual = *Operands[ActualIdx];
1486 unsigned Diag = validateOperandClass(Actual, Formal);
1487 if (Diag == Match_Success) {
1488 DEBUG_WITH_TYPE("asm-matcher",
1489 dbgs() << "match success using generic matcher\n");
1490 ++ActualIdx;
1491 continue;
1492 }
1493 // If the generic handler indicates an invalid operand
1494 // failure, check for a special case.
1495 if (Diag != Match_Success) {
1496 unsigned TargetDiag = validateTargetOperandClass(Actual, Formal);
1497 if (TargetDiag == Match_Success) {
1498 DEBUG_WITH_TYPE("asm-matcher",
1499 dbgs() << "match success using target matcher\n");
1500 ++ActualIdx;
1501 continue;
1502 }
1503 // If the target matcher returned a specific error code use
1504 // that, else use the one from the generic matcher.
1505 if (TargetDiag != Match_InvalidOperand && HasRequiredFeatures)
1506 Diag = TargetDiag;
1507 }
1508 // If current formal operand wasn't matched and it is optional
1509 // then try to match next formal operand
1510 if (Diag == Match_InvalidOperand && isSubclass(Formal, OptionalMatchClass)) {
1511 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "ignoring optional operand\n");
1512 continue;
1513 }
1514 // If this operand is broken for all of the instances of this
1515 // mnemonic, keep track of it so we can report loc info.
1516 // If we already had a match that only failed due to a
1517 // target predicate, that diagnostic is preferred.
1518 if (!HadMatchOtherThanPredicate &&
1519 (it == MnemonicRange.first || ErrorInfo <= ActualIdx)) {
1520 if (HasRequiredFeatures && (ErrorInfo != ActualIdx || Diag != Match_InvalidOperand))
1521 RetCode = Diag;
1522 ErrorInfo = ActualIdx;
1523 }
1524 // Otherwise, just reject this instance of the mnemonic.
1525 OperandsValid = false;
1526 break;
1527 }
1528
1529 if (!OperandsValid) {
1530 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Opcode result: multiple "
1531 "operand mismatches, ignoring "
1532 "this opcode\n");
1533 continue;
1534 }
1535 if (!HasRequiredFeatures) {
1536 HadMatchOtherThanFeatures = true;
1537 FeatureBitset NewMissingFeatures = RequiredFeatures & ~AvailableFeatures;
1538 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Missing target features:";
1539 for (unsigned I = 0, E = NewMissingFeatures.size(); I != E; ++I)
1540 if (NewMissingFeatures[I])
1541 dbgs() << ' ' << I;
1542 dbgs() << "\n");
1543 if (NewMissingFeatures.count() <=
1544 MissingFeatures.count())
1545 MissingFeatures = NewMissingFeatures;
1546 continue;
1547 }
1548
1549 Inst.clear();
1550
1551 Inst.setOpcode(it->Opcode);
1552 // We have a potential match but have not rendered the operands.
1553 // Check the target predicate to handle any context sensitive
1554 // constraints.
1555 // For example, Ties that are referenced multiple times must be
1556 // checked here to ensure the input is the same for each match
1557 // constraints. If we leave it any later the ties will have been
1558 // canonicalized
1559 unsigned MatchResult;
1560 if ((MatchResult = checkEarlyTargetMatchPredicate(Inst, Operands)) != Match_Success) {
1561 Inst.clear();
1562 DEBUG_WITH_TYPE(
1563 "asm-matcher",
1564 dbgs() << "Early target match predicate failed with diag code "
1565 << MatchResult << "\n");
1566 RetCode = MatchResult;
1567 HadMatchOtherThanPredicate = true;
1568 continue;
1569 }
1570
1571 if (matchingInlineAsm) {
1572 convertToMapAndConstraints(it->ConvertFn, Operands);
1573 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1574 return Match_InvalidTiedOperand;
1575
1576 return Match_Success;
1577 }
1578
1579 // We have selected a definite instruction, convert the parsed
1580 // operands into the appropriate MCInst.
1581 convertToMCInst(it->ConvertFn, Inst, it->Opcode, Operands);
1582
1583 // We have a potential match. Check the target predicate to
1584 // handle any context sensitive constraints.
1585 if ((MatchResult = checkTargetMatchPredicate(Inst)) != Match_Success) {
1586 DEBUG_WITH_TYPE("asm-matcher",
1587 dbgs() << "Target match predicate failed with diag code "
1588 << MatchResult << "\n");
1589 Inst.clear();
1590 RetCode = MatchResult;
1591 HadMatchOtherThanPredicate = true;
1592 continue;
1593 }
1594
1595 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1596 return Match_InvalidTiedOperand;
1597
1598 DEBUG_WITH_TYPE(
1599 "asm-matcher",
1600 dbgs() << "Opcode result: complete match, selecting this opcode\n");
1601 return Match_Success;
1602 }
1603
1604 // Okay, we had no match. Try to return a useful error code.
1605 if (HadMatchOtherThanPredicate || !HadMatchOtherThanFeatures)
1606 return RetCode;
1607
1608 ErrorInfo = 0;
1609 return Match_MissingFeature;
1610}

1658OperandMatchResultTy AVRAsmParser::
1659MatchOperandParserImpl(OperandVector &Operands,
1660 StringRef Mnemonic,
1661 bool ParseForAllFeatures) {
1662 // Get the current feature set.
1663 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1664
1665 // Get the next operand index.
1666 unsigned NextOpNum = Operands.size() - 1;
1667 // Search the table.
1668 auto MnemonicRange =
1669 std::equal_range(std::begin(OperandMatchTable), std::end(OperandMatchTable),
1670 Mnemonic, LessOpcodeOperand());
1671
1672 if (MnemonicRange.first == MnemonicRange.second)
1673 return MatchOperand_NoMatch;
1674
1675 for (const OperandMatchEntry *it = MnemonicRange.first,
1676 *ie = MnemonicRange.second; it != ie; ++it) {
1677 // equal_range guarantees that instruction mnemonic matches.
1678 assert(Mnemonic == it->getMnemonic());
1679
1680 // check if the available features match
1681 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1682 if (!ParseForAllFeatures && (AvailableFeatures & RequiredFeatures) != RequiredFeatures)
1683 continue;
1684
1685 // check if the operand in question has a custom parser.
1686 if (!(it->OperandMask & (1 << NextOpNum)))
1687 continue;
1688
1689 // call custom parse method to handle the operand
1690 OperandMatchResultTy Result = tryCustomParseOperand(Operands, it->Class);
1691 if (Result != MatchOperand_NoMatch)
1692 return Result;
1693 }
1694
1695 // Okay, we had no match.
1696 return MatchOperand_NoMatch;
1697}

1706 const unsigned MaxEditDist = 2;
1707 std::vector<StringRef> Candidates;
1708 StringRef Prev = "";
1709
1710 // Find the appropriate table for this asm variant.
1711 const MatchEntry *Start, *End;
1712 switch (VariantID) {
1713 default: llvm_unreachable("invalid variant!");
1714 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1715 }
1716
1717 for (auto I = Start; I < End; I++) {
1718 // Ignore unsupported instructions.
1719 const FeatureBitset &RequiredFeatures = FeatureBitsets[I->RequiredFeaturesIdx];
1720 if ((FBS & RequiredFeatures) != RequiredFeatures)
1721 continue;
1722
1723 StringRef T = I->getMnemonic();
1724 // Avoid recomputing the edit distance for the same string.
1725 if (T.equals(Prev))
1726 continue;
1727
1728 Prev = T;
1729 unsigned Dist = S.edit_distance(T, false, MaxEditDist);
1730 if (Dist <= MaxEditDist)
1731 Candidates.push_back(T);
1732 }
1733
1734 if (Candidates.empty())
1735 return "";
1736
1737 std::string Res = ", did you mean: ";
1738 unsigned i = 0;
1739 for (; i < Candidates.size() - 1; i++)
1740 Res += Candidates[i].str() + ", ";
1741 return Res + Candidates[i].str() + "?";
1742}
