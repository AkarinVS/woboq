634 struct MaskRolOp {
635 LaneBitmask Mask;
636 uint8_t RotateLeft;
637 };
638 static const MaskRolOp LaneMaskComposeSequences[] = {
639 { LaneBitmask(0xFFFFFFFFFFFFFFFF), 0 }, { LaneBitmask::getNone(), 0 } // Sequence 0
640 };
641 static const MaskRolOp *const CompositeSequences[] = {
642 &LaneMaskComposeSequences[0] // to subreg_8bit

644
645LaneBitmask MSP430GenRegisterInfo::composeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
646 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
647 LaneBitmask Result;
648 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
649 LaneBitmask::Type M = LaneMask.getAsInteger() & Ops->Mask.getAsInteger();
650 if (unsigned S = Ops->RotateLeft)
651 Result |= LaneBitmask((M << S) | (M >> (LaneBitmask::BitWidth - S)));
652 else
653 Result |= LaneBitmask(M);
654 }
655 return Result;
656}
657
658LaneBitmask MSP430GenRegisterInfo::reverseComposeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
659 LaneMask &= getSubRegIndexLaneMask(IdxA);
660 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
661 LaneBitmask Result;
662 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
663 LaneBitmask::Type M = LaneMask.getAsInteger();
664 if (unsigned S = Ops->RotateLeft)
665 Result |= LaneBitmask((M >> S) | (M << (LaneBitmask::BitWidth - S)));
666 else
667 Result |= LaneBitmask(M);
668 }
669 return Result;
670}
