135// ParseSubtargetFeatures - Parses features string setting specified
136// subtarget options.
137void llvm::MSP430Subtarget::ParseSubtargetFeatures(StringRef CPU, StringRef TuneCPU, StringRef FS) {
138 LLVM_DEBUG(dbgs() << "\nFeatures:" << FS);
139 LLVM_DEBUG(dbgs() << "\nCPU:" << CPU);
140 LLVM_DEBUG(dbgs() << "\nTuneCPU:" << TuneCPU << "\n\n");
141 InitMCProcessorInfo(CPU, TuneCPU, FS);
142 const FeatureBitset &Bits = getFeatureBits();
143 if (Bits[MSP430::FeatureHWMult16] && HWMultMode < HWMult16) HWMultMode = HWMult16;
144 if (Bits[MSP430::FeatureHWMult32] && HWMultMode < HWMult32) HWMultMode = HWMult32;
145 if (Bits[MSP430::FeatureHWMultF5] && HWMultMode < HWMultF5) HWMultMode = HWMultF5;
146 if (Bits[MSP430::FeatureX]) ExtendedInsts = true;
147}
