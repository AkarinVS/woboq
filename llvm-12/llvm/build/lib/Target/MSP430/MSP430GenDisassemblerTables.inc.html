18// Helper functions for extracting fields from encoded instructions.
19// InsnType must either be integral or an APInt-like object that must:
20// * Have a static const max_size_in_bits equal to the number of bits in the
21// encoding.
22// * be default-constructible and copy-constructible
23// * be constructible from a uint64_t
24// * be constructible from an APInt (this can be private)
25// * Support getBitsSet(loBit, hiBit)
26// * be convertible to uint64_t
27// * Support the ~, &, ==, !=, and |= operators with other objects of the same type
28// * Support shift (<<, >>) with signed and unsigned integers on the RHS
29// * Support put (<<) to raw_ostream&
30template <typename InsnType>
31#if defined(_MSC_VER) && !defined(__clang__)
32__declspec(noinline)
33#endif
34static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
35 unsigned numBits, std::true_type) {
36 assert(startBit + numBits <= 64 && "Cannot support >64-bit extractions!");
37 assert(startBit + numBits <= (sizeof(InsnType) * 8) &&
38 "Instruction field out of bounds!");
39 InsnType fieldMask;
40 if (numBits == sizeof(InsnType) * 8)
41 fieldMask = (InsnType)(-1LL);
42 else
43 fieldMask = (((InsnType)1 << numBits) - 1) << startBit;
44 return (insn & fieldMask) >> startBit;
45}

47template <typename InsnType>
48static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
49 unsigned numBits, std::false_type) {
50 assert(startBit + numBits <= InsnType::max_size_in_bits && "Instruction field out of bounds!");
51 InsnType fieldMask = InsnType::getBitsSet(0, numBits);
52 return (insn >> startBit) & fieldMask;
53}

1329template <typename InsnType>
1330static DecodeStatus decodeInstruction(const uint8_t DecodeTable[], MCInst &MI,
1331 InsnType insn, uint64_t Address,
1332 const void *DisAsm,
1333 const MCSubtargetInfo &STI) {
1334 const FeatureBitset &Bits = STI.getFeatureBits();
1335
1336 const uint8_t *Ptr = DecodeTable;
1337 InsnType CurFieldValue = 0;
1338 DecodeStatus S = MCDisassembler::Success;
1339 while (true) {
1340 ptrdiff_t Loc = Ptr - DecodeTable;
1341 switch (*Ptr) {
1342 default:
1343 errs() << Loc << ": Unexpected decode table opcode!\n";
1344 return MCDisassembler::Fail;
1345 case MCD::OPC_ExtractField: {
1346 unsigned Start = *++Ptr;
1347 unsigned Len = *++Ptr;
1348 ++Ptr;
1349 CurFieldValue = fieldFromInstruction(insn, Start, Len);
1350 LLVM_DEBUG(dbgs() << Loc << ": OPC_ExtractField(" << Start << ", "
1351 << Len << "): " << CurFieldValue << "\n");
1352 break;
1353 }
1354 case MCD::OPC_FilterValue: {
1355 // Decode the field value.
1356 unsigned Len;
1357 InsnType Val = decodeULEB128(++Ptr, &Len);
1358 Ptr += Len;
1359 // NumToSkip is a plain 24-bit integer.
1360 unsigned NumToSkip = *Ptr++;
1361 NumToSkip |= (*Ptr++) << 8;
1362 NumToSkip |= (*Ptr++) << 16;
1363
1364 // Perform the filter operation.
1365 if (Val != CurFieldValue)
1366 Ptr += NumToSkip;
1367 LLVM_DEBUG(dbgs() << Loc << ": OPC_FilterValue(" << Val << ", " << NumToSkip
1368 << "): " << ((Val != CurFieldValue) ? "FAIL:" : "PASS:")
1369 << " continuing at " << (Ptr - DecodeTable) << "\n");
1370
1371 break;
1372 }
1373 case MCD::OPC_CheckField: {
1374 unsigned Start = *++Ptr;
1375 unsigned Len = *++Ptr;
1376 InsnType FieldValue = fieldFromInstruction(insn, Start, Len);
1377 // Decode the field value.
1378 InsnType ExpectedValue = decodeULEB128(++Ptr, &Len);
1379 Ptr += Len;
1380 // NumToSkip is a plain 24-bit integer.
1381 unsigned NumToSkip = *Ptr++;
1382 NumToSkip |= (*Ptr++) << 8;
1383 NumToSkip |= (*Ptr++) << 16;
1384
1385 // If the actual and expected values don't match, skip.
1386 if (ExpectedValue != FieldValue)
1387 Ptr += NumToSkip;
1388 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckField(" << Start << ", "
1389 << Len << ", " << ExpectedValue << ", " << NumToSkip
1390 << "): FieldValue = " << FieldValue << ", ExpectedValue = "
1391 << ExpectedValue << ": "
1392 << ((ExpectedValue == FieldValue) ? "PASS\n" : "FAIL\n"));
1393 break;
1394 }
1395 case MCD::OPC_CheckPredicate: {
1396 unsigned Len;
1397 // Decode the Predicate Index value.
1398 unsigned PIdx = decodeULEB128(++Ptr, &Len);
1399 Ptr += Len;
1400 // NumToSkip is a plain 24-bit integer.
1401 unsigned NumToSkip = *Ptr++;
1402 NumToSkip |= (*Ptr++) << 8;
1403 NumToSkip |= (*Ptr++) << 16;
1404 // Check the predicate.
1405 bool Pred;
1406 if (!(Pred = checkDecoderPredicate(PIdx, Bits)))
1407 Ptr += NumToSkip;
1408 (void)Pred;
1409 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckPredicate(" << PIdx << "): "
1410 << (Pred ? "PASS\n" : "FAIL\n"));
1411
1412 break;
1413 }
1414 case MCD::OPC_Decode: {
1415 unsigned Len;
1416 // Decode the Opcode value.
1417 unsigned Opc = decodeULEB128(++Ptr, &Len);
1418 Ptr += Len;
1419 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
1420 Ptr += Len;
1421
1422 MI.clear();
1423 MI.setOpcode(Opc);
1424 bool DecodeComplete;
1425 S = decodeToMCInst(S, DecodeIdx, insn, MI, Address, DisAsm, DecodeComplete);
1426 assert(DecodeComplete);
1427
1428 LLVM_DEBUG(dbgs() << Loc << ": OPC_Decode: opcode " << Opc
1429 << ", using decoder " << DecodeIdx << ": "
1430 << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
1431 return S;
1432 }
1433 case MCD::OPC_TryDecode: {
1434 unsigned Len;
1435 // Decode the Opcode value.
1436 unsigned Opc = decodeULEB128(++Ptr, &Len);
1437 Ptr += Len;
1438 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
1439 Ptr += Len;
1440 // NumToSkip is a plain 24-bit integer.
1441 unsigned NumToSkip = *Ptr++;
1442 NumToSkip |= (*Ptr++) << 8;
1443 NumToSkip |= (*Ptr++) << 16;
1444
1445 // Perform the decode operation.
1446 MCInst TmpMI;
1447 TmpMI.setOpcode(Opc);
1448 bool DecodeComplete;
1449 S = decodeToMCInst(S, DecodeIdx, insn, TmpMI, Address, DisAsm, DecodeComplete);
1450 LLVM_DEBUG(dbgs() << Loc << ": OPC_TryDecode: opcode " << Opc
1451 << ", using decoder " << DecodeIdx << ": ");
1452
1453 if (DecodeComplete) {
1454 // Decoding complete.
1455 LLVM_DEBUG(dbgs() << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
1456 MI = TmpMI;
1457 return S;
1458 } else {
1459 assert(S == MCDisassembler::Fail);
1460 // If the decoding was incomplete, skip.
1461 Ptr += NumToSkip;
1462 LLVM_DEBUG(dbgs() << "FAIL: continuing at " << (Ptr - DecodeTable) << "\n");
1463 // Reset decode status. This also drops a SoftFail status that could be
1464 // set before the decode attempt.
1465 S = MCDisassembler::Success;
1466 }
1467 break;
1468 }
1469 case MCD::OPC_SoftFail: {
1470 // Decode the mask values.
1471 unsigned Len;
1472 InsnType PositiveMask = decodeULEB128(++Ptr, &Len);
1473 Ptr += Len;
1474 InsnType NegativeMask = decodeULEB128(Ptr, &Len);
1475 Ptr += Len;
1476 bool Fail = (insn & PositiveMask) || (~insn & NegativeMask);
1477 if (Fail)
1478 S = MCDisassembler::SoftFail;
1479 LLVM_DEBUG(dbgs() << Loc << ": OPC_SoftFail: " << (Fail ? "FAIL\n" : "PASS\n"));
1480 break;
1481 }
1482 case MCD::OPC_Fail: {
1483 LLVM_DEBUG(dbgs() << Loc << ": OPC_Fail\n");
1484 return MCDisassembler::Fail;
1485 }
1486 }
1487 }
1488 llvm_unreachable("bogosity detected in disassembler state machine!");
1489}
