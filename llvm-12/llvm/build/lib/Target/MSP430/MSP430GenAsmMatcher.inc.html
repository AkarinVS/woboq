326void MSP430AsmParser::
327convertToMCInst(unsigned Kind, MCInst &Inst, unsigned Opcode,
328 const OperandVector &Operands) {
329 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
330 const uint8_t *Converter = ConversionTable[Kind];
331 unsigned OpIdx;
332 Inst.setOpcode(Opcode);
333 for (const uint8_t *p = Converter; *p; p += 2) {
334 OpIdx = *(p + 1);
335 switch (*p) {
336 default: llvm_unreachable("invalid conversion entry!");
337 case CVT_Reg:
338 static_cast<MSP430Operand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
339 break;
340 case CVT_Tied: {
341 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
342 std::begin(TiedAsmOperandTable)) &&
343 "Tied operand not found");
344 unsigned TiedResOpnd = TiedAsmOperandTable[OpIdx][0];
345 if (TiedResOpnd != (uint8_t)-1)
346 Inst.addOperand(Inst.getOperand(TiedResOpnd));
347 break;
348 }
349 case CVT_95_Reg:
350 static_cast<MSP430Operand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
351 break;
352 case CVT_imm_95_0:
353 Inst.addOperand(MCOperand::createImm(0));
354 break;
355 case CVT_95_addMemOperands:
356 static_cast<MSP430Operand &>(*Operands[OpIdx]).addMemOperands(Inst, 2);
357 break;
358 case CVT_95_addImmOperands:
359 static_cast<MSP430Operand &>(*Operands[OpIdx]).addImmOperands(Inst, 1);
360 break;
361 case CVT_95_addRegOperands:
362 static_cast<MSP430Operand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
363 break;
364 case CVT_regSR:
365 Inst.addOperand(MCOperand::createReg(MSP430::SR));
366 break;
367 case CVT_imm_95_1:
368 Inst.addOperand(MCOperand::createImm(1));
369 break;
370 case CVT_imm_95_4:
371 Inst.addOperand(MCOperand::createImm(4));
372 break;
373 case CVT_imm_95_2:
374 Inst.addOperand(MCOperand::createImm(2));
375 break;
376 case CVT_imm_95_8:
377 Inst.addOperand(MCOperand::createImm(8));
378 break;
379 case CVT_imm_95__MINUS_1:
380 Inst.addOperand(MCOperand::createImm(-1));
381 break;
382 case CVT_regCG:
383 Inst.addOperand(MCOperand::createReg(MSP430::CG));
384 break;
385 }

507static unsigned validateOperandClass(MCParsedAsmOperand &GOp, MatchClassKind Kind) {
508 MSP430Operand &Operand = (MSP430Operand &)GOp;
509 if (Kind == InvalidMatchClass)
510 return MCTargetAsmParser::Match_InvalidOperand;
511
512 if (Operand.isToken() && Kind <= MCK_LAST_TOKEN)
513 return isSubclass(matchTokenString(Operand.getToken()), Kind) ?
514 MCTargetAsmParser::Match_Success :
515 MCTargetAsmParser::Match_InvalidOperand;
516
517 switch (Kind) {
518 default: break;
519 // 'CGImm' class
520 case MCK_CGImm: {
521 DiagnosticPredicate DP(Operand.isCGImm());
522 if (DP.isMatch())
523 return MCTargetAsmParser::Match_Success;
524 break;
525 }
526 // 'Imm' class
527 case MCK_Imm: {
528 DiagnosticPredicate DP(Operand.isImm());
529 if (DP.isMatch())
530 return MCTargetAsmParser::Match_Success;
531 break;
532 }
533 // 'IndReg' class
534 case MCK_IndReg: {
535 DiagnosticPredicate DP(Operand.isIndReg());
536 if (DP.isMatch())
537 return MCTargetAsmParser::Match_Success;
538 break;
539 }
540 // 'Mem' class
541 case MCK_Mem: {
542 DiagnosticPredicate DP(Operand.isMem());
543 if (DP.isMatch())
544 return MCTargetAsmParser::Match_Success;
545 break;
546 }
547 // 'PostIndReg' class
548 case MCK_PostIndReg: {
549 DiagnosticPredicate DP(Operand.isPostIndReg());
550 if (DP.isMatch())
551 return MCTargetAsmParser::Match_Success;
552 break;
553 }
554 } // end switch (Kind)
555
556 if (Operand.isReg()) {
557 MatchClassKind OpKind;
558 switch (Operand.getReg()) {
559 default: OpKind = InvalidMatchClass; break;
560 case MSP430::PCB: OpKind = MCK_GR8; break;
561 case MSP430::SPB: OpKind = MCK_GR8; break;
562 case MSP430::SRB: OpKind = MCK_GR8; break;
563 case MSP430::CGB: OpKind = MCK_GR8; break;
564 case MSP430::R4B: OpKind = MCK_GR8; break;
565 case MSP430::R5B: OpKind = MCK_GR8; break;
566 case MSP430::R6B: OpKind = MCK_GR8; break;
567 case MSP430::R7B: OpKind = MCK_GR8; break;
568 case MSP430::R8B: OpKind = MCK_GR8; break;

627static bool checkAsmTiedOperandConstraints(const MSP430AsmParser&AsmParser,
628 unsigned Kind,
629 const OperandVector &Operands,
630 uint64_t &ErrorInfo) {
631 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
632 const uint8_t *Converter = ConversionTable[Kind];
633 for (const uint8_t *p = Converter; *p; p += 2) {
634 switch (*p) {
635 case CVT_Tied: {
636 unsigned OpIdx = *(p + 1);
637 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
638 std::begin(TiedAsmOperandTable)) &&
639 "Tied operand not found");
640 unsigned OpndNum1 = TiedAsmOperandTable[OpIdx][1];
641 unsigned OpndNum2 = TiedAsmOperandTable[OpIdx][2];
642 if (OpndNum1 != OpndNum2) {
643 auto &SrcOp1 = Operands[OpndNum1];
644 auto &SrcOp2 = Operands[OpndNum2];
645 if (SrcOp1->isReg() && SrcOp2->isReg()) {
646 if (!AsmParser.regsEqual(*SrcOp1, *SrcOp2)) {
647 ErrorInfo = OpndNum2;
648 return false;
649 }
650 }
651 }
652 break;
653 }
654 default:
655 break;
656 }
657 }
658 return true;
659}

1097
1098unsigned MSP430AsmParser::
1099MatchInstructionImpl(const OperandVector &Operands,
1100 MCInst &Inst,
1101 uint64_t &ErrorInfo,
1102 FeatureBitset &MissingFeatures,
1103 bool matchingInlineAsm, unsigned VariantID) {
1104 // Eliminate obvious mismatches.
1105 if (Operands.size() > 3) {
1106 ErrorInfo = 3;
1107 return Match_InvalidOperand;
1108 }
1109
1110 // Get the current feature set.
1111 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1112
1113 // Get the instruction mnemonic, which is the first token.
1114 StringRef Mnemonic = ((MSP430Operand &)*Operands[0]).getToken();
1115
1116 // Some state to try to produce better error messages.
1117 bool HadMatchOtherThanFeatures = false;
1118 bool HadMatchOtherThanPredicate = false;
1119 unsigned RetCode = Match_InvalidOperand;
1120 MissingFeatures.set();
1121 // Set ErrorInfo to the operand that mismatches if it is
1122 // wrong for all instances of the instruction.
1123 ErrorInfo = ~0ULL;
1124 // Find the appropriate table for this asm variant.
1125 const MatchEntry *Start, *End;
1126 switch (VariantID) {
1127 default: llvm_unreachable("invalid variant!");
1128 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1129 }
1130 // Search the table.
1131 auto MnemonicRange = std::equal_range(Start, End, Mnemonic, LessOpcode());
1132
1133 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "AsmMatcher: found " <<
1134 std::distance(MnemonicRange.first, MnemonicRange.second) <<
1135 " encodings with mnemonic '" << Mnemonic << "'\n");
1136
1137 // Return a more specific error code if no mnemonics match.
1138 if (MnemonicRange.first == MnemonicRange.second)
1139 return Match_MnemonicFail;
1140
1141 for (const MatchEntry *it = MnemonicRange.first, *ie = MnemonicRange.second;
1142 it != ie; ++it) {
1143 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1144 bool HasRequiredFeatures =
1145 (AvailableFeatures & RequiredFeatures) == RequiredFeatures;
1146 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Trying to match opcode "
1147 << MII.getName(it->Opcode) << "\n");
1148 // equal_range guarantees that instruction mnemonic matches.
1149 assert(Mnemonic == it->getMnemonic());
1150 bool OperandsValid = true;
1151 for (unsigned FormalIdx = 0, ActualIdx = 1; FormalIdx != 2; ++FormalIdx) {
1152 auto Formal = static_cast<MatchClassKind>(it->Classes[FormalIdx]);
1153 DEBUG_WITH_TYPE("asm-matcher",
1154 dbgs() << " Matching formal operand class " << getMatchClassName(Formal)
1155 << " against actual operand at index " << ActualIdx);
1156 if (ActualIdx < Operands.size())
1157 DEBUG_WITH_TYPE("asm-matcher", dbgs() << " (";
1158 Operands[ActualIdx]->print(dbgs()); dbgs() << "): ");
1159 else
1160 DEBUG_WITH_TYPE("asm-matcher", dbgs() << ": ");
1161 if (ActualIdx >= Operands.size()) {
1162 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "actual operand index out of range ");
1163 OperandsValid = (Formal == InvalidMatchClass) || isSubclass(Formal, OptionalMatchClass);
1164 if (!OperandsValid) ErrorInfo = ActualIdx;
1165 break;
1166 }
1167 MCParsedAsmOperand &Actual = *Operands[ActualIdx];
1168 unsigned Diag = validateOperandClass(Actual, Formal);
1169 if (Diag == Match_Success) {
1170 DEBUG_WITH_TYPE("asm-matcher",
1171 dbgs() << "match success using generic matcher\n");
1172 ++ActualIdx;
1173 continue;
1174 }
1175 // If the generic handler indicates an invalid operand
1176 // failure, check for a special case.
1177 if (Diag != Match_Success) {
1178 unsigned TargetDiag = validateTargetOperandClass(Actual, Formal);
1179 if (TargetDiag == Match_Success) {
1180 DEBUG_WITH_TYPE("asm-matcher",
1181 dbgs() << "match success using target matcher\n");
1182 ++ActualIdx;
1183 continue;
1184 }
1185 // If the target matcher returned a specific error code use
1186 // that, else use the one from the generic matcher.
1187 if (TargetDiag != Match_InvalidOperand && HasRequiredFeatures)
1188 Diag = TargetDiag;
1189 }
1190 // If current formal operand wasn't matched and it is optional
1191 // then try to match next formal operand
1192 if (Diag == Match_InvalidOperand && isSubclass(Formal, OptionalMatchClass)) {
1193 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "ignoring optional operand\n");
1194 continue;
1195 }
1196 // If this operand is broken for all of the instances of this
1197 // mnemonic, keep track of it so we can report loc info.
1198 // If we already had a match that only failed due to a
1199 // target predicate, that diagnostic is preferred.
1200 if (!HadMatchOtherThanPredicate &&
1201 (it == MnemonicRange.first || ErrorInfo <= ActualIdx)) {
1202 if (HasRequiredFeatures && (ErrorInfo != ActualIdx || Diag != Match_InvalidOperand))
1203 RetCode = Diag;
1204 ErrorInfo = ActualIdx;
1205 }
1206 // Otherwise, just reject this instance of the mnemonic.
1207 OperandsValid = false;
1208 break;
1209 }
1210
1211 if (!OperandsValid) {
1212 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Opcode result: multiple "
1213 "operand mismatches, ignoring "
1214 "this opcode\n");
1215 continue;
1216 }
1217 if (!HasRequiredFeatures) {
1218 HadMatchOtherThanFeatures = true;
1219 FeatureBitset NewMissingFeatures = RequiredFeatures & ~AvailableFeatures;
1220 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Missing target features:";
1221 for (unsigned I = 0, E = NewMissingFeatures.size(); I != E; ++I)
1222 if (NewMissingFeatures[I])
1223 dbgs() << ' ' << I;
1224 dbgs() << "\n");
1225 if (NewMissingFeatures.count() <=
1226 MissingFeatures.count())
1227 MissingFeatures = NewMissingFeatures;
1228 continue;
1229 }
1230
1231 Inst.clear();
1232
1233 Inst.setOpcode(it->Opcode);
1234 // We have a potential match but have not rendered the operands.
1235 // Check the target predicate to handle any context sensitive
1236 // constraints.
1237 // For example, Ties that are referenced multiple times must be
1238 // checked here to ensure the input is the same for each match
1239 // constraints. If we leave it any later the ties will have been
1240 // canonicalized
1241 unsigned MatchResult;
1242 if ((MatchResult = checkEarlyTargetMatchPredicate(Inst, Operands)) != Match_Success) {
1243 Inst.clear();
1244 DEBUG_WITH_TYPE(
1245 "asm-matcher",
1246 dbgs() << "Early target match predicate failed with diag code "
1247 << MatchResult << "\n");
1248 RetCode = MatchResult;
1249 HadMatchOtherThanPredicate = true;
1250 continue;
1251 }
1252
1253 if (matchingInlineAsm) {
1254 convertToMapAndConstraints(it->ConvertFn, Operands);
1255 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1256 return Match_InvalidTiedOperand;
1257
1258 return Match_Success;
1259 }
1260
1261 // We have selected a definite instruction, convert the parsed
1262 // operands into the appropriate MCInst.
1263 convertToMCInst(it->ConvertFn, Inst, it->Opcode, Operands);
1264
1265 // We have a potential match. Check the target predicate to
1266 // handle any context sensitive constraints.
1267 if ((MatchResult = checkTargetMatchPredicate(Inst)) != Match_Success) {
1268 DEBUG_WITH_TYPE("asm-matcher",
1269 dbgs() << "Target match predicate failed with diag code "
1270 << MatchResult << "\n");
1271 Inst.clear();
1272 RetCode = MatchResult;
1273 HadMatchOtherThanPredicate = true;
1274 continue;
1275 }
1276
1277 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1278 return Match_InvalidTiedOperand;
1279
1280 DEBUG_WITH_TYPE(
1281 "asm-matcher",
1282 dbgs() << "Opcode result: complete match, selecting this opcode\n");
1283 return Match_Success;
1284 }
1285
1286 // Okay, we had no match. Try to return a useful error code.
1287 if (HadMatchOtherThanPredicate || !HadMatchOtherThanFeatures)
1288 return RetCode;
1289
1290 ErrorInfo = 0;
1291 return Match_MissingFeature;
1292}

1302 std::vector<StringRef> Candidates;
1303 StringRef Prev = "";
1304
1305 // Find the appropriate table for this asm variant.
1306 const MatchEntry *Start, *End;

1312 for (auto I = Start; I < End; I++) {
1313 // Ignore unsupported instructions.
1314 const FeatureBitset &RequiredFeatures = FeatureBitsets[I->RequiredFeaturesIdx];
1315 if ((FBS & RequiredFeatures) != RequiredFeatures)
1316 continue;
1317
1318 StringRef T = I->getMnemonic();
1319 // Avoid recomputing the edit distance for the same string.
1320 if (T.equals(Prev))
1321 continue;
1322
1323 Prev = T;
1324 unsigned Dist = S.edit_distance(T, false, MaxEditDist);
1325 if (Dist <= MaxEditDist)
1326 Candidates.push_back(T);
1327 }
1328
1329 if (Candidates.empty())
1330 return "";
1331
1332 std::string Res = ", did you mean: ";
1333 unsigned i = 0;
1334 for (; i < Candidates.size() - 1; i++)
1335 Res += Candidates[i].str() + ", ";
1336 return Res + Candidates[i].str() + "?";
1337}
