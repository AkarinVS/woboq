664/// printInstruction - This method is automatically generated by tablegen
665/// from the instruction set description.
666void MSP430InstPrinter::printInstruction(const MCInst *MI, uint64_t Address, raw_ostream &O) {
667 O << "\t";
668
669 auto MnemonicInfo = getMnemonic(MI);
670
671 O << MnemonicInfo.first;
672
673 uint32_t Bits = MnemonicInfo.second;
674 assert(Bits != 0 && "Cannot print this instruction.");
675
676 // Fragment 0 encoded into 4 bits for 16 unique commands.
677 switch ((Bits >> 10) & 15) {
678 default: llvm_unreachable("Invalid command number.");
679 case 0:
680 // DBG_VALUE, DBG_INSTR_REF, DBG_LABEL, BUNDLE, LIFETIME_START, LIFETIME_...
681 return;
682 break;
683 case 1:
684 // ADD16mc, ADD16mi, ADD16mr, ADD16rc, ADD16ri, ADD16rr, ADD8mc, ADD8mi, ...
685 printOperand(MI, 2, O);
686 O << ", ";
687 break;
688 case 2:
689 // ADD16mm, ADD16rm, ADD8mm, ADD8rm, ADDC16mm, ADDC16rm, ADDC8mm, ADDC8rm...
690 printSrcMemOperand(MI, 2, O);
691 O << ", ";
692 break;
693 case 3:
694 // ADD16mn, ADD16rn, ADD8mn, ADD8rn, ADDC16mn, ADDC16rn, ADDC8mn, ADDC8rn...
695 printIndRegOperand(MI, 2, O);
696 O << ", ";
697 break;
698 case 4:
699 // ADD16mp, ADD8mp, ADDC16mp, ADDC8mp, AND16mp, AND8mp, BIC16mp, BIC8mp, ...
700 printPostIndRegOperand(MI, 2, O);
701 O << ", ";
702 break;
703 case 5:
704 // ADD16rp, ADD8rp, ADDC16rp, ADDC8rp, AND16rp, AND8rp, BIC16rp, BIC8rp, ...
705 printPostIndRegOperand(MI, 3, O);
706 O << ", ";
707 printOperand(MI, 0, O);
708 return;
709 break;
710 case 6:
711 // ADJCALLSTACKDOWN, ADJCALLSTACKUP, Bi, Br, CALLi, CALLr, POP16r, PUSH16...
712 printOperand(MI, 0, O);
713 break;
714 case 7:
715 // BIT16rc, BIT16ri, BIT16rr, BIT8rc, BIT8ri, BIT8rr, CMP16rc, CMP16ri, C...
716 printOperand(MI, 1, O);
717 O << ", ";
718 printOperand(MI, 0, O);
719 return;
720 break;
721 case 8:
722 // BIT16rm, BIT8rm, CMP16rm, CMP8rm, MOV16rm, MOV8rm, MOVZX16rm8
723 printSrcMemOperand(MI, 1, O);
724 O << ", ";
725 printOperand(MI, 0, O);
726 return;
727 break;
728 case 9:
729 // BIT16rn, BIT8rn, CMP16rn, CMP8rn, MOV16rn, MOV8rn
730 printIndRegOperand(MI, 1, O);
731 O << ", ";
732 printOperand(MI, 0, O);
733 return;
734 break;
735 case 10:
736 // BIT16rp, BIT8rp, CMP16rp, CMP8rp
737 printPostIndRegOperand(MI, 1, O);
738 O << ", ";
739 printOperand(MI, 0, O);
740 return;
741 break;
742 case 11:
743 // Bm, CALLm, RRA16m, RRA8m, RRC16m, RRC8m, SEXT16m, SWPB16m
744 printSrcMemOperand(MI, 0, O);
745 return;
746 break;
747 case 12:
748 // CALLn, RRA16n, RRA8n, RRC16n, RRC8n, SEXT16n, SWPB16n
749 printIndRegOperand(MI, 0, O);
750 return;
751 break;
752 case 13:
753 // CALLp, RRA16p, RRA8p, RRC16p, RRC8p, SEXT16p, SWPB16p
754 printPostIndRegOperand(MI, 0, O);
755 return;
756 break;
757 case 14:
758 // JCC
759 printCCOperand(MI, 1, O);
760 O << "\t";
761 printPCRelImmOperand(MI, 0, O);
762 return;
763 break;
764 case 15:
765 // JMP
766 printPCRelImmOperand(MI, 0, O);
767 return;
768 break;
769 }
770
771
772 // Fragment 1 encoded into 2 bits for 4 unique commands.
773 switch ((Bits >> 14) & 3) {
774 default: llvm_unreachable("Invalid command number.");
775 case 0:
776 // ADD16mc, ADD16mi, ADD16mm, ADD16mn, ADD16mp, ADD16mr, ADD8mc, ADD8mi, ...
777 printSrcMemOperand(MI, 0, O);
778 return;
779 break;
780 case 1:
781 // ADD16rc, ADD16ri, ADD16rm, ADD16rn, ADD16rr, ADD8rc, ADD8ri, ADD8rm, A...
782 printOperand(MI, 0, O);
783 return;
784 break;
785 case 2:
786 // ADJCALLSTACKDOWN, ADJCALLSTACKUP
787 O << ' ';
788 printOperand(MI, 1, O);
789 return;
790 break;
791 case 3:
792 // Bi, Br, CALLi, CALLr, POP16r, PUSH16c, PUSH16i, PUSH16r, PUSH8r, RRA16...
793 return;
794 break;
795 }
796
797}
