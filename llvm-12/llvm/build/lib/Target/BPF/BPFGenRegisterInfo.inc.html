561 struct MaskRolOp {
562 LaneBitmask Mask;
563 uint8_t RotateLeft;
564 };
565 static const MaskRolOp LaneMaskComposeSequences[] = {
566 { LaneBitmask(0xFFFFFFFFFFFFFFFF), 0 }, { LaneBitmask::getNone(), 0 } // Sequence 0
567 };
568 static const MaskRolOp *const CompositeSequences[] = {
569 &LaneMaskComposeSequences[0] // to sub_32
570 };
571
572LaneBitmask BPFGenRegisterInfo::composeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
573 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
574 LaneBitmask Result;
575 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
576 LaneBitmask::Type M = LaneMask.getAsInteger() & Ops->Mask.getAsInteger();
577 if (unsigned S = Ops->RotateLeft)
578 Result |= LaneBitmask((M << S) | (M >> (LaneBitmask::BitWidth - S)));
579 else
580 Result |= LaneBitmask(M);
581 }
582 return Result;
583}
584
585LaneBitmask BPFGenRegisterInfo::reverseComposeSubRegIndexLaneMaskImpl(unsigned IdxA, LaneBitmask LaneMask) const {
586 LaneMask &= getSubRegIndexLaneMask(IdxA);
587 --IdxA; assert(IdxA < 1 && "Subregister index out of bounds");
588 LaneBitmask Result;
589 for (const MaskRolOp *Ops = CompositeSequences[IdxA]; Ops->Mask.any(); ++Ops) {
590 LaneBitmask::Type M = LaneMask.getAsInteger();
591 if (unsigned S = Ops->RotateLeft)
592 Result |= LaneBitmask((M >> S) | (M << (LaneBitmask::BitWidth - S)));
593 else
594 Result |= LaneBitmask(M);
595 }
596 return Result;
597}
