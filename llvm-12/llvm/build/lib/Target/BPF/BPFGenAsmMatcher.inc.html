278void BPFAsmParser::
279convertToMCInst(unsigned Kind, MCInst &Inst, unsigned Opcode,
280 const OperandVector &Operands) {
281 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
282 const uint8_t *Converter = ConversionTable[Kind];
283 unsigned OpIdx;
284 Inst.setOpcode(Opcode);
285 for (const uint8_t *p = Converter; *p; p += 2) {
286 OpIdx = *(p + 1);
287 switch (*p) {
288 default: llvm_unreachable("invalid conversion entry!");
289 case CVT_Reg:
290 static_cast<BPFOperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
291 break;
292 case CVT_Tied: {
293 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
294 std::begin(TiedAsmOperandTable)) &&
295 "Tied operand not found");
296 unsigned TiedResOpnd = TiedAsmOperandTable[OpIdx][0];
297 if (TiedResOpnd != (uint8_t)-1)
298 Inst.addOperand(Inst.getOperand(TiedResOpnd));
299 break;
300 }
301 case CVT_95_Reg:
302 static_cast<BPFOperand &>(*Operands[OpIdx]).addRegOperands(Inst, 1);
303 break;
304 case CVT_95_addImmOperands:
305 static_cast<BPFOperand &>(*Operands[OpIdx]).addImmOperands(Inst, 1);
306 break;
307 case CVT_imm_95_0:
308 Inst.addOperand(MCOperand::createImm(0));
309 break;
310 }
311 }

314void BPFAsmParser::
315convertToMapAndConstraints(unsigned Kind,
316 const OperandVector &Operands) {
317 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
318 unsigned NumMCOperands = 0;
319 const uint8_t *Converter = ConversionTable[Kind];
320 for (const uint8_t *p = Converter; *p; p += 2) {
321 switch (*p) {
322 default: llvm_unreachable("invalid conversion entry!");
323 case CVT_Reg:
324 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
325 Operands[*(p + 1)]->setConstraint("r");
326 ++NumMCOperands;
327 break;
328 case CVT_Tied:
329 ++NumMCOperands;
330 break;
331 case CVT_95_Reg:
332 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
333 Operands[*(p + 1)]->setConstraint("r");
334 NumMCOperands += 1;
335 break;
336 case CVT_95_addImmOperands:
337 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
338 Operands[*(p + 1)]->setConstraint("m");
339 NumMCOperands += 1;
340 break;
341 case CVT_imm_95_0:
342 Operands[*(p + 1)]->setMCOperandNum(NumMCOperands);
343 Operands[*(p + 1)]->setConstraint("");
344 ++NumMCOperands;
345 break;
346 }
347 }

770static bool checkAsmTiedOperandConstraints(const BPFAsmParser&AsmParser,
771 unsigned Kind,
772 const OperandVector &Operands,
773 uint64_t &ErrorInfo) {
774 assert(Kind < CVT_NUM_SIGNATURES && "Invalid signature!");
775 const uint8_t *Converter = ConversionTable[Kind];
776 for (const uint8_t *p = Converter; *p; p += 2) {
777 switch (*p) {
778 case CVT_Tied: {
779 unsigned OpIdx = *(p + 1);
780 assert(OpIdx < (size_t)(std::end(TiedAsmOperandTable) -
781 std::begin(TiedAsmOperandTable)) &&
782 "Tied operand not found");
783 unsigned OpndNum1 = TiedAsmOperandTable[OpIdx][1];
784 unsigned OpndNum2 = TiedAsmOperandTable[OpIdx][2];
785 if (OpndNum1 != OpndNum2) {
786 auto &SrcOp1 = Operands[OpndNum1];
787 auto &SrcOp2 = Operands[OpndNum2];
788 if (SrcOp1->isReg() && SrcOp2->isReg()) {
789 if (!AsmParser.regsEqual(*SrcOp1, *SrcOp2)) {
790 ErrorInfo = OpndNum2;
791 return false;
792 }
793 }
794 }
795 break;
796 }
797 default:
798 break;
799 }
800 }
801 return true;

990unsigned BPFAsmParser::
991MatchInstructionImpl(const OperandVector &Operands,
992 MCInst &Inst,
993 uint64_t &ErrorInfo,
994 FeatureBitset &MissingFeatures,
995 bool matchingInlineAsm, unsigned VariantID) {
996 // Eliminate obvious mismatches.
997 if (Operands.size() > 14) {
998 ErrorInfo = 14;
999 return Match_InvalidOperand;
1000 }
1001
1002 // Get the current feature set.
1003 const FeatureBitset &AvailableFeatures = getAvailableFeatures();
1004
1005 // Get the instruction mnemonic, which is the first token.
1006 StringRef Mnemonic;
1007 if (Operands[0]->isToken())
1008 Mnemonic = ((BPFOperand &)*Operands[0]).getToken();
1009
1010 // Some state to try to produce better error messages.
1011 bool HadMatchOtherThanFeatures = false;
1012 bool HadMatchOtherThanPredicate = false;
1013 unsigned RetCode = Match_InvalidOperand;
1014 MissingFeatures.set();
1015 // Set ErrorInfo to the operand that mismatches if it is
1016 // wrong for all instances of the instruction.
1017 ErrorInfo = ~0ULL;
1018 // Find the appropriate table for this asm variant.
1019 const MatchEntry *Start, *End;
1020 switch (VariantID) {
1021 default: llvm_unreachable("invalid variant!");
1022 case 0: Start = std::begin(MatchTable0); End = std::end(MatchTable0); break;
1023 }
1024 // Search the table.
1025 auto MnemonicRange = std::make_pair(Start, End);
1026 unsigned SIndex = Mnemonic.empty() ? 0 : 1;
1027 if (!Mnemonic.empty())
1028 MnemonicRange = std::equal_range(Start, End, Mnemonic.lower(), LessOpcode());
1029
1030 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "AsmMatcher: found " <<
1031 std::distance(MnemonicRange.first, MnemonicRange.second) <<
1032 " encodings with mnemonic '" << Mnemonic << "'\n");
1033
1034 // Return a more specific error code if no mnemonics match.
1035 if (MnemonicRange.first == MnemonicRange.second)
1036 return Match_MnemonicFail;
1037
1038 for (const MatchEntry *it = MnemonicRange.first, *ie = MnemonicRange.second;
1039 it != ie; ++it) {
1040 const FeatureBitset &RequiredFeatures = FeatureBitsets[it->RequiredFeaturesIdx];
1041 bool HasRequiredFeatures =
1042 (AvailableFeatures & RequiredFeatures) == RequiredFeatures;
1043 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Trying to match opcode "
1044 << MII.getName(it->Opcode) << "\n");
1045 bool OperandsValid = true;
1046 for (unsigned FormalIdx = SIndex, ActualIdx = SIndex; FormalIdx != 14; ++FormalIdx) {
1047 auto Formal = static_cast<MatchClassKind>(it->Classes[FormalIdx]);
1048 DEBUG_WITH_TYPE("asm-matcher",
1049 dbgs() << " Matching formal operand class " << getMatchClassName(Formal)
1050 << " against actual operand at index " << ActualIdx);
1051 if (ActualIdx < Operands.size())
1052 DEBUG_WITH_TYPE("asm-matcher", dbgs() << " (";
1053 Operands[ActualIdx]->print(dbgs()); dbgs() << "): ");
1054 else
1055 DEBUG_WITH_TYPE("asm-matcher", dbgs() << ": ");
1056 if (ActualIdx >= Operands.size()) {
1057 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "actual operand index out of range ");
1058 OperandsValid = (Formal == InvalidMatchClass) || isSubclass(Formal, OptionalMatchClass);
1059 if (!OperandsValid) ErrorInfo = ActualIdx;
1060 break;
1061 }
1062 MCParsedAsmOperand &Actual = *Operands[ActualIdx];
1063 unsigned Diag = validateOperandClass(Actual, Formal);
1064 if (Diag == Match_Success) {
1065 DEBUG_WITH_TYPE("asm-matcher",
1066 dbgs() << "match success using generic matcher\n");
1067 ++ActualIdx;
1068 continue;
1069 }
1070 // If the generic handler indicates an invalid operand
1071 // failure, check for a special case.
1072 if (Diag != Match_Success) {
1073 unsigned TargetDiag = validateTargetOperandClass(Actual, Formal);
1074 if (TargetDiag == Match_Success) {
1075 DEBUG_WITH_TYPE("asm-matcher",
1076 dbgs() << "match success using target matcher\n");
1077 ++ActualIdx;
1078 continue;
1079 }
1080 // If the target matcher returned a specific error code use
1081 // that, else use the one from the generic matcher.
1082 if (TargetDiag != Match_InvalidOperand && HasRequiredFeatures)
1083 Diag = TargetDiag;
1084 }
1085 // If current formal operand wasn't matched and it is optional
1086 // then try to match next formal operand
1087 if (Diag == Match_InvalidOperand && isSubclass(Formal, OptionalMatchClass)) {
1088 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "ignoring optional operand\n");
1089 continue;
1090 }
1091 // If this operand is broken for all of the instances of this
1092 // mnemonic, keep track of it so we can report loc info.
1093 // If we already had a match that only failed due to a
1094 // target predicate, that diagnostic is preferred.
1095 if (!HadMatchOtherThanPredicate &&
1096 (it == MnemonicRange.first || ErrorInfo <= ActualIdx)) {
1097 if (HasRequiredFeatures && (ErrorInfo != ActualIdx || Diag != Match_InvalidOperand))
1098 RetCode = Diag;
1099 ErrorInfo = ActualIdx;
1100 }
1101 // Otherwise, just reject this instance of the mnemonic.
1102 OperandsValid = false;
1103 break;
1104 }
1105
1106 if (!OperandsValid) {
1107 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Opcode result: multiple "
1108 "operand mismatches, ignoring "
1109 "this opcode\n");
1110 continue;
1111 }
1112 if (!HasRequiredFeatures) {
1113 HadMatchOtherThanFeatures = true;
1114 FeatureBitset NewMissingFeatures = RequiredFeatures & ~AvailableFeatures;
1115 DEBUG_WITH_TYPE("asm-matcher", dbgs() << "Missing target features:";
1116 for (unsigned I = 0, E = NewMissingFeatures.size(); I != E; ++I)
1117 if (NewMissingFeatures[I])
1118 dbgs() << ' ' << I;
1119 dbgs() << "\n");
1120 if (NewMissingFeatures.count() <=
1121 MissingFeatures.count())
1122 MissingFeatures = NewMissingFeatures;
1123 continue;
1124 }
1125
1126 Inst.clear();
1127
1128 Inst.setOpcode(it->Opcode);
1129 // We have a potential match but have not rendered the operands.
1130 // Check the target predicate to handle any context sensitive
1131 // constraints.
1132 // For example, Ties that are referenced multiple times must be
1133 // checked here to ensure the input is the same for each match
1134 // constraints. If we leave it any later the ties will have been
1135 // canonicalized
1136 unsigned MatchResult;
1137 if ((MatchResult = checkEarlyTargetMatchPredicate(Inst, Operands)) != Match_Success) {
1138 Inst.clear();
1139 DEBUG_WITH_TYPE(
1140 "asm-matcher",
1141 dbgs() << "Early target match predicate failed with diag code "
1142 << MatchResult << "\n");
1143 RetCode = MatchResult;
1144 HadMatchOtherThanPredicate = true;
1145 continue;
1146 }
1147
1148 if (matchingInlineAsm) {
1149 convertToMapAndConstraints(it->ConvertFn, Operands);
1150 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1151 return Match_InvalidTiedOperand;
1152
1153 return Match_Success;
1154 }
1155
1156 // We have selected a definite instruction, convert the parsed
1157 // operands into the appropriate MCInst.
1158 convertToMCInst(it->ConvertFn, Inst, it->Opcode, Operands);
1159
1160 // We have a potential match. Check the target predicate to
1161 // handle any context sensitive constraints.
1162 if ((MatchResult = checkTargetMatchPredicate(Inst)) != Match_Success) {
1163 DEBUG_WITH_TYPE("asm-matcher",
1164 dbgs() << "Target match predicate failed with diag code "
1165 << MatchResult << "\n");
1166 Inst.clear();
1167 RetCode = MatchResult;
1168 HadMatchOtherThanPredicate = true;
1169 continue;
1170 }
1171
1172 if (!checkAsmTiedOperandConstraints(*this, it->ConvertFn, Operands, ErrorInfo))
1173 return Match_InvalidTiedOperand;
1174
1175 DEBUG_WITH_TYPE(
1176 "asm-matcher",
1177 dbgs() << "Opcode result: complete match, selecting this opcode\n");
1178 return Match_Success;
1179 }
1180
1181 // Okay, we had no match. Try to return a useful error code.
1182 if (HadMatchOtherThanPredicate || !HadMatchOtherThanFeatures)
1183 return RetCode;
1184
1185 ErrorInfo = 0;
1186 return Match_MissingFeature;
1187}
