18// Helper functions for extracting fields from encoded instructions.
19// InsnType must either be integral or an APInt-like object that must:
20// * Have a static const max_size_in_bits equal to the number of bits in the
21// encoding.
22// * be default-constructible and copy-constructible
23// * be constructible from a uint64_t
24// * be constructible from an APInt (this can be private)
25// * Support getBitsSet(loBit, hiBit)
26// * be convertible to uint64_t
27// * Support the ~, &, ==, !=, and |= operators with other objects of the same type
28// * Support shift (<<, >>) with signed and unsigned integers on the RHS
29// * Support put (<<) to raw_ostream&
30template <typename InsnType>
31#if defined(_MSC_VER) && !defined(__clang__)
32__declspec(noinline)
33#endif
34static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
35 unsigned numBits, std::true_type) {
36 assert(startBit + numBits <= 64 && "Cannot support >64-bit extractions!");
37 assert(startBit + numBits <= (sizeof(InsnType) * 8) &&
38 "Instruction field out of bounds!");
39 InsnType fieldMask;
40 if (numBits == sizeof(InsnType) * 8)
41 fieldMask = (InsnType)(-1LL);
42 else
43 fieldMask = (((InsnType)1 << numBits) - 1) << startBit;
44 return (insn & fieldMask) >> startBit;
45}
46
47template <typename InsnType>
48static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
49 unsigned numBits, std::false_type) {
50 assert(startBit + numBits <= InsnType::max_size_in_bits && "Instruction field out of bounds!");
51 InsnType fieldMask = InsnType::getBitsSet(0, numBits);
52 return (insn >> startBit) & fieldMask;
53}

55template <typename InsnType>
56static InsnType fieldFromInstruction(InsnType insn, unsigned startBit,
57 unsigned numBits) {
58 return fieldFromInstruction(insn, startBit, numBits, std::is_integral<InsnType>());
59}

566template <typename InsnType>
567static DecodeStatus decodeInstruction(const uint8_t DecodeTable[], MCInst &MI,
568 InsnType insn, uint64_t Address,
569 const void *DisAsm,
570 const MCSubtargetInfo &STI) {
571 const FeatureBitset &Bits = STI.getFeatureBits();
572
573 const uint8_t *Ptr = DecodeTable;
574 InsnType CurFieldValue = 0;
575 DecodeStatus S = MCDisassembler::Success;
576 while (true) {
577 ptrdiff_t Loc = Ptr - DecodeTable;
578 switch (*Ptr) {
579 default:
580 errs() << Loc << ": Unexpected decode table opcode!\n";
581 return MCDisassembler::Fail;
582 case MCD::OPC_ExtractField: {
583 unsigned Start = *++Ptr;
584 unsigned Len = *++Ptr;
585 ++Ptr;
586 CurFieldValue = fieldFromInstruction(insn, Start, Len);
587 LLVM_DEBUG(dbgs() << Loc << ": OPC_ExtractField(" << Start << ", "
588 << Len << "): " << CurFieldValue << "\n");
589 break;
590 }
591 case MCD::OPC_FilterValue: {
592 // Decode the field value.
593 unsigned Len;
594 InsnType Val = decodeULEB128(++Ptr, &Len);
595 Ptr += Len;
596 // NumToSkip is a plain 24-bit integer.
597 unsigned NumToSkip = *Ptr++;
598 NumToSkip |= (*Ptr++) << 8;
599 NumToSkip |= (*Ptr++) << 16;
600
601 // Perform the filter operation.
602 if (Val != CurFieldValue)
603 Ptr += NumToSkip;
604 LLVM_DEBUG(dbgs() << Loc << ": OPC_FilterValue(" << Val << ", " << NumToSkip
605 << "): " << ((Val != CurFieldValue) ? "FAIL:" : "PASS:")
606 << " continuing at " << (Ptr - DecodeTable) << "\n");
607
608 break;
609 }
610 case MCD::OPC_CheckField: {
611 unsigned Start = *++Ptr;
612 unsigned Len = *++Ptr;
613 InsnType FieldValue = fieldFromInstruction(insn, Start, Len);
614 // Decode the field value.
615 InsnType ExpectedValue = decodeULEB128(++Ptr, &Len);
616 Ptr += Len;
617 // NumToSkip is a plain 24-bit integer.
618 unsigned NumToSkip = *Ptr++;
619 NumToSkip |= (*Ptr++) << 8;
620 NumToSkip |= (*Ptr++) << 16;
621
622 // If the actual and expected values don't match, skip.
623 if (ExpectedValue != FieldValue)
624 Ptr += NumToSkip;
625 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckField(" << Start << ", "
626 << Len << ", " << ExpectedValue << ", " << NumToSkip
627 << "): FieldValue = " << FieldValue << ", ExpectedValue = "
628 << ExpectedValue << ": "
629 << ((ExpectedValue == FieldValue) ? "PASS\n" : "FAIL\n"));
630 break;
631 }
632 case MCD::OPC_CheckPredicate: {
633 unsigned Len;
634 // Decode the Predicate Index value.
635 unsigned PIdx = decodeULEB128(++Ptr, &Len);
636 Ptr += Len;
637 // NumToSkip is a plain 24-bit integer.
638 unsigned NumToSkip = *Ptr++;
639 NumToSkip |= (*Ptr++) << 8;
640 NumToSkip |= (*Ptr++) << 16;
641 // Check the predicate.
642 bool Pred;
643 if (!(Pred = checkDecoderPredicate(PIdx, Bits)))
644 Ptr += NumToSkip;
645 (void)Pred;
646 LLVM_DEBUG(dbgs() << Loc << ": OPC_CheckPredicate(" << PIdx << "): "
647 << (Pred ? "PASS\n" : "FAIL\n"));
648
649 break;
650 }
651 case MCD::OPC_Decode: {
652 unsigned Len;
653 // Decode the Opcode value.
654 unsigned Opc = decodeULEB128(++Ptr, &Len);
655 Ptr += Len;
656 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
657 Ptr += Len;
658
659 MI.clear();
660 MI.setOpcode(Opc);
661 bool DecodeComplete;
662 S = decodeToMCInst(S, DecodeIdx, insn, MI, Address, DisAsm, DecodeComplete);
663 assert(DecodeComplete);
664
665 LLVM_DEBUG(dbgs() << Loc << ": OPC_Decode: opcode " << Opc
666 << ", using decoder " << DecodeIdx << ": "
667 << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
668 return S;
669 }
670 case MCD::OPC_TryDecode: {
671 unsigned Len;
672 // Decode the Opcode value.
673 unsigned Opc = decodeULEB128(++Ptr, &Len);
674 Ptr += Len;
675 unsigned DecodeIdx = decodeULEB128(Ptr, &Len);
676 Ptr += Len;
677 // NumToSkip is a plain 24-bit integer.
678 unsigned NumToSkip = *Ptr++;
679 NumToSkip |= (*Ptr++) << 8;
680 NumToSkip |= (*Ptr++) << 16;
681
682 // Perform the decode operation.
683 MCInst TmpMI;
684 TmpMI.setOpcode(Opc);
685 bool DecodeComplete;
686 S = decodeToMCInst(S, DecodeIdx, insn, TmpMI, Address, DisAsm, DecodeComplete);
687 LLVM_DEBUG(dbgs() << Loc << ": OPC_TryDecode: opcode " << Opc
688 << ", using decoder " << DecodeIdx << ": ");
689
690 if (DecodeComplete) {
691 // Decoding complete.
692 LLVM_DEBUG(dbgs() << (S != MCDisassembler::Fail ? "PASS" : "FAIL") << "\n");
693 MI = TmpMI;
694 return S;
695 } else {
696 assert(S == MCDisassembler::Fail);
697 // If the decoding was incomplete, skip.
698 Ptr += NumToSkip;
699 LLVM_DEBUG(dbgs() << "FAIL: continuing at " << (Ptr - DecodeTable) << "\n");
700 // Reset decode status. This also drops a SoftFail status that could be
701 // set before the decode attempt.
702 S = MCDisassembler::Success;
703 }
704 break;
705 }
706 case MCD::OPC_SoftFail: {
707 // Decode the mask values.
708 unsigned Len;
709 InsnType PositiveMask = decodeULEB128(++Ptr, &Len);
710 Ptr += Len;
711 InsnType NegativeMask = decodeULEB128(Ptr, &Len);
712 Ptr += Len;
713 bool Fail = (insn & PositiveMask) || (~insn & NegativeMask);
714 if (Fail)
715 S = MCDisassembler::SoftFail;
716 LLVM_DEBUG(dbgs() << Loc << ": OPC_SoftFail: " << (Fail ? "FAIL\n" : "PASS\n"));
717 break;
718 }
719 case MCD::OPC_Fail: {
720 LLVM_DEBUG(dbgs() << Loc << ": OPC_Fail\n");
721 return MCDisassembler::Fail;
722 }
723 }
724 }
725 llvm_unreachable("bogosity detected in disassembler state machine!");
726}
