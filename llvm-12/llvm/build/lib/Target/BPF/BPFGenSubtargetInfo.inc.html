135// ParseSubtargetFeatures - Parses features string setting specified
136// subtarget options.
137void llvm::BPFSubtarget::ParseSubtargetFeatures(StringRef CPU, StringRef TuneCPU, StringRef FS) {
138 LLVM_DEBUG(dbgs() << "\nFeatures:" << FS);
139 LLVM_DEBUG(dbgs() << "\nCPU:" << CPU);
140 LLVM_DEBUG(dbgs() << "\nTuneCPU:" << TuneCPU << "\n\n");
141 InitMCProcessorInfo(CPU, TuneCPU, FS);
142 const FeatureBitset &Bits = getFeatureBits();
143 if (Bits[BPF::ALU32]) HasAlu32 = true;
144 if (Bits[BPF::DummyFeature]) isDummyMode = true;
145 if (Bits[BPF::DwarfRIS]) UseDwarfRIS = true;
146}
