538template <class _A1, class _A2, class _A3>
539inline _LIBCPP_INLINE_VISIBILITY
540typename _EnableIf
541<
542 is_arithmetic<_A1>::value &&
543 is_arithmetic<_A2>::value &&
544 is_arithmetic<_A3>::value,
545 __promote<_A1, _A2, _A3>
546>::type
547hypot(_A1 __lcpp_x, _A2 __lcpp_y, _A3 __lcpp_z) _NOEXCEPT
548{
549 typedef typename __promote<_A1, _A2, _A3>::type __result_type;
550 static_assert((!(is_same<_A1, __result_type>::value &&
551 is_same<_A2, __result_type>::value &&
552 is_same<_A3, __result_type>::value)), "");
553 return hypot((__result_type)__lcpp_x, (__result_type)__lcpp_y, (__result_type)__lcpp_z);
554}

618template <typename _Fp>
619constexpr
620_Fp __lerp(_Fp __a, _Fp __b, _Fp __t) noexcept {
621 if ((__a <= 0 && __b >= 0) || (__a >= 0 && __b <= 0))
622 return __t * __b + (1 - __t) * __a;
623
624 if (__t == 1) return __b;
625 const _Fp __x = __a + __t * (__b - __a);
626 if (__t > 1 == __b > __a)
627 return __b < __x ? __x : __b;
628 else
629 return __x < __b ? __x : __b;
630}

643template <class _IntT, class _FloatT,
644 bool _FloatBigger = (numeric_limits<_FloatT>::digits > numeric_limits<_IntT>::digits),
645 int _Bits = (numeric_limits<_IntT>::digits - numeric_limits<_FloatT>::digits)>
646_LIBCPP_INLINE_VISIBILITY
647_LIBCPP_CONSTEXPR _IntT __max_representable_int_for_float() _NOEXCEPT {
648 static_assert(is_floating_point<_FloatT>::value, "must be a floating point type");
649 static_assert(is_integral<_IntT>::value, "must be an integral type");
650 static_assert(numeric_limits<_FloatT>::radix == 2, "FloatT has incorrect radix");
651 static_assert((_IsSame<_FloatT, float>::value || _IsSame<_FloatT, double>::value
652 || _IsSame<_FloatT,long double>::value), "unsupported floating point type");
653 return _FloatBigger ? numeric_limits<_IntT>::max() : (numeric_limits<_IntT>::max() >> _Bits << _Bits);
654}

656// Convert a floating point number to the specified integral type after
657// clamping to the integral types representable range.
658//
659// The behavior is undefined if `__r` is NaN.
660template <class _IntT, class _RealT>
661_LIBCPP_INLINE_VISIBILITY
662_IntT __clamp_to_integral(_RealT __r) _NOEXCEPT {
663 using _Lim = numeric_limits<_IntT>;
664 const _IntT _MaxVal = __max_representable_int_for_float<_IntT, _RealT>();
665 if (__r >= ::nextafter(static_cast<_RealT>(_MaxVal), INFINITY)) {
666 return _Lim::max();
667 } else if (__r <= _Lim::lowest()) {
668 return _Lim::min();
669 }
670 return static_cast<_IntT>(__r);
671}
