107inline
108int __libcpp_snprintf_l(char *__s, size_t __n, locale_t __l, const char *__format, ...) {
109 va_list __va;
110 va_start(__va, __format);
111 __libcpp_locale_guard __current(__l);
112 int __res = vsnprintf(__s, __n, __format, __va);
113 va_end(__va);
114 return __res;
115}
