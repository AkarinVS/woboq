105static inline bool hasCompatibleFnAttrs(const Function &Caller,
106 const Function &Callee) {
107 bool Ret = true;
108
109 Ret &= isEqual<SanitizeAddressAttr>(Caller, Callee);
110 Ret &= isEqual<SanitizeThreadAttr>(Caller, Callee);
111 Ret &= isEqual<SanitizeMemoryAttr>(Caller, Callee);
112 Ret &= isEqual<SanitizeHWAddressAttr>(Caller, Callee);
113 Ret &= isEqual<SanitizeMemTagAttr>(Caller, Callee);
114 Ret &= isEqual<SafeStackAttr>(Caller, Callee);
115 Ret &= isEqual<ShadowCallStackAttr>(Caller, Callee);
116 Ret &= isEqual<UseSampleProfileAttr>(Caller, Callee);
117
118 return Ret;
119}

121static inline void mergeFnAttrs(Function &Caller,
122 const Function &Callee) {
123 setAND<NoNansFPMathAttr>(Caller, Callee);
124 setAND<NoSignedZerosFPMathAttr>(Caller, Callee);
125 setAND<UnsafeFPMathAttr>(Caller, Callee);
126 setOR<NoImplicitFloatAttr>(Caller, Callee);
127 setOR<NoJumpTablesAttr>(Caller, Callee);
128 setOR<ProfileSampleAccurateAttr>(Caller, Callee);
129 setOR<SpeculativeLoadHardeningAttr>(Caller, Callee);
130 adjustCallerSSPLevel(Caller, Callee);
131 adjustCallerStackProbes(Caller, Callee);
132 adjustCallerStackProbeSize(Caller, Callee);
133 adjustMinLegalVectorWidth(Caller, Callee);
134 adjustNullPointerValidAttr(Caller, Callee);
135 setAND<MustProgressAttr>(Caller, Callee);
136 setAND<LessPreciseFPMADAttr>(Caller, Callee);
137 setAND<NoInfsFPMathAttr>(Caller, Callee);
138}
