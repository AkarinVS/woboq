268unsigned serialization::ComputeHash(Selector Sel) {
269 unsigned N = Sel.getNumArgs();
270 if (N == 0)
271 ++N;
272 unsigned R = 5381;
273 for (unsigned I = 0; I != N; ++I)
274 if (IdentifierInfo *II = Sel.getIdentifierInfoForSlot(I))
275 R = llvm::djbHash(II->getName(), R);
276 return R;
277}

279const DeclContext *
280serialization::getDefinitiveDeclContext(const DeclContext *DC) {
281 switch (DC->getDeclKind()) {
282 // These entities may have multiple definitions.
283 case Decl::TranslationUnit:
284 case Decl::ExternCContext:
285 case Decl::Namespace:
286 case Decl::LinkageSpec:
287 case Decl::Export:
288 return nullptr;
289
290 // C/C++ tag types can only be defined in one place.
291 case Decl::Enum:
292 case Decl::Record:
293 if (const TagDecl *Def = cast<TagDecl>(DC)->getDefinition())
294 return Def;
295 return nullptr;
296
297 // FIXME: These can be defined in one place... except special member
298 // functions and out-of-line definitions.
299 case Decl::CXXRecord:
300 case Decl::ClassTemplateSpecialization:
301 case Decl::ClassTemplatePartialSpecialization:
302 return nullptr;
303
304 // Each function, method, and block declaration is its own DeclContext.
305 case Decl::Function:
306 case Decl::CXXMethod:
307 case Decl::CXXConstructor:
308 case Decl::CXXDestructor:
309 case Decl::CXXConversion:
310 case Decl::ObjCMethod:
311 case Decl::Block:
312 case Decl::Captured:
313 // Objective C categories, category implementations, and class
314 // implementations can only be defined in one place.
315 case Decl::ObjCCategory:
316 case Decl::ObjCCategoryImpl:
317 case Decl::ObjCImplementation:
318 return DC;
319
320 case Decl::ObjCProtocol:
321 if (const ObjCProtocolDecl *Def
322 = cast<ObjCProtocolDecl>(DC)->getDefinition())
323 return Def;
324 return nullptr;
325
326 // FIXME: These are defined in one place, but properties in class extensions
327 // end up being back-patched into the main interface. See
328 // Sema::HandlePropertyInClassExtension for the offending code.
329 case Decl::ObjCInterface:
330 return nullptr;
331
332 default:
333 llvm_unreachable("Unhandled DeclContext in AST reader");
334 }
335
336 llvm_unreachable("Unhandled decl kind");
337}

438bool serialization::needsAnonymousDeclarationNumber(const NamedDecl *D) {
439 // Friend declarations in dependent contexts aren't anonymous in the usual
440 // sense, but they cannot be found by name lookup in their semantic context
441 // (or indeed in any context), so we treat them as anonymous.
442 //
443 // This doesn't apply to friend tag decls; Sema makes those available to name
444 // lookup in the surrounding context.
445 if (D->getFriendObjectKind() &&
446 D->getLexicalDeclContext()->isDependentContext() && !isa<TagDecl>(D)) {
447 // For function templates and class templates, the template is numbered and
448 // not its pattern.
449 if (auto *FD = dyn_cast<FunctionDecl>(D))
450 return !FD->getDescribedFunctionTemplate();
451 if (auto *RD = dyn_cast<CXXRecordDecl>(D))
452 return !RD->getDescribedClassTemplate();
453 return true;
454 }
455
456 // At block scope, we number everything that we need to deduplicate, since we
457 // can't just use name matching to keep things lined up.
458 // FIXME: This is only necessary for an inline function or a template or
459 // similar.
460 if (D->getLexicalDeclContext()->isFunctionOrMethod()) {
461 if (auto *VD = dyn_cast<VarDecl>(D))
462 return VD->isStaticLocal();
463 // FIXME: What about CapturedDecls (and declarations nested within them)?
464 return isa<TagDecl>(D) || isa<BlockDecl>(D);
465 }
466
467 // Otherwise, we only care about anonymous class members / block-scope decls.
468 // FIXME: We need to handle lambdas and blocks within inline / templated
469 // variables too.
470 if (D->getDeclName() || !isa<CXXRecordDecl>(D->getLexicalDeclContext()))
471 return false;
472 return isa<TagDecl>(D) || isa<FieldDecl>(D);
473}
