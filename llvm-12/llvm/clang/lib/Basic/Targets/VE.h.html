160 bool validateAsmConstraint(const char *&Name,
161 TargetInfo::ConstraintInfo &Info) const override {
162 switch (*Name) {
163 default:
164 return false;
165 case 'v':
166 Info.setAllowsRegister();
167 return true;
168 }
169 return false;
