21void VETargetInfo::getTargetDefines(const LangOptions &Opts,
22 MacroBuilder &Builder) const {
23 Builder.defineMacro("_LP64", "1");
24 Builder.defineMacro("unix", "1");
25 Builder.defineMacro("__unix__", "1");
26 Builder.defineMacro("__linux__", "1");
27 Builder.defineMacro("__ve", "1");
28 Builder.defineMacro("__ve__", "1");
29 Builder.defineMacro("__STDC_HOSTED__", "1");
30 Builder.defineMacro("__STDC__", "1");
31 Builder.defineMacro("__NEC__", "1");
32 // FIXME: define __FAST_MATH__ 1 if -ffast-math is enabled
33 // FIXME: define __OPTIMIZE__ n if -On is enabled
34 // FIXME: define __VECTOR__ n 1 if automatic vectorization is enabled
35}

38 return ArrayRef<Builtin::Info>();
