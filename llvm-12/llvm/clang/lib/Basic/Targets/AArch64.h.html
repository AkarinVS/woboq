23class LLVM_LIBRARY_VISIBILITY AArch64TargetInfo : public TargetInfo {
24 virtual void setDataLayout() = 0;
25 static const TargetInfo::GCCRegAlias GCCRegAliases[];
26 static const char *const GCCRegNames[];
27
28 enum FPUModeEnum { FPUMode, NeonMode = (1 << 0), SveMode = (1 << 1) };
29
30 unsigned FPU;
31 bool HasCRC;
32 bool HasCrypto;
33 bool HasUnaligned;
34 bool HasFullFP16;
35 bool HasDotProd;
36 bool HasFP16FML;
37 bool HasMTE;
38 bool HasTME;
39 bool HasPAuth;
40 bool HasLS64;
41 bool HasMatMul;
42 bool HasSVE2;
43 bool HasSVE2AES;
44 bool HasSVE2SHA3;
45 bool HasSVE2SM4;
46 bool HasSVE2BitPerm;
47 bool HasMatmulFP64;
48 bool HasMatmulFP32;
49 bool HasLSE;
50 bool HasFlagM;
51
52 llvm::AArch64::ArchKind ArchKind;
53
54 static const Builtin::Info BuiltinInfo[];
55
56 std::string ABI;
57
58public:
59 AArch64TargetInfo(const llvm::Triple &Triple, const TargetOptions &Opts);
60
61 StringRef getABI() const override;
62 bool setABI(const std::string &Name) override;
63
64 bool validateBranchProtection(StringRef, BranchProtectionInfo &,
65 StringRef &) const override;
66
67 bool isValidCPUName(StringRef Name) const override;
68 void fillValidCPUList(SmallVectorImpl<StringRef> &Values) const override;
69 bool setCPU(const std::string &Name) override;
70
71 bool useFP16ConversionIntrinsics() const override {
72 return false;
73 }
74
75 void getTargetDefinesARMV81A(const LangOptions &Opts,
76 MacroBuilder &Builder) const;
77 void getTargetDefinesARMV82A(const LangOptions &Opts,
78 MacroBuilder &Builder) const;
79 void getTargetDefinesARMV83A(const LangOptions &Opts,
80 MacroBuilder &Builder) const;
81 void getTargetDefinesARMV84A(const LangOptions &Opts,
82 MacroBuilder &Builder) const;
83 void getTargetDefinesARMV85A(const LangOptions &Opts,
84 MacroBuilder &Builder) const;
85 void getTargetDefinesARMV86A(const LangOptions &Opts,
86 MacroBuilder &Builder) const;
87 void getTargetDefinesARMV87A(const LangOptions &Opts,
88 MacroBuilder &Builder) const;
89 void getTargetDefines(const LangOptions &Opts,
90 MacroBuilder &Builder) const override;
91
92 ArrayRef<Builtin::Info> getTargetBuiltins() const override;
93
94 bool hasFeature(StringRef Feature) const override;
95 bool handleTargetFeatures(std::vector<std::string> &Features,
96 DiagnosticsEngine &Diags) override;
97
98 CallingConvCheckResult checkCallingConvention(CallingConv CC) const override;
99
100 bool isCLZForZeroUndef() const override;
101
102 BuiltinVaListKind getBuiltinVaListKind() const override;
103
104 ArrayRef<const char *> getGCCRegNames() const override;
105 ArrayRef<TargetInfo::GCCRegAlias> getGCCRegAliases() const override;
106
107 std::string convertConstraint(const char *&Constraint) const override {
108 std::string R;
109 switch (*Constraint) {
110 case 'U': // Three-character constraint; add "@3" hint for later parsing.
111 R = std::string("@3") + std::string(Constraint, 3);
112 Constraint += 2;
113 break;
114 default:
115 R = TargetInfo::convertConstraint(Constraint);
116 break;
117 }
118 return R;
119 }
120
121 bool validateAsmConstraint(const char *&Name,
122 TargetInfo::ConstraintInfo &Info) const override;
123 bool
124 validateConstraintModifier(StringRef Constraint, char Modifier, unsigned Size,
125 std::string &SuggestedModifier) const override;
126 const char *getClobbers() const override;
127
128 StringRef getConstraintRegister(StringRef Constraint,
129 StringRef Expression) const override {
130 return Expression;
131 }
132
133 int getEHDataRegisterNumber(unsigned RegNo) const override;
134
135 const char *getBFloat16Mangling() const override { return "u6__bf16"; };
136 bool hasInt128Type() const override;
137
138 bool hasExtIntType() const override { return true; }
139};
