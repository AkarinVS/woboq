18unsigned FunctionSummariesTy::getTotalNumBasicBlocks() {
19 unsigned Total = 0;
20 for (const auto &I : Map)
21 Total += I.second.TotalBasicBlocks;
22 return Total;
23}
24
25unsigned FunctionSummariesTy::getTotalNumVisitedBasicBlocks() {
26 unsigned Total = 0;
27 for (const auto &I : Map)
28 Total += I.second.VisitedBasicBlocks.count();
29 return Total;
30}
