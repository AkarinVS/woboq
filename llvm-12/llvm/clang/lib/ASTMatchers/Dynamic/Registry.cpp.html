61void RegistryMaps::registerMatcher(
62 StringRef MatcherName, std::unique_ptr<MatcherDescriptor> Callback) {
63 assert(Constructors.find(MatcherName) == Constructors.end());
64 Constructors[MatcherName] = std::move(Callback);
65}

67#define REGISTER_MATCHER(name) \
68 registerMatcher(#name, internal::makeMatcherAutoMarshall( \
69 ::clang::ast_matchers::name, #name));
70
71#define REGISTER_MATCHER_OVERLOAD(name) \
72 registerMatcher(#name, \
73 std::make_unique<internal::OverloadedMatcherDescriptor>(name##Callbacks))
74
75#define SPECIFIC_MATCHER_OVERLOAD(name, Id) \
76 static_cast<::clang::ast_matchers::name##_Type##Id>( \
77 ::clang::ast_matchers::name)
78
79#define MATCHER_OVERLOAD_ENTRY(name, Id) \
80 internal::makeMatcherAutoMarshall(SPECIFIC_MATCHER_OVERLOAD(name, Id), \
81 #name)
82
83#define REGISTER_OVERLOADED_2(name) \
84 do { \
85 std::unique_ptr<MatcherDescriptor> name##Callbacks[] = { \
86 MATCHER_OVERLOAD_ENTRY(name, 0), \
87 MATCHER_OVERLOAD_ENTRY(name, 1)}; \
88 REGISTER_MATCHER_OVERLOAD(name); \
89 } while (false)
90
91#define REGISTER_REGEX_MATCHER(name) \
92 registerMatcher(#name, internal::makeMatcherRegexMarshall(name, name))

561llvm::Optional<MatcherCtor> Registry::lookupMatcherCtor(StringRef MatcherName) {
562 auto it = RegistryData->constructors().find(MatcherName);
563 return it == RegistryData->constructors().end()
564 ? llvm::Optional<MatcherCtor>()
565 : it->second.get();
566}

568static llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
569 const std::set<ASTNodeKind> &KS) {
570 unsigned Count = 0;
571 for (std::set<ASTNodeKind>::const_iterator I = KS.begin(), E = KS.end();
572 I != E; ++I) {
573 if (I != KS.begin())
574 OS << "|";
575 if (Count++ == 3) {
576 OS << "...";
577 break;
578 }
579 OS << *I;
580 }
581 return OS;
582}

584std::vector<ArgKind> Registry::getAcceptedCompletionTypes(
585 ArrayRef<std::pair<MatcherCtor, unsigned>> Context) {
586 ASTNodeKind InitialTypes[] = {
587 ASTNodeKind::getFromNodeKind<Decl>(),
588 ASTNodeKind::getFromNodeKind<QualType>(),
589 ASTNodeKind::getFromNodeKind<Type>(),
590 ASTNodeKind::getFromNodeKind<Stmt>(),
591 ASTNodeKind::getFromNodeKind<NestedNameSpecifier>(),
592 ASTNodeKind::getFromNodeKind<NestedNameSpecifierLoc>(),
593 ASTNodeKind::getFromNodeKind<TypeLoc>()};
594
595 // Starting with the above seed of acceptable top-level matcher types, compute
596 // the acceptable type set for the argument indicated by each context element.
597 std::set<ArgKind> TypeSet(std::begin(InitialTypes), std::end(InitialTypes));
598 for (const auto &CtxEntry : Context) {
599 MatcherCtor Ctor = CtxEntry.first;
600 unsigned ArgNumber = CtxEntry.second;
601 std::vector<ArgKind> NextTypeSet;
602 for (const ArgKind &Kind : TypeSet) {
603 if (Kind.getArgKind() == Kind.AK_Matcher &&
604 Ctor->isConvertibleTo(Kind.getMatcherKind()) &&
605 (Ctor->isVariadic() || ArgNumber < Ctor->getNumArgs()))
606 Ctor->getArgKinds(Kind.getMatcherKind(), ArgNumber, NextTypeSet);
607 }
608 TypeSet.clear();
609 TypeSet.insert(NextTypeSet.begin(), NextTypeSet.end());
610 }
611 return std::vector<ArgKind>(TypeSet.begin(), TypeSet.end());
612}
613
614std::vector<MatcherCompletion>
615Registry::getMatcherCompletions(ArrayRef<ArgKind> AcceptedTypes) {
616 std::vector<MatcherCompletion> Completions;
617
618 // Search the registry for acceptable matchers.
619 for (const auto &M : RegistryData->constructors()) {
620 const MatcherDescriptor& Matcher = *M.getValue();
621 StringRef Name = M.getKey();
622
623 std::set<ASTNodeKind> RetKinds;
624 unsigned NumArgs = Matcher.isVariadic() ? 1 : Matcher.getNumArgs();
625 bool IsPolymorphic = Matcher.isPolymorphic();
626 std::vector<std::vector<ArgKind>> ArgsKinds(NumArgs);
627 unsigned MaxSpecificity = 0;
628 for (const ArgKind& Kind : AcceptedTypes) {
629 if (Kind.getArgKind() != Kind.AK_Matcher)
630 continue;
631 unsigned Specificity;
632 ASTNodeKind LeastDerivedKind;
633 if (Matcher.isConvertibleTo(Kind.getMatcherKind(), &Specificity,
634 &LeastDerivedKind)) {
635 if (MaxSpecificity < Specificity)
636 MaxSpecificity = Specificity;
637 RetKinds.insert(LeastDerivedKind);
638 for (unsigned Arg = 0; Arg != NumArgs; ++Arg)
639 Matcher.getArgKinds(Kind.getMatcherKind(), Arg, ArgsKinds[Arg]);
640 if (IsPolymorphic)
641 break;
642 }
643 }
644
645 if (!RetKinds.empty() && MaxSpecificity > 0) {
646 std::string Decl;
647 llvm::raw_string_ostream OS(Decl);
648
649 if (IsPolymorphic) {
650 OS << "Matcher<T> " << Name << "(Matcher<T>";
651 } else {
652 OS << "Matcher<" << RetKinds << "> " << Name << "(";
653 for (const std::vector<ArgKind> &Arg : ArgsKinds) {
654 if (&Arg != &ArgsKinds[0])
655 OS << ", ";
656
657 bool FirstArgKind = true;
658 std::set<ASTNodeKind> MatcherKinds;
659 // Two steps. First all non-matchers, then matchers only.
660 for (const ArgKind &AK : Arg) {
661 if (AK.getArgKind() == ArgKind::AK_Matcher) {
662 MatcherKinds.insert(AK.getMatcherKind());
663 } else {
664 if (!FirstArgKind) OS << "|";
665 FirstArgKind = false;
666 OS << AK.asString();
667 }
668 }
669 if (!MatcherKinds.empty()) {
670 if (!FirstArgKind) OS << "|";
671 OS << "Matcher<" << MatcherKinds << ">";
672 }
673 }
674 }
675 if (Matcher.isVariadic())
676 OS << "...";
677 OS << ")";
678
679 std::string TypedText = std::string(Name);
680 TypedText += "(";
681 if (ArgsKinds.empty())
682 TypedText += ")";
683 else if (ArgsKinds[0][0].getArgKind() == ArgKind::AK_String)
684 TypedText += "\"";
685
686 Completions.emplace_back(TypedText, OS.str(), MaxSpecificity);
687 }
688 }
689
690 return Completions;
691}

700VariantMatcher Registry::constructBoundMatcher(MatcherCtor Ctor,
701 SourceRange NameRange,
702 StringRef BindID,
703 ArrayRef<ParserValue> Args,
704 Diagnostics *Error) {
705 VariantMatcher Out = constructMatcher(Ctor, NameRange, Args, Error);
706 if (Out.isNull()) return Out;
707
708 llvm::Optional<DynTypedMatcher> Result = Out.getSingleMatcher();
709 if (Result.hasValue()) {
710 llvm::Optional<DynTypedMatcher> Bound = Result->tryBind(BindID);
711 if (Bound.hasValue()) {
712 return VariantMatcher::SingleMatcher(*Bound);
713 }
714 }
715 Error->addError(NameRange, Error->ET_RegistryNotBindable);
716 return VariantMatcher();
717}
