10#include "clang/AST/Expr.h"

12using namespace clang;
13using namespace clang::interp;

15SourceLocation SourceInfo::getLoc() const {
16 if (const Expr *E = asExpr())
17 return E->getExprLoc();
18 if (const Stmt *S = asStmt())
19 return S->getBeginLoc();
20 if (const Decl *D = asDecl())
21 return D->getBeginLoc();
22 return SourceLocation();
23}
24
25const Expr *SourceInfo::asExpr() const {
26 if (auto *S = Source.dyn_cast<const Stmt *>())
27 return dyn_cast<Expr>(S);
28 return nullptr;
29}
30
31const Expr *SourceMapper::getExpr(Function *F, CodePtr PC) const {
32 if (const Expr *E = getSource(F, PC).asExpr())
33 return E;
34 llvm::report_fatal_error("missing source expression");
35}
36
37SourceLocation SourceMapper::getLocation(Function *F, CodePtr PC) const {
38 return getSource(F, PC).getLoc();
39}
