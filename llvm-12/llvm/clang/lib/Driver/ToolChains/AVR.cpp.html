289 return Optional<StringRef>();

306/// AVR Toolchain
307AVRToolChain::AVRToolChain(const Driver &D, const llvm::Triple &Triple,
308 const ArgList &Args)
309 : Generic_ELF(D, Triple, Args), LinkStdlib(false) {
310 GCCInstallation.init(Triple, Args);
311
312 // Only add default libraries if the user hasn't explicitly opted out.
313 if (!Args.hasArg(options::OPT_nostdlib) &&
314 !Args.hasArg(options::OPT_nodefaultlibs) &&
315 !Args.hasArg(options::OPT_c /* does not apply when not linking */)) {
316 std::string CPU = getCPUName(Args, Triple);
317
318 if (CPU.empty()) {
319 // We cannot link any standard libraries without an MCU specified.
320 D.Diag(diag::warn_drv_avr_mcu_not_specified);
321 } else {
322 Optional<StringRef> FamilyName = GetMCUFamilyName(CPU);
323 Optional<std::string> AVRLibcRoot = findAVRLibcInstallation();
324
325 if (!FamilyName.hasValue()) {
326 // We do not have an entry for this CPU in the family
327 // mapping table yet.
328 D.Diag(diag::warn_drv_avr_family_linking_stdlibs_not_implemented)
329 << CPU;
330 } else if (!GCCInstallation.isValid()) {
331 // No avr-gcc found and so no runtime linked.
332 D.Diag(diag::warn_drv_avr_gcc_not_found);
333 } else if (!AVRLibcRoot.hasValue()) {
334 // No avr-libc found and so no runtime linked.
335 D.Diag(diag::warn_drv_avr_libc_not_found);
336 } else { // We have enough information to link stdlibs
337 std::string GCCRoot = std::string(GCCInstallation.getInstallPath());
338 std::string LibcRoot = AVRLibcRoot.getValue();
339 std::string SubPath = GetMCUSubPath(CPU);
340
341 getFilePaths().push_back(LibcRoot + std::string("/lib/") + SubPath);
342 getFilePaths().push_back(GCCRoot + std::string("/") + SubPath);
343
344 LinkStdlib = true;
345 }
346 }
347
348 if (!LinkStdlib)
349 D.Diag(diag::warn_drv_avr_stdlib_not_linked);
350 }
351}

357void AVR::Linker::ConstructJob(Compilation &C, const JobAction &JA,
358 const InputInfo &Output,
359 const InputInfoList &Inputs, const ArgList &Args,
360 const char *LinkingOutput) const {
361 // Compute information about the target AVR.
362 std::string CPU = getCPUName(Args, getToolChain().getTriple());
363 llvm::Optional<StringRef> FamilyName = GetMCUFamilyName(CPU);
364 llvm::Optional<unsigned> SectionAddressData = GetMCUSectionAddressData(CPU);
365
366 std::string Linker = getToolChain().GetProgramPath(getShortName());
367 ArgStringList CmdArgs;
368 AddLinkerInputs(getToolChain(), Inputs, Args, CmdArgs, JA);
369
370 CmdArgs.push_back("-o");
371 CmdArgs.push_back(Output.getFilename());
372
373 // Enable garbage collection of unused sections.
374 CmdArgs.push_back("--gc-sections");
375
376 // Add library search paths before we specify libraries.
377 Args.AddAllArgs(CmdArgs, options::OPT_L);
378 getToolChain().AddFilePathLibArgs(Args, CmdArgs);
379
380 if (SectionAddressData.hasValue()) {
381 std::string DataSectionArg = std::string("-Tdata=0x") +
382 llvm::utohexstr(SectionAddressData.getValue());
383 CmdArgs.push_back(Args.MakeArgString(DataSectionArg));
384 } else {
385 // We do not have an entry for this CPU in the address mapping table yet.
386 getToolChain().getDriver().Diag(
387 diag::warn_drv_avr_linker_section_addresses_not_implemented)
388 << CPU;
389 }
390
391 // If the family name is known, we can link with the device-specific libgcc.
392 // Without it, libgcc will simply not be linked. This matches avr-gcc
393 // behavior.
394 if (LinkStdlib) {
395 assert(!CPU.empty() && "CPU name must be known in order to link stdlibs");
396
397 // Add the object file for the CRT.
398 std::string CrtFileName = std::string("-l:crt") + CPU + std::string(".o");
399 CmdArgs.push_back(Args.MakeArgString(CrtFileName));
400
401 CmdArgs.push_back("-lgcc");
402 CmdArgs.push_back("-lm");
403 CmdArgs.push_back("-lc");
404
405 // Add the link library specific to the MCU.
406 CmdArgs.push_back(Args.MakeArgString(std::string("-l") + CPU));
407
408 // Specify the family name as the emulation mode to use.
409 // This is almost always required because otherwise avr-ld
410 // will assume 'avr2' and warn about the program being larger
411 // than the bare minimum supports.
412 CmdArgs.push_back(Args.MakeArgString(std::string("-m") + *FamilyName));
413 }
414
415 C.addCommand(std::make_unique<Command>(
416 JA, *this, ResponseFileSupport::AtFileCurCP(), Args.MakeArgString(Linker),
417 CmdArgs, Inputs, Output));
418}

420llvm::Optional<std::string> AVRToolChain::findAVRLibcInstallation() const {
421 for (StringRef PossiblePath : PossibleAVRLibcLocations) {
422 // Return the first avr-libc installation that exists.
423 if (llvm::sys::fs::is_directory(PossiblePath))
424 return Optional<std::string>(std::string(PossiblePath));
425 }
426
427 return llvm::None;
