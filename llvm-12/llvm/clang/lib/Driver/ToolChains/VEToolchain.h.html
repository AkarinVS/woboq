9#ifndef LLVM_CLANG_LIB_DRIVER_TOOLCHAINS_VE_H
10#define LLVM_CLANG_LIB_DRIVER_TOOLCHAINS_VE_H
11
12#include "Linux.h"
13#include "clang/Driver/ToolChain.h"
14
15namespace clang {
16namespace driver {
17namespace toolchains {
18
19class LLVM_LIBRARY_VISIBILITY VEToolChain : public Linux {
20public:
21 VEToolChain(const Driver &D, const llvm::Triple &Triple,
22 const llvm::opt::ArgList &Args);
23
24protected:
25 Tool *buildAssembler() const override;
26 Tool *buildLinker() const override;
27
28public:
29 bool IsIntegratedAssemblerDefault() const override { return true; }
30 bool isPICDefault() const override;
31 bool isPIEDefault() const override;
32 bool isPICDefaultForced() const override;
33 bool SupportsProfiling() const override;
34 bool hasBlocksRuntime() const override;
35 void
36 AddClangSystemIncludeArgs(const llvm::opt::ArgList &DriverArgs,
37 llvm::opt::ArgStringList &CC1Args) const override;
38 void
39 addClangTargetOptions(const llvm::opt::ArgList &DriverArgs,
40 llvm::opt::ArgStringList &CC1Args,
41 Action::OffloadKind DeviceOffloadKind) const override;
42 void AddClangCXXStdlibIncludeArgs(
43 const llvm::opt::ArgList &DriverArgs,
44 llvm::opt::ArgStringList &CC1Args) const override;
45 void AddCXXStdlibLibArgs(const llvm::opt::ArgList &Args,
46 llvm::opt::ArgStringList &CmdArgs) const override;
47
48 llvm::ExceptionHandling
49 GetExceptionModel(const llvm::opt::ArgList &Args) const override;
50
51 CXXStdlibType
52 GetCXXStdlibType(const llvm::opt::ArgList &Args) const override {
53 return ToolChain::CST_Libcxx;
54 }
55
56 RuntimeLibType GetDefaultRuntimeLibType() const override {
57 return ToolChain::RLT_CompilerRT;
58 }
59
60 const char *getDefaultLinker() const override { return "nld"; }
61};
62
63} // end namespace toolchains
64} // end namespace driver
65} // end namespace clang

67#endif // LLVM_CLANG_LIB_DRIVER_TOOLCHAINS_VE_H
