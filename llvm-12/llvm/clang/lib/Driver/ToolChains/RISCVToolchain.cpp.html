25static void addMultilibsFilePaths(const Driver &D, const MultilibSet &Multilibs,
26 const Multilib &Multilib,
27 StringRef InstallPath,
28 ToolChain::path_list &Paths) {
29 if (const auto &PathsCallback = Multilibs.filePathsCallback())
30 for (const auto &Path : PathsCallback(Multilib))
31 addPathIfExists(D, InstallPath + Path, Paths);
32}

34// This function tests whether a gcc installation is present either
35// through gcc-toolchain argument or in the same prefix where clang
36// is installed. This helps decide whether to instantiate this toolchain
37// or Baremetal toolchain.
38bool RISCVToolChain::hasGCCToolchain(const Driver &D,
39 const llvm::opt::ArgList &Args) {
40 if (Args.getLastArg(options::OPT_gcc_toolchain))
41 return true;
42
43 SmallString<128> GCCDir;
44 llvm::sys::path::append(GCCDir, D.Dir, "..", D.getTargetTriple(),
45 "lib/crt0.o");
46 return llvm::sys::fs::exists(GCCDir);
47}

49/// RISCV Toolchain
50RISCVToolChain::RISCVToolChain(const Driver &D, const llvm::Triple &Triple,
51 const ArgList &Args)
52 : Generic_ELF(D, Triple, Args) {
53 GCCInstallation.init(Triple, Args);
54 if (GCCInstallation.isValid()) {
55 Multilibs = GCCInstallation.getMultilibs();
56 SelectedMultilib = GCCInstallation.getMultilib();
57 path_list &Paths = getFilePaths();
58 // Add toolchain/multilib specific file paths.
59 addMultilibsFilePaths(D, Multilibs, SelectedMultilib,
60 GCCInstallation.getInstallPath(), Paths);
61 getFilePaths().push_back(GCCInstallation.getInstallPath().str());
62 ToolChain::path_list &PPaths = getProgramPaths();
63 // Multilib cross-compiler GCC installations put ld in a triple-prefixed
64 // directory off of the parent of the GCC installation.
65 PPaths.push_back(Twine(GCCInstallation.getParentLibPath() + "/../" +
66 GCCInstallation.getTriple().str() + "/bin")
67 .str());
68 PPaths.push_back((GCCInstallation.getParentLibPath() + "/../bin").str());
69 } else {
70 getProgramPaths().push_back(D.Dir);
71 }
72 getFilePaths().push_back(computeSysRoot() + "/lib");
73}

96void RISCVToolChain::AddClangSystemIncludeArgs(const ArgList &DriverArgs,
97 ArgStringList &CC1Args) const {
98 if (DriverArgs.hasArg(options::OPT_nostdinc))
99 return;
100
101 if (!DriverArgs.hasArg(options::OPT_nostdlibinc)) {
102 SmallString<128> Dir(computeSysRoot());
103 llvm::sys::path::append(Dir, "include");
104 addSystemInclude(DriverArgs, CC1Args, Dir.str());
105 }
106}

108void RISCVToolChain::addLibStdCxxIncludePaths(
109 const llvm::opt::ArgList &DriverArgs,
110 llvm::opt::ArgStringList &CC1Args) const {
111 const GCCVersion &Version = GCCInstallation.getVersion();
112 StringRef TripleStr = GCCInstallation.getTriple().str();
113 const Multilib &Multilib = GCCInstallation.getMultilib();
114 addLibStdCXXIncludePaths(computeSysRoot() + "/include/c++/" + Version.Text,
115 "", TripleStr, "", "", Multilib.includeSuffix(), DriverArgs, CC1Args);
116}

118std::string RISCVToolChain::computeSysRoot() const {
119 if (!getDriver().SysRoot.empty())
120 return getDriver().SysRoot;
121
122 SmallString<128> SysRootDir;
123 if (GCCInstallation.isValid()) {
124 StringRef LibDir = GCCInstallation.getParentLibPath();
125 StringRef TripleStr = GCCInstallation.getTriple().str();
126 llvm::sys::path::append(SysRootDir, LibDir, "..", TripleStr);
127 } else {
128 // Use the triple as provided to the driver. Unlike the parsed triple
129 // this has not been normalized to always contain every field.
130 llvm::sys::path::append(SysRootDir, getDriver().Dir, "..",
131 getDriver().getTargetTriple());
132 }
133
134 if (!llvm::sys::fs::exists(SysRootDir))
135 return std::string();
136
137 return std::string(SysRootDir.str());
138}

140void RISCV::Linker::ConstructJob(Compilation &C, const JobAction &JA,
141 const InputInfo &Output,
142 const InputInfoList &Inputs,
143 const ArgList &Args,
144 const char *LinkingOutput) const {
145 const ToolChain &ToolChain = getToolChain();
146 const Driver &D = ToolChain.getDriver();
147 ArgStringList CmdArgs;
148
149 if (!D.SysRoot.empty())
150 CmdArgs.push_back(Args.MakeArgString("--sysroot=" + D.SysRoot));
151
152 bool IsRV64 = ToolChain.getArch() == llvm::Triple::riscv64;
153 CmdArgs.push_back("-m");
154 if (IsRV64) {
155 CmdArgs.push_back("elf64lriscv");
156 } else {
157 CmdArgs.push_back("elf32lriscv");
158 }
159
160 std::string Linker = getToolChain().GetLinkerPath();
161
162 bool WantCRTs =
163 !Args.hasArg(options::OPT_nostdlib, options::OPT_nostartfiles);
164
165 const char *crtbegin, *crtend;
166 auto RuntimeLib = ToolChain.GetRuntimeLibType(Args);
167 if (RuntimeLib == ToolChain::RLT_Libgcc) {
168 crtbegin = "crtbegin.o";
169 crtend = "crtend.o";
170 } else {
171 assert (RuntimeLib == ToolChain::RLT_CompilerRT);
172 crtbegin = ToolChain.getCompilerRTArgString(Args, "crtbegin",
173 ToolChain::FT_Object);
174 crtend = ToolChain.getCompilerRTArgString(Args, "crtend",
175 ToolChain::FT_Object);
176 }
177
178 if (WantCRTs) {
179 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath("crt0.o")));
180 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crtbegin)));
181 }
182
183 Args.AddAllArgs(CmdArgs, options::OPT_L);
184 ToolChain.AddFilePathLibArgs(Args, CmdArgs);
185 Args.AddAllArgs(CmdArgs,
186 {options::OPT_T_Group, options::OPT_e, options::OPT_s,
187 options::OPT_t, options::OPT_Z_Flag, options::OPT_r});
188
189 AddLinkerInputs(ToolChain, Inputs, Args, CmdArgs, JA);
190
191 // TODO: add C++ includes and libs if compiling C++.
192
193 if (!Args.hasArg(options::OPT_nostdlib) &&
194 !Args.hasArg(options::OPT_nodefaultlibs)) {
195 if (ToolChain.ShouldLinkCXXStdlib(Args))
196 ToolChain.AddCXXStdlibLibArgs(Args, CmdArgs);
197 CmdArgs.push_back("--start-group");
198 CmdArgs.push_back("-lc");
199 CmdArgs.push_back("-lgloss");
200 CmdArgs.push_back("--end-group");
201 AddRunTimeLibs(ToolChain, ToolChain.getDriver(), CmdArgs, Args);
202 }
203
204 if (WantCRTs)
205 CmdArgs.push_back(Args.MakeArgString(ToolChain.GetFilePath(crtend)));
206
207 CmdArgs.push_back("-o");
208 CmdArgs.push_back(Output.getFilename());
209 C.addCommand(std::make_unique<Command>(
210 JA, *this, ResponseFileSupport::AtFileCurCP(), Args.MakeArgString(Linker),
211 CmdArgs, Inputs, Output));
212}
