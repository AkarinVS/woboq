248bool ClangTableGenMain(raw_ostream &OS, RecordKeeper &Records) {
249 switch (Action) {
250 case PrintRecords:
251 OS << Records; // No argument, dump all contents
252 break;
253 case DumpJSON:
254 EmitJSON(Records, OS);
255 break;
256 case GenClangAttrClasses:
257 EmitClangAttrClass(Records, OS);
258 break;
259 case GenClangAttrParserStringSwitches:
260 EmitClangAttrParserStringSwitches(Records, OS);
261 break;
262 case GenClangAttrSubjectMatchRulesParserStringSwitches:
263 EmitClangAttrSubjectMatchRulesParserStringSwitches(Records, OS);
264 break;
265 case GenClangAttrImpl:
266 EmitClangAttrImpl(Records, OS);
267 break;
268 case GenClangAttrList:
269 EmitClangAttrList(Records, OS);
270 break;
271 case GenClangAttrSubjectMatchRuleList:
272 EmitClangAttrSubjectMatchRuleList(Records, OS);
273 break;
274 case GenClangAttrPCHRead:
275 EmitClangAttrPCHRead(Records, OS);
276 break;
277 case GenClangAttrPCHWrite:
278 EmitClangAttrPCHWrite(Records, OS);
279 break;
280 case GenClangAttrHasAttributeImpl:
281 EmitClangAttrHasAttrImpl(Records, OS);
282 break;
283 case GenClangAttrSpellingListIndex:
284 EmitClangAttrSpellingListIndex(Records, OS);
285 break;
286 case GenClangAttrASTVisitor:
287 EmitClangAttrASTVisitor(Records, OS);
288 break;
289 case GenClangAttrTemplateInstantiate:
290 EmitClangAttrTemplateInstantiate(Records, OS);
291 break;
292 case GenClangAttrParsedAttrList:
293 EmitClangAttrParsedAttrList(Records, OS);
294 break;
295 case GenClangAttrParsedAttrImpl:
296 EmitClangAttrParsedAttrImpl(Records, OS);
297 break;
298 case GenClangAttrParsedAttrKinds:
299 EmitClangAttrParsedAttrKinds(Records, OS);
300 break;
301 case GenClangAttrTextNodeDump:
302 EmitClangAttrTextNodeDump(Records, OS);
303 break;
304 case GenClangAttrNodeTraverse:
305 EmitClangAttrNodeTraverse(Records, OS);
306 break;
307 case GenClangDiagsDefs:
308 EmitClangDiagsDefs(Records, OS, ClangComponent);
309 break;
310 case GenClangDiagGroups:
311 EmitClangDiagGroups(Records, OS);
312 break;
313 case GenClangDiagsIndexName:
314 EmitClangDiagsIndexName(Records, OS);
315 break;
316 case GenClangCommentNodes:
317 EmitClangASTNodes(Records, OS, CommentNodeClassName, "");
318 break;
319 case GenClangDeclNodes:
320 EmitClangASTNodes(Records, OS, DeclNodeClassName, "Decl");
321 EmitClangDeclContext(Records, OS);
322 break;
323 case GenClangStmtNodes:
324 EmitClangASTNodes(Records, OS, StmtNodeClassName, "");
325 break;
326 case GenClangTypeNodes:
327 EmitClangTypeNodes(Records, OS);
328 break;
329 case GenClangTypeReader:
330 EmitClangTypeReader(Records, OS);
331 break;
332 case GenClangTypeWriter:
333 EmitClangTypeWriter(Records, OS);
334 break;
335 case GenClangBasicReader:
336 EmitClangBasicReader(Records, OS);
337 break;
338 case GenClangBasicWriter:
339 EmitClangBasicWriter(Records, OS);
340 break;
341 case GenClangOpcodes:
342 EmitClangOpcodes(Records, OS);
343 break;
344 case GenClangSACheckers:
345 EmitClangSACheckers(Records, OS);
346 break;
347 case GenClangCommentHTMLTags:
348 EmitClangCommentHTMLTags(Records, OS);
349 break;
350 case GenClangCommentHTMLTagsProperties:
351 EmitClangCommentHTMLTagsProperties(Records, OS);
352 break;
353 case GenClangCommentHTMLNamedCharacterReferences:
354 EmitClangCommentHTMLNamedCharacterReferences(Records, OS);
355 break;
356 case GenClangCommentCommandInfo:
357 EmitClangCommentCommandInfo(Records, OS);
358 break;
359 case GenClangCommentCommandList:
360 EmitClangCommentCommandList(Records, OS);
361 break;
362 case GenClangOpenCLBuiltins:
363 EmitClangOpenCLBuiltins(Records, OS);
364 break;
365 case GenClangSyntaxNodeList:
366 EmitClangSyntaxNodeList(Records, OS);
367 break;
368 case GenClangSyntaxNodeClasses:
369 EmitClangSyntaxNodeClasses(Records, OS);
370 break;
371 case GenArmNeon:
372 EmitNeon(Records, OS);
373 break;
374 case GenArmFP16:
375 EmitFP16(Records, OS);
376 break;
377 case GenArmBF16:
378 EmitBF16(Records, OS);
379 break;
380 case GenArmNeonSema:
381 EmitNeonSema(Records, OS);
382 break;
383 case GenArmNeonTest:
384 EmitNeonTest(Records, OS);
385 break;
386 case GenArmMveHeader:
387 EmitMveHeader(Records, OS);
388 break;
389 case GenArmMveBuiltinDef:
390 EmitMveBuiltinDef(Records, OS);
391 break;
392 case GenArmMveBuiltinSema:
393 EmitMveBuiltinSema(Records, OS);
394 break;
395 case GenArmMveBuiltinCG:
396 EmitMveBuiltinCG(Records, OS);
397 break;
398 case GenArmMveBuiltinAliases:
399 EmitMveBuiltinAliases(Records, OS);
400 break;
401 case GenArmSveHeader:
402 EmitSveHeader(Records, OS);
403 break;
404 case GenArmSveBuiltins:
405 EmitSveBuiltins(Records, OS);
406 break;
407 case GenArmSveBuiltinCG:
408 EmitSveBuiltinCG(Records, OS);
409 break;
410 case GenArmSveTypeFlags:
411 EmitSveTypeFlags(Records, OS);
412 break;
413 case GenArmSveRangeChecks:
414 EmitSveRangeChecks(Records, OS);
415 break;
416 case GenArmCdeHeader:
417 EmitCdeHeader(Records, OS);
418 break;
419 case GenArmCdeBuiltinDef:
420 EmitCdeBuiltinDef(Records, OS);
421 break;
422 case GenArmCdeBuiltinSema:
423 EmitCdeBuiltinSema(Records, OS);
424 break;
425 case GenArmCdeBuiltinCG:
426 EmitCdeBuiltinCG(Records, OS);
427 break;
428 case GenArmCdeBuiltinAliases:
429 EmitCdeBuiltinAliases(Records, OS);
430 break;
431 case GenAttrDocs:
432 EmitClangAttrDocs(Records, OS);
433 break;
434 case GenDiagDocs:
435 EmitClangDiagDocs(Records, OS);
436 break;
437 case GenOptDocs:
438 EmitClangOptDocs(Records, OS);
439 break;
440 case GenDataCollectors:
441 EmitClangDataCollectors(Records, OS);
442 break;
443 case GenTestPragmaAttributeSupportedAttributes:
444 EmitTestPragmaAttributeSupportedAttributes(Records, OS);
445 break;
446 }
447
448 return false;
449}

452int main(int argc, char **argv) {
453 sys::PrintStackTraceOnErrorSignal(argv[0]);
454 PrettyStackTraceProgram X(argc, argv);
455 cl::ParseCommandLineOptions(argc, argv);
456
457 llvm_shutdown_obj Y;
458
459 return TableGenMain(argv[0], &ClangTableGenMain);
