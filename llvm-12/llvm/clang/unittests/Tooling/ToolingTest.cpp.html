76class FindClassDeclXConsumer : public clang::ASTConsumer {
77 public:
78 FindClassDeclXConsumer(bool *FoundClassDeclX)
79 : FoundClassDeclX(FoundClassDeclX) {}
80 bool HandleTopLevelDecl(clang::DeclGroupRef GroupRef) override {
81 if (CXXRecordDecl* Record = dyn_cast<clang::CXXRecordDecl>(
82 *GroupRef.begin())) {
83 if (Record->getName() == "X") {
84 *FoundClassDeclX = true;
85 }
86 }
87 return true;
88 }
89 private:
90 bool *FoundClassDeclX;
91};
92bool FindClassDeclX(ASTUnit *AST) {
93 for (std::vector<Decl *>::iterator i = AST->top_level_begin(),
94 e = AST->top_level_end();
95 i != e; ++i) {
96 if (CXXRecordDecl* Record = dyn_cast<clang::CXXRecordDecl>(*i)) {
97 if (Record->getName() == "X") {
98 return true;
99 }
100 }
101 }
102 return false;
103}

278TEST(runToolOnCode, TestSkipFunctionBody) {
279 std::vector<std::string> Args = {"-std=c++11"};
280 std::vector<std::string> Args2 = {"-fno-delayed-template-parsing"};
281
282 EXPECT_TRUE(runToolOnCode(std::make_unique<SkipBodyAction>(),
283 "int skipMe() { an_error_here }"));
284 EXPECT_FALSE(runToolOnCode(std::make_unique<SkipBodyAction>(),
285 "int skipMeNot() { an_error_here }"));
286
287 // Test constructors with initializers
288 EXPECT_TRUE(runToolOnCodeWithArgs(
289 std::make_unique<SkipBodyAction>(),
290 "struct skipMe { skipMe() : an_error() { more error } };", Args));
291 EXPECT_TRUE(runToolOnCodeWithArgs(
292 std::make_unique<SkipBodyAction>(), "struct skipMe { skipMe(); };"
293 "skipMe::skipMe() : an_error([](){;}) { more error }",
294 Args));
295 EXPECT_TRUE(runToolOnCodeWithArgs(
296 std::make_unique<SkipBodyAction>(), "struct skipMe { skipMe(); };"
297 "skipMe::skipMe() : an_error{[](){;}} { more error }",
298 Args));
299 EXPECT_TRUE(runToolOnCodeWithArgs(
300 std::make_unique<SkipBodyAction>(),
301 "struct skipMe { skipMe(); };"
302 "skipMe::skipMe() : a<b<c>(e)>>(), f{}, g() { error }",
303 Args));
304 EXPECT_TRUE(runToolOnCodeWithArgs(
305 std::make_unique<SkipBodyAction>(), "struct skipMe { skipMe() : bases()... { error } };",
306 Args));
307
308 EXPECT_FALSE(runToolOnCodeWithArgs(
309 std::make_unique<SkipBodyAction>(), "struct skipMeNot { skipMeNot() : an_error() { } };",
310 Args));
311 EXPECT_FALSE(runToolOnCodeWithArgs(std::make_unique<SkipBodyAction>(),
312 "struct skipMeNot { skipMeNot(); };"
313 "skipMeNot::skipMeNot() : an_error() { }",
314 Args));
315
316 // Try/catch
317 EXPECT_TRUE(runToolOnCode(
318 std::make_unique<SkipBodyAction>(),
319 "void skipMe() try { an_error() } catch(error) { error };"));
320 EXPECT_TRUE(runToolOnCode(
321 std::make_unique<SkipBodyAction>(),
322 "struct S { void skipMe() try { an_error() } catch(error) { error } };"));
323 EXPECT_TRUE(
324 runToolOnCode(std::make_unique<SkipBodyAction>(),
325 "void skipMe() try { an_error() } catch(error) { error; }"
326 "catch(error) { error } catch (error) { }"));
327 EXPECT_FALSE(runToolOnCode(
328 std::make_unique<SkipBodyAction>(),
329 "void skipMe() try something;")); // don't crash while parsing
330
331 // Template
332 EXPECT_TRUE(runToolOnCode(
333 std::make_unique<SkipBodyAction>(), "template<typename T> int skipMe() { an_error_here }"
334 "int x = skipMe<int>();"));
335 EXPECT_FALSE(runToolOnCodeWithArgs(
336 std::make_unique<SkipBodyAction>(),
337 "template<typename T> int skipMeNot() { an_error_here }", Args2));
338}

395TEST(ClangToolTest, ArgumentAdjusters) {
396 FixedCompilationDatabase Compilations("/", std::vector<std::string>());
397
398 ClangTool Tool(Compilations, std::vector<std::string>(1, "/a.cc"));
399 Tool.mapVirtualFile("/a.cc", "void a() {}");
400
401 std::unique_ptr<FrontendActionFactory> Action(
402 newFrontendActionFactory<SyntaxOnlyAction>());
403
404 bool Found = false;
405 bool Ran = false;
406 ArgumentsAdjuster CheckSyntaxOnlyAdjuster =
407 [&Found, &Ran](const CommandLineArguments &Args, StringRef /*unused*/) {
408 Ran = true;
409 if (llvm::is_contained(Args, "-fsyntax-only"))
410 Found = true;
411 return Args;
412 };
413 Tool.appendArgumentsAdjuster(CheckSyntaxOnlyAdjuster);
414 Tool.run(Action.get());
415 EXPECT_TRUE(Ran);
416 EXPECT_TRUE(Found);
417
418 Ran = Found = false;
419 Tool.clearArgumentsAdjusters();
420 Tool.appendArgumentsAdjuster(CheckSyntaxOnlyAdjuster);
421 Tool.appendArgumentsAdjuster(getClangSyntaxOnlyAdjuster());
422 Tool.run(Action.get());
423 EXPECT_TRUE(Ran);
424 EXPECT_FALSE(Found);
425}

635 llvm::InitializeAllTargets();
636 for (const auto &Target : llvm::TargetRegistry::targets()) {
637 std::string Error;
638 StringRef TargetName(Target.getName());
639 if (TargetName == "x86-64")
640 TargetName = "x86_64";
641 if (llvm::TargetRegistry::lookupTarget(std::string(TargetName), Error) ==
642 &Target) {
643 return std::string(TargetName);
644 }
645 }
646 return "";
647}
648}

747TEST(runToolOnCode, TestResetDiagnostics) {
748 // This is a tool that resets the diagnostic during the compilation.
749 struct ResetDiagnosticAction : public clang::ASTFrontendAction {
750 std::unique_ptr<ASTConsumer> CreateASTConsumer(CompilerInstance &Compiler,
751 StringRef) override {
752 struct Consumer : public clang::ASTConsumer {
753 bool HandleTopLevelDecl(clang::DeclGroupRef D) override {
754 auto &Diags = (*D.begin())->getASTContext().getDiagnostics();
755 // Ignore any error
756 Diags.Reset();
757 // Disable warnings because computing the CFG might crash.
758 Diags.setIgnoreAllWarnings(true);
759 return true;
760 }
761 };
762 return std::make_unique<Consumer>();
763 }
764 };
765
766 // Should not crash
767 EXPECT_FALSE(
768 runToolOnCode(std::make_unique<ResetDiagnosticAction>(),
769 "struct Foo { Foo(int); ~Foo(); struct Fwd _fwd; };"
770 "void func() { long x; Foo f(x); }"));
771}
