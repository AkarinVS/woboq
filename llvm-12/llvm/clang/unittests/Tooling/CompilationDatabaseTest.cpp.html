27static void expectFailure(StringRef JSONDatabase, StringRef Explanation) {
28 std::string ErrorMessage;
29 EXPECT_EQ(nullptr,
30 JSONCompilationDatabase::loadFromBuffer(JSONDatabase, ErrorMessage,
31 JSONCommandLineSyntax::Gnu))
32 << "Expected an error because of: " << Explanation.str();
33}

35TEST(JSONCompilationDatabase, ErrsOnInvalidFormat) {
36 expectFailure("", "Empty database");
37 expectFailure("{", "Invalid JSON");
38 expectFailure("[[]]", "Array instead of object");
39 expectFailure("[{\"a\":[]}]", "Array instead of value");
40 expectFailure("[{\"a\":\"b\"}]", "Unknown key");
41 expectFailure("[{[]:\"\"}]", "Incorrectly typed entry");
42 expectFailure("[{}]", "Empty entry");
43 expectFailure("[{\"directory\":\"\",\"command\":\"\"}]", "Missing file");
44 expectFailure("[{\"directory\":\"\",\"file\":\"\"}]", "Missing command or arguments");
45 expectFailure("[{\"command\":\"\",\"file\":\"\"}]", "Missing directory");
46 expectFailure("[{\"directory\":\"\",\"arguments\":[]}]", "Missing file");
47 expectFailure("[{\"arguments\":\"\",\"file\":\"\"}]", "Missing directory");
48 expectFailure("[{\"directory\":\"\",\"arguments\":\"\",\"file\":\"\"}]", "Arguments not array");
49 expectFailure("[{\"directory\":\"\",\"command\":[],\"file\":\"\"}]", "Command not string");
50 expectFailure("[{\"directory\":\"\",\"arguments\":[[]],\"file\":\"\"}]",
51 "Arguments contain non-string");
52 expectFailure("[{\"output\":[]}]", "Expected strings as value.");
53}

109TEST(JSONCompilationDatabase, GetAllCompileCommands) {
110 std::string ErrorMessage;
111 EXPECT_EQ(
112 0u, getAllCompileCommands(JSONCommandLineSyntax::Gnu, "[]", ErrorMessage)
113 .size())
114 << ErrorMessage;
115
116 StringRef Directory1("//net/dir1");
117 StringRef FileName1("file1");
118 StringRef Command1("command1");
119 StringRef Output1("file1.o");
120 StringRef Directory2("//net/dir2");
121 StringRef FileName2("file2");
122 StringRef Command2("command2");
123 StringRef Output2("");
124
125 std::vector<CompileCommand> Commands = getAllCompileCommands(
126 JSONCommandLineSyntax::Gnu,
127 ("[{\"directory\":\"" + Directory1 + "\"," + "\"command\":\"" + Command1 +
128 "\","
129 "\"file\":\"" +
130 FileName1 + "\", \"output\":\"" +
131 Output1 + "\"},"
132 " {\"directory\":\"" +
133 Directory2 + "\"," + "\"command\":\"" + Command2 + "\","
134 "\"file\":\"" +
135 FileName2 + "\"}]")
136 .str(),
137 ErrorMessage);
138 EXPECT_EQ(2U, Commands.size()) << ErrorMessage;
139 EXPECT_EQ(Directory1, Commands[0].Directory) << ErrorMessage;
140 EXPECT_EQ(FileName1, Commands[0].Filename) << ErrorMessage;
141 EXPECT_EQ(Output1, Commands[0].Output) << ErrorMessage;
142 ASSERT_EQ(1u, Commands[0].CommandLine.size());
143 EXPECT_EQ(Command1, Commands[0].CommandLine[0]) << ErrorMessage;
144 EXPECT_EQ(Directory2, Commands[1].Directory) << ErrorMessage;
145 EXPECT_EQ(FileName2, Commands[1].Filename) << ErrorMessage;
146 EXPECT_EQ(Output2, Commands[1].Output) << ErrorMessage;
147 ASSERT_EQ(1u, Commands[1].CommandLine.size());
148 EXPECT_EQ(Command2, Commands[1].CommandLine[0]) << ErrorMessage;
149
150 // Check that order is preserved.
151 Commands = getAllCompileCommands(
152 JSONCommandLineSyntax::Gnu,
153 ("[{\"directory\":\"" + Directory2 + "\"," + "\"command\":\"" + Command2 +
154 "\","
155 "\"file\":\"" +
156 FileName2 + "\"},"
157 " {\"directory\":\"" +
158 Directory1 + "\"," + "\"command\":\"" + Command1 + "\","
159 "\"file\":\"" +
160 FileName1 + "\"}]")
161 .str(),
162 ErrorMessage);
163 EXPECT_EQ(2U, Commands.size()) << ErrorMessage;
164 EXPECT_EQ(Directory2, Commands[0].Directory) << ErrorMessage;
165 EXPECT_EQ(FileName2, Commands[0].Filename) << ErrorMessage;
166 ASSERT_EQ(1u, Commands[0].CommandLine.size());
167 EXPECT_EQ(Command2, Commands[0].CommandLine[0]) << ErrorMessage;
168 EXPECT_EQ(Directory1, Commands[1].Directory) << ErrorMessage;
169 EXPECT_EQ(FileName1, Commands[1].Filename) << ErrorMessage;
170 ASSERT_EQ(1u, Commands[1].CommandLine.size());
171 EXPECT_EQ(Command1, Commands[1].CommandLine[0]) << ErrorMessage;
172}

174static CompileCommand findCompileArgsInJsonDatabase(StringRef FileName,
175 std::string JSONDatabase,
176 std::string &ErrorMessage) {
177 std::unique_ptr<CompilationDatabase> Database(
178 JSONCompilationDatabase::loadFromBuffer(JSONDatabase, ErrorMessage,
179 JSONCommandLineSyntax::Gnu));
180 if (!Database)
181 return CompileCommand();
182 // Overwrite the string to verify we're not reading from it later.
183 JSONDatabase.assign(JSONDatabase.size(), '*');
184 std::vector<CompileCommand> Commands = Database->getCompileCommands(FileName);
185 EXPECT_LE(Commands.size(), 1u);
186 if (Commands.empty())
187 return CompileCommand();
188 return Commands[0];
189}

387 {"gomacc clang++ foo.c", "clang++ foo.c"},
388 {"sccache clang++ foo.c", "clang++ foo.c"},
389 {"ccache gcc foo.c", "gcc foo.c"},

677struct MemCDB : public CompilationDatabase {
678 using EntryMap = llvm::StringMap<SmallVector<CompileCommand, 1>>;
679 EntryMap Entries;
680 MemCDB(const EntryMap &E) : Entries(E) {}
681
682 std::vector<CompileCommand> getCompileCommands(StringRef F) const override {
683 auto Ret = Entries.lookup(F);
684 return {Ret.begin(), Ret.end()};
685 }
686
687 std::vector<std::string> getAllFiles() const override {
688 std::vector<std::string> Result;
689 for (const auto &Entry : Entries)
690 Result.push_back(std::string(Entry.first()));
691 return Result;
692 }
693};

695class MemDBTest : public ::testing::Test {
696protected:
697 // Adds an entry to the underlying compilation database.
698 // A flag is injected: -D <File>, so the command used can be identified.
699 void add(StringRef File, StringRef Clang, StringRef Flags) {
700 SmallVector<StringRef, 8> Argv = {Clang, File, "-D", File};
701 llvm::SplitString(Flags, Argv);
702
703 SmallString<32> Dir;
704 llvm::sys::path::system_temp_directory(false, Dir);
705
706 Entries[path(File)].push_back(
707 {Dir, path(File), {Argv.begin(), Argv.end()}, "foo.o"});
708 }
709 void add(StringRef File, StringRef Flags = "") { add(File, "clang", Flags); }
710
711 // Turn a unix path fragment (foo/bar.h) into a native path (C:\tmp\foo\bar.h)
712 std::string path(llvm::SmallString<32> File) {
713 llvm::SmallString<32> Dir;
714 llvm::sys::path::system_temp_directory(false, Dir);
715 llvm::sys::path::native(File);
716 llvm::SmallString<64> Result;
717 llvm::sys::path::append(Result, Dir, File);
718 return std::string(Result.str());
719 }
720
721 MemCDB::EntryMap Entries;
722};

724class InterpolateTest : public MemDBTest {
725protected:
726 // Look up the command from a relative path, and return it in string form.
727 // The input file is not included in the returned command.
728 std::string getCommand(llvm::StringRef F) {
729 auto Results =
730 inferMissingCompileCommands(std::make_unique<MemCDB>(Entries))
731 ->getCompileCommands(path(F));
732 if (Results.empty())
733 return "none";
734 // drop the input file argument, so tests don't have to deal with path().
735 EXPECT_EQ(Results[0].CommandLine.back(), path(F))
736 << "Last arg should be the file";
737 Results[0].CommandLine.pop_back();
738 return llvm::join(Results[0].CommandLine, " ");
739 }
740
741 // Parse the file whose command was used out of the Heuristic string.
742 std::string getProxy(llvm::StringRef F) {
743 auto Results =
744 inferMissingCompileCommands(std::make_unique<MemCDB>(Entries))
745 ->getCompileCommands(path(F));
746 if (Results.empty())
747 return "none";
748 StringRef Proxy = Results.front().Heuristic;
749 if (!Proxy.consume_front("inferred from "))
750 return "";
751 // We have a proxy file, convert back to a unix relative path.
752 // This is a bit messy, but we do need to test these strings somehow...
753 llvm::SmallString<32> TempDir;
754 llvm::sys::path::system_temp_directory(false, TempDir);
755 Proxy.consume_front(TempDir);
756 Proxy.consume_front(llvm::sys::path::get_separator());
757 llvm::SmallString<32> Result = Proxy;
758 llvm::sys::path::native(Result, llvm::sys::path::Style::posix);
759 return std::string(Result.str());
760 }
761};

763TEST_F(InterpolateTest, Nearby) {
764 add("dir/foo.cpp");
765 add("dir/bar.cpp");
766 add("an/other/foo.cpp");
767
768 // great: dir and name both match (prefix or full, case insensitive)
769 EXPECT_EQ(getProxy("dir/f.cpp"), "dir/foo.cpp");
770 EXPECT_EQ(getProxy("dir/FOO.cpp"), "dir/foo.cpp");
771 // no name match. prefer matching dir, break ties by alpha
772 EXPECT_EQ(getProxy("dir/a.cpp"), "dir/bar.cpp");
773 // an exact name match beats one segment of directory match
774 EXPECT_EQ(getProxy("some/other/bar.h"), "dir/bar.cpp");
775 // two segments of directory match beat a prefix name match
776 EXPECT_EQ(getProxy("an/other/b.cpp"), "an/other/foo.cpp");
777 // if nothing matches at all, we still get the closest alpha match
778 EXPECT_EQ(getProxy("below/some/obscure/path.cpp"), "an/other/foo.cpp");
779}

781TEST_F(InterpolateTest, Language) {
782 add("dir/foo.cpp", "-std=c++17");
783 add("dir/bar.c", "");
784 add("dir/baz.cee", "-x c");
785 add("dir/aux.cpp", "-std=c++17 -x objective-c++");
786
787 // .h is ambiguous, so we add explicit language flags
788 EXPECT_EQ(getCommand("foo.h"),
789 "clang -D dir/foo.cpp -x c++-header -std=c++17");
790 // Same thing if we have no extension. (again, we treat as header).
791 EXPECT_EQ(getCommand("foo"), "clang -D dir/foo.cpp -x c++-header -std=c++17");
792 // and invalid extensions.
793 EXPECT_EQ(getCommand("foo.cce"),
794 "clang -D dir/foo.cpp -x c++-header -std=c++17");
795 // and don't add -x if the inferred language is correct.
796 EXPECT_EQ(getCommand("foo.hpp"), "clang -D dir/foo.cpp -std=c++17");
797 // respect -x if it's already there.
798 EXPECT_EQ(getCommand("baz.h"), "clang -D dir/baz.cee -x c-header");
799 // prefer a worse match with the right extension.
800 EXPECT_EQ(getCommand("foo.c"), "clang -D dir/bar.c");
801 Entries.erase(path(StringRef("dir/bar.c")));
802 // Now we transfer across languages, so drop -std too.
803 EXPECT_EQ(getCommand("foo.c"), "clang -D dir/foo.cpp");
804 // Prefer -x over -std when overriding language.
805 EXPECT_EQ(getCommand("aux.h"),
806 "clang -D dir/aux.cpp -x objective-c++-header -std=c++17");
807}
