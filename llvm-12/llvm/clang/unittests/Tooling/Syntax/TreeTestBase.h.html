28namespace clang {
29namespace syntax {
30class SyntaxTreeTest : public ::testing::Test,
31 public ::testing::WithParamInterface<TestClangConfig> {
32protected:
33 // Build a syntax tree for the code.
34 TranslationUnit *buildTree(StringRef Code,
35 const TestClangConfig &ClangConfig);
36
37 /// Finds the deepest node in the tree that covers exactly \p R.
38 /// FIXME: implement this efficiently and move to public syntax tree API.
39 syntax::Node *nodeByRange(llvm::Annotations::Range R, syntax::Node *Root);
40
41 // Data fields.
42 IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions();
43 IntrusiveRefCntPtr<DiagnosticsEngine> Diags =
44 new DiagnosticsEngine(new DiagnosticIDs, DiagOpts.get());
45 IntrusiveRefCntPtr<llvm::vfs::InMemoryFileSystem> FS =
46 new llvm::vfs::InMemoryFileSystem;
47 IntrusiveRefCntPtr<FileManager> FileMgr =
48 new FileManager(FileSystemOptions(), FS);
49 IntrusiveRefCntPtr<SourceManager> SourceMgr =
50 new SourceManager(*Diags, *FileMgr);
51 std::shared_ptr<CompilerInvocation> Invocation;
52 // Set after calling buildTree().
53 std::unique_ptr<syntax::TokenBuffer> TB;
54 std::unique_ptr<syntax::Arena> Arena;
55};

57std::vector<TestClangConfig> allTestClangConfigs();

60 if (arg.getRole() == R)
61 return true;
62 *result_listener << "role is " << llvm::to_string(arg.getRole());
63 return false;
64}
