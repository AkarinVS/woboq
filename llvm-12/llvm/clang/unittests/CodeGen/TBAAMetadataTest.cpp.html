51TEST(TBAAMetadataTest, BasicTypes) {
52 const char TestProgram[] = R"**(
53 void func(char *CP, short *SP, int *IP, long long *LP, void **VPP,
54 int **IPP) {
55 *CP = 4;
56 *SP = 11;
57 *IP = 601;
58 *LP = 604;
59 *VPP = CP;
60 *IPP = IP;
61 }
62 )**";
63
64 clang::LangOptions LO;
65 TBAATestCompiler Compiler(LO, TBAATestCompiler::getCommonCodeGenOpts());
66 Compiler.init(TestProgram);
67 const BasicBlock *BB = Compiler.compile();
68
69 const Instruction *I = match(BB,
70 MInstruction(Instruction::Store,
71 MConstInt(4, 8),
72 MMTuple(
73 OmnipotentCharC,
74 MSameAs(0),
75 MConstInt(0))));
76 ASSERT_TRUE(I);
77
78 I = matchNext(I,
79 MInstruction(Instruction::Store,
80 MConstInt(11, 16),
81 MMTuple(
82 MMTuple(
83 MMString("short"),
84 OmnipotentCharC,
85 MConstInt(0)),
86 MSameAs(0),
87 MConstInt(0))));
88 ASSERT_TRUE(I);
89
90 I = matchNext(I,
91 MInstruction(Instruction::Store,
92 MConstInt(601, 32),
93 MMTuple(
94 MMTuple(
95 MMString("int"),
96 OmnipotentCharC,

98 MSameAs(0),
99 MConstInt(0))));

101
