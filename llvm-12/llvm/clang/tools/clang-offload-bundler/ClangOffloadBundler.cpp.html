138/// Generic file handler interface.
139class FileHandler {
140public:
141 struct BundleInfo {
142 StringRef BundleID;
143 };
144
145 FileHandler() {}
146
147 virtual ~FileHandler() {}
148
149 /// Update the file handler with information from the header of the bundled
150 /// file.
151 virtual Error ReadHeader(MemoryBuffer &Input) = 0;
152
153 /// Read the marker of the next bundled to be read in the file. The bundle
154 /// name is returned if there is one in the file, or `None` if there are no
155 /// more bundles to be read.
156 virtual Expected<Optional<StringRef>>
157 ReadBundleStart(MemoryBuffer &Input) = 0;
158
159 /// Read the marker that closes the current bundle.
160 virtual Error ReadBundleEnd(MemoryBuffer &Input) = 0;
161
162 /// Read the current bundle and write the result into the stream \a OS.
163 virtual Error ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) = 0;
164
165 /// Write the header of the bundled file to \a OS based on the information
166 /// gathered from \a Inputs.
167 virtual Error WriteHeader(raw_fd_ostream &OS,
168 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) = 0;
169
170 /// Write the marker that initiates a bundle for the triple \a TargetTriple to
171 /// \a OS.
172 virtual Error WriteBundleStart(raw_fd_ostream &OS,
173 StringRef TargetTriple) = 0;
174
175 /// Write the marker that closes a bundle for the triple \a TargetTriple to \a
176 /// OS.
177 virtual Error WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) = 0;
178
179 /// Write the bundle from \a Input into \a OS.
180 virtual Error WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) = 0;
181
182 /// List bundle IDs in \a Input.
183 virtual Error listBundleIDs(MemoryBuffer &Input) {
184 if (Error Err = ReadHeader(Input))
185 return Err;
186
187 return forEachBundle(Input, [&](const BundleInfo &Info) -> Error {
188 llvm::outs() << Info.BundleID << '\n';
189 Error Err = listBundleIDsCallback(Input, Info);
190 if (Err)
191 return Err;
192 return Error::success();
193 });
194 }
195
196 /// For each bundle in \a Input, do \a Func.
197 Error forEachBundle(MemoryBuffer &Input,
198 std::function<Error(const BundleInfo &)> Func) {
199 while (true) {
200 Expected<Optional<StringRef>> CurTripleOrErr = ReadBundleStart(Input);
201 if (!CurTripleOrErr)
202 return CurTripleOrErr.takeError();
203
204 // No more bundles.
205 if (!*CurTripleOrErr)
206 break;
207
208 StringRef CurTriple = **CurTripleOrErr;
209 assert(!CurTriple.empty());
210
211 BundleInfo Info{CurTriple};
212 if (Error Err = Func(Info))
213 return Err;
214 }
215 return Error::success();
216 }
217
218protected:
219 virtual Error listBundleIDsCallback(MemoryBuffer &Input,
220 const BundleInfo &Info) {
221 return Error::success();
222 }
223};

225/// Handler for binary files. The bundled file will have the following format
226/// (all integers are stored in little-endian format):
227///
228/// "OFFLOAD_BUNDLER_MAGIC_STR" (ASCII encoding of the string)
229///
230/// NumberOfOffloadBundles (8-byte integer)
231///
232/// OffsetOfBundle1 (8-byte integer)
233/// SizeOfBundle1 (8-byte integer)
234/// NumberOfBytesInTripleOfBundle1 (8-byte integer)
235/// TripleOfBundle1 (byte length defined before)
236///
237/// ...
238///
239/// OffsetOfBundleN (8-byte integer)
240/// SizeOfBundleN (8-byte integer)
241/// NumberOfBytesInTripleOfBundleN (8-byte integer)
242/// TripleOfBundleN (byte length defined before)
243///
244/// Bundle1
245/// ...
246/// BundleN
247
248/// Read 8-byte integers from a buffer in little-endian format.
249static uint64_t Read8byteIntegerFromBuffer(StringRef Buffer, size_t pos) {
250 uint64_t Res = 0;
251 const char *Data = Buffer.data();
252
253 for (unsigned i = 0; i < 8; ++i) {
254 Res <<= 8;
255 uint64_t Char = (uint64_t)Data[pos + 7 - i];
256 Res |= 0xffu & Char;
257 }
258 return Res;
259}
260
261/// Write 8-byte integers to a buffer in little-endian format.
262static void Write8byteIntegerToBuffer(raw_fd_ostream &OS, uint64_t Val) {
263 for (unsigned i = 0; i < 8; ++i) {
264 char Char = (char)(Val & 0xffu);
265 OS.write(&Char, 1);
266 Val >>= 8;
267 }
268}

270class BinaryFileHandler final : public FileHandler {
271 /// Information about the bundles extracted from the header.
272 struct BinaryBundleInfo final : public BundleInfo {
273 /// Size of the bundle.
274 uint64_t Size = 0u;
275 /// Offset at which the bundle starts in the bundled file.
276 uint64_t Offset = 0u;
277
278 BinaryBundleInfo() {}
279 BinaryBundleInfo(uint64_t Size, uint64_t Offset)
280 : Size(Size), Offset(Offset) {}
281 };
282
283 /// Map between a triple and the corresponding bundle information.
284 StringMap<BinaryBundleInfo> BundlesInfo;
285
286 /// Iterator for the bundle information that is being read.
287 StringMap<BinaryBundleInfo>::iterator CurBundleInfo;
288 StringMap<BinaryBundleInfo>::iterator NextBundleInfo;
289
290 /// Current bundle target to be written.
291 std::string CurWriteBundleTarget;
292
293public:
294 BinaryFileHandler() : FileHandler() {}
295
296 ~BinaryFileHandler() final {}
297
298 Error ReadHeader(MemoryBuffer &Input) final {
299 StringRef FC = Input.getBuffer();
300
301 // Initialize the current bundle with the end of the container.
302 CurBundleInfo = BundlesInfo.end();
303
304 // Check if buffer is smaller than magic string.
305 size_t ReadChars = sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
306 if (ReadChars > FC.size())
307 return Error::success();
308
309 // Check if no magic was found.
310 StringRef Magic(FC.data(), sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1);
311 if (!Magic.equals(OFFLOAD_BUNDLER_MAGIC_STR))
312 return Error::success();
313
314 // Read number of bundles.
315 if (ReadChars + 8 > FC.size())
316 return Error::success();
317
318 uint64_t NumberOfBundles = Read8byteIntegerFromBuffer(FC, ReadChars);
319 ReadChars += 8;
320
321 // Read bundle offsets, sizes and triples.
322 for (uint64_t i = 0; i < NumberOfBundles; ++i) {
323
324 // Read offset.
325 if (ReadChars + 8 > FC.size())
326 return Error::success();
327
328 uint64_t Offset = Read8byteIntegerFromBuffer(FC, ReadChars);
329 ReadChars += 8;
330
331 // Read size.
332 if (ReadChars + 8 > FC.size())
333 return Error::success();
334
335 uint64_t Size = Read8byteIntegerFromBuffer(FC, ReadChars);
336 ReadChars += 8;
337
338 // Read triple size.
339 if (ReadChars + 8 > FC.size())
340 return Error::success();
341
342 uint64_t TripleSize = Read8byteIntegerFromBuffer(FC, ReadChars);
343 ReadChars += 8;
344
345 // Read triple.
346 if (ReadChars + TripleSize > FC.size())
347 return Error::success();
348
349 StringRef Triple(&FC.data()[ReadChars], TripleSize);
350 ReadChars += TripleSize;
351
352 // Check if the offset and size make sense.
353 if (!Offset || Offset + Size > FC.size())
354 return Error::success();
355
356 assert(BundlesInfo.find(Triple) == BundlesInfo.end() &&
357 "Triple is duplicated??");
358 BundlesInfo[Triple] = BinaryBundleInfo(Size, Offset);
359 }
360 // Set the iterator to where we will start to read.
361 CurBundleInfo = BundlesInfo.end();
362 NextBundleInfo = BundlesInfo.begin();
363 return Error::success();
364 }
365
366 Expected<Optional<StringRef>> ReadBundleStart(MemoryBuffer &Input) final {
367 if (NextBundleInfo == BundlesInfo.end())
368 return None;
369 CurBundleInfo = NextBundleInfo++;
370 return CurBundleInfo->first();
371 }
372
373 Error ReadBundleEnd(MemoryBuffer &Input) final {
374 assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
375 return Error::success();
376 }
377
378 Error ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
379 assert(CurBundleInfo != BundlesInfo.end() && "Invalid reader info!");
380 StringRef FC = Input.getBuffer();
381 OS.write(FC.data() + CurBundleInfo->second.Offset,
382 CurBundleInfo->second.Size);
383 return Error::success();
384 }
385
386 Error WriteHeader(raw_fd_ostream &OS,
387 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
388 // Compute size of the header.
389 uint64_t HeaderSize = 0;
390
391 HeaderSize += sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1;
392 HeaderSize += 8; // Number of Bundles
393
394 for (auto &T : TargetNames) {
395 HeaderSize += 3 * 8; // Bundle offset, Size of bundle and size of triple.
396 HeaderSize += T.size(); // The triple.
397 }
398
399 // Write to the buffer the header.
400 OS << OFFLOAD_BUNDLER_MAGIC_STR;
401
402 Write8byteIntegerToBuffer(OS, TargetNames.size());
403
404 unsigned Idx = 0;
405 for (auto &T : TargetNames) {
406 MemoryBuffer &MB = *Inputs[Idx++];
407 HeaderSize = alignTo(HeaderSize, BundleAlignment);
408 // Bundle offset.
409 Write8byteIntegerToBuffer(OS, HeaderSize);
410 // Size of the bundle (adds to the next bundle's offset)
411 Write8byteIntegerToBuffer(OS, MB.getBufferSize());
412 BundlesInfo[T] = BinaryBundleInfo(MB.getBufferSize(), HeaderSize);
413 HeaderSize += MB.getBufferSize();
414 // Size of the triple
415 Write8byteIntegerToBuffer(OS, T.size());
416 // Triple
417 OS << T;
418 }
419 return Error::success();
420 }
421
422 Error WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {
423 CurWriteBundleTarget = TargetTriple.str();
424 return Error::success();
425 }
426
427 Error WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
428 return Error::success();
429 }
430
431 Error WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
432 auto BI = BundlesInfo[CurWriteBundleTarget];
433 OS.seek(BI.Offset);
434 OS.write(Input.getBufferStart(), Input.getBufferSize());
435 return Error::success();
436 }
437};

441// This class implements a list of temporary files that are removed upon
442// object destruction.
443class TempFileHandlerRAII {
444public:
445 ~TempFileHandlerRAII() {
446 for (const auto &File : Files)
447 sys::fs::remove(File);
448 }
449
450 // Creates temporary file with given contents.
451 Expected<StringRef> Create(Optional<ArrayRef<char>> Contents) {
452 SmallString<128u> File;
453 if (std::error_code EC =
454 sys::fs::createTemporaryFile("clang-offload-bundler", "tmp", File))
455 return createFileError(File, EC);
456 Files.push_front(File);
457
458 if (Contents) {
459 std::error_code EC;
460 raw_fd_ostream OS(File, EC);
461 if (EC)
462 return createFileError(File, EC);
463 OS.write(Contents->data(), Contents->size());
464 }
465 return Files.front();
466 }
467
468private:
469 std::forward_list<SmallString<128u>> Files;
470};

474/// Handler for object files. The bundles are organized by sections with a
475/// designated name.
476///
477/// To unbundle, we just copy the contents of the designated section.
478class ObjectFileHandler final : public FileHandler {
479
480 /// The object file we are currently dealing with.
481 std::unique_ptr<ObjectFile> Obj;
482
483 /// Return the input file contents.
484 StringRef getInputFileContents() const { return Obj->getData(); }
485
486 /// Return bundle name (<kind>-<triple>) if the provided section is an offload
487 /// section.
488 static Expected<Optional<StringRef>> IsOffloadSection(SectionRef CurSection) {
489 Expected<StringRef> NameOrErr = CurSection.getName();
490 if (!NameOrErr)
491 return NameOrErr.takeError();
492
493 // If it does not start with the reserved suffix, just skip this section.
494 if (!NameOrErr->startswith(OFFLOAD_BUNDLER_MAGIC_STR))
495 return None;
496
497 // Return the triple that is right after the reserved prefix.
498 return NameOrErr->substr(sizeof(OFFLOAD_BUNDLER_MAGIC_STR) - 1);
499 }
500
501 /// Total number of inputs.
502 unsigned NumberOfInputs = 0;
503
504 /// Total number of processed inputs, i.e, inputs that were already
505 /// read from the buffers.
506 unsigned NumberOfProcessedInputs = 0;
507
508 /// Iterator of the current and next section.
509 section_iterator CurrentSection;
510 section_iterator NextSection;
511
512public:
513 ObjectFileHandler(std::unique_ptr<ObjectFile> ObjIn)
514 : FileHandler(), Obj(std::move(ObjIn)),
515 CurrentSection(Obj->section_begin()),
516 NextSection(Obj->section_begin()) {}
517
518 ~ObjectFileHandler() final {}
519
520 Error ReadHeader(MemoryBuffer &Input) final { return Error::success(); }
521
522 Expected<Optional<StringRef>> ReadBundleStart(MemoryBuffer &Input) final {
523 while (NextSection != Obj->section_end()) {
524 CurrentSection = NextSection;
525 ++NextSection;
526
527 // Check if the current section name starts with the reserved prefix. If
528 // so, return the triple.
529 Expected<Optional<StringRef>> TripleOrErr =
530 IsOffloadSection(*CurrentSection);
531 if (!TripleOrErr)
532 return TripleOrErr.takeError();
533 if (*TripleOrErr)
534 return **TripleOrErr;
535 }
536 return None;
537 }
538
539 Error ReadBundleEnd(MemoryBuffer &Input) final { return Error::success(); }
540
541 Error ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
542 Expected<StringRef> ContentOrErr = CurrentSection->getContents();
543 if (!ContentOrErr)
544 return ContentOrErr.takeError();
545 StringRef Content = *ContentOrErr;
546
547 // Copy fat object contents to the output when extracting host bundle.
548 if (Content.size() == 1u && Content.front() == 0)
549 Content = StringRef(Input.getBufferStart(), Input.getBufferSize());
550
551 OS.write(Content.data(), Content.size());
552 return Error::success();
553 }
554
555 Error WriteHeader(raw_fd_ostream &OS,
556 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
557 assert(HostInputIndex != ~0u && "Host input index not defined.");
558
559 // Record number of inputs.
560 NumberOfInputs = Inputs.size();
561 return Error::success();
562 }
563
564 Error WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {
565 ++NumberOfProcessedInputs;
566 return Error::success();
567 }
568
569 Error WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
570 assert(NumberOfProcessedInputs <= NumberOfInputs &&
571 "Processing more inputs that actually exist!");
572 assert(HostInputIndex != ~0u && "Host input index not defined.");
573
574 // If this is not the last output, we don't have to do anything.
575 if (NumberOfProcessedInputs != NumberOfInputs)
576 return Error::success();
577
578 // We will use llvm-objcopy to add target objects sections to the output
579 // fat object. These sections should have 'exclude' flag set which tells
580 // link editor to remove them from linker inputs when linking executable or
581 // shared library. llvm-objcopy currently does not support adding new
582 // section and changing flags for the added section in one invocation, and
583 // because of that we have to run it two times. First run adds sections and
584 // the second changes flags.
585 // TODO: change it to one run once llvm-objcopy starts supporting that.
586
587 // Find llvm-objcopy in order to create the bundle binary.
588 ErrorOr<std::string> Objcopy = sys::findProgramByName(
589 "llvm-objcopy", sys::path::parent_path(BundlerExecutable));
590 if (!Objcopy)
591 Objcopy = sys::findProgramByName("llvm-objcopy");
592 if (!Objcopy)
593 return createStringError(Objcopy.getError(),
594 "unable to find 'llvm-objcopy' in path");
595
596 // We write to the output file directly. So, we close it and use the name
597 // to pass down to llvm-objcopy.
598 OS.close();
599
600 // Temporary files that need to be removed.
601 TempFileHandlerRAII TempFiles;
602
603 // Create an intermediate temporary file to save object after the first
604 // llvm-objcopy run.
605 Expected<StringRef> IntermediateObjOrErr = TempFiles.Create(None);
606 if (!IntermediateObjOrErr)
607 return IntermediateObjOrErr.takeError();
608 StringRef IntermediateObj = *IntermediateObjOrErr;
609
610 // Compose llvm-objcopy command line for add target objects' sections.
611 BumpPtrAllocator Alloc;
612 StringSaver SS{Alloc};
613 SmallVector<StringRef, 8u> ObjcopyArgs{"llvm-objcopy"};
614 for (unsigned I = 0; I < NumberOfInputs; ++I) {
615 StringRef InputFile = InputFileNames[I];
616 if (I == HostInputIndex) {
617 // Special handling for the host bundle. We do not need to add a
618 // standard bundle for the host object since we are going to use fat
619 // object as a host object. Therefore use dummy contents (one zero byte)
620 // when creating section for the host bundle.
621 Expected<StringRef> TempFileOrErr = TempFiles.Create(ArrayRef<char>(0));
622 if (!TempFileOrErr)
623 return TempFileOrErr.takeError();
624 InputFile = *TempFileOrErr;
625 }
626
627 ObjcopyArgs.push_back(SS.save(Twine("--add-section=") +
628 OFFLOAD_BUNDLER_MAGIC_STR + TargetNames[I] +
629 "=" + InputFile));
630 }
631 ObjcopyArgs.push_back(InputFileNames[HostInputIndex]);
632 ObjcopyArgs.push_back(IntermediateObj);
633
634 if (Error Err = executeObjcopy(*Objcopy, ObjcopyArgs))
635 return Err;
636
637 // And run llvm-objcopy for the second time to update section flags.
638 ObjcopyArgs.resize(1);
639 for (unsigned I = 0; I < NumberOfInputs; ++I)
640 ObjcopyArgs.push_back(SS.save(Twine("--set-section-flags=") +
641 OFFLOAD_BUNDLER_MAGIC_STR + TargetNames[I] +
642 "=readonly,exclude"));
643 ObjcopyArgs.push_back(IntermediateObj);
644 ObjcopyArgs.push_back(OutputFileNames.front());
645
646 if (Error Err = executeObjcopy(*Objcopy, ObjcopyArgs))
647 return Err;
648
649 return Error::success();
650 }
651
652 Error WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
653 return Error::success();
654 }
655
656private:
657 static Error executeObjcopy(StringRef Objcopy, ArrayRef<StringRef> Args) {
658 // If the user asked for the commands to be printed out, we do that
659 // instead of executing it.
660 if (PrintExternalCommands) {
661 errs() << "\"" << Objcopy << "\"";
662 for (StringRef Arg : drop_begin(Args, 1))
663 errs() << " \"" << Arg << "\"";
664 errs() << "\n";
665 } else {
666 if (sys::ExecuteAndWait(Objcopy, Args))
667 return createStringError(inconvertibleErrorCode(),
668 "'llvm-objcopy' tool failed");
669 }
670 return Error::success();
671 }
672};

674/// Handler for text files. The bundled file will have the following format.
675///
676/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__START__ triple"
677/// Bundle 1
678/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__END__ triple"
679/// ...
680/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__START__ triple"
681/// Bundle N
682/// "Comment OFFLOAD_BUNDLER_MAGIC_STR__END__ triple"
683class TextFileHandler final : public FileHandler {
684 /// String that begins a line comment.
685 StringRef Comment;
686
687 /// String that initiates a bundle.
688 std::string BundleStartString;
689
690 /// String that closes a bundle.
691 std::string BundleEndString;
692
693 /// Number of chars read from input.
694 size_t ReadChars = 0u;
695
696protected:
697 Error ReadHeader(MemoryBuffer &Input) final { return Error::success(); }
698
699 Expected<Optional<StringRef>> ReadBundleStart(MemoryBuffer &Input) final {
700 StringRef FC = Input.getBuffer();
701
702 // Find start of the bundle.
703 ReadChars = FC.find(BundleStartString, ReadChars);
704 if (ReadChars == FC.npos)
705 return None;
706
707 // Get position of the triple.
708 size_t TripleStart = ReadChars = ReadChars + BundleStartString.size();
709
710 // Get position that closes the triple.
711 size_t TripleEnd = ReadChars = FC.find("\n", ReadChars);
712 if (TripleEnd == FC.npos)
713 return None;
714
715 // Next time we read after the new line.
716 ++ReadChars;
717
718 return StringRef(&FC.data()[TripleStart], TripleEnd - TripleStart);
719 }
720
721 Error ReadBundleEnd(MemoryBuffer &Input) final {
722 StringRef FC = Input.getBuffer();
723
724 // Read up to the next new line.
725 assert(FC[ReadChars] == '\n' && "The bundle should end with a new line.");
726
727 size_t TripleEnd = ReadChars = FC.find("\n", ReadChars + 1);
728 if (TripleEnd != FC.npos)
729 // Next time we read after the new line.
730 ++ReadChars;
731
732 return Error::success();
733 }
734
735 Error ReadBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
736 StringRef FC = Input.getBuffer();
737 size_t BundleStart = ReadChars;
738
739 // Find end of the bundle.
740 size_t BundleEnd = ReadChars = FC.find(BundleEndString, ReadChars);
741
742 StringRef Bundle(&FC.data()[BundleStart], BundleEnd - BundleStart);
743 OS << Bundle;
744
745 return Error::success();
746 }
747
748 Error WriteHeader(raw_fd_ostream &OS,
749 ArrayRef<std::unique_ptr<MemoryBuffer>> Inputs) final {
750 return Error::success();
751 }
752
753 Error WriteBundleStart(raw_fd_ostream &OS, StringRef TargetTriple) final {
754 OS << BundleStartString << TargetTriple << "\n";
755 return Error::success();
756 }
757
758 Error WriteBundleEnd(raw_fd_ostream &OS, StringRef TargetTriple) final {
759 OS << BundleEndString << TargetTriple << "\n";
760 return Error::success();
761 }
762
763 Error WriteBundle(raw_fd_ostream &OS, MemoryBuffer &Input) final {
764 OS << Input.getBuffer();
765 return Error::success();
766 }

790/// Return an appropriate object file handler. We use the specific object
791/// handler if we know how to deal with that format, otherwise we use a default
792/// binary file handler.
793static std::unique_ptr<FileHandler>
794CreateObjectFileHandler(MemoryBuffer &FirstInput) {
795 // Check if the input file format is one that we know how to deal with.
796 Expected<std::unique_ptr<Binary>> BinaryOrErr = createBinary(FirstInput);
797
798 // We only support regular object files. If failed to open the input as a
799 // known binary or this is not an object file use the default binary handler.
800 if (errorToBool(BinaryOrErr.takeError()) || !isa<ObjectFile>(*BinaryOrErr))
801 return std::make_unique<BinaryFileHandler>();
802
803 // Otherwise create an object file handler. The handler will be owned by the
804 // client of this function.
805 return std::make_unique<ObjectFileHandler>(
806 std::unique_ptr<ObjectFile>(cast<ObjectFile>(BinaryOrErr->release())));
807}

809/// Return an appropriate handler given the input files and options.
810static Expected<std::unique_ptr<FileHandler>>
811CreateFileHandler(MemoryBuffer &FirstInput) {
812 if (FilesType == "i")
813 return std::make_unique<TextFileHandler>(/*Comment=*/"//");
814 if (FilesType == "ii")
815 return std::make_unique<TextFileHandler>(/*Comment=*/"//");
816 if (FilesType == "cui")
817 return std::make_unique<TextFileHandler>(/*Comment=*/"//");
818 // TODO: `.d` should be eventually removed once `-M` and its variants are
819 // handled properly in offload compilation.
820 if (FilesType == "d")
821 return std::make_unique<TextFileHandler>(/*Comment=*/"#");
822 if (FilesType == "ll")
823 return std::make_unique<TextFileHandler>(/*Comment=*/";");
824 if (FilesType == "bc")
825 return std::make_unique<BinaryFileHandler>();
826 if (FilesType == "s")
827 return std::make_unique<TextFileHandler>(/*Comment=*/"#");
828 if (FilesType == "o")
829 return CreateObjectFileHandler(FirstInput);
830 if (FilesType == "gch")
831 return std::make_unique<BinaryFileHandler>();
832 if (FilesType == "ast")
833 return std::make_unique<BinaryFileHandler>();
834
835 return createStringError(errc::invalid_argument,
836 "'" + FilesType + "': invalid file type specified");
837}

839/// Bundle the files. Return true if an error was found.
840static Error BundleFiles() {
841 std::error_code EC;
842
843 // Create output file.
844 raw_fd_ostream OutputFile(OutputFileNames.front(), EC, sys::fs::OF_None);
845 if (EC)
846 return createFileError(OutputFileNames.front(), EC);
847
848 // Open input files.
849 SmallVector<std::unique_ptr<MemoryBuffer>, 8u> InputBuffers;
850 InputBuffers.reserve(InputFileNames.size());
851 for (auto &I : InputFileNames) {
852 ErrorOr<std::unique_ptr<MemoryBuffer>> CodeOrErr =
853 MemoryBuffer::getFileOrSTDIN(I);
854 if (std::error_code EC = CodeOrErr.getError())
855 return createFileError(I, EC);
856 InputBuffers.emplace_back(std::move(*CodeOrErr));
857 }
858
859 // Get the file handler. We use the host buffer as reference.
860 assert(HostInputIndex != ~0u && "Host input index undefined??");
861 Expected<std::unique_ptr<FileHandler>> FileHandlerOrErr =
862 CreateFileHandler(*InputBuffers[HostInputIndex]);
863 if (!FileHandlerOrErr)
864 return FileHandlerOrErr.takeError();
865
866 std::unique_ptr<FileHandler> &FH = *FileHandlerOrErr;
867 assert(FH);
868
869 // Write header.
870 if (Error Err = FH->WriteHeader(OutputFile, InputBuffers))
871 return Err;
872
873 // Write all bundles along with the start/end markers. If an error was found
874 // writing the end of the bundle component, abort the bundle writing.
875 auto Input = InputBuffers.begin();
876 for (auto &Triple : TargetNames) {
877 if (Error Err = FH->WriteBundleStart(OutputFile, Triple))
878 return Err;
879 if (Error Err = FH->WriteBundle(OutputFile, **Input))
880 return Err;
881 if (Error Err = FH->WriteBundleEnd(OutputFile, Triple))
882 return Err;
883 ++Input;
884 }
885 return Error::success();
886}

890 // Open Input file.
891 ErrorOr<std::unique_ptr<MemoryBuffer>> CodeOrErr =
892 MemoryBuffer::getFileOrSTDIN(InputFileName);
893 if (std::error_code EC = CodeOrErr.getError())

905 assert(FH);

907}

909// Unbundle the files. Return true if an error was found.
910static Error UnbundleFiles() {
911 // Open Input file.
912 ErrorOr<std::unique_ptr<MemoryBuffer>> CodeOrErr =
913 MemoryBuffer::getFileOrSTDIN(InputFileNames.front());
914 if (std::error_code EC = CodeOrErr.getError())
915 return createFileError(InputFileNames.front(), EC);
916
917 MemoryBuffer &Input = **CodeOrErr;
918
919 // Select the right files handler.
920 Expected<std::unique_ptr<FileHandler>> FileHandlerOrErr =
921 CreateFileHandler(Input);
922 if (!FileHandlerOrErr)
923 return FileHandlerOrErr.takeError();
924
925 std::unique_ptr<FileHandler> &FH = *FileHandlerOrErr;
926 assert(FH);
927
928 // Read the header of the bundled file.
929 if (Error Err = FH->ReadHeader(Input))
930 return Err;
931
932 // Create a work list that consist of the map triple/output file.
933 StringMap<StringRef> Worklist;
934 auto Output = OutputFileNames.begin();
935 for (auto &Triple : TargetNames) {
936 Worklist[Triple] = *Output;
937 ++Output;
938 }
939
940 // Read all the bundles that are in the work list. If we find no bundles we
941 // assume the file is meant for the host target.
942 bool FoundHostBundle = false;
943 while (!Worklist.empty()) {
944 Expected<Optional<StringRef>> CurTripleOrErr = FH->ReadBundleStart(Input);
945 if (!CurTripleOrErr)
946 return CurTripleOrErr.takeError();
947
948 // We don't have more bundles.
949 if (!*CurTripleOrErr)
950 break;
951
952 StringRef CurTriple = **CurTripleOrErr;
953 assert(!CurTriple.empty());
954
955 auto Output = Worklist.find(CurTriple);
956 // The file may have more bundles for other targets, that we don't care
957 // about. Therefore, move on to the next triple
958 if (Output == Worklist.end())
959 continue;
960
961 // Check if the output file can be opened and copy the bundle to it.
962 std::error_code EC;
963 raw_fd_ostream OutputFile(Output->second, EC, sys::fs::OF_None);
964 if (EC)
965 return createFileError(Output->second, EC);
966 if (Error Err = FH->ReadBundle(OutputFile, Input))
967 return Err;
968 if (Error Err = FH->ReadBundleEnd(Input))
969 return Err;
970 Worklist.erase(Output);
971
972 // Record if we found the host bundle.
973 if (hasHostKind(CurTriple))
974 FoundHostBundle = true;
975 }
976
977 if (!AllowMissingBundles && !Worklist.empty()) {
978 std::string ErrMsg = "Can't find bundles for";
979 std::set<StringRef> Sorted;
980 for (auto &E : Worklist)
981 Sorted.insert(E.first());
982 unsigned I = 0;
983 unsigned Last = Sorted.size() - 1;
984 for (auto &E : Sorted) {
985 if (I != 0 && Last > 1)
986 ErrMsg += ",";
987 ErrMsg += " ";
988 if (I == Last && I != 0)
989 ErrMsg += "and ";
990 ErrMsg += E.str();
991 ++I;
992 }
993 return createStringError(inconvertibleErrorCode(), ErrMsg);
994 }
995
996 // If no bundles were found, assume the input file is the host bundle and
997 // create empty files for the remaining targets.
998 if (Worklist.size() == TargetNames.size()) {
999 for (auto &E : Worklist) {
1000 std::error_code EC;
1001 raw_fd_ostream OutputFile(E.second, EC, sys::fs::OF_None);
1002 if (EC)
1003 return createFileError(E.second, EC);
1004
1005 // If this entry has a host kind, copy the input file to the output file.
1006 if (hasHostKind(E.first()))
1007 OutputFile.write(Input.getBufferStart(), Input.getBufferSize());
1008 }
1009 return Error::success();
1010 }
1011
1012 // If we found elements, we emit an error if none of those were for the host
1013 // in case host bundle name was provided in command line.
1014 if (!FoundHostBundle && HostInputIndex != ~0u)
1015 return createStringError(inconvertibleErrorCode(),
1016 "Can't find bundle for the host target");
1017
1018 // If we still have any elements in the worklist, create empty files for them.
1019 for (auto &E : Worklist) {
1020 std::error_code EC;
1021 raw_fd_ostream OutputFile(E.second, EC, sys::fs::OF_None);
1022 if (EC)
1023 return createFileError(E.second, EC);
1024 }
1025
1026 return Error::success();
1027}

1033int main(int argc, const char **argv) {
1034 sys::PrintStackTraceOnErrorSignal(argv[0]);
1035
1036 cl::HideUnrelatedOptions(ClangOffloadBundlerCategory);
1037 cl::SetVersionPrinter(PrintVersion);
1038 cl::ParseCommandLineOptions(
1039 argc, argv,
1040 "A tool to bundle several input files of the specified type <type> \n"
1041 "referring to the same source file but different targets into a single \n"
1042 "one. The resulting file can also be unbundled into different files by \n"
1043 "this tool if -unbundle is provided.\n");
1044
1045 if (Help) {
1046 cl::PrintHelpMessage();
1047 return 0;
1048 }
1049
1050 auto reportError = [argv](Error E) {
1051 logAllUnhandledErrors(std::move(E), WithColor::error(errs(), argv[0]));
1052 exit(1);
1053 };
1054
1055 auto doWork = [&](std::function<llvm::Error()> Work) {
1056 // Save the current executable directory as it will be useful to find other
1057 // tools.
1058 BundlerExecutable = argv[0];
1059 if (!llvm::sys::fs::exists(BundlerExecutable))
1060 BundlerExecutable =
1061 sys::fs::getMainExecutable(argv[0], &BundlerExecutable);
1062
1063 if (llvm::Error Err = Work()) {
1064 reportError(std::move(Err));
1065 }
1066 };
1067
1068 if (ListBundleIDs) {
1069 if (Unbundle) {
1070 reportError(
1071 createStringError(errc::invalid_argument,
1072 "-unbundle and -list cannot be used together"));
1073 }
1074 if (InputFileNames.size() != 1) {
1075 reportError(createStringError(errc::invalid_argument,
1076 "only one input file supported for -list"));
1077 }
1078 if (OutputFileNames.size()) {
1079 reportError(createStringError(errc::invalid_argument,
1080 "-outputs option is invalid for -list"));
1081 }
1082 if (TargetNames.size()) {
1083 reportError(createStringError(errc::invalid_argument,
1084 "-targets option is invalid for -list"));
1085 }
1086
1087 doWork([]() { return ListBundleIDsInFile(InputFileNames.front()); });
1088 return 0;
1089 }
1090
1091 if (OutputFileNames.getNumOccurrences() == 0) {
1092 reportError(createStringError(
1093 errc::invalid_argument,
1094 "for the --outputs option: must be specified at least once!"));
1095 }
1096 if (TargetNames.getNumOccurrences() == 0) {
1097 reportError(createStringError(
1098 errc::invalid_argument,
1099 "for the --targets option: must be specified at least once!"));
1100 }
1101 if (Unbundle) {
1102 if (InputFileNames.size() != 1) {
1103 reportError(createStringError(
1104 errc::invalid_argument,
1105 "only one input file supported in unbundling mode"));
1106 }
1107 if (OutputFileNames.size() != TargetNames.size()) {
1108 reportError(createStringError(errc::invalid_argument,
1109 "number of output files and targets should "
1110 "match in unbundling mode"));
1111 }
1112 } else {
1113 if (OutputFileNames.size() != 1) {
1114 reportError(createStringError(
1115 errc::invalid_argument,
1116 "only one output file supported in bundling mode"));
1117 }
1118 if (InputFileNames.size() != TargetNames.size()) {
1119 reportError(createStringError(
1120 errc::invalid_argument,
1121 "number of input files and targets should match in bundling mode"));
1122 }
1123 }
1124
1125 // Verify that the offload kinds and triples are known. We also check that we
1126 // have exactly one host target.
1127 unsigned Index = 0u;
1128 unsigned HostTargetNum = 0u;
1129 llvm::DenseSet<StringRef> ParsedTargets;
1130 for (StringRef Target : TargetNames) {
1131 if (ParsedTargets.contains(Target)) {
1132 reportError(createStringError(errc::invalid_argument,
1133 "Duplicate targets are not allowed"));
1134 }
1135 ParsedTargets.insert(Target);
1136
1137 StringRef Kind;
1138 StringRef Triple;
1139 getOffloadKindAndTriple(Target, Kind, Triple);
1140
1141 bool KindIsValid = !Kind.empty();
1142 KindIsValid = KindIsValid && StringSwitch<bool>(Kind)
1143 .Case("host", true)
1144 .Case("openmp", true)
1145 .Case("hip", true)
1146 .Default(false);
1147
1148 bool TripleIsValid = !Triple.empty();
1149 llvm::Triple T(Triple);
1150 TripleIsValid &= T.getArch() != Triple::UnknownArch;
1151
1152 if (!KindIsValid || !TripleIsValid) {
1153 SmallVector<char, 128u> Buf;
1154 raw_svector_ostream Msg(Buf);
1155 Msg << "invalid target '" << Target << "'";
1156 if (!KindIsValid)
1157 Msg << ", unknown offloading kind '" << Kind << "'";
1158 if (!TripleIsValid)
1159 Msg << ", unknown target triple '" << Triple << "'";
1160 reportError(createStringError(errc::invalid_argument, Msg.str()));
1161 }
1162
1163 if (KindIsValid && Kind == "host") {
1164 ++HostTargetNum;
1165 // Save the index of the input that refers to the host.
1166 HostInputIndex = Index;
1167 }
1168
1169 ++Index;
1170 }
1171
1172 // Host triple is not really needed for unbundling operation, so do not
1173 // treat missing host triple as error if we do unbundling.
1174 if ((Unbundle && HostTargetNum > 1) || (!Unbundle && HostTargetNum != 1)) {
1175 reportError(createStringError(errc::invalid_argument,
1176 "expecting exactly one host target but got " +
1177 Twine(HostTargetNum)));
1178 }
1179
1180 doWork([]() { return Unbundle ? UnbundleFiles() : BundleFiles(); });
1181 return 0;
1182}
