139CXCursor cxcursor::MakeCXCursor(const Decl *D, CXTranslationUnit TU,
140 SourceRange RegionOfInterest,
141 bool FirstInDeclGroup) {
142 assert(D && TU && "Invalid arguments!");
143
144 CXCursorKind K = getCursorKindForDecl(D);
145
146 if (K == CXCursor_ObjCClassMethodDecl ||
147 K == CXCursor_ObjCInstanceMethodDecl) {
148 int SelectorIdIndex = -1;
149 // Check if cursor points to a selector id.
150 if (RegionOfInterest.isValid() &&
151 RegionOfInterest.getBegin() == RegionOfInterest.getEnd()) {
152 SmallVector<SourceLocation, 16> SelLocs;
153 cast<ObjCMethodDecl>(D)->getSelectorLocs(SelLocs);
154 SmallVectorImpl<SourceLocation>::iterator I =
155 llvm::find(SelLocs, RegionOfInterest.getBegin());
156 if (I != SelLocs.end())
157 SelectorIdIndex = I - SelLocs.begin();
158 }
159 CXCursor C = {K,
160 SelectorIdIndex,
161 {D, (void *)(intptr_t)(FirstInDeclGroup ? 1 : 0), TU}};
162 return C;
163 }
164
165 CXCursor C = {K, 0, {D, (void *)(intptr_t)(FirstInDeclGroup ? 1 : 0), TU}};
166 return C;
167}

1177CXCursor cxcursor::getTypeRefCursor(CXCursor cursor) {
1178 if (cursor.kind != CXCursor_CallExpr)
1179 return cursor;
1180
1181 if (cursor.xdata == 0)
1182 return cursor;
1183
1184 const Expr *E = getCursorExpr(cursor);
1185 TypeSourceInfo *Type = nullptr;
1186 if (const CXXUnresolvedConstructExpr *UnCtor =
1187 dyn_cast<CXXUnresolvedConstructExpr>(E)) {
1188 Type = UnCtor->getTypeSourceInfo();
1189 } else if (const CXXTemporaryObjectExpr *Tmp =
1190 dyn_cast<CXXTemporaryObjectExpr>(E)) {
1191 Type = Tmp->getTypeSourceInfo();
1192 }
1193
1194 if (!Type)
1195 return cursor;
1196
1197 CXTranslationUnit TU = getCursorTU(cursor);
1198 QualType Ty = Type->getType();
1199 TypeLoc TL = Type->getTypeLoc();
1200 SourceLocation Loc = TL.getBeginLoc();
1201
1202 if (const ElaboratedType *ElabT = Ty->getAs<ElaboratedType>()) {
1203 Ty = ElabT->getNamedType();
1204 ElaboratedTypeLoc ElabTL = TL.castAs<ElaboratedTypeLoc>();
1205 Loc = ElabTL.getNamedTypeLoc().getBeginLoc();
1206 }
1207
1208 if (const TypedefType *Typedef = Ty->getAs<TypedefType>())
1209 return MakeCursorTypeRef(Typedef->getDecl(), Loc, TU);
1210 if (const TagType *Tag = Ty->getAs<TagType>())
1211 return MakeCursorTypeRef(Tag->getDecl(), Loc, TU);
1212 if (const TemplateTypeParmType *TemplP = Ty->getAs<TemplateTypeParmType>())
1213 return MakeCursorTypeRef(TemplP->getDecl(), Loc, TU);
1214
1215 return cursor;
1216}

1317enum CXGetTemplateArgumentStatus {
1318 /** The operation completed successfully */
1319 CXGetTemplateArgumentStatus_Success = 0,
1320
1321 /** The specified cursor did not represent a FunctionDecl. */
1322 CXGetTemplateArgumentStatus_CursorNotFunctionDecl = -1,
1323
1324 /** The specified cursor was not castable to a FunctionDecl. */
1325 CXGetTemplateArgumentStatus_BadFunctionDeclCast = -2,
1326
1327 /** A NULL FunctionTemplateSpecializationInfo was retrieved. */
1328 CXGetTemplateArgumentStatus_NullTemplSpecInfo = -3,
1329
1330 /** An invalid (OOB) argument index was specified */
1331 CXGetTemplateArgumentStatus_InvalidIndex = -4
1332};

1334static int clang_Cursor_getTemplateArgument(CXCursor C, unsigned I,
1335 TemplateArgument *TA) {
1336 if (clang_getCursorKind(C) != CXCursor_FunctionDecl) {
1337 return CXGetTemplateArgumentStatus_CursorNotFunctionDecl;
1338 }
1339
1340 const FunctionDecl *FD =
1341 llvm::dyn_cast_or_null<clang::FunctionDecl>(getCursorDecl(C));
1342 if (!FD) {
1343 return CXGetTemplateArgumentStatus_BadFunctionDeclCast;
1344 }
1345
1346 const FunctionTemplateSpecializationInfo *SpecInfo =
1347 FD->getTemplateSpecializationInfo();
1348 if (!SpecInfo) {
1349 return CXGetTemplateArgumentStatus_NullTemplSpecInfo;
1350 }
1351
1352 if (I >= SpecInfo->TemplateArguments->size()) {
1353 return CXGetTemplateArgumentStatus_InvalidIndex;
1354 }
1355
1356 *TA = SpecInfo->TemplateArguments->get(I);
1357 return 0;
1358}

1361 unsigned I) {

1364 return CXTemplateArgumentKind_Invalid;
1365 }

1368 case TemplateArgument::Null:
1369 return CXTemplateArgumentKind_Null;
1370 case TemplateArgument::Type:
1371 return CXTemplateArgumentKind_Type;
1372 case TemplateArgument::Declaration:
1373 return CXTemplateArgumentKind_Declaration;
1374 case TemplateArgument::NullPtr:
1375 return CXTemplateArgumentKind_NullPtr;
1376 case TemplateArgument::Integral:
1377 return CXTemplateArgumentKind_Integral;
1378 case TemplateArgument::Template:
1379 return CXTemplateArgumentKind_Template;
1380 case TemplateArgument::TemplateExpansion:
1381 return CXTemplateArgumentKind_TemplateExpansion;
1382 case TemplateArgument::Expression:
1383 return CXTemplateArgumentKind_Expression;
1384 case TemplateArgument::Pack:
1385 return CXTemplateArgumentKind_Pack;
1386 }
1387
1388 return CXTemplateArgumentKind_Invalid;
1389}

1554void clang_getOverriddenCursors(CXCursor cursor, CXCursor **overridden,
1555 unsigned *num_overridden) {
1556 if (overridden)
1557 *overridden = nullptr;
1558 if (num_overridden)
1559 *num_overridden = 0;
1560
1561 CXTranslationUnit TU = cxcursor::getCursorTU(cursor);
1562
1563 if (!overridden || !num_overridden || !TU)
1564 return;
1565
1566 if (!clang_isDeclaration(cursor.kind))
1567 return;
1568
1569 OverridenCursorsPool &pool =
1570 *static_cast<OverridenCursorsPool *>(TU->OverridenCursorsPool);
1571
1572 OverridenCursorsPool::CursorVec *Vec = nullptr;
1573
1574 if (!pool.AvailableCursors.empty()) {
1575 Vec = pool.AvailableCursors.back();
1576 pool.AvailableCursors.pop_back();
1577 } else {
1578 Vec = new OverridenCursorsPool::CursorVec();
1579 pool.AllCursors.push_back(Vec);
1580 }
1581
1582 // Clear out the vector, but don't free the memory contents. This
1583 // reduces malloc() traffic.
1584 Vec->clear();
1585
1586 // Use the first entry to contain a back reference to the vector.
1587 // This is a complete hack.
1588 CXCursor backRefCursor = MakeCXCursorInvalid(CXCursor_InvalidFile, TU);
1589 backRefCursor.data[0] = Vec;
1590 assert(cxcursor::getCursorTU(backRefCursor) == TU);
1591 Vec->push_back(backRefCursor);
1592
1593 // Get the overridden cursors.
1594 cxcursor::getOverriddenCursors(cursor, *Vec);
1595
1596 // Did we get any overridden cursors? If not, return Vec to the pool
1597 // of available cursor vectors.
1598 if (Vec->size() == 1) {
1599 pool.AvailableCursors.push_back(Vec);
1600 return;
1601 }
1602
1603 // Now tell the caller about the overridden cursors.
1604 assert(Vec->size() > 1);
1605 *overridden = &((*Vec)[1]);
1606 *num_overridden = Vec->size() - 1;
1607}

1609void clang_disposeOverriddenCursors(CXCursor *overridden) {
1610 if (!overridden)
1611 return;
1612
1613 // Use pointer arithmetic to get back the first faux entry
1614 // which has a back-reference to the TU and the vector.
1615 --overridden;
1616 OverridenCursorsPool::CursorVec *Vec =
1617 static_cast<OverridenCursorsPool::CursorVec *>(
1618 const_cast<void *>(overridden->data[0]));
1619 CXTranslationUnit TU = getCursorTU(*overridden);
1620
1621 assert(Vec && TU);
1622
1623 OverridenCursorsPool &pool =
1624 *static_cast<OverridenCursorsPool *>(TU->OverridenCursorsPool);
1625
1626 pool.AvailableCursors.push_back(Vec);
1627}

1629int clang_Cursor_isDynamicCall(CXCursor C) {
1630 const Expr *E = nullptr;
1631 if (clang_isExpression(C.kind))
1632 E = getCursorExpr(C);
1633 if (!E)
1634 return 0;
1635
1636 if (const ObjCMessageExpr *MsgE = dyn_cast<ObjCMessageExpr>(E)) {
1637 if (MsgE->getReceiverKind() != ObjCMessageExpr::Instance)
1638 return false;
1639 if (auto *RecE = dyn_cast<ObjCMessageExpr>(
1640 MsgE->getInstanceReceiver()->IgnoreParenCasts())) {
1641 if (RecE->getMethodFamily() == OMF_alloc)
1642 return false;
1643 }
1644 return true;
1645 }
1646
1647 if (auto *PropRefE = dyn_cast<ObjCPropertyRefExpr>(E)) {
1648 return !PropRefE->isSuperReceiver();
1649 }
1650
1651 const MemberExpr *ME = nullptr;
1652 if (isa<MemberExpr>(E))
1653 ME = cast<MemberExpr>(E);
1654 else if (const CallExpr *CE = dyn_cast<CallExpr>(E))
1655 ME = dyn_cast_or_null<MemberExpr>(CE->getCallee());
1656
1657 if (ME) {
1658 if (const CXXMethodDecl *MD =
1659 dyn_cast_or_null<CXXMethodDecl>(ME->getMemberDecl()))
1660 return MD->isVirtual() &&
1661 ME->performsVirtualDispatch(
1662 cxcursor::getCursorContext(C).getLangOpts());
1663 }
1664
1665 return 0;
1666}

1668CXType clang_Cursor_getReceiverType(CXCursor C) {
1669 CXTranslationUnit TU = cxcursor::getCursorTU(C);
1670 const Expr *E = nullptr;
1671 if (clang_isExpression(C.kind))
1672 E = getCursorExpr(C);
1673
1674 if (const ObjCMessageExpr *MsgE = dyn_cast_or_null<ObjCMessageExpr>(E))
1675 return cxtype::MakeCXType(MsgE->getReceiverType(), TU);
1676
1677 if (auto *PropRefE = dyn_cast<ObjCPropertyRefExpr>(E)) {
1678 return cxtype::MakeCXType(
1679 PropRefE->getReceiverType(cxcursor::getCursorContext(C)), TU);
1680 }
1681
1682 const MemberExpr *ME = nullptr;
1683 if (isa<MemberExpr>(E))
1684 ME = cast<MemberExpr>(E);
1685 else if (const CallExpr *CE = dyn_cast<CallExpr>(E))
1686 ME = dyn_cast_or_null<MemberExpr>(CE->getCallee());
1687
1688 if (ME) {
1689 if (dyn_cast_or_null<CXXMethodDecl>(ME->getMemberDecl())) {
1690 auto receiverTy = ME->getBase()->IgnoreImpCasts()->getType();
1691 return cxtype::MakeCXType(receiverTy, TU);
1692 }
1693 }
1694
1695 return cxtype::MakeCXType(QualType(), TU);
