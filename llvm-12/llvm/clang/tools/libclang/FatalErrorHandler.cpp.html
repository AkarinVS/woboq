10#include "clang-c/FatalErrorHandler.h"
11#include "llvm/Support/ErrorHandling.h"
12#include <stdlib.h>
13
14static void aborting_fatal_error_handler(void *, const std::string &reason,
15 bool) {
16 // Write the result out to stderr avoiding errs() because raw_ostreams can
17 // call report_fatal_error.
18 fprintf(stderr, "LIBCLANG FATAL ERROR: %s\n", reason.c_str());
19 ::abort();
20}
21
22extern "C" {
23void clang_install_aborting_llvm_fatal_error_handler(void) {
24 llvm::remove_fatal_error_handler();
25 llvm::install_fatal_error_handler(aborting_fatal_error_handler, nullptr);
26}
27
28void clang_uninstall_llvm_fatal_error_handler(void) {
29 llvm::remove_fatal_error_handler();
30}
31}
