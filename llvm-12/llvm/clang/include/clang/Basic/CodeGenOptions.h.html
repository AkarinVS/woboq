30/// Bitfields of CodeGenOptions, split out from CodeGenOptions to ensure
31/// that this large collection of bitfields is a trivial class type.
32class CodeGenOptionsBase {
33 friend class CompilerInvocation;
34
35public:
36#define CODEGENOPT(Name, Bits, Default) unsigned Name : Bits;
37#define ENUM_CODEGENOPT(Name, Type, Bits, Default)
38#include "clang/Basic/CodeGenOptions.def"
39
40protected:
41#define CODEGENOPT(Name, Bits, Default)
42#define ENUM_CODEGENOPT(Name, Type, Bits, Default) unsigned Name : Bits;
43#include "clang/Basic/CodeGenOptions.def"
44};

46/// CodeGenOptions - Track various options which control how the code
47/// is optimized and passed to the backend.
48class CodeGenOptions : public CodeGenOptionsBase {
49public:
50 enum InliningMethod {
51 NormalInlining, // Use the standard function inlining pass.
52 OnlyHintInlining, // Inline only (implicitly) hinted functions.
53 OnlyAlwaysInlining // Only run the always inlining pass.
54 };

56 enum VectorLibrary {
57 NoLibrary, // Don't use any vector library.
58 Accelerate, // Use the Accelerate framework.
59 LIBMVEC, // GLIBC vector math library.
60 MASSV, // IBM MASS vector library.
61 SVML // Intel short vector math library.
62 };
63
64 enum ObjCDispatchMethodKind {
65 Legacy = 0,
66 NonLegacy = 1,
67 Mixed = 2

71 GeneralDynamicTLSModel,

73 InitialExecTLSModel,
74 LocalExecTLSModel

77 /// Clang versions with different platform ABI conformance.
78 enum class ClangABI {
79 /// Attempt to be ABI-compatible with code generated by Clang 3.8.x
80 /// (SVN r257626). This causes <1 x long long> to be passed in an
81 /// integer register instead of an SSE register on x64_64.
82 Ver3_8,
83
84 /// Attempt to be ABI-compatible with code generated by Clang 4.0.x
85 /// (SVN r291814). This causes move operations to be ignored when
86 /// determining whether a class type can be passed or returned directly.
87 Ver4,
88
89 /// Conform to the underlying platform's C and C++ ABIs as closely
90 /// as we can.
91 Latest
92 };
93
94 enum StructReturnConventionKind {
95 SRCK_Default, // No special option was passed.
96 SRCK_OnStack, // Small structs on the stack (-fpcc-struct-return).
97 SRCK_InRegs // Small structs in registers (-freg-struct-return).
98 };
99
100 enum ProfileInstrKind {
101 ProfileNone, // Profile instrumentation is turned off.
102 ProfileClangInstr, // Clang instrumentation to generate execution counts
103 // to use with PGO.
104 ProfileIRInstr, // IR level PGO instrumentation in LLVM.
105 ProfileCSIRInstr, // IR level PGO context sensitive instrumentation in LLVM.
106 };
107
108 enum EmbedBitcodeKind {
109 Embed_Off, // No embedded bitcode.
110 Embed_All, // Embed both bitcode and commandline in the output.
111 Embed_Bitcode, // Embed just the bitcode in the output.
112 Embed_Marker // Embed a marker as a placeholder for bitcode.
113 };
114
115 // This field stores one of the allowed values for the option
116 // -fbasic-block-sections=. The allowed values with this option are:
117 // {"labels", "all", "list=<file>", "none"}.
118 //
119 // "labels": Only generate basic block symbols (labels) for all basic
120 // blocks, do not generate unique sections for basic blocks.
121 // Use the machine basic block id in the symbol name to
122 // associate profile info from virtual address to machine
123 // basic block.
124 // "all" : Generate basic block sections for all basic blocks.
125 // "list=<file>": Generate basic block sections for a subset of basic blocks.
126 // The functions and the machine basic block ids are specified
127 // in the file.
128 // "none": Disable sections/labels for basic blocks.
129 std::string BBSections;
130
131 // If set, override the default value of MCAsmInfo::BinutilsVersion. If
132 // DisableIntegratedAS is specified, the assembly output will consider GNU as
133 // support. "none" means that all ELF features can be used, regardless of
134 // binutils support.
135 std::string BinutilsVersion;
136
137 enum class FramePointerKind {
138 None, // Omit all frame pointers.
139 NonLeaf, // Keep non-leaf frame pointers.
140 All, // Keep all frame pointers.
141 };
142
143 enum FiniteLoopsKind {
144 Language, // Not specified, use language standard.
145 Always, // All loops are assumed to be finite.
146 Never, // No loop is assumed to be finite.
147 };

198 struct BitcodeFileToLink {
199 /// The filename of the bitcode file to link in.
200 std::string Filename;
201 /// If true, we set attributes functions in the bitcode library according to
202 /// our CodeGenOptions, much as we set attrs on functions that we generate
203 /// ourselves.
204 bool PropagateAttrs = false;
205 /// If true, we use LLVM module internalizer.
206 bool Internalize = false;
207 /// Bitwise combination of llvm::Linker::Flags, passed to the LLVM linker.
208 unsigned LinkFlags = 0;
209 };

287 /// Regular expression to select optimizations for which we should enable
288 /// optimization remarks. Transformation passes whose name matches this
289 /// expression (and support this feature), will emit a diagnostic
290 /// whenever they perform a transformation. This is enabled by the
291 /// -Rpass=regexp flag.
292 std::shared_ptr<llvm::Regex> OptimizationRemarkPattern;
293
294 /// Regular expression to select optimizations for which we should enable
295 /// missed optimization remarks. Transformation passes whose name matches this
296 /// expression (and support this feature), will emit a diagnostic
297 /// whenever they tried but failed to perform a transformation. This is
298 /// enabled by the -Rpass-missed=regexp flag.
299 std::shared_ptr<llvm::Regex> OptimizationRemarkMissedPattern;
300
301 /// Regular expression to select optimizations for which we should enable
302 /// optimization analyses. Transformation passes whose name matches this
303 /// expression (and support this feature), will emit a diagnostic
304 /// whenever they want to explain why they decided to apply or not apply
305 /// a given transformation. This is enabled by the -Rpass-analysis=regexp
306 /// flag.
307 std::shared_ptr<llvm::Regex> OptimizationRemarkAnalysisPattern;

359 /// Executable and command-line used to create a given CompilerInvocation.
360 /// Most of the time this will be the full -cc1 command.
361 const char *Argv0 = nullptr;
362 ArrayRef<const char *> CommandLineArgs;
363
364 /// The minimum hotness value a diagnostic needs in order to be included in
365 /// optimization diagnostics.
366 ///
367 /// The threshold is an Optional value, which maps to one of the 3 states:
368 /// 1. 0 => threshold disabled. All remarks will be printed.
369 /// 2. positive int => manual threshold by user. Remarks with hotness exceed
370 /// threshold will be printed.
371 /// 3. None => 'auto' threshold by user. The actual value is not
372 /// available at command line, but will be synced with
373 /// hotness threshold from profile summary during
374 /// compilation.
375 ///
376 /// If threshold option is not specified, it is disabled by default.
377 Optional<uint64_t> DiagnosticsHotnessThreshold = 0;
378
379public:
380 // Define accessors/mutators for code generation options of enumeration type.
381#define CODEGENOPT(Name, Bits, Default)
382#define ENUM_CODEGENOPT(Name, Type, Bits, Default) \
383 Type get##Name() const { return static_cast<Type>(Name); } \
384 void set##Name(Type Value) { Name = static_cast<unsigned>(Value); }
385#include "clang/Basic/CodeGenOptions.def"

387 CodeGenOptions();

389 /// Is this a libc/libm function that is no longer recognized as a
390 /// builtin because a -fno-builtin-* option has been specified?
391 bool isNoBuiltinFunc(const char *Name) const;
392
393 const std::vector<std::string> &getNoBuiltinFuncs() const {
394 return NoBuiltinFuncs;
395 }
396
397 /// Check if Clang profile instrumenation is on.
398 bool hasProfileClangInstr() const {
399 return getProfileInstr() == ProfileClangInstr;
400 }
401
402 /// Check if IR level profile instrumentation is on.
403 bool hasProfileIRInstr() const {
404 return getProfileInstr() == ProfileIRInstr;
405 }
406
407 /// Check if CS IR level profile instrumentation is on.
408 bool hasProfileCSIRInstr() const {
409 return getProfileInstr() == ProfileCSIRInstr;
410 }
411
412 /// Check if Clang profile use is on.
413 bool hasProfileClangUse() const {
414 return getProfileUse() == ProfileClangInstr;
415 }
416
417 /// Check if IR level profile use is on.
418 bool hasProfileIRUse() const {
419 return getProfileUse() == ProfileIRInstr ||
420 getProfileUse() == ProfileCSIRInstr;
421 }
422
423 /// Check if CSIR profile use is on.
424 bool hasProfileCSIRUse() const { return getProfileUse() == ProfileCSIRInstr; }
425
426 /// Check if type and variable info should be emitted.
427 bool hasReducedDebugInfo() const {
428 return getDebugInfo() >= codegenoptions::DebugInfoConstructor;
429 }
430
431 /// Check if maybe unused type info should be emitted.
432 bool hasMaybeUnusedDebugInfo() const {
433 return getDebugInfo() >= codegenoptions::UnusedTypeInfo;
434 }
435};
