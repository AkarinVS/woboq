89static inline bool IsNVIDIAGpuArch(CudaArch A) {
90 return A >= CudaArch::SM_20 && A < CudaArch::GFX600;
91}
92
93static inline bool IsAMDGpuArch(CudaArch A) {
94 return A >= CudaArch::GFX600 && A < CudaArch::LAST;
95}
