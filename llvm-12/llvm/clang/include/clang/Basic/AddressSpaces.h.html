15#ifndef LLVM_CLANG_BASIC_ADDRESSSPACES_H
16#define LLVM_CLANG_BASIC_ADDRESSSPACES_H
17
18#include <cassert>
19
20namespace clang {
21
22/// Defines the address space values used by the address space qualifier
23/// of QualType.
24///
25enum class LangAS : unsigned {
26 // The default value 0 is the value used in QualType for the situation
27 // where there is no address space qualifier.
28 Default = 0,
29
30 // OpenCL specific address spaces.
31 // In OpenCL each l-value must have certain non-default address space, each
32 // r-value must have no address space (i.e. the default address space). The
33 // pointee of a pointer must have non-default address space.
34 opencl_global,
35 opencl_local,
36 opencl_constant,
37 opencl_private,
38 opencl_generic,
39 opencl_global_device,
40 opencl_global_host,
41
42 // CUDA specific address spaces.
43 cuda_device,
44 cuda_constant,
45 cuda_shared,
46
47 // Pointer size and extension address spaces.
48 ptr32_sptr,
49 ptr32_uptr,
50 ptr64,
51
52 // This denotes the count of language-specific address spaces and also
53 // the offset added to the target-specific address spaces, which are usually
54 // specified by address space attributes __attribute__(address_space(n))).
55 FirstTargetAddressSpace
56};
57
58/// The type of a lookup table which maps from language-specific address spaces
59/// to target-specific ones.
60using LangASMap = unsigned[(unsigned)LangAS::FirstTargetAddressSpace];
61
62/// \return whether \p AS is a target-specific address space rather than a
63/// clang AST address space
64inline bool isTargetAddressSpace(LangAS AS) {
65 return (unsigned)AS >= (unsigned)LangAS::FirstTargetAddressSpace;
66}
67
68inline unsigned toTargetAddressSpace(LangAS AS) {
69 assert(isTargetAddressSpace(AS));
70 return (unsigned)AS - (unsigned)LangAS::FirstTargetAddressSpace;
71}
72
73inline LangAS getLangASFromTargetAS(unsigned TargetAS) {
74 return static_cast<LangAS>((TargetAS) +
75 (unsigned)LangAS::FirstTargetAddressSpace);
76}
77
78inline bool isPtrSizeAddressSpace(LangAS AS) {
79 return (AS == LangAS::ptr32_sptr || AS == LangAS::ptr32_uptr ||
80 AS == LangAS::ptr64);
81}
82
83} // namespace clang

85#endif // LLVM_CLANG_BASIC_ADDRESSSPACES_H
