10#ifndef BENCHMARK_UTILITIES_H
11#define BENCHMARK_UTILITIES_H

13#include <cassert>
14#include <type_traits>
15
16#include "benchmark/benchmark.h"
17
18namespace UtilitiesInternal {
19 template <class Container>
20 auto HaveDataImpl(int) -> decltype((std::declval<Container&>().data(), std::true_type{}));
21 template <class Container>
22 auto HaveDataImpl(long) -> std::false_type;
23 template <class T>
24 using HasData = decltype(HaveDataImpl<T>(0));
25} // namespace UtilitiesInternal
26
27template <class Container, std::enable_if_t<UtilitiesInternal::HasData<Container>::value>* = nullptr>
28void DoNotOptimizeData(Container &c) { benchmark::DoNotOptimize(c.data()); }
29template <class Container, std::enable_if_t<!UtilitiesInternal::HasData<Container>::value>* = nullptr>
30void DoNotOptimizeData(Container &c) { benchmark::DoNotOptimize(&c); }

33#endif // BENCHMARK_UTILITIES_H
