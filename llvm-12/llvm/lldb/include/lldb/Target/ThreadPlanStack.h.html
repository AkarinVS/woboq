24// The ThreadPlans have a thread for use when they are asked all the ThreadPlan
25// state machine questions, but they should never cache any pointers from their
26// owning lldb_private::Thread. That's because we want to be able to detach
27// them from an owning thread, then reattach them by TID.
28// The ThreadPlanStack holds the ThreadPlans for a given TID. All its methods
29// are private, and it should only be accessed through the owning thread. When
30// it is detached from a thread, all you can do is reattach it or delete it.

119class ThreadPlanStackMap {
120public:
121 ThreadPlanStackMap(Process &process) : m_process(process) {}
122 ~ThreadPlanStackMap() {}
123
124 // Prune the map using the current_threads list.
125 void Update(ThreadList &current_threads, bool delete_missing,
126 bool check_for_new = true);
127
128 void AddThread(Thread &thread) {
129 lldb::tid_t tid = thread.GetID();
130 m_plans_list.emplace(tid, thread);
131 }
132
133 bool RemoveTID(lldb::tid_t tid) {
134 auto result = m_plans_list.find(tid);
135 if (result == m_plans_list.end())
136 return false;
137 result->second.ThreadDestroyed(nullptr);
138 m_plans_list.erase(result);
139 return true;
140 }
141
142 ThreadPlanStack *Find(lldb::tid_t tid) {
143 auto result = m_plans_list.find(tid);
144 if (result == m_plans_list.end())
145 return nullptr;
146 else
147 return &result->second;
148 }
149
150 /// Clear the Thread* cache that each ThreadPlan contains.
151 ///
152 /// This is useful in situations like when a new Thread list is being
153 /// generated.
154 void ClearThreadCache() {
155 for (auto &plan_list : m_plans_list)
156 plan_list.second.ClearThreadCache();
157 }
158
159 void Clear() {
160 for (auto plan : m_plans_list)
161 plan.second.ThreadDestroyed(nullptr);
162 m_plans_list.clear();
163 }
164
165 // Implements Process::DumpThreadPlans
166 void DumpPlans(Stream &strm, lldb::DescriptionLevel desc_level, bool internal,
167 bool ignore_boring, bool skip_unreported);
168
169 // Implements Process::DumpThreadPlansForTID
170 bool DumpPlansForTID(Stream &strm, lldb::tid_t tid,
171 lldb::DescriptionLevel desc_level, bool internal,
172 bool ignore_boring, bool skip_unreported);
173
174 bool PrunePlansForTID(lldb::tid_t tid);
175
176private:
177 Process &m_process;
178 using PlansList = std::unordered_map<lldb::tid_t, ThreadPlanStack>;
179 PlansList m_plans_list;
180};
