15RecordsByName lldb_private::getRecordsByName(std::vector<Record *> Records,
16 StringRef Name) {
17 RecordsByName Result;
18 for (Record *R : Records)
19 Result[R->getValueAsString(Name).str()].push_back(R);
20 return Result;
21}
