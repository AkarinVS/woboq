13// on disk or by using "plugin load" command at the LLDB command line.

31bool lldb::PluginInitialize(lldb::SBDebugger debugger) {
32
33#ifdef BUILD_INTEL_PT
34 PTPluginInitialize(debugger);
35#endif
36
37#ifdef BUILD_INTEL_MPX
38 MPXPluginInitialize(debugger);
39#endif
40
41 return true;
42}
