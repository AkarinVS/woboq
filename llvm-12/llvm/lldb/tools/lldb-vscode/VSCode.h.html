75enum class PacketStatus {

78 JSONMalformed,
79 JSONNotObject
80};

82struct VSCode {
83 std::string debug_adaptor_path;
84 InputStream input;
85 OutputStream output;
86 lldb::SBDebugger debugger;
87 lldb::SBTarget target;
88 lldb::SBValueList variables;
89 lldb::SBBroadcaster broadcaster;
90 int64_t num_regs;
91 int64_t num_locals;
92 int64_t num_globals;
93 std::thread event_thread;
94 std::unique_ptr<std::ofstream> log;
95 llvm::DenseMap<lldb::addr_t, int64_t> addr_to_source_ref;
96 llvm::DenseMap<int64_t, SourceReference> source_map;
97 llvm::StringMap<SourceBreakpointMap> source_breakpoints;
98 FunctionBreakpointMap function_breakpoints;
99 std::vector<ExceptionBreakpoint> exception_breakpoints;
100 std::vector<std::string> init_commands;
101 std::vector<std::string> pre_run_commands;
102 std::vector<std::string> exit_commands;
103 std::vector<std::string> stop_commands;
104 std::vector<std::string> terminate_commands;
105 lldb::tid_t focus_tid;
106 bool sent_terminated_event;
107 bool stop_at_entry;
108 bool is_attach;
109 uint32_t reverse_request_seq;
110 std::map<std::string, RequestCallback> request_handlers;
111 bool waiting_for_run_in_terminal;
112 // Keep track of the last stop thread index IDs as threads won't go away
113 // unless we send a "thread" event to indicate the thread exited.
114 llvm::DenseSet<lldb::tid_t> thread_ids;
115 VSCode();
116 ~VSCode();
117 VSCode(const VSCode &rhs) = delete;
118 void operator=(const VSCode &rhs) = delete;
119 int64_t GetLineForPC(int64_t sourceReference, lldb::addr_t pc) const;
120 ExceptionBreakpoint *GetExceptionBreakpoint(const std::string &filter);
121 ExceptionBreakpoint *GetExceptionBreakpoint(const lldb::break_id_t bp_id);
122 // Send the JSON in "json_str" to the "out" stream. Correctly send the
123 // "Content-Length:" field followed by the length, followed by the raw
124 // JSON bytes.
125 void SendJSON(const std::string &json_str);
126
127 // Serialize the JSON value into a string and send the JSON packet to
128 // the "out" stream.
129 void SendJSON(const llvm::json::Value &json);
130
131 std::string ReadJSON();
132
133 void SendOutput(OutputType o, const llvm::StringRef output);
134
135 void __attribute__((format(printf, 3, 4)))
136 SendFormattedOutput(OutputType o, const char *format, ...);

142 lldb::SBThread GetLLDBThread(const llvm::json::Object &arguments);

144 lldb::SBFrame GetLLDBFrame(const llvm::json::Object &arguments);
145
146 llvm::json::Value CreateTopLevelScopes();
147
148 void RunLLDBCommands(llvm::StringRef prefix,
149 const std::vector<std::string> &commands);
150
151 void RunInitCommands();

154 void RunExitCommands();
155 void RunTerminateCommands();

157 /// Create a new SBTarget object from the given request arguments.
158 /// \param[in] arguments
159 /// Launch configuration arguments.
160 ///
161 /// \param[out] error
162 /// An SBError object that will contain an error description if
163 /// function failed to create the target.
164 ///
165 /// \return
166 /// An SBTarget object.
167 lldb::SBTarget CreateTargetFromArguments(const llvm::json::Object &arguments,
168 lldb::SBError &error);
169
170 /// Set given target object as a current target for lldb-vscode and start
171 /// listeing for its breakpoint events.
172 void SetTarget(const lldb::SBTarget target);
173
174 const std::map<std::string, RequestCallback> &GetRequestHandlers();
175
176 PacketStatus GetNextObject(llvm::json::Object &object);
177 bool HandleObject(const llvm::json::Object &object);
178
179 /// Send a Debug Adapter Protocol reverse request to the IDE
180 ///
181 /// \param[in] request
182 /// The payload of the request to send.
183 ///
184 /// \param[out] response
185 /// The response of the IDE. It might be undefined if there was an error.
186 ///
187 /// \return
188 /// A \a PacketStatus object indicating the sucess or failure of the
189 /// request.
190 PacketStatus SendReverseRequest(llvm::json::Object request,
191 llvm::json::Object &response);
192
193 /// Registers a callback handler for a Debug Adapter Protocol request
194 ///
195 /// \param[in] request
196 /// The name of the request following the Debug Adapter Protocol
197 /// specification.
198 ///
199 /// \param[in] callback
200 /// The callback to execute when the given request is triggered by the
201 /// IDE.
202 void RegisterRequestCallback(std::string request, RequestCallback callback);
203};
