10#include "BreakpointBase.h"
11#include "VSCode.h"

13namespace lldb_vscode {
14
15void ExceptionBreakpoint::SetBreakpoint() {
16 if (bp.IsValid())
17 return;
18 bool catch_value = filter.find("_catch") != std::string::npos;
19 bool throw_value = filter.find("_throw") != std::string::npos;
20 bp = g_vsc.target.BreakpointCreateForException(language, catch_value,
21 throw_value);
22 // See comments in BreakpointBase::GetBreakpointLabel() for details of why
23 // we add a label to our breakpoints.
24 bp.AddName(BreakpointBase::GetBreakpointLabel());
25}
26
27void ExceptionBreakpoint::ClearBreakpoint() {
28 if (!bp.IsValid())
29 return;
30 g_vsc.target.BreakpointDelete(bp.GetID());
31 bp = lldb::SBBreakpoint();
32}
