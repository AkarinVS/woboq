16size_t OptionValueArgs::GetArgs(Args &args) {
17 args.Clear();
18 for (auto value : m_values) {
19 llvm::StringRef string_value = value->GetStringValue();
20 if (!string_value.empty())
21 args.AppendArgument(string_value);
22 }
23
24 return args.GetArgumentCount();
25}
