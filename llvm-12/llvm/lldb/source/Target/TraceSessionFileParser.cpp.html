23void TraceSessionFileParser::NormalizePath(lldb_private::FileSpec &file_spec) {
24 if (file_spec.IsRelative())
25 file_spec.PrependPathComponent(m_session_file_dir);
26}

28Error TraceSessionFileParser::ParseModule(lldb::TargetSP &target_sp,
29 const JSONModule &module) {
30 FileSpec system_file_spec(module.system_path);
31 NormalizePath(system_file_spec);
32
33 FileSpec local_file_spec(module.file.hasValue() ? *module.file
34 : module.system_path);
35 NormalizePath(local_file_spec);
36
37 ModuleSpec module_spec;
38 module_spec.GetFileSpec() = local_file_spec;
39 module_spec.GetPlatformFileSpec() = system_file_spec;
40
41 if (module.uuid.hasValue())
42 module_spec.GetUUID().SetFromStringRef(*module.uuid);
43
44 Status error;
45 ModuleSP module_sp =
46 target_sp->GetOrCreateModule(module_spec, /*notify*/ false, &error);
47
48 if (error.Fail())
49 return error.ToError();
50
51 bool load_addr_changed = false;
52 module_sp->SetLoadAddress(*target_sp, module.load_address.value, false,
53 load_addr_changed);
54 return llvm::Error::success();
55}

57Error TraceSessionFileParser::CreateJSONError(json::Path::Root &root,
58 const json::Value &value) {
59 std::string err;
60 raw_string_ostream os(err);
61 root.printErrorContext(value, os);
62 return createStringError(
63 std::errc::invalid_argument, "%s\n\nContext:\n%s\n\nSchema:\n%s",
64 toString(root.getError()).c_str(), os.str().c_str(), m_schema.data());
65}

67std::string TraceSessionFileParser::BuildSchema(StringRef plugin_schema) {
68 std::ostringstream schema_builder;
69 schema_builder << "{\n \"trace\": ";
70 schema_builder << plugin_schema.data() << ",";
71 schema_builder << R"(
72 "processes": [
73 {
74 "pid": integer,
75 "triple": string, // llvm-triple
76 "threads": [
77 {
78 "tid": integer,
79 "traceFile": string
80 }
81 ],
82 "modules": [
83 {
84 "systemPath": string, // original path of the module at runtime
85 "file"?: string, // copy of the file if not available at "systemPath"
86 "loadAddress": string, // string address in hex or decimal form
87 "uuid"?: string,
88 }
89 ]
90 }
91 ]
92 // Notes:
93 // All paths are either absolute or relative to the session file.
94}
95)";
96 return schema_builder.str();
97}

99ThreadTraceSP TraceSessionFileParser::ParseThread(ProcessSP &process_sp,
100 const JSONThread &thread) {
101 lldb::tid_t tid = static_cast<lldb::tid_t>(thread.tid);
102
103 FileSpec trace_file(thread.trace_file);
104 NormalizePath(trace_file);
105
106 ThreadTraceSP thread_sp =
107 std::make_shared<ThreadTrace>(*process_sp, tid, trace_file);
108 process_sp->GetThreadList().AddThread(thread_sp);
109 return thread_sp;
110}

112Expected<TraceSessionFileParser::ParsedProcess>
113TraceSessionFileParser::ParseProcess(const JSONProcess &process) {
114 TargetSP target_sp;
115 Status error = m_debugger.GetTargetList().CreateTarget(
116 m_debugger, /*user_exe_path*/ StringRef(), process.triple,
117 eLoadDependentsNo,
118 /*platform_options*/ nullptr, target_sp);
119
120 if (!target_sp)
121 return error.ToError();
122
123 ParsedProcess parsed_process;
124 parsed_process.target_sp = target_sp;
125
126 ProcessSP process_sp = target_sp->CreateProcess(
127 /*listener*/ nullptr, "trace",
128 /*crash_file*/ nullptr,
129 /*can_connect*/ false);
130
131 process_sp->SetID(static_cast<lldb::pid_t>(process.pid));
132
133 for (const JSONThread &thread : process.threads)
134 parsed_process.threads.push_back(ParseThread(process_sp, thread));
135
136 for (const JSONModule &module : process.modules)
137 if (Error err = ParseModule(target_sp, module))
138 return std::move(err);
139
140 if (!process.threads.empty())
141 process_sp->GetThreadList().SetSelectedThreadByIndexID(0);
142
143 // We invoke DidAttach to create a correct stopped state for the process and
144 // its threads.
145 ArchSpec process_arch;
146 process_sp->DidAttach(process_arch);
147
148 return parsed_process;
149}

151Expected<std::vector<TraceSessionFileParser::ParsedProcess>>
152TraceSessionFileParser::ParseCommonSessionFile(
153 const JSONTraceSessionBase &session) {
154 std::vector<ParsedProcess> parsed_processes;
155
156 auto onError = [&]() {
157 // Delete all targets that were created so far in case of failures
158 for (ParsedProcess &parsed_process : parsed_processes)
159 m_debugger.GetTargetList().DeleteTarget(parsed_process.target_sp);
160 };
161
162 for (const JSONProcess &process : session.processes) {
163 if (Expected<ParsedProcess> parsed_process = ParseProcess(process))
164 parsed_processes.push_back(std::move(*parsed_process));
165 else {
166 onError();
167 return parsed_process.takeError();
168 }
169 }
170 return parsed_processes;
171}

176bool fromJSON(const Value &value, TraceSessionFileParser::JSONAddress &address,
177 Path path) {
178 Optional<StringRef> s = value.getAsString();
179 if (s.hasValue() && !s->getAsInteger(0, address.value))
180 return true;
181
182 path.report("expected numeric string");
183 return false;
184}

186bool fromJSON(const Value &value, TraceSessionFileParser::JSONModule &module,
187 Path path) {
188 ObjectMapper o(value, path);
189 return o && o.map("systemPath", module.system_path) &&
190 o.map("file", module.file) &&
191 o.map("loadAddress", module.load_address) &&
192 o.map("uuid", module.uuid);
193}
194
195bool fromJSON(const Value &value, TraceSessionFileParser::JSONThread &thread,
196 Path path) {
197 ObjectMapper o(value, path);
198 return o && o.map("tid", thread.tid) && o.map("traceFile", thread.trace_file);
199}

201bool fromJSON(const Value &value, TraceSessionFileParser::JSONProcess &process,
202 Path path) {
203 ObjectMapper o(value, path);
204 return o && o.map("pid", process.pid) && o.map("triple", process.triple) &&
205 o.map("threads", process.threads) && o.map("modules", process.modules);
206}
