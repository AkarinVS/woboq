9#ifndef LLDB_PLUGINS_SCRIPTINTERPRETER_PYTHON_LLDB_PYTHON_H
10#define LLDB_PLUGINS_SCRIPTINTERPRETER_PYTHON_LLDB_PYTHON_H
11
12#include "lldb/Host/Config.h"
13
14// Python.h needs to be included before any system headers in order to avoid
15// redefinition of macros
16
17#if LLDB_ENABLE_PYTHON
18#include "llvm/Support/Compiler.h"
19#if defined(_WIN32)
20// If anyone #includes Host/PosixApi.h later, it will try to typedef pid_t. We
21// need to ensure this doesn't happen. At the same time, Python.h will also try
22// to redefine a bunch of stuff that PosixApi.h defines. So define it all now
23// so that PosixApi.h doesn't redefine it.
24#define NO_PID_T
25#endif
26#if defined(__linux__)
27// features.h will define _POSIX_C_SOURCE if _GNU_SOURCE is defined. This value
28// may be different from the value that Python defines it to be which results
29// in a warning. Undefine _POSIX_C_SOURCE before including Python.h The same
30// holds for _XOPEN_SOURCE.
31#undef _POSIX_C_SOURCE
32#undef _XOPEN_SOURCE
33#endif
34
35// Include locale before Python so _PY_PORT_CTYPE_UTF8_ISSUE doesn't cause
36// macro redefinitions.
37#if defined(__APPLE__)
38#include <locale>
39#endif
40
41// Include python for non windows machines
42#include <Python.h>
43#endif
44
45#endif // LLDB_PLUGINS_SCRIPTINTERPRETER_PYTHON_LLDB_PYTHON_H
