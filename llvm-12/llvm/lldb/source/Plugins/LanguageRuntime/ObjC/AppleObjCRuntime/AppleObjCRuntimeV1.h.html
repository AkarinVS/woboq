121 class HashTableSignature {
122 public:
123 HashTableSignature()
124 : m_count(0), m_num_buckets(0), m_buckets_ptr(LLDB_INVALID_ADDRESS) {}
125
126 bool NeedsUpdate(uint32_t count, uint32_t num_buckets,
127 lldb::addr_t buckets_ptr) {
128 return m_count != count || m_num_buckets != num_buckets ||
129 m_buckets_ptr != buckets_ptr;
130 }
131
132 void UpdateSignature(uint32_t count, uint32_t num_buckets,
133 lldb::addr_t buckets_ptr) {
134 m_count = count;
135 m_num_buckets = num_buckets;
136 m_buckets_ptr = buckets_ptr;
137 }
138
139 protected:
140 uint32_t m_count;
141 uint32_t m_num_buckets;
142 lldb::addr_t m_buckets_ptr;
143 };

149 std::unique_ptr<DeclVendor> m_decl_vendor_up;
150
151private:
152 AppleObjCRuntimeV1(Process *process);
153};
