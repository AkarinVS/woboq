9#ifndef LLDB_SOURCE_PLUGINS_DYNAMICLOADER_POSIX_DYLD_DYNAMICLOADERPOSIXDYLD_H
10#define LLDB_SOURCE_PLUGINS_DYNAMICLOADER_POSIX_DYLD_DYNAMICLOADERPOSIXDYLD_H
11
12#include <map>
13#include <memory>
14
15#include "DYLDRendezvous.h"
16#include "Plugins/Process/Utility/AuxVector.h"
17#include "lldb/Breakpoint/StoppointCallbackContext.h"
18#include "lldb/Core/ModuleList.h"
19#include "lldb/Target/DynamicLoader.h"
20
21class AuxVector;
22
23class DynamicLoaderPOSIXDYLD : public lldb_private::DynamicLoader {
24public:
25 DynamicLoaderPOSIXDYLD(lldb_private::Process *process);
26
27 ~DynamicLoaderPOSIXDYLD() override;
28
29 static void Initialize();
30
31 static void Terminate();
32
33 static lldb_private::ConstString GetPluginNameStatic();
34
35 static const char *GetPluginDescriptionStatic();
36
37 static lldb_private::DynamicLoader *
38 CreateInstance(lldb_private::Process *process, bool force);
39
40 // DynamicLoader protocol
41
42 void DidAttach() override;
43
44 void DidLaunch() override;
45
46 lldb::ThreadPlanSP GetStepThroughTrampolinePlan(lldb_private::Thread &thread,
47 bool stop_others) override;
48
49 lldb_private::Status CanLoadImage() override;
50
51 lldb::addr_t GetThreadLocalData(const lldb::ModuleSP module,
52 const lldb::ThreadSP thread,
53 lldb::addr_t tls_file_addr) override;
54
55 // PluginInterface protocol
56 lldb_private::ConstString GetPluginName() override;
57
58 uint32_t GetPluginVersion() override;
59
60protected:
61 /// Runtime linker rendezvous structure.
62 DYLDRendezvous m_rendezvous;
63
64 /// Virtual load address of the inferior process.
65 lldb::addr_t m_load_offset;
66
67 /// Virtual entry address of the inferior process.
68 lldb::addr_t m_entry_point;
69
70 /// Auxiliary vector of the inferior process.
71 std::unique_ptr<AuxVector> m_auxv;
72
73 /// Rendezvous breakpoint.
74 lldb::break_id_t m_dyld_bid;
75
76 /// Contains AT_SYSINFO_EHDR, which means a vDSO has been
77 /// mapped to the address space
78 lldb::addr_t m_vdso_base;
79
80 /// Contains AT_BASE, which means a dynamic loader has been
81 /// mapped to the address space
82 lldb::addr_t m_interpreter_base;
83
84 /// Contains the pointer to the interpret module, if loaded.
85 std::weak_ptr<lldb_private::Module> m_interpreter_module;
86
87 /// Loaded module list. (link map for each module)
88 std::map<lldb::ModuleWP, lldb::addr_t, std::owner_less<lldb::ModuleWP>>
89 m_loaded_modules;
90
91 /// If possible sets a breakpoint on a function called by the runtime
92 /// linker each time a module is loaded or unloaded.
93 bool SetRendezvousBreakpoint();
94
95 /// Callback routine which updates the current list of loaded modules based
96 /// on the information supplied by the runtime linker.
97 static bool RendezvousBreakpointHit(
98 void *baton, lldb_private::StoppointCallbackContext *context,
99 lldb::user_id_t break_id, lldb::user_id_t break_loc_id);
100
101 /// Indicates whether the initial set of modules was reported added.
102 bool m_initial_modules_added;
103
104 /// Helper method for RendezvousBreakpointHit. Updates LLDB's current set
105 /// of loaded modules.
106 void RefreshModules();
107
108 /// Updates the load address of every allocatable section in \p module.
109 ///
110 /// \param module The module to traverse.
111 ///
112 /// \param link_map_addr The virtual address of the link map for the @p
113 /// module.
114 ///
115 /// \param base_addr The virtual base address \p module is loaded at.
116 void UpdateLoadedSections(lldb::ModuleSP module, lldb::addr_t link_map_addr,
117 lldb::addr_t base_addr,
118 bool base_addr_is_offset) override;
119
120 /// Removes the loaded sections from the target in \p module.
121 ///
122 /// \param module The module to traverse.
123 void UnloadSections(const lldb::ModuleSP module) override;
124
125 /// Resolves the entry point for the current inferior process and sets a
126 /// breakpoint at that address.
127 void ProbeEntry();
128
129 /// Callback routine invoked when we hit the breakpoint on process entry.
130 ///
131 /// This routine is responsible for resolving the load addresses of all
132 /// dependent modules required by the inferior and setting up the rendezvous
133 /// breakpoint.
134 static bool
135 EntryBreakpointHit(void *baton,
136 lldb_private::StoppointCallbackContext *context,
137 lldb::user_id_t break_id, lldb::user_id_t break_loc_id);
138
139 /// Helper for the entry breakpoint callback. Resolves the load addresses
140 /// of all dependent modules.
141 virtual void LoadAllCurrentModules();
142
143 void LoadVDSO();
144
145 // Loading an interpreter module (if present) assuming m_interpreter_base
146 // already points to its base address.
147 lldb::ModuleSP LoadInterpreterModule();
148
149 /// Computes a value for m_load_offset returning the computed address on
150 /// success and LLDB_INVALID_ADDRESS on failure.
151 lldb::addr_t ComputeLoadOffset();
152
153 /// Computes a value for m_entry_point returning the computed address on
154 /// success and LLDB_INVALID_ADDRESS on failure.
155 lldb::addr_t GetEntryPoint();
156
157 /// Evaluate if Aux vectors contain vDSO and LD information
158 /// in case they do, read and assign the address to m_vdso_base
159 /// and m_interpreter_base.
160 void EvalSpecialModulesStatus();
161
162 /// Loads Module from inferior process.
163 void ResolveExecutableModule(lldb::ModuleSP &module_sp);
164
165 bool AlwaysRelyOnEHUnwindInfo(lldb_private::SymbolContext &sym_ctx) override;
166
167private:
168 DynamicLoaderPOSIXDYLD(const DynamicLoaderPOSIXDYLD &) = delete;
169 const DynamicLoaderPOSIXDYLD &
170 operator=(const DynamicLoaderPOSIXDYLD &) = delete;
171};
172
173#endif // LLDB_SOURCE_PLUGINS_DYNAMICLOADER_POSIX_DYLD_DYNAMICLOADERPOSIXDYLD_H
