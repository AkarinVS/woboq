54// CreateInstance
55//
56// Platforms can register a callback to use when creating symbol vendors to
57// allow for complex debug information file setups, and to also allow for
58// finding separate debug information files.
59SymbolVendor *
60SymbolVendorWasm::CreateInstance(const lldb::ModuleSP &module_sp,
61 lldb_private::Stream *feedback_strm) {
62 if (!module_sp)
63 return nullptr;
64
65 ObjectFileWasm *obj_file =
66 llvm::dyn_cast_or_null<ObjectFileWasm>(module_sp->GetObjectFile());
67 if (!obj_file)
68 return nullptr;
69
70 // If the main object file already contains debug info, then we are done.
71 if (obj_file->GetSectionList()->FindSectionByType(
72 lldb::eSectionTypeDWARFDebugInfo, true))
73 return nullptr;
74
75 LLDB_SCOPED_TIMERF("SymbolVendorWasm::CreateInstance (module = %s)",
76 module_sp->GetFileSpec().GetPath().c_str());
77
78 ModuleSpec module_spec;
79 module_spec.GetFileSpec() = obj_file->GetFileSpec();
80 FileSystem::Instance().Resolve(module_spec.GetFileSpec());
81 module_spec.GetUUID() = obj_file->GetUUID();
82
83 // A Wasm module may have a custom section named "external_debug_info" whose
84 // content is the absolute or relative path of the Wasm module that contains
85 // debug symbols for this module.
86 llvm::Optional<FileSpec> symbol_file_spec =
87 obj_file->GetExternalDebugInfoFileSpec();
88 if (!symbol_file_spec)
89 return nullptr;
90 module_spec.GetSymbolFileSpec() = *symbol_file_spec;
91
92 FileSpecList search_paths = Target::GetDefaultDebugFileSearchPaths();
93 FileSpec sym_fspec =
94 Symbols::LocateExecutableSymbolFile(module_spec, search_paths);
95 if (!sym_fspec)
96 return nullptr;
97
98 DataBufferSP sym_file_data_sp;
99 lldb::offset_t sym_file_data_offset = 0;
100 ObjectFileSP sym_objfile_sp = ObjectFile::FindPlugin(
101 module_sp, &sym_fspec, 0, FileSystem::Instance().GetByteSize(sym_fspec),
102 sym_file_data_sp, sym_file_data_offset);
103 if (!sym_objfile_sp)
104 return nullptr;
105
106 // This objfile is for debugging purposes.
107 sym_objfile_sp->SetType(ObjectFile::eTypeDebugInfo);
108
109 SymbolVendorWasm *symbol_vendor = new SymbolVendorWasm(module_sp);
110
111 // Get the module unified section list and add our debug sections to
112 // that.
113 SectionList *module_section_list = module_sp->GetSectionList();
114 SectionList *objfile_section_list = sym_objfile_sp->GetSectionList();
115
116 static const SectionType g_sections[] = {
117 eSectionTypeDWARFDebugAbbrev, eSectionTypeDWARFDebugAddr,
118 eSectionTypeDWARFDebugAranges, eSectionTypeDWARFDebugCuIndex,
119 eSectionTypeDWARFDebugFrame, eSectionTypeDWARFDebugInfo,
120 eSectionTypeDWARFDebugLine, eSectionTypeDWARFDebugLineStr,
121 eSectionTypeDWARFDebugLoc, eSectionTypeDWARFDebugLocLists,
122 eSectionTypeDWARFDebugMacInfo, eSectionTypeDWARFDebugMacro,
123 eSectionTypeDWARFDebugPubNames, eSectionTypeDWARFDebugPubTypes,
124 eSectionTypeDWARFDebugRanges, eSectionTypeDWARFDebugRngLists,
125 eSectionTypeDWARFDebugStr, eSectionTypeDWARFDebugStrOffsets,
126 eSectionTypeDWARFDebugTypes};
127 for (SectionType section_type : g_sections) {
128 if (SectionSP section_sp =
129 objfile_section_list->FindSectionByType(section_type, true)) {
130 if (SectionSP module_section_sp =
131 module_section_list->FindSectionByType(section_type, true))
132 module_section_list->ReplaceSection(module_section_sp->GetID(),
133 section_sp);
134 else
135 module_section_list->AddSection(section_sp);
136 }
137 }
138
139 symbol_vendor->AddSymbolFileRepresentation(sym_objfile_sp);
140 return symbol_vendor;
141}
