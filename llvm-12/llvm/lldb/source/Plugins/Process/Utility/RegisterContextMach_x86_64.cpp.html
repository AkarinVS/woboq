18RegisterContextMach_x86_64::RegisterContextMach_x86_64(

24int RegisterContextMach_x86_64::DoReadGPR(lldb::tid_t tid, int flavor,
25 GPR &gpr) {
26 mach_msg_type_number_t count = GPRWordCount;
27 return ::thread_get_state(tid, flavor, (thread_state_t)&gpr, &count);
28}

30int RegisterContextMach_x86_64::DoReadFPU(lldb::tid_t tid, int flavor,

36int RegisterContextMach_x86_64::DoReadEXC(lldb::tid_t tid, int flavor,

42int RegisterContextMach_x86_64::DoWriteGPR(lldb::tid_t tid, int flavor,
43 const GPR &gpr) {
44 return ::thread_set_state(
45 tid, flavor, reinterpret_cast<thread_state_t>(const_cast<GPR *>(&gpr)),
46 GPRWordCount);
47}
