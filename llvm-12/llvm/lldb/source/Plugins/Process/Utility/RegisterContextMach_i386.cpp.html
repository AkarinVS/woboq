24int RegisterContextMach_i386::DoReadGPR(lldb::tid_t tid, int flavor, GPR &gpr) {
25 mach_msg_type_number_t count = GPRWordCount;
26 return ::thread_get_state(tid, flavor, (thread_state_t)&gpr, &count);
27}
