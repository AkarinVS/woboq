88RegisterContextFreeBSD_mips64::RegisterContextFreeBSD_mips64(
89 const ArchSpec &target_arch)
90 : RegisterInfoInterface(target_arch) {}

92size_t RegisterContextFreeBSD_mips64::GetGPRSize() const {
93 return sizeof(GPR_freebsd_mips);
94}
95
96const RegisterSet *
97RegisterContextFreeBSD_mips64::GetRegisterSet(size_t set) const {
98 // Check if RegisterSet is available
99 if (set < k_num_register_sets)
100 return &g_reg_sets_mips64[set];
101 return nullptr;
102}
103
104size_t
105RegisterContextFreeBSD_mips64::GetRegisterSetCount() const {
106 return k_num_register_sets;
107}
108
109const RegisterInfo *RegisterContextFreeBSD_mips64::GetRegisterInfo() const {
110 assert(m_target_arch.GetCore() == ArchSpec::eCore_mips64);
111 return g_register_infos_mips64;
112}
113
114uint32_t RegisterContextFreeBSD_mips64::GetRegisterCount() const {
115 return static_cast<uint32_t>(sizeof(g_register_infos_mips64) /
116 sizeof(g_register_infos_mips64[0]));
117}
