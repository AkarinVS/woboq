431 using ModuleCacheKey = std::pair<std::string, std::string>;
432 // KeyInfo for the cached module spec DenseMap.
433 // The invariant is that all real keys will have the file and architecture
434 // set.
435 // The empty key has an empty file and an empty arch.
436 // The tombstone key has an invalid arch and an empty file.
437 // The comparison and hash functions take the file name and architecture
438 // triple into account.
439 struct ModuleCacheInfo {
440 static ModuleCacheKey getEmptyKey() { return ModuleCacheKey(); }
441
442 static ModuleCacheKey getTombstoneKey() { return ModuleCacheKey("", "T"); }
443
444 static unsigned getHashValue(const ModuleCacheKey &key) {
445 return llvm::hash_combine(key.first, key.second);
446 }
447
448 static bool isEqual(const ModuleCacheKey &LHS, const ModuleCacheKey &RHS) {
449 return LHS == RHS;
450 }
451 };
452
453 llvm::DenseMap<ModuleCacheKey, ModuleSpec, ModuleCacheInfo>
454 m_cached_module_specs;
