18std::vector<CompilerType> DeclVendor::FindTypes(ConstString name,
19 uint32_t max_matches) {
20 // FIXME: This depends on clang, but should be able to support any
21 // TypeSystem.
22 std::vector<CompilerType> ret;
23 std::vector<CompilerDecl> decls;
24 if (FindDecls(name, /*append*/ true, max_matches, decls))
25 for (auto decl : decls)
26 if (auto type =
27 decl.GetTypeSystem()->GetTypeForDecl(decl.GetOpaqueDecl()))
28 ret.push_back(type);
29 return ret;
30}
