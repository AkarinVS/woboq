21Status fileLock(int fd, int cmd, int lock_type, const uint64_t start,
22 const uint64_t len) {
23 struct flock fl;
24
25 fl.l_type = lock_type;
26 fl.l_whence = SEEK_SET;
27 fl.l_start = start;
28 fl.l_len = len;
29 fl.l_pid = ::getpid();
30
31 Status error;
32 if (llvm::sys::RetryAfterSignal(-1, ::fcntl, fd, cmd, &fl) == -1)
33 error.SetErrorToErrno();
34
35 return error;
36}

38} // namespace

40LockFilePosix::LockFilePosix(int fd) : LockFileBase(fd) {}
41
42LockFilePosix::~LockFilePosix() { Unlock(); }

44Status LockFilePosix::DoWriteLock(const uint64_t start, const uint64_t len) {
45 return fileLock(m_fd, F_SETLKW, F_WRLCK, start, len);
46}
47
48Status LockFilePosix::DoTryWriteLock(const uint64_t start, const uint64_t len) {
49 return fileLock(m_fd, F_SETLK, F_WRLCK, start, len);
50}
51
52Status LockFilePosix::DoReadLock(const uint64_t start, const uint64_t len) {
53 return fileLock(m_fd, F_SETLKW, F_RDLCK, start, len);
54}
55
56Status LockFilePosix::DoTryReadLock(const uint64_t start, const uint64_t len) {
57 return fileLock(m_fd, F_SETLK, F_RDLCK, start, len);
58}
59
60Status LockFilePosix::DoUnlock() {
61 return fileLock(m_fd, F_SETLK, F_UNLCK, m_start, m_len);
62}
