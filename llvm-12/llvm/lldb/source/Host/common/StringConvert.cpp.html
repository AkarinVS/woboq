16int32_t ToSInt32(const char *s, int32_t fail_value, int base,
17 bool *success_ptr) {
18 if (s && s[0]) {
19 char *end = nullptr;
20 const long sval = ::strtol(s, &end, base);
21 if (*end == '\0') {
22 if (success_ptr)
23 *success_ptr = ((sval <= INT32_MAX) && (sval >= INT32_MIN));
24 return (int32_t)sval; // All characters were used, return the result
25 }
26 }
27 if (success_ptr)
28 *success_ptr = false;
29 return fail_value;
30}
