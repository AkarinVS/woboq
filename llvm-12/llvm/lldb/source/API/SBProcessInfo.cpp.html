30SBProcessInfo &SBProcessInfo::operator=(const SBProcessInfo &rhs) {
31 LLDB_RECORD_METHOD(lldb::SBProcessInfo &,

40ProcessInstanceInfo &SBProcessInfo::ref() {
41 if (m_opaque_up == nullptr) {
42 m_opaque_up = std::make_unique<ProcessInstanceInfo>();
43 }
44 return *m_opaque_up;
