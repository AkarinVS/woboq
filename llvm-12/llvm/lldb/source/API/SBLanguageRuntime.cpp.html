16lldb::LanguageType
17SBLanguageRuntime::GetLanguageTypeFromString(const char *string) {
18 LLDB_RECORD_STATIC_METHOD(lldb::LanguageType, SBLanguageRuntime,
19 GetLanguageTypeFromString, (const char *), string);
20
21 return Language::GetLanguageTypeFromString(
22 llvm::StringRef::withNullAsEmpty(string));
23}
24
25const char *
26SBLanguageRuntime::GetNameForLanguageType(lldb::LanguageType language) {
27 LLDB_RECORD_STATIC_METHOD(const char *, SBLanguageRuntime,
28 GetNameForLanguageType, (lldb::LanguageType),
29 language);
30
31 return Language::GetNameForLanguageType(language);
32}
