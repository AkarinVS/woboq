7template <typename SBClass>
8PyObject*
9SBTypeToSWIGWrapper (SBClass& sb_object)

12}

28template <>
29PyObject*
30SBTypeToSWIGWrapper (unsigned int* c_int)

32 if (!c_int)
33 return NULL;
34 return PyInt_FromLong(*c_int);
35}
