9#include "lldb/Utility/StreamCallback.h"
10#include "gtest/gtest.h"
11
12using namespace lldb;
13using namespace lldb_private;
14
15static char test_baton;
16static size_t callback_count = 0;
17static void TestCallback(const char *data, void *baton) {
18 EXPECT_STREQ("Foobar", data);
19 EXPECT_EQ(&test_baton, baton);
20 ++callback_count;
21}
22
23TEST(StreamCallbackTest, Callback) {
24 StreamCallback stream(TestCallback, &test_baton);
25 stream << "Foobar";
26 EXPECT_EQ(1u, callback_count);
27}
