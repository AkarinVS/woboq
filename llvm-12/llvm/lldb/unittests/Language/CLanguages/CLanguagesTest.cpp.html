20/// Returns the name of the LLDB plugin for the given language or a null
21/// ConstString if there is no fitting plugin.
22static ConstString GetPluginName(lldb::LanguageType language) {
23 Language *language_plugin = Language::FindPlugin(language);
24 if (language_plugin)
25 return language_plugin->GetPluginName();
26 return ConstString();
27}
28
29TEST(CLanguages, LookupCLanguagesByLanguageType) {
30 SubsystemRAII<CPlusPlusLanguage, ObjCPlusPlusLanguage, ObjCLanguage> langs;
31
32 // There is no plugin to find for C.
33 EXPECT_EQ(Language::FindPlugin(lldb::eLanguageTypeC), nullptr);
34 EXPECT_EQ(Language::FindPlugin(lldb::eLanguageTypeC89), nullptr);
35 EXPECT_EQ(Language::FindPlugin(lldb::eLanguageTypeC99), nullptr);
36 EXPECT_EQ(Language::FindPlugin(lldb::eLanguageTypeC11), nullptr);
37
38 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeC_plus_plus), "cplusplus");
39 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeC_plus_plus_03), "cplusplus");
40 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeC_plus_plus_11), "cplusplus");
41 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeC_plus_plus_14), "cplusplus");
42
43 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeObjC), "objc");
44
45 EXPECT_EQ(GetPluginName(lldb::eLanguageTypeObjC_plus_plus), "objcplusplus");
46}
