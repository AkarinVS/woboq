36TEST_F(XcodeSDKModuleTests, TestModuleGetXcodeSDK) {
37 const char *yamldata = R"(
38--- !ELF
39FileHeader:
40 Class: ELFCLASS64
41 Data: ELFDATA2LSB
42 Type: ET_EXEC
43 Machine: EM_386
44DWARF:
45 debug_str:
46 - MacOSX10.9.sdk
47 debug_abbrev:
48 - Table:
49 - Code: 0x00000001
50 Tag: DW_TAG_compile_unit
51 Children: DW_CHILDREN_no
52 Attributes:
53 - Attribute: DW_AT_language
54 Form: DW_FORM_data2
55 - Attribute: DW_AT_APPLE_sdk
56 Form: DW_FORM_strp
57 debug_info:
58 - Version: 2
59 AddrSize: 8
60 Entries:
61 - AbbrCode: 0x00000001
62 Values:
63 - Value: 0x000000000000000C
64 - Value: 0x0000000000000000
65 - AbbrCode: 0x00000000
66...
67)";
68
69 YAMLModuleTester t(yamldata);
70 DWARFUnit *dwarf_unit = t.GetDwarfUnit();
71 auto *dwarf_cu = llvm::cast<DWARFCompileUnit>(dwarf_unit);
72 ASSERT_TRUE(static_cast<bool>(dwarf_cu));
73 SymbolFileDWARF &sym_file = dwarf_cu->GetSymbolFileDWARF();
74 CompUnitSP comp_unit = sym_file.GetCompileUnitAtIndex(0);
75 ASSERT_TRUE(static_cast<bool>(comp_unit.get()));
76 ModuleSP module = t.GetModule();
77 ASSERT_EQ(module->GetSourceMappingList().GetSize(), 0u);
78 XcodeSDK sdk = sym_file.ParseXcodeSDK(*comp_unit);
79 ASSERT_EQ(sdk.GetType(), XcodeSDK::Type::MacOSX);
80 ASSERT_EQ(module->GetSourceMappingList().GetSize(), 1u);
81}
82#endif
