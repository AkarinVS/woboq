20extern const char *TestMainArgv0;
21
22std::string lldb_private::GetInputFilePath(const llvm::Twine &name) {
23 llvm::SmallString<128> result = llvm::sys::path::parent_path(TestMainArgv0);
24 llvm::sys::fs::make_absolute(result);
25 llvm::sys::path::append(result, "Inputs", name);
26 return std::string(result.str());
27}
28
29llvm::Expected<TestFile> TestFile::fromYaml(llvm::StringRef Yaml) {
30 std::string Buffer;
31 llvm::raw_string_ostream OS(Buffer);
32 llvm::yaml::Input YIn(Yaml);
33 if (!llvm::yaml::convertYAML(YIn, OS, [](const llvm::Twine &Msg) {}))
34 return llvm::createStringError(llvm::inconvertibleErrorCode(),
35 "convertYAML() failed");
36 return TestFile(std::move(Buffer));
37}
38
39llvm::Expected<TestFile> TestFile::fromYamlFile(const llvm::Twine &Name) {
40 auto BufferOrError =
41 llvm::MemoryBuffer::getFile(GetInputFilePath(Name), /*FileSize*/ -1,
42 /*RequiresNullTerminator*/ false);
43 if (!BufferOrError)
44 return llvm::errorCodeToError(BufferOrError.getError());
45 return fromYaml(BufferOrError.get()->getBuffer());
46}
