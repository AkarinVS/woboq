17YAMLModuleTester::YAMLModuleTester(llvm::StringRef yaml_data) {
18 llvm::Expected<TestFile> File = TestFile::fromYaml(yaml_data);
19 EXPECT_THAT_EXPECTED(File, llvm::Succeeded());
20 m_file = std::move(*File);
21
22 m_module_sp = std::make_shared<Module>(m_file->moduleSpec());
23 auto &symfile = *llvm::cast<SymbolFileDWARF>(m_module_sp->GetSymbolFile());
24
25 m_dwarf_unit = symfile.DebugInfo().GetUnitAtIndex(0);
26}
