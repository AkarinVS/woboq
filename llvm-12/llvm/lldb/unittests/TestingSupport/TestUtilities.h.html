20#define ASSERT_NO_ERROR(x) \
21 if (std::error_code ASSERT_NO_ERROR_ec = x) { \
22 llvm::SmallString<128> MessageStorage; \
23 llvm::raw_svector_ostream Message(MessageStorage); \
24 Message << #x ": did not return errc::success.\n" \
25 << "error number: " << ASSERT_NO_ERROR_ec.value() << "\n" \
26 << "error message: " << ASSERT_NO_ERROR_ec.message() << "\n"; \
27 GTEST_FATAL_FAILURE_(MessageStorage.c_str()); \
28 } else { \
29 }

31namespace lldb_private {
32std::string GetInputFilePath(const llvm::Twine &name);
33
34class TestFile {
35public:
36 static llvm::Expected<TestFile> fromYaml(llvm::StringRef Yaml);
37 static llvm::Expected<TestFile> fromYamlFile(const llvm::Twine &Name);
38
39 ModuleSpec moduleSpec() {
40 return ModuleSpec(FileSpec(), UUID(), dataBuffer());
41 }
42
43private:
44 TestFile(std::string &&Buffer) : Buffer(std::move(Buffer)) {}
45
46 lldb::DataBufferSP dataBuffer() {
47 auto *Data = reinterpret_cast<const uint8_t *>(Buffer.data());
48 return std::make_shared<DataBufferUnowned>(const_cast<uint8_t *>(Data),
49 Buffer.size());
50 }
51
52 std::string Buffer;
53};
54}
