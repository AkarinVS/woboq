20ALWAYS_INLINE uint64_t readTSC(uint8_t &CPU) XRAY_NEVER_INSTRUMENT {
21 unsigned LongCPU;
22 unsigned long Rax, Rdx;
23 __asm__ __volatile__("rdtscp\n" : "=a"(Rax), "=d"(Rdx), "=c"(LongCPU) ::);
24 CPU = LongCPU;
25 return (Rdx << 32) + Rax;
26}
