9#ifdef __BIONIC__
10#include <stdlib.h>
11extern "C" GWP_ASAN_WEAK void android_set_abort_message(const char *);
12#else // __BIONIC__
13#include <stdio.h>
14#endif
15
16namespace gwp_asan {
17void die(const char *Message) {
18#ifdef __BIONIC__
19 if (&android_set_abort_message != nullptr)
20 android_set_abort_message(Message);
21 abort();
22#else // __BIONIC__
23 fprintf(stderr, "%s", Message);
24 __builtin_trap();
25#endif // __BIONIC__
26}
27} // namespace gwp_asan
