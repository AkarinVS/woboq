12#include "tsan_ignoreset.h"
13
14namespace __tsan {
15
16const uptr IgnoreSet::kMaxSize;
17
18IgnoreSet::IgnoreSet()
19 : size_() {
20}
21
22void IgnoreSet::Add(u32 stack_id) {
23 if (size_ == kMaxSize)
24 return;
25 for (uptr i = 0; i < size_; i++) {
26 if (stacks_[i] == stack_id)
27 return;
28 }
29 stacks_[size_++] = stack_id;
30}
31
32void IgnoreSet::Reset() {
33 size_ = 0;
34}
35
36uptr IgnoreSet::Size() const {
37 return size_;
38}
39
40u32 IgnoreSet::At(uptr i) const {
41 CHECK_LT(i, size_);
42 CHECK_LE(size_, kMaxSize);
43 return stacks_[i];
44}
