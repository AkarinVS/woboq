22#if HWASAN_REPLACE_OPERATORS_NEW_AND_DELETE
23
24// TODO(alekseys): throw std::bad_alloc instead of dying on OOM.
25#define OPERATOR_NEW_BODY(nothrow) \
26 GET_MALLOC_STACK_TRACE; \
27 void *res = hwasan_malloc(size, &stack);\
28 if (!nothrow && UNLIKELY(!res)) ReportOutOfMemory(size, &stack);\
29 return res
30
31#define OPERATOR_DELETE_BODY \
32 GET_MALLOC_STACK_TRACE; \
33 if (ptr) hwasan_free(ptr, &stack)
34
35#elif defined(__ANDROID__)
36
37// We don't actually want to intercept operator new and delete on Android, but
38// since we previously released a runtime that intercepted these functions,
39// removing the interceptors would break ABI. Therefore we simply forward to
40// malloc and free.
41#define OPERATOR_NEW_BODY(nothrow) return malloc(size)
42#define OPERATOR_DELETE_BODY free(ptr)
43
44#endif

48using namespace __hwasan;

50// Fake std::nothrow_t to avoid including <new>.
51namespace std {
52 struct nothrow_t {};
53} // namespace std

57INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
58void *operator new(size_t size) { OPERATOR_NEW_BODY(false /*nothrow*/); }
59INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
60void *operator new[](size_t size) { OPERATOR_NEW_BODY(false /*nothrow*/); }
61INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
62void *operator new(size_t size, std::nothrow_t const&) {
63 OPERATOR_NEW_BODY(true /*nothrow*/);
64}
65INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
66void *operator new[](size_t size, std::nothrow_t const&) {
67 OPERATOR_NEW_BODY(true /*nothrow*/);
68}

70INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
71void operator delete(void *ptr) NOEXCEPT { OPERATOR_DELETE_BODY; }
72INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
73void operator delete[](void *ptr) NOEXCEPT { OPERATOR_DELETE_BODY; }
74INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
75void operator delete(void *ptr, std::nothrow_t const&) { OPERATOR_DELETE_BODY; }
76INTERCEPTOR_ATTRIBUTE SANITIZER_WEAK_ATTRIBUTE
77void operator delete[](void *ptr, std::nothrow_t const&) {
78 OPERATOR_DELETE_BODY;
79}
