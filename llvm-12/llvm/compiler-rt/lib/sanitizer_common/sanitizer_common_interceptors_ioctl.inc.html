16struct ioctl_desc {
17 unsigned req;
18 // FIXME: support read+write arguments. Currently READWRITE and WRITE do the
19 // same thing.
20 // XXX: The declarations below may use WRITE instead of READWRITE, unless
21 // explicitly noted.
22 enum {
23 NONE,
24 READ,
25 WRITE,
26 READWRITE,
27 CUSTOM
28 } type : 3;
29 unsigned size : 29;
30 const char* name;
31};
32
33const unsigned ioctl_table_max = 500;
34static ioctl_desc ioctl_table[ioctl_table_max];
35static unsigned ioctl_table_size = 0;
36
37// This can not be declared as a global, because references to struct_*_sz
38// require a global initializer. And this table must be available before global
39// initializers are run.
40static void ioctl_table_fill() {
41#define _(rq, tp, sz) \
42 if (IOCTL_##rq != IOCTL_NOT_PRESENT) { \
43 CHECK(ioctl_table_size < ioctl_table_max); \
44 ioctl_table[ioctl_table_size].req = IOCTL_##rq; \
45 ioctl_table[ioctl_table_size].type = ioctl_desc::tp; \
46 ioctl_table[ioctl_table_size].size = sz; \
47 ioctl_table[ioctl_table_size].name = #rq; \
48 ++ioctl_table_size; \
49 }

484static void ioctl_init() {
485 ioctl_table_fill();
486 Sort(ioctl_table, ioctl_table_size, ioctl_desc_compare());
487
488 bool bad = false;
489 for (unsigned i = 0; i < ioctl_table_size - 1; ++i) {
490 if (ioctl_table[i].req >= ioctl_table[i + 1].req) {
491 Printf("Duplicate or unsorted ioctl request id %x >= %x (%s vs %s)\n",
492 ioctl_table[i].req, ioctl_table[i + 1].req, ioctl_table[i].name,
493 ioctl_table[i + 1].name);
494 bad = true;
495 }
496 }
497
498 if (bad) Die();
499
500 ioctl_initialized = true;
501}

503// Handle the most evil ioctls that encode argument value as part of request id.
504static unsigned ioctl_request_fixup(unsigned req) {
505#if SANITIZER_LINUX
506 // Strip size and event number.
507 const unsigned kEviocgbitMask =
508 (IOC_SIZEMASK << IOC_SIZESHIFT) | EVIOC_EV_MAX;
509 if ((req & ~kEviocgbitMask) == IOCTL_EVIOCGBIT)
510 return IOCTL_EVIOCGBIT;
511 // Strip absolute axis number.
512 if ((req & ~EVIOC_ABS_MAX) == IOCTL_EVIOCGABS)
513 return IOCTL_EVIOCGABS;
514 if ((req & ~EVIOC_ABS_MAX) == IOCTL_EVIOCSABS)
515 return IOCTL_EVIOCSABS;
516#endif
517 return req;
518}

520static const ioctl_desc *ioctl_table_lookup(unsigned req) {
521 int left = 0;
522 int right = ioctl_table_size;
523 while (left < right) {
524 int mid = (left + right) / 2;
525 if (ioctl_table[mid].req < req)
526 left = mid + 1;
527 else
528 right = mid;
529 }
530 if (left == right && ioctl_table[left].req == req)
531 return ioctl_table + left;
532 else
533 return nullptr;
534}

536static bool ioctl_decode(unsigned req, ioctl_desc *desc) {
537 CHECK(desc);
538 desc->req = req;
539 desc->name = "<DECODED_IOCTL>";
540 desc->size = IOC_SIZE(req);
541 // Sanity check.
542 if (desc->size > 0xFFFF) return false;
543 unsigned dir = IOC_DIR(req);
544 switch (dir) {
545 case IOC_NONE:
546 desc->type = ioctl_desc::NONE;
547 break;
548 case IOC_READ | IOC_WRITE:
549 desc->type = ioctl_desc::READWRITE;
550 break;
551 case IOC_READ:
552 desc->type = ioctl_desc::WRITE;
553 break;
554 case IOC_WRITE:
555 desc->type = ioctl_desc::READ;
556 break;
557 default:
558 return false;
559 }
560 // Size can be 0 iff type is NONE.
561 if ((desc->type == IOC_NONE) != (desc->size == 0)) return false;
562 // Sanity check.
563 if (IOC_TYPE(req) == 0) return false;
564 return true;
565}
566
567static const ioctl_desc *ioctl_lookup(unsigned req) {
568 req = ioctl_request_fixup(req);
569 const ioctl_desc *desc = ioctl_table_lookup(req);
570 if (desc) return desc;
571
572 // Try stripping access size from the request id.
573 desc = ioctl_table_lookup(req & ~(IOC_SIZEMASK << IOC_SIZESHIFT));
574 // Sanity check: requests that encode access size are either read or write and
575 // have size of 0 in the table.
576 if (desc && desc->size == 0 &&
577 (desc->type == ioctl_desc::READWRITE || desc->type == ioctl_desc::WRITE ||
578 desc->type == ioctl_desc::READ))
579 return desc;
580 return nullptr;
581}

583static void ioctl_common_pre(void *ctx, const ioctl_desc *desc, int d,
584 unsigned request, void *arg) {
585 if (desc->type == ioctl_desc::READ || desc->type == ioctl_desc::READWRITE) {
586 unsigned size = desc->size ? desc->size : IOC_SIZE(request);
587 COMMON_INTERCEPTOR_READ_RANGE(ctx, arg, size);
588 }
589 if (desc->type != ioctl_desc::CUSTOM)
590 return;
591 if (request == IOCTL_SIOCGIFCONF) {
592 struct __sanitizer_ifconf *ifc = (__sanitizer_ifconf *)arg;
593 COMMON_INTERCEPTOR_READ_RANGE(ctx, (char*)&ifc->ifc_len,
594 sizeof(ifc->ifc_len));
595 }
596}

598static void ioctl_common_post(void *ctx, const ioctl_desc *desc, int res, int d,
599 unsigned request, void *arg) {
600 if (desc->type == ioctl_desc::WRITE || desc->type == ioctl_desc::READWRITE) {
601 // FIXME: add verbose output
602 unsigned size = desc->size ? desc->size : IOC_SIZE(request);
603 COMMON_INTERCEPTOR_WRITE_RANGE(ctx, arg, size);
604 }
605 if (desc->type != ioctl_desc::CUSTOM)
606 return;
607 if (request == IOCTL_SIOCGIFCONF) {
608 struct __sanitizer_ifconf *ifc = (__sanitizer_ifconf *)arg;
609 COMMON_INTERCEPTOR_WRITE_RANGE(ctx, ifc->ifc_ifcu.ifcu_req, ifc->ifc_len);
610 }
