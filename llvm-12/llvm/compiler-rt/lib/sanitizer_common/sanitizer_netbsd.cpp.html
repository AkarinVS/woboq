77static void *GetRealLibcAddress(const char *symbol) {
78 void *real = dlsym(RTLD_NEXT, symbol);
79 if (!real)
80 real = dlsym(RTLD_DEFAULT, symbol);
81 if (!real) {
82 Printf("GetRealLibcAddress failed for symbol=%s", symbol);
83 Die();
84 }
85 return real;
86}
87
88#define _REAL(func, ...) real##_##func(__VA_ARGS__)
89#define DEFINE__REAL(ret_type, func, ...) \
90 static ret_type (*real_##func)(__VA_ARGS__) = NULL; \
91 if (!real_##func) { \
92 real_##func = (ret_type(*)(__VA_ARGS__))GetRealLibcAddress(#func); \
93 } \
94 CHECK(real_##func);

212unsigned int internal_sleep(unsigned int seconds) {
213 struct timespec ts;
214 ts.tv_sec = seconds;
215 ts.tv_nsec = 0;
216 CHECK(&_sys___nanosleep50);
217 int res = _sys___nanosleep50(&ts, &ts);
218 if (res)
219 return ts.tv_sec;
220 return 0;
221}

240u64 NanoTime() {
241 timeval tv;
242 DEFINE__REAL(int, __gettimeofday50, void *a, void *b);
243 internal_memset(&tv, 0, sizeof(tv));
244 _REAL(__gettimeofday50, &tv, 0);
245 return (u64)tv.tv_sec * 1000 * 1000 * 1000 + tv.tv_usec * 1000;
246}
