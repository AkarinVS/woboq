136#if !SANITIZER_MAC
137#define ASAN_INTERCEPT_FUNC(name) \
138 do { \
139 if (!INTERCEPT_FUNCTION(name)) \
140 VReport(1, "AddressSanitizer: failed to intercept '%s'\n", #name); \
141 } while (0)
142#define ASAN_INTERCEPT_FUNC_VER(name, ver) \
143 do { \
144 if (!INTERCEPT_FUNCTION_VER(name, ver)) \
145 VReport(1, "AddressSanitizer: failed to intercept '%s@@%s'\n", #name, \
146 #ver); \
147 } while (0)
148#else
149// OS X interceptors don't need to be initialized with INTERCEPT_FUNCTION.
150#define ASAN_INTERCEPT_FUNC(name)
151#endif // SANITIZER_MAC
