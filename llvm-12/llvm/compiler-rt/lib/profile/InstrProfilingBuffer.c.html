16/* When continuous mode is enabled (%c), this parameter is set to 1.
17 *
18 * This parameter is defined here in InstrProfilingBuffer.o, instead of in
19 * InstrProfilingFile.o, to sequester all libc-dependent code in
20 * InstrProfilingFile.o. The test `instrprof-without-libc` will break if this
21 * layering is violated. */
22static int ContinuouslySyncProfile = 0;
23
24/* The system page size. Only valid when non-zero. If 0, the page size is
25 * unavailable. */
26static unsigned PageSize = 0;
27
28COMPILER_RT_VISIBILITY int __llvm_profile_is_continuous_mode_enabled(void) {
29 return ContinuouslySyncProfile && PageSize;
30}
31
32COMPILER_RT_VISIBILITY void __llvm_profile_enable_continuous_mode(void) {
33 ContinuouslySyncProfile = 1;
34}
35
36COMPILER_RT_VISIBILITY void __llvm_profile_set_page_size(unsigned PS) {
37 PageSize = PS;
38}

40COMPILER_RT_VISIBILITY
41uint64_t __llvm_profile_get_size_for_buffer(void) {
42 const __llvm_profile_data *DataBegin = __llvm_profile_begin_data();
43 const __llvm_profile_data *DataEnd = __llvm_profile_end_data();
44 const uint64_t *CountersBegin = __llvm_profile_begin_counters();
45 const uint64_t *CountersEnd = __llvm_profile_end_counters();
46 const char *NamesBegin = __llvm_profile_begin_names();
47 const char *NamesEnd = __llvm_profile_end_names();
48
49 return __llvm_profile_get_size_for_buffer_internal(
50 DataBegin, DataEnd, CountersBegin, CountersEnd, NamesBegin, NamesEnd);
51}

53COMPILER_RT_VISIBILITY
54uint64_t __llvm_profile_get_data_size(const __llvm_profile_data *Begin,
55 const __llvm_profile_data *End) {
56 intptr_t BeginI = (intptr_t)Begin, EndI = (intptr_t)End;
57 return ((EndI + sizeof(__llvm_profile_data) - 1) - BeginI) /
58 sizeof(__llvm_profile_data);
59}

61/// Calculate the number of padding bytes needed to add to \p Offset in order
62/// for (\p Offset + Padding) to be page-aligned.
63static uint64_t calculateBytesNeededToPageAlign(uint64_t Offset) {
64 uint64_t OffsetModPage = Offset % PageSize;
65 if (OffsetModPage > 0)
66 return PageSize - OffsetModPage;
67 return 0;
68}

70COMPILER_RT_VISIBILITY
71void __llvm_profile_get_padding_sizes_for_counters(
72 uint64_t DataSize, uint64_t CountersSize, uint64_t NamesSize,
73 uint64_t *PaddingBytesBeforeCounters, uint64_t *PaddingBytesAfterCounters,
74 uint64_t *PaddingBytesAfterNames) {
75 if (!__llvm_profile_is_continuous_mode_enabled() ||
76 lprofRuntimeCounterRelocation()) {
77 *PaddingBytesBeforeCounters = 0;
78 *PaddingBytesAfterCounters = 0;
79 *PaddingBytesAfterNames = __llvm_profile_get_num_padding_bytes(NamesSize);
80 return;
81 }
82
83 // In continuous mode, the file offsets for headers and for the start of
84 // counter sections need to be page-aligned.
85 uint64_t DataSizeInBytes = DataSize * sizeof(__llvm_profile_data);
86 uint64_t CountersSizeInBytes = CountersSize * sizeof(uint64_t);
87 *PaddingBytesBeforeCounters = calculateBytesNeededToPageAlign(
88 sizeof(__llvm_profile_header) + DataSizeInBytes);
89 *PaddingBytesAfterCounters =
90 calculateBytesNeededToPageAlign(CountersSizeInBytes);
91 *PaddingBytesAfterNames = calculateBytesNeededToPageAlign(NamesSize);
92}

94COMPILER_RT_VISIBILITY
95uint64_t __llvm_profile_get_size_for_buffer_internal(
96 const __llvm_profile_data *DataBegin, const __llvm_profile_data *DataEnd,
97 const uint64_t *CountersBegin, const uint64_t *CountersEnd,
98 const char *NamesBegin, const char *NamesEnd) {
99 /* Match logic in __llvm_profile_write_buffer(). */
100 const uint64_t NamesSize = (NamesEnd - NamesBegin) * sizeof(char);
101 uint64_t DataSize = __llvm_profile_get_data_size(DataBegin, DataEnd);
102 uint64_t CountersSize = CountersEnd - CountersBegin;
103
104 /* Determine how much padding is needed before/after the counters and after
105 * the names. */
106 uint64_t PaddingBytesBeforeCounters, PaddingBytesAfterCounters,
107 PaddingBytesAfterNames;
108 __llvm_profile_get_padding_sizes_for_counters(
109 DataSize, CountersSize, NamesSize, &PaddingBytesBeforeCounters,
110 &PaddingBytesAfterCounters, &PaddingBytesAfterNames);
111
112 return sizeof(__llvm_profile_header) +
113 (DataSize * sizeof(__llvm_profile_data)) + PaddingBytesBeforeCounters +
114 (CountersSize * sizeof(uint64_t)) + PaddingBytesAfterCounters +
115 NamesSize + PaddingBytesAfterNames;
116}

118COMPILER_RT_VISIBILITY
119void initBufferWriter(ProfDataWriter *BufferWriter, char *Buffer) {
120 BufferWriter->Write = lprofBufferWriter;
121 BufferWriter->WriterCtx = Buffer;
122}
123
124COMPILER_RT_VISIBILITY int __llvm_profile_write_buffer(char *Buffer) {
125 ProfDataWriter BufferWriter;
126 initBufferWriter(&BufferWriter, Buffer);
127 return lprofWriteData(&BufferWriter, 0, 0);
128}
129
130COMPILER_RT_VISIBILITY int __llvm_profile_write_buffer_internal(
131 char *Buffer, const __llvm_profile_data *DataBegin,
132 const __llvm_profile_data *DataEnd, const uint64_t *CountersBegin,
133 const uint64_t *CountersEnd, const char *NamesBegin, const char *NamesEnd) {
134 ProfDataWriter BufferWriter;
135 initBufferWriter(&BufferWriter, Buffer);
136 return lprofWriteDataImpl(&BufferWriter, DataBegin, DataEnd, CountersBegin,
137 CountersEnd, 0, NamesBegin, NamesEnd, 0);
138}
