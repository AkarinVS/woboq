23// Fatal internal map() or unmap() error (potentially OOM related).
24void NORETURN dieOnMapUnmapError(bool OutOfMemory) {
25 outputRaw("Scudo ERROR: internal map or unmap failure");
26 if (OutOfMemory)
27 outputRaw(" (OOM)");
28 outputRaw("\n");
29 die();
30}
