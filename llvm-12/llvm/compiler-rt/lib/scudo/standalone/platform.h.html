9#ifndef SCUDO_PLATFORM_H_
10#define SCUDO_PLATFORM_H_
11
12// Transitive includes of stdint.h specify some of the defines checked below.
13#include <stdint.h>
14
15#if defined(__linux__)
16#define SCUDO_LINUX 1
17#else
18#define SCUDO_LINUX 0
19#endif
20
21// See https://android.googlesource.com/platform/bionic/+/master/docs/defines.md
22#if defined(__BIONIC__)
23#define SCUDO_ANDROID 1
24#else
25#define SCUDO_ANDROID 0
26#endif
27
28#if defined(__Fuchsia__)
29#define SCUDO_FUCHSIA 1
30#else
31#define SCUDO_FUCHSIA 0
32#endif
33
34#if __LP64__
35#define SCUDO_WORDSIZE 64U
36#else
37#define SCUDO_WORDSIZE 32U
38#endif
39
40#if SCUDO_WORDSIZE == 64U
41#define FIRST_32_SECOND_64(a, b) (b)
42#else
43#define FIRST_32_SECOND_64(a, b) (a)
44#endif
45
46#ifndef SCUDO_CAN_USE_PRIMARY64
47#define SCUDO_CAN_USE_PRIMARY64 (SCUDO_WORDSIZE == 64U)
48#endif
49
50#ifndef SCUDO_MIN_ALIGNMENT_LOG
51// We force malloc-type functions to be aligned to std::max_align_t, but there
52// is no reason why the minimum alignment for all other functions can't be 8
53// bytes. Except obviously for applications making incorrect assumptions.
54// TODO(kostyak): define SCUDO_MIN_ALIGNMENT_LOG 3
55#define SCUDO_MIN_ALIGNMENT_LOG FIRST_32_SECOND_64(3, 4)
56#endif
57
58#if defined(__aarch64__)
59#define SCUDO_MMAP_RANGE_SIZE FIRST_32_SECOND_64(1ULL << 32, 1ULL << 48)
60#else
61#define SCUDO_MMAP_RANGE_SIZE FIRST_32_SECOND_64(1ULL << 32, 1ULL << 47)
62#endif
63
64// Older gcc have issues aligning to a constexpr, and require an integer.
65// See https://gcc.gnu.org/bugzilla/show_bug.cgi?id=56859 among others.
66#if defined(__powerpc__) || defined(__powerpc64__)
67#define SCUDO_CACHE_LINE_SIZE 128
68#else
69#define SCUDO_CACHE_LINE_SIZE 64
70#endif
71
72#define SCUDO_POINTER_FORMAT_LENGTH FIRST_32_SECOND_64(8, 12)
73
74#endif // SCUDO_PLATFORM_H_
