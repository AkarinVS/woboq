64inline bool systemDetectsMemoryTagFaultsTestOnly() {
65#ifndef PR_GET_TAGGED_ADDR_CTRL
66#define PR_GET_TAGGED_ADDR_CTRL 56
67#endif
68#ifndef PR_MTE_TCF_SHIFT
69#define PR_MTE_TCF_SHIFT 1
70#endif
71#ifndef PR_MTE_TCF_NONE
72#define PR_MTE_TCF_NONE (0UL << PR_MTE_TCF_SHIFT)
73#endif
74#ifndef PR_MTE_TCF_MASK
75#define PR_MTE_TCF_MASK (3UL << PR_MTE_TCF_SHIFT)
76#endif
77 return (static_cast<unsigned long>(
78 prctl(PR_GET_TAGGED_ADDR_CTRL, 0, 0, 0, 0)) &
79 PR_MTE_TCF_MASK) != PR_MTE_TCF_NONE;
80}

112inline uptr selectRandomTag(uptr Ptr, uptr ExcludeMask) {
113 uptr TaggedPtr;
114 __asm__ __volatile__(
115 ".arch_extension mte; irg %[TaggedPtr], %[Ptr], %[ExcludeMask]"
116 : [TaggedPtr] "=r"(TaggedPtr)
117 : [Ptr] "r"(Ptr), [ExcludeMask] "r"(ExcludeMask));
118 return TaggedPtr;
119}

121inline uptr storeTags(uptr Begin, uptr End) {
122 DCHECK(Begin % 16 == 0);
123 if (Begin != End) {
124 __asm__ __volatile__(
125 R"(
126 .arch_extension mte
127
128 1:
129 stzg %[Cur], [%[Cur]], #16
130 cmp %[Cur], %[End]
131 b.lt 1b
132 )"
133 : [Cur] "+&r"(Begin)
134 : [End] "r"(End)
135 : "memory");
136 }
137 return Begin;
138}

140inline void *prepareTaggedChunk(void *Ptr, uptr Size, uptr ExcludeMask,
141 uptr BlockEnd) {
142 // Prepare the granule before the chunk to store the chunk header by setting
143 // its tag to 0. Normally its tag will already be 0, but in the case where a
144 // chunk holding a low alignment allocation is reused for a higher alignment
145 // allocation, the chunk may already have a non-zero tag from the previous
146 // allocation.
147 __asm__ __volatile__(".arch_extension mte; stg %0, [%0, #-16]"
148 :
149 : "r"(Ptr)
150 : "memory");
151
152 uptr TaggedBegin, TaggedEnd;
153 setRandomTag(Ptr, Size, ExcludeMask, &TaggedBegin, &TaggedEnd);
154
155 // Finally, set the tag of the granule past the end of the allocation to 0,
156 // to catch linear overflows even if a previous larger allocation used the
157 // same block and tag. Only do this if the granule past the end is in our
158 // block, because this would otherwise lead to a SEGV if the allocation
159 // covers the entire block and our block is at the end of a mapping. The tag
160 // of the next block's header granule will be set to 0, so it will serve the
161 // purpose of catching linear overflows in this case.
162 uptr UntaggedEnd = untagPointer(TaggedEnd);
163 if (UntaggedEnd != BlockEnd)
164 __asm__ __volatile__(".arch_extension mte; stg %0, [%0]"
165 :
166 : "r"(UntaggedEnd)
167 : "memory");
168 return reinterpret_cast<void *>(TaggedBegin);
169}

171inline void resizeTaggedChunk(uptr OldPtr, uptr NewPtr, uptr BlockEnd) {
172 uptr RoundOldPtr = roundUpTo(OldPtr, 16);
173 if (RoundOldPtr >= NewPtr) {
174 // If the allocation is shrinking we just need to set the tag past the end
175 // of the allocation to 0. See explanation in prepareTaggedChunk above.
176 uptr RoundNewPtr = untagPointer(roundUpTo(NewPtr, 16));
177 if (RoundNewPtr != BlockEnd)
178 __asm__ __volatile__(".arch_extension mte; stg %0, [%0]"
179 :
180 : "r"(RoundNewPtr)
181 : "memory");
182 return;
183 }
184
185 __asm__ __volatile__(R"(
186 .arch_extension mte
187
188 // Set the memory tag of the region
189 // [roundUpTo(OldPtr, 16), roundUpTo(NewPtr, 16))
190 // to the pointer tag stored in OldPtr.
191 1:
192 stzg %[Cur], [%[Cur]], #16
193 cmp %[Cur], %[End]
194 b.lt 1b
195
196 // Finally, set the tag of the granule past the end of the allocation to 0.
197 and %[Cur], %[Cur], #(1 << 56) - 1
198 cmp %[Cur], %[BlockEnd]
199 b.eq 2f
200 stg %[Cur], [%[Cur]]
201
202 2:
203 )"
204 : [Cur] "+&r"(RoundOldPtr), [End] "+&r"(NewPtr)
205 : [BlockEnd] "r"(BlockEnd)
206 : "memory");
207}

209inline uptr loadTag(uptr Ptr) {
210 uptr TaggedPtr = Ptr;
211 __asm__ __volatile__(".arch_extension mte; ldg %0, [%0]"
212 : "+r"(TaggedPtr)
213 :
214 : "memory");
215 return TaggedPtr;
216}

275inline void setRandomTag(void *Ptr, uptr Size, uptr ExcludeMask,
276 uptr *TaggedBegin, uptr *TaggedEnd) {
277 *TaggedBegin = selectRandomTag(reinterpret_cast<uptr>(Ptr), ExcludeMask);
278 *TaggedEnd = storeTags(*TaggedBegin, *TaggedBegin + Size);
279}
