18// Hardware CRC32 is supported at compilation via the following:
19// - for i386 & x86_64: -msse4.2
20// - for ARM & AArch64: -march=armv8-a+crc or -mcrc
21// An additional check must be performed at runtime as well to make sure the
22// emitted instructions are valid on the target host.
23
24#if defined(__SSE4_2__) || defined(__ARM_FEATURE_CRC32)
25# ifdef __SSE4_2__
26# include <smmintrin.h>
27# define CRC32_INTRINSIC FIRST_32_SECOND_64(_mm_crc32_u32, _mm_crc32_u64)
28# endif
29# ifdef __ARM_FEATURE_CRC32
30# include <arm_acle.h>
31# define CRC32_INTRINSIC FIRST_32_SECOND_64(__crc32cw, __crc32cd)
32# endif
33#endif // defined(__SSE4_2__) || defined(__ARM_FEATURE_CRC32)

88inline u32 computeSoftwareCRC32(u32 Crc, uptr Data) {
89 for (uptr i = 0; i < sizeof(Data); i++) {
90 Crc = CRC32Table[(Crc ^ Data) & 0xff] ^ (Crc >> 8);
91 Data >>= 8;
92 }
93 return Crc;
94}
