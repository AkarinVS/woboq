11#include "FuzzerPlatform.h"
12#include <cstdint>
13
14#if LIBFUZZER_LINUX || LIBFUZZER_NETBSD || LIBFUZZER_FREEBSD || \
15 LIBFUZZER_FUCHSIA || LIBFUZZER_EMSCRIPTEN
16__attribute__((weak)) extern uint8_t __start___libfuzzer_extra_counters;
17__attribute__((weak)) extern uint8_t __stop___libfuzzer_extra_counters;
18
19namespace fuzzer {
20uint8_t *ExtraCountersBegin() { return &__start___libfuzzer_extra_counters; }
21uint8_t *ExtraCountersEnd() { return &__stop___libfuzzer_extra_counters; }
22ATTRIBUTE_NO_SANITIZE_ALL
23void ClearExtraCounters() { // hand-written memset, don't asan-ify.
24 uintptr_t *Beg = reinterpret_cast<uintptr_t*>(ExtraCountersBegin());
25 uintptr_t *End = reinterpret_cast<uintptr_t*>(ExtraCountersEnd());
26 for (; Beg < End; Beg++) {
27 *Beg = 0;
28 __asm__ __volatile__("" : : : "memory");
29 }
30}
31
32} // namespace fuzzer
33
34#else
35// TODO: implement for other platforms.
36namespace fuzzer {
37uint8_t *ExtraCountersBegin() { return nullptr; }
38uint8_t *ExtraCountersEnd() { return nullptr; }
39void ClearExtraCounters() {}
40} // namespace fuzzer
41
42#endif
