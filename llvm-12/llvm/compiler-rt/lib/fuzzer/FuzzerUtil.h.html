77inline std::pair<std::string, std::string> SplitBefore(std::string X,
78 std::string S) {
79 auto Pos = S.find(X);
80 if (Pos == std::string::npos)
81 return std::make_pair(S, "");
82 return std::make_pair(S.substr(0, Pos), S.substr(Pos));
83}

95inline size_t PageSize() { return 4096; }
96inline uint8_t *RoundUpByPage(uint8_t *P) {
97 uintptr_t X = reinterpret_cast<uintptr_t>(P);
98 size_t Mask = PageSize() - 1;
99 X = (X + Mask) & ~Mask;
100 return reinterpret_cast<uint8_t *>(X);
101}
102inline uint8_t *RoundDownByPage(uint8_t *P) {
103 uintptr_t X = reinterpret_cast<uintptr_t>(P);
104 size_t Mask = PageSize() - 1;
105 X = X & ~Mask;
106 return reinterpret_cast<uint8_t *>(X);
107}
