13#define DOUBLE_PRECISION
14#include "fp_lib.h"

16// Subtraction; flip the sign bit of b and add.
17COMPILER_RT_ABI fp_t __subdf3(fp_t a, fp_t b) {
18 return __adddf3(a, fromRep(toRep(b) ^ signBit));
19}
