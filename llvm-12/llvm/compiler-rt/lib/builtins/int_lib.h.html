109// Definitions for builtins unavailable on MSVC
110#if defined(_MSC_VER) && !defined(__clang__)
111#include <intrin.h>
112
113int __inline __builtin_ctz(uint32_t value) {
114 unsigned long trailing_zero = 0;
115 if (_BitScanForward(&trailing_zero, value))
116 return trailing_zero;
117 return 32;
118}
119
120int __inline __builtin_clz(uint32_t value) {
121 unsigned long leading_zero = 0;
122 if (_BitScanReverse(&leading_zero, value))
123 return 31 - leading_zero;
124 return 32;
125}
126
127#if defined(_M_ARM) || defined(_M_X64)
128int __inline __builtin_clzll(uint64_t value) {
129 unsigned long leading_zero = 0;
130 if (_BitScanReverse64(&leading_zero, value))
131 return 63 - leading_zero;
132 return 64;
133}
134#else
135int __inline __builtin_clzll(uint64_t value) {
136 if (value == 0)
137 return 64;
138 uint32_t msh = (uint32_t)(value >> 32);
139 uint32_t lsh = (uint32_t)(value & 0xFFFFFFFF);
140 if (msh != 0)
141 return __builtin_clz(msh);
142 return 32 + __builtin_clz(lsh);
143}
144#endif
145
146#define __builtin_clzl __builtin_clzll
147#endif // defined(_MSC_VER) && !defined(__clang__)
