15// Returns: -a
16
17// Effects: aborts if -a overflows
18
19COMPILER_RT_ABI di_int __negvdi2(di_int a) {
20 const di_int MIN = (di_int)1 << ((int)(sizeof(di_int) * CHAR_BIT) - 1);
21 if (a == MIN)
22 compilerrt_abort();
23 return -a;
24}
