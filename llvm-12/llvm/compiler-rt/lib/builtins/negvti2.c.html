13#include "int_lib.h"
14
15#ifdef CRT_HAS_128BIT
16
17// Returns: -a
18
19// Effects: aborts if -a overflows
20
21COMPILER_RT_ABI ti_int __negvti2(ti_int a) {
22 const ti_int MIN = (ti_int)1 << ((int)(sizeof(ti_int) * CHAR_BIT) - 1);
23 if (a == MIN)
24 compilerrt_abort();
25 return -a;
26}
27
28#endif // CRT_HAS_128BIT
