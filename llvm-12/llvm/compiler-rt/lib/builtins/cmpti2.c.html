13#include "int_lib.h"

15#ifdef CRT_HAS_128BIT
16
17// Returns: if (a < b) returns 0
18// if (a == b) returns 1
19// if (a > b) returns 2
20
21COMPILER_RT_ABI si_int __cmpti2(ti_int a, ti_int b) {
22 twords x;
23 x.all = a;
24 twords y;
25 y.all = b;
26 if (x.s.high < y.s.high)
27 return 0;
28 if (x.s.high > y.s.high)
29 return 2;
30 if (x.s.low < y.s.low)
31 return 0;
32 if (x.s.low > y.s.low)
33 return 2;
34 return 1;
35}
36
37#endif // CRT_HAS_128BIT
