17// Returns: convert a to a signed long long, rounding toward zero.
18
19// Assumption: long double is an intel 80 bit floating point type padded with 6
20// bytes di_int is a 64 bit integral type value in long double is representable
21// in di_int (no range checking performed)
22
23// gggg gggg gggg gggg gggg gggg gggg gggg | gggg gggg gggg gggg seee eeee eeee
24// eeee | 1mmm mmmm mmmm mmmm mmmm mmmm mmmm mmmm | mmmm mmmm mmmm mmmm mmmm
25// mmmm mmmm mmmm
26
27#if defined(_MSC_VER) && !defined(__clang__)
28// MSVC throws a warning about 'unitialized variable use' here,
29// disable it for builds that warn-as-error
30#pragma warning(push)
31#pragma warning(disable : 4700)
32#endif
33
34COMPILER_RT_ABI di_int __fixxfdi(long double a) {
35 const di_int di_max = (di_int)((~(du_int)0) / 2);
36 const di_int di_min = -di_max - 1;
37 long_double_bits fb;
38 fb.f = a;
39 int e = (fb.u.high.s.low & 0x00007FFF) - 16383;
40 if (e < 0)
41 return 0;
42 if ((unsigned)e >= sizeof(di_int) * CHAR_BIT)
43 return a > 0 ? di_max : di_min;
44 di_int s = -(si_int)((fb.u.high.s.low & 0x00008000) >> 15);
45 di_int r = fb.u.low.all;
46 r = (du_int)r >> (63 - e);
47 return (r ^ s) - s;
48}

51#pragma warning(pop)
52#endif
