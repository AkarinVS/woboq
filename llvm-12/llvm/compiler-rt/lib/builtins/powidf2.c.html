15// Returns: a ^ b

17COMPILER_RT_ABI double __powidf2(double a, int b) {
18 const int recip = b < 0;
19 double r = 1;
20 while (1) {
21 if (b & 1)
22 r *= a;
23 b /= 2;
24 if (b == 0)
25 break;
26 a *= a;
27 }
28 return recip ? 1 / r : r;
29}
