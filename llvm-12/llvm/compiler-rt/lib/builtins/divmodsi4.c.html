14#include "int_lib.h"

16// Returns: a / b, *rem = a % b
17
18COMPILER_RT_ABI si_int __divmodsi4(si_int a, si_int b, si_int *rem) {
19 const int bits_in_word_m1 = (int)(sizeof(si_int) * CHAR_BIT) - 1;
20 si_int s_a = a >> bits_in_word_m1; // s_a = a < 0 ? -1 : 0
21 si_int s_b = b >> bits_in_word_m1; // s_b = b < 0 ? -1 : 0
22 a = (a ^ s_a) - s_a; // negate if s_a == -1
23 b = (b ^ s_b) - s_b; // negate if s_b == -1
24 s_b ^= s_a; // sign of quotient
25 su_int r;
26 si_int q = (__udivmodsi4(a, b, &r) ^ s_b) - s_b; // negate if s_b == -1
27 *rem = (r ^ s_a) - s_a; // negate if s_a == -1
28 return q;
29}
