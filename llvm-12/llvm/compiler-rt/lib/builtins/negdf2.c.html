13#define DOUBLE_PRECISION
14#include "fp_lib.h"
15
16COMPILER_RT_ABI fp_t __negdf2(fp_t a) { return fromRep(toRep(a) ^ signBit); }
17
18#if defined(__ARM_EABI__)
19#if defined(COMPILER_RT_ARMHF_TARGET)
20AEABI_RTABI fp_t __aeabi_dneg(fp_t a) { return __negdf2(a); }

22COMPILER_RT_ALIAS(__negdf2, __aeabi_dneg)
