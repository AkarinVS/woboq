15// Returns: 1 if number of bits is odd else returns 0
16
17COMPILER_RT_ABI int __paritysi2(si_int a) {
18 su_int x = (su_int)a;
19 x ^= x >> 16;
20 x ^= x >> 8;
21 x ^= x >> 4;
22 return (0x6996 >> (x & 0xF)) & 1;
23}
