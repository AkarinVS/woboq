15// Returns: absolute value
16
17// Effects: aborts if abs(x) < 0
18
19COMPILER_RT_ABI di_int __absvdi2(di_int a) {
20 const int N = (int)(sizeof(di_int) * CHAR_BIT);
21 if (a == ((di_int)1 << (N - 1)))
22 compilerrt_abort();
23 const di_int t = a >> (N - 1);
24 return (a ^ t) - t;
25}
