15#ifdef CRT_HAS_128BIT
16
17// Returns: the number of leading 0-bits
18
19// Precondition: a != 0
20
21COMPILER_RT_ABI int __clzti2(ti_int a) {
22 twords x;
23 x.all = a;
24 const di_int f = -(x.s.high == 0);
25 return __builtin_clzll((x.s.high & ~f) | (x.s.low & f)) +
26 ((si_int)f & ((si_int)(sizeof(di_int) * CHAR_BIT)));
27}
28
29#endif // CRT_HAS_128BIT
