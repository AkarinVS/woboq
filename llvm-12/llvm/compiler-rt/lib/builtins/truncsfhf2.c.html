9#define SRC_SINGLE
10#define DST_HALF
11#include "fp_trunc_impl.inc"

13// Use a forwarding definition and noinline to implement a poor man's alias,
14// as there isn't a good cross-platform way of defining one.
15COMPILER_RT_ABI NOINLINE dst_t __truncsfhf2(float a) {
16 return __truncXfYf2__(a);
17}
18
19COMPILER_RT_ABI dst_t __gnu_f2h_ieee(float a) { return __truncsfhf2(a); }
20
