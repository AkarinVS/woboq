15COMPILER_RT_ABI uint32_t __bswapsi2(uint32_t u) {
16 return ((((u)&0xff000000) >> 24) |
17 (((u)&0x00ff0000) >> 8) |
18 (((u)&0x0000ff00) << 8) |
19 (((u)&0x000000ff) << 24));
20}
