17// Returns: 1 if number of bits is odd else returns 0
18
19COMPILER_RT_ABI int __parityti2(ti_int a) {
20 twords x;
21 dwords x2;
22 x.all = a;
23 x2.all = x.s.high ^ x.s.low;
24 su_int x3 = x2.s.high ^ x2.s.low;
25 x3 ^= x3 >> 16;
26 x3 ^= x3 >> 8;
27 x3 ^= x3 >> 4;
28 return (0x6996 >> (x3 & 0xF)) & 1;
29}
30
31#endif // CRT_HAS_128BIT
