15// Returns: a / b, *rem = a % b
16
17COMPILER_RT_ABI di_int __divmoddi4(di_int a, di_int b, di_int *rem) {
18 const int bits_in_dword_m1 = (int)(sizeof(di_int) * CHAR_BIT) - 1;
19 di_int s_a = a >> bits_in_dword_m1; // s_a = a < 0 ? -1 : 0
20 di_int s_b = b >> bits_in_dword_m1; // s_b = b < 0 ? -1 : 0
21 a = (a ^ s_a) - s_a; // negate if s_a == -1
22 b = (b ^ s_b) - s_b; // negate if s_b == -1
23 s_b ^= s_a; // sign of quotient
24 du_int r;
25 di_int q = (__udivmoddi4(a, b, &r) ^ s_b) - s_b; // negate if s_b == -1
26 *rem = (r ^ s_a) - s_a; // negate if s_a == -1
27 return q;
28}
