15// Returns: the index of the least significant 1-bit in a, or
16// the value zero if a is zero. The least significant bit is index one.
17
18COMPILER_RT_ABI int __ffsdi2(di_int a) {
19 dwords x;
20 x.all = a;
21 if (x.s.low == 0) {
22 if (x.s.high == 0)
23 return 0;
24 return ctzsi(x.s.high) + (1 + sizeof(si_int) * CHAR_BIT);
25 }
26 return ctzsi(x.s.low) + 1;
27}
