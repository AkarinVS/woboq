15// Returns: 1 if number of bits is odd else returns 0
16
17COMPILER_RT_ABI int __paritydi2(di_int a) {
18 dwords x;
19 x.all = a;
20 su_int x2 = x.s.high ^ x.s.low;
21 x2 ^= x2 >> 16;
22 x2 ^= x2 >> 8;
23 x2 ^= x2 >> 4;
24 return (0x6996 >> (x2 & 0xF)) & 1;
25}
