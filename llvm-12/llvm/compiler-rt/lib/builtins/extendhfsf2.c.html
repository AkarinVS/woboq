10#define DST_SINGLE
11#include "fp_extend_impl.inc"

13// Use a forwarding definition and noinline to implement a poor man's alias,
14// as there isn't a good cross-platform way of defining one.
15COMPILER_RT_ABI NOINLINE float __extendhfsf2(src_t a) {
16 return __extendXfYf2__(a);
17}
18
19COMPILER_RT_ABI float __gnu_h2f_ieee(src_t a) { return __extendhfsf2(a); }
