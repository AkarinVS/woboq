15// Returns: the index of the least significant 1-bit in a, or
16// the value zero if a is zero. The least significant bit is index one.
17
18COMPILER_RT_ABI int __ffssi2(si_int a) {
19 if (a == 0) {
20 return 0;
21 }
22 return ctzsi(a) + 1;
23}
