13#include "int_lib.h"

15// Returns: a - b
16
17// Effects: aborts if a - b overflows
18
19COMPILER_RT_ABI si_int __subvsi3(si_int a, si_int b) {
20 si_int s = (su_int)a - (su_int)b;
21 if (b >= 0) {
22 if (s > a)
23 compilerrt_abort();
24 } else {
25 if (s <= a)
26 compilerrt_abort();
27 }
28 return s;
29}
