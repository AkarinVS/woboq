7// long double __floatdixf(di_int a);

9#ifdef __i386__
10
11// This routine has some extra memory traffic, loading the 64-bit input via two
12// 32-bit loads, then immediately storing it back to the stack via a single 64-bit
13// store. This is to avoid a write-small, read-large stall.
14// However, if callers of this routine can be safely assumed to store the argument
15// via a 64-bt store, this is unnecessary memory traffic, and should be avoided.
16// It can be turned off by defining the TRUST_CALLERS_USE_64_BIT_STORES macro.
17
18.text
19.balign 4
20DEFINE_COMPILERRT_FUNCTION(__floatdixf)
21#ifndef TRUST_CALLERS_USE_64_BIT_STORES
22 movd 4(%esp), %xmm0
23 movd 8(%esp), %xmm1
24 punpckldq %xmm1, %xmm0
25 movq %xmm0, 4(%esp)
26#endif
27 fildll 4(%esp)
28 ret
29END_COMPILERRT_FUNCTION(__floatdixf)
30
31#endif // __i386__
32
33NO_EXEC_STACK_DIRECTIVE
