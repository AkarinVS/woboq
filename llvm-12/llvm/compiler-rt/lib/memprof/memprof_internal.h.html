79// Add convenient macro for interface functions that may be represented as
80// weak hooks.
81#define MEMPROF_MALLOC_HOOK(ptr, size) \
82 do { \
83 if (&__sanitizer_malloc_hook) \
84 __sanitizer_malloc_hook(ptr, size); \
85 RunMallocHooks(ptr, size); \
86 } while (false)
87#define MEMPROF_FREE_HOOK(ptr) \
88 do { \
89 if (&__sanitizer_free_hook) \
90 __sanitizer_free_hook(ptr); \
91 RunFreeHooks(ptr); \
92 } while (false)
