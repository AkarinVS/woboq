61void *MemprofDoesNotSupportStaticLinkage() {
62 // This will fail to link with -static.
63 return &_DYNAMIC;
64}
65
66uptr FindDynamicShadowStart() {
67 uptr shadow_size_bytes = MemToShadowSize(kHighMemEnd);
68 return MapDynamicShadow(shadow_size_bytes, SHADOW_SCALE,
69 /*min_shadow_base_alignment*/ 0, kHighMemEnd);
70}
71
72void ReadContextStack(void *context, uptr *stack, uptr *ssize) {
73 ucontext_t *ucp = (ucontext_t *)context;
74 *stack = (uptr)ucp->uc_stack.ss_sp;
75 *ssize = ucp->uc_stack.ss_size;
76}
