56class RecTy {
57public:
58 /// Subclass discriminator (for dyn_cast<> et al.)
59 enum RecTyKind {
60 BitRecTyKind,
61 BitsRecTyKind,
62 IntRecTyKind,
63 StringRecTyKind,
64 ListRecTyKind,
65 DagRecTyKind,
66 RecordRecTyKind
67 };
68
69private:
70 RecTyKind Kind;
71 /// ListRecTy of the list that has elements of this type.
72 ListRecTy *ListTy = nullptr;
73
74public:
75 RecTy(RecTyKind K) : Kind(K) {}
76 virtual ~RecTy() = default;
77
78 RecTyKind getRecTyKind() const { return Kind; }
79
80 virtual std::string getAsString() const = 0;
81 void print(raw_ostream &OS) const { OS << getAsString(); }
82 void dump() const;
83
84 /// Return true if all values of 'this' type can be converted to the specified
85 /// type.
86 virtual bool typeIsConvertibleTo(const RecTy *RHS) const;
87
88 /// Return true if 'this' type is equal to or a subtype of RHS. For example,
89 /// a bit set is not an int, but they are convertible.
90 virtual bool typeIsA(const RecTy *RHS) const;
91
92 /// Returns the type representing list<thistype>.
93 ListRecTy *getListTy();
94};

271class Init {
272protected:
273 /// Discriminator enum (for isa<>, dyn_cast<>, et al.)
274 ///
275 /// This enum is laid out by a preorder traversal of the inheritance
276 /// hierarchy, and does not contain an entry for abstract classes, as per
277 /// the recommendation in docs/HowToSetUpLLVMStyleRTTI.rst.
278 ///
279 /// We also explicitly include "first" and "last" values for each
280 /// interior node of the inheritance tree, to make it easier to read the
281 /// corresponding classof().
282 ///
283 /// We could pack these a bit tighter by not having the IK_FirstXXXInit
284 /// and IK_LastXXXInit be their own values, but that would degrade
285 /// readability for really no benefit.
286 enum InitKind : uint8_t {
287 IK_First, // unused; silence a spurious warning
288 IK_FirstTypedInit,
289 IK_BitInit,
290 IK_BitsInit,
291 IK_DagInit,
292 IK_DefInit,
293 IK_FieldInit,
294 IK_IntInit,
295 IK_ListInit,
296 IK_FirstOpInit,
297 IK_BinOpInit,
298 IK_TernOpInit,
299 IK_UnOpInit,
300 IK_LastOpInit,
301 IK_CondOpInit,
302 IK_FoldOpInit,
303 IK_IsAOpInit,
304 IK_StringInit,
305 IK_VarInit,
306 IK_VarListElementInit,
307 IK_VarBitInit,
308 IK_VarDefInit,
309 IK_LastTypedInit,
310 IK_UnsetInit
311 };
312
313private:
314 const InitKind Kind;
315
316protected:
317 uint8_t Opc; // Used by UnOpInit, BinOpInit, and TernOpInit
318
319private:
320 virtual void anchor();
321
322public:
323 /// Get the kind (type) of the value.
324 InitKind getKind() const { return Kind; }
325
326protected:
327 explicit Init(InitKind K, uint8_t Opc = 0) : Kind(K), Opc(Opc) {}
328
329public:
330 Init(const Init &) = delete;
331 Init &operator=(const Init &) = delete;
332 virtual ~Init() = default;
333
334 /// Is this a complete value with no unset (uninitialized) subvalues?
335 virtual bool isComplete() const { return true; }
336
337 /// Is this a concrete and fully resolved value without any references or
338 /// stuck operations? Unset values are concrete.
339 virtual bool isConcrete() const { return false; }
340
341 /// Print this value.
342 void print(raw_ostream &OS) const { OS << getAsString(); }
343
344 /// Convert this value to a literal form.
345 virtual std::string getAsString() const = 0;
346
347 /// Convert this value to a literal form,
348 /// without adding quotes around a string.
349 virtual std::string getAsUnquotedString() const { return getAsString(); }
350
351 /// Debugging method that may be called through a debugger; just
352 /// invokes print on stderr.
353 void dump() const;
354
355 /// If this value is convertible to type \p Ty, return a value whose
356 /// type is \p Ty, generating a !cast operation if required.
357 /// Otherwise, return null.
358 virtual Init *getCastTo(RecTy *Ty) const = 0;
359
360 /// Convert to a value whose type is \p Ty, or return null if this
361 /// is not possible. This can happen if the value's type is convertible
362 /// to \p Ty, but there are unresolved references.
363 virtual Init *convertInitializerTo(RecTy *Ty) const = 0;
364
365 /// This function is used to implement the bit range
366 /// selection operator. Given a value, it selects the specified bits,
367 /// returning them as a new \p Init of type \p bits. If it is not legal
368 /// to use the bit selection operator on this value, null is returned.
369 virtual Init *convertInitializerBitRange(ArrayRef<unsigned> Bits) const {
370 return nullptr;
371 }
372
373 /// This function is used to implement the list slice
374 /// selection operator. Given a value, it selects the specified list
375 /// elements, returning them as a new \p Init of type \p list. If it
376 /// is not legal to use the slice operator, null is returned.
377 virtual Init *convertInitListSlice(ArrayRef<unsigned> Elements) const {
378 return nullptr;
379 }
380
381 /// This function is used to implement the FieldInit class.
382 /// Implementors of this method should return the type of the named
383 /// field if they are of type record.
384 virtual RecTy *getFieldType(StringInit *FieldName) const {
385 return nullptr;
386 }
387
388 /// This function is used by classes that refer to other
389 /// variables which may not be defined at the time the expression is formed.
390 /// If a value is set for the variable later, this method will be called on
391 /// users of the value to allow the value to propagate out.
392 virtual Init *resolveReferences(Resolver &R) const {
393 return const_cast<Init *>(this);
394 }
395
396 /// Get the \p Init value of the specified bit.
397 virtual Init *getBit(unsigned Bit) const = 0;
398};

400inline raw_ostream &operator<<(raw_ostream &OS, const Init &I) {
401 I.print(OS); return OS;
402}

725class UnOpInit : public OpInit, public FoldingSetNode {
726public:
727 enum UnaryOp : uint8_t { CAST, NOT, HEAD, TAIL, SIZE, EMPTY, GETDAGOP };
728
729private:
730 Init *LHS;
731
732 UnOpInit(UnaryOp opc, Init *lhs, RecTy *Type)
733 : OpInit(IK_UnOpInit, Type, opc), LHS(lhs) {}
734
735public:
736 UnOpInit(const UnOpInit &) = delete;
737 UnOpInit &operator=(const UnOpInit &) = delete;
738
739 static bool classof(const Init *I) {
740 return I->getKind() == IK_UnOpInit;
741 }
742
743 static UnOpInit *get(UnaryOp opc, Init *lhs, RecTy *Type);
744
745 void Profile(FoldingSetNodeID &ID) const;
746
747 // Clone - Clone this operator, replacing arguments with the new list
748 OpInit *clone(ArrayRef<Init *> Operands) const override {
749 assert(Operands.size() == 1 &&
750 "Wrong number of operands for unary operation");
751 return UnOpInit::get(getOpcode(), *Operands.begin(), getType());
752 }
753
754 unsigned getNumOperands() const override { return 1; }
755
756 Init *getOperand(unsigned i) const override {
757 assert(i == 0 && "Invalid operand id for unary operator");
758 return getOperand();
759 }
760
761 UnaryOp getOpcode() const { return (UnaryOp)Opc; }
762 Init *getOperand() const { return LHS; }
763

1869struct LessRecordRegister {
1870 struct RecordParts {
1871 SmallVector<std::pair< bool, StringRef>, 4> Parts;
1872
1873 RecordParts(StringRef Rec) {
1874 if (Rec.empty())
1875 return;
1876
1877 size_t Len = 0;
1878 const char *Start = Rec.data();
1879 const char *Curr = Start;
1880 bool IsDigitPart = isDigit(Curr[0]);
1881 for (size_t I = 0, E = Rec.size(); I != E; ++I, ++Len) {
1882 bool IsDigit = isDigit(Curr[I]);
1883 if (IsDigit != IsDigitPart) {
1884 Parts.push_back(std::make_pair(IsDigitPart, StringRef(Start, Len)));
1885 Len = 0;
1886 Start = &Curr[I];
1887 IsDigitPart = isDigit(Curr[I]);
1888 }
1889 }
1890 // Push the last part.
1891 Parts.push_back(std::make_pair(IsDigitPart, StringRef(Start, Len)));
1892 }
1893
1894 size_t size() { return Parts.size(); }
1895
1896 std::pair<bool, StringRef> getPart(size_t i) {
1897 assert (i < Parts.size() && "Invalid idx!");
1898 return Parts[i];
1899 }
1900 };
1901
1902 bool operator()(const Record *Rec1, const Record *Rec2) const {
1903 RecordParts LHSParts(StringRef(Rec1->getName()));
1904 RecordParts RHSParts(StringRef(Rec2->getName()));
1905
1906 size_t LHSNumParts = LHSParts.size();
1907 size_t RHSNumParts = RHSParts.size();
1908 assert (LHSNumParts && RHSNumParts && "Expected at least one part!");
1909
1910 if (LHSNumParts != RHSNumParts)
1911 return LHSNumParts < RHSNumParts;
1912
1913 // We expect the registers to be of the form [_a-zA-Z]+([0-9]*[_a-zA-Z]*)*.
1914 for (size_t I = 0, E = LHSNumParts; I < E; I+=2) {
1915 std::pair<bool, StringRef> LHSPart = LHSParts.getPart(I);
1916 std::pair<bool, StringRef> RHSPart = RHSParts.getPart(I);
1917 // Expect even part to always be alpha.
1918 assert (LHSPart.first == false && RHSPart.first == false &&
1919 "Expected both parts to be alpha.");
1920 if (int Res = LHSPart.second.compare(RHSPart.second))
1921 return Res < 0;
1922 }
1923 for (size_t I = 1, E = LHSNumParts; I < E; I+=2) {
1924 std::pair<bool, StringRef> LHSPart = LHSParts.getPart(I);
1925 std::pair<bool, StringRef> RHSPart = RHSParts.getPart(I);
1926 // Expect odd part to always be numeric.
1927 assert (LHSPart.first == true && RHSPart.first == true &&
1928 "Expected both parts to be numeric.");
1929 if (LHSPart.second.size() != RHSPart.second.size())
1930 return LHSPart.second.size() < RHSPart.second.size();
1931
1932 unsigned LHSVal, RHSVal;
1933
1934 bool LHSFailed = LHSPart.second.getAsInteger(10, LHSVal); (void)LHSFailed;
1935 assert(!LHSFailed && "Unable to convert LHS to integer.");
1936 bool RHSFailed = RHSPart.second.getAsInteger(10, RHSVal); (void)RHSFailed;
1937 assert(!RHSFailed && "Unable to convert RHS to integer.");
1938
1939 if (LHSVal != RHSVal)
1940 return LHSVal < RHSVal;
1941 }
1942 return LHSNumParts < RHSNumParts;
