21/// Create a LinkGraph from a MachO relocatable object.
22///
23/// Note: The graph does not take ownership of the underlying buffer, nor copy
24/// its contents. The caller is responsible for ensuring that the object buffer
25/// outlives the graph.
26Expected<std::unique_ptr<LinkGraph>>
27createLinkGraphFromMachOObject(MemoryBufferRef ObjectBuffer);
28
29/// jit-link the given ObjBuffer, which must be a MachO object file.
30///
31/// Uses conservative defaults for GOT and stub handling based on the target
32/// platform.
33void link_MachO(std::unique_ptr<LinkGraph> G,
34 std::unique_ptr<JITLinkContext> Ctx);
