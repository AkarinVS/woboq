123 /// Returns true if RC is a sub-class of or equal to this class.
124 bool hasSubClassEq(const TargetRegisterClass *RC) const {

126 return (SubClassMask[ID / 32] >> (ID % 32)) & 1;

140 /// Returns a bit vector of subclasses, including this one.
141 /// The vector is indexed by class IDs.
142 ///
143 /// To use it, consider the returned array as a chunk of memory that
144 /// contains an array of bits of size NumRegClasses. Each 32-bit chunk
145 /// contains a bitset of the ID of the subclasses in big-endian style.
146
147 /// I.e., the representation of the memory from left to right at the
148 /// bit level looks like:
149 /// [31 30 ... 1 0] [ 63 62 ... 33 32] ...
150 /// [ XXX NumRegClasses NumRegClasses - 1 ... ]
151 /// Where the number represents the class ID and XXX bits that
152 /// should be ignored.
153 ///
154 /// See the implementation of hasSubClassEq for an example of how it
155 /// can be used.
156 const uint32_t *getSubClassMask() const {
157 return SubClassMask;
158 }
159
160 /// Returns a 0-terminated list of sub-register indices that project some
161 /// super-register class into this register class. The list has an entry for
162 /// each Idx such that:
163 ///
164 /// There exists SuperRC where:
165 /// For all Reg in SuperRC:
166 /// this->contains(Reg:Idx)
167 const uint16_t *getSuperRegIndices() const {
168 return SuperRegIndices;
169 }
170
171 /// Returns a NULL-terminated list of super-classes. The
172 /// classes are ordered by ID which is also a topological ordering from large
173 /// to small classes. The list does NOT include the current class.
174 sc_iterator getSuperClasses() const {
175 return SuperClasses;
176 }
177
178 /// Return true if this TargetRegisterClass is a subset
179 /// class of at least one other TargetRegisterClass.
180 bool isASubClass() const {
181 return SuperClasses[0] != nullptr;
182 }

184 /// Returns the preferred order for allocating registers from this register
185 /// class in MF. The raw order comes directly from the .td file and may
186 /// include reserved registers that are not allocatable.
187 /// Register allocators should also make sure to allocate
188 /// callee-saved registers only after all the volatiles are used. The
189 /// RegisterClassInfo class provides filtered allocation orders with
190 /// callee-saved registers moved to the end.
191 ///
192 /// The MachineFunction argument can be used to tune the allocatable
193 /// registers based on the characteristics of the function, subtarget, or
194 /// other criteria.
195 ///
196 /// By default, this method returns all registers in the class.
197 ArrayRef<MCPhysReg> getRawAllocationOrder(const MachineFunction &MF) const {
198 return OrderFunc ? OrderFunc(MF) : makeArrayRef(begin(), getNumRegs());
199 }

201 /// Returns the combination of all lane masks of register in this class.
202 /// The lane masks of the registers are the combination of all lane masks
203 /// of their subregisters. Returns 1 if there are no subregisters.
204 LaneBitmask getLaneMask() const {
205 return LaneMask;
206 }

261 // sometimes stack slots. The unsigned values are divided into these ranges:
262 //
263 // 0 Not a register, can be used as a sentinel.

266 // [2^31;2^32) Virtual registers assigned by MachineRegisterInfo.

268 // Further sentinels can be allocated from the small negative integers.

360 /// The lane masks returned by getSubRegIndexLaneMask() above can only be
361 /// used to determine if sub-registers overlap - they can't be used to
362 /// determine if a set of sub-registers completely cover another
363 /// sub-register.
364 ///
365 /// The X86 general purpose registers have two lanes corresponding to the
366 /// sub_8bit and sub_8bit_hi sub-registers. Both sub_32bit and sub_16bit have
367 /// lane masks '3', but the sub_16bit sub-register doesn't fully cover the
368 /// sub_32bit sub-register.
369 ///
370 /// On the other hand, the ARM NEON lanes fully cover their registers: The
371 /// dsub_0 sub-register is completely covered by the ssub_0 and ssub_1 lanes.
372 /// This is related to the CoveredBySubRegs property on register definitions.
373 ///
374 /// This function returns a bit mask of lanes that completely cover their
375 /// sub-registers. More precisely, given:
376 ///
377 /// Covering = getCoveringLanes();
378 /// MaskA = getSubRegIndexLaneMask(SubA);
379 /// MaskB = getSubRegIndexLaneMask(SubB);
380 ///
381 /// If (MaskA & ~(MaskB & Covering)) == 0, then SubA is completely covered by
382 /// SubB.
383 LaneBitmask getCoveringLanes() const { return CoveringLanes; }

385 /// Returns true if the two registers are equal or alias each other.
386 /// The registers may be virtual registers.
387 bool regsOverlap(Register regA, Register regB) const {
388 if (regA == regB) return true;
389 if (!regA.isPhysical() || !regB.isPhysical())
390 return false;
391
392 // Regunits are numerically ordered. Find a common unit.
393 MCRegUnitIterator RUA(regA.asMCReg(), this);
394 MCRegUnitIterator RUB(regB.asMCReg(), this);
395 do {
396 if (*RUA == *RUB) return true;
397 if (*RUA < *RUB) ++RUA;
398 else ++RUB;
399 } while (RUA.isValid() && RUB.isValid());
400 return false;
401 }

405 for (MCRegUnitIterator Units(Reg, this); Units.isValid(); ++Units)
406 if (Register(*Units) == RegUnit)
407 return true;
408 return false;
409 }

438 /// Return a mask of call-preserved registers for the given calling convention
439 /// on the current function. The mask should include all call-preserved
440 /// aliases. This is used by the register allocator to determine which
441 /// registers can be live across a call.
442 ///
443 /// The mask is an array containing (TRI::getNumRegs()+31)/32 entries.
444 /// A set bit indicates that all bits of the corresponding register are
445 /// preserved across the function call. The bit mask is expected to be
446 /// sub-register complete, i.e. if A is preserved, so are all its
447 /// sub-registers.
448 ///
449 /// Bits are numbered from the LSB, so the bit for physical register Reg can
450 /// be found as (Mask[Reg / 32] >> Reg % 32) & 1.
451 ///
452 /// A NULL pointer means that no register mask will be used, and call
453 /// instructions should use implicit-def operands to indicate call clobbered
454 /// registers.
455 ///
456 virtual const uint32_t *getCallPreservedMask(const MachineFunction &MF,
457 CallingConv::ID) const {
458 // The default mask clobbers everything. All targets should override.
459 return nullptr;
460 }

587 /// Return the subregister index you get from composing
588 /// two subregister indices.
589 ///
590 /// The special null sub-register index composes as the identity.
591 ///
592 /// If R:a:b is the same register as R:c, then composeSubRegIndices(a, b)
593 /// returns c. Note that composeSubRegIndices does not tell you about illegal
594 /// compositions. If R does not have a subreg a, or R:a does not have a subreg
595 /// b, composeSubRegIndices doesn't tell you.
596 ///
597 /// The ARM register Q0 has two D subregs dsub_0:D0 and dsub_1:D1. It also has
598 /// ssub_0:S0 - ssub_3:S3 subregs.
599 /// If you compose subreg indices dsub_1, ssub_0 you get ssub_2.
600 unsigned composeSubRegIndices(unsigned a, unsigned b) const {
601 if (!a) return b;
602 if (!b) return a;
603 return composeSubRegIndicesImpl(a, b);
604 }

652 /// Find a common super-register class if it exists.
653 ///
654 /// Find a register class, SuperRC and two sub-register indices, PreA and
655 /// PreB, such that:
656 ///

1061//
1062// Iterate over the possible super-registers for a given register class. The
1063// iterator will visit a list of pairs (Idx, Mask) corresponding to the
1064// possible classes of super-registers.
1065//
1066// Each bit mask will have at least one set bit, and each set bit in Mask
1067// corresponds to a SuperRC such that:
1068//
1069// For all Reg in SuperRC: Reg:Idx is in RC.
1070//
1071// The iterator can include (O, RC->getSubClassMask()) as the first entry which
1072// also satisfies the above requirement, assuming Reg:0 == Reg.
1073//
1074class SuperRegClassIterator {
1075 const unsigned RCMaskWords;
1076 unsigned SubReg = 0;
1077 const uint16_t *Idx;
1078 const uint32_t *Mask;
1079
1080public:
1081 /// Create a SuperRegClassIterator that visits all the super-register classes
1082 /// of RC. When IncludeSelf is set, also include the (0, sub-classes) entry.
1083 SuperRegClassIterator(const TargetRegisterClass *RC,
1084 const TargetRegisterInfo *TRI,
1085 bool IncludeSelf = false)
1086 : RCMaskWords((TRI->getNumRegClasses() + 31) / 32),
1087 Idx(RC->getSuperRegIndices()), Mask(RC->getSubClassMask()) {
1088 if (!IncludeSelf)
1089 ++*this;
1090 }
1091
1092 /// Returns true if this iterator is still pointing at a valid entry.
1093 bool isValid() const { return Idx; }
1094
1095 /// Returns the current sub-register index.
1096 unsigned getSubReg() const { return SubReg; }
1097
1098 /// Returns the bit mask of register classes that getSubReg() projects into
1099 /// RC.
1100 /// See TargetRegisterClass::getSubClassMask() for how to use it.
1101 const uint32_t *getMask() const { return Mask; }
1102
1103 /// Advance iterator to the next entry.
1104 void operator++() {
1105 assert(isValid() && "Cannot move iterator past end.");

1109 Idx = nullptr;

1113//===----------------------------------------------------------------------===//
1114// BitMaskClassIterator
1115//===----------------------------------------------------------------------===//
1116/// This class encapuslates the logic to iterate over bitmask returned by
1117/// the various RegClass related APIs.
1118/// E.g., this class can be used to iterate over the subclasses provided by
1119/// TargetRegisterClass::getSubClassMask or SuperRegClassIterator::getMask.
1120class BitMaskClassIterator {
1121 /// Total number of register classes.
1122 const unsigned NumRegClasses;
1123 /// Base index of CurrentChunk.
1124 /// In other words, the number of bit we read to get at the
1125 /// beginning of that chunck.
1126 unsigned Base = 0;
1127 /// Adjust base index of CurrentChunk.
1128 /// Base index + how many bit we read within CurrentChunk.
1129 unsigned Idx = 0;
1130 /// Current register class ID.
1131 unsigned ID = 0;
1132 /// Mask we are iterating over.
1133 const uint32_t *Mask;
1134 /// Current chunk of the Mask we are traversing.
1135 uint32_t CurrentChunk;
1136
1137 /// Move ID to the next set bit.
1138 void moveToNextID() {
1139 // If the current chunk of memory is empty, move to the next one,
1140 // while making sure we do not go pass the number of register
1141 // classes.
1142 while (!CurrentChunk) {
1143 // Move to the next chunk.
1144 Base += 32;
1145 if (Base >= NumRegClasses) {
1146 ID = NumRegClasses;
1147 return;
1148 }
1149 CurrentChunk = *++Mask;
1150 Idx = Base;
1151 }
1152 // Otherwise look for the first bit set from the right
1153 // (representation of the class ID is big endian).
1154 // See getSubClassMask for more details on the representation.
1155 unsigned Offset = countTrailingZeros(CurrentChunk);
1156 // Add the Offset to the adjusted base number of this chunk: Idx.
1157 // This is the ID of the register class.
1158 ID = Idx + Offset;
1159
1160 // Consume the zeros, if any, and the bit we just read
1161 // so that we are at the right spot for the next call.
1162 // Do not do Offset + 1 because Offset may be 31 and 32
1163 // will be UB for the shift, though in that case we could
1164 // have make the chunk being equal to 0, but that would
1165 // have introduced a if statement.
1166 moveNBits(Offset);
1167 moveNBits(1);
1168 }
1169
1170 /// Move \p NumBits Bits forward in CurrentChunk.
1171 void moveNBits(unsigned NumBits) {
1172 assert(NumBits < 32 && "Undefined behavior spotted!");
1173 // Consume the bit we read for the next call.
1174 CurrentChunk >>= NumBits;
1175 // Adjust the base for the chunk.
1176 Idx += NumBits;
1177 }
1178
1179public:
1180 /// Create a BitMaskClassIterator that visits all the register classes
1181 /// represented by \p Mask.
1182 ///
1183 /// \pre \p Mask != nullptr
1184 BitMaskClassIterator(const uint32_t *Mask, const TargetRegisterInfo &TRI)
1185 : NumRegClasses(TRI.getNumRegClasses()), Mask(Mask), CurrentChunk(*Mask) {
1186 // Move to the first ID.
1187 moveToNextID();
1188 }
1189
1190 /// Returns true if this iterator is still pointing at a valid entry.
1191 bool isValid() const { return getID() != NumRegClasses; }
1192
1193 /// Returns the current register class ID.
1194 unsigned getID() const { return ID; }
1195
1196 /// Advance iterator to the next entry.
1197 void operator++() {
1198 assert(isValid() && "Cannot move iterator past end.");
1199 moveToNextID();
1200 }
1201};
