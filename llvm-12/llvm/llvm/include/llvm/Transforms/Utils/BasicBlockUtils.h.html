181/// If this edge is a critical edge, insert a new node to split the critical
182/// edge. This will update the analyses passed in through the option struct.
183/// This returns the new block if the edge was split, null otherwise.
184///
185/// If MergeIdenticalEdges in the options struct is true (not the default),
186/// *all* edges from TI to the specified successor will be merged into the same
187/// critical edge block. This is most commonly interesting with switch
188/// instructions, which may have many edges to any one destination. This
189/// ensures that all edges to that dest go to one block instead of each going
190/// to a different block, but isn't the standard definition of a "critical
191/// edge".
192///
193/// It is invalid to call this function on a critical edge that starts at an
194/// IndirectBrInst. Splitting these edges will almost always create an invalid
195/// program because the address of the new block won't be the one that is jumped
196/// to.
197BasicBlock *SplitCriticalEdge(Instruction *TI, unsigned SuccNum,
198 const CriticalEdgeSplittingOptions &Options =
199 CriticalEdgeSplittingOptions(),
200 const Twine &BBName = "");
201
202inline BasicBlock *
203SplitCriticalEdge(BasicBlock *BB, succ_iterator SI,
204 const CriticalEdgeSplittingOptions &Options =
205 CriticalEdgeSplittingOptions()) {
206 return SplitCriticalEdge(BB->getTerminator(), SI.getSuccessorIndex(),
207 Options);
208}
209
210/// If the edge from *PI to BB is not critical, return false. Otherwise, split
211/// all edges between the two blocks and return true. This updates all of the
212/// same analyses as the other SplitCriticalEdge function. If P is specified, it
213/// updates the analyses described above.
214inline bool SplitCriticalEdge(BasicBlock *Succ, pred_iterator PI,
215 const CriticalEdgeSplittingOptions &Options =
216 CriticalEdgeSplittingOptions()) {
217 bool MadeChange = false;
218 Instruction *TI = (*PI)->getTerminator();
219 for (unsigned i = 0, e = TI->getNumSuccessors(); i != e; ++i)
220 if (TI->getSuccessor(i) == Succ)
221 MadeChange |= !!SplitCriticalEdge(TI, i, Options);
222 return MadeChange;
223}
224
225/// If an edge from Src to Dst is critical, split the edge and return true,
226/// otherwise return false. This method requires that there be an edge between
227/// the two blocks. It updates the analyses passed in the options struct
228inline BasicBlock *
229SplitCriticalEdge(BasicBlock *Src, BasicBlock *Dst,
230 const CriticalEdgeSplittingOptions &Options =
231 CriticalEdgeSplittingOptions()) {
232 Instruction *TI = Src->getTerminator();
233 unsigned i = 0;
234 while (true) {
235 assert(i != TI->getNumSuccessors() && "Edge doesn't exist!");
236 if (TI->getSuccessor(i) == Dst)
237 return SplitCriticalEdge(TI, i, Options);
238 ++i;
239 }
240}

468// Split critical edges where the source of the edge is an indirectbr
469// instruction. This isn't always possible, but we can handle some easy cases.
470// This is useful because MI is unable to split such critical edges,
471// which means it will not be able to sink instructions along those edges.
472// This is especially painful for indirect branches with many successors, where
473// we end up having to prepare all outgoing values in the origin block.
474//
475// Our normal algorithm for splitting critical edges requires us to update
476// the outgoing edges of the edge origin block, but for an indirectbr this
477// is hard, since it would require finding and updating the block addresses
478// the indirect branch uses. But if a block only has a single indirectbr
479// predecessor, with the others being regular branches, we can do it in a
480// different way.
481// Say we have A -> D, B -> D, I -> D where only I -> D is an indirectbr.
482// We can split D into D0 and D1, where D0 contains only the PHIs from D,
483// and D1 is the D block body. We can then duplicate D0 as D0A and D0B, and
484// create the following structure:
485// A -> D0A, B -> D0A, I -> D0B, D0A -> D1, D0B -> D1
486// If BPI and BFI aren't non-null, BPI/BFI will be updated accordingly.
487bool SplitIndirectBrCriticalEdges(Function &F,
488 BranchProbabilityInfo *BPI = nullptr,
489 BlockFrequencyInfo *BFI = nullptr);

491/// Given a set of incoming and outgoing blocks, create a "hub" such that every
492/// edge from an incoming block InBB to an outgoing block OutBB is now split
493/// into two edges, one from InBB to the hub and another from the hub to
494/// OutBB. The hub consists of a series of guard blocks, one for each outgoing
495/// block. Each guard block conditionally branches to the corresponding outgoing
496/// block, or the next guard block in the chain. These guard blocks are returned
497/// in the argument vector.
498///
499/// Since the control flow edges from InBB to OutBB have now been replaced, the
500/// function also updates any PHINodes in OutBB. For each such PHINode, the
501/// operands corresponding to incoming blocks are moved to a new PHINode in the
502/// hub, and the hub is made an operand of the original PHINode.
503///
504/// Input CFG:
505/// ----------
506///
507/// Def
508/// |
509/// v
510/// In1 In2
511/// | |
512/// | |
513/// v v
514/// Foo ---> Out1 Out2
515/// |
516/// v
517/// Use
518///
519///
520/// Create hub: Incoming = {In1, In2}, Outgoing = {Out1, Out2}
521/// ----------------------------------------------------------
522///
523/// Def
524/// |
525/// v
526/// In1 In2 Foo
527/// | Hub | |
528/// | + - - | - - + |
529/// | ' v ' V
530/// +------> Guard1 -----> Out1
531/// ' | '
532/// ' v '
533/// ' Guard2 -----> Out2
534/// ' ' |
535/// + - - - - - + |
536/// v
537/// Use
538///
539/// Limitations:
540/// -----------
541/// 1. This assumes that all terminators in the CFG are direct branches (the
542/// "br" instruction). The presence of any other control flow such as
543/// indirectbr, switch or callbr will cause an assert.
544///
545/// 2. The updates to the PHINodes are not sufficient to restore SSA
546/// form. Consider a definition Def, its use Use, incoming block In2 and
547/// outgoing block Out2, such that:
548/// a. In2 is reachable from D or contains D.
549/// b. U is reachable from Out2 or is contained in Out2.
550/// c. U is not a PHINode if U is contained in Out2.
551///
552/// Clearly, Def dominates Out2 since the program is valid SSA. But when the
553/// hub is introduced, there is a new path through the hub along which Use is
554/// reachable from entry without passing through Def, and SSA is no longer
555/// valid. To fix this, we need to look at all the blocks post-dominated by
556/// the hub on the one hand, and dominated by Out2 on the other. This is left
557/// for the caller to accomplish, since each specific use of this function
558/// may have additional information which simplifies this fixup. For example,
559/// see restoreSSA() in the UnifyLoopExits pass.
560BasicBlock *CreateControlFlowHub(DomTreeUpdater *DTU,
561 SmallVectorImpl<BasicBlock *> &GuardBlocks,
562 const SetVector<BasicBlock *> &Predecessors,
563 const SetVector<BasicBlock *> &Successors,
564 const StringRef Prefix);
