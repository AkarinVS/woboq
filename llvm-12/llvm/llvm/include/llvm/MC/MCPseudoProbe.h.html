9// This file contains the declaration of the MCPseudoProbe to support the pseudo
10// probe encoding for AutoFDO. Pseudo probes together with their inline context
11// are encoded in a DFS recursive way in the .pseudoprobe sections. For each
12// .pseudoprobe section, the encoded binary data consist of a single or mutiple
13// function records each for one outlined function. A function record has the
14// following format :
15//
16// FUNCTION BODY (one for each outlined function present in the text section)
17// GUID (uint64)
18// GUID of the function
19// NPROBES (ULEB128)
20// Number of probes originating from this function.
21// NUM_INLINED_FUNCTIONS (ULEB128)
22// Number of callees inlined into this function, aka number of
23// first-level inlinees
24// PROBE RECORDS
25// A list of NPROBES entries. Each entry contains:
26// INDEX (ULEB128)
27// TYPE (uint4)
28// 0 - block probe, 1 - indirect call, 2 - direct call
29// ATTRIBUTE (uint3)
30// reserved
31// ADDRESS_TYPE (uint1)
32// 0 - code address, 1 - address delta
33// CODE_ADDRESS (uint64 or ULEB128)
34// code address or address delta, depending on ADDRESS_TYPE
35// INLINED FUNCTION RECORDS
36// A list of NUM_INLINED_FUNCTIONS entries describing each of the inlined
37// callees. Each record contains:
38// INLINE SITE
39// GUID of the inlinee (uint64)
40// ID of the callsite probe (ULEB128)
41// FUNCTION BODY
42// A FUNCTION BODY entry describing the inlined function.
43//===----------------------------------------------------------------------===//

60enum class MCPseudoProbeFlag {
61 // If set, indicates that the probe is encoded as an address delta
62 // instead of a real code address.
63 AddressDelta = 0x1,
64};

66/// Instances of this class represent a pseudo probe instance for a pseudo probe
67/// table entry, which is created during a machine instruction is assembled and
68/// uses an address from a temporary label created at the current address in the
69/// current section.
70class MCPseudoProbe {
71 MCSymbol *Label;
72 uint64_t Guid;
73 uint64_t Index;
74 uint8_t Type;
75 uint8_t Attributes;
76
77public:
78 MCPseudoProbe(MCSymbol *Label, uint64_t Guid, uint64_t Index, uint64_t Type,
79 uint64_t Attributes)
80 : Label(Label), Guid(Guid), Index(Index), Type(Type),
81 Attributes(Attributes) {
82 assert(Type <= 0xFF && "Probe type too big to encode, exceeding 2^8");
83 assert(Attributes <= 0xFF &&
84 "Probe attributes too big to encode, exceeding 2^16");
85 }
86
87 MCSymbol *getLabel() const { return Label; }
88
89 uint64_t getGuid() const { return Guid; }
90
91 uint64_t getIndex() const { return Index; }
92
93 uint8_t getType() const { return Type; }
94
95 uint8_t getAttributes() const { return Attributes; }
96
97 void emit(MCObjectStreamer *MCOS, const MCPseudoProbe *LastProbe) const;
98};
99
100// An inline frame has the form <Guid, ProbeID>
101using InlineSite = std::tuple<uint64_t, uint32_t>;
102using MCPseudoProbeInlineStack = SmallVector<InlineSite, 8>;
103
104// A Tri-tree based data structure to group probes by inline stack.
105// A tree is allocated for a standalone .text section. A fake
106// instance is created as the root of a tree.
107// A real instance of this class is created for each function, either an
108// unlined function that has code in .text section or an inlined function.
109class MCPseudoProbeInlineTree {
110 uint64_t Guid;
111 // Set of probes that come with the function.
112 std::vector<MCPseudoProbe> Probes;
113 // Use std::map for a deterministic output.
114 std::map<InlineSite, MCPseudoProbeInlineTree *> Inlinees;
115
116 // Root node has a GUID 0.
117 bool isRoot() { return Guid == 0; }
118 MCPseudoProbeInlineTree *getOrAddNode(InlineSite Site);
119
120public:
121 MCPseudoProbeInlineTree() = default;
122 MCPseudoProbeInlineTree(uint64_t Guid) : Guid(Guid) {}
123 ~MCPseudoProbeInlineTree();
124 void addPseudoProbe(const MCPseudoProbe &Probe,
125 const MCPseudoProbeInlineStack &InlineStack);
126 void emit(MCObjectStreamer *MCOS, const MCPseudoProbe *&LastProbe);
127};
128
129/// Instances of this class represent the pseudo probes inserted into a compile
130/// unit.
131class MCPseudoProbeSection {
132public:
133 void addPseudoProbe(MCSection *Sec, const MCPseudoProbe &Probe,
134 const MCPseudoProbeInlineStack &InlineStack) {
135 MCProbeDivisions[Sec].addPseudoProbe(Probe, InlineStack);
136 }
137
138 // TODO: Sort by getOrdinal to ensure a determinstic section order
139 using MCProbeDivisionMap = std::map<MCSection *, MCPseudoProbeInlineTree>;
140
141private:
142 // A collection of MCPseudoProbe for each text section. The MCPseudoProbes
143 // are grouped by GUID of the functions where they are from and will be
144 // encoded by groups. In the comdat scenario where a text section really only
145 // contains the code of a function solely, the probes associated with a comdat
146 // function are still grouped by GUIDs due to inlining that can bring probes
147 // from different functions into one function.
148 MCProbeDivisionMap MCProbeDivisions;
149
150public:
151 const MCProbeDivisionMap &getMCProbes() const { return MCProbeDivisions; }
152
153 bool empty() const { return MCProbeDivisions.empty(); }
154
155 void emit(MCObjectStreamer *MCOS);
156};

158class MCPseudoProbeTable {
159 // A collection of MCPseudoProbe in the current module grouped by text
160 // sections. MCPseudoProbes will be encoded into a corresponding
161 // .pseudoprobe section. With functions emitted as separate comdats,
162 // a text section really only contains the code of a function solely, and the
163 // probes associated with the text section will be emitted into a standalone
164 // .pseudoprobe section that shares the same comdat group with the function.
165 MCPseudoProbeSection MCProbeSections;
166
167public:
168 static void emit(MCObjectStreamer *MCOS);
169
170 MCPseudoProbeSection &getProbeSections() { return MCProbeSections; }
171
172#ifndef NDEBUG
173 static int DdgPrintIndent;
174#endif
175};
