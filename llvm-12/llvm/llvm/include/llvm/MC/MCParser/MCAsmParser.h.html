36struct InlineAsmIdentifierInfo {
37 enum IdKind {
38 IK_Invalid, // Initial state. Unexpected after a successful parsing.
39 IK_Label, // Function/Label reference.
40 IK_EnumVal, // Value of enumeration type.
41 IK_Var // Variable.
42 };
43 // Represents an Enum value
44 struct EnumIdentifier {
45 int64_t EnumVal;
46 };
47 // Represents a label/function reference
48 struct LabelIdentifier {
49 void *Decl;
50 };
51 // Represents a variable
52 struct VariableIdentifier {
53 void *Decl;
54 bool IsGlobalLV;
55 unsigned Length;
56 unsigned Size;
57 unsigned Type;
58 };
59 // An InlineAsm identifier can only be one of those
60 union {
61 EnumIdentifier Enum;
62 LabelIdentifier Label;
63 VariableIdentifier Var;
64 };
65 bool isKind(IdKind kind) const { return Kind == kind; }
66 // Initializers
67 void setEnum(int64_t enumVal) {
68 assert(isKind(IK_Invalid) && "should be initialized only once");
69 Kind = IK_EnumVal;
70 Enum.EnumVal = enumVal;
71 }
72 void setLabel(void *decl) {
73 assert(isKind(IK_Invalid) && "should be initialized only once");
74 Kind = IK_Label;
75 Label.Decl = decl;
76 }
77 void setVar(void *decl, bool isGlobalLV, unsigned size, unsigned type) {
78 assert(isKind(IK_Invalid) && "should be initialized only once");
79 Kind = IK_Var;
80 Var.Decl = decl;
81 Var.IsGlobalLV = isGlobalLV;
82 Var.Size = size;
83 Var.Type = type;
84 Var.Length = size / type;
85 }
86 InlineAsmIdentifierInfo() : Kind(IK_Invalid) {}
87
88private:
89 // Discriminate using the current kind.
90 IdKind Kind;
91};

121/// Generic assembler parser interface, for use by target specific
122/// assembly parsers.
123class MCAsmParser {
124public:

126 using ExtensionDirectiveHandler =

138protected: // Can only create subclasses.

187 virtual bool defineMacro(StringRef Name, StringRef Value) { return true; }

212 /// Emit a warning at the location \p L, with the message \p Msg.

214 /// \return The return value is true, if warnings are fatal.

217 /// Return an error at the location \p L, with the message \p Msg. This

224 /// Emit an error at the location \p L, with the message \p Msg.
225 ///
226 /// \return The return value is always true, as an idiomatic convenience to
227 /// clients.
228 virtual bool printError(SMLoc L, const Twine &Msg, SMRange Range = None) = 0;

230 bool hasPendingError() { return !PendingErrors.empty(); }
231
232 bool printPendingErrors() {
233 bool rv = !PendingErrors.empty();
234 for (auto Err : PendingErrors) {
235 printError(Err.Loc, Twine(Err.Msg), Err.Range);
236 }
237 PendingErrors.clear();
238 return rv;
239 }
