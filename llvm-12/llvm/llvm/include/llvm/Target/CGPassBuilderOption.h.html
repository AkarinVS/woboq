14#ifndef LLVM_CODEGEN_PASSBUILDER_OPTION_H
15#define LLVM_CODEGEN_PASSBUILDER_OPTION_H
16
17#include "llvm/ADT/Optional.h"
18#include "llvm/ADT/StringRef.h"
19#include "llvm/Target/TargetOptions.h"
20#include <vector>
21
22namespace llvm {
23class TargetMachine;
24
25enum class RunOutliner { TargetDefault, AlwaysOutline, NeverOutline };
26enum class RegAllocType { Default, Basic, Fast, Greedy, PBQP };
27enum class CFLAAType { None, Steensgaard, Andersen, Both };
28
29// Not one-on-one but mostly corresponding to commandline options in
30// TargetPassConfig.cpp.
31struct CGPassBuilderOption {
32 Optional<bool> OptimizeRegAlloc;
33 Optional<bool> EnableIPRA;
34 bool DebugPM = false;
35 bool DisableVerify = false;
36 bool EnableImplicitNullChecks = false;
37 bool EnableBlockPlacementStats = false;
38 bool MISchedPostRA = false;
39 bool EarlyLiveIntervals = false;
40
41 bool DisableLSR = false;
42 bool DisableCGP = false;
43 bool PrintLSR = false;
44 bool DisableMergeICmps = false;
45 bool DisablePartialLibcallInlining = false;
46 bool DisableConstantHoisting = false;
47 bool PrintISelInput = false;
48 bool PrintGCInfo = false;
49 bool RequiresCodeGenSCCOrder = false;
50
51 RunOutliner EnableMachineOutliner = RunOutliner::TargetDefault;
52 RegAllocType RegAlloc = RegAllocType::Default;
53 CFLAAType UseCFLAA = CFLAAType::None;
54 Optional<GlobalISelAbortMode> EnableGlobalISelAbort;
55
56 Optional<bool> VerifyMachineCode;
57 Optional<bool> EnableFastISelOption;
58 Optional<bool> EnableGlobalISelOption;
59};
60
61CGPassBuilderOption getCGPassBuilderOption();
62
63} // namespace llvm
64
65#endif // LLVM_CODEGEN_PASSBUILDER_OPTION_H
