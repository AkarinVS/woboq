85/// Get the next available kind ID for a plugin diagnostic.
86/// Each time this function is called, it returns a different number.
87/// Therefore, a plugin that wants to "identify" its own classes
88/// with a dynamic identifier, just have to use this method to get a new ID
89/// and assign it to each of its classes.
90/// The returned ID will be greater than or equal to DK_FirstPluginKind.
91/// Thus, the plugin identifiers will not conflict with the
92/// DiagnosticKind values.

95/// This is the base abstract class for diagnostic reporting in
96/// the backend.
97/// The print method must be overloaded by the subclasses to print a
98/// user-friendly message in the client of the backend (let us call it a
99/// frontend).
100class DiagnosticInfo {
101private:
102 /// Kind defines the kind of report this is about.
103 const /* DiagnosticKind */ int Kind;
104 /// Severity gives the severity of the diagnostic.
105 const DiagnosticSeverity Severity;
106
107 virtual void anchor();
108public:
109 DiagnosticInfo(/* DiagnosticKind */ int Kind, DiagnosticSeverity Severity)
110 : Kind(Kind), Severity(Severity) {}
111
112 virtual ~DiagnosticInfo() = default;
113
114 /* DiagnosticKind */ int getKind() const { return Kind; }
115 DiagnosticSeverity getSeverity() const { return Severity; }
116
117 /// Print using the given \p DP a user-friendly message.
118 /// This is the default message that will be printed to the user.
119 /// It is used when the frontend does not directly take advantage
120 /// of the information contained in fields of the subclasses.
121 /// The printed message must not end with '.' nor start with a severity
122 /// keyword.
123 virtual void print(DiagnosticPrinter &DP) const = 0;
124};

128/// Diagnostic information for inline asm reporting.
129/// This is basically a message and an optional location.
130class DiagnosticInfoInlineAsm : public DiagnosticInfo {
131private:
132 /// Optional line information. 0 if not set.
133 unsigned LocCookie = 0;
134 /// Message to be reported.
135 const Twine &MsgStr;
136 /// Optional origin of the problem.
137 const Instruction *Instr = nullptr;
138
139public:
140 /// \p MsgStr is the message to be reported to the frontend.
141 /// This class does not copy \p MsgStr, therefore the reference must be valid
142 /// for the whole life time of the Diagnostic.
143 DiagnosticInfoInlineAsm(const Twine &MsgStr,
144 DiagnosticSeverity Severity = DS_Error)
145 : DiagnosticInfo(DK_InlineAsm, Severity), MsgStr(MsgStr) {}
146
147 /// \p LocCookie if non-zero gives the line number for this report.
148 /// \p MsgStr gives the message.
149 /// This class does not copy \p MsgStr, therefore the reference must be valid
150 /// for the whole life time of the Diagnostic.
151 DiagnosticInfoInlineAsm(unsigned LocCookie, const Twine &MsgStr,
152 DiagnosticSeverity Severity = DS_Error)
153 : DiagnosticInfo(DK_InlineAsm, Severity), LocCookie(LocCookie),
154 MsgStr(MsgStr) {}
155
156 /// \p Instr gives the original instruction that triggered the diagnostic.
157 /// \p MsgStr gives the message.
158 /// This class does not copy \p MsgStr, therefore the reference must be valid
159 /// for the whole life time of the Diagnostic.
160 /// Same for \p I.
161 DiagnosticInfoInlineAsm(const Instruction &I, const Twine &MsgStr,
162 DiagnosticSeverity Severity = DS_Error);
163
164 unsigned getLocCookie() const { return LocCookie; }
165 const Twine &getMsgStr() const { return MsgStr; }
166 const Instruction *getInstruction() const { return Instr; }
167
168 /// \see DiagnosticInfo::print.
169 void print(DiagnosticPrinter &DP) const override;
170
171 static bool classof(const DiagnosticInfo *DI) {
172 return DI->getKind() == DK_InlineAsm;
173 }

405/// Common features for diagnostics dealing with optimization remarks
406/// that are used by both IR and MIR passes.
407class DiagnosticInfoOptimizationBase : public DiagnosticInfoWithLocationBase {
408public:
409 /// Used to set IsVerbose via the stream interface.
410 struct setIsVerbose {};
411
412 /// When an instance of this is inserted into the stream, the arguments
413 /// following will not appear in the remark printed in the compiler output
414 /// (-Rpass) but only in the optimization record file
415 /// (-fsave-optimization-record).
416 struct setExtraArgs {};
417
418 /// Used in the streaming interface as the general argument type. It
419 /// internally converts everything into a key-value pair.
420 struct Argument {
421 std::string Key;
422 std::string Val;
423 // If set, the debug location corresponding to the value.
424 DiagnosticLocation Loc;
425
426 explicit Argument(StringRef Str = "") : Key("String"), Val(Str) {}
427 Argument(StringRef Key, const Value *V);
428 Argument(StringRef Key, const Type *T);
429 Argument(StringRef Key, StringRef S);
430 Argument(StringRef Key, const char *S) : Argument(Key, StringRef(S)) {};
431 Argument(StringRef Key, int N);
432 Argument(StringRef Key, float N);
433 Argument(StringRef Key, long N);
434 Argument(StringRef Key, long long N);
435 Argument(StringRef Key, unsigned N);
436 Argument(StringRef Key, unsigned long N);
437 Argument(StringRef Key, unsigned long long N);
438 Argument(StringRef Key, ElementCount EC);
439 Argument(StringRef Key, bool B) : Key(Key), Val(B ? "true" : "false") {}
440 Argument(StringRef Key, DebugLoc dl);
441 Argument(StringRef Key, InstructionCost C);
442 };
443
444 /// \p PassName is the name of the pass emitting this diagnostic. \p
445 /// RemarkName is a textual identifier for the remark (single-word,
446 /// camel-case). \p Fn is the function where the diagnostic is being emitted.
447 /// \p Loc is the location information to use in the diagnostic. If line table
448 /// information is available, the diagnostic will include the source code
449 /// location.
450 DiagnosticInfoOptimizationBase(enum DiagnosticKind Kind,
451 enum DiagnosticSeverity Severity,
452 const char *PassName, StringRef RemarkName,
453 const Function &Fn,
454 const DiagnosticLocation &Loc)
455 : DiagnosticInfoWithLocationBase(Kind, Severity, Fn, Loc),
456 PassName(PassName), RemarkName(RemarkName) {}
457
458 void insert(StringRef S);
459 void insert(Argument A);
460 void insert(setIsVerbose V);
461 void insert(setExtraArgs EA);
462
463 /// \see DiagnosticInfo::print.
464 void print(DiagnosticPrinter &DP) const override;
465
466 /// Return true if this optimization remark is enabled by one of
467 /// of the LLVM command line flags (-pass-remarks, -pass-remarks-missed,
468 /// or -pass-remarks-analysis). Note that this only handles the LLVM
469 /// flags. We cannot access Clang flags from here (they are handled
470 /// in BackendConsumer::OptimizationRemarkHandler).
471 virtual bool isEnabled() const = 0;
472
473 StringRef getPassName() const { return PassName; }
474 StringRef getRemarkName() const { return RemarkName; }
475 std::string getMsg() const;
476 Optional<uint64_t> getHotness() const { return Hotness; }
477 void setHotness(Optional<uint64_t> H) { Hotness = H; }
478
479 bool isVerbose() const { return IsVerbose; }
480
481 ArrayRef<Argument> getArgs() const { return Args; }
482
483 static bool classof(const DiagnosticInfo *DI) {
484 return (DI->getKind() >= DK_FirstRemark &&
485 DI->getKind() <= DK_LastRemark) ||
486 (DI->getKind() >= DK_FirstMachineRemark &&
487 DI->getKind() <= DK_LastMachineRemark);
488 }
489
490 bool isPassed() const {
491 return (getKind() == DK_OptimizationRemark ||
492 getKind() == DK_MachineOptimizationRemark);
493 }
494
495 bool isMissed() const {
496 return (getKind() == DK_OptimizationRemarkMissed ||
497 getKind() == DK_MachineOptimizationRemarkMissed);
498 }
499
500 bool isAnalysis() const {
501 return (getKind() == DK_OptimizationRemarkAnalysis ||
502 getKind() == DK_MachineOptimizationRemarkAnalysis);
503 }
504
505protected:
506 /// Name of the pass that triggers this report. If this matches the
507 /// regular expression given in -Rpass=regexp, then the remark will
508 /// be emitted.
509 const char *PassName;
510
511 /// Textual identifier for the remark (single-word, camel-case). Can be used
512 /// by external tools reading the output file for optimization remarks to
513 /// identify the remark.
514 StringRef RemarkName;
515
516 /// If profile information is available, this is the number of times the
517 /// corresponding code was executed in a profile instrumentation run.
518 Optional<uint64_t> Hotness;
519
520 /// Arguments collected via the streaming interface.
521 SmallVector<Argument, 4> Args;
522
523 /// The remark is expected to be noisy.
524 bool IsVerbose = false;
525
526 /// If positive, the index of the first argument that only appear in
527 /// the optimization records and not in the remark printed in the compiler
528 /// output.
529 int FirstExtraArgIndex = -1;
530};

614/// Common features for diagnostics dealing with optimization remarks
615/// that are used by IR passes.
616class DiagnosticInfoIROptimization : public DiagnosticInfoOptimizationBase {
617 void anchor() override;
618public:
619 /// \p PassName is the name of the pass emitting this diagnostic. \p
620 /// RemarkName is a textual identifier for the remark (single-word,
621 /// camel-case). \p Fn is the function where the diagnostic is being emitted.
622 /// \p Loc is the location information to use in the diagnostic. If line table
623 /// information is available, the diagnostic will include the source code
624 /// location. \p CodeRegion is IR value (currently basic block) that the
625 /// optimization operates on. This is currently used to provide run-time
626 /// hotness information with PGO.
627 DiagnosticInfoIROptimization(enum DiagnosticKind Kind,
628 enum DiagnosticSeverity Severity,
629 const char *PassName, StringRef RemarkName,
630 const Function &Fn,
631 const DiagnosticLocation &Loc,
632 const Value *CodeRegion = nullptr)
633 : DiagnosticInfoOptimizationBase(Kind, Severity, PassName, RemarkName, Fn,
634 Loc),
635 CodeRegion(CodeRegion) {}
636
637 /// This is ctor variant allows a pass to build an optimization remark
638 /// from an existing remark.
639 ///
640 /// This is useful when a transformation pass (e.g LV) wants to emit a remark
641 /// (\p Orig) generated by one of its analyses (e.g. LAA) as its own analysis
642 /// remark. The string \p Prepend will be emitted before the original
643 /// message.
644 DiagnosticInfoIROptimization(const char *PassName, StringRef Prepend,
645 const DiagnosticInfoIROptimization &Orig)
646 : DiagnosticInfoOptimizationBase(
647 (DiagnosticKind)Orig.getKind(), Orig.getSeverity(), PassName,
648 Orig.RemarkName, Orig.getFunction(), Orig.getLocation()),
649 CodeRegion(Orig.getCodeRegion()) {
650 *this << Prepend;
651 std::copy(Orig.Args.begin(), Orig.Args.end(), std::back_inserter(Args));
652 }
653
654 /// Legacy interface.
655 /// \p PassName is the name of the pass emitting this diagnostic.
656 /// \p Fn is the function where the diagnostic is being emitted. \p Loc is
657 /// the location information to use in the diagnostic. If line table
658 /// information is available, the diagnostic will include the source code
659 /// location. \p Msg is the message to show. Note that this class does not
660 /// copy this message, so this reference must be valid for the whole life time
661 /// of the diagnostic.
662 DiagnosticInfoIROptimization(enum DiagnosticKind Kind,
663 enum DiagnosticSeverity Severity,
664 const char *PassName, const Function &Fn,
665 const DiagnosticLocation &Loc, const Twine &Msg)
666 : DiagnosticInfoOptimizationBase(Kind, Severity, PassName, "", Fn, Loc) {
667 *this << Msg.str();
668 }
669
670 const Value *getCodeRegion() const { return CodeRegion; }
671
672 static bool classof(const DiagnosticInfo *DI) {
673 return DI->getKind() >= DK_FirstRemark && DI->getKind() <= DK_LastRemark;
674 }
675
676private:
677 /// The IR value (currently basic block) that the optimization operates on.
678 /// This is currently used to provide run-time hotness information with PGO.
679 const Value *CodeRegion = nullptr;
680};

682/// Diagnostic information for applied optimization remarks.
683class OptimizationRemark : public DiagnosticInfoIROptimization {
684public:
685 /// \p PassName is the name of the pass emitting this diagnostic. If this name
686 /// matches the regular expression given in -Rpass=, then the diagnostic will
687 /// be emitted. \p RemarkName is a textual identifier for the remark (single-
688 /// word, camel-case). \p Loc is the debug location and \p CodeRegion is the
689 /// region that the optimization operates on (currently only block is
690 /// supported).
691 OptimizationRemark(const char *PassName, StringRef RemarkName,
692 const DiagnosticLocation &Loc, const Value *CodeRegion);
693
694 /// Same as above, but the debug location and code region are derived from \p
695 /// Instr.
696 OptimizationRemark(const char *PassName, StringRef RemarkName,
697 const Instruction *Inst);
698
699 /// Same as above, but the debug location and code region are derived from \p
700 /// Func.
701 OptimizationRemark(const char *PassName, StringRef RemarkName,
702 const Function *Func);
703
704 static bool classof(const DiagnosticInfo *DI) {
705 return DI->getKind() == DK_OptimizationRemark;
706 }
707
708 /// \see DiagnosticInfoOptimizationBase::isEnabled.
709 bool isEnabled() const override;
710
711private:
712 /// This is deprecated now and only used by the function API below.
713 /// \p PassName is the name of the pass emitting this diagnostic. If
714 /// this name matches the regular expression given in -Rpass=, then the
