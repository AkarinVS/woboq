18#define EXPECT_VTY_EQ(LHS, RHS) \
19 ASSERT_NE(LHS, nullptr) << #LHS << " must not be null"; \
20 ASSERT_NE(RHS, nullptr) << #RHS << " must not be null"; \
21 EXPECT_EQ(LHS, RHS) << "Expect that " << #LHS << " == " << #RHS << " where " \
22 << #LHS << " = " << *LHS << " and " << #RHS << " = " \
23 << *RHS;
24
25#define EXPECT_VTY_NE(LHS, RHS) \
26 ASSERT_NE(LHS, nullptr) << #LHS << " must not be null"; \
27 ASSERT_NE(RHS, nullptr) << #RHS << " must not be null"; \
28 EXPECT_NE(LHS, RHS) << "Expect that " << #LHS << " != " << #RHS << " where " \
29 << #LHS << " = " << *LHS << " and " << #RHS << " = " \
30 << *RHS;

32TEST(VectorTypesTest, FixedLength) {
33 LLVMContext Ctx;
34
35 Type *Int8Ty = Type::getInt8Ty(Ctx);
36 Type *Int16Ty = Type::getInt16Ty(Ctx);
37 Type *Int32Ty = Type::getInt32Ty(Ctx);
38 Type *Int64Ty = Type::getInt64Ty(Ctx);
39 Type *Float64Ty = Type::getDoubleTy(Ctx);
40
41 auto *V16Int8Ty = FixedVectorType::get(Int8Ty, 16);
42 ASSERT_NE(nullptr, V16Int8Ty);
43 EXPECT_EQ(V16Int8Ty->getNumElements(), 16U);
44 EXPECT_EQ(V16Int8Ty->getElementType()->getScalarSizeInBits(), 8U);
45
46 auto *V8Int32Ty =
47 dyn_cast<FixedVectorType>(VectorType::get(Int32Ty, 8, false));
48 ASSERT_NE(nullptr, V8Int32Ty);
49 EXPECT_EQ(V8Int32Ty->getNumElements(), 8U);
50 EXPECT_EQ(V8Int32Ty->getElementType()->getScalarSizeInBits(), 32U);
51
52 auto *V8Int8Ty =
53 dyn_cast<FixedVectorType>(VectorType::get(Int8Ty, V8Int32Ty));
54 EXPECT_VTY_NE(V8Int32Ty, V8Int8Ty);
55 EXPECT_EQ(V8Int8Ty->getElementCount(), V8Int32Ty->getElementCount());
56 EXPECT_EQ(V8Int8Ty->getElementType()->getScalarSizeInBits(), 8U);
57
58 auto *V8Int32Ty2 =
59 dyn_cast<FixedVectorType>(VectorType::get(Int32Ty, V8Int32Ty));
60 EXPECT_VTY_EQ(V8Int32Ty, V8Int32Ty2);
61
62 auto *V8Int16Ty = dyn_cast<FixedVectorType>(
63 VectorType::get(Int16Ty, ElementCount::getFixed(8)));
64 ASSERT_NE(nullptr, V8Int16Ty);
65 EXPECT_EQ(V8Int16Ty->getNumElements(), 8U);
66 EXPECT_EQ(V8Int16Ty->getElementType()->getScalarSizeInBits(), 16U);
67
68 auto EltCnt = ElementCount::getFixed(4);
69 auto *V4Int64Ty = dyn_cast<FixedVectorType>(VectorType::get(Int64Ty, EltCnt));
70 ASSERT_NE(nullptr, V4Int64Ty);
71 EXPECT_EQ(V4Int64Ty->getNumElements(), 4U);
72 EXPECT_EQ(V4Int64Ty->getElementType()->getScalarSizeInBits(), 64U);
73
74 auto *V2Int64Ty = dyn_cast<FixedVectorType>(
75 VectorType::get(Int64Ty, EltCnt.divideCoefficientBy(2)));
76 ASSERT_NE(nullptr, V2Int64Ty);
77 EXPECT_EQ(V2Int64Ty->getNumElements(), 2U);
78 EXPECT_EQ(V2Int64Ty->getElementType()->getScalarSizeInBits(), 64U);
79
80 auto *V8Int64Ty =
81 dyn_cast<FixedVectorType>(VectorType::get(Int64Ty, EltCnt * 2));
82 ASSERT_NE(nullptr, V8Int64Ty);
83 EXPECT_EQ(V8Int64Ty->getNumElements(), 8U);
84 EXPECT_EQ(V8Int64Ty->getElementType()->getScalarSizeInBits(), 64U);
85
86 auto *V4Float64Ty =
87 dyn_cast<FixedVectorType>(VectorType::get(Float64Ty, EltCnt));
88 ASSERT_NE(nullptr, V4Float64Ty);
89 EXPECT_EQ(V4Float64Ty->getNumElements(), 4U);
90 EXPECT_EQ(V4Float64Ty->getElementType()->getScalarSizeInBits(), 64U);
91
92 auto *ExtTy = dyn_cast<FixedVectorType>(
93 VectorType::getExtendedElementVectorType(V8Int16Ty));
94 EXPECT_VTY_EQ(ExtTy, V8Int32Ty);
95 EXPECT_EQ(ExtTy->getNumElements(), 8U);
96 EXPECT_EQ(ExtTy->getElementType()->getScalarSizeInBits(), 32U);
97
98 auto *TruncTy = dyn_cast<FixedVectorType>(
99 VectorType::getTruncatedElementVectorType(V8Int32Ty));
100 EXPECT_VTY_EQ(TruncTy, V8Int16Ty);
101 EXPECT_EQ(TruncTy->getNumElements(), 8U);
102 EXPECT_EQ(TruncTy->getElementType()->getScalarSizeInBits(), 16U);
103
104 auto *HalvedTy = dyn_cast<FixedVectorType>(
105 VectorType::getHalfElementsVectorType(V4Int64Ty));
106 EXPECT_VTY_EQ(HalvedTy, V2Int64Ty);
107 EXPECT_EQ(HalvedTy->getNumElements(), 2U);
108 EXPECT_EQ(HalvedTy->getElementType()->getScalarSizeInBits(), 64U);
109
110 auto *DoubledTy = dyn_cast<FixedVectorType>(
111 VectorType::getDoubleElementsVectorType(V4Int64Ty));
112 EXPECT_VTY_EQ(DoubledTy, V8Int64Ty);
113 EXPECT_EQ(DoubledTy->getNumElements(), 8U);
114 EXPECT_EQ(DoubledTy->getElementType()->getScalarSizeInBits(), 64U);
115
116 auto *ConvTy = dyn_cast<FixedVectorType>(VectorType::getInteger(V4Float64Ty));
117 EXPECT_VTY_EQ(ConvTy, V4Int64Ty);
118 EXPECT_EQ(ConvTy->getNumElements(), 4U);
119 EXPECT_EQ(ConvTy->getElementType()->getScalarSizeInBits(), 64U);
120
121 EltCnt = V8Int64Ty->getElementCount();
122 EXPECT_EQ(EltCnt.getKnownMinValue(), 8U);
123 ASSERT_FALSE(EltCnt.isScalable());
124}

222TEST(VectorTypesTest, BaseVectorType) {
223 LLVMContext Ctx;
224
225 Type *Int16Ty = Type::getInt16Ty(Ctx);
226 Type *Int32Ty = Type::getInt32Ty(Ctx);
227
228 std::array<VectorType *, 8> VTys = {
229 VectorType::get(Int16Ty, ElementCount::getScalable(4)),
230 VectorType::get(Int16Ty, ElementCount::getFixed(4)),
231 VectorType::get(Int16Ty, ElementCount::getScalable(2)),
232 VectorType::get(Int16Ty, ElementCount::getFixed(2)),
233 VectorType::get(Int32Ty, ElementCount::getScalable(4)),
234 VectorType::get(Int32Ty, ElementCount::getFixed(4)),
235 VectorType::get(Int32Ty, ElementCount::getScalable(2)),
236 VectorType::get(Int32Ty, ElementCount::getFixed(2))};
237
238 /*
239 The comparison matrix is symmetric, so we only check the upper triangle:
240
241 (0,0) (0,1) (0,2) ... (0,7)
242 (1,0) (1,1) (1,2) .
243 (2,0) (2,1) (2,2) .
244 . . .
245 . .
246 . .
247 (7,0) ... (7,7)
248 */
249 for (size_t I = 0, IEnd = VTys.size(); I < IEnd; ++I) {
250 // test I == J
251 VectorType *VI = VTys[I];
252 ElementCount ECI = VI->getElementCount();
253 EXPECT_EQ(isa<ScalableVectorType>(VI), ECI.isScalable());
254
255 for (size_t J = I + 1, JEnd = VTys.size(); J < JEnd; ++J) {
256 // test I < J
257 VectorType *VJ = VTys[J];
258 EXPECT_VTY_NE(VI, VJ);
259
260 VectorType *VJPrime = VectorType::get(VI->getElementType(), VJ);
261 if (VI->getElementType() == VJ->getElementType()) {
262 EXPECT_VTY_EQ(VJ, VJPrime);
263 } else {
264 EXPECT_VTY_NE(VJ, VJPrime);
265 }
266
267 EXPECT_EQ(VJ->getTypeID(), VJPrime->getTypeID())
268 << "VJ and VJPrime are the same sort of vector";
269 }
270 }

273TEST(VectorTypesTest, FixedLenComparisons) {
274 LLVMContext Ctx;
275 DataLayout DL("");
276
277 Type *Int32Ty = Type::getInt32Ty(Ctx);
278 Type *Int64Ty = Type::getInt64Ty(Ctx);
279
280 auto *V2Int32Ty = FixedVectorType::get(Int32Ty, 2);
281 auto *V4Int32Ty = FixedVectorType::get(Int32Ty, 4);
282
283 auto *V2Int64Ty = FixedVectorType::get(Int64Ty, 2);
284
285 TypeSize V2I32Len = V2Int32Ty->getPrimitiveSizeInBits();
286 EXPECT_EQ(V2I32Len.getKnownMinSize(), 64U);
287 EXPECT_FALSE(V2I32Len.isScalable());
288
289 EXPECT_LT(V2Int32Ty->getPrimitiveSizeInBits().getFixedSize(),
290 V4Int32Ty->getPrimitiveSizeInBits().getFixedSize());
291 EXPECT_GT(V2Int64Ty->getPrimitiveSizeInBits().getFixedSize(),
292 V2Int32Ty->getPrimitiveSizeInBits().getFixedSize());
293 EXPECT_EQ(V4Int32Ty->getPrimitiveSizeInBits(),
294 V2Int64Ty->getPrimitiveSizeInBits());
295 EXPECT_NE(V2Int32Ty->getPrimitiveSizeInBits(),
296 V2Int64Ty->getPrimitiveSizeInBits());
297
298 // Check that a fixed-only comparison works for fixed size vectors.
299 EXPECT_EQ(V2Int64Ty->getPrimitiveSizeInBits().getFixedSize(),
300 V4Int32Ty->getPrimitiveSizeInBits().getFixedSize());
301
302 // Check the DataLayout interfaces.
303 EXPECT_EQ(DL.getTypeSizeInBits(V2Int64Ty), DL.getTypeSizeInBits(V4Int32Ty));
304 EXPECT_EQ(DL.getTypeSizeInBits(V2Int32Ty), 64U);
305 EXPECT_EQ(DL.getTypeSizeInBits(V2Int64Ty), 128U);
306 EXPECT_EQ(DL.getTypeStoreSize(V2Int64Ty), DL.getTypeStoreSize(V4Int32Ty));
307 EXPECT_NE(DL.getTypeStoreSizeInBits(V2Int32Ty),
308 DL.getTypeStoreSizeInBits(V2Int64Ty));
309 EXPECT_EQ(DL.getTypeStoreSizeInBits(V2Int32Ty), 64U);
310 EXPECT_EQ(DL.getTypeStoreSize(V2Int64Ty), 16U);
311 EXPECT_EQ(DL.getTypeAllocSize(V4Int32Ty), DL.getTypeAllocSize(V2Int64Ty));
312 EXPECT_NE(DL.getTypeAllocSizeInBits(V2Int32Ty),
313 DL.getTypeAllocSizeInBits(V2Int64Ty));
314 EXPECT_EQ(DL.getTypeAllocSizeInBits(V4Int32Ty), 128U);
315 EXPECT_EQ(DL.getTypeAllocSize(V2Int32Ty), 8U);
316 ASSERT_TRUE(DL.typeSizeEqualsStoreSize(V4Int32Ty));
317}

319TEST(VectorTypesTest, ScalableComparisons) {
320 LLVMContext Ctx;
321 DataLayout DL("");
322
323 Type *Int32Ty = Type::getInt32Ty(Ctx);
324 Type *Int64Ty = Type::getInt64Ty(Ctx);
325
326 auto *ScV2Int32Ty = ScalableVectorType::get(Int32Ty, 2);
327 auto *ScV4Int32Ty = ScalableVectorType::get(Int32Ty, 4);
328
329 auto *ScV2Int64Ty = ScalableVectorType::get(Int64Ty, 2);
330
331 TypeSize ScV2I32Len = ScV2Int32Ty->getPrimitiveSizeInBits();
332 EXPECT_EQ(ScV2I32Len.getKnownMinSize(), 64U);
333 EXPECT_TRUE(ScV2I32Len.isScalable());
334
335 EXPECT_LT(ScV2Int32Ty->getPrimitiveSizeInBits().getKnownMinSize(),
336 ScV4Int32Ty->getPrimitiveSizeInBits().getKnownMinSize());
337 EXPECT_GT(ScV2Int64Ty->getPrimitiveSizeInBits().getKnownMinSize(),
338 ScV2Int32Ty->getPrimitiveSizeInBits().getKnownMinSize());
339 EXPECT_EQ(ScV4Int32Ty->getPrimitiveSizeInBits().getKnownMinSize(),
340 ScV2Int64Ty->getPrimitiveSizeInBits().getKnownMinSize());
341 EXPECT_NE(ScV2Int32Ty->getPrimitiveSizeInBits().getKnownMinSize(),
342 ScV2Int64Ty->getPrimitiveSizeInBits().getKnownMinSize());
343
344 // Check the DataLayout interfaces.
345 EXPECT_EQ(DL.getTypeSizeInBits(ScV2Int64Ty),
346 DL.getTypeSizeInBits(ScV4Int32Ty));
347 EXPECT_EQ(DL.getTypeSizeInBits(ScV2Int32Ty).getKnownMinSize(), 64U);
348 EXPECT_EQ(DL.getTypeStoreSize(ScV2Int64Ty), DL.getTypeStoreSize(ScV4Int32Ty));
349 EXPECT_NE(DL.getTypeStoreSizeInBits(ScV2Int32Ty),
350 DL.getTypeStoreSizeInBits(ScV2Int64Ty));
351 EXPECT_EQ(DL.getTypeStoreSizeInBits(ScV2Int32Ty).getKnownMinSize(), 64U);
352 EXPECT_EQ(DL.getTypeStoreSize(ScV2Int64Ty).getKnownMinSize(), 16U);
353 EXPECT_EQ(DL.getTypeAllocSize(ScV4Int32Ty), DL.getTypeAllocSize(ScV2Int64Ty));
354 EXPECT_NE(DL.getTypeAllocSizeInBits(ScV2Int32Ty),
355 DL.getTypeAllocSizeInBits(ScV2Int64Ty));
356 EXPECT_EQ(DL.getTypeAllocSizeInBits(ScV4Int32Ty).getKnownMinSize(), 128U);
357 EXPECT_EQ(DL.getTypeAllocSize(ScV2Int32Ty).getKnownMinSize(), 8U);
358 ASSERT_TRUE(DL.typeSizeEqualsStoreSize(ScV4Int32Ty));
359}

361TEST(VectorTypesTest, CrossComparisons) {
362 LLVMContext Ctx;
363
364 Type *Int32Ty = Type::getInt32Ty(Ctx);
365
366 auto *V4Int32Ty = FixedVectorType::get(Int32Ty, 4);
367 auto *ScV4Int32Ty = ScalableVectorType::get(Int32Ty, 4);
368
369 // Even though the minimum size is the same, a scalable vector could be
370 // larger so we don't consider them to be the same size.
371 EXPECT_NE(V4Int32Ty->getPrimitiveSizeInBits(),
372 ScV4Int32Ty->getPrimitiveSizeInBits());
373 // If we are only checking the minimum, then they are the same size.
374 EXPECT_EQ(V4Int32Ty->getPrimitiveSizeInBits().getKnownMinSize(),
375 ScV4Int32Ty->getPrimitiveSizeInBits().getKnownMinSize());
376
377 // We can't use ordering comparisons (<,<=,>,>=) between scalable and
378 // non-scalable vector sizes.
379}
