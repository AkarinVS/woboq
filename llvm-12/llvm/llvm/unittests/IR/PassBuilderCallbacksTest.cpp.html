38/// A CRTP base for analysis mock handles
39///
40/// This class reconciles mocking with the value semantics implementation of the
41/// AnalysisManager. Analysis mock handles should derive from this class and
42/// call \c setDefault() in their constroctur for wiring up the defaults defined
43/// by this base with their mock run() and invalidate() implementations.
44template <typename DerivedT, typename IRUnitT,
45 typename AnalysisManagerT = AnalysisManager<IRUnitT>,
46 typename... ExtraArgTs>
47class MockAnalysisHandleBase {
48public:
49 class Analysis : public AnalysisInfoMixin<Analysis> {
50 friend AnalysisInfoMixin<Analysis>;
51 friend MockAnalysisHandleBase;
52 static AnalysisKey Key;
53
54 DerivedT *Handle;
55
56 Analysis(DerivedT &Handle) : Handle(&Handle) {
57 static_assert(std::is_base_of<MockAnalysisHandleBase, DerivedT>::value,
58 "Must pass the derived type to this template!");
59 }
60
61 public:
62 class Result {
63 friend MockAnalysisHandleBase;
64
65 DerivedT *Handle;
66
67 Result(DerivedT &Handle) : Handle(&Handle) {}
68
69 public:
70 // Forward invalidation events to the mock handle.
71 bool invalidate(IRUnitT &IR, const PreservedAnalyses &PA,
72 typename AnalysisManagerT::Invalidator &Inv) {
73 return Handle->invalidate(IR, PA, Inv);
74 }
75 };
76
77 Result run(IRUnitT &IR, AnalysisManagerT &AM, ExtraArgTs... ExtraArgs) {
78 return Handle->run(IR, AM, ExtraArgs...);
79 }
80 };
81
82 Analysis getAnalysis() { return Analysis(static_cast<DerivedT &>(*this)); }
83 typename Analysis::Result getResult() {
84 return typename Analysis::Result(static_cast<DerivedT &>(*this));
85 }
86 static StringRef getName() { return llvm::getTypeName<DerivedT>(); }
87
88protected:
89 // FIXME: MSVC seems unable to handle a lambda argument to Invoke from within
90 // the template, so we use a boring static function.
91 static bool invalidateCallback(IRUnitT &IR, const PreservedAnalyses &PA,
92 typename AnalysisManagerT::Invalidator &Inv) {
93 auto PAC = PA.template getChecker<Analysis>();
94 return !PAC.preserved() &&
95 !PAC.template preservedSet<AllAnalysesOn<IRUnitT>>();
96 }
97
98 /// Derived classes should call this in their constructor to set up default
99 /// mock actions. (We can't do this in our constructor because this has to
100 /// run after the DerivedT is constructed.)
101 void setDefaults() {
102 ON_CALL(static_cast<DerivedT &>(*this),
103 run(_, _, testing::Matcher<ExtraArgTs>(_)...))
104 .WillByDefault(Return(this->getResult()));
105 ON_CALL(static_cast<DerivedT &>(*this), invalidate(_, _, _))
106 .WillByDefault(Invoke(&invalidateCallback));
107 }
108};

135
136 public:

141 };

144

286/// Helper for HasName matcher that returns getName both for IRUnit and
287/// for IRUnit pointer wrapper into llvm::Any (wrapped by PassInstrumentation).
288template <typename IRUnitT> std::string getName(const IRUnitT &IR) {
289 return std::string(IR.getName());
290}
291
292template <> std::string getName(const StringRef &name) {
293 return std::string(name);
294}
295
296template <> std::string getName(const llvm::Any &WrappedIR) {
297 if (any_isa<const Module *>(WrappedIR))
298 return any_cast<const Module *>(WrappedIR)->getName().str();
299 if (any_isa<const Function *>(WrappedIR))
300 return any_cast<const Function *>(WrappedIR)->getName().str();
301 if (any_isa<const Loop *>(WrappedIR))
302 return any_cast<const Loop *>(WrappedIR)->getName().str();
303 if (any_isa<const LoopNest *>(WrappedIR))
304 return any_cast<const LoopNest *>(WrappedIR)->getName().str();
305 if (any_isa<const LazyCallGraph::SCC *>(WrappedIR))
306 return any_cast<const LazyCallGraph::SCC *>(WrappedIR)->getName();
307 return "<UNKNOWN>";
308}
309/// Define a custom matcher for objects which support a 'getName' method.
310///
311/// LLVM often has IR objects or analysis objects which expose a name
312/// and in tests it is convenient to match these by name for readability.
313/// Usually, this name is either a StringRef or a plain std::string. This
314/// matcher supports any type exposing a getName() method of this form whose
315/// return value is compatible with an std::ostream. For StringRef, this uses
316/// the shift operator defined above.
317///
318/// It should be used as:
319///
320/// HasName("my_function")
321///
322/// No namespace or other qualification is required.
323MATCHER_P(HasName, Name, "") {
324 *result_listener << "has name '" << getName(arg) << "'";
325 return Name == getName(arg);
326}
327
328MATCHER_P(HasNameRegex, Name, "") {
329 *result_listener << "has name '" << getName(arg) << "'";
330 llvm::Regex r(Name);
331 return r.match(getName(arg));
332}

334struct MockPassInstrumentationCallbacks {
335 PassInstrumentationCallbacks Callbacks;
336
337 MockPassInstrumentationCallbacks() {
338 ON_CALL(*this, runBeforePass(_, _)).WillByDefault(Return(true));
339 }
340 MOCK_METHOD2(runBeforePass, bool(StringRef PassID, llvm::Any));
341 MOCK_METHOD2(runBeforeSkippedPass, void(StringRef PassID, llvm::Any));
342 MOCK_METHOD2(runBeforeNonSkippedPass, void(StringRef PassID, llvm::Any));
343 MOCK_METHOD3(runAfterPass,
344 void(StringRef PassID, llvm::Any, const PreservedAnalyses &PA));
345 MOCK_METHOD2(runAfterPassInvalidated,
346 void(StringRef PassID, const PreservedAnalyses &PA));
347 MOCK_METHOD2(runBeforeAnalysis, void(StringRef PassID, llvm::Any));
348 MOCK_METHOD2(runAfterAnalysis, void(StringRef PassID, llvm::Any));
349
350 void registerPassInstrumentation() {
351 Callbacks.registerShouldRunOptionalPassCallback(
352 [this](StringRef P, llvm::Any IR) {
353 return this->runBeforePass(P, IR);
354 });
355 Callbacks.registerBeforeSkippedPassCallback(
356 [this](StringRef P, llvm::Any IR) {
357 this->runBeforeSkippedPass(P, IR);
358 });
359 Callbacks.registerBeforeNonSkippedPassCallback(
360 [this](StringRef P, llvm::Any IR) {
361 this->runBeforeNonSkippedPass(P, IR);
362 });
363 Callbacks.registerAfterPassCallback(
364 [this](StringRef P, llvm::Any IR, const PreservedAnalyses &PA) {
365 this->runAfterPass(P, IR, PA);
366 });
367 Callbacks.registerAfterPassInvalidatedCallback(
368 [this](StringRef P, const PreservedAnalyses &PA) {
369 this->runAfterPassInvalidated(P, PA);
370 });
371 Callbacks.registerBeforeAnalysisCallback([this](StringRef P, llvm::Any IR) {
372 return this->runBeforeAnalysis(P, IR);
373 });
374 Callbacks.registerAfterAnalysisCallback(
375 [this](StringRef P, llvm::Any IR) { this->runAfterAnalysis(P, IR); });
376 }

378 void ignoreNonMockPassInstrumentation(StringRef IRName) {
379 // Generic EXPECT_CALLs are needed to match instrumentation on unimportant
380 // parts of a pipeline that we do not care about (e.g. various passes added
381 // by default by PassBuilder - Verifier pass etc).
382 // Make sure to avoid ignoring Mock passes/analysis, we definitely want
383 // to check these explicitly.
384 EXPECT_CALL(*this,
385 runBeforePass(Not(HasNameRegex("Mock")), HasName(IRName)))
386 .Times(AnyNumber());
387 EXPECT_CALL(
388 *this, runBeforeSkippedPass(Not(HasNameRegex("Mock")), HasName(IRName)))
389 .Times(AnyNumber());
390 EXPECT_CALL(*this, runBeforeNonSkippedPass(Not(HasNameRegex("Mock")),
391 HasName(IRName)))
392 .Times(AnyNumber());
393 EXPECT_CALL(*this,
394 runAfterPass(Not(HasNameRegex("Mock")), HasName(IRName), _))
395 .Times(AnyNumber());
396 EXPECT_CALL(*this,
397 runBeforeAnalysis(Not(HasNameRegex("Mock")), HasName(IRName)))
398 .Times(AnyNumber());
399 EXPECT_CALL(*this,
400 runAfterAnalysis(Not(HasNameRegex("Mock")), HasName(IRName)))
401 .Times(AnyNumber());
402 }
403};

413/// takes care of setting up appropriate defaults.
414///
415/// The template specialization serves to extract the IRUnit and AM types from

422protected:
423 using IRUnitT = TestIRUnitT;
424 using AnalysisManagerT = AnalysisManager<TestIRUnitT, ExtraAnalysisArgTs...>;
425 using PassManagerT =

446 static PreservedAnalyses getAnalysisResult(IRUnitT &U, AnalysisManagerT &AM,
447 ExtraAnalysisArgTs &&... Args) {
448 (void)AM.template getResult<AnalysisT>(
449 U, std::forward<ExtraAnalysisArgTs>(Args)...);
450 return PreservedAnalyses::all();

452
453 PassBuilderCallbacksTest()
454 : M(parseIR(Context,
455 "declare void @bar()\n"
456 "define void @foo(i32 %n) {\n"
457 "entry:\n"
458 " br label %loop\n"
459 "loop:\n"
460 " %iv = phi i32 [ 0, %entry ], [ %iv.next, %loop ]\n"
461 " %iv.next = add i32 %iv, 1\n"
462 " tail call void @bar()\n"
463 " %cmp = icmp eq i32 %iv, %n\n"
464 " br i1 %cmp, label %exit, label %loop\n"
465 "exit:\n"
466 " ret void\n"
467 "}\n")),
468 CallbacksHandle(), PB(false, nullptr, PipelineTuningOptions(), None,
469 &CallbacksHandle.Callbacks),
470 PM(true), LAM(true), FAM(true), CGAM(true), AM(true) {
471
472 EXPECT_TRUE(&CallbacksHandle.Callbacks ==
473 PB.getPassInstrumentationCallbacks());
474
475 /// Register a callback for analysis registration.
476 ///
477 /// The callback is a function taking a reference to an AnalyisManager
478 /// object. When called, the callee gets to register its own analyses with
479 /// this PassBuilder instance.
480 PB.registerAnalysisRegistrationCallback([this](AnalysisManagerT &AM) {
481 // Register our mock analysis
482 AM.registerPass([this] { return AnalysisHandle.getAnalysis(); });
483 });
484
485 /// Register a callback for pipeline parsing.
486 ///
487 /// During parsing of a textual pipeline, the PassBuilder will call these
488 /// callbacks for each encountered pass name that it does not know. This
489 /// includes both simple pass names as well as names of sub-pipelines. In
490 /// the latter case, the InnerPipeline is not empty.
491 PB.registerPipelineParsingCallback(
492 [this](StringRef Name, PassManagerT &PM,
493 ArrayRef<PassBuilder::PipelineElement> InnerPipeline) {
494 /// Handle parsing of the names of analysis utilities such as
495 /// require<test-analysis> and invalidate<test-analysis> for our
496 /// analysis mock handle
497 if (parseAnalysisUtilityPasses<AnalysisT>("test-analysis", Name, PM))
498 return true;
499
500 /// Parse the name of our pass mock handle
501 if (Name == "test-transform") {
502 PM.addPass(PassHandle.getPass());
503 if (std::is_same<IRUnitT, Loop>::value)
504 PM.addPass(ExtraPassHandle.getPass());
505 return true;
506 }
507 return false;
508 });
509
510 /// Register builtin analyses and cross-register the analysis proxies
511 PB.registerModuleAnalyses(AM);
512 PB.registerCGSCCAnalyses(CGAM);
513 PB.registerFunctionAnalyses(FAM);
514 PB.registerLoopAnalyses(LAM);
515 PB.crossRegisterProxies(LAM, FAM, CGAM, AM);
516 }

584TEST_F(ModuleCallbacksTest, InstrumentedSkippedPasses) {
585 CallbacksHandle.registerPassInstrumentation();
586 // Non-mock instrumentation run here can safely be ignored.
587 CallbacksHandle.ignoreNonMockPassInstrumentation("<string>");
588 CallbacksHandle.ignoreNonMockPassInstrumentation("foo");
589
590 // Skip all passes by returning false. Pass managers and adaptor passes are
591 // also passes that observed by the callbacks.
592 EXPECT_CALL(CallbacksHandle, runBeforePass(_, _))
593 .WillRepeatedly(Return(false));
594
595 EXPECT_CALL(CallbacksHandle,
596 runBeforeSkippedPass(HasNameRegex("MockPassHandle"), _))
597 .Times(3);
598
599 EXPECT_CALL(AnalysisHandle, run(HasName("<string>"), _)).Times(0);
600 EXPECT_CALL(PassHandle, run(HasName("<string>"), _)).Times(0);
601
602 // As the pass is skipped there is no nonskippedpass/afterPass,
603 // beforeAnalysis/afterAnalysis as well.
604 EXPECT_CALL(CallbacksHandle,
605 runBeforeNonSkippedPass(HasNameRegex("MockPassHandle"), _))
606 .Times(0);
607 EXPECT_CALL(CallbacksHandle,
608 runAfterPass(HasNameRegex("MockPassHandle"), _, _))
609 .Times(0);
610 EXPECT_CALL(CallbacksHandle,
611 runBeforeAnalysis(HasNameRegex("MockAnalysisHandle"), _))
612 .Times(0);
613 EXPECT_CALL(CallbacksHandle,
614 runAfterAnalysis(HasNameRegex("MockAnalysisHandle"), _))
615 .Times(0);
616
617 // Order is important here. `Adaptor` expectations should be checked first
618 // because the its argument contains 'PassManager' (for example:
619 // ModuleToFunctionPassAdaptor{{.*}}PassManager{{.*}}). Check
620 // `runBeforeNonSkippedPass` and `runAfterPass` to show that they are not
621 // skipped.
622 //
623 // Pass managers are not ignored.
624 // 5 = (1) ModulePassManager + (2) FunctionPassMangers + (1) LoopPassManager +
625 // (1) CGSCCPassManager
626 EXPECT_CALL(CallbacksHandle,
627 runBeforeNonSkippedPass(HasNameRegex("PassManager"), _))
628 .Times(5);
629 EXPECT_CALL(
630 CallbacksHandle,
631 runBeforeNonSkippedPass(HasNameRegex("ModuleToFunctionPassAdaptor"), _))
632 .Times(1);
633 EXPECT_CALL(CallbacksHandle,
634 runBeforeNonSkippedPass(
635 HasNameRegex("ModuleToPostOrderCGSCCPassAdaptor"), _))
636 .Times(1);
637 EXPECT_CALL(
638 CallbacksHandle,
639 runBeforeNonSkippedPass(HasNameRegex("CGSCCToFunctionPassAdaptor"), _))
640 .Times(1);
641 EXPECT_CALL(
642 CallbacksHandle,
643 runBeforeNonSkippedPass(HasNameRegex("FunctionToLoopPassAdaptor"), _))
644 .Times(1);
645
646 // The `runAfterPass` checks are the same as these of
647 // `runBeforeNonSkippedPass`.
648 EXPECT_CALL(CallbacksHandle, runAfterPass(HasNameRegex("PassManager"), _, _))
649 .Times(5);
650 EXPECT_CALL(CallbacksHandle,
651 runAfterPass(HasNameRegex("ModuleToFunctionPassAdaptor"), _, _))
652 .Times(1);
653 EXPECT_CALL(
654 CallbacksHandle,
655 runAfterPass(HasNameRegex("ModuleToPostOrderCGSCCPassAdaptor"), _, _))
656 .Times(1);
657 EXPECT_CALL(CallbacksHandle,
658 runAfterPass(HasNameRegex("CGSCCToFunctionPassAdaptor"), _, _))
