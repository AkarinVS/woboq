43static const LLT V6P0 = LLT::vector(6, P0);

48TEST(GISelUtilsTest, getGCDType) {
49 EXPECT_EQ(S1, getGCDType(S1, S1));
50 EXPECT_EQ(S32, getGCDType(S32, S32));
51 EXPECT_EQ(S1, getGCDType(S1, S32));
52 EXPECT_EQ(S1, getGCDType(S32, S1));
53 EXPECT_EQ(S16, getGCDType(S16, S32));
54 EXPECT_EQ(S16, getGCDType(S32, S16));
55
56 EXPECT_EQ(V2S32, getGCDType(V2S32, V2S32));
57 EXPECT_EQ(S32, getGCDType(V3S32, V2S32));
58 EXPECT_EQ(S32, getGCDType(V2S32, V3S32));
59
60 EXPECT_EQ(V2S16, getGCDType(V4S16, V2S16));
61 EXPECT_EQ(V2S16, getGCDType(V2S16, V4S16));
62
63 EXPECT_EQ(V2S32, getGCDType(V4S32, V2S32));
64 EXPECT_EQ(V2S32, getGCDType(V2S32, V4S32));
65
66 EXPECT_EQ(S16, getGCDType(P0, S16));
67 EXPECT_EQ(S16, getGCDType(S16, P0));
68
69 EXPECT_EQ(S32, getGCDType(P0, S32));
70 EXPECT_EQ(S32, getGCDType(S32, P0));
71
72 EXPECT_EQ(P0, getGCDType(P0, S64));
73 EXPECT_EQ(S64, getGCDType(S64, P0));
74
75 EXPECT_EQ(S32, getGCDType(P0, P1));
76 EXPECT_EQ(S32, getGCDType(P1, P0));
77
78 EXPECT_EQ(P0, getGCDType(V3P0, V2P0));
79 EXPECT_EQ(P0, getGCDType(V2P0, V3P0));
80
81 EXPECT_EQ(P0, getGCDType(P0, V2P0));
82 EXPECT_EQ(P0, getGCDType(V2P0, P0));
83
84
85 EXPECT_EQ(V2P0, getGCDType(V2P0, V2P0));
86 EXPECT_EQ(P0, getGCDType(V3P0, V2P0));
87 EXPECT_EQ(P0, getGCDType(V2P0, V3P0));
88 EXPECT_EQ(V2P0, getGCDType(V4P0, V2P0));
89
90 EXPECT_EQ(V2P0, getGCDType(V2P0, V4P1));
91 EXPECT_EQ(V4P1, getGCDType(V4P1, V2P0));
92
93 EXPECT_EQ(V2P0, getGCDType(V4P0, V4P1));
94 EXPECT_EQ(V4P1, getGCDType(V4P1, V4P0));
95
96 // Elements have same size, but have different pointeriness, so prefer the
97 // original element type.
98 EXPECT_EQ(V2P0, getGCDType(V2P0, V4S64));
99 EXPECT_EQ(V2S64, getGCDType(V4S64, V2P0));
100
101 EXPECT_EQ(V2S16, getGCDType(V2S16, V4P1));
102 EXPECT_EQ(P1, getGCDType(V4P1, V2S16));
103 EXPECT_EQ(V2P1, getGCDType(V4P1, V4S16));
104 EXPECT_EQ(V4S16, getGCDType(V4S16, V2P1));
105
106 EXPECT_EQ(P0, getGCDType(P0, V2S64));
107 EXPECT_EQ(S64, getGCDType(V2S64, P0));
108
109 EXPECT_EQ(S16, getGCDType(V2S16, V3S16));
110 EXPECT_EQ(S16, getGCDType(V3S16, V2S16));
111 EXPECT_EQ(S16, getGCDType(V3S16, S16));
112 EXPECT_EQ(S16, getGCDType(S16, V3S16));
113
114 EXPECT_EQ(V2S16, getGCDType(V2S16, V2S32));
115 EXPECT_EQ(S32, getGCDType(V2S32, V2S16));
116
117 EXPECT_EQ(V4S8, getGCDType(V4S8, V2S32));
118 EXPECT_EQ(S32, getGCDType(V2S32, V4S8));
119
120 // Test cases where neither element type nicely divides.
121 EXPECT_EQ(LLT::scalar(3), getGCDType(LLT::vector(3, 5), LLT::vector(2, 6)));
122 EXPECT_EQ(LLT::scalar(3), getGCDType(LLT::vector(2, 6), LLT::vector(3, 5)));
123
124 // Have to go smaller than a pointer element.
125 EXPECT_EQ(LLT::scalar(3), getGCDType(LLT::vector(2, LLT::pointer(3, 6)),
126 LLT::vector(3, 5)));
127 EXPECT_EQ(LLT::scalar(3), getGCDType(LLT::vector(3, 5),
128 LLT::vector(2, LLT::pointer(3, 6))));
129
130 EXPECT_EQ(V4S8, getGCDType(V4S8, S32));
131 EXPECT_EQ(S32, getGCDType(S32, V4S8));
132 EXPECT_EQ(V4S8, getGCDType(V4S8, P1));
133 EXPECT_EQ(P1, getGCDType(P1, V4S8));
134
135 EXPECT_EQ(V2S8, getGCDType(V2S8, V4S16));
136 EXPECT_EQ(S16, getGCDType(V4S16, V2S8));
137
138 EXPECT_EQ(S8, getGCDType(V2S8, LLT::vector(4, 2)));
139 EXPECT_EQ(LLT::vector(4, 2), getGCDType(LLT::vector(4, 2), S8));
140
141
142 EXPECT_EQ(LLT::pointer(4, 8), getGCDType(LLT::vector(2, LLT::pointer(4, 8)),
143 LLT::vector(4, 2)));
144
145 EXPECT_EQ(LLT::vector(4, 2), getGCDType(LLT::vector(4, 2),
146 LLT::vector(2, LLT::pointer(4, 8))));
147
148 EXPECT_EQ(LLT::scalar(4), getGCDType(LLT::vector(3, 4), S8));
149 EXPECT_EQ(LLT::scalar(4), getGCDType(S8, LLT::vector(3, 4)));
150}

152TEST(GISelUtilsTest, getLCMType) {
153 EXPECT_EQ(S1, getLCMType(S1, S1));
154 EXPECT_EQ(S32, getLCMType(S32, S1));
155 EXPECT_EQ(S32, getLCMType(S1, S32));
156 EXPECT_EQ(S32, getLCMType(S32, S32));
157
158 EXPECT_EQ(S32, getLCMType(S32, S16));
159 EXPECT_EQ(S32, getLCMType(S16, S32));
160
161 EXPECT_EQ(S64, getLCMType(S64, P0));
162 EXPECT_EQ(P0, getLCMType(P0, S64));
163
164 EXPECT_EQ(P0, getLCMType(S32, P0));
165 EXPECT_EQ(P0, getLCMType(P0, S32));
166
167 EXPECT_EQ(S32, getLCMType(S32, P1));
168 EXPECT_EQ(P1, getLCMType(P1, S32));
169 EXPECT_EQ(P0, getLCMType(P0, P0));
170 EXPECT_EQ(P1, getLCMType(P1, P1));
171
172 EXPECT_EQ(P0, getLCMType(P0, P1));
173 EXPECT_EQ(P0, getLCMType(P1, P0));
174
175 EXPECT_EQ(V2S32, getLCMType(V2S32, V2S32));
176 EXPECT_EQ(V2S32, getLCMType(V2S32, S32));
177 EXPECT_EQ(V2S32, getLCMType(S32, V2S32));
178 EXPECT_EQ(V2S32, getLCMType(V2S32, V2S32));
179 EXPECT_EQ(V6S32, getLCMType(V2S32, V3S32));
180 EXPECT_EQ(V6S32, getLCMType(V3S32, V2S32));
181 EXPECT_EQ(LLT::vector(12, S32), getLCMType(V4S32, V3S32));
182 EXPECT_EQ(LLT::vector(12, S32), getLCMType(V3S32, V4S32));
183
184 EXPECT_EQ(V2P0, getLCMType(V2P0, V2P0));
185 EXPECT_EQ(V2P0, getLCMType(V2P0, P0));
186 EXPECT_EQ(V2P0, getLCMType(P0, V2P0));
187 EXPECT_EQ(V2P0, getLCMType(V2P0, V2P0));
188 EXPECT_EQ(V6P0, getLCMType(V2P0, V3P0));
189 EXPECT_EQ(V6P0, getLCMType(V3P0, V2P0));
190 EXPECT_EQ(LLT::vector(12, P0), getLCMType(V4P0, V3P0));
191 EXPECT_EQ(LLT::vector(12, P0), getLCMType(V3P0, V4P0));
192
193 EXPECT_EQ(LLT::vector(12, S64), getLCMType(V4S64, V3P0));
194 EXPECT_EQ(LLT::vector(12, P0), getLCMType(V3P0, V4S64));
195
196 EXPECT_EQ(LLT::vector(12, P0), getLCMType(V4P0, V3S64));
197 EXPECT_EQ(LLT::vector(12, S64), getLCMType(V3S64, V4P0));
198
199 EXPECT_EQ(V2P0, getLCMType(V2P0, S32));
200 EXPECT_EQ(V4S32, getLCMType(S32, V2P0));
201 EXPECT_EQ(V2P0, getLCMType(V2P0, S64));
202 EXPECT_EQ(V2S64, getLCMType(S64, V2P0));
203
204
205 EXPECT_EQ(V2P0, getLCMType(V2P0, V2P1));
206 EXPECT_EQ(V4P1, getLCMType(V2P1, V2P0));
207
208 EXPECT_EQ(V2P0, getLCMType(V2P0, V4P1));
209 EXPECT_EQ(V4P1, getLCMType(V4P1, V2P0));

211
212 EXPECT_EQ(V2S32, getLCMType(V2S32, S64));
213 EXPECT_EQ(S64, getLCMType(S64, V2S32));

215 EXPECT_EQ(V4S16, getLCMType(V4S16, V2S32));
216 EXPECT_EQ(V2S32, getLCMType(V2S32, V4S16));

218 EXPECT_EQ(V2S32, getLCMType(V2S32, V4S8));
219 EXPECT_EQ(V8S8, getLCMType(V4S8, V2S32));
220
221 EXPECT_EQ(V2S16, getLCMType(V2S16, V4S8));
222 EXPECT_EQ(V4S8, getLCMType(V4S8, V2S16));
223
224 EXPECT_EQ(LLT::vector(6, S16), getLCMType(V3S16, V4S8));
225 EXPECT_EQ(LLT::vector(12, S8), getLCMType(V4S8, V3S16));
226 EXPECT_EQ(V4S16, getLCMType(V4S16, V4S8));
227 EXPECT_EQ(V8S8, getLCMType(V4S8, V4S16));
228
229 EXPECT_EQ(LLT::vector(6, 4), getLCMType(LLT::vector(3, 4), S8));
230 EXPECT_EQ(LLT::vector(3, 8), getLCMType(S8, LLT::vector(3, 4)));

232 EXPECT_EQ(LLT::vector(6, 4),
233 getLCMType(LLT::vector(3, 4), LLT::pointer(4, 8)));
234 EXPECT_EQ(LLT::vector(3, LLT::pointer(4, 8)),
235 getLCMType(LLT::pointer(4, 8), LLT::vector(3, 4)));
236
237 EXPECT_EQ(V2S64, getLCMType(V2S64, P0));
238 EXPECT_EQ(V2P0, getLCMType(P0, V2S64));
239
240 EXPECT_EQ(V2S64, getLCMType(V2S64, P1));
241 EXPECT_EQ(V4P1, getLCMType(P1, V2S64));
