201// Check that we don't try to analysis PHIs progression.
202// Setting a deep enough max depth would allow to effectively simulate
203// what happens in the loop.
204// Thus, with a deep enough depth, we could actually figure out
205// that %14's zero known bits are actually at least what we know
206// for %10, right shifted by one.
207// However, this process is super expensive compile-time wise and
208// we don't want to reach that conclusion while playing with max depth.
209// For now, the analysis just stops and assumes it knows nothing
210// on PHIs, but eventually we could teach it how to properly track
211// phis that loop back without relying on the luck effect of max
212// depth.
213TEST_F(AArch64GISelMITest, TestKnownBitsDecreasingCstPHIWithLoop) {
214 StringRef MIRString = " bb.10:\n"
215 " %10:_(s8) = G_CONSTANT i8 5\n"
216 " %11:_(s8) = G_CONSTANT i8 1\n"
217 "\n"
218 " bb.12:\n"
219 " %13:_(s8) = PHI %10(s8), %bb.10, %14(s8), %bb.12\n"
220 " %14:_(s8) = G_LSHR %13, %11\n"
221 " %15:_(s8) = COPY %14\n"
222 " G_BR %bb.12\n";
223 setUp(MIRString);
224 if (!TM)
225 return;
226 Register CopyReg = Copies[Copies.size() - 1];
227 MachineInstr *FinalCopy = MRI->getVRegDef(CopyReg);
228 Register SrcReg = FinalCopy->getOperand(1).getReg();
229 Register DstReg = FinalCopy->getOperand(0).getReg();
230 GISelKnownBits Info(*MF, /*MaxDepth=*/24);
231 KnownBits Res = Info.getKnownBits(SrcReg);
232 EXPECT_EQ((uint64_t)0, Res.One.getZExtValue());
233 // A single iteration on the PHI (%13) gives:
234 // %10 has known zero of 0xFA
235 // %12 has known zero of 0x80 (we shift right by one so high bit is zero)
236 // Therefore, %14's known zero are 0x80 shifted by one 0xC0.
237 // If we had simulated the loop we could have more zero bits, basically
238 // up to 0xFC (count leading zero of 5, + 1).
239 EXPECT_EQ((uint64_t)0xC0, Res.Zero.getZExtValue());
240
241 KnownBits Res2 = Info.getKnownBits(DstReg);
242 EXPECT_EQ(Res.One.getZExtValue(), Res2.One.getZExtValue());
243 EXPECT_EQ(Res.Zero.getZExtValue(), Res2.Zero.getZExtValue());
244}

612TEST_F(AArch64GISelMITest, TestMetadata) {
613 StringRef MIRString = " %imp:_(p0) = G_IMPLICIT_DEF\n"
614 " %load:_(s8) = G_LOAD %imp(p0) :: (load 1)\n"
615 " %ext:_(s32) = G_ZEXT %load(s8)\n"
616 " %cst:_(s32) = G_CONSTANT i32 1\n"
617 " %and:_(s32) = G_AND %ext, %cst\n"
618 " %copy:_(s32) = COPY %and(s32)\n";
619 setUp(MIRString);
620 if (!TM)
621 return;
622
623 Register CopyReg = Copies[Copies.size() - 1];
624 MachineInstr *FinalCopy = MRI->getVRegDef(CopyReg);
625 Register SrcReg = FinalCopy->getOperand(1).getReg();
626
627 // We need a load with a metadata range for this to break. Fudge the load in
628 // the string and replace it with something we can work with.
629 MachineInstr *And = MRI->getVRegDef(SrcReg);
630 MachineInstr *Ext = MRI->getVRegDef(And->getOperand(1).getReg());
631 MachineInstr *Load = MRI->getVRegDef(Ext->getOperand(1).getReg());
632 IntegerType *Int8Ty = Type::getInt8Ty(Context);
633
634 // Value must be in [0, 2)
635 Metadata *LowAndHigh[] = {
636 ConstantAsMetadata::get(ConstantInt::get(Int8Ty, 0)),
637 ConstantAsMetadata::get(ConstantInt::get(Int8Ty, 2))};
638 auto NewMDNode = MDNode::get(Context, LowAndHigh);
639 const MachineMemOperand *OldMMO = *Load->memoperands_begin();
640 MachineMemOperand NewMMO(OldMMO->getPointerInfo(), OldMMO->getFlags(),
641 OldMMO->getSizeInBits(), OldMMO->getAlign(),
642 OldMMO->getAAInfo(), NewMDNode);
643 MachineIRBuilder MIB(*Load);
644 MIB.buildLoad(Load->getOperand(0), Load->getOperand(1), NewMMO);
645 Load->eraseFromParent();
646
647 GISelKnownBits Info(*MF);
648 KnownBits Res = Info.getKnownBits(And->getOperand(1).getReg());
649
650 // We don't know what the result of the load is, so we don't know any ones.
651 EXPECT_TRUE(Res.One.isNullValue());
652
653 // We know that the value is in [0, 2). So, we don't know if the first bit
654 // is 0 or not. However, we do know that every other bit must be 0.
655 APInt Mask(Res.getBitWidth(), 1);
656 Mask.flipAllBits();
657 EXPECT_EQ(Mask.getZExtValue(), Res.Zero.getZExtValue());
658}
