27void OptionalWorksInConstexpr() {
28 constexpr auto x1 = Optional<int>();
29 constexpr Optional<int> x2{};
30 static_assert(!x1.hasValue() && !x2.hasValue(),
31 "Default construction and hasValue() are contexpr");
32 constexpr auto y1 = Optional<int>(3);
33 constexpr Optional<int> y2{3};
34 static_assert(y1.getValue() == y2.getValue() && y1.getValue() == 3,
35 "Construction with value and getValue() are constexpr");
36 static_assert(Optional<int>{3} >= 2 && Optional<int>{1} < Optional<int>{2},
37 "Comparisons work in constexpr");
38}

394// Craft a class which is_trivially_copyable, but not
395// is_trivially_copy_constructible.
396struct NonTCopy {
397 NonTCopy() = default;
398
399 // Delete the volatile copy constructor to engage the "rule of 3" and delete
400 // any unspecified copy assignment or constructor.
401 NonTCopy(volatile NonTCopy const &) = delete;
402
403 // Leave the non-volatile default copy constructor unspecified (deleted by
404 // rule of 3)
405
406 // This template can serve as the copy constructor, but isn't chosen
407 // by =default in a class with a 'NonTCopy' member.
408 template <typename Self = NonTCopy>
409 NonTCopy(Self const &Other) : Val(Other.Val) {}
410
411 NonTCopy &operator=(NonTCopy const &) = default;
412
413 int Val{0};
414};
415
416#if defined(_MSC_VER) && _MSC_VER >= 1927 && !defined(__clang__)
417// Currently only true on recent MSVC releases.
418static_assert(std::is_trivially_copyable<NonTCopy>::value,
419 "Expect NonTCopy to be trivially copyable");
420
421static_assert(!std::is_trivially_copy_constructible<NonTCopy>::value,
422 "Expect NonTCopy not to be trivially copy constructible.");
423#endif // defined(_MSC_VER) && _MSC_VER >= 1927
424

435 // Check that the Optional can be copy assigned.

588template <typename OperatorT, typename T>
589void CheckRelation(const Optional<T> &Lhs, const Optional<T> &Rhs,
590 bool Expected) {
591 EXPECT_EQ(Expected, OperatorT::apply(Lhs, Rhs));
592
593 if (Lhs)
594 EXPECT_EQ(Expected, OperatorT::apply(*Lhs, Rhs));
595 else
596 EXPECT_EQ(Expected, OperatorT::apply(None, Rhs));
597
598 if (Rhs)
599 EXPECT_EQ(Expected, OperatorT::apply(Lhs, *Rhs));
600 else
601 EXPECT_EQ(Expected, OperatorT::apply(Lhs, None));
602}
