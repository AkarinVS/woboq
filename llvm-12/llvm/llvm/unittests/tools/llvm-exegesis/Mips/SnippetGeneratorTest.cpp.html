32class MipsSnippetGeneratorTest : public MipsTestBase {};

34template <typename SnippetGeneratorT>
35class SnippetGeneratorTest : public MipsSnippetGeneratorTest {
36protected:
37 SnippetGeneratorTest() : Generator(State, SnippetGenerator::Options()) {}
38
39 std::vector<CodeTemplate> checkAndGetCodeTemplates(unsigned Opcode) {
40 randomGenerator().seed(0); // Initialize seed.
41 const Instruction &Instr = State.getIC().getInstr(Opcode);
42 auto CodeTemplateOrError = Generator.generateCodeTemplates(
43 &Instr, State.getRATC().emptyRegisters());
44 EXPECT_FALSE(CodeTemplateOrError.takeError()); // Valid configuration.
45 return std::move(CodeTemplateOrError.get());
46 }
47
48 SnippetGeneratorT Generator;
49};

51using SerialSnippetGeneratorTest = SnippetGeneratorTest<SerialSnippetGenerator>;

53using ParallelSnippetGeneratorTest =
54 SnippetGeneratorTest<ParallelSnippetGenerator>;

56TEST_F(SerialSnippetGeneratorTest, ImplicitSelfDependencyThroughExplicitRegs) {
57 // - ADD
58 // - Op0 Explicit Def RegClass(GPR32)
59 // - Op1 Explicit Use RegClass(GPR32)
60 // - Op2 Explicit Use RegClass(GPR32)
61 // - Var0 [Op0]
62 // - Var1 [Op1]
63 // - Var2 [Op2]
64 // - hasAliasingRegisters
65 const unsigned Opcode = Mips::ADD;
66 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
67 ASSERT_THAT(CodeTemplates, SizeIs(1));
68 const auto &CT = CodeTemplates[0];
69 EXPECT_THAT(CT.Execution, ExecutionMode::SERIAL_VIA_EXPLICIT_REGS);
70 ASSERT_THAT(CT.Instructions, SizeIs(1));
71 const InstructionTemplate &IT = CT.Instructions[0];
72 EXPECT_THAT(IT.getOpcode(), Opcode);
73 ASSERT_THAT(IT.getVariableValues(), SizeIs(3));
74 EXPECT_THAT(IT.getVariableValues(),
75 AnyOf(ElementsAre(IsReg(), IsInvalid(), IsReg()),
76 ElementsAre(IsReg(), IsReg(), IsInvalid())))
77 << "Op0 is either set to Op1 or to Op2";
78}
79
80TEST_F(SerialSnippetGeneratorTest,
81 ImplicitSelfDependencyThroughExplicitRegsForbidAll) {
82 // - XOR
83 // - Op0 Explicit Def RegClass(GPR32)
84 // - Op1 Explicit Use RegClass(GPR32)
85 // - Op2 Explicit Use RegClass(GPR32)
86 // - Var0 [Op0]
87 // - Var1 [Op1]
88 // - Var2 [Op2]
89 // - hasAliasingRegisters
90 randomGenerator().seed(0); // Initialize seed.
91 const Instruction &Instr = State.getIC().getInstr(Mips::XOR);
92 auto AllRegisters = State.getRATC().emptyRegisters();
93 AllRegisters.flip();
94 auto Error =
95 Generator.generateCodeTemplates(&Instr, AllRegisters).takeError();
96 EXPECT_TRUE((bool)Error);
97 consumeError(std::move(Error));
98}

100TEST_F(ParallelSnippetGeneratorTest, MemoryUse) {
101 // LB reads from memory.
102 // - LB
103 // - Op0 Explicit Def RegClass(GPR32)
104 // - Op1 Explicit Use Memory RegClass(MSA128F16)
105 // - Op2 Explicit Use Memory
106 // - Var0 [Op0]
107 // - Var1 [Op1]
108 // - Var2 [Op2]
109 // - hasMemoryOperands
110 const unsigned Opcode = Mips::LB;
111 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
112 ASSERT_THAT(CodeTemplates, SizeIs(1));
113 const auto &CT = CodeTemplates[0];
114 EXPECT_THAT(CT.Info,
115 HasSubstr("instruction is parallel, repeating a random one."));
116 EXPECT_THAT(CT.Execution, ExecutionMode::UNKNOWN);
117 ASSERT_THAT(CT.Instructions,
118 SizeIs(ParallelSnippetGenerator::kMinNumDifferentAddresses));
119 const InstructionTemplate &IT = CT.Instructions[0];
120 EXPECT_THAT(IT.getOpcode(), Opcode);
121 ASSERT_THAT(IT.getVariableValues(), SizeIs(3));
122 EXPECT_EQ(IT.getVariableValues()[0].getReg(), 0u);
123 EXPECT_EQ(IT.getVariableValues()[2].getImm(), 0);
124}
