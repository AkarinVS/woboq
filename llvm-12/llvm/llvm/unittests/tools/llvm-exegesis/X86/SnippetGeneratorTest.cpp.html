50 std::vector<CodeTemplate> checkAndGetCodeTemplates(unsigned Opcode) {
51 randomGenerator().seed(0); // Initialize seed.
52 const Instruction &Instr = State.getIC().getInstr(Opcode);
53 auto CodeTemplateOrError = Generator.generateCodeTemplates(
54 &Instr, State.getRATC().emptyRegisters());
55 EXPECT_FALSE(CodeTemplateOrError.takeError()); // Valid configuration.
56 return std::move(CodeTemplateOrError.get());
57 }
58
59 SnippetGeneratorT Generator;
60};

67TEST_F(SerialSnippetGeneratorTest, ImplicitSelfDependencyThroughImplicitReg) {
68 // - ADC16i16
69 // - Op0 Explicit Use Immediate
70 // - Op1 Implicit Def Reg(AX)
71 // - Op2 Implicit Def Reg(EFLAGS)
72 // - Op3 Implicit Use Reg(AX)
73 // - Op4 Implicit Use Reg(EFLAGS)
74 // - Var0 [Op0]
75 // - hasAliasingImplicitRegisters (execution is always serial)
76 // - hasAliasingRegisters
77 const unsigned Opcode = X86::ADC16i16;
78 EXPECT_THAT(InstrInfo.get(Opcode).getImplicitDefs()[0], X86::AX);
79 EXPECT_THAT(InstrInfo.get(Opcode).getImplicitDefs()[1], X86::EFLAGS);
80 EXPECT_THAT(InstrInfo.get(Opcode).getImplicitUses()[0], X86::AX);
81 EXPECT_THAT(InstrInfo.get(Opcode).getImplicitUses()[1], X86::EFLAGS);
82 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
83 ASSERT_THAT(CodeTemplates, SizeIs(1));
84 const auto &CT = CodeTemplates[0];
85 EXPECT_THAT(CT.Execution, ExecutionMode::ALWAYS_SERIAL_IMPLICIT_REGS_ALIAS);
86 ASSERT_THAT(CT.Instructions, SizeIs(1));
87 const InstructionTemplate &IT = CT.Instructions[0];
88 EXPECT_THAT(IT.getOpcode(), Opcode);
89 ASSERT_THAT(IT.getVariableValues(), SizeIs(1)); // Imm.
90 EXPECT_THAT(IT.getVariableValues()[0], IsInvalid()) << "Immediate is not set";
91}

162TEST_F(SerialSnippetGeneratorTest, DependencyThroughOtherOpcode) {
163 // - CMP64rr
164 // - Op0 Explicit Use RegClass(GR64)
165 // - Op1 Explicit Use RegClass(GR64)
166 // - Op2 Implicit Def Reg(EFLAGS)
167 // - Var0 [Op0]
168 // - Var1 [Op1]
169 const unsigned Opcode = X86::CMP64rr;
170 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
171 ASSERT_THAT(CodeTemplates, SizeIs(Gt(1U))) << "Many templates are available";
172 for (const auto &CT : CodeTemplates) {
173 EXPECT_THAT(CT.Execution, ExecutionMode::SERIAL_VIA_NON_MEMORY_INSTR);
174 ASSERT_THAT(CT.Instructions, SizeIs(2));
175 const InstructionTemplate &IT = CT.Instructions[0];
176 EXPECT_THAT(IT.getOpcode(), Opcode);
177 ASSERT_THAT(IT.getVariableValues(), SizeIs(2));
178 EXPECT_THAT(IT.getVariableValues(),
179 AnyOf(ElementsAre(IsReg(), IsInvalid()),
180 ElementsAre(IsInvalid(), IsReg())));
181 EXPECT_THAT(CT.Instructions[1].getOpcode(), Not(Opcode));
182 // TODO: check that the two instructions alias each other.
183 }

260TEST_F(ParallelSnippetGeneratorTest, StaticRenaming) {
261 // CMOV32rr has tied variables, we enumerate the possible values to execute
262 // as many in parallel as possible.
263
264 // - CMOV32rr
265 // - Op0 Explicit Def RegClass(GR32)
266 // - Op1 Explicit Use RegClass(GR32) TiedToOp0
267 // - Op2 Explicit Use RegClass(GR32)
268 // - Op3 Explicit Use Immediate
269 // - Op3 Implicit Use Reg(EFLAGS)
270 // - Var0 [Op0,Op1]
271 // - Var1 [Op2]
272 // - hasTiedRegisters (execution is always serial)
273 // - hasAliasingRegisters
274 const unsigned Opcode = X86::CMOV32rr;
275 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
276 ASSERT_THAT(CodeTemplates, SizeIs(1));
277 const auto &CT = CodeTemplates[0];
278 EXPECT_THAT(CT.Info, HasSubstr("static renaming"));
279 EXPECT_THAT(CT.Execution, ExecutionMode::UNKNOWN);
280 constexpr const unsigned kInstructionCount = 15;
281 ASSERT_THAT(CT.Instructions, SizeIs(kInstructionCount));
282 std::unordered_set<unsigned> AllDefRegisters;
283 for (const auto &IT : CT.Instructions) {
284 ASSERT_THAT(IT.getVariableValues(), SizeIs(3));
285 AllDefRegisters.insert(IT.getVariableValues()[0].getReg());
286 }
287 EXPECT_THAT(AllDefRegisters, SizeIs(kInstructionCount))
288 << "Each instruction writes to a different register";
289}

291TEST_F(ParallelSnippetGeneratorTest, NoTiedVariables) {
292 // CMOV_GR32 has no tied variables, we make sure def and use are different
293 // from each other.
294
295 // - CMOV_GR32
296 // - Op0 Explicit Def RegClass(GR32)
297 // - Op1 Explicit Use RegClass(GR32)
298 // - Op2 Explicit Use RegClass(GR32)
299 // - Op3 Explicit Use Immediate
300 // - Op4 Implicit Use Reg(EFLAGS)
301 // - Var0 [Op0]
302 // - Var1 [Op1]
303 // - Var2 [Op2]
304 // - Var3 [Op3]
305 // - hasAliasingRegisters
306 const unsigned Opcode = X86::CMOV_GR32;
307 const auto CodeTemplates = checkAndGetCodeTemplates(Opcode);
308 ASSERT_THAT(CodeTemplates, SizeIs(1));
309 const auto &CT = CodeTemplates[0];
310 EXPECT_THAT(CT.Info, HasSubstr("no tied variables"));
311 EXPECT_THAT(CT.Execution, ExecutionMode::UNKNOWN);
312 ASSERT_THAT(CT.Instructions, SizeIs(1));
313 const InstructionTemplate &IT = CT.Instructions[0];
314 EXPECT_THAT(IT.getOpcode(), Opcode);
315 ASSERT_THAT(IT.getVariableValues(), SizeIs(4));
316 EXPECT_THAT(IT.getVariableValues()[0].getReg(),
317 Not(IT.getVariableValues()[1].getReg()))
318 << "Def is different from first Use";
319 EXPECT_THAT(IT.getVariableValues()[0].getReg(),
320 Not(IT.getVariableValues()[2].getReg()))
321 << "Def is different from second Use";
322 EXPECT_THAT(IT.getVariableValues()[3], IsInvalid());
323}

438TEST_F(FakeSnippetGeneratorTest, ComputeRegisterInitialValuesAdd64rr) {
439 // ADD64rr:
440 // mov64ri rax, 42
441 // add64rr rax, rax, rbx
442 // -> only rbx needs defining.
443 std::vector<InstructionTemplate> Snippet;
444 {
445 InstructionTemplate Mov = Generator.getInstructionTemplate(X86::MOV64ri);
446 Mov.getValueFor(Mov.getInstr().Variables[0]) =
447 MCOperand::createReg(X86::RAX);
448 Mov.getValueFor(Mov.getInstr().Variables[1]) = MCOperand::createImm(42);
449 Snippet.push_back(std::move(Mov));
450 }
451 {
452 InstructionTemplate Add = Generator.getInstructionTemplate(X86::ADD64rr);
453 Add.getValueFor(Add.getInstr().Variables[0]) =
454 MCOperand::createReg(X86::RAX);
455 Add.getValueFor(Add.getInstr().Variables[1]) =
456 MCOperand::createReg(X86::RBX);
457 Snippet.push_back(std::move(Add));
458 }
459
460 const auto RIV = Generator.computeRegisterInitialValues(Snippet);
461 EXPECT_THAT(RIV, ElementsAre(IsRegisterValue(X86::RBX, APInt())));
462}
