9#include "llvm/Support/FileSystem/UniqueID.h"
10#include "gtest/gtest.h"
11
12using namespace llvm;
13using namespace llvm::sys::fs;
14
15namespace {
16
17TEST(FSUniqueIDTest, construct) {
18 EXPECT_EQ(20u, UniqueID(20, 10).getDevice());
19 EXPECT_EQ(10u, UniqueID(20, 10).getFile());
20}
21
22TEST(FSUniqueIDTest, equals) {
23 EXPECT_TRUE(UniqueID(20, 10) == UniqueID(20, 10));
24 EXPECT_FALSE(UniqueID(20, 20) == UniqueID(20, 10));
25 EXPECT_FALSE(UniqueID(10, 10) == UniqueID(20, 10));
26}
27
28TEST(FSUniqueIDTest, less) {
29 EXPECT_FALSE(UniqueID(20, 2) < UniqueID(20, 2));
30 EXPECT_FALSE(UniqueID(20, 3) < UniqueID(20, 2));
31 EXPECT_FALSE(UniqueID(30, 2) < UniqueID(20, 2));
32 EXPECT_FALSE(UniqueID(30, 2) < UniqueID(20, 40));
33 EXPECT_TRUE(UniqueID(20, 2) < UniqueID(20, 3));
34 EXPECT_TRUE(UniqueID(20, 2) < UniqueID(30, 2));
35 EXPECT_TRUE(UniqueID(20, 40) < UniqueID(30, 2));
36}
37
38} // anonymous namespace
