9#include "llvm/Support/ELFAttributeParser.h"
10#include "llvm/Support/ELFAttributes.h"
11#include "gtest/gtest.h"
12#include <string>
13
14using namespace llvm;
15
16static const TagNameMap emptyTagNameMap;
17
18// This class is used to test the common part of the ELF attribute section.
19class AttributeHeaderParser : public ELFAttributeParser {
20 Error handler(uint64_t tag, bool &handled) override {
21 // Treat all attributes as handled.
22 handled = true;
23 return Error::success();
24 }
25
26public:
27 AttributeHeaderParser(ScopedPrinter *printer)
28 : ELFAttributeParser(printer, emptyTagNameMap, "test") {}
29 AttributeHeaderParser() : ELFAttributeParser(emptyTagNameMap, "test") {}
30};
31
32static void testParseError(ArrayRef<uint8_t> bytes, const char *msg) {
33 AttributeHeaderParser parser;
34 Error e = parser.parse(bytes, support::little);
35 EXPECT_STREQ(toString(std::move(e)).c_str(), msg);
36}
37
38TEST(AttributeHeaderParser, UnrecognizedFormatVersion) {
39 static const uint8_t bytes[] = {1};
40 testParseError(bytes, "unrecognized format-version: 0x1");
41}
42
43TEST(AttributeHeaderParser, InvalidSectionLength) {
44 static const uint8_t bytes[] = {'A', 3, 0, 0, 0};
45 testParseError(bytes, "invalid section length 3 at offset 0x1");
46}
47
48TEST(AttributeHeaderParser, UnrecognizedVendorName) {
49 static const uint8_t bytes[] = {'A', 7, 0, 0, 0, 'x', 'y', 0};
50 testParseError(bytes, "unrecognized vendor-name: xy");
51}
52
53TEST(AttributeHeaderParser, UnrecognizedTag) {
54 static const uint8_t bytes[] = {'A', 14, 0, 0, 0, 't', 'e', 's',
55 't', 0, 4, 5, 0, 0, 0};
56 testParseError(bytes, "unrecognized tag 0x4 at offset 0xa");
57}
58
59TEST(AttributeHeaderParser, InvalidAttributeSize) {
60 static const uint8_t bytes[] = {'A', 14, 0, 0, 0, 't', 'e', 's',
61 't', 0, 1, 4, 0, 0, 0};
62 testParseError(bytes, "invalid attribute size 4 at offset 0xa");
63}
