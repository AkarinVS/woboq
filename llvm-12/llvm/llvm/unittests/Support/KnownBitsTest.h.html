22template <typename FnTy> void ForeachKnownBits(unsigned Bits, FnTy Fn) {
23 unsigned Max = 1 << Bits;
24 KnownBits Known(Bits);
25 for (unsigned Zero = 0; Zero < Max; ++Zero) {
26 for (unsigned One = 0; One < Max; ++One) {
27 Known.Zero = Zero;
28 Known.One = One;
29 if (Known.hasConflict())
30 continue;
31
32 Fn(Known);
33 }
34 }
35}
36
37template <typename FnTy>
38void ForeachNumInKnownBits(const KnownBits &Known, FnTy Fn) {
39 unsigned Bits = Known.getBitWidth();
40 unsigned Max = 1 << Bits;
41 for (unsigned N = 0; N < Max; ++N) {
42 APInt Num(Bits, N);
43 if ((Num & Known.Zero) != 0 || (~Num & Known.One) != 0)
44 continue;
45
46 Fn(Num);
47 }
48}
