19template<typename T> std::string printToString(const T &Value) {
20 std::string res;
21 {
22 llvm::raw_string_ostream OS(res);
23 OS.SetBuffered();
24 OS << Value;
25 }
26 return res;
27}
28
29/// printToString - Print the given value to a stream which only has \arg
30/// BytesLeftInBuffer bytes left in the buffer. This is useful for testing edge
31/// cases in the buffer handling logic.
32template<typename T> std::string printToString(const T &Value,
33 unsigned BytesLeftInBuffer) {
34 // FIXME: This is relying on internal knowledge of how raw_ostream works to
35 // get the buffer position right.
36 SmallString<256> SVec;
37 assert(BytesLeftInBuffer < 256 && "Invalid buffer count!");
38 llvm::raw_svector_ostream OS(SVec);
39 unsigned StartIndex = 256 - BytesLeftInBuffer;
40 for (unsigned i = 0; i != StartIndex; ++i)
41 OS << '?';
42 OS << Value;
43 return std::string(OS.str().substr(StartIndex));
44}

46template<typename T> std::string printToStringUnbuffered(const T &Value) {
47 std::string res;
48 llvm::raw_string_ostream OS(res);
49 OS.SetUnbuffered();
50 OS << Value;
51 return res;
52}

58TEST(raw_ostreamTest, Types_Buffered) {
59 // Char
60 EXPECT_EQ("c", printToString('c'));
61
62 // String
63 EXPECT_EQ("hello", printToString("hello"));
64 EXPECT_EQ("hello", printToString(std::string("hello")));
65
66 // Int
67 EXPECT_EQ("0", printToString(0));
68 EXPECT_EQ("2425", printToString(2425));
69 EXPECT_EQ("-2425", printToString(-2425));
70
71 // Long long
72 EXPECT_EQ("0", printToString(0LL));
73 EXPECT_EQ("257257257235709", printToString(257257257235709LL));
74 EXPECT_EQ("-257257257235709", printToString(-257257257235709LL));
75
76 // Double
77 EXPECT_EQ("1.100000e+00", printToString(1.1));
78
79 // void*
80 EXPECT_EQ("0x0", printToString((void*) nullptr));
81 EXPECT_EQ("0xbeef", printToString((void*) 0xbeefLL));
82 EXPECT_EQ("0xdeadbeef", printToString((void*) 0xdeadbeefLL));
83
84 // Min and max.
85 EXPECT_EQ("18446744073709551615", printToString(UINT64_MAX));
86 EXPECT_EQ("-9223372036854775808", printToString(INT64_MIN));
87
88 // X, checking free operator<<().
89 EXPECT_EQ("X", printToString(X{}));
90}

203static std::string formatted_bytes_str(ArrayRef<uint8_t> Bytes,
204 llvm::Optional<uint64_t> Offset = None,
205 uint32_t NumPerLine = 16,
206 uint8_t ByteGroupSize = 4) {
207 std::string S;
208 raw_string_ostream Str(S);
209 Str << format_bytes(Bytes, Offset, NumPerLine, ByteGroupSize);
210 Str.flush();
211 return S;
212}

214static std::string format_bytes_with_ascii_str(ArrayRef<uint8_t> Bytes,
215 Optional<uint64_t> Offset = None,
216 uint32_t NumPerLine = 16,
217 uint8_t ByteGroupSize = 4) {
218 std::string S;
219 raw_string_ostream Str(S);
220 Str << format_bytes_with_ascii(Bytes, Offset, NumPerLine, ByteGroupSize);
221 Str.flush();
222 return S;
223}

225TEST(raw_ostreamTest, FormattedHexBytes) {
226 std::vector<uint8_t> Buf = {'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i',
227 'j', 'k', 'l', 'm', 'n', 'o', 'p', 'q', 'r',
228 's', 't', 'u', 'v', 'w', 'x', 'y', 'z', '0',
229 '1', '2', '3', '4', '5', '6', '7', '8', '9'};
230 ArrayRef<uint8_t> B(Buf);
231
232 // Test invalid input.
233 EXPECT_EQ("", formatted_bytes_str(ArrayRef<uint8_t>()));
234 EXPECT_EQ("", format_bytes_with_ascii_str(ArrayRef<uint8_t>()));
235 //----------------------------------------------------------------------
236 // Test hex byte output with the default 4 byte groups
237 //----------------------------------------------------------------------
238 EXPECT_EQ("61", formatted_bytes_str(B.take_front()));
239 EXPECT_EQ("61626364 65", formatted_bytes_str(B.take_front(5)));
240 // Test that 16 bytes get written to a line correctly.
241 EXPECT_EQ("61626364 65666768 696a6b6c 6d6e6f70",
242 formatted_bytes_str(B.take_front(16)));
243 // Test raw bytes with default 16 bytes per line wrapping.
244 EXPECT_EQ("61626364 65666768 696a6b6c 6d6e6f70\n71",
245 formatted_bytes_str(B.take_front(17)));
246 // Test raw bytes with 1 bytes per line wrapping.
247 EXPECT_EQ("61\n62\n63\n64\n65\n66",
248 formatted_bytes_str(B.take_front(6), None, 1));
249 // Test raw bytes with 7 bytes per line wrapping.
250 EXPECT_EQ("61626364 656667\n68696a6b 6c6d6e\n6f7071",
251 formatted_bytes_str(B.take_front(17), None, 7));
252 // Test raw bytes with 8 bytes per line wrapping.
253 EXPECT_EQ("61626364 65666768\n696a6b6c 6d6e6f70\n71",
254 formatted_bytes_str(B.take_front(17), None, 8));
255 //----------------------------------------------------------------------
256 // Test hex byte output with the 1 byte groups
257 //----------------------------------------------------------------------
258 EXPECT_EQ("61 62 63 64 65",
259 formatted_bytes_str(B.take_front(5), None, 16, 1));
260 // Test that 16 bytes get written to a line correctly.
261 EXPECT_EQ("61 62 63 64 65 66 67 68 69 6a 6b 6c 6d 6e 6f 70",
262 formatted_bytes_str(B.take_front(16), None, 16, 1));
263 // Test raw bytes with default 16 bytes per line wrapping.
264 EXPECT_EQ("61 62 63 64 65 66 67 68 69 6a 6b 6c 6d 6e 6f 70\n71",
265 formatted_bytes_str(B.take_front(17), None, 16, 1));
266 // Test raw bytes with 7 bytes per line wrapping.
267 EXPECT_EQ("61 62 63 64 65 66 67\n68 69 6a 6b 6c 6d 6e\n6f 70 71",
268 formatted_bytes_str(B.take_front(17), None, 7, 1));
269 // Test raw bytes with 8 bytes per line wrapping.
270 EXPECT_EQ("61 62 63 64 65 66 67 68\n69 6a 6b 6c 6d 6e 6f 70\n71",
271 formatted_bytes_str(B.take_front(17), None, 8, 1));
272
273 //----------------------------------------------------------------------
274 // Test hex byte output with the 2 byte groups
275 //----------------------------------------------------------------------
276 EXPECT_EQ("6162 6364 65", formatted_bytes_str(B.take_front(5), None, 16, 2));
277 // Test that 16 bytes get written to a line correctly.
278 EXPECT_EQ("6162 6364 6566 6768 696a 6b6c 6d6e 6f70",
279 formatted_bytes_str(B.take_front(16), None, 16, 2));
280 // Test raw bytes with default 16 bytes per line wrapping.
281 EXPECT_EQ("6162 6364 6566 6768 696a 6b6c 6d6e 6f70\n71",
282 formatted_bytes_str(B.take_front(17), None, 16, 2));
283 // Test raw bytes with 7 bytes per line wrapping.
284 EXPECT_EQ("6162 6364 6566 67\n6869 6a6b 6c6d 6e\n6f70 71",
285 formatted_bytes_str(B.take_front(17), None, 7, 2));
286 // Test raw bytes with 8 bytes per line wrapping.
287 EXPECT_EQ("6162 6364 6566 6768\n696a 6b6c 6d6e 6f70\n71",
288 formatted_bytes_str(B.take_front(17), None, 8, 2));
289
290 //----------------------------------------------------------------------
291 // Test hex bytes with offset with the default 4 byte groups.
292 //----------------------------------------------------------------------
293 EXPECT_EQ("0000: 61", formatted_bytes_str(B.take_front(), 0x0));
294 EXPECT_EQ("1000: 61", formatted_bytes_str(B.take_front(), 0x1000));
295 EXPECT_EQ("1000: 61\n1001: 62",
296 formatted_bytes_str(B.take_front(2), 0x1000, 1));
297 //----------------------------------------------------------------------
298 // Test hex bytes with ASCII with the default 4 byte groups.
299 //----------------------------------------------------------------------
300 EXPECT_EQ("61626364 65666768 696a6b6c 6d6e6f70 |abcdefghijklmnop|",
301 format_bytes_with_ascii_str(B.take_front(16)));
302 EXPECT_EQ("61626364 65666768 |abcdefgh|\n"
303 "696a6b6c 6d6e6f70 |ijklmnop|",
304 format_bytes_with_ascii_str(B.take_front(16), None, 8));
305 EXPECT_EQ("61626364 65666768 |abcdefgh|\n696a6b6c |ijkl|",
306 format_bytes_with_ascii_str(B.take_front(12), None, 8));
307 std::vector<uint8_t> Unprintable = {'a', '\x1e', 'b', '\x1f'};
308 // Make sure the ASCII is still lined up correctly when fewer bytes than 16
309 // bytes per line are available. The ASCII should still be aligned as if 16
310 // bytes of hex might be displayed.
311 EXPECT_EQ("611e621f |a.b.|",
312 format_bytes_with_ascii_str(Unprintable));
313 //----------------------------------------------------------------------
314 // Test hex bytes with ASCII with offsets with the default 4 byte groups.
315 //----------------------------------------------------------------------
316 EXPECT_EQ("0000: 61626364 65666768 "
317 "696a6b6c 6d6e6f70 |abcdefghijklmnop|",
318 format_bytes_with_ascii_str(B.take_front(16), 0));
319 EXPECT_EQ("0000: 61626364 65666768 |abcdefgh|\n"
320 "0008: 696a6b6c 6d6e6f70 |ijklmnop|",
321 format_bytes_with_ascii_str(B.take_front(16), 0, 8));
322 EXPECT_EQ("0000: 61626364 656667 |abcdefg|\n"
323 "0007: 68696a6b 6c |hijkl|",
324 format_bytes_with_ascii_str(B.take_front(12), 0, 7));
325
326 //----------------------------------------------------------------------
327 // Test hex bytes with ASCII with offsets with the default 2 byte groups.
328 //----------------------------------------------------------------------
329 EXPECT_EQ("0000: 6162 6364 6566 6768 "
330 "696a 6b6c 6d6e 6f70 |abcdefghijklmnop|",
331 format_bytes_with_ascii_str(B.take_front(16), 0, 16, 2));
332 EXPECT_EQ("0000: 6162 6364 6566 6768 |abcdefgh|\n"
333 "0008: 696a 6b6c 6d6e 6f70 |ijklmnop|",
334 format_bytes_with_ascii_str(B.take_front(16), 0, 8, 2));
335 EXPECT_EQ("0000: 6162 6364 6566 67 |abcdefg|\n"
336 "0007: 6869 6a6b 6c |hijkl|",
337 format_bytes_with_ascii_str(B.take_front(12), 0, 7, 2));
338
339 //----------------------------------------------------------------------
340 // Test hex bytes with ASCII with offsets with the default 1 byte groups.
341 //----------------------------------------------------------------------
342 EXPECT_EQ("0000: 61 62 63 64 65 66 67 68 "
343 "69 6a 6b 6c 6d 6e 6f 70 |abcdefghijklmnop|",
344 format_bytes_with_ascii_str(B.take_front(16), 0, 16, 1));
345 EXPECT_EQ("0000: 61 62 63 64 65 66 67 68 |abcdefgh|\n"
346 "0008: 69 6a 6b 6c 6d 6e 6f 70 |ijklmnop|",
347 format_bytes_with_ascii_str(B.take_front(16), 0, 8, 1));
348 EXPECT_EQ("0000: 61 62 63 64 65 66 67 |abcdefg|\n"
349 "0007: 68 69 6a 6b 6c |hijkl|",
350 format_bytes_with_ascii_str(B.take_front(12), 0, 7, 1));
351}

380TEST(raw_ostreamTest, flush_tied_to_stream_on_write) {
381 std::string TiedToBuffer;
382 raw_string_ostream TiedTo(TiedToBuffer);
383 TiedTo.SetBuffered();
384 TiedTo << "a";
385
386 std::string Buffer;
387 raw_string_ostream TiedStream(Buffer);
388 TiedStream.tie(&TiedTo);
389 // Sanity check that the stream hasn't already been flushed.
390 EXPECT_EQ("", TiedToBuffer);
391
392 // Empty string doesn't cause a flush of TiedTo.
393 TiedStream << "";
394 EXPECT_EQ("", TiedToBuffer);
395
396 // Non-empty strings trigger flush of TiedTo.
397 TiedStream << "abc";
398 EXPECT_EQ("a", TiedToBuffer);
399
400 // Single char write flushes TiedTo.
401 TiedTo << "c";
402 TiedStream << 'd';
403 EXPECT_EQ("ac", TiedToBuffer);
404
405 // Write to buffered stream without flush does not flush TiedTo.
406 TiedStream.SetBuffered();
407 TiedStream.SetBufferSize(2);
408 TiedTo << "e";
409 TiedStream << "f";
410 EXPECT_EQ("ac", TiedToBuffer);
411
412 // Explicit flush of buffered stream flushes TiedTo.
413 TiedStream.flush();
414 EXPECT_EQ("ace", TiedToBuffer);
415
416 // Explicit flush of buffered stream with empty buffer does not flush TiedTo.
417 TiedTo << "g";
418 TiedStream.flush();
419 EXPECT_EQ("ace", TiedToBuffer);
420
421 // Write of data to empty buffer that is greater than buffer size flushes
422 // TiedTo.
423 TiedStream << "hijklm";
424 EXPECT_EQ("aceg", TiedToBuffer);
425
426 // Write of data that overflows buffer size also flushes TiedTo.
427 TiedStream.flush();
428 TiedStream << "n";
429 TiedTo << "o";
430 TiedStream << "pq";
431 EXPECT_EQ("acego", TiedToBuffer);
432
433 // Streams can be tied to each other safely.
434 TiedStream.flush();
435 Buffer = "";
436 TiedTo.tie(&TiedStream);
437 TiedTo.SetBufferSize(2);
438 TiedStream << "r";
439 TiedTo << "s";
440 EXPECT_EQ("", Buffer);
441 EXPECT_EQ("acego", TiedToBuffer);
442 TiedTo << "tuv";
443 EXPECT_EQ("r", Buffer);
444 TiedStream << "wxy";
445 EXPECT_EQ("acegostuv", TiedToBuffer);
446 // The x remains in the buffer, since it was written after the flush of
447 // TiedTo.
448 EXPECT_EQ("rwx", Buffer);
449 TiedTo.tie(nullptr);
450
451 // Calling tie with nullptr unties stream.
452 TiedStream.SetUnbuffered();
453 TiedStream.tie(nullptr);
454 TiedTo << "y";
455 TiedStream << "0";
456 EXPECT_EQ("acegostuv", TiedToBuffer);
457}
