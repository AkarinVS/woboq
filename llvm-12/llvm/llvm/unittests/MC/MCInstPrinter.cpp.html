9#include "llvm/MC/MCInstPrinter.h"
10#include "llvm/MC/MCAsmInfo.h"
11#include "llvm/MC/MCInstrInfo.h"
12#include "llvm/MC/MCTargetOptions.h"
13#include "llvm/Support/TargetRegistry.h"
14#include "llvm/Support/TargetSelect.h"
15#include "llvm/Target/TargetMachine.h"
16#include "llvm/Target/TargetOptions.h"
17#include "gtest/gtest.h"
18
19using namespace llvm;
20
21namespace {
22class MCInstPrinterTest : public ::testing::Test {
23public:
24 std::unique_ptr<MCRegisterInfo> MRI;
25 std::unique_ptr<MCAsmInfo> MAI;
26 std::unique_ptr<const MCInstrInfo> MII;
27 std::unique_ptr<MCInstPrinter> Printer;
28
29 MCInstPrinterTest() {
30 llvm::InitializeAllTargetInfos();
31 llvm::InitializeAllTargetMCs();
32
33 std::string TripleName = "x86_64-pc-linux";
34 std::string ErrorStr;
35
36 const Target *TheTarget =
37 TargetRegistry::lookupTarget(TripleName, ErrorStr);
38
39 // If we didn't build x86, do not run the test.
40 if (!TheTarget)
41 return;
42
43 MRI.reset(TheTarget->createMCRegInfo(TripleName));
44 MCTargetOptions MCOptions;
45 MAI.reset(TheTarget->createMCAsmInfo(*MRI, TripleName, MCOptions));
46 MII.reset(TheTarget->createMCInstrInfo());
47 Printer.reset(TheTarget->createMCInstPrinter(
48 Triple(TripleName), MAI->getAssemblerDialect(), *MAI, *MII, *MRI));
49 }
50
51 template <typename T> std::string formatHex(T i) {
52 std::string Buffer;
53 raw_string_ostream OS(Buffer);
54 OS << Printer->formatHex(i);
55 OS.flush();
56 return Buffer;
57 }
58};
59} // namespace
60
61TEST_F(MCInstPrinterTest, formatHex) {
62 if (!Printer)
63 return;
64
65 EXPECT_EQ("0x1", formatHex<int64_t>(1));
66 EXPECT_EQ("0x7fffffffffffffff",
67 formatHex(std::numeric_limits<int64_t>::max()));
68 EXPECT_EQ("-0x8000000000000000",
69 formatHex(std::numeric_limits<int64_t>::min()));
70}
