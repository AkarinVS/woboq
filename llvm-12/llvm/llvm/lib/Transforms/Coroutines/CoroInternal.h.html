29// CoroEarly pass marks every function that has coro.begin with a string
30// attribute "coroutine.presplit"="0". CoroSplit pass processes the coroutine
31// twice. First, it lets it go through complete IPO optimization pipeline as a
32// single function. It forces restart of the pipeline by inserting an indirect
33// call to an empty function "coro.devirt.trigger" which is devirtualized by
34// CoroElide pass that triggers a restart of the pipeline by CGPassManager.
35// When CoroSplit pass sees the same coroutine the second time, it splits it up,
36// adds coroutine subfunctions to the SCC to be processed by IPO pipeline.
37// Async lowering similarily triggers a restart of the pipeline after it has
38// split the coroutine.
39#define CORO_PRESPLIT_ATTR "coroutine.presplit"
40#define UNPREPARED_FOR_SPLIT "0"
41#define PREPARED_FOR_SPLIT "1"
42#define ASYNC_RESTART_AFTER_SPLIT "2"
43
44#define CORO_DEVIRT_TRIGGER_FN "coro.devirt.trigger"

71enum class ABI {
72 /// The "resume-switch" lowering, where there are separate resume and
73 /// destroy functions that are shared between all suspend points. The
74 /// coroutine frame implicitly stores the resume and destroy functions,
75 /// the current index, and any promise value.
76 Switch,
77
78 /// The "returned-continuation" lowering, where each suspend point creates a
79 /// single continuation function that is used for both resuming and
80 /// destroying. Does not support promises.
81 Retcon,
82
83 /// The "unique returned-continuation" lowering, where each suspend point
84 /// creates a single continuation function that is used for both resuming
85 /// and destroying. Does not support promises. The function is known to
86 /// suspend at most once during its execution, and the return value of
87 /// the continuation is void.
88 RetconOnce,
89
90 /// The "async continuation" lowering, where each suspend point creates a
91 /// single continuation function. The continuation function is available as an
92 /// intrinsic.
93 Async,
94};

106 struct SwitchFieldIndex {
107 enum {
108 Resume,
109 Destroy
110
111 // The promise field is always at a fixed offset from the start of
112 // frame given its type, but the index isn't a constant for all
113 // possible frames.
114
115 // The switch-index field isn't at a fixed offset or index, either;
116 // we just work it in where it fits best.
117 };

165 CoroIdInst *getSwitchCoroId() const {

181 unsigned getSwitchIndexField() const {
182 assert(ABI == coro::ABI::Switch);
183 assert(FrameTy && "frame type not assigned");
184 return SwitchLowering.IndexField;
185 }
186 IntegerType *getIndexType() const {
187 assert(ABI == coro::ABI::Switch);
188 assert(FrameTy && "frame type not assigned");
189 return cast<IntegerType>(FrameTy->getElementType(getSwitchIndexField()));
190 }
191 ConstantInt *getIndex(uint64_t Value) const {
192 return ConstantInt::get(getIndexType(), Value);
193 }
194
195 PointerType *getSwitchResumePointerType() const {
196 assert(ABI == coro::ABI::Switch);
197 assert(FrameTy && "frame type not assigned");
198 return cast<PointerType>(FrameTy->getElementType(SwitchFieldIndex::Resume));
199 }
200
201 FunctionType *getResumeFunctionType() const {
202 switch (ABI) {
203 case coro::ABI::Switch: {
204 auto *FnPtrTy = getSwitchResumePointerType();
205 return cast<FunctionType>(FnPtrTy->getPointerElementType());
206 }
207 case coro::ABI::Retcon:
208 case coro::ABI::RetconOnce:
209 return RetconLowering.ResumePrototype->getFunctionType();
210 case coro::ABI::Async:
211 return AsyncLowering.AsyncFuncTy;
212 }
213
214 llvm_unreachable("Unknown coro::ABI enum");
215 }
216
217 ArrayRef<Type*> getRetconResultTypes() const {
218 assert(ABI == coro::ABI::Retcon ||
219 ABI == coro::ABI::RetconOnce);
220 auto FTy = CoroBegin->getFunction()->getFunctionType();
221
222 // The safety of all this is checked by checkWFRetconPrototype.
223 if (auto STy = dyn_cast<StructType>(FTy->getReturnType())) {
224 return STy->elements().slice(1);
225 } else {
226 return ArrayRef<Type*>();
227 }
228 }
229
230 ArrayRef<Type*> getRetconResumeTypes() const {
231 assert(ABI == coro::ABI::Retcon ||
232 ABI == coro::ABI::RetconOnce);
233
234 // The safety of all this is checked by checkWFRetconPrototype.
235 auto FTy = RetconLowering.ResumePrototype->getFunctionType();
236 return FTy->params().slice(1);
237 }
238
239 CallingConv::ID getResumeFunctionCC() const {
240 switch (ABI) {
241 case coro::ABI::Switch:
242 return CallingConv::Fast;
243
244 case coro::ABI::Retcon:
245 case coro::ABI::RetconOnce:
246 return RetconLowering.ResumePrototype->getCallingConv();
247 case coro::ABI::Async:
248 return CallingConv::Swift;
249 }
250 llvm_unreachable("Unknown coro::ABI enum");
251 }
