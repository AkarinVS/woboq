26static bool uniqueifyInternalLinkageNames(Module &M) {
27 llvm::MD5 Md5;
28 Md5.update(M.getSourceFileName());
29 llvm::MD5::MD5Result R;
30 Md5.final(R);
31 SmallString<32> Str;
32 llvm::MD5::stringifyResult(R, Str);
33 // Convert MD5hash to Decimal. Demangler suffixes can either contain numbers
34 // or characters but not both.
35 APInt IntHash = APInt(128, Str.str(), 16);
36 // Prepend "__uniq" before the hash for tools like profilers to understand that
37 // this symbol is of internal linkage type.
38 std::string ModuleNameHash = (Twine(".__uniq.") + Twine(IntHash.toString(10, false))).str();
39 bool Changed = false;
40 MDBuilder MDB(M.getContext());
41
42 // Append the module hash to all internal linkage functions.
43 for (auto &F : M) {
44 if (F.hasInternalLinkage()) {
45 F.setName(F.getName() + ModuleNameHash);
46 F.addFnAttr("sample-profile-suffix-elision-policy", "selected");
47 // Replace linkage names in the debug metadata.
48 if (DISubprogram *SP = F.getSubprogram()) {
49 if (SP->getRawLinkageName()) {
50 auto *Name = MDB.createString(F.getName());
51 SP->replaceRawLinkageName(Name);
52 if (DISubprogram *SPDecl = SP->getDeclaration()) {
53 if (SPDecl->getRawLinkageName())
54 SPDecl->replaceRawLinkageName(Name);
55 }
56 }
57 }
58 Changed = true;
59 }
60 }
61
62 // Append the module hash to all internal linkage globals.
63 for (auto &GV : M.globals()) {
64 if (GV.hasInternalLinkage()) {
65 GV.setName(GV.getName() + ModuleNameHash);
66 Changed = true;
67 }
68 }
69 return Changed;
70}

72namespace {

74// Legacy pass that provides a name to every anon globals.
75class UniqueInternalLinkageNamesLegacyPass : public ModulePass {
76
77public:
78 /// Pass identification, replacement for typeid
79 static char ID;
80
81 /// Specify pass name for debug output
82 StringRef getPassName() const override {
83 return "Unique Internal Linkage Names";
84 }
85
86 explicit UniqueInternalLinkageNamesLegacyPass() : ModulePass(ID) {
87 initializeUniqueInternalLinkageNamesLegacyPassPass(
88 *PassRegistry::getPassRegistry());
89 }
90
91 bool runOnModule(Module &M) override {
92 return uniqueifyInternalLinkageNames(M);
93 }
94};
95
96char UniqueInternalLinkageNamesLegacyPass::ID = 0;
97} // anonymous namespace
98
99PreservedAnalyses
100UniqueInternalLinkageNamesPass::run(Module &M, ModuleAnalysisManager &AM) {
101 if (!uniqueifyInternalLinkageNames(M))
102 return PreservedAnalyses::all();
103
104 return PreservedAnalyses::none();
105}

107INITIALIZE_PASS_BEGIN(UniqueInternalLinkageNamesLegacyPass,
108 "unique-internal-linkage-names",
109 "Uniqueify internal linkage names", false, false)
110INITIALIZE_PASS_END(UniqueInternalLinkageNamesLegacyPass,
111 "unique-internal-linkage-names",
112 "Uniqueify Internal linkage names", false, false)

114namespace llvm {
115ModulePass *createUniqueInternalLinkageNamesPass() {
116 return new UniqueInternalLinkageNamesLegacyPass();
117}
