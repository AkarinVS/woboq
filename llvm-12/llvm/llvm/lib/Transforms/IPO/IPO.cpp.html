120void LLVMAddInternalizePass(LLVMPassManagerRef PM, unsigned AllButMain) {
121 auto PreserveMain = [=](const GlobalValue &GV) {
122 return AllButMain && GV.getName() == "main";
123 };
124 unwrap(PM)->add(createInternalizePass(PreserveMain));
125}
126
127void LLVMAddInternalizePassWithMustPreservePredicate(
128 LLVMPassManagerRef PM,
129 void *Context,
130 LLVMBool (*Pred)(LLVMValueRef, void *)) {
131 unwrap(PM)->add(createInternalizePass([=](const GlobalValue &GV) {
132 return Pred(wrap(&GV), Context) == 0 ? false : true;
133 }));
134}
