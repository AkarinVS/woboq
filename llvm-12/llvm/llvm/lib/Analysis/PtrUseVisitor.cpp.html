21void detail::PtrUseVisitorBase::enqueueUsers(Instruction &I) {
22 for (Use &U : I.uses()) {
23 if (VisitedUses.insert(&U).second) {
24 UseToVisit NewU = {
25 UseToVisit::UseAndIsOffsetKnownPair(&U, IsOffsetKnown),
26 Offset
27 };
28 Worklist.push_back(std::move(NewU));
29 }
30 }
31}
32
33bool detail::PtrUseVisitorBase::adjustOffsetForGEP(GetElementPtrInst &GEPI) {
34 if (!IsOffsetKnown)
35 return false;
36
37 APInt TmpOffset(DL.getIndexTypeSizeInBits(GEPI.getType()), 0);
38 if (GEPI.accumulateConstantOffset(DL, TmpOffset)) {
39 Offset += TmpOffset.sextOrTrunc(Offset.getBitWidth());
40 return true;
41 }
42
43 return false;
44}
