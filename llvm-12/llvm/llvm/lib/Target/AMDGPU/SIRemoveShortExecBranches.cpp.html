9/// \file
10/// This pass optmizes the s_cbranch_execz instructions.
11/// The pass removes this skip instruction for short branches,
12/// if there is no unwanted sideeffect in the fallthrough code sequence.
13///

66bool SIRemoveShortExecBranches::getBlockDestinations(
67 MachineBasicBlock &SrcMBB, MachineBasicBlock *&TrueMBB,
68 MachineBasicBlock *&FalseMBB, SmallVectorImpl<MachineOperand> &Cond) {
69 if (TII->analyzeBranch(SrcMBB, TrueMBB, FalseMBB, Cond))
70 return false;
71
72 if (!FalseMBB)
73 FalseMBB = SrcMBB.getNextNode();
74
75 return true;
76}
77
78bool SIRemoveShortExecBranches::mustRetainExeczBranch(
79 const MachineBasicBlock &From, const MachineBasicBlock &To) const {
80 unsigned NumInstr = 0;
81 const MachineFunction *MF = From.getParent();
82
83 for (MachineFunction::const_iterator MBBI(&From), ToI(&To), End = MF->end();
84 MBBI != End && MBBI != ToI; ++MBBI) {
85 const MachineBasicBlock &MBB = *MBBI;
86
87 for (MachineBasicBlock::const_iterator I = MBB.begin(), E = MBB.end();
88 I != E; ++I) {
89 // When a uniform loop is inside non-uniform control flow, the branch
90 // leaving the loop might never be taken when EXEC = 0.
91 // Hence we should retain cbranch out of the loop lest it become infinite.
92 if (I->isConditionalBranch())
93 return true;
94
95 if (TII->hasUnwantedEffectsWhenEXECEmpty(*I))
96 return true;
97
98 if (TII->isKillTerminator(I->getOpcode()))
99 return true;
100
101 // These instructions are potentially expensive even if EXEC = 0.
102 if (TII->isSMRD(*I) || TII->isVMEM(*I) || TII->isFLAT(*I) ||
103 I->getOpcode() == AMDGPU::S_WAITCNT)
104 return true;
105
106 ++NumInstr;
107 if (NumInstr >= SkipThreshold)
108 return true;
109 }
110 }
111
112 return false;
113}
114
115// Returns true if the skip branch instruction is removed.
116bool SIRemoveShortExecBranches::removeExeczBranch(MachineInstr &MI,
117 MachineBasicBlock &SrcMBB) {
118 MachineBasicBlock *TrueMBB = nullptr;
119 MachineBasicBlock *FalseMBB = nullptr;
120 SmallVector<MachineOperand, 1> Cond;
121
122 if (!getBlockDestinations(SrcMBB, TrueMBB, FalseMBB, Cond))
123 return false;
124
125 // Consider only the forward branches.
126 if ((SrcMBB.getNumber() >= TrueMBB->getNumber()) ||
127 mustRetainExeczBranch(*FalseMBB, *TrueMBB))
128 return false;
129
130 LLVM_DEBUG(dbgs() << "Removing the execz branch: " << MI);
131 MI.eraseFromParent();
132 SrcMBB.removeSuccessor(TrueMBB);
133
134 return true;
135}
136
137bool SIRemoveShortExecBranches::runOnMachineFunction(MachineFunction &MF) {
138 const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
139 TII = ST.getInstrInfo();
140 MF.RenumberBlocks();
141 bool Changed = false;
142
143 for (MachineBasicBlock &MBB : MF) {
144 MachineBasicBlock::iterator MBBI = MBB.getFirstTerminator();
145 if (MBBI == MBB.end())
146 continue;
147
148 MachineInstr &MI = *MBBI;
149 switch (MI.getOpcode()) {
150 case AMDGPU::S_CBRANCH_EXECZ:
151 Changed = removeExeczBranch(MI, MBB);
152 break;
153 default:
154 break;
155 }
156 }
157
158 return Changed;
159}
