92void R600MCCodeEmitter::encodeInstruction(const MCInst &MI, raw_ostream &OS,
93 SmallVectorImpl<MCFixup> &Fixups,
94 const MCSubtargetInfo &STI) const {
95 verifyInstructionPredicates(MI,
96 computeAvailableFeatures(STI.getFeatureBits()));
97
98 const MCInstrDesc &Desc = MCII.get(MI.getOpcode());
99 if (MI.getOpcode() == R600::RETURN ||
100 MI.getOpcode() == R600::FETCH_CLAUSE ||
101 MI.getOpcode() == R600::ALU_CLAUSE ||
102 MI.getOpcode() == R600::BUNDLE ||
103 MI.getOpcode() == R600::KILL) {
104 return;
105 } else if (IS_VTX(Desc)) {
106 uint64_t InstWord01 = getBinaryCodeForInstr(MI, Fixups, STI);
107 uint32_t InstWord2 = MI.getOperand(2).getImm(); // Offset
108 if (!(STI.getFeatureBits()[R600::FeatureCaymanISA])) {
109 InstWord2 |= 1 << 19; // Mega-Fetch bit
110 }
111
112 Emit(InstWord01, OS);
113 Emit(InstWord2, OS);
114 Emit((uint32_t) 0, OS);
115 } else if (IS_TEX(Desc)) {
116 int64_t Sampler = MI.getOperand(14).getImm();
117
118 int64_t SrcSelect[4] = {
119 MI.getOperand(2).getImm(),
120 MI.getOperand(3).getImm(),
121 MI.getOperand(4).getImm(),
122 MI.getOperand(5).getImm()
123 };
124 int64_t Offsets[3] = {
125 MI.getOperand(6).getImm() & 0x1F,
126 MI.getOperand(7).getImm() & 0x1F,
127 MI.getOperand(8).getImm() & 0x1F
128 };
129
130 uint64_t Word01 = getBinaryCodeForInstr(MI, Fixups, STI);
131 uint32_t Word2 = Sampler << 15 | SrcSelect[ELEMENT_X] << 20 |
132 SrcSelect[ELEMENT_Y] << 23 | SrcSelect[ELEMENT_Z] << 26 |
133 SrcSelect[ELEMENT_W] << 29 | Offsets[0] << 0 | Offsets[1] << 5 |
134 Offsets[2] << 10;
135
136 Emit(Word01, OS);
137 Emit(Word2, OS);
138 Emit((uint32_t) 0, OS);
139 } else {
140 uint64_t Inst = getBinaryCodeForInstr(MI, Fixups, STI);
141 if ((STI.getFeatureBits()[R600::FeatureR600ALUInst]) &&
142 ((Desc.TSFlags & R600_InstFlag::OP1) ||
143 Desc.TSFlags & R600_InstFlag::OP2)) {
144 uint64_t ISAOpCode = Inst & (0x3FFULL << 39);
145 Inst &= ~(0x3FFULL << 39);
146 Inst |= ISAOpCode << 1;
147 }
148 Emit(Inst, OS);
149 }
150}

164uint64_t R600MCCodeEmitter::getMachineOpValue(const MCInst &MI,
165 const MCOperand &MO,
166 SmallVectorImpl<MCFixup> &Fixups,
167 const MCSubtargetInfo &STI) const {
168 if (MO.isReg()) {
169 if (HAS_NATIVE_OPERANDS(MCII.get(MI.getOpcode()).TSFlags))
170 return MRI.getEncodingValue(MO.getReg());
171 return getHWReg(MO.getReg());
172 }
173
174 if (MO.isExpr()) {
175 // We put rodata at the end of code section, then map the entire
176 // code secetion as vtx buf. Thus the section relative address is the
177 // correct one.
178 // Each R600 literal instruction has two operands
179 // We can't easily get the order of the current one, so compare against
180 // the first one and adjust offset.
181 const unsigned offset = (&MO == &MI.getOperand(0)) ? 0 : 4;
182 Fixups.push_back(MCFixup::create(offset, MO.getExpr(), FK_SecRel_4, MI.getLoc()));
183 return 0;
184 }
185
186 assert(MO.isImm());
187 return MO.getImm();
188}
