9/// \file
10/// \brief Try to reassign registers on GFX10+ to reduce register bank
11/// conflicts.
12///
13/// On GFX10 registers are organized in banks. VGPRs have 4 banks assigned in
14/// a round-robin fashion: v0, v4, v8... belong to bank 0. v1, v5, v9... to
15/// bank 1, etc. SGPRs have 8 banks and allocated in pairs, so that s0:s1,
16/// s16:s17, s32:s33 are at bank 0. s2:s3, s18:s19, s34:s35 are at bank 1 etc.
17///
18/// The shader can read one dword from each of these banks once per cycle.
19/// If an instruction has to read more register operands from the same bank
20/// an additional cycle is needed. HW attempts to pre-load registers through
21/// input operand gathering, but a stall cycle may occur if that fails. For
22/// example V_FMA_F32 V111 = V0 + V4 * V8 will need 3 cycles to read operands,
23/// potentially incuring 2 stall cycles.
24///
25/// The pass tries to reassign registers to reduce bank conflicts.
26///
27/// In this pass bank numbers 0-3 are VGPR banks and 4-11 are SGPR banks, so
28/// that 4 has to be subtracted from an SGPR bank number to get the real value.
29/// This also corresponds to bit numbers in bank masks used in the pass.
30///
31//===----------------------------------------------------------------------===//

295unsigned GCNRegBankReassign::getPhysRegBank(Register Reg,
296 unsigned SubReg) const {
297 assert(Reg.isPhysical());
298
299 const TargetRegisterClass *RC = TRI->getMinimalPhysRegClass(Reg);
300 unsigned Size = TRI->getRegSizeInBits(*RC);
301 if (Size == 16)
302 Reg = TRI->get32BitRegister(Reg);
303 else if (Size > 32) {
304 if (SubReg) {
305 const TargetRegisterClass *SubRC = TRI->getSubRegClass(RC, SubReg);
306 Reg = TRI->getSubReg(Reg, SubReg);
307 if (TRI->getRegSizeInBits(*SubRC) > 32)
308 Reg = TRI->getSubReg(Reg, AMDGPU::sub0);
309 } else {
310 Reg = TRI->getSubReg(Reg, AMDGPU::sub0);
311 }
312 }
313
314 if (TRI->hasVGPRs(RC)) {
315 unsigned RegNo = Reg - AMDGPU::VGPR0;
316 return RegNo % NUM_VGPR_BANKS;
317 }
318
319 unsigned RegNo = TRI->getEncodingValue(AMDGPU::getMCReg(Reg, *ST)) / 2;
320 return RegNo % NUM_SGPR_BANKS + SGPR_BANK_OFFSET;
321}

323uint32_t GCNRegBankReassign::getRegBankMask(Register Reg, unsigned SubReg,
324 int Bank) {
325 if (Reg.isVirtual()) {
326 if (!VRM->isAssignedReg(Reg))
327 return 0;
328
329 Reg = VRM->getPhys(Reg);
330 if (!Reg)
331 return 0;
332 if (SubReg)
333 Reg = TRI->getSubReg(Reg, SubReg);
334 }
335
336 const TargetRegisterClass *RC = TRI->getMinimalPhysRegClass(Reg);
337 unsigned Size = TRI->getRegSizeInBits(*RC);
338
339 if (Size == 16) {
340 Reg = TRI->get32BitRegister(Reg);
341 Size = 1;
342 } else {
343 Size /= 32;
344 if (Size > 1)
345 Reg = TRI->getSubReg(Reg, AMDGPU::sub0);
346 }
347
348 if (TRI->hasVGPRs(RC)) {
349 // VGPRs have 4 banks assigned in a round-robin fashion.
350 unsigned RegNo = Reg - AMDGPU::VGPR0;
351 uint32_t Mask = maskTrailingOnes<uint32_t>(Size);
352 unsigned Used = 0;
353 // Bitmask lacks an extract method
354 for (unsigned I = 0; I < Size; ++I)
355 if (RegsUsed.test(RegNo + I))
356 Used |= 1 << I;
357 RegsUsed.set(RegNo, RegNo + Size);
358 Mask &= ~Used;
359 Mask <<= (Bank == -1) ? RegNo % NUM_VGPR_BANKS : uint32_t(Bank);
360 return (Mask | (Mask >> NUM_VGPR_BANKS)) & VGPR_BANK_MASK;
361 }
362
363 // SGPRs have 8 banks holding 2 consequitive registers each.
364 unsigned RegNo = TRI->getEncodingValue(AMDGPU::getMCReg(Reg, *ST)) / 2;
365 unsigned StartBit = AMDGPU::VGPR_32RegClass.getNumRegs();
366 if (RegNo + StartBit >= RegsUsed.size())
367 return 0;
368
369 if (Size > 1)
370 Size /= 2;
371 unsigned Mask = (1 << Size) - 1;
372 unsigned Used = 0;
373 for (unsigned I = 0; I < Size; ++I)
374 if (RegsUsed.test(StartBit + RegNo + I))
375 Used |= 1 << I;
376 RegsUsed.set(StartBit + RegNo, StartBit + RegNo + Size);
377 Mask &= ~Used;
378 Mask <<= (Bank == -1) ? RegNo % NUM_SGPR_BANKS
379 : unsigned(Bank - SGPR_BANK_OFFSET);
380 Mask = (Mask | (Mask >> NUM_SGPR_BANKS)) & SGPR_BANK_SHIFTED_MASK;
381 // Reserve 4 bank ids for VGPRs.
382 return Mask << SGPR_BANK_OFFSET;
383}

385std::pair<unsigned, unsigned>
386GCNRegBankReassign::analyzeInst(const MachineInstr &MI, Register Reg,
387 unsigned SubReg, int Bank) {
388 unsigned StallCycles = 0;
389 unsigned UsedBanks = 0;
390
391 if (MI.isDebugValue())
392 return std::make_pair(StallCycles, UsedBanks);
393
394 RegsUsed.reset();
395 OperandMasks.clear();
396 for (const auto& Op : MI.explicit_uses()) {
397 // Undef can be assigned to any register, so two vregs can be assigned
398 // the same phys reg within the same instruction.
399 if (!Op.isReg() || Op.isUndef())
400 continue;
401
402 const Register R = Op.getReg();
403 const TargetRegisterClass *RC = TRI->getRegClassForReg(*MRI, R);
404
405 // Do not compute stalls for AGPRs
406 if (TRI->hasAGPRs(RC))
407 continue;
408
409 // Do not compute stalls if sub-register covers all banks
410 if (Op.getSubReg()) {
411 LaneBitmask LM = TRI->getSubRegIndexLaneMask(Op.getSubReg());
412 if (TRI->hasVGPRs(RC)) {
413 if (TRI->getNumCoveredRegs(LM) >= NUM_VGPR_BANKS)
414 continue;
415 } else {
416 if (TRI->getNumCoveredRegs(LM) / 2 >= NUM_SGPR_BANKS)
417 continue;
418 }
419 }
420
421 unsigned ShiftedBank = Bank;
422
423 if (Bank != -1 && R == Reg && (Op.getSubReg() || SubReg)) {
424 unsigned RegOffset =
425 TRI->getChannelFromSubReg(SubReg ? SubReg : (unsigned)AMDGPU::sub0);
426 unsigned Offset = TRI->getChannelFromSubReg(
427 Op.getSubReg() ? Op.getSubReg() : (unsigned)AMDGPU::sub0);
428 if (Bank < NUM_VGPR_BANKS) {
429 unsigned Shift = ((NUM_VGPR_BANKS + Offset) - RegOffset);
430 ShiftedBank = (Bank + Shift) % NUM_VGPR_BANKS;
431 } else if (Bank >= SGPR_BANK_OFFSET) {
432 unsigned Shift = (NUM_SGPR_BANKS + (Offset >> 1)) - (RegOffset >> 1);
433 ShiftedBank = SGPR_BANK_OFFSET +
434 (Bank - SGPR_BANK_OFFSET + Shift) % NUM_SGPR_BANKS;
435 }
436 }
437
438 uint32_t Mask = getRegBankMask(R, Op.getSubReg(),
439 (Reg == R) ? ShiftedBank : -1);
440 StallCycles += countPopulation(UsedBanks & Mask);
441 UsedBanks |= Mask;
442 OperandMasks.push_back(OperandMask(Op.getReg(), Op.getSubReg(), Mask));
443 }
444
445 return std::make_pair(StallCycles, UsedBanks);
446}

448unsigned GCNRegBankReassign::getOperandGatherWeight(const MachineInstr& MI,
449 Register Reg1,
450 Register Reg2,
451 unsigned StallCycles) const
452{
453 unsigned Defs = 0;
454 MachineBasicBlock::const_instr_iterator Def(MI.getIterator());
455 MachineBasicBlock::const_instr_iterator B(MI.getParent()->instr_begin());
456 for (unsigned S = StallCycles; S && Def != B && Defs != 3; --S) {
457 if (MI.isDebugInstr())
458 continue;
459 --Def;
460 if (Def->getOpcode() == TargetOpcode::IMPLICIT_DEF)
461 continue;
462 if (Def->modifiesRegister(Reg1, TRI))
463 Defs |= 1;
464 if (Def->modifiesRegister(Reg2, TRI))
465 Defs |= 2;
466 }
467 return countPopulation(Defs);
468}

470bool GCNRegBankReassign::isReassignable(Register Reg) const {
471 if (Reg.isPhysical() || !VRM->isAssignedReg(Reg))
472 return false;
473
474 const MachineInstr *Def = MRI->getUniqueVRegDef(Reg);
475
476 Register PhysReg = VRM->getPhys(Reg);
477
478 if (Def && Def->isCopy() && Def->getOperand(1).getReg() == PhysReg)
479 return false;
480
481 for (auto U : MRI->use_nodbg_operands(Reg)) {
482 if (U.isImplicit())
483 return false;
484 const MachineInstr *UseInst = U.getParent();
485 if (UseInst->isCopy() && UseInst->getOperand(0).getReg() == PhysReg)
486 return false;
487 }
488
489 const TargetRegisterClass *RC = TRI->getMinimalPhysRegClass(PhysReg);
490 unsigned Size = TRI->getRegSizeInBits(*RC);
491
492 // TODO: Support 16 bit registers. Those needs to be moved with their
493 // parent VGPR_32 and potentially a sibling 16 bit sub-register.
494 if (Size < 32)
495 return false;
496
497 if (TRI->hasVGPRs(RC))
498 return true;
499
500 if (Size == 16)
501 return AMDGPU::SGPR_LO16RegClass.contains(PhysReg);
502
503 if (Size > 32)
504 PhysReg = TRI->getSubReg(PhysReg, AMDGPU::sub0);
505
506 return AMDGPU::SGPR_32RegClass.contains(PhysReg);
507}

509unsigned GCNRegBankReassign::getFreeBanks(unsigned Mask,
510 unsigned UsedBanks) const {
511 unsigned Size = countPopulation(Mask);
512 unsigned FreeBanks = 0;
513 unsigned Bank = findFirstSet(Mask);
514
515 UsedBanks &= ~Mask;
516
517 // Find free VGPR banks
518 if ((Mask & VGPR_BANK_MASK) && (Size < NUM_VGPR_BANKS)) {
519 for (unsigned I = 0; I < NUM_VGPR_BANKS; ++I) {
520 if (Bank == I)
521 continue;
522 unsigned NewMask = ((1 << Size) - 1) << I;
523 NewMask = (NewMask | (NewMask >> NUM_VGPR_BANKS)) & VGPR_BANK_MASK;
524 if (!(UsedBanks & NewMask))
525 FreeBanks |= 1 << I;
526 }
527 return FreeBanks;
528 }
529
530 // Find free SGPR banks
531 // SGPR tuples must be aligned, so step is size in banks it
532 // crosses.
533 Bank -= SGPR_BANK_OFFSET;
534 for (unsigned I = 0; I < NUM_SGPR_BANKS; I += Size) {
535 if (Bank == I)
536 continue;
537 unsigned NewMask = ((1 << Size) - 1) << I;
538 NewMask = (NewMask | (NewMask >> NUM_SGPR_BANKS)) & SGPR_BANK_SHIFTED_MASK;
539 if (!(UsedBanks & (NewMask << SGPR_BANK_OFFSET)))
540 FreeBanks |= (1 << SGPR_BANK_OFFSET) << I;
541 }
542
543 return FreeBanks;
544}

546unsigned GCNRegBankReassign::getFreeBanks(Register Reg,
547 unsigned SubReg,
548 unsigned Mask,
549 unsigned UsedBanks) const {
550 if (!isReassignable(Reg))
551 return 0;
552
553 unsigned FreeBanks = getFreeBanks(Mask, UsedBanks);
554
555 unsigned Offset = TRI->getChannelFromSubReg(SubReg);
556 if (Offset && (Mask & VGPR_BANK_MASK)) {
557 unsigned Shift = Offset;
558 if (Shift >= NUM_VGPR_BANKS)
559 return 0;
560 unsigned VB = FreeBanks & VGPR_BANK_MASK;
561 FreeBanks = ((VB >> Shift) | (VB << (NUM_VGPR_BANKS - Shift))) &
562 VGPR_BANK_MASK;
563 } else if (Offset > 1 && (Mask & SGPR_BANK_MASK)) {
564 unsigned Shift = Offset >> 1;
565 if (Shift >= NUM_SGPR_BANKS)
566 return 0;
567 unsigned SB = FreeBanks >> SGPR_BANK_OFFSET;
568 FreeBanks = ((SB >> Shift) | (SB << (NUM_SGPR_BANKS - Shift))) &
569 SGPR_BANK_SHIFTED_MASK;
570 FreeBanks <<= SGPR_BANK_OFFSET;
571 }
572
573 LLVM_DEBUG(if (FreeBanks) {
574 dbgs() << "Potential reassignments of " << printReg(Reg, SubReg)
575 << " to banks: "; dumpFreeBanks(FreeBanks);
576 dbgs() << '\n'; });
577
578 return FreeBanks;
579}

581void GCNRegBankReassign::collectCandidates(MachineInstr& MI,
582 unsigned UsedBanks,
583 unsigned StallCycles) {
584 LLVM_DEBUG(MI.dump());
585
586 if (!StallCycles)
587 return;
588
589 LLVM_DEBUG(dbgs() << "Stall cycles = " << StallCycles << '\n');
590
591 for (unsigned I = 0, E = OperandMasks.size(); I + 1 < E; ++I) {
592 for (unsigned J = I + 1; J != E; ++J) {
593 if (!(OperandMasks[I].Mask & OperandMasks[J].Mask))
594 continue;
595
596 Register Reg1 = OperandMasks[I].Reg;
597 Register Reg2 = OperandMasks[J].Reg;
598 unsigned SubReg1 = OperandMasks[I].SubReg;
599 unsigned SubReg2 = OperandMasks[J].SubReg;
600 unsigned Mask1 = OperandMasks[I].Mask;
601 unsigned Mask2 = OperandMasks[J].Mask;
602 unsigned Size1 = countPopulation(Mask1);
603 unsigned Size2 = countPopulation(Mask2);
604
605 LLVM_DEBUG(dbgs() << "Conflicting operands: " << printReg(Reg1, SubReg1) <<
606 " and " << printReg(Reg2, SubReg2) << '\n');
607
608 unsigned Weight = getOperandGatherWeight(MI, Reg1, Reg2, StallCycles);
609 Weight += MLI->getLoopDepth(MI.getParent()) * 10;
610
611 LLVM_DEBUG(dbgs() << "Stall weight = " << Weight << '\n');
612
613 unsigned FreeBanks1 = getFreeBanks(Reg1, SubReg1, Mask1, UsedBanks);
614 unsigned FreeBanks2 = getFreeBanks(Reg2, SubReg2, Mask2, UsedBanks);
615 if (FreeBanks1)
616 Candidates.push(Weight + ((Size2 > Size1) ? 1 : 0),
617 Candidate(&MI, Reg1, SubReg1, FreeBanks1));
618 if (FreeBanks2)
619 Candidates.push(Weight + ((Size1 > Size2) ? 1 : 0),
620 Candidate(&MI, Reg2, SubReg2, FreeBanks2));
621 }
622 }

625unsigned GCNRegBankReassign::computeStallCycles(Register SrcReg, Register Reg,
626 unsigned SubReg, int Bank,
627 bool Collect) {
628 unsigned TotalStallCycles = 0;
629 SmallSet<const MachineInstr *, 16> Visited;
630
631 for (auto &MI : MRI->use_nodbg_instructions(SrcReg)) {
632 if (MI.isBundle())
633 continue;
634 if (!Visited.insert(&MI).second)
635 continue;
636 unsigned StallCycles;
637 unsigned UsedBanks;
638 std::tie(StallCycles, UsedBanks) = analyzeInst(MI, Reg, SubReg, Bank);
639 TotalStallCycles += StallCycles;
640 if (Collect)
641 collectCandidates(MI, UsedBanks, StallCycles);
642 }
643
644 return TotalStallCycles;
645}

647MCRegister GCNRegBankReassign::scavengeReg(LiveInterval &LI, unsigned Bank,
648 unsigned SubReg) const {
649 const TargetRegisterClass *RC = MRI->getRegClass(LI.reg());
650 unsigned MaxNumRegs = (Bank < NUM_VGPR_BANKS) ? MaxNumVGPRs
651 : MaxNumSGPRs;
652 unsigned MaxReg = MaxNumRegs + (Bank < NUM_VGPR_BANKS ? AMDGPU::VGPR0
653 : AMDGPU::SGPR0);
654
655 for (MCRegister Reg : RC->getRegisters()) {
656 // Check occupancy limit.
657 if (TRI->isSubRegisterEq(Reg, MaxReg))
658 break;
659
660 if (!MRI->isAllocatable(Reg) || getPhysRegBank(Reg, SubReg) != Bank)
661 continue;
662
663 for (unsigned I = 0; CSRegs[I]; ++I)
664 if (TRI->isSubRegisterEq(Reg, CSRegs[I]) &&
665 !LRM->isPhysRegUsed(CSRegs[I]))
666 return MCRegister::from(AMDGPU::NoRegister);
667
668 LLVM_DEBUG(dbgs() << "Trying register " << printReg(Reg) << '\n');
669
670 if (!LRM->checkInterference(LI, Reg))
671 return Reg;
672 }
673
674 return MCRegister::from(AMDGPU::NoRegister);
675}

677unsigned GCNRegBankReassign::tryReassign(Candidate &C) {
678 if (!LIS->hasInterval(C.Reg))
679 return 0;
680
681 LiveInterval &LI = LIS->getInterval(C.Reg);
682 LLVM_DEBUG(dbgs() << "Try reassign " << printReg(C.Reg) << " in "; C.MI->dump();
683 LI.dump());
684
685 // For each candidate bank walk all instructions in the range of live
686 // interval and check if replacing the register with one belonging to
687 // the candidate bank reduces conflicts.
688
689 unsigned OrigStalls = computeStallCycles(C.Reg);
690 LLVM_DEBUG(dbgs() << "--- Stall cycles in range = " << OrigStalls << '\n');
691 if (!OrigStalls)
692 return 0;
693
694 struct BankStall {
695 BankStall(unsigned b, unsigned s) : Bank(b), Stalls(s) {};
696 bool operator<(const BankStall &RHS) const {
697 if (Stalls == RHS.Stalls)
698 return Bank < RHS.Bank;
699 return Stalls > RHS.Stalls;
700 }
701 unsigned Bank;
702 unsigned Stalls;
703 };
704 SmallVector<BankStall, 8> BankStalls;
705
706 for (int Bank = 0; Bank < NUM_BANKS; ++Bank) {
707 if (C.FreeBanks & (1 << Bank)) {
708 LLVM_DEBUG(dbgs() << "Trying bank " << printBank(Bank) << '\n');
709 unsigned Stalls = computeStallCycles(C.Reg, C.Reg, C.SubReg, Bank);
710 if (Stalls < OrigStalls) {
711 LLVM_DEBUG(dbgs() << "With bank " << printBank(Bank) << " -> "
712 << Stalls << '\n');
713 BankStalls.push_back(BankStall((unsigned)Bank, Stalls));
714 }
715 }
716 }
717 llvm::sort(BankStalls);
718
719 MCRegister OrigReg = VRM->getPhys(C.Reg);
720 LRM->unassign(LI);
721 while (!BankStalls.empty()) {
722 BankStall BS = BankStalls.pop_back_val();
723 MCRegister Reg = scavengeReg(LI, BS.Bank, C.SubReg);
724 if (Reg == AMDGPU::NoRegister) {
725 LLVM_DEBUG(dbgs() << "No free registers in bank " << printBank(BS.Bank)
726 << '\n');
727 continue;
728 }
729 LLVM_DEBUG(dbgs() << "Found free register " << printReg(Reg)
730 << (LRM->isPhysRegUsed(Reg) ? "" : " (new)")
731 << " in bank " << printBank(BS.Bank) << '\n');
732
733 LRM->assign(LI, Reg);
734
735 LLVM_DEBUG(dbgs() << "--- Cycles saved: " << OrigStalls - BS.Stalls << '\n');
736
737 return OrigStalls - BS.Stalls;
738 }
739 LRM->assign(LI, OrigReg);
740
741 return 0;
742}

744unsigned GCNRegBankReassign::collectCandidates(MachineFunction &MF,
745 bool Collect) {
746 unsigned TotalStallCycles = 0;
747
748 for (MachineBasicBlock &MBB : MF) {
749
750 LLVM_DEBUG(if (Collect) {
751 if (MBB.getName().empty()) dbgs() << "bb." << MBB.getNumber();
752 else dbgs() << MBB.getName(); dbgs() << ":\n";
753 });
754
755 for (MachineInstr &MI : MBB.instrs()) {
756 if (MI.isBundle())
757 continue; // we analyze the instructions inside the bundle individually
758
759 unsigned StallCycles;
760 unsigned UsedBanks;
761 std::tie(StallCycles, UsedBanks) = analyzeInst(MI);
762
763 if (Collect)
764 collectCandidates(MI, UsedBanks, StallCycles);
765
766 TotalStallCycles += StallCycles;
767 }
768
769 LLVM_DEBUG(if (Collect) { dbgs() << '\n'; });
770 }
771
772 return TotalStallCycles;
773}

775void GCNRegBankReassign::removeCandidates(Register Reg) {
776 typename CandidateList::iterator Next;
777 for (auto I = Candidates.begin(), E = Candidates.end(); I != E; I = Next) {
778 Next = std::next(I);
779 I->second.remove_if([Reg, this](const Candidate& C) {
780 return C.MI->readsRegister(Reg, TRI);
781 });
782 if (I->second.empty())
783 Candidates.erase(I);
784 }
785}

787bool GCNRegBankReassign::verifyCycles(MachineFunction &MF,
788 unsigned OriginalCycles,
789 unsigned CyclesSaved) {
790 unsigned StallCycles = collectCandidates(MF, false);
791 LLVM_DEBUG(dbgs() << "=== After the pass " << StallCycles
792 << " stall cycles left\n");
793 return StallCycles + CyclesSaved == OriginalCycles;
794}

796bool GCNRegBankReassign::runOnMachineFunction(MachineFunction &MF) {
797 ST = &MF.getSubtarget<GCNSubtarget>();
798 if (!ST->hasRegisterBanking() || skipFunction(MF.getFunction()))
799 return false;
800
801 MRI = &MF.getRegInfo();
802 TRI = ST->getRegisterInfo();
803 MLI = &getAnalysis<MachineLoopInfo>();
804 VRM = &getAnalysis<VirtRegMap>();
805 LRM = &getAnalysis<LiveRegMatrix>();
806 LIS = &getAnalysis<LiveIntervals>();
807
808 const SIMachineFunctionInfo *MFI = MF.getInfo<SIMachineFunctionInfo>();
809 unsigned Occupancy = MFI->getOccupancy();
810 MaxNumVGPRs = ST->getMaxNumVGPRs(MF);
811 MaxNumSGPRs = ST->getMaxNumSGPRs(MF);
812 MaxNumVGPRs = std::min(ST->getMaxNumVGPRs(Occupancy), MaxNumVGPRs);
813 MaxNumSGPRs = std::min(ST->getMaxNumSGPRs(Occupancy, true), MaxNumSGPRs);
814
815 CSRegs = MRI->getCalleeSavedRegs();
816 unsigned NumRegBanks = AMDGPU::VGPR_32RegClass.getNumRegs() +
817 // Not a tight bound
818 AMDGPU::SReg_32RegClass.getNumRegs() / 2 + 1;
819 RegsUsed.resize(NumRegBanks);
820
821 LLVM_DEBUG(dbgs() << "=== RegBanks reassign analysis on function " << MF.getName()
822 << '\n');
823
824 unsigned StallCycles = collectCandidates(MF);
825 NumStallsDetected += StallCycles;
826
827 LLVM_DEBUG(dbgs() << "=== " << StallCycles << " stall cycles detected in "
828 "function " << MF.getName() << '\n');
829
830 LLVM_DEBUG(Candidates.dump(this));
831
832 unsigned CyclesSaved = 0;
833 while (!Candidates.empty()) {
834 Candidate C = Candidates.back();
835 unsigned LocalCyclesSaved = tryReassign(C);
836 CyclesSaved += LocalCyclesSaved;
837
838 if (VerifyStallCycles > 1 && !verifyCycles(MF, StallCycles, CyclesSaved))
839 report_fatal_error("RegBank reassign stall cycles verification failed.");
840
841 Candidates.pop_back();
842 if (LocalCyclesSaved) {
843 removeCandidates(C.Reg);
844 computeStallCycles(C.Reg, AMDGPU::NoRegister, 0, -1, true);
845
846 LLVM_DEBUG(Candidates.dump(this));
847 }
848 }
849 NumStallsRecovered += CyclesSaved;
850
851 LLVM_DEBUG(dbgs() << "=== After the pass " << CyclesSaved
852 << " cycles saved in function " << MF.getName() << '\n');
853
854 Candidates.clear();
855
856 if (VerifyStallCycles == 1 && !verifyCycles(MF, StallCycles, CyclesSaved))
857 report_fatal_error("RegBank reassign stall cycles verification failed.");
858
859 RegsUsed.clear();
860
861 return CyclesSaved > 0;
862}
