284 // \returns number of 32 bit registers covered by a \p LM
285 static unsigned getNumCoveredRegs(LaneBitmask LM) {
286 // The assumption is that every lo16 subreg is an even bit and every hi16
287 // is an adjacent odd bit or vice versa.
288 uint64_t Mask = LM.getAsInteger();
289 uint64_t Even = Mask & 0xAAAAAAAAAAAAAAAAULL;
290 Mask = (Even >> 1) | Mask;
291 uint64_t Odd = Mask & 0x5555555555555555ULL;
292 return countPopulation(Odd);
293 }
