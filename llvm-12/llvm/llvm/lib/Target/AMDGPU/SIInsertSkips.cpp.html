87static bool opcodeEmitsNoInsts(const MachineInstr &MI) {
88 if (MI.isMetaInstruction())
89 return true;
90
91 // Handle target specific opcodes.
92 switch (MI.getOpcode()) {
93 case AMDGPU::SI_MASK_BRANCH:
94 return true;
95 default:
96 return false;
97 }
98}

100bool SIInsertSkips::shouldSkip(const MachineBasicBlock &From,
101 const MachineBasicBlock &To) const {
102 unsigned NumInstr = 0;
103 const MachineFunction *MF = From.getParent();
104
105 for (MachineFunction::const_iterator MBBI(&From), ToI(&To), End = MF->end();
106 MBBI != End && MBBI != ToI; ++MBBI) {
107 const MachineBasicBlock &MBB = *MBBI;
108
109 for (MachineBasicBlock::const_iterator I = MBB.begin(), E = MBB.end();
110 NumInstr < SkipThreshold && I != E; ++I) {
111 if (opcodeEmitsNoInsts(*I))
112 continue;
113
114 // FIXME: Since this is required for correctness, this should be inserted
115 // during SILowerControlFlow.
116
117 // When a uniform loop is inside non-uniform control flow, the branch
118 // leaving the loop might be an S_CBRANCH_VCCNZ, which is never taken
119 // when EXEC = 0. We should skip the loop lest it becomes infinite.
120 if (I->getOpcode() == AMDGPU::S_CBRANCH_VCCNZ ||
121 I->getOpcode() == AMDGPU::S_CBRANCH_VCCZ)
122 return true;
123
124 if (TII->hasUnwantedEffectsWhenEXECEmpty(*I))
125 return true;
126
127 // These instructions are potentially expensive even if EXEC = 0.
128 if (TII->isSMRD(*I) || TII->isVMEM(*I) || TII->isFLAT(*I) ||
129 I->getOpcode() == AMDGPU::S_WAITCNT)
130 return true;
131
132 ++NumInstr;
133 if (NumInstr >= SkipThreshold)
134 return true;
135 }
136 }
137
138 return false;
139}

141/// Check whether \p MBB dominates all blocks that are reachable from it.
142bool SIInsertSkips::dominatesAllReachable(MachineBasicBlock &MBB) {
143 for (MachineBasicBlock *Other : depth_first(&MBB)) {
144 if (!MDT->dominates(&MBB, Other))
145 return false;
146 }
147 return true;
148}

150static void generateEndPgm(MachineBasicBlock &MBB,
151 MachineBasicBlock::iterator I, DebugLoc DL,
152 const SIInstrInfo *TII, bool IsPS) {
153 // "null export"
154 if (IsPS) {
155 BuildMI(MBB, I, DL, TII->get(AMDGPU::EXP_DONE))
156 .addImm(AMDGPU::Exp::ET_NULL)
157 .addReg(AMDGPU::VGPR0, RegState::Undef)
158 .addReg(AMDGPU::VGPR0, RegState::Undef)
159 .addReg(AMDGPU::VGPR0, RegState::Undef)
160 .addReg(AMDGPU::VGPR0, RegState::Undef)
161 .addImm(1) // vm
162 .addImm(0) // compr
163 .addImm(0); // en
164 }
165 // s_endpgm
166 BuildMI(MBB, I, DL, TII->get(AMDGPU::S_ENDPGM)).addImm(0);
167}

169void SIInsertSkips::ensureEarlyExitBlock(MachineBasicBlock &MBB,
170 bool ClearExec) {
171 MachineFunction *MF = MBB.getParent();
172 DebugLoc DL;
173
174 if (!EarlyExitBlock) {
175 EarlyExitBlock = MF->CreateMachineBasicBlock();
176 MF->insert(MF->end(), EarlyExitBlock);
177 generateEndPgm(*EarlyExitBlock, EarlyExitBlock->end(), DL, TII,
178 MF->getFunction().getCallingConv() ==
179 CallingConv::AMDGPU_PS);
180 EarlyExitClearsExec = false;
181 }
182
183 if (ClearExec && !EarlyExitClearsExec) {
184 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
185 unsigned Mov = ST.isWave32() ? AMDGPU::S_MOV_B32 : AMDGPU::S_MOV_B64;
186 Register Exec = ST.isWave32() ? AMDGPU::EXEC_LO : AMDGPU::EXEC;
187 auto ExitI = EarlyExitBlock->getFirstNonPHI();
188 BuildMI(*EarlyExitBlock, ExitI, DL, TII->get(Mov), Exec).addImm(0);
189 EarlyExitClearsExec = true;
190 }
191}

193static void splitBlock(MachineBasicBlock &MBB, MachineInstr &MI,
194 MachineDominatorTree *MDT) {
195 MachineBasicBlock *SplitBB = MBB.splitAt(MI, /*UpdateLiveIns*/ true);
196
197 // Update dominator tree
198 using DomTreeT = DomTreeBase<MachineBasicBlock>;
199 SmallVector<DomTreeT::UpdateType, 16> DTUpdates;
200 for (MachineBasicBlock *Succ : SplitBB->successors()) {
201 DTUpdates.push_back({DomTreeT::Insert, SplitBB, Succ});
202 DTUpdates.push_back({DomTreeT::Delete, &MBB, Succ});
203 }
204 DTUpdates.push_back({DomTreeT::Insert, &MBB, SplitBB});
205 MDT->getBase().applyUpdates(DTUpdates);
206}

208/// Insert an "if exec=0 { null export; s_endpgm }" sequence before the given
209/// iterator. Only applies to pixel shaders.
210void SIInsertSkips::skipIfDead(MachineBasicBlock &MBB,
211 MachineBasicBlock::iterator I, DebugLoc DL) {
212 MachineFunction *MF = MBB.getParent();
213 (void)MF;
214 assert(MF->getFunction().getCallingConv() == CallingConv::AMDGPU_PS);
215
216 // It is possible for an SI_KILL_*_TERMINATOR to sit at the bottom of a
217 // basic block that has no further successors (e.g., there was an
218 // `unreachable` there in IR). This can happen with original source of the
219 // form:
220 //
221 // if (uniform_condition) {
222 // write_to_memory();
223 // discard;
224 // }
225 //
226 // In this case, we write the "null_export; s_endpgm" skip code in the
227 // already-existing basic block.
228 auto NextBBI = std::next(MBB.getIterator());
229 bool NoSuccessor =
230 I == MBB.end() && !llvm::is_contained(MBB.successors(), &*NextBBI);
231
232 if (NoSuccessor) {
233 generateEndPgm(MBB, I, DL, TII, true);
234 } else {
235 ensureEarlyExitBlock(MBB, false);
236
237 MachineInstr *BranchMI =
238 BuildMI(MBB, I, DL, TII->get(AMDGPU::S_CBRANCH_EXECZ))
239 .addMBB(EarlyExitBlock);
240
241 // Split the block if the branch will not come at the end.
242 auto Next = std::next(BranchMI->getIterator());
243 if (Next != MBB.end() && !Next->isTerminator())
244 splitBlock(MBB, *BranchMI, MDT);
245
246 MBB.addSuccessor(EarlyExitBlock);
247 MDT->getBase().insertEdge(&MBB, EarlyExitBlock);
248 }
249}

251/// Translate a SI_KILL_*_TERMINATOR into exec-manipulating instructions.
252/// Return true unless the terminator is a no-op.
253bool SIInsertSkips::kill(MachineInstr &MI) {
254 MachineBasicBlock &MBB = *MI.getParent();
255 DebugLoc DL = MI.getDebugLoc();
256
257 switch (MI.getOpcode()) {
258 case AMDGPU::SI_KILL_F32_COND_IMM_TERMINATOR: {
259 unsigned Opcode = 0;
260
261 // The opcodes are inverted because the inline immediate has to be
262 // the first operand, e.g. from "x < imm" to "imm > x"
263 switch (MI.getOperand(2).getImm()) {
264 case ISD::SETOEQ:
265 case ISD::SETEQ:
266 Opcode = AMDGPU::V_CMPX_EQ_F32_e64;
267 break;
268 case ISD::SETOGT:
269 case ISD::SETGT:
270 Opcode = AMDGPU::V_CMPX_LT_F32_e64;
271 break;
272 case ISD::SETOGE:
273 case ISD::SETGE:
274 Opcode = AMDGPU::V_CMPX_LE_F32_e64;
275 break;
276 case ISD::SETOLT:
277 case ISD::SETLT:
278 Opcode = AMDGPU::V_CMPX_GT_F32_e64;
279 break;
280 case ISD::SETOLE:
281 case ISD::SETLE:
282 Opcode = AMDGPU::V_CMPX_GE_F32_e64;
283 break;
284 case ISD::SETONE:
285 case ISD::SETNE:
286 Opcode = AMDGPU::V_CMPX_LG_F32_e64;
287 break;
288 case ISD::SETO:
289 Opcode = AMDGPU::V_CMPX_O_F32_e64;
290 break;
291 case ISD::SETUO:
292 Opcode = AMDGPU::V_CMPX_U_F32_e64;
293 break;
294 case ISD::SETUEQ:
295 Opcode = AMDGPU::V_CMPX_NLG_F32_e64;
296 break;
297 case ISD::SETUGT:
298 Opcode = AMDGPU::V_CMPX_NGE_F32_e64;
299 break;
300 case ISD::SETUGE:
301 Opcode = AMDGPU::V_CMPX_NGT_F32_e64;
302 break;
303 case ISD::SETULT:
304 Opcode = AMDGPU::V_CMPX_NLE_F32_e64;
305 break;
306 case ISD::SETULE:
307 Opcode = AMDGPU::V_CMPX_NLT_F32_e64;
308 break;
309 case ISD::SETUNE:
310 Opcode = AMDGPU::V_CMPX_NEQ_F32_e64;
311 break;
312 default:
313 llvm_unreachable("invalid ISD:SET cond code");
314 }
315
316 const GCNSubtarget &ST = MBB.getParent()->getSubtarget<GCNSubtarget>();
317 if (ST.hasNoSdstCMPX())
318 Opcode = AMDGPU::getVCMPXNoSDstOp(Opcode);
319
320 assert(MI.getOperand(0).isReg());
321
322 if (TRI->isVGPR(MBB.getParent()->getRegInfo(),
323 MI.getOperand(0).getReg())) {
324 Opcode = AMDGPU::getVOPe32(Opcode);
325 BuildMI(MBB, &MI, DL, TII->get(Opcode))
326 .add(MI.getOperand(1))
327 .add(MI.getOperand(0));
328 } else {
329 auto I = BuildMI(MBB, &MI, DL, TII->get(Opcode));
330 if (!ST.hasNoSdstCMPX())
331 I.addReg(AMDGPU::VCC, RegState::Define);
332
333 I.addImm(0) // src0 modifiers
334 .add(MI.getOperand(1))
335 .addImm(0) // src1 modifiers
336 .add(MI.getOperand(0));
337
338 I.addImm(0); // omod
339 }
340 return true;
341 }
342 case AMDGPU::SI_KILL_I1_TERMINATOR: {
343 const MachineFunction *MF = MI.getParent()->getParent();
344 const GCNSubtarget &ST = MF->getSubtarget<GCNSubtarget>();
345 unsigned Exec = ST.isWave32() ? AMDGPU::EXEC_LO : AMDGPU::EXEC;
346 const MachineOperand &Op = MI.getOperand(0);
347 int64_t KillVal = MI.getOperand(1).getImm();
348 assert(KillVal == 0 || KillVal == -1);
349
350 // Kill all threads if Op0 is an immediate and equal to the Kill value.
351 if (Op.isImm()) {
352 int64_t Imm = Op.getImm();
353 assert(Imm == 0 || Imm == -1);
354
355 if (Imm == KillVal) {
356 BuildMI(MBB, &MI, DL, TII->get(ST.isWave32() ? AMDGPU::S_MOV_B32
357 : AMDGPU::S_MOV_B64), Exec)
358 .addImm(0);
359 return true;
360 }
361 return false;
362 }
363
364 unsigned Opcode = KillVal ? AMDGPU::S_ANDN2_B64 : AMDGPU::S_AND_B64;
365 if (ST.isWave32())
366 Opcode = KillVal ? AMDGPU::S_ANDN2_B32 : AMDGPU::S_AND_B32;
367 BuildMI(MBB, &MI, DL, TII->get(Opcode), Exec)
368 .addReg(Exec)
369 .add(Op);
370 return true;
371 }
372 default:
373 llvm_unreachable("invalid opcode, expected SI_KILL_*_TERMINATOR");
374 }
375}

377void SIInsertSkips::earlyTerm(MachineInstr &MI) {
378 MachineBasicBlock &MBB = *MI.getParent();
379 const DebugLoc DL = MI.getDebugLoc();
380
381 ensureEarlyExitBlock(MBB, true);
382
383 auto BranchMI = BuildMI(MBB, MI, DL, TII->get(AMDGPU::S_CBRANCH_SCC0))
384 .addMBB(EarlyExitBlock);
385 auto Next = std::next(MI.getIterator());
386
387 if (Next != MBB.end() && !Next->isTerminator())
388 splitBlock(MBB, *BranchMI, MDT);
389
390 MBB.addSuccessor(EarlyExitBlock);
391 MDT->getBase().insertEdge(&MBB, EarlyExitBlock);
392}

394// Returns true if a branch over the block was inserted.
395bool SIInsertSkips::skipMaskBranch(MachineInstr &MI,
396 MachineBasicBlock &SrcMBB) {
397 MachineBasicBlock *DestBB = MI.getOperand(0).getMBB();
398
399 if (!shouldSkip(**SrcMBB.succ_begin(), *DestBB))
400 return false;
401
402 const DebugLoc &DL = MI.getDebugLoc();
403 MachineBasicBlock::iterator InsPt = std::next(MI.getIterator());
404
405 BuildMI(SrcMBB, InsPt, DL, TII->get(AMDGPU::S_CBRANCH_EXECZ))
406 .addMBB(DestBB);
407
408 return true;
409}

411bool SIInsertSkips::runOnMachineFunction(MachineFunction &MF) {
412 const GCNSubtarget &ST = MF.getSubtarget<GCNSubtarget>();
413 TII = ST.getInstrInfo();
414 TRI = &TII->getRegisterInfo();
415 MDT = &getAnalysis<MachineDominatorTree>();
416 SkipThreshold = SkipThresholdFlag;
417
418 SmallVector<MachineInstr *, 4> KillInstrs;
419 SmallVector<MachineInstr *, 4> EarlyTermInstrs;
420 bool MadeChange = false;
421
422 for (MachineBasicBlock &MBB : MF) {
423 MachineBasicBlock::iterator I, Next;
424 for (I = MBB.begin(); I != MBB.end(); I = Next) {
425 Next = std::next(I);
426 MachineInstr &MI = *I;
427
428 switch (MI.getOpcode()) {
429 case AMDGPU::SI_MASK_BRANCH:
430 MadeChange |= skipMaskBranch(MI, MBB);
431 break;
432
433 case AMDGPU::S_BRANCH:
434 // Optimize out branches to the next block.
435 // FIXME: Shouldn't this be handled by BranchFolding?
436 if (MBB.isLayoutSuccessor(MI.getOperand(0).getMBB())) {
437 assert(&MI == &MBB.back());
438 MI.eraseFromParent();
439 MadeChange = true;
440 }
441 break;
442
443 case AMDGPU::SI_KILL_F32_COND_IMM_TERMINATOR:
444 case AMDGPU::SI_KILL_I1_TERMINATOR: {
445 MadeChange = true;
446 bool CanKill = kill(MI);
447
448 // Check if we can add an early "if exec=0 { end shader }".
449 //
450 // Note that we _always_ do this if it is correct, even if the kill
451 // happens fairly late in the shader, because the null export should
452 // generally still be cheaper than normal export(s).
453 //
454 // TODO: The dominatesAllReachable check is conservative: if the
455 // dominance is only missing due to _uniform_ branches, we could
456 // in fact insert the early-exit as well.
457 if (CanKill &&
458 MF.getFunction().getCallingConv() == CallingConv::AMDGPU_PS &&
459 dominatesAllReachable(MBB)) {
460 // Mark the instruction for kill-if-dead insertion. We delay this
461 // change because it modifies the CFG.
462 KillInstrs.push_back(&MI);
463 } else {
464 MI.eraseFromParent();
465 }
466 break;
467 }
468
469 case AMDGPU::SI_KILL_CLEANUP:
470 if (MF.getFunction().getCallingConv() == CallingConv::AMDGPU_PS &&
471 dominatesAllReachable(MBB)) {
472 KillInstrs.push_back(&MI);
473 } else {
474 MI.eraseFromParent();
475 }
476 break;
477
478 case AMDGPU::SI_EARLY_TERMINATE_SCC0:
479 EarlyTermInstrs.push_back(&MI);
480 break;
481
482 default:
483 break;
484 }
485 }
486 }
487
488 for (MachineInstr *Instr : EarlyTermInstrs) {
489 // Early termination in GS does nothing
490 if (MF.getFunction().getCallingConv() != CallingConv::AMDGPU_GS)
491 earlyTerm(*Instr);
492 Instr->eraseFromParent();
493 }
494 for (MachineInstr *Kill : KillInstrs) {
495 skipIfDead(*Kill->getParent(), std::next(Kill->getIterator()),
496 Kill->getDebugLoc());
497 Kill->eraseFromParent();
498 }
499 KillInstrs.clear();
500 EarlyTermInstrs.clear();
501 EarlyExitBlock = nullptr;
502
503 return MadeChange;
504}
