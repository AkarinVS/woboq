23//===----------------------------------------------------------------------===//
24// AMDGPU Target Machine (R600+)
25//===----------------------------------------------------------------------===//
26
27class AMDGPUTargetMachine : public LLVMTargetMachine {
28protected:
29 std::unique_ptr<TargetLoweringObjectFile> TLOF;
30
31 StringRef getGPUName(const Function &F) const;
32 StringRef getFeatureString(const Function &F) const;
33
34public:
35 static bool EnableLateStructurizeCFG;
36 static bool EnableFunctionCalls;
37 static bool EnableFixedFunctionABI;
38
39 AMDGPUTargetMachine(const Target &T, const Triple &TT, StringRef CPU,
40 StringRef FS, TargetOptions Options,
41 Optional<Reloc::Model> RM, Optional<CodeModel::Model> CM,
42 CodeGenOpt::Level OL);
43 ~AMDGPUTargetMachine() override;
44
45 const TargetSubtargetInfo *getSubtargetImpl() const;
46 const TargetSubtargetInfo *getSubtargetImpl(const Function &) const override = 0;
47
48 TargetLoweringObjectFile *getObjFileLowering() const override {
49 return TLOF.get();
50 }
51
52 void adjustPassManager(PassManagerBuilder &) override;
53
54 void registerPassBuilderCallbacks(PassBuilder &PB,
55 bool DebugPassManager) override;
56 void registerDefaultAliasAnalyses(AAManager &) override;
57
58 /// Get the integer value of a null pointer in the given address space.
59 static int64_t getNullPointerValue(unsigned AddrSpace);
60
61 bool isNoopAddrSpaceCast(unsigned SrcAS, unsigned DestAS) const override;
62
63 unsigned getAssumedAddrSpace(const Value *V) const override;
64};
