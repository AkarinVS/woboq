41MSP430Subtarget &
42MSP430Subtarget::initializeSubtargetDependencies(StringRef CPU, StringRef FS) {
43 ExtendedInsts = false;
44 HWMultMode = NoHWMult;
45
46 StringRef CPUName = CPU;
47 if (CPUName.empty())
48 CPUName = "msp430";
49
50 ParseSubtargetFeatures(CPUName, /*TuneCPU*/ CPUName, FS);
51
52 if (HWMultModeOption != NoHWMult)
53 HWMultMode = HWMultModeOption;
54
55 return *this;
56}
