21RISCVCallLowering::RISCVCallLowering(const RISCVTargetLowering &TLI)
22 : CallLowering(&TLI) {}
23
24bool RISCVCallLowering::lowerReturn(MachineIRBuilder &MIRBuilder,
25 const Value *Val, ArrayRef<Register> VRegs,
26 FunctionLoweringInfo &FLI) const {
27
28 MachineInstrBuilder Ret = MIRBuilder.buildInstrNoInsert(RISCV::PseudoRET);
29
30 if (Val != nullptr) {
31 return false;
32 }
33 MIRBuilder.insertInstr(Ret);
34 return true;
35}
36
37bool RISCVCallLowering::lowerFormalArguments(MachineIRBuilder &MIRBuilder,
38 const Function &F,
39 ArrayRef<ArrayRef<Register>> VRegs,
40 FunctionLoweringInfo &FLI) const {
41
42 if (F.arg_empty())
43 return true;
44
45 return false;
46}

48bool RISCVCallLowering::lowerCall(MachineIRBuilder &MIRBuilder,
49 CallLoweringInfo &Info) const {
50 return false;
51}
