20/// \returns true if \p C is a legal immediate operand for an arithmetic
21/// instruction.
22constexpr bool isLegalArithImmed(const uint64_t C) {
23 return (C >> 12 == 0) || ((C & 0xFFFULL) == 0 && C >> 24 == 0);
24}
