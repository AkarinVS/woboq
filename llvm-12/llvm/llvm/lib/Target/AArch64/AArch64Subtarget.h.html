531 bool useSmallAddressing() const {
532 switch (TLInfo.getTargetMachine().getCodeModel()) {
533 case CodeModel::Kernel:
534 // Kernel is currently allowed only for Fuchsia targets,
535 // where it is the same as Small for almost all purposes.
536 case CodeModel::Small:
537 return true;
