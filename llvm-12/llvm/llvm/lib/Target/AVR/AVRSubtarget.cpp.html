30AVRSubtarget::AVRSubtarget(const Triple &TT, const std::string &CPU,
31 const std::string &FS, const AVRTargetMachine &TM)
32 : AVRGenSubtargetInfo(TT, CPU, /*TuneCPU*/ CPU, FS), ELFArch(0),
33
34 // Subtarget features
35 m_hasSRAM(false), m_hasJMPCALL(false), m_hasIJMPCALL(false),
36 m_hasEIJMPCALL(false), m_hasADDSUBIW(false), m_hasSmallStack(false),
37 m_hasMOVW(false), m_hasLPM(false), m_hasLPMX(false), m_hasELPM(false),
38 m_hasELPMX(false), m_hasSPM(false), m_hasSPMX(false), m_hasDES(false),
39 m_supportsRMW(false), m_supportsMultiplication(false), m_hasBREAK(false),
40 m_hasTinyEncoding(false), m_hasMemMappedGPR(false),
41 m_FeatureSetDummy(false),
42
43 InstrInfo(), FrameLowering(),
44 TLInfo(TM, initializeSubtargetDependencies(CPU, FS, TM)), TSInfo() {
45 // Parse features string.
46 ParseSubtargetFeatures(CPU, /*TuneCPU*/ CPU, FS);
47}
48
49AVRSubtarget &
50AVRSubtarget::initializeSubtargetDependencies(StringRef CPU, StringRef FS,
51 const TargetMachine &TM) {
52 // Parse features string.
53 ParseSubtargetFeatures(CPU, /*TuneCPU*/ CPU, FS);
54 return *this;
55}
