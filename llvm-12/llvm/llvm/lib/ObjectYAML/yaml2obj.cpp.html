21bool convertYAML(yaml::Input &YIn, raw_ostream &Out, ErrorHandler ErrHandler,
22 unsigned DocNum, uint64_t MaxSize) {
23 unsigned CurDocNum = 0;
24 do {
25 if (++CurDocNum != DocNum)
26 continue;
27
28 yaml::YamlObjectFile Doc;
29 YIn >> Doc;
30 if (std::error_code EC = YIn.error()) {
31 ErrHandler("failed to parse YAML input: " + EC.message());
32 return false;
33 }
34
35 if (Doc.Arch)
36 return yaml2archive(*Doc.Arch, Out, ErrHandler);
37 if (Doc.Elf)
38 return yaml2elf(*Doc.Elf, Out, ErrHandler, MaxSize);
39 if (Doc.Coff)
40 return yaml2coff(*Doc.Coff, Out, ErrHandler);
41 if (Doc.MachO || Doc.FatMachO)
42 return yaml2macho(Doc, Out, ErrHandler);
43 if (Doc.Minidump)
44 return yaml2minidump(*Doc.Minidump, Out, ErrHandler);
45 if (Doc.Wasm)
46 return yaml2wasm(*Doc.Wasm, Out, ErrHandler);
47
48 ErrHandler("unknown document type");
49 return false;
50
51 } while (YIn.nextDocument());
52
53 ErrHandler("cannot find the " + Twine(DocNum) +
54 getOrdinalSuffix(DocNum).data() + " document");
55 return false;
56}
57
58std::unique_ptr<object::ObjectFile>
59yaml2ObjectFile(SmallVectorImpl<char> &Storage, StringRef Yaml,
60 ErrorHandler ErrHandler) {
61 Storage.clear();
62 raw_svector_ostream OS(Storage);
63
64 yaml::Input YIn(Yaml);
65 if (!convertYAML(YIn, OS, ErrHandler))
66 return {};
67
68 Expected<std::unique_ptr<object::ObjectFile>> ObjOrErr =
69 object::ObjectFile::createObjectFile(
70 MemoryBufferRef(OS.str(), "YamlObject"));
71 if (ObjOrErr)
72 return std::move(*ObjOrErr);
73
74 ErrHandler(toString(ObjOrErr.takeError()));
75 return {};
76}
