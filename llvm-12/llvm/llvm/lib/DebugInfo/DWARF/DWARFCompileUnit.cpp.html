17void DWARFCompileUnit::dump(raw_ostream &OS, DIDumpOptions DumpOpts) {
18 int OffsetDumpWidth = 2 * dwarf::getDwarfOffsetByteSize(getFormat());
19 OS << format("0x%08" PRIx64, getOffset()) << ": Compile Unit:"
20 << " length = " << format("0x%0*" PRIx64, OffsetDumpWidth, getLength())
21 << ", format = " << dwarf::FormatString(getFormat())
22 << ", version = " << format("0x%04x", getVersion());
23 if (getVersion() >= 5)
24 OS << ", unit_type = " << dwarf::UnitTypeString(getUnitType());
25 OS << ", abbr_offset = " << format("0x%04" PRIx64, getAbbrOffset());
26 if (!getAbbreviations())
27 OS << " (invalid)";
28 OS << ", addr_size = " << format("0x%02x", getAddressByteSize());
29 if (getVersion() >= 5 && getUnitType() != dwarf::DW_UT_compile)
30 OS << ", DWO_id = " << format("0x%016" PRIx64, *getDWOId());
31 OS << " (next unit at " << format("0x%08" PRIx64, getNextUnitOffset())
32 << ")\n";
33
34 if (DWARFDie CUDie = getUnitDIE(false))
35 CUDie.dump(OS, 0, DumpOpts);
36 else
37 OS << "<compile unit can't be parsed!>\n\n";
38}
