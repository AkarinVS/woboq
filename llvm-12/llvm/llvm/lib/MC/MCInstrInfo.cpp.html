15bool MCInstrInfo::getDeprecatedInfo(MCInst &MI, const MCSubtargetInfo &STI,
16 std::string &Info) const {
17 unsigned Opcode = MI.getOpcode();
18 if (ComplexDeprecationInfos && ComplexDeprecationInfos[Opcode])
19 return ComplexDeprecationInfos[Opcode](MI, STI, Info);
20 if (DeprecatedFeatures && DeprecatedFeatures[Opcode] != uint8_t(-1U) &&
21 STI.getFeatureBits()[DeprecatedFeatures[Opcode]]) {
22 // FIXME: it would be nice to include the subtarget feature here.
23 Info = "deprecated";
24 return true;
25 }
26 return false;
27}
