9#include "llvm/ExecutionEngine/Orc/ObjectTransformLayer.h"
10#include "llvm/Support/MemoryBuffer.h"
11
12namespace llvm {
13namespace orc {
14
15ObjectTransformLayer::ObjectTransformLayer(ExecutionSession &ES,
16 ObjectLayer &BaseLayer,
17 TransformFunction Transform)
18 : ObjectLayer(ES), BaseLayer(BaseLayer), Transform(std::move(Transform)) {}
19
20void ObjectTransformLayer::emit(
21 std::unique_ptr<MaterializationResponsibility> R,
22 std::unique_ptr<MemoryBuffer> O) {
23 assert(O && "Module must not be null");
24
25 // If there is a transform set then apply it.
26 if (Transform) {
27 if (auto TransformedObj = Transform(std::move(O)))
28 O = std::move(*TransformedObj);
29 else {
30 R->failMaterialization();
31 getExecutionSession().reportError(TransformedObj.takeError());
32 return;
33 }
34 }
35
36 BaseLayer.emit(std::move(R), std::move(O));
37}
