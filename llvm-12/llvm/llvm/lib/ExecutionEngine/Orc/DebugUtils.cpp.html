45// SetPrinter predicate that prints every element.
46template <typename T> struct PrintAll {
47 bool operator()(const T &E) { return true; }
48};
49
50bool anyPrintSymbolOptionSet() {
51#ifndef NDEBUG
52 return PrintHidden || PrintCallable || PrintData;
53#else
54 return false;
55#endif // NDEBUG
56}
57
58bool flagsMatchCLOpts(const JITSymbolFlags &Flags) {
59#ifndef NDEBUG
60 // Bail out early if this is a hidden symbol and we're not printing hiddens.
61 if (!PrintHidden && !Flags.isExported())
62 return false;
63
64 // Return true if this is callable and we're printing callables.
65 if (PrintCallable && Flags.isCallable())
66 return true;
67
68 // Return true if this is data and we're printing data.
69 if (PrintData && !Flags.isCallable())
70 return true;
71
72 // otherwise return false.
73 return false;
74#else
75 return false;
76#endif // NDEBUG
77}

79// Prints a sequence of items, filtered by an user-supplied predicate.
80template <typename Sequence,
81 typename Pred = PrintAll<typename Sequence::value_type>>
82class SequencePrinter {
83public:
84 SequencePrinter(const Sequence &S, char OpenSeq, char CloseSeq,
85 Pred ShouldPrint = Pred())
86 : S(S), OpenSeq(OpenSeq), CloseSeq(CloseSeq),
87 ShouldPrint(std::move(ShouldPrint)) {}
88
89 void printTo(llvm::raw_ostream &OS) const {
90 bool PrintComma = false;
91 OS << OpenSeq;
92 for (auto &E : S) {
93 if (ShouldPrint(E)) {
94 if (PrintComma)
95 OS << ',';
96 OS << ' ' << E;
97 PrintComma = true;
98 }
99 }
100 OS << ' ' << CloseSeq;
101 }
102
103private:
104 const Sequence &S;
105 char OpenSeq;
106 char CloseSeq;
107 mutable Pred ShouldPrint;
108};
109
110template <typename Sequence, typename Pred>
111SequencePrinter<Sequence, Pred> printSequence(const Sequence &S, char OpenSeq,
112 char CloseSeq, Pred P = Pred()) {
113 return SequencePrinter<Sequence, Pred>(S, OpenSeq, CloseSeq, std::move(P));
114}
115
116// Render a SequencePrinter by delegating to its printTo method.
117template <typename Sequence, typename Pred>
118llvm::raw_ostream &operator<<(llvm::raw_ostream &OS,
119 const SequencePrinter<Sequence, Pred> &Printer) {
120 Printer.printTo(OS);
121 return OS;
122}

157raw_ostream &operator<<(raw_ostream &OS, const JITSymbolFlags &Flags) {
158 if (Flags.hasError())
159 OS << "[*ERROR*]";
160 if (Flags.isCallable())
161 OS << "[Callable]";
162 else
163 OS << "[Data]";
164 if (Flags.isWeak())
165 OS << "[Weak]";
166 else if (Flags.isCommon())
167 OS << "[Common]";
168
169 if (!Flags.isExported())
170 OS << "[Hidden]";
171
172 return OS;
173}

256raw_ostream &operator<<(raw_ostream &OS,
257 const JITDylibSearchOrder &SearchOrder) {
258 OS << "[";
259 if (!SearchOrder.empty()) {
260 assert(SearchOrder.front().first &&
261 "JITDylibList entries must not be null");
262 OS << " (\"" << SearchOrder.front().first->getName() << "\", "
263 << SearchOrder.begin()->second << ")";
264 for (auto &KV :
265 make_range(std::next(SearchOrder.begin(), 1), SearchOrder.end())) {
266 assert(KV.first && "JITDylibList entries must not be null");
267 OS << ", (\"" << KV.first->getName() << "\", " << KV.second << ")";
268 }
269 }
270 OS << " ]";
271 return OS;

301DumpObjects::DumpObjects(std::string DumpDir, std::string IdentifierOverride)
302 : DumpDir(std::move(DumpDir)),
303 IdentifierOverride(std::move(IdentifierOverride)) {
304
305 /// Discard any trailing separators.
306 while (!this->DumpDir.empty() &&
307 sys::path::is_separator(this->DumpDir.back()))
308 this->DumpDir.pop_back();
309}
310
311Expected<std::unique_ptr<MemoryBuffer>>
312DumpObjects::operator()(std::unique_ptr<MemoryBuffer> Obj) {
313 size_t Idx = 1;
314
315 std::string DumpPathStem;
316 raw_string_ostream(DumpPathStem)
317 << DumpDir << (DumpDir.empty() ? "" : "/") << getBufferIdentifier(*Obj);
318
319 std::string DumpPath = DumpPathStem + ".o";
320 while (sys::fs::exists(DumpPath)) {
321 DumpPath.clear();
322 raw_string_ostream(DumpPath) << DumpPathStem << "." << (++Idx) << ".o";
323 }
324
325 LLVM_DEBUG({
326 dbgs() << "Dumping object buffer [ " << (const void *)Obj->getBufferStart()
327 << " -- " << (const void *)(Obj->getBufferEnd() - 1) << " ] to "
328 << DumpPath << "\n";
329 });
330
331 std::error_code EC;
332 raw_fd_ostream DumpStream(DumpPath, EC);
333 if (EC)
334 return errorCodeToError(EC);
335 DumpStream.write(Obj->getBufferStart(), Obj->getBufferSize());
336
337 return std::move(Obj);
338}
339
340StringRef DumpObjects::getBufferIdentifier(MemoryBuffer &B) {
341 if (!IdentifierOverride.empty())
342 return IdentifierOverride;
343 StringRef Identifier = B.getBufferIdentifier();
344 Identifier.consume_back(".o");
345 return Identifier;
346}
