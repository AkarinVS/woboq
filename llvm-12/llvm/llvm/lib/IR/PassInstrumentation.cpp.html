32bool isSpecialPass(StringRef PassID, const std::vector<StringRef> &Specials) {
33 size_t Pos = PassID.find('<');
34 StringRef Prefix = PassID;
35 if (Pos != StringRef::npos)
36 Prefix = PassID.substr(0, Pos);
37 return any_of(Specials, [Prefix](StringRef S) { return Prefix.endswith(S); });
38}
