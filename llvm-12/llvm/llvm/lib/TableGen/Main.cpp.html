58static int reportError(const char *ProgName, Twine Msg) {
59 errs() << ProgName << ": " << Msg;
60 errs().flush();
61 return 1;
62}

64/// Create a dependency file for `-d` option.
65///
66/// This functionality is really only for the benefit of the build system.
67/// It is similar to GCC's `-M*` family of options.
68static int createDependencyFile(const TGParser &Parser, const char *argv0) {
69 if (OutputFilename == "-")
70 return reportError(argv0, "the option -d must be used together with -o\n");
71
72 std::error_code EC;
73 ToolOutputFile DepOut(DependFilename, EC, sys::fs::OF_None);
74 if (EC)
75 return reportError(argv0, "error opening " + DependFilename + ":" +
76 EC.message() + "\n");
77 DepOut.os() << OutputFilename << ":";
78 for (const auto &Dep : Parser.getDependencies()) {
79 DepOut.os() << ' ' << Dep;
80 }
81 DepOut.os() << "\n";
82 DepOut.keep();
83 return 0;
84}

86int llvm::TableGenMain(const char *argv0, TableGenMainFn *MainFn) {
87 RecordKeeper Records;
88
89 if (TimePhases)
90 Records.startPhaseTiming();
91
92 // Parse the input file.
93
94 Records.startTimer("Parse, build records");
95 ErrorOr<std::unique_ptr<MemoryBuffer>> FileOrErr =
96 MemoryBuffer::getFileOrSTDIN(InputFilename);
97 if (std::error_code EC = FileOrErr.getError())
98 return reportError(argv0, "Could not open input file '" + InputFilename +
99 "': " + EC.message() + "\n");
100
101 Records.saveInputFilename(InputFilename);
102
103 // Tell SrcMgr about this buffer, which is what TGParser will pick up.
104 SrcMgr.AddNewSourceBuffer(std::move(*FileOrErr), SMLoc());
105
106 // Record the location of the include directory so that the lexer can find
107 // it later.
108 SrcMgr.setIncludeDirs(IncludeDirs);
109
110 TGParser Parser(SrcMgr, MacroNames, Records);
111
112 if (Parser.ParseFile())
113 return 1;
114 Records.stopTimer();
115
116 // Write output to memory.
117 Records.startBackendTimer("Backend overall");
118 std::string OutString;
119 raw_string_ostream Out(OutString);
120 unsigned status = MainFn(Out, Records);
121 Records.stopBackendTimer();
122 if (status)
123 return 1;
124
125 // Always write the depfile, even if the main output hasn't changed.
126 // If it's missing, Ninja considers the output dirty. If this was below
127 // the early exit below and someone deleted the .inc.d file but not the .inc
128 // file, tablegen would never write the depfile.
129 if (!DependFilename.empty()) {
130 if (int Ret = createDependencyFile(Parser, argv0))
131 return Ret;
132 }
133
134 Records.startTimer("Write output");
135 bool WriteFile = true;
136 if (WriteIfChanged) {
137 // Only updates the real output file if there are any differences.
138 // This prevents recompilation of all the files depending on it if there
139 // aren't any.
140 if (auto ExistingOrErr = MemoryBuffer::getFile(OutputFilename))
141 if (std::move(ExistingOrErr.get())->getBuffer() == Out.str())
142 WriteFile = false;
143 }
144 if (WriteFile) {
145 std::error_code EC;
146 ToolOutputFile OutFile(OutputFilename, EC, sys::fs::OF_None);
147 if (EC)
148 return reportError(argv0, "error opening " + OutputFilename + ": " +
149 EC.message() + "\n");
150 OutFile.os() << Out.str();
151 if (ErrorsPrinted == 0)
152 OutFile.keep();
153 }
154
155 Records.stopTimer();
156 Records.stopPhaseTiming();
157
158 if (ErrorsPrinted > 0)
159 return reportError(argv0, Twine(ErrorsPrinted) + " errors.\n");
160 return 0;
161}
