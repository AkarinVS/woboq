59std::string SDNode::getOperationName(const SelectionDAG *G) const {
60 switch (getOpcode()) {
61 default:
62 if (getOpcode() < ISD::BUILTIN_OP_END)
63 return "<<Unknown DAG Node>>";
64 if (isMachineOpcode()) {
65 if (G)
66 if (const TargetInstrInfo *TII = G->getSubtarget().getInstrInfo())
67 if (getMachineOpcode() < TII->getNumOpcodes())
68 return std::string(TII->getName(getMachineOpcode()));
69 return "<<Unknown Machine Node #" + utostr(getOpcode()) + ">>";
70 }
71 if (G) {
72 const TargetLowering &TLI = G->getTargetLoweringInfo();
73 const char *Name = TLI.getTargetNodeName(getOpcode());
74 if (Name) return Name;
75 return "<<Unknown Target Node #" + utostr(getOpcode()) + ">>";
76 }
77 return "<<Unknown Node #" + utostr(getOpcode()) + ">>";

867/// Return true if this node is so simple that we should just print it inline
868/// if it appears as an operand.
869static bool shouldPrintInline(const SDNode &Node, const SelectionDAG *G) {
870 // Avoid lots of cluttering when inline printing nodes with associated
871 // DbgValues in verbose mode.
872 if (VerboseDAGDumping && G && !G->GetDbgValues(&Node).empty())
873 return false;
874 if (Node.getOpcode() == ISD::EntryToken)
875 return false;
876 return Node.getNumOperands() == 0;
877}

927static bool printOperand(raw_ostream &OS, const SelectionDAG *G,
928 const SDValue Value) {
929 if (!Value.getNode()) {
930 OS << "<null>";
931 return false;
932 } else if (shouldPrintInline(*Value.getNode(), G)) {
933 OS << Value->getOperationName(G) << ':';
934 Value->print_types(OS, G);
935 Value->print_details(OS, G);
936 return true;
937 } else {
938 OS << PrintNodeId(*Value.getNode());
939 if (unsigned RN = Value.getResNo())
940 OS << ':' << RN;
941 return false;
942 }
943}

946using VisitedSDNodeSet = SmallPtrSet<const SDNode *, 32>;
947
948static void DumpNodesr(raw_ostream &OS, const SDNode *N, unsigned indent,
949 const SelectionDAG *G, VisitedSDNodeSet &once) {
950 if (!once.insert(N).second) // If we've been here before, return now.
951 return;
952
953 // Dump the current SDNode, but don't end the line yet.
954 OS.indent(indent);
955 N->printr(OS, G);
956
957 // Having printed this SDNode, walk the children:
958 for (unsigned i = 0, e = N->getNumOperands(); i != e; ++i) {
959 if (i) OS << ",";
960 OS << " ";
961
962 const SDValue Op = N->getOperand(i);
963 bool printedInline = printOperand(OS, G, Op);
964 if (printedInline)
965 once.insert(Op.getNode());
966 }
967
968 OS << "\n";
969
970 // Dump children that have grandchildren on their own line(s).
971 for (const SDValue &Op : N->op_values())
972 DumpNodesr(OS, Op.getNode(), indent+2, G, once);
973}

986static void printrWithDepthHelper(raw_ostream &OS, const SDNode *N,
987 const SelectionDAG *G, unsigned depth,
988 unsigned indent) {
989 if (depth == 0)
990 return;
991
992 OS.indent(indent);
993
994 N->print(OS, G);
995
996 if (depth < 1)
997 return;
998
999 for (const SDValue &Op : N->op_values()) {
1000 // Don't follow chain operands.
1001 if (Op.getValueType() == MVT::Other)
1002 continue;
1003 OS << '\n';
1004 printrWithDepthHelper(OS, Op.getNode(), G, depth-1, indent+2);
1005 }
