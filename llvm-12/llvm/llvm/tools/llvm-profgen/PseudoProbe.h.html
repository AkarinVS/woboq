27enum PseudoProbeAttributes { TAILCALL = 1, DANGLING = 2 };
28
29// Use func GUID and index as the location info of the inline site
30using InlineSite = std::tuple<uint64_t, uint32_t>;
31
32struct PseudoProbe;
33
34// Tree node to represent the inline relation and its inline site, we use a
35// dummy root in the PseudoProbeDecoder to lead the tree, the outlined
36// function will directly be the children of the dummy root. For the inlined
37// function, all the inlinee will be connected to its inlineer, then further to
38// its outlined function. Pseudo probes originating from the function stores the
39// tree's leaf node which we can process backwards to get its inline context
40class PseudoProbeInlineTree {
41 std::vector<PseudoProbe *> ProbeVector;
42
43 struct InlineSiteHash {
44 uint64_t operator()(const InlineSite &Site) const {
45 return std::get<0>(Site) ^ std::get<1>(Site);
46 }
47 };
48 std::unordered_map<InlineSite, std::unique_ptr<PseudoProbeInlineTree>,
49 InlineSiteHash>
50 Children;
51
52public:
53 // Inlinee function GUID
54 uint64_t GUID = 0;
55 // Inline site to indicate the location in its inliner. As the node could also
56 // be an outlined function, it will use a dummy InlineSite whose GUID and
57 // Index is 0 connected to the dummy root
58 InlineSite ISite;
59 // Used for decoding
60 uint32_t ChildrenToProcess = 0;
61 // Caller node of the inline site
62 PseudoProbeInlineTree *Parent;
63
64 PseudoProbeInlineTree(){};
65 PseudoProbeInlineTree(const InlineSite &Site) : ISite(Site){};
66
67 PseudoProbeInlineTree *getOrAddNode(const InlineSite &Site) {
68 auto Ret =
69 Children.emplace(Site, std::make_unique<PseudoProbeInlineTree>(Site));
70 Ret.first->second->Parent = this;
71 return Ret.first->second.get();
72 }
73
74 void addProbes(PseudoProbe *Probe) { ProbeVector.push_back(Probe); }
75 // Return false if it's a dummy inline site
76 bool hasInlineSite() const { return std::get<0>(ISite) != 0; }
77};

79// Function descriptor decoded from .pseudo_probe_desc section
80struct PseudoProbeFuncDesc {
81 uint64_t FuncGUID = 0;
82 uint64_t FuncHash = 0;
83 std::string FuncName;
84
85 PseudoProbeFuncDesc(uint64_t GUID, uint64_t Hash, StringRef Name)
86 : FuncGUID(GUID), FuncHash(Hash), FuncName(Name){};
87
88 void print(raw_ostream &OS);
89};
90
91// GUID to PseudoProbeFuncDesc map
92using GUIDProbeFunctionMap = std::unordered_map<uint64_t, PseudoProbeFuncDesc>;
93// Address to pseudo probes map.
94using AddressProbesMap = std::unordered_map<uint64_t, std::vector<PseudoProbe>>;
95
96/*
97A pseudo probe has the format like below:
98 INDEX (ULEB128)
99 TYPE (uint4)
100 0 - block probe, 1 - indirect call, 2 - direct call
101 ATTRIBUTE (uint3)
102 1 - tail call, 2 - dangling
103 ADDRESS_TYPE (uint1)
104 0 - code address, 1 - address delta
105 CODE_ADDRESS (uint64 or ULEB128)
106 code address or address delta, depending on Flag
107*/
108struct PseudoProbe {
109 uint64_t Address;
110 uint64_t GUID;
111 uint32_t Index;
112 PseudoProbeType Type;
113 uint8_t Attribute;
114 PseudoProbeInlineTree *InlineTree;
115 const static uint32_t PseudoProbeFirstId =
116 static_cast<uint32_t>(PseudoProbeReservedId::Last) + 1;
117
118 PseudoProbe(uint64_t Ad, uint64_t G, uint32_t I, PseudoProbeType K,
119 uint8_t At, PseudoProbeInlineTree *Tree)
120 : Address(Ad), GUID(G), Index(I), Type(K), Attribute(At),
121 InlineTree(Tree){};
122
123 bool isEntry() const { return Index == PseudoProbeFirstId; }
124
125 bool isDangling() const {
126 return Attribute & static_cast<uint8_t>(PseudoProbeAttributes::DANGLING);
127 }
128
129 bool isTailCall() const {
130 return Attribute & static_cast<uint8_t>(PseudoProbeAttributes::TAILCALL);
131 }
132
133 bool isBlock() const { return Type == PseudoProbeType::Block; }
134 bool isIndirectCall() const { return Type == PseudoProbeType::IndirectCall; }
135 bool isDirectCall() const { return Type == PseudoProbeType::DirectCall; }
136 bool isCall() const { return isIndirectCall() || isDirectCall(); }
137
138 // Get the inlined context by traversing current inline tree backwards,
139 // each tree node has its InlineSite which is taken as the context.
140 // \p ContextStack is populated in root to leaf order
141 void getInlineContext(SmallVectorImpl<std::string> &ContextStack,
142 const GUIDProbeFunctionMap &GUID2FuncMAP,
143 bool ShowName) const;
144 // Helper function to get the string from context stack
145 std::string getInlineContextStr(const GUIDProbeFunctionMap &GUID2FuncMAP,
146 bool ShowName) const;
147 // Print pseudo probe while disassembling
148 void print(raw_ostream &OS, const GUIDProbeFunctionMap &GUID2FuncMAP,
149 bool ShowName);
150};

152/*
153Decode pseudo probe info from ELF section, used along with ELF reader
154Two sections are decoded here:
155 1) \fn buildGUID2FunctionMap is responsible for .pseudo_probe_desc
156 section which encodes all function descriptors.
157 2) \fn buildAddress2ProbeMap is responsible for .pseudoprobe section
158 which encodes an inline function forest and each tree includes its
159 inlined function and all pseudo probes inside the function.
160see \file MCPseudoProbe.h for the details of the section encoding format.
161*/
162class PseudoProbeDecoder {
163 // GUID to PseudoProbeFuncDesc map.
164 GUIDProbeFunctionMap GUID2FuncDescMap;
165
166 // Address to probes map.
167 AddressProbesMap Address2ProbesMap;
168
169 // The dummy root of the inline trie, all the outlined function will directly
170 // be the children of the dummy root, all the inlined function will be the
171 // children of its inlineer. So the relation would be like:
172 // DummyRoot --> OutlinedFunc --> InlinedFunc1 --> InlinedFunc2
173 PseudoProbeInlineTree DummyInlineRoot;
174
175 /// Points to the current location in the buffer.
176 const uint8_t *Data = nullptr;
177
178 /// Points to the end of the buffer.
179 const uint8_t *End = nullptr;
180
181 /// SectionName used for debug
182 std::string SectionName;
183
184 // Decoding helper function
185 template <typename T> T readUnencodedNumber();
186 template <typename T> T readUnsignedNumber();
187 template <typename T> T readSignedNumber();
188 StringRef readString(uint32_t Size);
189
190public:
191 // Decode pseudo_probe_desc section to build GUID to PseudoProbeFuncDesc map.
192 void buildGUID2FuncDescMap(const uint8_t *Start, std::size_t Size);
193
194 // Decode pseudo_probe section to build address to probes map.
195 void buildAddress2ProbeMap(const uint8_t *Start, std::size_t Size);
196
197 // Print pseudo_probe_desc section info
198 void printGUID2FuncDescMap(raw_ostream &OS);
199
200 // Print pseudo_probe section info, used along with show-disassembly
201 void printProbeForAddress(raw_ostream &OS, uint64_t Address);
202
203 // Look up the probe of a call for the input address
204 const PseudoProbe *getCallProbeForAddr(uint64_t Address) const;
205
206 const PseudoProbeFuncDesc *getFuncDescForGUID(uint64_t GUID) const;
207
208 // Helper function to populate one probe's inline stack into
209 // \p InlineContextStack.
210 // Current leaf location info will be added if IncludeLeaf is true
211 // Example:
212 // Current probe(bar:3) inlined at foo:2 then inlined at main:1
213 // IncludeLeaf = true, Output: [main:1, foo:2, bar:3]
214 // IncludeLeaf = false, Output: [main:1, foo:2]
215 void
216 getInlineContextForProbe(const PseudoProbe *Probe,
217 SmallVectorImpl<std::string> &InlineContextStack,
218 bool IncludeLeaf) const;
219
220 const AddressProbesMap &getAddress2ProbesMap() const {
221 return Address2ProbesMap;
222 }
223
224 const PseudoProbeFuncDesc *
225 getInlinerDescForProbe(const PseudoProbe *Probe) const;
226};
