59class CSProfileGenerator : public ProfileGenerator {
60protected:
61 const BinarySampleCounterMap &BinarySampleCounters;
62
63public:
64 CSProfileGenerator(const BinarySampleCounterMap &Counters)
65 : BinarySampleCounters(Counters){};
66
67public:
68 void generateProfile() override {
69 for (const auto &BI : BinarySampleCounters) {
70 ProfiledBinary *Binary = BI.first;
71 for (const auto &CI : BI.second) {
72 const StringBasedCtxKey *CtxKey =
73 dyn_cast<StringBasedCtxKey>(CI.first.getPtr());
74 StringRef ContextId(CtxKey->Context);
75 // Get or create function profile for the range
76 FunctionSamples &FunctionProfile =
77 getFunctionProfileForContext(ContextId);
78
79 // Fill in function body samples
80 populateFunctionBodySamples(FunctionProfile, CI.second.RangeCounter,
81 Binary);
82 // Fill in boundary sample counts as well as call site samples for calls
83 populateFunctionBoundarySamples(ContextId, FunctionProfile,
84 CI.second.BranchCounter, Binary);
85 }
86 }
87 // Fill in call site value sample for inlined calls and also use context to
88 // infer missing samples. Since we don't have call count for inlined
89 // functions, we estimate it from inlinee's profile using the entry of the
90 // body sample.
91 populateInferredFunctionSamples();
92 }

94 // Remove adjacent repeated context sequences up to a given sequence length,
95 // -1 means no size limit. Note that repeated sequences are identified based
96 // on the exact call site, this is finer granularity than function recursion.
97 template <typename T>
98 static void compressRecursionContext(SmallVectorImpl<T> &Context,
99 int32_t CSize = MaxCompressionSize) {
100 uint32_t I = 1;
101 uint32_t HS = static_cast<uint32_t>(Context.size() / 2);
102 uint32_t MaxDedupSize =
103 CSize == -1 ? HS : std::min(static_cast<uint32_t>(CSize), HS);
104 auto BeginIter = Context.begin();
105 // Use an in-place algorithm to save memory copy
106 // End indicates the end location of current iteration's data
107 uint32_t End = 0;
108 // Deduplicate from length 1 to the max possible size of a repeated
109 // sequence.
110 while (I <= MaxDedupSize) {
111 // This is a linear algorithm that deduplicates adjacent repeated
112 // sequences of size I. The deduplication detection runs on a sliding
113 // window whose size is 2*I and it keeps sliding the window to deduplicate
114 // the data inside. Once duplication is detected, deduplicate it by
115 // skipping the right half part of the window, otherwise just copy back
116 // the new one by appending them at the back of End pointer(for the next
117 // iteration).
118 //
119 // For example:
120 // Input: [a1, a2, b1, b2]
121 // (Added index to distinguish the same char, the origin is [a, a, b,
122 // b], the size of the dedup window is 2(I = 1) at the beginning)
123 //
124 // 1) The initial status is a dummy window[null, a1], then just copy the
125 // right half of the window(End = 0), then slide the window.
126 // Result: [a1], a2, b1, b2 (End points to the element right before ],
127 // after ] is the data of the previous iteration)
128 //
129 // 2) Next window is [a1, a2]. Since a1 == a2, then skip the right half of
130 // the window i.e the duplication happen. Only slide the window.
131 // Result: [a1], a2, b1, b2
132 //
133 // 3) Next window is [a2, b1], copy the right half of the window(b1 is
134 // new) to the End and slide the window.
135 // Result: [a1, b1], b1, b2
136 //
137 // 4) Next window is [b1, b2], same to 2), skip b2.
138 // Result: [a1, b1], b1, b2
139 // After resize, it will be [a, b]
140
141 // Use pointers like below to do comparison inside the window
142 // [a b c a b c]
143 // | | | | |
144 // LeftBoundary Left Right Left+I Right+I
145 // A duplication found if Left < LeftBoundry.
146
147 int32_t Right = I - 1;
148 End = I;
149 int32_t LeftBoundary = 0;
150 while (Right + I < Context.size()) {
151 // To avoids scanning a part of a sequence repeatedly, it finds out
152 // the common suffix of two hald in the window. The common suffix will
153 // serve as the common prefix of next possible pair of duplicate
154 // sequences. The non-common part will be ignored and never scanned
155 // again.
156
157 // For example.
158 // Input: [a, b1], c1, b2, c2
159 // I = 2
160 //
161 // 1) For the window [a, b1, c1, b2], non-common-suffix for the right
162 // part is 'c1', copy it and only slide the window 1 step.
163 // Result: [a, b1, c1], b2, c2
164 //
165 // 2) Next window is [b1, c1, b2, c2], so duplication happen.
166 // Result after resize: [a, b, c]
167
168 int32_t Left = Right;
169 while (Left >= LeftBoundary && Context[Left] == Context[Left + I]) {
170 // Find the longest suffix inside the window. When stops, Left points
171 // at the diverging point in the current sequence.
172 Left--;
173 }
174
175 bool DuplicationFound = (Left < LeftBoundary);
176 // Don't need to recheck the data before Right
177 LeftBoundary = Right + 1;
178 if (DuplicationFound) {
179 // Duplication found, skip right half of the window.
180 Right += I;
181 } else {
182 // Copy the non-common-suffix part of the adjacent sequence.
183 std::copy(BeginIter + Right + 1, BeginIter + Left + I + 1,
184 BeginIter + End);
185 End += Left + I - Right;
186 // Only slide the window by the size of non-common-suffix
187 Right = Left + I;
188 }
189 }
190 // Don't forget the remaining part that's not scanned.
191 std::copy(BeginIter + Right + 1, Context.end(), BeginIter + End);
192 End += Context.size() - Right - 1;
193 I++;
194 Context.resize(End);
195 MaxDedupSize = std::min(static_cast<uint32_t>(End / 2), MaxDedupSize);
196 }
197 }

223public:

225 // length. -1 means no size limit.
226 static int32_t MaxCompressionSize;
227};
