9#ifndef LLVM_TOOLS_LLVM_PROFGEN_ERRORHANDLING_H
10#define LLVM_TOOLS_LLVM_PROFGEN_ERRORHANDLING_H

12#include "llvm/ADT/Twine.h"
13#include "llvm/Support/Errc.h"
14#include "llvm/Support/Error.h"
15#include "llvm/Support/ErrorOr.h"
16#include "llvm/Support/WithColor.h"
17#include <system_error>
18
19using namespace llvm;
20
21LLVM_ATTRIBUTE_NORETURN inline void
22exitWithError(const Twine &Message, StringRef Whence = StringRef(),
23 StringRef Hint = StringRef()) {
24 WithColor::error(errs(), "llvm-profgen");
25 if (!Whence.empty())
26 errs() << Whence.str() << ": ";
27 errs() << Message << "\n";
28 if (!Hint.empty())
29 WithColor::note() << Hint.str() << "\n";
30 ::exit(EXIT_FAILURE);
31}
32
33LLVM_ATTRIBUTE_NORETURN inline void
34exitWithError(std::error_code EC, StringRef Whence = StringRef()) {
35 exitWithError(EC.message(), Whence);
36}
37
38LLVM_ATTRIBUTE_NORETURN inline void exitWithError(Error E, StringRef Whence) {
39 exitWithError(errorToErrorCode(std::move(E)), Whence);
40}
41
42template <typename T, typename... Ts>
43T unwrapOrError(Expected<T> EO, Ts &&... Args) {
44 if (EO)
45 return std::move(*EO);
46 exitWithError(EO.takeError(), std::forward<Ts>(Args)...);
47}
