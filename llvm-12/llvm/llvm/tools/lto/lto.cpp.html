71struct LTOToolDiagnosticHandler : public DiagnosticHandler {
72 bool handleDiagnostics(const DiagnosticInfo &DI) override {
73 if (DI.getSeverity() != DS_Error) {
74 DiagnosticPrinterRawOStream DP(errs());
75 DI.print(DP);
76 errs() << '\n';
77 return true;
78 }
79 sLastErrorString = "";
80 {
81 raw_string_ostream Stream(sLastErrorString);
82 DiagnosticPrinterRawOStream DP(Stream);
83 DI.print(DP);
84 }
85 return true;
86 }
87};

89// Initialize the configured targets if they have not been initialized.
90static void lto_initialize() {
91 if (!initialized) {
92#ifdef _WIN32
93 // Dialog box on crash disabling doesn't work across DLL boundaries, so do
94 // it here.
95 llvm::sys::DisableSystemDialogsOnCrash();
96#endif
97
98 InitializeAllTargetInfos();
99 InitializeAllTargets();
100 InitializeAllTargetMCs();
101 InitializeAllAsmParsers();
102 InitializeAllAsmPrinters();
103 InitializeAllDisassemblers();
104
105 static LLVMContext Context;
106 LTOContext = &Context;
107 LTOContext->setDiagnosticHandler(
108 std::make_unique<LTOToolDiagnosticHandler>(), true);
109 initialized = true;
110 }
111}

120// This derived class owns the native object file. This helps implement the
121// libLTO API semantics, which require that the code generator owns the object
122// file.
123struct LibLTOCodeGenerator : LTOCodeGenerator {
124 LibLTOCodeGenerator() : LTOCodeGenerator(*LTOContext) { init(); }
125 LibLTOCodeGenerator(std::unique_ptr<LLVMContext> Context)
126 : LTOCodeGenerator(*Context), OwnedContext(std::move(Context)) {
127 init();
128 }
129
130 // Reset the module first in case MergedModule is created in OwnedContext.
131 // Module must be destructed before its context gets destructed.
132 ~LibLTOCodeGenerator() { resetMergedModule(); }
133
134 void init() { setDiagnosticHandler(handleLibLTODiagnostic, nullptr); }
135
136 std::unique_ptr<MemoryBuffer> NativeObjectFile;
137 std::unique_ptr<LLVMContext> OwnedContext;
138};

146// Convert the subtarget features into a string to pass to LTOCodeGenerator.
147static void lto_add_attrs(lto_code_gen_t cg) {
148 LTOCodeGenerator *CG = unwrap(cg);
149 CG->setAttrs(codegen::getMAttrs());
150
151 if (OptLevel < '0' || OptLevel > '3')
152 report_fatal_error("Optimization level must be between 0 and 3");
153 CG->setOptLevel(OptLevel - '0');
154 CG->setFreestanding(EnableFreestanding);
155 CG->setDisableVerify(DisableVerify);
156}

265 const char *path) {

269
270 // Create a local context. Ownership will be transferred to LTOModule.
271 std::unique_ptr<LLVMContext> Context = std::make_unique<LLVMContext>();
272 Context->setDiagnosticHandler(std::make_unique<LTOToolDiagnosticHandler>(),
273 true);

276 std::move(Context), mem, length, Options, StringRef(path));
277 if (!M)
278 return nullptr;
279 return wrap(M->release());
280}

321lto_bool_t lto_module_get_macho_cputype(lto_module_t mod,
322 unsigned int *out_cputype,
323 unsigned int *out_cpusubtype) {
324 LTOModule *M = unwrap(mod);
325 Expected<uint32_t> CPUType = M->getMachOCPUType();
326 if (!CPUType) {
327 sLastErrorString = toString(CPUType.takeError());
328 return true;
329 }
330 *out_cputype = *CPUType;
331
332 Expected<uint32_t> CPUSubType = M->getMachOCPUSubType();
333 if (!CPUSubType) {
334 sLastErrorString = toString(CPUSubType.takeError());
335 return true;
336 }
337 *out_cpusubtype = *CPUSubType;
338
339 return false;
340}

494thinlto_code_gen_t thinlto_create_codegen(void) {
495 lto_initialize();
496 ThinLTOCodeGenerator *CodeGen = new ThinLTOCodeGenerator();
497 CodeGen->setTargetOptions(
498 codegen::InitTargetOptionsFromCodeGenFlags(Triple()));
499 CodeGen->setFreestanding(EnableFreestanding);
500
501 if (OptLevel.getNumOccurrences()) {
502 if (OptLevel < '0' || OptLevel > '3')
503 report_fatal_error("Optimization level must be between 0 and 3");
504 CodeGen->setOptLevel(OptLevel - '0');
505 switch (OptLevel) {
506 case '0':
507 CodeGen->setCodeGenOptLevel(CodeGenOpt::None);
508 break;
509 case '1':
510 CodeGen->setCodeGenOptLevel(CodeGenOpt::Less);
511 break;
512 case '2':
513 CodeGen->setCodeGenOptLevel(CodeGenOpt::Default);
514 break;
515 case '3':
516 CodeGen->setCodeGenOptLevel(CodeGenOpt::Aggressive);
517 break;
518 }
519 }
520 return wrap(CodeGen);
521}

645 unwrap(cg)->setCodePICModel(Reloc::PIC_);
646 return false;
647 case LTO_CODEGEN_PIC_MODEL_DYNAMIC_NO_PIC:
648 unwrap(cg)->setCodePICModel(Reloc::DynamicNoPIC);
649 return false;
650 case LTO_CODEGEN_PIC_MODEL_DEFAULT:
651 unwrap(cg)->setCodePICModel(None);
652 return false;
653 }
654 sLastErrorString = "Unknown PIC model";
655 return true;
656}
