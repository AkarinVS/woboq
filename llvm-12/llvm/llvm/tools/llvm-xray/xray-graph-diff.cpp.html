213Expected<GraphDiffRenderer> GraphDiffRenderer::Factory::getGraphDiffRenderer() {
214 GraphDiffRenderer R;
215
216 for (int i = 0; i < N; ++i) {
217 const auto &G = this->G[i].get();
218 for (const auto &V : G.vertices()) {
219 const auto &VAttr = V.second;
220 R.G[VAttr.SymbolName].CorrVertexPtr[i] = &V;
221 }
222 for (const auto &E : G.edges()) {
223 auto &EdgeTailID = E.first.first;
224 auto &EdgeHeadID = E.first.second;
225 auto EdgeTailAttrOrErr = G.at(EdgeTailID);
226 auto EdgeHeadAttrOrErr = G.at(EdgeHeadID);
227 if (!EdgeTailAttrOrErr)
228 return EdgeTailAttrOrErr.takeError();
229 if (!EdgeHeadAttrOrErr)
230 return EdgeHeadAttrOrErr.takeError();
231 GraphT::EdgeIdentifier ID{EdgeTailAttrOrErr->SymbolName,
232 EdgeHeadAttrOrErr->SymbolName};
233 R.G[ID].CorrEdgePtr[i] = &E;
234 }
235 }
236
237 return R;
238}
239// Returns the Relative change With respect to LeftStat between LeftStat
240// and RightStat.
241static double statRelDiff(const GraphDiffRenderer::TimeStat &LeftStat,
242 const GraphDiffRenderer::TimeStat &RightStat,
243 GraphDiffRenderer::StatType T) {
244 double LeftAttr = LeftStat.getDouble(T);
245 double RightAttr = RightStat.getDouble(T);
246
247 return RightAttr / LeftAttr - 1.0;
248}

250static std::string getColor(const GraphDiffRenderer::GraphT::EdgeValueType &E,
251 const GraphDiffRenderer::GraphT &G, ColorHelper H,
252 GraphDiffRenderer::StatType T) {
253 auto &EdgeAttr = E.second;
254 if (EdgeAttr.CorrEdgePtr[0] == nullptr)
255 return H.getColorString(2.0); // A number greater than 1.0
256 if (EdgeAttr.CorrEdgePtr[1] == nullptr)
257 return H.getColorString(-2.0); // A number less than -1.0
258
259 if (T == GraphDiffRenderer::StatType::NONE)
260 return H.getDefaultColorString();
261
262 const auto &LeftStat = EdgeAttr.CorrEdgePtr[0]->second.S;
263 const auto &RightStat = EdgeAttr.CorrEdgePtr[1]->second.S;
264
265 double RelDiff = statRelDiff(LeftStat, RightStat, T);
266 double CappedRelDiff = std::min(1.0, std::max(-1.0, RelDiff));
267
268 return H.getColorString(CappedRelDiff);
269}

292static Twine truncateString(const StringRef &S, size_t n) {
293 return (S.size() > n) ? Twine(S.substr(0, n)) + "..." : Twine(S);
294}

296template <typename T> static bool containsNullptr(const T &Collection) {
297 for (const auto &E : Collection)
298 if (E == nullptr)
299 return true;
300 return false;
301}

303static std::string getLabel(const GraphDiffRenderer::GraphT::EdgeValueType &E,
304 GraphDiffRenderer::StatType EL) {
305 auto &EdgeAttr = E.second;
306 switch (EL) {
307 case GraphDiffRenderer::StatType::NONE:
308 return "";
309 default:
310 if (containsNullptr(EdgeAttr.CorrEdgePtr))
311 return "";
312
313 const auto &LeftStat = EdgeAttr.CorrEdgePtr[0]->second.S;
314 const auto &RightStat = EdgeAttr.CorrEdgePtr[1]->second.S;
315
316 double RelDiff = statRelDiff(LeftStat, RightStat, EL);
317 return std::string(formatv(R"({0:P})", RelDiff));
318 }
319}
320
321static std::string getLabel(const GraphDiffRenderer::GraphT::VertexValueType &V,
322 GraphDiffRenderer::StatType VL, int TrunLen) {
323 const auto &VertexId = V.first;
324 const auto &VertexAttr = V.second;
325 switch (VL) {
326 case GraphDiffRenderer::StatType::NONE:
327 return std::string(
328 formatv(R"({0})", truncateString(VertexId, TrunLen).str()));
329 default:
330 if (containsNullptr(VertexAttr.CorrVertexPtr))
331 return std::string(
332 formatv(R"({0})", truncateString(VertexId, TrunLen).str()));
333
334 const auto &LeftStat = VertexAttr.CorrVertexPtr[0]->second.S;
335 const auto &RightStat = VertexAttr.CorrVertexPtr[1]->second.S;
336
337 double RelDiff = statRelDiff(LeftStat, RightStat, VL);
338 return std::string(formatv(
339 R"({{{0}|{1:P}})", truncateString(VertexId, TrunLen).str(), RelDiff));
340 }
341}

343static double getLineWidth(const GraphDiffRenderer::GraphT::EdgeValueType &E,
344 GraphDiffRenderer::StatType EL) {
345 auto &EdgeAttr = E.second;
346 switch (EL) {
347 case GraphDiffRenderer::StatType::NONE:
348 return 1.0;
349 default:
350 if (containsNullptr(EdgeAttr.CorrEdgePtr))
351 return 1.0;
352
353 const auto &LeftStat = EdgeAttr.CorrEdgePtr[0]->second.S;
354 const auto &RightStat = EdgeAttr.CorrEdgePtr[1]->second.S;
355
356 double RelDiff = statRelDiff(LeftStat, RightStat, EL);
357 return (RelDiff > 1.0) ? RelDiff : 1.0;
358 }

361void GraphDiffRenderer::exportGraphAsDOT(raw_ostream &OS, StatType EdgeLabel,
362 StatType EdgeColor,
363 StatType VertexLabel,
364 StatType VertexColor, int TruncLen) {
365 // Get numbering of vertices for dot output.
366 StringMap<int32_t> VertexNo;
367
368 int i = 0;
369 for (const auto &V : G.vertices()) {
370 VertexNo[V.first] = i++;
371 }
372
373 ColorHelper H(ColorHelper::DivergingScheme::PiYG);
374
375 OS << "digraph xrayDiff {\n";
376
377 if (VertexLabel != StatType::NONE)
378 OS << "node [shape=record]\n";
379
380 for (const auto &E : G.edges()) {
381 const auto &HeadId = E.first.first;
382 const auto &TailId = E.first.second;
383 OS << formatv(R"(F{0} -> F{1} [tooltip="{2} -> {3}" label="{4}" )"
384 R"(color="{5}" labelfontcolor="{5}" penwidth={6}])"
385 "\n",
386 VertexNo[HeadId], VertexNo[TailId],
387 (HeadId.equals("")) ? static_cast<StringRef>("F0") : HeadId,
388 TailId, getLabel(E, EdgeLabel), getColor(E, G, H, EdgeColor),
389 getLineWidth(E, EdgeColor));
390 }
391
392 for (const auto &V : G.vertices()) {
393 const auto &VertexId = V.first;
394 if (VertexId.equals("")) {
395 OS << formatv(R"(F{0} [label="F0"])"
396 "\n",
397 VertexNo[VertexId]);
398 continue;
399 }
400 OS << formatv(R"(F{0} [label="{1}" color="{2}"])"
401 "\n",
402 VertexNo[VertexId], getLabel(V, VertexLabel, TruncLen),
403 getColor(V, G, H, VertexColor));
404 }
405
406 OS << "}\n";
407}

409template <typename T> static T &ifSpecified(T &A, cl::alias &AA, T &B) {
410 if (A.getPosition() == 0 && AA.getPosition() == 0)
411 return B;
412
413 return A;
414}

416static CommandRegistration Unused(&GraphDiff, []() -> Error {
417 std::array<GraphRenderer::Factory, 2> Factories{
418 {{ifSpecified(GraphDiffKeepGoing1, GraphDiffKeepGoing1A,
419 GraphDiffKeepGoing),
420 ifSpecified(GraphDiffDeduceSiblingCalls1, GraphDiffDeduceSiblingCalls1A,
421 GraphDiffDeduceSiblingCalls),
422 ifSpecified(GraphDiffInstrMap1, GraphDiffInstrMap1A, GraphDiffInstrMap),
423 Trace()},
424 {ifSpecified(GraphDiffKeepGoing2, GraphDiffKeepGoing2A,
425 GraphDiffKeepGoing),
426 ifSpecified(GraphDiffDeduceSiblingCalls2, GraphDiffDeduceSiblingCalls2A,
427 GraphDiffDeduceSiblingCalls),
428 ifSpecified(GraphDiffInstrMap2, GraphDiffInstrMap2A, GraphDiffInstrMap),
429 Trace()}}};
430
431 std::array<std::string, 2> Inputs{{GraphDiffInput1, GraphDiffInput2}};
432
433 std::array<GraphRenderer::GraphT, 2> Graphs;
434
435 for (int i = 0; i < 2; i++) {
436 auto TraceOrErr = loadTraceFile(Inputs[i], true);
437 if (!TraceOrErr)
438 return make_error<StringError>(
439 Twine("Failed Loading Input File '") + Inputs[i] + "'",
440 make_error_code(llvm::errc::invalid_argument));
441 Factories[i].Trace = std::move(*TraceOrErr);
442
443 auto GraphRendererOrErr = Factories[i].getGraphRenderer();
444
445 if (!GraphRendererOrErr)
446 return GraphRendererOrErr.takeError();
447
448 auto GraphRenderer = *GraphRendererOrErr;
449
450 Graphs[i] = GraphRenderer.getGraph();
451 }
452
453 GraphDiffRenderer::Factory DGF(Graphs[0], Graphs[1]);
454
455 auto GDROrErr = DGF.getGraphDiffRenderer();
456 if (!GDROrErr)
457 return GDROrErr.takeError();
458
459 auto &GDR = *GDROrErr;
460
461 std::error_code EC;
462 raw_fd_ostream OS(GraphDiffOutput, EC, sys::fs::OpenFlags::OF_Text);
463 if (EC)
464 return make_error<StringError>(
465 Twine("Cannot open file '") + GraphDiffOutput + "' for writing.", EC);
466
467 GDR.exportGraphAsDOT(OS, GraphDiffEdgeLabel, GraphDiffEdgeColor,
468 GraphDiffVertexLabel, GraphDiffVertexColor,
469 GraphDiffVertexLabelTrunc);
470
471 return Error::success();
472});
