195static Error applyHarnessPromotions(Session &S, LinkGraph &G) {
196
197 // If this graph is part of the test harness there's nothing to do.
198 if (S.HarnessFiles.empty() || S.HarnessFiles.count(G.getName()))
199 return Error::success();
200
201 LLVM_DEBUG(dbgs() << "Appling promotions to graph " << G.getName() << "\n");
202
203 // If this graph is part of the test then promote any symbols referenced by
204 // the harness to default scope, remove all symbols that clash with harness
205 // definitions.
206 std::vector<Symbol *> DefinitionsToRemove;
207 for (auto *Sym : G.defined_symbols()) {
208
209 if (!Sym->hasName())
210 continue;
211
212 if (Sym->getLinkage() == Linkage::Weak) {
213 if (!S.CanonicalWeakDefs.count(Sym->getName()) ||
214 S.CanonicalWeakDefs[Sym->getName()] != G.getName()) {
215 LLVM_DEBUG({
216 dbgs() << " Externalizing weak symbol " << Sym->getName() << "\n";
217 });
218 DefinitionsToRemove.push_back(Sym);
219 } else {
220 LLVM_DEBUG({
221 dbgs() << " Making weak symbol " << Sym->getName() << " strong\n";
222 });
223 if (S.HarnessExternals.count(Sym->getName()))
224 Sym->setScope(Scope::Default);
225 else
226 Sym->setScope(Scope::Hidden);
227 Sym->setLinkage(Linkage::Strong);
228 }
229 } else if (S.HarnessExternals.count(Sym->getName())) {
230 LLVM_DEBUG(dbgs() << " Promoting " << Sym->getName() << "\n");
231 Sym->setScope(Scope::Default);
232 Sym->setLive(true);
233 continue;
234 } else if (S.HarnessDefinitions.count(Sym->getName())) {
235 LLVM_DEBUG(dbgs() << " Externalizing " << Sym->getName() << "\n");
236 DefinitionsToRemove.push_back(Sym);
237 }
238 }
239
240 for (auto *Sym : DefinitionsToRemove)
241 G.makeExternal(*Sym);
242
243 return Error::success();
244}

253static void dumpSectionContents(raw_ostream &OS, LinkGraph &G) {
254 constexpr JITTargetAddress DumpWidth = 16;
255 static_assert(isPowerOf2_64(DumpWidth), "DumpWidth must be a power of two");
256
257 // Put sections in address order.
258 std::vector<Section *> Sections;
259 for (auto &S : G.sections())
260 Sections.push_back(&S);
261
262 llvm::sort(Sections, [](const Section *LHS, const Section *RHS) {
263 if (llvm::empty(LHS->symbols()) && llvm::empty(RHS->symbols()))
264 return false;
265 if (llvm::empty(LHS->symbols()))
266 return false;
267 if (llvm::empty(RHS->symbols()))
268 return true;
269 SectionRange LHSRange(*LHS);
270 SectionRange RHSRange(*RHS);
271 return LHSRange.getStart() < RHSRange.getStart();
272 });
273
274 for (auto *S : Sections) {
275 OS << S->getName() << " content:";
276 if (llvm::empty(S->symbols())) {
277 OS << "\n section empty\n";
278 continue;
279 }
280
281 // Sort symbols into order, then render.
282 std::vector<Symbol *> Syms(S->symbols().begin(), S->symbols().end());
283 llvm::sort(Syms, [](const Symbol *LHS, const Symbol *RHS) {
284 return LHS->getAddress() < RHS->getAddress();
285 });
286
287 JITTargetAddress NextAddr = Syms.front()->getAddress() & ~(DumpWidth - 1);
288 for (auto *Sym : Syms) {
289 bool IsZeroFill = Sym->getBlock().isZeroFill();
290 JITTargetAddress SymStart = Sym->getAddress();
291 JITTargetAddress SymSize = Sym->getSize();
292 JITTargetAddress SymEnd = SymStart + SymSize;
293 const uint8_t *SymData =
294 IsZeroFill ? nullptr : Sym->getSymbolContent().bytes_begin();
295
296 // Pad any space before the symbol starts.
297 while (NextAddr != SymStart) {
298 if (NextAddr % DumpWidth == 0)
299 OS << formatv("\n{0:x16}:", NextAddr);
300 OS << " ";
301 ++NextAddr;
302 }
303
304 // Render the symbol content.
305 while (NextAddr != SymEnd) {
306 if (NextAddr % DumpWidth == 0)
307 OS << formatv("\n{0:x16}:", NextAddr);
308 if (IsZeroFill)
309 OS << " 00";
310 else
311 OS << formatv(" {0:x-2}", SymData[NextAddr - SymStart]);
312 ++NextAddr;
313 }
314 }
315 OS << "\n";
316 }

319class JITLinkSlabAllocator final : public JITLinkMemoryManager {
320public:
321 static Expected<std::unique_ptr<JITLinkSlabAllocator>>
322 Create(uint64_t SlabSize) {
323 Error Err = Error::success();
324 std::unique_ptr<JITLinkSlabAllocator> Allocator(
325 new JITLinkSlabAllocator(SlabSize, Err));
326 if (Err)
327 return std::move(Err);
328 return std::move(Allocator);
329 }
330
331 Expected<std::unique_ptr<JITLinkMemoryManager::Allocation>>
332 allocate(const JITLinkDylib *JD, const SegmentsRequestMap &Request) override {
333
334 using AllocationMap = DenseMap<unsigned, sys::MemoryBlock>;
335
336 // Local class for allocation.
337 class IPMMAlloc : public Allocation {
338 public:
339 IPMMAlloc(JITLinkSlabAllocator &Parent, AllocationMap SegBlocks)
340 : Parent(Parent), SegBlocks(std::move(SegBlocks)) {}
341 MutableArrayRef<char> getWorkingMemory(ProtectionFlags Seg) override {
342 assert(SegBlocks.count(Seg) && "No allocation for segment");
343 return {static_cast<char *>(SegBlocks[Seg].base()),
344 SegBlocks[Seg].allocatedSize()};
345 }
346 JITTargetAddress getTargetMemory(ProtectionFlags Seg) override {
347 assert(SegBlocks.count(Seg) && "No allocation for segment");
348 return pointerToJITTargetAddress(SegBlocks[Seg].base()) +
349 Parent.TargetDelta;
350 }
351 void finalizeAsync(FinalizeContinuation OnFinalize) override {
352 OnFinalize(applyProtections());
353 }
354 Error deallocate() override {
355 for (auto &KV : SegBlocks)
356 if (auto EC = sys::Memory::releaseMappedMemory(KV.second))
357 return errorCodeToError(EC);
358 return Error::success();
359 }
360
361 private:
362 Error applyProtections() {
363 for (auto &KV : SegBlocks) {
364 auto &Prot = KV.first;
365 auto &Block = KV.second;
366 if (auto EC = sys::Memory::protectMappedMemory(Block, Prot))
367 return errorCodeToError(EC);
368 if (Prot & sys::Memory::MF_EXEC)
369 sys::Memory::InvalidateInstructionCache(Block.base(),
370 Block.allocatedSize());
371 }
372 return Error::success();
373 }
374
375 JITLinkSlabAllocator &Parent;
376 AllocationMap SegBlocks;
377 };
378
379 AllocationMap Blocks;
380
381 for (auto &KV : Request) {
382 auto &Seg = KV.second;
383
384 if (Seg.getAlignment() > PageSize)
385 return make_error<StringError>("Cannot request higher than page "
386 "alignment",
387 inconvertibleErrorCode());
388
389 if (PageSize % Seg.getAlignment() != 0)
390 return make_error<StringError>("Page size is not a multiple of "
391 "alignment",
392 inconvertibleErrorCode());
393
394 uint64_t ZeroFillStart = Seg.getContentSize();
395 uint64_t SegmentSize = ZeroFillStart + Seg.getZeroFillSize();
396
397 // Round segment size up to page boundary.
398 SegmentSize = (SegmentSize + PageSize - 1) & ~(PageSize - 1);
399
400 // Take segment bytes from the front of the slab.
401 void *SlabBase = SlabRemaining.base();
402 uint64_t SlabRemainingSize = SlabRemaining.allocatedSize();
403
404 if (SegmentSize > SlabRemainingSize)
405 return make_error<StringError>("Slab allocator out of memory",
406 inconvertibleErrorCode());
407
408 sys::MemoryBlock SegMem(SlabBase, SegmentSize);
409 SlabRemaining =
410 sys::MemoryBlock(reinterpret_cast<char *>(SlabBase) + SegmentSize,
411 SlabRemainingSize - SegmentSize);
412
413 // Zero out the zero-fill memory.
414 memset(static_cast<char *>(SegMem.base()) + ZeroFillStart, 0,
415 Seg.getZeroFillSize());
416
417 // Record the block for this segment.
418 Blocks[KV.first] = std::move(SegMem);
419 }
420 return std::unique_ptr<InProcessMemoryManager::Allocation>(
421 new IPMMAlloc(*this, std::move(Blocks)));
422 }
423
424private:
425 JITLinkSlabAllocator(uint64_t SlabSize, Error &Err) {
426 ErrorAsOutParameter _(&Err);
427
428 PageSize = sys::Process::getPageSizeEstimate();
429
430 if (!isPowerOf2_64(PageSize)) {
431 Err = make_error<StringError>("Page size is not a power of 2",
432 inconvertibleErrorCode());
433 return;
434 }
435
436 // Round slab request up to page size.
437 SlabSize = (SlabSize + PageSize - 1) & ~(PageSize - 1);
438
439 const sys::Memory::ProtectionFlags ReadWrite =
440 static_cast<sys::Memory::ProtectionFlags>(sys::Memory::MF_READ |
441 sys::Memory::MF_WRITE);
442
443 std::error_code EC;
444 SlabRemaining =
445 sys::Memory::allocateMappedMemory(SlabSize, nullptr, ReadWrite, EC);
446
447 if (EC) {
448 Err = errorCodeToError(EC);
449 return;
450 }
451
452 // Calculate the target address delta to link as-if slab were at
453 // SlabAddress.
454 if (SlabAddress != ~0ULL)
455 TargetDelta =
456 SlabAddress - pointerToJITTargetAddress(SlabRemaining.base());
457 }
458
459 sys::MemoryBlock SlabRemaining;
460 uint64_t PageSize = 0;
461 int64_t TargetDelta = 0;
462};

464Expected<uint64_t> getSlabAllocSize(StringRef SizeString) {
465 SizeString = SizeString.trim();
466
467 uint64_t Units = 1024;
468
469 if (SizeString.endswith_lower("kb"))
470 SizeString = SizeString.drop_back(2).rtrim();
471 else if (SizeString.endswith_lower("mb")) {
472 Units = 1024 * 1024;
473 SizeString = SizeString.drop_back(2).rtrim();
474 } else if (SizeString.endswith_lower("gb")) {
475 Units = 1024 * 1024 * 1024;
476 SizeString = SizeString.drop_back(2).rtrim();
477 }
478
479 uint64_t SlabSize = 0;
480 if (SizeString.getAsInteger(10, SlabSize))
481 return make_error<StringError>("Invalid numeric format for slab size",
482 inconvertibleErrorCode());
483
484 return SlabSize * Units;
485}

499Error LLVMJITLinkObjectLinkingLayer::add(ResourceTrackerSP RT,
500 std::unique_ptr<MemoryBuffer> O) {
501
502 if (S.HarnessFiles.empty() || S.HarnessFiles.count(O->getBufferIdentifier()))
503 return ObjectLinkingLayer::add(std::move(RT), std::move(O));
504
505 // Use getObjectSymbolInfo to compute the init symbol, but ignore
506 // the symbols field. We'll handle that manually to include promotion.
507 auto ObjSymInfo =
508 getObjectSymbolInfo(getExecutionSession(), O->getMemBufferRef());
509
510 if (!ObjSymInfo)
511 return ObjSymInfo.takeError();
512
513 auto &InitSymbol = ObjSymInfo->second;
514
515 // If creating an object file was going to fail it would have happened above,
516 // so we can 'cantFail' this.
517 auto Obj =
518 cantFail(object::ObjectFile::createObjectFile(O->getMemBufferRef()));
519
520 SymbolFlagsMap SymbolFlags;
521
522 // The init symbol must be included in the SymbolFlags map if present.
523 if (InitSymbol)
524 SymbolFlags[InitSymbol] = JITSymbolFlags::MaterializationSideEffectsOnly;
525
526 for (auto &Sym : Obj->symbols()) {
527 Expected<uint32_t> SymFlagsOrErr = Sym.getFlags();
528 if (!SymFlagsOrErr)
529 // TODO: Test this error.
530 return SymFlagsOrErr.takeError();
531
532 // Skip symbols not defined in this object file.
533 if ((*SymFlagsOrErr & object::BasicSymbolRef::SF_Undefined))
534 continue;
535
536 auto Name = Sym.getName();
537 if (!Name)
538 return Name.takeError();
539
540 // Skip symbols that have type SF_File.
541 if (auto SymType = Sym.getType()) {
542 if (*SymType == object::SymbolRef::ST_File)
543 continue;
544 } else
545 return SymType.takeError();
546
547 auto SymFlags = JITSymbolFlags::fromObjectSymbol(Sym);
548 if (!SymFlags)
549 return SymFlags.takeError();
550
551 if (SymFlags->isWeak()) {
552 // If this is a weak symbol that's not defined in the harness then we
553 // need to either mark it as strong (if this is the first definition
554 // that we've seen) or discard it.
555 if (S.HarnessDefinitions.count(*Name) || S.CanonicalWeakDefs.count(*Name))
556 continue;
557 S.CanonicalWeakDefs[*Name] = O->getBufferIdentifier();
558 *SymFlags &= ~JITSymbolFlags::Weak;
559 if (!S.HarnessExternals.count(*Name))
560 *SymFlags &= ~JITSymbolFlags::Exported;
561 } else if (S.HarnessExternals.count(*Name)) {
562 *SymFlags |= JITSymbolFlags::Exported;
563 } else if (S.HarnessDefinitions.count(*Name) ||
564 !(*SymFlagsOrErr & object::BasicSymbolRef::SF_Global))
565 continue;
566
567 auto InternedName = S.ES.intern(*Name);
568 SymbolFlags[InternedName] = std::move(*SymFlags);
569 }
570
571 auto MU = std::make_unique<BasicObjectLayerMaterializationUnit>(
572 *this, std::move(O), std::move(SymbolFlags), std::move(InitSymbol));
573
574 auto &JD = RT->getJITDylib();
575 return JD.define(std::move(MU), std::move(RT));
576}

578Expected<std::unique_ptr<TargetProcessControl>>
579LLVMJITLinkRemoteTargetProcessControl::LaunchExecutor() {
580#ifndef LLVM_ON_UNIX
581 // FIXME: Add support for Windows.
582 return make_error<StringError>("-" + OutOfProcessExecutor.ArgStr +
583 " not supported on non-unix platforms",
584 inconvertibleErrorCode());
585#else
586
587 shared::registerStringError<LLVMJITLinkChannel>();
588
589 constexpr int ReadEnd = 0;
590 constexpr int WriteEnd = 1;
591
592 // Pipe FDs.
593 int ToExecutor[2];
594 int FromExecutor[2];
595
596 pid_t ChildPID;
597
598 // Create pipes to/from the executor..
599 if (pipe(ToExecutor) != 0 || pipe(FromExecutor) != 0)
600 return make_error<StringError>("Unable to create pipe for executor",
601 inconvertibleErrorCode());
602
603 ChildPID = fork();
604
605 if (ChildPID == 0) {
606 // In the child...
607
608 // Close the parent ends of the pipes
609 close(ToExecutor[WriteEnd]);
610 close(FromExecutor[ReadEnd]);
611
612 // Execute the child process.
613 std::unique_ptr<char[]> ExecutorPath, FDSpecifier;
614 {
615 ExecutorPath = std::make_unique<char[]>(OutOfProcessExecutor.size() + 1);
616 strcpy(ExecutorPath.get(), OutOfProcessExecutor.data());
617
618 std::string FDSpecifierStr("filedescs=");
619 FDSpecifierStr += utostr(ToExecutor[ReadEnd]);
620 FDSpecifierStr += ',';
621 FDSpecifierStr += utostr(FromExecutor[WriteEnd]);
622 FDSpecifier = std::make_unique<char[]>(FDSpecifierStr.size() + 1);
623 strcpy(FDSpecifier.get(), FDSpecifierStr.c_str());
624 }
625
626 char *const Args[] = {ExecutorPath.get(), FDSpecifier.get(), nullptr};
627 int RC = execvp(ExecutorPath.get(), Args);
628 if (RC != 0) {
629 errs() << "unable to launch out-of-process executor \""
630 << ExecutorPath.get() << "\"\n";
631 exit(1);
632 }
633 }
634 // else we're the parent...
635
636 // Close the child ends of the pipes
637 close(ToExecutor[ReadEnd]);
638 close(FromExecutor[WriteEnd]);
639
640 // Return an RPC channel connected to our end of the pipes.
641 auto SSP = std::make_shared<SymbolStringPool>();
642 auto Channel = std::make_unique<shared::FDRawByteChannel>(
643 FromExecutor[ReadEnd], ToExecutor[WriteEnd]);
644 auto Endpoint = std::make_unique<LLVMJITLinkRPCEndpoint>(*Channel, true);
645
646 auto ReportError = [](Error Err) {
647 logAllUnhandledErrors(std::move(Err), errs(), "");
648 };
649
650 Error Err = Error::success();
651 std::unique_ptr<LLVMJITLinkRemoteTargetProcessControl> RTPC(
652 new LLVMJITLinkRemoteTargetProcessControl(
653 std::move(SSP), std::move(Channel), std::move(Endpoint),
654 std::move(ReportError), Err));
655 if (Err)
656 return std::move(Err);
657 return std::move(RTPC);
658#endif
659}

661Expected<std::unique_ptr<TargetProcessControl>>
662LLVMJITLinkRemoteTargetProcessControl::ConnectToExecutor() {
663#ifndef LLVM_ON_UNIX
664 // FIXME: Add TCP support for Windows.
665 return make_error<StringError>("-" + OutOfProcessExecutorConnect.ArgStr +
666 " not supported on non-unix platforms",
667 inconvertibleErrorCode());
668#else
669
670 shared::registerStringError<LLVMJITLinkChannel>();
671
672 StringRef HostNameStr, PortStr;
673 std::tie(HostNameStr, PortStr) =
674 StringRef(OutOfProcessExecutorConnect).split(':');
675
676 if (HostNameStr.empty())
677 return make_error<StringError>("host name for -" +
678 OutOfProcessExecutorConnect.ArgStr +
679 " can not be empty",
680 inconvertibleErrorCode());
681 if (PortStr.empty())
682 return make_error<StringError>(
683 "port for -" + OutOfProcessExecutorConnect.ArgStr + " can not be empty",
684 inconvertibleErrorCode());
685
686 std::string HostName = HostNameStr.str();
687 int Port = 0;
688 if (PortStr.getAsInteger(10, Port))
689 return make_error<StringError>("port number " + PortStr +
690 " is not a valid integer",
691 inconvertibleErrorCode());
692
693 int SockFD = socket(PF_INET, SOCK_STREAM, 0);
694 hostent *Server = gethostbyname(HostName.c_str());
695 sockaddr_in ServAddr;
696 memset(&ServAddr, 0, sizeof(ServAddr));
697 ServAddr.sin_family = PF_INET;
698 memmove(&Server->h_addr, &ServAddr.sin_addr.s_addr, Server->h_length);
699 ServAddr.sin_port = htons(Port);
700 if (connect(SockFD, reinterpret_cast<sockaddr *>(&ServAddr),
701 sizeof(ServAddr)) < 0)
702 return make_error<StringError>("Failed to connect to " + HostName + ":" +
703 Twine(Port),
704 inconvertibleErrorCode());
705
706 auto SSP = std::make_shared<SymbolStringPool>();
707 auto Channel = std::make_unique<shared::FDRawByteChannel>(SockFD, SockFD);
708 auto Endpoint = std::make_unique<LLVMJITLinkRPCEndpoint>(*Channel, true);
709
710 auto ReportError = [](Error Err) {
711 logAllUnhandledErrors(std::move(Err), errs(), "");
712 };
713
714 Error Err = Error::success();
715 std::unique_ptr<LLVMJITLinkRemoteTargetProcessControl> RTPC(
716 new LLVMJITLinkRemoteTargetProcessControl(
717 std::move(SSP), std::move(Channel), std::move(Endpoint),
718 std::move(ReportError), Err));
719 if (Err)
720 return std::move(Err);
721 return std::move(RTPC);
722#endif
723}
724
725Error LLVMJITLinkRemoteTargetProcessControl::disconnect() {
726 std::promise<MSVCPError> P;
727 auto F = P.get_future();
728 auto Err = closeConnection([&](Error Err) -> Error {
729 P.set_value(std::move(Err));
730 Finished = true;
731 return Error::success();
732 });
733 ListenerThread.join();
734 return joinErrors(std::move(Err), F.get());
735}

749Expected<std::unique_ptr<Session>> Session::Create(Triple TT) {
750
751 auto PageSize = sys::Process::getPageSize();
752 if (!PageSize)
753 return PageSize.takeError();
754
755 /// If -oop-executor is passed then launch the executor.
756 std::unique_ptr<TargetProcessControl> TPC;
757 if (OutOfProcessExecutor.getNumOccurrences()) {
758 if (auto RTPC = LLVMJITLinkRemoteTargetProcessControl::LaunchExecutor())
759 TPC = std::move(*RTPC);
760 else
761 return RTPC.takeError();
762 } else if (OutOfProcessExecutorConnect.getNumOccurrences()) {
763 if (auto RTPC = LLVMJITLinkRemoteTargetProcessControl::ConnectToExecutor())
764 TPC = std::move(*RTPC);
765 else
766 return RTPC.takeError();
767 } else
768 TPC = std::make_unique<SelfTargetProcessControl>(
769 std::make_shared<SymbolStringPool>(), std::move(TT), *PageSize,
770 createMemoryManager());
771
772 Error Err = Error::success();
773 std::unique_ptr<Session> S(new Session(std::move(TPC), Err));
774 if (Err)
775 return std::move(Err);
776 return std::move(S);

988static Error sanitizeArguments(const Triple &TT, const char *ArgV0) {
989 // Set the entry point name if not specified.
990 if (EntryPointName.empty()) {
991 if (TT.getObjectFormat() == Triple::MachO)
992 EntryPointName = "_main";
993 else
994 EntryPointName = "main";
995 }
996
997 // -noexec and --args should not be used together.
998 if (NoExec && !InputArgv.empty())
999 outs() << "Warning: --args passed to -noexec run will be ignored.\n";
1000
1001 // If -slab-address is passed, require -slab-allocate and -noexec
1002 if (SlabAddress != ~0ULL) {
1003 if (SlabAllocateSizeString == "" || !NoExec)
1004 return make_error<StringError>(
1005 "-slab-address requires -slab-allocate and -noexec",
1006 inconvertibleErrorCode());
1007 }
1008
1009 // Only one of -oop-executor and -oop-executor-connect can be used.
1010 if (!!OutOfProcessExecutor.getNumOccurrences() &&
1011 !!OutOfProcessExecutorConnect.getNumOccurrences())
1012 return make_error<StringError>(
1013 "Only one of -" + OutOfProcessExecutor.ArgStr + " and -" +
1014 OutOfProcessExecutorConnect.ArgStr + " can be specified",
1015 inconvertibleErrorCode());
1016
1017 // If -oop-executor was used but no value was specified then use a sensible
1018 // default.
1019 if (!!OutOfProcessExecutor.getNumOccurrences() &&
1020 OutOfProcessExecutor.empty()) {
1021 SmallString<256> OOPExecutorPath(sys::fs::getMainExecutable(
1022 ArgV0, reinterpret_cast<void *>(&sanitizeArguments)));
1023 sys::path::remove_filename(OOPExecutorPath);
1024 if (OOPExecutorPath.back() != '/')
1025 OOPExecutorPath += '/';
1026 OOPExecutorPath += "llvm-jitlink-executor";
1027 OutOfProcessExecutor = OOPExecutorPath.str().str();
1028 }
1029
1030 return Error::success();
1031}

1060static Error loadObjects(Session &S) {
1061 std::map<unsigned, JITDylib *> IdxToJLD;
1062
1063 // First, set up JITDylibs.
1064 LLVM_DEBUG(dbgs() << "Creating JITDylibs...\n");
1065 {
1066 // Create a "main" JITLinkDylib.
1067 IdxToJLD[0] = S.MainJD;
1068 S.JDSearchOrder.push_back(S.MainJD);
1069 LLVM_DEBUG(dbgs() << " 0: " << S.MainJD->getName() << "\n");
1070
1071 // Add any extra JITLinkDylibs from the command line.
1072 std::string JDNamePrefix("lib");
1073 for (auto JLDItr = JITLinkDylibs.begin(), JLDEnd = JITLinkDylibs.end();
1074 JLDItr != JLDEnd; ++JLDItr) {
1075 auto JD = S.ES.createJITDylib(JDNamePrefix + *JLDItr);
1076 if (!JD)
1077 return JD.takeError();
1078 unsigned JDIdx =
1079 JITLinkDylibs.getPosition(JLDItr - JITLinkDylibs.begin());
1080 IdxToJLD[JDIdx] = &*JD;
1081 S.JDSearchOrder.push_back(&*JD);
1082 LLVM_DEBUG(dbgs() << " " << JDIdx << ": " << JD->getName() << "\n");
1083 }
1084
1085 // Set every dylib to link against every other, in command line order.
1086 for (auto *JD : S.JDSearchOrder) {
1087 auto LookupFlags = JITDylibLookupFlags::MatchExportedSymbolsOnly;
1088 JITDylibSearchOrder LinkOrder;
1089 for (auto *JD2 : S.JDSearchOrder) {
1090 if (JD2 == JD)
1091 continue;
1092 LinkOrder.push_back(std::make_pair(JD2, LookupFlags));
1093 }
1094 JD->setLinkOrder(std::move(LinkOrder));
1095 }
1096 }
1097
1098 LLVM_DEBUG(dbgs() << "Adding test harness objects...\n");
1099 for (auto HarnessFile : TestHarnesses) {
1100 LLVM_DEBUG(dbgs() << " " << HarnessFile << "\n");
1101 auto ObjBuffer =
1102 ExitOnErr(errorOrToExpected(MemoryBuffer::getFile(HarnessFile)));
1103 ExitOnErr(S.ObjLayer.add(*S.MainJD, std::move(ObjBuffer)));
1104 }
1105
1106 // Load each object into the corresponding JITDylib..
1107 LLVM_DEBUG(dbgs() << "Adding objects...\n");
1108 for (auto InputFileItr = InputFiles.begin(), InputFileEnd = InputFiles.end();
1109 InputFileItr != InputFileEnd; ++InputFileItr) {
1110 unsigned InputFileArgIdx =
1111 InputFiles.getPosition(InputFileItr - InputFiles.begin());
1112 const std::string &InputFile = *InputFileItr;
1113 auto &JD = *std::prev(IdxToJLD.lower_bound(InputFileArgIdx))->second;
1114 LLVM_DEBUG(dbgs() << " " << InputFileArgIdx << ": \"" << InputFile
1115 << "\" to " << JD.getName() << "\n";);
1116 auto ObjBuffer =
1117 ExitOnErr(errorOrToExpected(MemoryBuffer::getFile(InputFile)));
1118
1119 auto Magic = identify_magic(ObjBuffer->getBuffer());
1120 if (Magic == file_magic::archive ||
1121 Magic == file_magic::macho_universal_binary)
1122 JD.addGenerator(ExitOnErr(StaticLibraryDefinitionGenerator::Load(
1123 S.ObjLayer, InputFile.c_str(), S.TPC->getTargetTriple())));
1124 else
1125 ExitOnErr(S.ObjLayer.add(JD, std::move(ObjBuffer)));
1126 }
1127
1128 // Define absolute symbols.
1129 LLVM_DEBUG(dbgs() << "Defining absolute symbols...\n");
1130 for (auto AbsDefItr = AbsoluteDefs.begin(), AbsDefEnd = AbsoluteDefs.end();
1131 AbsDefItr != AbsDefEnd; ++AbsDefItr) {
1132 unsigned AbsDefArgIdx =
1133 AbsoluteDefs.getPosition(AbsDefItr - AbsoluteDefs.begin());
1134 auto &JD = *std::prev(IdxToJLD.lower_bound(AbsDefArgIdx))->second;
1135
1136 StringRef AbsDefStmt = *AbsDefItr;
1137 size_t EqIdx = AbsDefStmt.find_first_of('=');
1138 if (EqIdx == StringRef::npos)
1139 return make_error<StringError>("Invalid absolute define \"" + AbsDefStmt +
1140 "\". Syntax: <name>=<addr>",
1141 inconvertibleErrorCode());
1142 StringRef Name = AbsDefStmt.substr(0, EqIdx).trim();
1143 StringRef AddrStr = AbsDefStmt.substr(EqIdx + 1).trim();
1144
1145 uint64_t Addr;
1146 if (AddrStr.getAsInteger(0, Addr))
1147 return make_error<StringError>("Invalid address expression \"" + AddrStr +
1148 "\" in absolute define \"" + AbsDefStmt +
1149 "\"",
1150 inconvertibleErrorCode());
1151 JITEvaluatedSymbol AbsDef(Addr, JITSymbolFlags::Exported);
1152 if (auto Err = JD.define(absoluteSymbols({{S.ES.intern(Name), AbsDef}})))
1153 return Err;
1154
1155 // Register the absolute symbol with the session symbol infos.
1156 S.SymbolInfos[Name] = { StringRef(), Addr };
1157 }
1158
1159 LLVM_DEBUG({
1160 dbgs() << "Dylib search order is [ ";
1161 for (auto *JD : S.JDSearchOrder)
1162 dbgs() << JD->getName() << " ";
1163 dbgs() << "]\n";
1164 });
1165
1166 return Error::success();
1167}

1169static Error runChecks(Session &S) {
1170
1171 auto TripleName = S.TPC->getTargetTriple().str();
1172 std::string ErrorStr;
1173 const Target *TheTarget = TargetRegistry::lookupTarget(TripleName, ErrorStr);
1174 if (!TheTarget)
1175 ExitOnErr(make_error<StringError>("Error accessing target '" + TripleName +
1176 "': " + ErrorStr,
1177 inconvertibleErrorCode()));
1178
1179 std::unique_ptr<MCSubtargetInfo> STI(
1180 TheTarget->createMCSubtargetInfo(TripleName, "", ""));
1181 if (!STI)
1182 ExitOnErr(
1183 make_error<StringError>("Unable to create subtarget for " + TripleName,
1184 inconvertibleErrorCode()));
1185
1186 std::unique_ptr<MCRegisterInfo> MRI(TheTarget->createMCRegInfo(TripleName));
1187 if (!MRI)
1188 ExitOnErr(make_error<StringError>("Unable to create target register info "
1189 "for " +
1190 TripleName,
1191 inconvertibleErrorCode()));
1192
1193 MCTargetOptions MCOptions;
1194 std::unique_ptr<MCAsmInfo> MAI(
1195 TheTarget->createMCAsmInfo(*MRI, TripleName, MCOptions));
1196 if (!MAI)
1197 ExitOnErr(make_error<StringError>("Unable to create target asm info " +
1198 TripleName,
1199 inconvertibleErrorCode()));
1200
1201 MCContext Ctx(MAI.get(), MRI.get(), nullptr);
1202
1203 std::unique_ptr<MCDisassembler> Disassembler(
1204 TheTarget->createMCDisassembler(*STI, Ctx));
1205 if (!Disassembler)
1206 ExitOnErr(make_error<StringError>("Unable to create disassembler for " +
1207 TripleName,
1208 inconvertibleErrorCode()));
1209
1210 std::unique_ptr<MCInstrInfo> MII(TheTarget->createMCInstrInfo());
1211
1212 std::unique_ptr<MCInstPrinter> InstPrinter(
1213 TheTarget->createMCInstPrinter(Triple(TripleName), 0, *MAI, *MII, *MRI));
1214
1215 auto IsSymbolValid = [&S](StringRef Symbol) {
1216 return S.isSymbolRegistered(Symbol);
1217 };
1218
1219 auto GetSymbolInfo = [&S](StringRef Symbol) {
1220 return S.findSymbolInfo(Symbol, "Can not get symbol info");
1221 };
1222
1223 auto GetSectionInfo = [&S](StringRef FileName, StringRef SectionName) {
1224 return S.findSectionInfo(FileName, SectionName);
1225 };
1226
1227 auto GetStubInfo = [&S](StringRef FileName, StringRef SectionName) {
1228 return S.findStubInfo(FileName, SectionName);
1229 };
1230
1231 auto GetGOTInfo = [&S](StringRef FileName, StringRef SectionName) {
1232 return S.findGOTEntryInfo(FileName, SectionName);
1233 };
1234
1235 RuntimeDyldChecker Checker(
1236 IsSymbolValid, GetSymbolInfo, GetSectionInfo, GetStubInfo, GetGOTInfo,
1237 S.TPC->getTargetTriple().isLittleEndian() ? support::little
1238 : support::big,
1239 Disassembler.get(), InstPrinter.get(), dbgs());
1240
1241 std::string CheckLineStart = "# " + CheckName + ":";
1242 for (auto &CheckFile : CheckFiles) {
1243 auto CheckerFileBuf =
1244 ExitOnErr(errorOrToExpected(MemoryBuffer::getFile(CheckFile)));
1245 if (!Checker.checkAllRulesInBuffer(CheckLineStart, &*CheckerFileBuf))
1246 ExitOnErr(make_error<StringError>(
1247 "Some checks in " + CheckFile + " failed", inconvertibleErrorCode()));
1248 }
1249
1250 return Error::success();
1251}

1273int main(int argc, char *argv[]) {
1274 InitLLVM X(argc, argv);
1275
1276 InitializeAllTargetInfos();
1277 InitializeAllTargetMCs();
1278 InitializeAllDisassemblers();
1279
1280 cl::ParseCommandLineOptions(argc, argv, "llvm jitlink tool");
1281 ExitOnErr.setBanner(std::string(argv[0]) + ": ");
1282
1283 /// If timers are enabled, create a JITLinkTimers instance.
1284 std::unique_ptr<JITLinkTimers> Timers =
1285 ShowTimes ? std::make_unique<JITLinkTimers>() : nullptr;
1286
1287 ExitOnErr(sanitizeArguments(getFirstFileTriple(), argv[0]));
1288
1289 auto S = ExitOnErr(Session::Create(getFirstFileTriple()));
1290
1291 {
1292 TimeRegion TR(Timers ? &Timers->LoadObjectsTimer : nullptr);
1293 ExitOnErr(loadObjects(*S));
1294 }
1295
1296 if (!NoProcessSymbols)
1297 ExitOnErr(loadProcessSymbols(*S));
1298 ExitOnErr(loadDylibs(*S));
1299
1300 if (PhonyExternals)
1301 addPhonyExternalsGenerator(*S);
1302
1303
1304 if (ShowInitialExecutionSessionState)
1305 S->ES.dump(outs());
1306
1307 JITEvaluatedSymbol EntryPoint = 0;
1308 {
1309 TimeRegion TR(Timers ? &Timers->LinkTimer : nullptr);
1310 EntryPoint = ExitOnErr(getMainEntryPoint(*S));
1311 }
1312
1313 if (ShowAddrs)
