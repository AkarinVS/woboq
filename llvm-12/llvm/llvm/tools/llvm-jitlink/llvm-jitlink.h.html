59class LLVMJITLinkRemoteTargetProcessControl
60 : public orc::OrcRPCTargetProcessControlBase<LLVMJITLinkRPCEndpoint> {
61public:
62 using BaseT = orc::OrcRPCTargetProcessControlBase<LLVMJITLinkRPCEndpoint>;
63 static Expected<std::unique_ptr<TargetProcessControl>> LaunchExecutor();
64
65 static Expected<std::unique_ptr<TargetProcessControl>> ConnectToExecutor();
66
67 Error disconnect() override;
68
69private:
70 using LLVMJITLinkRemoteMemoryAccess =
71 orc::OrcRPCTPCMemoryAccess<LLVMJITLinkRemoteTargetProcessControl>;
72
73 using LLVMJITLinkRemoteMemoryManager =
74 orc::OrcRPCTPCJITLinkMemoryManager<LLVMJITLinkRemoteTargetProcessControl>;
75
76 LLVMJITLinkRemoteTargetProcessControl(
77 std::shared_ptr<orc::SymbolStringPool> SSP,
78 std::unique_ptr<LLVMJITLinkChannel> Channel,
79 std::unique_ptr<LLVMJITLinkRPCEndpoint> Endpoint,
80 ErrorReporter ReportError, Error &Err)
81 : BaseT(std::move(SSP), *Endpoint, std::move(ReportError)),
82 Channel(std::move(Channel)), Endpoint(std::move(Endpoint)) {
83 ErrorAsOutParameter _(&Err);
84
85 ListenerThread = std::thread([&]() {
86 while (!Finished) {
87 if (auto Err = this->Endpoint->handleOne()) {
88 reportError(std::move(Err));
89 return;
90 }
91 }
92 });
93
94 if (auto Err2 = initializeORCRPCTPCBase()) {
95 Err = joinErrors(std::move(Err2), disconnect());
96 return;
97 }
98
99 OwnedMemAccess = std::make_unique<LLVMJITLinkRemoteMemoryAccess>(*this);
100 MemAccess = OwnedMemAccess.get();
101 OwnedMemMgr = std::make_unique<LLVMJITLinkRemoteMemoryManager>(*this);
102 MemMgr = OwnedMemMgr.get();
103 }
104
105 std::unique_ptr<LLVMJITLinkChannel> Channel;
106 std::unique_ptr<LLVMJITLinkRPCEndpoint> Endpoint;
107 std::unique_ptr<TargetProcessControl::MemoryAccess> OwnedMemAccess;
108 std::unique_ptr<jitlink::JITLinkMemoryManager> OwnedMemMgr;
109 std::atomic<bool> Finished{false};
110 std::thread ListenerThread;
111};
