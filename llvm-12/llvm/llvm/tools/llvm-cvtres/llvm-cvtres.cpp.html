69static LLVM_ATTRIBUTE_NORETURN void reportError(Twine Msg) {
70 errs() << Msg;
71 exit(1);
72}
73
74static void reportError(StringRef Input, std::error_code EC) {
75 reportError(Twine(Input) + ": " + EC.message() + ".\n");
76}
77
78static void error(Error EC) {
79 if (!EC)
80 return;
81 handleAllErrors(std::move(EC),
82 [&](const ErrorInfoBase &EI) { reportError(EI.message()); });
83}
84
85static uint32_t getTime() {
86 std::time_t Now = time(nullptr);
87 if (Now < 0 || !isUInt<32>(Now))
88 return UINT32_MAX;
89 return static_cast<uint32_t>(Now);
90}
91
92template <typename T> T error(Expected<T> EC) {
93 if (!EC)
94 error(EC.takeError());
95 return std::move(EC.get());
96}

98int main(int Argc, const char **Argv) {
99 InitLLVM X(Argc, Argv);
100
101 CvtResOptTable T;
102 unsigned MAI, MAC;
103 ArrayRef<const char *> ArgsArr = makeArrayRef(Argv + 1, Argc - 1);
104 opt::InputArgList InputArgs = T.ParseArgs(ArgsArr, MAI, MAC);
105
106 if (InputArgs.hasArg(OPT_HELP)) {
107 T.PrintHelp(outs(), "llvm-cvtres [options] file...", "Resource Converter");
108 return 0;
109 }
110
111 bool Verbose = InputArgs.hasArg(OPT_VERBOSE);
112
113 COFF::MachineTypes MachineType;
114
115 if (opt::Arg *Arg = InputArgs.getLastArg(OPT_MACHINE)) {
116 MachineType = getMachineType(Arg->getValue());
117 if (MachineType == COFF::IMAGE_FILE_MACHINE_UNKNOWN) {
118 reportError(Twine("Unsupported machine architecture ") + Arg->getValue() +
119 "\n");
120 }
121 } else {
122 if (Verbose)
123 outs() << "Machine architecture not specified; assumed X64.\n";
124 MachineType = COFF::IMAGE_FILE_MACHINE_AMD64;
125 }
126
127 std::vector<std::string> InputFiles = InputArgs.getAllArgValues(OPT_INPUT);
128
129 if (InputFiles.size() == 0) {
130 reportError("No input file specified.\n");
131 }
132
133 SmallString<128> OutputFile;
134
135 if (opt::Arg *Arg = InputArgs.getLastArg(OPT_OUT)) {
136 OutputFile = Arg->getValue();
137 } else {
138 OutputFile = sys::path::filename(StringRef(InputFiles[0]));
139 sys::path::replace_extension(OutputFile, ".obj");
140 }
141
142 uint32_t DateTimeStamp;
143 if (llvm::opt::Arg *Arg = InputArgs.getLastArg(OPT_TIMESTAMP)) {
144 StringRef Value(Arg->getValue());
145 if (Value.getAsInteger(0, DateTimeStamp))
146 reportError(Twine("invalid timestamp: ") + Value +
147 ". Expected 32-bit integer\n");
148 } else {
149 DateTimeStamp = getTime();
150 }
151
152 if (Verbose)
153 outs() << "Machine: " << machineToStr(MachineType) << '\n';
154
155 WindowsResourceParser Parser;
156
157 for (const auto &File : InputFiles) {
158 Expected<OwningBinary<Binary>> BinaryOrErr = createBinary(File);
159 if (!BinaryOrErr)
160 reportError(File, errorToErrorCode(BinaryOrErr.takeError()));
161
162 Binary &Binary = *BinaryOrErr.get().getBinary();
163
164 WindowsResource *RF = dyn_cast<WindowsResource>(&Binary);
165 if (!RF)
166 reportError(File + ": unrecognized file format.\n");
167
168 if (Verbose) {
169 int EntryNumber = 0;
170 ResourceEntryRef Entry = error(RF->getHeadEntry());
171 bool End = false;
172 while (!End) {
173 error(Entry.moveNext(End));
174 EntryNumber++;
175 }
176 outs() << "Number of resources: " << EntryNumber << "\n";
177 }
178
179 std::vector<std::string> Duplicates;
180 error(Parser.parse(RF, Duplicates));
181 for (const auto& DupeDiag : Duplicates)
182 reportError(DupeDiag);
183 }
184
185 if (Verbose) {
186 Parser.printTree(outs());
187 }
188
189 std::unique_ptr<MemoryBuffer> OutputBuffer =
190 error(llvm::object::writeWindowsResourceCOFF(MachineType, Parser,
191 DateTimeStamp));
192 auto FileOrErr =
193 FileOutputBuffer::create(OutputFile, OutputBuffer->getBufferSize());
194 if (!FileOrErr)
195 reportError(OutputFile, errorToErrorCode(FileOrErr.takeError()));
196 std::unique_ptr<FileOutputBuffer> FileBuffer = std::move(*FileOrErr);
197 std::copy(OutputBuffer->getBufferStart(), OutputBuffer->getBufferEnd(),
198 FileBuffer->getBufferStart());
199 error(FileBuffer->commit());
200
201 if (Verbose) {
202 Expected<OwningBinary<Binary>> BinaryOrErr = createBinary(OutputFile);
203 if (!BinaryOrErr)
204 reportError(OutputFile, errorToErrorCode(BinaryOrErr.takeError()));
205 Binary &Binary = *BinaryOrErr.get().getBinary();
206 ScopedPrinter W(errs());
207 W.printBinaryBlock("Output File Raw Data", Binary.getData());
208 }
209
210 return 0;
211}
