109static Error copySwiftInterfaces(
110 const std::map<std::string, std::string> &ParseableSwiftInterfaces,
111 StringRef Architecture, const LinkOptions &Options) {
112 std::error_code EC;
113 SmallString<128> InputPath;
114 SmallString<128> Path;
115 sys::path::append(Path, *Options.ResourceDir, "Swift", Architecture);
116 if ((EC = sys::fs::create_directories(Path.str(), true,
117 sys::fs::perms::all_all)))
118 return make_error<StringError>(
119 "cannot create directory: " + toString(errorCodeToError(EC)), EC);
120 unsigned BaseLength = Path.size();
121
122 for (auto &I : ParseableSwiftInterfaces) {
123 StringRef ModuleName = I.first;
124 StringRef InterfaceFile = I.second;
125 if (!Options.PrependPath.empty()) {
126 InputPath.clear();
127 sys::path::append(InputPath, Options.PrependPath, InterfaceFile);
128 InterfaceFile = InputPath;
129 }
130 sys::path::append(Path, ModuleName);
131 Path.append(".swiftinterface");
132 if (Options.Verbose)
133 outs() << "copy parseable Swift interface " << InterfaceFile << " -> "
134 << Path.str() << '\n';
135
136 // copy_file attempts an APFS clone first, so this should be cheap.
137 if ((EC = sys::fs::copy_file(InterfaceFile, Path.str())))
138 warn(Twine("cannot copy parseable Swift interface ") + InterfaceFile +
139 ": " + toString(errorCodeToError(EC)));
140 Path.resize(BaseLength);
141 }
142 return Error::success();
143}

145/// Report a warning to the user, optionally including information about a
146/// specific \p DIE related to the warning.
147void DwarfLinkerForBinary::reportWarning(const Twine &Warning,
148 StringRef Context,
149 const DWARFDie *DIE) const {
150
151 warn(Warning, Context);
152
153 if (!Options.Verbose || !DIE)
154 return;
155
156 DIDumpOptions DumpOpts;
157 DumpOpts.ChildRecurseDepth = 0;
158 DumpOpts.Verbose = Options.Verbose;
159
160 WithColor::note() << " in DIE:\n";
161 DIE->dump(errs(), 6 /* Indent */, DumpOpts);
162}

164bool DwarfLinkerForBinary::createStreamer(const Triple &TheTriple,
165 raw_fd_ostream &OutFile) {
166 if (Options.NoOutput)
167 return true;
168
169 Streamer = std::make_unique<DwarfStreamer>(
170 Options.FileType, OutFile, Options.Translator, Options.Minimize,
171 [&](const Twine &Error, StringRef Context, const DWARFDie *) {
172 error(Error, Context);
173 },
174 [&](const Twine &Warning, StringRef Context, const DWARFDie *) {
175 warn(Warning, Context);
176 });
177 return Streamer->init(TheTriple);

180ErrorOr<const object::ObjectFile &>
181DwarfLinkerForBinary::loadObject(const DebugMapObject &Obj,
182 const Triple &Triple) {
183 auto ObjectEntry =
184 BinHolder.getObjectEntry(Obj.getObjectFilename(), Obj.getTimestamp());
185 if (!ObjectEntry) {
186 auto Err = ObjectEntry.takeError();
187 reportWarning(Twine(Obj.getObjectFilename()) + ": " +
188 toString(std::move(Err)),
189 Obj.getObjectFilename());
190 return errorToErrorCode(std::move(Err));
191 }
192
193 auto Object = ObjectEntry->getObject(Triple);
194 if (!Object) {
195 auto Err = Object.takeError();
196 reportWarning(Twine(Obj.getObjectFilename()) + ": " +
197 toString(std::move(Err)),
198 Obj.getObjectFilename());
199 return errorToErrorCode(std::move(Err));
200 }
201
202 return *Object;
203}

205static Error remarksErrorHandler(const DebugMapObject &DMO,
206 DwarfLinkerForBinary &Linker,
207 std::unique_ptr<FileError> FE) {
208 bool IsArchive = DMO.getObjectFilename().endswith(")");
209 // Don't report errors for missing remark files from static
210 // archives.
211 if (!IsArchive)
212 return Error(std::move(FE));
213
214 std::string Message = FE->message();
215 Error E = FE->takeError();
216 Error NewE = handleErrors(std::move(E), [&](std::unique_ptr<ECError> EC) {
217 if (EC->convertToErrorCode() != std::errc::no_such_file_or_directory)
218 return Error(std::move(EC));
219
220 Linker.reportWarning(Message, DMO.getObjectFilename());
221 return Error(Error::success());
222 });
223
224 if (!NewE)
225 return Error::success();
226
227 return createFileError(FE->getFileName(), std::move(NewE));
228}

230static Error emitRemarks(const LinkOptions &Options, StringRef BinaryPath,
231 StringRef ArchName, const remarks::RemarkLinker &RL) {
232 // Make sure we don't create the directories and the file if there is nothing
233 // to serialize.
234 if (RL.empty())
235 return Error::success();
236
237 SmallString<128> InputPath;
238 SmallString<128> Path;
239 // Create the "Remarks" directory in the "Resources" directory.
240 sys::path::append(Path, *Options.ResourceDir, "Remarks");
241 if (std::error_code EC = sys::fs::create_directories(Path.str(), true,
242 sys::fs::perms::all_all))
243 return errorCodeToError(EC);
244
245 // Append the file name.
246 // For fat binaries, also append a dash and the architecture name.
247 sys::path::append(Path, sys::path::filename(BinaryPath));
248 if (Options.NumDebugMaps > 1) {
249 // More than one debug map means we have a fat binary.
250 Path += '-';
251 Path += ArchName;
252 }
253
254 std::error_code EC;
255 raw_fd_ostream OS(Options.NoOutput ? "-" : Path.str(), EC, sys::fs::OF_None);
256 if (EC)
257 return errorCodeToError(EC);
258
259 if (Error E = RL.serialize(OS, Options.RemarksFormat))
260 return E;
261
262 return Error::success();
263}

265ErrorOr<DWARFFile &>
266DwarfLinkerForBinary::loadObject(const DebugMapObject &Obj,
267 const DebugMap &DebugMap,
268 remarks::RemarkLinker &RL) {
269 auto ErrorOrObj = loadObject(Obj, DebugMap.getTriple());
270
271 if (ErrorOrObj) {
272 ContextForLinking.push_back(
273 std::unique_ptr<DWARFContext>(DWARFContext::create(*ErrorOrObj)));
274 AddressMapForLinking.push_back(
275 std::make_unique<AddressManager>(*this, *ErrorOrObj, Obj));
276
277 ObjectsForLinking.push_back(std::make_unique<DWARFFile>(
278 Obj.getObjectFilename(), ContextForLinking.back().get(),
279 AddressMapForLinking.back().get(),
280 Obj.empty() ? Obj.getWarnings() : EmptyWarnings));
281
282 Error E = RL.link(*ErrorOrObj);
283 if (Error NewE = handleErrors(
284 std::move(E), [&](std::unique_ptr<FileError> EC) -> Error {
285 return remarksErrorHandler(Obj, *this, std::move(EC));
286 }))
287 return errorToErrorCode(std::move(NewE));
288
289 return *ObjectsForLinking.back();
290 }
291
292 return ErrorOrObj.getError();
293}

295bool DwarfLinkerForBinary::link(const DebugMap &Map) {
296 if (!createStreamer(Map.getTriple(), OutFile))
297 return false;
298
299 ObjectsForLinking.clear();
300 ContextForLinking.clear();
301 AddressMapForLinking.clear();
302
303 DebugMap DebugMap(Map.getTriple(), Map.getBinaryPath());
304
305 DWARFLinker GeneralLinker(Streamer.get(), DwarfLinkerClient::Dsymutil);
306
307 remarks::RemarkLinker RL;
308 if (!Options.RemarksPrependPath.empty())
309 RL.setExternalFilePrependPath(Options.RemarksPrependPath);
310 GeneralLinker.setObjectPrefixMap(&Options.ObjectPrefixMap);
311
312 std::function<StringRef(StringRef)> TranslationLambda = [&](StringRef Input) {
313 assert(Options.Translator);
314 return Options.Translator(Input);
315 };
316
317 GeneralLinker.setVerbosity(Options.Verbose);
318 GeneralLinker.setStatistics(Options.Statistics);
319 GeneralLinker.setNoOutput(Options.NoOutput);
320 GeneralLinker.setNoODR(Options.NoODR);
321 GeneralLinker.setUpdate(Options.Update);
322 GeneralLinker.setNumThreads(Options.Threads);
323 GeneralLinker.setAccelTableKind(Options.TheAccelTableKind);
324 GeneralLinker.setPrependPath(Options.PrependPath);
325 if (Options.Translator)
326 GeneralLinker.setStringsTranslator(TranslationLambda);
327 GeneralLinker.setWarningHandler(
328 [&](const Twine &Warning, StringRef Context, const DWARFDie *DIE) {
329 reportWarning(Warning, Context, DIE);
330 });
331 GeneralLinker.setErrorHandler(
332 [&](const Twine &Error, StringRef Context, const DWARFDie *) {
333 error(Error, Context);
334 });
335 GeneralLinker.setObjFileLoader(
336 [&DebugMap, &RL, this](StringRef ContainerName,
337 StringRef Path) -> ErrorOr<DWARFFile &> {
338 auto &Obj = DebugMap.addDebugMapObject(
339 Path, sys::TimePoint<std::chrono::seconds>(), MachO::N_OSO);
340
341 if (auto ErrorOrObj = loadObject(Obj, DebugMap, RL)) {
342 return *ErrorOrObj;
343 } else {
344 // Try and emit more helpful warnings by applying some heuristics.
345 StringRef ObjFile = ContainerName;
346 bool IsClangModule = sys::path::extension(Path).equals(".pcm");
347 bool IsArchive = ObjFile.endswith(")");
348
349 if (IsClangModule) {
350 StringRef ModuleCacheDir = sys::path::parent_path(Path);
351 if (sys::fs::exists(ModuleCacheDir)) {
352 // If the module's parent directory exists, we assume that the
353 // module cache has expired and was pruned by clang. A more
354 // adventurous dsymutil would invoke clang to rebuild the module
355 // now.
356 if (!ModuleCacheHintDisplayed) {
357 WithColor::note()
358 << "The clang module cache may have expired since "
359 "this object file was built. Rebuilding the "
360 "object file will rebuild the module cache.\n";
361 ModuleCacheHintDisplayed = true;
362 }
363 } else if (IsArchive) {
364 // If the module cache directory doesn't exist at all and the
365 // object file is inside a static library, we assume that the
366 // static library was built on a different machine. We don't want
367 // to discourage module debugging for convenience libraries within
368 // a project though.
369 if (!ArchiveHintDisplayed) {
370 WithColor::note()
371 << "Linking a static library that was built with "
372 "-gmodules, but the module cache was not found. "
373 "Redistributable static libraries should never be "
374 "built with module debugging enabled. The debug "
375 "experience will be degraded due to incomplete "
376 "debug information.\n";
377 ArchiveHintDisplayed = true;
378 }
379 }
380 }
381
382 return ErrorOrObj.getError();
383 }
384
385 llvm_unreachable("Unhandled DebugMap object");
386 });
387 GeneralLinker.setSwiftInterfacesMap(&ParseableSwiftInterfaces);
388
389 for (const auto &Obj : Map.objects()) {
390 // N_AST objects (swiftmodule files) should get dumped directly into the
391 // appropriate DWARF section.
392 if (Obj->getType() == MachO::N_AST) {
393 if (Options.Verbose)
394 outs() << "DEBUG MAP OBJECT: " << Obj->getObjectFilename() << "\n";
395
396 StringRef File = Obj->getObjectFilename();
397 auto ErrorOrMem = MemoryBuffer::getFile(File);
398 if (!ErrorOrMem) {
399 warn("Could not open '" + File + "'\n");
400 continue;
401 }
402 sys::fs::file_status Stat;
403 if (auto Err = sys::fs::status(File, Stat)) {
404 warn(Err.message());
405 continue;
406 }
407 if (!Options.NoTimestamp) {
408 // The modification can have sub-second precision so we need to cast
409 // away the extra precision that's not present in the debug map.
410 auto ModificationTime =
411 std::chrono::time_point_cast<std::chrono::seconds>(
412 Stat.getLastModificationTime());
413 if (ModificationTime != Obj->getTimestamp()) {
414 // Not using the helper here as we can easily stream TimePoint<>.
415 WithColor::warning()
416 << File << ": timestamp mismatch between swift interface file ("
417 << sys::TimePoint<>(Obj->getTimestamp()) << ") and debug map ("
418 << sys::TimePoint<>(Obj->getTimestamp()) << ")\n";
419 continue;
420 }
421 }
422
423 // Copy the module into the .swift_ast section.
424 if (!Options.NoOutput)
425 Streamer->emitSwiftAST((*ErrorOrMem)->getBuffer());
426
427 continue;
428 }
429
430 if (auto ErrorOrObj = loadObject(*Obj, Map, RL))
431 GeneralLinker.addObjectFile(*ErrorOrObj);
432 else {
433 ObjectsForLinking.push_back(std::make_unique<DWARFFile>(
434 Obj->getObjectFilename(), nullptr, nullptr,
435 Obj->empty() ? Obj->getWarnings() : EmptyWarnings));
436 GeneralLinker.addObjectFile(*ObjectsForLinking.back());
437 }
438 }
439
440 // link debug info for loaded object files.
441 GeneralLinker.link();
442
443 StringRef ArchName = Map.getTriple().getArchName();
444 if (Error E = emitRemarks(Options, Map.getBinaryPath(), ArchName, RL))
445 return error(toString(std::move(E)));
446
447 if (Options.NoOutput)
448 return true;
449
450 if (Options.ResourceDir && !ParseableSwiftInterfaces.empty()) {
451 StringRef ArchName = Triple::getArchTypeName(Map.getTriple().getArch());
452 if (auto E =
453 copySwiftInterfaces(ParseableSwiftInterfaces, ArchName, Options))
454 return error(toString(std::move(E)));
455 }
456
457 if (Map.getTriple().isOSDarwin() && !Map.getBinaryPath().empty() &&
458 Options.FileType == OutputFileType::Object)
459 return MachOUtils::generateDsymCompanion(
460 Options.VFS, Map, Options.Translator,
461 *Streamer->getAsmPrinter().OutStreamer, OutFile);
462
463 Streamer->finish();
464 return true;
465}

467static bool isMachOPairedReloc(uint64_t RelocType, uint64_t Arch) {
468 switch (Arch) {
469 case Triple::x86:
470 return RelocType == MachO::GENERIC_RELOC_SECTDIFF ||
471 RelocType == MachO::GENERIC_RELOC_LOCAL_SECTDIFF;
472 case Triple::x86_64:
473 return RelocType == MachO::X86_64_RELOC_SUBTRACTOR;
474 case Triple::arm:
475 case Triple::thumb:
476 return RelocType == MachO::ARM_RELOC_SECTDIFF ||
477 RelocType == MachO::ARM_RELOC_LOCAL_SECTDIFF ||
478 RelocType == MachO::ARM_RELOC_HALF ||
479 RelocType == MachO::ARM_RELOC_HALF_SECTDIFF;
480 case Triple::aarch64:
481 return RelocType == MachO::ARM64_RELOC_SUBTRACTOR;
482 default:
483 return false;
484 }

487/// Iterate over the relocations of the given \p Section and
488/// store the ones that correspond to debug map entries into the
489/// ValidRelocs array.
490void DwarfLinkerForBinary::AddressManager::findValidRelocsMachO(
491 const object::SectionRef &Section, const object::MachOObjectFile &Obj,
492 const DebugMapObject &DMO, std::vector<ValidReloc> &ValidRelocs) {
493 Expected<StringRef> ContentsOrErr = Section.getContents();
494 if (!ContentsOrErr) {
495 consumeError(ContentsOrErr.takeError());
496 Linker.reportWarning("error reading section", DMO.getObjectFilename());
497 return;
498 }
499 DataExtractor Data(*ContentsOrErr, Obj.isLittleEndian(), 0);
500 bool SkipNext = false;
501
502 for (const object::RelocationRef &Reloc : Section.relocations()) {
503 if (SkipNext) {
504 SkipNext = false;
505 continue;
506 }
507
508 object::DataRefImpl RelocDataRef = Reloc.getRawDataRefImpl();
509 MachO::any_relocation_info MachOReloc = Obj.getRelocation(RelocDataRef);
510
511 if (isMachOPairedReloc(Obj.getAnyRelocationType(MachOReloc),
512 Obj.getArch())) {
513 SkipNext = true;
514 Linker.reportWarning("unsupported relocation in " + *Section.getName() +
515 " section.",
516 DMO.getObjectFilename());
517 continue;
518 }
519
520 unsigned RelocSize = 1 << Obj.getAnyRelocationLength(MachOReloc);
521 uint64_t Offset64 = Reloc.getOffset();
522 if ((RelocSize != 4 && RelocSize != 8)) {
523 Linker.reportWarning("unsupported relocation in " + *Section.getName() +
524 " section.",
525 DMO.getObjectFilename());
526 continue;
527 }
528 uint64_t OffsetCopy = Offset64;
529 // Mach-o uses REL relocations, the addend is at the relocation offset.
530 uint64_t Addend = Data.getUnsigned(&OffsetCopy, RelocSize);
531 uint64_t SymAddress;
532 int64_t SymOffset;
533
534 if (Obj.isRelocationScattered(MachOReloc)) {
535 // The address of the base symbol for scattered relocations is
536 // stored in the reloc itself. The actual addend will store the
537 // base address plus the offset.
538 SymAddress = Obj.getScatteredRelocationValue(MachOReloc);
539 SymOffset = int64_t(Addend) - SymAddress;
540 } else {
541 SymAddress = Addend;
542 SymOffset = 0;
543 }
544
545 auto Sym = Reloc.getSymbol();
546 if (Sym != Obj.symbol_end()) {
547 Expected<StringRef> SymbolName = Sym->getName();
548 if (!SymbolName) {
549 consumeError(SymbolName.takeError());
550 Linker.reportWarning("error getting relocation symbol name.",
551 DMO.getObjectFilename());
552 continue;
553 }
554 if (const auto *Mapping = DMO.lookupSymbol(*SymbolName))
555 ValidRelocs.emplace_back(Offset64, RelocSize, Addend, Mapping);
556 } else if (const auto *Mapping = DMO.lookupObjectAddress(SymAddress)) {
557 // Do not store the addend. The addend was the address of the symbol in
558 // the object file, the address in the binary that is stored in the debug
559 // map doesn't need to be offset.
560 ValidRelocs.emplace_back(Offset64, RelocSize, SymOffset, Mapping);
561 }
562 }
563}

565/// Dispatch the valid relocation finding logic to the
566/// appropriate handler depending on the object file format.
567bool DwarfLinkerForBinary::AddressManager::findValidRelocs(
568 const object::SectionRef &Section, const object::ObjectFile &Obj,
569 const DebugMapObject &DMO, std::vector<ValidReloc> &Relocs) {
570 // Dispatch to the right handler depending on the file type.
571 if (auto *MachOObj = dyn_cast<object::MachOObjectFile>(&Obj))
572 findValidRelocsMachO(Section, *MachOObj, DMO, Relocs);
573 else
574 Linker.reportWarning(Twine("unsupported object file type: ") +
575 Obj.getFileName(),
576 DMO.getObjectFilename());
577 if (Relocs.empty())
578 return false;
579
580 // Sort the relocations by offset. We will walk the DIEs linearly in
581 // the file, this allows us to just keep an index in the relocation
582 // array that we advance during our walk, rather than resorting to
583 // some associative container. See DwarfLinkerForBinary::NextValidReloc.
584 llvm::sort(Relocs);
585 return true;
586}

588/// Look for relocations in the debug_info and debug_addr section that match
589/// entries in the debug map. These relocations will drive the Dwarf link by
590/// indicating which DIEs refer to symbols present in the linked binary.
591/// \returns whether there are any valid relocations in the debug info.
592bool DwarfLinkerForBinary::AddressManager::findValidRelocsInDebugSections(
593 const object::ObjectFile &Obj, const DebugMapObject &DMO) {
594 // Find the debug_info section.
595 bool FoundValidRelocs = false;
596 for (const object::SectionRef &Section : Obj.sections()) {
597 StringRef SectionName;
598 if (Expected<StringRef> NameOrErr = Section.getName())
599 SectionName = *NameOrErr;
600 else
601 consumeError(NameOrErr.takeError());
602
603 SectionName = SectionName.substr(SectionName.find_first_not_of("._"));
604 if (SectionName == "debug_info")
605 FoundValidRelocs |=
606 findValidRelocs(Section, Obj, DMO, ValidDebugInfoRelocs);
607 if (SectionName == "debug_addr")
608 FoundValidRelocs |=
609 findValidRelocs(Section, Obj, DMO, ValidDebugAddrRelocs);
610 }
611 return FoundValidRelocs;
612}

614bool DwarfLinkerForBinary::AddressManager::hasValidDebugAddrRelocationAt(
615 uint64_t Offset) {
616 auto It = std::lower_bound(ValidDebugAddrRelocs.begin(),
617 ValidDebugAddrRelocs.end(), Offset);
618 return It != ValidDebugAddrRelocs.end();

621bool DwarfLinkerForBinary::AddressManager::hasValidDebugInfoRelocationAt(
622 uint64_t StartOffset, uint64_t EndOffset, CompileUnit::DIEInfo &Info) {
623 assert(NextValidReloc == 0 ||
624 StartOffset > ValidDebugInfoRelocs[NextValidReloc - 1].Offset);
625 if (NextValidReloc >= ValidDebugInfoRelocs.size())
626 return false;
627
628 uint64_t RelocOffset = ValidDebugInfoRelocs[NextValidReloc].Offset;
629
630 // We might need to skip some relocs that we didn't consider. For
631 // example the high_pc of a discarded DIE might contain a reloc that
632 // is in the list because it actually corresponds to the start of a
633 // function that is in the debug map.
634 while (RelocOffset < StartOffset &&
635 NextValidReloc < ValidDebugInfoRelocs.size() - 1)
636 RelocOffset = ValidDebugInfoRelocs[++NextValidReloc].Offset;
637
638 if (RelocOffset < StartOffset || RelocOffset >= EndOffset)
639 return false;
640
641 const auto &ValidReloc = ValidDebugInfoRelocs[NextValidReloc++];
642 const auto &Mapping = ValidReloc.Mapping->getValue();
643 const uint64_t BinaryAddress = Mapping.BinaryAddress;
644 const uint64_t ObjectAddress = Mapping.ObjectAddress
645 ? uint64_t(*Mapping.ObjectAddress)
646 : std::numeric_limits<uint64_t>::max();
647 if (Linker.Options.Verbose)
648 outs() << "Found valid debug map entry: " << ValidReloc.Mapping->getKey()
649 << "\t"
650 << format("0x%016" PRIx64 " => 0x%016" PRIx64 "\n", ObjectAddress,
651 BinaryAddress);
652
653 Info.AddrAdjust = BinaryAddress + ValidReloc.Addend;
654 if (Mapping.ObjectAddress)
655 Info.AddrAdjust -= ObjectAddress;
656 Info.InDebugMap = true;
657 return true;
658}
659
660/// Get the starting and ending (exclusive) offset for the
661/// attribute with index \p Idx descibed by \p Abbrev. \p Offset is
662/// supposed to point to the position of the first attribute described
663/// by \p Abbrev.
664/// \return [StartOffset, EndOffset) as a pair.
665static std::pair<uint64_t, uint64_t>
666getAttributeOffsets(const DWARFAbbreviationDeclaration *Abbrev, unsigned Idx,
667 uint64_t Offset, const DWARFUnit &Unit) {
668 DataExtractor Data = Unit.getDebugInfoExtractor();
669
670 for (unsigned I = 0; I < Idx; ++I)
671 DWARFFormValue::skipValue(Abbrev->getFormByIndex(I), Data, &Offset,
672 Unit.getFormParams());
673
674 uint64_t End = Offset;
675 DWARFFormValue::skipValue(Abbrev->getFormByIndex(Idx), Data, &End,
676 Unit.getFormParams());
677
678 return std::make_pair(Offset, End);
679}

681bool DwarfLinkerForBinary::AddressManager::hasLiveMemoryLocation(
682 const DWARFDie &DIE, CompileUnit::DIEInfo &MyInfo) {
683 const auto *Abbrev = DIE.getAbbreviationDeclarationPtr();
684
685 Optional<uint32_t> LocationIdx =
686 Abbrev->findAttributeIndex(dwarf::DW_AT_location);
687 if (!LocationIdx)
688 return false;
689
690 uint64_t Offset = DIE.getOffset() + getULEB128Size(Abbrev->getCode());
691 uint64_t LocationOffset, LocationEndOffset;
692 std::tie(LocationOffset, LocationEndOffset) =
693 getAttributeOffsets(Abbrev, *LocationIdx, Offset, *DIE.getDwarfUnit());
694
695 // FIXME: Support relocations debug_addr.
696 return hasValidDebugInfoRelocationAt(LocationOffset, LocationEndOffset,
697 MyInfo);
698}

700bool DwarfLinkerForBinary::AddressManager::hasLiveAddressRange(
701 const DWARFDie &DIE, CompileUnit::DIEInfo &MyInfo) {
702 const auto *Abbrev = DIE.getAbbreviationDeclarationPtr();
703
704 Optional<uint32_t> LowPcIdx = Abbrev->findAttributeIndex(dwarf::DW_AT_low_pc);
705 if (!LowPcIdx)
706 return false;
707
708 dwarf::Form Form = Abbrev->getFormByIndex(*LowPcIdx);
709
710 if (Form == dwarf::DW_FORM_addr) {
711 uint64_t Offset = DIE.getOffset() + getULEB128Size(Abbrev->getCode());
712 uint64_t LowPcOffset, LowPcEndOffset;
713 std::tie(LowPcOffset, LowPcEndOffset) =
714 getAttributeOffsets(Abbrev, *LowPcIdx, Offset, *DIE.getDwarfUnit());
715 return hasValidDebugInfoRelocationAt(LowPcOffset, LowPcEndOffset, MyInfo);
716 }
717
718 if (Form == dwarf::DW_FORM_addrx) {
719 Optional<DWARFFormValue> AddrValue = DIE.find(dwarf::DW_AT_low_pc);
720 return hasValidDebugAddrRelocationAt(*AddrValue->getAsAddress());
721 }
722
723 return false;
724}
725/// Apply the valid relocations found by findValidRelocs() to
726/// the buffer \p Data, taking into account that Data is at \p BaseOffset
727/// in the debug_info section.
728///
729/// Like for findValidRelocs(), this function must be called with
730/// monotonic \p BaseOffset values.
731///
732/// \returns whether any reloc has been applied.
733bool DwarfLinkerForBinary::AddressManager::applyValidRelocs(
734 MutableArrayRef<char> Data, uint64_t BaseOffset, bool IsLittleEndian) {
735 assert(areRelocationsResolved());
736 assert((NextValidReloc == 0 ||
737 BaseOffset > ValidDebugInfoRelocs[NextValidReloc - 1].Offset) &&
738 "BaseOffset should only be increasing.");
739 if (NextValidReloc >= ValidDebugInfoRelocs.size())
740 return false;
741
742 // Skip relocs that haven't been applied.
743 while (NextValidReloc < ValidDebugInfoRelocs.size() &&
744 ValidDebugInfoRelocs[NextValidReloc].Offset < BaseOffset)
745 ++NextValidReloc;
746
747 bool Applied = false;
748 uint64_t EndOffset = BaseOffset + Data.size();
749 while (NextValidReloc < ValidDebugInfoRelocs.size() &&
750 ValidDebugInfoRelocs[NextValidReloc].Offset >= BaseOffset &&
751 ValidDebugInfoRelocs[NextValidReloc].Offset < EndOffset) {
752 const auto &ValidReloc = ValidDebugInfoRelocs[NextValidReloc++];
753 assert(ValidReloc.Offset - BaseOffset < Data.size());
754 assert(ValidReloc.Offset - BaseOffset + ValidReloc.Size <= Data.size());
755 char Buf[8];
756 uint64_t Value = ValidReloc.Mapping->getValue().BinaryAddress;
757 Value += ValidReloc.Addend;
758 for (unsigned I = 0; I != ValidReloc.Size; ++I) {
759 unsigned Index = IsLittleEndian ? I : (ValidReloc.Size - I - 1);
760 Buf[I] = uint8_t(Value >> (Index * 8));
761 }
762 assert(ValidReloc.Size <= sizeof(Buf));
763 memcpy(&Data[ValidReloc.Offset - BaseOffset], Buf, ValidReloc.Size);
764 Applied = true;
765 }
766
767 return Applied;
768}

770llvm::Expected<uint64_t>
771DwarfLinkerForBinary::AddressManager::relocateIndexedAddr(uint64_t Offset) {
772 auto It = std::lower_bound(ValidDebugAddrRelocs.begin(),
773 ValidDebugAddrRelocs.end(), Offset);
774 if (It == ValidDebugAddrRelocs.end())
775 return createStringError(
776 std::make_error_code(std::errc::invalid_argument),
777 "no relocation for offset %llu in debug_addr section", Offset);
778 return It->Mapping->getValue().BinaryAddress + It->Addend;
779}
