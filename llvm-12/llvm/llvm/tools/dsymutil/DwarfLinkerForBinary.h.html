26/// The core of the Dsymutil Dwarf linking logic.
27///
28/// The link of the dwarf information from the object files will be
29/// driven by DWARFLinker. DwarfLinkerForBinary reads DebugMap objects
30/// and pass information to the DWARFLinker. DWARFLinker
31/// optimizes DWARF taking into account valid relocations.
32/// Finally, optimized DWARF is passed to DwarfLinkerForBinary through
33/// DWARFEmitter interface.
34class DwarfLinkerForBinary {
35public:
36 DwarfLinkerForBinary(raw_fd_ostream &OutFile, BinaryHolder &BinHolder,
37 LinkOptions Options)
38 : OutFile(OutFile), BinHolder(BinHolder), Options(std::move(Options)) {}
39
40 /// Link the contents of the DebugMap.
41 bool link(const DebugMap &);
42
43 void reportWarning(const Twine &Warning, StringRef Context,
44 const DWARFDie *DIE = nullptr) const;
45
46 /// Flags passed to DwarfLinker::lookForDIEsToKeep
47 enum TraversalFlags {
48 TF_Keep = 1 << 0, ///< Mark the traversed DIEs as kept.
49 TF_InFunctionScope = 1 << 1, ///< Current scope is a function scope.
50 TF_DependencyWalk = 1 << 2, ///< Walking the dependencies of a kept DIE.
51 TF_ParentWalk = 1 << 3, ///< Walking up the parents of a kept DIE.
52 TF_ODR = 1 << 4, ///< Use the ODR while keeping dependents.
53 TF_SkipPC = 1 << 5, ///< Skip all location attributes.
54 };
55
56private:
57
58 /// Keeps track of relocations.
59 class AddressManager : public AddressesMap {
60 struct ValidReloc {
61 uint64_t Offset;
62 uint32_t Size;
63 uint64_t Addend;
64 const DebugMapObject::DebugMapEntry *Mapping;
65
66 ValidReloc(uint64_t Offset, uint32_t Size, uint64_t Addend,
67 const DebugMapObject::DebugMapEntry *Mapping)
68 : Offset(Offset), Size(Size), Addend(Addend), Mapping(Mapping) {}
69
70 bool operator<(const ValidReloc &RHS) const {
71 return Offset < RHS.Offset;
72 }
73 bool operator<(uint64_t RHS) const { return Offset < RHS; }
74 };
75
76 const DwarfLinkerForBinary &Linker;
77
78 /// The valid relocations for the current DebugMapObject.
79 /// This vector is sorted by relocation offset.
80 /// {
81 std::vector<ValidReloc> ValidDebugInfoRelocs;
82 std::vector<ValidReloc> ValidDebugAddrRelocs;
83 /// }
84
85 /// Index into ValidRelocs of the next relocation to consider. As we walk
86 /// the DIEs in acsending file offset and as ValidRelocs is sorted by file
87 /// offset, keeping this index up to date is all we have to do to have a
88 /// cheap lookup during the root DIE selection and during DIE cloning.
89 unsigned NextValidReloc = 0;
90
91 RangesTy AddressRanges;
92
93 public:
94 AddressManager(DwarfLinkerForBinary &Linker, const object::ObjectFile &Obj,
95 const DebugMapObject &DMO)
96 : Linker(Linker) {
97 findValidRelocsInDebugSections(Obj, DMO);
98
99 // Iterate over the debug map entries and put all the ones that are
100 // functions (because they have a size) into the Ranges map. This map is
101 // very similar to the FunctionRanges that are stored in each unit, with 2
102 // notable differences:
103 //
104 // 1. Obviously this one is global, while the other ones are per-unit.
105 //
106 // 2. This one contains not only the functions described in the DIE
107 // tree, but also the ones that are only in the debug map.
108 //
109 // The latter information is required to reproduce dsymutil's logic while
110 // linking line tables. The cases where this information matters look like
111 // bugs that need to be investigated, but for now we need to reproduce
112 // dsymutil's behavior.
113 // FIXME: Once we understood exactly if that information is needed,
114 // maybe totally remove this (or try to use it to do a real
115 // -gline-tables-only on Darwin.
116 for (const auto &Entry : DMO.symbols()) {
117 const auto &Mapping = Entry.getValue();
118 if (Mapping.Size && Mapping.ObjectAddress)
119 AddressRanges[*Mapping.ObjectAddress] = ObjFileAddressRange(
120 *Mapping.ObjectAddress + Mapping.Size,
121 int64_t(Mapping.BinaryAddress) - *Mapping.ObjectAddress);
122 }
123 }
124 virtual ~AddressManager() override { clear(); }
125
126 virtual bool areRelocationsResolved() const override { return true; }
127
128 bool hasValidRelocs(bool ResetRelocsPtr = true) override {
129 if (ResetRelocsPtr)
130 NextValidReloc = 0;
131 return !ValidDebugInfoRelocs.empty() || !ValidDebugAddrRelocs.empty();
132 }
133
134 /// \defgroup FindValidRelocations Translate debug map into a list
135 /// of relevant relocations
136 ///
137 /// @{
138 bool findValidRelocsInDebugSections(const object::ObjectFile &Obj,
139 const DebugMapObject &DMO);
140
141 bool findValidRelocs(const object::SectionRef &Section,
142 const object::ObjectFile &Obj,
143 const DebugMapObject &DMO,
144 std::vector<ValidReloc> &ValidRelocs);
145
146 void findValidRelocsMachO(const object::SectionRef &Section,
147 const object::MachOObjectFile &Obj,
148 const DebugMapObject &DMO,
149 std::vector<ValidReloc> &ValidRelocs);
150 /// @}
151
152 /// Checks that there is a relocation in the debug_addr section against a
153 /// debug map entry between \p StartOffset and \p NextOffset.
154 ///
155 /// This function must be called with offsets in strictly ascending order
156 /// because it never looks back at relocations it already 'went past'.
157 /// \returns true and sets Info.InDebugMap if it is the case.
158 bool hasValidDebugInfoRelocationAt(uint64_t StartOffset, uint64_t EndOffset,
159 CompileUnit::DIEInfo &Info);
160
161 /// Checks that there is a relocation in the debug_addr section against a
162 /// debug map entry at the given offset.
163 bool hasValidDebugAddrRelocationAt(uint64_t Offset);
164
165 bool hasLiveMemoryLocation(const DWARFDie &DIE,
166 CompileUnit::DIEInfo &Info) override;
167 bool hasLiveAddressRange(const DWARFDie &DIE,
168 CompileUnit::DIEInfo &Info) override;
169
170 bool applyValidRelocs(MutableArrayRef<char> Data, uint64_t BaseOffset,
171 bool IsLittleEndian) override;
172
173 llvm::Expected<uint64_t> relocateIndexedAddr(uint64_t Offset) override;
174
175 RangesTy &getValidAddressRanges() override { return AddressRanges; }
176
177 void clear() override {
178 AddressRanges.clear();
179 ValidDebugInfoRelocs.clear();
180 ValidDebugAddrRelocs.clear();
181 NextValidReloc = 0;
182 }
183 };

213 bool ModuleCacheHintDisplayed = false;
