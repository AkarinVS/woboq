9// This tool verifies Control Flow Integrity (CFI) instrumentation by static
10// binary anaylsis. See the design document in /docs/CFIVerify.rst for more
11// information.
12//
13// This tool is currently incomplete. It currently only does disassembly for
14// object files, and searches through the code for indirect control flow
15// instructions, printing them once found.

35cl::opt<std::string> InputFilename(cl::Positional, cl::desc("<input file>"),
36 cl::Required);
37cl::opt<std::string> BlacklistFilename(cl::Positional,
38 cl::desc("[blacklist file]"),
39 cl::init("-"));
40cl::opt<bool> PrintGraphs(
41 "print-graphs",
42 cl::desc("Print graphs around indirect CF instructions in DOT format."),
43 cl::init(false));
44cl::opt<unsigned> PrintBlameContext(
45 "blame-context",
46 cl::desc("Print the blame context (if possible) for BAD instructions. This "
47 "specifies the number of lines of context to include, where zero "
48 "disables this feature."),
49 cl::init(0));
50cl::opt<unsigned> PrintBlameContextAll(
51 "blame-context-all",
52 cl::desc("Prints the blame context (if possible) for ALL instructions. "
53 "This specifies the number of lines of context for non-BAD "
54 "instructions (see --blame-context). If --blame-context is "
55 "unspecified, it prints this number of contextual lines for BAD "
56 "instructions as well."),
57 cl::init(0));
58cl::opt<bool> Summarize("summarize", cl::desc("Print the summary only."),
59 cl::init(false));

61ExitOnError ExitOnErr;

63static void printBlameContext(const DILineInfo &LineInfo, unsigned Context) {
64 auto FileOrErr = MemoryBuffer::getFile(LineInfo.FileName);
65 if (!FileOrErr) {
66 errs() << "Could not open file: " << LineInfo.FileName << "\n";
67 return;
68 }
69
70 std::unique_ptr<MemoryBuffer> File = std::move(FileOrErr.get());
71 SmallVector<StringRef, 100> Lines;
72 File->getBuffer().split(Lines, '\n');
73
74 for (unsigned i = std::max<size_t>(1, LineInfo.Line - Context);
75 i <
76 std::min<size_t>(Lines.size() + 1, LineInfo.Line + Context + 1);
77 ++i) {
78 if (i == LineInfo.Line)
79 outs() << ">";
80 else
81 outs() << " ";
82
83 outs() << i << ": " << Lines[i - 1] << "\n";
84 }
85}

87static void printInstructionInformation(const FileAnalysis &Analysis,
88 const Instr &InstrMeta,
89 const GraphResult &Graph,
90 CFIProtectionStatus ProtectionStatus) {
91 outs() << "Instruction: " << format_hex(InstrMeta.VMAddress, 2) << " ("
92 << stringCFIProtectionStatus(ProtectionStatus) << "): ";
93 Analysis.printInstruction(InstrMeta, outs());
94 outs() << " \n";
95
96 if (PrintGraphs)
97 Graph.printToDOT(Analysis, outs());
98}
99
100static void printInstructionStatus(unsigned BlameLine, bool CFIProtected,
101 const DILineInfo &LineInfo) {
102 if (BlameLine) {
103 outs() << "Blacklist Match: " << BlacklistFilename << ":" << BlameLine
104 << "\n";
105 if (CFIProtected)
106 outs() << "====> Unexpected Protected\n";
107 else
108 outs() << "====> Expected Unprotected\n";
109
110 if (PrintBlameContextAll)
111 printBlameContext(LineInfo, PrintBlameContextAll);
112 } else {
113 if (CFIProtected) {
114 outs() << "====> Expected Protected\n";
115 if (PrintBlameContextAll)
116 printBlameContext(LineInfo, PrintBlameContextAll);
117 } else {
118 outs() << "====> Unexpected Unprotected (BAD)\n";
119 if (PrintBlameContext)
120 printBlameContext(LineInfo, PrintBlameContext);
121 }
122 }
123}

125static void
126printIndirectCFInstructions(FileAnalysis &Analysis,
127 const SpecialCaseList *SpecialCaseList) {
128 uint64_t ExpectedProtected = 0;
129 uint64_t UnexpectedProtected = 0;
130 uint64_t ExpectedUnprotected = 0;
131 uint64_t UnexpectedUnprotected = 0;
132
133 std::map<unsigned, uint64_t> BlameCounter;
134
135 for (object::SectionedAddress Address : Analysis.getIndirectInstructions()) {
136 const auto &InstrMeta = Analysis.getInstructionOrDie(Address.Address);
137 GraphResult Graph = GraphBuilder::buildFlowGraph(Analysis, Address);
138
139 CFIProtectionStatus ProtectionStatus =
140 Analysis.validateCFIProtection(Graph);
141 bool CFIProtected = (ProtectionStatus == CFIProtectionStatus::PROTECTED);
142
143 if (!Summarize) {
144 outs() << "-----------------------------------------------------\n";
145 printInstructionInformation(Analysis, InstrMeta, Graph, ProtectionStatus);
146 }
147
148 if (IgnoreDWARFFlag) {
149 if (CFIProtected)
150 ExpectedProtected++;
151 else
152 UnexpectedUnprotected++;
153 continue;
154 }
155
156 auto InliningInfo = Analysis.symbolizeInlinedCode(Address);
157 if (!InliningInfo || InliningInfo->getNumberOfFrames() == 0) {
158 errs() << "Failed to symbolise " << format_hex(Address.Address, 2)
159 << " with line tables from " << InputFilename << "\n";
160 exit(EXIT_FAILURE);
161 }
162
163 const auto &LineInfo = InliningInfo->getFrame(0);
164
165 // Print the inlining symbolisation of this instruction.
166 if (!Summarize) {
167 for (uint32_t i = 0; i < InliningInfo->getNumberOfFrames(); ++i) {
168 const auto &Line = InliningInfo->getFrame(i);
169 outs() << " " << format_hex(Address.Address, 2) << " = "
170 << Line.FileName << ":" << Line.Line << ":" << Line.Column
171 << " (" << Line.FunctionName << ")\n";
172 }
173 }
174
175 if (!SpecialCaseList) {
176 if (CFIProtected) {
177 if (PrintBlameContextAll && !Summarize)
178 printBlameContext(LineInfo, PrintBlameContextAll);
179 ExpectedProtected++;
180 } else {
181 if (PrintBlameContext && !Summarize)
182 printBlameContext(LineInfo, PrintBlameContext);
183 UnexpectedUnprotected++;
184 }
185 continue;
186 }
187
188 unsigned BlameLine = 0;
189 for (auto &K : {"cfi-icall", "cfi-vcall"}) {
190 if (!BlameLine)
191 BlameLine =
192 SpecialCaseList->inSectionBlame(K, "src", LineInfo.FileName);
193 if (!BlameLine)
194 BlameLine =
195 SpecialCaseList->inSectionBlame(K, "fun", LineInfo.FunctionName);
196 }
197
198 if (BlameLine) {
199 BlameCounter[BlameLine]++;
200 if (CFIProtected)
201 UnexpectedProtected++;
202 else
203 ExpectedUnprotected++;
204 } else {
205 if (CFIProtected)
206 ExpectedProtected++;
207 else
208 UnexpectedUnprotected++;
209 }
210
211 if (!Summarize)
212 printInstructionStatus(BlameLine, CFIProtected, LineInfo);
213 }
214
215 uint64_t IndirectCFInstructions = ExpectedProtected + UnexpectedProtected +
216 ExpectedUnprotected + UnexpectedUnprotected;
217
218 if (IndirectCFInstructions == 0) {
219 outs() << "No indirect CF instructions found.\n";
220 return;
221 }
222
223 outs() << formatv("\nTotal Indirect CF Instructions: {0}\n"
224 "Expected Protected: {1} ({2:P})\n"
225 "Unexpected Protected: {3} ({4:P})\n"
226 "Expected Unprotected: {5} ({6:P})\n"
227 "Unexpected Unprotected (BAD): {7} ({8:P})\n",
228 IndirectCFInstructions, ExpectedProtected,
229 ((double)ExpectedProtected) / IndirectCFInstructions,
230 UnexpectedProtected,
231 ((double)UnexpectedProtected) / IndirectCFInstructions,
232 ExpectedUnprotected,
233 ((double)ExpectedUnprotected) / IndirectCFInstructions,
234 UnexpectedUnprotected,
235 ((double)UnexpectedUnprotected) / IndirectCFInstructions);
236
237 if (!SpecialCaseList)
238 return;
239
240 outs() << "\nBlacklist Results:\n";
241 for (const auto &KV : BlameCounter) {
242 outs() << " " << BlacklistFilename << ":" << KV.first << " affects "
243 << KV.second << " indirect CF instructions.\n";
244 }
245}

247int main(int argc, char **argv) {
248 cl::ParseCommandLineOptions(
249 argc, argv,
250 "Identifies whether Control Flow Integrity protects all indirect control "
251 "flow instructions in the provided object file, DSO or binary.\nNote: "
252 "Anything statically linked into the provided file *must* be compiled "
253 "with '-g'. This can be relaxed through the '--ignore-dwarf' flag.");
254
255 InitializeAllTargetInfos();
256 InitializeAllTargetMCs();
257 InitializeAllAsmParsers();
258 InitializeAllDisassemblers();
259
260 if (PrintBlameContextAll && !PrintBlameContext)
261 PrintBlameContext.setValue(PrintBlameContextAll);
262
263 std::unique_ptr<SpecialCaseList> SpecialCaseList;
264 if (BlacklistFilename != "-") {
265 std::string Error;
266 SpecialCaseList = SpecialCaseList::create({BlacklistFilename},
267 *vfs::getRealFileSystem(), Error);
268 if (!SpecialCaseList) {
269 errs() << "Failed to get blacklist: " << Error << "\n";
270 exit(EXIT_FAILURE);
271 }
272 }
273
274 FileAnalysis Analysis = ExitOnErr(FileAnalysis::Create(InputFilename));
275 printIndirectCFInstructions(Analysis, SpecialCaseList.get());
276
277 return EXIT_SUCCESS;
278}
