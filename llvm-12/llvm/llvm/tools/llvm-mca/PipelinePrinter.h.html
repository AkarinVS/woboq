24#define DEBUG_TYPE "llvm-mca"

26namespace llvm {
27namespace mca {
28
29/// A printer class that knows how to collects statistics on the
30/// code analyzed by the llvm-mca tool.
31///
32/// This class knows how to print out the analysis information collected
33/// during the execution of the code. Internally, it delegates to other
34/// classes the task of printing out timeline information as well as
35/// resource pressure.
36class PipelinePrinter {
37 Pipeline &P;
38 llvm::SmallVector<std::unique_ptr<View>, 8> Views;
39 View::OutputKind OutputKind;
40
41public:
42 PipelinePrinter(Pipeline &pipeline, View::OutputKind OutputKind)
43 : P(pipeline), OutputKind(OutputKind) {}
44
45 void addView(std::unique_ptr<View> V) {
46 P.addEventListener(V.get());
47 Views.emplace_back(std::move(V));
48 }
49
50 void printReport(llvm::raw_ostream &OS) const;
51};
52} // namespace mca
53} // namespace llvm

55#endif // LLVM_TOOLS_LLVM_MCA_PIPELINEPRINTER_H
