8/// \file
9///
10/// This file implements the instruction info view.
11///
12/// The goal fo the instruction info view is to print the latency and reciprocal
13/// throughput information for every instruction in the input sequence.
14/// This section also reports extra information related to the number of micro
15/// opcodes, and opcode properties (i.e. 'MayLoad', 'MayStore', 'HasSideEffects)
16///
17/// Example:
18///
19/// Instruction Info:
20/// [1]: #uOps
21/// [2]: Latency
22/// [3]: RThroughput
23/// [4]: MayLoad
24/// [5]: MayStore
25/// [6]: HasSideEffects
26///
27/// [1] [2] [3] [4] [5] [6] Instructions:
28/// 1 2 1.00 vmulps %xmm0, %xmm1, %xmm2
29/// 1 3 1.00 vhaddps %xmm2, %xmm2, %xmm3
30/// 1 3 1.00 vhaddps %xmm3, %xmm3, %xmm4
31//
32//===----------------------------------------------------------------------===//

49namespace llvm {
50namespace mca {
51
52/// A view that prints out generic instruction information.
53class InstructionInfoView : public InstructionView {
54 const llvm::MCInstrInfo &MCII;
55 CodeEmitter &CE;
56 bool PrintEncodings;
57
58 struct InstructionInfoViewData {
59 unsigned NumMicroOpcodes = 0;
60 unsigned Latency = 0;
61 Optional<double> RThroughput = 0.0;
62 bool mayLoad = false;
63 bool mayStore = false;
64 bool hasUnmodeledSideEffects = false;
65 };
66 using IIVDVec = SmallVector<InstructionInfoViewData, 16>;
67
68 /// Place the data into the array of InstructionInfoViewData IIVD.
69 void collectData(MutableArrayRef<InstructionInfoViewData> IIVD) const;
70
71public:
72 InstructionInfoView(const llvm::MCSubtargetInfo &ST,
73 const llvm::MCInstrInfo &II, CodeEmitter &C,
74 bool ShouldPrintEncodings, llvm::ArrayRef<llvm::MCInst> S,
75 llvm::MCInstPrinter &IP)
76 : InstructionView(ST, IP, S), MCII(II), CE(C),
77 PrintEncodings(ShouldPrintEncodings) {}
78
79 void printView(llvm::raw_ostream &OS) const override;
80 StringRef getNameAsString() const override { return "InstructionInfoView"; }
81 json::Value toJSON() const override;
82 json::Object toJSON(const InstructionInfoViewData &IIVD) const;
83};
84} // namespace mca
85} // namespace llvm
