20double Finish(Counter const& c, double cpu_time, double num_threads) {
21 double v = c.value;
22 if (c.flags & Counter::kIsRate) {
23 v /= cpu_time;
24 }
25 if (c.flags & Counter::kAvgThreads) {
26 v /= num_threads;
27 }
28 return v;
29}
30
31void Finish(UserCounters *l, double cpu_time, double num_threads) {
32 for (auto &c : *l) {
33 c.second.value = Finish(c.second, cpu_time, num_threads);
34 }
35}
36
37void Increment(UserCounters *l, UserCounters const& r) {
38 // add counters present in both or just in *l
39 for (auto &c : *l) {
40 auto it = r.find(c.first);
41 if (it != r.end()) {
42 c.second.value = c.second + it->second;
43 }
44 }
45 // add counters present in r, but not in *l
46 for (auto const &tc : r) {
47 auto it = l->find(tc.first);
48 if (it == l->end()) {
49 (*l)[tc.first] = tc.second;
50 }
51 }
52}
53
54bool SameNames(UserCounters const& l, UserCounters const& r) {
55 if (&l == &r) return true;
56 if (l.size() != r.size()) {
57 return false;
58 }
59 for (auto const& c : l) {
60 if (r.find(c.first) == r.end()) {
61 return false;
62 }
63 }
64 return true;
65}
