33// Some synthetic sections (e.g. "name" and "linking") have subsections.
34// Just like the synthetic sections themselves these need to be created before
35// they can be written out (since they are preceded by their length). This
36// class is used to create subsections and then write them into the stream
37// of the parent section.
38class SubSection {
39public:
40 explicit SubSection(uint32_t type) : type(type) {}
41
42 void writeTo(raw_ostream &to) {
43 os.flush();
44 writeUleb128(to, type, "subsection type");
45 writeUleb128(to, body.size(), "subsection size");
46 to.write(body.data(), body.size());
47 }
48
49private:
50 uint32_t type;
51 std::string body;
52
53public:
54 raw_string_ostream os{body};
55};
56
57} // namespace

71uint32_t TypeSection::registerType(const WasmSignature &sig) {
72 auto pair = typeIndices.insert(std::make_pair(sig, types.size()));
73 if (pair.second) {
74 LLVM_DEBUG(llvm::dbgs() << "type " << toString(sig) << "\n");
75 types.push_back(&sig);
76 }
77 return pair.first->second;
78}
79
80uint32_t TypeSection::lookupType(const WasmSignature &sig) {
81 auto it = typeIndices.find(sig);
82 if (it == typeIndices.end()) {
83 error("type not found: " + toString(sig));
84 return 0;
85 }
86 return it->second;
87}
88
89void TypeSection::writeBody() {
90 writeUleb128(bodyOutputStream, types.size(), "type count");
91 for (const WasmSignature *sig : types)
92 writeSig(bodyOutputStream, *sig);
93}

95uint32_t ImportSection::getNumImports() const {
96 assert(isSealed);
97 uint32_t numImports = importedSymbols.size() + gotSymbols.size();
98 if (config->importMemory)
99 ++numImports;
100 return numImports;

103void ImportSection::addGOTEntry(Symbol *sym) {
104 assert(!isSealed);
105 if (sym->hasGOTIndex())
106 return;
107 LLVM_DEBUG(dbgs() << "addGOTEntry: " << toString(*sym) << "\n");
108 sym->setGOTIndex(numImportedGlobals++);
109 gotSymbols.push_back(sym);
110}

112void ImportSection::addImport(Symbol *sym) {
113 assert(!isSealed);
114 importedSymbols.emplace_back(sym);
115 if (auto *f = dyn_cast<FunctionSymbol>(sym))
116 f->setFunctionIndex(numImportedFunctions++);
117 else if (auto *g = dyn_cast<GlobalSymbol>(sym))
118 g->setGlobalIndex(numImportedGlobals++);
119 else if (auto *e = dyn_cast<EventSymbol>(sym))
120 e->setEventIndex(numImportedEvents++);
121 else
122 cast<TableSymbol>(sym)->setTableNumber(numImportedTables++);
123}

125void ImportSection::writeBody() {
126 raw_ostream &os = bodyOutputStream;
127
128 writeUleb128(os, getNumImports(), "import count");
129
130 if (config->importMemory) {
131 WasmImport import;
132 import.Module = defaultModule;
133 import.Field = "memory";
134 import.Kind = WASM_EXTERNAL_MEMORY;
135 import.Memory.Flags = 0;
136 import.Memory.Initial = out.memorySec->numMemoryPages;
137 if (out.memorySec->maxMemoryPages != 0 || config->sharedMemory) {
138 import.Memory.Flags |= WASM_LIMITS_FLAG_HAS_MAX;
139 import.Memory.Maximum = out.memorySec->maxMemoryPages;
140 }
141 if (config->sharedMemory)
142 import.Memory.Flags |= WASM_LIMITS_FLAG_IS_SHARED;
143 if (config->is64.getValueOr(false))
144 import.Memory.Flags |= WASM_LIMITS_FLAG_IS_64;
145 writeImport(os, import);
146 }
147
148 for (const Symbol *sym : importedSymbols) {
149 WasmImport import;
150 if (auto *f = dyn_cast<UndefinedFunction>(sym)) {
151 import.Field = f->importName ? *f->importName : sym->getName();
152 import.Module = f->importModule ? *f->importModule : defaultModule;
153 } else if (auto *g = dyn_cast<UndefinedGlobal>(sym)) {
154 import.Field = g->importName ? *g->importName : sym->getName();
155 import.Module = g->importModule ? *g->importModule : defaultModule;
156 } else if (auto *t = dyn_cast<UndefinedTable>(sym)) {
157 import.Field = t->importName ? *t->importName : sym->getName();
158 import.Module = t->importModule ? *t->importModule : defaultModule;
159 } else {
160 import.Field = sym->getName();
161 import.Module = defaultModule;
162 }
163
164 if (auto *functionSym = dyn_cast<FunctionSymbol>(sym)) {
165 import.Kind = WASM_EXTERNAL_FUNCTION;
166 import.SigIndex = out.typeSec->lookupType(*functionSym->signature);
167 } else if (auto *globalSym = dyn_cast<GlobalSymbol>(sym)) {
168 import.Kind = WASM_EXTERNAL_GLOBAL;
169 import.Global = *globalSym->getGlobalType();
170 } else if (auto *eventSym = dyn_cast<EventSymbol>(sym)) {
171 import.Kind = WASM_EXTERNAL_EVENT;
172 import.Event.Attribute = eventSym->getEventType()->Attribute;
173 import.Event.SigIndex = out.typeSec->lookupType(*eventSym->signature);
174 } else {
175 auto *tableSym = cast<TableSymbol>(sym);
176 import.Kind = WASM_EXTERNAL_TABLE;
177 import.Table = *tableSym->getTableType();
178 }
179 writeImport(os, import);
180 }
181
182 for (const Symbol *sym : gotSymbols) {
183 WasmImport import;
184 import.Kind = WASM_EXTERNAL_GLOBAL;
185 import.Global = {WASM_TYPE_I32, true};
186 if (isa<DataSymbol>(sym))
187 import.Module = "GOT.mem";
188 else
189 import.Module = "GOT.func";
190 import.Field = sym->getName();
191 writeImport(os, import);
192 }
193}

203void FunctionSection::addFunction(InputFunction *func) {
204 if (!func->live)
205 return;
206 uint32_t functionIndex =
207 out.importSec->getNumImportedFunctions() + inputFunctions.size();
208 inputFunctions.emplace_back(func);
209 func->setFunctionIndex(functionIndex);
210}

229void MemorySection::writeBody() {
230 raw_ostream &os = bodyOutputStream;
231
232 bool hasMax = maxMemoryPages != 0 || config->sharedMemory;
233 writeUleb128(os, 1, "memory count");
234 unsigned flags = 0;
235 if (hasMax)
236 flags |= WASM_LIMITS_FLAG_HAS_MAX;
237 if (config->sharedMemory)
238 flags |= WASM_LIMITS_FLAG_IS_SHARED;
239 if (config->is64.getValueOr(false))
240 flags |= WASM_LIMITS_FLAG_IS_64;
241 writeUleb128(os, flags, "memory limits flags");
242 writeUleb128(os, numMemoryPages, "initial pages");
243 if (hasMax)
244 writeUleb128(os, maxMemoryPages, "max pages");
245}

267void GlobalSection::assignIndexes() {
268 uint32_t globalIndex = out.importSec->getNumImportedGlobals();
269 for (InputGlobal *g : inputGlobals)
270 g->setGlobalIndex(globalIndex++);
271 for (Symbol *sym : internalGotSymbols)
272 sym->setGOTIndex(globalIndex++);
273 isSealed = true;
274}

276void GlobalSection::addInternalGOTEntry(Symbol *sym) {
277 assert(!isSealed);
278 if (sym->requiresGOT)
279 return;
280 LLVM_DEBUG(dbgs() << "addInternalGOTEntry: " << sym->getName() << " "
281 << toString(sym->kind()) << "\n");
282 sym->requiresGOT = true;
283 if (auto *F = dyn_cast<FunctionSymbol>(sym))
284 out.elemSec->addEntry(F);
285 internalGotSymbols.push_back(sym);
286}

288void GlobalSection::generateRelocationCode(raw_ostream &os) const {
289 unsigned opcode_ptr_const = config->is64.getValueOr(false)
290 ? WASM_OPCODE_I64_CONST
291 : WASM_OPCODE_I32_CONST;
292 unsigned opcode_ptr_add = config->is64.getValueOr(false)
293 ? WASM_OPCODE_I64_ADD
294 : WASM_OPCODE_I32_ADD;
295
296 for (const Symbol *sym : internalGotSymbols) {
297 if (auto *d = dyn_cast<DefinedData>(sym)) {
298 // Get __memory_base
299 writeU8(os, WASM_OPCODE_GLOBAL_GET, "GLOBAL_GET");
300 writeUleb128(os, WasmSym::memoryBase->getGlobalIndex(), "__memory_base");
301
302 // Add the virtual address of the data symbol
303 writeU8(os, opcode_ptr_const, "CONST");
304 writeSleb128(os, d->getVirtualAddress(), "offset");
305 } else if (auto *f = dyn_cast<FunctionSymbol>(sym)) {
306 if (f->isStub)
307 continue;
308 // Get __table_base
309 writeU8(os, WASM_OPCODE_GLOBAL_GET, "GLOBAL_GET");
310 writeUleb128(os, WasmSym::tableBase->getGlobalIndex(), "__table_base");
311
312 // Add the table index to __table_base
313 writeU8(os, opcode_ptr_const, "CONST");
314 writeSleb128(os, f->getTableIndex(), "offset");
315 } else {
316 assert(isa<UndefinedData>(sym));
317 continue;
318 }
319 writeU8(os, opcode_ptr_add, "ADD");
320 writeU8(os, WASM_OPCODE_GLOBAL_SET, "GLOBAL_SET");
321 writeUleb128(os, sym->getGOTIndex(), "got_entry");
322 }
323}

325void GlobalSection::writeBody() {
326 raw_ostream &os = bodyOutputStream;
327
328 writeUleb128(os, numGlobals(), "global count");
329 for (InputGlobal *g : inputGlobals)
330 writeGlobal(os, g->global);
331 // TODO(wvo): when do these need I64_CONST?
332 for (const Symbol *sym : internalGotSymbols) {
333 WasmGlobal global;
334 // In the case of dynamic linking, internal GOT entries
335 // need to be mutable since they get updated to the correct
336 // runtime value during `__wasm_apply_relocs`.
337 bool mutable_ = config->isPic & !sym->isStub;
338 global.Type = {WASM_TYPE_I32, mutable_};
339 global.InitExpr.Opcode = WASM_OPCODE_I32_CONST;
340 if (auto *d = dyn_cast<DefinedData>(sym))
341 global.InitExpr.Value.Int32 = d->getVirtualAddress();
342 else if (auto *f = dyn_cast<FunctionSymbol>(sym))
343 global.InitExpr.Value.Int32 = f->isStub ? 0 : f->getTableIndex();
344 else {
345 assert(isa<UndefinedData>(sym));
346 global.InitExpr.Value.Int32 = 0;
347 }
348 writeGlobal(os, global);
349 }
350 for (const DefinedData *sym : dataAddressGlobals) {
351 WasmGlobal global;
352 global.Type = {WASM_TYPE_I32, false};
353 global.InitExpr.Opcode = WASM_OPCODE_I32_CONST;
354 global.InitExpr.Value.Int32 = sym->getVirtualAddress();
355 writeGlobal(os, global);
356 }
357}

384void ElemSection::addEntry(FunctionSymbol *sym) {
385 // Don't add stub functions to the wasm table. The address of all stub
386 // functions should be zero and they should they don't appear in the table.
387 // They only exist so that the calls to missing functions can validate.
388 if (sym->hasTableIndex() || sym->isStub)
389 return;
390 sym->setTableIndex(config->tableBase + indirectFunctions.size());
391 indirectFunctions.emplace_back(sym);
392}
393
394void ElemSection::writeBody() {
395 raw_ostream &os = bodyOutputStream;
396
397 writeUleb128(os, 1, "segment count");
398 writeUleb128(os, 0, "table index");
399 WasmInitExpr initExpr;
400 if (config->isPic) {
401 initExpr.Opcode = WASM_OPCODE_GLOBAL_GET;
402 initExpr.Value.Global = WasmSym::tableBase->getGlobalIndex();
403 } else {
404 initExpr.Opcode = WASM_OPCODE_I32_CONST;
405 initExpr.Value.Int32 = config->tableBase;
406 }
407 writeInitExpr(os, initExpr);
408 writeUleb128(os, indirectFunctions.size(), "elem count");
409
410 uint32_t tableIndex = config->tableBase;
411 for (const FunctionSymbol *sym : indirectFunctions) {
412 assert(sym->getTableIndex() == tableIndex);
413 writeUleb128(os, sym->getFunctionIndex(), "function index");
414 ++tableIndex;
415 }
416}

432void LinkingSection::writeBody() {
433 raw_ostream &os = bodyOutputStream;
434
435 writeUleb128(os, WasmMetadataVersion, "Version");
436
437 if (!symtabEntries.empty()) {
438 SubSection sub(WASM_SYMBOL_TABLE);
439 writeUleb128(sub.os, symtabEntries.size(), "num symbols");
440
441 for (const Symbol *sym : symtabEntries) {
442 assert(sym->isDefined() || sym->isUndefined());
443 WasmSymbolType kind = sym->getWasmType();
444 uint32_t flags = sym->flags;
445
446 writeU8(sub.os, kind, "sym kind");
447 writeUleb128(sub.os, flags, "sym flags");
448
449 if (auto *f = dyn_cast<FunctionSymbol>(sym)) {
450 writeUleb128(sub.os, f->getFunctionIndex(), "index");
451 if (sym->isDefined() || (flags & WASM_SYMBOL_EXPLICIT_NAME) != 0)
452 writeStr(sub.os, sym->getName(), "sym name");
453 } else if (auto *g = dyn_cast<GlobalSymbol>(sym)) {
454 writeUleb128(sub.os, g->getGlobalIndex(), "index");
455 if (sym->isDefined() || (flags & WASM_SYMBOL_EXPLICIT_NAME) != 0)
456 writeStr(sub.os, sym->getName(), "sym name");
457 } else if (auto *e = dyn_cast<EventSymbol>(sym)) {
458 writeUleb128(sub.os, e->getEventIndex(), "index");
459 if (sym->isDefined() || (flags & WASM_SYMBOL_EXPLICIT_NAME) != 0)
460 writeStr(sub.os, sym->getName(), "sym name");
461 } else if (auto *t = dyn_cast<TableSymbol>(sym)) {
462 writeUleb128(sub.os, t->getTableNumber(), "table number");
463 if (sym->isDefined() || (flags & WASM_SYMBOL_EXPLICIT_NAME) != 0)
464 writeStr(sub.os, sym->getName(), "sym name");
465 } else if (isa<DataSymbol>(sym)) {
466 writeStr(sub.os, sym->getName(), "sym name");
467 if (auto *dataSym = dyn_cast<DefinedData>(sym)) {
468 writeUleb128(sub.os, dataSym->getOutputSegmentIndex(), "index");
469 writeUleb128(sub.os, dataSym->getOutputSegmentOffset(),
470 "data offset");
471 writeUleb128(sub.os, dataSym->getSize(), "data size");
472 }
473 } else {
474 auto *s = cast<OutputSectionSymbol>(sym);
475 writeUleb128(sub.os, s->section->sectionIndex, "sym section index");
476 }
477 }
478
479 sub.writeTo(os);
480 }
481
482 if (dataSegments.size()) {
483 SubSection sub(WASM_SEGMENT_INFO);
484 writeUleb128(sub.os, dataSegments.size(), "num data segments");
485 for (const OutputSegment *s : dataSegments) {
486 writeStr(sub.os, s->name, "segment name");
487 writeUleb128(sub.os, s->alignment, "alignment");
488 writeUleb128(sub.os, 0, "flags");
489 }
490 sub.writeTo(os);
491 }
492
493 if (!initFunctions.empty()) {
494 SubSection sub(WASM_INIT_FUNCS);
495 writeUleb128(sub.os, initFunctions.size(), "num init functions");
496 for (const WasmInitEntry &f : initFunctions) {
497 writeUleb128(sub.os, f.priority, "priority");
498 writeUleb128(sub.os, f.sym->getOutputSymbolIndex(), "function index");
499 }
500 sub.writeTo(os);
501 }
502
503 struct ComdatEntry {
504 unsigned kind;
505 uint32_t index;
506 };
507 std::map<StringRef, std::vector<ComdatEntry>> comdats;
508
509 for (const InputFunction *f : out.functionSec->inputFunctions) {
510 StringRef comdat = f->getComdatName();
511 if (!comdat.empty())
512 comdats[comdat].emplace_back(
513 ComdatEntry{WASM_COMDAT_FUNCTION, f->getFunctionIndex()});
514 }
515 for (uint32_t i = 0; i < dataSegments.size(); ++i) {
516 const auto &inputSegments = dataSegments[i]->inputSegments;
517 if (inputSegments.empty())
518 continue;
519 StringRef comdat = inputSegments[0]->getComdatName();
520#ifndef NDEBUG
521 for (const InputSegment *isec : inputSegments)
522 assert(isec->getComdatName() == comdat);
523#endif
524 if (!comdat.empty())
525 comdats[comdat].emplace_back(ComdatEntry{WASM_COMDAT_DATA, i});
526 }
527
528 if (!comdats.empty()) {
529 SubSection sub(WASM_COMDAT_INFO);
530 writeUleb128(sub.os, comdats.size(), "num comdats");
531 for (const auto &c : comdats) {
532 writeStr(sub.os, c.first, "comdat name");
533 writeUleb128(sub.os, 0, "comdat flags"); // flags for future use
534 writeUleb128(sub.os, c.second.size(), "num entries");
535 for (const ComdatEntry &entry : c.second) {
536 writeU8(sub.os, entry.kind, "entry kind");
537 writeUleb128(sub.os, entry.index, "entry index");
538 }
539 }
540 sub.writeTo(os);
541 }
542}

549unsigned NameSection::numNamedFunctions() const {
550 unsigned numNames = out.importSec->getNumImportedFunctions();
551
552 for (const InputFunction *f : out.functionSec->inputFunctions)
553 if (!f->getName().empty() || !f->getDebugName().empty())
554 ++numNames;
555
556 return numNames;
557}

559unsigned NameSection::numNamedGlobals() const {
560 unsigned numNames = out.importSec->getNumImportedGlobals();
561
562 for (const InputGlobal *g : out.globalSec->inputGlobals)
563 if (!g->getName().empty())
564 ++numNames;
565
566 numNames += out.globalSec->internalGotSymbols.size();
567 return numNames;
568}

570unsigned NameSection::numNamedDataSegments() const {

580// Create the custom "name" section containing debug symbol names.
581void NameSection::writeBody() {
582 unsigned count = numNamedFunctions();
583 if (count) {
584 SubSection sub(WASM_NAMES_FUNCTION);
585 writeUleb128(sub.os, count, "name count");
586
587 // Function names appear in function index order. As it happens
588 // importedSymbols and inputFunctions are numbered in order with imported
589 // functions coming first.
590 for (const Symbol *s : out.importSec->importedSymbols) {
591 if (auto *f = dyn_cast<FunctionSymbol>(s)) {
592 writeUleb128(sub.os, f->getFunctionIndex(), "func index");
593 writeStr(sub.os, toString(*s), "symbol name");
594 }
595 }
596 for (const InputFunction *f : out.functionSec->inputFunctions) {
597 if (!f->getName().empty()) {
598 writeUleb128(sub.os, f->getFunctionIndex(), "func index");
599 if (!f->getDebugName().empty()) {
600 writeStr(sub.os, f->getDebugName(), "symbol name");
601 } else {
602 writeStr(sub.os, maybeDemangleSymbol(f->getName()), "symbol name");
603 }
604 }
605 }
606 sub.writeTo(bodyOutputStream);
607 }
608
609 count = numNamedGlobals();
610 if (count) {
611 SubSection sub(WASM_NAMES_GLOBAL);
612 writeUleb128(sub.os, count, "name count");
613
614 for (const Symbol *s : out.importSec->importedSymbols) {
615 if (auto *g = dyn_cast<GlobalSymbol>(s)) {
616 writeUleb128(sub.os, g->getGlobalIndex(), "global index");
617 writeStr(sub.os, toString(*s), "symbol name");
618 }
619 }
620 for (const Symbol *s : out.importSec->gotSymbols) {
621 writeUleb128(sub.os, s->getGOTIndex(), "global index");
622 writeStr(sub.os, toString(*s), "symbol name");
623 }
624 for (const InputGlobal *g : out.globalSec->inputGlobals) {
625 if (!g->getName().empty()) {
626 writeUleb128(sub.os, g->getGlobalIndex(), "global index");
627 writeStr(sub.os, maybeDemangleSymbol(g->getName()), "symbol name");
628 }
629 }
630 for (Symbol *s : out.globalSec->internalGotSymbols) {
631 writeUleb128(sub.os, s->getGOTIndex(), "global index");
632 if (isa<FunctionSymbol>(s))
633 writeStr(sub.os, "GOT.func.internal." + toString(*s), "symbol name");
634 else
635 writeStr(sub.os, "GOT.data.internal." + toString(*s), "symbol name");
636 }
637
638 sub.writeTo(bodyOutputStream);
639 }
640
641 count = numNamedDataSegments();
642 if (count) {
643 SubSection sub(WASM_NAMES_DATA_SEGMENT);
644 writeUleb128(sub.os, count, "name count");
645
646 for (OutputSegment *s : segments) {
647 if (!s->name.empty() && !s->isBss) {
648 writeUleb128(sub.os, s->index, "global index");
649 writeStr(sub.os, s->name, "segment name");
650 }
651 }
652
653 sub.writeTo(bodyOutputStream);
654 }
655}

657void ProducersSection::addInfo(const WasmProducerInfo &info) {
658 for (auto &producers :
659 {std::make_pair(&info.Languages, &languages),
660 std::make_pair(&info.Tools, &tools), std::make_pair(&info.SDKs, &sDKs)})
661 for (auto &producer : *producers.first)
662 if (producers.second->end() ==
663 llvm::find_if(*producers.second,
664 [&](std::pair<std::string, std::string> seen) {
665 return seen.first == producer.first;
666 }))
667 producers.second->push_back(producer);
668}

670void ProducersSection::writeBody() {
671 auto &os = bodyOutputStream;
672 writeUleb128(os, fieldCount(), "field count");
673 for (auto &field :
674 {std::make_pair("language", languages),
675 std::make_pair("processed-by", tools), std::make_pair("sdk", sDKs)}) {
676 if (field.second.empty())
677 continue;
678 writeStr(os, field.first, "field name");
679 writeUleb128(os, field.second.size(), "number of entries");
680 for (auto &entry : field.second) {
681 writeStr(os, entry.first, "producer name");
682 writeStr(os, entry.second, "producer version");
683 }
684 }
685}

687void TargetFeaturesSection::writeBody() {
688 SmallVector<std::string, 8> emitted(features.begin(), features.end());
689 llvm::sort(emitted);
690 auto &os = bodyOutputStream;
691 writeUleb128(os, emitted.size(), "feature count");
692 for (auto &feature : emitted) {
693 writeU8(os, WASM_FEATURE_PREFIX_USED, "feature used prefix");
694 writeStr(os, feature, "feature name");
695 }
696}
