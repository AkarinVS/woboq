29std::string toString(const wasm::Symbol &sym) {
30 return maybeDemangleSymbol(sym.getName());
31}

33std::string maybeDemangleSymbol(StringRef name) {
34 // WebAssembly requires caller and callee signatures to match, so we mangle
35 // `main` in the case where we need to pass it arguments.
36 if (name == "__main_argc_argv")
37 return "main";
38 if (wasm::config->demangle)
39 return demangleItanium(name);
40 return std::string(name);
41}

43std::string toString(wasm::Symbol::Kind kind) {
44 switch (kind) {
45 case wasm::Symbol::DefinedFunctionKind:
46 return "DefinedFunction";
47 case wasm::Symbol::DefinedDataKind:
48 return "DefinedData";
49 case wasm::Symbol::DefinedGlobalKind:
50 return "DefinedGlobal";
51 case wasm::Symbol::DefinedTableKind:
52 return "DefinedTable";
53 case wasm::Symbol::DefinedEventKind:
54 return "DefinedEvent";
55 case wasm::Symbol::UndefinedFunctionKind:
56 return "UndefinedFunction";
57 case wasm::Symbol::UndefinedDataKind:
58 return "UndefinedData";
59 case wasm::Symbol::UndefinedGlobalKind:
60 return "UndefinedGlobal";
61 case wasm::Symbol::UndefinedTableKind:
62 return "UndefinedTable";
63 case wasm::Symbol::LazyKind:
64 return "LazyKind";
65 case wasm::Symbol::SectionKind:
66 return "SectionKind";
67 case wasm::Symbol::OutputSectionKind:
68 return "OutputSectionKind";
69 }
70 llvm_unreachable("invalid symbol kind");
71}

96WasmSymbolType Symbol::getWasmType() const {
97 if (isa<FunctionSymbol>(this))
98 return WASM_SYMBOL_TYPE_FUNCTION;
99 if (isa<DataSymbol>(this))
100 return WASM_SYMBOL_TYPE_DATA;
101 if (isa<GlobalSymbol>(this))
102 return WASM_SYMBOL_TYPE_GLOBAL;
103 if (isa<EventSymbol>(this))
104 return WASM_SYMBOL_TYPE_EVENT;
105 if (isa<TableSymbol>(this))
106 return WASM_SYMBOL_TYPE_TABLE;
107 if (isa<SectionSymbol>(this) || isa<OutputSectionSymbol>(this))
108 return WASM_SYMBOL_TYPE_SECTION;
109 llvm_unreachable("invalid symbol kind");
110}

120InputChunk *Symbol::getChunk() const {
121 if (auto *f = dyn_cast<DefinedFunction>(this))
122 return f->function;
123 if (auto *f = dyn_cast<UndefinedFunction>(this))
124 if (f->stubFunction)
125 return f->stubFunction->function;
126 if (auto *d = dyn_cast<DefinedData>(this))
127 return d->segment;
128 return nullptr;
129}

131bool Symbol::isDiscarded() const {
132 if (InputChunk *c = getChunk())
133 return c->discarded;
134 return false;
135}

137bool Symbol::isLive() const {
138 if (auto *g = dyn_cast<DefinedGlobal>(this))
139 return g->global->live;
140 if (auto *e = dyn_cast<DefinedEvent>(this))
141 return e->event->live;
142 if (auto *t = dyn_cast<DefinedTable>(this))
143 return t->table->live;
144 if (InputChunk *c = getChunk())
145 return c->live;
146 return referenced;
147}
148
149void Symbol::markLive() {
150 assert(!isDiscarded());
151 if (file != NULL && isDefined())
152 file->markLive();
153 if (auto *g = dyn_cast<DefinedGlobal>(this))
154 g->global->live = true;
155 if (auto *e = dyn_cast<DefinedEvent>(this))
156 e->event->live = true;
157 if (auto *t = dyn_cast<DefinedTable>(this))
158 t->table->live = true;
159 if (InputChunk *c = getChunk())
160 c->live = true;
161 referenced = true;
162}

176void Symbol::setGOTIndex(uint32_t index) {
177 LLVM_DEBUG(dbgs() << "setGOTIndex " << name << " -> " << index << "\n");
178 assert(gotIndex == INVALID_INDEX);
179 if (config->isPic) {
180 // Any symbol that is assigned a GOT entry must be exported otherwise the
181 // dynamic linker won't be able create the entry that contains it.
182 forceExport = true;
183 }
184 gotIndex = index;
185}

199void Symbol::setHidden(bool isHidden) {
200 LLVM_DEBUG(dbgs() << "setHidden: " << name << " -> " << isHidden << "\n");
201 flags &= ~WASM_SYMBOL_VISIBILITY_MASK;
202 if (isHidden)
203 flags |= WASM_SYMBOL_VISIBILITY_HIDDEN;
204 else
205 flags |= WASM_SYMBOL_VISIBILITY_DEFAULT;
206}
207
208bool Symbol::isExported() const {
209 if (!isDefined() || isLocal())
210 return false;
211
212 if (forceExport || config->exportAll)
213 return true;
214
215 if (config->exportDynamic && !isHidden())
216 return true;
217
218 return flags & WASM_SYMBOL_EXPORTED;
219}

262void FunctionSymbol::setTableIndex(uint32_t index) {
263 // For imports, we set the table index here on the Symbol; for defined
264 // functions we set the index on the InputFunction so that we don't export
265 // the same thing twice (keeps the table size down).
266 if (auto *f = dyn_cast<DefinedFunction>(this)) {
267 f->function->setTableIndex(index);
268 return;
269 }
270 LLVM_DEBUG(dbgs() << "setTableIndex " << name << " -> " << index << "\n");
271 assert(tableIndex == INVALID_INDEX);
272 tableIndex = index;
273}

281uint64_t DefinedData::getVirtualAddress() const {
282 LLVM_DEBUG(dbgs() << "getVirtualAddress: " << getName() << "\n");

285 return offset;

409void printTraceSymbolUndefined(StringRef name, const InputFile* file) {
410 message(toString(file) + ": reference to " + name);
411}

413// Print out a log message for --trace-symbol.
414void printTraceSymbol(Symbol *sym) {
415 // Undefined symbols are traced via printTraceSymbolUndefined
416 if (sym->isUndefined())
417 return;
418
419 std::string s;
420 if (sym->isLazy())
421 s = ": lazy definition of ";
422 else
423 s = ": definition of ";
424
425 message(toString(sym->getFile()) + s + sym->getName());
426}
