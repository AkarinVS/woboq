43// The base class for real symbol classes.
44class Symbol {
45public:
46 enum Kind : uint8_t {
47 DefinedFunctionKind,
48 DefinedDataKind,
49 DefinedGlobalKind,
50 DefinedEventKind,
51 DefinedTableKind,
52 SectionKind,
53 OutputSectionKind,
54 UndefinedFunctionKind,
55 UndefinedDataKind,
56 UndefinedGlobalKind,
57 UndefinedTableKind,
58 LazyKind,
59 };
60
61 Kind kind() const { return symbolKind; }
62
63 bool isDefined() const { return !isLazy() && !isUndefined(); }
64
65 bool isUndefined() const {
66 return symbolKind == UndefinedFunctionKind ||
67 symbolKind == UndefinedDataKind ||
68 symbolKind == UndefinedGlobalKind ||
69 symbolKind == UndefinedTableKind;
70 }
71
72 bool isLazy() const { return symbolKind == LazyKind; }
73
74 bool isLocal() const;
75 bool isWeak() const;
76 bool isHidden() const;
77
78 // Returns true if this symbol exists in a discarded (due to COMDAT) section
79 bool isDiscarded() const;
80
81 // True if this is an undefined weak symbol. This only works once
82 // all input files have been added.
83 bool isUndefWeak() const {
84 // See comment on lazy symbols for details.
85 return isWeak() && (isUndefined() || isLazy());
86 }
87
88 // Returns the symbol name.
89 StringRef getName() const { return name; }
90
91 // Returns the file from which this symbol was created.
92 InputFile *getFile() const { return file; }
93
94 InputChunk *getChunk() const;
95
96 // Indicates that the section or import for this symbol will be included in
97 // the final image.
98 bool isLive() const;
99
100 // Marks the symbol's InputChunk as Live, so that it will be included in the
101 // final image.
102 void markLive();
103
104 void setHidden(bool isHidden);
105
106 // Get/set the index in the output symbol table. This is only used for
107 // relocatable output.
108 uint32_t getOutputSymbolIndex() const;
109 void setOutputSymbolIndex(uint32_t index);
110
111 WasmSymbolType getWasmType() const;
112 bool isExported() const;
113
114 // Indicates that the symbol is used in an __attribute__((used)) directive
115 // or similar.
116 bool isNoStrip() const;
117
118 const WasmSignature* getSignature() const;
119
120 uint32_t getGOTIndex() const {
121 assert(gotIndex != INVALID_INDEX);
122 return gotIndex;
123 }
124
125 void setGOTIndex(uint32_t index);
126 bool hasGOTIndex() const { return gotIndex != INVALID_INDEX; }
127
128protected:
129 Symbol(StringRef name, Kind k, uint32_t flags, InputFile *f)
130 : name(name), file(f), symbolKind(k), referenced(!config->gcSections),
131 requiresGOT(false), isUsedInRegularObj(false), forceExport(false),
132 canInline(false), traced(false), isStub(false), flags(flags) {}
133
134 StringRef name;
135 InputFile *file;
136 uint32_t outputSymbolIndex = INVALID_INDEX;
137 uint32_t gotIndex = INVALID_INDEX;
138 Kind symbolKind;
139
140public:
141 bool referenced : 1;
142
143 // True for data symbols that needs a dummy GOT entry. Used for static
144 // linking of GOT accesses.
145 bool requiresGOT : 1;
146
147 // True if the symbol was used for linking and thus need to be added to the
148 // output file's symbol table. This is true for all symbols except for
149 // unreferenced DSO symbols, lazy (archive) symbols, and bitcode symbols that
150 // are unreferenced except by other bitcode objects.
151 bool isUsedInRegularObj : 1;
152
153 // True if ths symbol is explicitly marked for export (i.e. via the
154 // -e/--export command line flag)
155 bool forceExport : 1;
156
157 // False if LTO shouldn't inline whatever this symbol points to. If a symbol
158 // is overwritten after LTO, LTO shouldn't inline the symbol because it
159 // doesn't know the final contents of the symbol.
160 bool canInline : 1;
161
162 // True if this symbol is specified by --trace-symbol option.
163 bool traced : 1;
164
165 // True if this symbol is a linker-synthesized stub function (traps when
166 // called) and should otherwise be treated as missing/undefined. See
167 // SymbolTable::replaceWithUndefined.
168 // These stubs never appear in the table and any table index relocations
169 // against them will produce address 0 (The table index representing
170 // the null function pointer).
171 bool isStub : 1;
172
173 uint32_t flags;
174};

416// Wasm events are features that suspend the current execution and transfer the
417// control flow to a corresponding handler. Currently the only supported event
418// kind is exceptions.
419//
420// Event tags are values to distinguish different events. For exceptions, they
421// can be used to distinguish different language's exceptions, i.e., all C++
422// exceptions have the same tag. Wasm can generate code capable of doing
423// different handling actions based on the tag of caught exceptions.
424//
425// A single EventSymbol object represents a single tag. C++ exception event
426// symbol is a weak symbol generated in every object file in which exceptions
427// are used, and has name '__cpp_exception' for linking.
428class EventSymbol : public Symbol {
429public:
430 static bool classof(const Symbol *s) { return s->kind() == DefinedEventKind; }
431
432 const WasmEventType *getEventType() const { return eventType; }
433
434 // Get/set the event index
435 uint32_t getEventIndex() const;
436 void setEventIndex(uint32_t index);
437 bool hasEventIndex() const;
438
439 const WasmSignature *signature;
440
441protected:
442 EventSymbol(StringRef name, Kind k, uint32_t flags, InputFile *f,
443 const WasmEventType *eventType, const WasmSignature *sig)
444 : Symbol(name, k, flags, f), signature(sig), eventType(eventType) {}
445
446 const WasmEventType *eventType;
447 uint32_t eventIndex = INVALID_INDEX;
448};

460// LazySymbol represents a symbol that is not yet in the link, but we know where
461// to find it if needed. If the resolver finds both Undefined and Lazy for the
462// same name, it will ask the Lazy to load a file.
463//
464// A special complication is the handling of weak undefined symbols. They should
465// not load a file, but we have to remember we have seen both the weak undefined
466// and the lazy. We represent that with a lazy symbol with a weak binding. This
467// means that code looking for undefined symbols normally also has to take lazy
468// symbols into consideration.
469class LazySymbol : public Symbol {
470public:
471 LazySymbol(StringRef name, uint32_t flags, InputFile *file,
472 const llvm::object::Archive::Symbol &sym)
473 : Symbol(name, LazyKind, flags, file), archiveSymbol(sym) {}
474
475 static bool classof(const Symbol *s) { return s->kind() == LazyKind; }
476 void fetch();
477 void setWeak();
478 MemoryBufferRef getMemberBuffer();
479
480 // Lazy symbols can have a signature because they can replace an
481 // UndefinedFunction which which case we need to be able to preserve the
482 // signature.
483 // TODO(sbc): This repetition of the signature field is inelegant. Revisit
484 // the use of class hierarchy to represent symbol taxonomy.
485 const WasmSignature *signature = nullptr;
486
487private:
488 llvm::object::Archive::Symbol archiveSymbol;
489};

491// linker-generated symbols
492struct WasmSym {
493 // __global_base
494 // Symbol marking the start of the global section.
495 static DefinedData *globalBase;
496
497 // __stack_pointer
498 // Global that holds the address of the top of the explicit value stack in
499 // linear memory.
500 static GlobalSymbol *stackPointer;
501
502 // __tls_base
503 // Global that holds the address of the base of the current thread's
504 // TLS block.
505 static GlobalSymbol *tlsBase;
506
507 // __tls_size
508 // Symbol whose value is the size of the TLS block.
509 static GlobalSymbol *tlsSize;
510
511 // __tls_size
512 // Symbol whose value is the alignment of the TLS block.
513 static GlobalSymbol *tlsAlign;
514
515 // __data_end
516 // Symbol marking the end of the data and bss.
517 static DefinedData *dataEnd;
518
519 // __heap_base
520 // Symbol marking the end of the data, bss and explicit stack. Any linear
521 // memory following this address is not used by the linked code and can
522 // therefore be used as a backing store for brk()/malloc() implementations.
523 static DefinedData *heapBase;
524
525 // __wasm_init_memory_flag
526 // Symbol whose contents are nonzero iff memory has already been initialized.
527 static DefinedData *initMemoryFlag;
528
529 // __wasm_init_memory
530 // Function that initializes passive data segments during instantiation.
531 static DefinedFunction *initMemory;
532
533 // __wasm_call_ctors
534 // Function that directly calls all ctors in priority order.
535 static DefinedFunction *callCtors;
536
537 // __wasm_call_dtors
538 // Function that calls the libc/etc. cleanup function.
539 static DefinedFunction *callDtors;
540
541 // __wasm_apply_data_relocs
542 // Function that applies relocations to data segment post-instantiation.
543 static DefinedFunction *applyDataRelocs;
544
545 // __wasm_apply_global_relocs
546 // Function that applies relocations to data segment post-instantiation.
547 // Unlike __wasm_apply_data_relocs this needs to run on every thread.
548 static DefinedFunction *applyGlobalRelocs;
549
550 // __wasm_init_tls
551 // Function that allocates thread-local storage and initializes it.
552 static DefinedFunction *initTLS;
553
554 // Pointer to the function that is to be used in the start section.
555 // (normally an alias of initMemory, or applyGlobalRelocs).
556 static DefinedFunction *startFunction;
557
558 // __dso_handle
559 // Symbol used in calls to __cxa_atexit to determine current DLL
560 static DefinedData *dsoHandle;
561
562 // __table_base
563 // Used in PIC code for offset of indirect function table
564 static UndefinedGlobal *tableBase;
565 static DefinedData *definedTableBase;
566
567 // __memory_base
568 // Used in PIC code for offset of global data
569 static UndefinedGlobal *memoryBase;
570 static DefinedData *definedMemoryBase;
571
572 // __indirect_function_table
573 // Used as an address space for function pointers, with each function that is
574 // used as a function pointer being allocated a slot.
575 static TableSymbol *indirectFunctionTable;
576};

578// A buffer class that is large enough to hold any Symbol-derived
579// object. We allocate memory using this class and instantiate a symbol
580// using the placement new.
581union SymbolUnion {
582 alignas(DefinedFunction) char a[sizeof(DefinedFunction)];
583 alignas(DefinedData) char b[sizeof(DefinedData)];
584 alignas(DefinedGlobal) char c[sizeof(DefinedGlobal)];
585 alignas(DefinedEvent) char d[sizeof(DefinedEvent)];
586 alignas(DefinedTable) char e[sizeof(DefinedTable)];
587 alignas(LazySymbol) char f[sizeof(LazySymbol)];
588 alignas(UndefinedFunction) char g[sizeof(UndefinedFunction)];
589 alignas(UndefinedData) char h[sizeof(UndefinedData)];
590 alignas(UndefinedGlobal) char i[sizeof(UndefinedGlobal)];
591 alignas(UndefinedTable) char j[sizeof(UndefinedTable)];
592 alignas(SectionSymbol) char k[sizeof(SectionSymbol)];
593};
594
595// It is important to keep the size of SymbolUnion small for performance and
596// memory usage reasons. 96 bytes is a soft limit based on the size of
597// UndefinedFunction on a 64-bit system.
598static_assert(sizeof(SymbolUnion) <= 120, "SymbolUnion too large");

603template <typename T, typename... ArgT>
604T *replaceSymbol(Symbol *s, ArgT &&... arg) {
605 static_assert(std::is_trivially_destructible<T>(),
606 "Symbol types must be trivially destructible");
607 static_assert(sizeof(T) <= sizeof(SymbolUnion), "SymbolUnion too small");
608 static_assert(alignof(T) <= alignof(SymbolUnion),
609 "SymbolUnion not aligned enough");
610 assert(static_cast<Symbol *>(static_cast<T *>(nullptr)) == nullptr &&
611 "Not a Symbol");
612
613 Symbol symCopy = *s;
614
615 T *s2 = new (s) T(std::forward<ArgT>(arg)...);
616 s2->isUsedInRegularObj = symCopy.isUsedInRegularObj;
617 s2->forceExport = symCopy.forceExport;
618 s2->canInline = symCopy.canInline;
619 s2->traced = symCopy.traced;
620
621 // Print out a log message if --trace-symbol was specified.
622 // This is for debugging.
623 if (s2->traced)
624 printTraceSymbol(s2);
625
626 return s2;
627}
