30class OutputSection;
31class InputSection;
32
33class OutputSegment {
34public:
35 const OutputSection *firstSection() const { return sections.front(); }
36 const OutputSection *lastSection() const { return sections.back(); }
37
38 void addOutputSection(OutputSection *os);
39 void sortOutputSections(
40 llvm::function_ref<bool(OutputSection *, OutputSection *)> comparator) {
41 llvm::stable_sort(sections, comparator);
42 }
43
44 const std::vector<OutputSection *> &getSections() const { return sections; }
45 size_t numNonHiddenSections() const;
46
47 uint64_t fileOff = 0;
48 StringRef name;
49 uint32_t maxProt = 0;
50 uint32_t initProt = 0;
51 uint8_t index;
52
53private:
54 std::vector<OutputSection *> sections;
55};
