36#define TINTERLACE_FLAG_VLPF 01
37#define TINTERLACE_FLAG_CVLPF 2
38#define TINTERLACE_FLAG_EXACT_TB 4
39#define TINTERLACE_FLAG_BYPASS_IL 8
40
41enum VLPFilter {
42 VLPF_OFF = 0,
43 VLPF_LIN = 1,
44 VLPF_CMP = 2,
45};
46
47enum TInterlaceMode {
48 MODE_MERGE = 0,
49 MODE_DROP_EVEN,
50 MODE_DROP_ODD,
51 MODE_PAD,
52 MODE_INTERLEAVE_TOP,
53 MODE_INTERLEAVE_BOTTOM,
54 MODE_INTERLACEX2,
55 MODE_MERGEX2,
56 MODE_NB,
57};
58
59enum InterlaceScanMode {
60 MODE_TFF = 0,
61 MODE_BFF,
62};
63
64typedef struct TInterlaceContext {
65 const AVClass *class;
66 int mode; ///< TInterlaceMode, interlace mode selected
67 AVRational preout_time_base;
68 int flags; ///< flags affecting interlacing algorithm
69 int lowpass; ///< legacy interlace filter lowpass mode
70 int vsub; ///< chroma vertical subsampling
71 AVFrame *cur;
72 AVFrame *next;
73 uint8_t *black_data[4]; ///< buffer used to fill padded lines
74 int black_linesize[4];
75 FFDrawContext draw;
76 FFDrawColor color;
77 const AVPixFmtDescriptor *csp;
78 void (*lowpass_line)(uint8_t *dstp, ptrdiff_t width, const uint8_t *srcp,
79 ptrdiff_t mref, ptrdiff_t pref, int clip_max);
80} TInterlaceContext;
81
82void ff_tinterlace_init_x86(TInterlaceContext *interlace);
