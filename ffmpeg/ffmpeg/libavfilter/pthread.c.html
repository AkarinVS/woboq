36typedef struct ThreadContext {
37 AVFilterGraph *graph;
38 AVSliceThread *thread;
39 avfilter_action_func *func;
40
41 /* per-execute parameters */
42 AVFilterContext *ctx;
43 void *arg;
44 int *rets;
45} ThreadContext;
46
47static void worker_func(void *priv, int jobnr, int threadnr, int nb_jobs, int nb_threads)
48{
49 ThreadContext *c = priv;
50 int ret = c->func(c->ctx, c->arg, jobnr, nb_jobs);
51 if (c->rets)
52 c->rets[jobnr] = ret;
53}
54
55static void slice_thread_uninit(ThreadContext *c)
56{
57 avpriv_slicethread_free(&c->thread);
58}
59
60static int thread_execute(AVFilterContext *ctx, avfilter_action_func *func,
61 void *arg, int *ret, int nb_jobs)
62{
63 ThreadContext *c = ctx->graph->internal->thread;
64
65 if (nb_jobs <= 0)
66 return 0;
67 c->ctx = ctx;
68 c->arg = arg;
69 c->func = func;
70 c->rets = ret;
71
72 avpriv_slicethread_execute(c->thread, nb_jobs, 0);
73 return 0;
74}
75
76static int thread_init_internal(ThreadContext *c, int nb_threads)
77{
78 nb_threads = avpriv_slicethread_create(&c->thread, c, worker_func, NULL, nb_threads);
79 if (nb_threads <= 1)
80 avpriv_slicethread_free(&c->thread);
81 return FFMAX(nb_threads, 1);
82}
83
84int ff_graph_thread_init(AVFilterGraph *graph)
85{
86 int ret;
87
88 if (graph->nb_threads == 1) {
89 graph->thread_type = 0;
90 return 0;
91 }
92
93 graph->internal->thread = av_mallocz(sizeof(ThreadContext));
94 if (!graph->internal->thread)
95 return AVERROR(ENOMEM);
96
97 ret = thread_init_internal(graph->internal->thread, graph->nb_threads);
98 if (ret <= 1) {
99 av_freep(&graph->internal->thread);
100 graph->thread_type = 0;
101 graph->nb_threads = 1;
102 return (ret < 0) ? ret : 0;
103 }
104 graph->nb_threads = ret;
105
106 graph->internal->thread_execute = thread_execute;
107
108 return 0;
109}
110
111void ff_graph_thread_free(AVFilterGraph *graph)
112{
113 if (graph->internal->thread)
114 slice_thread_uninit(graph->internal->thread);
115 av_freep(&graph->internal->thread);
116}
