30typedef struct MaskedThresholdContext {
31 const AVClass *class;
32
33 int threshold;
34 int planes;
35
36 int linesize[4];
37 int planewidth[4], planeheight[4];
38 int nb_planes;
39 int depth;
40 FFFrameSync fs;
41
42 void (*maskedthreshold)(const uint8_t *src, const uint8_t *ref, uint8_t *dst, int threshold, int w);
43} MaskedThresholdContext;

85static void threshold8(const uint8_t *src, const uint8_t *ref, uint8_t *dst, int threshold, int w)
86{
87 for (int x = 0; x < w; x++)
88 dst[x] = FFABS(src[x] - ref[x]) <= threshold ? src[x] : ref[x];
89}
90
91static void threshold16(const uint8_t *ssrc, const uint8_t *rref, uint8_t *ddst, int threshold, int w)
92{
93 const uint16_t *src = (const uint16_t *)ssrc;
94 const uint16_t *ref = (const uint16_t *)rref;
95 uint16_t *dst = (uint16_t *)ddst;
96
97 for (int x = 0; x < w; x++)
98 dst[x] = FFABS(src[x] - ref[x]) <= threshold ? src[x] : ref[x];
99}

101static int config_input(AVFilterLink *inlink)
102{
103 AVFilterContext *ctx = inlink->dst;
104 MaskedThresholdContext *s = ctx->priv;
105 const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(inlink->format);
106 int vsub, hsub, ret;
107
108 s->nb_planes = av_pix_fmt_count_planes(inlink->format);
109
110 if ((ret = av_image_fill_linesizes(s->linesize, inlink->format, inlink->w)) < 0)
111 return ret;
112
113 hsub = desc->log2_chroma_w;
114 vsub = desc->log2_chroma_h;
115 s->planeheight[1] = s->planeheight[2] = AV_CEIL_RSHIFT(inlink->h, vsub);
116 s->planeheight[0] = s->planeheight[3] = inlink->h;
117 s->planewidth[1] = s->planewidth[2] = AV_CEIL_RSHIFT(inlink->w, hsub);
118 s->planewidth[0] = s->planewidth[3] = inlink->w;
119
120 s->depth = desc->comp[0].depth;
121
122 if (desc->comp[0].depth == 8)
123 s->maskedthreshold = threshold8;
124 else
125 s->maskedthreshold = threshold16;
126
127 return 0;
128}

130static int threshold_slice(AVFilterContext *ctx, void *arg, int jobnr, int nb_jobs)
131{
132 MaskedThresholdContext *s = ctx->priv;
133 const int threshold = s->threshold;
134 ThreadData *td = arg;
135
136 for (int p = 0; p < s->nb_planes; p++) {
137 const ptrdiff_t src_linesize = td->src->linesize[p];
138 const ptrdiff_t ref_linesize = td->ref->linesize[p];
139 const ptrdiff_t dst_linesize = td->dst->linesize[p];
140 const int w = s->planewidth[p];
141 const int h = s->planeheight[p];
142 const int slice_start = (h * jobnr) / nb_jobs;
143 const int slice_end = (h * (jobnr+1)) / nb_jobs;
144 const uint8_t *src = td->src->data[p] + slice_start * src_linesize;
145 const uint8_t *ref = td->ref->data[p] + slice_start * ref_linesize;
146 uint8_t *dst = td->dst->data[p] + slice_start * dst_linesize;
147
148 if (!((1 << p) & s->planes)) {
149 av_image_copy_plane(dst, dst_linesize, ref, ref_linesize,
150 s->linesize[p], slice_end - slice_start);
151 continue;
152 }
153
154 for (int y = slice_start; y < slice_end; y++) {
155 s->maskedthreshold(src, ref, dst, threshold, w);
156
157 dst += dst_linesize;
158 src += src_linesize;
159 ref += ref_linesize;
160 }
161 }
162
163 return 0;
164}

166static int process_frame(FFFrameSync *fs)
167{
168 AVFilterContext *ctx = fs->parent;
169 MaskedThresholdContext *s = fs->opaque;
170 AVFilterLink *outlink = ctx->outputs[0];
171 AVFrame *out, *src, *ref;
172 int ret;
173
174 if ((ret = ff_framesync_get_frame(&s->fs, 0, &src, 0)) < 0 ||
175 (ret = ff_framesync_get_frame(&s->fs, 1, &ref, 0)) < 0)
176 return ret;
177
178 if (ctx->is_disabled) {
179 out = av_frame_clone(src);
180 if (!out)
181 return AVERROR(ENOMEM);
182 } else {
183 ThreadData td;
184
185 out = ff_get_video_buffer(outlink, outlink->w, outlink->h);
186 if (!out)
187 return AVERROR(ENOMEM);
188 av_frame_copy_props(out, src);
189
190 td.src = src;
191 td.ref = ref;
192 td.dst = out;
193
194 ctx->internal->execute(ctx, threshold_slice, &td, NULL, FFMIN(s->planeheight[2],
195 ff_filter_get_nb_threads(ctx)));
196 }
197 out->pts = av_rescale_q(s->fs.pts, s->fs.time_base, outlink->time_base);
198
199 return ff_filter_frame(outlink, out);
200}

202static int config_output(AVFilterLink *outlink)
203{
204 AVFilterContext *ctx = outlink->src;
205 MaskedThresholdContext *s = ctx->priv;
206 AVFilterLink *source = ctx->inputs[0];
207 AVFilterLink *ref = ctx->inputs[1];
208 FFFrameSyncIn *in;
209 int ret;
210
211 if (source->format != ref->format) {
212 av_log(ctx, AV_LOG_ERROR, "inputs must be of same pixel format\n");
213 return AVERROR(EINVAL);
214 }
215 if (source->w != ref->w || source->h != ref->h) {
216 av_log(ctx, AV_LOG_ERROR, "First input link %s parameters "
217 "(size %dx%d) do not match the corresponding "
218 "second input link %s parameters (%dx%d)\n",
219 ctx->input_pads[0].name, source->w, source->h,
220 ctx->input_pads[1].name, ref->w, ref->h);
221 return AVERROR(EINVAL);
222 }
223
224 outlink->w = source->w;
225 outlink->h = source->h;
226 outlink->sample_aspect_ratio = source->sample_aspect_ratio;
227 outlink->frame_rate = source->frame_rate;
228
229 if ((ret = ff_framesync_init(&s->fs, ctx, 2)) < 0)
230 return ret;
231
232 in = s->fs.in;
233 in[0].time_base = source->time_base;
234 in[1].time_base = ref->time_base;
235 in[0].sync = 1;
236 in[0].before = EXT_STOP;
237 in[0].after = EXT_INFINITY;
238 in[1].sync = 1;
239 in[1].before = EXT_STOP;
240 in[1].after = EXT_INFINITY;
241 s->fs.opaque = s;
242 s->fs.on_event = process_frame;
243
244 ret = ff_framesync_configure(&s->fs);
245 outlink->time_base = s->fs.time_base;
246
247 return ret;
248}
