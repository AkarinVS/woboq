22#ifndef AVFILTER_AUDIO_H
23#define AVFILTER_AUDIO_H
24
25#include "avfilter.h"
26#include "internal.h"
27
28/** default handler for get_audio_buffer() for audio inputs */
29AVFrame *ff_default_get_audio_buffer(AVFilterLink *link, int nb_samples);
30
31/** get_audio_buffer() handler for filters which simply pass audio along */
32AVFrame *ff_null_get_audio_buffer(AVFilterLink *link, int nb_samples);
33
34/**
35 * Request an audio samples buffer with a specific set of permissions.
36 *
37 * @param link the output link to the filter from which the buffer will
38 * be requested
39 * @param nb_samples the number of samples per channel
40 * @return A reference to the samples. This must be unreferenced with
41 * avfilter_unref_buffer when you are finished with it.
42 */
43AVFrame *ff_get_audio_buffer(AVFilterLink *link, int nb_samples);
44
45#endif /* AVFILTER_AUDIO_H */
