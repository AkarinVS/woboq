75static inline unsigned pow_2(unsigned base)
76{
77 return base*base;
78}
79
80static inline double get_psnr(double mse, uint64_t nb_frames, int max)
81{
82 return 10.0 * log10(pow_2(max) / (mse / nb_frames));
83}
84
85static uint64_t sse_line_8bit(const uint8_t *main_line, const uint8_t *ref_line, int outw)
86{
87 int j;
88 unsigned m2 = 0;
89
90 for (j = 0; j < outw; j++)
91 m2 += pow_2(main_line[j] - ref_line[j]);
92
93 return m2;
94}
95
96static uint64_t sse_line_16bit(const uint8_t *_main_line, const uint8_t *_ref_line, int outw)
97{
98 int j;
99 uint64_t m2 = 0;
100 const uint16_t *main_line = (const uint16_t *) _main_line;
101 const uint16_t *ref_line = (const uint16_t *) _ref_line;
102
103 for (j = 0; j < outw; j++)
104 m2 += pow_2(main_line[j] - ref_line[j]);
105
106 return m2;
107}

109typedef struct ThreadData {
110 const uint8_t *main_data[4];
111 const uint8_t *ref_data[4];
112 int main_linesize[4];
113 int ref_linesize[4];
114 int planewidth[4];
115 int planeheight[4];
116 uint64_t **score;
117 int nb_components;
118 PSNRDSPContext *dsp;
119} ThreadData;

121static
122int compute_images_mse(AVFilterContext *ctx, void *arg,
123 int jobnr, int nb_jobs)
124{
125 ThreadData *td = arg;
126 uint64_t *score = td->score[jobnr];
127
128 for (int c = 0; c < td->nb_components; c++) {
129 const int outw = td->planewidth[c];
130 const int outh = td->planeheight[c];
131 const int slice_start = (outh * jobnr) / nb_jobs;
132 const int slice_end = (outh * (jobnr+1)) / nb_jobs;
133 const int ref_linesize = td->ref_linesize[c];
134 const int main_linesize = td->main_linesize[c];
135 const uint8_t *main_line = td->main_data[c] + main_linesize * slice_start;
136 const uint8_t *ref_line = td->ref_data[c] + ref_linesize * slice_start;
137 uint64_t m = 0;
138 for (int i = slice_start; i < slice_end; i++) {
139 m += td->dsp->sse_line(main_line, ref_line, outw);
140 ref_line += ref_linesize;
141 main_line += main_linesize;
142 }
143 score[c] = m;
144 }
145
146 return 0;
147}

149static void set_meta(AVDictionary **metadata, const char *key, char comp, float d)
150{
151 char value[128];
152 snprintf(value, sizeof(value), "%f", d);
153 if (comp) {
154 char key2[128];
155 snprintf(key2, sizeof(key2), "%s%c", key, comp);
156 av_dict_set(metadata, key2, value, 0);
157 } else {
158 av_dict_set(metadata, key, value, 0);
159 }
160}

162static int do_psnr(FFFrameSync *fs)
163{
164 AVFilterContext *ctx = fs->parent;
165 PSNRContext *s = ctx->priv;
166 AVFrame *master, *ref;
167 double comp_mse[4], mse = 0.;
168 uint64_t comp_sum[4] = { 0 };
169 AVDictionary **metadata;
170 ThreadData td;
171 int ret;
172
173 ret = ff_framesync_dualinput_get(fs, &master, &ref);
174 if (ret < 0)
175 return ret;
176 if (ctx->is_disabled || !ref)
177 return ff_filter_frame(ctx->outputs[0], master);
178 metadata = &master->metadata;
179
180 td.nb_components = s->nb_components;
181 td.dsp = &s->dsp;
182 td.score = s->score;
183 for (int c = 0; c < s->nb_components; c++) {
184 td.main_data[c] = master->data[c];
185 td.ref_data[c] = ref->data[c];
186 td.main_linesize[c] = master->linesize[c];
187 td.ref_linesize[c] = ref->linesize[c];
188 td.planewidth[c] = s->planewidth[c];
189 td.planeheight[c] = s->planeheight[c];
190 }
191
192 ctx->internal->execute(ctx, compute_images_mse, &td, NULL, FFMIN(s->planeheight[1], s->nb_threads));
193
194 for (int j = 0; j < s->nb_threads; j++) {
195 for (int c = 0; c < s->nb_components; c++)
196 comp_sum[c] += s->score[j][c];
197 }
198
199 for (int c = 0; c < s->nb_components; c++)
200 comp_mse[c] = comp_sum[c] / ((double)s->planewidth[c] * s->planeheight[c]);
201
202 for (int c = 0; c < s->nb_components; c++)
203 mse += comp_mse[c] * s->planeweight[c];
204
205 s->min_mse = FFMIN(s->min_mse, mse);
206 s->max_mse = FFMAX(s->max_mse, mse);
207
208 s->mse += mse;
209
210 for (int j = 0; j < s->nb_components; j++)
211 s->mse_comp[j] += comp_mse[j];
212 s->nb_frames++;
213
214 for (int j = 0; j < s->nb_components; j++) {
215 int c = s->is_rgb ? s->rgba_map[j] : j;
216 set_meta(metadata, "lavfi.psnr.mse.", s->comps[j], comp_mse[c]);
217 set_meta(metadata, "lavfi.psnr.psnr.", s->comps[j], get_psnr(comp_mse[c], 1, s->max[c]));
218 }
219 set_meta(metadata, "lavfi.psnr.mse_avg", 0, mse);
220 set_meta(metadata, "lavfi.psnr.psnr_avg", 0, get_psnr(mse, 1, s->average_max));
221
222 if (s->stats_file) {
223 if (s->stats_version == 2 && !s->stats_header_written) {
224 fprintf(s->stats_file, "psnr_log_version:2 fields:n");
225 fprintf(s->stats_file, ",mse_avg");
226 for (int j = 0; j < s->nb_components; j++) {
227 fprintf(s->stats_file, ",mse_%c", s->comps[j]);
228 }
229 fprintf(s->stats_file, ",psnr_avg");
230 for (int j = 0; j < s->nb_components; j++) {
231 fprintf(s->stats_file, ",psnr_%c", s->comps[j]);
232 }
233 if (s->stats_add_max) {
234 fprintf(s->stats_file, ",max_avg");
235 for (int j = 0; j < s->nb_components; j++) {
236 fprintf(s->stats_file, ",max_%c", s->comps[j]);
237 }
238 }
239 fprintf(s->stats_file, "\n");
240 s->stats_header_written = 1;
241 }
242 fprintf(s->stats_file, "n:%"PRId64" mse_avg:%0.2f ", s->nb_frames, mse);
243 for (int j = 0; j < s->nb_components; j++) {
244 int c = s->is_rgb ? s->rgba_map[j] : j;
245 fprintf(s->stats_file, "mse_%c:%0.2f ", s->comps[j], comp_mse[c]);
246 }
247 fprintf(s->stats_file, "psnr_avg:%0.2f ", get_psnr(mse, 1, s->average_max));
248 for (int j = 0; j < s->nb_components; j++) {
249 int c = s->is_rgb ? s->rgba_map[j] : j;
250 fprintf(s->stats_file, "psnr_%c:%0.2f ", s->comps[j],
251 get_psnr(comp_mse[c], 1, s->max[c]));
252 }
253 if (s->stats_version == 2 && s->stats_add_max) {
254 fprintf(s->stats_file, "max_avg:%d ", s->average_max);
255 for (int j = 0; j < s->nb_components; j++) {
256 int c = s->is_rgb ? s->rgba_map[j] : j;
257 fprintf(s->stats_file, "max_%c:%d ", s->comps[j], s->max[c]);
258 }
259 }
260 fprintf(s->stats_file, "\n");
261 }
262
263 return ff_filter_frame(ctx->outputs[0], master);
264}

266static av_cold int init(AVFilterContext *ctx)
267{
268 PSNRContext *s = ctx->priv;
269
270 s->min_mse = +INFINITY;
271 s->max_mse = -INFINITY;
272
273 if (s->stats_file_str) {
274 if (s->stats_version < 2 && s->stats_add_max) {
275 av_log(ctx, AV_LOG_ERROR,
276 "stats_add_max was specified but stats_version < 2.\n" );
277 return AVERROR(EINVAL);
278 }
279 if (!strcmp(s->stats_file_str, "-")) {
280 s->stats_file = stdout;
281 } else {
282 s->stats_file = fopen(s->stats_file_str, "w");
283 if (!s->stats_file) {
284 int err = AVERROR(errno);
285 char buf[128];
286 av_strerror(err, buf, sizeof(buf));
287 av_log(ctx, AV_LOG_ERROR, "Could not open stats file %s: %s\n",
288 s->stats_file_str, buf);
289 return err;
290 }
291 }
292 }
293
294 s->fs.on_event = do_psnr;
295 return 0;
296}

298static int query_formats(AVFilterContext *ctx)
299{
300 static const enum AVPixelFormat pix_fmts[] = {
301 AV_PIX_FMT_GRAY8, AV_PIX_FMT_GRAY9, AV_PIX_FMT_GRAY10, AV_PIX_FMT_GRAY12, AV_PIX_FMT_GRAY14, AV_PIX_FMT_GRAY16,
302#define PF_NOALPHA(suf) AV_PIX_FMT_YUV420##suf, AV_PIX_FMT_YUV422##suf, AV_PIX_FMT_YUV444##suf
303#define PF_ALPHA(suf) AV_PIX_FMT_YUVA420##suf, AV_PIX_FMT_YUVA422##suf, AV_PIX_FMT_YUVA444##suf
304#define PF(suf) PF_NOALPHA(suf), PF_ALPHA(suf)
305 PF(P), PF(P9), PF(P10), PF_NOALPHA(P12), PF_NOALPHA(P14), PF(P16),
306 AV_PIX_FMT_YUV440P, AV_PIX_FMT_YUV411P, AV_PIX_FMT_YUV410P,
307 AV_PIX_FMT_YUVJ411P, AV_PIX_FMT_YUVJ420P, AV_PIX_FMT_YUVJ422P,
308 AV_PIX_FMT_YUVJ440P, AV_PIX_FMT_YUVJ444P,
309 AV_PIX_FMT_GBRP, AV_PIX_FMT_GBRP9, AV_PIX_FMT_GBRP10,
310 AV_PIX_FMT_GBRP12, AV_PIX_FMT_GBRP14, AV_PIX_FMT_GBRP16,
311 AV_PIX_FMT_GBRAP, AV_PIX_FMT_GBRAP10, AV_PIX_FMT_GBRAP12, AV_PIX_FMT_GBRAP16,
312 AV_PIX_FMT_NONE
313 };
314
315 AVFilterFormats *fmts_list = ff_make_format_list(pix_fmts);
316 if (!fmts_list)
317 return AVERROR(ENOMEM);
318 return ff_set_common_formats(ctx, fmts_list);
319}

321static int config_input_ref(AVFilterLink *inlink)
322{
323 const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(inlink->format);
324 AVFilterContext *ctx = inlink->dst;
325 PSNRContext *s = ctx->priv;
326 double average_max;
327 unsigned sum;
328 int j;
329
330 s->nb_threads = ff_filter_get_nb_threads(ctx);
331 s->nb_components = desc->nb_components;
332 if (ctx->inputs[0]->w != ctx->inputs[1]->w ||
333 ctx->inputs[0]->h != ctx->inputs[1]->h) {
334 av_log(ctx, AV_LOG_ERROR, "Width and height of input videos must be same.\n");
335 return AVERROR(EINVAL);
336 }
337 if (ctx->inputs[0]->format != ctx->inputs[1]->format) {
338 av_log(ctx, AV_LOG_ERROR, "Inputs must be of same pixel format.\n");
339 return AVERROR(EINVAL);
340 }
341
342 s->max[0] = (1 << desc->comp[0].depth) - 1;
343 s->max[1] = (1 << desc->comp[1].depth) - 1;
344 s->max[2] = (1 << desc->comp[2].depth) - 1;
345 s->max[3] = (1 << desc->comp[3].depth) - 1;
346
347 s->is_rgb = ff_fill_rgba_map(s->rgba_map, inlink->format) >= 0;
348 s->comps[0] = s->is_rgb ? 'r' : 'y' ;
349 s->comps[1] = s->is_rgb ? 'g' : 'u' ;
350 s->comps[2] = s->is_rgb ? 'b' : 'v' ;
351 s->comps[3] = 'a';
352
353 s->planeheight[1] = s->planeheight[2] = AV_CEIL_RSHIFT(inlink->h, desc->log2_chroma_h);
354 s->planeheight[0] = s->planeheight[3] = inlink->h;
355 s->planewidth[1] = s->planewidth[2] = AV_CEIL_RSHIFT(inlink->w, desc->log2_chroma_w);
356 s->planewidth[0] = s->planewidth[3] = inlink->w;
357 sum = 0;
358 for (j = 0; j < s->nb_components; j++)
359 sum += s->planeheight[j] * s->planewidth[j];
360 average_max = 0;
361 for (j = 0; j < s->nb_components; j++) {
362 s->planeweight[j] = (double) s->planeheight[j] * s->planewidth[j] / sum;
363 average_max += s->max[j] * s->planeweight[j];
364 }
365 s->average_max = lrint(average_max);
366
367 s->dsp.sse_line = desc->comp[0].depth > 8 ? sse_line_16bit : sse_line_8bit;
368 if (ARCH_X86)
369 ff_psnr_init_x86(&s->dsp, desc->comp[0].depth);
370
371 s->score = av_calloc(s->nb_threads, sizeof(*s->score));
372 if (!s->score)
373 return AVERROR(ENOMEM);
374
375 for (int t = 0; t < s->nb_threads && s->score; t++) {
376 s->score[t] = av_calloc(s->nb_components, sizeof(*s->score[0]));
377 if (!s->score[t])
378 return AVERROR(ENOMEM);
379 }
380
381 return 0;
382}

384static int config_output(AVFilterLink *outlink)
385{
386 AVFilterContext *ctx = outlink->src;
387 PSNRContext *s = ctx->priv;
388 AVFilterLink *mainlink = ctx->inputs[0];
389 int ret;
390
391 ret = ff_framesync_init_dualinput(&s->fs, ctx);
392 if (ret < 0)
393 return ret;
394 outlink->w = mainlink->w;
395 outlink->h = mainlink->h;
396 outlink->time_base = mainlink->time_base;
397 outlink->sample_aspect_ratio = mainlink->sample_aspect_ratio;
398 outlink->frame_rate = mainlink->frame_rate;
399 if ((ret = ff_framesync_configure(&s->fs)) < 0)
400 return ret;
401
402 outlink->time_base = s->fs.time_base;
403
404 if (av_cmp_q(mainlink->time_base, outlink->time_base) ||
405 av_cmp_q(ctx->inputs[1]->time_base, outlink->time_base))
406 av_log(ctx, AV_LOG_WARNING, "not matching timebases found between first input: %d/%d and second input %d/%d, results may be incorrect!\n",
407 mainlink->time_base.num, mainlink->time_base.den,
408 ctx->inputs[1]->time_base.num, ctx->inputs[1]->time_base.den);
409
410 return 0;
411}

419static av_cold void uninit(AVFilterContext *ctx)
420{
421 PSNRContext *s = ctx->priv;
422
423 if (s->nb_frames > 0) {
424 int j;
425 char buf[256];
426
427 buf[0] = 0;
428 for (j = 0; j < s->nb_components; j++) {
429 int c = s->is_rgb ? s->rgba_map[j] : j;
430 av_strlcatf(buf, sizeof(buf), " %c:%f", s->comps[j],
431 get_psnr(s->mse_comp[c], s->nb_frames, s->max[c]));
432 }
433 av_log(ctx, AV_LOG_INFO, "PSNR%s average:%f min:%f max:%f\n",
434 buf,
435 get_psnr(s->mse, s->nb_frames, s->average_max),
436 get_psnr(s->max_mse, 1, s->average_max),
437 get_psnr(s->min_mse, 1, s->average_max));
438 }
439
440 ff_framesync_uninit(&s->fs);
441 for (int t = 0; t < s->nb_threads && s->score; t++)
442 av_freep(&s->score[t]);
443 av_freep(&s->score);
444
445 if (s->stats_file && s->stats_file != stdout)
446 fclose(s->stats_file);
447}
