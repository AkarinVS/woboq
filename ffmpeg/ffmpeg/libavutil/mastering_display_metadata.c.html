27AVMasteringDisplayMetadata *av_mastering_display_metadata_alloc(void)
28{
29 return av_mallocz(sizeof(AVMasteringDisplayMetadata));
30}
31
32AVMasteringDisplayMetadata *av_mastering_display_metadata_create_side_data(AVFrame *frame)
33{
34 AVFrameSideData *side_data = av_frame_new_side_data(frame,
35 AV_FRAME_DATA_MASTERING_DISPLAY_METADATA,
36 sizeof(AVMasteringDisplayMetadata));
37 if (!side_data)
38 return NULL;
39
40 memset(side_data->data, 0, sizeof(AVMasteringDisplayMetadata));
41
42 return (AVMasteringDisplayMetadata *)side_data->data;
43}

45AVContentLightMetadata *av_content_light_metadata_alloc(size_t *size)
46{
47 AVContentLightMetadata *metadata = av_mallocz(sizeof(AVContentLightMetadata));
48
49 if (size)
50 *size = sizeof(*metadata);
51
52 return metadata;
53}
