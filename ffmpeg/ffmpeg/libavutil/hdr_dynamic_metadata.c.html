21#include "hdr_dynamic_metadata.h"
22#include "mem.h"
23
24AVDynamicHDRPlus *av_dynamic_hdr_plus_alloc(size_t *size)
25{
26 AVDynamicHDRPlus *hdr_plus = av_mallocz(sizeof(AVDynamicHDRPlus));
27 if (!hdr_plus)
28 return NULL;
29
30 if (size)
31 *size = sizeof(*hdr_plus);
32
33 return hdr_plus;
34}
35
36AVDynamicHDRPlus *av_dynamic_hdr_plus_create_side_data(AVFrame *frame)
37{
38 AVFrameSideData *side_data = av_frame_new_side_data(frame,
39 AV_FRAME_DATA_DYNAMIC_HDR_PLUS,
40 sizeof(AVDynamicHDRPlus));
41 if (!side_data)
42 return NULL;
43
44 memset(side_data->data, 0, sizeof(AVDynamicHDRPlus));
45
46 return (AVDynamicHDRPlus *)side_data->data;
47}
