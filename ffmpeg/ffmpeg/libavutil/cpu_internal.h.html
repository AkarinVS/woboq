26#define CPUEXT_SUFFIX(flags, suffix, cpuext) \
27 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext))
28
29#define CPUEXT_SUFFIX_FAST2(flags, suffix, cpuext, slow_cpuext) \
30 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext) && \
31 !((flags) & AV_CPU_FLAG_ ## slow_cpuext ## SLOW))
32
33#define CPUEXT_SUFFIX_SLOW2(flags, suffix, cpuext, slow_cpuext) \
34 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext) && \
35 ((flags) & AV_CPU_FLAG_ ## slow_cpuext ## SLOW))
36
37#define CPUEXT_SUFFIX_FAST(flags, suffix, cpuext) CPUEXT_SUFFIX_FAST2(flags, suffix, cpuext, cpuext)
38#define CPUEXT_SUFFIX_SLOW(flags, suffix, cpuext) CPUEXT_SUFFIX_SLOW2(flags, suffix, cpuext, cpuext)
39
40#define CPUEXT(flags, cpuext) CPUEXT_SUFFIX(flags, , cpuext)
41#define CPUEXT_FAST(flags, cpuext) CPUEXT_SUFFIX_FAST(flags, , cpuext)
42#define CPUEXT_SLOW(flags, cpuext) CPUEXT_SUFFIX_SLOW(flags, , cpuext)
