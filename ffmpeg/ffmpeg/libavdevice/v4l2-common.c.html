73uint32_t ff_fmt_ff2v4l(enum AVPixelFormat pix_fmt, enum AVCodecID codec_id)
74{
75 int i;
76
77 for (i = 0; ff_fmt_conversion_table[i].codec_id != AV_CODEC_ID_NONE; i++) {
78 if ((codec_id == AV_CODEC_ID_NONE ||
79 ff_fmt_conversion_table[i].codec_id == codec_id) &&
80 (pix_fmt == AV_PIX_FMT_NONE ||
81 ff_fmt_conversion_table[i].ff_fmt == pix_fmt)) {
82 return ff_fmt_conversion_table[i].v4l2_fmt;
83 }
84 }
85
86 return 0;
87}

89enum AVPixelFormat ff_fmt_v4l2ff(uint32_t v4l2_fmt, enum AVCodecID codec_id)
90{
91 int i;
92
93 for (i = 0; ff_fmt_conversion_table[i].codec_id != AV_CODEC_ID_NONE; i++) {
94 if (ff_fmt_conversion_table[i].v4l2_fmt == v4l2_fmt &&
95 ff_fmt_conversion_table[i].codec_id == codec_id) {
96 return ff_fmt_conversion_table[i].ff_fmt;
97 }
98 }
99
100 return AV_PIX_FMT_NONE;
101}
