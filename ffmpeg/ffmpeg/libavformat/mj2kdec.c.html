26#if CONFIG_MJPEG_2000_DEMUXER
27static int mjpeg2000_probe(const AVProbeData *p)
28{
29 const uint8_t *b = p->buf;
30 int i, marker, marker_size;
31 int frames = 0, invalid = 0;
32
33 for (i = 0; i < p->buf_size - 5; i++) {
34 if (AV_RB32(b) == 0xff4fff51){
35 marker_size = AV_RB16(b+4);
36 if (marker_size + i < p->buf_size - 4) {
37 marker = AV_RB8(b+4+marker_size);
38 if (marker == 0xff)
39 frames++;
40 else
41 invalid++;
42 }
43 }
44 b += 1;
45 }
46 if (invalid*4 + 1 < frames) {
47 if (invalid == 0 && frames > 2)
48 return AVPROBE_SCORE_EXTENSION / 2;
49 return AVPROBE_SCORE_EXTENSION / 4;
50 }
51 return 0;
52}
53FF_DEF_RAWVIDEO_DEMUXER2(mjpeg_2000, "raw MJPEG 2000 video", mjpeg2000_probe, "j2k", AV_CODEC_ID_JPEG2000, AVFMT_GENERIC_INDEX|AVFMT_NOTIMESTAMPS)
54#endif
