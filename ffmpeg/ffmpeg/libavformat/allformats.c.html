581static void av_format_init_next(void)
582{
583 AVOutputFormat *prevout = NULL, *out;
584 AVInputFormat *previn = NULL, *in;
585
586 ff_mutex_lock(&avpriv_register_devices_mutex);
587
588 for (int i = 0; (out = (AVOutputFormat*)muxer_list[i]); i++) {

603 if (previn)

609 for (int i = 0; (in = (AVInputFormat*)indev_list[i]); i++) {
