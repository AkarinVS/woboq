323int ff_get_qtpalette(int codec_id, AVIOContext *pb, uint32_t *palette)
324{
325 int tmp, bit_depth, color_table_id, greyscale, i;
326
327 avio_seek(pb, 82, SEEK_CUR);
328
329 /* Get the bit depth and greyscale state */
330 tmp = avio_rb16(pb);
331 bit_depth = tmp & 0x1F;
332 greyscale = tmp & 0x20;
333
334 /* Get the color table ID */
335 color_table_id = avio_rb16(pb);
336
337 /* Do not create a greyscale palette for Cinepak */
338 if (greyscale && codec_id == AV_CODEC_ID_CINEPAK)
339 return 0;
340
341 /* If the depth is 1, 2, 4, or 8 bpp, file is palettized. */
342 if ((bit_depth == 1 || bit_depth == 2 || bit_depth == 4 || bit_depth == 8)) {
343 uint32_t color_count, color_start, color_end;
344 uint32_t r, g, b;
345
346 /* Ignore the greyscale bit for 1-bit video and sample
347 * descriptions containing a color table. */
348 if (greyscale && bit_depth > 1 && color_table_id) {
349 int color_index, color_dec;
350 /* compute the greyscale palette */
351 color_count = 1 << bit_depth;
352 color_index = 255;
353 color_dec = 256 / (color_count - 1);
354 for (i = 0; i < color_count; i++) {
355 r = g = b = color_index;
356 palette[i] = (0xFFU << 24) | (r << 16) | (g << 8) | (b);
357 color_index -= color_dec;
358 if (color_index < 0)
359 color_index = 0;
360 }
361 } else if (color_table_id) {
362 /* The color table ID is non-zero. Interpret this as
363 * being -1, which means use the default Macintosh
364 * color table */
365 const uint8_t *color_table;
366 color_count = 1 << bit_depth;
367 if (bit_depth == 1)
368 color_table = qt_default_palette_2;
369 else if (bit_depth == 2)
370 color_table = qt_default_palette_4;
371 else if (bit_depth == 4)
372 color_table = qt_default_palette_16;
373 else
374 color_table = qt_default_palette_256;
375 for (i = 0; i < color_count; i++) {
376 r = color_table[i * 3 + 0];
377 g = color_table[i * 3 + 1];
378 b = color_table[i * 3 + 2];
379 palette[i] = (0xFFU << 24) | (r << 16) | (g << 8) | (b);
380 }
381 } else {
382 /* The color table ID is 0; the color table is in the sample
383 * description */
384 color_start = avio_rb32(pb);
385 avio_rb16(pb); /* color table flags */
386 color_end = avio_rb16(pb);
387 if ((color_start <= 255) && (color_end <= 255)) {
388 for (i = color_start; i <= color_end; i++) {
389 /* Each color is made of four unsigned 16 bit integers. The
390 * first integer is 0, the remaining integers are the red,
391 * the green and the blue values. We only use the top 8 bit. */
392 avio_skip(pb, 2);
393 r = avio_r8(pb);
394 avio_r8(pb);
395 g = avio_r8(pb);
396 avio_r8(pb);
397 b = avio_r8(pb);
398 avio_r8(pb);
399 palette[i] = (0xFFU << 24) | (r << 16) | (g << 8) | (b);
400 }
401 }
402 }
403
404 return 1;
405 }
406
407 return 0;
408}
