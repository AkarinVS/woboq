40static int subviewer_probe(const AVProbeData *p)
41{
42 char c;
43 const unsigned char *ptr = p->buf;
44
45 if (AV_RB24(ptr) == 0xEFBBBF)
46 ptr += 3; /* skip UTF-8 BOM */
47 if (sscanf(ptr, "%*u:%*u:%*u.%*u,%*u:%*u:%*u.%*u%c", &c) == 1)
48 return AVPROBE_SCORE_EXTENSION;
49 if (!strncmp(ptr, "[INFORMATION]", 13))
50 return AVPROBE_SCORE_MAX/3;
51 return 0;
52}

54static int read_ts(const char *s, int64_t *start, int *duration)
55{
56 int64_t end;
57 int hh1, mm1, ss1, ms1;
58 int hh2, mm2, ss2, ms2;
59 int multiplier = 1;
60
61 if (sscanf(s, "%u:%u:%u.%2u,%u:%u:%u.%2u",
62 &hh1, &mm1, &ss1, &ms1, &hh2, &mm2, &ss2, &ms2) == 8) {
63 multiplier = 10;
64 } else if (sscanf(s, "%u:%u:%u.%1u,%u:%u:%u.%1u",
65 &hh1, &mm1, &ss1, &ms1, &hh2, &mm2, &ss2, &ms2) == 8) {
66 multiplier = 100;
67 }
68 if (sscanf(s, "%u:%u:%u.%u,%u:%u:%u.%u",
69 &hh1, &mm1, &ss1, &ms1, &hh2, &mm2, &ss2, &ms2) == 8) {
70 ms1 = FFMIN(ms1, 999);
71 ms2 = FFMIN(ms2, 999);
72 end = (hh2*3600LL + mm2*60LL + ss2) * 1000LL + ms2 * multiplier;
73 *start = (hh1*3600LL + mm1*60LL + ss1) * 1000LL + ms1 * multiplier;
74 *duration = end - *start;
75 return 0;
76 }
77 return -1;
78}

80static int subviewer_read_header(AVFormatContext *s)
81{
82 SubViewerContext *subviewer = s->priv_data;
83 AVStream *st = avformat_new_stream(s, NULL);
84 AVBPrint header;
85 int res = 0, new_event = 1;
86 int64_t pts_start = AV_NOPTS_VALUE;
87 int duration = -1;
88 AVPacket *sub = NULL;
89
90 if (!st)
91 return AVERROR(ENOMEM);
92 res = ffio_ensure_seekback(s->pb, 3);
93 if (res < 0)
94 return res;
95 if (avio_rb24(s->pb) != 0xefbbbf)
96 avio_seek(s->pb, -3, SEEK_CUR);
97 avpriv_set_pts_info(st, 64, 1, 1000);
98 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
99 st->codecpar->codec_id = AV_CODEC_ID_SUBVIEWER;
100
101 av_bprint_init(&header, 0, AV_BPRINT_SIZE_UNLIMITED);
102
103 while (!avio_feof(s->pb)) {
104 char line[2048];
105 int64_t pos = 0;
106 int len = ff_get_line(s->pb, line, sizeof(line));
107
108 if (!len)
109 break;
110
111 line[strcspn(line, "\r\n")] = 0;
112
113 if (line[0] == '[' && strncmp(line, "[br]", 4)) {
114
115 /* ignore event style, XXX: add to side_data? */
116 if (strstr(line, "[COLF]") || strstr(line, "[SIZE]") ||
117 strstr(line, "[FONT]") || strstr(line, "[STYLE]"))
118 continue;
119
120 if (!st->codecpar->extradata) { // header not finalized yet
121 av_bprintf(&header, "%s\n", line);
122 if (!strncmp(line, "[END INFORMATION]", 17) || !strncmp(line, "[SUBTITLE]", 10)) {
123 /* end of header */
124 res = ff_bprint_to_codecpar_extradata(st->codecpar, &header);
125 if (res < 0)
126 goto end;
127 } else if (strncmp(line, "[INFORMATION]", 13)) {
128 /* assume file metadata at this point */
129 int i, j = 0;
130 char key[32], value[128];
131
132 for (i = 1; i < sizeof(key) - 1 && line[i] && line[i] != ']'; i++)
133 key[i - 1] = av_tolower(line[i]);
134 key[i - 1] = 0;
135
136 if (line[i] == ']')
137 i++;
138 while (line[i] == ' ')
139 i++;
140 while (j < sizeof(value) - 1 && line[i] && line[i] != ']')
141 value[j++] = line[i++];
142 value[j] = 0;
143
144 av_dict_set(&s->metadata, key, value, 0);
145 }
146 }
147 } else if (read_ts(line, &pts_start, &duration) >= 0) {
148 new_event = 1;
149 pos = avio_tell(s->pb);
150 } else if (*line) {
151 if (pts_start == AV_NOPTS_VALUE) {
152 res = AVERROR_INVALIDDATA;
153 goto end;
154 }
155 if (!new_event) {
156 sub = ff_subtitles_queue_insert(&subviewer->q, "\n", 1, 1);
157 if (!sub) {
158 res = AVERROR(ENOMEM);
159 goto end;
160 }
161 }
162 sub = ff_subtitles_queue_insert(&subviewer->q, line, strlen(line), !new_event);
163 if (!sub) {
164 res = AVERROR(ENOMEM);
165 goto end;
166 }
167 if (new_event) {
168 sub->pos = pos;
169 sub->pts = pts_start;
170 sub->duration = duration;
171 }
172 new_event = 0;
173 }
174 }
175
176 ff_subtitles_queue_finalize(s, &subviewer->q);
177
178end:
179 if (res < 0)
180 ff_subtitles_queue_clean(&subviewer->q);
181 av_bprint_finalize(&header, NULL);
182 return res;
183}

185static int subviewer_read_packet(AVFormatContext *s, AVPacket *pkt)
186{
187 SubViewerContext *subviewer = s->priv_data;
188 return ff_subtitles_queue_read_packet(&subviewer->q, pkt);
189}
190
191static int subviewer_read_seek(AVFormatContext *s, int stream_index,
192 int64_t min_ts, int64_t ts, int64_t max_ts, int flags)
193{
194 SubViewerContext *subviewer = s->priv_data;
195 return ff_subtitles_queue_seek(&subviewer->q, s, stream_index,
196 min_ts, ts, max_ts, flags);
197}
198
199static int subviewer_read_close(AVFormatContext *s)
200{
201 SubViewerContext *subviewer = s->priv_data;
202 ff_subtitles_queue_clean(&subviewer->q);
203 return 0;
204}
