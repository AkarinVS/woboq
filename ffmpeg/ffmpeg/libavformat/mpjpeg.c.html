21#include "libavutil/opt.h"
22#include "avformat.h"
23
24/* Multipart JPEG */
25
26#define BOUNDARY_TAG "ffmpeg"
27
28typedef struct MPJPEGContext {
29 AVClass *class;
30 char *boundary_tag;
31} MPJPEGContext;
32
33static int mpjpeg_write_header(AVFormatContext *s)
34{
35 MPJPEGContext *mpj = s->priv_data;
36 avio_printf(s->pb, "--%s\r\n", mpj->boundary_tag);
37 return 0;
38}
39
40static int mpjpeg_write_packet(AVFormatContext *s, AVPacket *pkt)
41{
42 MPJPEGContext *mpj = s->priv_data;
43 avio_printf(s->pb, "Content-type: image/jpeg\r\n");
44 avio_printf(s->pb, "Content-length: %d\r\n\r\n",
45 pkt->size);
46 avio_write(s->pb, pkt->data, pkt->size);
47
48 avio_printf(s->pb, "\r\n--%s\r\n", mpj->boundary_tag);
49 return 0;
50}
51
52static const AVOption options[] = {
53 { "boundary_tag", "Boundary tag", offsetof(MPJPEGContext, boundary_tag), AV_OPT_TYPE_STRING, {.str = BOUNDARY_TAG}, .flags = AV_OPT_FLAG_ENCODING_PARAM },
54 { NULL },
55};
56
57static const AVClass mpjpeg_muxer_class = {
58 .class_name = "mpjpeg_muxer",
59 .item_name = av_default_item_name,
60 .option = options,
61 .version = LIBAVUTIL_VERSION_INT,
62};
63
64AVOutputFormat ff_mpjpeg_muxer = {
65 .name = "mpjpeg",
66 .long_name = NULL_IF_CONFIG_SMALL("MIME multipart JPEG"),
67 .mime_type = "multipart/x-mixed-replace;boundary=" BOUNDARY_TAG,
68 .extensions = "mjpg",
69 .priv_data_size = sizeof(MPJPEGContext),
70 .audio_codec = AV_CODEC_ID_NONE,
71 .video_codec = AV_CODEC_ID_MJPEG,
72 .write_header = mpjpeg_write_header,
73 .write_packet = mpjpeg_write_packet,
74 .flags = AVFMT_NOTIMESTAMPS,
75 .priv_class = &mpjpeg_muxer_class,
76};
