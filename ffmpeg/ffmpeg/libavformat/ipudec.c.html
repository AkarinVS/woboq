30static int ipu_read_probe(const AVProbeData *p)
31{
32 if (AV_RB32(p->buf) != MKBETAG('i', 'p', 'u', 'm'))
33 return 0;
34
35 if (AV_RL32(p->buf + 4) == 0)
36 return 0;
37
38 if (AV_RL16(p->buf + 8) == 0)
39 return 0;
40
41 if (AV_RL16(p->buf + 10) == 0)
42 return 0;
43
44 if (AV_RL32(p->buf + 12) == 0)
45 return 0;
46
47 return AVPROBE_SCORE_MAX;
48}
49
50static int ipu_read_header(AVFormatContext *s)
51{
52 AVIOContext *pb = s->pb;
53 AVStream *st = avformat_new_stream(s, NULL);
54
55 if (!st)
56 return AVERROR(ENOMEM);
57 avio_skip(pb, 8);
58 st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
59 st->codecpar->codec_id = AV_CODEC_ID_IPU;
60 st->codecpar->width = avio_rl16(pb);
61 st->codecpar->height = avio_rl16(pb);
62 st->start_time = 0;
63 st->duration =
64 st->nb_frames = avio_rl32(pb);
65 st->need_parsing = AVSTREAM_PARSE_FULL_RAW;
66 avpriv_set_pts_info(st, 64, 1, 25);
67
68 return 0;
69}
