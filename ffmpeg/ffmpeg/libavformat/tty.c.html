37static int isansicode(int x)
38{
39 return x == 0x1B || x == 0x0A || x == 0x0D || (x >= 0x20 && x < 0x7f);
40}

42static const char tty_extensions[31] = "ans,art,asc,diz,ice,nfo,txt,vt";
43
44typedef struct TtyDemuxContext {
45 AVClass *class;
46 int chars_per_frame;
47 uint64_t fsize; /**< file size less metadata buffer */
48 int width, height; /**< Set by a private option. */
49 AVRational framerate; /**< Set by a private option. */
50} TtyDemuxContext;

52static int read_probe(const AVProbeData *p)
53{
54 int cnt = 0;
55
56 if (!p->buf_size)
57 return 0;
58
59 for (int i = 0; i < 8 && i < p->buf_size; i++)
60 cnt += !!isansicode(p->buf[i]);
61
62 if (cnt != 8)
63 return 0;
64
65 for (int i = 8; i < p->buf_size; i++)
66 cnt += !!isansicode(p->buf[i]);
67
68 return (cnt * 99LL / p->buf_size) * (cnt > 400) *
69 !!av_match_ext(p->filename, tty_extensions);
70}

72/**
73 * Parse EFI header
74 */
75static int efi_read(AVFormatContext *avctx, uint64_t start_pos)
76{
77 TtyDemuxContext *s = avctx->priv_data;
78 AVIOContext *pb = avctx->pb;
79 char buf[37];
80 int len;
81
82 avio_seek(pb, start_pos, SEEK_SET);
83 if (avio_r8(pb) != 0x1A)
84 return -1;
85
86#define GET_EFI_META(name,size) \
87 len = avio_r8(pb); \
88 if (len < 1 || len > size) \
89 return -1; \
90 if (avio_read(pb, buf, size) == size) { \
91 buf[len] = 0; \
92 av_dict_set(&avctx->metadata, name, buf, 0); \
93 }
94
95 GET_EFI_META("filename", 12)
96 GET_EFI_META("title", 36)
97
98 s->fsize = start_pos;
99 return 0;
100}

102static int read_header(AVFormatContext *avctx)
103{
104 TtyDemuxContext *s = avctx->priv_data;
105 int ret = 0;
106 AVStream *st = avformat_new_stream(avctx, NULL);
107
108 if (!st) {
109 ret = AVERROR(ENOMEM);
110 goto fail;
111 }
112 st->codecpar->codec_tag = 0;
113 st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
114 st->codecpar->codec_id = AV_CODEC_ID_ANSI;
115
116 st->codecpar->width = s->width;
117 st->codecpar->height = s->height;
118 avpriv_set_pts_info(st, 60, s->framerate.den, s->framerate.num);
119 st->avg_frame_rate = s->framerate;
120
121 /* simulate tty display speed */
122 s->chars_per_frame = FFMAX(av_q2d(st->time_base)*s->chars_per_frame, 1);
123
124 if (avctx->pb->seekable & AVIO_SEEKABLE_NORMAL) {
125 s->fsize = avio_size(avctx->pb);
126 st->duration = (s->fsize + s->chars_per_frame - 1) / s->chars_per_frame;
127
128 if (ff_sauce_read(avctx, &s->fsize, 0, 0) < 0)
129 efi_read(avctx, s->fsize - 51);
130
131 avio_seek(avctx->pb, 0, SEEK_SET);
132 }
133
134fail:
135 return ret;
136}

138static int read_packet(AVFormatContext *avctx, AVPacket *pkt)
139{
140 TtyDemuxContext *s = avctx->priv_data;
141 int n;
142
143 if (avio_feof(avctx->pb))
144 return AVERROR_EOF;
145
146 n = s->chars_per_frame;
147 if (s->fsize) {
148 // ignore metadata buffer
149 uint64_t p = avio_tell(avctx->pb);
150 if (p == s->fsize)
151 return AVERROR_EOF;
152 if (p + s->chars_per_frame > s->fsize)
153 n = s->fsize - p;
154 }
155
156 pkt->size = av_get_packet(avctx->pb, pkt, n);
157 if (pkt->size < 0)
158 return pkt->size;
159 pkt->stream_index = 0;
160 pkt->pts = pkt->pos / s->chars_per_frame;
161 pkt->flags |= AV_PKT_FLAG_KEY;
162 return 0;
163}

165#define OFFSET(x) offsetof(TtyDemuxContext, x)
166#define DEC AV_OPT_FLAG_DECODING_PARAM
167static const AVOption options[] = {
168 { "chars_per_frame", "", offsetof(TtyDemuxContext, chars_per_frame), AV_OPT_TYPE_INT, {.i64 = 6000}, 1, INT_MAX, AV_OPT_FLAG_DECODING_PARAM},
169 { "video_size", "A string describing frame size, such as 640x480 or hd720.", OFFSET(width), AV_OPT_TYPE_IMAGE_SIZE, {.str = NULL}, 0, 0, DEC },
170 { "framerate", "", OFFSET(framerate), AV_OPT_TYPE_VIDEO_RATE, {.str = "25"}, 0, INT_MAX, DEC },
171 { NULL },
172};
