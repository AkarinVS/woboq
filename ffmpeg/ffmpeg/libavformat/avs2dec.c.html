24#include "rawdec.h"
25#include "libavcodec/internal.h"
26#include "libavutil/intreadwrite.h"

28#define AVS2_ISSQH(x) ((x) == 0xB0)
29#define AVS2_ISEND(x) ((x) == 0xB1)
30#define AVS2_ISPIC(x) ((x) == 0xB3 || (x) == 0xB6)
31#define AVS2_ISUNIT(x) (AVS2_ISSQH(x) || AVS2_ISEND(x) || (x) == 0xB2 || AVS2_ISPIC(x) || (x) == 0xB5 || (x) == 0xB7)
32#define AVS2_ISPROFILE(x) ((x) == 0x20 || (x) == 0x22 || (x) == 0x30 || (x) == 0x32)
33
34static int avs2_probe(const AVProbeData *p)
35{
36 uint32_t code = -1, hds = 0, pic = 0, seq = 0;
37 uint8_t state = 0;
38 const uint8_t *ptr = p->buf, *end = p->buf + p->buf_size, *sqb = 0;
39 if (AV_RB32(p->buf) != 0x1B0){
40 return 0;
41 }
42
43 while (ptr < end) {
44 ptr = avpriv_find_start_code(ptr, end, &code);
45 state = code & 0xFF;
46 if ((code & 0xffffff00) == 0x100) {
47 if (AVS2_ISUNIT(state)) {
48 if (sqb && !hds) {
49 hds = ptr - sqb;
50 }
51 if (AVS2_ISSQH(state)) {
52 if (!AVS2_ISPROFILE(*ptr))
53 return 0;
54 sqb = ptr;
55 seq++;
56 } else if (AVS2_ISPIC(state)) {
57 pic++;
58 } else if (AVS2_ISEND(state)) {
59 break;
60 }
61 }
62 }
63 }
64 if (seq && hds >= 21 && pic){
65 return AVPROBE_SCORE_EXTENSION + 2; // more than cavs
66 }
67
68 return 0;
69}
