22#include "avformat.h"
23#include "internal.h"
24
25#define CDG_PACKET_SIZE 24
26#define CDG_COMMAND 0x09
27#define CDG_MASK 0x3F
28
29static int read_header(AVFormatContext *s)
30{
31 AVStream *vst;
32 int ret;
33
34 vst = avformat_new_stream(s, NULL);
35 if (!vst)
36 return AVERROR(ENOMEM);
37
38 vst->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
39 vst->codecpar->codec_id = AV_CODEC_ID_CDGRAPHICS;
40
41 /// 75 sectors/sec * 4 packets/sector = 300 packets/sec
42 avpriv_set_pts_info(vst, 32, 1, 300);
43
44 ret = avio_size(s->pb);
45 if (ret < 0) {
46 av_log(s, AV_LOG_WARNING, "Cannot calculate duration as file size cannot be determined\n");
47 } else
48 vst->duration = (ret * (int64_t)vst->time_base.den) / (CDG_PACKET_SIZE * 300);
49
50 return 0;
51}
52
53static int read_packet(AVFormatContext *s, AVPacket *pkt)
54{
55 int ret;
56
57 ret = av_get_packet(s->pb, pkt, CDG_PACKET_SIZE);
58 pkt->stream_index = 0;
59 pkt->dts=
60 pkt->pts= pkt->pos / CDG_PACKET_SIZE;
61
62 if (!pkt->pos || (ret > 5 &&
63 (pkt->data[0] & CDG_MASK) == CDG_COMMAND &&
64 (pkt->data[1] & CDG_MASK) == 1 && !(pkt->data[2+2+1] & 0x0F))) {
65 pkt->flags = AV_PKT_FLAG_KEY;
66 }
67 return ret;
68}

70AVInputFormat ff_cdg_demuxer = {
71 .name = "cdg",
72 .long_name = NULL_IF_CONFIG_SMALL("CD Graphics"),
73 .read_header = read_header,
74 .read_packet = read_packet,
75 .flags = AVFMT_GENERIC_INDEX,
76 .extensions = "cdg",
77};
