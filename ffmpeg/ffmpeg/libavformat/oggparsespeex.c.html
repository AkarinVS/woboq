37struct speex_params {
38 int packet_size;
39 int final_packet_duration;
40 int seq;
41};

43static int speex_header(AVFormatContext *s, int idx) {
44 struct ogg *ogg = s->priv_data;
45 struct ogg_stream *os = ogg->streams + idx;
46 struct speex_params *spxp = os->private;
47 AVStream *st = s->streams[idx];
48 uint8_t *p = os->buf + os->pstart;
49 int ret;
50
51 if (!spxp) {
52 spxp = av_mallocz(sizeof(*spxp));
53 if (!spxp)
54 return AVERROR(ENOMEM);
55 os->private = spxp;
56 }
57
58 if (spxp->seq > 1)
59 return 0;
60
61 if (spxp->seq == 0) {
62 int frames_per_packet;
63 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
64 st->codecpar->codec_id = AV_CODEC_ID_SPEEX;
65
66 if (os->psize < 68) {
67 av_log(s, AV_LOG_ERROR, "speex packet too small\n");
68 return AVERROR_INVALIDDATA;
69 }
70
71 st->codecpar->sample_rate = AV_RL32(p + 36);
72 if (st->codecpar->sample_rate <= 0) {
73 av_log(s, AV_LOG_ERROR, "Invalid sample rate %d\n", st->codecpar->sample_rate);
74 return AVERROR_INVALIDDATA;
75 }
76 st->codecpar->channels = AV_RL32(p + 48);
77 if (st->codecpar->channels < 1 || st->codecpar->channels > 2) {
78 av_log(s, AV_LOG_ERROR, "invalid channel count. Speex must be mono or stereo.\n");
79 return AVERROR_INVALIDDATA;
80 }
81 st->codecpar->channel_layout = st->codecpar->channels == 1 ? AV_CH_LAYOUT_MONO :
82 AV_CH_LAYOUT_STEREO;
83
84 spxp->packet_size = AV_RL32(p + 56);
85 frames_per_packet = AV_RL32(p + 64);
86 if (spxp->packet_size < 0 ||
87 frames_per_packet < 0 ||
88 spxp->packet_size * (int64_t)frames_per_packet > INT32_MAX / 256) {
89 av_log(s, AV_LOG_ERROR, "invalid packet_size, frames_per_packet %d %d\n", spxp->packet_size, frames_per_packet);
90 spxp->packet_size = 0;
91 return AVERROR_INVALIDDATA;
92 }
93 if (frames_per_packet)
94 spxp->packet_size *= frames_per_packet;
95
96 if ((ret = ff_alloc_extradata(st->codecpar, os->psize)) < 0)
97 return ret;
98 memcpy(st->codecpar->extradata, p, st->codecpar->extradata_size);
99
100 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
101 } else
102 ff_vorbis_stream_comment(s, st, p, os->psize);
103
104 spxp->seq++;
105 return 1;
106}

108static int ogg_page_packets(struct ogg_stream *os)
109{
110 int i;
111 int packets = 0;
112 for (i = 0; i < os->nsegs; i++)
113 if (os->segments[i] < 255)
114 packets++;
115 return packets;
116}
117
118static int speex_packet(AVFormatContext *s, int idx)
119{
120 struct ogg *ogg = s->priv_data;
121 struct ogg_stream *os = ogg->streams + idx;
122 struct speex_params *spxp = os->private;
123 int packet_size = spxp->packet_size;
124
125 if (os->flags & OGG_FLAG_EOS && os->lastpts != AV_NOPTS_VALUE &&
126 os->granule > 0) {
127 /* first packet of final page. we have to calculate the final packet
128 duration here because it is the only place we know the next-to-last
129 granule position. */
130 spxp->final_packet_duration = os->granule - os->lastpts -
131 packet_size * (ogg_page_packets(os) - 1);
132 }
133
134 if (!os->lastpts && os->granule > 0)
135 /* first packet */
136 os->lastpts = os->lastdts = os->granule - packet_size *
137 ogg_page_packets(os);
138 if (os->flags & OGG_FLAG_EOS && os->segp == os->nsegs &&
139 spxp->final_packet_duration)
140 /* final packet */
141 os->pduration = spxp->final_packet_duration;
142 else
143 os->pduration = packet_size;
144
145 return 0;
146}
