19#include "avformat.h"
20#include "avio.h"
21#include "isom.h"
22
23int ff_mov_read_esds(AVFormatContext *fc, AVIOContext *pb)
24{
25 AVStream *st;
26 int tag, ret = 0;
27
28 if (fc->nb_streams < 1)
29 return 0;
30 st = fc->streams[fc->nb_streams-1];
31
32 avio_rb32(pb); /* version + flags */
33 ff_mp4_read_descr(fc, pb, &tag);
34 if (tag == MP4ESDescrTag) {
35 ff_mp4_parse_es_descr(pb, NULL);
36 } else
37 avio_rb16(pb); /* ID */
38
39 ff_mp4_read_descr(fc, pb, &tag);
40 if (tag == MP4DecConfigDescrTag)
41 ret = ff_mp4_read_dec_config_descr(fc, st, pb);
42
43 return ret;
44}
