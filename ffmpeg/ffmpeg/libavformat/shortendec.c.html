28static int shn_probe(const AVProbeData *p)
29{
30 GetBitContext gb;
31 int version, internal_ftype, channels, blocksize;
32
33 if (AV_RB32(p->buf) != 0x616a6b67)
34 return 0;
35 version = p->buf[4];
36 if (init_get_bits8(&gb, p->buf + 5, p->buf_size - 5 - AV_INPUT_BUFFER_PADDING_SIZE) < 0)
37 return 0;
38 if (!version) {
39 internal_ftype = get_ur_golomb_shorten(&gb, 4);
40 channels = get_ur_golomb_shorten(&gb, 0);
41 blocksize = 256;
42 } else {
43 unsigned k;
44 k = get_ur_golomb_shorten(&gb, 2);
45 if (k > 31)
46 return 0;
47 internal_ftype = get_ur_golomb_shorten(&gb, k);
48 k = get_ur_golomb_shorten(&gb, 2);
49 if (k > 31)
50 return 0;
51 channels = get_ur_golomb_shorten(&gb, k);
52 k = get_ur_golomb_shorten(&gb, 2);
53 if (k > 31)
54 return 0;
55 blocksize = get_ur_golomb_shorten(&gb, k);
56 }
57
58 if (internal_ftype != 2 && internal_ftype != 3 && internal_ftype != 5)
59 return 0;
60 if (channels < 1 || channels > 8)
61 return 0;
62 if (blocksize < 1 || blocksize > 65535)
63 return 0;
64
65 return AVPROBE_SCORE_EXTENSION + 1;
66}

75 .flags = AVFMT_NOBINSEARCH | AVFMT_NOGENSEARCH | AVFMT_NO_BYTE_SEEK | AVFMT_NOTIMESTAMPS,
76 .extensions = "shn",
77 .raw_codec_id = AV_CODEC_ID_SHORTEN,
78 .priv_data_size = sizeof(FFRawDemuxerContext),
79 .priv_class = &shorten_demuxer_class,
80};
