28static int apc_probe(const AVProbeData *p)
29{
30 if (!strncmp(p->buf, "CRYO_APC", 8))
31 return AVPROBE_SCORE_MAX;
32
33 return 0;
34}
35
36static int apc_read_header(AVFormatContext *s)
37{
38 AVIOContext *pb = s->pb;
39 AVStream *st;
40 int ret;
41
42 avio_rl32(pb); /* CRYO */
43 avio_rl32(pb); /* _APC */
44 avio_rl32(pb); /* 1.20 */
45
46 st = avformat_new_stream(s, NULL);
47 if (!st)
48 return AVERROR(ENOMEM);
49
50 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
51 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_IMA_APC;
52
53 avio_rl32(pb); /* number of samples */
54 st->codecpar->sample_rate = avio_rl32(pb);
55
56 /* initial predictor values for adpcm decoder */
57 if ((ret = ff_get_extradata(s, st->codecpar, pb, 2 * 4)) < 0)
58 return ret;
59
60 if (avio_rl32(pb)) {
61 st->codecpar->channels = 2;
62 st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
63 } else {
64 st->codecpar->channels = 1;
65 st->codecpar->channel_layout = AV_CH_LAYOUT_MONO;
66 }
67
68 st->codecpar->bits_per_coded_sample = 4;
69 st->codecpar->bit_rate = (int64_t)st->codecpar->bits_per_coded_sample * st->codecpar->channels
70 * st->codecpar->sample_rate;
71 st->codecpar->block_align = 1;
72
73 return 0;
74}
75
76#define MAX_READ_SIZE 4096
77
78static int apc_read_packet(AVFormatContext *s, AVPacket *pkt)
79{
80 if (av_get_packet(s->pb, pkt, MAX_READ_SIZE) <= 0)
81 return AVERROR(EIO);
82 pkt->stream_index = 0;
83 return 0;
84}

86AVInputFormat ff_apc_demuxer = {
87 .name = "apc",
88 .long_name = NULL_IF_CONFIG_SMALL("CRYO APC"),
89 .read_probe = apc_probe,
90 .read_header = apc_read_header,
91 .read_packet = apc_read_packet,
92};
