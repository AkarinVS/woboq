233int ff_mov_iso639_to_lang(const char lang[4], int mp4)
234{
235 int i, code = 0;
236
237 /* old way, only for QT? */
238 for (i = 0; lang[0] && !mp4 && i < FF_ARRAY_ELEMS(mov_mdhd_language_map); i++) {
239 if (!strcmp(lang, mov_mdhd_language_map[i]))
240 return i;
241 }
242 /* XXX:can we do that in mov too? */
243 if (!mp4)
244 return -1;
245 /* handle undefined as such */
246 if (lang[0] == '\0')
247 lang = "und";
248 /* 5 bits ASCII */
249 for (i = 0; i < 3; i++) {
250 uint8_t c = lang[i];
251 c -= 0x60;
252 if (c > 0x1f)
253 return -1;
254 code <<= 5;
255 code |= c;
256 }
257 return code;
258}

260int ff_mov_lang_to_iso639(unsigned code, char to[4])
261{
262 int i;
263 memset(to, 0, 4);
264 /* is it the mangled iso code? */
265 /* see http://www.geocities.com/xhelmboyx/quicktime/formats/mp4-layout.txt */
266 if (code >= 0x400 && code != 0x7fff) {
267 for (i = 2; i >= 0; i--) {
268 to[i] = 0x60 + (code & 0x1f);
269 code >>= 5;
270 }
271 return 1;
272 }
273 /* old fashion apple lang code */
274 if (code >= FF_ARRAY_ELEMS(mov_mdhd_language_map))
275 return 0;
276 if (!mov_mdhd_language_map[code][0])
277 return 0;
278 memcpy(to, mov_mdhd_language_map[code], 4);
279 return 1;
280}

282int ff_mp4_read_descr_len(AVIOContext *pb)
283{
284 int len = 0;
285 int count = 4;
286 while (count--) {
287 int c = avio_r8(pb);
288 len = (len << 7) | (c & 0x7f);
289 if (!(c & 0x80))
290 break;
291 }
292 return len;
293}

295int ff_mp4_read_descr(AVFormatContext *fc, AVIOContext *pb, int *tag)
296{
297 int len;
298 *tag = avio_r8(pb);
299 len = ff_mp4_read_descr_len(pb);
300 av_log(fc, AV_LOG_TRACE, "MPEG-4 description: tag=0x%02x len=%d\n", *tag, len);
301 return len;
302}

304void ff_mp4_parse_es_descr(AVIOContext *pb, int *es_id)
305{
306 int flags;
307 if (es_id) *es_id = avio_rb16(pb);
308 else avio_rb16(pb);
309 flags = avio_r8(pb);
310 if (flags & 0x80) //streamDependenceFlag
311 avio_rb16(pb);
312 if (flags & 0x40) { //URL_Flag
313 int len = avio_r8(pb);
314 avio_skip(pb, len);
315 }
316 if (flags & 0x20) //OCRstreamFlag
317 avio_rb16(pb);
318}

329int ff_mp4_read_dec_config_descr(AVFormatContext *fc, AVStream *st, AVIOContext *pb)
330{
331 enum AVCodecID codec_id;
332 unsigned v;
333 int len, tag;
334 int ret;
335 int object_type_id = avio_r8(pb);
336 avio_r8(pb); /* stream type */
337 avio_rb24(pb); /* buffer size db */
338
339 v = avio_rb32(pb);
340
341 // TODO: fix this with codecpar
342#if FF_API_LAVF_AVCTX
343FF_DISABLE_DEPRECATION_WARNINGS
344 if (v < INT32_MAX)
345 st->codec->rc_max_rate = v;
346FF_ENABLE_DEPRECATION_WARNINGS
347#endif
348
349 st->codecpar->bit_rate = avio_rb32(pb); /* avg bitrate */
350
351 codec_id= ff_codec_get_id(ff_mp4_obj_type, object_type_id);
352 if (codec_id)
353 st->codecpar->codec_id = codec_id;
354 av_log(fc, AV_LOG_TRACE, "esds object type id 0x%02x\n", object_type_id);
355 len = ff_mp4_read_descr(fc, pb, &tag);
356 if (tag == MP4DecSpecificDescrTag) {
357 av_log(fc, AV_LOG_TRACE, "Specific MPEG-4 header len=%d\n", len);
358 /* As per 14496-3:2009 9.D.2.2, No decSpecificInfo is defined
359 for MPEG-1 Audio or MPEG-2 Audio; MPEG-2 AAC excluded. */
360 if (object_type_id == 0x69 || object_type_id == 0x6b)
361 return 0;
362 if (!len || (uint64_t)len > (1<<30))
363 return AVERROR_INVALIDDATA;
364 if ((ret = ff_get_extradata(fc, st->codecpar, pb, len)) < 0)
365 return ret;
366 if (st->codecpar->codec_id == AV_CODEC_ID_AAC) {
367 MPEG4AudioConfig cfg = {0};
368 ret = avpriv_mpeg4audio_get_config2(&cfg, st->codecpar->extradata,
369 st->codecpar->extradata_size, 1, fc);
370 if (ret < 0)
371 return ret;
372 st->codecpar->channels = cfg.channels;
373 if (cfg.object_type == 29 && cfg.sampling_index < 3) // old mp3on4
374 st->codecpar->sample_rate = avpriv_mpa_freq_tab[cfg.sampling_index];
375 else if (cfg.ext_sample_rate)
376 st->codecpar->sample_rate = cfg.ext_sample_rate;
377 else
378 st->codecpar->sample_rate = cfg.sample_rate;
379 av_log(fc, AV_LOG_TRACE, "mp4a config channels %d obj %d ext obj %d "
380 "sample rate %d ext sample rate %d\n", st->codecpar->channels,
381 cfg.object_type, cfg.ext_object_type,
382 cfg.sample_rate, cfg.ext_sample_rate);
383 if (!(st->codecpar->codec_id = ff_codec_get_id(mp4_audio_types,
384 cfg.object_type)))
385 st->codecpar->codec_id = AV_CODEC_ID_AAC;
386 }
387 }
388 return 0;
389}

422void ff_mov_write_chan(AVIOContext *pb, int64_t channel_layout)
423{
424 const MovChannelLayout *layouts;
425 uint32_t layout_tag = 0;
426
427 for (layouts = mov_channel_layout; layouts->channel_layout; layouts++)
428 if (channel_layout == layouts->channel_layout) {
429 layout_tag = layouts->layout_tag;
430 break;
431 }
432
433 if (layout_tag) {
434 avio_wb32(pb, layout_tag); // mChannelLayoutTag
435 avio_wb32(pb, 0); // mChannelBitmap
436 } else {
437 avio_wb32(pb, 0x10000); // kCAFChannelLayoutTag_UseChannelBitmap
438 avio_wb32(pb, channel_layout);
439 }
440 avio_wb32(pb, 0); // mNumberChannelDescriptions
441}
