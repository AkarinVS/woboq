375/**
376 * Compute codec id for 'lpcm' tag.
377 * See CoreAudioTypes and AudioStreamBasicDescription at Apple.
378 */
379static inline enum AVCodecID ff_mov_get_lpcm_codec_id(int bps, int flags)
380{
381 /* lpcm flags:
382 * 0x1 = float
383 * 0x2 = big-endian
384 * 0x4 = signed
385 */
386 return ff_get_pcm_codec_id(bps, flags & 1, flags & 2, flags & 4 ? -1 : 0);
387}
