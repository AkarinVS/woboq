43static int timed_line(const char *ptr)
44{
45 char c;
46 int fs, fe;
47 return (sscanf(ptr, "%*u:%*u:%*u.%*u %*u:%*u:%*u.%*u %c", &c) == 1 ||
48 (sscanf(ptr, "@%u @%u %c", &fs, &fe, &c) == 3 && fs < fe));
49}

51static int jacosub_probe(const AVProbeData *p)
52{
53 const char *ptr = p->buf;
54 const char *ptr_end = p->buf + p->buf_size;
55
56 if (AV_RB24(ptr) == 0xEFBBBF)
57 ptr += 3; /* skip UTF-8 BOM */
58
59 while (ptr < ptr_end) {
60 while (jss_whitespace(*ptr))
61 ptr++;
62 if (*ptr != '#' && *ptr != '\n') {
63 if (timed_line(ptr))
64 return AVPROBE_SCORE_EXTENSION + 1;
65 return 0;
66 }
67 ptr += ff_subtitles_next_line(ptr);
68 }
69 return 0;
70}

72static const char * const cmds[] = {
73 "CLOCKPAUSE",
74 "DIRECTIVE",
75 "FONT",
76 "HRES",
77 "INCLUDE",
78 "PALETTE",
79 "QUANTIZE",
80 "RAMP",
81 "SHIFT",
82 "TIMERES",
83};

85static int get_jss_cmd(char k)
86{
87 int i;
88
89 k = av_toupper(k);
90 for (i = 0; i < FF_ARRAY_ELEMS(cmds); i++)
91 if (k == cmds[i][0])
92 return i;
93 return -1;
94}

103static const char *read_ts(JACOsubContext *jacosub, const char *buf,
104 int64_t *start, int64_t *duration)
105{
106 int len;
107 unsigned hs, ms, ss, fs; // hours, minutes, seconds, frame start
108 unsigned he, me, se, fe; // hours, minutes, seconds, frame end
109 int ts_start, ts_end;
110 int64_t ts_start64, ts_end64;
111
112 /* timed format */
113 if (sscanf(buf, "%u:%u:%u.%u %u:%u:%u.%u %n",
114 &hs, &ms, &ss, &fs,
115 &he, &me, &se, &fe, &len) == 8) {
116 ts_start = (hs*3600 + ms*60 + ss) * jacosub->timeres + fs;
117 ts_end = (he*3600 + me*60 + se) * jacosub->timeres + fe;
118 goto shift_and_ret;
119 }
120
121 /* timestamps format */
122 if (sscanf(buf, "@%u @%u %n", &ts_start, &ts_end, &len) == 2)
123 goto shift_and_ret;
124
125 return NULL;
126
127shift_and_ret:
128 ts_start64 = (ts_start + (int64_t)jacosub->shift) * 100LL / jacosub->timeres;
129 ts_end64 = (ts_end + (int64_t)jacosub->shift) * 100LL / jacosub->timeres;
130 *start = ts_start64;
131 *duration = ts_end64 - ts_start64;
132 return buf + len;
133}

135static int get_shift(int timeres, const char *buf)
136{
137 int sign = 1;
138 int a = 0, b = 0, c = 0, d = 0;
139 int64_t ret;
140#define SSEP "%*1[.:]"
141 int n = sscanf(buf, "%d"SSEP"%d"SSEP"%d"SSEP"%d", &a, &b, &c, &d);
142#undef SSEP
143
144 if (*buf == '-' || a < 0) {
145 sign = -1;
146 a = FFABS(a);
147 }
148
149 ret = 0;
150 switch (n) {
151 case 4:
152 ret = sign * (((int64_t)a*3600 + b*60 + c) * timeres + d);
153 break;
154 case 3:
155 ret = sign * (( (int64_t)a*60 + b) * timeres + c);
156 break;
157 case 2:
158 ret = sign * (( (int64_t)a) * timeres + b);
159 break;
160 }
161 if ((int)ret != ret)
162 ret = 0;
163
164 return ret;
165}

167static int jacosub_read_header(AVFormatContext *s)
168{
169 AVBPrint header;
170 AVIOContext *pb = s->pb;
171 char line[JSS_MAX_LINESIZE];
172 JACOsubContext *jacosub = s->priv_data;
173 int shift_set = 0; // only the first shift matters
174 int merge_line = 0;
175 int i, ret;
176
177 AVStream *st = avformat_new_stream(s, NULL);
178 if (!st)
179 return AVERROR(ENOMEM);
180 avpriv_set_pts_info(st, 64, 1, 100);
181 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
182 st->codecpar->codec_id = AV_CODEC_ID_JACOSUB;
183
184 jacosub->timeres = 30;
185
186 av_bprint_init(&header, 1024+AV_INPUT_BUFFER_PADDING_SIZE, 4096);
187
188 while (!avio_feof(pb)) {
189 int cmd_len;
190 const char *p = line;
191 int64_t pos = avio_tell(pb);
192 int len = ff_get_line(pb, line, sizeof(line));
193
194 p = jss_skip_whitespace(p);
195
196 /* queue timed line */
197 if (merge_line || timed_line(p)) {
198 AVPacket *sub;
199
200 sub = ff_subtitles_queue_insert(&jacosub->q, line, len, merge_line);
201 if (!sub) {
202 av_bprint_finalize(&header, NULL);
203 ret = AVERROR(ENOMEM);
204 goto fail;
205 }
206 sub->pos = pos;
207 merge_line = len > 1 && !strcmp(&line[len - 2], "\\\n");
208 continue;
209 }
210
211 /* skip all non-compiler commands and focus on the command */
212 if (*p != '#')
213 continue;
214 p++;
215 i = get_jss_cmd(p[0]);
216 if (i == -1)
217 continue;
218
219 /* trim command + spaces */
220 cmd_len = strlen(cmds[i]);
221 if (av_strncasecmp(p, cmds[i], cmd_len) == 0)
222 p += cmd_len;
223 else
224 p++;
225 p = jss_skip_whitespace(p);
226
227 /* handle commands which affect the whole script */
228 switch (cmds[i][0]) {
229 case 'S': // SHIFT command affect the whole script...
230 if (!shift_set) {
231 jacosub->shift = get_shift(jacosub->timeres, p);
232 shift_set = 1;
233 }
234 av_bprintf(&header, "#S %s", p);
235 break;
236 case 'T': // ...but must be placed after TIMERES
237 jacosub->timeres = strtol(p, NULL, 10);
238 if (!jacosub->timeres)
239 jacosub->timeres = 30;
240 else
241 av_bprintf(&header, "#T %s", p);
242 break;
243 }
244 }
245
246 /* general/essential directives in the extradata */
247 ret = ff_bprint_to_codecpar_extradata(st->codecpar, &header);
248 if (ret < 0)
249 goto fail;
250
251 /* SHIFT and TIMERES affect the whole script so packet timing can only be
252 * done in a second pass */
253 for (i = 0; i < jacosub->q.nb_subs; i++) {
254 AVPacket *sub = jacosub->q.subs[i];
255 read_ts(jacosub, sub->data, &sub->pts, &sub->duration);
256 }
257 ff_subtitles_queue_finalize(s, &jacosub->q);
258
259 return 0;
260fail:
261 jacosub_read_close(s);
262 return ret;
263}

265static int jacosub_read_packet(AVFormatContext *s, AVPacket *pkt)
266{
267 JACOsubContext *jacosub = s->priv_data;
268 return ff_subtitles_queue_read_packet(&jacosub->q, pkt);
269}
