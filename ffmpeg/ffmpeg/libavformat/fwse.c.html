27static int fwse_probe(const AVProbeData *p)
28{
29 if (AV_RL32(p->buf) != MKTAG('F','W','S','E'))
30 return 0;
31 if (AV_RL32(p->buf+4) != 2 && AV_RL32(p->buf+4) != 3)
32 return 0;
33 if (AV_RL32(p->buf+16) != 1 && AV_RL32(p->buf+16) != 2)
34 return 0;
35
36 return AVPROBE_SCORE_MAX / 4 * 3;
37}
38
39static int fwse_read_header(AVFormatContext *s)
40{
41 unsigned start_offset, version;
42 AVIOContext *pb = s->pb;
43 AVCodecParameters *par;
44 AVStream *st;
45
46 avio_skip(pb, 4);
47 version = avio_rl32(pb);
48 if (version != 2 && version != 3)
49 return AVERROR_INVALIDDATA;
50 avio_skip(pb, 4);
51 start_offset = avio_rl32(pb);
52
53 st = avformat_new_stream(s, NULL);
54 if (!st)
55 return AVERROR(ENOMEM);
56
57 par = st->codecpar;
58 par->codec_type = AVMEDIA_TYPE_AUDIO;
59 par->codec_id = AV_CODEC_ID_ADPCM_IMA_MTF;
60 par->format = AV_SAMPLE_FMT_S16;
61 par->channels = avio_rl32(pb);
62 if (par->channels != 1 && par->channels != 2)
63 return AVERROR_INVALIDDATA;
64 if (par->channels == 1)
65 par->channel_layout = AV_CH_LAYOUT_MONO;
66 else if (par->channels == 2)
67 par->channel_layout = AV_CH_LAYOUT_STEREO;
68 st->duration = avio_rl32(pb);
69 par->sample_rate = avio_rl32(pb);
70 if (par->sample_rate <= 0 || par->sample_rate > INT_MAX)
71 return AVERROR_INVALIDDATA;
72
73 par->block_align = 1;
74 avio_skip(pb, start_offset - avio_tell(pb));
75
76 avpriv_set_pts_info(st, 64, 1, par->sample_rate);
77
78 return 0;
79}
