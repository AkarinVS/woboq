19#ifndef AVCODEC_BLOCKDSP_H

22#include <stddef.h>

25#include "avcodec.h"
26#include "version.h"

28/* add and put pixel (decoding)
29 * Block sizes for op_pixels_func are 8x4,8x8 16x8 16x16.
30 * h for op_pixels_func is limited to { width / 2, width },
31 * but never larger than 16 and never smaller than 4. */
32typedef void (*op_fill_func)(uint8_t *block /* align width (8 or 16) */,
33 uint8_t value, ptrdiff_t line_size, int h);
34
35typedef struct BlockDSPContext {
36 void (*clear_block)(int16_t *block /* align 32 */);
37 void (*clear_blocks)(int16_t *blocks /* align 32 */);
38
39 op_fill_func fill_block_tab[2];
40} BlockDSPContext;
41
42void ff_blockdsp_init(BlockDSPContext *c, AVCodecContext *avctx);
43
44void ff_blockdsp_init_alpha(BlockDSPContext *c);
45void ff_blockdsp_init_arm(BlockDSPContext *c);
46void ff_blockdsp_init_ppc(BlockDSPContext *c);
47void ff_blockdsp_init_x86(BlockDSPContext *c, AVCodecContext *avctx);
48void ff_blockdsp_init_mips(BlockDSPContext *c);
49
50#endif /* AVCODEC_BLOCKDSP_H */
