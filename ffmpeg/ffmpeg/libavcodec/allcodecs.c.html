852{
853 for (int i = 0; codec_list[i]; i++) {

859const AVCodec *av_codec_iterate(void **opaque)
860{
861 uintptr_t i = (uintptr_t)*opaque;
862 const AVCodec *c = codec_list[i];
863
864 ff_thread_once(&av_codec_static_init, av_codec_init_static);
865
866 if (c)
867 *opaque = (void*)(i + 1);
868
869 return c;
870}

920static AVCodec *find_codec(enum AVCodecID id, int (*x)(const AVCodec *))
921{
922 const AVCodec *p, *experimental = NULL;
923 void *i = 0;
924
925 id = remap_deprecated_codec_id(id);
926
927 while ((p = av_codec_iterate(&i))) {
928 if (!x(p))
929 continue;
930 if (p->id == id) {
931 if (p->capabilities & AV_CODEC_CAP_EXPERIMENTAL && !experimental) {
932 experimental = p;
933 } else
934 return (AVCodec*)p;
935 }
936 }
937
938 return (AVCodec*)experimental;
939}

951static AVCodec *find_codec_by_name(const char *name, int (*x)(const AVCodec *))
952{
953 void *i = 0;
954 const AVCodec *p;
955
956 if (!name)
957 return NULL;
958
959 while ((p = av_codec_iterate(&i))) {
960 if (!x(p))
961 continue;
962 if (strcmp(name, p->name) == 0)
963 return (AVCodec*)p;
964 }
965
966 return NULL;
967}
