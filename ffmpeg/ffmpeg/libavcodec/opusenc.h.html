22#ifndef AVCODEC_OPUSENC_H
23#define AVCODEC_OPUSENC_H
24
25#include "internal.h"
26#include "opus_celt.h"
27
28/* Determines the maximum delay the psychoacoustic system will use for lookahead */
29#define FF_BUFQUEUE_SIZE 145
30#include "libavfilter/bufferqueue.h"
31
32#define OPUS_MAX_LOOKAHEAD ((FF_BUFQUEUE_SIZE - 1)*2.5f)
33
34#define OPUS_MAX_CHANNELS 2
35
36/* 120 ms / 2.5 ms = 48 frames (extremely improbable, but the encoder'll work) */
37#define OPUS_MAX_FRAMES_PER_PACKET 48
38
39#define OPUS_BLOCK_SIZE(x) (2 * 15 * (1 << ((x) + 2)))
40
41#define OPUS_SAMPLES_TO_BLOCK_SIZE(x) (ff_log2((x) / (2 * 15)) - 2)
42
43typedef struct OpusEncOptions {
44 float max_delay_ms;
45 int apply_phase_inv;
46} OpusEncOptions;
47
48typedef struct OpusPacketInfo {
49 enum OpusMode mode;
50 enum OpusBandwidth bandwidth;
51 int framesize;
52 int frames;
53} OpusPacketInfo;
54
55#endif /* AVCODEC_OPUSENC_H */
