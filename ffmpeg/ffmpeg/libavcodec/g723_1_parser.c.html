24#include "parser.h"
25#include "g723_1.h"
26
27typedef struct G723_1ParseContext {
28 ParseContext pc;
29} G723_1ParseContext;
30
31static int g723_1_parse(AVCodecParserContext *s1, AVCodecContext *avctx,
32 const uint8_t **poutbuf, int *poutbuf_size,
33 const uint8_t *buf, int buf_size)
34{
35 G723_1ParseContext *s = s1->priv_data;
36 ParseContext *pc = &s->pc;
37 int next = END_NOT_FOUND;
38
39 if (buf_size > 0)
40 next = frame_size[buf[0] & 3] * FFMAX(1, avctx->channels);
41
42 if (ff_combine_frame(pc, next, &buf, &buf_size) < 0 || !buf_size) {
43 *poutbuf = NULL;
44 *poutbuf_size = 0;
45 return buf_size;
46 }
47
48 s1->duration = 240;
49
50 *poutbuf = buf;
51 *poutbuf_size = buf_size;
52 return next;
53}
54
55AVCodecParser ff_g723_1_parser = {
56 .codec_ids = { AV_CODEC_ID_G723_1 },
57 .priv_data_size = sizeof(G723_1ParseContext),
58 .parser_parse = g723_1_parse,
59 .parser_close = ff_parse_close,
60};
