50// H.264 table A-2 plus values from A-1.
51static const struct {
52 int profile_idc;
53 int cpb_br_vcl_factor;
54 int cpb_br_nal_factor;
55} h264_br_factors[] = {

66// We are only ever interested in the NAL bitrate factor.
67static int h264_get_br_factor(int profile_idc)
68{
69 int i;
70 for (i = 0; i < FF_ARRAY_ELEMS(h264_br_factors); i++) {
71 if (h264_br_factors[i].profile_idc == profile_idc)
72 return h264_br_factors[i].cpb_br_nal_factor;
73 }
74 // Default to the non-high profile value if not specified.
75 return 1200;
76}

78const H264LevelDescriptor *ff_h264_guess_level(int profile_idc,
79 int64_t bitrate,
80 int framerate,
81 int width, int height,
82 int max_dec_frame_buffering)
83{
84 int width_mbs = (width + 15) / 16;
85 int height_mbs = (height + 15) / 16;
86 int no_cs3f = !(profile_idc == 66 ||
87 profile_idc == 77 ||
88 profile_idc == 88);
89 int i;
90
91 for (i = 0; i < FF_ARRAY_ELEMS(h264_levels); i++) {
92 const H264LevelDescriptor *level = &h264_levels[i];
93
94 if (level->constraint_set3_flag && no_cs3f)
95 continue;
96
97 if (bitrate > (int64_t)level->max_br * h264_get_br_factor(profile_idc))
98 continue;
99
100 if (width_mbs * height_mbs > level->max_fs)
101 continue;
102 if (width_mbs * width_mbs > 8 * level->max_fs)
103 continue;
104 if (height_mbs * height_mbs > 8 * level->max_fs)
105 continue;
106
107 if (width_mbs && height_mbs) {
108 int max_dpb_frames =
109 FFMIN(level->max_dpb_mbs / (width_mbs * height_mbs), 16);
110 if (max_dec_frame_buffering > max_dpb_frames)
111 continue;
112
113 if (framerate > (level->max_mbps / (width_mbs * height_mbs)))
114 continue;
115 }
116
117 return level;
118 }
119
120 // No usable levels found - frame is too big or bitrate is too high.
121 return NULL;
122}
