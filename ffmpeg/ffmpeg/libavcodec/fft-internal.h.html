43#define CMUL(dre, dim, are, aim, bre, bim) do { \
44 int64_t accu; \
45 (accu) = (int64_t)(bre) * (are); \
46 (accu) -= (int64_t)(bim) * (aim); \
47 (dre) = (int)(((accu) + 0x40000000) >> 31); \
48 (accu) = (int64_t)(bre) * (aim); \
49 (accu) += (int64_t)(bim) * (are); \
50 (dim) = (int)(((accu) + 0x40000000) >> 31); \
51 } while (0)
