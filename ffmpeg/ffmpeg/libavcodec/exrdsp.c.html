21#include <stdint.h>

23#include "libavutil/attributes.h"
24#include "exrdsp.h"

27static void reorder_pixels_scalar(uint8_t *dst, const uint8_t *src, ptrdiff_t size)
28{
29 const uint8_t *t1 = src;
30 int half_size = size / 2;
31 const uint8_t *t2 = src + half_size;
32 uint8_t *s = dst;
33 int i;
34
35 for (i = 0; i < half_size; i++) {
36 *(s++) = *(t1++);
37 *(s++) = *(t2++);
38 }
39}
40
41static void predictor_scalar(uint8_t *src, ptrdiff_t size)
42{
43 ptrdiff_t i;
44
45 for (i = 1; i < size; i++)
46 src[i] += src[i-1] - 128;
47}
48
49av_cold void ff_exrdsp_init(ExrDSPContext *c)
50{
51 c->reorder_pixels = reorder_pixels_scalar;
52 c->predictor = predictor_scalar;
53
54 if (ARCH_X86)
55 ff_exrdsp_init_x86(c);
56}
