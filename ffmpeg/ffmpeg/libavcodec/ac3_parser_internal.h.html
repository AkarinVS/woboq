27/**
28 * Parse AC-3 frame header.
29 * Parse the header up to the lfeon element, which is the first 52 or 54 bits
30 * depending on the audio coding mode.
31 * @param[in] gbc BitContext containing the first 54 bits of the frame.
32 * @param[out] hdr Pointer to struct where header info is written.
33 * @return Returns 0 on success, -1 if there is a sync word mismatch,
34 * -2 if the bsid (version) element is invalid, -3 if the fscod (sample rate)
35 * element is invalid, or -4 if the frmsizecod (bit rate) element is invalid.
36 */
37int ff_ac3_parse_header(GetBitContext *gbc, AC3HeaderInfo *hdr);
38
39int avpriv_ac3_parse_header(AC3HeaderInfo **hdr, const uint8_t *buf,
40 size_t size);
