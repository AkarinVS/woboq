36enum AVFieldOrder {
37 AV_FIELD_UNKNOWN,
38 AV_FIELD_PROGRESSIVE,
39 AV_FIELD_TT, //< Top coded_first, top displayed first
40 AV_FIELD_BB, //< Bottom coded first, bottom displayed first
41 AV_FIELD_TB, //< Top coded first, bottom displayed first
42 AV_FIELD_BT, //< Bottom coded first, top displayed first
43};
44
45/**
46 * This struct describes the properties of an encoded stream.
47 *
48 * sizeof(AVCodecParameters) is not a part of the public ABI, this struct must
49 * be allocated with avcodec_parameters_alloc() and freed with
50 * avcodec_parameters_free().
51 */
52typedef struct AVCodecParameters {
53 /**
54 * General type of the encoded data.
55 */
56 enum AVMediaType codec_type;
57 /**
58 * Specific type of the encoded data (the codec used).
59 */
60 enum AVCodecID codec_id;
61 /**
62 * Additional information about the codec (corresponds to the AVI FOURCC).
63 */
64 uint32_t codec_tag;
65
66 /**
67 * Extra binary data needed for initializing the decoder, codec-dependent.
68 *
69 * Must be allocated with av_malloc() and will be freed by
70 * avcodec_parameters_free(). The allocated size of extradata must be at
71 * least extradata_size + AV_INPUT_BUFFER_PADDING_SIZE, with the padding
72 * bytes zeroed.
73 */
74 uint8_t *extradata;
75 /**
76 * Size of the extradata content in bytes.
77 */
78 int extradata_size;
79
80 /**
81 * - video: the pixel format, the value corresponds to enum AVPixelFormat.
82 * - audio: the sample format, the value corresponds to enum AVSampleFormat.
83 */
84 int format;
85
86 /**
87 * The average bitrate of the encoded data (in bits per second).
88 */
89 int64_t bit_rate;
90
91 /**
92 * The number of bits per sample in the codedwords.
93 *
94 * This is basically the bitrate per sample. It is mandatory for a bunch of
95 * formats to actually decode them. It's the number of bits for one sample in
96 * the actual coded bitstream.
97 *
98 * This could be for example 4 for ADPCM
99 * For PCM formats this matches bits_per_raw_sample
100 * Can be 0
101 */
102 int bits_per_coded_sample;
103
104 /**
105 * This is the number of valid bits in each output sample. If the
106 * sample format has more bits, the least significant bits are additional
107 * padding bits, which are always 0. Use right shifts to reduce the sample
108 * to its actual size. For example, audio formats with 24 bit samples will
109 * have bits_per_raw_sample set to 24, and format set to AV_SAMPLE_FMT_S32.
110 * To get the original sample use "(int32_t)sample >> 8"."
111 *
112 * For ADPCM this might be 12 or 16 or similar
113 * Can be 0
114 */
115 int bits_per_raw_sample;
116
117 /**
118 * Codec-specific bitstream restrictions that the stream conforms to.
119 */
120 int profile;
121 int level;
122
123 /**
124 * Video only. The dimensions of the video frame in pixels.
125 */
126 int width;
127 int height;
128
129 /**
130 * Video only. The aspect ratio (width / height) which a single pixel
131 * should have when displayed.
132 *
133 * When the aspect ratio is unknown / undefined, the numerator should be
134 * set to 0 (the denominator may have any value).
135 */
136 AVRational sample_aspect_ratio;
137
138 /**
139 * Video only. The order of the fields in interlaced video.
140 */
141 enum AVFieldOrder field_order;
142
143 /**
144 * Video only. Progressive frame flag.
145 */
146 int progressive_frame;
147
148 /**
149 * Video only. Additional colorspace characteristics.
150 */
151 enum AVColorRange color_range;
152 enum AVColorPrimaries color_primaries;
153 enum AVColorTransferCharacteristic color_trc;
154 enum AVColorSpace color_space;
155 enum AVChromaLocation chroma_location;
156
157 /**
158 * Video only. Number of delayed frames.
159 */
160 int video_delay;
161
162 /**
163 * Audio only. The channel layout bitmask. May be 0 if the channel layout is
164 * unknown or unspecified, otherwise the number of bits set must be equal to
165 * the channels field.
166 */
167 uint64_t channel_layout;
168 /**
169 * Audio only. The number of audio channels.
170 */
171 int channels;
172 /**
173 * Audio only. The number of audio samples per second.
174 */
175 int sample_rate;
176 /**
177 * Audio only. The number of bytes per coded audio frame, required by some
178 * formats.
179 *
180 * Corresponds to nBlockAlign in WAVEFORMATEX.
181 */
182 int block_align;
183 /**
184 * Audio only. Audio frame size, if known. Required by some formats to be static.
185 */
186 int frame_size;
187
188 /**
189 * Audio only. The amount of padding (in samples) inserted by the encoder at
190 * the beginning of the audio. I.e. this number of leading decoded samples
191 * must be discarded by the caller to get the original audio without leading
192 * padding.
193 */
194 int initial_padding;
195 /**
196 * Audio only. The amount of padding (in samples) appended by the encoder to
197 * the end of the audio. I.e. this number of decoded samples must be
198 * discarded by the caller from the end of the stream to get the original
199 * audio without any trailing padding.
200 */
201 int trailing_padding;
202 /**
203 * Audio only. Number of samples to skip after a discontinuity.
204 */
205 int seek_preroll;
206} AVCodecParameters;
207
208/**
209 * Allocate a new AVCodecParameters and set its fields to default values
210 * (unknown/invalid/0). The returned struct must be freed with
211 * avcodec_parameters_free().
212 */
213AVCodecParameters *avcodec_parameters_alloc(void);
214
215/**
216 * Free an AVCodecParameters instance and everything associated with it and
217 * write NULL to the supplied pointer.
218 */
219void avcodec_parameters_free(AVCodecParameters **par);
220
221/**
222 * Copy the contents of src to dst. Any allocated fields in dst are freed and
223 * replaced with newly allocated duplicates of the corresponding fields in src.
224 *
225 * @return >= 0 on success, a negative AVERROR code on failure.
226 */
227int avcodec_parameters_copy(AVCodecParameters *dst, const AVCodecParameters *src);
228
229
230/**
231 * @}
232 */

234#endif // AVCODEC_CODEC_PAR_H
