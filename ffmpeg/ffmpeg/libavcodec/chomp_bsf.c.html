22#include "bsf.h"
23#include "bsf_internal.h"
24
25static int chomp_filter(AVBSFContext *ctx, AVPacket *pkt)
26{
27 int ret;
28
29 ret = ff_bsf_get_packet_ref(ctx, pkt);
30 if (ret < 0)
31 return ret;
32
33 while (pkt->size > 0 && !pkt->data[pkt->size - 1])
34 pkt->size--;
35
36 return 0;
37}
38
39/**
40 * This filter removes a string of NULL bytes from the end of a packet.
41 */
42const AVBitStreamFilter ff_chomp_bsf = {
43 .name = "chomp",
44 .filter = chomp_filter,
45};
