21static void postfilter_c(float *data, int period, float *gains, int len)
22{
23 const float g0 = gains[0];
24 const float g1 = gains[1];
25 const float g2 = gains[2];
26
27 float x4 = data[-period - 2];
28 float x3 = data[-period - 1];
29 float x2 = data[-period + 0];
30 float x1 = data[-period + 1];
31
32 for (int i = 0; i < len; i++) {
33 float x0 = data[i - period + 2];
34 data[i] += g0 * x2 +
35 g1 * (x1 + x3) +
36 g2 * (x0 + x4);
37 x4 = x3;
38 x3 = x2;
39 x2 = x1;
40 x1 = x0;
41 }
42}
43
44static float deemphasis_c(float *y, float *x, float coeff, int len)
45{
46 for (int i = 0; i < len; i++)
47 coeff = y[i] = x[i] + coeff*CELT_EMPH_COEFF;
48
49 return coeff;
50}
51
52av_cold void ff_opus_dsp_init(OpusDSP *ctx)
53{
54 ctx->postfilter = postfilter_c;
55 ctx->deemphasis = deemphasis_c;
56
57 if (ARCH_X86)
58 ff_opus_dsp_init_x86(ctx);
59
60 if (ARCH_AARCH64)
61 ff_opus_dsp_init_aarch64(ctx);
62}
