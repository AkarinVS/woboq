40static inline int decode_dc(GetBitContext *gb, int component)
41{
42 int code, diff;
43
44 if (component == 0) {
45 code = get_vlc2(gb, ff_dc_lum_vlc.table, DC_VLC_BITS, 2);
46 } else {
47 code = get_vlc2(gb, ff_dc_chroma_vlc.table, DC_VLC_BITS, 2);
48 }
49 if (code == 0) {
50 diff = 0;
51 } else {
52 diff = get_xbits(gb, code);
53 }
54 return diff;
55}
