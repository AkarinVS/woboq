42/** Used to minimize the amount of memory used in order to
43 * optimize cache performance. **/
44typedef struct slice_buffer_s {
45 IDWTELEM **line; ///< For use by idwt and predict_slices.

48 int line_count;
49 int line_width;
50 int data_count;

52} slice_buffer;
