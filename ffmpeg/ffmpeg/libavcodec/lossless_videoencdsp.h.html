19#ifndef AVCODEC_LOSSLESS_VIDEOENCDSP_H
20#define AVCODEC_LOSSLESS_VIDEOENCDSP_H

22#include <stdint.h>
23
24#include "avcodec.h"
25
26typedef struct LLVidEncDSPContext {
27 void (*diff_bytes)(uint8_t *dst /* align 16 */,
28 const uint8_t *src1 /* align 16 */,
29 const uint8_t *src2 /* align 1 */,
30 intptr_t w);
31 /**
32 * Subtract HuffYUV's variant of median prediction.
33 * Note, this might read from src1[-1], src2[-1].
34 */
35 void (*sub_median_pred)(uint8_t *dst, const uint8_t *src1,
36 const uint8_t *src2, intptr_t w,
37 int *left, int *left_top);
38
39 void (*sub_left_predict)(uint8_t *dst, uint8_t *src,
40 ptrdiff_t stride, ptrdiff_t width, int height);
41} LLVidEncDSPContext;
42
43void ff_llvidencdsp_init(LLVidEncDSPContext *c);
44void ff_llvidencdsp_init_x86(LLVidEncDSPContext *c);
45
46#endif /* AVCODEC_LOSSLESS_VIDEOENCDSP_H */
