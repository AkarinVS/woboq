31typedef struct LLVidDSPContext {
32 void (*add_bytes)(uint8_t *dst /* align 32 */, uint8_t *src /* align 32 */,
33 ptrdiff_t w);
34 void (*add_median_pred)(uint8_t *dst, const uint8_t *top,
35 const uint8_t *diff, ptrdiff_t w,
36 int *left, int *left_top);
37 int (*add_left_pred)(uint8_t *dst, const uint8_t *src,
38 ptrdiff_t w, int left);
39
40 int (*add_left_pred_int16)(uint16_t *dst, const uint16_t *src,
41 unsigned mask, ptrdiff_t w, unsigned left);
42 void (*add_gradient_pred)(uint8_t *src /* align 32 */, const ptrdiff_t stride, const ptrdiff_t width);
43} LLVidDSPContext;

45void ff_llviddsp_init(LLVidDSPContext *llviddsp);
46void ff_llviddsp_init_x86(LLVidDSPContext *llviddsp);
