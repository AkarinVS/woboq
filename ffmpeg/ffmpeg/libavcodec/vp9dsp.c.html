86av_cold void ff_vp9dsp_init(VP9DSPContext *dsp, int bpp, int bitexact)
87{
88 if (bpp == 8) {
89 ff_vp9dsp_init_8(dsp);
90 } else if (bpp == 10) {
91 ff_vp9dsp_init_10(dsp);
92 } else {
93 av_assert0(bpp == 12);
94 ff_vp9dsp_init_12(dsp);
95 }
96
97 if (ARCH_AARCH64) ff_vp9dsp_init_aarch64(dsp, bpp);
98 if (ARCH_ARM) ff_vp9dsp_init_arm(dsp, bpp);
99 if (ARCH_X86) ff_vp9dsp_init_x86(dsp, bpp, bitexact);
100 if (ARCH_MIPS) ff_vp9dsp_init_mips(dsp, bpp);
101}
