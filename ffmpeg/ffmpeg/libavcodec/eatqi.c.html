41typedef struct TqiContext {
42 AVCodecContext *avctx;
43 GetBitContext gb;
44 BlockDSPContext bdsp;
45 BswapDSPContext bsdsp;
46 IDCTDSPContext idsp;
47 ScanTable intra_scantable;
48
49 void *bitstream_buf;
50 unsigned int bitstream_buf_size;
51
52 int mb_x, mb_y;
53 uint16_t intra_matrix[64];
54 int last_dc[3];
55
56 DECLARE_ALIGNED(32, int16_t, block)[6][64];
57} TqiContext;
58
59static av_cold int tqi_decode_init(AVCodecContext *avctx)
60{
61 TqiContext *t = avctx->priv_data;
62
63 ff_blockdsp_init(&t->bdsp, avctx);
64 ff_bswapdsp_init(&t->bsdsp);
65 ff_idctdsp_init(&t->idsp, avctx);
66 ff_init_scantable_permutation(t->idsp.idct_permutation, FF_IDCT_PERM_NONE);
67 ff_init_scantable(t->idsp.idct_permutation, &t->intra_scantable, ff_zigzag_direct);
68
69 avctx->framerate = (AVRational){ 15, 1 };
70 avctx->pix_fmt = AV_PIX_FMT_YUV420P;
71 ff_mpeg12_init_vlcs();
72 return 0;
73}

75static int tqi_decode_mb(TqiContext *t, int16_t (*block)[64])
76{
77 int n;
78
79 t->bdsp.clear_blocks(block[0]);
80 for (n = 0; n < 6; n++) {
81 int ret = ff_mpeg1_decode_block_intra(&t->gb,
82 t->intra_matrix,
83 t->intra_scantable.permutated,
84 t->last_dc, block[n], n, 1);
85 if (ret < 0) {
86 av_log(t->avctx, AV_LOG_ERROR, "ac-tex damaged at %d %d\n",
87 t->mb_x, t->mb_y);
88 return ret;
89 }
90 }
91
92 return 0;
93}

95static inline void tqi_idct_put(AVCodecContext *avctx, AVFrame *frame,
96 int16_t (*block)[64])
97{
98 TqiContext *t = avctx->priv_data;
99 ptrdiff_t linesize = frame->linesize[0];
100 uint8_t *dest_y = frame->data[0] + t->mb_y * 16 * linesize + t->mb_x * 16;
101 uint8_t *dest_cb = frame->data[1] + t->mb_y * 8 * frame->linesize[1] + t->mb_x * 8;
102 uint8_t *dest_cr = frame->data[2] + t->mb_y * 8 * frame->linesize[2] + t->mb_x * 8;
103
104 ff_ea_idct_put_c(dest_y , linesize, block[0]);
105 ff_ea_idct_put_c(dest_y + 8, linesize, block[1]);
106 ff_ea_idct_put_c(dest_y + 8*linesize , linesize, block[2]);
107 ff_ea_idct_put_c(dest_y + 8*linesize + 8, linesize, block[3]);
108
109 if (!(avctx->flags & AV_CODEC_FLAG_GRAY)) {
110 ff_ea_idct_put_c(dest_cb, frame->linesize[1], block[4]);
111 ff_ea_idct_put_c(dest_cr, frame->linesize[2], block[5]);
112 }
113}

115static void tqi_calculate_qtable(TqiContext *t, int quant)
116{
117 const int64_t qscale = (215 - 2*quant)*5;
118 int i;
119
120 t->intra_matrix[0] = (ff_inv_aanscales[0] * ff_mpeg1_default_intra_matrix[0]) >> 11;
121 for(i=1; i<64; i++)
122 t->intra_matrix[i] = (ff_inv_aanscales[i] * ff_mpeg1_default_intra_matrix[i] * qscale + 32) >> 14;
123}

125static int tqi_decode_frame(AVCodecContext *avctx,
126 void *data, int *got_frame,
127 AVPacket *avpkt)
128{
129 const uint8_t *buf = avpkt->data;
130 int buf_size = avpkt->size;
131 const uint8_t *buf_end = buf+buf_size;
132 TqiContext *t = avctx->priv_data;
133 AVFrame *frame = data;
134 int ret, w, h;
135
136 if (buf_size < 12)
137 return AVERROR_INVALIDDATA;
138
139 t->avctx = avctx;
140
141 w = AV_RL16(&buf[0]);
142 h = AV_RL16(&buf[2]);
143 tqi_calculate_qtable(t, buf[4]);
144 buf += 8;
145
146 ret = ff_set_dimensions(avctx, w, h);
147 if (ret < 0)
148 return ret;
149
150 if ((ret = ff_get_buffer(avctx, frame, 0)) < 0)
151 return ret;
152
153 av_fast_padded_malloc(&t->bitstream_buf, &t->bitstream_buf_size,
154 buf_end - buf);
155 if (!t->bitstream_buf)
156 return AVERROR(ENOMEM);
157 t->bsdsp.bswap_buf(t->bitstream_buf, (const uint32_t *) buf,
158 (buf_end - buf) / 4);
159 init_get_bits(&t->gb, t->bitstream_buf, 8 * (buf_end - buf));
160
161 t->last_dc[0] =
162 t->last_dc[1] =
163 t->last_dc[2] = 0;
164 for (t->mb_y = 0; t->mb_y < (h + 15) / 16; t->mb_y++) {
165 for (t->mb_x = 0; t->mb_x < (w + 15) / 16; t->mb_x++) {
166 if (tqi_decode_mb(t, t->block) < 0)
167 goto end;
168 tqi_idct_put(avctx, frame, t->block);
169 }
170 }
171 end:
172
173 *got_frame = 1;
174 return buf_size;
175}
176
177static av_cold int tqi_decode_end(AVCodecContext *avctx)
178{
179 TqiContext *t = avctx->priv_data;
180 av_freep(&t->bitstream_buf);
181 return 0;
182}
