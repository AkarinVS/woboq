67static int h264_find_frame_end(H264ParseContext *p, const uint8_t *buf,
68 int buf_size, void *logctx)
69{
70 int i, j;
71 uint32_t state;
72 ParseContext *pc = &p->pc;
73
74 int next_avc = p->is_avc ? 0 : buf_size;
75// mb_addr= pc->mb_addr - 1;
76 state = pc->state;
77 if (state > 13)
78 state = 7;
79
80 if (p->is_avc && !p->nal_length_size)
81 av_log(logctx, AV_LOG_ERROR, "AVC-parser: nal length size invalid\n");
82
83 for (i = 0; i < buf_size; i++) {
84 if (i >= next_avc) {
85 int nalsize = 0;
86 i = next_avc;
87 for (j = 0; j < p->nal_length_size; j++)
88 nalsize = (nalsize << 8) | buf[i++];
89 if (nalsize <= 0 || nalsize > buf_size - i) {
90 av_log(logctx, AV_LOG_ERROR, "AVC-parser: nal size %d remaining %d\n", nalsize, buf_size - i);
91 return buf_size;
92 }
93 next_avc = i + nalsize;
94 state = 5;
95 }
96
97 if (state == 7) {
98 i += p->h264dsp.startcode_find_candidate(buf + i, next_avc - i);
99 if (i < next_avc)
100 state = 2;
101 } else if (state <= 2) {
102 if (buf[i] == 1)
103 state ^= 5; // 2->7, 1->4, 0->5
104 else if (buf[i])
105 state = 7;
106 else
107 state >>= 1; // 2->1, 1->0, 0->0
108 } else if (state <= 5) {
109 int nalu_type = buf[i] & 0x1F;
110 if (nalu_type == H264_NAL_SEI || nalu_type == H264_NAL_SPS ||
111 nalu_type == H264_NAL_PPS || nalu_type == H264_NAL_AUD) {
112 if (pc->frame_start_found) {
113 i++;
114 goto found;
115 }
116 } else if (nalu_type == H264_NAL_SLICE || nalu_type == H264_NAL_DPA ||
117 nalu_type == H264_NAL_IDR_SLICE) {
118 state += 8;
119 continue;
120 }
121 state = 7;
122 } else {
123 unsigned int mb, last_mb = p->parse_last_mb;
124 GetBitContext gb;
125 p->parse_history[p->parse_history_count++] = buf[i];
126
127 init_get_bits(&gb, p->parse_history, 8*p->parse_history_count);
128 mb= get_ue_golomb_long(&gb);
129 if (get_bits_left(&gb) > 0 || p->parse_history_count > 5) {
130 p->parse_last_mb = mb;
131 if (pc->frame_start_found) {
132 if (mb <= last_mb) {
133 i -= p->parse_history_count - 1;
134 p->parse_history_count = 0;
135 goto found;
136 }
137 } else
138 pc->frame_start_found = 1;
139 p->parse_history_count = 0;
140 state = 7;
141 }
142 }
143 }
144 pc->state = state;
145 if (p->is_avc)
146 return next_avc;
147 return END_NOT_FOUND;
148
149found:
150 pc->state = 7;
151 pc->frame_start_found = 0;
152 if (p->is_avc)
153 return next_avc;
154 return i - (state & 5);
155}

157static int scan_mmco_reset(AVCodecParserContext *s, GetBitContext *gb,
158 void *logctx)
159{
160 H264PredWeightTable pwt;
161 int slice_type_nos = s->pict_type & 3;
162 H264ParseContext *p = s->priv_data;
163 int list_count, ref_count[2];
164
165
166 if (p->ps.pps->redundant_pic_cnt_present)
167 get_ue_golomb(gb); // redundant_pic_count
168
169 if (slice_type_nos == AV_PICTURE_TYPE_B)
170 get_bits1(gb); // direct_spatial_mv_pred
171
172 if (ff_h264_parse_ref_count(&list_count, ref_count, gb, p->ps.pps,
173 slice_type_nos, p->picture_structure, logctx) < 0)
174 return AVERROR_INVALIDDATA;
175
176 if (slice_type_nos != AV_PICTURE_TYPE_I) {
177 int list;
178 for (list = 0; list < list_count; list++) {
179 if (get_bits1(gb)) {
180 int index;
181 for (index = 0; ; index++) {
182 unsigned int reordering_of_pic_nums_idc = get_ue_golomb_31(gb);
183
184 if (reordering_of_pic_nums_idc < 3)
185 get_ue_golomb_long(gb);
186 else if (reordering_of_pic_nums_idc > 3) {
187 av_log(logctx, AV_LOG_ERROR,
188 "illegal reordering_of_pic_nums_idc %d\n",
189 reordering_of_pic_nums_idc);
190 return AVERROR_INVALIDDATA;
191 } else
192 break;
193
194 if (index >= ref_count[list]) {
195 av_log(logctx, AV_LOG_ERROR,
196 "reference count %d overflow\n", index);
197 return AVERROR_INVALIDDATA;
198 }
199 }
200 }
201 }
202 }
203
204 if ((p->ps.pps->weighted_pred && slice_type_nos == AV_PICTURE_TYPE_P) ||
205 (p->ps.pps->weighted_bipred_idc == 1 && slice_type_nos == AV_PICTURE_TYPE_B))
206 ff_h264_pred_weight_table(gb, p->ps.sps, ref_count, slice_type_nos,
207 &pwt, p->picture_structure, logctx);
208
209 if (get_bits1(gb)) { // adaptive_ref_pic_marking_mode_flag
210 int i;
211 for (i = 0; i < MAX_MMCO_COUNT; i++) {
212 MMCOOpcode opcode = get_ue_golomb_31(gb);
213 if (opcode > (unsigned) MMCO_LONG) {
214 av_log(logctx, AV_LOG_ERROR,
215 "illegal memory management control operation %d\n",
216 opcode);
217 return AVERROR_INVALIDDATA;
218 }
219 if (opcode == MMCO_END)
220 return 0;
221 else if (opcode == MMCO_RESET)
222 return 1;
223
224 if (opcode == MMCO_SHORT2UNUSED || opcode == MMCO_SHORT2LONG)
225 get_ue_golomb_long(gb); // difference_of_pic_nums_minus1
226 if (opcode == MMCO_SHORT2LONG || opcode == MMCO_LONG2UNUSED ||
227 opcode == MMCO_LONG || opcode == MMCO_SET_MAX_LONG)
228 get_ue_golomb_31(gb);
229 }
230 }
231
232 return 0;
233}

235/**
236 * Parse NAL units of found picture and decode some basic information.
237 *
238 * @param s parser context.
239 * @param avctx codec context.
240 * @param buf buffer with field/frame data.
241 * @param buf_size size of the buffer.
242 */
243static inline int parse_nal_units(AVCodecParserContext *s,
244 AVCodecContext *avctx,
245 const uint8_t * const buf, int buf_size)
246{
247 H264ParseContext *p = s->priv_data;
248 H2645RBSP rbsp = { NULL };
249 H2645NAL nal = { NULL };
250 int buf_index, next_avc;
251 unsigned int pps_id;
252 unsigned int slice_type;
253 int state = -1, got_reset = 0;
254 int q264 = buf_size >=4 && !memcmp("Q264", buf, 4);
255 int field_poc[2];
256 int ret;
257
258 /* set some sane default values */
259 s->pict_type = AV_PICTURE_TYPE_I;
260 s->key_frame = 0;
261 s->picture_structure = AV_PICTURE_STRUCTURE_UNKNOWN;
262
263 ff_h264_sei_uninit(&p->sei);
264 p->sei.frame_packing.arrangement_cancel_flag = -1;
265
266 if (!buf_size)
267 return 0;
268
269 av_fast_padded_malloc(&rbsp.rbsp_buffer, &rbsp.rbsp_buffer_alloc_size, buf_size);
270 if (!rbsp.rbsp_buffer)
271 return AVERROR(ENOMEM);
272
273 buf_index = 0;
274 next_avc = p->is_avc ? 0 : buf_size;
275 for (;;) {
276 const SPS *sps;
277 int src_length, consumed, nalsize = 0;
278
279 if (buf_index >= next_avc) {
280 nalsize = get_nalsize(p->nal_length_size, buf, buf_size, &buf_index, avctx);
281 if (nalsize < 0)
282 break;
283 next_avc = buf_index + nalsize;
284 } else {
285 buf_index = find_start_code(buf, buf_size, buf_index, next_avc);
286 if (buf_index >= buf_size)
287 break;
288 if (buf_index >= next_avc)
289 continue;
290 }
291 src_length = next_avc - buf_index;
292
293 state = buf[buf_index];
294 switch (state & 0x1f) {
295 case H264_NAL_SLICE:
296 case H264_NAL_IDR_SLICE:
297 // Do not walk the whole buffer just to decode slice header
298 if ((state & 0x1f) == H264_NAL_IDR_SLICE || ((state >> 5) & 0x3) == 0) {
299 /* IDR or disposable slice
300 * No need to decode many bytes because MMCOs shall not be present. */
301 if (src_length > 60)
302 src_length = 60;
303 } else {
304 /* To decode up to MMCOs */
305 if (src_length > 1000)
306 src_length = 1000;
307 }
308 break;
309 }
310 consumed = ff_h2645_extract_rbsp(buf + buf_index, src_length, &rbsp, &nal, 1);
311 if (consumed < 0)
312 break;
313
314 buf_index += consumed;
315
316 ret = init_get_bits8(&nal.gb, nal.data, nal.size);
317 if (ret < 0)
318 goto fail;
319 get_bits1(&nal.gb);
320 nal.ref_idc = get_bits(&nal.gb, 2);
321 nal.type = get_bits(&nal.gb, 5);
322
323 switch (nal.type) {
324 case H264_NAL_SPS:
325 ff_h264_decode_seq_parameter_set(&nal.gb, avctx, &p->ps, 0);
326 break;
327 case H264_NAL_PPS:
328 ff_h264_decode_picture_parameter_set(&nal.gb, avctx, &p->ps,
329 nal.size_bits);
330 break;
331 case H264_NAL_SEI:
332 ff_h264_sei_decode(&p->sei, &nal.gb, &p->ps, avctx);
333 break;
334 case H264_NAL_IDR_SLICE:
335 s->key_frame = 1;
336
337 p->poc.prev_frame_num = 0;
338 p->poc.prev_frame_num_offset = 0;
339 p->poc.prev_poc_msb =
340 p->poc.prev_poc_lsb = 0;
341 /* fall through */
342 case H264_NAL_SLICE:
343 get_ue_golomb_long(&nal.gb); // skip first_mb_in_slice
344 slice_type = get_ue_golomb_31(&nal.gb);
345 s->pict_type = ff_h264_golomb_to_pict_type[slice_type % 5];
346 if (p->sei.recovery_point.recovery_frame_cnt >= 0) {
347 /* key frame, since recovery_frame_cnt is set */
348 s->key_frame = 1;
349 }
350 pps_id = get_ue_golomb(&nal.gb);
351 if (pps_id >= MAX_PPS_COUNT) {
352 av_log(avctx, AV_LOG_ERROR,
353 "pps_id %u out of range\n", pps_id);
354 goto fail;
355 }
356 if (!p->ps.pps_list[pps_id]) {
357 av_log(avctx, AV_LOG_ERROR,
358 "non-existing PPS %u referenced\n", pps_id);
359 goto fail;
360 }
361
362 av_buffer_unref(&p->ps.pps_ref);
363 p->ps.pps = NULL;
364 p->ps.sps = NULL;
365 p->ps.pps_ref = av_buffer_ref(p->ps.pps_list[pps_id]);
366 if (!p->ps.pps_ref)
367 goto fail;
368 p->ps.pps = (const PPS*)p->ps.pps_ref->data;
369 p->ps.sps = p->ps.pps->sps;
370 sps = p->ps.sps;
371
372 // heuristic to detect non marked keyframes
373 if (p->ps.sps->ref_frame_count <= 1 && p->ps.pps->ref_count[0] <= 1 && s->pict_type == AV_PICTURE_TYPE_I)
374 s->key_frame = 1;
375
376 p->poc.frame_num = get_bits(&nal.gb, sps->log2_max_frame_num);
377
378 s->coded_width = 16 * sps->mb_width;
379 s->coded_height = 16 * sps->mb_height;
380 s->width = s->coded_width - (sps->crop_right + sps->crop_left);
381 s->height = s->coded_height - (sps->crop_top + sps->crop_bottom);
382 if (s->width <= 0 || s->height <= 0) {
383 s->width = s->coded_width;
384 s->height = s->coded_height;
385 }
386
387 switch (sps->bit_depth_luma) {
388 case 9:
389 if (sps->chroma_format_idc == 3) s->format = AV_PIX_FMT_YUV444P9;
390 else if (sps->chroma_format_idc == 2) s->format = AV_PIX_FMT_YUV422P9;
391 else s->format = AV_PIX_FMT_YUV420P9;
392 break;
393 case 10:
394 if (sps->chroma_format_idc == 3) s->format = AV_PIX_FMT_YUV444P10;
395 else if (sps->chroma_format_idc == 2) s->format = AV_PIX_FMT_YUV422P10;
396 else s->format = AV_PIX_FMT_YUV420P10;
397 break;
398 case 8:
399 if (sps->chroma_format_idc == 3) s->format = AV_PIX_FMT_YUV444P;
400 else if (sps->chroma_format_idc == 2) s->format = AV_PIX_FMT_YUV422P;
401 else s->format = AV_PIX_FMT_YUV420P;
402 break;
403 default:
404 s->format = AV_PIX_FMT_NONE;
405 }
406
407 avctx->profile = ff_h264_get_profile(sps);
408 avctx->level = sps->level_idc;
409
410 if (sps->frame_mbs_only_flag) {
411 p->picture_structure = PICT_FRAME;
412 } else {
413 if (get_bits1(&nal.gb)) { // field_pic_flag
414 p->picture_structure = PICT_TOP_FIELD + get_bits1(&nal.gb); // bottom_field_flag
415 } else {
416 p->picture_structure = PICT_FRAME;
417 }
418 }
419
420 if (nal.type == H264_NAL_IDR_SLICE)
421 get_ue_golomb_long(&nal.gb); /* idr_pic_id */
422 if (sps->poc_type == 0) {
423 p->poc.poc_lsb = get_bits(&nal.gb, sps->log2_max_poc_lsb);
424
425 if (p->ps.pps->pic_order_present == 1 &&
426 p->picture_structure == PICT_FRAME)
427 p->poc.delta_poc_bottom = get_se_golomb(&nal.gb);
428 }
429
430 if (sps->poc_type == 1 &&
431 !sps->delta_pic_order_always_zero_flag) {
432 p->poc.delta_poc[0] = get_se_golomb(&nal.gb);
433
434 if (p->ps.pps->pic_order_present == 1 &&
435 p->picture_structure == PICT_FRAME)
436 p->poc.delta_poc[1] = get_se_golomb(&nal.gb);
437 }
438
439 /* Decode POC of this picture.
440 * The prev_ values needed for decoding POC of the next picture are not set here. */
441 field_poc[0] = field_poc[1] = INT_MAX;
442 ret = ff_h264_init_poc(field_poc, &s->output_picture_number, sps,
443 &p->poc, p->picture_structure, nal.ref_idc);
444 if (ret < 0)
445 goto fail;
446
447 /* Continue parsing to check if MMCO_RESET is present.
448 * FIXME: MMCO_RESET could appear in non-first slice.
449 * Maybe, we should parse all undisposable non-IDR slice of this
450 * picture until encountering MMCO_RESET in a slice of it. */
451 if (nal.ref_idc && nal.type != H264_NAL_IDR_SLICE) {
452 got_reset = scan_mmco_reset(s, &nal.gb, avctx);
453 if (got_reset < 0)
454 goto fail;
455 }
456
457 /* Set up the prev_ values for decoding POC of the next picture. */
458 p->poc.prev_frame_num = got_reset ? 0 : p->poc.frame_num;
459 p->poc.prev_frame_num_offset = got_reset ? 0 : p->poc.frame_num_offset;
460 if (nal.ref_idc != 0) {
461 if (!got_reset) {
462 p->poc.prev_poc_msb = p->poc.poc_msb;
463 p->poc.prev_poc_lsb = p->poc.poc_lsb;
464 } else {
465 p->poc.prev_poc_msb = 0;
466 p->poc.prev_poc_lsb =
467 p->picture_structure == PICT_BOTTOM_FIELD ? 0 : field_poc[0];
468 }
469 }
470
471 if (p->sei.picture_timing.present) {
472 ret = ff_h264_sei_process_picture_timing(&p->sei.picture_timing,
473 sps, avctx);
474 if (ret < 0) {
475 av_log(avctx, AV_LOG_ERROR, "Error processing the picture timing SEI\n");
476 p->sei.picture_timing.present = 0;
477 }
478 }
479
480 if (sps->pic_struct_present_flag && p->sei.picture_timing.present) {
481 switch (p->sei.picture_timing.pic_struct) {
482 case H264_SEI_PIC_STRUCT_TOP_FIELD:
483 case H264_SEI_PIC_STRUCT_BOTTOM_FIELD:
484 s->repeat_pict = 0;
485 break;
486 case H264_SEI_PIC_STRUCT_FRAME:
487 case H264_SEI_PIC_STRUCT_TOP_BOTTOM:
488 case H264_SEI_PIC_STRUCT_BOTTOM_TOP:
489 s->repeat_pict = 1;
490 break;
491 case H264_SEI_PIC_STRUCT_TOP_BOTTOM_TOP:
492 case H264_SEI_PIC_STRUCT_BOTTOM_TOP_BOTTOM:
493 s->repeat_pict = 2;
494 break;
495 case H264_SEI_PIC_STRUCT_FRAME_DOUBLING:
496 s->repeat_pict = 3;
497 break;
498 case H264_SEI_PIC_STRUCT_FRAME_TRIPLING:
499 s->repeat_pict = 5;
500 break;
501 default:
502 s->repeat_pict = p->picture_structure == PICT_FRAME ? 1 : 0;
503 break;
504 }
505 } else {
506 s->repeat_pict = p->picture_structure == PICT_FRAME ? 1 : 0;
507 }
508
509 if (p->picture_structure == PICT_FRAME) {
510 s->picture_structure = AV_PICTURE_STRUCTURE_FRAME;
511 if (sps->pic_struct_present_flag && p->sei.picture_timing.present) {
512 switch (p->sei.picture_timing.pic_struct) {
513 case H264_SEI_PIC_STRUCT_TOP_BOTTOM:
514 case H264_SEI_PIC_STRUCT_TOP_BOTTOM_TOP:
515 s->field_order = AV_FIELD_TT;
516 break;
517 case H264_SEI_PIC_STRUCT_BOTTOM_TOP:
518 case H264_SEI_PIC_STRUCT_BOTTOM_TOP_BOTTOM:
519 s->field_order = AV_FIELD_BB;
520 break;
521 default:
522 s->field_order = AV_FIELD_PROGRESSIVE;
523 break;
524 }
525 } else {
526 if (field_poc[0] < field_poc[1])
527 s->field_order = AV_FIELD_TT;
528 else if (field_poc[0] > field_poc[1])
529 s->field_order = AV_FIELD_BB;
530 else
531 s->field_order = AV_FIELD_PROGRESSIVE;
532 }
533 } else {
534 if (p->picture_structure == PICT_TOP_FIELD)
535 s->picture_structure = AV_PICTURE_STRUCTURE_TOP_FIELD;
536 else
537 s->picture_structure = AV_PICTURE_STRUCTURE_BOTTOM_FIELD;
538 if (p->poc.frame_num == p->last_frame_num &&
539 p->last_picture_structure != AV_PICTURE_STRUCTURE_UNKNOWN &&
540 p->last_picture_structure != AV_PICTURE_STRUCTURE_FRAME &&
541 p->last_picture_structure != s->picture_structure) {
542 if (p->last_picture_structure == AV_PICTURE_STRUCTURE_TOP_FIELD)
543 s->field_order = AV_FIELD_TT;
544 else
545 s->field_order = AV_FIELD_BB;
546 } else {
547 s->field_order = AV_FIELD_UNKNOWN;
548 }
549 p->last_picture_structure = s->picture_structure;
550 p->last_frame_num = p->poc.frame_num;
551 }
552
553 av_freep(&rbsp.rbsp_buffer);
554 return 0; /* no need to evaluate the rest */
555 }
556 }
557 if (q264) {
558 av_freep(&rbsp.rbsp_buffer);
559 return 0;
560 }
561 /* didn't find a picture! */
562 av_log(avctx, AV_LOG_ERROR, "missing picture in access unit with size %d\n", buf_size);
563fail:
564 av_freep(&rbsp.rbsp_buffer);
565 return -1;
566}

568static int h264_parse(AVCodecParserContext *s,
569 AVCodecContext *avctx,
570 const uint8_t **poutbuf, int *poutbuf_size,
571 const uint8_t *buf, int buf_size)
572{
573 H264ParseContext *p = s->priv_data;
574 ParseContext *pc = &p->pc;
575 int next;
576
577 if (!p->got_first) {
578 p->got_first = 1;
579 if (avctx->extradata_size) {
580 ff_h264_decode_extradata(avctx->extradata, avctx->extradata_size,
581 &p->ps, &p->is_avc, &p->nal_length_size,
582 avctx->err_recognition, avctx);
583 }
584 }
585
586 if (s->flags & PARSER_FLAG_COMPLETE_FRAMES) {
587 next = buf_size;
588 } else {
589 next = h264_find_frame_end(p, buf, buf_size, avctx);
590
591 if (ff_combine_frame(pc, next, &buf, &buf_size) < 0) {
592 *poutbuf = NULL;
593 *poutbuf_size = 0;
594 return buf_size;
595 }
596
597 if (next < 0 && next != END_NOT_FOUND) {
598 av_assert1(pc->last_index + next >= 0);
599 h264_find_frame_end(p, &pc->buffer[pc->last_index + next], -next, avctx); // update state
600 }
601 }
602
603 parse_nal_units(s, avctx, buf, buf_size);
604
605 if (avctx->framerate.num)
606 avctx->time_base = av_inv_q(av_mul_q(avctx->framerate, (AVRational){avctx->ticks_per_frame, 1}));
607 if (p->sei.picture_timing.cpb_removal_delay >= 0) {
608 s->dts_sync_point = p->sei.buffering_period.present;
609 s->dts_ref_dts_delta = p->sei.picture_timing.cpb_removal_delay;
610 s->pts_dts_delta = p->sei.picture_timing.dpb_output_delay;
611 } else {
612 s->dts_sync_point = INT_MIN;
613 s->dts_ref_dts_delta = INT_MIN;
614 s->pts_dts_delta = INT_MIN;
615 }
616
617 if (s->flags & PARSER_FLAG_ONCE) {
618 s->flags &= PARSER_FLAG_COMPLETE_FRAMES;
619 }
620
621 *poutbuf = buf;
622 *poutbuf_size = buf_size;
623 return next;
624}

626static int h264_split(AVCodecContext *avctx,
627 const uint8_t *buf, int buf_size)
628{
629 uint32_t state = -1;
630 int has_sps = 0;
631 int has_pps = 0;
632 const uint8_t *ptr = buf, *end = buf + buf_size;
633 int nalu_type;
634
635 while (ptr < end) {
636 ptr = avpriv_find_start_code(ptr, end, &state);
637 if ((state & 0xFFFFFF00) != 0x100)
638 break;
639 nalu_type = state & 0x1F;
640 if (nalu_type == H264_NAL_SPS) {
641 has_sps = 1;
642 } else if (nalu_type == H264_NAL_PPS)
643 has_pps = 1;
644 /* else if (nalu_type == 0x01 ||
645 * nalu_type == 0x02 ||
646 * nalu_type == 0x05) {
647 * }
648 */
649 else if ((nalu_type != H264_NAL_SEI || has_pps) &&
650 nalu_type != H264_NAL_AUD && nalu_type != H264_NAL_SPS_EXT &&
651 nalu_type != 0x0f) {
652 if (has_sps) {
653 while (ptr - 4 > buf && ptr[-5] == 0)
654 ptr--;
655 return ptr - 4 - buf;
656 }
657 }
658 }
659
660 return 0;
661}
