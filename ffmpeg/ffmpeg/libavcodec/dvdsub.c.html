23#include "libavutil/avstring.h"
24#include <stdlib.h>

26void ff_dvdsub_parse_palette(uint32_t *palette, const char *p)
27{
28 for (int i = 0; i < 16; i++) {
29 palette[i] = strtoul(p, (char **)&p, 16);
30 while (*p == ',' || av_isspace(*p))
31 p++;
32 }
33}
