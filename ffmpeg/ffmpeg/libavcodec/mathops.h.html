95/* median of 3 */
96#ifndef mid_pred
97#define mid_pred mid_pred
98static inline av_const int mid_pred(int a, int b, int c)
99{
100 if(a>b){
101 if(c>b){
102 if(c>a) b=a;
103 else b=c;
104 }
105 }else{
106 if(b>c){
107 if(c>a) b=c;
108 else b=a;
109 }
110 }
111 return b;
112}
113#endif

115#ifndef median4
116#define median4 median4
117static inline av_const int median4(int a, int b, int c, int d)
118{
119 if (a < b) {
120 if (c < d) return (FFMIN(b, d) + FFMAX(a, c)) / 2;
121 else return (FFMIN(b, c) + FFMAX(a, d)) / 2;
122 } else {
123 if (c < d) return (FFMIN(a, d) + FFMAX(b, c)) / 2;
124 else return (FFMIN(a, c) + FFMAX(b, d)) / 2;
125 }
126}
127#endif

129#ifndef sign_extend
130static inline av_const int sign_extend(int val, unsigned bits)
131{
132 unsigned shift = 8 * sizeof(int) - bits;
133 union { unsigned u; int s; } v = { (unsigned) val << shift };
134 return v.s >> shift;
135}
136#endif

138#ifndef zero_extend
139static inline av_const unsigned zero_extend(unsigned val, unsigned bits)
140{
141 return (val << ((8 * sizeof(int)) - bits)) >> ((8 * sizeof(int)) - bits);
142}
143#endif

205#ifndef ff_sqrt
206#define ff_sqrt ff_sqrt
207static inline av_const unsigned int ff_sqrt(unsigned int a)
208{
209 unsigned int b;
210
211 if (a < 255) return (ff_sqrt_tab[a + 1] - 1) >> 4;
212 else if (a < (1 << 12)) b = ff_sqrt_tab[a >> 4] >> 2;
213#if !CONFIG_SMALL
214 else if (a < (1 << 14)) b = ff_sqrt_tab[a >> 6] >> 1;
215 else if (a < (1 << 16)) b = ff_sqrt_tab[a >> 8] ;
216#endif
217 else {
218 int s = av_log2_16bit(a >> 16) >> 1;
219 unsigned int c = a >> (s + 2);
220 b = ff_sqrt_tab[c >> (s + 8)];
221 b = FASTDIV(c,b) + (b << s);
222 }
223
224 return b - (a < b * b);
225}
226#endif

233static inline int8_t ff_u8_to_s8(uint8_t a)
234{
235 union {
236 uint8_t u8;
237 int8_t s8;
238 } b;
239 b.u8 = a;
240 return b.s8;

243static av_always_inline uint32_t bitswap_32(uint32_t x)
244{
245 return (uint32_t)ff_reverse[ x & 0xFF] << 24 |
246 (uint32_t)ff_reverse[(x >> 8) & 0xFF] << 16 |
247 (uint32_t)ff_reverse[(x >> 16) & 0xFF] << 8 |
248 (uint32_t)ff_reverse[ x >> 24];
249}
