27#include "parser.h"
28
29typedef struct IPUParseContext {
30 ParseContext pc;
31} IPUParseContext;
32
33static int ipu_parse(AVCodecParserContext *s, AVCodecContext *avctx,
34 const uint8_t **poutbuf, int *poutbuf_size,
35 const uint8_t *buf, int buf_size)
36{
37 IPUParseContext *ipc = s->priv_data;
38 uint32_t state = ipc->pc.state;
39 int next = END_NOT_FOUND, i = 0;
40
41 s->pict_type = AV_PICTURE_TYPE_NONE;
42 s->duration = 1;
43
44 *poutbuf_size = 0;
45 *poutbuf = NULL;
46
47 if (s->flags & PARSER_FLAG_COMPLETE_FRAMES) {
48 next = buf_size;
49 } else {
50 for (; i < buf_size; i++) {
51 state = (state << 8) | buf[i];
52 if (state == 0x1b0) {
53 next = i + 1;
54 break;
55 }
56 }
57
58 ipc->pc.state = state;
59 if (ff_combine_frame(&ipc->pc, next, &buf, &buf_size) < 0) {
60 *poutbuf = NULL;
61 *poutbuf_size = 0;
62 return buf_size;
63 }
64 }
65
66 *poutbuf = buf;
67 *poutbuf_size = buf_size;
68
69 return next;
70}
71
72AVCodecParser ff_ipu_parser = {
73 .codec_ids = { AV_CODEC_ID_IPU },
74 .priv_data_size = sizeof(IPUParseContext),
75 .parser_parse = ipu_parse,
76 .parser_close = ff_parse_close,
77};
