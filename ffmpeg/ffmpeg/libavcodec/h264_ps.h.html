37#define MAX_SPS_COUNT 32

41/**
42 * Sequence parameter set
43 */
44typedef struct SPS {
45 unsigned int sps_id;
46 int profile_idc;
47 int level_idc;
48 int chroma_format_idc;
49 int transform_bypass; ///< qpprime_y_zero_transform_bypass_flag
50 int log2_max_frame_num; ///< log2_max_frame_num_minus4 + 4
51 int poc_type; ///< pic_order_cnt_type
52 int log2_max_poc_lsb; ///< log2_max_pic_order_cnt_lsb_minus4
53 int delta_pic_order_always_zero_flag;
54 int offset_for_non_ref_pic;
55 int offset_for_top_to_bottom_field;
56 int poc_cycle_length; ///< num_ref_frames_in_pic_order_cnt_cycle
57 int ref_frame_count; ///< num_ref_frames
58 int gaps_in_frame_num_allowed_flag;
59 int mb_width; ///< pic_width_in_mbs_minus1 + 1
60 ///< (pic_height_in_map_units_minus1 + 1) * (2 - frame_mbs_only_flag)
61 int mb_height;
62 int frame_mbs_only_flag;
63 int mb_aff; ///< mb_adaptive_frame_field_flag
64 int direct_8x8_inference_flag;
65 int crop; ///< frame_cropping_flag
66
67 /* those 4 are already in luma samples */
68 unsigned int crop_left; ///< frame_cropping_rect_left_offset
69 unsigned int crop_right; ///< frame_cropping_rect_right_offset
70 unsigned int crop_top; ///< frame_cropping_rect_top_offset
71 unsigned int crop_bottom; ///< frame_cropping_rect_bottom_offset
72 int vui_parameters_present_flag;
73 AVRational sar;
74 int video_signal_type_present_flag;
75 int full_range;
76 int colour_description_present_flag;
77 enum AVColorPrimaries color_primaries;
78 enum AVColorTransferCharacteristic color_trc;
79 enum AVColorSpace colorspace;
80 enum AVChromaLocation chroma_location;
81
82 int timing_info_present_flag;
83 uint32_t num_units_in_tick;
84 uint32_t time_scale;
85 int fixed_frame_rate_flag;
86 int32_t offset_for_ref_frame[256];
87 int bitstream_restriction_flag;
88 int num_reorder_frames;
89 int scaling_matrix_present;
90 uint8_t scaling_matrix4[6][16];
91 uint8_t scaling_matrix8[6][64];
92 int nal_hrd_parameters_present_flag;
93 int vcl_hrd_parameters_present_flag;
94 int pic_struct_present_flag;
95 int time_offset_length;
96 int cpb_cnt; ///< See H.264 E.1.2
97 int initial_cpb_removal_delay_length; ///< initial_cpb_removal_delay_length_minus1 + 1
98 int cpb_removal_delay_length; ///< cpb_removal_delay_length_minus1 + 1
99 int dpb_output_delay_length; ///< dpb_output_delay_length_minus1 + 1
100 int bit_depth_luma; ///< bit_depth_luma_minus8 + 8
101 int bit_depth_chroma; ///< bit_depth_chroma_minus8 + 8
102 int residual_color_transform_flag; ///< residual_colour_transform_flag
103 int constraint_set_flags; ///< constraint_set[0-3]_flag
104 uint8_t data[4096];
105 size_t data_size;
106} SPS;
107
108/**
109 * Picture parameter set
110 */
111typedef struct PPS {
112 unsigned int sps_id;
113 int cabac; ///< entropy_coding_mode_flag
114 int pic_order_present; ///< pic_order_present_flag
115 int slice_group_count; ///< num_slice_groups_minus1 + 1
116 int mb_slice_group_map_type;
117 unsigned int ref_count[2]; ///< num_ref_idx_l0/1_active_minus1 + 1
118 int weighted_pred; ///< weighted_pred_flag
119 int weighted_bipred_idc;
120 int init_qp; ///< pic_init_qp_minus26 + 26
121 int init_qs; ///< pic_init_qs_minus26 + 26
122 int chroma_qp_index_offset[2];
123 int deblocking_filter_parameters_present; ///< deblocking_filter_parameters_present_flag
124 int constrained_intra_pred; ///< constrained_intra_pred_flag
125 int redundant_pic_cnt_present; ///< redundant_pic_cnt_present_flag
126 int transform_8x8_mode; ///< transform_8x8_mode_flag
127 uint8_t scaling_matrix4[6][16];
128 uint8_t scaling_matrix8[6][64];
129 uint8_t chroma_qp_table[2][QP_MAX_NUM+1]; ///< pre-scaled (with chroma_qp_index_offset) version of qp_table
130 int chroma_qp_diff;
131 uint8_t data[4096];
132 size_t data_size;
133
134 uint32_t dequant4_buffer[6][QP_MAX_NUM + 1][16];
135 uint32_t dequant8_buffer[6][QP_MAX_NUM + 1][64];
136 uint32_t(*dequant4_coeff[6])[16];
137 uint32_t(*dequant8_coeff[6])[64];
138
139 AVBufferRef *sps_ref;
140 const SPS *sps;
141} PPS;
142
143typedef struct H264ParamSets {
144 AVBufferRef *sps_list[MAX_SPS_COUNT];
145 AVBufferRef *pps_list[MAX_PPS_COUNT];
146
147 AVBufferRef *pps_ref;
148 /* currently active parameters sets */
149 const PPS *pps;
150 const SPS *sps;
151
152 int overread_warning_printed[2];
153} H264ParamSets;
154
155/**
156 * Decode SPS

158int ff_h264_decode_seq_parameter_set(GetBitContext *gb, AVCodecContext *avctx,
159 H264ParamSets *ps, int ignore_truncation);

161/**
162 * Decode PPS

164int ff_h264_decode_picture_parameter_set(GetBitContext *gb, AVCodecContext *avctx,
165 H264ParamSets *ps, int bit_length);

167/**
168 * Uninit H264 param sets structure.
169 */
170void ff_h264_ps_uninit(H264ParamSets *ps);
