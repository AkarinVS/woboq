39static void vc1_extract_headers(AVCodecParserContext *s, AVCodecContext *avctx,
40 const uint8_t *buf, int buf_size)
41{
42 VC1ParseContext *vpc = s->priv_data;
43 GetBitContext gb;
44 const uint8_t *start, *end, *next;
45 uint8_t *buf2 = av_mallocz(buf_size + AV_INPUT_BUFFER_PADDING_SIZE);
46
47 vpc->v.s.avctx = avctx;
48 vpc->v.parse_only = 1;
49 vpc->v.first_pic_header_flag = 1;
50 next = buf;
51 s->repeat_pict = 0;
52
53 for(start = buf, end = buf + buf_size; next < end; start = next){
54 int buf2_size, size;
55 int ret;
56
57 next = find_next_marker(start + 4, end);
58 size = next - start - 4;
59 buf2_size = vc1_unescape_buffer(start + 4, size, buf2);
60 init_get_bits8(&gb, buf2, buf2_size);
61 if(size <= 0) continue;
62 switch(AV_RB32(start)){
63 case VC1_CODE_SEQHDR:
64 ff_vc1_decode_sequence_header(avctx, &vpc->v, &gb);
65 break;
66 case VC1_CODE_ENTRYPOINT:
67 ff_vc1_decode_entry_point(avctx, &vpc->v, &gb);
68 break;
69 case VC1_CODE_FRAME:
70 if(vpc->v.profile < PROFILE_ADVANCED)
71 ret = ff_vc1_parse_frame_header (&vpc->v, &gb);
72 else
73 ret = ff_vc1_parse_frame_header_adv(&vpc->v, &gb);
74
75 if (ret < 0)
76 break;
77
78 /* keep AV_PICTURE_TYPE_BI internal to VC1 */
79 if (vpc->v.s.pict_type == AV_PICTURE_TYPE_BI)
80 s->pict_type = AV_PICTURE_TYPE_B;
81 else
82 s->pict_type = vpc->v.s.pict_type;
83
84 if (avctx->ticks_per_frame > 1){
85 // process pulldown flags
86 s->repeat_pict = 1;
87 // Pulldown flags are only valid when 'broadcast' has been set.
88 // So ticks_per_frame will be 2
89 if (vpc->v.rff){
90 // repeat field
91 s->repeat_pict = 2;
92 }else if (vpc->v.rptfrm){
93 // repeat frames
94 s->repeat_pict = vpc->v.rptfrm * 2 + 1;
95 }
96 }
97
98 if (vpc->v.broadcast && vpc->v.interlace && !vpc->v.psf)
99 s->field_order = vpc->v.tff ? AV_FIELD_TT : AV_FIELD_BB;
100 else
101 s->field_order = AV_FIELD_PROGRESSIVE;
102
103 break;
104 }
105 if (avctx->framerate.num)
106 avctx->time_base = av_inv_q(av_mul_q(avctx->framerate, (AVRational){avctx->ticks_per_frame, 1}));
107 s->format = vpc->v.chromaformat == 1 ? AV_PIX_FMT_YUV420P
108 : AV_PIX_FMT_NONE;
109 if (avctx->width && avctx->height) {
110 s->width = avctx->width;
111 s->height = avctx->height;
112 s->coded_width = FFALIGN(avctx->coded_width, 16);
113 s->coded_height = FFALIGN(avctx->coded_height, 16);
114 }
115 }
116
117 av_free(buf2);
118}

120/**
121 * Find the end of the current frame in the bitstream.
122 * @return the position of the first byte of the next frame, or -1
123 */
124static int vc1_find_frame_end(ParseContext *pc, const uint8_t *buf,
125 int buf_size) {
126 int pic_found, i;
127 uint32_t state;
128
129 pic_found= pc->frame_start_found;
130 state= pc->state;
131
132 i=0;
133 if(!pic_found){
134 for(i=0; i<buf_size; i++){
135 state= (state<<8) | buf[i];
136 if(state == VC1_CODE_FRAME || state == VC1_CODE_FIELD){
137 i++;
138 pic_found=1;
139 break;
140 }
141 }
142 }
143
144 if(pic_found){
145 /* EOF considered as end of frame */
146 if (buf_size == 0)
147 return 0;
148 for(; i<buf_size; i++){
149 state= (state<<8) | buf[i];
150 if(IS_MARKER(state) && state != VC1_CODE_FIELD && state != VC1_CODE_SLICE){
151 pc->frame_start_found=0;
152 pc->state=-1;
153 return i-3;
154 }
155 }
156 }
157 pc->frame_start_found= pic_found;
158 pc->state= state;
159 return END_NOT_FOUND;
160}

162static int vc1_parse(AVCodecParserContext *s,
163 AVCodecContext *avctx,
164 const uint8_t **poutbuf, int *poutbuf_size,
165 const uint8_t *buf, int buf_size)
166{
167 VC1ParseContext *vpc = s->priv_data;
168 int next;
169
170 if(s->flags & PARSER_FLAG_COMPLETE_FRAMES){
171 next= buf_size;
172 }else{
173 next= vc1_find_frame_end(&vpc->pc, buf, buf_size);
174
175 if (ff_combine_frame(&vpc->pc, next, &buf, &buf_size) < 0) {
176 *poutbuf = NULL;
177 *poutbuf_size = 0;
178 return buf_size;
179 }
180 }
181
182 vc1_extract_headers(s, avctx, buf, buf_size);
183
184 *poutbuf = buf;
185 *poutbuf_size = buf_size;
186 return next;
187}

189static int vc1_split(AVCodecContext *avctx,
190 const uint8_t *buf, int buf_size)
191{
192 uint32_t state = -1;
193 int charged = 0;
194 const uint8_t *ptr = buf, *end = buf + buf_size;
195
196 while (ptr < end) {
197 ptr = avpriv_find_start_code(ptr, end, &state);
198 if (state == VC1_CODE_SEQHDR || state == VC1_CODE_ENTRYPOINT)
199 charged = 1;
200 else if (charged && IS_MARKER(state))
201 return ptr - 4 - buf;
202 }
203
204 return 0;
205}

207static av_cold int vc1_parse_init(AVCodecParserContext *s)
208{
209 VC1ParseContext *vpc = s->priv_data;
210 vpc->v.s.slice_context_count = 1;
211 return ff_vc1_init_common(&vpc->v);
212}
