31#define IS_FIL(a) ((a) & MB_TYPE_H261_FIL)

33static void h261_loop_filter(uint8_t *src, int stride)
34{
35 int x, y, xy, yz;
36 int temp[64];
37
38 for (x = 0; x < 8; x++) {
39 temp[x] = 4 * src[x];
40 temp[x + 7 * 8] = 4 * src[x + 7 * stride];
41 }
42 for (y = 1; y < 7; y++) {
43 for (x = 0; x < 8; x++) {
44 xy = y * stride + x;
45 yz = y * 8 + x;
46 temp[yz] = src[xy - stride] + 2 * src[xy] + src[xy + stride];
47 }
48 }
49
50 for (y = 0; y < 8; y++) {
51 src[y * stride] = (temp[y * 8] + 2) >> 2;
52 src[y * stride + 7] = (temp[y * 8 + 7] + 2) >> 2;
53 for (x = 1; x < 7; x++) {
54 xy = y * stride + x;
55 yz = y * 8 + x;
56 src[xy] = (temp[yz - 1] + 2 * temp[yz] + temp[yz + 1] + 8) >> 4;
57 }
58 }
59}

61void ff_h261_loop_filter(MpegEncContext *s)
62{
63 H261Context *h = (H261Context *)s;
64 const int linesize = s->linesize;
65 const int uvlinesize = s->uvlinesize;
66 uint8_t *dest_y = s->dest[0];
67 uint8_t *dest_cb = s->dest[1];
68 uint8_t *dest_cr = s->dest[2];
69
70 if (!(IS_FIL(h->mtype)))
71 return;
72
73 h261_loop_filter(dest_y, linesize);
74 h261_loop_filter(dest_y + 8, linesize);
75 h261_loop_filter(dest_y + 8 * linesize, linesize);
76 h261_loop_filter(dest_y + 8 * linesize + 8, linesize);
77 h261_loop_filter(dest_cb, uvlinesize);
78 h261_loop_filter(dest_cr, uvlinesize);
79}
