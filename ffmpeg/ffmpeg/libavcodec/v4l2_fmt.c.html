112uint32_t ff_v4l2_format_avcodec_to_v4l2(enum AVCodecID avcodec)
113{
114 int i;
115 for (i = 0; i < FF_ARRAY_ELEMS(fmt_map); i++) {
116 if (fmt_map[i].avcodec == avcodec)
117 return fmt_map[i].v4l2_fmt;
118 }
119 return 0;
120}
121
122uint32_t ff_v4l2_format_avfmt_to_v4l2(enum AVPixelFormat avfmt)
123{
124 int i;
125 for (i = 0; i < FF_ARRAY_ELEMS(fmt_map); i++) {
126 if (fmt_map[i].avfmt == avfmt)
127 return fmt_map[i].v4l2_fmt;
128 }
129 return 0;
130}
131
132enum AVPixelFormat ff_v4l2_format_v4l2_to_avfmt(uint32_t v4l2_fmt, enum AVCodecID avcodec)
133{
134 int i;
135 for (i = 0; i < FF_ARRAY_ELEMS(fmt_map); i++) {
136 if (fmt_map[i].avcodec == avcodec &&
137 fmt_map[i].v4l2_fmt == v4l2_fmt)
138 return fmt_map[i].avfmt;
139 }
140 return AV_PIX_FMT_NONE;
141}
