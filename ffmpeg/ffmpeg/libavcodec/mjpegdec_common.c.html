24#include <stdint.h>
25#include "libavutil/avassert.h"
26#include "mjpegdec.h"
27#include "vlc.h"

29static int build_huffman_codes(uint8_t *huff_size, const uint8_t *bits_table)
30{
31 int nb_codes = 0;
32 for (int i = 1, j = 0; i <= 16; i++) {
33 nb_codes += bits_table[i];
34 av_assert1(nb_codes <= 256);
35 for (; j < nb_codes; j++)
36 huff_size[j] = i;
37 }
38 return nb_codes;
39}
40
41int ff_mjpeg_build_vlc(VLC *vlc, const uint8_t *bits_table,
42 const uint8_t *val_table, int is_ac, void *logctx)
43{
44 uint8_t huff_size[256];
45 uint16_t huff_sym[256];
46 int nb_codes = build_huffman_codes(huff_size, bits_table);
47
48 for (int i = 0; i < nb_codes; i++) {
49 huff_sym[i] = val_table[i] + 16 * is_ac;
50
51 if (is_ac && !val_table[i])
52 huff_sym[i] = 16 * 256;
53 }
54
55 return ff_init_vlc_from_lengths(vlc, 9, nb_codes, huff_size, 1,
56 huff_sym, 2, 2, 0, 0, logctx);
57}
