58// Generate a sine window.
59av_cold void ff_sine_window_init(float *window, int n)
60{
61 int i;
62 for(i = 0; i < n; i++)
63 window[i] = sinf((i + 0.5) * (M_PI / (2.0 * n)));
64}

101av_cold void ff_init_ff_sine_windows(int index)
102{
103 assert(index >= 5 && index < FF_ARRAY_ELEMS(ff_sine_windows));
104#if !CONFIG_HARDCODED_TABLES
105#ifdef BUILD_TABLES
106 ff_sine_window_init(ff_sine_windows[index], 1 << index);
107#else
108 ff_thread_once(&init_sine_window_once[index - 5], sine_window_init_func_array[index - 5]);
109#endif
110#endif
111}
