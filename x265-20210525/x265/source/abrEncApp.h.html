111 class Scaler : public Thread
112 {
113 public:
114 PassEncoder *m_parentEnc;
115 int m_id;
116 int m_scalePlanes[3];
117 int m_scaleFrameSize;
118 uint32_t m_threadId;
119 uint32_t m_threadTotal;
120 ThreadSafeInteger m_scaledWriteCnt;
121 VideoDesc* m_srcFormat;
122 VideoDesc* m_dstFormat;
123 int m_threadActive;
124 ScalerFilterManager* m_filterManager;
125
126 Scaler(int threadId, int threadNum, int id, VideoDesc *src, VideoDesc * dst, PassEncoder *parentEnc);
127 bool scalePic(x265_picture *destination, x265_picture *source);
128 void threadMain();
129 void destroy()
130 {
131 if (m_filterManager)
132 {
133 delete m_filterManager;
134 m_filterManager = NULL;
135 }
136 }
137 };
