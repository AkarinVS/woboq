32#if _WIN32
33#define popen _popen
34#define pclose _pclose
35#define pipemode "wb"
36#else
37#define pipemode "w"
38#endif

40bool ReconPlay::pipeValid;
41
42#ifndef _WIN32
43static void sigpipe_handler(int)
44{
45 if (ReconPlay::pipeValid)
46 general_log(NULL, "exec", X265_LOG_ERROR, "pipe closed\n");
47 ReconPlay::pipeValid = false;
48}
49#endif
50
51ReconPlay::ReconPlay(const char* commandLine, x265_param& param)
52{
53#ifndef _WIN32
54 if (signal(SIGPIPE, sigpipe_handler) == SIG_ERR)
55 general_log(&param, "exec", X265_LOG_ERROR, "Unable to register SIGPIPE handler: %s\n", strerror(errno));
56#endif
57
58 width = param.sourceWidth;
59 height = param.sourceHeight;
60 colorSpace = param.internalCsp;
61
62 frameSize = 0;
63 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
64 frameSize += (uint32_t)((width >> x265_cli_csps[colorSpace].width[i]) * (height >> x265_cli_csps[colorSpace].height[i]));
65
66 for (int i = 0; i < RECON_BUF_SIZE; i++)
67 {
68 poc[i] = -1;
69 CHECKED_MALLOC(frameData[i], pixel, frameSize);
70 }
71
72 outputPipe = popen(commandLine, pipemode);
73 if (outputPipe)
74 {
75 const char* csp = (colorSpace >= X265_CSP_I444) ? "444" : (colorSpace >= X265_CSP_I422) ? "422" : "420";
76 const char* depth = (param.internalBitDepth == 10) ? "p10" : "";
77
78 fprintf(outputPipe, "YUV4MPEG2 W%d H%d F%d:%d Ip C%s%s\n", width, height, param.fpsNum, param.fpsDenom, csp, depth);
79
80 pipeValid = true;
81 threadActive = true;
82 start();
83 return;
84 }
85 else
86 general_log(&param, "exec", X265_LOG_ERROR, "popen(%s) failed\n", commandLine);
87
88fail:
89 threadActive = false;
90}

92ReconPlay::~ReconPlay()
93{
94 if (threadActive)
95 {
96 threadActive = false;
97 writeCount.poke();
98 stop();
99 }
100
101 if (outputPipe)
102 pclose(outputPipe);
103
104 for (int i = 0; i < RECON_BUF_SIZE; i++)
105 X265_FREE(frameData[i]);
106}

108bool ReconPlay::writePicture(const x265_picture& pic)
109{
110 if (!threadActive || !pipeValid)
111 return false;
112
113 int written = writeCount.get();
114 int read = readCount.get();
115 int currentCursor = pic.poc % RECON_BUF_SIZE;
116
117 /* TODO: it's probably better to drop recon pictures when the ring buffer is
118 * backed up on the display app */
119 while (written - read > RECON_BUF_SIZE - 2 || poc[currentCursor] != -1)
120 {
121 read = readCount.waitForChange(read);
122 if (!threadActive)
123 return false;
124 }
125
126 X265_CHECK(pic.colorSpace == colorSpace, "invalid color space\n");
127 X265_CHECK(pic.bitDepth == X265_DEPTH, "invalid bit depth\n");
128
129 pixel* buf = frameData[currentCursor];
130 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
131 {
132 char* src = (char*)pic.planes[i];
133 int pwidth = width >> x265_cli_csps[colorSpace].width[i];
134
135 for (int h = 0; h < height >> x265_cli_csps[colorSpace].height[i]; h++)
136 {
137 memcpy(buf, src, pwidth * sizeof(pixel));
138 src += pic.stride[i];
139 buf += pwidth;
140 }
141 }
142
143 poc[currentCursor] = pic.poc;
144 writeCount.incr();
145
146 return true;
147}

149void ReconPlay::threadMain()
150{
151 THREAD_NAME("ReconPlayOutput", 0);
152
153 do
154 {
155 /* extract the next output picture in display order and write to pipe */
156 if (!outputFrame())
157 break;
158 }
159 while (threadActive);
160
161 threadActive = false;
162 readCount.poke();
163}
164
165bool ReconPlay::outputFrame()
166{
167 int written = writeCount.get();
168 int read = readCount.get();
169 int currentCursor = read % RECON_BUF_SIZE;
170
171 while (poc[currentCursor] != read)
172 {
173 written = writeCount.waitForChange(written);
174 if (!threadActive)
175 return false;
176 }
177
178 char* buf = (char*)frameData[currentCursor];
179 intptr_t remainSize = frameSize * sizeof(pixel);
180
181 fprintf(outputPipe, "FRAME\n");
182 while (remainSize > 0)
183 {
184 intptr_t retCount = (intptr_t)fwrite(buf, sizeof(char), remainSize, outputPipe);
185
186 if (retCount < 0 || !pipeValid)
187 /* pipe failure, stop writing and start dropping recon pictures */
188 return false;
189
190 buf += retCount;
191 remainSize -= retCount;
192 }
193
194 poc[currentCursor] = -1;
195 readCount.incr();
196 return true;
197}
