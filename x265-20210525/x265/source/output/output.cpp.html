33ReconFile* ReconFile::open(const char *fname, int width, int height, uint32_t bitdepth, uint32_t fpsNum, uint32_t fpsDenom, int csp)
34{
35 const char * s = strrchr(fname, '.');
36
37 if (s && !strcmp(s, ".y4m"))
38 return new Y4MOutput(fname, width, height, fpsNum, fpsDenom, csp);
39 else
40 return new YUVOutput(fname, width, height, bitdepth, csp);
41}
42
43OutputFile* OutputFile::open(const char *fname, InputFileInfo& inputInfo)
44{
45 return new RAWOutput(fname, inputInfo);
46}
