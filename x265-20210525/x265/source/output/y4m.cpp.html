31Y4MOutput::Y4MOutput(const char *filename, int w, int h, uint32_t fpsNum, uint32_t fpsDenom, int csp)
32 : width(w)
33 , height(h)
34 , colorSpace(csp)
35 , frameSize(0)
36{
37 ofs.open(filename, ios::binary | ios::out);
38 buf = new char[width];
39
40 const char *cf = (csp >= X265_CSP_I444) ? "444" : (csp >= X265_CSP_I422) ? "422" : "420";
41
42 if (ofs)
43 {
44 ofs << "YUV4MPEG2 W" << width << " H" << height << " F" << fpsNum << ":" << fpsDenom << " Ip" << " C" << cf << "\n";
45 header = ofs.tellp();
46 }
47
48 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
49 frameSize += (uint32_t)((width >> x265_cli_csps[colorSpace].width[i]) * (height >> x265_cli_csps[colorSpace].height[i]));
50}
51
52Y4MOutput::~Y4MOutput()
53{
54 ofs.close();
55 delete [] buf;
56}
57
58bool Y4MOutput::writePicture(const x265_picture& pic)
59{
60 std::ofstream::pos_type outPicPos = header;
61 outPicPos += (uint64_t)pic.poc * (6 + frameSize);
62 ofs.seekp(outPicPos);
63 ofs << "FRAME\n";
64
65#if HIGH_BIT_DEPTH
66 if (pic.bitDepth > 8 && pic.poc == 0)
67 x265_log(NULL, X265_LOG_WARNING, "y4m: down-shifting reconstructed pixels to 8 bits\n");
68#else
69 if (pic.bitDepth > 8 && pic.poc == 0)
70 x265_log(NULL, X265_LOG_WARNING, "y4m: forcing reconstructed pixels to 8 bits\n");
71#endif
72
73 X265_CHECK(pic.colorSpace == colorSpace, "invalid chroma subsampling\n");
74
75#if HIGH_BIT_DEPTH
76
77 // encoder gave us short pixels, downshift, then write
78 X265_CHECK(pic.bitDepth > 8, "invalid bit depth\n");
79 int shift = pic.bitDepth - 8;
80 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
81 {
82 uint16_t *src = (uint16_t*)pic.planes[i];
83 for (int h = 0; h < height >> x265_cli_csps[colorSpace].height[i]; h++)
84 {
85 for (int w = 0; w < width >> x265_cli_csps[colorSpace].width[i]; w++)
86 buf[w] = (char)(src[w] >> shift);
87
88 ofs.write(buf, width >> x265_cli_csps[colorSpace].width[i]);
89 src += pic.stride[i] / sizeof(*src);
90 }
91 }
92
93#else // if HIGH_BIT_DEPTH
94
95 X265_CHECK(pic.bitDepth == 8, "invalid bit depth\n");
96 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
97 {
98 char *src = (char*)pic.planes[i];
99 for (int h = 0; h < height >> x265_cli_csps[colorSpace].height[i]; h++)
100 {
101 ofs.write(src, width >> x265_cli_csps[colorSpace].width[i]);
102 src += pic.stride[i] / sizeof(*src);
103 }
104 }
105
106#endif // if HIGH_BIT_DEPTH
107
108 return true;
109}
