43YUVInput::YUVInput(InputFileInfo& info)
44{
45 for (int i = 0; i < QUEUE_SIZE; i++)
46 buf[i] = NULL;
47
48 depth = info.depth;
49 width = info.width;
50 height = info.height;
51 colorSpace = info.csp;
52 threadActive = false;
53 ifs = NULL;
54
55 uint32_t pixelbytes = depth > 8 ? 2 : 1;
56 framesize = 0;
57 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
58 {
59 uint32_t w = width >> x265_cli_csps[colorSpace].width[i];
60 uint32_t h = height >> x265_cli_csps[colorSpace].height[i];
61 framesize += w * h * pixelbytes;
62 }
63
64 if (width == 0 || height == 0 || info.fpsNum == 0 || info.fpsDenom == 0)
65 {
66 x265_log(NULL, X265_LOG_ERROR, "yuv: width, height, and FPS must be specified\n");
67 return;
68 }
69 if (!strcmp(info.filename, "-"))
70 {
71 ifs = stdin;
72#if _WIN32
73 setmode(fileno(stdin), O_BINARY);
74#endif
75 }
76 else
77 ifs = x265_fopen(info.filename, "rb");
78 if (ifs && !ferror(ifs))
79 threadActive = true;
80 else
81 {
82 if (ifs && ifs != stdin)
83 fclose(ifs);
84 ifs = NULL;
85 return;
86 }
87
88 for (uint32_t i = 0; i < QUEUE_SIZE; i++)
89 {
90 buf[i] = X265_MALLOC(char, framesize);
91 if (buf[i] == NULL)
92 {
93 x265_log(NULL, X265_LOG_ERROR, "yuv: buffer allocation failure, aborting\n");
94 threadActive = false;
95 return;
96 }
97 }
98
99 info.frameCount = -1;
100 /* try to estimate frame count, if this is not stdin */
101 if (ifs != stdin)
102 {
103 int64_t cur = ftello(ifs);
104 if (cur >= 0)
105 {
106 fseeko(ifs, 0, SEEK_END);
107 int64_t size = ftello(ifs);
108 fseeko(ifs, cur, SEEK_SET);
109 if (size > 0)
110 info.frameCount = (int)((size - cur) / framesize);
111 }
112 }
113 if (info.skipFrames)
114 {
115 if (ifs != stdin)
116 fseeko(ifs, (int64_t)framesize * info.skipFrames, SEEK_CUR);
117 else
118 for (int i = 0; i < info.skipFrames; i++)
119 if (fread(buf[0], framesize, 1, ifs) != 1)
120 break;
121 }
122}
123YUVInput::~YUVInput()
124{
125 if (ifs && ifs != stdin)
126 fclose(ifs);
127 for (int i = 0; i < QUEUE_SIZE; i++)
128 X265_FREE(buf[i]);
129}

131void YUVInput::release()
132{
133 threadActive = false;
134 readCount.poke();
135 stop();
136 delete this;
137}

147void YUVInput::threadMain()
148{
149 THREAD_NAME("YUVRead", 0);
150 while (threadActive)
151 {
152 if (!populateFrameQueue())
153 break;
154 }
155
156 threadActive = false;
157 writeCount.poke();
158}
159bool YUVInput::populateFrameQueue()
160{
161 if (!ifs || ferror(ifs))
162 return false;
163 /* wait for room in the ring buffer */
164 int written = writeCount.get();
165 int read = readCount.get();
166 while (written - read > QUEUE_SIZE - 2)
167 {
168 read = readCount.waitForChange(read);
169 if (!threadActive)
170 // release() has been called
171 return false;
172 }
173 ProfileScopeEvent(frameRead);
174 if (fread(buf[written % QUEUE_SIZE], framesize, 1, ifs) == 1)
175 {
176 writeCount.incr();
177 return true;
178 }
179 else
180 return false;
181}

183bool YUVInput::readPicture(x265_picture& pic)
184{
185 int read = readCount.get();
186 int written = writeCount.get();
187
188#if ENABLE_THREADING
189
190 /* only wait if the read thread is still active */
191 while (threadActive && read == written)
192 written = writeCount.waitForChange(written);
193
194#else
195
196 populateFrameQueue();
197
198#endif // if ENABLE_THREADING
199
200 if (read < written)
201 {
202 uint32_t pixelbytes = depth > 8 ? 2 : 1;
203 pic.colorSpace = colorSpace;
204 pic.bitDepth = depth;
205 pic.framesize = framesize;
206 pic.height = height;
207 pic.width = width;
208 pic.stride[0] = width * pixelbytes;
209 pic.stride[1] = pic.stride[0] >> x265_cli_csps[colorSpace].width[1];
210 pic.stride[2] = pic.stride[0] >> x265_cli_csps[colorSpace].width[2];
211 pic.planes[0] = buf[read % QUEUE_SIZE];
212 pic.planes[1] = (char*)pic.planes[0] + pic.stride[0] * height;
213 pic.planes[2] = (char*)pic.planes[1] + pic.stride[1] * (height >> x265_cli_csps[colorSpace].height[1]);
214 readCount.incr();
215 return true;
216 }
217 else
218 return false;
219}
