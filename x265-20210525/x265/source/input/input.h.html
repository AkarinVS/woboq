24#ifndef X265_INPUT_H
25#define X265_INPUT_H

27#define MIN_FRAME_WIDTH 64
28#define MAX_FRAME_WIDTH 8192
29#define MIN_FRAME_HEIGHT 64
30#define MAX_FRAME_HEIGHT 4320
31#define MIN_FRAME_RATE 1
32#define MAX_FRAME_RATE 300

34#include "common.h"
35
36namespace X265_NS {
37// private x265 namespace
38
39struct InputFileInfo
40{
41 /* possibly user-supplied, possibly read from file header */
42 int width;
43 int height;
44 int csp;
45 int depth;
46 int fpsNum;
47 int fpsDenom;
48 int sarWidth;
49 int sarHeight;
50 int frameCount;
51 int timebaseNum;
52 int timebaseDenom;
53
54 /* user supplied */
55 int skipFrames;
56 const char *filename;
57};
58
59class InputFile
60{
61protected:
62
63 virtual ~InputFile() {}
64
65public:
66
67 InputFile() {}
68
69 static InputFile* open(InputFileInfo& info, bool bForceY4m);
70
71 virtual void startReader() = 0;
72
73 virtual void release() = 0;
74
75 virtual bool readPicture(x265_picture& pic) = 0;
76
77 virtual bool isEof() const = 0;
78
79 virtual bool isFail() = 0;
80
81 virtual const char *getName() const = 0;
82
83 virtual int getWidth() const = 0;
84
85 virtual int getHeight() const = 0;
86};
87}
