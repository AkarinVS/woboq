24#ifndef X265_Y4M_H
25#define X265_Y4M_H
26
27#include "input.h"
28#include "threading.h"
29#include <fstream>
30
31#define QUEUE_SIZE 5
32
33namespace X265_NS {
34// x265 private namespace
35
36class Y4MInput : public InputFile, public Thread
37{
38protected:
39
40 uint32_t rateNum;
41
42 uint32_t rateDenom;
43
44 uint32_t sarWidth;
45
46 uint32_t sarHeight;
47
48 size_t framesize;
49
50 int depth;
51
52 int width;
53
54 int height;
55
56 int colorSpace;
57
58 bool threadActive;
59
60 ThreadSafeInteger readCount;
61
62 ThreadSafeInteger writeCount;
63 char* buf[QUEUE_SIZE];
64 FILE *ifs;
65 bool parseHeader();
66 void threadMain();
67
68 bool populateFrameQueue();
69
70public:
71
72 Y4MInput(InputFileInfo& info);
73
74 virtual ~Y4MInput();
75 void release();
76 bool isEof() const { return ifs && feof(ifs); }
77 bool isFail() { return !(ifs && !ferror(ifs) && threadActive); }
78 void startReader();
79 bool readPicture(x265_picture&);
80
81 const char *getName() const { return "y4m"; }
82
83 int getWidth() const { return width; }
84
85 int getHeight() const { return height; }
86};
87}
