30InputFile* InputFile::open(InputFileInfo& info, bool bForceY4m)
31{
32 const char * s = strrchr(info.filename, '.');
33
34 if (bForceY4m || (s && !strcmp(s, ".y4m")))
35 return new Y4MInput(info);
36 else
37 return new YUVInput(info);
38}
