42static const char header[] = {'F','R','A','M','E'};
43Y4MInput::Y4MInput(InputFileInfo& info)
44{
45 for (int i = 0; i < QUEUE_SIZE; i++)
46 buf[i] = NULL;
47
48 threadActive = false;
49 colorSpace = info.csp;
50 sarWidth = info.sarWidth;
51 sarHeight = info.sarHeight;
52 width = info.width;
53 height = info.height;
54 rateNum = info.fpsNum;
55 rateDenom = info.fpsDenom;
56 depth = info.depth;
57 framesize = 0;
58
59 ifs = NULL;
60 if (!strcmp(info.filename, "-"))
61 {
62 ifs = stdin;
63#if _WIN32
64 setmode(fileno(stdin), O_BINARY);
65#endif
66 }
67 else
68 ifs = x265_fopen(info.filename, "rb");
69 if (ifs && !ferror(ifs) && parseHeader())
70 {
71 int pixelbytes = depth > 8 ? 2 : 1;
72 for (int i = 0; i < x265_cli_csps[colorSpace].planes; i++)
73 {
74 int stride = (width >> x265_cli_csps[colorSpace].width[i]) * pixelbytes;
75 framesize += (stride * (height >> x265_cli_csps[colorSpace].height[i]));
76 }
77
78 threadActive = true;
79 for (int q = 0; q < QUEUE_SIZE; q++)
80 {
81 buf[q] = X265_MALLOC(char, framesize);
82 if (!buf[q])
83 {
84 x265_log(NULL, X265_LOG_ERROR, "y4m: buffer allocation failure, aborting");
85 threadActive = false;
86 break;
87 }
88 }
89 }
90 if (!threadActive)
91 {
92 if (ifs && ifs != stdin)
93 fclose(ifs);
94 ifs = NULL;
95 return;
96 }
97
98 info.width = width;
99 info.height = height;
100 info.sarHeight = sarHeight;
101 info.sarWidth = sarWidth;
102 info.fpsNum = rateNum;
103 info.fpsDenom = rateDenom;
104 info.csp = colorSpace;
105 info.depth = depth;
106 info.frameCount = -1;
107 size_t estFrameSize = framesize + sizeof(header) + 1; /* assume basic FRAME\n headers */
108 /* try to estimate frame count, if this is not stdin */
109 if (ifs != stdin)
110 {
111 int64_t cur = ftello(ifs);
112 if (cur >= 0)
113 {
114 fseeko(ifs, 0, SEEK_END);
115 int64_t size = ftello(ifs);
116 fseeko(ifs, cur, SEEK_SET);
117 if (size > 0)
118 info.frameCount = (int)((size - cur) / estFrameSize);
119 }
120 }
121 if (info.skipFrames)
122 {
123 if (ifs != stdin)
124 fseeko(ifs, (int64_t)estFrameSize * info.skipFrames, SEEK_CUR);
125 else
126 for (int i = 0; i < info.skipFrames; i++)
127 if (fread(buf[0], estFrameSize - framesize, 1, ifs) + fread(buf[0], framesize, 1, ifs) != 2)
128 break;
129 }
130}
131Y4MInput::~Y4MInput()
132{
133 if (ifs && ifs != stdin)
134 fclose(ifs);
135 for (int i = 0; i < QUEUE_SIZE; i++)
136 X265_FREE(buf[i]);
137}

147bool Y4MInput::parseHeader()
148{
149 if (!ifs)
150 return false;
151
152 int csp = 0;
153 int d = 0;
154 int c;
155 while ((c = fgetc(ifs)) != EOF)
156 {
157 // Skip Y4MPEG string
158 while ((c != EOF) && (c != ' ') && (c != '\n'))
159 c = fgetc(ifs);
160 while (c == ' ')
161 {
162 // read parameter identifier
163 switch (fgetc(ifs))
164 {
165 case 'W':
166 width = 0;
167 while ((c = fgetc(ifs)) != EOF)
168 {
169 if (c == ' ' || c == '\n')
170 break;
171 else
172 width = width * 10 + (c - '0');
173 }
174 break;
175 case 'H':
176 height = 0;
177 while ((c = fgetc(ifs)) != EOF)
178 {
179 if (c == ' ' || c == '\n')
180 break;
181 else
182 height = height * 10 + (c - '0');
183 }
184 break;
185
186 case 'F':
187 rateNum = 0;
188 rateDenom = 0;
189 while ((c = fgetc(ifs)) != EOF)
190 {
191 if (c == '.')
192 {
193 rateDenom = 1;
194 while ((c = fgetc(ifs)) != EOF)
195 {
196 if (c == ' ' || c == '\n')
197 break;
198 else
199 {
200 rateNum = rateNum * 10 + (c - '0');
201 rateDenom = rateDenom * 10;
202 }
203 }
204 break;
205 }
206 else if (c == ':')
207 {
208 while ((c = fgetc(ifs)) != EOF)
209 {
210 if (c == ' ' || c == '\n')
211 break;
212 else
213 rateDenom = rateDenom * 10 + (c - '0');
214 }
215 break;
216 }
217 else
218 rateNum = rateNum * 10 + (c - '0');
219 }
220 break;
221
222 case 'A':
223 sarWidth = 0;
224 sarHeight = 0;
225 while ((c = fgetc(ifs)) != EOF)
226 {
227 if (c == ':')
228 {
229 while ((c = fgetc(ifs)) != EOF)
230 {
231 if (c == ' ' || c == '\n')
232 break;
233 else
234 sarHeight = sarHeight * 10 + (c - '0');
235 }
236 break;
237 }
238 else
239 sarWidth = sarWidth * 10 + (c - '0');
240 }
241 break;
242
243 case 'C':
244 csp = 0;
245 d = 0;
246 while ((c = fgetc(ifs)) != EOF)
247 {
248 if (c <= 'o' && c >= '0')
249 csp = csp * 10 + (c - '0');
250 else if (c == 'p')
251 {
252 // example: C420p16
253 while ((c = fgetc(ifs)) != EOF)
254 {
255 if (c <= '9' && c >= '0')
256 d = d * 10 + (c - '0');
257 else
258 break;
259 }
260 break;
261 }
262 else
263 break;
264 }
265
266 if (csp / 100 == ('m'-'0')*1000 + ('o'-'0')*100 + ('n'-'0')*10 + ('o'-'0'))
267 {
268 colorSpace = X265_CSP_I400;
269 d = csp % 100;
270 }
271 else if (csp / 10 == ('m'-'0')*1000 + ('o'-'0')*100 + ('n'-'0')*10 + ('o'-'0'))
272 {
273 colorSpace = X265_CSP_I400;
274 d = csp % 10;
275 }
276 else if (csp == ('m'-'0')*1000 + ('o'-'0')*100 + ('n'-'0')*10 + ('o'-'0'))
277 {
278 colorSpace = X265_CSP_I400;
279 d = 8;
280 }
281 else
282 colorSpace = (csp == 444) ? X265_CSP_I444 : (csp == 422) ? X265_CSP_I422 : X265_CSP_I420;
283
284 if (d >= 8 && d <= 16)
285 depth = d;
286 break;
287 default:
288 while ((c = fgetc(ifs)) != EOF)
289 {
290 // consume this unsupported configuration word
291 if (c == ' ' || c == '\n')
292 break;
293 }
294 break;
295 }
296 }
297
298 if (c == '\n')
299 break;
300 }
301
302 if (width < MIN_FRAME_WIDTH || width > MAX_FRAME_WIDTH ||
303 height < MIN_FRAME_HEIGHT || height > MAX_FRAME_HEIGHT ||
304 (rateNum / rateDenom) < 1 || (rateNum / rateDenom) > MAX_FRAME_RATE ||
305 colorSpace < X265_CSP_I400 || colorSpace >= X265_CSP_COUNT)
306 return false;
307
308 return true;
309}

319void Y4MInput::threadMain()
320{
321 THREAD_NAME("Y4MRead", 0);
322 do
323 {
324 if (!populateFrameQueue())
325 break;
326 }
327 while (threadActive);
328
329 threadActive = false;
330 writeCount.poke();
331}
332bool Y4MInput::populateFrameQueue()
333{
334 if (!ifs || ferror(ifs))
335 return false;
336 /* strip off the FRAME\n header */
337 char hbuf[sizeof(header) + 1];
338 if (fread(hbuf, sizeof(hbuf), 1, ifs) != 1 || memcmp(hbuf, header, sizeof(header)))
339 {
340 if (!feof(ifs))
341 x265_log(NULL, X265_LOG_ERROR, "y4m: frame header missing\n");
342 return false;
343 }
344 /* consume bytes up to line feed */
345 int c = hbuf[sizeof(header)];
346 while (c != '\n')
347 if ((c = fgetc(ifs)) == EOF)
348 break;
349 /* wait for room in the ring buffer */
350 int written = writeCount.get();
351 int read = readCount.get();
352 while (written - read > QUEUE_SIZE - 2)
353 {
354 read = readCount.waitForChange(read);
355 if (!threadActive)
356 return false;
357 }
358 ProfileScopeEvent(frameRead);
359 if (fread(buf[written % QUEUE_SIZE], framesize, 1, ifs) == 1)
360 {
361 writeCount.incr();
362 return true;
363 }
364 else
365 return false;
366}

368bool Y4MInput::readPicture(x265_picture& pic)
369{
370 int read = readCount.get();
371 int written = writeCount.get();
372
373#if ENABLE_THREADING
374
375 /* only wait if the read thread is still active */
376 while (threadActive && read == written)
377 written = writeCount.waitForChange(written);
378
379#else
380
381 populateFrameQueue();
382
383#endif // if ENABLE_THREADING
384
385 if (read < written)
386 {
387 int pixelbytes = depth > 8 ? 2 : 1;
388 pic.bitDepth = depth;
389 pic.framesize = framesize;
390 pic.height = height;
391 pic.width = width;
392 pic.colorSpace = colorSpace;
393 pic.stride[0] = width * pixelbytes;
394 pic.stride[1] = pic.stride[0] >> x265_cli_csps[colorSpace].width[1];
395 pic.stride[2] = pic.stride[0] >> x265_cli_csps[colorSpace].width[2];
396 pic.planes[0] = buf[read % QUEUE_SIZE];
397 pic.planes[1] = (char*)pic.planes[0] + pic.stride[0] * height;
398 pic.planes[2] = (char*)pic.planes[1] + pic.stride[1] * (height >> x265_cli_csps[colorSpace].height[1]);
399 readCount.incr();
400 return true;
401 }
402 else
403 return false;
404}
