63/* determine minimum decoder level required to decode the described video */
64void determineLevel(const x265_param &param, VPS& vps)
65{
66 vps.ptl.onePictureOnlyConstraintFlag = param.totalFrames == 1;
67 vps.ptl.intraConstraintFlag = param.keyframeMax <= 1 || vps.ptl.onePictureOnlyConstraintFlag;
68 vps.ptl.bitDepthConstraint = param.internalBitDepth;
69 vps.ptl.chromaFormatConstraint = param.internalCsp;
70
71 /* TODO: figure out HighThroughput signaling, aka: HbrFactor in section A.4.2, only available
72 * for intra-only profiles (vps.ptl.intraConstraintFlag) */
73 vps.ptl.lowerBitRateConstraintFlag = true;
74
75 vps.maxTempSubLayers = param.bEnableTemporalSubLayers ? 2 : 1;
76
77 if (param.internalCsp == X265_CSP_I420 && param.internalBitDepth <= 10)
78 {
79 /* Probably an HEVC v1 profile, but must check to be sure */
80 if (param.internalBitDepth <= 8)
81 {
82 if (vps.ptl.onePictureOnlyConstraintFlag)
83 vps.ptl.profileIdc = Profile::MAINSTILLPICTURE;
84 else if (vps.ptl.intraConstraintFlag)
85 vps.ptl.profileIdc = Profile::MAINREXT; /* Main Intra */
86 else
87 vps.ptl.profileIdc = Profile::MAIN;
88 }
89 else if (param.internalBitDepth <= 10)
90 {
91 /* note there is no 10bit still picture profile */
92 if (vps.ptl.intraConstraintFlag)
93 vps.ptl.profileIdc = Profile::MAINREXT; /* Main10 Intra */
94 else
95 vps.ptl.profileIdc = Profile::MAIN10;
96 }
97 }
98 else
99 vps.ptl.profileIdc = Profile::MAINREXT;
100
101 /* determine which profiles are compatible with this stream */
102
103 memset(vps.ptl.profileCompatibilityFlag, 0, sizeof(vps.ptl.profileCompatibilityFlag));
104 vps.ptl.profileCompatibilityFlag[vps.ptl.profileIdc] = true;
105 if (vps.ptl.profileIdc == Profile::MAIN10 && param.internalBitDepth == 8)
106 vps.ptl.profileCompatibilityFlag[Profile::MAIN] = true;
107 else if (vps.ptl.profileIdc == Profile::MAIN)
108 vps.ptl.profileCompatibilityFlag[Profile::MAIN10] = true;
109 else if (vps.ptl.profileIdc == Profile::MAINSTILLPICTURE)
110 {
111 vps.ptl.profileCompatibilityFlag[Profile::MAIN] = true;
112 vps.ptl.profileCompatibilityFlag[Profile::MAIN10] = true;
113 }
114 else if (vps.ptl.profileIdc == Profile::MAINREXT)
115 vps.ptl.profileCompatibilityFlag[Profile::MAINREXT] = true;
116
117 uint32_t lumaSamples = param.sourceWidth * param.sourceHeight;
118 uint32_t samplesPerSec = (uint32_t)(lumaSamples * ((double)param.fpsNum / param.fpsDenom));
119 uint32_t bitrate = param.rc.vbvMaxBitrate ? param.rc.vbvMaxBitrate : param.rc.bitrate;
120
121 const uint32_t MaxDpbPicBuf = 6;
122 vps.ptl.levelIdc = Level::NONE;
123 vps.ptl.tierFlag = Level::MAIN;
124
125 const size_t NumLevels = sizeof(levels) / sizeof(levels[0]);
126 uint32_t i;
127 if (param.bLossless)
128 {
129 i = 13;
130 vps.ptl.minCrForLevel = 1;
131 vps.ptl.maxLumaSrForLevel = MAX_UINT;
132 vps.ptl.levelIdc = Level::LEVEL8_5;
133 vps.ptl.tierFlag = Level::MAIN;
134 }
135 else if (param.uhdBluray)
136 {
137 i = 8;
138 vps.ptl.levelIdc = levels[i].levelEnum;
139 vps.ptl.tierFlag = Level::HIGH;
140 vps.ptl.minCrForLevel = levels[i].minCompressionRatio;
141 vps.ptl.maxLumaSrForLevel = levels[i].maxLumaSamplesPerSecond;
142 }
143 else for (i = 0; i < NumLevels; i++)
144 {
145 if (lumaSamples > levels[i].maxLumaSamples)
146 continue;
147 else if (samplesPerSec > levels[i].maxLumaSamplesPerSecond)
148 continue;
149 else if (bitrate > levels[i].maxBitrateMain && levels[i].maxBitrateHigh == MAX_UINT)
150 continue;
151 else if (bitrate > levels[i].maxBitrateHigh)
152 continue;
153 else if (param.sourceWidth > sqrt(levels[i].maxLumaSamples * 8.0f))
154 continue;
155 else if (param.sourceHeight > sqrt(levels[i].maxLumaSamples * 8.0f))
156 continue;
157 else if (param.levelIdc && param.levelIdc != levels[i].levelIdc)
158 continue;
159 uint32_t maxDpbSize = MaxDpbPicBuf;
160
161 if (lumaSamples <= (levels[i].maxLumaSamples >> 2))
162 maxDpbSize = X265_MIN(4 * MaxDpbPicBuf, 16);
163 else if (lumaSamples <= (levels[i].maxLumaSamples >> 1))
164 maxDpbSize = X265_MIN(2 * MaxDpbPicBuf, 16);
165 else if (lumaSamples <= ((3 * levels[i].maxLumaSamples) >> 2))
166 maxDpbSize = X265_MIN((4 * MaxDpbPicBuf) / 3, 16);
167
168 /* The value of sps_max_dec_pic_buffering_minus1[ HighestTid ] + 1 shall be less than
169 * or equal to MaxDpbSize */
170 if (vps.maxDecPicBuffering > maxDpbSize)
171 continue;
172
173 /* For level 5 and higher levels, the value of CtbSizeY shall be equal to 32 or 64 */
174 if (levels[i].levelEnum >= Level::LEVEL5 && param.maxCUSize < 32)
175 {
176 x265_log(&param, X265_LOG_WARNING, "level %s detected, but CTU size 16 is non-compliant\n", levels[i].name);
177 vps.ptl.profileIdc = Profile::NONE;
178 vps.ptl.levelIdc = Level::NONE;
179 vps.ptl.tierFlag = Level::MAIN;
180 x265_log(&param, X265_LOG_INFO, "NONE profile, Level-NONE (Main tier)\n");
181 return;
182 }
183
184 /* The value of NumPocTotalCurr shall be less than or equal to 8 */
185 int numPocTotalCurr = param.maxNumReferences + vps.numReorderPics;
186 if (numPocTotalCurr > 8)
187 {
188 x265_log(&param, X265_LOG_WARNING, "level %s detected, but NumPocTotalCurr (total references) is non-compliant\n", levels[i].name);
189 vps.ptl.profileIdc = Profile::NONE;
190 vps.ptl.levelIdc = Level::NONE;
191 vps.ptl.tierFlag = Level::MAIN;
192 x265_log(&param, X265_LOG_INFO, "NONE profile, Level-NONE (Main tier)\n");
193 return;
194 }
195
196#define CHECK_RANGE(value, main, high) (high != MAX_UINT && value > main && value <= high)
197
198 if (CHECK_RANGE(bitrate, levels[i].maxBitrateMain, levels[i].maxBitrateHigh) ||
199 CHECK_RANGE((uint32_t)param.rc.vbvBufferSize, levels[i].maxCpbSizeMain, levels[i].maxCpbSizeHigh))
200 {
201 /* The bitrate or buffer size are out of range for Main tier, but in
202 * range for High tier. If the user allowed High tier then give
203 * them High tier at this level. Otherwise allow the loop to
204 * progress to the Main tier of the next level */
205 if (param.bHighTier)
206 vps.ptl.tierFlag = Level::HIGH;
207 else
208 continue;
209 }
210 else
211 vps.ptl.tierFlag = Level::MAIN;
212#undef CHECK_RANGE
213
214 vps.ptl.levelIdc = levels[i].levelEnum;
215 vps.ptl.minCrForLevel = levels[i].minCompressionRatio;
216 vps.ptl.maxLumaSrForLevel = levels[i].maxLumaSamplesPerSecond;
217 break;
218 }
219
220 static const char *profiles[] = { "None", "Main", "Main 10", "Main Still Picture", "RExt" };
221 static const char *tiers[] = { "Main", "High" };
222
223 char profbuf[64];
224 strcpy(profbuf, profiles[vps.ptl.profileIdc]);
225
226 bool bStillPicture = false;
227 if (vps.ptl.profileIdc == Profile::MAINREXT)
228 {
229 if (vps.ptl.bitDepthConstraint > 12 && vps.ptl.intraConstraintFlag)
230 {
231 if (vps.ptl.onePictureOnlyConstraintFlag)
232 {
233 strcpy(profbuf, "Main 4:4:4 16 Still Picture");
234 bStillPicture = true;
235 }
236 else
237 strcpy(profbuf, "Main 4:4:4 16");
238 }
239 else if (param.internalCsp == X265_CSP_I420)
240 {
241 X265_CHECK(vps.ptl.intraConstraintFlag || vps.ptl.bitDepthConstraint > 10, "rext fail\n");
242 if (vps.ptl.bitDepthConstraint <= 8)
243 strcpy(profbuf, "Main");
244 else if (vps.ptl.bitDepthConstraint <= 10)
245 strcpy(profbuf, "Main 10");
246 else if (vps.ptl.bitDepthConstraint <= 12)
247 strcpy(profbuf, "Main 12");
248 }
249 else if (param.internalCsp == X265_CSP_I422)
250 {
251 /* there is no Main 4:2:2 profile, so it must be signaled as Main10 4:2:2 */
252 if (param.internalBitDepth <= 10)
253 strcpy(profbuf, "Main 4:2:2 10");
254 else if (vps.ptl.bitDepthConstraint <= 12)
255 strcpy(profbuf, "Main 4:2:2 12");
256 }
257 else if (param.internalCsp == X265_CSP_I444)
258 {
259 if (vps.ptl.bitDepthConstraint <= 8)
260 {
261 if (vps.ptl.onePictureOnlyConstraintFlag)
262 {
263 strcpy(profbuf, "Main 4:4:4 Still Picture");
264 bStillPicture = true;
265 }
266 else
267 strcpy(profbuf, "Main 4:4:4");
268 }
269 else if (vps.ptl.bitDepthConstraint <= 10)
270 strcpy(profbuf, "Main 4:4:4 10");
271 else if (vps.ptl.bitDepthConstraint <= 12)
272 strcpy(profbuf, "Main 4:4:4 12");
273 }
274 else
275 strcpy(profbuf, "Unknown");
276
277 if (vps.ptl.intraConstraintFlag && !bStillPicture)
278 strcat(profbuf, " Intra");
279 }
280 x265_log(&param, X265_LOG_INFO, "%s profile, Level-%s (%s tier)\n",
281 profbuf, levels[i].name, tiers[vps.ptl.tierFlag]);
282}

284/* enforce a maximum decoder level requirement, in other words assure that a
285 * decoder of the specified level may decode the video about to be created.
286 * Lower parameters where necessary to ensure the video will be decodable by a
287 * decoder meeting this level of requirement. Some parameters (resolution and
288 * frame rate) are non-negotiable and thus this function may fail. In those
289 * circumstances it will be quite noisy */
290bool enforceLevel(x265_param& param, VPS& vps)
291{
292 vps.numReorderPics = (param.bBPyramid && param.bframes > 1) ? 2 : !!param.bframes;
293 vps.maxDecPicBuffering = X265_MIN(MAX_NUM_REF, X265_MAX(vps.numReorderPics + 2, (uint32_t)param.maxNumReferences) + 1);
294
295 /* no level specified by user, just auto-detect from the configuration */
296 if (param.levelIdc <= 0)
297 return true;
298
299 uint32_t level = 0;
300 while (levels[level].levelIdc != param.levelIdc && level + 1 < sizeof(levels) / sizeof(levels[0]))
301 level++;
302 if (levels[level].levelIdc != param.levelIdc)
303 {
304 x265_log(&param, X265_LOG_ERROR, "specified level %d does not exist\n", param.levelIdc);
305 return false;
306 }
307
308 LevelSpec& l = levels[level];
309
310 //highTier is allowed for this level and has not been explicitly disabled. This does not mean it is the final chosen tier
311 bool allowHighTier = l.maxBitrateHigh < MAX_UINT && param.bHighTier;
312
313 uint32_t lumaSamples = param.sourceWidth * param.sourceHeight;
314 uint32_t samplesPerSec = (uint32_t)(lumaSamples * ((double)param.fpsNum / param.fpsDenom));
315 bool ok = true;
316 if (lumaSamples > l.maxLumaSamples)
317 ok = false;
318 else if (param.sourceWidth > sqrt(l.maxLumaSamples * 8.0f))
319 ok = false;
320 else if (param.sourceHeight > sqrt(l.maxLumaSamples * 8.0f))
321 ok = false;
322 if (!ok)
323 {
324 x265_log(&param, X265_LOG_ERROR, "picture dimensions are out of range for specified level\n");
325 return false;
326 }
327 else if (samplesPerSec > l.maxLumaSamplesPerSecond)
328 {
329 x265_log(&param, X265_LOG_ERROR, "frame rate is out of range for specified level\n");
330 return false;
331 }
332
333 /* Adjustments of Bitrate, VBV buffer size, refs will be triggered only if specified params do not fit
334 * within the max limits of that level (high tier if allowed, main otherwise)
335 */
336
337 if ((uint32_t)param.rc.vbvMaxBitrate > (allowHighTier ? l.maxBitrateHigh : l.maxBitrateMain))
338 {
339 param.rc.vbvMaxBitrate = allowHighTier ? l.maxBitrateHigh : l.maxBitrateMain;
340 x265_log(&param, X265_LOG_WARNING, "lowering VBV max bitrate to %dKbps\n", param.rc.vbvMaxBitrate);
341 }
342 if ((uint32_t)param.rc.vbvBufferSize > (allowHighTier ? l.maxCpbSizeHigh : l.maxCpbSizeMain))
343 {
344 param.rc.vbvBufferSize = allowHighTier ? l.maxCpbSizeHigh : l.maxCpbSizeMain;
345 x265_log(&param, X265_LOG_WARNING, "lowering VBV buffer size to %dKb\n", param.rc.vbvBufferSize);
346 }
347
348 switch (param.rc.rateControlMode)
349 {
350 case X265_RC_ABR:
351 if ((uint32_t)param.rc.bitrate > (allowHighTier ? l.maxBitrateHigh : l.maxBitrateMain))
352 {
353 param.rc.bitrate = allowHighTier ? l.maxBitrateHigh : l.maxBitrateMain;
354 x265_log(&param, X265_LOG_WARNING, "lowering target bitrate to High tier limit of %dKbps\n", param.rc.bitrate);
355 }
356 break;
357
358 case X265_RC_CQP:
359 x265_log(&param, X265_LOG_ERROR, "Constant QP is inconsistent with specifying a decoder level, no bitrate guarantee is possible.\n");
360 return false;
361
362 case X265_RC_CRF:
363 if (!param.rc.vbvBufferSize || !param.rc.vbvMaxBitrate)
364 {
365 if (!param.rc.vbvMaxBitrate)
366 param.rc.vbvMaxBitrate = allowHighTier ? l.maxBitrateHigh : l.maxBitrateMain;
367 if (!param.rc.vbvBufferSize)
368 param.rc.vbvBufferSize = allowHighTier ? l.maxCpbSizeHigh : l.maxCpbSizeMain;
369 x265_log(&param, X265_LOG_WARNING, "Specifying a decoder level with constant rate factor rate-control requires\n");
370 x265_log(&param, X265_LOG_WARNING, "enabling VBV with vbv-bufsize=%dkb vbv-maxrate=%dkbps. VBV outputs are non-deterministic!\n",
371 param.rc.vbvBufferSize, param.rc.vbvMaxBitrate);
372 }
373 break;
374
375 default:
376 x265_log(&param, X265_LOG_ERROR, "Unknown rate control mode is inconsistent with specifying a decoder level\n");
377 return false;
378 }
379
380 /* The value of sps_max_dec_pic_buffering_minus1[ HighestTid ] + 1 shall be less than or equal to MaxDpbSize */
381 const uint32_t MaxDpbPicBuf = 6;
382 uint32_t maxDpbSize = MaxDpbPicBuf;
383 if (!param.uhdBluray) /* Do not change MaxDpbPicBuf for UHD-Bluray */
384 {
385 if (lumaSamples <= (l.maxLumaSamples >> 2))
386 maxDpbSize = X265_MIN(4 * MaxDpbPicBuf, 16);
387 else if (lumaSamples <= (l.maxLumaSamples >> 1))
388 maxDpbSize = X265_MIN(2 * MaxDpbPicBuf, 16);
389 else if (lumaSamples <= ((3 * l.maxLumaSamples) >> 2))
390 maxDpbSize = X265_MIN((4 * MaxDpbPicBuf) / 3, 16);
391 }
392
393 int savedRefCount = param.maxNumReferences;
394 while (vps.maxDecPicBuffering > maxDpbSize && param.maxNumReferences > 1)
395 {
396 param.maxNumReferences--;
397 vps.maxDecPicBuffering = X265_MIN(MAX_NUM_REF, X265_MAX(vps.numReorderPics + 1, (uint32_t)param.maxNumReferences) + 1);
398 }
399 if (param.maxNumReferences != savedRefCount)
400 x265_log(&param, X265_LOG_WARNING, "Lowering max references to %d to meet level requirement\n", param.maxNumReferences);
401
402 /* For level 5 and higher levels, the value of CtbSizeY shall be equal to 32 or 64 */
403 if (param.levelIdc >= 50 && param.maxCUSize < 32)
404 {
405 param.maxCUSize = 32;
406 x265_log(&param, X265_LOG_WARNING, "Levels 5.0 and above require a maximum CTU size of at least 32, using --ctu 32\n");
407 }
408
409 /* The value of NumPocTotalCurr shall be less than or equal to 8 */
410 int numPocTotalCurr = param.maxNumReferences + !!param.bframes;
411 if (numPocTotalCurr > 8)
412 {
413 param.maxNumReferences = 8 - !!param.bframes;
414 x265_log(&param, X265_LOG_WARNING, "Lowering max references to %d to meet numPocTotalCurr requirement\n", param.maxNumReferences);
415 }
416
417 return true;
418}

434int x265_param_apply_profile(x265_param *param, const char *profile)
435{
436 if (!param || !profile)
437 return 0;
438
439#ifdef SVT_HEVC
440 if (param->bEnableSvtHevc)
441 {
442 EB_H265_ENC_CONFIGURATION* svtParam = (EB_H265_ENC_CONFIGURATION*)param->svtHevcParam;
443 if (!strcmp(profile, "main")) svtParam->profile = 1;
444 else if (!strcmp(profile, "main10")) svtParam->profile = 2;
445 else
446 {
447 x265_log(param, X265_LOG_ERROR, "SVT-HEVC encoder: Unsupported profile %s \n", profile);
448 return -1;
449 }
450 return 0;
451 }
452#endif
453
454 /* Check if profile bit-depth requirement is exceeded by internal bit depth */
455 bool bInvalidDepth = false;
456#if X265_DEPTH > 8
457 if (!strcmp(profile, "main") || !strcmp(profile, "mainstillpicture") || !strcmp(profile, "msp") ||
458 !strcmp(profile, "main444-8") || !strcmp(profile, "main-intra") ||
459 !strcmp(profile, "main444-intra") || !strcmp(profile, "main444-stillpicture"))
460 bInvalidDepth = true;
461#endif
462#if X265_DEPTH > 10
463 if (!strcmp(profile, "main10") || !strcmp(profile, "main422-10") || !strcmp(profile, "main444-10") ||
464 !strcmp(profile, "main10-intra") || !strcmp(profile, "main422-10-intra") || !strcmp(profile, "main444-10-intra"))
465 bInvalidDepth = true;
466#endif
467#if X265_DEPTH > 12
468 if (!strcmp(profile, "main12") || !strcmp(profile, "main422-12") || !strcmp(profile, "main444-12") ||
469 !strcmp(profile, "main12-intra") || !strcmp(profile, "main422-12-intra") || !strcmp(profile, "main444-12-intra"))
470 bInvalidDepth = true;
471#endif
472
473 if (bInvalidDepth)
474 {
475 x265_log(param, X265_LOG_ERROR, "%s profile not supported, internal bit depth %d.\n", profile, X265_DEPTH);
476 return -1;
477 }
478
479 size_t l = strlen(profile);
480 bool bBoolIntra = (l > 6 && !strcmp(profile + l - 6, "-intra")) ||
481 !strcmp(profile, "mainstillpicture") || !strcmp(profile, "msp");
482 if (bBoolIntra)
483 {
484 /* The profile may be detected as still picture if param->totalFrames is 1 */
485 param->keyframeMax = 1;
486 }
487
488 /* check that input color space is supported by profile */
489 if (!strcmp(profile, "main") || !strcmp(profile, "main-intra") ||
490 !strcmp(profile, "main10") || !strcmp(profile, "main10-intra") ||
491 !strcmp(profile, "main12") || !strcmp(profile, "main12-intra") ||
492 !strcmp(profile, "mainstillpicture") || !strcmp(profile, "msp"))
493 {
494 if (param->internalCsp != X265_CSP_I420)
495 {
496 x265_log(param, X265_LOG_ERROR, "%s profile not compatible with %s input chroma subsampling.\n",
497 profile, x265_source_csp_names[param->internalCsp]);
498 return -1;
499 }
500 }
501 else if (!strcmp(profile, "main422-10") || !strcmp(profile, "main422-10-intra") ||
502 !strcmp(profile, "main422-12") || !strcmp(profile, "main422-12-intra"))
503 {
504 if (param->internalCsp != X265_CSP_I420 && param->internalCsp != X265_CSP_I422)
505 {
506 x265_log(param, X265_LOG_ERROR, "%s profile not compatible with %s input chroma subsampling.\n",
507 profile, x265_source_csp_names[param->internalCsp]);
508 return -1;
509 }
510 }
511 else if (!strcmp(profile, "main444-8") ||
512 !strcmp(profile, "main444-intra") || !strcmp(profile, "main444-stillpicture") ||
513 !strcmp(profile, "main444-10") || !strcmp(profile, "main444-10-intra") ||
514 !strcmp(profile, "main444-12") || !strcmp(profile, "main444-12-intra") ||
515 !strcmp(profile, "main444-16-intra") || !strcmp(profile, "main444-16-stillpicture"))
516 {
517 /* any color space allowed */
518 }
519 else
520 {
521 x265_log(param, X265_LOG_ERROR, "unknown profile <%s>\n", profile);
522 return -1;
523 }
524
525 return 0;
526}
