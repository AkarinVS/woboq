35DPB::~DPB()
36{
37 while (!m_freeList.empty())
38 {
39 Frame* curFrame = m_freeList.popFront();
40 curFrame->destroy();
41 delete curFrame;
42 }
43
44 while (!m_picList.empty())
45 {
46 Frame* curFrame = m_picList.popFront();
47 curFrame->destroy();
48 delete curFrame;
49 }
50
51 while (m_frameDataFreeList)
52 {
53 FrameData* next = m_frameDataFreeList->m_freeListNext;
54 m_frameDataFreeList->destroy();
55
56 m_frameDataFreeList->m_reconPic->destroy();
57 delete m_frameDataFreeList->m_reconPic;
58
59 delete m_frameDataFreeList;
60 m_frameDataFreeList = next;
61 }
62}

64// move unreferenced pictures from picList to freeList for recycle
65void DPB::recycleUnreferenced()
66{
67 Frame *iterFrame = m_picList.first();
68
69 while (iterFrame)
70 {
71 Frame *curFrame = iterFrame;
72 iterFrame = iterFrame->m_next;
73 if (!curFrame->m_encData->m_bHasReferences && !curFrame->m_countRefEncoders)
74 {
75 curFrame->m_bChromaExtended = false;
76
77 // Reset column counter
78 X265_CHECK(curFrame->m_reconRowFlag != NULL, "curFrame->m_reconRowFlag check failure");
79 X265_CHECK(curFrame->m_reconColCount != NULL, "curFrame->m_reconColCount check failure");
80 X265_CHECK(curFrame->m_numRows > 0, "curFrame->m_numRows check failure");
81
82 for(int32_t row = 0; row < curFrame->m_numRows; row++)
83 {
84 curFrame->m_reconRowFlag[row].set(0);
85 curFrame->m_reconColCount[row].set(0);
86 }
87
88 // iterator is invalidated by remove, restart scan
89 m_picList.remove(*curFrame);
90 iterFrame = m_picList.first();
91
92 m_freeList.pushBack(*curFrame);
93 curFrame->m_encData->m_freeListNext = m_frameDataFreeList;
94 m_frameDataFreeList = curFrame->m_encData;
95 for (int i = 0; i < INTEGRAL_PLANE_NUM; i++)
96 {
97 if (curFrame->m_encData->m_meBuffer[i] != NULL)
98 {
99 X265_FREE(curFrame->m_encData->m_meBuffer[i]);
100 curFrame->m_encData->m_meBuffer[i] = NULL;
101 }
102 }
103 if (curFrame->m_ctuInfo != NULL)
104 {
105 uint32_t widthInCU = (curFrame->m_param->sourceWidth + curFrame->m_param->maxCUSize - 1) >> curFrame->m_param->maxLog2CUSize;
106 uint32_t heightInCU = (curFrame->m_param->sourceHeight + curFrame->m_param->maxCUSize - 1) >> curFrame->m_param->maxLog2CUSize;
107 uint32_t numCUsInFrame = widthInCU * heightInCU;
108 for (uint32_t i = 0; i < numCUsInFrame; i++)
109 {
110 X265_FREE((*curFrame->m_ctuInfo + i)->ctuInfo);
111 (*curFrame->m_ctuInfo + i)->ctuInfo = NULL;
112 }
113 X265_FREE(*curFrame->m_ctuInfo);
114 *(curFrame->m_ctuInfo) = NULL;
115 X265_FREE(curFrame->m_ctuInfo);
116 curFrame->m_ctuInfo = NULL;
117 X265_FREE(curFrame->m_prevCtuInfoChange);
118 curFrame->m_prevCtuInfoChange = NULL;
119 }
120 curFrame->m_encData = NULL;
121 curFrame->m_reconPic = NULL;
122 }
123 }

126void DPB::prepareEncode(Frame *newFrame)
127{
128 Slice* slice = newFrame->m_encData->m_slice;
129 slice->m_poc = newFrame->m_poc;
130 slice->m_fieldNum = newFrame->m_fieldNum;
131
132 int pocCurr = slice->m_poc;
133 int type = newFrame->m_lowres.sliceType;
134 bool bIsKeyFrame = newFrame->m_lowres.bKeyframe;
135 slice->m_nalUnitType = getNalUnitType(pocCurr, bIsKeyFrame);
136 if (slice->m_nalUnitType == NAL_UNIT_CODED_SLICE_IDR_W_RADL || slice->m_nalUnitType == NAL_UNIT_CODED_SLICE_IDR_N_LP)
137 m_lastIDR = pocCurr;
138 slice->m_lastIDR = m_lastIDR;
139 slice->m_sliceType = IS_X265_TYPE_B(type) ? B_SLICE : (type == X265_TYPE_P) ? P_SLICE : I_SLICE;
140
141 if (type == X265_TYPE_B)
142 {
143 newFrame->m_encData->m_bHasReferences = false;
144
145 // Adjust NAL type for unreferenced B frames (change from _R "referenced"
146 // to _N "non-referenced" NAL unit type)
147 switch (slice->m_nalUnitType)
148 {
149 case NAL_UNIT_CODED_SLICE_TRAIL_R:
150 slice->m_nalUnitType = m_bTemporalSublayer ? NAL_UNIT_CODED_SLICE_TSA_N : NAL_UNIT_CODED_SLICE_TRAIL_N;
151 break;
152 case NAL_UNIT_CODED_SLICE_RADL_R:
153 slice->m_nalUnitType = NAL_UNIT_CODED_SLICE_RADL_N;
154 break;
155 case NAL_UNIT_CODED_SLICE_RASL_R:
156 slice->m_nalUnitType = NAL_UNIT_CODED_SLICE_RASL_N;
157 break;
158 default:
159 break;
160 }
161 }
162 else
163 {
164 /* m_bHasReferences starts out as true for non-B pictures, and is set to false
165 * once no more pictures reference it */
166 newFrame->m_encData->m_bHasReferences = true;
167 }
168
169 m_picList.pushFront(*newFrame);
170
171 // Do decoding refresh marking if any
172 decodingRefreshMarking(pocCurr, slice->m_nalUnitType);
173
174 computeRPS(pocCurr, slice->isIRAP(), &slice->m_rps, slice->m_sps->maxDecPicBuffering);
175
176 // Mark pictures in m_piclist as unreferenced if they are not included in RPS
177 applyReferencePictureSet(&slice->m_rps, pocCurr);
178
179 if (slice->m_sliceType != I_SLICE)
180 slice->m_numRefIdx[0] = x265_clip3(1, newFrame->m_param->maxNumReferences, slice->m_rps.numberOfNegativePictures);
181 else
182 slice->m_numRefIdx[0] = X265_MIN(newFrame->m_param->maxNumReferences, slice->m_rps.numberOfNegativePictures); // Ensuring L0 contains just the -ve POC
183 slice->m_numRefIdx[1] = X265_MIN(newFrame->m_param->bBPyramid ? 2 : 1, slice->m_rps.numberOfPositivePictures);
184 slice->setRefPicList(m_picList);
185
186 X265_CHECK(slice->m_sliceType != B_SLICE || slice->m_numRefIdx[1], "B slice without L1 references (non-fatal)\n");
187
188 if (slice->m_sliceType == B_SLICE)
189 {
190 /* TODO: the lookahead should be able to tell which reference picture
191 * had the least motion residual. We should be able to use that here to
192 * select a colocation reference list and index */
193 slice->m_colFromL0Flag = false;
194 slice->m_colRefIdx = 0;
195 slice->m_bCheckLDC = false;
196 }
197 else
198 {
199 slice->m_bCheckLDC = true;
200 slice->m_colFromL0Flag = true;
201 slice->m_colRefIdx = 0;
202 }
203
204 // Disable Loopfilter in bound area, because we will do slice-parallelism in future
205 slice->m_sLFaseFlag = (newFrame->m_param->maxSlices > 1) ? false : ((SLFASE_CONSTANT & (1 << (pocCurr % 31))) > 0);
206
207 /* Increment reference count of all motion-referenced frames to prevent them
208 * from being recycled. These counts are decremented at the end of
209 * compressFrame() */
210 int numPredDir = slice->isInterP() ? 1 : slice->isInterB() ? 2 : 0;
211 for (int l = 0; l < numPredDir; l++)
212 {
213 for (int ref = 0; ref < slice->m_numRefIdx[l]; ref++)
214 {
215 Frame *refpic = slice->m_refFrameList[l][ref];
216 ATOMIC_INC(&refpic->m_countRefEncoders);
217 }
218 }
219}

221void DPB::computeRPS(int curPoc, bool isRAP, RPS * rps, unsigned int maxDecPicBuffer)
222{
223 unsigned int poci = 0, numNeg = 0, numPos = 0;
224
225 Frame* iterPic = m_picList.first();
226
227 while (iterPic && (poci < maxDecPicBuffer - 1))
228 {
229 if ((iterPic->m_poc != curPoc) && iterPic->m_encData->m_bHasReferences)
230 {
231 if ((m_lastIDR >= curPoc) || (m_lastIDR <= iterPic->m_poc))
232 {
233 rps->poc[poci] = iterPic->m_poc;
234 rps->deltaPOC[poci] = rps->poc[poci] - curPoc;
235 (rps->deltaPOC[poci] < 0) ? numNeg++ : numPos++;
236 rps->bUsed[poci] = !isRAP;
237 poci++;
238 }
239 }
240 iterPic = iterPic->m_next;
241 }
242
243 rps->numberOfPictures = poci;
244 rps->numberOfPositivePictures = numPos;
245 rps->numberOfNegativePictures = numNeg;
246
247 rps->sortDeltaPOC();
248}

250/* Marking reference pictures when an IDR/CRA is encountered. */
251void DPB::decodingRefreshMarking(int pocCurr, NalUnitType nalUnitType)
252{
253 if (nalUnitType == NAL_UNIT_CODED_SLICE_IDR_W_RADL || nalUnitType == NAL_UNIT_CODED_SLICE_IDR_N_LP)
254 {
255 /* If the nal_unit_type is IDR, all pictures in the reference picture
256 * list are marked as "unused for reference" */
257 Frame* iterFrame = m_picList.first();
258 while (iterFrame)
259 {
260 if (iterFrame->m_poc != pocCurr)
261 iterFrame->m_encData->m_bHasReferences = false;
262 iterFrame = iterFrame->m_next;
263 }
264 }
265 else // CRA or No DR
266 {
267 if (m_bRefreshPending && pocCurr > m_pocCRA)
268 {
269 /* If the bRefreshPending flag is true (a deferred decoding refresh
270 * is pending) and the current temporal reference is greater than
271 * the temporal reference of the latest CRA picture (pocCRA), mark
272 * all reference pictures except the latest CRA picture as "unused
273 * for reference" and set the bRefreshPending flag to false */
274 Frame* iterFrame = m_picList.first();
275 while (iterFrame)
276 {
277 if (iterFrame->m_poc != pocCurr && iterFrame->m_poc != m_pocCRA)
278 iterFrame->m_encData->m_bHasReferences = false;
279 iterFrame = iterFrame->m_next;
280 }
281
282 m_bRefreshPending = false;
283 }
284 if (nalUnitType == NAL_UNIT_CODED_SLICE_CRA)
285 {
286 /* If the nal_unit_type is CRA, set the bRefreshPending flag to true
287 * and pocCRA to the temporal reference of the current picture */
288 m_bRefreshPending = true;
289 m_pocCRA = pocCurr;
290 }
291 }
292
293 /* Note that the current picture is already placed in the reference list and
294 * its marking is not changed. If the current picture has a nal_ref_idc
295 * that is not 0, it will remain marked as "used for reference" */
296}

298/** Function for applying picture marking based on the Reference Picture Set */
299void DPB::applyReferencePictureSet(RPS *rps, int curPoc)
300{
301 // loop through all pictures in the reference picture buffer
302 Frame* iterFrame = m_picList.first();
303 while (iterFrame)
304 {
305 if (iterFrame->m_poc != curPoc && iterFrame->m_encData->m_bHasReferences)
306 {
307 // loop through all pictures in the Reference Picture Set
308 // to see if the picture should be kept as reference picture
309 bool referenced = false;
310 for (int i = 0; i < rps->numberOfPositivePictures + rps->numberOfNegativePictures; i++)
311 {
312 if (iterFrame->m_poc == curPoc + rps->deltaPOC[i])
313 {
314 referenced = true;
315 break;
316 }
317 }
318 if (!referenced)
319 iterFrame->m_encData->m_bHasReferences = false;
320 }
321 iterFrame = iterFrame->m_next;
322 }

325/* deciding the nal_unit_type */
326NalUnitType DPB::getNalUnitType(int curPOC, bool bIsKeyFrame)
327{
328 if (!curPOC)
329 return NAL_UNIT_CODED_SLICE_IDR_N_LP;
330 if (bIsKeyFrame)
331 return m_bOpenGOP ? NAL_UNIT_CODED_SLICE_CRA : m_bhasLeadingPicture ? NAL_UNIT_CODED_SLICE_IDR_W_RADL : NAL_UNIT_CODED_SLICE_IDR_N_LP;
332 if (m_pocCRA && curPOC < m_pocCRA)
333 // All leading pictures are being marked as TFD pictures here since
334 // current encoder uses all reference pictures while encoding leading
335 // pictures. An encoder can ensure that a leading picture can be still
336 // decodable when random accessing to a CRA/CRANT/BLA/BLANT picture by
337 // controlling the reference pictures used for encoding that leading
338 // picture. Such a leading picture need not be marked as a TFD picture.
339 return NAL_UNIT_CODED_SLICE_RASL_R;
340
341 if (m_lastIDR && curPOC < m_lastIDR)
342 return NAL_UNIT_CODED_SLICE_RADL_R;
343
344 return NAL_UNIT_CODED_SLICE_TRAIL_R;
345}
