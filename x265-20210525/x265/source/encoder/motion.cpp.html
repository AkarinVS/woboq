86/* sum of absolute differences between MV candidates, used for adaptive ME range */
87inline int predictorDifference(const MV *mvc, intptr_t numCandidates)
88{
89 int sum = 0;
90
91 for (int i = 0; i < numCandidates - 1; i++)
92 {
93 sum += abs(mvc[i].x - mvc[i + 1].x)
94 + abs(mvc[i].y - mvc[i + 1].y);
95 }
96
97 return sum;
98}

194void MotionEstimate::setSourcePU(const Yuv& srcFencYuv, int _ctuAddr, int cuPartIdx, int puPartIdx, int pwidth, int pheight, const int method, const int refine, bool bChroma)
195{
196 partEnum = partitionFromSizes(pwidth, pheight);
197 X265_CHECK(LUMA_4x4 != partEnum, "4x4 inter partition detected!\n");
198 sad = primitives.pu[partEnum].sad;
199 ads = primitives.pu[partEnum].ads;
200 satd = primitives.pu[partEnum].satd;
201 sad_x3 = primitives.pu[partEnum].sad_x3;
202 sad_x4 = primitives.pu[partEnum].sad_x4;
203
204 chromaSatd = primitives.chroma[fencPUYuv.m_csp].pu[partEnum].satd;
205
206 /* Set search characteristics */
207 searchMethod = method;
208 subpelRefine = refine;
209
210 /* Enable chroma residual cost if subpelRefine level is greater than 2 and chroma block size
211 * is an even multiple of 4x4 pixels (indicated by non-null chromaSatd pointer) */
212 bChromaSATD = subpelRefine > 2 && chromaSatd && (srcFencYuv.m_csp != X265_CSP_I400 && bChroma);
213 X265_CHECK(!(bChromaSATD && !workload[subpelRefine].hpel_satd), "Chroma SATD cannot be used with SAD hpel\n");

215 ctuAddr = _ctuAddr;
216 absPartIdx = cuPartIdx + puPartIdx;
217 blockwidth = pwidth;
218 blockOffset = 0;
219
220 /* copy PU from CU Yuv */
221 fencPUYuv.copyPUFromYuv(srcFencYuv, puPartIdx, partEnum, bChromaSATD);
222}

362void MotionEstimate::StarPatternSearch(ReferencePlanes *ref,
363 const MV & mvmin,
364 const MV & mvmax,
365 MV & bmv,
366 int & bcost,
367 int & bPointNr,
368 int & bDistance,
369 int earlyExitIters,
370 int merange,
371 int hme)
372{
373 ALIGN_VAR_16(int, costs[16]);
374 pixel* fenc = fencPUYuv.m_buf[0];
375 pixel* fref = (hme? ref->fpelLowerResPlane[0] : ref->fpelPlane[0]) + blockOffset;
376 intptr_t stride = hme? ref->lumaStride / 2 : ref->lumaStride;
377
378 MV omv = bmv;
379 int saved = bcost;
380 int rounds = 0;
381
382 {
383 int16_t dist = 1;
384
385 /* bPointNr
386 2
387 4 * 5
388 7
389 */
390 const int32_t top = omv.y - dist;
391 const int32_t bottom = omv.y + dist;
392 const int32_t left = omv.x - dist;
393 const int32_t right = omv.x + dist;
394
395 if (top >= mvmin.y && left >= mvmin.x && right <= mvmax.x && bottom <= mvmax.y)
396 {
397 COST_MV_PT_DIST_X4(omv.x, top, 2, dist,
398 left, omv.y, 4, dist,
399 right, omv.y, 5, dist,
400 omv.x, bottom, 7, dist);
401 }
402 else
403 {
404 if (top >= mvmin.y) // check top
405 {
406 COST_MV_PT_DIST(omv.x, top, 2, dist);
407 }
408 if (left >= mvmin.x) // check middle left
409 {
410 COST_MV_PT_DIST(left, omv.y, 4, dist);
411 }
412 if (right <= mvmax.x) // check middle right
413 {
414 COST_MV_PT_DIST(right, omv.y, 5, dist);
415 }
416 if (bottom <= mvmax.y) // check bottom
417 {
418 COST_MV_PT_DIST(omv.x, bottom, 7, dist);
419 }
420 }
421 if (bcost < saved)
422 rounds = 0;
423 else if (++rounds >= earlyExitIters)
424 return;
425 }
426
427 for (int16_t dist = 2; dist <= 8; dist <<= 1)
428 {
429 /* bPointNr
430 2
431 1 3
432 4 * 5
433 6 8
434 7
435 Points 2, 4, 5, 7 are dist
436 Points 1, 3, 6, 8 are dist>>1
437 */
438 const int32_t top = omv.y - dist;
439 const int32_t bottom = omv.y + dist;
440 const int32_t left = omv.x - dist;
441 const int32_t right = omv.x + dist;
442 const int32_t top2 = omv.y - (dist >> 1);
443 const int32_t bottom2 = omv.y + (dist >> 1);
444 const int32_t left2 = omv.x - (dist >> 1);
445 const int32_t right2 = omv.x + (dist >> 1);
446 saved = bcost;
447
448 if (top >= mvmin.y && left >= mvmin.x &&
449 right <= mvmax.x && bottom <= mvmax.y) // check border
450 {
451 COST_MV_PT_DIST_X4(omv.x, top, 2, dist,
452 left2, top2, 1, dist >> 1,
453 right2, top2, 3, dist >> 1,
454 left, omv.y, 4, dist);
455 COST_MV_PT_DIST_X4(right, omv.y, 5, dist,
456 left2, bottom2, 6, dist >> 1,
457 right2, bottom2, 8, dist >> 1,
458 omv.x, bottom, 7, dist);
459 }
460 else // check border for each mv
461 {
462 if (top >= mvmin.y) // check top
463 {
464 COST_MV_PT_DIST(omv.x, top, 2, dist);
465 }
466 if (top2 >= mvmin.y) // check half top
467 {
468 if (left2 >= mvmin.x) // check half left
469 {
470 COST_MV_PT_DIST(left2, top2, 1, (dist >> 1));
471 }
472 if (right2 <= mvmax.x) // check half right
473 {
474 COST_MV_PT_DIST(right2, top2, 3, (dist >> 1));
475 }
476 }
477 if (left >= mvmin.x) // check left
478 {
479 COST_MV_PT_DIST(left, omv.y, 4, dist);
480 }
481 if (right <= mvmax.x) // check right
482 {
483 COST_MV_PT_DIST(right, omv.y, 5, dist);
484 }
485 if (bottom2 <= mvmax.y) // check half bottom
486 {
487 if (left2 >= mvmin.x) // check half left
488 {
489 COST_MV_PT_DIST(left2, bottom2, 6, (dist >> 1));
490 }
491 if (right2 <= mvmax.x) // check half right
492 {
493 COST_MV_PT_DIST(right2, bottom2, 8, (dist >> 1));
494 }
495 }
496 if (bottom <= mvmax.y) // check bottom
497 {
498 COST_MV_PT_DIST(omv.x, bottom, 7, dist);
499 }
500 }
501
502 if (bcost < saved)
503 rounds = 0;
504 else if (++rounds >= earlyExitIters)
505 return;
506 }
507
508 for (int16_t dist = 16; dist <= (int16_t)merange; dist <<= 1)
509 {
510 const int32_t top = omv.y - dist;
511 const int32_t bottom = omv.y + dist;
512 const int32_t left = omv.x - dist;
513 const int32_t right = omv.x + dist;
514
515 saved = bcost;
516 if (top >= mvmin.y && left >= mvmin.x &&
517 right <= mvmax.x && bottom <= mvmax.y) // check border
518 {
519 /* index
520 0
521 3
522 2
523 1
524 0 3 2 1 * 1 2 3 0
525 1
526 2
527 3
528 0
529 */
530
531 COST_MV_PT_DIST_X4(omv.x, top, 0, dist,
532 left, omv.y, 0, dist,
533 right, omv.y, 0, dist,
534 omv.x, bottom, 0, dist);
535
536 for (int16_t index = 1; index < 4; index++)
537 {
538 int32_t posYT = top + ((dist >> 2) * index);
539 int32_t posYB = bottom - ((dist >> 2) * index);
540 int32_t posXL = omv.x - ((dist >> 2) * index);
541 int32_t posXR = omv.x + ((dist >> 2) * index);
542
543 COST_MV_PT_DIST_X4(posXL, posYT, 0, dist,
544 posXR, posYT, 0, dist,
545 posXL, posYB, 0, dist,
546 posXR, posYB, 0, dist);
547 }
548 }
549 else // check border for each mv
550 {
551 if (top >= mvmin.y) // check top
552 {
553 COST_MV_PT_DIST(omv.x, top, 0, dist);
554 }
555 if (left >= mvmin.x) // check left
556 {
557 COST_MV_PT_DIST(left, omv.y, 0, dist);
558 }
559 if (right <= mvmax.x) // check right
560 {
561 COST_MV_PT_DIST(right, omv.y, 0, dist);
562 }
563 if (bottom <= mvmax.y) // check bottom
564 {
565 COST_MV_PT_DIST(omv.x, bottom, 0, dist);
566 }
567 for (int16_t index = 1; index < 4; index++)
568 {
569 int32_t posYT = top + ((dist >> 2) * index);
570 int32_t posYB = bottom - ((dist >> 2) * index);
571 int32_t posXL = omv.x - ((dist >> 2) * index);
572 int32_t posXR = omv.x + ((dist >> 2) * index);
573
574 if (posYT >= mvmin.y) // check top
575 {
576 if (posXL >= mvmin.x) // check left
577 {
578 COST_MV_PT_DIST(posXL, posYT, 0, dist);
579 }
580 if (posXR <= mvmax.x) // check right
581 {
582 COST_MV_PT_DIST(posXR, posYT, 0, dist);
583 }
584 }
585 if (posYB <= mvmax.y) // check bottom
586 {
587 if (posXL >= mvmin.x) // check left
588 {
589 COST_MV_PT_DIST(posXL, posYB, 0, dist);
590 }
591 if (posXR <= mvmax.x) // check right
592 {
593 COST_MV_PT_DIST(posXR, posYB, 0, dist);
594 }
595 }
596 }
597 }
598
599 if (bcost < saved)
600 rounds = 0;
601 else if (++rounds >= earlyExitIters)
602 return;
603 }

606void MotionEstimate::refineMV(ReferencePlanes* ref,
607 const MV& mvmin,
608 const MV& mvmax,
609 const MV& qmvp,
610 MV& outQMv)
611{
612 ALIGN_VAR_16(int, costs[16]);
613 if (ctuAddr >= 0)
614 blockOffset = ref->reconPic->getLumaAddr(ctuAddr, absPartIdx) - ref->reconPic->getLumaAddr(0);
615 intptr_t stride = ref->lumaStride;
616 pixel* fenc = fencPUYuv.m_buf[0];
617 pixel* fref = ref->fpelPlane[0] + blockOffset;
618
619 setMVP(qmvp);
620
621 MV qmvmin = mvmin.toQPel();
622 MV qmvmax = mvmax.toQPel();
623
624 /* The term cost used here means satd/sad values for that particular search.
625 * The costs used in ME integer search only includes the SAD cost of motion
626 * residual and sqrtLambda times MVD bits. The subpel refine steps use SATD
627 * cost of residual and sqrtLambda * MVD bits.
628 */
629
630 // measure SATD cost at clipped QPEL MVP
631 MV pmv = qmvp.clipped(qmvmin, qmvmax);
632 MV bestpre = pmv;
633 int bprecost;
634
635 bprecost = subpelCompare(ref, pmv, sad);
636
637 /* re-measure full pel rounded MVP with SAD as search start point */
638 MV bmv = pmv.roundToFPel();
639 int bcost = bprecost;
640 if (pmv.isSubpel())
641 bcost = sad(fenc, FENC_STRIDE, fref + bmv.x + bmv.y * stride, stride) + mvcost(bmv << 2);
642
643 /* square refine */
644 int dir = 0;
645 COST_MV_X4_DIR(0, -1, 0, 1, -1, 0, 1, 0, costs);
646 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
647 COPY2_IF_LT(bcost, costs[0], dir, 1);
648 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
649 COPY2_IF_LT(bcost, costs[1], dir, 2);
650 COPY2_IF_LT(bcost, costs[2], dir, 3);
651 COPY2_IF_LT(bcost, costs[3], dir, 4);
652 COST_MV_X4_DIR(-1, -1, -1, 1, 1, -1, 1, 1, costs);
653 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
654 COPY2_IF_LT(bcost, costs[0], dir, 5);
655 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
656 COPY2_IF_LT(bcost, costs[1], dir, 6);
657 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
658 COPY2_IF_LT(bcost, costs[2], dir, 7);
659 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
660 COPY2_IF_LT(bcost, costs[3], dir, 8);
661 bmv += square1[dir];
662
663 if (bprecost < bcost)
664 {
665 bmv = bestpre;
666 bcost = bprecost;
667 }
668 else
669 bmv = bmv.toQPel(); // promote search bmv to qpel
670
671 // TO DO: Change SubpelWorkload to fine tune MV
672 // Now it is set to 5 for experiment.
673 // const SubpelWorkload& wl = workload[this->subpelRefine];
674 const SubpelWorkload& wl = workload[5];
675
676 pixelcmp_t hpelcomp;
677
678 if (wl.hpel_satd)
679 {
680 bcost = subpelCompare(ref, bmv, satd) + mvcost(bmv);
681 hpelcomp = satd;
682 }
683 else
684 hpelcomp = sad;
685
686 for (int iter = 0; iter < wl.hpel_iters; iter++)
687 {
688 int bdir = 0;
689 for (int i = 1; i <= wl.hpel_dirs; i++)
690 {
691 MV qmv = bmv + square1[i] * 2;
692
693 // check mv range for slice bound
694 if ((qmv.y < qmvmin.y) | (qmv.y > qmvmax.y))
695 continue;
696
697 int cost = subpelCompare(ref, qmv, hpelcomp) + mvcost(qmv);
698 COPY2_IF_LT(bcost, cost, bdir, i);
699 }
700
701 if (bdir)
702 bmv += square1[bdir] * 2;
703 else
704 break;
705 }
706
707 /* if HPEL search used SAD, remeasure with SATD before QPEL */
708 if (!wl.hpel_satd)
709 bcost = subpelCompare(ref, bmv, satd) + mvcost(bmv);
710
711 for (int iter = 0; iter < wl.qpel_iters; iter++)
712 {
713 int bdir = 0;
714 for (int i = 1; i <= wl.qpel_dirs; i++)
715 {
716 MV qmv = bmv + square1[i];
717
718 // check mv range for slice bound
719 if ((qmv.y < qmvmin.y) | (qmv.y > qmvmax.y))
720 continue;
721
722 int cost = subpelCompare(ref, qmv, satd) + mvcost(qmv);
723 COPY2_IF_LT(bcost, cost, bdir, i);
724 }
725
726 if (bdir)
727 bmv += square1[bdir];
728 else
729 break;
730 }
731
732 // check mv range for slice bound
733 X265_CHECK(((pmv.y >= qmvmin.y) & (pmv.y <= qmvmax.y)), "mv beyond range!");
734
735 x265_emms();
736 outQMv = bmv;
737}

739int MotionEstimate::motionEstimate(ReferencePlanes *ref,
740 const MV & mvmin,
741 const MV & mvmax,
742 const MV & qmvp,
743 int numCandidates,
744 const MV * mvc,
745 int merange,
746 MV & outQMv,
747 uint32_t maxSlices,
748 pixel * srcReferencePlane)
749{
750 ALIGN_VAR_16(int, costs[16]);
751 bool hme = srcReferencePlane && srcReferencePlane == ref->fpelLowerResPlane[0];
752 if (ctuAddr >= 0)
753 blockOffset = ref->reconPic->getLumaAddr(ctuAddr, absPartIdx) - ref->reconPic->getLumaAddr(0);
754 intptr_t stride = hme ? ref->lumaStride / 2 : ref->lumaStride;
755 pixel* fenc = fencPUYuv.m_buf[0];
756 pixel* fref = srcReferencePlane == 0 ? ref->fpelPlane[0] + blockOffset : srcReferencePlane + blockOffset;
757
758 setMVP(qmvp);
759
760 MV qmvmin = mvmin.toQPel();
761 MV qmvmax = mvmax.toQPel();
762
763 /* The term cost used here means satd/sad values for that particular search.
764 * The costs used in ME integer search only includes the SAD cost of motion
765 * residual and sqrtLambda times MVD bits. The subpel refine steps use SATD
766 * cost of residual and sqrtLambda * MVD bits. Mode decision will be based
767 * on video distortion cost (SSE/PSNR) plus lambda times all signaling bits
768 * (mode + MVD bits). */
769
770 // measure SAD cost at clipped QPEL MVP
771 MV pmv = qmvp.clipped(qmvmin, qmvmax);
772 MV bestpre = pmv;
773 int bprecost;
774
775 if (ref->isLowres)
776 bprecost = ref->lowresQPelCost(fenc, blockOffset, pmv, sad, hme);
777 else
778 bprecost = subpelCompare(ref, pmv, sad);
779
780 /* re-measure full pel rounded MVP with SAD as search start point */
781 MV bmv = pmv.roundToFPel();
782 int bcost = bprecost;
783 if (pmv.isSubpel())
784 bcost = sad(fenc, FENC_STRIDE, fref + bmv.x + bmv.y * stride, stride) + mvcost(bmv << 2);
785
786 // measure SAD cost at MV(0) if MVP is not zero
787 if (pmv.notZero())
788 {
789 int cost = sad(fenc, FENC_STRIDE, fref, stride) + mvcost(MV(0, 0));
790 if (cost < bcost)
791 {
792 bcost = cost;
793 bmv = 0;
794 bmv.y = X265_MAX(X265_MIN(0, mvmax.y), mvmin.y);
795 }
796 }
797
798 X265_CHECK(!(ref->isLowres && numCandidates), "lowres motion candidates not allowed\n")
799 // measure SAD cost at each QPEL motion vector candidate
800 for (int i = 0; i < numCandidates; i++)
801 {
802 MV m = mvc[i].clipped(qmvmin, qmvmax);
803 if (m.notZero() & (m != pmv ? 1 : 0) & (m != bestpre ? 1 : 0)) // check already measured
804 {
805 int cost = subpelCompare(ref, m, sad) + mvcost(m);
806 if (cost < bprecost)
807 {
808 bprecost = cost;
809 bestpre = m;
810 }
811 }
812 }
813
814 pmv = pmv.roundToFPel();
815 MV omv = bmv; // current search origin or starting point
816
817 int search = ref->isHMELowres ? (hme ? searchMethodL0 : searchMethodL1) : searchMethod;
818 switch (search)
819 {
820 case X265_DIA_SEARCH:
821 {
822 /* diamond search, radius 1 */
823 bcost <<= 4;
824 int i = merange;
825 do
826 {
827 COST_MV_X4_DIR(0, -1, 0, 1, -1, 0, 1, 0, costs);
828 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
829 COPY1_IF_LT(bcost, (costs[0] << 4) + 1);
830 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
831 COPY1_IF_LT(bcost, (costs[1] << 4) + 3);
832 COPY1_IF_LT(bcost, (costs[2] << 4) + 4);
833 COPY1_IF_LT(bcost, (costs[3] << 4) + 12);
834 if (!(bcost & 15))
835 break;
836 bmv.x -= (bcost << 28) >> 30;
837 bmv.y -= (bcost << 30) >> 30;
838 bcost &= ~15;
839 }
840 while (--i && bmv.checkRange(mvmin, mvmax));
841 bcost >>= 4;
842 break;
843 }
844
845 case X265_HEX_SEARCH:
846 {
847me_hex2:
848 /* hexagon search, radius 2 */
849#if 0
850 for (int i = 0; i < merange / 2; i++)
851 {

854 COST_MV(omv.x - 1, omv.y + 2);

871 if ((bmv.y + 2 >= mvmin.y) & (bmv.y + 2 <= mvmax.y))

873 COPY1_IF_LT(bcost, (costs[1] << 3) + 3);
874 COPY1_IF_LT(bcost, (costs[2] << 3) + 4);

877 COST_MV_X3_DIR(2, 0, 1, -2, -1, -2, costs);
878 if ((bmv.y >= mvmin.y) & (bmv.y <= mvmax.y))
879 COPY1_IF_LT(bcost, (costs[0] << 3) + 5);
880 if ((bmv.y - 2 >= mvmin.y) & (bmv.y - 2 <= mvmax.y))
881 {
882 COPY1_IF_LT(bcost, (costs[1] << 3) + 6);
883 COPY1_IF_LT(bcost, (costs[2] << 3) + 7);
884 }

895 for (int i = (merange >> 1) - 1; i > 0 && bmv.checkRange(mvmin, mvmax); i--)
896 {
897 COST_MV_X3_DIR(hex2[dir + 0].x, hex2[dir + 0].y,
898 hex2[dir + 1].x, hex2[dir + 1].y,
899 hex2[dir + 2].x, hex2[dir + 2].y,
900 costs);
901 bcost &= ~7;
902
903 if ((bmv.y + hex2[dir + 0].y >= mvmin.y) & (bmv.y + hex2[dir + 0].y <= mvmax.y))
904 COPY1_IF_LT(bcost, (costs[0] << 3) + 1);
905
906 if ((bmv.y + hex2[dir + 1].y >= mvmin.y) & (bmv.y + hex2[dir + 1].y <= mvmax.y))
907 COPY1_IF_LT(bcost, (costs[1] << 3) + 2);
908
909 if ((bmv.y + hex2[dir + 2].y >= mvmin.y) & (bmv.y + hex2[dir + 2].y <= mvmax.y))
910 COPY1_IF_LT(bcost, (costs[2] << 3) + 3);
911
912 if (!(bcost & 7))
913 break;
914
915 dir += (bcost & 7) - 2;
916 dir = mod6m1[dir + 1];
917 bmv += hex2[dir + 1];
918 }
919 } // if ((bmv.y + hex2[dir + 1].y >= mvmin.y) & (bmv.y + hex2[dir + 1].y <= mvmax.y))
920 }
921 bcost >>= 3;
922#endif // if 0

924 /* square refine */
925 int dir = 0;
926 COST_MV_X4_DIR(0, -1, 0, 1, -1, 0, 1, 0, costs);
927 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
928 COPY2_IF_LT(bcost, costs[0], dir, 1);
929 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
930 COPY2_IF_LT(bcost, costs[1], dir, 2);
931 COPY2_IF_LT(bcost, costs[2], dir, 3);
932 COPY2_IF_LT(bcost, costs[3], dir, 4);
933 COST_MV_X4_DIR(-1, -1, -1, 1, 1, -1, 1, 1, costs);
934 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
935 COPY2_IF_LT(bcost, costs[0], dir, 5);
936 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
937 COPY2_IF_LT(bcost, costs[1], dir, 6);
938 if ((bmv.y - 1 >= mvmin.y) & (bmv.y - 1 <= mvmax.y))
939 COPY2_IF_LT(bcost, costs[2], dir, 7);
940 if ((bmv.y + 1 >= mvmin.y) & (bmv.y + 1 <= mvmax.y))
941 COPY2_IF_LT(bcost, costs[3], dir, 8);
942 bmv += square1[dir];
943 break;
944 }

946 case X265_UMH_SEARCH:
947 {
948 int ucost1, ucost2;
949 int16_t cross_start = 1;

951 /* refine predictors */
952 omv = bmv;
953 ucost1 = bcost;
954 X265_CHECK(((pmv.y >= mvmin.y) & (pmv.y <= mvmax.y)), "pmv outside of search range!");
955 DIA1_ITER(pmv.x, pmv.y);
956 if (pmv.notZero())
957 DIA1_ITER(0, 0);
958
959 ucost2 = bcost;
960 if (bmv.notZero() && bmv != pmv)
961 DIA1_ITER(bmv.x, bmv.y);
962 if (bcost == ucost2)
963 cross_start = 3;
964
965 /* Early Termination */
966 omv = bmv;
967 if (bcost == ucost2 && SAD_THRESH(2000))
968 {
969 COST_MV_X4(0, -2, -1, -1, 1, -1, -2, 0);
970 COST_MV_X4(2, 0, -1, 1, 1, 1, 0, 2);
971 if (bcost == ucost1 && SAD_THRESH(500))
972 break;
973 if (bcost == ucost2)
974 {
975 int16_t range = (int16_t)(merange >> 1) | 1;
976 CROSS(3, range, range);
977 COST_MV_X4(-1, -2, 1, -2, -2, -1, 2, -1);
978 COST_MV_X4(-2, 1, 2, 1, -1, 2, 1, 2);
979 if (bcost == ucost2)
980 break;
981 cross_start = range + 2;

986 // have special cases here for 16x16, and whether they apply to HEVC CTU

989 if (numCandidates)

992 * average distance between current predictor and final mv found by ESA.
993 * these have not been tuned much by actual encoding. */

995 {
996 { 3, 3, 4, 4 },
997 { 3, 4, 4, 4 },
998 { 4, 4, 4, 5 },
999 { 4, 4, 5, 6 },

1007 {

1010 * but prediction usually isn't too bad, so just use medium range */

1012 else

1014 }

1020 * so don't count mvp separately. */
1021
1022 denom = numCandidates - 1;
1023 mvd = 0;
1024 if (partEnum != LUMA_64x64)
1025 {
1026 mvd = abs(qmvp.x - mvc[0].x) + abs(qmvp.y - mvc[0].y);
1027 denom++;
1028 }
1029 mvd += predictorDifference(mvc, numCandidates);
1030 }
1031
1032 sad_ctx = SAD_THRESH(1000) ? 0
1033 : SAD_THRESH(2000) ? 1
1034 : SAD_THRESH(4000) ? 2 : 3;
1035 mvd_ctx = mvd < 10 * denom ? 0
1036 : mvd < 20 * denom ? 1
1037 : mvd < 40 * denom ? 2 : 3;
1038
1039 merange = (merange * range_mul[mvd_ctx][sad_ctx]) >> 2;
1040 }
1041
1042 /* FIXME if the above DIA2/OCT2/CROSS found a new mv, it has not updated omx/omy.
1043 * we are still centered on the same place as the DIA2. is this desirable? */
1044 CROSS(cross_start, merange, merange >> 1);
1045 COST_MV_X4(-2, -2, -2, 2, 2, -2, 2, 2);
1046
1047 /* hexagon grid */
1048 omv = bmv;
1049 const uint16_t *p_cost_omvx = m_cost_mvx + omv.x * 4;
1050 const uint16_t *p_cost_omvy = m_cost_mvy + omv.y * 4;
1051 uint16_t i = 1;
1052 do
1053 {
1054 if (4 * i > X265_MIN4(mvmax.x - omv.x, omv.x - mvmin.x,
1055 mvmax.y - omv.y, omv.y - mvmin.y))
1056 {
1057 for (int j = 0; j < 16; j++)
1058 {
1059 MV mv = omv + (hex4[j] * i);
1060 if (mv.checkRange(mvmin, mvmax))
1061 COST_MV(mv.x, mv.y);
1062 }
1063 }
1064 else
1065 {
1066 int16_t dir = 0;
1067 pixel *fref_base = fref + omv.x + (omv.y - 4 * i) * stride;
1068 size_t dy = (size_t)i * stride;
1069#define SADS(k, x0, y0, x1, y1, x2, y2, x3, y3) \
1070 sad_x4(fenc, \
1071 fref_base x0 * i + (y0 - 2 * k + 4) * dy, \

1073 fref_base x2 * i + (y2 - 2 * k + 4) * dy, \
1074 fref_base x3 * i + (y3 - 2 * k + 4) * dy, \

1076 fref_base += 2 * dy;

1078#define MIN_MV(k, dx, dy) if ((omv.y + (dy) >= mvmin.y) & (omv.y + (dy) <= mvmax.y)) { COPY2_IF_LT(bcost, costs[k], dir, dx * 16 + (dy & 15)) }
