39 static void integral_init4h_c(uint32_t *sum, pixel *pix, intptr_t stride)
40 {
41 int32_t v = pix[0] + pix[1] + pix[2] + pix[3];
42 for (int16_t x = 0; x < stride - 4; x++)
43 {
44 sum[x] = v + sum[x - stride];
45 v += pix[x + 4] - pix[x];
46 }
47 }
48
49 static void integral_init8h_c(uint32_t *sum, pixel *pix, intptr_t stride)
50 {
51 int32_t v = pix[0] + pix[1] + pix[2] + pix[3] + pix[4] + pix[5] + pix[6] + pix[7];
52 for (int16_t x = 0; x < stride - 8; x++)
53 {
54 sum[x] = v + sum[x - stride];
55 v += pix[x + 8] - pix[x];
56 }
57 }
58
59 static void integral_init12h_c(uint32_t *sum, pixel *pix, intptr_t stride)
60 {
61 int32_t v = pix[0] + pix[1] + pix[2] + pix[3] + pix[4] + pix[5] + pix[6] + pix[7] +
62 pix[8] + pix[9] + pix[10] + pix[11];
63 for (int16_t x = 0; x < stride - 12; x++)
64 {
65 sum[x] = v + sum[x - stride];
66 v += pix[x + 12] - pix[x];
67 }
68 }

72 int32_t v = pix[0] + pix[1] + pix[2] + pix[3] + pix[4] + pix[5] + pix[6] + pix[7] +
73 pix[8] + pix[9] + pix[10] + pix[11] + pix[12] + pix[13] + pix[14] + pix[15];

76 sum[x] = v + sum[x - stride];

78 }

85 pix[16] + pix[17] + pix[18] + pix[19] + pix[20] + pix[21] + pix[22] + pix[23];

97 pix[16] + pix[17] + pix[18] + pix[19] + pix[20] + pix[21] + pix[22] + pix[23] +
98 pix[24] + pix[25] + pix[26] + pix[27] + pix[28] + pix[29] + pix[30] + pix[31];
99 for (int16_t x = 0; x < stride - 32; x++)
100 {
101 sum[x] = v + sum[x - stride];
102 v += pix[x + 32] - pix[x];
103 }
104 }

159void FrameFilter::destroy()
160{
161 X265_FREE(m_ssimBuf);
162
163 if (m_parallelFilter)
164 {
165 if (m_useSao)
166 {
167 for(int row = 0; row < m_numRows; row++)
168 m_parallelFilter[row].m_sao.destroy((row == 0 ? 1 : 0));
169 }
170
171 delete[] m_parallelFilter;
172 m_parallelFilter = NULL;
173 }

176void FrameFilter::init(Encoder *top, FrameEncoder *frame, int numRows, uint32_t numCols)
177{
178 m_param = frame->m_param;
179 m_frameEncoder = frame;
180 m_useSao = 1;
181 m_numRows = numRows;
182 m_numCols = numCols;
183 m_hChromaShift = CHROMA_H_SHIFT(m_param->internalCsp);
184 m_vChromaShift = CHROMA_V_SHIFT(m_param->internalCsp);
185 m_pad[0] = top->m_sps.conformanceWindow.rightOffset;
186 m_pad[1] = top->m_sps.conformanceWindow.bottomOffset;
187 m_saoRowDelay = m_param->bEnableLoopFilter ? 1 : 0;
188 m_lastHeight = (m_param->sourceHeight % m_param->maxCUSize) ? (m_param->sourceHeight % m_param->maxCUSize) : m_param->maxCUSize;
189 m_lastWidth = (m_param->sourceWidth % m_param->maxCUSize) ? (m_param->sourceWidth % m_param->maxCUSize) : m_param->maxCUSize;
190 integralCompleted.set(0);
191
192 if (m_param->bEnableSsim)
193 m_ssimBuf = X265_MALLOC(int, 8 * (m_param->sourceWidth / 4 + 3));
194
195 m_parallelFilter = new ParallelFilter[numRows];
196
197 if (m_parallelFilter)
198 {
199 if (m_useSao)
200 {
201 for(int row = 0; row < numRows; row++)
202 {
203 if (!m_parallelFilter[row].m_sao.create(m_param, (row == 0 ? 1 : 0)))
204 m_useSao = 0;
205 else
206 {
207 if (row != 0)
208 m_parallelFilter[row].m_sao.createFromRootNode(&m_parallelFilter[0].m_sao);
209 }
210
211 }
212 }
213
214 for(int row = 0; row < numRows; row++)
215 {
216 // Setting maximum bound information
217 m_parallelFilter[row].m_rowHeight = (row == numRows - 1) ? m_lastHeight : m_param->maxCUSize;
218 m_parallelFilter[row].m_row = row;
219 m_parallelFilter[row].m_rowAddr = row * numCols;
220 m_parallelFilter[row].m_frameFilter = this;
221
222 if (row > 0)
223 m_parallelFilter[row].m_prevRow = &m_parallelFilter[row - 1];
224 }
225 }

229void FrameFilter::start(Frame *frame, Entropy& initState)
230{
231 m_frame = frame;
232
233 // Reset Filter Data Struct
234 if (m_parallelFilter)
235 {
236 for(int row = 0; row < m_numRows; row++)
237 {
238 if (m_useSao)
239 m_parallelFilter[row].m_sao.startSlice(frame, initState);
240
241 m_parallelFilter[row].m_lastCol.set(0);
242 m_parallelFilter[row].m_allowedCol.set(0);
243 m_parallelFilter[row].m_lastDeblocked.set(-1);
244 m_parallelFilter[row].m_encData = frame->m_encData;
245 }
246
247 // Reset SAO common statistics
248 if (m_useSao)
249 m_parallelFilter[0].m_sao.resetStats();
250 }

253/* restore original YUV samples to recon after SAO (if lossless) */
254static void restoreOrigLosslessYuv(const CUData* cu, Frame& frame, uint32_t absPartIdx)
255{
256 const int size = cu->m_log2CUSize[absPartIdx] - 2;
257 const uint32_t cuAddr = cu->m_cuAddr;
258
259 PicYuv* reconPic = frame.m_reconPic;
260 PicYuv* fencPic = frame.m_fencPic;
261
262 pixel* dst = reconPic->getLumaAddr(cuAddr, absPartIdx);
263 pixel* src = fencPic->getLumaAddr(cuAddr, absPartIdx);
264
265 primitives.cu[size].copy_pp(dst, reconPic->m_stride, src, fencPic->m_stride);
266
267 if (cu->m_chromaFormat != X265_CSP_I400)
268 {
269 pixel* dstCb = reconPic->getCbAddr(cuAddr, absPartIdx);
270 pixel* srcCb = fencPic->getCbAddr(cuAddr, absPartIdx);
271 pixel* dstCr = reconPic->getCrAddr(cuAddr, absPartIdx);
272 pixel* srcCr = fencPic->getCrAddr(cuAddr, absPartIdx);
273
274 const int csp = fencPic->m_picCsp;
275 primitives.chroma[csp].cu[size].copy_pp(dstCb, reconPic->m_strideC, srcCb, fencPic->m_strideC);
276 primitives.chroma[csp].cu[size].copy_pp(dstCr, reconPic->m_strideC, srcCr, fencPic->m_strideC);
277 }
278}

280/* Original YUV restoration for CU in lossless coding */
281static void origCUSampleRestoration(const CUData* cu, const CUGeom& cuGeom, Frame& frame)
282{
283 uint32_t absPartIdx = cuGeom.absPartIdx;
284 if (cu->m_cuDepth[absPartIdx] > cuGeom.depth)
285 {
286 for (int subPartIdx = 0; subPartIdx < 4; subPartIdx++)
287 {
288 const CUGeom& childGeom = *(&cuGeom + cuGeom.childOffset + subPartIdx);
289 if (childGeom.flags & CUGeom::PRESENT)
290 origCUSampleRestoration(cu, childGeom, frame);
291 }
292 return;
293 }
294
295 // restore original YUV samples
296 if (cu->m_tqBypass[absPartIdx])
297 restoreOrigLosslessYuv(cu, frame, absPartIdx);
298}

300void FrameFilter::ParallelFilter::copySaoAboveRef(const CUData *ctu, PicYuv* reconPic, uint32_t cuAddr, int col)
301{
302 // Copy SAO Top Reference Pixels
303 int ctuWidth = ctu->m_encData->m_param->maxCUSize;
304 const pixel* recY = reconPic->getPlaneAddr(0, cuAddr) - (ctu->m_bFirstRowInSlice ? 0 : reconPic->m_stride);
305
306 // Luma
307 memcpy(&m_sao.m_tmpU[0][col * ctuWidth], recY, ctuWidth * sizeof(pixel));
308 X265_CHECK(col * ctuWidth + ctuWidth <= m_sao.m_numCuInWidth * ctuWidth, "m_tmpU buffer beyond bound write detected");
309
310 // Chroma
311 if (m_frameFilter->m_param->internalCsp != X265_CSP_I400)
312 {
313 ctuWidth >>= m_sao.m_hChromaShift;
314
315 const pixel* recU = reconPic->getPlaneAddr(1, cuAddr) - (ctu->m_bFirstRowInSlice ? 0 : reconPic->m_strideC);
316 const pixel* recV = reconPic->getPlaneAddr(2, cuAddr) - (ctu->m_bFirstRowInSlice ? 0 : reconPic->m_strideC);
317 memcpy(&m_sao.m_tmpU[1][col * ctuWidth], recU, ctuWidth * sizeof(pixel));
318 memcpy(&m_sao.m_tmpU[2][col * ctuWidth], recV, ctuWidth * sizeof(pixel));
319
320 X265_CHECK(col * ctuWidth + ctuWidth <= m_sao.m_numCuInWidth * ctuWidth, "m_tmpU buffer beyond bound write detected");
321 }
322}

342 }

345// NOTE: MUST BE delay a row when Deblock enabled, the Deblock will modify above pixels in Horizon pass
346void FrameFilter::ParallelFilter::processPostCu(int col) const
347{
348 // Update finished CU cursor
349 m_frameFilter->m_frame->m_reconColCount[m_row].set(col);
350
351 // shortcut path for non-border area
352 if ((col != 0) & (col != m_frameFilter->m_numCols - 1) & (m_row != 0) & (m_row != m_frameFilter->m_numRows - 1))
353 return;
354
355 PicYuv *reconPic = m_frameFilter->m_frame->m_reconPic;
356 const uint32_t lineStartCUAddr = m_rowAddr + col;
357 const int realH = getCUHeight();
358 const int realW = m_frameFilter->getCUWidth(col);
359
360 const uint32_t lumaMarginX = reconPic->m_lumaMarginX;
361 const uint32_t lumaMarginY = reconPic->m_lumaMarginY;
362 const uint32_t chromaMarginX = reconPic->m_chromaMarginX;
363 const uint32_t chromaMarginY = reconPic->m_chromaMarginY;
364 const int hChromaShift = reconPic->m_hChromaShift;
365 const int vChromaShift = reconPic->m_vChromaShift;
366 const intptr_t stride = reconPic->m_stride;
367 const intptr_t strideC = reconPic->m_strideC;
368 pixel *pixY = reconPic->getLumaAddr(lineStartCUAddr);
369 // // MUST BE check I400 since m_picOrg uninitialize in that case
370 pixel *pixU = (m_frameFilter->m_param->internalCsp != X265_CSP_I400) ? reconPic->getCbAddr(lineStartCUAddr) : NULL;
371 pixel *pixV = (m_frameFilter->m_param->internalCsp != X265_CSP_I400) ? reconPic->getCrAddr(lineStartCUAddr) : NULL;
372 int copySizeY = realW;
373 int copySizeC = (realW >> hChromaShift);
374
375 if ((col == 0) | (col == m_frameFilter->m_numCols - 1))
376 {
377 // TODO: improve by process on Left or Right only
378 primitives.extendRowBorder(reconPic->getLumaAddr(m_rowAddr), stride, reconPic->m_picWidth, realH, reconPic->m_lumaMarginX);
379
380 if (m_frameFilter->m_param->internalCsp != X265_CSP_I400)
381 {
382 primitives.extendRowBorder(reconPic->getCbAddr(m_rowAddr), strideC, reconPic->m_picWidth >> hChromaShift, realH >> vChromaShift, reconPic->m_chromaMarginX);
383 primitives.extendRowBorder(reconPic->getCrAddr(m_rowAddr), strideC, reconPic->m_picWidth >> hChromaShift, realH >> vChromaShift, reconPic->m_chromaMarginX);
384 }
385 }
386
387 // Extra Left and Right border on first and last CU
388 if ((col == 0) | (col == m_frameFilter->m_numCols - 1))
389 {
390 copySizeY += lumaMarginX;
391 copySizeC += chromaMarginX;
392 }
393
394 // First column need extension left padding area and first CU
395 if (col == 0)
396 {
397 pixY -= lumaMarginX;
398 pixU -= chromaMarginX;
399 pixV -= chromaMarginX;
400 }
401
402 // Border extend Top
403 if (m_row == 0)
404 {
405 for (uint32_t y = 0; y < lumaMarginY; y++)
406 memcpy(pixY - (y + 1) * stride, pixY, copySizeY * sizeof(pixel));
407
408 if (m_frameFilter->m_param->internalCsp != X265_CSP_I400)
409 {
410 for (uint32_t y = 0; y < chromaMarginY; y++)
411 {
412 memcpy(pixU - (y + 1) * strideC, pixU, copySizeC * sizeof(pixel));
413 memcpy(pixV - (y + 1) * strideC, pixV, copySizeC * sizeof(pixel));
414 }
415 }
416 }
417
418 // Border extend Bottom
419 if (m_row == m_frameFilter->m_numRows - 1)
420 {
421 pixY += (realH - 1) * stride;
422 pixU += ((realH >> vChromaShift) - 1) * strideC;
423 pixV += ((realH >> vChromaShift) - 1) * strideC;
424 for (uint32_t y = 0; y < lumaMarginY; y++)
425 memcpy(pixY + (y + 1) * stride, pixY, copySizeY * sizeof(pixel));
426
427 if (m_frameFilter->m_param->internalCsp != X265_CSP_I400)
428 {
429 for (uint32_t y = 0; y < chromaMarginY; y++)
430 {
431 memcpy(pixU + (y + 1) * strideC, pixU, copySizeC * sizeof(pixel));
432 memcpy(pixV + (y + 1) * strideC, pixV, copySizeC * sizeof(pixel));
433 }
434 }
435 }

438// NOTE: Single Threading only
439void FrameFilter::ParallelFilter::processTasks(int /*workerThreadId*/)
440{
441 SAOParam* saoParam = m_encData->m_saoParam;
442 const CUGeom* cuGeoms = m_frameFilter->m_frameEncoder->m_cuGeoms;
443 const uint32_t* ctuGeomMap = m_frameFilter->m_frameEncoder->m_ctuGeomMap;
444 PicYuv* reconPic = m_encData->m_reconPic;
445 const int colStart = m_lastCol.get();
446 const int numCols = m_frameFilter->m_numCols;
447 // TODO: Waiting previous row finish or simple clip on it?
448 int colEnd = m_allowedCol.get();
449
450 // Avoid threading conflict
451 if (!m_encData->getPicCTU(m_rowAddr)->m_bFirstRowInSlice && colEnd > m_prevRow->m_lastDeblocked.get())
452 colEnd = m_prevRow->m_lastDeblocked.get();
453
454 if (colStart >= colEnd)
455 return;
456
457 for (uint32_t col = (uint32_t)colStart; col < (uint32_t)colEnd; col++)
458 {
459 const uint32_t cuAddr = m_rowAddr + col;
460 const CUData* ctu = m_encData->getPicCTU(cuAddr);
461
462 if (m_frameFilter->m_param->bEnableLoopFilter)
463 {
464 deblockCTU(ctu, cuGeoms[ctuGeomMap[cuAddr]], Deblock::EDGE_VER);
465 }
466
467 if (col >= 1)
468 {
469 const CUData* ctuPrev = m_encData->getPicCTU(cuAddr - 1);
470 if (m_frameFilter->m_param->bEnableLoopFilter)
471 {
472 deblockCTU(ctuPrev, cuGeoms[ctuGeomMap[cuAddr - 1]], Deblock::EDGE_HOR);
473
474 // When SAO Disable, setting column counter here
475 if (!m_frameFilter->m_useSao & !ctuPrev->m_bFirstRowInSlice)
476 m_prevRow->processPostCu(col - 1);
477 }
478
479 if (m_frameFilter->m_useSao)
480 {
481 // Save SAO bottom row reference pixels
482 copySaoAboveRef(ctuPrev, reconPic, cuAddr - 1, col - 1);
483
484 // SAO Decide
485 if (col >= 2)
486 {
487 // NOTE: Delay 2 column to avoid mistake on below case, it is Deblock sync logic issue, less probability but still alive
488 // ... H V |
489 // ..S H V |
490 m_sao.rdoSaoUnitCu(saoParam, (ctu->m_bFirstRowInSlice ? 0 : m_rowAddr), col - 2, cuAddr - 2);
491 }
492
493 // Process Previous Row SAO CU
494 if (!ctu->m_bFirstRowInSlice && col >= 3)
495 {
496 // Must delay 1 row to avoid thread data race conflict
497 m_prevRow->processSaoCTU(saoParam, col - 3);
498 m_prevRow->processPostCu(col - 3);
499 }
500 }
501
502 m_lastDeblocked.set(col);
503 }
504 m_lastCol.incr();
505 }
506
507 if (colEnd == numCols)
508 {
509 const uint32_t cuAddr = m_rowAddr + numCols - 1;
510 const CUData* ctuPrev = m_encData->getPicCTU(cuAddr);
511
512 if (m_frameFilter->m_param->bEnableLoopFilter)
513 {
514 deblockCTU(ctuPrev, cuGeoms[ctuGeomMap[cuAddr]], Deblock::EDGE_HOR);
515
516 // When SAO Disable, setting column counter here
517 if (!m_frameFilter->m_useSao & !ctuPrev->m_bFirstRowInSlice)
518 m_prevRow->processPostCu(numCols - 1);
519 }
520
521 // TODO: move processPostCu() into processSaoUnitCu()
522 if (m_frameFilter->m_useSao)
523 {
524 const CUData* ctu = m_encData->getPicCTU(m_rowAddr + numCols - 2);
525
526 // Save SAO bottom row reference pixels
527 copySaoAboveRef(ctuPrev, reconPic, cuAddr, numCols - 1);
528
529 // SAO Decide
530 // NOTE: reduce condition check for 1 CU only video, Why someone play with it?
531 if (numCols >= 2)
532 m_sao.rdoSaoUnitCu(saoParam, (ctu->m_bFirstRowInSlice ? 0 : m_rowAddr), numCols - 2, cuAddr - 1);
533
534 if (numCols >= 1)
535 m_sao.rdoSaoUnitCu(saoParam, (ctuPrev->m_bFirstRowInSlice ? 0 : m_rowAddr), numCols - 1, cuAddr);
536
537 // Process Previous Rows SAO CU
538 if (!ctuPrev->m_bFirstRowInSlice & (numCols >= 3))
539 {
540 m_prevRow->processSaoCTU(saoParam, numCols - 3);
541 m_prevRow->processPostCu(numCols - 3);
542 }
543
544 if (!ctuPrev->m_bFirstRowInSlice & (numCols >= 2))
545 {
546 m_prevRow->processSaoCTU(saoParam, numCols - 2);
547 m_prevRow->processPostCu(numCols - 2);
548 }
549
550 if (!ctuPrev->m_bFirstRowInSlice & (numCols >= 1))
551 {
552 m_prevRow->processSaoCTU(saoParam, numCols - 1);
553 m_prevRow->processPostCu(numCols - 1);
554 }
555
556 // Setting column sync counter
557 if (!ctuPrev->m_bFirstRowInSlice)
558 m_frameFilter->m_frame->m_reconColCount[m_row - 1].set(numCols - 1);
559 }
560 m_lastDeblocked.set(numCols);
561 }
562}

564void FrameFilter::processRow(int row)
565{
566 ProfileScopeEvent(filterCTURow);
567
568#if DETAILED_CU_STATS
569 ScopedElapsedTime filterPerfScope(m_frameEncoder->m_cuStats.loopFilterElapsedTime);
570 m_frameEncoder->m_cuStats.countLoopFilter++;
571#endif
572
573 if (!m_param->bEnableLoopFilter && !m_useSao)
574 {
575 processPostRow(row);
576 return;
577 }
578 FrameData& encData = *m_frame->m_encData;
579
580 // SAO: was integrate into encode loop
581 SAOParam* saoParam = encData.m_saoParam;
582 CUData* ctu = encData.getPicCTU(m_parallelFilter[row].m_rowAddr);
583
584 /* Processing left block Deblock with current threading */
585 {
586 /* Check to avoid previous row process slower than current row */
587 X265_CHECK(ctu->m_bFirstRowInSlice || m_parallelFilter[row - 1].m_lastDeblocked.get() == m_numCols, "previous row not finish");
588
589 m_parallelFilter[row].m_allowedCol.set(m_numCols);
590 m_parallelFilter[row].processTasks(-1);
591
592 if (ctu->m_bLastRowInSlice)
593 {
594 /* TODO: Early start last row */
595 if ((!ctu->m_bFirstRowInSlice) && (m_parallelFilter[row - 1].m_lastDeblocked.get() != m_numCols))
596 x265_log(m_param, X265_LOG_WARNING, "detected ParallelFilter race condition on last row\n");
597
598 /* Apply SAO on last row of CUs, because we always apply SAO on row[X-1] */
599 if (m_useSao)
600 {
601 for(int col = 0; col < m_numCols; col++)
602 {
603 // NOTE: must use processSaoUnitCu(), it include TQBypass logic
604 m_parallelFilter[row].processSaoCTU(saoParam, col);
605 }
606 }
607
608 // Process border extension on last row
609 for(int col = 0; col < m_numCols; col++)
610 {
611 // m_reconColCount will be set in processPostCu()
612 m_parallelFilter[row].processPostCu(col);
613 }
614 }
615 }
616
617 // this row of CTUs has been encoded
618 if (!ctu->m_bFirstRowInSlice)
619 processPostRow(row - 1);
620
621 // NOTE: slices parallelism will be execute out-of-order
622 int numRowFinished = 0;
623 if (m_frame->m_reconRowFlag)
624 {
625 for (numRowFinished = 0; numRowFinished < m_numRows; numRowFinished++)
626 {
627 if (!m_frame->m_reconRowFlag[numRowFinished].get())
628 break;
629
630 if (numRowFinished == row)
631 continue;
632 }
633 }
634
635 if (numRowFinished == m_numRows)
636 {
637 if (m_useSao)
638 {
639 // Merge numNoSao into RootNode (Node0)
640 for(int i = 1; i < m_numRows; i++)
641 {
642 m_parallelFilter[0].m_sao.m_numNoSao[0] += m_parallelFilter[i].m_sao.m_numNoSao[0];
643 m_parallelFilter[0].m_sao.m_numNoSao[1] += m_parallelFilter[i].m_sao.m_numNoSao[1];
644 }
645
646 m_parallelFilter[0].m_sao.rdoSaoUnitRowEnd(saoParam, encData.m_slice->m_sps->numCUsInFrame);
647 }
648 }
649
650 if (ctu->m_bLastRowInSlice)
651 processPostRow(row);
652}

654void FrameFilter::processPostRow(int row)
655{
656 PicYuv *reconPic = m_frame->m_reconPic;
657 const uint32_t numCols = m_frame->m_encData->m_slice->m_sps->numCuInWidth;
658 const uint32_t lineStartCUAddr = row * numCols;
659
660 /* Generate integral planes for SEA motion search */
661 if(m_param->searchMethod == X265_SEA)
662 computeMEIntegral(row);
663 // Notify other FrameEncoders that this row of reconstructed pixels is available
664 m_frame->m_reconRowFlag[row].set(1);
665
666 uint32_t cuAddr = lineStartCUAddr;
667 if (m_param->bEnablePsnr)
668 {
669 PicYuv* fencPic = m_frame->m_fencPic;
670
671 intptr_t stride = reconPic->m_stride;
672 uint32_t width = reconPic->m_picWidth - m_pad[0];
673 uint32_t height = m_parallelFilter[row].getCUHeight();
674
675 uint64_t ssdY = m_frameEncoder->m_top->computeSSD(fencPic->getLumaAddr(cuAddr), reconPic->getLumaAddr(cuAddr), stride, width, height, m_param);
676 m_frameEncoder->m_SSDY += ssdY;
677
678 if (m_param->internalCsp != X265_CSP_I400)
679 {
680 height >>= m_vChromaShift;
681 width >>= m_hChromaShift;
682 stride = reconPic->m_strideC;
683
684 uint64_t ssdU = m_frameEncoder->m_top->computeSSD(fencPic->getCbAddr(cuAddr), reconPic->getCbAddr(cuAddr), stride, width, height, m_param);
685 uint64_t ssdV = m_frameEncoder->m_top->computeSSD(fencPic->getCrAddr(cuAddr), reconPic->getCrAddr(cuAddr), stride, width, height, m_param);
686
687 m_frameEncoder->m_SSDU += ssdU;
688 m_frameEncoder->m_SSDV += ssdV;
689 }
690 }
691
692 if (m_param->bEnableSsim && m_ssimBuf)
693 {
694 pixel *rec = reconPic->m_picOrg[0];
695 pixel *fenc = m_frame->m_fencPic->m_picOrg[0];
696 intptr_t stride1 = reconPic->m_stride;
697 intptr_t stride2 = m_frame->m_fencPic->m_stride;
698 uint32_t bEnd = ((row) == (this->m_numRows - 1));
699 uint32_t bStart = (row == 0);
700 uint32_t minPixY = row * m_param->maxCUSize - 4 * !bStart;
701 uint32_t maxPixY = X265_MIN((row + 1) * m_param->maxCUSize - 4 * !bEnd, (uint32_t)m_param->sourceHeight);
702 uint32_t ssim_cnt;
703 x265_emms();
704
705 /* SSIM is done for each row in blocks of 4x4 . The First blocks are offset by 2 pixels to the right
706 * to avoid alignment of ssim blocks with DCT blocks. */
707 minPixY += bStart ? 2 : -6;
708 m_frameEncoder->m_ssim += calculateSSIM(rec + 2 + minPixY * stride1, stride1, fenc + 2 + minPixY * stride2, stride2,
709 m_param->sourceWidth - 2, maxPixY - minPixY, m_ssimBuf, ssim_cnt);
710 m_frameEncoder->m_ssimCnt += ssim_cnt;
711 }
712
713 if (m_param->maxSlices == 1)
714 {
715 uint32_t height = m_parallelFilter[row].getCUHeight();
716 m_frameEncoder->initDecodedPictureHashSEI(row, cuAddr, height);
717 } // end of (m_param->maxSlices == 1)
718
719 if (ATOMIC_INC(&m_frameEncoder->m_completionCount) == 2 * (int)m_frameEncoder->m_numRows)
720 {
721 m_frameEncoder->m_completionEvent.trigger();
722 }
723}

725void FrameFilter::computeMEIntegral(int row)
726{
727 int lastRow = row == (int)m_frame->m_encData->m_slice->m_sps->numCuInHeight - 1;
728 if (m_frame->m_lowres.sliceType != X265_TYPE_B)
729 {
730 /* If WPP, other than first row, integral calculation for current row needs to wait till the
731 * integral for the previous row is computed */
732 if (m_param->bEnableWavefront && row)
733 {
734 while (m_parallelFilter[row - 1].m_frameFilter->integralCompleted.get() == 0)
735 {
736 m_parallelFilter[row - 1].m_frameFilter->integralCompleted.waitForChange(0);
737 }
738 }
739
740 int stride = (int)m_frame->m_reconPic->m_stride;
741 int padX = m_param->maxCUSize + 32;
742 int padY = m_param->maxCUSize + 16;
743 int numCuInHeight = m_frame->m_encData->m_slice->m_sps->numCuInHeight;
744 int maxHeight = numCuInHeight * m_param->maxCUSize;
745 int startRow = 0;
746
747 if (m_param->interlaceMode)
748 startRow = (row * m_param->maxCUSize >> 1);
749 else
750 startRow = row * m_param->maxCUSize;
751
752 int height = lastRow ? (maxHeight + m_param->maxCUSize * m_param->interlaceMode) : (((row + m_param->interlaceMode) * m_param->maxCUSize) + m_param->maxCUSize);
753
754 if (!row)
755 {
756 for (int i = 0; i < INTEGRAL_PLANE_NUM; i++)
757 memset(m_frame->m_encData->m_meIntegral[i] - padY * stride - padX, 0, stride * sizeof(uint32_t));
758 startRow = -padY;
759 }
760
761 if (lastRow)
762 height += padY - 1;
763
764 for (int y = startRow; y < height; y++)
765 {
766 pixel *pix = m_frame->m_reconPic->m_picOrg[0] + y * stride - padX;
767 uint32_t *sum32x32 = m_frame->m_encData->m_meIntegral[0] + (y + 1) * stride - padX;
768 uint32_t *sum32x24 = m_frame->m_encData->m_meIntegral[1] + (y + 1) * stride - padX;
769 uint32_t *sum32x8 = m_frame->m_encData->m_meIntegral[2] + (y + 1) * stride - padX;
770 uint32_t *sum24x32 = m_frame->m_encData->m_meIntegral[3] + (y + 1) * stride - padX;
771 uint32_t *sum16x16 = m_frame->m_encData->m_meIntegral[4] + (y + 1) * stride - padX;
772 uint32_t *sum16x12 = m_frame->m_encData->m_meIntegral[5] + (y + 1) * stride - padX;
773 uint32_t *sum16x4 = m_frame->m_encData->m_meIntegral[6] + (y + 1) * stride - padX;
774 uint32_t *sum12x16 = m_frame->m_encData->m_meIntegral[7] + (y + 1) * stride - padX;
775 uint32_t *sum8x32 = m_frame->m_encData->m_meIntegral[8] + (y + 1) * stride - padX;
776 uint32_t *sum8x8 = m_frame->m_encData->m_meIntegral[9] + (y + 1) * stride - padX;
777 uint32_t *sum4x16 = m_frame->m_encData->m_meIntegral[10] + (y + 1) * stride - padX;
778 uint32_t *sum4x4 = m_frame->m_encData->m_meIntegral[11] + (y + 1) * stride - padX;
779
780 /*For width = 32 */
781 primitives.integral_inith[INTEGRAL_32](sum32x32, pix, stride);
782 if (y >= 32 - padY)
783 primitives.integral_initv[INTEGRAL_32](sum32x32 - 32 * stride, stride);
784 primitives.integral_inith[INTEGRAL_32](sum32x24, pix, stride);
785 if (y >= 24 - padY)
786 primitives.integral_initv[INTEGRAL_24](sum32x24 - 24 * stride, stride);
787 primitives.integral_inith[INTEGRAL_32](sum32x8, pix, stride);
788 if (y >= 8 - padY)
789 primitives.integral_initv[INTEGRAL_8](sum32x8 - 8 * stride, stride);
790 /*For width = 24 */
791 primitives.integral_inith[INTEGRAL_24](sum24x32, pix, stride);
792 if (y >= 32 - padY)
793 primitives.integral_initv[INTEGRAL_32](sum24x32 - 32 * stride, stride);
794 /*For width = 16 */
795 primitives.integral_inith[INTEGRAL_16](sum16x16, pix, stride);
796 if (y >= 16 - padY)
797 primitives.integral_initv[INTEGRAL_16](sum16x16 - 16 * stride, stride);
798 primitives.integral_inith[INTEGRAL_16](sum16x12, pix, stride);
799 if (y >= 12 - padY)
800 primitives.integral_initv[INTEGRAL_12](sum16x12 - 12 * stride, stride);
801 primitives.integral_inith[INTEGRAL_16](sum16x4, pix, stride);
802 if (y >= 4 - padY)
803 primitives.integral_initv[INTEGRAL_4](sum16x4 - 4 * stride, stride);
804 /*For width = 12 */
805 primitives.integral_inith[INTEGRAL_12](sum12x16, pix, stride);
806 if (y >= 16 - padY)
807 primitives.integral_initv[INTEGRAL_16](sum12x16 - 16 * stride, stride);
808 /*For width = 8 */
809 primitives.integral_inith[INTEGRAL_8](sum8x32, pix, stride);
810 if (y >= 32 - padY)
811 primitives.integral_initv[INTEGRAL_32](sum8x32 - 32 * stride, stride);
812 primitives.integral_inith[INTEGRAL_8](sum8x8, pix, stride);
813 if (y >= 8 - padY)
814 primitives.integral_initv[INTEGRAL_8](sum8x8 - 8 * stride, stride);
815 /*For width = 4 */
816 primitives.integral_inith[INTEGRAL_4](sum4x16, pix, stride);
817 if (y >= 16 - padY)
818 primitives.integral_initv[INTEGRAL_16](sum4x16 - 16 * stride, stride);
819 primitives.integral_inith[INTEGRAL_4](sum4x4, pix, stride);
820 if (y >= 4 - padY)
821 primitives.integral_initv[INTEGRAL_4](sum4x4 - 4 * stride, stride);
822 }
823 m_parallelFilter[row].m_frameFilter->integralCompleted.set(1);
824 }
825}

827/* Function to calculate SSIM for each row */
828static float calculateSSIM(pixel *pix1, intptr_t stride1, pixel *pix2, intptr_t stride2, uint32_t width, uint32_t height, void *buf, uint32_t& cnt)
829{
830 uint32_t z = 0;
831 float ssim = 0.0;
832
833 int(*sum0)[4] = (int(*)[4])buf;
834 int(*sum1)[4] = sum0 + (width >> 2) + 3;
835 width >>= 2;
836 height >>= 2;
837
838 for (uint32_t y = 1; y < height; y++)
839 {
840 for (; z <= y; z++)
841 {
842 std::swap(sum0, sum1);
843 for (uint32_t x = 0; x < width; x += 2)
844 primitives.ssim_4x4x2_core(&pix1[4 * (x + (z * stride1))], stride1, &pix2[4 * (x + (z * stride2))], stride2, &sum0[x]);
845 }
846
847 for (uint32_t x = 0; x < width - 1; x += 4)
848 ssim += primitives.ssim_end_4(sum0 + x, sum1 + x, X265_MIN(4, width - x - 1));
849 }
850
851 cnt = (height - 1) * (width - 1);
852 return ssim;
853}
