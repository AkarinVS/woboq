31NALList::NALList()
32 : m_numNal(0)
33 , m_buffer(NULL)
34 , m_occupancy(0)
35 , m_allocSize(0)
36 , m_extraBuffer(NULL)
37 , m_extraOccupancy(0)
38 , m_extraAllocSize(0)
39 , m_annexB(true)
40{}

42void NALList::takeContents(NALList& other)
43{
44 /* take other NAL buffer, discard our old one */
45 X265_FREE(m_buffer);
46 m_buffer = other.m_buffer;
47 m_allocSize = other.m_allocSize;
48 m_occupancy = other.m_occupancy;
49
50 /* copy packet data */
51 m_numNal = other.m_numNal;
52 memcpy(m_nal, other.m_nal, sizeof(x265_nal) * m_numNal);
53
54 /* reset other list, re-allocate their buffer with same size */
55 other.m_numNal = 0;
56 other.m_occupancy = 0;
57 other.m_buffer = X265_MALLOC(uint8_t, m_allocSize);
58}

60void NALList::serialize(NalUnitType nalUnitType, const Bitstream& bs)
61{
62 static const char startCodePrefix[] = { 0, 0, 0, 1 };
63
64 uint32_t payloadSize = bs.getNumberOfWrittenBytes();
65 const uint8_t* bpayload = bs.getFIFO();
66 if (!bpayload)
67 return;
68
69 uint32_t nextSize = m_occupancy + sizeof(startCodePrefix) + 2 + payloadSize + (payloadSize >> 1) + m_extraOccupancy;
70 if (nextSize > m_allocSize)
71 {
72 uint8_t *temp = X265_MALLOC(uint8_t, nextSize);
73 if (temp)
74 {
75 memcpy(temp, m_buffer, m_occupancy);
76
77 /* fixup existing payload pointers */
78 for (uint32_t i = 0; i < m_numNal; i++)
79 m_nal[i].payload = temp + (m_nal[i].payload - m_buffer);
80
81 X265_FREE(m_buffer);
82 m_buffer = temp;
83 m_allocSize = nextSize;
84 }
85 else
86 {
87 x265_log(NULL, X265_LOG_ERROR, "Unable to realloc access unit buffer\n");
88 return;
89 }
90 }
91
92 uint8_t *out = m_buffer + m_occupancy;
93 uint32_t bytes = 0;
94
95 if (!m_annexB)
96 {
97 /* Will write size later */
98 bytes += 4;
99 }
100 else if (!m_numNal || nalUnitType == NAL_UNIT_VPS || nalUnitType == NAL_UNIT_SPS || nalUnitType == NAL_UNIT_PPS || nalUnitType == NAL_UNIT_UNSPECIFIED)
101 {
102 memcpy(out, startCodePrefix, 4);
103 bytes += 4;
104 }
105 else
106 {
107 memcpy(out, startCodePrefix + 1, 3);
108 bytes += 3;
109 }
110
111 /* 16 bit NAL header:
112 * forbidden_zero_bit 1-bit
113 * nal_unit_type 6-bits
114 * nuh_reserved_zero_6bits 6-bits
115 * nuh_temporal_id_plus1 3-bits */
116 out[bytes++] = (uint8_t)nalUnitType << 1;
117 out[bytes++] = 1 + (nalUnitType == NAL_UNIT_CODED_SLICE_TSA_N);
118
119 /* 7.4.1 ...
120 * Within the NAL unit, the following three-byte sequences shall not occur at
121 * any byte-aligned position:
122 * - 0x000000
123 * - 0x000001
124 * - 0x000002 */
125 for (uint32_t i = 0; i < payloadSize; i++)
126 {
127 if (i > 2 && !out[bytes - 2] && !out[bytes - 3] && out[bytes - 1] <= 0x03 && nalUnitType != NAL_UNIT_UNSPECIFIED)
128 {
129 /* inject 0x03 to prevent emulating a start code */
130 out[bytes] = out[bytes - 1];
131 out[bytes - 1] = 0x03;
132 bytes++;
133 }
134
135 out[bytes++] = bpayload[i];
136 }
137
138 X265_CHECK(bytes <= 4 + 2 + payloadSize + (payloadSize >> 1), "NAL buffer overflow\n");
139
140 if (m_extraOccupancy)
141 {
142 /* these bytes were escaped by serializeSubstreams */
143 memcpy(out + bytes, m_extraBuffer, m_extraOccupancy);
144 bytes += m_extraOccupancy;
145 m_extraOccupancy = 0;
146 }
147
148 /* 7.4.1.1
149 * ... when the last byte of the RBSP data is equal to 0x00 (which can
150 * only occur when the RBSP ends in a cabac_zero_word), a final byte equal
151 * to 0x03 is appended to the end of the data. */
152 if (!out[bytes - 1])
153 out[bytes++] = 0x03;
154
155 if (!m_annexB)
156 {
157 uint32_t dataSize = bytes - 4;
158 out[0] = (uint8_t)(dataSize >> 24);
159 out[1] = (uint8_t)(dataSize >> 16);
160 out[2] = (uint8_t)(dataSize >> 8);
161 out[3] = (uint8_t)dataSize;
162 }
163
164 m_occupancy += bytes;
165
166 X265_CHECK(m_numNal < (uint32_t)MAX_NAL_UNITS, "NAL count overflow\n");
167
168 x265_nal& nal = m_nal[m_numNal++];
169 nal.type = nalUnitType;
170 nal.sizeBytes = bytes;
171 nal.payload = out;
172}

174/* concatenate and escape WPP sub-streams, return escaped row lengths.
175 * These streams will be appended to the next serialized NAL */
176uint32_t NALList::serializeSubstreams(uint32_t* streamSizeBytes, uint32_t streamCount, const Bitstream* streams)
177{
178 uint32_t maxStreamSize = 0;
179 uint32_t estSize = 0;
180 for (uint32_t s = 0; s < streamCount; s++)
181 estSize += streams[s].getNumberOfWrittenBytes();
182 estSize += estSize >> 1;
183
184 if (estSize > m_extraAllocSize)
185 {
186 uint8_t *temp = X265_MALLOC(uint8_t, estSize);
187 if (temp)
188 {
189 X265_FREE(m_extraBuffer);
190 m_extraBuffer = temp;
191 m_extraAllocSize = estSize;
192 }
193 else
194 {
195 x265_log(NULL, X265_LOG_ERROR, "Unable to realloc WPP substream concatenation buffer\n");
196 return 0;
197 }
198 }
199
200 uint32_t bytes = 0;
201 uint8_t *out = m_extraBuffer;
202 for (uint32_t s = 0; s < streamCount; s++)
203 {
204 const Bitstream& stream = streams[s];
205 uint32_t inSize = stream.getNumberOfWrittenBytes();
206 const uint8_t *inBytes = stream.getFIFO();
207 uint32_t prevBufSize = bytes;
208
209 if (inBytes)
210 {
211 for (uint32_t i = 0; i < inSize; i++)
212 {
213 if (bytes >= 2 && !out[bytes - 2] && !out[bytes - 1] && inBytes[i] <= 0x03)
214 {
215 /* inject 0x03 to prevent emulating a start code */
216 out[bytes++] = 3;
217 }
218
219 out[bytes++] = inBytes[i];
220 }
221 }
222
223 if (s < streamCount - 1)
224 {
225 streamSizeBytes[s] = bytes - prevBufSize;
226 if (streamSizeBytes[s] > maxStreamSize)
227 maxStreamSize = streamSizeBytes[s];
228 }
229 }
230
231 m_extraOccupancy = bytes;
232 return maxStreamSize;
233}
