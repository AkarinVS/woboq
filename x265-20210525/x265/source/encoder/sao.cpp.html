34inline int32_t roundIBDI(int32_t num, int32_t den)
35{
36 return num >= 0 ? ((num * 2 + den) / (den * 2)) : -((-num * 2 + den) / (den * 2));
37}

39/* get the sign of input variable (TODO: this is a dup, make common) */
40inline int8_t signOf(int x)
41{
42 return (x >> 31) | ((int)((((uint32_t)-x)) >> 31));
43}

45inline int signOf2(const int a, const int b)
46{
47 // NOTE: don't reorder below compare, both ICL, VC, GCC optimize strong depends on order!
48 int r = 0;
49 if (a < b)
50 r = -1;
51 if (a > b)
52 r = 1;
53 return r;
54}

56inline int64_t estSaoDist(int32_t count, int32_t offset, int32_t offsetOrg)
57{
58 return (count * offset - offsetOrg * 2) * offset;

94bool SAO::create(x265_param* param, int initCommon)
95{
96 m_param = param;
97 m_chromaFormat = param->internalCsp;
98 m_hChromaShift = CHROMA_H_SHIFT(param->internalCsp);
99 m_vChromaShift = CHROMA_V_SHIFT(param->internalCsp);
100
101 m_numCuInWidth = (m_param->sourceWidth + m_param->maxCUSize - 1) / m_param->maxCUSize;
102 m_numCuInHeight = (m_param->sourceHeight + m_param->maxCUSize - 1) / m_param->maxCUSize;
103
104 const pixel maxY = (1 << X265_DEPTH) - 1;
105 const pixel rangeExt = maxY >> 1;
106 int numCtu = m_numCuInWidth * m_numCuInHeight;
107
108 for (int i = 0; i < (param->internalCsp != X265_CSP_I400 ? 3 : 1); i++)
109 {
110 CHECKED_MALLOC(m_tmpL1[i], pixel, m_param->maxCUSize + 1);
111 CHECKED_MALLOC(m_tmpL2[i], pixel, m_param->maxCUSize + 1);
112
113 // SAO asm code will read 1 pixel before and after, so pad by 2
114 // NOTE: m_param->sourceWidth+2 enough, to avoid condition check in copySaoAboveRef(), I alloc more up to 63 bytes in here
115 CHECKED_MALLOC(m_tmpU[i], pixel, m_numCuInWidth * m_param->maxCUSize + 2 + 32);
116 m_tmpU[i] += 1;
117 }
118
119 if (initCommon)
120 {
121 if (m_param->bSaoNonDeblocked)
122 {
123 CHECKED_MALLOC(m_countPreDblk, PerPlane, numCtu);
124 CHECKED_MALLOC(m_offsetOrgPreDblk, PerPlane, numCtu);
125 }
126 CHECKED_MALLOC(m_depthSaoRate, double, 2 * SAO_DEPTHRATE_SIZE);
127
128 m_depthSaoRate[0 * SAO_DEPTHRATE_SIZE + 0] = 0;
129 m_depthSaoRate[0 * SAO_DEPTHRATE_SIZE + 1] = 0;
130 m_depthSaoRate[0 * SAO_DEPTHRATE_SIZE + 2] = 0;
131 m_depthSaoRate[0 * SAO_DEPTHRATE_SIZE + 3] = 0;
132 m_depthSaoRate[1 * SAO_DEPTHRATE_SIZE + 0] = 0;
133 m_depthSaoRate[1 * SAO_DEPTHRATE_SIZE + 1] = 0;
134 m_depthSaoRate[1 * SAO_DEPTHRATE_SIZE + 2] = 0;
135 m_depthSaoRate[1 * SAO_DEPTHRATE_SIZE + 3] = 0;
136
137 CHECKED_MALLOC(m_clipTableBase, pixel, maxY + 2 * rangeExt);
138 m_clipTable = &(m_clipTableBase[rangeExt]);
139
140 // Share with fast clip lookup table
141
142 for (int i = 0; i < rangeExt; i++)
143 m_clipTableBase[i] = 0;
144
145 for (int i = 0; i < maxY; i++)
146 m_clipTable[i] = (pixel)i;
147
148 for (int i = maxY; i < maxY + rangeExt; i++)
149 m_clipTable[i] = maxY;
150
151 }
152 else
153 {
154 // must initialize these common pointer outside of function
155 m_countPreDblk = NULL;
156 m_offsetOrgPreDblk = NULL;
157 m_clipTableBase = NULL;
158 m_clipTable = NULL;
159 }
160
161 return true;
162
163fail:
164 return false;
165}

183{
184 for (int i = 0; i < 3; i++)
185 {
186 if (m_tmpL1[i])
187 {
188 X265_FREE(m_tmpL1[i]);

204

227void SAO::startSlice(Frame* frame, Entropy& initState)
228{
229 m_frame = frame;
230 Slice* slice = m_frame->m_encData->m_slice;
231
232 switch (slice->m_sliceType)
233 {
234 case I_SLICE:
235 m_refDepth = 0;
236 break;
237 case P_SLICE:
238 m_refDepth = 1;
239 break;
240 case B_SLICE:
241 m_refDepth = 2 + !IS_REFERENCED(frame);
242 break;
243 }
244
245 m_entropyCoder.load(initState);
246 m_rdContexts.next.load(initState);
247 m_rdContexts.cur.load(initState);
248
249 SAOParam* saoParam = frame->m_encData->m_saoParam;
250 if (!saoParam)
251 {
252 saoParam = new SAOParam;
253 allocSaoParam(saoParam);
254 frame->m_encData->m_saoParam = saoParam;
255 }
256
257 saoParam->bSaoFlag[0] = true;
258 saoParam->bSaoFlag[1] = m_param->internalCsp != X265_CSP_I400 && m_frame->m_fencPic->m_picCsp != X265_CSP_I400;
259
260 m_numNoSao[0] = 0; // Luma
261 m_numNoSao[1] = 0; // Chroma
262
263 // NOTE: Allow SAO automatic turn-off only when frame parallelism is disabled.
264 if (m_param->frameNumThreads == 1)
265 {
266 if (m_refDepth > 0 && m_depthSaoRate[0 * SAO_DEPTHRATE_SIZE + m_refDepth - 1] > SAO_ENCODING_RATE)
267 saoParam->bSaoFlag[0] = false;
268 if (m_refDepth > 0 && m_depthSaoRate[1 * SAO_DEPTHRATE_SIZE + m_refDepth - 1] > SAO_ENCODING_RATE_CHROMA)
269 saoParam->bSaoFlag[1] = false;
270 }
271}

273// CTU-based SAO process without slice granularity
274void SAO::applyPixelOffsets(int addr, int typeIdx, int plane)
275{
276 PicYuv* reconPic = m_frame->m_reconPic;
277 pixel* rec = reconPic->getPlaneAddr(plane, addr);
278 intptr_t stride = plane ? reconPic->m_strideC : reconPic->m_stride;
279 uint32_t picWidth = m_param->sourceWidth;
280 uint32_t picHeight = m_param->sourceHeight;
281 const CUData* cu = m_frame->m_encData->getPicCTU(addr);
282 int ctuWidth = m_param->maxCUSize;
283 int ctuHeight = m_param->maxCUSize;
284 uint32_t lpelx = cu->m_cuPelX;
285 uint32_t tpely = cu->m_cuPelY;
286 const uint32_t firstRowInSlice = cu->m_bFirstRowInSlice;
287 const uint32_t lastRowInSlice = cu->m_bLastRowInSlice;
288 const uint32_t bAboveUnavail = (!tpely) | firstRowInSlice;
289
290 // NOTE: Careful! the picHeight for Equal operator only, so I may safe to hack it
291 if (lastRowInSlice)
292 {
293 picHeight = x265_min(picHeight, (tpely + ctuHeight));
294 }
295
296 if (plane)
297 {
298 picWidth >>= m_hChromaShift;
299 picHeight >>= m_vChromaShift;
300 ctuWidth >>= m_hChromaShift;
301 ctuHeight >>= m_vChromaShift;
302 lpelx >>= m_hChromaShift;
303 tpely >>= m_vChromaShift;
304 }
305 uint32_t rpelx = x265_min(lpelx + ctuWidth, picWidth);
306 uint32_t bpely = x265_min(tpely + ctuHeight, picHeight);
307 ctuWidth = rpelx - lpelx;
308 ctuHeight = bpely - tpely;
309
310 int8_t _upBuff1[MAX_CU_SIZE + 2], *upBuff1 = _upBuff1 + 1, signLeft1[2];
311 int8_t _upBufft[MAX_CU_SIZE + 2], *upBufft = _upBufft + 1;
312
313 memset(_upBuff1 + MAX_CU_SIZE, 0, 2 * sizeof(int8_t)); /* avoid valgrind uninit warnings */
314
315 pixel* tmpL = m_tmpL1[plane];
316 pixel* tmpU = &(m_tmpU[plane][lpelx]);
317
318 int8_t* offsetEo = m_offsetEo[plane];
319
320 switch (typeIdx)
321 {
322 case SAO_EO_0: // dir: -
323 {
324 pixel firstPxl = 0, lastPxl = 0, row1FirstPxl = 0, row1LastPxl = 0;
325 int startX = !lpelx;
326 int endX = (rpelx == picWidth) ? ctuWidth - 1 : ctuWidth;
327 if (ctuWidth & 15)
328 {
329 for (int y = 0; y < ctuHeight; y++, rec += stride)
330 {
331 int signLeft = signOf(rec[startX] - tmpL[y]);
332 for (int x = startX; x < endX; x++)
333 {
334 int signRight = signOf(rec[x] - rec[x + 1]);
335 int edgeType = signRight + signLeft + 2;
336 signLeft = -signRight;
337
338 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
339 }
340 }
341 }
342 else
343 {
344 for (int y = 0; y < ctuHeight; y += 2, rec += 2 * stride)
345 {
346 signLeft1[0] = signOf(rec[startX] - tmpL[y]);
347 signLeft1[1] = signOf(rec[stride + startX] - tmpL[y + 1]);
348
349 if (!lpelx)
350 {
351 firstPxl = rec[0];
352 row1FirstPxl = rec[stride];
353 }
354
355 if (rpelx == picWidth)
356 {
357 lastPxl = rec[ctuWidth - 1];
358 row1LastPxl = rec[stride + ctuWidth - 1];
359 }
360
361 primitives.saoCuOrgE0(rec, offsetEo, ctuWidth, signLeft1, stride);
362
363 if (!lpelx)
364 {
365 rec[0] = firstPxl;
366 rec[stride] = row1FirstPxl;
367 }
368
369 if (rpelx == picWidth)
370 {
371 rec[ctuWidth - 1] = lastPxl;
372 rec[stride + ctuWidth - 1] = row1LastPxl;
373 }
374 }
375 }
376 break;
377 }
378 case SAO_EO_1: // dir: |
379 {
380 int startY = bAboveUnavail;
381 int endY = (bpely == picHeight) ? ctuHeight - 1 : ctuHeight;
382 if (startY)
383 rec += stride;
384
385 if (ctuWidth & 15)
386 {
387 for (int x = 0; x < ctuWidth; x++)
388 upBuff1[x] = signOf(rec[x] - tmpU[x]);
389
390 for (int y = startY; y < endY; y++, rec += stride)
391 {
392 for (int x = 0; x < ctuWidth; x++)
393 {
394 int8_t signDown = signOf(rec[x] - rec[x + stride]);
395 int edgeType = signDown + upBuff1[x] + 2;
396 upBuff1[x] = -signDown;
397
398 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
399 }
400 }
401 }
402 else
403 {
404 primitives.sign(upBuff1, rec, tmpU, ctuWidth);
405
406 int diff = (endY - startY) % 2;
407 for (int y = startY; y < endY - diff; y += 2, rec += 2 * stride)
408 primitives.saoCuOrgE1_2Rows(rec, upBuff1, offsetEo, stride, ctuWidth);
409
410 if (diff & 1)
411 primitives.saoCuOrgE1(rec, upBuff1, offsetEo, stride, ctuWidth);
412 }
413
414 break;
415 }
416 case SAO_EO_2: // dir: 135
417 {
418 int startX = !lpelx;
419 int endX = (rpelx == picWidth) ? ctuWidth - 1 : ctuWidth;
420
421 int startY = bAboveUnavail;
422 int endY = (bpely == picHeight) ? ctuHeight - 1 : ctuHeight;
423
424 if (startY)
425 rec += stride;
426
427 if (!(ctuWidth & 15))
428 {
429 int8_t firstSign, lastSign;
430
431 if (!lpelx)
432 firstSign = upBuff1[0];
433
434 if (rpelx == picWidth)
435 lastSign = upBuff1[ctuWidth - 1];
436
437 primitives.sign(upBuff1, rec, &tmpU[- 1], ctuWidth);
438
439 if (!lpelx)
440 upBuff1[0] = firstSign;
441
442 if (rpelx == picWidth)
443 upBuff1[ctuWidth - 1] = lastSign;
444 }
445 else
446 {
447 for (int x = startX; x < endX; x++)
448 upBuff1[x] = signOf(rec[x] - tmpU[x - 1]);
449 }
450
451 if (ctuWidth & 15)
452 {
453 for (int y = startY; y < endY; y++, rec += stride)
454 {
455 upBufft[startX] = signOf(rec[stride + startX] - tmpL[y]);
456 for (int x = startX; x < endX; x++)
457 {
458 int8_t signDown = signOf(rec[x] - rec[x + stride + 1]);
459 int edgeType = signDown + upBuff1[x] + 2;
460 upBufft[x + 1] = -signDown;
461 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
462 }
463
464 std::swap(upBuff1, upBufft);
465 }
466 }
467 else
468 {
469 for (int y = startY; y < endY; y++, rec += stride)
470 {
471 int8_t iSignDown2 = signOf(rec[stride + startX] - tmpL[y]);
472
473 primitives.saoCuOrgE2[endX > 16](rec + startX, upBufft + startX, upBuff1 + startX, offsetEo, endX - startX, stride);
474
475 upBufft[startX] = iSignDown2;
476
477 std::swap(upBuff1, upBufft);
478 }
479 }
480 break;
481 }
482 case SAO_EO_3: // dir: 45
483 {
484 int startX = !lpelx;
485 int endX = (rpelx == picWidth) ? ctuWidth - 1 : ctuWidth;
486
487 int startY = bAboveUnavail;
488 int endY = (bpely == picHeight) ? ctuHeight - 1 : ctuHeight;
489
490 if (startY)
491 rec += stride;
492
493 if (ctuWidth & 15)
494 {
495 for (int x = startX - 1; x < endX; x++)
496 upBuff1[x] = signOf(rec[x] - tmpU[x + 1]);
497
498 for (int y = startY; y < endY; y++, rec += stride)
499 {
500 int x = startX;
501 int8_t signDown = signOf(rec[x] - tmpL[y + 1]);
502 int edgeType = signDown + upBuff1[x] + 2;
503 upBuff1[x - 1] = -signDown;
504 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
505
506 for (x = startX + 1; x < endX; x++)
507 {
508 signDown = signOf(rec[x] - rec[x + stride - 1]);
509 edgeType = signDown + upBuff1[x] + 2;
510 upBuff1[x - 1] = -signDown;
511 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
512 }
513
514 upBuff1[endX - 1] = signOf(rec[endX - 1 + stride] - rec[endX]);
515 }
516 }
517 else
518 {
519 int8_t firstSign, lastSign;
520
521 if (lpelx)
522 firstSign = signOf(rec[-1] - tmpU[0]);
523 if (rpelx == picWidth)
524 lastSign = upBuff1[ctuWidth - 1];
525
526 primitives.sign(upBuff1, rec, &tmpU[1], ctuWidth);
527
528 if (lpelx)
529 upBuff1[-1] = firstSign;
530 if (rpelx == picWidth)
531 upBuff1[ctuWidth - 1] = lastSign;
532
533 for (int y = startY; y < endY; y++, rec += stride)
534 {
535 int x = startX;
536 int8_t signDown = signOf(rec[x] - tmpL[y + 1]);
537 int edgeType = signDown + upBuff1[x] + 2;
538 upBuff1[x - 1] = -signDown;
539 rec[x] = m_clipTable[rec[x] + offsetEo[edgeType]];
540
541 primitives.saoCuOrgE3[endX > 16](rec, upBuff1, offsetEo, stride - 1, startX, endX);
542
543 upBuff1[endX - 1] = signOf(rec[endX - 1 + stride] - rec[endX]);
544 }
545 }
546
547 break;
548 }
549 case SAO_BO:
550 {
551 const int8_t* offsetBo = m_offsetBo[plane];
552
553 if (ctuWidth & 15)
554 {
555 #define SAO_BO_BITS 5
556 const int boShift = X265_DEPTH - SAO_BO_BITS;
557
558 for (int y = 0; y < ctuHeight; y++, rec += stride)
559 for (int x = 0; x < ctuWidth; x++)
560 rec[x] = x265_clip(rec[x] + offsetBo[rec[x] >> boShift]);
561 }
562 else
563 primitives.saoCuOrgB0(rec, offsetBo, ctuWidth, ctuHeight, stride);
564
565 break;
566 }
567 default: break;
568 }
569}

571/* Process SAO unit */
572void SAO::generateLumaOffsets(SaoCtuParam* ctuParam, int idxY, int idxX)
573{
574 PicYuv* reconPic = m_frame->m_reconPic;
575 intptr_t stride = reconPic->m_stride;
576 int ctuWidth = m_param->maxCUSize;
577 int ctuHeight = m_param->maxCUSize;
578
579 int addr = idxY * m_numCuInWidth + idxX;
580 pixel* rec = reconPic->getLumaAddr(addr);
581
582 if (idxX == 0)
583 {
584 for (int i = 0; i < ctuHeight + 1; i++)
585 {
586 m_tmpL1[0][i] = rec[0];
587 rec += stride;
588 }
589 }
590
591 bool mergeLeftFlag = (ctuParam[addr].mergeMode == SAO_MERGE_LEFT);
592 int typeIdx = ctuParam[addr].typeIdx;
593
594 if (idxX != (m_numCuInWidth - 1))
595 {
596 rec = reconPic->getLumaAddr(addr);
597 for (int i = 0; i < ctuHeight + 1; i++)
598 {
599 m_tmpL2[0][i] = rec[ctuWidth - 1];
600 rec += stride;
601 }
602 }
603
604 if (typeIdx >= 0)
605 {
606 if (!mergeLeftFlag)
607 {
608 if (typeIdx == SAO_BO)
609 {
610 memset(m_offsetBo[0], 0, sizeof(m_offsetBo[0]));
611
612 for (int i = 0; i < SAO_NUM_OFFSET; i++)
613 m_offsetBo[0][((ctuParam[addr].bandPos + i) & (MAX_NUM_SAO_CLASS - 1))] = (int8_t)(ctuParam[addr].offset[i] << SAO_BIT_INC);
614 }
615 else // if (typeIdx == SAO_EO_0 || typeIdx == SAO_EO_1 || typeIdx == SAO_EO_2 || typeIdx == SAO_EO_3)
616 {
617 int offset[NUM_EDGETYPE];
618 offset[0] = 0;
619 for (int i = 0; i < SAO_NUM_OFFSET; i++)
620 offset[i + 1] = ctuParam[addr].offset[i] << SAO_BIT_INC;
621
622 for (int edgeType = 0; edgeType < NUM_EDGETYPE; edgeType++)
623 m_offsetEo[0][edgeType] = (int8_t)offset[s_eoTable[edgeType]];
624 }
625 }
626 applyPixelOffsets(addr, typeIdx, 0);
627 }
628 std::swap(m_tmpL1[0], m_tmpL2[0]);
629}

631/* Process SAO unit (Chroma only) */
632void SAO::generateChromaOffsets(SaoCtuParam* ctuParam[3], int idxY, int idxX)
633{
634 PicYuv* reconPic = m_frame->m_reconPic;
635 intptr_t stride = reconPic->m_strideC;
636 int ctuWidth = m_param->maxCUSize;
637 int ctuHeight = m_param->maxCUSize;
638
639 {
640 ctuWidth >>= m_hChromaShift;
641 ctuHeight >>= m_vChromaShift;
642 }
643
644 int addr = idxY * m_numCuInWidth + idxX;
645 pixel* recCb = reconPic->getCbAddr(addr);
646 pixel* recCr = reconPic->getCrAddr(addr);
647
648 if (idxX == 0)
649 {
650 for (int i = 0; i < ctuHeight + 1; i++)
651 {
652 m_tmpL1[1][i] = recCb[0];
653 m_tmpL1[2][i] = recCr[0];
654 recCb += stride;
655 recCr += stride;
656 }
657 }
658
659 bool mergeLeftFlagCb = (ctuParam[1][addr].mergeMode == SAO_MERGE_LEFT);
660 int typeIdxCb = ctuParam[1][addr].typeIdx;
661

1244 bool chroma = m_param->internalCsp != X265_CSP_I400 && m_frame->m_fencPic->m_picCsp != X265_CSP_I400;
1245 int planes = chroma ? 3 : 1;
1246
1247 // reset stats Y, Cb, Cr
1248 X265_CHECK(sizeof(PerPlane) == (sizeof(int32_t) * (NUM_PLANE * MAX_NUM_SAO_TYPE * MAX_NUM_SAO_CLASS)), "Found Padding space in struct PerPlane");
1249
1250 // TODO: Confirm the address space is continuous
1251 if (m_param->bSaoNonDeblocked)
1252 {
1253 memcpy(m_count, m_countPreDblk[addr], sizeof(m_count));
1254 memcpy(m_offsetOrg, m_offsetOrgPreDblk[addr], sizeof(m_offsetOrg));
1255 }
1256 else
1257 {
1258 memset(m_count, 0, sizeof(m_count));
1259 memset(m_offsetOrg, 0, sizeof(m_offsetOrg));
1260 }
1261
1262 for (int i = 0; i < planes; i++)
1263 saoParam->ctuParam[i][addr].reset();
1264 // SAO distortion calculation
1265 m_entropyCoder.load(m_rdContexts.cur);
1266 m_entropyCoder.resetBits();
1267 if (allowMerge[0])
1268 m_entropyCoder.codeSaoMerge(0);
1269 if (allowMerge[1])
1270 m_entropyCoder.codeSaoMerge(0);
1271 m_entropyCoder.store(m_rdContexts.temp);
1272 memset(m_offset, 0, sizeof(m_offset));
1273 int64_t bestCost = 0;
1274 int64_t rateDist = 0;
1275
1276 bool bAboveLeftAvail = true;
1277 for (int mergeIdx = 0; mergeIdx < 2; ++mergeIdx)
1278 {
1279 if (!allowMerge[mergeIdx])
1280 continue;
1281
1282 SaoCtuParam* mergeSrcParam = &(saoParam->ctuParam[0][addrMerge[mergeIdx]]);
1283 bAboveLeftAvail = bAboveLeftAvail && (mergeSrcParam->typeIdx == -1);
1284 }
1285 // Don't apply sao if ctu is skipped or ajacent ctus are sao off
1286 bool bSaoOff = (slice->m_sliceType == B_SLICE) && (cu->isSkipped(0) || bAboveLeftAvail);
1287
1288 // Estimate distortion and cost of new SAO params
1289 if (saoParam->bSaoFlag[0])
1290 {
1291 if (!m_param->bLimitSAO || !bSaoOff)
1292 {
1293 calcSaoStatsCTU(addr, 0);
1294 saoStatsInitialOffset(addr, 0);
1295 saoLumaComponentParamDist(saoParam, addr, rateDist, lambda, bestCost);
1296 }
1297 }
1298
1299 SaoCtuParam* lclCtuParam = &saoParam->ctuParam[0][addr];
1300 if (saoParam->bSaoFlag[1])
1301 {
1302 if (!m_param->bLimitSAO || ((lclCtuParam->typeIdx != -1) && !bSaoOff))
1303 {
1304 calcSaoStatsCTU(addr, 1);
1305 calcSaoStatsCTU(addr, 2);
1306 saoStatsInitialOffset(addr, 1);
1307 saoChromaComponentParamDist(saoParam, addr, rateDist, lambda, bestCost);
1308 }
1309 }
1310 if (saoParam->bSaoFlag[0] || saoParam->bSaoFlag[1])
1311 {
1312 // Cost of merge left or Up
1313 for (int mergeIdx = 0; mergeIdx < 2; ++mergeIdx)
1314 {
1315 if (!allowMerge[mergeIdx])
1316 continue;
1317
1318 int64_t mergeDist = 0;
1319 for (int plane = 0; plane < planes; plane++)
1320 {
1321 int64_t estDist = 0;
1322 SaoCtuParam* mergeSrcParam = &(saoParam->ctuParam[plane][addrMerge[mergeIdx]]);
1323 int typeIdx = mergeSrcParam->typeIdx;
1324 if (typeIdx >= 0)
1325 {
1326 int bandPos = (typeIdx == SAO_BO) ? mergeSrcParam->bandPos : 1;
1327 for (int classIdx = 0; classIdx < SAO_NUM_OFFSET; classIdx++)
1328 {
1329 int mergeOffset = mergeSrcParam->offset[classIdx];
1330 estDist += estSaoDist(m_count[plane][typeIdx][classIdx + bandPos], mergeOffset, m_offsetOrg[plane][typeIdx][classIdx + bandPos]);
1331 }
1332 }
1333 mergeDist += (estDist << 8) / lambda[!!plane];
1334 }
1335
1336 m_entropyCoder.load(m_rdContexts.cur);
1337 m_entropyCoder.resetBits();
1338 if (allowMerge[0])
1339 m_entropyCoder.codeSaoMerge(1 - mergeIdx);
1340 if (allowMerge[1] && (mergeIdx == 1))
1341 m_entropyCoder.codeSaoMerge(1);
1342
1343 uint32_t estRate = m_entropyCoder.getNumberOfWrittenBits();
1344 int64_t mergeCost = mergeDist + estRate;
1345 if (mergeCost < bestCost)
1346 {
1347 SaoMergeMode mergeMode = mergeIdx ? SAO_MERGE_UP : SAO_MERGE_LEFT;
1348 bestCost = mergeCost;
1349 m_entropyCoder.store(m_rdContexts.temp);
1350 for (int plane = 0; plane < planes; plane++)
1351 {
1352 if (saoParam->bSaoFlag[plane > 0])
1353 {
1354 SaoCtuParam* dstCtuParam = &saoParam->ctuParam[plane][addr];
1355 SaoCtuParam* mergeSrcParam = &(saoParam->ctuParam[plane][addrMerge[mergeIdx]]);
1356 dstCtuParam->mergeMode = mergeMode;
1357 dstCtuParam->typeIdx = mergeSrcParam->typeIdx;
1358 dstCtuParam->bandPos = mergeSrcParam->bandPos;
1359
1360 for (int i = 0; i < SAO_NUM_OFFSET; i++)
1361 dstCtuParam->offset[i] = mergeSrcParam->offset[i];
1362 }
1363 }
1364 }
1365 }
1366
1367 if (saoParam->ctuParam[0][addr].typeIdx < 0)
1368 m_numNoSao[0]++;
1369 if (chroma && saoParam->ctuParam[1][addr].typeIdx < 0)
1370 m_numNoSao[1]++;
1371 m_entropyCoder.load(m_rdContexts.temp);
1372 m_entropyCoder.store(m_rdContexts.cur);
1373 }

1435inline int64_t SAO::calcSaoRdoCost(int64_t distortion, uint32_t bits, int64_t lambda)
1436{
1437#if X265_DEPTH < 10
1438 X265_CHECK(bits <= (INT64_MAX - 128) / lambda,
1439 "calcRdCost wrap detected dist: " X265_LL ", bits %u, lambda: " X265_LL "\n",
1440 distortion, bits, lambda);
1441#else
1442 X265_CHECK(bits <= (INT64_MAX - 128) / lambda,
1443 "calcRdCost wrap detected dist: " X265_LL ", bits %u, lambda: " X265_LL "\n",
1444 distortion, bits, lambda);
1445#endif
1446 return distortion + ((bits * lambda + 128) >> 8);
1447}

1449void SAO::estIterOffset(int typeIdx, int64_t lambda, int32_t count, int32_t offsetOrg, int32_t& offset, int32_t& distClasses, int64_t& costClasses)
1450{
1451 int bestOffset = 0;
1452 distClasses = 0;
1453
1454 // Assuming sending quantized value 0 results in zero offset and sending the value zero needs 1 bit.
1455 // entropy coder can be used to measure the exact rate here.
1456 int64_t bestCost = calcSaoRdoCost(0, 1, lambda);
1457 while (offset != 0)
1458 {
1459 // Calculate the bits required for signalling the offset
1460 uint32_t rate = (typeIdx == SAO_BO) ? (abs(offset) + 2) : (abs(offset) + 1);
1461 if (abs(offset) == OFFSET_THRESH - 1)
1462 rate--;
1463
1464 // Do the dequntization before distorion calculation
1465 int64_t dist = estSaoDist(count, offset << SAO_BIT_INC, offsetOrg);
1466 int64_t cost = calcSaoRdoCost(dist, rate, lambda);
1467 if (cost < bestCost)
1468 {
1469 bestCost = cost;
1470 bestOffset = offset;
1471 distClasses = (int)dist;
1472 }
1473 offset = (offset > 0) ? (offset - 1) : (offset + 1);
1474 }
1475
1476 costClasses = bestCost;
1477 offset = bestOffset;
1478}
1479void SAO::saoLumaComponentParamDist(SAOParam* saoParam, int32_t addr, int64_t& rateDist, int64_t* lambda, int64_t &bestCost)

1588 bestDist = estDist;

1598 m_entropyCoder.load(m_rdContexts.temp);

1600 m_entropyCoder.store(m_rdContexts.temp);

1605 }

1780void saoCuStatsE0_c(const int16_t *diff, const pixel *rec, intptr_t stride, int endX, int endY, int32_t *stats, int32_t *count)
1781{

1783 int32_t tmp_count[SAO::NUM_EDGETYPE];
1784
1785 X265_CHECK(endX <= MAX_CU_SIZE, "endX too big\n");

1788 memset(tmp_count, 0, sizeof(tmp_count));
