25#ifndef X265_FRAMEFILTER_H
26#define X265_FRAMEFILTER_H
27
28#include "common.h"
29#include "frame.h"
30#include "deblock.h"
31#include "sao.h"
32#include "threadpool.h" // class BondedTaskGroup
33
34namespace X265_NS {
35// private x265 namespace
36
37class Encoder;
38class Entropy;
39class FrameEncoder;
40struct ThreadLocalData;
41
42// Manages the processing of a single frame loopfilter
43class FrameFilter
44{
45public:
46
47 x265_param* m_param;
48 Frame* m_frame;
49 int m_useSao;
50 FrameEncoder* m_frameEncoder;
51 int m_hChromaShift;
52 int m_vChromaShift;
53 int m_pad[2];
54
55 int m_numRows;
56 int m_numCols;
57 int m_saoRowDelay;
58 int m_lastHeight;
59 int m_lastWidth;
60
61 ThreadSafeInteger integralCompleted; /* check if integral calculation is completed in this row */
62
63 void* m_ssimBuf; /* Temp storage for ssim computation */
64
65#define MAX_PFILTER_CUS (4) /* maximum CUs for every thread */
66 class ParallelFilter : public Deblock
67 {
68 public:
69 uint32_t m_rowHeight;
70 int m_row;
71 uint32_t m_rowAddr;
72 FrameFilter* m_frameFilter;
73 FrameData* m_encData;
74 ParallelFilter* m_prevRow;
75 SAO m_sao;
76 ThreadSafeInteger m_lastCol; /* The column that next to process */
77 ThreadSafeInteger m_allowedCol; /* The column that processed from Encode pipeline */
78 ThreadSafeInteger m_lastDeblocked; /* The column that finished all of Deblock stages */
79
80 ParallelFilter()
81 : m_rowHeight(0)
82 , m_row(0)
83 , m_rowAddr(0)
84 , m_frameFilter(NULL)
85 , m_encData(NULL)
86 , m_prevRow(NULL)
87 {
88 }
89
90 ~ParallelFilter()
91 { }
92
93 void processTasks(int workerThreadId);
94
95 // Apply SAO on a CU in current row
96 void processSaoCTU(SAOParam *saoParam, int col);
97
98 // Copy and Save SAO reference pixels for SAO Rdo decide
99 void copySaoAboveRef(const CUData *ctu, PicYuv* reconPic, uint32_t cuAddr, int col);
100
101 // Post-Process (Border extension)
102 void processPostCu(int col) const;
103
104 uint32_t getCUHeight() const
105 {
106 return m_rowHeight;
107 }
108 };
109
110 ParallelFilter* m_parallelFilter;
111
112 FrameFilter()
113 : m_param(NULL)
114 , m_frame(NULL)
115 , m_frameEncoder(NULL)
116 , m_ssimBuf(NULL)
117 , m_parallelFilter(NULL)
118 {
119 }
120
121 uint32_t getCUWidth(int colNum) const
122 {
123 return (colNum == (int)m_numCols - 1) ? m_lastWidth : m_param->maxCUSize;
124 }
125
126 void init(Encoder *top, FrameEncoder *frame, int numRows, uint32_t numCols);
127 void destroy();
128
129 void start(Frame *pic, Entropy& initState);
130
131 void processRow(int row);
132 void processPostRow(int row);
133 void computeMEIntegral(int row);
134};
135}

137#endif // ifndef X265_FRAMEFILTER_H
