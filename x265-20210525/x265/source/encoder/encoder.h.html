111struct cuLocation
112{
113 bool skipWidth;
114 bool skipHeight;
115 uint32_t heightInCU;
116 uint32_t widthInCU;
117 uint32_t oddRowIndex;
118 uint32_t evenRowIndex;
119 uint32_t switchCondition;
120
121 void init(x265_param* param)
122 {
123 skipHeight = false;
124 skipWidth = false;
125 heightInCU = (param->sourceHeight + param->maxCUSize - 1) >> param->maxLog2CUSize;
126 widthInCU = (param->sourceWidth + param->maxCUSize - 1) >> param->maxLog2CUSize;
127 evenRowIndex = 0;
128 oddRowIndex = param->num4x4Partitions * widthInCU;
129 switchCondition = 0; // To switch between odd and even rows
130 }
131};
