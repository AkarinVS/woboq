31void BitCost::setQP(unsigned int qp)
32{
33 if (!s_costs[qp])
34 {
35 ScopedLock s(s_costCalcLock);
36
37 // Now that we have acquired the lock, check again if another thread calculated
38 // this row while we were blocked
39 if (!s_costs[qp])
40 {
41 x265_emms(); // just to be safe
42
43 CalculateLogs();
44 s_costs[qp] = X265_MALLOC(uint16_t, 4 * BC_MAX_MV + 1) + 2 * BC_MAX_MV;
45 if (!s_costs[qp])
46 {
47 x265_log(NULL, X265_LOG_ERROR, "BitCost s_costs buffer allocation failure\n");
48 return;
49 }
50 double lambda = x265_lambda_tab[qp];
51
52 // estimate same cost for negative and positive MVD
53 for (int i = 0; i <= 2 * BC_MAX_MV; i++)
54 s_costs[qp][i] = s_costs[qp][-i] = (uint16_t)X265_MIN(s_bitsizes[i] * lambda + 0.5f, (1 << 15) - 1);
55 }
56 }
57 for (int j = 0; j < 4; j++)
58 {
59 if (!s_fpelMvCosts[qp][j])
60 {
61 ScopedLock s(s_costCalcLock);
62 if (!s_fpelMvCosts[qp][j])
63 {
64 s_fpelMvCosts[qp][j] = X265_MALLOC(uint16_t, BC_MAX_MV + 1) + (BC_MAX_MV >> 1);
65 if (!s_fpelMvCosts[qp][j])
66 {
67 x265_log(NULL, X265_LOG_ERROR, "BitCost s_fpelMvCosts buffer allocation failure\n");
68 return;
69 }
70 for (int i = -(BC_MAX_MV >> 1); i < (BC_MAX_MV >> 1); i++)
71 {
72 s_fpelMvCosts[qp][j][i] = s_costs[qp][i * 4 + j];
73 }
74 }
75 }
76 }
77 m_cost = s_costs[qp];
78 for (int j = 0; j < 4; j++)
79 {
80 m_fpelMvCosts[j] = s_fpelMvCosts[qp][j];
81 }
82}

95void BitCost::CalculateLogs()
96{
97 if (!s_bitsizes)
98 {
99 s_bitsizes = X265_MALLOC(float, 4 * BC_MAX_MV + 1) + 2 * BC_MAX_MV;
100 if (!s_bitsizes)
101 {
102 x265_log(NULL, X265_LOG_ERROR, "BitCost s_bitsizes buffer allocation failure\n");
103 return;
104 }
105 s_bitsizes[0] = 0.718f;
106 float log2_2 = 2.0f / log(2.0f); // 2 x 1/log(2)
107 for (int i = 1; i <= 2 * BC_MAX_MV; i++)
108 s_bitsizes[i] = s_bitsizes[-i] = log((float)(i + 1)) * log2_2 + 1.718f;
109 }
110}

112void BitCost::destroy()
113{
114 for (int i = 0; i < BC_MAX_QP; i++)
115 {
116 if (s_costs[i])
117 {
118 X265_FREE(s_costs[i] - 2 * BC_MAX_MV);
119
120 s_costs[i] = NULL;
121 }
122 }
123 for (int i = 0; i < BC_MAX_QP; i++)
124 {
125 for (int j = 0; j < 4; j++)
126 {
127 if (s_fpelMvCosts[i][j])
128 {
129 X265_FREE(s_fpelMvCosts[i][j] - (BC_MAX_MV >> 1));
130 s_fpelMvCosts[i][j] = NULL;
131 }
132 }
133 }
134
135 if (s_bitsizes)
136 {
137 X265_FREE(s_bitsizes - 2 * BC_MAX_MV);
138 s_bitsizes = NULL;
139 }
140}
