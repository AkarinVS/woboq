37/* marshal a single SEI message sei, storing the marshalled representation
38* in bitstream bs */
39void SEI::writeSEImessages(Bitstream& bs, const SPS& sps, NalUnitType nalUnitType, NALList& list, int isNested)
40{
41 if (!isNested)
42 bs.resetBits();
43
44 BitCounter counter;
45 m_bitIf = &counter;
46 writeSEI(sps);
47 /* count the size of the payload and return the size in bits */
48 X265_CHECK(0 == (counter.getNumberOfWrittenBits() & 7), "payload unaligned\n");
49 uint32_t payloadData = counter.getNumberOfWrittenBits() >> 3;
50
51 // set bitstream
52 m_bitIf = &bs;
53
54 uint32_t payloadType = m_payloadType;
55 for (; payloadType >= 0xff; payloadType -= 0xff)
56 WRITE_CODE(0xff, 8, "payload_type");
57 WRITE_CODE(payloadType, 8, "payload_type");
58
59 uint32_t payloadSize = payloadData;
60 for (; payloadSize >= 0xff; payloadSize -= 0xff)
61 WRITE_CODE(0xff, 8, "payload_size");
62 WRITE_CODE(payloadSize, 8, "payload_size");
63
64 // virtual writeSEI method, write to bs
65 writeSEI(sps);
66
67 if (!isNested)
68 {
69 if (nalUnitType != NAL_UNIT_UNSPECIFIED)
70 bs.writeByteAlignment();
71 list.serialize(nalUnitType, bs);
72 }
73}

75void SEI::writeByteAlign()
76{
77 // TODO: expose bs.writeByteAlignment() as virtual function
78 if (m_bitIf->getNumberOfWrittenBits() % 8 != 0)
79 {
80 WRITE_FLAG(1, "bit_equal_to_one");
81 while (m_bitIf->getNumberOfWrittenBits() % 8 != 0)
82 {
83 WRITE_FLAG(0, "bit_equal_to_zero");
84 }
85 }
86}

93/* charSet = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/" */
94
95char* SEI::base64Decode(char encodedString[], int base64EncodeLength)
96{
97 char* decodedString;
98 decodedString = (char*)malloc(sizeof(char) * ((base64EncodeLength / 4) * 3));
99 int i, j, k = 0;
100 // stores the bitstream
101 int bitstream = 0;
102 // countBits stores current number of bits in bitstream
103 int countBits = 0;
104 // selects 4 characters from encodedString at a time. Find the position of each encoded character in charSet and stores in bitstream
105 for (i = 0; i < base64EncodeLength; i += 4)
106 {
107 bitstream = 0, countBits = 0;
108 for (j = 0; j < 4; j++)
109 {
110 // make space for 6 bits
111 if (encodedString[i + j] != '=')
112 {
113 bitstream = bitstream << 6;
114 countBits += 6;
115 }
116 // Finding the position of each encoded character in charSet and storing in bitstream, use OR '|' operator to store bits
117
118 if (encodedString[i + j] >= 'A' && encodedString[i + j] <= 'Z')
119 bitstream = bitstream | (encodedString[i + j] - 'A');
120
121 else if (encodedString[i + j] >= 'a' && encodedString[i + j] <= 'z')
122 bitstream = bitstream | (encodedString[i + j] - 'a' + 26);
123
124 else if (encodedString[i + j] >= '0' && encodedString[i + j] <= '9')
125 bitstream = bitstream | (encodedString[i + j] - '0' + 52);
126
127 // '+' occurs in 62nd position in charSet
128 else if (encodedString[i + j] == '+')
129 bitstream = bitstream | 62;
130
131 // '/' occurs in 63rd position in charSet
132 else if (encodedString[i + j] == '/')
133 bitstream = bitstream | 63;
134
135 // to delete appended bits during encoding
136 else
137 {
138 bitstream = bitstream >> 2;
139 countBits -= 2;
140 }
141 }
142
143 while (countBits != 0)
144 {
145 countBits -= 8;
146 decodedString[k++] = (bitstream >> countBits) & 255;
147 }
148 }
149 return decodedString;
150}
