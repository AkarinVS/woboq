393/* Used to pass pictures into the encoder, and to get picture data back out of
394 * the encoder. The input and output semantics are different */
395typedef struct x265_picture
396{
397 /* presentation time stamp: user-specified, returned on output */
398 int64_t pts;
399
400 /* display time stamp: ignored on input, copied from reordered pts. Returned
401 * on output */
402 int64_t dts;
403
404 /* force quantizer for != X265_QP_AUTO */
405 /* The value provided on input is returned with the same picture (POC) on
406 * output */
407 void* userData;
408
409 /* Must be specified on input pictures, the number of planes is determined
410 * by the colorSpace value */
411 void* planes[3];
412
413 /* Stride is the number of bytes between row starts */
414 int stride[3];
415
416 /* Must be specified on input pictures. x265_picture_init() will set it to
417 * the encoder's internal bit depth, but this field must describe the depth
418 * of the input pictures. Must be between 8 and 16. Values larger than 8
419 * imply 16bits per input sample. If input bit depth is larger than the
420 * internal bit depth, the encoder will down-shift pixels. Input samples
421 * larger than 8bits will be masked to internal bit depth. On output the
422 * bitDepth will be the internal encoder bit depth */
423 int bitDepth;
424
425 /* Must be specified on input pictures: X265_TYPE_AUTO or other.
426 * x265_picture_init() sets this to auto, returned on output */
427 int sliceType;
428
429 /* Ignored on input, set to picture count, returned on output */
430 int poc;
431
432 /* Must be specified on input pictures: X265_CSP_I420 or other. It must
433 * match the internal color space of the encoder. x265_picture_init() will
434 * initialize this value to the internal color space */
435 int colorSpace;
436
437 /* Force the slice base QP for this picture within the encoder. Set to 0
438 * to allow the encoder to determine base QP */
439 int forceqp;
440
441 /* If param.analysisLoad and param.analysisSave are disabled, this field is
442 * ignored on input and output. Else the user must call x265_alloc_analysis_data()
443 * to allocate analysis buffers for every picture passed to the encoder.
444 *
445 * On input when param.analysisLoad is enabled and analysisData
446 * member pointers are valid, the encoder will use the data stored here to
447 * reduce encoder work.
448 *
449 * On output when param.analysisSave is enabled and analysisData
450 * member pointers are valid, the encoder will write output analysis into
451 * this data structure */
452 x265_analysis_data analysisData;
453
454 /* An array of quantizer offsets to be applied to this image during encoding.
455 * These are added on top of the decisions made by rateControl.
456 * Adaptive quantization must be enabled to use this feature. These quantizer
457 * offsets should be given for each 16x16 block (8x8 block, when qg-size is 8).
458 * Behavior if quant offsets differ between encoding passes is undefined. */
459 float *quantOffsets;
460
461 /* Frame level statistics */
462 x265_frame_stats frameData;
463
464 /* User defined SEI */
465 x265_sei userSEI;
466
467 /* Ratecontrol statistics for collecting the ratecontrol information.
468 * It is not used for collecting the last pass ratecontrol data in
469 * multi pass ratecontrol mode. */
470 void* rcData;
471
472 size_t framesize;
473
474 int height;
475
476 // pts is reordered in the order of encoding.
477 int64_t reorderedPts;
478
479 //Dolby Vision RPU metadata
480 x265_dolby_vision_rpu rpu;
481
482 int fieldNum;
483
484 //SEI picture structure message
485 uint32_t picStruct;
486
487 int width;
488} x265_picture;

513#define X265_CPU_AVX (1 << 9) /* Requires OS support even if YMM registers aren't used. */
514#define X265_CPU_XOP (1 << 10) /* AMD XOP */
515#define X265_CPU_FMA4 (1 << 11) /* AMD FMA4 */
516#define X265_CPU_FMA3 (1 << 12) /* Intel FMA3 */
517#define X265_CPU_BMI1 (1 << 13) /* BMI1 */
518#define X265_CPU_BMI2 (1 << 14) /* BMI2 */

520#define X265_CPU_AVX512 (1 << 16) /* AVX-512 {F, CD, BW, DQ, VL}, requires OS support */
521/* x86 modifiers */
522#define X265_CPU_CACHELINE_32 (1 << 17) /* avoid memory loads that span the border between two cachelines */
523#define X265_CPU_CACHELINE_64 (1 << 18) /* 32/64 is the size of a cacheline in bytes */
524#define X265_CPU_SSE2_IS_SLOW (1 << 19) /* avoid most SSE2 functions on Athlon64 */
525#define X265_CPU_SSE2_IS_FAST (1 << 20) /* a few functions are only faster on Core2 and Phenom */
526#define X265_CPU_SLOW_SHUFFLE (1 << 21) /* The Conroe has a slow shuffle unit (relative to overall SSE performance) */
527#define X265_CPU_STACK_MOD4 (1 << 22) /* if stack is only mod4 and not mod16 */
528#define X265_CPU_SLOW_ATOM (1 << 23) /* The Atom is terrible: slow SSE unaligned loads, slow
529 * SIMD multiplies, slow SIMD variable shifts, slow pshufb,
530 * cacheline split penalties -- gather everything here that
531 * isn't shared by other CPUs to avoid making half a dozen
532 * new SLOW flags. */
533#define X265_CPU_SLOW_PSHUFB (1 << 24) /* such as on the Intel Atom */
534#define X265_CPU_SLOW_PALIGNR (1 << 25) /* such as on the AMD Bobcat */
