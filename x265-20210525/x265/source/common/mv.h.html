24#ifndef X265_MV_H
25#define X265_MV_H
26
27#include "common.h"
28#include "primitives.h"
29
30namespace X265_NS {
31// private x265 namespace
32
33#if _MSC_VER
34#pragma warning(disable: 4201) // non-standard extension used (nameless struct/union)
35#endif
36
37struct MV
38{
39public:
40
41 union {
42 struct { int32_t x, y; };
43
44 int64_t word;
45 };
46
47 MV() {}
48 MV(int64_t w) : word(w) {}
49 MV(int32_t _x, int32_t _y) : x(_x), y(_y) {}
50
51 MV& operator =(uint64_t w) { word = w; return *this; }
52
53 MV& operator +=(const MV& other) { x += other.x; y += other.y; return *this; }
54
55 MV& operator -=(const MV& other) { x -= other.x; y -= other.y; return *this; }
56
57 MV& operator >>=(int i) { x >>= i; y >>= i; return *this; }
58
59#if USING_FTRAPV
60 /* avoid signed left-shifts when -ftrapv is enabled */
61 MV& operator <<=(int i) { x *= (1 << i); y *= (1 << i); return *this; }
62 MV operator <<(int i) const { return MV(x * (1 << i), y * (1 << i)); }
63#else
64 MV& operator <<=(int i) { x <<= i; y <<= i; return *this; }
65 MV operator <<(int i) const { return MV(x << i, y << i); }
66#endif
67
68 MV operator >>(int i) const { return MV(x >> i, y >> i); }
69
70 MV operator *(int32_t i) const { return MV(x * i, y * i); }
71
72 MV operator -(const MV& other) const { return MV(x - other.x, y - other.y); }
73
74 MV operator +(const MV& other) const { return MV(x + other.x, y + other.y); }
75
76 bool operator ==(const MV& other) const { return word == other.word; }
77
78 bool operator !=(const MV& other) const { return word != other.word; }
79
80 bool operator !() const { return !word; }
81
82 // Scale down a QPEL mv to FPEL mv, rounding up by one HPEL offset
83 MV roundToFPel() const { return MV((x + 2) >> 2, (y + 2) >> 2); }
84
85 // Scale up an FPEL mv to QPEL by shifting up two bits
86 MV toQPel() const { return *this << 2; }
87
88 bool inline notZero() const { return this->word != 0; }
89
90 bool inline isSubpel() const { return (this->word & 0x0000000300000003ll) != 0; }
91
92 MV mvmin(const MV& m) const { return MV(x > m.x ? m.x : x, y > m.y ? m.y : y); }
93
94 MV mvmax(const MV& m) const { return MV(x < m.x ? m.x : x, y < m.y ? m.y : y); }
95
96 MV clipped(const MV& _min, const MV& _max) const
97 {
98 MV cl = mvmin(_max);
99
100 return cl.mvmax(_min);
101 }
102
103 // returns true if MV is within range (inclusive)
104 bool checkRange(const MV& _min, const MV& _max) const
105 {
106 return x >= _min.x && x <= _max.x && y >= _min.y && y <= _max.y;
107 }
108};
109}
