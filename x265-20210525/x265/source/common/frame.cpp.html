69bool Frame::create(x265_param *param, float* quantOffsets)
70{
71 m_fencPic = new PicYuv;
72 m_param = param;
73 CHECKED_MALLOC_ZERO(m_rcData, RcStats, 1);
74
75 if (param->bCTUInfo)
76 {
77 uint32_t widthInCTU = (m_param->sourceWidth + param->maxCUSize - 1) >> m_param->maxLog2CUSize;
78 uint32_t heightInCTU = (m_param->sourceHeight + param->maxCUSize - 1) >> m_param->maxLog2CUSize;
79 uint32_t numCTUsInFrame = widthInCTU * heightInCTU;
80 CHECKED_MALLOC_ZERO(m_addOnDepth, uint8_t *, numCTUsInFrame);
81 CHECKED_MALLOC_ZERO(m_addOnCtuInfo, uint8_t *, numCTUsInFrame);
82 CHECKED_MALLOC_ZERO(m_addOnPrevChange, int *, numCTUsInFrame);
83 for (uint32_t i = 0; i < numCTUsInFrame; i++)
84 {
85 CHECKED_MALLOC_ZERO(m_addOnDepth[i], uint8_t, uint32_t(param->num4x4Partitions));
86 CHECKED_MALLOC_ZERO(m_addOnCtuInfo[i], uint8_t, uint32_t(param->num4x4Partitions));
87 CHECKED_MALLOC_ZERO(m_addOnPrevChange[i], int, uint32_t(param->num4x4Partitions));
88 }
89 }
90
91 if (param->bAnalysisType == AVC_INFO)
92 {
93 m_analysisData.wt = NULL;
94 m_analysisData.intraData = NULL;
95 m_analysisData.interData = NULL;
96 m_analysisData.distortionData = NULL;
97 }
98
99 if (param->bDynamicRefine)
100 {
101 int size = m_param->maxCUDepth * X265_REFINE_INTER_LEVELS;
102 CHECKED_MALLOC_ZERO(m_classifyRd, uint64_t, size);
103 CHECKED_MALLOC_ZERO(m_classifyVariance, uint64_t, size);
104 CHECKED_MALLOC_ZERO(m_classifyCount, uint32_t, size);
105 }
106
107 if (param->rc.aqMode == X265_AQ_EDGE || (param->rc.zonefileCount && param->rc.aqMode != 0))
108 {
109 uint32_t numCuInWidth = (param->sourceWidth + param->maxCUSize - 1) / param->maxCUSize;
110 uint32_t numCuInHeight = (param->sourceHeight + param->maxCUSize - 1) / param->maxCUSize;
111 uint32_t m_lumaMarginX = param->maxCUSize + 32; // search margin and 8-tap filter half-length, padded for 32-byte alignment
112 uint32_t m_lumaMarginY = param->maxCUSize + 16; // margin for 8-tap filter and infinite padding
113 intptr_t m_stride = (numCuInWidth * param->maxCUSize) + (m_lumaMarginX << 1);
114 int maxHeight = numCuInHeight * param->maxCUSize;
115
116 m_edgePic = X265_MALLOC(pixel, m_stride * (maxHeight + (m_lumaMarginY * 2)));
117 m_gaussianPic = X265_MALLOC(pixel, m_stride * (maxHeight + (m_lumaMarginY * 2)));
118 m_thetaPic = X265_MALLOC(pixel, m_stride * (maxHeight + (m_lumaMarginY * 2)));
119 }
120
121 if (param->recursionSkipMode == EDGE_BASED_RSKIP)
122 {
123 uint32_t numCuInWidth = (param->sourceWidth + param->maxCUSize - 1) / param->maxCUSize;
124 uint32_t numCuInHeight = (param->sourceHeight + param->maxCUSize - 1) / param->maxCUSize;
125 uint32_t lumaMarginX = param->maxCUSize + 32;
126 uint32_t lumaMarginY = param->maxCUSize + 16;
127 uint32_t stride = (numCuInWidth * param->maxCUSize) + (lumaMarginX << 1);
128 uint32_t maxHeight = numCuInHeight * param->maxCUSize;
129 uint32_t bitPlaneSize = stride * (maxHeight + (lumaMarginY * 2));
130 CHECKED_MALLOC_ZERO(m_edgeBitPlane, pixel, bitPlaneSize);
131 m_edgeBitPic = m_edgeBitPlane + lumaMarginY * stride + lumaMarginX;
132 }
133
134 if (m_fencPic->create(param, !!m_param->bCopyPicToFrame) && m_lowres.create(param, m_fencPic, param->rc.qgSize))
135 {
136 X265_CHECK((m_reconColCount == NULL), "m_reconColCount was initialized");
137 m_numRows = (m_fencPic->m_picHeight + param->maxCUSize - 1) / param->maxCUSize;
138 m_reconRowFlag = new ThreadSafeInteger[m_numRows];
139 m_reconColCount = new ThreadSafeInteger[m_numRows];
140
141 if (quantOffsets)
142 {
143 int32_t cuCount = (param->rc.qgSize == 8) ? m_lowres.maxBlocksInRowFullRes * m_lowres.maxBlocksInColFullRes :
144 m_lowres.maxBlocksInRow * m_lowres.maxBlocksInCol;
145 m_quantOffsets = new float[cuCount];
146 }
147 return true;
148 }
149 return false;
150fail:
151 return false;
152}

154bool Frame::allocEncodeData(x265_param *param, const SPS& sps)
155{
156 m_encData = new FrameData;
157 m_reconPic = new PicYuv;
158 m_param = param;
159 m_encData->m_reconPic = m_reconPic;
160 bool ok = m_encData->create(*param, sps, m_fencPic->m_picCsp) && m_reconPic->create(param);
161 if (ok)
162 {
163 /* initialize right border of m_reconpicYuv as SAO may read beyond the
164 * end of the picture accessing uninitialized pixels */
165 int maxHeight = sps.numCuInHeight * param->maxCUSize;
166 memset(m_reconPic->m_picOrg[0], 0, sizeof(pixel)* m_reconPic->m_stride * maxHeight);
167
168 /* use pre-calculated cu/pu offsets cached in the SPS structure */
169 m_reconPic->m_cuOffsetY = sps.cuOffsetY;
170 m_reconPic->m_buOffsetY = sps.buOffsetY;
171
172 if (param->internalCsp != X265_CSP_I400)
173 {
174 memset(m_reconPic->m_picOrg[1], 0, sizeof(pixel) * m_reconPic->m_strideC * (maxHeight >> m_reconPic->m_vChromaShift));
175 memset(m_reconPic->m_picOrg[2], 0, sizeof(pixel) * m_reconPic->m_strideC * (maxHeight >> m_reconPic->m_vChromaShift));
176
177 /* use pre-calculated cu/pu offsets cached in the SPS structure */
178 m_reconPic->m_cuOffsetC = sps.cuOffsetC;
179 m_reconPic->m_buOffsetC = sps.buOffsetC;
180 }
181 }
182 return ok;
183}

185/* prepare to re-use a FrameData instance to encode a new picture */
186void Frame::reinit(const SPS& sps)
187{
188 m_bChromaExtended = false;
189 m_reconPic = m_encData->m_reconPic;
190 m_encData->reinit(sps);
191}

193void Frame::destroy()
194{
195 if (m_encData)
196 {
197 m_encData->destroy();
198 delete m_encData;
199 m_encData = NULL;
200 }
201
202 if (m_fencPic)
203 {
204 if (m_param->bCopyPicToFrame)
205 m_fencPic->destroy();
206 delete m_fencPic;
207 m_fencPic = NULL;
208 }
209
210 if (m_reconPic)
211 {
212 m_reconPic->destroy();
213 delete m_reconPic;
214 m_reconPic = NULL;
215 }
216
217 if (m_reconRowFlag)
218 {
219 delete[] m_reconRowFlag;
220 m_reconRowFlag = NULL;
221 }
222
223 if (m_reconColCount)
224 {
225 delete[] m_reconColCount;
226 m_reconColCount = NULL;
227 }
228
229 if (m_quantOffsets)
230 {
231 delete[] m_quantOffsets;
232 }
233
234 if (m_userSEI.numPayloads)
235 {
236 for (int i = 0; i < m_userSEI.numPayloads; i++)
237 delete[] m_userSEI.payloads[i].payload;
238 delete[] m_userSEI.payloads;
239 }
240
241 if (m_ctuInfo)
242 {
243 uint32_t widthInCU = (m_param->sourceWidth + m_param->maxCUSize - 1) >> m_param->maxLog2CUSize;
244 uint32_t heightInCU = (m_param->sourceHeight + m_param->maxCUSize - 1) >> m_param->maxLog2CUSize;
245 uint32_t numCUsInFrame = widthInCU * heightInCU;
246 for (uint32_t i = 0; i < numCUsInFrame; i++)
247 {
248 X265_FREE((*m_ctuInfo + i)->ctuInfo);
249 (*m_ctuInfo + i)->ctuInfo = NULL;
250 X265_FREE(m_addOnDepth[i]);
251 m_addOnDepth[i] = NULL;
252 X265_FREE(m_addOnCtuInfo[i]);
253 m_addOnCtuInfo[i] = NULL;
254 X265_FREE(m_addOnPrevChange[i]);
255 m_addOnPrevChange[i] = NULL;
256 }
257 X265_FREE(*m_ctuInfo);
258 *m_ctuInfo = NULL;
259 X265_FREE(m_ctuInfo);
260 m_ctuInfo = NULL;
261 X265_FREE(m_prevCtuInfoChange);
262 m_prevCtuInfoChange = NULL;
263 X265_FREE(m_addOnDepth);
264 m_addOnDepth = NULL;
265 X265_FREE(m_addOnCtuInfo);
266 m_addOnCtuInfo = NULL;
267 X265_FREE(m_addOnPrevChange);
268 m_addOnPrevChange = NULL;
269 }
270 m_lowres.destroy();
271 X265_FREE(m_rcData);
272
273 if (m_param->bDynamicRefine)
274 {
275 X265_FREE_ZERO(m_classifyRd);
276 X265_FREE_ZERO(m_classifyVariance);
277 X265_FREE_ZERO(m_classifyCount);
278 }
279
280 if (m_param->rc.aqMode == X265_AQ_EDGE || (m_param->rc.zonefileCount && m_param->rc.aqMode != 0))
281 {
282 X265_FREE(m_edgePic);
283 X265_FREE(m_gaussianPic);
284 X265_FREE(m_thetaPic);
285 }
286
287 if (m_param->recursionSkipMode == EDGE_BASED_RSKIP)
288 {
289 X265_FREE_ZERO(m_edgeBitPlane);
290 m_edgeBitPic = NULL;
291 }
292}
