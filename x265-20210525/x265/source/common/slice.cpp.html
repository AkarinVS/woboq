32void Slice::setRefPicList(PicList& picList)
33{
34 if (m_sliceType == I_SLICE)
35 {
36 memset(m_refFrameList, 0, sizeof(m_refFrameList));
37 memset(m_refReconPicList, 0, sizeof(m_refReconPicList));
38 memset(m_refPOCList, 0, sizeof(m_refPOCList));
39 m_numRefIdx[1] = m_numRefIdx[0] = 0;
40 return;
41 }
42
43 Frame* refPic = NULL;
44 Frame* refPicSetStCurr0[MAX_NUM_REF];
45 Frame* refPicSetStCurr1[MAX_NUM_REF];
46 Frame* refPicSetLtCurr[MAX_NUM_REF];
47 int numPocStCurr0 = 0;
48 int numPocStCurr1 = 0;
49 int numPocLtCurr = 0;
50 int i;
51
52 for (i = 0; i < m_rps.numberOfNegativePictures; i++)
53 {
54 if (m_rps.bUsed[i])
55 {
56 refPic = picList.getPOC(m_poc + m_rps.deltaPOC[i]);
57 refPicSetStCurr0[numPocStCurr0] = refPic;
58 numPocStCurr0++;
59 }
60 }
61
62 for (; i < m_rps.numberOfNegativePictures + m_rps.numberOfPositivePictures; i++)
63 {
64 if (m_rps.bUsed[i])
65 {
66 refPic = picList.getPOC(m_poc + m_rps.deltaPOC[i]);
67 refPicSetStCurr1[numPocStCurr1] = refPic;
68 numPocStCurr1++;
69 }
70 }
71
72 X265_CHECK(m_rps.numberOfPictures == m_rps.numberOfNegativePictures + m_rps.numberOfPositivePictures,
73 "unexpected picture in RPS\n");
74
75 // ref_pic_list_init
76 Frame* rpsCurrList0[MAX_NUM_REF + 1];
77 Frame* rpsCurrList1[MAX_NUM_REF + 1];
78 int numPocTotalCurr = numPocStCurr0 + numPocStCurr1 + numPocLtCurr;
79
80 int cIdx = 0;
81 for (i = 0; i < numPocStCurr0; i++, cIdx++)
82 rpsCurrList0[cIdx] = refPicSetStCurr0[i];
83
84 for (i = 0; i < numPocStCurr1; i++, cIdx++)
85 rpsCurrList0[cIdx] = refPicSetStCurr1[i];
86
87 for (i = 0; i < numPocLtCurr; i++, cIdx++)
88 rpsCurrList0[cIdx] = refPicSetLtCurr[i];
89
90 X265_CHECK(cIdx == numPocTotalCurr, "RPS index check fail\n");
91
92 if (m_sliceType == B_SLICE)
93 {
94 cIdx = 0;
95 for (i = 0; i < numPocStCurr1; i++, cIdx++)
96 rpsCurrList1[cIdx] = refPicSetStCurr1[i];
97
98 for (i = 0; i < numPocStCurr0; i++, cIdx++)
99 rpsCurrList1[cIdx] = refPicSetStCurr0[i];
100
101 for (i = 0; i < numPocLtCurr; i++, cIdx++)
102 rpsCurrList1[cIdx] = refPicSetLtCurr[i];
103
104 X265_CHECK(cIdx == numPocTotalCurr, "RPS index check fail\n");
105 }
106
107 for (int rIdx = 0; rIdx < m_numRefIdx[0]; rIdx++)
108 {
109 cIdx = rIdx % numPocTotalCurr;
110 X265_CHECK(cIdx >= 0 && cIdx < numPocTotalCurr, "RPS index check fail\n");
111 m_refFrameList[0][rIdx] = rpsCurrList0[cIdx];
112 }
113
114 if (m_sliceType != B_SLICE)
115 {
116 m_numRefIdx[1] = 0;
117 memset(m_refFrameList[1], 0, sizeof(m_refFrameList[1]));
118 }
119 else
120 {
121 for (int rIdx = 0; rIdx < m_numRefIdx[1]; rIdx++)
122 {
123 cIdx = rIdx % numPocTotalCurr;
124 X265_CHECK(cIdx >= 0 && cIdx < numPocTotalCurr, "RPS index check fail\n");
125 m_refFrameList[1][rIdx] = rpsCurrList1[cIdx];
126 }
127 }
128
129 for (int dir = 0; dir < 2; dir++)
130 for (int numRefIdx = 0; numRefIdx < m_numRefIdx[dir]; numRefIdx++)
131 m_refPOCList[dir][numRefIdx] = m_refFrameList[dir][numRefIdx]->m_poc;
132}

134void Slice::disableWeights()
135{
136 for (int l = 0; l < 2; l++)
137 for (int i = 0; i < MAX_NUM_REF; i++)
138 for (int yuv = 0; yuv < 3; yuv++)
139 {
140 WeightParam& wp = m_weightPredTable[l][i][yuv];
141 wp.wtPresent = 0;
142 wp.log2WeightDenom = 0;
143 wp.inputWeight = 1;
144 wp.inputOffset = 0;
145 }
146}

148/* Sorts the deltaPOC and Used by current values in the RPS based on the
149 * deltaPOC values. deltaPOC values are sorted with -ve values before the +ve
150 * values. -ve values are in decreasing order. +ve values are in increasing
151 * order */
152void RPS::sortDeltaPOC()
153{
154 // sort in increasing order (smallest first)
155 for (int j = 1; j < numberOfPictures; j++)
156 {
157 int dPOC = deltaPOC[j];
158 bool used = bUsed[j];
159 for (int k = j - 1; k >= 0; k--)
160 {
161 int temp = deltaPOC[k];
162 if (dPOC < temp)
163 {
164 deltaPOC[k + 1] = temp;
165 bUsed[k + 1] = bUsed[k];
166 deltaPOC[k] = dPOC;
167 bUsed[k] = used;
168 }
169 }
170 }
171
172 // flip the negative values to largest first
173 int numNegPics = numberOfNegativePictures;
174 for (int j = 0, k = numNegPics - 1; j < numNegPics >> 1; j++, k--)
175 {
176 int dPOC = deltaPOC[j];
177 bool used = bUsed[j];
178 deltaPOC[j] = deltaPOC[k];
179 bUsed[j] = bUsed[k];
180 deltaPOC[k] = dPOC;
181 bUsed[k] = used;
182 }
183}

185uint32_t Slice::realEndAddress(uint32_t endCUAddr) const
186{
187 // Calculate end address
188 uint32_t internalAddress = (endCUAddr - 1) % m_param->num4x4Partitions;
189 uint32_t externalAddress = (endCUAddr - 1) / m_param->num4x4Partitions;
190 uint32_t xmax = m_sps->picWidthInLumaSamples - (externalAddress % m_sps->numCuInWidth) * m_param->maxCUSize;
191 uint32_t ymax = m_sps->picHeightInLumaSamples - (externalAddress / m_sps->numCuInWidth) * m_param->maxCUSize;
192
193 while (g_zscanToPelX[internalAddress] >= xmax || g_zscanToPelY[internalAddress] >= ymax)
194 internalAddress--;
195
196 internalAddress++;
197 if (internalAddress == m_param->num4x4Partitions)
198 {
199 internalAddress = 0;
200 externalAddress++;
201 }
202
203 return externalAddress * m_param->num4x4Partitions + internalAddress;
204}
