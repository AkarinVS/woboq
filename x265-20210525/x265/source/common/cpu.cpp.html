45static sigjmp_buf jmpbuf;
46static volatile sig_atomic_t canjump = 0;
47
48static void sigill_handler(int sig)
49{
50 if (!canjump)
51 {
52 signal(sig, SIG_DFL);
53 raise(sig);
54 }
55
56 canjump = 0;
57 siglongjmp(jmpbuf, 1);
58}

133uint32_t cpu_detect(bool benableavx512 )
134{
135
136 uint32_t cpu = 0;
137 uint32_t eax, ebx, ecx, edx;
138 uint32_t vendor[4] = { 0 };
139 uint32_t max_extended_cap, max_basic_cap;
140 uint64_t xcr0 = 0;
141
142#if !X86_64
143 if (!PFX(cpu_cpuid_test)())
144 return 0;
145#endif
146
147 PFX(cpu_cpuid)(0, &max_basic_cap, vendor + 0, vendor + 2, vendor + 1);
148 if (max_basic_cap == 0)
149 return 0;
150
151 PFX(cpu_cpuid)(1, &eax, &ebx, &ecx, &edx);
152 if (edx & 0x00800000)
153 cpu |= X265_CPU_MMX;
154 else
155 return cpu;
156 if (edx & 0x02000000)
157 cpu |= X265_CPU_MMX2 | X265_CPU_SSE;
158 if (edx & 0x04000000)
159 cpu |= X265_CPU_SSE2;
160 if (ecx & 0x00000001)
161 cpu |= X265_CPU_SSE3;
162 if (ecx & 0x00000200)
163 cpu |= X265_CPU_SSSE3 | X265_CPU_SSE2_IS_FAST;
164 if (ecx & 0x00080000)
165 cpu |= X265_CPU_SSE4;
166 if (ecx & 0x00100000)
167 cpu |= X265_CPU_SSE42;
168
169 if (ecx & 0x08000000) /* XGETBV supported and XSAVE enabled by OS */
170 {
171 /* Check for OS support */
172 xcr0 = PFX(cpu_xgetbv)(0);
173 if ((xcr0 & 0x6) == 0x6) /* XMM/YMM state */
174 {
175 if (ecx & 0x10000000)
176 cpu |= X265_CPU_AVX;
177 if (ecx & 0x00001000)
178 cpu |= X265_CPU_FMA3;
179 }
180 }
181
182 if (max_basic_cap >= 7)
183 {
184 PFX(cpu_cpuid)(7, &eax, &ebx, &ecx, &edx);
185 /* AVX2 requires OS support, but BMI1/2 don't. */
186 if (ebx & 0x00000008)
187 cpu |= X265_CPU_BMI1;
188 if (ebx & 0x00000100)
189 cpu |= X265_CPU_BMI2;
190
191 if ((xcr0 & 0x6) == 0x6) /* XMM/YMM state */
192 {
193 if (ebx & 0x00000020)
194 cpu |= X265_CPU_AVX2;
195 if (benableavx512)
196 {
197 if ((xcr0 & 0xE0) == 0xE0) /* OPMASK/ZMM state */
198 {
199 if ((ebx & 0xD0030000) == 0xD0030000)
200 {
201 cpu |= X265_CPU_AVX512;
202 enable512 = true;
203 }
204 }
205 }
206 }
207 }
208
209 PFX(cpu_cpuid)(0x80000000, &eax, &ebx, &ecx, &edx);
210 max_extended_cap = eax;
211
212 if (max_extended_cap >= 0x80000001)
213 {
214 PFX(cpu_cpuid)(0x80000001, &eax, &ebx, &ecx, &edx);
215
216 if (ecx & 0x00000020)
217 cpu |= X265_CPU_LZCNT; /* Supported by Intel chips starting with Haswell */
218 if (ecx & 0x00000040) /* SSE4a, AMD only */
219 {
220 int family = ((eax >> 8) & 0xf) + ((eax >> 20) & 0xff);
221 cpu |= X265_CPU_SSE2_IS_FAST; /* Phenom and later CPUs have fast SSE units */
222 if (family == 0x14)
223 {
224 cpu &= ~X265_CPU_SSE2_IS_FAST; /* SSSE3 doesn't imply fast SSE anymore... */
225 cpu |= X265_CPU_SSE2_IS_SLOW; /* Bobcat has 64-bit SIMD units */
226 cpu |= X265_CPU_SLOW_PALIGNR; /* palignr is insanely slow on Bobcat */
227 }
228 if (family == 0x16)
229 {
230 cpu |= X265_CPU_SLOW_PSHUFB; /* Jaguar's pshufb isn't that slow, but it's slow enough
231 * compared to alternate instruction sequences that this
232 * is equal or faster on almost all such functions. */
233 }
234 }
235
236 if (cpu & X265_CPU_AVX)
237 {
238 if (ecx & 0x00000800) /* XOP */
239 cpu |= X265_CPU_XOP;
240 if (ecx & 0x00010000) /* FMA4 */
241 cpu |= X265_CPU_FMA4;
242 }
243
244 if (!strcmp((char*)vendor, "AuthenticAMD"))
245 {
246 if (edx & 0x00400000)
247 cpu |= X265_CPU_MMX2;
248 if ((cpu & X265_CPU_SSE2) && !(cpu & X265_CPU_SSE2_IS_FAST))
249 cpu |= X265_CPU_SSE2_IS_SLOW; /* AMD CPUs come in two types: terrible at SSE and great at it */
250 }
251 }
252
253 if (!strcmp((char*)vendor, "GenuineIntel"))
254 {
255 PFX(cpu_cpuid)(1, &eax, &ebx, &ecx, &edx);
256 int family = ((eax >> 8) & 0xf) + ((eax >> 20) & 0xff);
257 int model = ((eax >> 4) & 0xf) + ((eax >> 12) & 0xf0);
258 if (family == 6)
259 {
260 /* Detect Atom CPU */
261 if (model == 28)
262 {
263 cpu |= X265_CPU_SLOW_ATOM;
264 cpu |= X265_CPU_SLOW_PSHUFB;
265 }
266
267 /* Conroe has a slow shuffle unit. Check the model number to make sure not
268 * to include crippled low-end Penryns and Nehalems that don't have SSE4. */
269 else if ((cpu & X265_CPU_SSSE3) && !(cpu & X265_CPU_SSE4) && model < 23)
270 cpu |= X265_CPU_SLOW_SHUFFLE;
271 }
272 }
273
274 if ((!strcmp((char*)vendor, "GenuineIntel") || !strcmp((char*)vendor, "CyrixInstead")) && !(cpu & X265_CPU_SSE42))
275 {
276 /* cacheline size is specified in 3 places, any of which may be missing */
277 PFX(cpu_cpuid)(1, &eax, &ebx, &ecx, &edx);
278 int cache = (ebx & 0xff00) >> 5; // cflush size
279 if (!cache && max_extended_cap >= 0x80000006)
280 {
281 PFX(cpu_cpuid)(0x80000006, &eax, &ebx, &ecx, &edx);
282 cache = ecx & 0xff; // cacheline size
283 }
284 if (!cache && max_basic_cap >= 2)
285 {
286 // Cache and TLB Information
287 static const char cache32_ids[] = { '\x0a','\x0c','\x41','\x42','\x43','\x44','\x45','\x82','\x83','\x84','\x85','\0' };
288 static const char cache64_ids[] = { '\x22','\x23','\x25','\x29','\x2c','\x46','\x47','\x49','\x60','\x66','\x67',
289 '\x68','\x78','\x79','\x7a','\x7b','\x7c','\x7c','\x7f','\x86','\x87','\0' };
290 uint32_t buf[4];
291 int max, i = 0;
292 do
293 {
294 PFX(cpu_cpuid)(2, buf + 0, buf + 1, buf + 2, buf + 3);
295 max = buf[0] & 0xff;
296 buf[0] &= ~0xff;
297 for (int j = 0; j < 4; j++)
298 {
299 if (!(buf[j] >> 31))
300 while (buf[j])
301 {
302 if (strchr(cache32_ids, buf[j] & 0xff))
303 cache = 32;
304 if (strchr(cache64_ids, buf[j] & 0xff))
305 cache = 64;
306 buf[j] >>= 8;
307 }
308 }
309 }
310 while (++i < max);
311 }
312
313 if (cache == 32)
314 cpu |= X265_CPU_CACHELINE_32;
315 else if (cache == 64)
316 cpu |= X265_CPU_CACHELINE_64;
317 else
318 x265_log(NULL, X265_LOG_WARNING, "unable to determine cacheline size\n");
319 }
320
321#if BROKEN_STACK_ALIGNMENT
322 cpu |= X265_CPU_STACK_MOD4;
323#endif
324
325 return cpu;
326}

335uint32_t cpu_detect(bool benableavx512)
336{
337 int flags = 0;
338
339#if HAVE_ARMV6
340 flags |= X265_CPU_ARMV6;
341
342 // don't do this hack if compiled with -mfpu=neon
343#if !HAVE_NEON
344 static void (* oldsig)(int);
345 oldsig = signal(SIGILL, sigill_handler);
346 if (sigsetjmp(jmpbuf, 1))
347 {
348 signal(SIGILL, oldsig);
349 return flags;
350 }
351
352 canjump = 1;
353 PFX(cpu_neon_test)();
354 canjump = 0;
355 signal(SIGILL, oldsig);
356#endif // if !HAVE_NEON
357
358 flags |= X265_CPU_NEON;
359
360 // fast neon -> arm (Cortex-A9) detection relies on user access to the
361 // cycle counter; this assumes ARMv7 performance counters.
362 // NEON requires at least ARMv7, ARMv8 may require changes here, but
363 // hopefully this hacky detection method will have been replaced by then.
364 // Note that there is potential for a race condition if another program or
365 // x264 instance disables or reinits the counters while x264 is using them,
366 // which may result in incorrect detection and the counters stuck enabled.
367 // right now Apple does not seem to support performance counters for this test
368#ifndef __MACH__
369 flags |= PFX(cpu_fast_neon_mrc_test)() ? X265_CPU_FAST_NEON_MRC : 0;
370#endif
371 // TODO: write dual issue test? currently it's A8 (dual issue) vs. A9 (fast mrc)
372#elif X265_ARCH_ARM64
373 flags |= X265_CPU_NEON;
374#endif // if HAVE_ARMV6
375 return flags;
376}
