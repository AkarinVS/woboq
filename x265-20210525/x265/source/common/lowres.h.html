65 /* lowres motion compensation, you must provide a buffer and stride for QPEL averaged pixels
66 * in case QPEL is required. Else it returns a pointer to the HPEL pixels */
67 inline pixel *lowresMC(intptr_t blockOffset, const MV& qmv, pixel *buf, intptr_t& outstride, bool hme)
68 {
69 intptr_t YStride = hme ? lumaStride / 2 : lumaStride;
70 pixel *plane[4];
71 for (int i = 0; i < 4; i++)
72 {
73 plane[i] = hme ? lowerResPlane[i] : lowresPlane[i];
74 }
75 if ((qmv.x | qmv.y) & 1)
76 {
77 int hpelA = (qmv.y & 2) | ((qmv.x & 2) >> 1);
78 pixel *frefA = plane[hpelA] + blockOffset + (qmv.x >> 2) + (qmv.y >> 2) * YStride;
79 int qmvx = qmv.x + (qmv.x & 1);
80 int qmvy = qmv.y + (qmv.y & 1);
81 int hpelB = (qmvy & 2) | ((qmvx & 2) >> 1);
82 pixel *frefB = plane[hpelB] + blockOffset + (qmvx >> 2) + (qmvy >> 2) * YStride;
83 primitives.pu[LUMA_8x8].pixelavg_pp[(outstride % 64 == 0) && (YStride % 64 == 0)](buf, outstride, frefA, YStride, frefB, YStride, 32);
84 return buf;
85 }
86 else
87 {
88 outstride = YStride;
89 int hpel = (qmv.y & 2) | ((qmv.x & 2) >> 1);
90 return plane[hpel] + blockOffset + (qmv.x >> 2) + (qmv.y >> 2) * YStride;
91 }
92 }

94 inline int lowresQPelCost(pixel *fenc, intptr_t blockOffset, const MV& qmv, pixelcmp_t comp, bool hme)
95 {
96 intptr_t YStride = hme ? lumaStride / 2 : lumaStride;
97 pixel *plane[4];
98 for (int i = 0; i < 4; i++)
99 {
100 plane[i] = hme ? lowerResPlane[i] : lowresPlane[i];
101 }
102 if ((qmv.x | qmv.y) & 1)
103 {
104 ALIGN_VAR_16(pixel, subpelbuf[8 * 8]);
105 int hpelA = (qmv.y & 2) | ((qmv.x & 2) >> 1);
106 pixel *frefA = plane[hpelA] + blockOffset + (qmv.x >> 2) + (qmv.y >> 2) * YStride;
107 int qmvx = qmv.x + (qmv.x & 1);
108 int qmvy = qmv.y + (qmv.y & 1);
109 int hpelB = (qmvy & 2) | ((qmvx & 2) >> 1);
110 pixel *frefB = plane[hpelB] + blockOffset + (qmvx >> 2) + (qmvy >> 2) * YStride;
111 primitives.pu[LUMA_8x8].pixelavg_pp[NONALIGNED](subpelbuf, 8, frefA, YStride, frefB, YStride, 32);
112 return comp(fenc, FENC_STRIDE, subpelbuf, 8);
113 }
114 else
115 {
116 int hpel = (qmv.y & 2) | ((qmv.x & 2) >> 1);
117 pixel *fref = plane[hpel] + blockOffset + (qmv.x >> 2) + (qmv.y >> 2) * YStride;
118 return comp(fenc, FENC_STRIDE, fref, YStride);
119 }
120 }
