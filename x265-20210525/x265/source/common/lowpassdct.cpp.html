29/* standard dct transformations */
30static dct_t* s_dct4x4;
31static dct_t* s_dct8x8;
32static dct_t* s_dct16x16;
33
34static void lowPassDct8_c(const int16_t* src, int16_t* dst, intptr_t srcStride)
35{
36 ALIGN_VAR_32(int16_t, coef[4 * 4]);
37 ALIGN_VAR_32(int16_t, avgBlock[4 * 4]);
38 int16_t totalSum = 0;
39 int16_t sum = 0;
40
41 for (int i = 0; i < 4; i++)
42 for (int j =0; j < 4; j++)
43 {
44 // Calculate average of 2x2 cells
45 sum = src[2*i*srcStride + 2*j] + src[2*i*srcStride + 2*j + 1]
46 + src[(2*i+1)*srcStride + 2*j] + src[(2*i+1)*srcStride + 2*j + 1];
47 avgBlock[i*4 + j] = sum >> 2;
48
49 totalSum += sum; // use to calculate total block average
50 }
51
52 //dct4
53 (*s_dct4x4)(avgBlock, coef, 4);
54 memset(dst, 0, 64 * sizeof(int16_t));
55 for (int i = 0; i < 4; i++)
56 {
57 memcpy(&dst[i * 8], &coef[i * 4], 4 * sizeof(int16_t));
58 }
59
60 // replace first coef with total block average
61 dst[0] = totalSum << 1;
62}

64static void lowPassDct16_c(const int16_t* src, int16_t* dst, intptr_t srcStride)
65{
66 ALIGN_VAR_32(int16_t, coef[8 * 8]);
67 ALIGN_VAR_32(int16_t, avgBlock[8 * 8]);
68 int32_t totalSum = 0;
69 int16_t sum = 0;
70 for (int i = 0; i < 8; i++)
71 for (int j =0; j < 8; j++)
72 {
73 sum = src[2*i*srcStride + 2*j] + src[2*i*srcStride + 2*j + 1]
74 + src[(2*i+1)*srcStride + 2*j] + src[(2*i+1)*srcStride + 2*j + 1];
75 avgBlock[i*8 + j] = sum >> 2;
76
77 totalSum += sum;
78 }
79
80 (*s_dct8x8)(avgBlock, coef, 8);
81 memset(dst, 0, 256 * sizeof(int16_t));
82 for (int i = 0; i < 8; i++)
83 {
84 memcpy(&dst[i * 16], &coef[i * 8], 8 * sizeof(int16_t));
85 }
86 dst[0] = static_cast<int16_t>(totalSum >> 1);
87}
88
89static void lowPassDct32_c(const int16_t* src, int16_t* dst, intptr_t srcStride)
90{
91 ALIGN_VAR_32(int16_t, coef[16 * 16]);
92 ALIGN_VAR_32(int16_t, avgBlock[16 * 16]);
93 int32_t totalSum = 0;
94 int16_t sum = 0;
95 for (int i = 0; i < 16; i++)
96 for (int j =0; j < 16; j++)
97 {
98 sum = src[2*i*srcStride + 2*j] + src[2*i*srcStride + 2*j + 1]
99 + src[(2*i+1)*srcStride + 2*j] + src[(2*i+1)*srcStride + 2*j + 1];
100 avgBlock[i*16 + j] = sum >> 2;
101
102 totalSum += sum;
103 }
104
105 (*s_dct16x16)(avgBlock, coef, 16);
106 memset(dst, 0, 1024 * sizeof(int16_t));
107 for (int i = 0; i < 16; i++)
108 {
109 memcpy(&dst[i * 32], &coef[i * 16], 16 * sizeof(int16_t));
110 }
111 dst[0] = static_cast<int16_t>(totalSum >> 3);
112}

117void setupLowPassPrimitives_c(EncoderPrimitives& p)
118{
119 s_dct4x4 = &(p.cu[BLOCK_4x4].standard_dct);
120 s_dct8x8 = &(p.cu[BLOCK_8x8].standard_dct);
121 s_dct16x16 = &(p.cu[BLOCK_16x16].standard_dct);
122
123 p.cu[BLOCK_8x8].lowpass_dct = lowPassDct8_c;
124 p.cu[BLOCK_16x16].lowpass_dct = lowPassDct16_c;
125 p.cu[BLOCK_32x32].lowpass_dct = lowPassDct32_c;
126}
