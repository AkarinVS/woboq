40template<int width, int height>
41void filterPixelToShort_c(const pixel* src, intptr_t srcStride, int16_t* dst, intptr_t dstStride)
42{
43 int shift = IF_INTERNAL_PREC - X265_DEPTH;
44 int row, col;
45
46 for (row = 0; row < height; row++)
47 {
48 for (col = 0; col < width; col++)
49 {
50 int16_t val = src[col] << shift;
51 dst[col] = val - (int16_t)IF_INTERNAL_OFFS;
52 }
53
54 src += srcStride;
55 dst += dstStride;
56 }
57}

59static void extendCURowColBorder(pixel* txt, intptr_t stride, int width, int height, int marginX)
60{
61 for (int y = 0; y < height; y++)
62 {
63#if HIGH_BIT_DEPTH
64 for (int x = 0; x < marginX; x++)
65 {
66 txt[-marginX + x] = txt[0];
67 txt[width + x] = txt[width - 1];
68 }
69
70#else
71 memset(txt - marginX, txt[0], marginX);
72 memset(txt + width, txt[width - 1], marginX);
73#endif
74
75 txt += stride;
76 }
77}

79template<int N, int width, int height>
80void interp_horiz_pp_c(const pixel* src, intptr_t srcStride, pixel* dst, intptr_t dstStride, int coeffIdx)
81{
82 const int16_t* coeff = (N == 4) ? g_chromaFilter[coeffIdx] : g_lumaFilter[coeffIdx];
83 int headRoom = IF_FILTER_PREC;
84 int offset = (1 << (headRoom - 1));
85 uint16_t maxVal = (1 << X265_DEPTH) - 1;
86 int cStride = 1;
87
88 src -= (N / 2 - 1) * cStride;
89
90 int row, col;
91 for (row = 0; row < height; row++)
92 {
93 for (col = 0; col < width; col++)
94 {
95 int sum;
96
97 sum = src[col + 0 * cStride] * coeff[0];
98 sum += src[col + 1 * cStride] * coeff[1];
99 sum += src[col + 2 * cStride] * coeff[2];
100 sum += src[col + 3 * cStride] * coeff[3];
101 if (N == 8)
102 {
103 sum += src[col + 4 * cStride] * coeff[4];
104 sum += src[col + 5 * cStride] * coeff[5];
105 sum += src[col + 6 * cStride] * coeff[6];
106 sum += src[col + 7 * cStride] * coeff[7];
107 }
108 int16_t val = (int16_t)((sum + offset) >> headRoom);
109
110 if (val < 0) val = 0;
111 if (val > maxVal) val = maxVal;
112 dst[col] = (pixel)val;
113 }
114
115 src += srcStride;
116 dst += dstStride;
117 }
118}

120template<int N, int width, int height>
121void interp_horiz_ps_c(const pixel* src, intptr_t srcStride, int16_t* dst, intptr_t dstStride, int coeffIdx, int isRowExt)
122{
123 const int16_t* coeff = (N == 4) ? g_chromaFilter[coeffIdx] : g_lumaFilter[coeffIdx];
124 int headRoom = IF_INTERNAL_PREC - X265_DEPTH;
125 int shift = IF_FILTER_PREC - headRoom;
126 int offset = (unsigned)-IF_INTERNAL_OFFS << shift;
127 int blkheight = height;
128 src -= N / 2 - 1;
129
130 if (isRowExt)
131 {
132 src -= (N / 2 - 1) * srcStride;
133 blkheight += N - 1;
134 }
135
136 int row, col;
137 for (row = 0; row < blkheight; row++)
138 {
139 for (col = 0; col < width; col++)
140 {
141 int sum;
142
143 sum = src[col + 0] * coeff[0];
144 sum += src[col + 1] * coeff[1];
145 sum += src[col + 2] * coeff[2];
146 sum += src[col + 3] * coeff[3];
147 if (N == 8)
148 {
149 sum += src[col + 4] * coeff[4];
150 sum += src[col + 5] * coeff[5];
151 sum += src[col + 6] * coeff[6];
152 sum += src[col + 7] * coeff[7];
153 }
154
155 int16_t val = (int16_t)((sum + offset) >> shift);
156 dst[col] = val;
157 }
158
159 src += srcStride;
160 dst += dstStride;
161 }

362template<int N, int width, int height>
363void interp_hv_pp_c(const pixel* src, intptr_t srcStride, pixel* dst, intptr_t dstStride, int idxX, int idxY)
364{
365 ALIGN_VAR_32(int16_t, immed[width * (height + N - 1)]);
366
367 interp_horiz_ps_c<N, width, height>(src, srcStride, immed, width, idxX, 1);
368 filterVertical_sp_c<N>(immed + (N / 2 - 1) * width, width, dst, dstStride, width, height, idxY);
369}
