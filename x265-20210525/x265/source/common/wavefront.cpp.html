33bool WaveFront::init(int numRows)
34{
35 m_numRows = numRows;
36
37 m_numWords = (numRows + 31) >> 5;
38 m_internalDependencyBitmap = X265_MALLOC(uint32_t, m_numWords);
39 if (m_internalDependencyBitmap)
40 memset((void*)m_internalDependencyBitmap, 0, sizeof(uint32_t) * m_numWords);
41
42 m_externalDependencyBitmap = X265_MALLOC(uint32_t, m_numWords);
43 if (m_externalDependencyBitmap)
44 memset((void*)m_externalDependencyBitmap, 0, sizeof(uint32_t) * m_numWords);
45
46 m_row_to_idx = X265_MALLOC(uint32_t, m_numRows);
47 m_idx_to_row = X265_MALLOC(uint32_t, m_numRows);
48
49 return m_internalDependencyBitmap && m_externalDependencyBitmap;
50}
51
52WaveFront::~WaveFront()
53{
54 x265_free((void*)m_row_to_idx);
55 x265_free((void*)m_idx_to_row);
56
57 x265_free((void*)m_internalDependencyBitmap);
58 x265_free((void*)m_externalDependencyBitmap);
59}
60
61void WaveFront::clearEnabledRowMask()
62{
63 memset((void*)m_externalDependencyBitmap, 0, sizeof(uint32_t) * m_numWords);
64 memset((void*)m_internalDependencyBitmap, 0, sizeof(uint32_t) * m_numWords);
65}
66
67void WaveFront::enqueueRow(int row)
68{
69 uint32_t bit = 1 << (row & 31);
70 ATOMIC_OR(&m_internalDependencyBitmap[row >> 5], bit);
71}
72
73void WaveFront::enableRow(int row)
74{
75 uint32_t bit = 1 << (row & 31);
76 ATOMIC_OR(&m_externalDependencyBitmap[row >> 5], bit);
77}
78
79void WaveFront::enableAllRows()
80{
81 memset((void*)m_externalDependencyBitmap, ~0, sizeof(uint32_t) * m_numWords);
82}
83
84bool WaveFront::dequeueRow(int row)
85{
86 uint32_t bit = 1 << (row & 31);
87 return !!(ATOMIC_AND(&m_internalDependencyBitmap[row >> 5], ~bit) & bit);
88}
89
90void WaveFront::findJob(int threadId)
91{
92 unsigned long id;
93
94 /* Loop over each word until all available rows are finished */
95 for (int w = 0; w < m_numWords; w++)
96 {
97 uint32_t oldval = m_internalDependencyBitmap[w] & m_externalDependencyBitmap[w];
98 while (oldval)
99 {
100 CTZ(id, oldval);
101
102 uint32_t bit = 1 << id;
103 if (ATOMIC_AND(&m_internalDependencyBitmap[w], ~bit) & bit)
104 {
105 /* we cleared the bit, we get to process the row */
106 processRow(w * 32 + id, threadId);
107 m_helpWanted = true;
108 return; /* check for a higher priority task */
109 }
110
111 oldval = m_internalDependencyBitmap[w] & m_externalDependencyBitmap[w];
112 }
113 }
114
115 m_helpWanted = false;
116}
