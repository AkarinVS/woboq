31bool PicQPAdaptationLayer::create(uint32_t width, uint32_t height, uint32_t partWidth, uint32_t partHeight, uint32_t numAQPartInWidthExt, uint32_t numAQPartInHeightExt)
32{
33 aqPartWidth = partWidth;
34 aqPartHeight = partHeight;
35 numAQPartInWidth = (width + partWidth - 1) / partWidth;
36 numAQPartInHeight = (height + partHeight - 1) / partHeight;
37
38 CHECKED_MALLOC_ZERO(dActivity, double, numAQPartInWidthExt * numAQPartInHeightExt);
39 CHECKED_MALLOC_ZERO(dQpOffset, double, numAQPartInWidthExt * numAQPartInHeightExt);
40 CHECKED_MALLOC_ZERO(dCuTreeOffset, double, numAQPartInWidthExt * numAQPartInHeightExt);
41
42 if (bQpSize)
43 CHECKED_MALLOC_ZERO(dCuTreeOffset8x8, double, numAQPartInWidthExt * numAQPartInHeightExt);
44
45 return true;
46fail:
47 return false;
48}

50bool Lowres::create(x265_param* param, PicYuv *origPic, uint32_t qgSize)
51{
52 isLowres = true;
53 bframes = param->bframes;
54 widthFullRes = origPic->m_picWidth;
55 heightFullRes = origPic->m_picHeight;
56 width = origPic->m_picWidth / 2;
57 lines = origPic->m_picHeight / 2;
58 bEnableHME = param->bEnableHME ? 1 : 0;
59 lumaStride = width + 2 * origPic->m_lumaMarginX;
60 if (lumaStride & 31)
61 lumaStride += 32 - (lumaStride & 31);
62 maxBlocksInRow = (width + X265_LOWRES_CU_SIZE - 1) >> X265_LOWRES_CU_BITS;
63 maxBlocksInCol = (lines + X265_LOWRES_CU_SIZE - 1) >> X265_LOWRES_CU_BITS;
64 maxBlocksInRowFullRes = maxBlocksInRow * 2;
65 maxBlocksInColFullRes = maxBlocksInCol * 2;
66 int cuCount = maxBlocksInRow * maxBlocksInCol;
67 int cuCountFullRes = (qgSize > 8) ? cuCount : cuCount << 2;
68 isHMELowres = param->bEnableHME ? 1 : 0;
69
70 /* rounding the width to multiple of lowres CU size */
71 width = maxBlocksInRow * X265_LOWRES_CU_SIZE;
72 lines = maxBlocksInCol * X265_LOWRES_CU_SIZE;
73
74 size_t planesize = lumaStride * (lines + 2 * origPic->m_lumaMarginY);
75 size_t padoffset = lumaStride * origPic->m_lumaMarginY + origPic->m_lumaMarginX;
76 if (!!param->rc.aqMode || !!param->rc.hevcAq || !!param->bAQMotion)
77 {
78 CHECKED_MALLOC_ZERO(qpAqOffset, double, cuCountFullRes);
79 CHECKED_MALLOC_ZERO(invQscaleFactor, int, cuCountFullRes);
80 CHECKED_MALLOC_ZERO(qpCuTreeOffset, double, cuCountFullRes);
81 if (qgSize == 8)
82 CHECKED_MALLOC_ZERO(invQscaleFactor8x8, int, cuCount);
83 CHECKED_MALLOC_ZERO(edgeInclined, int, cuCountFullRes);
84 }
85
86 if (origPic->m_param->bAQMotion)
87 CHECKED_MALLOC_ZERO(qpAqMotionOffset, double, cuCountFullRes);
88 if (origPic->m_param->bDynamicRefine || origPic->m_param->bEnableFades)
89 CHECKED_MALLOC_ZERO(blockVariance, uint32_t, cuCountFullRes);
90
91 if (!!param->rc.hevcAq)
92 {
93 m_maxCUSize = param->maxCUSize;
94 m_qgSize = qgSize;
95
96 uint32_t partWidth, partHeight, nAQPartInWidth, nAQPartInHeight;
97
98 pAQLayer = new PicQPAdaptationLayer[4];
99 maxAQDepth = 0;
100 for (uint32_t d = 0; d < 4; d++)
101 {
102 int ctuSizeIdx = 6 - g_log2Size[param->maxCUSize];
103 int aqDepth = g_log2Size[param->maxCUSize] - g_log2Size[qgSize];
104 if (!aqLayerDepth[ctuSizeIdx][aqDepth][d])
105 continue;
106
107 pAQLayer->minAQDepth = d;
108 partWidth = param->maxCUSize >> d;
109 partHeight = param->maxCUSize >> d;
110
111 if (minAQSize[ctuSizeIdx] == d)
112 {
113 pAQLayer[d].bQpSize = true;
114 nAQPartInWidth = maxBlocksInRow * 2;
115 nAQPartInHeight = maxBlocksInCol * 2;
116 }
117 else
118 {
119 pAQLayer[d].bQpSize = false;
120 nAQPartInWidth = (origPic->m_picWidth + partWidth - 1) / partWidth;
121 nAQPartInHeight = (origPic->m_picHeight + partHeight - 1) / partHeight;
122 }
123
124 maxAQDepth++;
125
126 pAQLayer[d].create(origPic->m_picWidth, origPic->m_picHeight, partWidth, partHeight, nAQPartInWidth, nAQPartInHeight);
127 }
128 }
129 CHECKED_MALLOC(propagateCost, uint16_t, cuCount);
130
131 /* allocate lowres buffers */
132 CHECKED_MALLOC_ZERO(buffer[0], pixel, 4 * planesize);
133
134 buffer[1] = buffer[0] + planesize;
135 buffer[2] = buffer[1] + planesize;
136 buffer[3] = buffer[2] + planesize;
137
138 lowresPlane[0] = buffer[0] + padoffset;
139 lowresPlane[1] = buffer[1] + padoffset;
140 lowresPlane[2] = buffer[2] + padoffset;
141 lowresPlane[3] = buffer[3] + padoffset;
142
143 if (bEnableHME)
144 {
145 intptr_t lumaStrideHalf = lumaStride / 2;
146 if (lumaStrideHalf & 31)
147 lumaStrideHalf += 32 - (lumaStrideHalf & 31);
148 size_t planesizeHalf = planesize / 2;
149 size_t padoffsetHalf = padoffset / 2;
150 /* allocate lower-res buffers */
151 CHECKED_MALLOC_ZERO(lowerResBuffer[0], pixel, 4 * planesizeHalf);
152
153 lowerResBuffer[1] = lowerResBuffer[0] + planesizeHalf;
154 lowerResBuffer[2] = lowerResBuffer[1] + planesizeHalf;
155 lowerResBuffer[3] = lowerResBuffer[2] + planesizeHalf;
156
157 lowerResPlane[0] = lowerResBuffer[0] + padoffsetHalf;
158 lowerResPlane[1] = lowerResBuffer[1] + padoffsetHalf;
159 lowerResPlane[2] = lowerResBuffer[2] + padoffsetHalf;
160 lowerResPlane[3] = lowerResBuffer[3] + padoffsetHalf;
161 }
162
163 CHECKED_MALLOC(intraCost, int32_t, cuCount);
164 CHECKED_MALLOC(intraMode, uint8_t, cuCount);
165
166 for (int i = 0; i < bframes + 2; i++)
167 {
168 for (int j = 0; j < bframes + 2; j++)
169 {
170 CHECKED_MALLOC(rowSatds[i][j], int32_t, maxBlocksInCol);
171 CHECKED_MALLOC(lowresCosts[i][j], uint16_t, cuCount);
172 }
173 }
174
175 for (int i = 0; i < bframes + 2; i++)
176 {
177 CHECKED_MALLOC(lowresMvs[0][i], MV, cuCount);
178 CHECKED_MALLOC(lowresMvs[1][i], MV, cuCount);
179 CHECKED_MALLOC(lowresMvCosts[0][i], int32_t, cuCount);
180 CHECKED_MALLOC(lowresMvCosts[1][i], int32_t, cuCount);
181 if (bEnableHME)
182 {
183 int maxBlocksInRowLowerRes = ((width/2) + X265_LOWRES_CU_SIZE - 1) >> X265_LOWRES_CU_BITS;
184 int maxBlocksInColLowerRes = ((lines/2) + X265_LOWRES_CU_SIZE - 1) >> X265_LOWRES_CU_BITS;
185 int cuCountLowerRes = maxBlocksInRowLowerRes * maxBlocksInColLowerRes;
186 CHECKED_MALLOC(lowerResMvs[0][i], MV, cuCountLowerRes);
187 CHECKED_MALLOC(lowerResMvs[1][i], MV, cuCountLowerRes);
188 CHECKED_MALLOC(lowerResMvCosts[0][i], int32_t, cuCountLowerRes);
189 CHECKED_MALLOC(lowerResMvCosts[1][i], int32_t, cuCountLowerRes);
190 }
191 }
192
193 return true;
194
195fail:
196 return false;
197}

199void Lowres::destroy()
200{
201 X265_FREE(buffer[0]);
202 if(bEnableHME)
203 X265_FREE(lowerResBuffer[0]);
204 X265_FREE(intraCost);
205 X265_FREE(intraMode);
206
207 for (int i = 0; i < bframes + 2; i++)
208 {
209 for (int j = 0; j < bframes + 2; j++)
210 {
211 X265_FREE(rowSatds[i][j]);
212 X265_FREE(lowresCosts[i][j]);
213 }
214 }
215
216 for (int i = 0; i < bframes + 2; i++)
217 {
218 X265_FREE(lowresMvs[0][i]);
219 X265_FREE(lowresMvs[1][i]);
220 X265_FREE(lowresMvCosts[0][i]);
221 X265_FREE(lowresMvCosts[1][i]);
222 if (bEnableHME)
223 {
224 X265_FREE(lowerResMvs[0][i]);
225 X265_FREE(lowerResMvs[1][i]);
226 X265_FREE(lowerResMvCosts[0][i]);
227 X265_FREE(lowerResMvCosts[1][i]);
228 }
229 }
230 X265_FREE(qpAqOffset);
231 X265_FREE(invQscaleFactor);
232 X265_FREE(qpCuTreeOffset);
233 X265_FREE(propagateCost);
234 X265_FREE(invQscaleFactor8x8);
235 X265_FREE(edgeInclined);
236 X265_FREE(qpAqMotionOffset);
237 X265_FREE(blockVariance);
238 if (maxAQDepth > 0)
239 {
240 for (uint32_t d = 0; d < 4; d++)
241 {
242 int ctuSizeIdx = 6 - g_log2Size[m_maxCUSize];
243 int aqDepth = g_log2Size[m_maxCUSize] - g_log2Size[m_qgSize];
244 if (!aqLayerDepth[ctuSizeIdx][aqDepth][d])
245 continue;
246
247 X265_FREE(pAQLayer[d].dActivity);
248 X265_FREE(pAQLayer[d].dQpOffset);
249 X265_FREE(pAQLayer[d].dCuTreeOffset);
250
251 if (pAQLayer[d].bQpSize == true)
252 X265_FREE(pAQLayer[d].dCuTreeOffset8x8);
253 }
254
255 delete[] pAQLayer;
256 }
257}
258// (re) initialize lowres state
259void Lowres::init(PicYuv *origPic, int poc)
260{
261 bLastMiniGopBFrame = false;
262 bKeyframe = false; // Not a keyframe unless identified by lookahead
263 bIsFadeEnd = false;
264 frameNum = poc;
265 leadingBframes = 0;
266 indB = 0;
267 memset(costEst, -1, sizeof(costEst));
268 memset(weightedCostDelta, 0, sizeof(weightedCostDelta));
269 interPCostPercDiff = 0.0;
270 intraCostPercDiff = 0.0;
271 m_bIsMaxThres = false;
272 m_bIsHardScenecut = false;
273
274 if (qpAqOffset && invQscaleFactor)
275 memset(costEstAq, -1, sizeof(costEstAq));
276
277 for (int y = 0; y < bframes + 2; y++)
278 for (int x = 0; x < bframes + 2; x++)
279 rowSatds[y][x][0] = -1;
280
281 for (int i = 0; i < bframes + 2; i++)
282 {
283 lowresMvs[0][i][0].x = 0x7FFF;
284 lowresMvs[1][i][0].x = 0x7FFF;
285 }
286
287 for (int i = 0; i < bframes + 2; i++)
288 intraMbs[i] = 0;
289 if (origPic->m_param->rc.vbvBufferSize)
290 for (int i = 0; i < X265_LOOKAHEAD_MAX + 1; i++)
291 plannedType[i] = X265_TYPE_AUTO;
292
293 /* downscale and generate 4 hpel planes for lookahead */
294 primitives.frameInitLowres(origPic->m_picOrg[0],
295 lowresPlane[0], lowresPlane[1], lowresPlane[2], lowresPlane[3],
296 origPic->m_stride, lumaStride, width, lines);
297
298 /* extend hpel planes for motion search */
299 extendPicBorder(lowresPlane[0], lumaStride, width, lines, origPic->m_lumaMarginX, origPic->m_lumaMarginY);
300 extendPicBorder(lowresPlane[1], lumaStride, width, lines, origPic->m_lumaMarginX, origPic->m_lumaMarginY);
301 extendPicBorder(lowresPlane[2], lumaStride, width, lines, origPic->m_lumaMarginX, origPic->m_lumaMarginY);
302 extendPicBorder(lowresPlane[3], lumaStride, width, lines, origPic->m_lumaMarginX, origPic->m_lumaMarginY);
303
304 if (origPic->m_param->bEnableHME)
305 {
306 primitives.frameInitLowerRes(lowresPlane[0],
307 lowerResPlane[0], lowerResPlane[1], lowerResPlane[2], lowerResPlane[3],
308 lumaStride, lumaStride/2, (width / 2), (lines / 2));
309 extendPicBorder(lowerResPlane[0], lumaStride/2, width/2, lines/2, origPic->m_lumaMarginX/2, origPic->m_lumaMarginY/2);
310 extendPicBorder(lowerResPlane[1], lumaStride/2, width/2, lines/2, origPic->m_lumaMarginX/2, origPic->m_lumaMarginY/2);
311 extendPicBorder(lowerResPlane[2], lumaStride/2, width/2, lines/2, origPic->m_lumaMarginX/2, origPic->m_lumaMarginY/2);
312 extendPicBorder(lowerResPlane[3], lumaStride/2, width/2, lines/2, origPic->m_lumaMarginX/2, origPic->m_lumaMarginY/2);
313 fpelLowerResPlane[0] = lowerResPlane[0];
314 }
315
316 fpelPlane[0] = lowresPlane[0];
317}
