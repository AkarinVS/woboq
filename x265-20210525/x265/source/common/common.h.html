167template<typename T>
168inline T x265_min(T a, T b) { return a < b ? a : b; }
169
170template<typename T>
171inline T x265_max(T a, T b) { return a > b ? a : b; }
172
173template<typename T>
174inline T x265_clip3(T minVal, T maxVal, T a) { return x265_min(x265_max(minVal, a), maxVal); }
175
176template<typename T> /* clip to pixel range, 0..255 or 0..1023 */
177inline pixel x265_clip(T x) { return (pixel)x265_min<T>(T((1 << X265_DEPTH) - 1), x265_max<T>(T(0), x)); }

183#define COPY1_IF_LT(x, y) {if ((y) < (x)) (x) = (y);}
184#define COPY2_IF_LT(x, y, a, b) \
185 if ((y) < (x)) \
186 { \
187 (x) = (y); \
188 (a) = (b); \
189 }
190#define COPY3_IF_LT(x, y, a, b, c, d) \
191 if ((y) < (x)) \
192 { \
193 (x) = (y); \
194 (a) = (b); \
195 (c) = (d); \
196 }
197#define COPY4_IF_LT(x, y, a, b, c, d, e, f) \
198 if ((y) < (x)) \
199 { \
200 (x) = (y); \
201 (a) = (b); \
202 (c) = (d); \
203 (e) = (f); \
204 }
205#define X265_MIN3(a, b, c) X265_MIN((a), X265_MIN((b), (c)))

223#define CHECKED_MALLOC(var, type, count) \
224 { \
225 var = (type*)x265_malloc(sizeof(type) * (count)); \
226 if (!var) \
227 { \
228 x265_log(NULL, X265_LOG_ERROR, "malloc of size %d failed\n", sizeof(type) * (count)); \
229 goto fail; \
230 } \
231 }
232#define CHECKED_MALLOC_ZERO(var, type, count) \
233 { \
234 var = (type*)x265_malloc(sizeof(type) * (count)); \
