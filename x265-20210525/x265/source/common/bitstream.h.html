31class BitInterface
32{
33public:
34
35 virtual void write(uint32_t val, uint32_t numBits) = 0;
36 virtual void writeByte(uint32_t val) = 0;
37 virtual void resetBits() = 0;
38 virtual uint32_t getNumberOfWrittenBits() const = 0;
39 virtual void writeAlignOne() = 0;
40 virtual void writeAlignZero() = 0;
41 virtual ~BitInterface() {}
42};
43
44class BitCounter : public BitInterface
45{
46protected:
47
48 uint32_t m_bitCounter;
49
50public:
51
52 BitCounter() : m_bitCounter(0) {}
53
54 void write(uint32_t, uint32_t num) { m_bitCounter += num; }
55 void writeByte(uint32_t) { m_bitCounter += 8; }
56 void resetBits() { m_bitCounter = 0; }
57 uint32_t getNumberOfWrittenBits() const { return m_bitCounter; }
58 void writeAlignOne() { }
59 void writeAlignZero() { }
60};
61
62
63class Bitstream : public BitInterface
64{
65public:
66
67 Bitstream();
68 ~Bitstream() { X265_FREE(m_fifo); }
69
70 void resetBits() { m_partialByteBits = m_byteOccupancy = 0; m_partialByte = 0; }
71 uint32_t getNumberOfWrittenBytes() const { return m_byteOccupancy; }
72 uint32_t getNumberOfWrittenBits() const { return m_byteOccupancy * 8 + m_partialByteBits; }
73 const uint8_t* getFIFO() const { return m_fifo; }
74 void copyBits(Bitstream* stream) { m_partialByteBits = stream->m_partialByteBits; m_byteOccupancy = stream->m_byteOccupancy; m_partialByte = stream->m_partialByte; }
75
76 void write(uint32_t val, uint32_t numBits);
77 void writeByte(uint32_t val);
78
79 void writeAlignOne(); // insert one bits until the bitstream is byte-aligned
80 void writeAlignZero(); // insert zero bits until the bitstream is byte-aligned
81 void writeByteAlignment(); // insert 1 bit, then pad to byte-align with zero
82
83private:
84
85 uint8_t *m_fifo;
86 uint32_t m_byteAlloc;
87 uint32_t m_byteOccupancy;
88 uint32_t m_partialByteBits;
89 uint8_t m_partialByte;
90
91 void push_back(uint8_t val);
92};

114static inline int bs_size_ue(unsigned int val)
115{
116 return bitSize[val + 1];
117}
118
119static inline int bs_size_ue_big(unsigned int val)
120{
121 if (val < 255)
122 return bitSize[val + 1];
123 else
124 return bitSize[(val + 1) >> 8] + 16;
125}
126
127static inline int bs_size_se(int val)
128{
129 int tmp = 1 - val * 2;
130
131 if (tmp < 0) tmp = val * 2;
132 if (tmp < 256)
133 return bitSize[tmp];
134 else
135 return bitSize[tmp >> 8] + 16;
136}

138class SyntaxElementWriter
139{
140public:
141
142 BitInterface* m_bitIf;
143
144 SyntaxElementWriter() : m_bitIf(NULL) {}
145
146 /* silently discard the name of the syntax element */
147 inline void WRITE_CODE(uint32_t code, uint32_t length, const char *) { writeCode(code, length); }
148 inline void WRITE_UVLC(uint32_t code, const char *) { writeUvlc(code); }
149 inline void WRITE_SVLC(int32_t code, const char *) { writeSvlc(code); }
150 inline void WRITE_FLAG(bool flag, const char *) { writeFlag(flag); }
151
152 void writeCode(uint32_t code, uint32_t length) { m_bitIf->write(code, length); }
153 void writeUvlc(uint32_t code);
154 void writeSvlc(int32_t code) { uint32_t ucode = (code <= 0) ? -code << 1 : (code << 1) - 1; writeUvlc(ucode); }
155 void writeFlag(bool code) { m_bitIf->write(code, 1); }
156};
