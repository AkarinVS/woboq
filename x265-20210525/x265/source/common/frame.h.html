34
35class FrameData;
36class PicYuv;
37struct SPS;
38
39#define IS_REFERENCED(frame) (frame->m_lowres.sliceType != X265_TYPE_B)
40
41/* Ratecontrol statistics */
42struct RcStats
43{
44 double qpaRc;
45 double qpAq;
46 double qRceq;
47 double qpNoVbv;
48 double newQScale;
49 double iCuCount;
50 double pCuCount;
51 double skipCuCount;
52 double qScale;
53 double cumulativePQp;
54 double cumulativePNorm;
55 double lastQScaleFor[3];
56 int mvBits;
57 int miscBits;
58 int coeffBits;
59 int poc;
60 int encodeOrder;
61 int sliceType;
62 int keptAsRef;
63 double wantedBitsWindow;
64 double cplxrSum;
65 double shortTermCplxSum;
66 double shortTermCplxCount;
67 int64_t totalBits;
68 int64_t encodedBits;
69 double coeff[4];
70 double count[4];
71 double offset[4];
72 double bufferFillFinal;
73};
74
75class Frame
76{
77public:
78
79 /* These two items will be NULL until the Frame begins to be encoded, at which point
80 * it will be assigned a FrameData instance, which comes with a reconstructed image PicYuv */
81 FrameData* m_encData;
82 PicYuv* m_reconPic;
83
84 /* Data associated with x265_picture */
85 PicYuv* m_fencPic;
86 int m_poc;
87 int m_encodeOrder;
88 int64_t m_pts; // user provided presentation time stamp
89 int64_t m_reorderedPts;
90 int64_t m_dts;
91 int32_t m_forceqp; // Force to use the qp specified in qp file
92 void* m_userData; // user provided pointer passed in with this picture
93
94 Lowres m_lowres;
95 bool m_lowresInit; // lowres init complete (pre-analysis)
96 bool m_bChromaExtended; // orig chroma planes motion extended for weight analysis
97 bool m_reconfigureRc;
98
99 float* m_quantOffsets; // points to quantOffsets in x265_picture
100 x265_sei m_userSEI;
101 uint32_t m_picStruct; // picture structure SEI message
102 x265_dolby_vision_rpu m_rpu;
103
104 /* Frame Parallelism - notification between FrameEncoders of available motion reference rows */
105 ThreadSafeInteger* m_reconRowFlag; // flag of CTU rows completely reconstructed and extended for motion reference
106 ThreadSafeInteger* m_reconColCount; // count of CTU cols completely reconstructed and extended for motion reference
107 int32_t m_numRows;
108 volatile uint32_t m_countRefEncoders; // count of FrameEncoder threads monitoring m_reconRowCount
109
110 Frame* m_next; // PicList doubly linked list pointers
111 Frame* m_prev;
112 x265_param* m_param; // Points to the latest param set for the frame.
113 x265_analysis_data m_analysisData;
114 RcStats* m_rcData;
115
116 Event m_copyMVType;
117
118 x265_ctu_info_t** m_ctuInfo;
119 Event m_copied;
120 int* m_prevCtuInfoChange;
121 int64_t m_encodeStartTime;
122
123 uint8_t** m_addOnDepth;
124 uint8_t** m_addOnCtuInfo;
125 int** m_addOnPrevChange;
126
127 /* Average feature values of frames being considered for classification */
128 uint64_t* m_classifyRd;
129 uint64_t* m_classifyVariance;
130 uint32_t* m_classifyCount;
131
132 bool m_classifyFrame;
133 int m_fieldNum;
134
135 /* aq-mode 4 : Gaussian, edge and theta frames for edge information */
136 pixel* m_edgePic;
137 pixel* m_gaussianPic;
138 pixel* m_thetaPic;
139
140 /* edge bit plane for rskips 2 and 3 */
141 pixel* m_edgeBitPlane;
142 pixel* m_edgeBitPic;
143
144 int m_isInsideWindow;
145
146 Frame();
147
148 bool create(x265_param *param, float* quantOffsets);
149 bool allocEncodeData(x265_param *param, const SPS& sps);
150 void reinit(const SPS& sps);
151 void destroy();
152};
153}

155#endif // ifndef X265_FRAME_H
