35/* for all bcast* and copy* functions, dst and src are aligned to MIN(size, 32) */
36
37static void bcast1(uint8_t* dst, uint8_t val) { dst[0] = val; }
38
39static void copy4(uint8_t* dst, uint8_t* src) { ((uint32_t*)dst)[0] = ((uint32_t*)src)[0]; }
40static void bcast4(uint8_t* dst, uint8_t val) { ((uint32_t*)dst)[0] = 0x01010101u * val; }
41
42static void copy16(uint8_t* dst, uint8_t* src) { ((uint64_t*)dst)[0] = ((uint64_t*)src)[0]; ((uint64_t*)dst)[1] = ((uint64_t*)src)[1]; }
43static void bcast16(uint8_t* dst, uint8_t val) { uint64_t bval = 0x0101010101010101ULL * val; ((uint64_t*)dst)[0] = bval; ((uint64_t*)dst)[1] = bval; }
44
45static void copy64(uint8_t* dst, uint8_t* src) { ((uint64_t*)dst)[0] = ((uint64_t*)src)[0]; ((uint64_t*)dst)[1] = ((uint64_t*)src)[1];
46 ((uint64_t*)dst)[2] = ((uint64_t*)src)[2]; ((uint64_t*)dst)[3] = ((uint64_t*)src)[3];
47 ((uint64_t*)dst)[4] = ((uint64_t*)src)[4]; ((uint64_t*)dst)[5] = ((uint64_t*)src)[5];
48 ((uint64_t*)dst)[6] = ((uint64_t*)src)[6]; ((uint64_t*)dst)[7] = ((uint64_t*)src)[7]; }
49static void bcast64(uint8_t* dst, uint8_t val) { uint64_t bval = 0x0101010101010101ULL * val;
50 ((uint64_t*)dst)[0] = bval; ((uint64_t*)dst)[1] = bval; ((uint64_t*)dst)[2] = bval; ((uint64_t*)dst)[3] = bval;
51 ((uint64_t*)dst)[4] = bval; ((uint64_t*)dst)[5] = bval; ((uint64_t*)dst)[6] = bval; ((uint64_t*)dst)[7] = bval; }
52
53/* at 256 bytes, memset/memcpy will probably use SIMD more effectively than our uint64_t hack,
54 * but hand-written assembly would beat it. */
55static void copy256(uint8_t* dst, uint8_t* src) { memcpy(dst, src, 256); }
56static void bcast256(uint8_t* dst, uint8_t val) { memset(dst, val, 256); }

61/* Check whether 2 addresses point to the same column */
62inline bool isEqualCol(int addrA, int addrB)
63{
64 return ((addrA ^ addrB) & (RASTER_SIZE - 1)) == 0;
65}
66
67/* Check whether 2 addresses point to the same row */
68inline bool isEqualRow(int addrA, int addrB)
69{
70 return ((addrA ^ addrB) < RASTER_SIZE);
71}
72
73/* Check whether 2 addresses point to the same row or column */
74inline bool isEqualRowOrCol(int addrA, int addrB)
75{
76 return isEqualCol(addrA, addrB) | isEqualRow(addrA, addrB);
77}
78
79/* Check whether one address points to the first column */
80inline bool isZeroCol(int addr)
81{
82 return (addr & (RASTER_SIZE - 1)) == 0;
83}
84
85/* Check whether one address points to the first row */
86inline bool isZeroRow(int addr)
87{
88 return (addr < RASTER_SIZE);
89}
90
91/* Check whether one address points to a column whose index is smaller than a given value */
92inline bool lessThanCol(int addr, int val)
93{
94 return (addr & (RASTER_SIZE - 1)) < val;
95}
96
97/* Check whether one address points to a row whose index is smaller than a given value */
98inline bool lessThanRow(int addr, int val)
99{
100 // addr / numUnits < val
101 return (addr >> LOG2_RASTER_SIZE) < val;
102}
103
104inline MV scaleMv(MV mv, int scale)
105{
106 int mvx = x265_clip3(-32768, 32767, (scale * mv.x + 127 + (scale * mv.x < 0)) >> 8);
107 int mvy = x265_clip3(-32768, 32767, (scale * mv.y + 127 + (scale * mv.y < 0)) >> 8);
108
109 return MV((int32_t)mvx, (int32_t)mvy);
110}

119void CUData::initialize(const CUDataMemPool& dataPool, uint32_t depth, const x265_param& param, int instance)
120{
121 int csp = param.internalCsp;
122 m_chromaFormat = csp;
123 m_hChromaShift = CHROMA_H_SHIFT(csp);
124 m_vChromaShift = CHROMA_V_SHIFT(csp);
125 m_numPartitions = param.num4x4Partitions >> (depth * 2);
126
127 if (!s_partSet[0])
128 {
129 s_numPartInCUSize = 1 << param.unitSizeDepth;
130 switch (param.maxLog2CUSize)
131 {
132 case 6:
133 s_partSet[0] = bcast256;
134 s_partSet[1] = bcast64;
135 s_partSet[2] = bcast16;
136 s_partSet[3] = bcast4;
137 s_partSet[4] = bcast1;
138 break;
139 case 5:
140 s_partSet[0] = bcast64;
141 s_partSet[1] = bcast16;
142 s_partSet[2] = bcast4;
143 s_partSet[3] = bcast1;
144 s_partSet[4] = NULL;
145 break;
146 case 4:
147 s_partSet[0] = bcast16;
148 s_partSet[1] = bcast4;
149 s_partSet[2] = bcast1;
150 s_partSet[3] = NULL;
151 s_partSet[4] = NULL;
152 break;
153 default:
154 X265_CHECK(0, "unexpected CTU size\n");
155 break;
156 }
157 }
158
159 switch (m_numPartitions)
160 {
161 case 256: // 64x64 CU
162 m_partCopy = copy256;
163 m_partSet = bcast256;
164 m_subPartCopy = copy64;
165 m_subPartSet = bcast64;
166 break;
167 case 64: // 32x32 CU
168 m_partCopy = copy64;
169 m_partSet = bcast64;
170 m_subPartCopy = copy16;
171 m_subPartSet = bcast16;
172 break;
173 case 16: // 16x16 CU
174 m_partCopy = copy16;
175 m_partSet = bcast16;
176 m_subPartCopy = copy4;
177 m_subPartSet = bcast4;
178 break;
179 case 4: // 8x8 CU
180 m_partCopy = copy4;
181 m_partSet = bcast4;
182 m_subPartCopy = NULL;
183 m_subPartSet = NULL;
184 break;
185 default:
186 X265_CHECK(0, "unexpected CU partition count\n");
187 break;
188 }
189
190 if (csp == X265_CSP_I400)
191 {
192 /* Each CU's data is layed out sequentially within the charMemBlock */
193 uint8_t *charBuf = dataPool.charMemBlock + (m_numPartitions * (BytesPerPartition - 4)) * instance;
194
195 m_qp = (int8_t*)charBuf; charBuf += m_numPartitions;
196 m_qpAnalysis = (int8_t*)charBuf; charBuf += m_numPartitions;
197 m_log2CUSize = charBuf; charBuf += m_numPartitions;
198 m_lumaIntraDir = charBuf; charBuf += m_numPartitions;
199 m_tqBypass = charBuf; charBuf += m_numPartitions;
200 m_refIdx[0] = (int8_t*)charBuf; charBuf += m_numPartitions;
201 m_refIdx[1] = (int8_t*)charBuf; charBuf += m_numPartitions;
202 m_cuDepth = charBuf; charBuf += m_numPartitions;
203 m_predMode = charBuf; charBuf += m_numPartitions; /* the order up to here is important in initCTU() and initSubCU() */
204 m_partSize = charBuf; charBuf += m_numPartitions;
205 m_skipFlag[0] = charBuf; charBuf += m_numPartitions;
206 m_skipFlag[1] = charBuf; charBuf += m_numPartitions;
207 m_mergeFlag = charBuf; charBuf += m_numPartitions;
208 m_interDir = charBuf; charBuf += m_numPartitions;
209 m_mvpIdx[0] = charBuf; charBuf += m_numPartitions;
210 m_mvpIdx[1] = charBuf; charBuf += m_numPartitions;
211 m_tuDepth = charBuf; charBuf += m_numPartitions;
212 m_transformSkip[0] = charBuf; charBuf += m_numPartitions;
213 m_cbf[0] = charBuf; charBuf += m_numPartitions;
214 m_chromaIntraDir = charBuf; charBuf += m_numPartitions;
215
216 X265_CHECK(charBuf == dataPool.charMemBlock + (m_numPartitions * (BytesPerPartition - 4)) * (instance + 1), "CU data layout is broken\n"); //BytesPerPartition
217
218 m_mv[0] = dataPool.mvMemBlock + (instance * 4) * m_numPartitions;
219 m_mv[1] = m_mv[0] + m_numPartitions;
220 m_mvd[0] = m_mv[1] + m_numPartitions;
221 m_mvd[1] = m_mvd[0] + m_numPartitions;
222
223 m_distortion = dataPool.distortionMemBlock + instance * m_numPartitions;
224
225 uint32_t cuSize = param.maxCUSize >> depth;
226 m_trCoeff[0] = dataPool.trCoeffMemBlock + instance * (cuSize * cuSize);
227 m_trCoeff[1] = m_trCoeff[2] = 0;
228 m_transformSkip[1] = m_transformSkip[2] = m_cbf[1] = m_cbf[2] = 0;
229 m_fAc_den[0] = m_fDc_den[0] = 0;
230 }
231 else
232 {
233 /* Each CU's data is layed out sequentially within the charMemBlock */
234 uint8_t *charBuf = dataPool.charMemBlock + (m_numPartitions * BytesPerPartition) * instance;
235
236 m_qp = (int8_t*)charBuf; charBuf += m_numPartitions;
237 m_qpAnalysis = (int8_t*)charBuf; charBuf += m_numPartitions;
238 m_log2CUSize = charBuf; charBuf += m_numPartitions;
239 m_lumaIntraDir = charBuf; charBuf += m_numPartitions;
240 m_tqBypass = charBuf; charBuf += m_numPartitions;
241 m_refIdx[0] = (int8_t*)charBuf; charBuf += m_numPartitions;
242 m_refIdx[1] = (int8_t*)charBuf; charBuf += m_numPartitions;
243 m_cuDepth = charBuf; charBuf += m_numPartitions;
244 m_predMode = charBuf; charBuf += m_numPartitions; /* the order up to here is important in initCTU() and initSubCU() */
245 m_partSize = charBuf; charBuf += m_numPartitions;
246 m_skipFlag[0] = charBuf; charBuf += m_numPartitions;
247 m_skipFlag[1] = charBuf; charBuf += m_numPartitions;
248 m_mergeFlag = charBuf; charBuf += m_numPartitions;
249 m_interDir = charBuf; charBuf += m_numPartitions;
250 m_mvpIdx[0] = charBuf; charBuf += m_numPartitions;
251 m_mvpIdx[1] = charBuf; charBuf += m_numPartitions;
252 m_tuDepth = charBuf; charBuf += m_numPartitions;
253 m_transformSkip[0] = charBuf; charBuf += m_numPartitions;
254 m_transformSkip[1] = charBuf; charBuf += m_numPartitions;
255 m_transformSkip[2] = charBuf; charBuf += m_numPartitions;
256 m_cbf[0] = charBuf; charBuf += m_numPartitions;
257 m_cbf[1] = charBuf; charBuf += m_numPartitions;
258 m_cbf[2] = charBuf; charBuf += m_numPartitions;
259 m_chromaIntraDir = charBuf; charBuf += m_numPartitions;
260
261 X265_CHECK(charBuf == dataPool.charMemBlock + (m_numPartitions * BytesPerPartition) * (instance + 1), "CU data layout is broken\n");
262
263 m_mv[0] = dataPool.mvMemBlock + (instance * 4) * m_numPartitions;
264 m_mv[1] = m_mv[0] + m_numPartitions;
265 m_mvd[0] = m_mv[1] + m_numPartitions;
266 m_mvd[1] = m_mvd[0] + m_numPartitions;
267
268 m_distortion = dataPool.distortionMemBlock + instance * m_numPartitions;
269
270 uint32_t cuSize = param.maxCUSize >> depth;
271 uint32_t sizeL = cuSize * cuSize;
272 uint32_t sizeC = sizeL >> (m_hChromaShift + m_vChromaShift); // block chroma part
273 m_trCoeff[0] = dataPool.trCoeffMemBlock + instance * (sizeL + sizeC * 2);
274 m_trCoeff[1] = m_trCoeff[0] + sizeL;
275 m_trCoeff[2] = m_trCoeff[0] + sizeL + sizeC;
276 for (int i = 0; i < 3; i++)
277 m_fAc_den[i] = m_fDc_den[i] = 0;
278 }

281void CUData::initCTU(const Frame& frame, uint32_t cuAddr, int qp, uint32_t firstRowInSlice, uint32_t lastRowInSlice, uint32_t lastCuInSlice)
282{
283 m_encData = frame.m_encData;
284 m_slice = m_encData->m_slice;
285 m_cuAddr = cuAddr;
286 m_cuPelX = (cuAddr % m_slice->m_sps->numCuInWidth) << m_slice->m_param->maxLog2CUSize;
287 m_cuPelY = (cuAddr / m_slice->m_sps->numCuInWidth) << m_slice->m_param->maxLog2CUSize;
288 m_absIdxInCTU = 0;
289 m_numPartitions = m_encData->m_param->num4x4Partitions;
290 m_bFirstRowInSlice = (uint8_t)firstRowInSlice;
291 m_bLastRowInSlice = (uint8_t)lastRowInSlice;
292 m_bLastCuInSlice = (uint8_t)lastCuInSlice;
293
294 /* sequential memsets */
295 m_partSet((uint8_t*)m_qp, (uint8_t)qp);
296 m_partSet((uint8_t*)m_qpAnalysis, (uint8_t)qp);
297 m_partSet(m_log2CUSize, (uint8_t)m_slice->m_param->maxLog2CUSize);
298 m_partSet(m_lumaIntraDir, (uint8_t)ALL_IDX);
299 m_partSet(m_chromaIntraDir, (uint8_t)ALL_IDX);
300 m_partSet(m_tqBypass, (uint8_t)frame.m_encData->m_param->bLossless);
301 if (m_slice->m_sliceType != I_SLICE)
302 {
303 m_partSet((uint8_t*)m_refIdx[0], (uint8_t)REF_NOT_VALID);
304 m_partSet((uint8_t*)m_refIdx[1], (uint8_t)REF_NOT_VALID);
305 }
306
307 X265_CHECK(!(frame.m_encData->m_param->bLossless && !m_slice->m_pps->bTransquantBypassEnabled), "lossless enabled without TQbypass in PPS\n");
308
309 /* initialize the remaining CU data in one memset */
310 memset(m_cuDepth, 0, (frame.m_param->internalCsp == X265_CSP_I400 ? BytesPerPartition - 12 : BytesPerPartition - 8) * m_numPartitions);
311
312 for (int8_t i = 0; i < NUM_TU_DEPTH; i++)
313 m_refTuDepth[i] = -1;
314
315 m_vbvAffected = false;
316
317 uint32_t widthInCU = m_slice->m_sps->numCuInWidth;
318 m_cuLeft = (m_cuAddr % widthInCU) ? m_encData->getPicCTU(m_cuAddr - 1) : NULL;
319 m_cuAbove = (m_cuAddr >= widthInCU) && !m_bFirstRowInSlice ? m_encData->getPicCTU(m_cuAddr - widthInCU) : NULL;
320 m_cuAboveLeft = (m_cuLeft && m_cuAbove) ? m_encData->getPicCTU(m_cuAddr - widthInCU - 1) : NULL;
321 m_cuAboveRight = (m_cuAbove && ((m_cuAddr % widthInCU) < (widthInCU - 1))) ? m_encData->getPicCTU(m_cuAddr - widthInCU + 1) : NULL;
322 memset(m_distortion, 0, m_numPartitions * sizeof(sse_t));
323}

326void CUData::initSubCU(const CUData& ctu, const CUGeom& cuGeom, int qp)
327{
328 m_absIdxInCTU = cuGeom.absPartIdx;
329 m_encData = ctu.m_encData;
330 m_slice = ctu.m_slice;
331 m_cuAddr = ctu.m_cuAddr;
332 m_cuPelX = ctu.m_cuPelX + g_zscanToPelX[cuGeom.absPartIdx];
333 m_cuPelY = ctu.m_cuPelY + g_zscanToPelY[cuGeom.absPartIdx];
334 m_cuLeft = ctu.m_cuLeft;
335 m_cuAbove = ctu.m_cuAbove;
336 m_cuAboveLeft = ctu.m_cuAboveLeft;
337 m_cuAboveRight = ctu.m_cuAboveRight;
338 m_bFirstRowInSlice = ctu.m_bFirstRowInSlice;
339 m_bLastRowInSlice = ctu.m_bLastRowInSlice;
340 m_bLastCuInSlice = ctu.m_bLastCuInSlice;
341 for (int i = 0; i < 3; i++)
342 {
343 m_fAc_den[i] = ctu.m_fAc_den[i];
344 m_fDc_den[i] = ctu.m_fDc_den[i];
345 }
346
347 X265_CHECK(m_numPartitions == cuGeom.numPartitions, "initSubCU() size mismatch\n");
348
349 m_partSet((uint8_t*)m_qp, (uint8_t)qp);
350 m_partSet((uint8_t*)m_qpAnalysis, (uint8_t)qp);
351
352 m_partSet(m_log2CUSize, (uint8_t)cuGeom.log2CUSize);
353 m_partSet(m_lumaIntraDir, (uint8_t)ALL_IDX);
354 m_partSet(m_chromaIntraDir, (uint8_t)ALL_IDX);
355 m_partSet(m_tqBypass, (uint8_t)m_encData->m_param->bLossless);
356 m_partSet((uint8_t*)m_refIdx[0], (uint8_t)REF_NOT_VALID);
357 m_partSet((uint8_t*)m_refIdx[1], (uint8_t)REF_NOT_VALID);
358 m_partSet(m_cuDepth, (uint8_t)cuGeom.depth);
359
360 /* initialize the remaining CU data in one memset */
361 memset(m_predMode, 0, (ctu.m_chromaFormat == X265_CSP_I400 ? BytesPerPartition - 13 : BytesPerPartition - 9) * m_numPartitions);
362 memset(m_distortion, 0, m_numPartitions * sizeof(sse_t));
363}

365/* Copy the results of a sub-part (split) CU to the parent CU */
366void CUData::copyPartFrom(const CUData& subCU, const CUGeom& childGeom, uint32_t subPartIdx)
367{
368 X265_CHECK(subPartIdx < 4, "part unit should be less than 4\n");
369
370 uint32_t offset = childGeom.numPartitions * subPartIdx;
371
372 m_bFirstRowInSlice = subCU.m_bFirstRowInSlice;
373 m_bLastCuInSlice = subCU.m_bLastCuInSlice;
374
375 m_subPartCopy((uint8_t*)m_qp + offset, (uint8_t*)subCU.m_qp);
376 m_subPartCopy((uint8_t*)m_qpAnalysis + offset, (uint8_t*)subCU.m_qpAnalysis);
377 m_subPartCopy(m_log2CUSize + offset, subCU.m_log2CUSize);
378 m_subPartCopy(m_lumaIntraDir + offset, subCU.m_lumaIntraDir);
379 m_subPartCopy(m_tqBypass + offset, subCU.m_tqBypass);
380 m_subPartCopy((uint8_t*)m_refIdx[0] + offset, (uint8_t*)subCU.m_refIdx[0]);
381 m_subPartCopy((uint8_t*)m_refIdx[1] + offset, (uint8_t*)subCU.m_refIdx[1]);
382 m_subPartCopy(m_cuDepth + offset, subCU.m_cuDepth);
383 m_subPartCopy(m_predMode + offset, subCU.m_predMode);
384 m_subPartCopy(m_partSize + offset, subCU.m_partSize);
385 m_subPartCopy(m_mergeFlag + offset, subCU.m_mergeFlag);
386 m_subPartCopy(m_interDir + offset, subCU.m_interDir);
387 m_subPartCopy(m_mvpIdx[0] + offset, subCU.m_mvpIdx[0]);
388 m_subPartCopy(m_mvpIdx[1] + offset, subCU.m_mvpIdx[1]);
389 m_subPartCopy(m_tuDepth + offset, subCU.m_tuDepth);
390
391 m_subPartCopy(m_transformSkip[0] + offset, subCU.m_transformSkip[0]);
392 m_subPartCopy(m_cbf[0] + offset, subCU.m_cbf[0]);
393
394 memcpy(m_mv[0] + offset, subCU.m_mv[0], childGeom.numPartitions * sizeof(MV));
395 memcpy(m_mv[1] + offset, subCU.m_mv[1], childGeom.numPartitions * sizeof(MV));
396 memcpy(m_mvd[0] + offset, subCU.m_mvd[0], childGeom.numPartitions * sizeof(MV));
397 memcpy(m_mvd[1] + offset, subCU.m_mvd[1], childGeom.numPartitions * sizeof(MV));
398
399 memcpy(m_distortion + offset, subCU.m_distortion, childGeom.numPartitions * sizeof(sse_t));
400
401 uint32_t tmp = 1 << ((m_slice->m_param->maxLog2CUSize - childGeom.depth) * 2);
402 uint32_t tmp2 = subPartIdx * tmp;
403 memcpy(m_trCoeff[0] + tmp2, subCU.m_trCoeff[0], sizeof(coeff_t)* tmp);
404
405 if (subCU.m_chromaFormat != X265_CSP_I400)
406 {
407 m_subPartCopy(m_transformSkip[1] + offset, subCU.m_transformSkip[1]);
408 m_subPartCopy(m_transformSkip[2] + offset, subCU.m_transformSkip[2]);
409 m_subPartCopy(m_cbf[1] + offset, subCU.m_cbf[1]);
410 m_subPartCopy(m_cbf[2] + offset, subCU.m_cbf[2]);
411 m_subPartCopy(m_chromaIntraDir + offset, subCU.m_chromaIntraDir);
412
413 uint32_t tmpC = tmp >> (m_hChromaShift + m_vChromaShift);
414 uint32_t tmpC2 = tmp2 >> (m_hChromaShift + m_vChromaShift);
415 memcpy(m_trCoeff[1] + tmpC2, subCU.m_trCoeff[1], sizeof(coeff_t) * tmpC);
416 memcpy(m_trCoeff[2] + tmpC2, subCU.m_trCoeff[2], sizeof(coeff_t) * tmpC);
417 }

429/* Copy all CU data from one instance to the next, except set lossless flag
430 * This will only get used when --cu-lossless is enabled but --lossless is not. */
431void CUData::initLosslessCU(const CUData& cu, const CUGeom& cuGeom)
432{
433 /* Start by making an exact copy */
434 m_encData = cu.m_encData;
435 m_slice = cu.m_slice;
436 m_cuAddr = cu.m_cuAddr;
437 m_cuPelX = cu.m_cuPelX;
438 m_cuPelY = cu.m_cuPelY;
439 m_cuLeft = cu.m_cuLeft;
440 m_cuAbove = cu.m_cuAbove;
441 m_cuAboveLeft = cu.m_cuAboveLeft;
442 m_cuAboveRight = cu.m_cuAboveRight;
443 m_absIdxInCTU = cuGeom.absPartIdx;
444 m_numPartitions = cuGeom.numPartitions;
445 memcpy(m_qp, cu.m_qp, BytesPerPartition * m_numPartitions);
446 memcpy(m_mv[0], cu.m_mv[0], m_numPartitions * sizeof(MV));
447 memcpy(m_mv[1], cu.m_mv[1], m_numPartitions * sizeof(MV));
448 memcpy(m_mvd[0], cu.m_mvd[0], m_numPartitions * sizeof(MV));
449 memcpy(m_mvd[1], cu.m_mvd[1], m_numPartitions * sizeof(MV));

912 const CUData* tempCU;
913 uint32_t tempPartIdx;
914 uint32_t leftIntraDir, aboveIntraDir;
915
916 // Get intra direction of left PU
917 tempCU = getPULeft(tempPartIdx, m_absIdxInCTU + absPartIdx);
918
919 leftIntraDir = (tempCU && tempCU->isIntra(tempPartIdx)) ? tempCU->m_lumaIntraDir[tempPartIdx] : DC_IDX;
920
921 // Get intra direction of above PU
922 tempCU = g_zscanToPelY[m_absIdxInCTU + absPartIdx] > 0 ? getPUAbove(tempPartIdx, m_absIdxInCTU + absPartIdx) : NULL;
923
924 aboveIntraDir = (tempCU && tempCU->isIntra(tempPartIdx)) ? tempCU->m_lumaIntraDir[tempPartIdx] : DC_IDX;
925
926 if (leftIntraDir == aboveIntraDir)
927 {
928 if (leftIntraDir >= 2) // angular modes
929 {
930 intraDirPred[0] = leftIntraDir;
931 intraDirPred[1] = ((leftIntraDir - 2 + 31) & 31) + 2;
932 intraDirPred[2] = ((leftIntraDir - 2 + 1) & 31) + 2;
933 }
934 else //non-angular
935 {
936 intraDirPred[0] = PLANAR_IDX;
937 intraDirPred[1] = DC_IDX;
938 intraDirPred[2] = VER_IDX;
939 }
940 return 1;
941 }
942 else
943 {
944 intraDirPred[0] = leftIntraDir;
945 intraDirPred[1] = aboveIntraDir;
946
947 if (leftIntraDir && aboveIntraDir) //both modes are non-planar
948 intraDirPred[2] = PLANAR_IDX;
949 else
950 intraDirPred[2] = (leftIntraDir + aboveIntraDir) < 2 ? VER_IDX : DC_IDX;
951 return 2;
952 }

1098 memset(m_interDir + absPartIdx + curPartNumQ + (curPartNumQ >> 1), dir, (curPartNumQ >> 2));

1130
1131 case SIZE_2NxN:

1135 break;
1136

1158 }

1164
1165 pT = p + curPartNumQ;
1166 for (i = 0; i < ((curPartNumQ >> 1) + (curPartNumQ << 1)); i++)
1167 pT[i] = val;
1168 }
1169 break;
1170 }
1171
1172 case SIZE_2NxnD:
1173 {
1174 int curPartNumQ = numElements >> 2;

1179 pT[i] = val;
1180
1181 pT = p + (numElements - curPartNumQ);
1182 for (i = 0; i < (curPartNumQ >> 1); i++)
1183 pT[i] = val;
1184 }
1185 else
1186 {
1187 T *pT = p;
1188 T *pT2 = p + curPartNumQ;
1189 for (i = 0; i < (curPartNumQ >> 1); i++)
1190 {
1191 pT[i] = val;
1192 pT2[i] = val;
1193 }
1194 }
1195 break;
1196 }
1197
1198 case SIZE_nLx2N:
1199 {

1204 T *pT2 = p + (curPartNumQ << 1);

1214 }
1215 }

1218 T *pT = p;

1221 {
1222 pT[i] = val;
1223 pT2[i] = val;
1224 }
1225

1227 pT2 = p + (curPartNumQ << 1) + (curPartNumQ >> 1);
1228 for (i = 0; i < ((curPartNumQ >> 2) + curPartNumQ); i++)
1229 {
1230 pT[i] = val;
1231 pT2[i] = val;
1232 }
1233 }
1234 break;
1235 }
1236
1237 case SIZE_nRx2N:
1238 {
1239 int curPartNumQ = numElements >> 2;
1240 if (!puIdx)
1241 {
1242 T *pT = p;
1243 T *pT2 = p + (curPartNumQ << 1);
1244 for (i = 0; i < ((curPartNumQ >> 2) + curPartNumQ); i++)
1245 {
1246 pT[i] = val;
1247 pT2[i] = val;
1248 }
1249
1250 pT = p + curPartNumQ + (curPartNumQ >> 1);
1251 pT2 = p + numElements - curPartNumQ + (curPartNumQ >> 1);
1252 for (i = 0; i < (curPartNumQ >> 2); i++)
1253 {
1254 pT[i] = val;
1255 pT2[i] = val;
1256 }
1257 }
1258 else
1259 {
1260 T *pT = p;

1263 T *pT4 = p + (curPartNumQ << 1) + (curPartNumQ >> 1);

1265 {
1266 pT[i] = val;
1267 pT2[i] = val;
1268 pT3[i] = val;
1269 pT4[i] = val;

1279 }

1487 uint32_t count = 0;
1488
1489 uint32_t partIdxLT, partIdxRT, partIdxLB = deriveLeftBottomIdx(puIdx);
1490 PartSize curPS = (PartSize)m_partSize[absPartIdx];
1491
1492 // left
1493 uint32_t leftPartIdx = 0;
1494 const CUData* cuLeft = getPULeft(leftPartIdx, partIdxLB);
1495 bool isAvailableA1 = cuLeft &&
1496 cuLeft->isDiffMER(xP - 1, yP + nPSH - 1, xP, yP) &&
1497 !(puIdx == 1 && (curPS == SIZE_Nx2N || curPS == SIZE_nLx2N || curPS == SIZE_nRx2N)) &&
1498 cuLeft->isInter(leftPartIdx);
1499 if (isAvailableA1)
1500 {
1501 // get Inter Dir
1502 candDir[count] = cuLeft->m_interDir[leftPartIdx];
1503 // get Mv from Left
1504 cuLeft->getMvField(cuLeft, leftPartIdx, 0, candMvField[count][0]);
1505 if (isInterB)
1506 cuLeft->getMvField(cuLeft, leftPartIdx, 1, candMvField[count][1]);
1507
1508 if (++count == maxNumMergeCand)
1509 return maxNumMergeCand;
1510 }
1511
1512 deriveLeftRightTopIdx(puIdx, partIdxLT, partIdxRT);
1513
1514 // above
1515 uint32_t abovePartIdx = 0;
1516 const CUData* cuAbove = getPUAbove(abovePartIdx, partIdxRT);
1517 bool isAvailableB1 = cuAbove &&
1518 cuAbove->isDiffMER(xP + nPSW - 1, yP - 1, xP, yP) &&
1519 !(puIdx == 1 && (curPS == SIZE_2NxN || curPS == SIZE_2NxnU || curPS == SIZE_2NxnD)) &&
1520 cuAbove->isInter(abovePartIdx);
1521 if (isAvailableB1 && (!isAvailableA1 || !cuLeft->hasEqualMotion(leftPartIdx, *cuAbove, abovePartIdx)))
1522 {
1523 // get Inter Dir
1524 candDir[count] = cuAbove->m_interDir[abovePartIdx];
1525 // get Mv from Left
1526 cuAbove->getMvField(cuAbove, abovePartIdx, 0, candMvField[count][0]);
1527 if (isInterB)
1528 cuAbove->getMvField(cuAbove, abovePartIdx, 1, candMvField[count][1]);
1529
1530 if (++count == maxNumMergeCand)
1531 return maxNumMergeCand;
1532 }
1533
1534 // above right
1535 uint32_t aboveRightPartIdx = 0;
1536 const CUData* cuAboveRight = getPUAboveRight(aboveRightPartIdx, partIdxRT);
1537 bool isAvailableB0 = cuAboveRight &&
1538 cuAboveRight->isDiffMER(xP + nPSW, yP - 1, xP, yP) &&
1539 cuAboveRight->isInter(aboveRightPartIdx);
1540 if (isAvailableB0 && (!isAvailableB1 || !cuAbove->hasEqualMotion(abovePartIdx, *cuAboveRight, aboveRightPartIdx)))
1541 {
1542 // get Inter Dir
1543 candDir[count] = cuAboveRight->m_interDir[aboveRightPartIdx];
1544 // get Mv from Left
1545 cuAboveRight->getMvField(cuAboveRight, aboveRightPartIdx, 0, candMvField[count][0]);
1546 if (isInterB)
1547 cuAboveRight->getMvField(cuAboveRight, aboveRightPartIdx, 1, candMvField[count][1]);
1548
1549 if (++count == maxNumMergeCand)
1550 return maxNumMergeCand;
1551 }
1552
1553 // left bottom
1554 uint32_t leftBottomPartIdx = 0;
1555 const CUData* cuLeftBottom = this->getPUBelowLeft(leftBottomPartIdx, partIdxLB);
1556 bool isAvailableA0 = cuLeftBottom &&
1557 cuLeftBottom->isDiffMER(xP - 1, yP + nPSH, xP, yP) &&
1558 cuLeftBottom->isInter(leftBottomPartIdx);
1559 if (isAvailableA0 && (!isAvailableA1 || !cuLeft->hasEqualMotion(leftPartIdx, *cuLeftBottom, leftBottomPartIdx)))
1560 {
1561 // get Inter Dir
1562 candDir[count] = cuLeftBottom->m_interDir[leftBottomPartIdx];
1563 // get Mv from Left
1564 cuLeftBottom->getMvField(cuLeftBottom, leftBottomPartIdx, 0, candMvField[count][0]);
1565 if (isInterB)
1566 cuLeftBottom->getMvField(cuLeftBottom, leftBottomPartIdx, 1, candMvField[count][1]);
1567
1568 if (++count == maxNumMergeCand)
1569 return maxNumMergeCand;
1570 }
1571
1572 // above left
1573 if (count < 4)
1574 {
1575 uint32_t aboveLeftPartIdx = 0;
1576 const CUData* cuAboveLeft = getPUAboveLeft(aboveLeftPartIdx, absPartAddr);
1577 bool isAvailableB2 = cuAboveLeft &&
1578 cuAboveLeft->isDiffMER(xP - 1, yP - 1, xP, yP) &&
1579 cuAboveLeft->isInter(aboveLeftPartIdx);
1580 if (isAvailableB2 && (!isAvailableA1 || !cuLeft->hasEqualMotion(leftPartIdx, *cuAboveLeft, aboveLeftPartIdx))
1581 && (!isAvailableB1 || !cuAbove->hasEqualMotion(abovePartIdx, *cuAboveLeft, aboveLeftPartIdx)))
1582 {
1583 // get Inter Dir
1584 candDir[count] = cuAboveLeft->m_interDir[aboveLeftPartIdx];
1585 // get Mv from Left
1586 cuAboveLeft->getMvField(cuAboveLeft, aboveLeftPartIdx, 0, candMvField[count][0]);
1587 if (isInterB)
1588 cuAboveLeft->getMvField(cuAboveLeft, aboveLeftPartIdx, 1, candMvField[count][1]);
1589
1590 if (++count == maxNumMergeCand)
1591 return maxNumMergeCand;
1592 }
1593 }
1594 if (m_slice->m_sps->bTemporalMVPEnabled)
1595 {
1596 uint32_t partIdxRB = deriveRightBottomIdx(puIdx);
1597 MV colmv;
1598 int ctuIdx = -1;
1599
1600 // image boundary check
1601 if (m_encData->getPicCTU(m_cuAddr)->m_cuPelX + g_zscanToPelX[partIdxRB] + UNIT_SIZE < m_slice->m_sps->picWidthInLumaSamples &&
1602 m_encData->getPicCTU(m_cuAddr)->m_cuPelY + g_zscanToPelY[partIdxRB] + UNIT_SIZE < m_slice->m_sps->picHeightInLumaSamples)
1603 {
1604 uint32_t absPartIdxRB = g_zscanToRaster[partIdxRB];
1605 uint32_t numUnits = s_numPartInCUSize;
1606 bool bNotLastCol = lessThanCol(absPartIdxRB, numUnits - 1); // is not at the last column of CTU
1607 bool bNotLastRow = lessThanRow(absPartIdxRB, numUnits - 1); // is not at the last row of CTU
1608
1609 if (bNotLastCol && bNotLastRow)
1610 {
1611 absPartAddr = g_rasterToZscan[absPartIdxRB + RASTER_SIZE + 1];
1612 ctuIdx = m_cuAddr;
1613 }
1614 else if (bNotLastCol)
1615 absPartAddr = g_rasterToZscan[(absPartIdxRB + 1) & (numUnits - 1)];
1616 else if (bNotLastRow)
1617 {
1618 absPartAddr = g_rasterToZscan[absPartIdxRB + RASTER_SIZE - numUnits + 1];
1619 ctuIdx = m_cuAddr + 1;
1620 }
1621 else // is the right bottom corner of CTU
1622 absPartAddr = 0;
1623 }
1624
1625 int maxList = isInterB ? 2 : 1;
1626 int dir = 0, refIdx = 0;
1627 for (int list = 0; list < maxList; list++)
1628 {
1629 bool bExistMV = ctuIdx >= 0 && getColMVP(colmv, refIdx, list, ctuIdx, absPartAddr);
1630 if (!bExistMV)
1631 {
1632 uint32_t partIdxCenter = deriveCenterIdx(puIdx);
1633 bExistMV = getColMVP(colmv, refIdx, list, m_cuAddr, partIdxCenter);
1634 }
1635 if (bExistMV)
1636 {
1637 dir |= (1 << list);
1638 candMvField[count][list].mv = colmv;
1639 candMvField[count][list].refIdx = refIdx;
1640 }
1641 }
1642
1643 if (dir != 0)
1644 {
1645 candDir[count] = (uint8_t)dir;
1646
1647 if (++count == maxNumMergeCand)
1648 return maxNumMergeCand;
1649 }
1650 }
1651
1652 if (isInterB)
1653 {
1654 const uint32_t cutoff = count * (count - 1);
1655 uint32_t priorityList0 = 0xEDC984; // { 0, 1, 0, 2, 1, 2, 0, 3, 1, 3, 2, 3 }
1656 uint32_t priorityList1 = 0xB73621; // { 1, 0, 2, 0, 2, 1, 3, 0, 3, 1, 3, 2 }
1657
1658 for (uint32_t idx = 0; idx < cutoff; idx++, priorityList0 >>= 2, priorityList1 >>= 2)
1659 {
1660 int i = priorityList0 & 3;
1661 int j = priorityList1 & 3;
1662
1663 if ((candDir[i] & 0x1) && (candDir[j] & 0x2))
1664 {
1665 // get Mv from cand[i] and cand[j]
1666 int refIdxL0 = candMvField[i][0].refIdx;
1667 int refIdxL1 = candMvField[j][1].refIdx;
1668 int refPOCL0 = m_slice->m_refPOCList[0][refIdxL0];
1669 int refPOCL1 = m_slice->m_refPOCList[1][refIdxL1];
1670 if (!(refPOCL0 == refPOCL1 && candMvField[i][0].mv == candMvField[j][1].mv))
1671 {
1672 candMvField[count][0].mv = candMvField[i][0].mv;
1673 candMvField[count][0].refIdx = refIdxL0;
1674 candMvField[count][1].mv = candMvField[j][1].mv;
1675 candMvField[count][1].refIdx = refIdxL1;
1676 candDir[count] = 3;
1677
1678 if (++count == maxNumMergeCand)
1679 return maxNumMergeCand;
1680 }
1681 }
1682 }
1683 }
1684 int numRefIdx = (isInterB) ? X265_MIN(m_slice->m_numRefIdx[0], m_slice->m_numRefIdx[1]) : m_slice->m_numRefIdx[0];
1685 int r = 0;
1686 int refcnt = 0;
1687 while (count < maxNumMergeCand)
1688 {
1689 candDir[count] = 1;
1690 candMvField[count][0].mv.word = 0;
1691 candMvField[count][0].refIdx = r;
1692
1693 if (isInterB)
1694 {
1695 candDir[count] = 3;
1696 candMvField[count][1].mv.word = 0;
1697 candMvField[count][1].refIdx = r;
1698 }
1699
1700 count++;
1701
1702 if (refcnt == numRefIdx - 1)
1703 r = 0;
1704 else
1705 {
1706 ++r;
1707 ++refcnt;
1708 }
1709 }
1710
1711 return count;
1712}

1714// Create the PMV list. Called for each reference index.
1715int CUData::getPMV(InterNeighbourMV *neighbours, uint32_t picList, uint32_t refIdx, MV* amvpCand, MV* pmv) const
1716{
1717 MV directMV[MD_ABOVE_LEFT + 1];
1718 MV indirectMV[MD_ABOVE_LEFT + 1];
1719 bool validDirect[MD_ABOVE_LEFT + 1];
1720 bool validIndirect[MD_ABOVE_LEFT + 1];
1721
1722 // Left candidate.
1723 validDirect[MD_BELOW_LEFT] = getDirectPMV(directMV[MD_BELOW_LEFT], neighbours + MD_BELOW_LEFT, picList, refIdx);
1724 validDirect[MD_LEFT] = getDirectPMV(directMV[MD_LEFT], neighbours + MD_LEFT, picList, refIdx);
1725 // Top candidate.
1726 validDirect[MD_ABOVE_RIGHT] = getDirectPMV(directMV[MD_ABOVE_RIGHT], neighbours + MD_ABOVE_RIGHT, picList, refIdx);
1727 validDirect[MD_ABOVE] = getDirectPMV(directMV[MD_ABOVE], neighbours + MD_ABOVE, picList, refIdx);
1728 validDirect[MD_ABOVE_LEFT] = getDirectPMV(directMV[MD_ABOVE_LEFT], neighbours + MD_ABOVE_LEFT, picList, refIdx);
1729
1730 // Left candidate.
1731 validIndirect[MD_BELOW_LEFT] = getIndirectPMV(indirectMV[MD_BELOW_LEFT], neighbours + MD_BELOW_LEFT, picList, refIdx);
1732 validIndirect[MD_LEFT] = getIndirectPMV(indirectMV[MD_LEFT], neighbours + MD_LEFT, picList, refIdx);
1733 // Top candidate.
1734 validIndirect[MD_ABOVE_RIGHT] = getIndirectPMV(indirectMV[MD_ABOVE_RIGHT], neighbours + MD_ABOVE_RIGHT, picList, refIdx);
1735 validIndirect[MD_ABOVE] = getIndirectPMV(indirectMV[MD_ABOVE], neighbours + MD_ABOVE, picList, refIdx);
1736 validIndirect[MD_ABOVE_LEFT] = getIndirectPMV(indirectMV[MD_ABOVE_LEFT], neighbours + MD_ABOVE_LEFT, picList, refIdx);
1737
1738 int num = 0;
1739 // Left predictor search
1740 if (validDirect[MD_BELOW_LEFT])
1741 amvpCand[num++] = directMV[MD_BELOW_LEFT];
1742 else if (validDirect[MD_LEFT])
1743 amvpCand[num++] = directMV[MD_LEFT];
1744 else if (validIndirect[MD_BELOW_LEFT])
1745 amvpCand[num++] = indirectMV[MD_BELOW_LEFT];
1746 else if (validIndirect[MD_LEFT])
1747 amvpCand[num++] = indirectMV[MD_LEFT];
1748
1749 bool bAddedSmvp = num > 0;
1750
1751 // Above predictor search
1752 if (validDirect[MD_ABOVE_RIGHT])
1753 amvpCand[num++] = directMV[MD_ABOVE_RIGHT];
1754 else if (validDirect[MD_ABOVE])
1755 amvpCand[num++] = directMV[MD_ABOVE];
1756 else if (validDirect[MD_ABOVE_LEFT])
1757 amvpCand[num++] = directMV[MD_ABOVE_LEFT];
1758
1759 if (!bAddedSmvp)
1760 {
1761 if (validIndirect[MD_ABOVE_RIGHT])
1762 amvpCand[num++] = indirectMV[MD_ABOVE_RIGHT];
1763 else if (validIndirect[MD_ABOVE])
1764 amvpCand[num++] = indirectMV[MD_ABOVE];
1765 else if (validIndirect[MD_ABOVE_LEFT])
1766 amvpCand[num++] = indirectMV[MD_ABOVE_LEFT];
1767 }
1768
1769 int numMvc = 0;
1770 for (int dir = MD_LEFT; dir <= MD_ABOVE_LEFT; dir++)
1771 {
1772 if (validDirect[dir] && directMV[dir].notZero())
1773 pmv[numMvc++] = directMV[dir];
1774
1775 if (validIndirect[dir] && indirectMV[dir].notZero())
1776 pmv[numMvc++] = indirectMV[dir];
1777 }
1778
1779 if (num == 2)
1780 num -= amvpCand[0] == amvpCand[1];
1781
1782 // Get the collocated candidate. At this step, either the first candidate
1783 // was found or its value is 0.
1784 if (m_slice->m_sps->bTemporalMVPEnabled && num < 2)
1785 {
1786 int tempRefIdx = neighbours[MD_COLLOCATED].refIdx[picList];
1787 if (tempRefIdx != -1)
1788 {
1789 uint32_t cuAddr = neighbours[MD_COLLOCATED].cuAddr[picList];
1790 const Frame* colPic = m_slice->m_refFrameList[m_slice->isInterB() && !m_slice->m_colFromL0Flag][m_slice->m_colRefIdx];
1791 const CUData* colCU = colPic->m_encData->getPicCTU(cuAddr);
1792
1793 // Scale the vector
1794 int colRefPOC = colCU->m_slice->m_refPOCList[tempRefIdx >> 4][tempRefIdx & 0xf];
1795 int colPOC = colCU->m_slice->m_poc;
1796
1797 int curRefPOC = m_slice->m_refPOCList[picList][refIdx];
1798 int curPOC = m_slice->m_poc;
1799 pmv[numMvc++] = amvpCand[num++] = scaleMvByPOCDist(neighbours[MD_COLLOCATED].mv[picList], curPOC, curRefPOC, colPOC, colRefPOC);
1800 }
1801 }
1802
1803 while (num < AMVP_NUM_CANDS)
1804 amvpCand[num++] = 0;
1805
1806 return numMvc;
1807}

1809/* Constructs a list of candidates for AMVP, and a larger list of motion candidates */
1810void CUData::getNeighbourMV(uint32_t puIdx, uint32_t absPartIdx, InterNeighbourMV* neighbours) const
1811{
1812 // Set the temporal neighbour to unavailable by default.
1813 neighbours[MD_COLLOCATED].unifiedRef = -1;
1814
1815 uint32_t partIdxLT, partIdxRT, partIdxLB = deriveLeftBottomIdx(puIdx);
1816 deriveLeftRightTopIdx(puIdx, partIdxLT, partIdxRT);
1817
1818 // Load the spatial MVs.
1819 getInterNeighbourMV(neighbours + MD_BELOW_LEFT, partIdxLB, MD_BELOW_LEFT);
1820 getInterNeighbourMV(neighbours + MD_LEFT, partIdxLB, MD_LEFT);
1821 getInterNeighbourMV(neighbours + MD_ABOVE_RIGHT,partIdxRT, MD_ABOVE_RIGHT);
1822 getInterNeighbourMV(neighbours + MD_ABOVE, partIdxRT, MD_ABOVE);
1823 getInterNeighbourMV(neighbours + MD_ABOVE_LEFT, partIdxLT, MD_ABOVE_LEFT);
1824
1825 if (m_slice->m_sps->bTemporalMVPEnabled)
1826 {
1827 uint32_t absPartAddr = m_absIdxInCTU + absPartIdx;
1828 uint32_t partIdxRB = deriveRightBottomIdx(puIdx);
1829
1830 // co-located RightBottom temporal predictor (H)
1831 int ctuIdx = -1;
1832
1833 // image boundary check
1834 if (m_encData->getPicCTU(m_cuAddr)->m_cuPelX + g_zscanToPelX[partIdxRB] + UNIT_SIZE < m_slice->m_sps->picWidthInLumaSamples &&
1835 m_encData->getPicCTU(m_cuAddr)->m_cuPelY + g_zscanToPelY[partIdxRB] + UNIT_SIZE < m_slice->m_sps->picHeightInLumaSamples)
1836 {
1837 uint32_t absPartIdxRB = g_zscanToRaster[partIdxRB];
1838 uint32_t numUnits = s_numPartInCUSize;
1839 bool bNotLastCol = lessThanCol(absPartIdxRB, numUnits - 1); // is not at the last column of CTU
1840 bool bNotLastRow = lessThanRow(absPartIdxRB, numUnits - 1); // is not at the last row of CTU
1841
1842 if (bNotLastCol && bNotLastRow)
1843 {
1844 absPartAddr = g_rasterToZscan[absPartIdxRB + RASTER_SIZE + 1];
1845 ctuIdx = m_cuAddr;
1846 }
1847 else if (bNotLastCol)
1848 absPartAddr = g_rasterToZscan[(absPartIdxRB + 1) & (numUnits - 1)];
1849 else if (bNotLastRow)
1850 {
1851 absPartAddr = g_rasterToZscan[absPartIdxRB + RASTER_SIZE - numUnits + 1];
1852 ctuIdx = m_cuAddr + 1;
1853 }
1854 else // is the right bottom corner of CTU
1855 absPartAddr = 0;
1856 }
1857
1858 if (!(ctuIdx >= 0 && getCollocatedMV(ctuIdx, absPartAddr, neighbours + MD_COLLOCATED)))
1859 {
1860 uint32_t partIdxCenter = deriveCenterIdx(puIdx);
1861 uint32_t curCTUIdx = m_cuAddr;
1862 getCollocatedMV(curCTUIdx, partIdxCenter, neighbours + MD_COLLOCATED);
1863 }
1864 }
1865}

1867void CUData::getInterNeighbourMV(InterNeighbourMV *neighbour, uint32_t partUnitIdx, MVP_DIR dir) const
1868{
1869 const CUData* tmpCU = NULL;
1870 uint32_t idx = 0;
1871
1872 switch (dir)
1873 {
1874 case MD_LEFT:
1875 tmpCU = getPULeft(idx, partUnitIdx);
1876 break;
1877 case MD_ABOVE:
1878 tmpCU = getPUAbove(idx, partUnitIdx);
1879 break;
1880 case MD_ABOVE_RIGHT:
1881 tmpCU = getPUAboveRight(idx, partUnitIdx);
1882 break;
1883 case MD_BELOW_LEFT:
1884 tmpCU = getPUBelowLeft(idx, partUnitIdx);
1885 break;
1886 case MD_ABOVE_LEFT:
1887 tmpCU = getPUAboveLeft(idx, partUnitIdx);
1888 break;
1889 default:
1890 break;
1891 }
1892
1893 if (!tmpCU)
1894 {
1895 // Mark the PMV as unavailable.
1896 for (int i = 0; i < 2; i++)
1897 neighbour->refIdx[i] = -1;
1898 return;
1899 }
1900
1901 for (int i = 0; i < 2; i++)
1902 {
1903 // Get the MV.
1904 neighbour->mv[i] = tmpCU->m_mv[i][idx];
1905
1906 // Get the reference idx.
1907 neighbour->refIdx[i] = tmpCU->m_refIdx[i][idx];
1908 }
1909}

1911/* Clip motion vector to within slightly padded boundary of picture (the
1912 * MV may reference a block that is completely within the padded area).
1913 * Note this function is unaware of how much of this picture is actually
1914 * available for use (re: frame parallelism) */
1915void CUData::clipMv(MV& outMV) const
1916{
1917 const uint32_t mvshift = 2;
1918 uint32_t offset = 8;
1919
1920 int32_t xmax = (int32_t)((m_slice->m_sps->picWidthInLumaSamples + offset - m_cuPelX - 1) << mvshift);
1921 int32_t xmin = -(int32_t)((m_encData->m_param->maxCUSize + offset + m_cuPelX - 1) << mvshift);
1922
1923 int32_t ymax = (int32_t)((m_slice->m_sps->picHeightInLumaSamples + offset - m_cuPelY - 1) << mvshift);
1924 int32_t ymin = -(int32_t)((m_encData->m_param->maxCUSize + offset + m_cuPelY - 1) << mvshift);
1925
1926 outMV.x = X265_MIN(xmax, X265_MAX(xmin, outMV.x));
1927 outMV.y = X265_MIN(ymax, X265_MAX(ymin, outMV.y));
1928}

1931bool CUData::getDirectPMV(MV& pmv, InterNeighbourMV *neighbours, uint32_t picList, uint32_t refIdx) const

1934 for (int i = 0; i < 2; i++, picList = !picList)
1935 {
1936 int partRefIdx = neighbours->refIdx[picList];
1937 if (partRefIdx >= 0 && curRefPOC == m_slice->m_refPOCList[picList][partRefIdx])
1938 {
1939 pmv = neighbours->mv[picList];
1940 return true;
1941 }
1942 }
1943 return false;

1968bool CUData::getColMVP(MV& outMV, int& outRefIdx, int picList, int cuAddr, int partUnitIdx) const
1969{
1970 const Frame* colPic = m_slice->m_refFrameList[m_slice->isInterB() && !m_slice->m_colFromL0Flag][m_slice->m_colRefIdx];
1971 const CUData* colCU = colPic->m_encData->getPicCTU(cuAddr);
1972
1973 uint32_t absPartAddr = partUnitIdx & TMVP_UNIT_MASK;
1974 if (colCU->m_predMode[partUnitIdx] == MODE_NONE || colCU->isIntra(absPartAddr))
1975 return false;
1976
1977 int colRefPicList = m_slice->m_bCheckLDC ? picList : m_slice->m_colFromL0Flag;
1978
1979 int colRefIdx = colCU->m_refIdx[colRefPicList][absPartAddr];
1980
1981 if (colRefIdx < 0)
1982 {
1983 colRefPicList = !colRefPicList;
1984 colRefIdx = colCU->m_refIdx[colRefPicList][absPartAddr];
1985
1986 if (colRefIdx < 0)
1987 return false;
1988 }
1989
1990 // Scale the vector
1991 int colRefPOC = colCU->m_slice->m_refPOCList[colRefPicList][colRefIdx];
1992 int colPOC = colCU->m_slice->m_poc;
1993 MV colmv = colCU->m_mv[colRefPicList][absPartAddr];
1994
1995 int curRefPOC = m_slice->m_refPOCList[picList][outRefIdx];
1996 int curPOC = m_slice->m_poc;
1997
1998 outMV = scaleMvByPOCDist(colmv, curPOC, curRefPOC, colPOC, colRefPOC);
1999 return true;
2000}

2002// Cache the collocated MV.
2003bool CUData::getCollocatedMV(int cuAddr, int partUnitIdx, InterNeighbourMV *neighbour) const
2004{
2005 const Frame* colPic = m_slice->m_refFrameList[m_slice->isInterB() && !m_slice->m_colFromL0Flag][m_slice->m_colRefIdx];
2006 const CUData* colCU = colPic->m_encData->getPicCTU(cuAddr);
2007
2008 uint32_t absPartAddr = partUnitIdx & TMVP_UNIT_MASK;
2009 if (colCU->m_predMode[partUnitIdx] == MODE_NONE || colCU->isIntra(absPartAddr))

2104void CUData::calcCTUGeoms(uint32_t ctuWidth, uint32_t ctuHeight, uint32_t maxCUSize, uint32_t minCUSize, CUGeom cuDataArray[CUGeom::MAX_GEOMS])
2105{
2106 uint32_t num4x4Partition = (1U << ((g_log2Size[maxCUSize] - LOG2_UNIT_SIZE) << 1));
2107
2108 // Initialize the coding blocks inside the CTB
2109 for (uint32_t log2CUSize = g_log2Size[maxCUSize], rangeCUIdx = 0; log2CUSize >= g_log2Size[minCUSize]; log2CUSize--)
2110 {
2111 uint32_t blockSize = 1 << log2CUSize;
2112 uint32_t sbWidth = 1 << (g_log2Size[maxCUSize] - log2CUSize);
2113 int32_t lastLevelFlag = log2CUSize == g_log2Size[minCUSize];
2114
2115 for (uint32_t sbY = 0; sbY < sbWidth; sbY++)
2116 {
2117 for (uint32_t sbX = 0; sbX < sbWidth; sbX++)
2118 {
2119 uint32_t depthIdx = g_depthScanIdx[sbY][sbX];
2120 uint32_t cuIdx = rangeCUIdx + depthIdx;
2121 uint32_t childIdx = rangeCUIdx + sbWidth * sbWidth + (depthIdx << 2);
2122 uint32_t px = sbX * blockSize;
2123 uint32_t py = sbY * blockSize;
2124 int32_t presentFlag = px < ctuWidth && py < ctuHeight;
2125 int32_t splitMandatoryFlag = presentFlag && !lastLevelFlag && (px + blockSize > ctuWidth || py + blockSize > ctuHeight);
2126
2127 /* Offset of the luma CU in the X, Y direction in terms of pixels from the CTU origin */
2128 uint32_t xOffset = (sbX * blockSize) >> 3;
2129 uint32_t yOffset = (sbY * blockSize) >> 3;
2130 X265_CHECK(cuIdx < CUGeom::MAX_GEOMS, "CU geom index bug\n");
2131
2132 CUGeom *cu = cuDataArray + cuIdx;
2133 cu->log2CUSize = log2CUSize;
2134 cu->childOffset = childIdx - cuIdx;
2135 cu->absPartIdx = g_depthScanIdx[yOffset][xOffset] * 4;
2136 cu->numPartitions = (num4x4Partition >> ((g_log2Size[maxCUSize] - cu->log2CUSize) * 2));
2137 cu->depth = g_log2Size[maxCUSize] - log2CUSize;
2138 cu->geomRecurId = cuIdx;
2139
2140 cu->flags = 0;
2141 CU_SET_FLAG(cu->flags, CUGeom::PRESENT, presentFlag);
2142 CU_SET_FLAG(cu->flags, CUGeom::SPLIT_MANDATORY | CUGeom::SPLIT, splitMandatoryFlag);
2143 CU_SET_FLAG(cu->flags, CUGeom::LEAF, lastLevelFlag);
2144 }
2145 }
2146 rangeCUIdx += sbWidth * sbWidth;
2147 }
