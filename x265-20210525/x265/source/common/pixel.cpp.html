44

121template<int lx, int ly>
122int ads_x4(int encDC[4], uint32_t *sums, int delta, uint16_t *costMvX, int16_t *mvs, int width, int thresh)
123{
124 int nmv = 0;
125 for (int16_t i = 0; i < width; i++, sums++)
126 {
127 int ads = abs(encDC[0] - long(sums[0]))
128 + abs(encDC[1] - long(sums[lx >> 1]))
129 + abs(encDC[2] - long(sums[delta]))
130 + abs(encDC[3] - long(sums[delta + (lx >> 1)]))
131 + costMvX[i];
132 if (ads < thresh)
133 mvs[nmv++] = i;
134 }
135 return nmv;
136}

167template<int lx, int ly, class T1, class T2>
168sse_t sse(const T1* pix1, intptr_t stride_pix1, const T2* pix2, intptr_t stride_pix2)
169{
170 sse_t sum = 0;
171 int tmp;
172
173 for (int y = 0; y < ly; y++)
174 {
175 for (int x = 0; x < lx; x++)
176 {
177 tmp = pix1[x] - pix2[x];
178 sum += (tmp * tmp);
179 }
180
181 pix1 += stride_pix1;
182 pix2 += stride_pix2;
183 }
184
185 return sum;
186}

188#define BITS_PER_SUM (8 * sizeof(sum_t))
189
190#define HADAMARD4(d0, d1, d2, d3, s0, s1, s2, s3) { \
191 sum2_t t0 = s0 + s1; \
192 sum2_t t1 = s0 - s1; \
193 sum2_t t2 = s2 + s3; \
194 sum2_t t3 = s2 - s3; \
195 d0 = t0 + t2; \
196 d2 = t0 - t2; \
197 d1 = t1 + t3; \
198 d3 = t1 - t3; \
199}

201// in: a pseudo-simd number of the form x+(y<<16)
202// return: abs(x)+(abs(y)<<16)
203inline sum2_t abs2(sum2_t a)
204{
205 sum2_t s = ((a >> (BITS_PER_SUM - 1)) & (((sum2_t)1 << BITS_PER_SUM) + 1)) * ((sum_t)-1);
206
207 return (a + s) ^ s;
208}

210static int satd_4x4(const pixel* pix1, intptr_t stride_pix1, const pixel* pix2, intptr_t stride_pix2)
211{
212 sum2_t tmp[4][2];
213 sum2_t a0, a1, a2, a3, b0, b1;
214 sum2_t sum = 0;
215
216 for (int i = 0; i < 4; i++, pix1 += stride_pix1, pix2 += stride_pix2)
217 {
218 a0 = pix1[0] - pix2[0];
219 a1 = pix1[1] - pix2[1];
220 b0 = (a0 + a1) + ((a0 - a1) << BITS_PER_SUM);
221 a2 = pix1[2] - pix2[2];
222 a3 = pix1[3] - pix2[3];
223 b1 = (a2 + a3) + ((a2 - a3) << BITS_PER_SUM);
224 tmp[i][0] = b0 + b1;
225 tmp[i][1] = b0 - b1;
226 }
227
228 for (int i = 0; i < 2; i++)
229 {
230 HADAMARD4(a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i]);
231 a0 = abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
232 sum += ((sum_t)a0) + (a0 >> BITS_PER_SUM);
233 }
234
235 return (int)(sum >> 1);
236}

238// x264's SWAR version of satd 8x4, performs two 4x4 SATDs at once
239static int satd_8x4(const pixel* pix1, intptr_t stride_pix1, const pixel* pix2, intptr_t stride_pix2)
240{
241 sum2_t tmp[4][4];
242 sum2_t a0, a1, a2, a3;
243 sum2_t sum = 0;
244
245 for (int i = 0; i < 4; i++, pix1 += stride_pix1, pix2 += stride_pix2)
246 {
247 a0 = (pix1[0] - pix2[0]) + ((sum2_t)(pix1[4] - pix2[4]) << BITS_PER_SUM);
248 a1 = (pix1[1] - pix2[1]) + ((sum2_t)(pix1[5] - pix2[5]) << BITS_PER_SUM);
249 a2 = (pix1[2] - pix2[2]) + ((sum2_t)(pix1[6] - pix2[6]) << BITS_PER_SUM);
250 a3 = (pix1[3] - pix2[3]) + ((sum2_t)(pix1[7] - pix2[7]) << BITS_PER_SUM);
251 HADAMARD4(tmp[i][0], tmp[i][1], tmp[i][2], tmp[i][3], a0, a1, a2, a3);
252 }
253
254 for (int i = 0; i < 4; i++)
255 {
256 HADAMARD4(a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i]);
257 sum += abs2(a0) + abs2(a1) + abs2(a2) + abs2(a3);
258 }
259
260 return (((sum_t)sum) + (sum >> BITS_PER_SUM)) >> 1;
261}

263template<int w, int h>
264// calculate satd in blocks of 4x4
265int satd4(const pixel* pix1, intptr_t stride_pix1, const pixel* pix2, intptr_t stride_pix2)
266{
267 int satd = 0;
268
269#if ENABLE_ASSEMBLY && X265_ARCH_ARM64
270 pixelcmp_t satd_4x4 = x265_pixel_satd_4x4_neon;
271#endif
272
273 for (int row = 0; row < h; row += 4)
274 for (int col = 0; col < w; col += 4)
275 satd += satd_4x4(pix1 + row * stride_pix1 + col, stride_pix1,
276 pix2 + row * stride_pix2 + col, stride_pix2);
277
278 return satd;
279}

299inline int _sa8d_8x8(const pixel* pix1, intptr_t i_pix1, const pixel* pix2, intptr_t i_pix2)
300{
301 sum2_t tmp[8][4];
302 sum2_t a0, a1, a2, a3, a4, a5, a6, a7, b0, b1, b2, b3;
303 sum2_t sum = 0;
304
305 for (int i = 0; i < 8; i++, pix1 += i_pix1, pix2 += i_pix2)
306 {
307 a0 = pix1[0] - pix2[0];
308 a1 = pix1[1] - pix2[1];
309 b0 = (a0 + a1) + ((a0 - a1) << BITS_PER_SUM);
310 a2 = pix1[2] - pix2[2];
311 a3 = pix1[3] - pix2[3];
312 b1 = (a2 + a3) + ((a2 - a3) << BITS_PER_SUM);
313 a4 = pix1[4] - pix2[4];
314 a5 = pix1[5] - pix2[5];
315 b2 = (a4 + a5) + ((a4 - a5) << BITS_PER_SUM);
316 a6 = pix1[6] - pix2[6];
317 a7 = pix1[7] - pix2[7];
318 b3 = (a6 + a7) + ((a6 - a7) << BITS_PER_SUM);
319 HADAMARD4(tmp[i][0], tmp[i][1], tmp[i][2], tmp[i][3], b0, b1, b2, b3);
320 }
321
322 for (int i = 0; i < 4; i++)
323 {
324 HADAMARD4(a0, a1, a2, a3, tmp[0][i], tmp[1][i], tmp[2][i], tmp[3][i]);
325 HADAMARD4(a4, a5, a6, a7, tmp[4][i], tmp[5][i], tmp[6][i], tmp[7][i]);
326 b0 = abs2(a0 + a4) + abs2(a0 - a4);
327 b0 += abs2(a1 + a5) + abs2(a1 - a5);
328 b0 += abs2(a2 + a6) + abs2(a2 - a6);
329 b0 += abs2(a3 + a7) + abs2(a3 - a7);
330 sum += (sum_t)b0 + (b0 >> BITS_PER_SUM);
331 }
332
333 return (int)sum;
334}

341static int sa8d_16x16(const pixel* pix1, intptr_t i_pix1, const pixel* pix2, intptr_t i_pix2)
342{
343 int sum = _sa8d_8x8(pix1, i_pix1, pix2, i_pix2)
344 + _sa8d_8x8(pix1 + 8, i_pix1, pix2 + 8, i_pix2)
345 + _sa8d_8x8(pix1 + 8 * i_pix1, i_pix1, pix2 + 8 * i_pix2, i_pix2)
346 + _sa8d_8x8(pix1 + 8 + 8 * i_pix1, i_pix1, pix2 + 8 + 8 * i_pix2, i_pix2);
347
348 // This matches x264 sa8d_16x16, but is slightly different from HM's behavior because
349 // this version only rounds once at the end
350 return (sum + 2) >> 2;
351}

493static void weight_sp_c(const int16_t* src, pixel* dst, intptr_t srcStride, intptr_t dstStride, int width, int height, int w0, int round, int shift, int offset)
494{
495 int x, y;
496
497#if CHECKED_BUILD || _DEBUG
498 const int correction = (IF_INTERNAL_PREC - X265_DEPTH);
499 X265_CHECK(!((w0 << 6) > 32767), "w0 using more than 16 bits, asm output will mismatch\n");
500 X265_CHECK(!(round > 32767), "round using more than 16 bits, asm output will mismatch\n");
501 X265_CHECK((shift >= correction), "shift must be include factor correction, please update ASM ABI\n");
502#endif
503
504 for (y = 0; y <= height - 1; y++)
505 {
506 for (x = 0; x <= width - 1; )
507 {
508 // note: width can be odd
509 dst[x] = x265_clip(((w0 * (src[x] + IF_INTERNAL_OFFS) + round) >> shift) + offset);
510 x++;
511 }
512
513 src += srcStride;
514 dst += dstStride;
515 }
516}

518static void weight_pp_c(const pixel* src, pixel* dst, intptr_t stride, int width, int height, int w0, int round, int shift, int offset)
519{
520 int x, y;
521
522 const int correction = (IF_INTERNAL_PREC - X265_DEPTH);
523
524 X265_CHECK(!(width & 15), "weightp alignment error\n");
525 X265_CHECK(!((w0 << 6) > 32767), "w0 using more than 16 bits, asm output will mismatch\n");
526 X265_CHECK(!(round > 32767), "round using more than 16 bits, asm output will mismatch\n");
527 X265_CHECK((shift >= correction), "shift must be include factor correction, please update ASM ABI\n");
528 X265_CHECK(!(round & ((1 << correction) - 1)), "round must be include factor correction, please update ASM ABI\n");
529
530 for (y = 0; y <= height - 1; y++)
531 {
532 for (x = 0; x <= width - 1; )
533 {
534 // simulating pixel to short conversion
535 int16_t val = src[x] << correction;
536 dst[x] = x265_clip(((w0 * (val) + round) >> shift) + offset);
537 x++;
538 }
539
540 src += stride;
541 dst += stride;
542 }
543}

559static void scale1D_128to64(pixel *dst, const pixel *src)
560{
561 int x;
562 const pixel* src1 = src;
563 const pixel* src2 = src + 128;
564
565 pixel* dst1 = dst;
566 pixel* dst2 = dst + 64/*128*/;
567
568 for (x = 0; x < 128; x += 2)
569 {
570 // Top pixel
571 pixel pix0 = src1[(x + 0)];
572 pixel pix1 = src1[(x + 1)];
573
574 // Left pixel
575 pixel pix2 = src2[(x + 0)];
576 pixel pix3 = src2[(x + 1)];
577 int sum1 = pix0 + pix1;
578 int sum2 = pix2 + pix3;
579
580 dst1[x >> 1] = (pixel)((sum1 + 1) >> 1);
581 dst2[x >> 1] = (pixel)((sum2 + 1) >> 1);
582 }
583}

585static void scale2D_64to32(pixel* dst, const pixel* src, intptr_t stride)
586{
587 uint32_t x, y;
588
589 for (y = 0; y < 64; y += 2)
590 {
591 for (x = 0; x < 64; x += 2)
592 {
593 pixel pix0 = src[(y + 0) * stride + (x + 0)];
594 pixel pix1 = src[(y + 0) * stride + (x + 1)];
595 pixel pix2 = src[(y + 1) * stride + (x + 0)];
596 pixel pix3 = src[(y + 1) * stride + (x + 1)];
597 int sum = pix0 + pix1 + pix2 + pix3;
598
599 dst[y / 2 * 32 + x / 2] = (pixel)((sum + 2) >> 2);
600 }
601 }

631static void ssim_4x4x2_core(const pixel* pix1, intptr_t stride1, const pixel* pix2, intptr_t stride2, int sums[2][4])
632{

637 {

641 int b = pix2[x + y * stride2];
642 s1 += a;
643 s2 += b;
644 ss += a * a;
645 ss += b * b;
646 s12 += a * b;
647 }
648 }

650 sums[z][0] = s1;
651 sums[z][1] = s2;
652 sums[z][2] = ss;
653 sums[z][3] = s12;
654 pix1 += 4;
655 pix2 += 4;
656 }

659static float ssim_end_1(int s1, int s2, int ss, int s12)
660{
661/* Maximum value for 10-bit is: ss*64 = (2^10-1)^2*16*4*64 = 4286582784, which will overflow in some cases.
662 * s1*s1, s2*s2, and s1*s2 also obtain this value for edge cases: ((2^10-1)*16*4)^2 = 4286582784.
663 * Maximum value for 9-bit is: ss*64 = (2^9-1)^2*16*4*64 = 1069551616, which will not overflow. */
664
665#if HIGH_BIT_DEPTH
666 X265_CHECK((X265_DEPTH == 10) || (X265_DEPTH == 12), "ssim invalid depth\n");
667#define type float
668 static const float ssim_c1 = (float)(.01 * .01 * PIXEL_MAX * PIXEL_MAX * 64);
669 static const float ssim_c2 = (float)(.03 * .03 * PIXEL_MAX * PIXEL_MAX * 64 * 63);
670#else
671 X265_CHECK(X265_DEPTH == 8, "ssim invalid depth\n");
672#define type int
673 static const int ssim_c1 = (int)(.01 * .01 * PIXEL_MAX * PIXEL_MAX * 64 + .5);
674 static const int ssim_c2 = (int)(.03 * .03 * PIXEL_MAX * PIXEL_MAX * 64 * 63 + .5);
675#endif
676 type fs1 = (type)s1;
677 type fs2 = (type)s2;
678 type fss = (type)ss;
679 type fs12 = (type)s12;
680 type vars = (type)(fss * 64 - fs1 * fs1 - fs2 * fs2);
681 type covar = (type)(fs12 * 64 - fs1 * fs2);
682 return (float)(2 * fs1 * fs2 + ssim_c1) * (float)(2 * covar + ssim_c2)
683 / ((float)(fs1 * fs1 + fs2 * fs2 + ssim_c1) * (float)(vars + ssim_c2));
684#undef type
685#undef PIXEL_MAX
686}

688static float ssim_end_4(int sum0[5][4], int sum1[5][4], int width)
689{
690 float ssim = 0.0;
691
692 for (int i = 0; i < width; i++)
693 {
694 ssim += ssim_end_1(sum0[i][0] + sum0[i + 1][0] + sum1[i][0] + sum1[i + 1][0],
695 sum0[i][1] + sum0[i + 1][1] + sum1[i][1] + sum1[i + 1][1],
696 sum0[i][2] + sum0[i + 1][2] + sum1[i][2] + sum1[i + 1][2],
697 sum0[i][3] + sum0[i + 1][3] + sum1[i][3] + sum1[i + 1][3]);
698 }
699
700 return ssim;

705{
706 uint32_t sum = 0, sqr = 0;
707
708 for (int y = 0; y < size; y++)
709 {
710 for (int x = 0; x < size; x++)
711 {
712 sum += pix[x];

726template<int size>
727int psyCost_pp(const pixel* source, intptr_t sstride, const pixel* recon, intptr_t rstride)
728{
729 static pixel zeroBuf[8] /* = { 0 } */;
730
731 if (size)
732 {
733 int dim = 1 << (size + 2);
734 uint32_t totEnergy = 0;
735 for (int i = 0; i < dim; i += 8)
736 {
737 for (int j = 0; j < dim; j+= 8)
738 {
739 /* AC energy, measured by sa8d (AC + DC) minus SAD (DC) */
740 int sourceEnergy = sa8d_8x8(source + i * sstride + j, sstride, zeroBuf, 0) -
741 (sad<8, 8>(source + i * sstride + j, sstride, zeroBuf, 0) >> 2);
742 int reconEnergy = sa8d_8x8(recon + i * rstride + j, rstride, zeroBuf, 0) -
743 (sad<8, 8>(recon + i * rstride + j, rstride, zeroBuf, 0) >> 2);
744
745 totEnergy += abs(sourceEnergy - reconEnergy);
746 }
747 }
748 return totEnergy;
749 }
750 else
751 {
752 /* 4x4 is too small for sa8d */
753 int sourceEnergy = satd_4x4(source, sstride, zeroBuf, 0) - (sad<4, 4>(source, sstride, zeroBuf, 0) >> 2);
754 int reconEnergy = satd_4x4(recon, rstride, zeroBuf, 0) - (sad<4, 4>(recon, rstride, zeroBuf, 0) >> 2);
755 return abs(sourceEnergy - reconEnergy);
756 }
757}

914static void estimateCUPropagateCost(int* dst, const uint16_t* propagateIn, const int32_t* intraCosts, const uint16_t* interCosts,
915 const int32_t* invQscales, const double* fpsFactor, int len)

918 for (int i = 0; i < len; i++)

920 int intraCost = intraCosts[i];
921 int interCost = X265_MIN(intraCosts[i], interCosts[i] & LOWRES_COST_MASK);
922 double propagateIntra = intraCost * invQscales[i]; // Q16 x Q8.8 = Q24.8
923 double propagateAmount = (double)propagateIn[i] + propagateIntra * fps; // Q16.0 + Q24.8 x Q0.x = Q25.0
924 double propagateNum = (double)(intraCost - interCost); // Q32 - Q32 = Q33.0
925
926#if 0

928 float intraRcp = (float)1.0f / intraCost; // VC can't mapping this into RCPPS
929 float intraRcpError1 = (float)intraCost * (float)intraRcp;
930 intraRcpError1 *= (float)intraRcp;
931 float intraRcpError2 = intraRcp + intraRcp;
932 float propagateDenom = intraRcpError2 - intraRcpError1;
933 dst[i] = (int)(propagateAmount * propagateNum * (double)propagateDenom + 0.5);
934#else
