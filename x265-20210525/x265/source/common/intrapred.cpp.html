31template<int tuSize>
32void intraFilter(const pixel* samples, pixel* filtered) /* 1:2:1 filtering of left and top reference samples */
33{
34 const int tuSize2 = tuSize << 1;
35
36 pixel topLeft = samples[0], topLast = samples[tuSize2], leftLast = samples[tuSize2 + tuSize2];
37
38 // filtering top
39 for (int i = 1; i < tuSize2; i++)
40 filtered[i] = ((samples[i] << 1) + samples[i - 1] + samples[i + 1] + 2) >> 2;
41 filtered[tuSize2] = topLast;
42
43 // filtering top-left
44 filtered[0] = ((topLeft << 1) + samples[1] + samples[tuSize2 + 1] + 2) >> 2;
45
46 // filtering left
47 filtered[tuSize2 + 1] = ((samples[tuSize2 + 1] << 1) + topLeft + samples[tuSize2 + 2] + 2) >> 2;
48 for (int i = tuSize2 + 2; i < tuSize2 + tuSize2; i++)
49 filtered[i] = ((samples[i] << 1) + samples[i - 1] + samples[i + 1] + 2) >> 2;
50 filtered[tuSize2 + tuSize2] = leftLast;
51}

53static void dcPredFilter(const pixel* above, const pixel* left, pixel* dst, intptr_t dststride, int size)
54{
55 // boundary pixels processing
56 dst[0] = (pixel)((above[0] + left[0] + 2 * dst[0] + 2) >> 2);
57
58 for (int x = 1; x < size; x++)
59 dst[x] = (pixel)((above[x] + 3 * dst[x] + 2) >> 2);
60
61 dst += dststride;
62 for (int y = 1; y < size; y++)
63 {
64 *dst = (pixel)((left[y] + 3 * *dst + 2) >> 2);
65 dst += dststride;
66 }
67}

69template<int width>
70void intra_pred_dc_c(pixel* dst, intptr_t dstStride, const pixel* srcPix, int /*dirMode*/, int bFilter)
71{
72 int k, l;
73
74 int dcVal = width;
75 for (int i = 0; i < width; i++)
76 dcVal += srcPix[1 + i] + srcPix[2 * width + 1 + i];
77
78 dcVal = dcVal / (width + width);
79 for (k = 0; k < width; k++)
80 for (l = 0; l < width; l++)
81 dst[k * dstStride + l] = (pixel)dcVal;
82
83 if (bFilter)
84 dcPredFilter(srcPix + 1, srcPix + (2 * width + 1), dst, dstStride, width);
85}

87template<int log2Size>
88void planar_pred_c(pixel* dst, intptr_t dstStride, const pixel* srcPix, int /*dirMode*/, int /*bFilter*/)
89{
90 const int blkSize = 1 << log2Size;
91
92 const pixel* above = srcPix + 1;
93 const pixel* left = srcPix + (2 * blkSize + 1);
94
95 pixel topRight = above[blkSize];
96 pixel bottomLeft = left[blkSize];
97 for (int y = 0; y < blkSize; y++)
98 for (int x = 0; x < blkSize; x++)
99 dst[y * dstStride + x] = (pixel) (((blkSize - 1 - x) * left[y] + (blkSize - 1 -y) * above[x] + (x + 1) * topRight + (y + 1) * bottomLeft + blkSize) >> (log2Size + 1));
100}

102template<int width>
103void intra_pred_ang_c(pixel* dst, intptr_t dstStride, const pixel *srcPix0, int dirMode, int bFilter)
104{
105 int width2 = width << 1;
106 // Flip the neighbours in the horizontal case.
107 int horMode = dirMode < 18;
108 pixel neighbourBuf[129];
109 const pixel *srcPix = srcPix0;
110
111 if (horMode)
112 {
113 neighbourBuf[0] = srcPix[0];
114 for (int i = 0; i < width << 1; i++)
115 {
116 neighbourBuf[1 + i] = srcPix[width2 + 1 + i];
117 neighbourBuf[width2 + 1 + i] = srcPix[1 + i];
118 }
119 srcPix = neighbourBuf;
120 }
121
122 // Intra prediction angle and inverse angle tables.
123 const int8_t angleTable[17] = { -32, -26, -21, -17, -13, -9, -5, -2, 0, 2, 5, 9, 13, 17, 21, 26, 32 };
124 const int16_t invAngleTable[8] = { 4096, 1638, 910, 630, 482, 390, 315, 256 };
125
126 // Get the prediction angle.
127 int angleOffset = horMode ? 10 - dirMode : dirMode - 26;
128 int angle = angleTable[8 + angleOffset];
129
130 // Vertical Prediction.
131 if (!angle)
132 {
133 for (int y = 0; y < width; y++)
134 for (int x = 0; x < width; x++)
135 dst[y * dstStride + x] = srcPix[1 + x];
136
137 if (bFilter)
138 {
139 int topLeft = srcPix[0], top = srcPix[1];
140 for (int y = 0; y < width; y++)
141 dst[y * dstStride] = x265_clip((int16_t)(top + ((srcPix[width2 + 1 + y] - topLeft) >> 1)));
142 }
143 }
144 else // Angular prediction.
145 {
146 // Get the reference pixels. The reference base is the first pixel to the top (neighbourBuf[1]).
147 pixel refBuf[64];
148 const pixel *ref;
149
150 // Use the projected left neighbours and the top neighbours.
151 if (angle < 0)
152 {
153 // Number of neighbours projected.
154 int nbProjected = -((width * angle) >> 5) - 1;
155 pixel *ref_pix = refBuf + nbProjected + 1;
156
157 // Project the neighbours.
158 int invAngle = invAngleTable[- angleOffset - 1];
159 int invAngleSum = 128;
160 for (int i = 0; i < nbProjected; i++)
161 {
162 invAngleSum += invAngle;
163 ref_pix[- 2 - i] = srcPix[width2 + (invAngleSum >> 8)];
164 }
165
166 // Copy the top-left and top pixels.
167 for (int i = 0; i < width + 1; i++)
168 ref_pix[-1 + i] = srcPix[i];
169 ref = ref_pix;
170 }
171 else // Use the top and top-right neighbours.
172 ref = srcPix + 1;
173
174 // Pass every row.
175 int angleSum = 0;
176 for (int y = 0; y < width; y++)
177 {
178 angleSum += angle;
179 int offset = angleSum >> 5;
180 int fraction = angleSum & 31;
181
182 if (fraction) // Interpolate
183 for (int x = 0; x < width; x++)
184 dst[y * dstStride + x] = (pixel)(((32 - fraction) * ref[offset + x] + fraction * ref[offset + x + 1] + 16) >> 5);
185 else // Copy.
186 for (int x = 0; x < width; x++)
187 dst[y * dstStride + x] = ref[offset + x];
188 }
189 }
190
191 // Flip for horizontal.
192 if (horMode)
193 {
194 for (int y = 0; y < width - 1; y++)
195 {
196 for (int x = y + 1; x < width; x++)
197 {
198 pixel tmp = dst[y * dstStride + x];
199 dst[y * dstStride + x] = dst[x * dstStride + y];
200 dst[x * dstStride + y] = tmp;
201 }
202 }
203 }
204}

206template<int log2Size>
207void all_angs_pred_c(pixel *dest, pixel *refPix, pixel *filtPix, int bLuma)
208{
209 const int size = 1 << log2Size;
210 for (int mode = 2; mode <= 34; mode++)
211 {
212 pixel *srcPix = (g_intraFilterFlags[mode] & size ? filtPix : refPix);
213 pixel *out = dest + ((mode - 2) << (log2Size * 2));
214
215 intra_pred_ang_c<size>(out, size, srcPix, mode, bLuma);
216
217 // Optimize code don't flip buffer
218 bool modeHor = (mode < 18);
219
220 // transpose the block if this is a horizontal mode
221 if (modeHor)
222 {
223 for (int k = 0; k < size - 1; k++)
224 {
225 for (int l = k + 1; l < size; l++)
226 {
227 pixel tmp = out[k * size + l];
228 out[k * size + l] = out[l * size + k];
229 out[l * size + k] = tmp;
230 }
231 }
232 }
233 }
234}

240void setupIntraPrimitives_c(EncoderPrimitives& p)
241{
242 p.cu[BLOCK_4x4].intra_filter = intraFilter<4>;
243 p.cu[BLOCK_8x8].intra_filter = intraFilter<8>;
244 p.cu[BLOCK_16x16].intra_filter = intraFilter<16>;
245 p.cu[BLOCK_32x32].intra_filter = intraFilter<32>;
246
247 p.cu[BLOCK_4x4].intra_pred[PLANAR_IDX] = planar_pred_c<2>;
248 p.cu[BLOCK_8x8].intra_pred[PLANAR_IDX] = planar_pred_c<3>;
249 p.cu[BLOCK_16x16].intra_pred[PLANAR_IDX] = planar_pred_c<4>;
250 p.cu[BLOCK_32x32].intra_pred[PLANAR_IDX] = planar_pred_c<5>;
251
252 p.cu[BLOCK_4x4].intra_pred[DC_IDX] = intra_pred_dc_c<4>;
253 p.cu[BLOCK_8x8].intra_pred[DC_IDX] = intra_pred_dc_c<8>;
254 p.cu[BLOCK_16x16].intra_pred[DC_IDX] = intra_pred_dc_c<16>;
255 p.cu[BLOCK_32x32].intra_pred[DC_IDX] = intra_pred_dc_c<32>;
256
257 for (int i = 2; i < NUM_INTRA_MODE; i++)
258 {
259 p.cu[BLOCK_4x4].intra_pred[i] = intra_pred_ang_c<4>;
260 p.cu[BLOCK_8x8].intra_pred[i] = intra_pred_ang_c<8>;
261 p.cu[BLOCK_16x16].intra_pred[i] = intra_pred_ang_c<16>;
262 p.cu[BLOCK_32x32].intra_pred[i] = intra_pred_ang_c<32>;
263 }
264
265 p.cu[BLOCK_4x4].intra_pred_allangs = all_angs_pred_c<2>;
266 p.cu[BLOCK_8x8].intra_pred_allangs = all_angs_pred_c<3>;
267 p.cu[BLOCK_16x16].intra_pred_allangs = all_angs_pred_c<4>;
268 p.cu[BLOCK_32x32].intra_pred_allangs = all_angs_pred_c<5>;
269}
