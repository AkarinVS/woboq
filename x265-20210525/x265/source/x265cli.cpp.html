432 void CLIOptions::printStatus(uint32_t frameNum)
433 {
434 char buf[200];
435 int64_t time = x265_mdate();
436
437 if (!bProgress || !frameNum || (prevUpdateTime && time - prevUpdateTime < UPDATE_INTERVAL))
438 return;
439
440 int64_t elapsed = time - startTime;
441 double fps = elapsed > 0 ? frameNum * 1000000. / elapsed : 0;
442 float bitrate = 0.008f * totalbytes * (param->fpsNum / param->fpsDenom) / ((float)frameNum);
443 if (framesToBeEncoded)
444 {
445 int eta = (int)(elapsed * (framesToBeEncoded - frameNum) / ((int64_t)frameNum * 1000000));
446 sprintf(buf, "x265 [%.1f%%] %d/%d frames, %.2f fps, %.2f kb/s, eta %d:%02d:%02d",
447 100. * frameNum / (param->chunkEnd ? param->chunkEnd : param->totalFrames), frameNum, (param->chunkEnd ? param->chunkEnd : param->totalFrames), fps, bitrate,
448 eta / 3600, (eta / 60) % 60, eta % 60);
449 }
450 else
451 sprintf(buf, "x265 %d frames: %.2f fps, %.2f kb/s", frameNum, fps, bitrate);
452
453 fprintf(stderr, "%s \r", buf + 5);
454 SetConsoleTitle(buf);
455 fflush(stderr); // needed in windows
456 prevUpdateTime = time;
457 }

459 bool CLIOptions::parseZoneParam(int argc, char **argv, x265_param* globalParam, int zonefileCount)
460 {
461 bool bError = false;
462 int bShowHelp = false;
463 int outputBitDepth = 0;
464 const char *profile = NULL;
465
466 /* Presets are applied before all other options. */
467 for (optind = 0;;)
468 {
469 int c = getopt_long(argc, argv, short_options, long_options, NULL);
470 if (c == -1)
471 break;
472 else if (c == 'D')
473 outputBitDepth = atoi(optarg);
474 else if (c == 'P')
475 profile = optarg;
476 else if (c == '?')
477 bShowHelp = true;
478 }
479
480 if (!outputBitDepth && profile)
481 {
482 /* try to derive the output bit depth from the requested profile */
483 if (strstr(profile, "10"))
484 outputBitDepth = 10;
485 else if (strstr(profile, "12"))
486 outputBitDepth = 12;
487 else
488 outputBitDepth = 8;
489 }
490
491 api = x265_api_get(outputBitDepth);
492 if (!api)
493 {
494 x265_log(NULL, X265_LOG_WARNING, "falling back to default bit-depth\n");
495 api = x265_api_get(0);
496 }
497
498 if (bShowHelp)
499 {
500 printVersion(globalParam, api);
501 showHelp(globalParam);
502 }
503
504 globalParam->rc.zones[zonefileCount].zoneParam = api->param_alloc();
505 if (!globalParam->rc.zones[zonefileCount].zoneParam)
506 {
507 x265_log(NULL, X265_LOG_ERROR, "param alloc failed\n");
508 return true;
509 }
510
511 memcpy(globalParam->rc.zones[zonefileCount].zoneParam, globalParam, sizeof(x265_param));
512
513 for (optind = 0;;)
514 {
515 int long_options_index = -1;
516 int c = getopt_long(argc, argv, short_options, long_options, &long_options_index);
517 if (c == -1)
518 break;
519
520 if (long_options_index < 0 && c > 0)
521 {
522 for (size_t i = 0; i < sizeof(long_options) / sizeof(long_options[0]); i++)
523 {
524 if (long_options[i].val == c)
525 {
526 long_options_index = (int)i;
527 break;
528 }
529 }
530
531 if (long_options_index < 0)
532 {
533 /* getopt_long might have already printed an error message */
534 if (c != 63)
535 x265_log(NULL, X265_LOG_WARNING, "internal error: short option '%c' has no long option\n", c);
536 return true;
537 }
538 }
539 if (long_options_index < 0)
540 {
541 x265_log(NULL, X265_LOG_WARNING, "short option '%c' unrecognized\n", c);
542 return true;
543 }
544
545 bError |= !!api->zone_param_parse(globalParam->rc.zones[zonefileCount].zoneParam, long_options[long_options_index].name, optarg);
546
547 if (bError)
548 {
549 const char *name = long_options_index > 0 ? long_options[long_options_index].name : argv[optind - 2];
550 x265_log(NULL, X265_LOG_ERROR, "invalid argument: %s = %s\n", name, optarg);
551 return true;
552 }
553 }
554
555 if (optind < argc)
556 {
557 x265_log(param, X265_LOG_WARNING, "extra unused command arguments given <%s>\n", argv[optind]);
558 return true;
559 }
560 return false;
561 }

951 bool CLIOptions::parseQPFile(x265_picture &pic_org)
952 {
953 int32_t num = -1, qp, ret;
954 char type;
955 uint32_t filePos;
956 pic_org.forceqp = 0;
957 pic_org.sliceType = X265_TYPE_AUTO;
958 while (num < pic_org.poc)
959 {
960 filePos = ftell(qpfile);
961 qp = -1;
962 ret = fscanf(qpfile, "%d %c%*[ \t]%d\n", &num, &type, &qp);
963
964 if (num > pic_org.poc || ret == EOF)
965 {
966 fseek(qpfile, filePos, SEEK_SET);
967 break;
968 }
969 if (num < pic_org.poc && ret >= 2)
970 continue;
971 if (ret == 3 && qp >= 0)
972 pic_org.forceqp = qp + 1;
973 if (type == 'I') pic_org.sliceType = X265_TYPE_IDR;
974 else if (type == 'i') pic_org.sliceType = X265_TYPE_I;
975 else if (type == 'K') pic_org.sliceType = param->bOpenGOP ? X265_TYPE_I : X265_TYPE_IDR;
976 else if (type == 'P') pic_org.sliceType = X265_TYPE_P;
977 else if (type == 'B') pic_org.sliceType = X265_TYPE_BREF;
978 else if (type == 'b') pic_org.sliceType = X265_TYPE_B;
979 else ret = 0;
980 if (ret < 2 || qp < -1 || qp > 51)
981 return 0;
982 }
983 return 1;
984 }

1039 /* Parse the RPU file and extract the RPU corresponding to the current picture
1040 * and fill the rpu field of the input picture */
1041 int CLIOptions::rpuParser(x265_picture * pic)
1042 {
1043 uint8_t byteVal;
1044 uint32_t code = 0;
1045 int bytesRead = 0;
1046 pic->rpu.payloadSize = 0;
1047
1048 if (!pic->pts)
1049 {
1050 while (bytesRead++ < 4 && fread(&byteVal, sizeof(uint8_t), 1, dolbyVisionRpu))
1051 code = (code << 8) | byteVal;
1052
1053 if (code != START_CODE)
1054 {
1055 x265_log(NULL, X265_LOG_ERROR, "Invalid Dolby Vision RPU startcode in POC %d\n", pic->pts);
1056 return 1;
1057 }
1058 }
1059
1060 bytesRead = 0;
1061 while (fread(&byteVal, sizeof(uint8_t), 1, dolbyVisionRpu))
1062 {
1063 code = (code << 8) | byteVal;
1064 if (bytesRead++ < 3)
1065 continue;
1066 if (bytesRead >= 1024)
1067 {
1068 x265_log(NULL, X265_LOG_ERROR, "Invalid Dolby Vision RPU size in POC %d\n", pic->pts);
1069 return 1;
1070 }
1071
1072 if (code != START_CODE)
1073 pic->rpu.payload[pic->rpu.payloadSize++] = (code >> (3 * 8)) & 0xFF;
1074 else
1075 return 0;
1076 }
1077
1078 int ShiftBytes = START_CODE_BYTES - (bytesRead - pic->rpu.payloadSize);
1079 int bytesLeft = bytesRead - pic->rpu.payloadSize;
1080 code = (code << ShiftBytes * 8);
1081 for (int i = 0; i < bytesLeft; i++)
1082 {
1083 pic->rpu.payload[pic->rpu.payloadSize++] = (code >> (3 * 8)) & 0xFF;
1084 code = (code << 8);
1085 }
1086 if (!pic->rpu.payloadSize)
1087 x265_log(NULL, X265_LOG_WARNING, "Dolby Vision RPU not found for POC %d\n", pic->pts);
1088 return 0;
1089 }
